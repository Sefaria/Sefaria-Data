פרשת במדבר




במדבר פרק-א

{א} וַיְדַבֵּ֨ר יְהוָ֧ה אֶל-מֹשֶׁ֛ה בְּמִדְבַּ֥ר סִינַ֖י בְּאֹ֣הֶל מוֹעֵ֑ד בְּאֶחָד֩ לַחֹ֨דֶשׁ הַשֵּׁנִ֜י בַּשָּׁנָ֣ה הַשֵּׁנִ֗ית לְצֵאתָ֛ם מֵאֶ֥רֶץ מִצְרַ֖יִם לֵאמֹֽר:  מ  וַיְדַבֵּ֨ר יְהוָ֧ה אֶל-מֹשֶׁ֛ה בְּמִדְבַּ֥ר סִינַ֖י בְּאֹ֣הֶל מוֹעֵ֑ד בְּאֶחָד֩ לַחֹ֨דֶשׁ הַשֵּׁנִ֜י בַּשָּׁנָ֣ה הַשֵּׁנִ֗ית לְצֵאתָ֛ם מֵאֶ֥רֶץ מִצְרַ֖יִם לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה בְּמַדְבְּרָא דְסִינַי בְּמַשְׁכַּן זִמְנָא בְּחַד לְיַרְחָא תִנְיָנָא בְּשַׁתָּא תִנְיֵתָא לְמִפַּקְהוֹן מֵאַרְעָא דְמִצְרַיִם לְמֵימָר:

{ב} שְׂא֗וּ אֶת-רֹאשׁ֙ כָּל-עֲדַ֣ת בְּנֵֽי-יִשְׂרָאֵ֔ל לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֔וֹת כָּל-זָכָ֖ר לְגֻלְגְּלֹתָֽם:  מ  שְׂא֗וּ אֶת-רֹאשׁ֙ כָּל-עֲדַ֣ת בְּנֵֽי-יִשְׂרָאֵ֔ל לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֔וֹת כָּל-זָכָ֖ר לְגֻלְגְּלֹתָֽם:  ת  קַבִּילוּ יָת חֻשְׁבַּן כָּל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל לְזַרְעֲ יַתְהוֹן לְבֵית אֲבָהַתְהוֹן בְּמִנְיַן שְׁמָהָן כָּל דְכוּרָא לְגֻלְגְלַתְהוֹן:

{ג} מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כָּל-יֹצֵ֥א צָבָ֖א בְּיִשְׂרָאֵ֑ל תִּפְקְד֥וּ אֹתָ֛ם לְצִבְאֹתָ֖ם אַתָּ֥ה וְאַהֲרֹֽן:  מ  מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כָּל-יֹצֵ֥א צָבָ֖א בְּיִשְׂרָאֵ֑ל תִּפְקְד֥וּ אֹתָ֛ם לְצִבְאֹתָ֖ם אַתָּ֥ה וְאַהֲרֹֽן:  ת  מִבַּר עַשְׂרִין שְׁנִין וּלְעֵלָא כָּל נָפֵק חֵילָא בְּיִשְׂרָאֵל תִּמְנוּן יָתְהוֹן לְחֵילֵיהוֹן אַתְּ וְאַהֲרֹן:

{ד} וְאִתְּכֶ֣ם יִהְי֔וּ אִ֥ישׁ אִ֖ישׁ לַמַּטֶּ֑ה אִ֛ישׁ רֹ֥אשׁ לְבֵית-אֲבֹתָ֖יו הֽוּא:  מ  וְאִתְּכֶ֣ם יִהְי֔וּ אִ֥ישׁ אִ֖ישׁ לַמַּטֶּ֑ה אִ֛ישׁ רֹ֥אשׁ לְבֵית-אֲבֹתָ֖יו הֽוּא:  ת  וְעִמְכוֹן יְהוֹן גַבְרָא גַבְרָא לְשִׁבְטָא גְבַר רֵישׁ לְבֵית אֲבָהָתוֹהִי הוּא:

{ה} וְאֵ֙לֶּה֙ שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁ֥ר יַֽעַמְד֖וּ אִתְּכֶ֑ם לִרְאוּבֵ֕ן אֱלִיצ֖וּר בֶּן-שְׁדֵיאֽוּר:  מ  וְאֵ֙לֶּה֙ שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁ֥ר יַֽעַמְד֖וּ אִתְּכֶ֑ם לִרְאוּבֵ֕ן אֱלִיצ֖וּר בֶּן-שְׁדֵיאֽוּר:  ת  וְאִלֵין שְׁמָהַת גֻבְרַיָא דִי יְקוּמוּן עִמְכוֹן לִרְאוּבֵן אֶלִיצוּר בַּר שְׁדֵיאוּר:

{ו} לְשִׁמְע֕וֹן שְׁלֻמִיאֵ֖ל בֶּן-צוּרִֽישַׁדָּֽי:  מ  לְשִׁמְע֕וֹן שְׁלֻמִיאֵ֖ל בֶּן-צוּרִֽישַׁדָּֽי:  ת  לְשִׁמְעוֹן שְׁלֻמִיאֵל בַּר צוּרִישַׁדָי:

{ז} לִֽיהוּדָ֕ה נַחְשׁ֖וֹן בֶּן-עַמִּינָדָֽב:  מ  לִֽיהוּדָ֕ה נַחְשׁ֖וֹן בֶּן-עַמִּינָדָֽב:  ת  לִיהוּדָה נַחְשׁוֹן בַּר עַמִינָדָב:

{ח} לְיִ֨שָּׂשכָ֔ר נְתַנְאֵ֖ל בֶּן-צוּעָֽר:  מ  לְיִ֨שָּׂשכָ֔ר נְתַנְאֵ֖ל בֶּן-צוּעָֽר:  ת  לְיִשָׂשׂכָר נְתַנְאֵל בַּר צוּעָר:

{ט} לִזְבוּלֻ֕ן אֱלִיאָ֖ב בֶּן-חֵלֹֽן:  מ  לִזְבוּלֻ֕ן אֱלִיאָ֖ב בֶּן-חֵלֹֽן:  ת  לִזְבוּלֻן אֱלִיאָב בַּר חֵלֹן:

{י} לִבְנֵ֣י יוֹסֵ֔ף לְאֶפְרַ֕יִם אֱלִישָׁמָ֖ע בֶּן-עַמִּיה֑וּד לִמְנַשֶּׁ֕ה גַּמְלִיאֵ֖ל בֶּן-פְּדָהצֽוּר:  מ  לִבְנֵ֣י יוֹסֵ֔ף לְאֶפְרַ֕יִם אֱלִישָׁמָ֖ע בֶּן-עַמִּיה֑וּד לִמְנַשֶּׁ֕ה גַּמְלִיאֵ֖ל בֶּן-פְּדָהצֽוּר:  ת  לִבְנֵי יוֹסֵף לְאֶפְרַיִם אֱלִישָׁמָע בַּר עַמִיהוּד לִמְנַשֶׁה גַמְלִיאֵל בַּר פְּדָהצוּר:

{יא} לְבִ֨נְיָמִ֔ן אֲבִידָ֖ן בֶּן-גִּדְעֹנִֽי:  מ  לְבִ֨נְיָמִ֔ן אֲבִידָ֖ן בֶּן-גִּדְעֹנִֽי:  ת  לְבִנְיָמִן אֲבִידָן בַּר גִדְעֹנִי:

{יב} לְדָ֕ן אֲחִיעֶ֖זֶר בֶּן-עַמִּֽישַׁדָּֽי:  מ  לְדָ֕ן אֲחִיעֶ֖זֶר בֶּן-עַמִּֽישַׁדָּֽי:  ת  לְדָן אֲחִיעֶזֶר בַּר עַמִישַׁדָי:

{יג} לְאָשֵׁ֕ר פַּגְעִיאֵ֖ל בֶּן-עָכְרָֽן:  מ  לְאָשֵׁ֕ר פַּגְעִיאֵ֖ל בֶּן-עָכְרָֽן:  ת  לְאָשֵׁר פַּגְעִיאֵל בַּר עָכְרָן:

{יד} לְגָ֕ד אֶלְיָסָ֖ף בֶּן-דְּעוּאֵֽל:  מ  לְגָ֕ד אֶלְיָסָ֖ף בֶּן-דְּעוּאֵֽל:  ת  לְגָד אֶלְיָסָף בַּר דְעוּאֵל:

{טו} לְנַ֨פְתָּלִ֔י אֲחִירַ֖ע בֶּן-עֵינָֽן:  מ  לְנַ֨פְתָּלִ֔י אֲחִירַ֖ע בֶּן-עֵינָֽן:  ת  לְנַפְתָּלִי אֲחִירַע בַּר עֵינָן:

{טז} אֵ֚לֶּה (קריאי) קְרוּאֵ֣י הָעֵדָ֔ה נְשִׂיאֵ֖י מַטּ֣וֹת אֲבוֹתָ֑ם רָאשֵׁ֛י אַלְפֵ֥י יִשְׂרָאֵ֖ל הֵֽם:  מ  אֵ֚לֶּה (קריאי) קְרוּאֵ֣י הָעֵדָ֔ה נְשִׂיאֵ֖י מַטּ֣וֹת אֲבוֹתָ֑ם רָאשֵׁ֛י אַלְפֵ֥י יִשְׂרָאֵ֖ל הֵֽם:  ת  אִלֵין מְעַרְעֵי כְנִשְׁתָּא רַבְרְבֵי שִׁבְטֵי אֲבָהַתְהוֹן רֵישֵׁי אַלְפַיָא דְיִשְׂרָאֵל אִנוּן:

{יז} וַיִּקַּ֥ח מֹשֶׁ֖ה וְאַהֲרֹ֑ן אֵ֚ת הָאֲנָשִׁ֣ים הָאֵ֔לֶּה אֲשֶׁ֥ר נִקְּב֖וּ בְּשֵׁמֽוֹת:  מ  וַיִּקַּ֥ח מֹשֶׁ֖ה וְאַהֲרֹ֑ן אֵ֚ת הָאֲנָשִׁ֣ים הָאֵ֔לֶּה אֲשֶׁ֥ר נִקְּב֖וּ בְּשֵׁמֽוֹת:  ת  וּדְבַר משֶׁה וְאַהֲרֹן יָת גֻבְרַיָא הָאִלֵין דִי אִתְפָּרָשׁוּ בִּשְׁמָהָן:

{יח} וְאֵ֨ת כָּל-הָעֵדָ֜ה הִקְהִ֗ילוּ בְּאֶחָד֙ לַחֹ֣דֶשׁ הַשֵּׁנִ֔י וַיִּתְיַֽלְד֥וּ עַל-מִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֗וֹת מִבֶּ֨ן עֶשְׂרִ֥ים שָׁנָ֛ה וָמַ֖עְלָה לְגֻלְגְּלֹתָֽם:  מ  וְאֵ֨ת כָּל-הָעֵדָ֜ה הִקְהִ֗ילוּ בְּאֶחָד֙ לַחֹ֣דֶשׁ הַשֵּׁנִ֔י וַיִּתְיַֽלְד֥וּ עַל-מִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֗וֹת מִבֶּ֨ן עֶשְׂרִ֥ים שָׁנָ֛ה וָמַ֖עְלָה לְגֻלְגְּלֹתָֽם:  ת  וְיָת כָּל כְּנִשְׁתָּא אַכְנָשׁוּ בְּחַד לְיַרְחָא תִנְיָנָא וְאִתְיַחֲסוּ עַל זַרְעֲיַתְהוֹן לְבֵית אֲבָהַתְהוֹן בְּמִנְיַן שְׁמָהָן מִבַּר עַשְׂרִין שְׁנִין וּלְעֵלָא לְגֻלְגְלַתְהוֹן:

{יט} כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת-מֹשֶׁ֑ה וַֽיִּפְקְדֵ֖ם בְּמִדְבַּ֥ר סִינָֽי:  מ  כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת-מֹשֶׁ֑ה וַֽיִּפְקְדֵ֖ם בְּמִדְבַּ֥ר סִינָֽי:  ת  כְּמָא דִי פַקֵד יְיָ יָת משֶׁה וּמְנָנוּן בְּמַדְבְּרָא דְסִינָי: (ס)

{כ}  שני  וַיִּהְי֤וּ בְנֵֽי-רְאוּבֵן֙ בְּכֹ֣ר יִשְׂרָאֵ֔ל תּוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֤ר שֵׁמוֹת֙ לְגֻלְגְּלֹתָ֔ם כָּל-זָכָ֗ר מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא:  מ  וַיִּהְי֤וּ בְנֵֽי-רְאוּבֵן֙ בְּכֹ֣ר יִשְׂרָאֵ֔ל תּוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֤ר שֵׁמוֹת֙ לְגֻלְגְּלֹתָ֔ם כָּל-זָכָ֗ר מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא:  ת  וַהֲווֹ בְנֵי רְאוּבֵן בּוּכְרָא דְיִשְׂרָאֵל תּוּלְדַתְהוֹן לְזַרְעֲיַתְהוֹן לְבֵית אֲבָהַתְהוֹן בְּמִנְיַן שְׁמָהָן לְגֻלְגְלַתְהוֹן כָּלּ דְכוּרָא מִבַּר עַשְׂרִין שְׁנִין וּלְעֵלָא כֹּל נָפֵק חֵילָא:

{כא} פְּקֻדֵיהֶ֖ם לְמַטֵּ֣ה רְאוּבֵ֑ן שִׁשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת:  מ  פְּקֻדֵיהֶ֖ם לְמַטֵּ֣ה רְאוּבֵ֑ן שִׁשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת:  ת  מִנְיָנֵיהוֹן לְשִׁבְטָא דִרְאוּבֵן אַרְבְּעִין וְשִׁתָּא אַלְפִין וַחֲמֵשׁ מְאָה: (פ)

{מט} אַ֣ךְ אֶת-מַטֵּ֤ה לֵוִי֙ לֹ֣א תִפְקֹ֔ד וְאֶת-רֹאשָׁ֖ם לֹ֣א תִשָּׂ֑א בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל:  מ  אַ֣ךְ אֶת-מַטֵּ֤ה לֵוִי֙ לֹ֣א תִפְקֹ֔ד וְאֶת-רֹאשָׁ֖ם לֹ֣א תִשָּׂ֑א בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל:  ת  בְּרַם יָת שִׁבְטָא דְלֵוִי לָא תִמְנֵי וְיָת חֻשְׁבַּנְהוֹן לָא תְקַבֵּל בְּגוֹ בְּנֵי יִשְׂרָאֵל:

{נ} וְאַתָּ֡ה הַפְקֵ֣ד אֶת-הַלְוִיִּם֩ עַל-מִשְׁכַּ֨ן הָעֵדֻ֜ת וְעַ֣ל כָּל-כֵּלָיו֮ וְעַ֣ל כָּל-אֲשֶׁר-לוֹ֒ הֵ֜מָּה יִשְׂא֤וּ אֶת-הַמִּשְׁכָּן֙ וְאֶת-כָּל-כֵּלָ֔יו וְהֵ֖ם יְשָׁרְתֻ֑הוּ וְסָבִ֥יב לַמִּשְׁכָּ֖ן יַחֲנֽוּ:  מ  וְאַתָּ֡ה הַפְקֵ֣ד אֶת-הַלְוִיִּם֩ עַל-מִשְׁכַּ֨ן הָעֵדֻ֜ת וְעַ֣ל כָּל-כֵּלָיו֮ וְעַ֣ל כָּל-אֲשֶׁר-לוֹ֒ הֵ֜מָּה יִשְׂא֤וּ אֶת-הַמִּשְׁכָּן֙ וְאֶת-כָּל-כֵּלָ֔יו וְהֵ֖ם יְשָׁרְתֻ֑הוּ וְסָבִ֥יב לַמִּשְׁכָּ֖ן יַחֲנֽוּ:  ת  וְאַתְּ מַנִי יָת לֵוָאֵי עַל מַשְׁכְּנָא דְסַהֲדוּתָא וְעַל כָּל מָנוֹהִי וְעַל כָּל דִּי לֵהּ אִנּוּן יִטְלוּן יָת מַשְׁכְּנָא וְיָת כָּל מָנוֹהִי וְאִנוּן יְשַׁמְשֻׁנֵהּ וּסְחוֹר סְחוֹר לְמַשְׁכְּנָא יִשְׁרוּן:

{נא} וּבִנְסֹ֣עַ הַמִּשְׁכָּ֗ן יוֹרִ֤ידוּ אֹתוֹ֙ הַלְוִיִּ֔ם וּבַחֲנֹת֙ הַמִּשְׁכָּ֔ן יָקִ֥ימוּ אֹת֖וֹ הַלְוִיִּ֑ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת:  מ  וּבִנְסֹ֣עַ הַמִּשְׁכָּ֗ן יוֹרִ֤ידוּ אֹתוֹ֙ הַלְוִיִּ֔ם וּבַחֲנֹת֙ הַמִּשְׁכָּ֔ן יָקִ֥ימוּ אֹת֖וֹ הַלְוִיִּ֑ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת:  ת  וּבְמֵטַל מַשְׁכְּנָא יְפַרְקוּן יָתֵהּ לֵוָאֵי וּבְמִשְׁרֵי מַשְׁכְּנָא יְקִימוּן יָתֵהּ לֵוָאֵי וְחִלוֹנַי דְיִקְרַב יִתְקְטֵל:

{נב} וְחָנ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל אִ֧ישׁ עַֽל-מַחֲנֵ֛הוּ וְאִ֥ישׁ עַל-דִּגְל֖וֹ לְצִבְאֹתָֽם:  מ  וְחָנ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל אִ֧ישׁ עַֽל-מַחֲנֵ֛הוּ וְאִ֥ישׁ עַל-דִּגְל֖וֹ לְצִבְאֹתָֽם:  ת  וְיִשְׁרוּן בְּנֵי יִשְׂרָאֵל גְבַר עַל מַשְׁרוֹהִי וּגְבַר עַל טִּקְסֵהּ לְחֵילֵיהוֹן:

{נג} וְהַלְוִיִּ֞ם יַחֲנ֤וּ סָבִיב֙ לְמִשְׁכַּ֣ן הָעֵדֻ֔ת וְלֹֽא-יִהְיֶ֣ה קֶ֔צֶף עַל-עֲדַ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל וְשָׁמְרוּ֙ הַלְוִיִּ֔ם אֶת-מִשְׁמֶ֖רֶת מִשְׁכַּ֥ן הָעֵדֽוּת:  מ  וְהַלְוִיִּ֞ם יַחֲנ֤וּ סָבִיב֙ לְמִשְׁכַּ֣ן הָעֵדֻ֔ת וְלֹֽא-יִהְיֶ֣ה קֶ֔צֶף עַל-עֲדַ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל וְשָׁמְרוּ֙ הַלְוִיִּ֔ם אֶת-מִשְׁמֶ֖רֶת מִשְׁכַּ֥ן הָעֵדֽוּת:  ת  וְלֵוָאֵי יִשְׁרוּן סְחוֹר סְחוֹר לְמַשְׁכְּנָא דְסַהַדוּתָא וְלָא יְהֵי רוּגְזָא עַל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל וְיִטְרוּן לֵוָאֵי יָת מַטְרַת מַשְׁכְּנָא דְסַהֲדוּתָא:

{נד} וַֽיַּעֲשׂ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל כְּ֠כֹל אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֶת-מֹשֶׁ֖ה כֵּ֥ן עָשֽׂוּ:  מ  וַֽיַּעֲשׂ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל כְּ֠כֹל אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֶת-מֹשֶׁ֖ה כֵּ֥ן עָשֽׂוּ:  ת  וַעֲבָדוּ בְּנֵי יִשְׂרָאֵל כְּכֹל דִּי פַּקֵיד יְיָ יָת משֶׁה כֵּן עֲבָדוּ: (פ)




במדבר פרק-ב

{א}  שלישי  וַיְדַבֵּ֣ר יְהוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  מ  וַיְדַבֵּ֣ר יְהוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה וְעִם אַהֲרֹן לְמֵימָר:

{ב} אִ֣ישׁ עַל-דִּגְל֤וֹ בְאֹתֹת֙ לְבֵ֣ית אֲבֹתָ֔ם יַחֲנ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל מִנֶּ֕גֶד סָבִ֥יב לְאֹֽהֶל-מוֹעֵ֖ד יַחֲנֽוּ:  מ  אִ֣ישׁ עַל-דִּגְל֤וֹ בְאֹתֹת֙ לְבֵ֣ית אֲבֹתָ֔ם יַחֲנ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל מִנֶּ֕גֶד סָבִ֥יב לְאֹֽהֶל-מוֹעֵ֖ד יַחֲנֽוּ:  ת  גְבַר עַל טִּקְסֵהּ בְּאַתְוָן לְבֵית אֲבָהַתְהוֹן יִשְׁרוּן בְּנֵי יִשְׂרָאֵל מִלָקֳבֵל סְחוֹר סְחוֹר לְמַשְׁכַּן זִמְנָא יִשְׁרוּן:

{ג} וְהַחֹנִים֙ קֵ֣דְמָה מִזְרָ֔חָה דֶּ֛גֶל מַחֲנֵ֥ה יְהוּדָ֖ה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י יְהוּדָ֔ה נַחְשׁ֖וֹן בֶּן-עַמִּינָדָֽב:  מ  וְהַחֹנִים֙ קֵ֣דְמָה מִזְרָ֔חָה דֶּ֛גֶל מַחֲנֵ֥ה יְהוּדָ֖ה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י יְהוּדָ֔ה נַחְשׁ֖וֹן בֶּן-עַמִּינָדָֽב:  ת  וְדִי שְׁרַן קִדוּמָא מַדִינְחָא טֵקַס מַשְׁרִית יְהוּדָה לְחֵילֵיהוֹן וְרַבָּא לִבְנֵי יְהוּדָה נַחְשׁוֹן בַּר עַמִינָדָב:

{ד} וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁבְעִ֛ים אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת:  מ  וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁבְעִ֛ים אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת:  ת  וְחֵילֵיהּ וּמִנְיָנֵיהוֹן שַׁבְעִין וְאַרְבְּעָא אַלְפִין וְשִׁית מְאָה:

{ה} וְהַחֹנִ֥ים עָלָ֖יו מַטֵּ֣ה יִשָּׂשכָ֑ר וְנָשִׂיא֙ לִבְנֵ֣י יִשָּׂשכָ֔ר נְתַנְאֵ֖ל בֶּן-צוּעָֽר:  מ  וְהַחֹנִ֥ים עָלָ֖יו מַטֵּ֣ה יִשָּׂשכָ֑ר וְנָשִׂיא֙ לִבְנֵ֣י יִשָּׂשכָ֔ר נְתַנְאֵ֖ל בֶּן-צוּעָֽר:  ת  וְדִי שְׁרַן סְמִיכִין עֲלוֹהִי שִׁבְטָא דְיִשָׂשכָר וְרַבָּא לִבְנֵי יִשָׂשכָר נְתַנְאֵל בַּר צוּעָר:

{ו} וּצְבָא֖וֹ וּפְקֻדָ֑יו אַרְבָּעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  מ  וּצְבָא֖וֹ וּפְקֻדָ֑יו אַרְבָּעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  ת  וְחֵילֵהּ וּמִנְיָנוֹהִי חַמְשִׁין וְאַרְבְּעָא אַלְפִין וְאַרְבַּע מְאָה:

{ז} מַטֵּ֖ה זְבוּלֻ֑ן וְנָשִׂיא֙ לִבְנֵ֣י זְבוּלֻ֔ן אֱלִיאָ֖ב בֶּן-חֵלֹֽן:  מ  מַטֵּ֖ה זְבוּלֻ֑ן וְנָשִׂיא֙ לִבְנֵ֣י זְבוּלֻ֔ן אֱלִיאָ֖ב בֶּן-חֵלֹֽן:  ת  שִׁבְטָא דִזְבוּלֻן וְרַבָּא לִבְנֵי זְבוּלֻן אֱלִיאָב בַּר חֵלוֹן:

{ח} וּצְבָא֖וֹ וּפְקֻדָ֑יו שִׁבְעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  מ  וּצְבָא֖וֹ וּפְקֻדָ֑יו שִׁבְעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  ת  וְחֵילֵהּ וּמִנְיָנוֹהִי חַמְשִׁין וְשַׁבְעָא אַלְפִין וְאַרְבַּע מְאָה:

{ט} כָּֽל-הַפְּקֻדִ֞ים לְמַחֲנֵ֣ה יְהוּדָ֗ה מְאַ֨ת אֶ֜לֶף וּשְׁמֹנִ֥ים אֶ֛לֶף וְשֵֽׁשֶׁת-אֲלָפִ֥ים וְאַרְבַּע-מֵא֖וֹת לְצִבְאֹתָ֑ם רִאשֹׁנָ֖ה יִסָּֽעוּ:  מ  כָּֽל-הַפְּקֻדִ֞ים לְמַחֲנֵ֣ה יְהוּדָ֗ה מְאַ֨ת אֶ֜לֶף וּשְׁמֹנִ֥ים אֶ֛לֶף וְשֵֽׁשֶׁת-אֲלָפִ֥ים וְאַרְבַּע-מֵא֖וֹת לְצִבְאֹתָ֑ם רִאשֹׁנָ֖ה יִסָּֽעוּ:  ת  כָּל מִנְיָנַיָא לְמַשְׁרִית יְהוּדָה מְאָה וּתְמָנָן וְשִׁתָּא אַלְפִין וְאַרְבַּע מְאָה לְחֵילֵיהוֹן בְּקַדְמֵיתָא נַטְלִין: (ס)

{י} דֶּ֣גֶל מַחֲנֵ֧ה רְאוּבֵ֛ן תֵּימָ֖נָה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י רְאוּבֵ֔ן אֱלִיצ֖וּר בֶּן-שְׁדֵיאֽוּר:  מ  דֶּ֣גֶל מַחֲנֵ֧ה רְאוּבֵ֛ן תֵּימָ֖נָה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י רְאוּבֵ֔ן אֱלִיצ֖וּר בֶּן-שְׁדֵיאֽוּר:  ת  טֵקַס מַשְׁרִית רְאוּבֵן דָרוֹמָא לְחֵילֵיהוֹן וְרַבָּא לִבְנֵי רְאוּבֵן אֱלִיצוּר בַּר שְׁדֵיאוּר:

{יא} וּצְבָא֖וֹ וּפְקֻדָ֑יו שִׁשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת:  מ  וּצְבָא֖וֹ וּפְקֻדָ֑יו שִׁשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת:  ת  וְחֵילֵהּ וּמִנְיָנוֹהִי אַרְבְּעִין וְשִׁתָּא אַלְפִין וַחֲמֵשׁ מְאָה:

{יב} וְהַחוֹנִ֥ם עָלָ֖יו מַטֵּ֣ה שִׁמְע֑וֹן וְנָשִׂיא֙ לִבְנֵ֣י שִׁמְע֔וֹן שְׁלֻמִיאֵ֖ל בֶּן-צוּרִֽי-שַׁדָּֽי:  מ  וְהַחוֹנִ֥ם עָלָ֖יו מַטֵּ֣ה שִׁמְע֑וֹן וְנָשִׂיא֙ לִבְנֵ֣י שִׁמְע֔וֹן שְׁלֻמִיאֵ֖ל בֶּן-צוּרִֽי-שַׁדָּֽי:  ת  וְדִי שְׁרַן סְמִיכִין עֲלוֹהִי שִׁבְטָא דְשִׁמְעוֹן וְרַבָּא לִבְנֵי שִׁמְעוֹן שְׁלֻמִיאֵל בַּר צוּרִישַׁדָי:

{יג} וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם תִּשְׁעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וּשְׁלֹ֥שׁ מֵאֽוֹת:  מ  וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם תִּשְׁעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וּשְׁלֹ֥שׁ מֵאֽוֹת:  ת  וְחֵילֵהּ וּמִנְיָנֵיהוֹן חַמְשִׁין וּתְשַׁע אַלְפִין וּתְלַת מְאָה:

{יד} וְמַטֵּ֖ה גָּ֑ד וְנָשִׂיא֙ לִבְנֵ֣י גָ֔ד אֶלְיָסָ֖ף בֶּן-רְעוּאֵֽל:  מ  וְמַטֵּ֖ה גָּ֑ד וְנָשִׂיא֙ לִבְנֵ֣י גָ֔ד אֶלְיָסָ֖ף בֶּן-רְעוּאֵֽל:  ת  וְשִׁבְטָא דְגָד וְרַבָּא לִבְנֵי גָד אֶלְיָסָף בַּר רְעוּאֵל:

{טו} וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם חֲמִשָּׁ֤ה וְאַרְבָּעִים֙ אֶ֔לֶף וְשֵׁ֥שׁ מֵא֖וֹת וַחֲמִשִּֽׁים:  מ  וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם חֲמִשָּׁ֤ה וְאַרְבָּעִים֙ אֶ֔לֶף וְשֵׁ֥שׁ מֵא֖וֹת וַחֲמִשִּֽׁים:  ת  וְחֵילֵהּ וּמִנְיָנֵיהוֹן אַרְבְּעִין וְחַמְשָׁא אַלְפִין וְשִׁית מְאָה וְחַמְשִׁין:

{טז} כָּֽל-הַפְּקֻדִ֞ים לְמַחֲנֵ֣ה רְאוּבֵ֗ן מְאַ֨ת אֶ֜לֶף וְאֶחָ֨ד וַחֲמִשִּׁ֥ים אֶ֛לֶף וְאַרְבַּע-מֵא֥וֹת וַחֲמִשִּׁ֖ים לְצִבְאֹתָ֑ם וּשְׁנִיִּ֖ם יִסָּֽעוּ:  מ  כָּֽל-הַפְּקֻדִ֞ים לְמַחֲנֵ֣ה רְאוּבֵ֗ן מְאַ֨ת אֶ֜לֶף וְאֶחָ֨ד וַחֲמִשִּׁ֥ים אֶ֛לֶף וְאַרְבַּע-מֵא֥וֹת וַחֲמִשִּׁ֖ים לְצִבְאֹתָ֑ם וּשְׁנִיִּ֖ם יִסָּֽעוּ:  ת  כָּל מִנְיָנַיָא לְמַשְׁרִית רְאוּבֵן מְאָה וְחַמְשִׁין וְחַד אַלְפִין וְאַרְבַּע מְאָה וְחַמְשִׁין לְחֵילֵיהוֹן בְּתִנְיֵתָא נַטְלִין: (ס)

{יז} וְנָסַ֧ע אֹֽהֶל-מוֹעֵ֛ד מַחֲנֵ֥ה הַלְוִיִּ֖ם בְּת֣וֹךְ הַֽמַּחֲנֹ֑ת כַּאֲשֶׁ֤ר יַחֲנוּ֙ כֵּ֣ן יִסָּ֔עוּ אִ֥ישׁ עַל-יָד֖וֹ לְדִגְלֵיהֶֽם:  מ  וְנָסַ֧ע אֹֽהֶל-מוֹעֵ֛ד מַחֲנֵ֥ה הַלְוִיִּ֖ם בְּת֣וֹךְ הַֽמַּחֲנֹ֑ת כַּאֲשֶׁ֤ר יַחֲנוּ֙ כֵּ֣ן יִסָּ֔עוּ אִ֥ישׁ עַל-יָד֖וֹ לְדִגְלֵיהֶֽם:  ת  וְנַטֵל מַשְׁכַּן זִמְנָא מַשְׁרִית לֵוָאֵי בְּגוֹ מַשִּׁרְיָתָא כְּמָא דְשָׁרַן כֵּן נָטְלִין גְבַר עַל אַתְרֵהּ לְטִקְסֵיהוֹן: (ס)

{יח} דֶּ֣גֶל מַחֲנֵ֥ה אֶפְרַ֛יִם לְצִבְאֹתָ֖ם יָ֑מָּה וְנָשִׂיא֙ לִבְנֵ֣י אֶפְרַ֔יִם אֱלִישָׁמָ֖ע בֶּן-עַמִּיהֽוּד:  מ  דֶּ֣גֶל מַחֲנֵ֥ה אֶפְרַ֛יִם לְצִבְאֹתָ֖ם יָ֑מָּה וְנָשִׂיא֙ לִבְנֵ֣י אֶפְרַ֔יִם אֱלִישָׁמָ֖ע בֶּן-עַמִּיהֽוּד:  ת  טֵּקַס מַשְׁרִית אֶפְרַיִם לְחֵילֵיהוֹן מַעַרְבָא וְרַבָּא לִבְנֵי אֶפְרַיִם אֱלִישָׁמָע בַּר עַמִיהוּד:

{יט} וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם אַרְבָּעִ֥ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת:  מ  וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם אַרְבָּעִ֥ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת:  ת  וְחֵילֵהּ וּמִנְיָנֵיהוֹן אַרְבְּעִין אַלְפִין וַחֲמֵשׁ מְאָה:

{כ} וְעָלָ֖יו מַטֵּ֣ה מְנַשֶּׁ֑ה וְנָשִׂיא֙ לִבְנֵ֣י מְנַשֶּׁ֔ה גַּמְלִיאֵ֖ל בֶּן-פְּדָהצֽוּר:  מ  וְעָלָ֖יו מַטֵּ֣ה מְנַשֶּׁ֑ה וְנָשִׂיא֙ לִבְנֵ֣י מְנַשֶּׁ֔ה גַּמְלִיאֵ֖ל בֶּן-פְּדָהצֽוּר:  ת  וְדִסְמִיכִין עֲלוֹהִי שִׁבְטָא דִמְנַשֶׁה וְרַבָּא לִבְנֵי מְנַשֶׁה גַמְלִיאֵל בַּר פְדָהצוּר:

{כא} וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם שְׁנַ֧יִם וּשְׁלֹשִׁ֛ים אֶ֖לֶף וּמָאתָֽיִם:  מ  וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם שְׁנַ֧יִם וּשְׁלֹשִׁ֛ים אֶ֖לֶף וּמָאתָֽיִם:  ת  וְחֵילֵהּ וּמִנְיָנֵיהוֹן תְּלָתִין וּתְרֵין אַלְפִין וּמָאתָן:

{כב} וּמַטֵּ֖ה בִּנְיָמִ֑ן וְנָשִׂיא֙ לִבְנֵ֣י בִנְיָמִ֔ן אֲבִידָ֖ן בֶּן-גִּדְעֹנִֽי:  מ  וּמַטֵּ֖ה בִּנְיָמִ֑ן וְנָשִׂיא֙ לִבְנֵ֣י בִנְיָמִ֔ן אֲבִידָ֖ן בֶּן-גִּדְעֹנִֽי:  ת  וְשִׁבְטָא דְבִנְיָמִן וְרַבָּא לִבְנֵי בִנְיָמִן אֲבִידָן בַּר גִדְעוֹנִי:

{כג} וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם חֲמִשָּׁ֧ה וּשְׁלֹשִׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  מ  וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם חֲמִשָּׁ֧ה וּשְׁלֹשִׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  ת  וְחֵילֵהּ וּמִנְיָנֵיהוֹן תְּלָתִין וְחַמְשָׁא אַלְפִין וְאַרְבַּע מְאָה:

{כד} כָּֽל-הַפְּקֻדִ֞ים לְמַחֲנֵ֣ה אֶפְרַ֗יִם מְאַ֥ת אֶ֛לֶף וּשְׁמֹֽנַת-אֲלָפִ֥ים וּמֵאָ֖ה לְצִבְאֹתָ֑ם וּשְׁלִשִׁ֖ים יִסָּֽעוּ:  מ  כָּֽל-הַפְּקֻדִ֞ים לְמַחֲנֵ֣ה אֶפְרַ֗יִם מְאַ֥ת אֶ֛לֶף וּשְׁמֹֽנַת-אֲלָפִ֥ים וּמֵאָ֖ה לְצִבְאֹתָ֑ם וּשְׁלִשִׁ֖ים יִסָּֽעוּ:  ת  כָּל מִנְיָנַיָא לְמַשְׁרִית אֶפְרַיִם מְאָה וְתַמְנֵי אַלְפִין וּמְאָה לְחֵילֵיהוֹן בִּתְלִיתֵתָא נָטְלִין: (ס)

{כה} דֶּ֣גֶל מַחֲנֵ֥ה דָ֛ן צָפֹ֖נָה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י דָ֔ן אֲחִיעֶ֖זֶר בֶּן-עַמִּֽישַׁדָּֽי:  מ  דֶּ֣גֶל מַחֲנֵ֥ה דָ֛ן צָפֹ֖נָה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י דָ֔ן אֲחִיעֶ֖זֶר בֶּן-עַמִּֽישַׁדָּֽי:  ת  טֵקַס מַשְׁרִית דָן צִפּוּנָא לְחֵילֵיהוֹן וְרַבָּא לִבְנֵי דָן אֲחִיעֶזֶר בַּר עַמִישַׁדָי:

{כו} וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם שְׁנַ֧יִם וְשִׁשִּׁ֛ים אֶ֖לֶף וּשְׁבַ֥ע מֵאֽוֹת:  מ  וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם שְׁנַ֧יִם וְשִׁשִּׁ֛ים אֶ֖לֶף וּשְׁבַ֥ע מֵאֽוֹת:  ת  וְחֵילֵהּ וּמִנְיָנֵיהוֹן שִׁתִּין וּתְרֵין אַלְפִין וּשְׁבַע מְאָה:

{כז} וְהַחֹנִ֥ים עָלָ֖יו מַטֵּ֣ה אָשֵׁ֑ר וְנָשִׂיא֙ לִבְנֵ֣י אָשֵׁ֔ר פַּגְעִיאֵ֖ל בֶּן-עָכְרָֽן:  מ  וְהַחֹנִ֥ים עָלָ֖יו מַטֵּ֣ה אָשֵׁ֑ר וְנָשִׂיא֙ לִבְנֵ֣י אָשֵׁ֔ר פַּגְעִיאֵ֖ל בֶּן-עָכְרָֽן:  ת  וְדִי שְׁרַן םְמִכִין עֲלוֹהִי שִׁבְטָא דְאָשֵׁר וְרַבָּא לִבְנֵי אָשֵׁר פַּגְעִיאֵל בַּר עָכְרָן:

{כח} וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם אֶחָ֧ד וְאַרְבָּעִ֛ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת:  מ  וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם אֶחָ֧ד וְאַרְבָּעִ֛ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת:  ת  וְחֵילֵהּ וּמִנְיָנֵיהוֹן אַרְבְּעִין וְחַד אַלְפִין וַחֲמֵשׁ מְאָה:

{כט} וּמַטֵּ֖ה נַפְתָּלִ֑י וְנָשִׂיא֙ לִבְנֵ֣י נַפְתָּלִ֔י אֲחִירַ֖ע בֶּן-עֵינָֽן:  מ  וּמַטֵּ֖ה נַפְתָּלִ֑י וְנָשִׂיא֙ לִבְנֵ֣י נַפְתָּלִ֔י אֲחִירַ֖ע בֶּן-עֵינָֽן:  ת  וְשִׁבְטָא דְנַפְתָּלִי וְרַבָּא לִבְנֵי נַפְתָּלִי אֲחִירַע בַּר עֵינָן:

{ל} וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם שְׁלֹשָׁ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  מ  וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם שְׁלֹשָׁ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  ת  וְחֵילֵהּ וּמִנְיָנֵיהוֹן חַמְשִׁין וּתְלָתָא אַלְפִין וְאַרְבַּע מְאָה:

{לא} כָּל-הַפְּקֻדִים֙ לְמַ֣חֲנֵה דָ֔ן מְאַ֣ת אֶ֗לֶף וְשִׁבְעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְשֵׁ֣שׁ מֵא֑וֹת לָאַחֲרֹנָ֥ה יִסְע֖וּ לְדִגְלֵיהֶֽם:  מ  כָּל-הַפְּקֻדִים֙ לְמַ֣חֲנֵה דָ֔ן מְאַ֣ת אֶ֗לֶף וְשִׁבְעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְשֵׁ֣שׁ מֵא֑וֹת לָאַחֲרֹנָ֥ה יִסְע֖וּ לְדִגְלֵיהֶֽם:  ת  כָּל מִנְיָנַיָא לְמַשְׁרִית דָן מְאָה וְחַמְשִׁין וְשַׁבְעָא אַלְפִין וְשִׁית מְאָה בְּבַתְרֵתָא נָטְלִין לְטִקְסֵיהוֹן: (פ)

{לב} אֵ֛לֶּה פְּקוּדֵ֥י בְנֵֽי-יִשְׂרָאֵ֖ל לְבֵ֣ית אֲבֹתָ֑ם כָּל-פְּקוּדֵ֤י הַֽמַּחֲנֹת֙ לְצִבְאֹתָ֔ם שֵׁשׁ-מֵא֥וֹת אֶ֙לֶף֙ וּשְׁלֹ֣שֶׁת אֲלָפִ֔ים וַחֲמֵ֥שׁ מֵא֖וֹת וַחֲמִשִּֽׁים:  מ  אֵ֛לֶּה פְּקוּדֵ֥י בְנֵֽי-יִשְׂרָאֵ֖ל לְבֵ֣ית אֲבֹתָ֑ם כָּל-פְּקוּדֵ֤י הַֽמַּחֲנֹת֙ לְצִבְאֹתָ֔ם שֵׁשׁ-מֵא֥וֹת אֶ֙לֶף֙ וּשְׁלֹ֣שֶׁת אֲלָפִ֔ים וַחֲמֵ֥שׁ מֵא֖וֹת וַחֲמִשִּֽׁים:  ת  אִלֵין מִנְיָנֵי בְנֵי יִשְׂרָאֵל לְבֵית אֲבָהַתְהוֹן כָּל מִנְיָנֵי מַשִׁרְיָתָא לְחֵילֵיהוֹן שִׁית מְאָהּ וּתְלָתָא אַלְפִין וַחֲמֵשׁ מְאָה וְחַמְשִׁין:

{לג} וְהַ֨לְוִיִּ֔ם לֹ֣א הָתְפָּקְד֔וּ בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת-מֹשֶֽׁה:  מ  וְהַ֨לְוִיִּ֔ם לֹ֣א הָתְפָּקְד֔וּ בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת-מֹשֶֽׁה:  ת  וְלֵוָאֵי לָא אִתְמְנִיאוּ בְּגוֹ בְּנֵי יִשְׂרָאֵל כְּמָא דְפַקֵיד יְיָ יָת משֶׁה:

{לד} וַֽיַּעֲשׂ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל כְּ֠כֹל אֲשֶׁר-צִוָּ֨ה יְהוָ֜ה אֶת-מֹשֶׁ֗ה כֵּֽן-חָנ֤וּ לְדִגְלֵיהֶם֙ וְכֵ֣ן נָסָ֔עוּ אִ֥ישׁ לְמִשְׁפְּחֹתָ֖יו עַל-בֵּ֥ית אֲבֹתָֽיו:  מ  וַֽיַּעֲשׂ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל כְּ֠כֹל אֲשֶׁר-צִוָּ֨ה יְהוָ֜ה אֶת-מֹשֶׁ֗ה כֵּֽן-חָנ֤וּ לְדִגְלֵיהֶם֙ וְכֵ֣ן נָסָ֔עוּ אִ֥ישׁ לְמִשְׁפְּחֹתָ֖יו עַל-בֵּ֥ית אֲבֹתָֽיו:  ת  וַעֲבָדוּ בְּנֵי יִשְׂרָאֵל כְּכֹל דִי פַקֵיד יְיָ יָת משֶׁה כֵּן שָׁרַן לְטִּקְסֵיהוֹן וְכֵן נָטְלִין גְבַר לְזַרְעִיתֵיהּ עַל בֵּית אֲבָהָתוֹהִי: (פ)




במדבר פרק-ג

{א}  רביעי  וְאֵ֛לֶּה תּוֹלְדֹ֥ת אַהֲרֹ֖ן וּמֹשֶׁ֑ה בְּי֗וֹם דִּבֶּ֧ר יְהוָ֛ה אֶת-מֹשֶׁ֖ה בְּהַ֥ר סִינָֽי:  מ  וְאֵ֛לֶּה תּוֹלְדֹ֥ת אַהֲרֹ֖ן וּמֹשֶׁ֑ה בְּי֗וֹם דִּבֶּ֧ר יְהוָ֛ה אֶת-מֹשֶׁ֖ה בְּהַ֥ר סִינָֽי:  ת  וְאִלֵין תּוּלְדַת אַהֲרֹן וּמשֶׁה בְּיוֹמָא דְמַלִיל יְיָ עִם משֶׁה בְּטוּרָא דְסִינָי:

{ב} וְאֵ֛לֶּה שְׁמ֥וֹת בְּֽנֵי-אַהֲרֹ֖ן הַבְּכ֣וֹר | נָדָ֑ב וַאֲבִיה֕וּא אֶלְעָזָ֖ר וְאִיתָמָֽר:  מ  וְאֵ֛לֶּה שְׁמ֥וֹת בְּֽנֵי-אַהֲרֹ֖ן הַבְּכ֣וֹר | נָדָ֑ב וַאֲבִיה֕וּא אֶלְעָזָ֖ר וְאִיתָמָֽר:  ת  וְאִלֵין שְׁמָהַת בְּנֵי אַהֲרֹן בּוּכְרָא נָדָב וַאֲבִיהוּא אֶלְעָזָר וְאִיתָמָר:

{ג} אֵ֗לֶּה שְׁמוֹת֙ בְּנֵ֣י אַהֲרֹ֔ן הַכֹּהֲנִ֖ים הַמְּשֻׁחִ֑ים אֲשֶׁר-מִלֵּ֥א יָדָ֖ם לְכַהֵֽן:  מ  אֵ֗לֶּה שְׁמוֹת֙ בְּנֵ֣י אַהֲרֹ֔ן הַכֹּהֲנִ֖ים הַמְּשֻׁחִ֑ים אֲשֶׁר-מִלֵּ֥א יָדָ֖ם לְכַהֵֽן:  ת  אִלֵין שְׁמָהַת בְּנֵי אַהֲרֹן כַּהֲנַיָא דְאִתְרַבִּיאוּ דִי אִתְקָרֵב קֻרְבַּנְהוֹן לְשַׁמָשָׁא:

{ד} וַיָּ֣מָת נָדָ֣ב וַאֲבִיה֣וּא לִפְנֵ֣י יְהוָ֡ה בְּֽהַקְרִבָם֩ אֵ֨שׁ זָרָ֜ה לִפְנֵ֤י יְהוָה֙ בְּמִדְבַּ֣ר סִינַ֔י וּבָנִ֖ים לֹא-הָי֣וּ לָהֶ֑ם וַיְכַהֵ֤ן אֶלְעָזָר֙ וְאִ֣יתָמָ֔ר עַל-פְּנֵ֖י אַהֲרֹ֥ן אֲבִיהֶֽם:  מ  וַיָּ֣מָת נָדָ֣ב וַאֲבִיה֣וּא לִפְנֵ֣י יְהוָ֡ה בְּֽהַקְרִבָם֩ אֵ֨שׁ זָרָ֜ה לִפְנֵ֤י יְהוָה֙ בְּמִדְבַּ֣ר סִינַ֔י וּבָנִ֖ים לֹא-הָי֣וּ לָהֶ֑ם וַיְכַהֵ֤ן אֶלְעָזָר֙ וְאִ֣יתָמָ֔ר עַל-פְּנֵ֖י אַהֲרֹ֥ן אֲבִיהֶֽם:  ת  וּמִית נָדָב וַאֲבִיהוּא קֳדָם יְיָ בְּקָרוֹבֵיהוֹן אֶשָׁתָא נוּכְרֵיתָא קֳדָם יְיָ בְּמַדְבְּרָא דְסִינַי וּבְנִין לָא הֲווֹ לְהוֹן וְשַׁמֵּשׁ אֶלְעָזָר וְאִיתָמָר עַל אַפֵּי אַהֲרֹן אֲבוּהוֹן: (פ)

{ה} וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{ו} הַקְרֵב֙ אֶת-מַטֵּ֣ה לֵוִ֔י וְֽהַעֲמַדְתָּ֣ אֹת֔וֹ לִפְנֵ֖י אַהֲרֹ֣ן הַכֹּהֵ֑ן וְשֵׁרְת֖וּ אֹתֽוֹ:  מ  הַקְרֵב֙ אֶת-מַטֵּ֣ה לֵוִ֔י וְֽהַעֲמַדְתָּ֣ אֹת֔וֹ לִפְנֵ֖י אַהֲרֹ֣ן הַכֹּהֵ֑ן וְשֵׁרְת֖וּ אֹתֽוֹ:  ת  קָרֵב יָת שִׁבְטָּא דְלֵוִי וּתְקֵים יָתֵהּ קֳדָם אַהֲרֹן כַּהֲנָא וִישַׁמְשׁוּן יָתֵהּ:

{ז} וְשָׁמְר֣וּ אֶת-מִשְׁמַרְתּ֗וֹ וְאֶת-מִשְׁמֶ֙רֶת֙ כָּל-הָ֣עֵדָ֔ה לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד לַעֲבֹ֖ד אֶת-עֲבֹדַ֥ת הַמִּשְׁכָּֽן:  מ  וְשָׁמְר֣וּ אֶת-מִשְׁמַרְתּ֗וֹ וְאֶת-מִשְׁמֶ֙רֶת֙ כָּל-הָ֣עֵדָ֔ה לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד לַעֲבֹ֖ד אֶת-עֲבֹדַ֥ת הַמִּשְׁכָּֽן:  ת  וְיִטְרוּן יָת מַטַרְתֵּהּ וְיָת מַטְרַת כָּל כְּנִשְׁתָּא קֳדָם מַשְׁכַּן זִמְנָא לְמִפְלַח יָת פָּלְחַן מַשְׁכְּנָא:

{ח} וְשָׁמְר֗וּ אֶֽת-כָּל-כְּלֵי֙ אֹ֣הֶל מוֹעֵ֔ד וְאֶת-מִשְׁמֶ֖רֶת בְּנֵ֣י יִשְׂרָאֵ֑ל לַעֲבֹ֖ד אֶת-עֲבֹדַ֥ת הַמִּשְׁכָּֽן:  מ  וְשָׁמְר֗וּ אֶֽת-כָּל-כְּלֵי֙ אֹ֣הֶל מוֹעֵ֔ד וְאֶת-מִשְׁמֶ֖רֶת בְּנֵ֣י יִשְׂרָאֵ֑ל לַעֲבֹ֖ד אֶת-עֲבֹדַ֥ת הַמִּשְׁכָּֽן:  ת  וְיִטְרוּן יָת כָּל מָאנֵי מַשְׁכַּן זִמְנָא וְיָת מַטְּרַת בְּנֵי יִשְׂרָאֵל לְמִפְלַח יָת פָּלְחַן מַשְׁכְּנָא:

{ט} וְנָתַתָּה֙ אֶת-הַלְוִיִּ֔ם לְאַהֲרֹ֖ן וּלְבָנָ֑יו נְתוּנִ֨ם נְתוּנִ֥ם הֵ֙מָּה֙ ל֔וֹ מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וְנָתַתָּה֙ אֶת-הַלְוִיִּ֔ם לְאַהֲרֹ֖ן וּלְבָנָ֑יו נְתוּנִ֨ם נְתוּנִ֥ם הֵ֙מָּה֙ ל֔וֹ מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וְתִתֵּן יָת לֵוָאֵי לְאַהֲרֹן וְלִבְנוֹהִי מְסִירִין יְהִיבִין אִנוּן לֵהּ מִן בְּנֵי יִשְׂרָאֵל:

{י} וְאֶת-אַהֲרֹ֤ן וְאֶת-בָּנָיו֙ תִּפְקֹ֔ד וְשָׁמְר֖וּ אֶת-כְּהֻנָּתָ֑ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת:  מ  וְאֶת-אַהֲרֹ֤ן וְאֶת-בָּנָיו֙ תִּפְקֹ֔ד וְשָׁמְר֖וּ אֶת-כְּהֻנָּתָ֑ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת:  ת  וְיָת אַהֲרֹן וְיָת בְּנוֹהִי תְּמַנֵי וְיִטְרוּן יָת כְּהֻנַתְהוֹן וְחִלוֹנַי דְיִקְרַב יִתְקְטֵל: (פ)

{יא} וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{יב} וַאֲנִ֞י הִנֵּ֧ה לָקַ֣חְתִּי אֶת-הַלְוִיִּ֗ם מִתּוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל תַּ֧חַת כָּל-בְּכ֛וֹר פֶּ֥טֶר רֶ֖חֶם מִבְּנֵ֣י יִשְׂרָאֵ֑ל וְהָ֥יוּ לִ֖י הַלְוִיִּֽם:  מ  וַאֲנִ֞י הִנֵּ֧ה לָקַ֣חְתִּי אֶת-הַלְוִיִּ֗ם מִתּוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל תַּ֧חַת כָּל-בְּכ֛וֹר פֶּ֥טֶר רֶ֖חֶם מִבְּנֵ֣י יִשְׂרָאֵ֑ל וְהָ֥יוּ לִ֖י הַלְוִיִּֽם:  ת  וַאֲנָא הָא קְרֵבִית יָת לֵוָאֵי מִגוֹ בְּנֵי יִשְׂרָאֵל חֲלַף כָּל בּוּכְרָא פְּתַח וַלְדָא מִבְּנֵי יִשְׂרָאֵל וִיהוֹן מְשַׁמְשִׁין קֳדָמַי לֵוָאֵי:

{יג} כִּ֣י לִי֮ כָּל-בְּכוֹר֒ בְּיוֹם֩ הַכֹּתִ֨י כָל-בְּכ֜וֹר בְּאֶ֣רֶץ מִצְרַ֗יִם הִקְדַּ֨שְׁתִּי לִ֤י כָל-בְּכוֹר֙ בְּיִשְׂרָאֵ֔ל מֵאָדָ֖ם עַד-בְּהֵמָ֑ה לִ֥י יִהְי֖וּ אֲנִ֥י יְהוָֽה:  מ  כִּ֣י לִי֮ כָּל-בְּכוֹר֒ בְּיוֹם֩ הַכֹּתִ֨י כָל-בְּכ֜וֹר בְּאֶ֣רֶץ מִצְרַ֗יִם הִקְדַּ֨שְׁתִּי לִ֤י כָל-בְּכוֹר֙ בְּיִשְׂרָאֵ֔ל מֵאָדָ֖ם עַד-בְּהֵמָ֑ה לִ֥י יִהְי֖וּ אֲנִ֥י יְהוָֽה:  ת  אֲרֵי דִילִי כָּל בּוּכְרָא בְּיוֹמָא דִקְטָלִית כָּל בּוּכְרָא בְּאַרְעָא דְמִצְרַיִם אַקְדֵשִׁית קֳדָמַי כָּל בּוּכְרָא בְּיִשְׂרָאֵל מֵאֱנָשָׁא עַד בְּעִירָא דִילִי יְהוֹן אֲנָא יְיָ: (פ)

{יד}  חמישי  וַיְדַבֵּ֤ר יְהוָה֙ אֶל-מֹשֶׁ֔ה בְּמִדְבַּ֥ר סִינַ֖י לֵאמֹֽר:  מ  וַיְדַבֵּ֤ר יְהוָה֙ אֶל-מֹשֶׁ֔ה בְּמִדְבַּ֥ר סִינַ֖י לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה בְּמַדְבְּרָא דְסִינַי לְמֵימָר:

{טו} פְּקֹד֙ אֶת-בְּנֵ֣י לֵוִ֔י לְבֵ֥ית אֲבֹתָ֖ם לְמִשְׁפְּחֹתָ֑ם כָּל-זָכָ֛ר מִבֶּן-חֹ֥דֶשׁ וָמַ֖עְלָה תִּפְקְדֵֽם:  מ  פְּקֹד֙ אֶת-בְּנֵ֣י לֵוִ֔י לְבֵ֥ית אֲבֹתָ֖ם לְמִשְׁפְּחֹתָ֑ם כָּל-זָכָ֛ר מִבֶּן-חֹ֥דֶשׁ וָמַ֖עְלָה תִּפְקְדֵֽם:  ת  מְנֵי יָת בְּנֵי לֵוִי לְבֵית אֲבָהַתְהוֹן לְזַרְעֲיַתְהוֹן כָּל דְכוּרָא מִבַּר יַרְחָא וּלְעֵלָא תִּמְנִנוּן:

{טז} וַיִּפְקֹ֥ד אֹתָ֛ם מֹשֶׁ֖ה עַל-פִּ֣י יְהוָ֑ה כַּאֲשֶׁ֖ר צֻוָּֽה:  מ  וַיִּפְקֹ֥ד אֹתָ֛ם מֹשֶׁ֖ה עַל-פִּ֣י יְהוָ֑ה כַּאֲשֶׁ֖ר צֻוָּֽה:  ת  וּמְנָא יָתְהוֹן משֶׁה עַל מֵימְרָא דַיְיָ כְּמָא דְאִתְפַּקָד:

{יז} וַיִּֽהְיוּ-אֵ֥לֶּה בְנֵֽי-לֵוִ֖י בִּשְׁמֹתָ֑ם גֵּרְשׁ֕וֹן וּקְהָ֖ת וּמְרָרִֽי:  מ  וַיִּֽהְיוּ-אֵ֥לֶּה בְנֵֽי-לֵוִ֖י בִּשְׁמֹתָ֑ם גֵּרְשׁ֕וֹן וּקְהָ֖ת וּמְרָרִֽי:  ת  וַהֲווֹ אִלֵין בְּנֵי לֵוִי בִּשְׁמָהַתְהוֹן גֵרְשׁוֹן וּקְהָת וּמְרָרִי:

{יח} וְאֵ֛לֶּה שְׁמ֥וֹת בְּֽנֵי-גֵרְשׁ֖וֹן לְמִשְׁפְּחֹתָ֑ם לִבְנִ֖י וְשִׁמְעִֽי:  מ  וְאֵ֛לֶּה שְׁמ֥וֹת בְּֽנֵי-גֵרְשׁ֖וֹן לְמִשְׁפְּחֹתָ֑ם לִבְנִ֖י וְשִׁמְעִֽי:  ת  וְאִלֵין שְׁמָהַת בְּנֵי גֵרְשׁוֹן לְזַרְעֲיַתְהוֹן לִבְנִי וְשִׁמְעִי:

{יט} וּבְנֵ֥י קְהָ֖ת לְמִשְׁפְּחֹתָ֑ם עַמְרָ֣ם וְיִצְהָ֔ר חֶבְר֖וֹן וְעֻזִּיאֵֽל:  מ  וּבְנֵ֥י קְהָ֖ת לְמִשְׁפְּחֹתָ֑ם עַמְרָ֣ם וְיִצְהָ֔ר חֶבְר֖וֹן וְעֻזִּיאֵֽל:  ת  וּבְנֵי קְהָת לְזַרְעֲיַתְהוֹן עַמְרָם וְיִצְהָר חֶבְרוֹן וְעֻזִיאֵל:

{כ} וּבְנֵ֧י מְרָרִ֛י לְמִשְׁפְּחֹתָ֖ם מַחְלִ֣י וּמוּשִׁ֑י אֵ֥לֶּה הֵ֛ם מִשְׁפְּחֹ֥ת הַלֵּוִ֖י לְבֵ֥ית אֲבֹתָֽם:  מ  וּבְנֵ֧י מְרָרִ֛י לְמִשְׁפְּחֹתָ֖ם מַחְלִ֣י וּמוּשִׁ֑י אֵ֥לֶּה הֵ֛ם מִשְׁפְּחֹ֥ת הַלֵּוִ֖י לְבֵ֥ית אֲבֹתָֽם:  ת  וּבְנֵי מְרָרִי לְזַרְעֲיַתְהוֹן מַחְלִי וּמוּשִׁי אִלֵין אִנוּן זַרְעֲיַת לֵוָאֵי לְבֵית אֲבָהַתְהוֹן:

{כא} לְגֵ֣רְשׁ֔וֹן מִשְׁפַּ֙חַת֙ הַלִּבְנִ֔י וּמִשְׁפַּ֖חַת הַשִּׁמְעִ֑י אֵ֣לֶּה הֵ֔ם מִשְׁפְּחֹ֖ת הַגֵּרְשֻׁנִּֽי:  מ  לְגֵ֣רְשׁ֔וֹן מִשְׁפַּ֙חַת֙ הַלִּבְנִ֔י וּמִשְׁפַּ֖חַת הַשִּׁמְעִ֑י אֵ֣לֶּה הֵ֔ם מִשְׁפְּחֹ֖ת הַגֵּרְשֻׁנִּֽי:  ת  לְגֵרְשׁוֹן זַרְעִית לִבְנִי וְזַרְעִית שִׁמְעִי אִלֵין אִנוּן זַרְעֲיַת גֵרְשׁוֹן:

{כב} פְּקֻדֵיהֶם֙ בְּמִסְפַּ֣ר כָּל-זָכָ֔ר מִבֶּן-חֹ֖דֶשׁ וָמָ֑עְלָה פְּקֻ֣דֵיהֶ֔ם שִׁבְעַ֥ת אֲלָפִ֖ים וַחֲמֵ֥שׁ מֵאֽוֹת:  מ  פְּקֻדֵיהֶם֙ בְּמִסְפַּ֣ר כָּל-זָכָ֔ר מִבֶּן-חֹ֖דֶשׁ וָמָ֑עְלָה פְּקֻ֣דֵיהֶ֔ם שִׁבְעַ֥ת אֲלָפִ֖ים וַחֲמֵ֥שׁ מֵאֽוֹת:  ת  מִנְיָנֵיהוֹן בְּמִנְיַן כָּל דְכוּרָא מִבַּר יַרְחָא וּלְעֵלָא מִנְיָנֵיהוֹן שַׁבְעָא אַלְפִין וַחֲמֵשׁ מְאָה:

{כג} מִשְׁפְּחֹ֖ת הַגֵּרְשֻׁנִּ֑י אַחֲרֵ֧י הַמִּשְׁכָּ֛ן יַחֲנ֖וּ יָֽמָּה:  מ  מִשְׁפְּחֹ֖ת הַגֵּרְשֻׁנִּ֑י אַחֲרֵ֧י הַמִּשְׁכָּ֛ן יַחֲנ֖וּ יָֽמָּה:  ת  זַרְעֲיַת גֵרְשׁוֹן אֲחוֹרֵי מַשְׁכְּנָא יִשְּׁרוּן מַעַרְבָא:

{כד} וּנְשִׂ֥יא בֵֽית-אָ֖ב לַגֵּרְשֻׁנִּ֑י אֶלְיָסָ֖ף בֶּן-לָאֵֽל:  מ  וּנְשִׂ֥יא בֵֽית-אָ֖ב לַגֵּרְשֻׁנִּ֑י אֶלְיָסָ֖ף בֶּן-לָאֵֽל:  ת  וְרַב בֵּית אַבָּא לְבֵית גֵרְשׁוֹן אֶלְיָסָף בַּר לָאֵל:

{כה} וּמִשְׁמֶ֤רֶת בְּנֵֽי-גֵרְשׁוֹן֙ בְּאֹ֣הֶל מוֹעֵ֔ד הַמִּשְׁכָּ֖ן וְהָאֹ֑הֶל מִכְסֵ֕הוּ וּמָסַ֕ךְ פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  מ  וּמִשְׁמֶ֤רֶת בְּנֵֽי-גֵרְשׁוֹן֙ בְּאֹ֣הֶל מוֹעֵ֔ד הַמִּשְׁכָּ֖ן וְהָאֹ֑הֶל מִכְסֵ֕הוּ וּמָסַ֕ךְ פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  ת  וּמַטְרַת בְּנֵי גֵרְשׁוֹן בְּמַשְׁכַּן זִמְנָא מַשְׁכְּנָא וּפְרָסָא חוֹפָאֵהּ וּפְרָסָא דִתְרַע מַשְׁכַן זִמְנָא:

{כו} וְקַלְעֵ֣י הֶֽחָצֵ֗ר וְאֶת-מָסַךְ֙ פֶּ֣תַח הֶֽחָצֵ֔ר אֲשֶׁ֧ר עַל-הַמִּשְׁכָּ֛ן וְעַל-הַמִּזְבֵּ֖חַ סָבִ֑יב וְאֵת֙ מֵֽיתָרָ֔יו לְכֹ֖ל עֲבֹדָתֽוֹ:  מ  וְקַלְעֵ֣י הֶֽחָצֵ֗ר וְאֶת-מָסַךְ֙ פֶּ֣תַח הֶֽחָצֵ֔ר אֲשֶׁ֧ר עַל-הַמִּשְׁכָּ֛ן וְעַל-הַמִּזְבֵּ֖חַ סָבִ֑יב וְאֵת֙ מֵֽיתָרָ֔יו לְכֹ֖ל עֲבֹדָתֽוֹ:  ת  וּסְרָדֵי דְדַרְתָּא וְיָת פְּרָסָא דִתְרַע דַרְתָּא דִי עַל מַשְׁכְּנָא וְעַל מַדְבְּחָא סְחוֹר סְחוֹר וְיָת אַטוּנוֹהִי לְכֹל פּוּלְחָנֵהּ: (ס)

{כז} וְלִקְהָ֗ת מִשְׁפַּ֤חַת הַֽעַמְרָמִי֙ וּמִשְׁפַּ֣חַת הַיִּצְהָרִ֔י וּמִשְׁפַּ֙חַת֙ הַֽחֶבְרֹנִ֔י וּמִשְׁפַּ֖חַת הָֽעָזִּיאֵלִ֑י אֵ֥לֶּה הֵ֖ם מִשְׁפְּחֹ֥ת הַקְּהָתִֽי:  מ  וְלִקְהָ֗ת מִשְׁפַּ֤חַת הַֽעַמְרָמִי֙ וּמִשְׁפַּ֣חַת הַיִּצְהָרִ֔י וּמִשְׁפַּ֙חַת֙ הַֽחֶבְרֹנִ֔י וּמִשְׁפַּ֖חַת הָֽעָזִּיאֵלִ֑י אֵ֥לֶּה הֵ֖ם מִשְׁפְּחֹ֥ת הַקְּהָתִֽי:  ת  וְלִקְהָת זַרְעִית עַמְרָם וְזַרְעִית יִצְהָר וְזַרְעִית חֶבְרוֹן וְזַרְעִית עֻזִיאֵל אִלֵין אִנוּן זַרְעֲיַת קְהָת:

{כח} בְּמִסְפַּר֙ כָּל-זָכָ֔ר מִבֶּן-חֹ֖דֶשׁ וָמָ֑עְלָה שְׁמֹנַ֤ת אֲלָפִים֙ וְשֵׁ֣שׁ מֵא֔וֹת שֹׁמְרֵ֖י מִשְׁמֶ֥רֶת הַקֹּֽדֶשׁ:  מ  בְּמִסְפַּר֙ כָּל-זָכָ֔ר מִבֶּן-חֹ֖דֶשׁ וָמָ֑עְלָה שְׁמֹנַ֤ת אֲלָפִים֙ וְשֵׁ֣שׁ מֵא֔וֹת שֹׁמְרֵ֖י מִשְׁמֶ֥רֶת הַקֹּֽדֶשׁ:  ת  בְּמִנְיַן כָּל דְכוּרָא מִבַּר יַרְחָא וּלְעֵלָא תְּמַנְיָא אַלְפִין וְשִׁית מְאָה נַטְּרֵי מַטַרְתָּא דְקוּדְשָׁא:

{כט} מִשְׁפְּחֹ֥ת בְּנֵי-קְהָ֖ת יַחֲנ֑וּ עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן תֵּימָֽנָה:  מ  מִשְׁפְּחֹ֥ת בְּנֵי-קְהָ֖ת יַחֲנ֑וּ עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן תֵּימָֽנָה:  ת  זַרְעֲיַת בְּנֵי קְהָת יִשְׁרוּן עַל צִדָא דְמַשְׁכְּנָא דָרוֹמָא:

{ל} וּנְשִׂ֥יא בֵֽית-אָ֖ב לְמִשְׁפְּחֹ֣ת הַקְּהָתִ֑י אֶלִיצָפָ֖ן בֶּן-עֻזִּיאֵֽל:  מ  וּנְשִׂ֥יא בֵֽית-אָ֖ב לְמִשְׁפְּחֹ֣ת הַקְּהָתִ֑י אֶלִיצָפָ֖ן בֶּן-עֻזִּיאֵֽל:  ת  וְרַב בֵּית אַבָּא לְזַרְעֲיַת קְהָת אֱלִיצָפָן בַּר עֻזִיאֵל:

{לא} וּמִשְׁמַרְתָּ֗ם הָאָרֹ֤ן וְהַשֻּׁלְחָן֙ וְהַמְּנֹרָ֣ה וְהַֽמִּזְבְּחֹ֔ת וּכְלֵ֣י הַקֹּ֔דֶשׁ אֲשֶׁ֥ר יְשָׁרְת֖וּ בָּהֶ֑ם וְהַ֨מָּסָ֔ךְ וְכֹ֖ל עֲבֹדָתֽוֹ:  מ  וּמִשְׁמַרְתָּ֗ם הָאָרֹ֤ן וְהַשֻּׁלְחָן֙ וְהַמְּנֹרָ֣ה וְהַֽמִּזְבְּחֹ֔ת וּכְלֵ֣י הַקֹּ֔דֶשׁ אֲשֶׁ֥ר יְשָׁרְת֖וּ בָּהֶ֑ם וְהַ֨מָּסָ֔ךְ וְכֹ֖ל עֲבֹדָתֽוֹ:  ת  וּמַטַרְתְּהוֹן אֲרוֹנָא וּפְתוֹרָא וּמְנַרְתָּא וּמַדְבְּחַיָא וּמָנֵי קוּדְשָׁא דִי יְשַׁמְשׁוּן בְּהוֹן וּפְרָסָא וְכֹל פוּלְחָנֵהּ:

{לב} וּנְשִׂיא֙ נְשִׂיאֵ֣י הַלֵּוִ֔י אֶלְעָזָ֖ר בֶּן-אַהֲרֹ֣ן הַכֹּהֵ֑ן פְּקֻדַּ֕ת שֹׁמְרֵ֖י מִשְׁמֶ֥רֶת הַקֹּֽדֶשׁ:  מ  וּנְשִׂיא֙ נְשִׂיאֵ֣י הַלֵּוִ֔י אֶלְעָזָ֖ר בֶּן-אַהֲרֹ֣ן הַכֹּהֵ֑ן פְּקֻדַּ֕ת שֹׁמְרֵ֖י מִשְׁמֶ֥רֶת הַקֹּֽדֶשׁ:  ת  וַאֲמַרְכְּלָא דִמְמַנָא עַל רַבְרְבֵי לֵוָאֵי אֶלְעָזָר בַּר אַהֲרֹן כַּהֲנָא דְמִתְּחוֹת יְדוֹהִי מְמַנָן נָטְרֵי מַטְרַת קוּדְשָׁא:

{לג} לִמְרָרִ֕י מִשְׁפַּ֙חַת֙ הַמַּחְלִ֔י וּמִשְׁפַּ֖חַת הַמּוּשִׁ֑י אֵ֥לֶּה הֵ֖ם מִשְׁפְּחֹ֥ת מְרָרִֽי:  מ  לִמְרָרִ֕י מִשְׁפַּ֙חַת֙ הַמַּחְלִ֔י וּמִשְׁפַּ֖חַת הַמּוּשִׁ֑י אֵ֥לֶּה הֵ֖ם מִשְׁפְּחֹ֥ת מְרָרִֽי:  ת  לִמְרָרִי זַרְעִית מַחְלִי וְזַרְעִית מוּשִׁי אִלֵין אִנוּן זַרְעֲיַת מְרָרִי:

{לד} וּפְקֻדֵיהֶם֙ בְּמִסְפַּ֣ר כָּל-זָכָ֔ר מִבֶּן-חֹ֖דֶשׁ וָמָ֑עְלָה שֵׁ֥שֶׁת אֲלָפִ֖ים וּמָאתָֽיִם:  מ  וּפְקֻדֵיהֶם֙ בְּמִסְפַּ֣ר כָּל-זָכָ֔ר מִבֶּן-חֹ֖דֶשׁ וָמָ֑עְלָה שֵׁ֥שֶׁת אֲלָפִ֖ים וּמָאתָֽיִם:  ת  וּמִנְיָנֵיהוֹן בְּמִנְיַן כָּל דְכוּרָא מִבַּר יַרְחָא וּלְעֵלָא שִׁתָּא אַלְפִין וּמָאתָן:

{לה} וּנְשִׂ֤יא בֵֽית-אָב֙ לְמִשְׁפְּחֹ֣ת מְרָרִ֔י צוּרִיאֵ֖ל בֶּן-אֲבִיחָ֑יִל עַ֣ל יֶ֧רֶךְ הַמִּשְׁכָּ֛ן יַחֲנ֖וּ צָפֹֽנָה:  מ  וּנְשִׂ֤יא בֵֽית-אָב֙ לְמִשְׁפְּחֹ֣ת מְרָרִ֔י צוּרִיאֵ֖ל בֶּן-אֲבִיחָ֑יִל עַ֣ל יֶ֧רֶךְ הַמִּשְׁכָּ֛ן יַחֲנ֖וּ צָפֹֽנָה:  ת  וְרַב בֵּית אַבָּא לְזַרְעֲיַת מְרָרִי צוּרִיאֵל בַּר אֲבִיחָיִל עַל צִדָא דְמַשְׁכְּנָא יִשְׁרוּן צִפּוּנָא:

{לו} וּפְקֻדַּ֣ת מִשְׁמֶרֶת֮ בְּנֵ֣י מְרָרִי֒ קַרְשֵׁי֙ הַמִּשְׁכָּ֔ן וּבְרִיחָ֖יו וְעַמֻּדָ֣יו וַאֲדָנָ֑יו וְכָל-כֵּלָ֔יו וְכֹ֖ל עֲבֹדָתֽוֹ:  מ  וּפְקֻדַּ֣ת מִשְׁמֶרֶת֮ בְּנֵ֣י מְרָרִי֒ קַרְשֵׁי֙ הַמִּשְׁכָּ֔ן וּבְרִיחָ֖יו וְעַמֻּדָ֣יו וַאֲדָנָ֑יו וְכָל-כֵּלָ֔יו וְכֹ֖ל עֲבֹדָתֽוֹ:  ת  וְדִי מְסִיר לְמַטְרַת בְּנֵי מְרָרִי דַפֵּי מַשְׁכְּנָא וְעַבְרוֹהִי וְעַמוּדוֹהִי וְסַמְכוֹהִי וְכָל מָנוֹהִי וְכֹל פּוּלְחָנֵהּ:

{לז} וְעַמֻּדֵ֧י הֶחָצֵ֛ר סָבִ֖יב וְאַדְנֵיהֶ֑ם וִיתֵדֹתָ֖ם וּמֵֽיתְרֵיהֶֽם:  מ  וְעַמֻּדֵ֧י הֶחָצֵ֛ר סָבִ֖יב וְאַדְנֵיהֶ֑ם וִיתֵדֹתָ֖ם וּמֵֽיתְרֵיהֶֽם:  ת  וְעַמוּדֵי דְדַרְתָּא סְחוֹר סְחוֹר וְסַמְכֵיהוֹן וְסִכֵּיהוֹן וְאַטוּנֵיהוֹן:

{לח} וְהַחֹנִ֣ים לִפְנֵ֣י הַמִּשְׁכָּ֡ן קֵ֣דְמָה לִפְנֵי֩ אֹֽהֶל-מוֹעֵ֨ד | מִזְרָ֜חָה מֹשֶׁ֣ה | וְאַהֲרֹ֣ן וּבָנָ֗יו שֹֽׁמְרִים֙ מִשְׁמֶ֣רֶת הַמִּקְדָּ֔שׁ לְמִשְׁמֶ֖רֶת בְּנֵ֣י יִשְׂרָאֵ֑ל וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת:  מ  וְהַחֹנִ֣ים לִפְנֵ֣י הַמִּשְׁכָּ֡ן קֵ֣דְמָה לִפְנֵי֩ אֹֽהֶל-מוֹעֵ֨ד | מִזְרָ֜חָה מֹשֶׁ֣ה | וְאַהֲרֹ֣ן וּבָנָ֗יו שֹֽׁמְרִים֙ מִשְׁמֶ֣רֶת הַמִּקְדָּ֔שׁ לְמִשְׁמֶ֖רֶת בְּנֵ֣י יִשְׂרָאֵ֑ל וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת:  ת  וְדִי שְׁרַן קֳדָם מַשְׁכְּנָא קִדוּמָא קֳדָם מַשְׁכַּן זִמְנָא מַדִינְחָא משֶׁה וְאַהֲרֹן וּבְנוֹהִי נָטְרִין מַטְרַת מַקְדְשָׁא לְמַטְרַת בְּנֵי יִשְׂרָאֵל וְחִילוֹנַי דְיִקְרַב יִתְקְטֵל:

{לט} כָּל-פְּקוּדֵ֨י הַלְוִיִּ֜ם אֲשֶׁר֩ פָּקַ֨ד מֹשֶׁ֧ה וְׄאַׄהֲׄרֹ֛ׄןׄ עַל-פִּ֥י יְהוָ֖ה לְמִשְׁפְּחֹתָ֑ם כָּל-זָכָר֙ מִבֶּן-חֹ֣דֶשׁ וָמַ֔עְלָה שְׁנַ֥יִם וְעֶשְׂרִ֖ים אָֽלֶף:  מ  כָּל-פְּקוּדֵ֨י הַלְוִיִּ֜ם אֲשֶׁר֩ פָּקַ֨ד מֹשֶׁ֧ה וְׄאַׄהֲׄרֹ֛ׄןׄ עַל-פִּ֥י יְהוָ֖ה לְמִשְׁפְּחֹתָ֑ם כָּל-זָכָר֙ מִבֶּן-חֹ֣דֶשׁ וָמַ֔עְלָה שְׁנַ֥יִם וְעֶשְׂרִ֖ים אָֽלֶף:  ת  כָּל מִנְיָנֵי לֵוָאֵי דִי מְנָא משֶׁה וְאַהֲרֹן עַל מֵימְרָא דַיְיָ לְזַרְעֲיַתְהוֹן כָּל דְכוּרָא מִבַּר יַרְחָא וּלְעֵלָא עַשְׂרִין וּתְרֵין אַלְפִין: (ס)

{מ}  שישי  וַיֹּ֨אמֶר יְהוָ֜ה אֶל-מֹשֶׁ֗ה פְּקֹ֨ד כָּל-בְּכֹ֤ר זָכָר֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל מִבֶּן-חֹ֖דֶשׁ וָמָ֑עְלָה וְשָׂ֕א אֵ֖ת מִסְפַּ֥ר שְׁמֹתָֽם:  מ  וַיֹּ֨אמֶר יְהוָ֜ה אֶל-מֹשֶׁ֗ה פְּקֹ֨ד כָּל-בְּכֹ֤ר זָכָר֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל מִבֶּן-חֹ֖דֶשׁ וָמָ֑עְלָה וְשָׂ֕א אֵ֖ת מִסְפַּ֥ר שְׁמֹתָֽם:  ת  וַאֲמַר יְיָ לְמשֶׁה מְנֵי כָּל בּוּכְרַיָא דִכְרַיָא לִבְנֵי יִשְׂרָאֵל מִבַּר יַרְחָא וּלְעֵלָא וְקַבֵּל יָת מִנְיַן שְׁמָהַתְהוֹן:

{מא} וְלָקַחְתָּ֨ אֶת-הַלְוִיִּ֥ם לִי֙ אֲנִ֣י יְהוָ֔ה תַּ֥חַת כָּל-בְּכֹ֖ר בִּבְנֵ֣י יִשְׂרָאֵ֑ל וְאֵת֙ בֶּהֱמַ֣ת הַלְוִיִּ֔ם תַּ֣חַת כָּל-בְּכ֔וֹר בְּבֶהֱמַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וְלָקַחְתָּ֨ אֶת-הַלְוִיִּ֥ם לִי֙ אֲנִ֣י יְהוָ֔ה תַּ֥חַת כָּל-בְּכֹ֖ר בִּבְנֵ֣י יִשְׂרָאֵ֑ל וְאֵת֙ בֶּהֱמַ֣ת הַלְוִיִּ֔ם תַּ֣חַת כָּל-בְּכ֔וֹר בְּבֶהֱמַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וּתְקָרֵב יָת לֵוָאֵי קֳדָמַי אֲנָא יְיָ חֲלַף כָּל בּוּכְרָא בִּבְנֵי יִשְׂרָאֵל וְיָת בְּעִירָא דְלֵוָאֵי חֲלַף כָּל בּוּכְרָא בִּבְעִירָא דִבְנֵי יִשְׂרָאֵל:

{מב} וַיִּפְקֹ֣ד מֹשֶׁ֔ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֹת֑וֹ אֶֽת-כָּל-בְּכֹ֖ר בִּבְנֵ֥י יִשְׂרָאֵֽל:  מ  וַיִּפְקֹ֣ד מֹשֶׁ֔ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֹת֑וֹ אֶֽת-כָּל-בְּכֹ֖ר בִּבְנֵ֥י יִשְׂרָאֵֽל:  ת  וּמְנָא משֶׁה כְּמָא דִי פַקֵיד יְיָ יָתֵהּ יָת כָּל בּוּכְרָא בִּבְנֵי יִשְׂרָאֵל:

{מג} וַיְהִי֩ כָל-בְּכ֨וֹר זָכָ֜ר בְּמִסְפַּ֥ר שֵׁמ֛וֹת מִבֶּן-חֹ֥דֶשׁ וָמַ֖עְלָה לִפְקֻדֵיהֶ֑ם שְׁנַ֤יִם וְעֶשְׂרִים֙ אֶ֔לֶף שְׁלֹשָׁ֥ה וְשִׁבְעִ֖ים וּמָאתָֽיִם:  מ  וַיְהִי֩ כָל-בְּכ֨וֹר זָכָ֜ר בְּמִסְפַּ֥ר שֵׁמ֛וֹת מִבֶּן-חֹ֥דֶשׁ וָמַ֖עְלָה לִפְקֻדֵיהֶ֑ם שְׁנַ֤יִם וְעֶשְׂרִים֙ אֶ֔לֶף שְׁלֹשָׁ֥ה וְשִׁבְעִ֖ים וּמָאתָֽיִם:  ת  וַהֲווֹ כָל בּוּכְרַיָא דִכְרַיָא בְּמִנְיַן שְׁמָהָן מִבַּר יַרְחָא וּלְעֵלָא לְמִנְיָנֵיהוֹן עַשְׂרִין וּתְרֵין אַלְפִין מָאתָן וְשַׁבְעִין וּתְלָתָא: (פ)

{מד} וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְּמֵימָר:

{מה} קַ֣ח אֶת-הַלְוִיִּ֗ם תַּ֤חַת כָּל-בְּכוֹר֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל וְאֶת-בֶּהֱמַ֥ת הַלְוִיִּ֖ם תַּ֣חַת בְּהֶמְתָּ֑ם וְהָיוּ-לִ֥י הַלְוִיִּ֖ם אֲנִ֥י יְהוָֽה:  מ  קַ֣ח אֶת-הַלְוִיִּ֗ם תַּ֤חַת כָּל-בְּכוֹר֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל וְאֶת-בֶּהֱמַ֥ת הַלְוִיִּ֖ם תַּ֣חַת בְּהֶמְתָּ֑ם וְהָיוּ-לִ֥י הַלְוִיִּ֖ם אֲנִ֥י יְהוָֽה:  ת  קָרֵב יָת לֵוָאֵי חֲלַף כָּל בּוּכְרָא בִּבְנֵי יִשְׂרָאֵל וְיָת בְּעִירָא דְלֵוָאֵי חֲלַף בְּעִירְהוֹן וִיהוֹן מְשַּׁמְשִׁין קֳדָמַי לֵוָאֵי אֲנָא יְיָ:

{מו} וְאֵת֙ פְּדוּיֵ֣י הַשְּׁלֹשָׁ֔ה וְהַשִּׁבְעִ֖ים וְהַמָּאתָ֑יִם הָעֹֽדְפִים֙ עַל-הַלְוִיִּ֔ם מִבְּכ֖וֹר בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וְאֵת֙ פְּדוּיֵ֣י הַשְּׁלֹשָׁ֔ה וְהַשִּׁבְעִ֖ים וְהַמָּאתָ֑יִם הָעֹֽדְפִים֙ עַל-הַלְוִיִּ֔ם מִבְּכ֖וֹר בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וְיָת פּוּרְקַן מָאתָן וְשַׁבְעִין וּתְלָתָא דְיַתִּירִין עַל לֵוָאֵי מִבּוּכְרַיָא דִבְנֵי יִשְׂרָאֵל:

{מז} וְלָקַחְתָּ֗ חֲמֵ֧שֶׁת חֲמֵ֛שֶׁת שְׁקָלִ֖ים לַגֻּלְגֹּ֑לֶת בְּשֶׁ֤קֶל הַקֹּ֙דֶשׁ֙ תִּקָּ֔ח עֶשְׂרִ֥ים גֵּרָ֖ה הַשָּֽׁקֶל:  מ  וְלָקַחְתָּ֗ חֲמֵ֧שֶׁת חֲמֵ֛שֶׁת שְׁקָלִ֖ים לַגֻּלְגֹּ֑לֶת בְּשֶׁ֤קֶל הַקֹּ֙דֶשׁ֙ תִּקָּ֔ח עֶשְׂרִ֥ים גֵּרָ֖ה הַשָּֽׁקֶל:  ת  וְתִסַב חָמֵשׁ חָמֵשׁ סִלְעִין לְגֻלְגַלְתָּא בְּסִלְעֵי קוּדְשָׁא תִּסַב עַשְׂרִין מָעִין סִלְעָא:

{מח} וְנָתַתָּ֣ה הַכֶּ֔סֶף לְאַהֲרֹ֖ן וּלְבָנָ֑יו פְּדוּיֵ֕י הָעֹדְפִ֖ים בָּהֶֽם:  מ  וְנָתַתָּ֣ה הַכֶּ֔סֶף לְאַהֲרֹ֖ן וּלְבָנָ֑יו פְּדוּיֵ֕י הָעֹדְפִ֖ים בָּהֶֽם:  ת  וְתִתֵּן כַּסְפָּא לְאַהֲרֹן וְלִבְנוֹהִי פֻּרְקַן דְיַתִּירִין בְּהוֹן:

{מט} וַיִּקַּ֣ח מֹשֶׁ֔ה אֵ֖ת כֶּ֣סֶף הַפִּדְי֑וֹם מֵאֵת֙ הָעֹ֣דְפִ֔ים עַ֖ל פְּדוּיֵ֥י הַלְוִיִּֽם:  מ  וַיִּקַּ֣ח מֹשֶׁ֔ה אֵ֖ת כֶּ֣סֶף הַפִּדְי֑וֹם מֵאֵת֙ הָעֹ֣דְפִ֔ים עַ֖ל פְּדוּיֵ֥י הַלְוִיִּֽם:  ת  וּנְסֵיב משֶׁה יָת כְּסַף פּוּרְקָנָא מִן דְיַתִּירִין עַל פְּרִיקֵי לֵוָאֵי:

{נ} מֵאֵ֗ת בְּכ֛וֹר בְּנֵ֥י יִשְׂרָאֵ֖ל לָקַ֣ח אֶת-הַכָּ֑סֶף חֲמִשָּׁ֨ה וְשִׁשִּׁ֜ים וּשְׁלֹ֥שׁ מֵא֛וֹת וָאֶ֖לֶף בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ:  מ  מֵאֵ֗ת בְּכ֛וֹר בְּנֵ֥י יִשְׂרָאֵ֖ל לָקַ֣ח אֶת-הַכָּ֑סֶף חֲמִשָּׁ֨ה וְשִׁשִּׁ֜ים וּשְׁלֹ֥שׁ מֵא֛וֹת וָאֶ֖לֶף בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ:  ת  מִן בּוּכְרַיָא דִבְנֵי יִשְׂרָאֵל נְסִיב יָת כַּסְפָּא אֲלַף וּתְלַת מְאָה וְשִׁתִּין וְחָמֵשׁ סִלְעִין בְּסִלְעֵי קוּדְשָׁא:

{נא} וַיִּתֵּ֨ן מֹשֶׁ֜ה אֶת-כֶּ֧סֶף הַפְּדֻיִ֛ם לְאַהֲרֹ֥ן וּלְבָנָ֖יו עַל-פִּ֣י יְהוָ֑ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת-מֹשֶֽׁה:  מ  וַיִּתֵּ֨ן מֹשֶׁ֜ה אֶת-כֶּ֧סֶף הַפְּדֻיִ֛ם לְאַהֲרֹ֥ן וּלְבָנָ֖יו עַל-פִּ֣י יְהוָ֑ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת-מֹשֶֽׁה:  ת  וִיהַב משֶׁה יָת כְּסַף פְּרִיקַיָא לְאַהֲרֹן וְלִבְנוֹהִי עַל מֵימְרָא דַיְיָ כְּמָא דִי פַקֵיד יְיָ יָת משֶׁה: (פ)

{א}  שביעי  וַיְדַבֵּ֣ר יְהוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  מ  וַיְדַבֵּ֣ר יְהוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה וְעִם אַהֲרֹן לְמֵימָר:

{ב} נָשֹׂ֗א אֶת-רֹאשׁ֙ בְּנֵ֣י קְהָ֔ת מִתּ֖וֹךְ בְּנֵ֣י לֵוִ֑י לְמִשְׁפְּחֹתָ֖ם לְבֵ֥ית אֲבֹתָֽם:  מ  נָשֹׂ֗א אֶת-רֹאשׁ֙ בְּנֵ֣י קְהָ֔ת מִתּ֖וֹךְ בְּנֵ֣י לֵוִ֑י לְמִשְׁפְּחֹתָ֖ם לְבֵ֥ית אֲבֹתָֽם:  ת  קַבִּילוּ יָת חֻשְׁבַּן בְּנֵי קְהָת מִגוֹ בְּנֵי לֵוִי לְזַרְעֲיַתְהוֹן לְבֵית אֲבָהַתְהוֹן:

{ג} מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן-חֲמִשִּׁ֣ים שָׁנָ֑ה כָּל-בָּא֙ לַצָּבָ֔א לַעֲשׂ֥וֹת מְלָאכָ֖ה בְּאֹ֥הֶל מוֹעֵֽד:  מ  מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן-חֲמִשִּׁ֣ים שָׁנָ֑ה כָּל-בָּא֙ לַצָּבָ֔א לַעֲשׂ֥וֹת מְלָאכָ֖ה בְּאֹ֥הֶל מוֹעֵֽד:  ת  מִבַּר תְּלָתִין שְׁנִין וּלְעֵלָא וְעַד בַּר חַמְשִׁין שְׁנִין כָּל דְאָתֵי לְחֵילָא לְמֶעְבַּד עִבִדְתָּא בְּמַשְׁכַּן זִ מְ נָא:

{ד} זֹ֛את עֲבֹדַ֥ת בְּנֵי-קְהָ֖ת בְּאֹ֣הֶל מוֹעֵ֑ד קֹ֖דֶשׁ הַקֳּדָשִֽׁים:  מ  זֹ֛את עֲבֹדַ֥ת בְּנֵי-קְהָ֖ת בְּאֹ֣הֶל מוֹעֵ֑ד קֹ֖דֶשׁ הַקֳּדָשִֽׁים:  ת  דֵין פּוּלְחַן בְּנֵי קְהָת בְּמַשְׁכַּן זִמְנָא קוֹדֶשׁ קוּדְשַׁיָא:

{ה} וּבָ֨א אַהֲרֹ֤ן וּבָנָיו֙ בִּנְסֹ֣עַ הַֽמַּחֲנֶ֔ה וְהוֹרִ֕דוּ אֵ֖ת פָּרֹ֣כֶת הַמָּסָ֑ךְ וְכִ֨סּוּ-בָ֔הּ אֵ֖ת אֲרֹ֥ן הָעֵדֻֽת:  מ  וּבָ֨א אַהֲרֹ֤ן וּבָנָיו֙ בִּנְסֹ֣עַ הַֽמַּחֲנֶ֔ה וְהוֹרִ֕דוּ אֵ֖ת פָּרֹ֣כֶת הַמָּסָ֑ךְ וְכִ֨סּוּ-בָ֔הּ אֵ֖ת אֲרֹ֥ן הָעֵדֻֽת:  ת  וְיֵיעוֹל אַהֲרֹן וּבְנוֹהִי בְּמֵטַל מַשְׁרִיתָא וִיפָרְקוּן יָת פָּרֻכְתָּא דִפְרָסָא וִיכַסוּן בַּהּ יָת אֲרוֹנָא דְסַהֲדוּתָא:

{ו} וְנָתְנ֣וּ עָלָ֗יו כְּסוּי֙ ע֣וֹר תַּ֔חַשׁ וּפָרְשׂ֧וּ בֶֽגֶד-כְּלִ֛יל תְּכֵ֖לֶת מִלְמָ֑עְלָה וְשָׂמ֖וּ בַּדָּֽיו:  מ  וְנָתְנ֣וּ עָלָ֗יו כְּסוּי֙ ע֣וֹר תַּ֔חַשׁ וּפָרְשׂ֧וּ בֶֽגֶד-כְּלִ֛יל תְּכֵ֖לֶת מִלְמָ֑עְלָה וְשָׂמ֖וּ בַּדָּֽיו:  ת  וְיִתְּנוּן עֲלוֹהִי חוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וְיִפְרְסוּן לְבוּשׁ גְמִיר תִּכְלָא מִלְעֵלָא וִישַׁוּוּן אֲרִיחוֹהִי:

{ז} וְעַ֣ל | שֻׁלְחַ֣ן הַפָּנִ֗ים יִפְרְשׂוּ֮ בֶּ֣גֶד תְּכֵלֶת֒ וְנָתְנ֣וּ עָ֠לָיו אֶת-הַקְּעָרֹ֤ת וְאֶת-הַכַּפֹּת֙ וְאֶת-הַמְּנַקִּיֹּ֔ת וְאֵ֖ת קְשׂ֣וֹת הַנָּ֑סֶךְ וְלֶ֥חֶם הַתָּמִ֖יד עָלָ֥יו יִהְיֶֽה:  מ  וְעַ֣ל | שֻׁלְחַ֣ן הַפָּנִ֗ים יִפְרְשׂוּ֮ בֶּ֣גֶד תְּכֵלֶת֒ וְנָתְנ֣וּ עָ֠לָיו אֶת-הַקְּעָרֹ֤ת וְאֶת-הַכַּפֹּת֙ וְאֶת-הַמְּנַקִּיֹּ֔ת וְאֵ֖ת קְשׂ֣וֹת הַנָּ֑סֶךְ וְלֶ֥חֶם הַתָּמִ֖יד עָלָ֥יו יִהְיֶֽה:  ת  וְעַל פָּתוֹרָא דִלְחֵם אַפַּיָא יִפְרְסוּן לְבוּשׁ תִּכְלָא וְיִתְּנוּן עֲלוֹהִי יָת מָגִּיסַיָא וְיָת בָּזִיכַיָא וְיָת מְכִילָתָא וְיָת קַסְווֹת נִסוּכָא וְלַחְמָא תְּדִירָא עֲלוֹהִי יְהֵי:

{ח} וּפָרְשׂ֣וּ עֲלֵיהֶ֗ם בֶּ֚גֶד תּוֹלַ֣עַת שָׁנִ֔י וְכִסּ֣וּ אֹת֔וֹ בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְשָׂמ֖וּ אֶת-בַּדָּֽיו:  מ  וּפָרְשׂ֣וּ עֲלֵיהֶ֗ם בֶּ֚גֶד תּוֹלַ֣עַת שָׁנִ֔י וְכִסּ֣וּ אֹת֔וֹ בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְשָׂמ֖וּ אֶת-בַּדָּֽיו:  ת  וְיִפְרְסוּן עֲלֵיהוֹן לְבוּשׁ צְבַע זְהוֹרִי וִיכַסוּן יָתֵהּ בְּחוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וִישַׁוּוּן יָת אֲרִיחוֹהִי:

{ט} וְלָקְח֣וּ | בֶּ֣גֶד תְּכֵ֗לֶת וְכִסּ֞וּ אֶת-מְנֹרַ֤ת הַמָּאוֹר֙ וְאֶת-נֵ֣רֹתֶ֔יהָ וְאֶת-מַלְקָחֶ֖יהָ וְאֶת-מַחְתֹּתֶ֑יהָ וְאֵת֙ כָּל-כְּלֵ֣י שַׁמְנָ֔הּ אֲשֶׁ֥ר יְשָׁרְתוּ-לָ֖הּ בָּהֶֽם:  מ  וְלָקְח֣וּ | בֶּ֣גֶד תְּכֵ֗לֶת וְכִסּ֞וּ אֶת-מְנֹרַ֤ת הַמָּאוֹר֙ וְאֶת-נֵ֣רֹתֶ֔יהָ וְאֶת-מַלְקָחֶ֖יהָ וְאֶת-מַחְתֹּתֶ֑יהָ וְאֵת֙ כָּל-כְּלֵ֣י שַׁמְנָ֔הּ אֲשֶׁ֥ר יְשָׁרְתוּ-לָ֖הּ בָּהֶֽם:  ת  וְיִסְבוּן לְבוּשׁ תִּכְלָא וִיכַסוּן יָת מְנַרְתָּא דְאַנְהוֹרֵי וְיָת בּוֹצִינָהָא וְיָת צִבְתָהָא וְיָת מַחְתְּיָתָהָא וְיָת כָּל מָנֵי מִשְׁחָא דִי יְשַׁמְשׁוּן לָהּ בְּהוֹן:

{י} וְנָתְנ֤וּ אֹתָהּ֙ וְאֶת-כָּל-כֵּלֶ֔יהָ אֶל-מִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְנָתְנ֖וּ עַל-הַמּֽוֹט:  מ  וְנָתְנ֤וּ אֹתָהּ֙ וְאֶת-כָּל-כֵּלֶ֔יהָ אֶל-מִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְנָתְנ֖וּ עַל-הַמּֽוֹט:  ת  וְיִתְּנוּן יָתָהּ וְיָת כָּל מָנָהָא לְחוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וְיִתְּנוּן עַל אֲרִיחָא:

{יא} וְעַ֣ל | מִזְבַּ֣ח הַזָּהָ֗ב יִפְרְשׂוּ֙ בֶּ֣גֶד תְּכֵ֔לֶת וְכִסּ֣וּ אֹת֔וֹ בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְשָׂמ֖וּ אֶת-בַּדָּֽיו:  מ  וְעַ֣ל | מִזְבַּ֣ח הַזָּהָ֗ב יִפְרְשׂוּ֙ בֶּ֣גֶד תְּכֵ֔לֶת וְכִסּ֣וּ אֹת֔וֹ בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְשָׂמ֖וּ אֶת-בַּדָּֽיו:  ת  וְעַל מַדְבְּחָא דְדַהֲבָא יִפְרְסוּן לְבוּשׁ תִּכְלָא וִיכַסוּן יָתֵהּ בְּחוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וִישַׁוּוּן יָת אֲרִיחוֹהִי:

{יב} וְלָקְחוּ֩ אֶת-כָּל-כְּלֵ֨י הַשָּׁרֵ֜ת אֲשֶׁ֧ר יְשָֽׁרְתוּ-בָ֣ם בַּקֹּ֗דֶשׁ וְנָֽתְנוּ֙ אֶל-בֶּ֣גֶד תְּכֵ֔לֶת וְכִסּ֣וּ אוֹתָ֔ם בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְנָתְנ֖וּ עַל-הַמּֽוֹט:  מ  וְלָקְחוּ֩ אֶת-כָּל-כְּלֵ֨י הַשָּׁרֵ֜ת אֲשֶׁ֧ר יְשָֽׁרְתוּ-בָ֣ם בַּקֹּ֗דֶשׁ וְנָֽתְנוּ֙ אֶל-בֶּ֣גֶד תְּכֵ֔לֶת וְכִסּ֣וּ אוֹתָ֔ם בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְנָתְנ֖וּ עַל-הַמּֽוֹט:  ת  וְיִסְבוּן יָת כָּל מָנֵי שִׁמוּשָׁא דִי יְשַׁמְשׁוּן בְּהוֹן בְּקוּדְשָׁא וְיִתְּנוּן לִלְבוּשׁ תִּכְלָא וִיכַסוּן יָתְהוֹן בְּחוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וְיִתְּנוּן עַל אֲרִיחָא:

{יג} וְדִשְּׁנ֖וּ אֶת-הַמִּזְבֵּ֑חַ וּפָרְשׂ֣וּ עָלָ֔יו בֶּ֖גֶד אַרְגָּמָֽן:  מ  וְדִשְּׁנ֖וּ אֶת-הַמִּזְבֵּ֑חַ וּפָרְשׂ֣וּ עָלָ֔יו בֶּ֖גֶד אַרְגָּמָֽן:  ת  וְיִסְפוּן יָת קִטְמָא דְמַדְבְּחָא וְיִפְרְסוּן עֲלוֹהִי לְבוּשׁ אַרְגְוָן:

{יד} וְנָתְנ֣וּ עָ֠לָיו אֶֽת-כָּל-כֵּלָ֞יו אֲשֶׁ֣ר יְֽשָׁרְת֧וּ עָלָ֣יו בָּהֶ֗ם אֶת-הַמַּחְתֹּ֤ת אֶת-הַמִּזְלָגֹת֙ וְאֶת-הַיָּעִ֣ים וְאֶת-הַמִּזְרָקֹ֔ת כֹּ֖ל כְּלֵ֣י הַמִּזְבֵּ֑חַ וּפָרְשׂ֣וּ עָלָ֗יו כְּס֛וּי ע֥וֹר תַּ֖חַשׁ וְשָׂמ֥וּ בַדָּֽיו:  מ  וְנָתְנ֣וּ עָ֠לָיו אֶֽת-כָּל-כֵּלָ֞יו אֲשֶׁ֣ר יְֽשָׁרְת֧וּ עָלָ֣יו בָּהֶ֗ם אֶת-הַמַּחְתֹּ֤ת אֶת-הַמִּזְלָגֹת֙ וְאֶת-הַיָּעִ֣ים וְאֶת-הַמִּזְרָקֹ֔ת כֹּ֖ל כְּלֵ֣י הַמִּזְבֵּ֑חַ וּפָרְשׂ֣וּ עָלָ֗יו כְּס֛וּי ע֥וֹר תַּ֖חַשׁ וְשָׂמ֥וּ בַדָּֽיו:  ת  וְיִתְּנוּן עֲלוֹהִי יָת כָּל מָנוֹהִי דִי יְשַׁמְשׁוּן עֲלוֹהִי בְהוֹן יָת מַחְתְּיָתָא וְיָת צִנוֹרְיָתָא וְיָת מַגְרוֹפְיָתָא וְיָת מִזְרְקַיָא כֹּל מָנֵי מַדְבְּחָא וְיִפְרְסוּן עֲלוֹהִי חוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וִישַׁווּן אֲרִיחוֹהִי:

{טו} וְכִלָּ֣ה אַֽהֲרֹן-וּ֠בָנָיו לְכַסֹּ֨ת אֶת-הַקֹּ֜דֶשׁ וְאֶת-כָּל-כְּלֵ֣י הַקֹּדֶשׁ֮ בִּנְסֹ֣עַ הַֽמַּחֲנֶה֒ וְאַחֲרֵי-כֵ֗ן יָבֹ֤אוּ בְנֵי-קְהָת֙ לָשֵׂ֔את וְלֹֽא-יִגְּע֥וּ אֶל-הַקֹּ֖דֶשׁ וָמֵ֑תוּ אֵ֛לֶּה מַשָּׂ֥א בְנֵֽי-קְהָ֖ת בְּאֹ֥הֶל מוֹעֵֽד:  מ  וְכִלָּ֣ה אַֽהֲרֹן-וּ֠בָנָיו לְכַסֹּ֨ת אֶת-הַקֹּ֜דֶשׁ וְאֶת-כָּל-כְּלֵ֣י הַקֹּדֶשׁ֮ בִּנְסֹ֣עַ הַֽמַּחֲנֶה֒ וְאַחֲרֵי-כֵ֗ן יָבֹ֤אוּ בְנֵי-קְהָת֙ לָשֵׂ֔את וְלֹֽא-יִגְּע֥וּ אֶל-הַקֹּ֖דֶשׁ וָמֵ֑תוּ אֵ֛לֶּה מַשָּׂ֥א בְנֵֽי-קְהָ֖ת בְּאֹ֥הֶל מוֹעֵֽד:  ת  וִישֵׁיצֵי אַהֲרֹן וּבְנוֹהִי לְכַסָאָה יָת קוּדְשָׁא וְיָת כָּל מָנֵי קוּדְשָׁא בְּמֵטַל מַשְׁרִיתָא וּבָתַר כֵּן יַעֲלוּן בְּנֵי קְהָת לְמִטַּל וְלָא יִקְרְבוּן לְקוּדְשָׁא וְלָא יְמוּתוּן אִלֵין מַטוּל בְּנֵי קְהָת בְּמַשְׁכַּן זִמְנָא:

{טז} וּפְקֻדַּ֞ת אֶלְעָזָ֣ר | בֶּן-אַהֲרֹ֣ן הַכֹּהֵ֗ן שֶׁ֤מֶן הַמָּאוֹר֙ וּקְטֹ֣רֶת הַסַּמִּ֔ים וּמִנְחַ֥ת הַתָּמִ֖יד וְשֶׁ֣מֶן הַמִּשְׁחָ֑ה פְּקֻדַּ֗ת כָּל-הַמִּשְׁכָּן֙ וְכָל-אֲשֶׁר-בּ֔וֹ בְּקֹ֖דֶשׁ וּבְכֵלָֽיו:  מ  וּפְקֻדַּ֞ת אֶלְעָזָ֣ר | בֶּן-אַהֲרֹ֣ן הַכֹּהֵ֗ן שֶׁ֤מֶן הַמָּאוֹר֙ וּקְטֹ֣רֶת הַסַּמִּ֔ים וּמִנְחַ֥ת הַתָּמִ֖יד וְשֶׁ֣מֶן הַמִּשְׁחָ֑ה פְּקֻדַּ֗ת כָּל-הַמִּשְׁכָּן֙ וְכָל-אֲשֶׁר-בּ֔וֹ בְּקֹ֖דֶשׁ וּבְכֵלָֽיו:  ת  וְדִי מְסִיר לְאֶלְעָזָר בַּר אַהֲרֹן כַּהֲנָא מִשְׁחָא דְאַנְהָרוּתָא וּקְטוֹרֶת בּוּסְמַיָא וּמִנְחָתָּא תְדִירָא וּמִשְׁחָא דִרְבוּתָא מַטְרַת כָּל מַשְׁכְּנָא וְכָל דִי בֵהּ בְּקוּדְשָׁא וּבְמָנוֹהִי: (פ)

{יז}  מפטיר  וַיְדַבֵּ֣ר יְהוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  מ  וַיְדַבֵּ֣ר יְהוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה וְעִם אַהֲרֹן לְמֵימָר:

{יח} אַל-תַּכְרִ֕יתוּ אֶת-שֵׁ֖בֶט מִשְׁפְּחֹ֣ת הַקְּהָתִ֑י מִתּ֖וֹךְ הַלְוִיִּֽם:  מ  אַל-תַּכְרִ֕יתוּ אֶת-שֵׁ֖בֶט מִשְׁפְּחֹ֣ת הַקְּהָתִ֑י מִתּ֖וֹךְ הַלְוִיִּֽם:  ת  לָא תְשֵׁיצוּן יָת שִׁבְטָא זַרְעֲיַת קְהָת מִגוֹ לֵוָאֵי:

{יט} וְזֹ֣את | עֲשׂ֣וּ לָהֶ֗ם וְחָיוּ֙ וְלֹ֣א יָמֻ֔תוּ בְּגִשְׁתָּ֖ם אֶת-קֹ֣דֶשׁ הַקֳּדָשִׁ֑ים אַהֲרֹ֤ן וּבָנָיו֙ יָבֹ֔אוּ וְשָׂמ֣וּ אוֹתָ֗ם אִ֥ישׁ אִ֛ישׁ עַל-עֲבֹדָת֖וֹ וְאֶל-מַשָּׂאֽוֹ:  מ  וְזֹ֣את | עֲשׂ֣וּ לָהֶ֗ם וְחָיוּ֙ וְלֹ֣א יָמֻ֔תוּ בְּגִשְׁתָּ֖ם אֶת-קֹ֣דֶשׁ הַקֳּדָשִׁ֑ים אַהֲרֹ֤ן וּבָנָיו֙ יָבֹ֔אוּ וְשָׂמ֣וּ אוֹתָ֗ם אִ֥ישׁ אִ֛ישׁ עַל-עֲבֹדָת֖וֹ וְאֶל-מַשָּׂאֽוֹ:  ת  וְדָא עִבִידוּ לְהוֹן וְיֵחוּן וְלָא יְמוּתוּן בְּמִקְרַבְהוֹן לְקֹדֶשׁ קוּדְשַּׁיָא אַהֲרֹן וּבְנוֹהִי יַעֲלוּן וִימַנוּן יָתְהוֹן גְבַר גְבַר עַל פֻּלְחָנֵהּ וּלְמַטוּלֵהּ:

{כ} וְלֹא-יָבֹ֧אוּ לִרְא֛וֹת כְּבַלַּ֥ע אֶת-הַקֹּ֖דֶשׁ וָמֵֽתוּ:  מ  וְלֹא-יָבֹ֧אוּ לִרְא֛וֹת כְּבַלַּ֥ע אֶת-הַקֹּ֖דֶשׁ וָמֵֽתוּ:  ת  וְלָא יֵעֲלוּן לְמֶחֱזֵי כַּד מְכַסָן יָת מָנֵי קוּדְשָׁא וְלָא יְמוּתוּן: (פפפ)
