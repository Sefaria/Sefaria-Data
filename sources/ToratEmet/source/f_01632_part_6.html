פרשת כי תצא


{י} כִּי תֵצֵא לַמִּלְחָמָה עַל אֹיְבֶיךָ וּנְתָנוֹ יְהוָה אֱלֹהֶיךָ בְּיָדֶךָ וְשָׁבִיתָ שִׁבְיוֹ:  מ  כִּי תֵצֵא לַמִּלְחָמָה עַל אֹיְבֶיךָ וּנְתָנוֹ יְהוָה אֱלֹהֶיךָ בְּיָדֶךָ וְשָׁבִיתָ שִׁבְיוֹ:  ת  אֲרֵי תִפּוֹק לַאֲגָחָא קְרָבָא עַל בַּעֲלֵי דְבָבָךְ וְיִמְסְרִנוּן יְיָ אֱלָהָךְ בִּידָךְ וְתִשְׁבֵּי שִׁבְיְהוֹן:

{יא} וְרָאִיתָ בַּשִּׁבְיָה אֵשֶׁת יְפַת תֹּאַר וְחָשַׁקְתָּ בָהּ וְלָקַחְתָּ לְךָ לְאִשָּׁה:  מ  וְרָאִיתָ בַּשִּׁבְיָה אֵשֶׁת יְפַת תֹּאַר וְחָשַׁקְתָּ בָהּ וְלָקַחְתָּ לְךָ לְאִשָּׁה:  ת  וְתֶחֱזֵי בְּשִׁבְיָה אִתְּתָא שַׁפִּירַת חֵזוּ וְתִתִּרְעֵי בַהּ וְתִסְבַהּ לָךְ לְאִנְתּוּ:

{יב} וַהֲבֵאתָהּ אֶל תּוֹךְ בֵּיתֶךָ וְגִלְּחָה אֶת רֹאשָׁהּ וְעָשְׂתָה אֶת צִפָּרְנֶיהָ:  מ  וַהֲבֵאתָהּ אֶל תּוֹךְ בֵּיתֶךָ וְגִלְּחָה אֶת רֹאשָׁהּ וְעָשְׂתָה אֶת צִפָּרְנֶיהָ:  ת  וְתָעֵלִנַהּ לְגוֹ בֵיתָךְ וּתְגַלַח יָת רֵישַׁהּ וּתְרַבִּי יָת טוּפְרָנָהָא:

{יג} וְהֵסִירָה אֶת שִׂמְלַת שִׁבְיָהּ מֵעָלֶיהָ וְיָשְׁבָה בְּבֵיתֶךָ וּבָכְתָה אֶת אָבִיהָ וְאֶת אִמָּהּ יֶרַח יָמִים וְאַחַר כֵּן תָּבוֹא אֵלֶיהָ וּבְעַלְתָּהּ וְהָיְתָה לְךָ לְאִשָּׁה:  מ  וְהֵסִירָה אֶת שִׂמְלַת שִׁבְיָהּ מֵעָלֶיהָ וְיָשְׁבָה בְּבֵיתֶךָ וּבָכְתָה אֶת אָבִיהָ וְאֶת אִמָּהּ יֶרַח יָמִים וְאַחַר כֵּן תָּבוֹא אֵלֶיהָ וּבְעַלְתָּהּ וְהָיְתָה לְךָ לְאִשָּׁה:  ת  וְתֶעְדֵי יָת כְּסוּת שִׁבְיַהּ מִנַהּ וְתֵיתַב בְּבֵיתָךְ וְתִבְכֵּי יָת אֲבוּהָא וְיָת אִמַהּ יְרַח יוֹמִין וּבָתַר כֵּן תֵּיעוּל לְוָתַהּ וְתִבְעֲלִנַהּ וּתְהֵי לָךְ לְאִנְתּוּ:

{יד} וְהָיָה אִם לֹא חָפַצְתָּ בָּהּ וְשִׁלַּחְתָּהּ לְנַפְשָׁהּ וּמָכֹר לֹא תִמְכְּרֶנָּה בַּכָּסֶף לֹא תִתְעַמֵּר בָּהּ תַּחַת אֲשֶׁר עִנִּיתָהּ:  מ  וְהָיָה אִם לֹא חָפַצְתָּ בָּהּ וְשִׁלַּחְתָּהּ לְנַפְשָׁהּ וּמָכֹר לֹא תִמְכְּרֶנָּה בַּכָּסֶף לֹא תִתְעַמֵּר בָּהּ תַּחַת אֲשֶׁר עִנִּיתָהּ:  ת  וִיהֵי אִם לָא תִתִּרְעֵי בַהּ וְתִפְטְּרִנַהּ לְנַפְשַׁהּ וְזַבָּנָא לָא תְזַבְּנִנַהּ בְּכַסְפָּא לָא תִתָּגַר בַּהּ חֳלַף דִי עִינִיתַהּ: (ס)

{טו} כִּי תִהְיֶיןָ לְאִישׁ שְׁתֵּי נָשִׁים הָאַחַת אֲהוּבָה וְהָאַחַת שְׂנוּאָה וְיָלְדוּ לוֹ בָנִים הָאֲהוּבָה וְהַשְּׂנוּאָה וְהָיָה הַבֵּן הַבְּכוֹר לַשְּׂנִיאָה:  מ  כִּי תִהְיֶיןָ לְאִישׁ שְׁתֵּי נָשִׁים הָאַחַת אֲהוּבָה וְהָאַחַת שְׂנוּאָה וְיָלְדוּ לוֹ בָנִים הָאֲהוּבָה וְהַשְּׂנוּאָה וְהָיָה הַבֵּן הַבְּכוֹר לַשְּׂנִיאָה:  ת  אֲרֵי תְהֶוְיָן לִגְבַר תַּרְתֵּין נְשִׁין חֲדָא רְחֶמְתָּא וַחֲדָא שְׂנִיאֲתָא וִילִידָן לֵהּ בְּנִין רְחֶמְתָּא וּשְׂנִיאֲתָא וִיהֵי בְּרָא בוּכְרָא לִשְׂנִיאֲתָא:

{טז} וְהָיָה בְּיוֹם הַנְחִילוֹ אֶת בָּנָיו אֵת אֲשֶׁר יִהְיֶה לוֹ לֹא יוּכַל לְבַכֵּר אֶת בֶּן הָאֲהוּבָה עַל פְּנֵי בֶן הַשְּׂנוּאָה הַבְּכֹר:  מ  וְהָיָה בְּיוֹם הַנְחִילוֹ אֶת בָּנָיו אֵת אֲשֶׁר יִהְיֶה לוֹ לֹא יוּכַל לְבַכֵּר אֶת בֶּן הָאֲהוּבָה עַל פְּנֵי בֶן הַשְּׂנוּאָה הַבְּכֹר:  ת  וִיהֵי בְּיוֹמָא דְיַחְסֵן לִבְנוֹהִי יָת דִי יְהֵי לֵהּ לֵית לֵהּ רְשׁוּ לְבַכָּרָא יָת בַּר רְחֶמְתָּא עַל אַפֵּי בַר שְׂנִיאֲתָא בּוּכְרָא:

{יז} כִּי אֶת הַבְּכֹר בֶּן הַשְּׂנוּאָה יַכִּיר לָתֶת לוֹ פִּי שְׁנַיִם בְּכֹל אֲשֶׁר יִמָּצֵא לוֹ כִּי הוּא רֵאשִׁית אֹנוֹ לוֹ מִשְׁפַּט הַבְּכֹרָה:  מ  כִּי אֶת הַבְּכֹר בֶּן הַשְּׂנוּאָה יַכִּיר לָתֶת לוֹ פִּי שְׁנַיִם בְּכֹל אֲשֶׁר יִמָּצֵא לוֹ כִּי הוּא רֵאשִׁית אֹנוֹ לוֹ מִשְׁפַּט הַבְּכֹרָה:  ת  אֲרֵי יָת בּוּכְרָא בַר שְׂנִיאֲתָא יַפְרַשׁ לְמִתַּן לֵהּ תַּרְתֵּין חָלָקִין בְּכֹל דִי יִשְׁתְּכַח לֵהּ אֲרֵי הוּא רֵישׁ תָּקְפֵּהּ לֵהּ חָזְיָא בְכֵרוּתָא: (ס)

{יח} כִּי יִהְיֶה לְאִישׁ בֵּן סוֹרֵר וּמוֹרֶה אֵינֶנּוּ שֹׁמֵעַ בְּקוֹל אָבִיו וּבְקוֹל אִמּוֹ וְיִסְּרוּ אֹתוֹ וְלֹא יִשְׁמַע אֲלֵיהֶם:  מ  כִּי יִהְיֶה לְאִישׁ בֵּן סוֹרֵר וּמוֹרֶה אֵינֶנּוּ שֹׁמֵעַ בְּקוֹל אָבִיו וּבְקוֹל אִמּוֹ וְיִסְּרוּ אֹתוֹ וְלֹא יִשְׁמַע אֲלֵיהֶם:  ת  אֲרֵי יְהֵי לִגְבַר בַּר סָטֵי וּמָרֵי לֵיתוֹהִי מְקַבַּל לְמֵימַר אֲבוּהִי וּלְמֵימַר אִמֵהּ וּמַלְפָן יָתֵהּ וְלָא יְקַבֵּל מִנְהוֹן:

{יט} וְתָפְשׂוּ בוֹ אָבִיו וְאִמּוֹ וְהוֹצִיאוּ אֹתוֹ אֶל זִקְנֵי עִירוֹ וְאֶל שַׁעַר מְקֹמוֹ:  מ  וְתָפְשׂוּ בוֹ אָבִיו וְאִמּוֹ וְהוֹצִיאוּ אֹתוֹ אֶל זִקְנֵי עִירוֹ וְאֶל שַׁעַר מְקֹמוֹ:  ת  וְיֶחֱדוּן בֵּהּ אֲבוּהִי וְאִמֵּהּ וְיַפְּקוּן יָתֵהּ לָקֳדָם סָבֵי קַרְתֵּהּ וְלִתְרַע בֵּית דִין אַתְרֵהּ:

{כ} וְאָמְרוּ אֶל זִקְנֵי עִירוֹ בְּנֵנוּ זֶה סוֹרֵר וּמֹרֶה אֵינֶנּוּ שֹׁמֵעַ בְּקֹלֵנוּ זוֹלֵל וְסֹבֵא:  מ  וְאָמְרוּ אֶל זִקְנֵי עִירוֹ בְּנֵנוּ זֶה סוֹרֵר וּמֹרֶה אֵינֶנּוּ שֹׁמֵעַ בְּקֹלֵנוּ זוֹלֵל וְסֹבֵא:  ת  וְיֵימְרוּן לְסָבֵי קַרְתֵּהּ בְּנָנָא דֵין סָטֵי וּמָרֵי לֵיתוֹהִי מְקַבֵּל לְמֵימְרָנָא זָלֵל בְּסָר וְסָבֵי חֲמַר:

{כא} וּרְגָמֻהוּ כָּל אַנְשֵׁי עִירוֹ בָאֲבָנִים וָמֵת וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ וְכָל יִשְׂרָאֵל יִשְׁמְעוּ וְיִרָאוּ:  מ  וּרְגָמֻהוּ כָּל אַנְשֵׁי עִירוֹ בָאֲבָנִים וָמֵת וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ וְכָל יִשְׂרָאֵל יִשְׁמְעוּ וְיִרָאוּ:  ת  וְיִרְגְמֻנֵהּ כָּל אֶנָשֵׁי קַרְתֵּהּ בְּאַבְנַיָא וִימוּת וּתְפַלֵי עָבֵד דְבִישׁ מִבֵּינָךְ וְכָל יִשְׂרָאֵל יִשְׁמְעוּן וְיִדְחֲלוּן: (ס)

{כב}  שני  וְכִי יִהְיֶה בְאִישׁ חֵטְא מִשְׁפַּט מָוֶת וְהוּמָת וְתָלִיתָ אֹתוֹ עַל עֵץ:  מ  וְכִי יִהְיֶה בְאִישׁ חֵטְא מִשְׁפַּט מָוֶת וְהוּמָת וְתָלִיתָ אֹתוֹ עַל עֵץ:  ת  וַאֲרֵי יְהֵי בִגְבַר חוֹבַת דִין דִקְטוֹל וְיִתְקְטֵל וְתִצְלוֹב יָתֵהּ עַל צְלִיבָא:

{כג} לֹא תָלִין נִבְלָתוֹ עַל הָעֵץ כִּי קָבוֹר תִּקְבְּרֶנּוּ בַּיּוֹם הַהוּא כִּי קִלְלַת אֱלֹהִים תָּלוּי וְלֹא תְטַמֵּא אֶת אַדְמָתְךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה:  מ  לֹא תָלִין נִבְלָתוֹ עַל הָעֵץ כִּי קָבוֹר תִּקְבְּרֶנּוּ בַּיּוֹם הַהוּא כִּי קִלְלַת אֱלֹהִים תָּלוּי וְלֹא תְטַמֵּא אֶת אַדְמָתְךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה:  ת  לָא תְבִית נְבִלְתֵּהּ עַל צְלִיבָא אֲרֵי מִקְבַּר תִּקְבְּרִנֵהּ בְּיוֹמָא הַהוּא אֲרֵי עַל דְחָב קֳדָם יְיָ אִצְטְלַב וְלָא תְסָאֵב יָת אַרְעָךְ דִי יְיָ אֱלָהָךְ יָהֵב לָךְ אַחֲסָנָא: (ס)




דברים פרק-כב

{א} לֹא תִרְאֶה אֶת שׁוֹר אָחִיךָ אוֹ אֶת שֵׂיוֹ נִדָּחִים וְהִתְעַלַּמְתָּ מֵהֶם הָשֵׁב תְּשִׁיבֵם לְאָחִיךָ:  מ  לֹא תִרְאֶה אֶת שׁוֹר אָחִיךָ אוֹ אֶת שֵׂיוֹ נִדָּחִים וְהִתְעַלַּמְתָּ מֵהֶם הָשֵׁב תְּשִׁיבֵם לְאָחִיךָ:  ת  לָא תֶחֱזֵי יָת תּוֹרָא דְאָחוּךְ אוֹ יַת אִמְרֵהּ דְטָעָן וְתִתְכְּבֵשׁ מִנְהוֹן אָתָבָא תָתֵבִנוּן לְאָחוּךְ:

{ב} וְאִם לֹא קָרוֹב אָחִיךָ אֵלֶיךָ וְלֹא יְדַעְתּוֹ וַאֲסַפְתּוֹ אֶל תּוֹךְ בֵּיתֶךָ וְהָיָה עִמְּךָ עַד דְּרֹשׁ אָחִיךָ אֹתוֹ וַהֲשֵׁבֹתוֹ לוֹ:  מ  וְאִם לֹא קָרוֹב אָחִיךָ אֵלֶיךָ וְלֹא יְדַעְתּוֹ וַאֲסַפְתּוֹ אֶל תּוֹךְ בֵּיתֶךָ וְהָיָה עִמְּךָ עַד דְּרֹשׁ אָחִיךָ אֹתוֹ וַהֲשֵׁבֹתוֹ לוֹ:  ת  וְאִם לָא קָרֵב אֲחוּךְ לְוָתָךְ וְלָא יְדַעַתְּ לֵהּ וְתַכְנְשִׁנֵהּ לְגוֹ בֵיתָךְ וִיהֵי עִמָךְ עַד דְיִתְבַּע אֲחוּךְ יָתֵהּ וְתָתֵיבִנֵהּ לֵהּ:

{ג} וְכֵן תַּעֲשֶׂה לַחֲמֹרוֹ וְכֵן תַּעֲשֶׂה לְשִׂמְלָתוֹ וְכֵן תַּעֲשֶׂה לְכָל אֲבֵדַת אָחִיךָ אֲשֶׁר תֹּאבַד מִמֶּנּוּ וּמְצָאתָהּ לֹא תוּכַל לְהִתְעַלֵּם:  מ  וְכֵן תַּעֲשֶׂה לַחֲמֹרוֹ וְכֵן תַּעֲשֶׂה לְשִׂמְלָתוֹ וְכֵן תַּעֲשֶׂה לְכָל אֲבֵדַת אָחִיךָ אֲשֶׁר תֹּאבַד מִמֶּנּוּ וּמְצָאתָהּ לֹא תוּכַל לְהִתְעַלֵּם:  ת  וְכֵן תַּעְבֵּד לַחֲמָרֵהּ וְכֵן תַּעְבֵּד לִכְסוּתֵהּ וְכֵן תַּעְבֵּד לְכָל אֲבֵדַת אֲחוּךְ דִי תֵיבַד מִנֵהּ וְתַשְׁכְּחִנַהּ לֵית לָךְ רְשׁוּ לְאִתְכַּסָאָה: (ס)

{ד} לֹא תִרְאֶה אֶת חֲמוֹר אָחִיךָ אוֹ שׁוֹרוֹ נֹפְלִים בַּדֶּרֶךְ וְהִתְעַלַּמְתָּ מֵהֶם הָקֵם תָּקִים עִמּוֹ:  מ  לֹא תִרְאֶה אֶת חֲמוֹר אָחִיךָ אוֹ שׁוֹרוֹ נֹפְלִים בַּדֶּרֶךְ וְהִתְעַלַּמְתָּ מֵהֶם הָקֵם תָּקִים עִמּוֹ:  ת  לָא תֶחֱזֵי יָת חֲמָרָא דְאָחוּךְ אוֹ תוֹרֵהּ רָמָן בְּאָרְחָא וְתִתְכְּבֵשׁ מִנְהוֹן אֲקָמָא תְקִים עִמֵהּ: (ס)

{ה} לֹא יִהְיֶה כְלִי גֶבֶר עַל אִשָּׁה וְלֹא יִלְבַּשׁ גֶּבֶר שִׂמְלַת אִשָּׁה כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ כָּל עֹשֵׂה אֵלֶּה:  מ  לֹא יִהְיֶה כְלִי גֶבֶר עַל אִשָּׁה וְלֹא יִלְבַּשׁ גֶּבֶר שִׂמְלַת אִשָּׁה כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ כָּל עֹשֵׂה אֵלֶּה:  ת  לָא יְהֵי תִקוּן זֵין דִגְבַר עַל אִתְּתָא וְלָא יְתַקֵן גְבַר בְּתִקוּנֵי אִתְּתָא אֲרֵי מְרָחָק קֳדָם יְיָ אֱלָהָךְ כָּל עָבֵד אִלֵין: (פ)

{ו} כִּי יִקָּרֵא קַן צִפּוֹר לְפָנֶיךָ בַּדֶּרֶךְ בְּכָל עֵץ אוֹ עַל הָאָרֶץ אֶפְרֹחִים אוֹ בֵיצִים וְהָאֵם רֹבֶצֶת עַל הָאֶפְרֹחִים אוֹ עַל הַבֵּיצִים לֹא תִקַּח הָאֵם עַל הַבָּנִים:  מ  כִּי יִקָּרֵא קַן צִפּוֹר לְפָנֶיךָ בַּדֶּרֶךְ בְּכָל עֵץ אוֹ עַל הָאָרֶץ אֶפְרֹחִים אוֹ בֵיצִים וְהָאֵם רֹבֶצֶת עַל הָאֶפְרֹחִים אוֹ עַל הַבֵּיצִים לֹא תִקַּח הָאֵם עַל הַבָּנִים:  ת  אֲרֵי יְעָרַע קִנָא דְצִפְּרָא קֳדָמָךְ בְּאָרְחָא בְּכָל אִילָן אוֹ עַל אַרְעָא אֶפְרוֹחִין אוֹ בֵיעִין וְאִמָא רְבֵעָא עַל אֶפְרֹחִין אוֹ עַל בֵּיעִין לָא תִסַב אִמָא עַל בְּנַיָא:

{ז} שַׁלֵּחַ תְּשַׁלַּח אֶת הָאֵם וְאֶת הַבָּנִים תִּקַּח לָךְ לְמַעַן יִיטַב לָךְ וְהַאֲרַכְתָּ יָמִים:  מ  שַׁלֵּחַ תְּשַׁלַּח אֶת הָאֵם וְאֶת הַבָּנִים תִּקַּח לָךְ לְמַעַן יִיטַב לָךְ וְהַאֲרַכְתָּ יָמִים:  ת  שַׁלָחָא תְשַׁלַח יָת אִמָא וְיָת בְּנַיָא תִּסַב לָךְ בְּדִיל דְיֵיטַב לָךְ וְתוֹרִיךְ יוֹמִין: (ס)

{ח}  שלישי  כִּי תִבְנֶה בַּיִת חָדָשׁ וְעָשִׂיתָ מַעֲקֶה לְגַגֶּךָ וְלֹא תָשִׂים דָּמִים בְּבֵיתֶךָ כִּי יִפֹּל הַנֹּפֵל מִמֶּנּוּ:  מ  כִּי תִבְנֶה בַּיִת חָדָשׁ וְעָשִׂיתָ מַעֲקֶה לְגַגֶּךָ וְלֹא תָשִׂים דָּמִים בְּבֵיתֶךָ כִּי יִפֹּל הַנֹּפֵל מִמֶּנּוּ:  ת  אֲרֵי תִבְנֵי בֵּיתָא חַדְתָּא וְתַעְבֵּד תְּיָקָא לְאִגְרָךְ וְלָא תְשַׁוִי חוֹבַת דִין דִקְטוֹל בְּבֵיתָךְ אֲרֵי יִפֵּל דְנָפֵל מִנֵהּ:

{ט} לֹא תִזְרַע כַּרְמְךָ כִּלְאָיִם פֶּן תִּקְדַּשׁ הַמְלֵאָה הַזֶּרַע אֲשֶׁר תִּזְרָע וּתְבוּאַת הַכָּרֶם:  מ  לֹא תִזְרַע כַּרְמְךָ כִּלְאָיִם פֶּן תִּקְדַּשׁ הַמְלֵאָה הַזֶּרַע אֲשֶׁר תִּזְרָע וּתְבוּאַת הַכָּרֶם:  ת  לָא תִזְרַע כַּרְמָךְ עֵרוּבִין דִילְמָא תִסְתָּאַב דִמְעַת זַרְעָא דִי תִזְרַע וַעֲלָלַת כַּרְמָא: (ס)

{י} לֹא תַחֲרֹשׁ בְּשׁוֹר וּבַחֲמֹר יַחְדָּו:  מ  לֹא תַחֲרֹשׁ בְּשׁוֹר וּבַחֲמֹר יַחְדָּו:  ת  לָא תִרְדֵי בְתוֹרָא וּבַחֲמָרָא כַּחֲדָא:

{יא} לֹא תִלְבַּשׁ שַׁעַטְנֵז צֶמֶר וּפִשְׁתִּים יַחְדָּו:  מ  לֹא תִלְבַּשׁ שַׁעַטְנֵז צֶמֶר וּפִשְׁתִּים יַחְדָּו:  ת  לָא תִלְבַּשּׁ שַׁעַטְנֵזָא עֲמַר וְכִתָּן מְחֻבַּר כַּחֲדָא: (ס)

{יב} גְּדִלִים תַּעֲשֶׂה לָּךְ עַל אַרְבַּע כַּנְפוֹת כְּסוּתְךָ אֲשֶׁר תְּכַסֶּה בָּהּ:  מ  גְּדִלִים תַּעֲשֶׂה לָּךְ עַל אַרְבַּע כַּנְפוֹת כְּסוּתְךָ אֲשֶׁר תְּכַסֶּה בָּהּ:  ת  כְּרֻסְפְּדִין תַּעְבֶּד לָךְ עַל אַרְבַּע כַּנְפֵי כְסוּתָךְ דִי תְכַסֵי בַהּ: (ס)

{יג} כִּי יִקַּח אִישׁ אִשָּׁה וּבָא אֵלֶיהָ וּשְׂנֵאָהּ:  מ  כִּי יִקַּח אִישׁ אִשָּׁה וּבָא אֵלֶיהָ וּשְׂנֵאָהּ:  ת  אֲרֵי יִסַב גְבַר אִתְּתָא וְיֵעוּל לְוָתַהּ וְיִסְנִנַהּ:

{יד} וְשָׂם לָהּ עֲלִילֹת דְּבָרִים וְהוֹצִיא עָלֶיהָ שֵׁם רָע וְאָמַר אֶת הָאִשָּׁה הַזֹּאת לָקַחְתִּי וָאֶקְרַב אֵלֶיהָ וְלֹא מָצָאתִי לָהּ בְּתוּלִים:  מ  וְשָׂם לָהּ עֲלִילֹת דְּבָרִים וְהוֹצִיא עָלֶיהָ שֵׁם רָע וְאָמַר אֶת הָאִשָּׁה הַזֹּאת לָקַחְתִּי וָאֶקְרַב אֵלֶיהָ וְלֹא מָצָאתִי לָהּ בְּתוּלִים:  ת  וִישַׁוִי לַהּ תַּסְקוֹפֵי מִלִין וְיַפֵּק עֲלַהּ שׁוּם בִּישׁ וְיֵימַר יָת אִתְּתָא הָדָא נְסֵבִית וְעַלִית לְוָתַהּ וְלָא אַשְׁכָּחִית לַהּ בְּתוּלִין:

{טו} וְלָקַח אֲבִי (הנער) הַנַּעֲרָה וְאִמָּהּ וְהוֹצִיאוּ אֶת בְּתוּלֵי (הנער) הַנַּעֲרָה אֶל זִקְנֵי הָעִיר הַשָּׁעְרָה:  מ  וְלָקַח אֲבִי (הנער) הַנַּעֲרָה וְאִמָּהּ וְהוֹצִיאוּ אֶת בְּתוּלֵי (הנער) הַנַּעֲרָה אֶל זִקְנֵי הָעִיר הַשָּׁעְרָה:  ת  וְיִסַב אֲבוּהָא (נ''א אֲבוּהִי) דְעוּלֶמְתָּא וְאִמַהּ וְיַפְּקוּן יָת בְּתוּלֵי עוּלֶמְתָּא לָקֳדָם סָבֵי קַרְתָּא לִתְרַע בֵּית דִין אַתְרָא:

{טז} וְאָמַר אֲבִי (הנער) הַנַּעֲרָה אֶל הַזְּקֵנִים אֶת בִּתִּי נָתַתִּי לָאִישׁ הַזֶּה לְאִשָּׁה וַיִּשְׂנָאֶהָ:  מ  וְאָמַר אֲבִי (הנער) הַנַּעֲרָה אֶל הַזְּקֵנִים אֶת בִּתִּי נָתַתִּי לָאִישׁ הַזֶּה לְאִשָּׁה וַיִּשְׂנָאֶהָ:  ת  וְיֵימַר אֲבוּהָא דְעוּלֶמְתָּא לְסָבַיָא יָת בְּרַתִּי יְהָבִית לְגַבְרָא הָדֵין לְאִנְתּוּ וּסְנִנַהּ:

{יז} וְהִנֵּה הוּא שָׂם עֲלִילֹת דְּבָרִים לֵאמֹר לֹא מָצָאתִי לְבִתְּךָ בְּתוּלִים וְאֵלֶּה בְּתוּלֵי בִתִּי וּפָרְשׂוּ הַשִּׂמְלָה לִפְנֵי זִקְנֵי הָעִיר:  מ  וְהִנֵּה הוּא שָׂם עֲלִילֹת דְּבָרִים לֵאמֹר לֹא מָצָאתִי לְבִתְּךָ בְּתוּלִים וְאֵלֶּה בְּתוּלֵי בִתִּי וּפָרְשׂוּ הַשִּׂמְלָה לִפְנֵי זִקְנֵי הָעִיר:  ת  וְהָא הוּא שַׁוִי תַּסְקוֹפֵי מִלִין לְמֵימַר לָא אַשְׁכָּחִית לִבְרַתָּךְ בְּתוּלִין וְאִלֵין בְּתוּלֵי בְרַתִּי וְיִפְרְסוּן שׁוֹשִׁיפָא קֳדָם סָבֵי קַרְתָּא:

{יח} וְלָקְחוּ זִקְנֵי הָעִיר הַהִוא אֶת הָאִישׁ וְיִסְּרוּ אֹתוֹ:  מ  וְלָקְחוּ זִקְנֵי הָעִיר הַהִוא אֶת הָאִישׁ וְיִסְּרוּ אֹתוֹ:  ת  וְיִדְבְּרוּן סָבֵי קַרְתָּא הַהִיא יָת גַבְרָא וְיַלְקוּן יָתֵהּ:

{יט} וְעָנְשׁוּ אֹתוֹ מֵאָה כֶסֶף וְנָתְנוּ לַאֲבִי הַנַּעֲרָה כִּי הוֹצִיא שֵׁם רָע עַל בְּתוּלַת יִשְׂרָאֵל וְלוֹ תִהְיֶה לְאִשָּׁה לֹא יוּכַל לְשַׁלְּחָהּ כָּל יָמָיו:  מ  וְעָנְשׁוּ אֹתוֹ מֵאָה כֶסֶף וְנָתְנוּ לַאֲבִי הַנַּעֲרָה כִּי הוֹצִיא שֵׁם רָע עַל בְּתוּלַת יִשְׂרָאֵל וְלוֹ תִהְיֶה לְאִשָּׁה לֹא יוּכַל לְשַׁלְּחָהּ כָּל יָמָיו:  ת  וְיִגְבּוּן מִנֵהּ מְאָה סִלְעִין דִכְסָף וְיִתְּנוּן לַאֲבוּהָא דְעוּלֶמְתָּא אֲרֵי אַפֵּק שׁוּם בִּישׁ עַל בְּתוּלְתָּא דְיִשְׂרָאֵל וְלֵהּ תְּהֵי לְאִנְתּוּ לֵית לֵהּ רְשׁוּ לְמִפְטְּרַהּ כָּל יוֹמוֹהִי: (ס)

{כ} וְאִם אֱמֶת הָיָה הַדָּבָר הַזֶּה לֹא נִמְצְאוּ בְתוּלִים (לנער) לַנַּעֲרָה:  מ  וְאִם אֱמֶת הָיָה הַדָּבָר הַזֶּה לֹא נִמְצְאוּ בְתוּלִים (לנער) לַנַּעֲרָה:  ת  וְאִם קֻשְׁטָא הֲוָה פִּתְגָמָא הָדֵין לָא אִשְׁתְּכָחוּ בְתוּלִין לְעוּלֶמְתָּא:

{כא} וְהוֹצִיאוּ אֶת (הנער) הַנַּעֲרָה אֶל פֶּתַח בֵּית אָבִיהָ וּסְקָלוּהָ אַנְשֵׁי עִירָהּ בָּאֲבָנִים וָמֵתָה כִּי עָשְׂתָה נְבָלָה בְּיִשְׂרָאֵל לִזְנוֹת בֵּית אָבִיהָ וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  מ  וְהוֹצִיאוּ אֶת (הנער) הַנַּעֲרָה אֶל פֶּתַח בֵּית אָבִיהָ וּסְקָלוּהָ אַנְשֵׁי עִירָהּ בָּאֲבָנִים וָמֵתָה כִּי עָשְׂתָה נְבָלָה בְּיִשְׂרָאֵל לִזְנוֹת בֵּית אָבִיהָ וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  ת  וְיַפְּקוּן יָת עוּלֶמְתָּא לִתְרַע בֵּית אָבוּהָא וְיִרְגְמֻנַהּ אֱנָשֵׁי קַרְתַּהּ בְּאַבְנַיָא וּתְמוּת אֲרֵי עָבְדַת קְלָנָא בְּיִשְׂרָאֵל לְזַנָאָה בֵּית אֲבוּהָא וּתְפַלֵי עָבֵד דְבִישׁ מִבֵּינָךְ: (ס)

{כב} כִּי יִמָּצֵא אִישׁ שֹׁכֵב עִם אִשָּׁה בְעֻלַת בַּעַל וּמֵתוּ גַּם שְׁנֵיהֶם הָאִישׁ הַשֹּׁכֵב עִם הָאִשָּׁה וְהָאִשָּׁה וּבִעַרְתָּ הָרָע מִיִּשְׂרָאֵל:  מ  כִּי יִמָּצֵא אִישׁ שֹׁכֵב עִם אִשָּׁה בְעֻלַת בַּעַל וּמֵתוּ גַּם שְׁנֵיהֶם הָאִישׁ הַשֹּׁכֵב עִם הָאִשָּׁה וְהָאִשָּׁה וּבִעַרְתָּ הָרָע מִיִּשְׂרָאֵל:  ת  אֲרֵי יִשְׁתְּכַח גְבַר דְשָׁכֵב עִם אִתְּתָא אִתַּת גְבַר וְיִתְקַטְלוּן אַף תַּרְוֵיהוֹן גַבְרָא דְשָׁכֵב עִם אִתְּתָא וְאִתְּתָא וּתְפַלֵי עָבֵד דְבִישׁ מִיִשְׂרָאֵל: (ס)

{כג} כִּי יִהְיֶה (נער) נַעֲרָה בְתוּלָה מְאֹרָשָׂה לְאִישׁ וּמְצָאָהּ אִישׁ בָּעִיר וְשָׁכַב עִמָּהּ:  מ  כִּי יִהְיֶה (נער) נַעֲרָה בְתוּלָה מְאֹרָשָׂה לְאִישׁ וּמְצָאָהּ אִישׁ בָּעִיר וְשָׁכַב עִמָּהּ:  ת  אֲרֵי תְהֵי עוּלֶמְתָּא בְתֻלְתָּא דִמְאָרְסָא לִגְבַר וְיַשְׁכְּחִנַהּ גְבַר בְּקַרְתָּא וְיִשְׁכּוּב עִמַהּ:

{כד} וְהוֹצֵאתֶם אֶת שְׁנֵיהֶם אֶל שַׁעַר הָעִיר הַהִוא וּסְקַלְתֶּם אֹתָם בָּאֲבָנִים וָמֵתוּ אֶת (הנער) הַנַּעֲרָה עַל דְּבַר אֲשֶׁר לֹא צָעֲקָה בָעִיר וְאֶת הָאִישׁ עַל דְּבַר אֲשֶׁר עִנָּה אֶת אֵשֶׁת רֵעֵהוּ וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  מ  וְהוֹצֵאתֶם אֶת שְׁנֵיהֶם אֶל שַׁעַר הָעִיר הַהִוא וּסְקַלְתֶּם אֹתָם בָּאֲבָנִים וָמֵתוּ אֶת (הנער) הַנַּעֲרָה עַל דְּבַר אֲשֶׁר לֹא צָעֲקָה בָעִיר וְאֶת הָאִישׁ עַל דְּבַר אֲשֶׁר עִנָּה אֶת אֵשֶׁת רֵעֵהוּ וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  ת  וְתַפְּקוּן יָת תַּרְוֵיהוֹן לִתְרַע קַרְתָּא הַהִיא וְתִרְגְמוּן יָתְהוֹן בְּאַבְנַיָא וִימוּתוּן יָת עוּלֶמְתָּא עַל עֵסַק דִי לָא צְוָחַת בְּקַרְתָּא וְיָת גַבְרָא עַל עֵסַק דְעַנִי יָת אִתַּת חַבְרֵהּ וּתְפַלֵי עָבֵד דְבִּישׁ מִבֵּינָךְ: (ס)

{כה} וְאִם בַּשָּׂדֶה יִמְצָא הָאִישׁ אֶת (הנער) הַנַּעֲרָה הַמְאֹרָשָׂה וְהֶחֱזִיק בָּהּ הָאִישׁ וְשָׁכַב עִמָּהּ וּמֵת הָאִישׁ אֲשֶׁר שָׁכַב עִמָּהּ לְבַדּוֹ:  מ  וְאִם בַּשָּׂדֶה יִמְצָא הָאִישׁ אֶת (הנער) הַנַּעֲרָה הַמְאֹרָשָׂה וְהֶחֱזִיק בָּהּ הָאִישׁ וְשָׁכַב עִמָּהּ וּמֵת הָאִישׁ אֲשֶׁר שָׁכַב עִמָּהּ לְבַדּוֹ:  ת  וְאִם בְּחַקְלָא יַשְׁכַּח גְבַר יָת עוּלֶמְתָּא דִמְאָרְסָא וְיִתְקֶף בַּהּ גַבְרָא וְיִשְׁכּוּב עִמַהּ וְיִתְקְטֵּל גַבְרָא דְשָׁכֵב עִמַהּ בִּלְחוֹדוֹהִי:

{כו} (ולנער) וְלַנַּעֲרָה לֹא תַעֲשֶׂה דָבָר אֵין (לנער) לַנַּעֲרָה חֵטְא מָוֶת כִּי כַּאֲשֶׁר יָקוּם אִישׁ עַל רֵעֵהוּ וּרְצָחוֹ נֶפֶשׁ כֵּן הַדָּבָר הַזֶּה:  מ  (ולנער) וְלַנַּעֲרָה לֹא תַעֲשֶׂה דָבָר אֵין (לנער) לַנַּעֲרָה חֵטְא מָוֶת כִּי כַּאֲשֶׁר יָקוּם אִישׁ עַל רֵעֵהוּ וּרְצָחוֹ נֶפֶשׁ כֵּן הַדָּבָר הַזֶּה:  ת  וּלְעוּלֶמְתָּא לָא תַעְבֵּד מִדַעַם לֵית לְעוּלֶמְתָּא חוֹבַת דִין דִקְטוֹל אֲרֵי כְּמָא דִיקוּם גַבְרָא עַל חַבְרֵהּ וְיִקְטְּלִנֵהּ נְפַשׁ כֵּן פִּתְגָמָא הָדֵין:

{כז} כִּי בַשָּׂדֶה מְצָאָהּ צָעֲקָה (הנער) הַנַּעֲרָה הַמְאֹרָשָׂה וְאֵין מוֹשִׁיעַ לָהּ:  מ  כִּי בַשָּׂדֶה מְצָאָהּ צָעֲקָה (הנער) הַנַּעֲרָה הַמְאֹרָשָׂה וְאֵין מוֹשִׁיעַ לָהּ:  ת  אֲרֵי בְחַקְלָא אַשְׁכְּחַהּ צְוָחַת עוּלֶמְתָּא דִמְאָרְסָא וְלֵית דְפָרִיק לַהּ: (ס)

{כח} כִּי יִמְצָא אִישׁ (נער) נַעֲרָה בְתוּלָה אֲשֶׁר לֹא אֹרָשָׂה וּתְפָשָׂהּ וְשָׁכַב עִמָּהּ וְנִמְצָאוּ:  מ  כִּי יִמְצָא אִישׁ (נער) נַעֲרָה בְתוּלָה אֲשֶׁר לֹא אֹרָשָׂה וּתְפָשָׂהּ וְשָׁכַב עִמָּהּ וְנִמְצָאוּ:  ת  אֲרֵי יַשְׁכַּח גְבַר עוּלֶמְתָּא בְתֻלְתָּא דִי לָא מְאָרְסָא וְיֶחֱדִנַהּ וְיִשְׁכּוּב עִמַהּ וְיִשְׁתַּכְּחוּן:

{כט} וְנָתַן הָאִישׁ הַשֹּׁכֵב עִמָּהּ לַאֲבִי (הנער) הַנַּעֲרָה חֲמִשִּׁים כָּסֶף וְלוֹ תִהְיֶה לְאִשָּׁה תַּחַת אֲשֶׁר עִנָּהּ לֹא יוּכַל שַׁלְּחָהּ כָּל יָמָיו:  מ  וְנָתַן הָאִישׁ הַשֹּׁכֵב עִמָּהּ לַאֲבִי (הנער) הַנַּעֲרָה חֲמִשִּׁים כָּסֶף וְלוֹ תִהְיֶה לְאִשָּׁה תַּחַת אֲשֶׁר עִנָּהּ לֹא יוּכַל שַׁלְּחָהּ כָּל יָמָיו:  ת  וְיִתֵּן גַבְרָא דְיִשְׁכֻּב עִמַהּ לַאֲבוּהָא דְעוּלֶמְתָּא חַמְשִׁין סִלְעִין דִכְסַף וְלֵהּ תְּהֵי לְאִנְתּוּ חֳלַף דִי עַנְיַהּ לֵית לֵהּ רְשׁוּ לְמִפְטְרַהּ כָּל יוֹמוֹהִי: (ס)




דברים פרק-כג

{א} לֹא יִקַּח אִישׁ אֶת אֵשֶׁת אָבִיו וְלֹא יְגַלֶּה כְּנַף אָבִיו:  מ  לֹא יִקַּח אִישׁ אֶת אֵשֶׁת אָבִיו וְלֹא יְגַלֶּה כְּנַף אָבִיו:  ת  לָא יִסַב גְבַר יָת אִתַּת אֲבוּהִי וְלָא יְגַלֵי כַּנְפָא דַאֲבוּהִי: (ס)

{ב} לֹא יָבֹא פְצוּעַ דַּכָּא וּכְרוּת שָׁפְכָה בִּקְהַל יְהוָה:  מ  לֹא יָבֹא פְצוּעַ דַּכָּא וּכְרוּת שָׁפְכָה בִּקְהַל יְהוָה:  ת  לָא יִדְכֵּי דְפָסִיק וְדִמְחַבַּל לְמֵעַל בִּקְהָלָא דַיְיָ: (ס)

{ג} לֹא יָבֹא מַמְזֵר בִּקְהַל יְהוָה גַּם דּוֹר עֲשִׂירִי לֹא יָבֹא לוֹ בִּקְהַל יְהוָה:  מ  לֹא יָבֹא מַמְזֵר בִּקְהַל יְהוָה גַּם דּוֹר עֲשִׂירִי לֹא יָבֹא לוֹ בִּקְהַל יְהוָה:  ת  לָא יִדְכֵּי מַמְזֵרָא לְמֵעַל בִּקְהָלָא דַיְיָ אַף דָרָא עֲשִׂירָאָה לָא יִדְכֵּי לֵהּ לְמֵעַל בִּקְהָלָא דַיְיָ: (ס)

{ד} לֹא יָבֹא עַמּוֹנִי וּמוֹאָבִי בִּקְהַל יְהוָה גַּם דּוֹר עֲשִׂירִי לֹא יָבֹא לָהֶם בִּקְהַל יְהוָה עַד עוֹלָם:  מ  לֹא יָבֹא עַמּוֹנִי וּמוֹאָבִי בִּקְהַל יְהוָה גַּם דּוֹר עֲשִׂירִי לֹא יָבֹא לָהֶם בִּקְהַל יְהוָה עַד עוֹלָם:  ת  לָא יִדְכּוּן עַמוֹנָאֵי וּמוֹאָבָאֵי לְמֵעַל בִּקְהָלָא דַיְיָ אַף דָרָא עֲשִׂירָאָה לָא יִדְכֵּי לְהוֹן לְמֵעַל בִּקְהָלָא דַיְיָ עַד עָלָם:

{ה} עַל דְּבַר אֲשֶׁר לֹא קִדְּמוּ אֶתְכֶם בַּלֶּחֶם וּבַמַּיִם בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם וַאֲשֶׁר שָׂכַר עָלֶיךָ אֶת בִּלְעָם בֶּן בְּעוֹר מִפְּתוֹר אֲרַם נַהֲרַיִם לְקַלְלֶךָּ:  מ  עַל דְּבַר אֲשֶׁר לֹא קִדְּמוּ אֶתְכֶם בַּלֶּחֶם וּבַמַּיִם בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם וַאֲשֶׁר שָׂכַר עָלֶיךָ אֶת בִּלְעָם בֶּן בְּעוֹר מִפְּתוֹר אֲרַם נַהֲרַיִם לְקַלְלֶךָּ:  ת  עַל עֵסַק דִי לָא עֲרָעוּ יָתְכוֹן בְּלַחְמָא וּבְמַיָא בְּאָרְחָא בְּמִפָּקְכוֹן מִמִצְרָיִם וְדִי אֲגַר עֲלָךְ יָת בִּלְעָם בַּר בְּעוֹר מִפְּתוֹר אֲרַם דִי עַל פְּרָת לְלַטָּיוּתָךְ:

{ו} וְלֹא אָבָה יְהוָה אֱלֹהֶיךָ לִשְׁמֹעַ אֶל בִּלְעָם וַיַּהֲפֹךְ יְהוָה אֱלֹהֶיךָ לְּךָ אֶת הַקְּלָלָה לִבְרָכָה כִּי אֲהֵבְךָ יְהוָה אֱלֹהֶיךָ:  מ  וְלֹא אָבָה יְהוָה אֱלֹהֶיךָ לִשְׁמֹעַ אֶל בִּלְעָם וַיַּהֲפֹךְ יְהוָה אֱלֹהֶיךָ לְּךָ אֶת הַקְּלָלָה לִבְרָכָה כִּי אֲהֵבְךָ יְהוָה אֱלֹהֶיךָ:  ת  וְלָא אָבֵי יְיָ אֱלָהָךְ לְקַבָּלָא מִן בִּלְעָם וַהֲפַךְ יְיָ אֱלָהָךְ לָךְ יָת לְוָטִין לְבִרְכָן אֲרֵי רָחֲמָךְ יְיָ אֱלָהָךְ:

{ז} לֹא תִדְרֹשׁ שְׁלֹמָם וְטֹבָתָם כָּל יָמֶיךָ לְעוֹלָם:  מ  לֹא תִדְרֹשׁ שְׁלֹמָם וְטֹבָתָם כָּל יָמֶיךָ לְעוֹלָם:  ת  לָא תִתְבַּע שְׁלָמְהוֹן וְטָבָתְהוֹן כָּל יוֹמָיךְ לְעָלָם: (ס)

{ח}  רביעי  לֹא תְתַעֵב אֲדֹמִי כִּי אָחִיךָ הוּא לֹא תְתַעֵב מִצְרִי כִּי גֵר הָיִיתָ בְאַרְצוֹ:  מ  לֹא תְתַעֵב אֲדֹמִי כִּי אָחִיךָ הוּא לֹא תְתַעֵב מִצְרִי כִּי גֵר הָיִיתָ בְאַרְצוֹ:  ת  לָא תְרַחַק אֱדוֹמָאָה אֲרֵי אֲחוּךְ הוּא לָּא תְרַחַק מִצְרָאָה אֲרֵי דַיָר הֲוֵיתָא בְאַרְעֵהּ:

{ט} בָּנִים אֲשֶׁר יִוָּלְדוּ לָהֶם דּוֹר שְׁלִישִׁי יָבֹא לָהֶם בִּקְהַל יְהוָה:  מ  בָּנִים אֲשֶׁר יִוָּלְדוּ לָהֶם דּוֹר שְׁלִישִׁי יָבֹא לָהֶם בִּקְהַל יְהוָה:  ת  בְּנִין דִי יִתְיְלִידוּ לְהוֹן דָרָא תְלִיתָאָה יִדְכֵּי לְהוֹן לְמֵעַל בִּקְהָלָה דַיְיָ: (ס)

{י} כִּי תֵצֵא מַחֲנֶה עַל אֹיְבֶיךָ וְנִשְׁמַרְתָּ מִכֹּל דָּבָר רָע:  מ  כִּי תֵצֵא מַחֲנֶה עַל אֹיְבֶיךָ וְנִשְׁמַרְתָּ מִכֹּל דָּבָר רָע:  ת  אֲרֵי תִפּוֹק מַשְׁרִיתָא עַל בַּעֲלֵי דְבָבָךְ וְתִסְתַּמַר מִכֹּל מִדַעַם בִּישׁ:

{יא} כִּי יִהְיֶה בְךָ אִישׁ אֲשֶׁר לֹא יִהְיֶה טָהוֹר מִקְּרֵה לָיְלָה וְיָצָא אֶל מִחוּץ לַמַּחֲנֶה לֹא יָבֹא אֶל תּוֹךְ הַמַּחֲנֶה:  מ  כִּי יִהְיֶה בְךָ אִישׁ אֲשֶׁר לֹא יִהְיֶה טָהוֹר מִקְּרֵה לָיְלָה וְיָצָא אֶל מִחוּץ לַמַּחֲנֶה לֹא יָבֹא אֶל תּוֹךְ הַמַּחֲנֶה:  ת  אֲרֵי יְהֵי בָךְ גְבַר דִי לָא יְהֵּי דְכֵי מִקְרֵהּ לֵילְיָא וְיִפּוֹק לְמִבָּרָא לְמַשְׁרִיתָא לָא יֵעוּל לְגוֹ מַשְׁרִיתָא:

{יב} וְהָיָה לִפְנוֹת עֶרֶב יִרְחַץ בַּמָּיִם וּכְבֹא הַשֶּׁמֶשׁ יָבֹא אֶל תּוֹךְ הַמַּחֲנֶה:  מ  וְהָיָה לִפְנוֹת עֶרֶב יִרְחַץ בַּמָּיִם וּכְבֹא הַשֶּׁמֶשׁ יָבֹא אֶל תּוֹךְ הַמַּחֲנֶה:  ת  וִיהֵי לְמִפְנֵי רַמְשָׁא יַסְחֵי בְמַיָא וּכְמֵעַל שִׁמְשָׁא יֵעוּל לְגוֹ מַשְׁרִיתָא:

{יג} וְיָד תִּהְיֶה לְךָ מִחוּץ לַמַּחֲנֶה וְיָצָאתָ שָׁמָּה חוּץ:  מ  וְיָד תִּהְיֶה לְךָ מִחוּץ לַמַּחֲנֶה וְיָצָאתָ שָׁמָּה חוּץ:  ת  וַאֲתַר מְתַקֵן יְהֵי לָךְ מִבָּרָא לְמַשְׁרִיתָא וְתִפּוֹק תַּמָן לְבָרָא:

{יד} וְיָתֵד תִּהְיֶה לְךָ עַל אֲזֵנֶךָ וְהָיָה בְּשִׁבְתְּךָ חוּץ וְחָפַרְתָּה בָהּ וְשַׁבְתָּ וְכִסִּיתָ אֶת צֵאָתֶךָ:  מ  וְיָתֵד תִּהְיֶה לְךָ עַל אֲזֵנֶךָ וְהָיָה בְּשִׁבְתְּךָ חוּץ וְחָפַרְתָּה בָהּ וְשַׁבְתָּ וְכִסִּיתָ אֶת צֵאָתֶךָ:  ת  וְסִכְּתָא תְּהֵי לָךְ עַל זֵינָךְ וִיהֵי בְּמִיתְבָךְ לְבָרָא וְתַחְפַּר בַּהּ וּתְתוּב וּתְכַסֵי יָת מִפְּקָתָךְ:

{טו} כִּי יְהוָה אֱלֹהֶיךָ מִתְהַלֵּךְ בְּקֶרֶב מַחֲנֶךָ לְהַצִּילְךָ וְלָתֵת אֹיְבֶיךָ לְפָנֶיךָ וְהָיָה מַחֲנֶיךָ קָדוֹשׁ וְלֹא יִרְאֶה בְךָ עֶרְוַת דָּבָר וְשָׁב מֵאַחֲרֶיךָ:  מ  כִּי יְהוָה אֱלֹהֶיךָ מִתְהַלֵּךְ בְּקֶרֶב מַחֲנֶךָ לְהַצִּילְךָ וְלָתֵת אֹיְבֶיךָ לְפָנֶיךָ וְהָיָה מַחֲנֶיךָ קָדוֹשׁ וְלֹא יִרְאֶה בְךָ עֶרְוַת דָּבָר וְשָׁב מֵאַחֲרֶיךָ:  ת  אֲרֵי יְיָ אֱלָהָךְ שְׁכִינְתֵּהּ מְהַלְכָא בְּגוֹ מַשְׁרִיתָךְ לְשֵׁזָבוּתָךְ וּלְמִמְסַר בַּעֲלֵי דְבָבָךְ קֳדָמָךְ וּתְהֵי מַשְׁרִיתָךְ קַדִישָׁא וִלָא יִתְחֲזֵי בָךְ עֲבֵרַת פִּתְגָם וִיתוּב מֵימְרֵהּ מִלְאוֹטָבָא לָךְ: (ס)

{טז} לֹא תַסְגִּיר עֶבֶד אֶל אֲדֹנָיו אֲשֶׁר יִנָּצֵל אֵלֶיךָ מֵעִם אֲדֹנָיו:  מ  לֹא תַסְגִּיר עֶבֶד אֶל אֲדֹנָיו אֲשֶׁר יִנָּצֵל אֵלֶיךָ מֵעִם אֲדֹנָיו:  ת  לָא תִמְסַר עֶבֶד עַמְמִין לְיַד רִבּוֹנֵהּ דְיִשְׁתֵּזֵב לְוָתָךְ מִלְוָת רִבּוֹנֵהּ:

{יז} עִמְּךָ יֵשֵׁב בְּקִרְבְּךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר בְּאַחַד שְׁעָרֶיךָ בַּטּוֹב לוֹ לֹא תּוֹנֶנּוּ:  מ  עִמְּךָ יֵשֵׁב בְּקִרְבְּךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר בְּאַחַד שְׁעָרֶיךָ בַּטּוֹב לוֹ לֹא תּוֹנֶנּוּ:  ת  עִמָךְ יִתֵּב בֵּינָךְ בְּאַתְרָא דִי יִתִּרְעֵי בַּחֲדָא מִן קִרְוָיךְ בִּדְיֵיטַּב לֵהּ לָא תוֹנִנֵהּ: (ס)

{יח} לֹא תִהְיֶה קְדֵשָׁה מִבְּנוֹת יִשְׂרָאֵל וְלֹא יִהְיֶה קָדֵשׁ מִבְּנֵי יִשְׂרָאֵל:  מ  לֹא תִהְיֶה קְדֵשָׁה מִבְּנוֹת יִשְׂרָאֵל וְלֹא יִהְיֶה קָדֵשׁ מִבְּנֵי יִשְׂרָאֵל:  ת  לָא תְהֵי אִתְּתָא מִבְּנַת יִשְׂרָאֵל לִגְבָר עָבֵד וְלָא יִסַב גַבְרָא מִבְּנֵי יִשְׂרָאֵל אִתְּתָא אָמָא:

{יט} לֹא תָבִיא אֶתְנַן זוֹנָה וּמְחִיר כֶּלֶב בֵּית יְהוָה אֱלֹהֶיךָ לְכָל נֶדֶר כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ גַּם שְׁנֵיהֶם:  מ  לֹא תָבִיא אֶתְנַן זוֹנָה וּמְחִיר כֶּלֶב בֵּית יְהוָה אֱלֹהֶיךָ לְכָל נֶדֶר כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ גַּם שְׁנֵיהֶם:  ת  לָא תָעֵל אֲגַר זַנִיתָא וְחוּלְפַן כַּלְבָא לְבֵית מַקְדְשָׁא דַיְיָ אֱלָהָךְ לְכָל נִדְרָא אֲרֵי מְרָחָק קֳדָם יְיָ אֱלָהָךְ אַף תַּרְוֵיהוֹן: (ס)

{כ} לֹא תַשִּׁיךְ לְאָחִיךָ נֶשֶׁךְ כֶּסֶף נֶשֶׁךְ אֹכֶל נֶשֶׁךְ כָּל דָּבָר אֲשֶׁר יִשָּׁךְ:  מ  לֹא תַשִּׁיךְ לְאָחִיךָ נֶשֶׁךְ כֶּסֶף נֶשֶׁךְ אֹכֶל נֶשֶׁךְ כָּל דָּבָר אֲשֶׁר יִשָּׁךְ:  ת  לָא תְרַבֵּי לַאֲחוּךְ רִבִּית כְּסַף רִבִּית עִבוּר רִבִּית כָּל מִדַעַם דְמִתְרַבֵּי:

{כא} לַנָּכְרִי תַשִּׁיךְ וּלְאָחִיךָ לֹא תַשִּׁיךְ לְמַעַן יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל מִשְׁלַח יָדֶךָ עַל הָאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ:  מ  לַנָּכְרִי תַשִּׁיךְ וּלְאָחִיךָ לֹא תַשִּׁיךְ לְמַעַן יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל מִשְׁלַח יָדֶךָ עַל הָאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ:  ת  לְבַר עַמְמִין תְּרַבֵּי וְלַאֲחוּךְ לָא תְרַבֵּי בְּדִיל דִיבָרֵכִנָךְ יְיָ אֱלָהָךְ בְּכֹל אוֹשָׁטוּת יְדָךְ עַל אַרְעָא דִי אַתְּ עָלֵל:לתַמָן לְמֵירְתַהּ: (ס)

{כב} כִּי תִדֹּר נֶדֶר לַיהוָה אֱלֹהֶיךָ לֹא תְאַחֵר לְשַׁלְּמוֹ כִּי דָּרֹשׁ יִדְרְשֶׁנּוּ יְהוָה אֱלֹהֶיךָ מֵעִמָּךְ וְהָיָה בְךָ חֵטְא:  מ  כִּי תִדֹּר נֶדֶר לַיהוָה אֱלֹהֶיךָ לֹא תְאַחֵר לְשַׁלְּמוֹ כִּי דָּרֹשׁ יִדְרְשֶׁנּוּ יְהוָה אֱלֹהֶיךָ מֵעִמָּךְ וְהָיָה בְךָ חֵטְא:  ת  אֲרֵי תִדַר נְדַר קֳדָם יְיָ אֱלָהָךְ לָא תְאַחַר לְשַׁלָמוּתֵהּ אֲרֵי מִתְבַּע יִתְבְּעִנֵהּ יְיָ אֱלָהָךְ מִנָךְ וִיהֵי בָךְ חוֹבָא:

{כג} וְכִי תֶחְדַּל לִנְדֹּר לֹא יִהְיֶה בְךָ חֵטְא:  מ  וְכִי תֶחְדַּל לִנְדֹּר לֹא יִהְיֶה בְךָ חֵטְא:  ת  וַאֲרֵי תִתִּמְנַע לְמִנְדַר לָא יְהֵי בָךְ חוֹבָא:

{כד} מוֹצָא שְׂפָתֶיךָ תִּשְׁמֹר וְעָשִׂיתָ כַּאֲשֶׁר נָדַרְתָּ לַיהוָה אֱלֹהֶיךָ נְדָבָה אֲשֶׁר דִּבַּרְתָּ בְּפִיךָ:  מ  מוֹצָא שְׂפָתֶיךָ תִּשְׁמֹר וְעָשִׂיתָ כַּאֲשֶׁר נָדַרְתָּ לַיהוָה אֱלֹהֶיךָ נְדָבָה אֲשֶׁר דִּבַּרְתָּ בְּפִיךָ:  ת  אַפָּקוּת סִפְוָתָךְ תִּטַר וְתַעְבֵּד כְּמָא דִי נְדַרְתָּא קֳדָם יְיָ אֱלָהָךְ נְדַבְתָּא דִי מַלֶלְתָּא בְּפוּמָךְ: (ס)

{כה}  חמישי  כִּי תָבֹא בְּכֶרֶם רֵעֶךָ וְאָכַלְתָּ עֲנָבִים כְּנַפְשְׁךָ שָׂבְעֶךָ וְאֶל כֶּלְיְךָ לֹא תִתֵּן:  מ  כִּי תָבֹא בְּכֶרֶם רֵעֶךָ וְאָכַלְתָּ עֲנָבִים כְּנַפְשְׁךָ שָׂבְעֶךָ וְאֶל כֶּלְיְךָ לֹא תִתֵּן:  ת  אֲרֵי תִתָּגַר בְּכַרְמָא דְחַבְרָךְ וְתֵיכוּל עִנְבִין כְּנַפְשָׁךְ שִׂבְעָךְ וּלְמָאנָךְ לָא תִתֵּן: (ס)

{כו} כִּי תָבֹא בְּקָמַת רֵעֶךָ וְקָטַפְתָּ מְלִילֹת בְּיָדֶךָ וְחֶרְמֵשׁ לֹא תָנִיף עַל קָמַת רֵעֶךָ:  מ  כִּי תָבֹא בְּקָמַת רֵעֶךָ וְקָטַפְתָּ מְלִילֹת בְּיָדֶךָ וְחֶרְמֵשׁ לֹא תָנִיף עַל קָמַת רֵעֶךָ:  ת  אֲרֵי תִתָּגַר בְּקַמְתָא דְחַבְרָךְ וְתִקְטוֹף מְלִילָן בִּידָךְ וּמַגְלָא לָא תְרִים עַל קַמְתָא דְחַבְרָךְ: (ס)




דברים פרק-כד

{א} כִּי יִקַּח אִישׁ אִשָּׁה וּבְעָלָהּ וְהָיָה אִם לֹא תִמְצָא חֵן בְּעֵינָיו כִּי מָצָא בָהּ עֶרְוַת דָּבָר וְכָתַב לָהּ סֵפֶר כְּרִיתֻת וְנָתַן בְּיָדָהּ וְשִׁלְּחָהּ מִבֵּיתוֹ:  מ  כִּי יִקַּח אִישׁ אִשָּׁה וּבְעָלָהּ וְהָיָה אִם לֹא תִמְצָא חֵן בְּעֵינָיו כִּי מָצָא בָהּ עֶרְוַת דָּבָר וְכָתַב לָהּ סֵפֶר כְּרִיתֻת וְנָתַן בְּיָדָהּ וְשִׁלְּחָהּ מִבֵּיתוֹ:  ת  אֲרֵי יִסַב גְבַר אִתְּתָא וְיִבְעֲלִנַהּ וִיהֵי אִם לָא תַשְׁכַּח רַחֲמִין בְּעֵינוֹהִי אֲרֵי אַשְׁכַּח בַּהּ עֲבֵרַת פִתְגָם וְיִכְתּוֹב לַהּ גֵט פִּטוּרִין וִיהַב בִּידַהּ וְיִפְטְּרִנַהּ מִבֵּיתֵהּ:

{ב} וְיָצְאָה מִבֵּיתוֹ וְהָלְכָה וְהָיְתָה לְאִישׁ אַחֵר:  מ  וְיָצְאָה מִבֵּיתוֹ וְהָלְכָה וְהָיְתָה לְאִישׁ אַחֵר:  ת  וְתִפּוֹק מִבֵּיתֵהּ וּתְהַךְ וּתְהֵי לִגְבַר אָחֳרָן:

{ג} וּשְׂנֵאָהּ הָאִישׁ הָאַחֲרוֹן וְכָתַב לָהּ סֵפֶר כְּרִיתֻת וְנָתַן בְּיָדָהּ וְשִׁלְּחָהּ מִבֵּיתוֹ אוֹ כִי יָמוּת הָאִישׁ הָאַחֲרוֹן אֲשֶׁר לְקָחָהּ לוֹ לְאִשָּׁה:  מ  וּשְׂנֵאָהּ הָאִישׁ הָאַחֲרוֹן וְכָתַב לָהּ סֵפֶר כְּרִיתֻת וְנָתַן בְּיָדָהּ וְשִׁלְּחָהּ מִבֵּיתוֹ אוֹ כִי יָמוּת הָאִישׁ הָאַחֲרוֹן אֲשֶׁר לְקָחָהּ לוֹ לְאִשָּׁה:  ת  וְיִסְנִנַהּ גַבְרָא בַתְרָאָה וְיִכְתּוֹב לַהּ גֵטּ פִטּוּרִין וְיִתֵּן בִּידַהּ וְיִפְטְרִנַהּ מִבֵּיתֵהּ אוֹ אֲרֵי יְמוּת גַבְרָא בַתְרָאָה דְנָסְבַהּ לֵה לְאִנְתּוּ:

{ד} לֹא יוּכַל בַּעְלָהּ הָרִאשׁוֹן אֲשֶׁר שִׁלְּחָהּ לָשׁוּב לְקַחְתָּהּ לִהְיוֹת לוֹ לְאִשָּׁה אַחֲרֵי אֲשֶׁר הֻטַּמָּאָה כִּי תוֹעֵבָה הִוא לִפְנֵי יְהוָה וְלֹא תַחֲטִיא אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה:  מ  לֹא יוּכַל בַּעְלָהּ הָרִאשׁוֹן אֲשֶׁר שִׁלְּחָהּ לָשׁוּב לְקַחְתָּהּ לִהְיוֹת לוֹ לְאִשָּׁה אַחֲרֵי אֲשֶׁר הֻטַּמָּאָה כִּי תוֹעֵבָה הִוא לִפְנֵי יְהוָה וְלֹא תַחֲטִיא אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה:  ת  לֵית לֵהּ רְשׁוּ לְבַעֲלַהּ קַדְמָאָה דִי פָטְרַהּ לְמִתֻּב לְמִסְבַהּ לְמֶהֱוֵי לֵה לְאִנְתּוּ בָּתַר דִי אִסְתָּאָבַת אֲרֵי מְרָחָקָא הִיא קֳדָם יְיָ וְלָא תְחַיַב יָת אַרְעָא דִי יְיָ אֱלָהָךְ יָהֵב לָךְ אַחֲסָנָא: (ס)

{ה}  שישי  כִּי יִקַּח אִישׁ אִשָּׁה חֲדָשָׁה לֹא יֵצֵא בַּצָּבָא וְלֹא יַעֲבֹר עָלָיו לְכָל דָּבָר נָקִי יִהְיֶה לְבֵיתוֹ שָׁנָה אֶחָת וְשִׂמַּח אֶת אִשְׁתּוֹ אֲשֶׁר לָקָח:  מ  כִּי יִקַּח אִישׁ אִשָּׁה חֲדָשָׁה לֹא יֵצֵא בַּצָּבָא וְלֹא יַעֲבֹר עָלָיו לְכָל דָּבָר נָקִי יִהְיֶה לְבֵיתוֹ שָׁנָה אֶחָת וְשִׂמַּח אֶת אִשְׁתּוֹ אֲשֶׁר לָקָח:  ת  אֲרֵי יִסַב גְבַר אִתְּתָא חַדְתָּא לָא יִפּוֹק בְּחֵילָא וְלָא יֵעִבַר עֲלוֹהִי לְכָל מִדָעַם פָּנֵי יְהֵי לְבֵיתֵהּ שַׁתָּא חֲדָא וְיַחְדֵי יָת אִתְּתֵהּ דִי נְסַב:

{ו} לֹא יַחֲבֹל רֵחַיִם וָרָכֶב כִּי נֶפֶשׁ הוּא חֹבֵל:  מ  לֹא יַחֲבֹל רֵחַיִם וָרָכֶב כִּי נֶפֶשׁ הוּא חֹבֵל:  ת  לָא יִסַב מַשְׁכּוֹנָא רֵחַיָא וְרִכְבָּא אֲרֵי בְהוֹן מִתְעֲבֵד מָזוֹן לְכָל נְפַשׁ: (ס)

{ז} כִּי יִמָּצֵא אִישׁ גֹּנֵב נֶפֶשׁ מֵאֶחָיו מִבְּנֵי יִשְׂרָאֵל וְהִתְעַמֶּר בּוֹ וּמְכָרוֹ וּמֵת הַגַּנָּב הַהוּא וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  מ  כִּי יִמָּצֵא אִישׁ גֹּנֵב נֶפֶשׁ מֵאֶחָיו מִבְּנֵי יִשְׂרָאֵל וְהִתְעַמֶּר בּוֹ וּמְכָרוֹ וּמֵת הַגַּנָּב הַהוּא וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  ת  אֲרֵי יִשְׁתְּכַח גְבַר גָנֵב נַפְשָׁא מֵאֲחוֹהִי מִבְּנֵי יִשְׂרָאֵל וְיִתָּגַר בֵּהּ וִיזַבְּנִנֵהּ וְיִתְקְטֵל גַנָבָא הַהוּא וּתְפַלֵי עָבֵד דְבִישׁ מִבֵּינָךְ: (ס)

{ח} הִשָּׁמֶר בְּנֶגַע הַצָּרַעַת לִשְׁמֹר מְאֹד וְלַעֲשׂוֹת כְּכֹל אֲשֶׁר יוֹרוּ אֶתְכֶם הַכֹּהֲנִים הַלְוִיִּם כַּאֲשֶׁר צִוִּיתִם תִּשְׁמְרוּ לַעֲשׂוֹת:  מ  הִשָּׁמֶר בְּנֶגַע הַצָּרַעַת לִשְׁמֹר מְאֹד וְלַעֲשׂוֹת כְּכֹל אֲשֶׁר יוֹרוּ אֶתְכֶם הַכֹּהֲנִים הַלְוִיִּם כַּאֲשֶׁר צִוִּיתִם תִּשְׁמְרוּ לַעֲשׂוֹת:  ת  אִסְתַּמַר בְּמַכְתַּשׁ סְגִירוּ לְמִטַּר לַחֲדָא וּלְמֶעְבָּד כְּכֹל דִי יַלְפוּן יָתְכוֹן כָּהֲנַיָא לֵוָאֵי כְּמָא דִי פַקֶדְתִּנוּן תִּטְרוּן לְמֶעְבָּד:

{ט} זָכוֹר אֵת אֲשֶׁר עָשָׂה יְהוָה אֱלֹהֶיךָ לְמִרְיָם בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם:  מ  זָכוֹר אֵת אֲשֶׁר עָשָׂה יְהוָה אֱלֹהֶיךָ לְמִרְיָם בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם:  ת  הֱוֵי דְכִיר יָת דִי עֲבַד יְיָ אֱלָהָךְ לְמִרְיָם בְּאָרְחָא בְּמִפָקְכוֹן מִמִצְרָיִם: (ס)

{י} כִּי תַשֶּׁה בְרֵעֲךָ מַשַּׁאת מְאוּמָה לֹא תָבֹא אֶל בֵּיתוֹ לַעֲבֹט עֲבֹטוֹ:  מ  כִּי תַשֶּׁה בְרֵעֲךָ מַשַּׁאת מְאוּמָה לֹא תָבֹא אֶל בֵּיתוֹ לַעֲבֹט עֲבֹטוֹ:  ת  אֲרֵי תַרְשֵׁי בְחַבְרָךְ רְשׁוּ מִדָעַם לָא תֵעוּל לְבֵיתֵהּ לְמִסַב מַשְׁכּוֹנֵהּ:

{יא} בַּחוּץ תַּעֲמֹד וְהָאִישׁ אֲשֶׁר אַתָּה נֹשֶׁה בוֹ יוֹצִיא אֵלֶיךָ אֶת הַעֲבוֹט הַחוּצָה:  מ  בַּחוּץ תַּעֲמֹד וְהָאִישׁ אֲשֶׁר אַתָּה נֹשֶׁה בוֹ יוֹצִיא אֵלֶיךָ אֶת הַעֲבוֹט הַחוּצָה:  ת  בְּבָרָא תְּקוּם וְגַבְרָא דִי אַתְּ רָשֵׁי בֵהּ יַפֵק לְוָתָךְ יָת מַשְׁכּוֹנָא לְבָרָא:

{יב} וְאִם אִישׁ עָנִי הוּא לֹא תִשְׁכַּב בַּעֲבֹטוֹ:  מ  וְאִם אִישׁ עָנִי הוּא לֹא תִשְׁכַּב בַּעֲבֹטוֹ:  ת  וְאִם גְבַר מִסְכֵּן הוּא לָא תִשְׁכּוּב בְּמַשְׁכּוֹנֵהּ:

{יג} הָשֵׁב תָּשִׁיב לוֹ אֶת הַעֲבוֹט כְּבֹא הַשֶּׁמֶשׁ וְשָׁכַב בְּשַׂלְמָתוֹ וּבֵרֲכֶךָּ וּלְךָ תִּהְיֶה צְדָקָה לִפְנֵי יְהוָה אֱלֹהֶיךָ:  מ  הָשֵׁב תָּשִׁיב לוֹ אֶת הַעֲבוֹט כְּבֹא הַשֶּׁמֶשׁ וְשָׁכַב בְּשַׂלְמָתוֹ וּבֵרֲכֶךָּ וּלְךָ תִּהְיֶה צְדָקָה לִפְנֵי יְהוָה אֱלֹהֶיךָ:  ת  אָתָבָא תָתֵב לֵהּ יָת מַשְׁכּוֹנָא כְּמֵעַל שִׁמְשָׁא וְיִשְׁכּוּב בִּכְסוּתֵהּ וִיבָרֵכִנָךְ וְלָךְ תְּהֵי זְכוּתָא קֳדָם יְיָ אֱלָהָךְ: (ס)

{יד}  שביעי  לֹא תַעֲשֹׁק שָׂכִיר עָנִי וְאֶבְיוֹן מֵאַחֶיךָ אוֹ מִגֵּרְךָ אֲשֶׁר בְּאַרְצְךָ בִּשְׁעָרֶיךָ:  מ  לֹא תַעֲשֹׁק שָׂכִיר עָנִי וְאֶבְיוֹן מֵאַחֶיךָ אוֹ מִגֵּרְךָ אֲשֶׁר בְּאַרְצְךָ בִּשְׁעָרֶיךָ:  ת  לָא תַעֲשׁוֹק אֲגִירָא עַנְיָא וּמִסְכֵּנָא מֵאַחָיךְ אוֹ מִגִיוֹרָךְ דִי בְאַרְעָךְ בְּקִרְוָיךְ:

{טו} בְּיוֹמוֹ תִתֵּן שְׂכָרוֹ וְלֹא תָבוֹא עָלָיו הַשֶּׁמֶשׁ כִּי עָנִי הוּא וְאֵלָיו הוּא נֹשֵׂא אֶת נַפְשׁוֹ וְלֹא יִקְרָא עָלֶיךָ אֶל יְהוָה וְהָיָה בְךָ חֵטְא:  מ  בְּיוֹמוֹ תִתֵּן שְׂכָרוֹ וְלֹא תָבוֹא עָלָיו הַשֶּׁמֶשׁ כִּי עָנִי הוּא וְאֵלָיו הוּא נֹשֵׂא אֶת נַפְשׁוֹ וְלֹא יִקְרָא עָלֶיךָ אֶל יְהוָה וְהָיָה בְךָ חֵטְא:  ת  בְּיוֹמֵהּ תִּתֵּן אַגְרֵהּ וְלָא תֵעוּל עֲלוֹהִי שִׁמְשָׁא אֲרֵי עַנְיָא הוּא וְלֵהּ הוּא מְסַר יָת נַפְשֵׁהּ וְלָא יִקְרֵי עֲלָךְ קֳדָם יְיָ וִיהֵי בָךְ חוֹבָא: (ס)

{טז} לֹא יוּמְתוּ אָבוֹת עַל בָּנִים וּבָנִים לֹא יוּמְתוּ עַל אָבוֹת אִישׁ בְּחֶטְאוֹ יוּמָתוּ:  מ  לֹא יוּמְתוּ אָבוֹת עַל בָּנִים וּבָנִים לֹא יוּמְתוּ עַל אָבוֹת אִישׁ בְּחֶטְאוֹ יוּמָתוּ:  ת  לָא יְמוּתוּן אֲבָהָן עַל פּוּם בְּנִין וּבְנִין לָא יְמוּתוּן עַל פּוּם אֲבָהָן אֱנַשׁ בְּחוֹבֵהּ יְמוּתוּן: (ס)

{יז} לֹא תַטֶּה מִשְׁפַּט גֵּר יָתוֹם וְלֹא תַחֲבֹל בֶּגֶד אַלְמָנָה:  מ  לֹא תַטֶּה מִשְׁפַּט גֵּר יָתוֹם וְלֹא תַחֲבֹל בֶּגֶד אַלְמָנָה:  ת  לָא תַצְלֵי דִין גִיוֹרָא וְיִתַּמָא וְלָא תִסַב מַשְׁכּוֹנָא לְבוּשׁ אַרְמְלָא:

{יח} וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּמִצְרַיִם וַיִּפְדְּךָ יְהוָה אֱלֹהֶיךָ מִשָּׁם עַל כֵּן אָנֹכִי מְצַוְּךָ לַעֲשׂוֹת אֶת הַדָּבָר הַזֶּה:  מ  וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּמִצְרַיִם וַיִּפְדְּךָ יְהוָה אֱלֹהֶיךָ מִשָּׁם עַל כֵּן אָנֹכִי מְצַוְּךָ לַעֲשׂוֹת אֶת הַדָּבָר הַזֶּה:  ת  וְתִדְכַּר אֲרֵי עַבְדָא הֲוֵיתָא בְּמִצְרַיִם וּפָרְקָךְ יְיָ אֱלָהָךְ מִתַּמָן עַל כֵּן אֲנָא מְפַקְדָךְ לְמֶעְבַּד יָת פִּתְגָמָא הָדֵין: (ס)

{יט} כִּי תִקְצֹר קְצִירְךָ בְשָׂדֶךָ וְשָׁכַחְתָּ עֹמֶר בַּשָּׂדֶה לֹא תָשׁוּב לְקַחְתּוֹ לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה יִהְיֶה לְמַעַן יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל מַעֲשֵׂה יָדֶיךָ:  מ  כִּי תִקְצֹר קְצִירְךָ בְשָׂדֶךָ וְשָׁכַחְתָּ עֹמֶר בַּשָּׂדֶה לֹא תָשׁוּב לְקַחְתּוֹ לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה יִהְיֶה לְמַעַן יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל מַעֲשֵׂה יָדֶיךָ:  ת  אֲרֵי תַחְצוֹד חֲצָדָךְ בְּחַקְלָךְ וְתִנְשֵׁי עֻמְרָא בְחַקְלָא לָא תְתוּב לְמִסְבֵהּ לְגִיוֹרָא לְיִתַּמָא וּלְאַרְמְלָא יְהֵי בְּדִיל דִיבָרֵכִנָךְ יְיָ אֱלָהָךְ בְּכֹל עוֹבָדֵי יְדָךְ: (ס)

{כ} כִּי תַחְבֹּט זֵיתְךָ לֹא תְפָאֵר אַחֲרֶיךָ לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה יִהְיֶה:  מ  כִּי תַחְבֹּט זֵיתְךָ לֹא תְפָאֵר אַחֲרֶיךָ לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה יִהְיֶה:  ת  אֲרֵי תַחְבּוֹט זֵיתָךְ לָא תְפַלֵי בַּתְרָךְ לְגִיוֹרָא לְיִתַּמָא וּלְאַרְמְלָא יְהֵי:

{כא} כִּי תִבְצֹר כַּרְמְךָ לֹא תְעוֹלֵל אַחֲרֶיךָ לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה יִהְיֶה:  מ  כִּי תִבְצֹר כַּרְמְךָ לֹא תְעוֹלֵל אַחֲרֶיךָ לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה יִהְיֶה:  ת  אֲרֵי תִקְטוֹף כַּרְמָךְ לָא תְעָלֵל בַּתְרָךְ לְגִיוֹרָא לְיִתַּמָא וּלְאַרְמְלָא יְהֵי:

{כב} וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרָיִם עַל כֵּן אָנֹכִי מְצַוְּךָ לַעֲשׂוֹת אֶת הַדָּבָר הַזֶּה:  מ  וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרָיִם עַל כֵּן אָנֹכִי מְצַוְּךָ לַעֲשׂוֹת אֶת הַדָּבָר הַזֶּה:  ת  וְתִדְכַּר אֲרֵי עַבְדָא הֲוֵיתָא בְּאַרְעָא דְמִצְרָיִם עַל כֵּן אֲנָא מְפַקְדָךְ לְמֵעְבַּד יָת פִּתְגָמָא הָדֵין: (ס)




דברים פרק-כה

{א} כִּי יִהְיֶה רִיב בֵּין אֲנָשִׁים וְנִגְּשׁוּ אֶל הַמִּשְׁפָּט וּשְׁפָטוּם וְהִצְדִּיקוּ אֶת הַצַּדִּיק וְהִרְשִׁיעוּ אֶת הָרָשָׁע:  מ  כִּי יִהְיֶה רִיב בֵּין אֲנָשִׁים וְנִגְּשׁוּ אֶל הַמִּשְׁפָּט וּשְׁפָטוּם וְהִצְדִּיקוּ אֶת הַצַּדִּיק וְהִרְשִׁיעוּ אֶת הָרָשָׁע:  ת  אֲרֵי יְהֵי דִין בֵּין גֻבְרַיָא וְיִתְקָרְבוּן לְדִינָא וִידוּנֻנוּן וִיזַכּוּן יָת זַכָּאָה וִיחַיְבוּן יָת חַיָבָא:

{ב} וְהָיָה אִם בִּן הַכּוֹת הָרָשָׁע וְהִפִּילוֹ הַשֹּׁפֵט וְהִכָּהוּ לְפָנָיו כְּדֵי רִשְׁעָתוֹ בְּמִסְפָּר:  מ  וְהָיָה אִם בִּן הַכּוֹת הָרָשָׁע וְהִפִּילוֹ הַשֹּׁפֵט וְהִכָּהוּ לְפָנָיו כְּדֵי רִשְׁעָתוֹ בְּמִסְפָּר:  ת  וִיהֵי אִם בַּר חַיָבָא לְאַלְקָאָה חַיָבָא וְיִרְמִנֵהּ דַיָנָא וְיַלְקִנֵהּ קֳדָמוֹהִי כְּמִסַת חוֹבְתֵהּ בְּמִנְיָן:

{ג} אַרְבָּעִים יַכֶּנּוּ לֹא יֹסִיף פֶּן יֹסִיף לְהַכֹּתוֹ עַל אֵלֶּה מַכָּה רַבָּה וְנִקְלָה אָחִיךָ לְעֵינֶיךָ:  מ  אַרְבָּעִים יַכֶּנּוּ לֹא יֹסִיף פֶּן יֹסִיף לְהַכֹּתוֹ עַל אֵלֶּה מַכָּה רַבָּה וְנִקְלָה אָחִיךָ לְעֵינֶיךָ:  ת  אַרְבְּעִין יַלְקִנֵהּ לָא יוֹסֵף דִילְמָא יוֹסֵף לְאַלְקָיוּתֵהּ עַל אִלֵין מָחָא רַבָּא וְיֵקַל אֲחוּךְ לְעֵינָיךְ:

{ד} לֹא תַחְסֹם שׁוֹר בְּדִישׁוֹ:  מ  לֹא תַחְסֹם שׁוֹר בְּדִישׁוֹ:  ת  לָא תֵחוֹד פּוּם תּוֹרָא בִּדְיָשֵׁהּ: (ס)

{ה} כִּי יֵשְׁבוּ אַחִים יַחְדָּו וּמֵת אַחַד מֵהֶם וּבֵן אֵין לוֹ לֹא תִהְיֶה אֵשֶׁת הַמֵּת הַחוּצָה לְאִישׁ זָר יְבָמָהּ יָבֹא עָלֶיהָ וּלְקָחָהּ לוֹ לְאִשָּׁה וְיִבְּמָהּ:  מ  כִּי יֵשְׁבוּ אַחִים יַחְדָּו וּמֵת אַחַד מֵהֶם וּבֵן אֵין לוֹ לֹא תִהְיֶה אֵשֶׁת הַמֵּת הַחוּצָה לְאִישׁ זָר יְבָמָהּ יָבֹא עָלֶיהָ וּלְקָחָהּ לוֹ לְאִשָּׁה וְיִבְּמָהּ:  ת  אֲרֵי יֵתְבוּן אַחִין כַּחֲדָא וִימוּת חַד מִנְהוֹן וּבַר לֵית לֵהּ לָא תְהֵי אִתַּת מֵתָנָא לְבָרָא לִגְבַר אוֹחֲרָן יְבָמַהּ יֵעוּל עֲלַהּ וְיִסְבַהּ לֵהּ לְאִנְתּוּ וְיִבְּמַהּ:

{ו} וְהָיָה הַבְּכוֹר אֲשֶׁר תֵּלֵד יָקוּם עַל שֵׁם אָחִיו הַמֵּת וְלֹא יִמָּחֶה שְׁמוֹ מִיִּשְׂרָאֵל:  מ  וְהָיָה הַבְּכוֹר אֲשֶׁר תֵּלֵד יָקוּם עַל שֵׁם אָחִיו הַמֵּת וְלֹא יִמָּחֶה שְׁמוֹ מִיִּשְׂרָאֵל:  ת  וִיהֵי בּוּכְרָא דִי תְלִיד יְקוּם עַל שְׁמָא דַאֲחוּהִי מֵתָנָא וְלָא יִתִּמְחֵי שְׁמֵהּ מִיִשְׂרָאֵל:

{ז} וְאִם לֹא יַחְפֹּץ הָאִישׁ לָקַחַת אֶת יְבִמְתּוֹ וְעָלְתָה יְבִמְתּוֹ הַשַּׁעְרָה אֶל הַזְּקֵנִים וְאָמְרָה מֵאֵין יְבָמִי לְהָקִים לְאָחִיו שֵׁם בְּיִשְׂרָאֵל לֹא אָבָה יַבְּמִי:  מ  וְאִם לֹא יַחְפֹּץ הָאִישׁ לָקַחַת אֶת יְבִמְתּוֹ וְעָלְתָה יְבִמְתּוֹ הַשַּׁעְרָה אֶל הַזְּקֵנִים וְאָמְרָה מֵאֵין יְבָמִי לְהָקִים לְאָחִיו שֵׁם בְּיִשְׂרָאֵל לֹא אָבָה יַבְּמִי:  ת  וְאִם לָא יִצְבֵי גַבְרָא לְמִסַב יָת יְבִמְתֵּהּ וְתִסַק יְבִמְתֵהּ לִתְרַע בֵּית דִינָא לָקֳדָם סָבַיָא וְתֵימַר לָא צָבֵי יְבָמִי לַאֲקָמָא לַאֲחוּהִי שְׁמָא בְּיִשְׂרָאֵל לָא אָבֵי לְיַבָּמוּתִי:

{ח} וְקָרְאוּ לוֹ זִקְנֵי עִירוֹ וְדִבְּרוּ אֵלָיו וְעָמַד וְאָמַר לֹא חָפַצְתִּי לְקַחְתָּהּ:  מ  וְקָרְאוּ לוֹ זִקְנֵי עִירוֹ וְדִבְּרוּ אֵלָיו וְעָמַד וְאָמַר לֹא חָפַצְתִּי לְקַחְתָּהּ:  ת  וִיקְרוּן לֵהּ סָבֵי קַרְתֵּהּ וִימַלְלוּן עִמֵהּ וִיקוּם וְיֵימַר לָא רְעֵנָא לְמִסְבַהּ:

{ט} וְנִגְּשָׁה יְבִמְתּוֹ אֵלָיו לְעֵינֵי הַזְּקֵנִים וְחָלְצָה נַעֲלוֹ מֵעַל רַגְלוֹ וְיָרְקָה בְּפָנָיו וְעָנְתָה וְאָמְרָה כָּכָה יֵעָשֶׂה לָאִישׁ אֲשֶׁר לֹא יִבְנֶה אֶת בֵּית אָחִיו:  מ  וְנִגְּשָׁה יְבִמְתּוֹ אֵלָיו לְעֵינֵי הַזְּקֵנִים וְחָלְצָה נַעֲלוֹ מֵעַל רַגְלוֹ וְיָרְקָה בְּפָנָיו וְעָנְתָה וְאָמְרָה כָּכָה יֵעָשֶׂה לָאִישׁ אֲשֶׁר לֹא יִבְנֶה אֶת בֵּית אָחִיו:  ת  וְתִתְקְרַב יְבִמְתֵּהּ לְוָתֵה לָקֳדָם סָבַיָא וְתַשְׁרֵי סֵינֵהּ מֵעַל רַגְלֵהּ וְתֵרוֹק בְּאַנְפּוֹהִי וְתָתֵב וְתֵימַר כְּדֵין יִתְעֲבֵד לִגְבַר דִי לָא יִבְנֵי יָת בֵּיתָא דַאֲחוּהִי:

{י} וְנִקְרָא שְׁמוֹ בְּיִשְׂרָאֵל בֵּית חֲלוּץ הַנָּעַל:  מ  וְנִקְרָא שְׁמוֹ בְּיִשְׂרָאֵל בֵּית חֲלוּץ הַנָּעַל:  ת  וְיִתְקְרֵי שְׁמֵהּ בְּיִשְׂרָאֵל בֵּית שְׁרִי סֵינָא: (ס)

{יא} כִּי יִנָּצוּ אֲנָשִׁים יַחְדָּו אִישׁ וְאָחִיו וְקָרְבָה אֵשֶׁת הָאֶחָד לְהַצִּיל אֶת אִישָׁהּ מִיַּד מַכֵּהוּ וְשָׁלְחָה יָדָהּ וְהֶחֱזִיקָה בִּמְבֻשָׁיו:  מ  כִּי יִנָּצוּ אֲנָשִׁים יַחְדָּו אִישׁ וְאָחִיו וְקָרְבָה אֵשֶׁת הָאֶחָד לְהַצִּיל אֶת אִישָׁהּ מִיַּד מַכֵּהוּ וְשָׁלְחָה יָדָהּ וְהֶחֱזִיקָה בִּמְבֻשָׁיו:  ת  אֲרֵי יִנְצוּן גֻבְרִין כַּחֲדָא גְבַר וַאֲחוּהִי וְתִתְקְרֵב אִתַּת חַד לְשֵׁזָבָא יָת בַּעֲלַהּ מִיַד מָחוּהִי וְתוֹשֵׁט יְדַהּ וְתִתְקֵף בְּבֵית בַּהֲתְּתֵהּ:

{יב} וְקַצֹּתָה אֶת כַּפָּהּ לֹא תָחוֹס עֵינֶךָ:  מ  וְקַצֹּתָה אֶת כַּפָּהּ לֹא תָחוֹס עֵינֶךָ:  ת  וּתְקוּץ יָת יְדַהּ לָא תְחוּס עֵינָךְ: (ס)

{יג} לֹא יִהְיֶה לְךָ בְּכִיסְךָ אֶבֶן וָאָבֶן גְּדוֹלָה וּקְטַנָּה:  מ  לֹא יִהְיֶה לְךָ בְּכִיסְךָ אֶבֶן וָאָבֶן גְּדוֹלָה וּקְטַנָּה:  ת  לָא יְהֵי לָךְ בְּכִיסָךְ מַתְקַל וּמַתְקָל רַבְּתָא וּזְעֶרְתָּא:

{יד} לֹא יִהְיֶה לְךָ בְּבֵיתְךָ אֵיפָה וְאֵיפָה גְּדוֹלָה וּקְטַנָּה:  מ  לֹא יִהְיֶה לְךָ בְּבֵיתְךָ אֵיפָה וְאֵיפָה גְּדוֹלָה וּקְטַנָּה:  ת  לָא יְהֵי לָךְ בְּבֵיתָךְ מְכִילָא וּמְכִילָא רַבְּתָא וּזְעֶרְתָּא:

{טו} אֶבֶן שְׁלֵמָה וָצֶדֶק יִהְיֶה לָּךְ אֵיפָה שְׁלֵמָה וָצֶדֶק יִהְיֶה לָּךְ לְמַעַן יַאֲרִיכוּ יָמֶיךָ עַל הָאֲדָמָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:  מ  אֶבֶן שְׁלֵמָה וָצֶדֶק יִהְיֶה לָּךְ אֵיפָה שְׁלֵמָה וָצֶדֶק יִהְיֶה לָּךְ לְמַעַן יַאֲרִיכוּ יָמֶיךָ עַל הָאֲדָמָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:  ת  מַתְקִלִין שָׁלְמִין דִקְשׁוֹט יְהוֹן לָךְ מְכִילָן שָׁלְמִין דִקְשׁוֹט יְהוֹן לָךְ בְּדִיל דְיוֹרְכוּן יוֹמָיךְ עַל אַרְעָא דַיְיָ אֱלָהָךְ יָהֵב לָךְ:

{טז} כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ כָּל עֹשֵׂה אֵלֶּה כֹּל עֹשֵׂה עָוֶל:  מ  כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ כָּל עֹשֵׂה אֵלֶּה כֹּל עֹשֵׂה עָוֶל:  ת  אֲרֵי מְרָחָק קֳדָם יְיָ אֱלהָךְ כָּל עָבֵד אִלֵין כֹּל עָבֵד שְׁקַר: (פ)

{יז}  מפטיר  זָכוֹר אֵת אֲשֶׁר עָשָׂה לְךָ עֲמָלֵק בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם:  מ  זָכוֹר אֵת אֲשֶׁר עָשָׂה לְךָ עֲמָלֵק בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם:  ת  הֲוֵי דְכִיר יָת דִי עֲבַד לָךְ עֲמָלֵק בְּאָרְחָא בְּמִפָּקְכוֹן מִמִצְרָיִם:

{יח} אֲשֶׁר קָרְךָ בַּדֶּרֶךְ וַיְזַנֵּב בְּךָ כָּל הַנֶּחֱשָׁלִים אַחַרֶיךָ וְאַתָּה עָיֵף וְיָגֵעַ וְלֹא יָרֵא אֱלֹהִים:  מ  אֲשֶׁר קָרְךָ בַּדֶּרֶךְ וַיְזַנֵּב בְּךָ כָּל הַנֶּחֱשָׁלִים אַחַרֶיךָ וְאַתָּה עָיֵף וְיָגֵעַ וְלֹא יָרֵא אֱלֹהִים:  ת  דִי עָרְעָךְ בְּאָרְחָא וְקָטֵל בָּךְ כָּל דַהֲווֹ מִתְאַחֲרִין בַּתְרָךְ וְאַתְּ מְשַׁלְהֵי וְלָאֵי וְלָא דָחֵיל מִן קֳדָם יְיָ:

{יט} וְהָיָה בְּהָנִיחַ יְהוָה אֱלֹהֶיךָ לְךָ מִכָּל אֹיְבֶיךָ מִסָּבִיב בָּאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה לְרִשְׁתָּהּ תִּמְחֶה אֶת זֵכֶר עֲמָלֵק מִתַּחַת הַשָּׁמָיִם לֹא תִּשְׁכָּח:  מ  וְהָיָה בְּהָנִיחַ יְהוָה אֱלֹהֶיךָ לְךָ מִכָּל אֹיְבֶיךָ מִסָּבִיב בָּאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה לְרִשְׁתָּהּ תִּמְחֶה אֶת זֵכֶר עֲמָלֵק מִתַּחַת הַשָּׁמָיִם לֹא תִּשְׁכָּח:  ת  וִיהֵי כַּד יְנִיחַ יְיָ אֱלָהָךְ לָךְ מִכָּל בַּעֲלֵי דְבָבָךְ מִסְחוֹר סְחוֹר בְּאַרְעָא דִי יְיָ אֱלָהָךְ יָהֵב לָךְ אַחֲסָנָא לְמֵירְתַהּ תִּמְחֵי יָת דוּכְרָנָא דַעֲמָלֵק מִתְּחוֹת שְׁמַיָא לָא תִּתִּנְשֵׁי: (פפפ)
