פרשת קדושים




ויקרא פרק-יט

{א} וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} דַּבֵּ֞ר אֶל-כָּל-עֲדַ֧ת בְּנֵי-יִשְׂרָאֵ֛ל וְאָמַרְתָּ֥ אֲלֵהֶ֖ם קְדֹשִׁ֣ים תִּהְי֑וּ כִּ֣י קָד֔וֹשׁ אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  מ  דַּבֵּ֞ר אֶל-כָּל-עֲדַ֧ת בְּנֵי-יִשְׂרָאֵ֛ל וְאָמַרְתָּ֥ אֲלֵהֶ֖ם קְדֹשִׁ֣ים תִּהְי֑וּ כִּ֣י קָד֔וֹשׁ אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  ת  מַלֵל עִם כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן קַדִישִׁין תְּהוֹן אֲרֵי קַדִישׁ אֲנָא יְיָ אֱלָהָכוֹן:

{ג} אִ֣ישׁ אִמּ֤וֹ וְאָבִיו֙ תִּירָ֔אוּ וְאֶת-שַׁבְּתֹתַ֖י תִּשְׁמֹ֑רוּ אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  מ  אִ֣ישׁ אִמּ֤וֹ וְאָבִיו֙ תִּירָ֔אוּ וְאֶת-שַׁבְּתֹתַ֖י תִּשְׁמֹ֑רוּ אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  ת  גְבַר מִן אִמֵהּ וּמִן אֲבוּהִי תְּהוֹן דָחֲלִין וְיָת יוֹמֵי שַׁבַּיָא דִילִי תִּטְרוּן אֲנָא יְיָ אֱלָהָכוֹן:

{ד} אַל-תִּפְנוּ֙ אֶל-הָ֣אֱלִילִ֔ים וֵֽאלֹהֵי֙ מַסֵּכָ֔ה לֹ֥א תַעֲשׂ֖וּ לָכֶ֑ם אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  מ  אַל-תִּפְנוּ֙ אֶל-הָ֣אֱלִילִ֔ים וֵֽאלֹהֵי֙ מַסֵּכָ֔ה לֹ֥א תַעֲשׂ֖וּ לָכֶ֑ם אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  ת  לָא תִתְפְּנוּן בָּתַר טַעֲוָן וְדַחֲלָן דְמַתְּכָא לָא תַעְבְּדוּן לְכוֹן אֲנָא יְיָ אֱלָהָכוֹן:

{ה} וְכִ֧י תִזְבְּח֛וּ זֶ֥בַח שְׁלָמִ֖ים לַיהוָ֑ה לִֽרְצֹנְכֶ֖ם תִּזְבָּחֻֽהוּ:  מ  וְכִ֧י תִזְבְּח֛וּ זֶ֥בַח שְׁלָמִ֖ים לַיהוָ֑ה לִֽרְצֹנְכֶ֖ם תִּזְבָּחֻֽהוּ:  ת  וַאֲרֵי תִכְסוּן נִכְסַת קוּדְשַׁיָא קֳדָם יְיָ לְרַעֲוָא לְכוֹן תִּכְסֻנֵהּ:

{ו} בְּי֧וֹם זִבְחֲכֶ֛ם יֵאָכֵ֖ל וּמִֽמָּחֳרָ֑ת וְהַנּוֹתָר֙ עַד-י֣וֹם הַשְּׁלִישִׁ֔י בָּאֵ֖שׁ יִשָּׂרֵֽף:  מ  בְּי֧וֹם זִבְחֲכֶ֛ם יֵאָכֵ֖ל וּמִֽמָּחֳרָ֑ת וְהַנּוֹתָר֙ עַד-י֣וֹם הַשְּׁלִישִׁ֔י בָּאֵ֖שׁ יִשָּׂרֵֽף:  ת  בְּיוֹמָא דְיִתְנְכֵס יִתְאֲכֵל וּבְיוֹמָא דְבַתְרוֹהִי וּדְאִשְׁתָּאַר עַד יוֹמָא תְלִיתָאָה בְּנוּרָא יִתּוֹקָד:

{ז} וְאִ֛ם הֵאָכֹ֥ל יֵאָכֵ֖ל בַּיּ֣וֹם הַשְּׁלִישִׁ֑י פִּגּ֥וּל ה֖וּא לֹ֥א יֵרָצֶֽה:  מ  וְאִ֛ם הֵאָכֹ֥ל יֵאָכֵ֖ל בַּיּ֣וֹם הַשְּׁלִישִׁ֑י פִּגּ֥וּל ה֖וּא לֹ֥א יֵרָצֶֽה:  ת  וְאִם אִתְאֲכָלָא יִתְאֲכֵל בְּיוֹמָא תְלִיתָאָה מְרָחַק הוּא לָא יְהֵי לְרַעַוָא:

{ח} וְאֹֽכְלָיו֙ עֲוֹנ֣וֹ יִשָּׂ֔א כִּֽי-אֶת-קֹ֥דֶשׁ יְהוָ֖ה חִלֵּ֑ל וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵעַמֶּֽיהָ:  מ  וְאֹֽכְלָיו֙ עֲוֹנ֣וֹ יִשָּׂ֔א כִּֽי-אֶת-קֹ֥דֶשׁ יְהוָ֖ה חִלֵּ֑ל וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵעַמֶּֽיהָ:  ת  וּדְיֵיכְלִנֵהּ חוֹבֵהּ יְקַבֵּל אֲרֵי יָת קוּדְשָׁא דַיְיָ אַחֵל וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא מֵעַמֵהּ:

{ט} וּֽבְקֻצְרְכֶם֙ אֶת-קְצִ֣יר אַרְצְכֶ֔ם לֹ֧א תְכַלֶּ֛ה פְּאַ֥ת שָׂדְךָ֖ לִקְצֹ֑ר וְלֶ֥קֶט קְצִֽירְךָ֖ לֹ֥א תְלַקֵּֽט:  מ  וּֽבְקֻצְרְכֶם֙ אֶת-קְצִ֣יר אַרְצְכֶ֔ם לֹ֧א תְכַלֶּ֛ה פְּאַ֥ת שָׂדְךָ֖ לִקְצֹ֑ר וְלֶ֥קֶט קְצִֽירְךָ֖ לֹ֥א תְלַקֵּֽט:  ת  וּבְמֶחְצָדְכוֹן יָת חֲצָדָא דְאַרְעֲכוֹן לָא תְשֵׁיצֵי פָּאתָא דְחַקְלָךְ לְמֶחְצָד וּלְקָטָא דַחֲצָדָךְ לָא תְלַקֵט:

{י} וְכַרְמְךָ֙ לֹ֣א תְעוֹלֵ֔ל וּפֶ֥רֶט כַּרְמְךָ֖ לֹ֣א תְלַקֵּ֑ט לֶֽעָנִ֤י וְלַגֵּר֙ תַּעֲזֹ֣ב אֹתָ֔ם אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  מ  וְכַרְמְךָ֙ לֹ֣א תְעוֹלֵ֔ל וּפֶ֥רֶט כַּרְמְךָ֖ לֹ֣א תְלַקֵּ֑ט לֶֽעָנִ֤י וְלַגֵּר֙ תַּעֲזֹ֣ב אֹתָ֔ם אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  ת  וְכַרְמָךְ לָא תְעַלֵל וְנִתְרָא דְכַרְמָךְ לָא תְלַקֵט לְעַנְיֵי וּלְגִיוֹרֵי תִּשְׁבּוֹק יָתְהוֹן אֲנָא יְיָ אֱלָהָכוֹן:

{יא} לֹ֖א תִּגְנֹ֑בוּ וְלֹא-תְכַחֲשׁ֥וּ וְלֹֽא-תְשַׁקְּר֖וּ אִ֥ישׁ בַּעֲמִיתֽוֹ:  מ  לֹ֖א תִּגְנֹ֑בוּ וְלֹא-תְכַחֲשׁ֥וּ וְלֹֽא-תְשַׁקְּר֖וּ אִ֥ישׁ בַּעֲמִיתֽוֹ:  ת  לָא תִּגְנְבוּן וְלָא תְכַדְבוּן וְלָא תְשַׁקְרוּן אֱנַשׁ בְּחַבְרֵהּ:

{יב} וְלֹֽא-תִשָּׁבְע֥וּ בִשְׁמִ֖י לַשָּׁ֑קֶר וְחִלַּלְתָּ֛ אֶת-שֵׁ֥ם אֱלֹהֶ֖יךָ אֲנִ֥י יְהוָֽה:  מ  וְלֹֽא-תִשָּׁבְע֥וּ בִשְׁמִ֖י לַשָּׁ֑קֶר וְחִלַּלְתָּ֛ אֶת-שֵׁ֥ם אֱלֹהֶ֖יךָ אֲנִ֥י יְהוָֽה:  ת  וְלָא תִשָׁבְעוּן בִּשְׁמִי לְשִׁקְרָא וְתַחֵל יָת שְׁמָא דֶאֱלָהָךְ אֲנָא יְיָ:

{יג} לֹֽא-תַעֲשֹׁ֥ק אֶת-רֵֽעֲךָ֖ וְלֹ֣א תִגְזֹ֑ל לֹֽא-תָלִ֞ין פְּעֻלַּ֥ת שָׂכִ֛יר אִתְּךָ֖ עַד-בֹּֽקֶר:  מ  לֹֽא-תַעֲשֹׁ֥ק אֶת-רֵֽעֲךָ֖ וְלֹ֣א תִגְזֹ֑ל לֹֽא-תָלִ֞ין פְּעֻלַּ֥ת שָׂכִ֛יר אִתְּךָ֖ עַד-בֹּֽקֶר:  ת  לָא תַעֲשׁוֹק יָת חַבְרָךְ וְלָא תְאַנֵס לָא תְבִית אַגְרָא דַאֲגִירָא לְוָתָךְ עַד צַפְרָא:

{יד} לֹא-תְקַלֵּ֣ל חֵרֵ֔שׁ וְלִפְנֵ֣י עִוֵּ֔ר לֹ֥א תִתֵּ֖ן מִכְשֹׁ֑ל וְיָרֵ֥אתָ מֵּאֱלֹהֶ֖יךָ אֲנִ֥י יְהוָֽה:  מ  לֹא-תְקַלֵּ֣ל חֵרֵ֔שׁ וְלִפְנֵ֣י עִוֵּ֔ר לֹ֥א תִתֵּ֖ן מִכְשֹׁ֑ל וְיָרֵ֥אתָ מֵּאֱלֹהֶ֖יךָ אֲנִ֥י יְהוָֽה:  ת  לָא תְלוּטּ דְלָא שָׁמַע וְקֳדָם דְלָא חָזֵי לָא תְשִׂים תַּקְלָא וְתִדְחַל מֵאֱלָהָךְ אֲנָא יְיָ:

{טו}  שני  לֹא-תַעֲשׂ֥וּ עָ֙וֶל֙ בַּמִּשְׁפָּ֔ט לֹא-תִשָּׂ֣א פְנֵי-דָ֔ל וְלֹ֥א תֶהְדַּ֖ר פְּנֵ֣י גָד֑וֹל בְּצֶ֖דֶק תִּשְׁפֹּ֥ט עֲמִיתֶֽךָ:  מ  לֹא-תַעֲשׂ֥וּ עָ֙וֶל֙ בַּמִּשְׁפָּ֔ט לֹא-תִשָּׂ֣א פְנֵי-דָ֔ל וְלֹ֥א תֶהְדַּ֖ר פְּנֵ֣י גָד֑וֹל בְּצֶ֖דֶק תִּשְׁפֹּ֥ט עֲמִיתֶֽךָ:  ת  לָא תַעְבְּדוּן שְׁקַר בְּדִינָא לָא תִסַב אַפֵּי מַסְכֵּנָא וְלָא תֶהְדַר אַפֵּי רַבָּא בְּקוּשְׁטָא תְּדִינֵהּ לְחַבְרָךְ:

{טז} לֹא-תֵלֵ֤ךְ רָכִיל֙ בְּעַמֶּ֔יךָ לֹ֥א תַעֲמֹ֖ד עַל-דַּ֣ם רֵעֶ֑ךָ אֲנִ֖י יְהוָֽה:  מ  לֹא-תֵלֵ֤ךְ רָכִיל֙ בְּעַמֶּ֔יךָ לֹ֥א תַעֲמֹ֖ד עַל-דַּ֣ם רֵעֶ֑ךָ אֲנִ֖י יְהוָֽה:  ת  לָא תֵיכוּל קוּרְצִין בְּעַמָךְ לָא תְקוּם עַל דְמָא דְחַבְרָךְ אֲנָא יְיָ:

{יז} לֹֽא-תִשְׂנָ֥א אֶת-אָחִ֖יךָ בִּלְבָבֶ֑ךָ הוֹכֵ֤חַ תּוֹכִ֙יחַ֙ אֶת-עֲמִיתֶ֔ךָ וְלֹא-תִשָּׂ֥א עָלָ֖יו חֵֽטְא:  מ  לֹֽא-תִשְׂנָ֥א אֶת-אָחִ֖יךָ בִּלְבָבֶ֑ךָ הוֹכֵ֤חַ תּוֹכִ֙יחַ֙ אֶת-עֲמִיתֶ֔ךָ וְלֹא-תִשָּׂ֥א עָלָ֖יו חֵֽטְא:  ת  לָא תִשְׂנֵי יָת אֲחוּךְ בְּלִבָּךְ אוֹכָחָא תוֹכַח יָת חַבְרָךְ וְלָא תְקַבֵּל עַל דִי לֵהּ חוֹבָא:

{יח} לֹֽא-תִקֹּ֤ם וְלֹֽא-תִטֹּר֙ אֶת-בְּנֵ֣י עַמֶּ֔ךָ וְאָֽהַבְתָּ֥ לְרֵעֲךָ֖ כָּמ֑וֹךָ אֲנִ֖י יְהוָֽה:  מ  לֹֽא-תִקֹּ֤ם וְלֹֽא-תִטֹּר֙ אֶת-בְּנֵ֣י עַמֶּ֔ךָ וְאָֽהַבְתָּ֥ לְרֵעֲךָ֖ כָּמ֑וֹךָ אֲנִ֖י יְהוָֽה:  ת  לָא תִקוֹם וְלָא תִטַר דְבָבוּ לִבְנֵי עַמָךְ וּתְרַחֵם לְחַבְרָךְ כְּוָתָךְ אֲנָא יְיָ:

{יט} אֶֽת-חֻקֹּתַי֮ תִּשְׁמֹרוּ֒ בְּהֶמְתְּךָ֙ לֹא-תַרְבִּ֣יעַ כִּלְאַ֔יִם שָׂדְךָ֖ לֹא-תִזְרַ֣ע כִּלְאָ֑יִם וּבֶ֤גֶד כִּלְאַ֙יִם֙ שַֽׁעַטְנֵ֔ז לֹ֥א יַעֲלֶ֖ה עָלֶֽיךָ:  מ  אֶֽת-חֻקֹּתַי֮ תִּשְׁמֹרוּ֒ בְּהֶמְתְּךָ֙ לֹא-תַרְבִּ֣יעַ כִּלְאַ֔יִם שָׂדְךָ֖ לֹא-תִזְרַ֣ע כִּלְאָ֑יִם וּבֶ֤גֶד כִּלְאַ֙יִם֙ שַֽׁעַטְנֵ֔ז לֹ֥א יַעֲלֶ֖ה עָלֶֽיךָ:  ת  יָת קְיָמַי תִּטְרוּן בְּעִירָךְ לָא תַרְכֵּב עֵרוּבִין חַקְלָךְ לָא תִזְרַע עֵרוּבִין וּלְבוּשׁ עֵרוּבִין שַׁעַטְנֵזָא לָא יִסַק עֲלָךְ:

{כ} וְ֠אִישׁ כִּֽי-יִשְׁכַּ֨ב אֶת-אִשָּׁ֜ה שִׁכְבַת-זֶ֗רַע וְהִ֤וא שִׁפְחָה֙ נֶחֱרֶ֣פֶת לְאִ֔ישׁ וְהָפְדֵּה֙ לֹ֣א נִפְדָּ֔תָה א֥וֹ חֻפְשָׁ֖ה לֹ֣א נִתַּן-לָ֑הּ בִּקֹּ֧רֶת תִּהְיֶ֛ה לֹ֥א יוּמְת֖וּ כִּי-לֹ֥א חֻפָּֽשָׁה:  מ  וְ֠אִישׁ כִּֽי-יִשְׁכַּ֨ב אֶת-אִשָּׁ֜ה שִׁכְבַת-זֶ֗רַע וְהִ֤וא שִׁפְחָה֙ נֶחֱרֶ֣פֶת לְאִ֔ישׁ וְהָפְדֵּה֙ לֹ֣א נִפְדָּ֔תָה א֥וֹ חֻפְשָׁ֖ה לֹ֣א נִתַּן-לָ֑הּ בִּקֹּ֧רֶת תִּהְיֶ֛ה לֹ֥א יוּמְת֖וּ כִּי-לֹ֥א חֻפָּֽשָׁה:  ת  וּגְבַר אֲרֵי יִשְׁכּוּב עִם אִתְּתָא שִׁכְבַת זַרְעָא וְהִיא אַמְתָא אֲחִידָא לִגְבַר וְאִתְפְּרָקָא לָא אִתְפְּרִיקַת בְּכַסְפָּא אוֹ חֵרוּתָא לָא אִתְיְהִיבַת לַהּ בִּשְׁטָר בִּקֻּרְתָּא תְהֵי לָא יוּמְתוּן אֲרֵי לָא אִתְחָרָרַת:

{כא} וְהֵבִ֤יא אֶת-אֲשָׁמוֹ֙ לַֽיהוָ֔ה אֶל-פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד אֵ֖יל אָשָֽׁם:  מ  וְהֵבִ֤יא אֶת-אֲשָׁמוֹ֙ לַֽיהוָ֔ה אֶל-פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד אֵ֖יל אָשָֽׁם:  ת  וְיַיְתִי יָת אַשְׁמֵהּ קֳדָם יְיָ לִתְרַע מַשְׁכַּן זִמְנָא דְכַר לַאֲשָׁמָא:

{כב} וְכִפֶּר֩ עָלָ֨יו הַכֹּהֵ֜ן בְּאֵ֤יל הָֽאָשָׁם֙ לִפְנֵ֣י יְהוָ֔ה עַל-חַטָּאת֖וֹ אֲשֶׁ֣ר חָטָ֑א וְנִסְלַ֣ח ל֔וֹ מֵחַטָּאת֖וֹ אֲשֶׁ֥ר חָטָֽא:  מ  וְכִפֶּר֩ עָלָ֨יו הַכֹּהֵ֜ן בְּאֵ֤יל הָֽאָשָׁם֙ לִפְנֵ֣י יְהוָ֔ה עַל-חַטָּאת֖וֹ אֲשֶׁ֣ר חָטָ֑א וְנִסְלַ֣ח ל֔וֹ מֵחַטָּאת֖וֹ אֲשֶׁ֥ר חָטָֽא:  ת  וִיכַפֵּר עֲלוֹהִי כַּהֲנָא בְּדִכְרָא דַאֲשָׁמָא קֳדָם יְיָ עַל חוֹבְתֵהּ דִי חָב וְיִשְׁתְּבֵק לֵהּ מְחוֹבְתֵהּ דִי חַב: (פ)

{כג}  שלישי  וְכִי-תָבֹ֣אוּ אֶל-הָאָ֗רֶץ וּנְטַעְתֶּם֙ כָּל-עֵ֣ץ מַאֲכָ֔ל וַעֲרַלְתֶּ֥ם עָרְלָת֖וֹ אֶת-פִּרְי֑וֹ שָׁלֹ֣שׁ שָׁנִ֗ים יִהְיֶ֥ה לָכֶ֛ם עֲרֵלִ֖ים לֹ֥א יֵאָכֵֽל:  מ  וְכִי-תָבֹ֣אוּ אֶל-הָאָ֗רֶץ וּנְטַעְתֶּם֙ כָּל-עֵ֣ץ מַאֲכָ֔ל וַעֲרַלְתֶּ֥ם עָרְלָת֖וֹ אֶת-פִּרְי֑וֹ שָׁלֹ֣שׁ שָׁנִ֗ים יִהְיֶ֥ה לָכֶ֛ם עֲרֵלִ֖ים לֹ֥א יֵאָכֵֽל:  ת  וַאֲרֵי תֵעֲלוּן לְאַרְעָא וְתִצְבוּן כָּל אִלָן דְמֵיכַל וּתְרַחֲקוּן רָחָקָא יַת אִבֵּהּ תְּלַת שְׁנִין יְהֵי לְכוֹן מְרָחָק לַאֲבָדָא לָא יִתְאֲכֵל:

{כד} וּבַשָּׁנָה֙ הָרְבִיעִ֔ת יִהְיֶ֖ה כָּל-פִּרְי֑וֹ קֹ֥דֶשׁ הִלּוּלִ֖ים לַיהוָֽה:  מ  וּבַשָּׁנָה֙ הָרְבִיעִ֔ת יִהְיֶ֖ה כָּל-פִּרְי֑וֹ קֹ֥דֶשׁ הִלּוּלִ֖ים לַיהוָֽה:  ת  וּבְשַׁתָּא רְבִיעֵתָא יְהֵי כָּל אִבֵּהּ קֹדֶשׁ תּוּשְׁבְּחָן קֳדָם יְיָ:

{כה} וּבַשָּׁנָ֣ה הַחֲמִישִׁ֗ת תֹּֽאכְלוּ֙ אֶת-פִּרְי֔וֹ לְהוֹסִ֥יף לָכֶ֖ם תְּבוּאָת֑וֹ אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  מ  וּבַשָּׁנָ֣ה הַחֲמִישִׁ֗ת תֹּֽאכְלוּ֙ אֶת-פִּרְי֔וֹ לְהוֹסִ֥יף לָכֶ֖ם תְּבוּאָת֑וֹ אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  ת  וּבְשַׁתָּא חֲמִישֵׁתָא תֵּיכְלוּן יָת אִבֵּהּ לְאוֹסָפָא לְכוֹן עֲלַלְתֵּהּ אֲנָא יְיָ אֱלָהָכוֹן:

{כו} לֹ֥א תֹאכְל֖וּ עַל-הַדָּ֑ם לֹ֥א תְנַחֲשׁ֖וּ וְלֹ֥א תְעוֹנֵֽנוּ:  מ  לֹ֥א תֹאכְל֖וּ עַל-הַדָּ֑ם לֹ֥א תְנַחֲשׁ֖וּ וְלֹ֥א תְעוֹנֵֽנוּ:  ת  לָא תֵיכְלוּן עַל דְמָא לָא תְנַחֲשׁוּן וְלָא תְעוֹנוּן:

{כז} לֹ֣א תַקִּ֔פוּ פְּאַ֖ת רֹאשְׁכֶ֑ם וְלֹ֣א תַשְׁחִ֔ית אֵ֖ת פְּאַ֥ת זְקָנֶֽךָ:  מ  לֹ֣א תַקִּ֔פוּ פְּאַ֖ת רֹאשְׁכֶ֑ם וְלֹ֣א תַשְׁחִ֔ית אֵ֖ת פְּאַ֥ת זְקָנֶֽךָ:  ת  לָא תַקְפוּן פָּאתָא דְרֵישְׁכוֹן וְלָא תְחַבֵּל יָת פָּאתָא דְדִקְנָךְ:

{כח} וְשֶׂ֣רֶט לָנֶ֗פֶשׁ לֹ֤א תִתְּנוּ֙ בִּבְשַׂרְכֶ֔ם וּכְתֹ֣בֶת קַֽעֲקַ֔ע לֹ֥א תִתְּנ֖וּ בָּכֶ֑ם אֲנִ֖י יְהוָֽה:  מ  וְשֶׂ֣רֶט לָנֶ֗פֶשׁ לֹ֤א תִתְּנוּ֙ בִּבְשַׂרְכֶ֔ם וּכְתֹ֣בֶת קַֽעֲקַ֔ע לֹ֥א תִתְּנ֖וּ בָּכֶ֑ם אֲנִ֖י יְהוָֽה:  ת  וְחִבּוּל עַל מִית לָא תִתְּנוּן בְּבִסְרְכוֹן וְרוּשְׁמִין חֲרִיתִין לָא תִתְּנוּן בְכוֹן אֲנָא יְיָ:

{כט} אַל-תְּחַלֵּ֥ל אֶֽת-בִּתְּךָ֖ לְהַזְנוֹתָ֑הּ וְלֹא-תִזְנֶ֣ה הָאָ֔רֶץ וּמָלְאָ֥ה הָאָ֖רֶץ זִמָּֽה:  מ  אַל-תְּחַלֵּ֥ל אֶֽת-בִּתְּךָ֖ לְהַזְנוֹתָ֑הּ וְלֹא-תִזְנֶ֣ה הָאָ֔רֶץ וּמָלְאָ֥ה הָאָ֖רֶץ זִמָּֽה:  ת  לָא תַחֵל יָת בְּרַתָּךְ לְאַטְעֲיוּתַהּ וְלָא תִטְעֵי אַרְעָא וְתִתְמְלֵי אַרְעָא עֵצַת חִטְאִין:

{ל} אֶת-שַׁבְּתֹתַ֣י תִּשְׁמֹ֔רוּ וּמִקְדָּשִׁ֖י תִּירָ֑אוּ אֲנִ֖י יְהוָֽה:  מ  אֶת-שַׁבְּתֹתַ֣י תִּשְׁמֹ֔רוּ וּמִקְדָּשִׁ֖י תִּירָ֑אוּ אֲנִ֖י יְהוָֽה:  ת  יָת יוֹמֵי שַׁבַּיָא דִילִי תִּטְרוּן וּלְבֵית מַקְדְשִׁי תְּהוֹן דָחֲלִין אֲנָא יְיָ:

{לא} אַל-תִּפְנ֤וּ אֶל-הָאֹבֹת֙ וְאֶל-הַיִּדְּעֹנִ֔ים אַל-תְּבַקְשׁ֖וּ לְטָמְאָ֣ה בָהֶ֑ם אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  מ  אַל-תִּפְנ֤וּ אֶל-הָאֹבֹת֙ וְאֶל-הַיִּדְּעֹנִ֔ים אַל-תְּבַקְשׁ֖וּ לְטָמְאָ֣ה בָהֶ֑ם אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  ת  לָא תִתְפְּנוּן בָּתַר בִּדִין וּדְכוּרוּ לָא תִתְבְּעוּן לְאִסְתָּאָבָא בְהוֹן אֲנָא יְיָ אֱלָהָכוֹן:

{לב} מִפְּנֵ֤י שֵׂיבָה֙ תָּק֔וּם וְהָדַרְתָּ֖ פְּנֵ֣י זָקֵ֑ן וְיָרֵ֥אתָ מֵּאֱלֹהֶ֖יךָ אֲנִ֥י יְהוָֽה:  מ  מִפְּנֵ֤י שֵׂיבָה֙ תָּק֔וּם וְהָדַרְתָּ֖ פְּנֵ֣י זָקֵ֑ן וְיָרֵ֥אתָ מֵּאֱלֹהֶ֖יךָ אֲנִ֥י יְהוָֽה:  ת  מִן קֳדָם דְסַבַּר בְּאוֹרַיְתָא תְּקוּם וּתְהַדַר אַפֵּי סָבָא וְתִדְחַל מֵאֳלָהָךְ אֲנָא יְיָ: (ס)

{לג}  רביעי - שישי במחוברין  וְכִֽי-יָג֧וּר אִתְּךָ֛ גֵּ֖ר בְּאַרְצְכֶ֑ם לֹ֥א תוֹנ֖וּ אֹתֽוֹ:  מ  וְכִֽי-יָג֧וּר אִתְּךָ֛ גֵּ֖ר בְּאַרְצְכֶ֑ם לֹ֥א תוֹנ֖וּ אֹתֽוֹ:  ת  וַאֲרֵי יִתְגַיַר עִמָךְ גִיוֹרָא בְּאַרְעֲכוֹן לָא תוֹנוּן יָתֵהּ:

{לד} כְּאֶזְרָ֣ח מִכֶּם֩ יִהְיֶ֨ה לָכֶ֜ם הַגֵּ֣ר | הַגָּ֣ר אִתְּכֶ֗ם וְאָהַבְתָּ֥ לוֹ֙ כָּמ֔וֹךָ כִּֽי-גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֣רֶץ מִצְרָ֑יִם אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  מ  כְּאֶזְרָ֣ח מִכֶּם֩ יִהְיֶ֨ה לָכֶ֜ם הַגֵּ֣ר | הַגָּ֣ר אִתְּכֶ֗ם וְאָהַבְתָּ֥ לוֹ֙ כָּמ֔וֹךָ כִּֽי-גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֣רֶץ מִצְרָ֑יִם אֲנִ֖י יְהוָ֥ה אֱלֹהֵיכֶֽם:  ת  כְּיַצִיבָא מִנְכוֹן יְהֵי לְכוֹן גִיוֹרָא דְיִתְגַיַר בֵּינֵיכוֹן וּתְרַחֵם לֵהּ כְּוָתָךְ אֲרֵי דַיָרִין הֲוֵיתוּן בְּאַרְעָא דְמִצְרָיִם אֲנָא יְיָ אֱלָהָכוֹן:

{לה} לֹא-תַעֲשׂ֥וּ עָ֖וֶל בַּמִּשְׁפָּ֑ט בַּמִּדָּ֕ה בַּמִּשְׁקָ֖ל וּבַמְּשׂוּרָֽה:  מ  לֹא-תַעֲשׂ֥וּ עָ֖וֶל בַּמִּשְׁפָּ֑ט בַּמִּדָּ֕ה בַּמִּשְׁקָ֖ל וּבַמְּשׂוּרָֽה:  ת  לָא תַעְבְּדוּן שְׁקַר בְּדִין בִּמְשַׁחְתָּא בְּמַתְקְלָא וּבִמְכִלְתָּא:

{לו} מֹ֧אזְנֵי צֶ֣דֶק אַבְנֵי-צֶ֗דֶק אֵ֥יפַת צֶ֛דֶק וְהִ֥ין צֶ֖דֶק יִהְיֶ֣ה לָכֶ֑ם אֲנִי֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁר-הוֹצֵ֥אתִי אֶתְכֶ֖ם מֵאֶ֥רֶץ מִצְרָֽיִם:  מ  מֹ֧אזְנֵי צֶ֣דֶק אַבְנֵי-צֶ֗דֶק אֵ֥יפַת צֶ֛דֶק וְהִ֥ין צֶ֖דֶק יִהְיֶ֣ה לָכֶ֑ם אֲנִי֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁר-הוֹצֵ֥אתִי אֶתְכֶ֖ם מֵאֶ֥רֶץ מִצְרָֽיִם:  ת  מֹאזְנָוָן דִקְשׁוֹט מַתְקְלָן דִקְשׁוֹט מְכִילָן דִקְשׁוֹט וְהִינִין דִקְשׁוֹט יְהוֹן לְכוֹן אֲנָא יְיָ אֱלָהָכוֹן דִי אֲפֵּקִית יָתְכוֹן מֵאַרְעָא דְמִצְרָיִם:

{לז} וּשְׁמַרְתֶּ֤ם אֶת-כָּל-חֻקֹּתַי֙ וְאֶת-כָּל-מִשְׁפָּטַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם אֲנִ֖י יְהוָֽה:  מ  וּשְׁמַרְתֶּ֤ם אֶת-כָּל-חֻקֹּתַי֙ וְאֶת-כָּל-מִשְׁפָּטַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם אֲנִ֖י יְהוָֽה:  ת  וְתִטְרוּן יָת כָּל קְיָמַי וְיָת כָּל דִינַי וְתַעְבְּדוּן יָתְהוֹן אֲנָא יְיָ: (פ)




ויקרא פרק-כ

{א}  חמישי  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} וְאֶל-בְּנֵ֣י יִשְׂרָאֵל֮ תֹּאמַר֒ אִ֣ישׁ אִישׁ֩ מִבְּנֵ֨י יִשְׂרָאֵ֜ל וּמִן-הַגֵּ֣ר | הַגָּ֣ר בְּיִשְׂרָאֵ֗ל אֲשֶׁ֨ר יִתֵּ֧ן מִזַּרְע֛וֹ לַמֹּ֖לֶךְ מ֣וֹת יוּמָ֑ת עַ֥ם הָאָ֖רֶץ יִרְגְּמֻ֥הוּ בָאָֽבֶן:  מ  וְאֶל-בְּנֵ֣י יִשְׂרָאֵל֮ תֹּאמַר֒ אִ֣ישׁ אִישׁ֩ מִבְּנֵ֨י יִשְׂרָאֵ֜ל וּמִן-הַגֵּ֣ר | הַגָּ֣ר בְּיִשְׂרָאֵ֗ל אֲשֶׁ֨ר יִתֵּ֧ן מִזַּרְע֛וֹ לַמֹּ֖לֶךְ מ֣וֹת יוּמָ֑ת עַ֥ם הָאָ֖רֶץ יִרְגְּמֻ֥הוּ בָאָֽבֶן:  ת  וְלִבְנֵי יִשְׂרָאֵל תֵּימַר גְבַר גְבַר מִבְּנֵי יִשְׂרָאֵל וּמִן גִיוֹרָא דְיִתְגַיְרוּן בְּיִשְׂרָאֵל דִי יִתֵּן מִזַרְעֵהּ לְמוֹלֶךְ אִתְקְטָלָא יִתְקְטֵל עַמָא בֵית יִשְׂרָאֵל יִרְגְמֻנֵהּ בְּאַבְנָא:

{ג} וַאֲנִ֞י אֶתֵּ֤ן אֶת-פָּנַי֙ בָּאִ֣ישׁ הַה֔וּא וְהִכְרַתִּ֥י אֹת֖וֹ מִקֶּ֣רֶב עַמּ֑וֹ כִּ֤י מִזַּרְעוֹ֙ נָתַ֣ן לַמֹּ֔לֶךְ לְמַ֗עַן טַמֵּא֙ אֶת-מִקְדָּשִׁ֔י וּלְחַלֵּ֖ל אֶת-שֵׁ֥ם קָדְשִֽׁי:  מ  וַאֲנִ֞י אֶתֵּ֤ן אֶת-פָּנַי֙ בָּאִ֣ישׁ הַה֔וּא וְהִכְרַתִּ֥י אֹת֖וֹ מִקֶּ֣רֶב עַמּ֑וֹ כִּ֤י מִזַּרְעוֹ֙ נָתַ֣ן לַמֹּ֔לֶךְ לְמַ֗עַן טַמֵּא֙ אֶת-מִקְדָּשִׁ֔י וּלְחַלֵּ֖ל אֶת-שֵׁ֥ם קָדְשִֽׁי:  ת  וַאֲנָא אֶתֵּן יָת רוּגְזִי בְגַבְרָא הַּהוּא וְאֱשֵׁיצֵי יָתֵהּ מִגוֹ עַמֵהּ אֲרֵי מִזַרְעֵהּ יְהַב לְמוֹלֶךְ בְּדִיל לְסָאָבָא יָת מַקְדְשִׁי וּלְאַחַלָא יָת שְׁמָא דְקוּדְשִׁי:

{ד} וְאִ֡ם הַעְלֵ֣ם יַעְלִימֽוּ֩ עַ֨ם הָאָ֜רֶץ אֶת-עֵֽינֵיהֶם֙ מִן-הָאִ֣ישׁ הַה֔וּא בְּתִתּ֥וֹ מִזַּרְע֖וֹ לַמֹּ֑לֶךְ לְבִלְתִּ֖י הָמִ֥ית אֹתֽוֹ:  מ  וְאִ֡ם הַעְלֵ֣ם יַעְלִימֽוּ֩ עַ֨ם הָאָ֜רֶץ אֶת-עֵֽינֵיהֶם֙ מִן-הָאִ֣ישׁ הַה֔וּא בְּתִתּ֥וֹ מִזַּרְע֖וֹ לַמֹּ֑לֶךְ לְבִלְתִּ֖י הָמִ֥ית אֹתֽוֹ:  ת  וְאִם מִכְבַּשׁ יִכְבְּשׁוּן עַמָא בֵית יִשְׂרָאֵל יָת עֵינֵיהוֹן מִן גַבְרָא הַהוּא בִּדְיָהֵב מִזַרְעֵהּ לְמוֹלֶךְ בְּדִיל דְלָא לְקַטָלָא יָתֵהּ:

{ה} וְשַׂמְתִּ֨י אֲנִ֧י אֶת-פָּנַ֛י בָּאִ֥ישׁ הַה֖וּא וּבְמִשְׁפַּחְתּ֑וֹ וְהִכְרַתִּ֨י אֹת֜וֹ וְאֵ֣ת | כָּל-הַזֹּנִ֣ים אַחֲרָ֗יו לִזְנ֛וֹת אַחֲרֵ֥י הַמֹּ֖לֶךְ מִקֶּ֥רֶב עַמָּֽם:  מ  וְשַׂמְתִּ֨י אֲנִ֧י אֶת-פָּנַ֛י בָּאִ֥ישׁ הַה֖וּא וּבְמִשְׁפַּחְתּ֑וֹ וְהִכְרַתִּ֨י אֹת֜וֹ וְאֵ֣ת | כָּל-הַזֹּנִ֣ים אַחֲרָ֗יו לִזְנ֛וֹת אַחֲרֵ֥י הַמֹּ֖לֶךְ מִקֶּ֥רֶב עַמָּֽם:  ת  וֶאֱשַׁוֵי אֲנָא יָת רוּגְזִי בְגַבְרָא הַהוּא וּבִסְעָדוֹהִי וֶאֱשֵׁיצֵי יָתֵהּ וְיָת כָּל דְטָעָן בַּתְרוֹהִי לְמִטְעֵי בָּתַר מוֹלֶךְ מִגוֹ עַמְהוֹן:

{ו} וְהַנֶּ֗פֶשׁ אֲשֶׁ֨ר תִּפְנֶ֤ה אֶל-הָֽאֹבֹת֙ וְאֶל-הַיִּדְּעֹנִ֔ים לִזְנ֖וֹת אַחֲרֵיהֶ֑ם וְנָתַתִּ֤י אֶת-פָּנַי֙ בַּנֶּ֣פֶשׁ הַהִ֔וא וְהִכְרַתִּ֥י אֹת֖וֹ מִקֶּ֥רֶב עַמּֽוֹ:  מ  וְהַנֶּ֗פֶשׁ אֲשֶׁ֨ר תִּפְנֶ֤ה אֶל-הָֽאֹבֹת֙ וְאֶל-הַיִּדְּעֹנִ֔ים לִזְנ֖וֹת אַחֲרֵיהֶ֑ם וְנָתַתִּ֤י אֶת-פָּנַי֙ בַּנֶּ֣פֶשׁ הַהִ֔וא וְהִכְרַתִּ֥י אֹת֖וֹ מִקֶּ֥רֶב עַמּֽוֹ:  ת  וֶאֱנַשׁ דִי יִתְפְּנֵי בָּתַר בִּדִין וּדְכוּרוּ לְמִטְעֵי בַתְרֵיהוֹן וְאֶתֵּן יָת רוּגְזִי בַּאֲנָשָׁא הַהִיא וֶאֱשֵׁיצֵי יָתֵהּ מִגוֹ עַמֵהּ:

{ז} וְהִ֨תְקַדִּשְׁתֶּ֔ם וִהְיִיתֶ֖ם קְדֹשִׁ֑ים כִּ֛י אֲנִ֥י יְהוָ֖ה אֱלֹהֵיכֶֽם:  מ  וְהִ֨תְקַדִּשְׁתֶּ֔ם וִהְיִיתֶ֖ם קְדֹשִׁ֑ים כִּ֛י אֲנִ֥י יְהוָ֖ה אֱלֹהֵיכֶֽם:  ת  וְתִתְקַדְשׁוּן וּתְהוֹן קַדִישִׁין אֲרֵי אֲנָא יְיָ אֱלָהָכוֹן:

{ח}  שישי - שביעי במחוברין  וּשְׁמַרְתֶּם֙ אֶת-חֻקֹּתַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם אֲנִ֥י יְהוָ֖ה מְקַדִּשְׁכֶֽם:  מ  וּשְׁמַרְתֶּם֙ אֶת-חֻקֹּתַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם אֲנִ֥י יְהוָ֖ה מְקַדִּשְׁכֶֽם:  ת  וְתִטְּרוּן יָת קְיָמַי וְתַעְבְּדוּן יָתְהוֹן אֲנָא:ייָ מַקְדִשְׁכוֹן:

{ט} כִּֽי-אִ֣ישׁ אִ֗ישׁ אֲשֶׁ֨ר יְקַלֵּ֧ל אֶת-אָבִ֛יו וְאֶת-אִמּ֖וֹ מ֣וֹת יוּמָ֑ת אָבִ֧יו וְאִמּ֛וֹ קִלֵּ֖ל דָּמָ֥יו בּֽוֹ:  מ  כִּֽי-אִ֣ישׁ אִ֗ישׁ אֲשֶׁ֨ר יְקַלֵּ֧ל אֶת-אָבִ֛יו וְאֶת-אִמּ֖וֹ מ֣וֹת יוּמָ֑ת אָבִ֧יו וְאִמּ֛וֹ קִלֵּ֖ל דָּמָ֥יו בּֽוֹ:  ת  אֲרֵי גְבַר גְבַר דִי יְלוּט יָת אֲבוּהִי וְיָת אִמֵהּ אִתְקְטָּלָא יִתְקְטֵל אֲבוּהִי וְאִמֵהּ לַטּ קְטָלָא חַיָב:

{י} וְאִ֗ישׁ אֲשֶׁ֤ר יִנְאַף֙ אֶת-אֵ֣שֶׁת אִ֔ישׁ אֲשֶׁ֥ר יִנְאַ֖ף אֶת-אֵ֣שֶׁת רֵעֵ֑הוּ מֽוֹת-יוּמַ֥ת הַנֹּאֵ֖ף וְהַנֹּאָֽפֶת:  מ  וְאִ֗ישׁ אֲשֶׁ֤ר יִנְאַף֙ אֶת-אֵ֣שֶׁת אִ֔ישׁ אֲשֶׁ֥ר יִנְאַ֖ף אֶת-אֵ֣שֶׁת רֵעֵ֑הוּ מֽוֹת-יוּמַ֥ת הַנֹּאֵ֖ף וְהַנֹּאָֽפֶת:  ת  וּגְבַר דִי יְגוּף יָת אִתַּת גְבַר דִי יְגוּף יָת אִתַּת חַבְרֵהּ אִתְקְטָלָא יִתְקְטֵל גַיָפָא וְגַיַפְתָּא:

{יא} וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת-אֵ֣שֶׁת אָבִ֔יו עֶרְוַ֥ת אָבִ֖יו גִּלָּ֑ה מֽוֹת-יוּמְת֥וּ שְׁנֵיהֶ֖ם דְּמֵיהֶ֥ם בָּֽם:  מ  וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת-אֵ֣שֶׁת אָבִ֔יו עֶרְוַ֥ת אָבִ֖יו גִּלָּ֑ה מֽוֹת-יוּמְת֥וּ שְׁנֵיהֶ֖ם דְּמֵיהֶ֥ם בָּֽם:  ת  וּגְבַר דִי יִשְׁכּוּב עִם אִתַּת אֲבוּהִי עֶרְיְתָא דַאֲבוּהִי גַלִי אִתְקַטָלָא יִתְקַטְלוּן תַּרְוֵיהוֹן קְטָלָא חַיָבִין:

{יב} וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת-כַּלָּת֔וֹ מ֥וֹת יוּמְת֖וּ שְׁנֵיהֶ֑ם תֶּ֥בֶל עָשׂ֖וּ דְּמֵיהֶ֥ם בָּֽם:  מ  וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת-כַּלָּת֔וֹ מ֥וֹת יוּמְת֖וּ שְׁנֵיהֶ֑ם תֶּ֥בֶל עָשׂ֖וּ דְּמֵיהֶ֥ם בָּֽם:  ת  וּגְבַר דִי יִשְׁכּוּב יָת כַּלָתֵהּ אִתְקְטָלָא יִתְקַטְלוּן תַּרְוֵיהוֹן תִּבְלָא עֲבָדוּ קְטָלָא חַיָבִין:

{יג} וְאִ֗ישׁ אֲשֶׁ֨ר יִשְׁכַּ֤ב אֶת-זָכָר֙ מִשְׁכְּבֵ֣י אִשָּׁ֔ה תּוֹעֵבָ֥ה עָשׂ֖וּ שְׁנֵיהֶ֑ם מ֥וֹת יוּמָ֖תוּ דְּמֵיהֶ֥ם בָּֽם:  מ  וְאִ֗ישׁ אֲשֶׁ֨ר יִשְׁכַּ֤ב אֶת-זָכָר֙ מִשְׁכְּבֵ֣י אִשָּׁ֔ה תּוֹעֵבָ֥ה עָשׂ֖וּ שְׁנֵיהֶ֑ם מ֥וֹת יוּמָ֖תוּ דְּמֵיהֶ֥ם בָּֽם:  ת  וּגְבַר דִי יִשְׁכּוּב יָת דְכוּרָא מִשְׁכְּבֵי אִתְּתָא תּוֹעֶבְתָּא עֲבָדוּ תַּרְוֵיהוֹן אִתְקְטָלָא יִתְקַטְלוּן קְטָלָא חַיָבִין:

{יד} וְאִ֗ישׁ אֲשֶׁ֨ר יִקַּ֧ח אֶת-אִשָּׁ֛ה וְאֶת-אִמָּ֖הּ זִמָּ֣ה הִ֑וא בָּאֵ֞שׁ יִשְׂרְפ֤וּ אֹתוֹ֙ וְאֶתְהֶ֔ן וְלֹא-תִהְיֶ֥ה זִמָּ֖ה בְּתוֹכְכֶֽם:  מ  וְאִ֗ישׁ אֲשֶׁ֨ר יִקַּ֧ח אֶת-אִשָּׁ֛ה וְאֶת-אִמָּ֖הּ זִמָּ֣ה הִ֑וא בָּאֵ֞שׁ יִשְׂרְפ֤וּ אֹתוֹ֙ וְאֶתְהֶ֔ן וְלֹא-תִהְיֶ֥ה זִמָּ֖ה בְּתוֹכְכֶֽם:  ת  וּגְבַר דִי יִסַב יָת אִתְּתָא וְיָת אִמַהּ עֵצַת חִטְאִין הִיא בְּנוּרָא יוֹקְדוּן יָתֵהּ וְיָתְהֶן וְלָא תְהֵי עֵצַת חִטְאִין בֵּינֵיכוֹן:

{טו} וְאִ֗ישׁ אֲשֶׁ֨ר יִתֵּ֧ן שְׁכָבְתּ֛וֹ בִּבְהֵמָ֖ה מ֣וֹת יוּמָ֑ת וְאֶת-הַבְּהֵמָ֖ה תַּהֲרֹֽגוּ:  מ  וְאִ֗ישׁ אֲשֶׁ֨ר יִתֵּ֧ן שְׁכָבְתּ֛וֹ בִּבְהֵמָ֖ה מ֣וֹת יוּמָ֑ת וְאֶת-הַבְּהֵמָ֖ה תַּהֲרֹֽגוּ:  ת  וּגְבַר דִי יִתֵּן שְׁכָבְתֵּהּ בִּבְעִירָא אִתְקְטָלָא יִתְקְטֵל וְיָת בְּעִירָא תִּקְטְלוּן:

{טז} וְאִשָּׁ֗ה אֲשֶׁ֨ר תִּקְרַ֤ב אֶל-כָּל-בְּהֵמָה֙ לְרִבְעָ֣ה אֹתָ֔הּ וְהָרַגְתָּ֥ אֶת-הָאִשָּׁ֖ה וְאֶת-הַבְּהֵמָ֑ה מ֥וֹת יוּמָ֖תוּ דְּמֵיהֶ֥ם בָּֽם:  מ  וְאִשָּׁ֗ה אֲשֶׁ֨ר תִּקְרַ֤ב אֶל-כָּל-בְּהֵמָה֙ לְרִבְעָ֣ה אֹתָ֔הּ וְהָרַגְתָּ֥ אֶת-הָאִשָּׁ֖ה וְאֶת-הַבְּהֵמָ֑ה מ֥וֹת יוּמָ֖תוּ דְּמֵיהֶ֥ם בָּֽם:  ת  וְאִתְּתָא דִי תִקְרַב לְכָל בְּעִירָא לְמִשְׁלַט בַּהּ וְתִקְטוֹל יָת אִתְּתָא וְיָת בְּעִירָא אִתְקְטָלָא יִתְקַטְלוּן קְטָלָא חַיָבִין:

{יז} וְאִ֣ישׁ אֲשֶׁר-יִקַּ֣ח אֶת-אֲחֹת֡וֹ בַּת-אָבִ֣יו א֣וֹ בַת-אִ֠מּוֹ וְרָאָ֨ה אֶת-עֶרְוָתָ֜הּ וְהִֽיא-תִרְאֶ֤ה אֶת-עֶרְוָתוֹ֙ חֶ֣סֶד ה֔וּא וְנִ֨כְרְת֔וּ לְעֵינֵ֖י בְּנֵ֣י עַמָּ֑ם עֶרְוַ֧ת אֲחֹת֛וֹ גִּלָּ֖ה עֲוֹנ֥וֹ יִשָּֽׂא:  מ  וְאִ֣ישׁ אֲשֶׁר-יִקַּ֣ח אֶת-אֲחֹת֡וֹ בַּת-אָבִ֣יו א֣וֹ בַת-אִ֠מּוֹ וְרָאָ֨ה אֶת-עֶרְוָתָ֜הּ וְהִֽיא-תִרְאֶ֤ה אֶת-עֶרְוָתוֹ֙ חֶ֣סֶד ה֔וּא וְנִ֨כְרְת֔וּ לְעֵינֵ֖י בְּנֵ֣י עַמָּ֑ם עֶרְוַ֧ת אֲחֹת֛וֹ גִּלָּ֖ה עֲוֹנ֥וֹ יִשָּֽׂא:  ת  וּגְבַר דִי יִסַב יָת אֲחָתֵהּ בַּת אֲבוּהִי אוֹ בַת אִמֵהּ וְיֶחֱזֵי יָת עֶרְיְתַהּ וְהִיא תֶחֱזֵי יָת עֶרְיְתֵהּ קְלָנָא הוּא וְיִשְׁתֵּיצוּן לְעֵינֵי בְּנֵי עַמְהוֹן עֶרְיַת אֲחָתֵהּ גַלִי חוֹבֵהּ יְקַבֵּל:

{יח} וְ֠אִישׁ אֲשֶׁר-יִשְׁכַּ֨ב אֶת-אִשָּׁ֜ה דָּוָ֗ה וְגִלָּ֤ה אֶת-עֶרְוָתָהּ֙ אֶת-מְקֹרָ֣הּ הֶֽעֱרָ֔ה וְהִ֕יא גִּלְּתָ֖ה אֶת-מְק֣וֹר דָּמֶ֑יהָ וְנִכְרְת֥וּ שְׁנֵיהֶ֖ם מִקֶּ֥רֶב עַמָּֽם:  מ  וְ֠אִישׁ אֲשֶׁר-יִשְׁכַּ֨ב אֶת-אִשָּׁ֜ה דָּוָ֗ה וְגִלָּ֤ה אֶת-עֶרְוָתָהּ֙ אֶת-מְקֹרָ֣הּ הֶֽעֱרָ֔ה וְהִ֕יא גִּלְּתָ֖ה אֶת-מְק֣וֹר דָּמֶ֑יהָ וְנִכְרְת֥וּ שְׁנֵיהֶ֖ם מִקֶּ֥רֶב עַמָּֽם:  ת  וּגְבַר דִי יִשְׁכּוּב יָת אִתְּתָא סוֹאֲבָתָא וִיגַלֵי יָת עֶרְיְתַהּ יָת קְלָנַהּ גַלִי וְהִיא תְגַלֵי יָת סוֹאֲבַת דְמָהָא וְיִשְׁתֵּיצוּן תַּרְוֵיהוֹן מִגוֹ עַמְהוֹן:

{יט} וְעֶרְוַ֨ת אֲח֧וֹת אִמְּךָ֛ וַאֲח֥וֹת אָבִ֖יךָ לֹ֣א תְגַלֵּ֑ה כִּ֧י אֶת-שְׁאֵר֛וֹ הֶעֱרָ֖ה עֲוֹנָ֥ם יִשָּֽׂאוּ:  מ  וְעֶרְוַ֨ת אֲח֧וֹת אִמְּךָ֛ וַאֲח֥וֹת אָבִ֖יךָ לֹ֣א תְגַלֵּ֑ה כִּ֧י אֶת-שְׁאֵר֛וֹ הֶעֱרָ֖ה עֲוֹנָ֥ם יִשָּֽׂאוּ:  ת  וְעֶרְיַת אֲחַת אִמָךְ וַאֲחַת אָבוּךְ לָא תְגַלֵי אֲרֵי יָת קָרִיבֵהּ גַלִי חוֹבְהֶן יְקַבְּלוּן:

{כ} וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת-דֹּ֣דָת֔וֹ עֶרְוַ֥ת דֹּד֖וֹ גִּלָּ֑ה חֶטְאָ֥ם יִשָּׂ֖אוּ עֲרִירִ֥ים יָמֻֽתוּ:  מ  וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת-דֹּ֣דָת֔וֹ עֶרְוַ֥ת דֹּד֖וֹ גִּלָּ֑ה חֶטְאָ֥ם יִשָּׂ֖אוּ עֲרִירִ֥ים יָמֻֽתוּ:  ת  וּגְבַר דִי יִשְׁכּוּב יָת אִתַּת אַח אֲבוּהִי עֶרְיַת אַח אֲבוּהִי גַלִי חוֹבְהֶן יְקַבְּלּוּן בְּלָּא וְלָד יְמוּתוּן:

{כא} וְאִ֗ישׁ אֲשֶׁ֥ר יִקַּ֛ח אֶת-אֵ֥שֶׁת אָחִ֖יו נִדָּ֣ה הִ֑וא עֶרְוַ֥ת אָחִ֛יו גִּלָּ֖ה עֲרִירִ֥ים יִהְיֽוּ:  מ  וְאִ֗ישׁ אֲשֶׁ֥ר יִקַּ֛ח אֶת-אֵ֥שֶׁת אָחִ֖יו נִדָּ֣ה הִ֑וא עֶרְוַ֥ת אָחִ֛יו גִּלָּ֖ה עֲרִירִ֥ים יִהְיֽוּ:  ת  וּגְבַר דִי יִסַב יָת אִתַּת אֲחוּהִי מְרָחָקָא הִיא עֶרְיְתָא דַאֲחוּהִי גַלִי בְּלָא וְלָד יְהוֹן:

{כב} וּשְׁמַרְתֶּ֤ם אֶת-כָּל-חֻקֹּתַי֙ וְאֶת-כָּל-מִשְׁפָּטַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם וְלֹא-תָקִ֤יא אֶתְכֶם֙ הָאָ֔רֶץ אֲשֶׁ֨ר אֲנִ֜י מֵבִ֥יא אֶתְכֶ֛ם שָׁ֖מָּה לָשֶׁ֥בֶת בָּֽהּ:  מ  וּשְׁמַרְתֶּ֤ם אֶת-כָּל-חֻקֹּתַי֙ וְאֶת-כָּל-מִשְׁפָּטַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם וְלֹא-תָקִ֤יא אֶתְכֶם֙ הָאָ֔רֶץ אֲשֶׁ֨ר אֲנִ֜י מֵבִ֥יא אֶתְכֶ֛ם שָׁ֖מָּה לָשֶׁ֥בֶת בָּֽהּ:  ת  וְתִטְרוּן יָת כָּל קְיָמַי וְיָת כָּל דִינַי וְתַעְבְּדוּן יָתְהוֹן וְלָא תְרוֹקֵן יָתְכוֹן אַרְעָא דִי אֲנָא מָעֵיל יָתְכוֹן תַּמָן לְמִתַּב בַּהּ:

{כג}  שביעי  וְלֹ֤א תֵֽלְכוּ֙ בְּחֻקֹּ֣ת הַגּ֔וֹי אֲשֶׁר-אֲנִ֥י מְשַׁלֵּ֖חַ מִפְּנֵיכֶ֑ם כִּ֤י אֶת-כָּל-אֵ֙לֶּה֙ עָשׂ֔וּ וָאָקֻ֖ץ בָּֽם:  מ  וְלֹ֤א תֵֽלְכוּ֙ בְּחֻקֹּ֣ת הַגּ֔וֹי אֲשֶׁר-אֲנִ֥י מְשַׁלֵּ֖חַ מִפְּנֵיכֶ֑ם כִּ֤י אֶת-כָּל-אֵ֙לֶּה֙ עָשׂ֔וּ וָאָקֻ֖ץ בָּֽם:  ת  וְלָא תְהָכוּן בְּנִימוֹסֵי עַמְמַיָא דִי אֲנָא מַגְלֵי מִן קֳדָמֵיכוֹן אֲרֵי יָת כָּל אִלֵין עֲבָדוּ וְרָחִיק מֵימְרִי יָתְהוֹן:

{כד} וָאֹמַ֣ר לָכֶ֗ם אַתֶּם֮ תִּֽירְשׁ֣וּ אֶת-אַדְמָתָם֒ וַאֲנִ֞י אֶתְּנֶ֤נָּה לָכֶם֙ לָרֶ֣שֶׁת אֹתָ֔הּ אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ אֲנִי֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁר-הִבְדַּ֥לְתִּי אֶתְכֶ֖ם מִן-הָֽעַמִּֽים:  מ  וָאֹמַ֣ר לָכֶ֗ם אַתֶּם֮ תִּֽירְשׁ֣וּ אֶת-אַדְמָתָם֒ וַאֲנִ֞י אֶתְּנֶ֤נָּה לָכֶם֙ לָרֶ֣שֶׁת אֹתָ֔הּ אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ אֲנִי֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁר-הִבְדַּ֥לְתִּי אֶתְכֶ֖ם מִן-הָֽעַמִּֽים:  ת  וַאֲמָרִית לְכוֹן אַתּוּן תִּירְתוּן יָת אַרְעֲהוֹן וַאֲנָא אֶתְּנִנַהּ לְכוֹן לְמֵירַת יָתַהּ אַרְעָא עַבְדָא חַלָב וּדְבָשׁ אֲנָא יְיָ אֱלָהָכוֹן דִי אַפְרֵשִׁית יָתְכוֹן מִן עַמְמַיָא:

{כה}  מפטיר  וְהִבְדַּלְתֶּ֞ם בֵּֽין-הַבְּהֵמָ֤ה הַטְּהֹרָה֙ לַטְּמֵאָ֔ה וּבֵין-הָע֥וֹף הַטָּמֵ֖א לַטָּהֹ֑ר וְלֹֽא-תְשַׁקְּצ֨וּ אֶת-נַפְשֹֽׁתֵיכֶ֜ם בַּבְּהֵמָ֣ה וּבָע֗וֹף וּבְכֹל֙ אֲשֶׁ֣ר תִּרְמֹ֣שׂ הָֽאֲדָמָ֔ה אֲשֶׁר-הִבְדַּ֥לְתִּי לָכֶ֖ם לְטַמֵּֽא:  מ  וְהִבְדַּלְתֶּ֞ם בֵּֽין-הַבְּהֵמָ֤ה הַטְּהֹרָה֙ לַטְּמֵאָ֔ה וּבֵין-הָע֥וֹף הַטָּמֵ֖א לַטָּהֹ֑ר וְלֹֽא-תְשַׁקְּצ֨וּ אֶת-נַפְשֹֽׁתֵיכֶ֜ם בַּבְּהֵמָ֣ה וּבָע֗וֹף וּבְכֹל֙ אֲשֶׁ֣ר תִּרְמֹ֣שׂ הָֽאֲדָמָ֔ה אֲשֶׁר-הִבְדַּ֥לְתִּי לָכֶ֖ם לְטַמֵּֽא:  ת  וְתַפְרְשׁוּן בֵּין בְּעִירָא דַכְיָא לִמְסָאָבָא וּבֵין עוֹפָא מְסָאָבָא לִדְכֵי וְלָא תְשַׁקְצוּן יָת נַפְשָׁתֵיכוֹן בִּבְעִירָא וּבְעוֹפָא וּבְכֹל דִי תַרְחֵשׁ אַרְעָא דִי אַפְרֵשִׁית לְכוֹן לְסָאָבָא:

{כו} וִהְיִ֤יתֶם לִי֙ קְדֹשִׁ֔ים כִּ֥י קָד֖וֹשׁ אֲנִ֣י יְהוָ֑ה וָאַבְדִּ֥ל אֶתְכֶ֛ם מִן-הָֽעַמִּ֖ים לִהְי֥וֹת לִֽי:  מ  וִהְיִ֤יתֶם לִי֙ קְדֹשִׁ֔ים כִּ֥י קָד֖וֹשׁ אֲנִ֣י יְהוָ֑ה וָאַבְדִּ֥ל אֶתְכֶ֛ם מִן-הָֽעַמִּ֖ים לִהְי֥וֹת לִֽי:  ת  וּתְהוֹן קֳדָמַי קַדִישִׁין אֲרֵי קַדִישׁ אֲנָא יְיָ וְאַפְרֵשִׁית יָתְכוֹן מִן עַמְמַיָא לְמֶהֱוֵי פָלְחִין קֳדָמַי:

{כז} וְאִ֣ישׁ אֽוֹ-אִשָּׁ֗ה כִּי-יִהְיֶ֨ה בָהֶ֥ם א֛וֹב א֥וֹ יִדְּעֹנִ֖י מ֣וֹת יוּמָ֑תוּ בָּאֶ֛בֶן יִרְגְּמ֥וּ אֹתָ֖ם דְּמֵיהֶ֥ם בָּֽם:  מ  וְאִ֣ישׁ אֽוֹ-אִשָּׁ֗ה כִּי-יִהְיֶ֨ה בָהֶ֥ם א֛וֹב א֥וֹ יִדְּעֹנִ֖י מ֣וֹת יוּמָ֑תוּ בָּאֶ֛בֶן יִרְגְּמ֥וּ אֹתָ֖ם דְּמֵיהֶ֥ם בָּֽם:  ת  וּגְבַר אוֹ אִתְּתָא אֲרֵי יְהֵי בְהוֹן בִּדִין אוֹ דְכוּרוּ אִתְקְטָלָא יִתְקַטְלוּן בְּאַבְנָא יִרְגְמוּן יָתְהוֹן קְטָלָא חַיָבִין: (פפפ)
