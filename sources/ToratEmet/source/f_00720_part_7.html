ז

ז-א


[במדבר פרק-ד-כד]  זֹאת עֲבֹדַת מִשְׁפְּחֹת הַגֵּרְשֻׁנִּי לַעֲבֹד וּלְמַשָּׂא:


ז-ב


[ויקרא פרק-ו-יג]  זֶה קָרְבַּן אַהֲרֹן וּבָנָיו אֲשֶׁר יַקְרִיבוּ לַיהוָה בְּיוֹם הִמָּשַׁח אֹתוֹ עֲשִׂירִת הָאֵפָה סֹלֶת מִנְחָה תָּמִיד מַחֲצִיתָהּ בַּבֹּקֶר וּמַחֲצִיתָהּ בָּעָרֶב:

[ישעיה פרק-ט-יד]  זָקֵן וּנְשׂוּא פָנִים הוּא הָרֹאשׁ וְנָבִיא מוֹרֶה שֶּׁקֶר הוּא הַזָּנָב:

[ישעיה פרק-מו-ח]  זִכְרוּ זֹאת וְהִתְאֹשָׁשׁוּ הָשִׁיבוּ פוֹשְׁעִים עַל לֵב:

[הושע פרק-ד-יא]  זְנוּת וְיַיִן וְתִירוֹשׁ יִקַּח לֵב:

[חגי פרק-א-ו]  זְרַעְתֶּם הַרְבֵּה וְהָבֵא מְעָט אָכוֹל וְאֵין לְשָׂבְעָה שָׁתוֹ וְאֵין לְשָׁכְרָה לָבוֹשׁ וְאֵין לְחֹם לוֹ וְהַמִּשְׂתַּכֵּר מִשְׂתַּכֵּר אֶל צְרוֹר נָקוּב:

[תהילים פרק-נח-ד]  זֹרוּ רְשָׁעִים מֵרָחֶם תָּעוּ מִבֶּטֶן דֹּבְרֵי כָזָב:

[משלי פרק-י-ז]  זֵכֶר צַדִּיק לִבְרָכָה וְשֵׁם רְשָׁעִים יִרְקָב:

[איוב פרק-ז-ז]  זְכֹר כִּי רוּחַ חַיָּי לֹא תָשׁוּב עֵינִי לִרְאוֹת טוֹב:


ז-ד


[במדבר פרק-ח-כד]  זֹאת אֲשֶׁר לַלְוִיִּם מִבֶּן חָמֵשׁ וְעֶשְׂרִים שָׁנָה וָמַעְלָה יָבוֹא לִצְבֹא צָבָא בַּעֲבֹדַת אֹהֶל מוֹעֵד:

[איכה פרק-ב-ז]  זָנַח אֲדֹנָי מִזְבְּחוֹ נִאֵר מִקְדָּשׁוֹ הִסְגִּיר בְּיַד אוֹיֵב חוֹמֹת אַרְמְנוֹתֶיהָ קוֹל נָתְנוּ בְּבֵית יְהוָה כְּיוֹם מוֹעֵד:

[דברי הימים א פרק-ט-כא]  זְכַרְיָה בֶּן מְשֶׁלֶמְיָה שֹׁעֵר פֶּתַח לְאֹהֶל מוֹעֵד:


ז-ה


[שמות פרק-ל-יג]  זֶה יִתְּנוּ כָּל הָעֹבֵר עַל הַפְּקֻדִים מַחֲצִית הַשֶּׁקֶל בְּשֶׁקֶל הַקֹּדֶשׁ עֶשְׂרִים גֵּרָה הַשֶּׁקֶל מַחֲצִית הַשֶּׁקֶל תְּרוּמָה לַיהוָה:

[ויקרא פרק-ז-לה]  זֹאת מִשְׁחַת אַהֲרֹן וּמִשְׁחַת בָּנָיו מֵאִשֵּׁי יְהוָה בְּיוֹם הִקְרִיב אֹתָם לְכַהֵן לַיהוָה:

[ויקרא פרק-טו-לב]  זֹאת תּוֹרַת הַזָּב וַאֲשֶׁר תֵּצֵא מִמֶּנּוּ שִׁכְבַת זֶרַע לְטָמְאָה בָהּ:

[במדבר פרק-ה-כט]  זֹאת תּוֹרַת הַקְּנָאֹת אֲשֶׁר תִּשְׂטֶה אִשָּׁה תַּחַת אִישָׁהּ וְנִטְמָאָה:

[במדבר פרק-ז-פד]  זֹאת חֲנֻכַּת הַמִּזְבֵּחַ בְּיוֹם הִמָּשַׁח אֹתוֹ מֵאֵת נְשִׂיאֵי יִשְׂרָאֵל קַעֲרֹת כֶּסֶף שְׁתֵּים עֶשְׂרֵה מִזְרְקֵי כֶסֶף שְׁנֵים עָשָׂר כַּפּוֹת זָהָב שְׁתֵּים עֶשְׂרֵה:

[דברים פרק-א-לו]  זוּלָתִי כָּלֵב בֶּן יְפֻנֶּה הוּא יִרְאֶנָּה וְלוֹ אֶתֵּן אֶת הָאָרֶץ אֲשֶׁר דָּרַךְ בָּהּ וּלְבָנָיו יַעַן אֲשֶׁר מִלֵּא אַחֲרֵי יְהוָה:

[דברים פרק-ט-ז]  זְכֹר אַל תִּשְׁכַּח אֵת אֲשֶׁר הִקְצַפְתָּ אֶת יְהוָה אֱלֹהֶיךָ בַּמִּדְבָּר לְמִן הַיּוֹם אֲשֶׁר יָצָאתָ מֵאֶרֶץ מִצְרַיִם עַד בֹּאֲכֶם עַד הַמָּקוֹם הַזֶּה מַמְרִים הֱיִיתֶם עִם יְהוָה:

[דברים פרק-כח-לח]  זֶרַע רַב תּוֹצִיא הַשָּׂדֶה וּמְעַט תֶּאֱסֹף כִּי יַחְסְלֶנּוּ הָאַרְבֶּה:

[שופטים פרק-ה-יח]  זְבֻלוּן עַם חֵרֵף נַפְשׁוֹ לָמוּת וְנַפְתָּלִי עַל מְרוֹמֵי שָׂדֶה:

[ישעיה פרק-מד-ה]  זֶה יֹאמַר לַיהוָה אָנִי וְזֶה יִקְרָא בְשֵׁם יַעֲקֹב וְזֶה יִכְתֹּב יָדוֹ לַיהוָה וּבְשֵׁם יִשְׂרָאֵל יְכַנֶּה:

[ישעיה פרק-סה-כה]  זְאֵב וְטָלֶה יִרְעוּ כְאֶחָד וְאַרְיֵה כַּבָּקָר יֹאכַל תֶּבֶן וְנָחָשׁ עָפָר לַחְמוֹ לֹא יָרֵעוּ וְלֹא יַשְׁחִיתוּ בְּכָל הַר קָדְשִׁי אָמַר יְהוָה:

[ירמיה פרק-יב-יג]  זָרְעוּ חִטִּים וְקֹצִים קָצָרוּ נֶחְלוּ לֹא יוֹעִלוּ וּבֹשׁוּ מִתְּבוּאֹתֵיכֶם מֵחֲרוֹן אַף יְהוָה:

[ירמיה פרק-נב-כח]  זֶה הָעָם אֲשֶׁר הֶגְלָה נְבוּכַדְרֶאצַּר בִּשְׁנַת שֶׁבַע יְהוּדִים שְׁלֹשֶׁת אֲלָפִים וְעֶשְׂרִים וּשְׁלֹשָׁה:

[יחזקאל פרק-מח-כט]  זֹאת הָאָרֶץ אֲשֶׁר תַּפִּילוּ מִנַּחֲלָה לְשִׁבְטֵי יִשְׂרָאֵל וְאֵלֶּה מַחְלְקוֹתָם נְאֻם אֲדֹנָי יְהוִה:

[תהילים פרק-ד-ו]  זִבְחוּ זִבְחֵי צֶדֶק וּבִטְחוּ אֶל יְהוָה:

[תהילים פרק-כד-ו]  זֶה דּוֹר (דרשו) דֹּרְשָׁיו מְבַקְשֵׁי פָנֶיךָ יַעֲקֹב סֶלָה:

[תהילים פרק-כה-ו]  זְכֹר רַחֲמֶיךָ יְהוָה וַחֲסָדֶיךָ כִּי מֵעוֹלָם הֵמָּה:

[תהילים פרק-מט-יד]  זֶה דַרְכָּם כֵּסֶל לָמוֹ וְאַחֲרֵיהֶם בְּפִיהֶם יִרְצוּ סֶלָה:

[תהילים פרק-נא-יט]  זִבְחֵי אֱלֹהִים רוּחַ נִשְׁבָּרָה לֵב נִשְׁבָּר וְנִדְכֶּה אֱלֹהִים לֹא תִבְזֶה:

[תהילים פרק-צח-ה]  זַמְּרוּ לַיהוָה בְּכִנּוֹר בְּכִנּוֹר וְקוֹל זִמְרָה:

[תהילים פרק-קיא-ד]  זֵכֶר עָשָׂה לְנִפְלְאֹתָיו חַנּוּן וְרַחוּם יְהוָה:

[תהילים פרק-קלב-יד]  זֹאת מְנוּחָתִי עֲדֵי עַד פֹּה אֵשֵׁב כִּי אִוִּתִיהָ:

[תהילים פרק-קלז-ז]  זְכֹר יְהוָה לִבְנֵי אֱדוֹם אֵת יוֹם יְרוּשָׁלִָם הָאֹמְרִים עָרוּ עָרוּ עַד הַיְסוֹד בָּהּ:

[משלי פרק-כב-ח]  זוֹרֵעַ עַוְלָה (יקצור) יִקְצָר אָוֶן וְשֵׁבֶט עֶבְרָתוֹ יִכְלֶה:

[איכה פרק-א-ז]  זָכְרָה יְרוּשָׁלִַם יְמֵי עָנְיָהּ וּמְרוּדֶיהָ כֹּל מַחֲמֻדֶיהָ אֲשֶׁר הָיוּ מִימֵי קֶדֶם בִּנְפֹל עַמָּהּ בְּיַד צָר וְאֵין עוֹזֵר לָהּ רָאוּהָ צָרִים שָׂחֲקוּ עַל מִשְׁבַּתֶּהָ:

[נחמיה פרק-ה-יט]  זָכְרָה לִּי אֱלֹהַי לְטוֹבָה כֹּל אֲשֶׁר עָשִׂיתִי עַל הָעָם הַזֶּה:

[נחמיה פרק-י-יג]  זַכּוּר שֵׁרֵבְיָה שְׁבַנְיָה:


ז-ו


[בראשית פרק-ה-א]  זֶה סֵפֶר תּוֹלְדֹת אָדָם בְּיוֹם בְּרֹא אֱלֹהִים אָדָם בִּדְמוּת אֱלֹהִים עָשָׂה אֹתוֹ:

[שמות פרק-טז-טז]  זֶה הַדָּבָר אֲשֶׁר צִוָּה יְהוָה לִקְטוּ מִמֶּנּוּ אִישׁ לְפִי אָכְלוֹ עֹמֶר לַגֻּלְגֹּלֶת מִסְפַּר נַפְשֹׁתֵיכֶם אִישׁ לַאֲשֶׁר בְּאָהֳלוֹ תִּקָּחוּ:

[שמות פרק-כ-ח]  זָכוֹר אֶת יוֹם הַשַּׁבָּת לְקַדְּשׁוֹ:

[שמות פרק-כב-יט]  זֹבֵחַ לָאֱלֹהִים יָחֳרָם בִּלְתִּי לַיהוָה לְבַדּוֹ:

[ויקרא פרק-יג-נט]  זֹאת תּוֹרַת נֶגַע צָרַעַת בֶּגֶד הַצֶּמֶר אוֹ הַפִּשְׁתִּים אוֹ הַשְּׁתִי אוֹ הָעֵרֶב אוֹ כָּל כְּלִי עוֹר לְטַהֲרוֹ אוֹ לְטַמְּאוֹ:

[ויקרא פרק-יד-לב]  זֹאת תּוֹרַת אֲשֶׁר בּוֹ נֶגַע צָרָעַת אֲשֶׁר לֹא תַשִּׂיג יָדוֹ בְּטָהֳרָתוֹ:

[במדבר פרק-ו-כא]  זֹאת תּוֹרַת הַנָּזִיר אֲשֶׁר יִדֹּר קָרְבָּנוֹ לַיהוָה עַל נִזְרוֹ מִלְּבַד אֲשֶׁר תַּשִּׂיג יָדוֹ כְּפִי נִדְרוֹ אֲשֶׁר יִדֹּר כֵּן יַעֲשֶׂה עַל תּוֹרַת נִזְרוֹ:

[במדבר פרק-טז-ו]  זֹאת עֲשׂוּ קְחוּ לָכֶם מַחְתּוֹת קֹרַח וְכָל עֲדָתוֹ:

[במדבר פרק-יז-ה]  זִכָּרוֹן לִבְנֵי יִשְׂרָאֵל לְמַעַן אֲשֶׁר לֹא יִקְרַב אִישׁ זָר אֲשֶׁר לֹא מִזֶּרַע אַהֲרֹן הוּא לְהַקְטִיר קְטֹרֶת לִפְנֵי יְהוָה וְלֹא יִהְיֶה כְקֹרַח וְכַעֲדָתוֹ כַּאֲשֶׁר דִּבֶּר יְהוָה בְּיַד מֹשֶׁה לוֹ:

[דברים פרק-ט-כז]  זְכֹר לַעֲבָדֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב אַל תֵּפֶן אֶל קְשִׁי הָעָם הַזֶּה וְאֶל רִשְׁעוֹ וְאֶל חַטָּאתוֹ:

[ירמיה פרק-יא-טז]  זַיִת רַעֲנָן יְפֵה פְרִי תֹאַר קָרָא יְהוָה שְׁמֵךְ לְקוֹל הֲמוּלָּה גְדֹלָה הִצִּית אֵשׁ עָלֶיהָ וְרָעוּ דָּלִיּוֹתָיו:

[הושע פרק-ח-ג]  זָנַח יִשְׂרָאֵל טוֹב אוֹיֵב יִרְדְּפוֹ:

[הושע פרק-ח-יג]  זִבְחֵי הַבְהָבַי יִזְבְּחוּ בָשָׂר וַיֹּאכֵלוּ יְהוָה לֹא רָצָם עַתָּה יִזְכֹּר עֲוֹנָם וְיִפְקֹד חַטֹּאותָם הֵמָּה מִצְרַיִם יָשׁוּבוּ:

[מיכה פרק-ז-ט]  זַעַף יְהוָה אֶשָּׂא כִּי חָטָאתִי לוֹ עַד אֲשֶׁר יָרִיב רִיבִי וְעָשָׂה מִשְׁפָּטִי יוֹצִיאֵנִי לָאוֹר אֶרְאֶה בְּצִדְקָתוֹ:

[צפניה פרק-ב-טו]  זֹאת הָעִיר הָעַלִּיזָה הַיּוֹשֶׁבֶת לָבֶטַח הָאֹמְרָה בִּלְבָבָהּ אֲנִי וְאַפְסִי עוֹד אֵיךְ הָיְתָה לְשַׁמָּה מַרְבֵּץ לַחַיָּה כֹּל עוֹבֵר עָלֶיהָ יִשְׁרֹק יָנִיעַ יָדוֹ:

[תהילים פרק-ט-יב]  זַמְּרוּ לַיהוָה יֹשֵׁב צִיּוֹן הַגִּידוּ בָעַמִּים עֲלִילוֹתָיו:

[תהילים פרק-ל-ה]  זַמְּרוּ לַיהוָה חֲסִידָיו וְהוֹדוּ לְזֵכֶר קָדְשׁוֹ:

[תהילים פרק-לד-ז]  זֶה עָנִי קָרָא וַיהוָה שָׁמֵעַ וּמִכָּל צָרוֹתָיו הוֹשִׁיעוֹ:

[תהילים פרק-לז-יב]  זֹמֵם רָשָׁע לַצַּדִּיק וְחֹרֵק עָלָיו שִׁנָּיו:

[תהילים פרק-מז-ז]  זַמְּרוּ אֱלֹהִים זַמֵּרוּ זַמְּרוּ לְמַלְכֵּנוּ זַמֵּרוּ:

[תהילים פרק-סו-ב]  זַמְּרוּ כְבוֹד שְׁמוֹ שִׂימוּ כָבוֹד תְּהִלָּתוֹ:

[תהילים פרק-עד-ב]  זְכֹר עֲדָתְךָ קָנִיתָ קֶּדֶם גָּאַלְתָּ שֵׁבֶט נַחֲלָתֶךָ הַר צִיּוֹן זֶה שָׁכַנְתָּ בּוֹ:

[תהילים פרק-עז-יח]  זֹרְמוּ מַיִם עָבוֹת קוֹל נָתְנוּ שְׁחָקִים אַף חֲצָצֶיךָ יִתְהַלָּכוּ:

[תהילים פרק-צח-ג]  זָכַר חַסְדּוֹ וֶאֱמוּנָתוֹ לְבֵית יִשְׂרָאֵל רָאוּ כָל אַפְסֵי אָרֶץ אֵת יְשׁוּעַת אֱלֹהֵינוּ:

[תהילים פרק-קה-ה]  זִכְרוּ נִפְלְאוֹתָיו אֲשֶׁר עָשָׂה מֹפְתָיו וּמִשְׁפְּטֵי פִיו:

[תהילים פרק-קה-ו]  זֶרַע אַבְרָהָם עַבְדּוֹ בְּנֵי יַעֲקֹב בְּחִירָיו:

[תהילים פרק-קיח-כ]  זֶה הַשַּׁעַר לַיהוָה צַדִּיקִים יָבֹאוּ בוֹ:

[תהילים פרק-קיח-כד]  זֶה הַיּוֹם עָשָׂה יְהוָה נָגִילָה וְנִשְׂמְחָה בוֹ:

[תהילים פרק-קמה-ז]  זֵכֶר רַב טוּבְךָ יַבִּיעוּ וְצִדְקָתְךָ יְרַנֵּנוּ:

[משלי פרק-טו-ח]  זֶבַח רְשָׁעִים תּוֹעֲבַת יְהוָה וּתְפִלַּת יְשָׁרִים רְצוֹנוֹ:

[משלי פרק-כא-כז]  זֶבַח רְשָׁעִים תּוֹעֵבָה אַף כִּי בְזִמָּה יְבִיאֶנּוּ:

[משלי פרק-ל-לא]  זַרְזִיר מָתְנַיִם אוֹ תָיִשׁ וּמֶלֶךְ אַלְקוּם עִמּוֹ:

[איוב פרק-ד-ז]  זְכָר נָא מִי הוּא נָקִי אָבָד וְאֵיפֹה יְשָׁרִים נִכְחָדוּ:

[איוב פרק-כא-כג]  זֶה יָמוּת בְּעֶצֶם תֻּמּוֹ כֻּלּוֹ שַׁלְאֲנַן וְשָׁלֵיו:

[איוב פרק-כז-יג]  זֶה חֵלֶק אָדָם רָשָׁע עִם אֵל וְנַחֲלַת עָרִיצִים מִשַּׁדַּי יִקָּחוּ:

[איכה פרק-ה-א]  זְכֹר יְהוָה מֶה הָיָה לָנוּ (הביט) הַבִּיטָה וּרְאֵה אֶת חֶרְפָּתֵנוּ:

[נחמיה פרק-יג-יד]  זָכְרָה לִּי אֱלֹהַי עַל זֹאת וְאַל תֶּמַח חֲסָדַי אֲשֶׁר עָשִׂיתִי בְּבֵית אֱלֹהַי וּבְמִשְׁמָרָיו:

[דברי הימים א פרק-טז-יב]  זִכְרוּ נִפְלְאֹתָיו אֲשֶׁר עָשָׂה מֹפְתָיו וּמִשְׁפְּטֵי פִיהוּ:

[דברי הימים א פרק-טז-יג]  זֶרַע יִשְׂרָאֵל עַבְדּוֹ בְּנֵי יַעֲקֹב בְּחִירָיו:


ז-ז


[ישעיה פרק-טז-יג]  זֶה הַדָּבָר אֲשֶׁר דִּבֶּר יְהוָה אֶל מוֹאָב מֵאָז:


ז-ח


[תהילים פרק-קמג-ה]  זָכַרְתִּי יָמִים מִקֶּדֶם הָגִיתִי בְכָל פָּעֳלֶךָ בְּמַעֲשֵׂה יָדֶיךָ אֲשׂוֹחֵחַ:


ז-ט


[קהלת פרק-י-א]  זְבוּבֵי מָוֶת יַבְאִישׁ יַבִּיעַ שֶׁמֶן רוֹקֵחַ יָקָר מֵחָכְמָה מִכָּבוֹד סִכְלוּת מְעָט:


ז-י


[בראשית פרק-לא-לח]  זֶה עֶשְׂרִים שָׁנָה אָנֹכִי עִמָּךְ רְחֵלֶיךָ וְעִזֶּיךָ לֹא שִׁכֵּלוּ וְאֵילֵי צֹאנְךָ לֹא אָכָלְתִּי:

[יהושע פרק-יג-ב]  זֹאת הָאָרֶץ הַנִּשְׁאָרֶת כָּל גְּלִילוֹת הַפְּלִשְׁתִּים וְכָל הַגְּשׁוּרִי:

[ישעיה פרק-מד-כא]  זְכָר אֵלֶּה יַעֲקֹב וְיִשְׂרָאֵל כִּי עַבְדִּי אָתָּה יְצַרְתִּיךָ עֶבֶד לִי אַתָּה יִשְׂרָאֵל לֹא תִנָּשֵׁנִי:

[ישעיה פרק-מו-ט]  זִכְרוּ רִאשֹׁנוֹת מֵעוֹלָם כִּי אָנֹכִי אֵל וְאֵין עוֹד אֱלֹהִים וְאֶפֶס כָּמוֹנִי:

[תהילים פרק-פט-לז]  זַרְעוֹ לְעוֹלָם יִהְיֶה וְכִסְאוֹ כַשֶּׁמֶשׁ נֶגְדִּי:

[תהילים פרק-קט-כ]  זֹאת פְּעֻלַּת שֹׂטְנַי מֵאֵת יְהוָה וְהַדֹּבְרִים רָע עַל נַפְשִׁי:

[תהילים פרק-קיט-מט]  זְכֹר דָּבָר לְעַבְדֶּךָ עַל אֲשֶׁר יִחַלְתָּנִי:

[תהילים פרק-קיט-נ]  זֹאת נֶחָמָתִי בְעָנְיִי כִּי אִמְרָתְךָ חִיָּתְנִי:

[תהילים פרק-קיט-נא]  זֵדִים הֱלִיצֻנִי עַד מְאֹד מִתּוֹרָתְךָ לֹא נָטִיתִי:

[תהילים פרק-קיט-נד]  זְמִרוֹת הָיוּ לִי חֻקֶּיךָ בְּבֵית מְגוּרָי:

[תהילים פרק-קיט-נו]  זֹאת הָיְתָה לִּי כִּי פִקֻּדֶיךָ נָצָרְתִּי:

[משלי פרק-ז-יד]  זִבְחֵי שְׁלָמִים עָלָי הַיּוֹם שִׁלַּמְתִּי נְדָרָי:

[איוב פרק-י-ט]  זְכָר נָא כִּי כַחֹמֶר עֲשִׂיתָנִי וְאֶל עָפָר תְּשִׁיבֵנִי:

[איוב פרק-יט-ג]  זֶה עֶשֶׂר פְּעָמִים תַּכְלִימוּנִי לֹא תֵבֹשׁוּ תַּהְכְּרוּ לִי:

[איוב פרק-לג-ט]  זַךְ אֲנִי בְּלִי פָשַׁע חַף אָנֹכִי וְלֹא עָוֹן לִי:

[איכה פרק-ג-כ]  זָכוֹר תִּזְכּוֹר (ותשיח) וְתָשׁוֹחַ עָלַי נַפְשִׁי:

[נחמיה פרק-ו-יד]  זָכְרָה אֱלֹהַי לְטוֹבִיָּה וּלְסַנְבַלַּט כְּמַעֲשָׂיו אֵלֶּה וְגַם לְנוֹעַדְיָה הַנְּבִיאָה וּלְיֶתֶר הַנְּבִיאִים אֲשֶׁר הָיוּ מְיָרְאִים אוֹתִי:


ז-כ


[במדבר פרק-יח-ט]  זֶה יִהְיֶה לְךָ מִקֹּדֶשׁ הַקֳּדָשִׁים מִן הָאֵשׁ כָּל קָרְבָּנָם לְכָל מִנְחָתָם וּלְכָל חַטָּאתָם וּלְכָל אֲשָׁמָם אֲשֶׁר יָשִׁיבוּ לִי קֹדֶשׁ קָדָשִׁים לְךָ הוּא וּלְבָנֶיךָ:

[דברים פרק-כח-מ]  זֵיתִים יִהְיוּ לְךָ בְּכָל גְּבוּלֶךָ וְשֶׁמֶן לֹא תָסוּךְ כִּי יִשַּׁל זֵיתֶךָ:

[דברים פרק-לב-ז]  זְכֹר יְמוֹת עוֹלָם בִּינוּ שְׁנוֹת דּוֹר וָדוֹר שְׁאַל אָבִיךָ וְיַגֵּדְךָ זְקֵנֶיךָ וְיֹאמְרוּ לָךְ:

[יחזקאל פרק-כא-יז]  זְעַק וְהֵילֵל בֶּן אָדָם כִּי הִיא הָיתָה בְעַמִּי הִיא בְּכָל נְשִׂיאֵי יִשְׂרָאֵל מְגוּרֵי אֶל חֶרֶב הָיוּ אֶת עַמִּי לָכֵן סְפֹק אֶל יָרֵךְ:

[יחזקאל פרק-כז-ט]  זִקְנֵי גְבַל וַחֲכָמֶיהָ הָיוּ בָךְ מַחֲזִיקֵי בִּדְקֵךְ כָּל אֳנִיּוֹת הַיָּם וּמַלָּחֵיהֶם הָיוּ בָךְ לַעֲרֹב מַעֲרָבֵךְ:

[תהילים פרק-נ-יד]  זְבַח לֵאלֹהִים תּוֹדָה וְשַׁלֵּם לְעֶלְיוֹן נְדָרֶיךָ:

[תהילים פרק-עד-יח]  זְכָר זֹאת אוֹיֵב חֵרֵף יְהוָה וְעַם נָבָל נִאֲצוּ שְׁמֶךָ:

[תהילים פרק-קו-ד]  זָכְרֵנִי יְהוָה בִּרְצוֹן עַמֶּךָ פָּקְדֵנִי בִּישׁוּעָתֶךָ:

[תהילים פרק-קיט-נג]  זַלְעָפָה אֲחָזַתְנִי מֵרְשָׁעִים עֹזְבֵי תּוֹרָתֶךָ:

[תהילים פרק-קיט-נה]  זָכַרְתִּי בַלַּיְלָה שִׁמְךָ יְהוָה וָאֶשְׁמְרָה תּוֹרָתֶךָ:


ז-ל


[במדבר פרק-יט-ב]  זֹאת חֻקַּת הַתּוֹרָה אֲשֶׁר צִוָּה יְהוָה לֵאמֹר דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְיִקְחוּ אֵלֶיךָ פָרָה אֲדֻמָּה תְּמִימָה אֲשֶׁר אֵין בָּהּ מוּם אֲשֶׁר לֹא עָלָה עָלֶיהָ עֹל:

[איוב פרק-כ-כט]  זֶה חֵלֶק אָדָם רָשָׁע מֵאֱלֹהִים וְנַחֲלַת אִמְרוֹ מֵאֵל:

[איכה פרק-ג-כא]  זֹאת אָשִׁיב אֶל לִבִּי עַל כֵּן אוֹחִיל:


ז-מ


[בראשית פרק-ה-ב]  זָכָר וּנְקֵבָה בְּרָאָם וַיְבָרֶךְ אֹתָם וַיִּקְרָא אֶת שְׁמָם אָדָם בְּיוֹם הִבָּרְאָם:

[בראשית פרק-לא-מא]  זֶה לִּי עֶשְׂרִים שָׁנָה בְּבֵיתֶךָ עֲבַדְתִּיךָ אַרְבַּע עֶשְׂרֵה שָׁנָה בִּשְׁתֵּי בְנֹתֶיךָ וְשֵׁשׁ שָׁנִים בְּצֹאנֶךָ וַתַּחֲלֵף אֶת מַשְׂכֻּרְתִּי עֲשֶׂרֶת מֹנִים:

[שמות פרק-לב-יג]  זְכֹר לְאַבְרָהָם לְיִצְחָק וּלְיִשְׂרָאֵל עֲבָדֶיךָ אֲשֶׁר נִשְׁבַּעְתָּ לָהֶם בָּךְ וַתְּדַבֵּר אֲלֵהֶם אַרְבֶּה אֶת זַרְעֲכֶם כְּכוֹכְבֵי הַשָּׁמָיִם וְכָל הָאָרֶץ הַזֹּאת אֲשֶׁר אָמַרְתִּי אֶתֵּן לְזַרְעֲכֶם וְנָחֲלוּ לְעֹלָם:

[ויקרא פרק-ז-לז]  זֹאת הַתּוֹרָה לָעֹלָה לַמִּנְחָה וְלַחַטָּאת וְלָאָשָׁם וְלַמִּלּוּאִים וּלְזֶבַח הַשְּׁלָמִים:

[במדבר פרק-ד-ד]  זֹאת עֲבֹדַת בְּנֵי קְהָת בְּאֹהֶל מוֹעֵד קֹדֶשׁ הַקֳּדָשִׁים:

[במדבר פרק-יא-ה]  זָכַרְנוּ אֶת הַדָּגָה אֲשֶׁר נֹאכַל בְּמִצְרַיִם חִנָּם אֵת הַקִּשֻּׁאִים וְאֵת הָאֲבַטִּחִים וְאֶת הֶחָצִיר וְאֶת הַבְּצָלִים וְאֶת הַשּׁוּמִים:

[במדבר פרק-יט-יד]  זֹאת הַתּוֹרָה אָדָם כִּי יָמוּת בְּאֹהֶל כָּל הַבָּא אֶל הָאֹהֶל וְכָל אֲשֶׁר בָּאֹהֶל יִטְמָא שִׁבְעַת יָמִים:

[במדבר פרק-לו-ו]  זֶה הַדָּבָר אֲשֶׁר צִוָּה יְהוָה לִבְנוֹת צְלָפְחָד לֵאמֹר לַטּוֹב בְּעֵינֵיהֶם תִּהְיֶינָה לְנָשִׁים אַךְ לְמִשְׁפַּחַת מַטֵּה אֲבִיהֶם תִּהְיֶינָה לְנָשִׁים:

[דברים פרק-יד-ד]  זֹאת הַבְּהֵמָה אֲשֶׁר תֹּאכֵלוּ שׁוֹר שֵׂה כְשָׂבִים וְשֵׂה עִזִּים:

[דברים פרק-כד-ט]  זָכוֹר אֵת אֲשֶׁר עָשָׂה יְהוָה אֱלֹהֶיךָ לְמִרְיָם בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם:

[דברים פרק-כה-יז]  זָכוֹר אֵת אֲשֶׁר עָשָׂה לְךָ עֲמָלֵק בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם:

[יהושע פרק-ט-יב]  זֶה לַחְמֵנוּ חָם הִצְטַיַּדְנוּ אֹתוֹ מִבָּתֵּינוּ בְּיוֹם צֵאתֵנוּ לָלֶכֶת אֲלֵיכֶם וְעַתָּה הִנֵּה יָבֵשׁ וְהָיָה נִקֻּדִים:

[יהושע פרק-ט-כ]  זֹאת נַעֲשֶׂה לָהֶם וְהַחֲיֵה אוֹתָם וְלֹא יִהְיֶה עָלֵינוּ קֶצֶף עַל הַשְּׁבוּעָה אֲשֶׁר נִשְׁבַּעְנוּ לָהֶם:

[יהושע פרק-יג-כח]  זֹאת נַחֲלַת בְּנֵי גָד לְמִשְׁפְּחֹתָם הֶעָרִים וְחַצְרֵיהֶם:

[יהושע פרק-טו-כ]  זֹאת נַחֲלַת מַטֵּה בְנֵי יְהוּדָה לְמִשְׁפְּחֹתָם:

[מלכים ב פרק-יט-כא]  זֶה הַדָּבָר אֲשֶׁר דִּבֶּר יְהוָה עָלָיו בָּזָה לְךָ לָעֲגָה לְךָ בְּתוּלַת בַּת צִיּוֹן אַחֲרֶיךָ רֹאשׁ הֵנִיעָה בַּת יְרוּשָׁלִָם:

[ישעיה פרק-יד-כו]  זֹאת הָעֵצָה הַיְּעוּצָה עַל כָּל הָאָרֶץ וְזֹאת הַיָּד הַנְּטוּיָה עַל כָּל הַגּוֹיִם:

[ישעיה פרק-לז-כב]  זֶה הַדָּבָר אֲשֶׁר דִּבֶּר יְהוָה עָלָיו בָּזָה לְךָ לָעֲגָה לְךָ בְּתוּלַת בַּת צִיּוֹן אַחֲרֶיךָ רֹאשׁ הֵנִיעָה בַּת יְרוּשָׁלִָם:

[יחזקאל פרק-מה-יג]  זֹאת הַתְּרוּמָה אֲשֶׁר תָּרִימוּ שִׁשִּׁית הָאֵיפָה מֵחֹמֶר הַחִטִּים וְשִׁשִּׁיתֶם הָאֵיפָה מֵחֹמֶר הַשְּׂעֹרִים:

[הושע פרק-י-יב]  זִרְעוּ לָכֶם לִצְדָקָה קִצְרוּ לְפִי חֶסֶד נִירוּ לָכֶם נִיר וְעֵת לִדְרוֹשׁ אֶת יְהוָה עַד יָבוֹא וְיֹרֶה צֶדֶק לָכֶם:

[מלאכי פרק-ג-כב]  זִכְרוּ תּוֹרַת מֹשֶׁה עַבְדִּי אֲשֶׁר צִוִּיתִי אוֹתוֹ בְחֹרֵב עַל כָּל יִשְׂרָאֵל חֻקִּים וּמִשְׁפָּטִים:

[תהילים פרק-נ-כג]  זֹבֵחַ תּוֹדָה יְכַבְּדָנְנִי וְשָׂם דֶּרֶךְ אַרְאֶנּוּ בְּיֵשַׁע אֱלֹהִים:

[תהילים פרק-פט-מח]  זְכָר אֲנִי מֶה חָלֶד עַל מַה שָּׁוְא בָּרָאתָ כָל בְּנֵי אָדָם:

[תהילים פרק-פט-נא]  זְכֹר אֲדֹנָי חֶרְפַּת עֲבָדֶיךָ שְׂאֵתִי בְחֵיקִי כָּל רַבִּים עַמִּים:

[תהילים פרק-קיט-נב]  זָכַרְתִּי מִשְׁפָּטֶיךָ מֵעוֹלָם יְהוָה וָאֶתְנֶחָם:

[תהילים פרק-קמב-ו]  זָעַקְתִּי אֵלֶיךָ יְהוָה אָמַרְתִּי אַתָּה מַחְסִי חֶלְקִי בְּאֶרֶץ הַחַיִּים:

[משלי פרק-לא-טז]  זָמְמָה שָׂדֶה וַתִּקָּחֵהוּ מִפְּרִי כַפֶּיהָ (נטע) נָטְעָה כָּרֶם:

[איוב פרק-יג-יב]  זִכְרֹנֵיכֶם מִשְׁלֵי אֵפֶר לְגַבֵּי חֹמֶר גַּבֵּיכֶם:

[איוב פרק-כא-ח]  זַרְעָם נָכוֹן לִפְנֵיהֶם עִמָּם וְצֶאֱצָאֵיהֶם לְעֵינֵיהֶם:

[איוב פרק-לו-כד]  זְכֹר כִּי תַשְׂגִּיא פָעֳלוֹ אֲשֶׁר שֹׁרְרוּ אֲנָשִׁים:

[איכה פרק-ד-ז]  זַכּוּ נְזִירֶיהָ מִשֶּׁלֶג צַחוּ מֵחָלָב אָדְמוּ עֶצֶם מִפְּנִינִים סַפִּיר גִּזְרָתָם:

[איכה פרק-ה-יד]  זְקֵנִים מִשַּׁעַר שָׁבָתוּ בַּחוּרִים מִנְּגִינָתָם:

[קהלת פרק-ט-ג]  זֶה רָע בְּכֹל אֲשֶׁר נַעֲשָׂה תַּחַת הַשֶּׁמֶשׁ כִּי מִקְרֶה אֶחָד לַכֹּל וְגַם לֵב בְּנֵי הָאָדָם מָלֵא רָע וְהוֹלֵלוֹת בִּלְבָבָם בְּחַיֵּיהֶם וְאַחֲרָיו אֶל הַמֵּתִים:

[נחמיה פרק-א-ח]  זְכָר נָא אֶת הַדָּבָר אֲשֶׁר צִוִּיתָ אֶת מֹשֶׁה עַבְדְּךָ לֵאמֹר אַתֶּם תִּמְעָלוּ אֲנִי אָפִיץ אֶתְכֶם בָּעַמִּים:

[נחמיה פרק-יא-ל]  זָנֹחַ עֲדֻלָּם וְחַצְרֵיהֶם לָכִישׁ וּשְׂדֹתֶיהָ עֲזֵקָה וּבְנֹתֶיהָ וַיַּחֲנוּ מִבְּאֵר שֶׁבַע עַד גֵּיא הִנֹּם:

[נחמיה פרק-יג-כט]  זָכְרָה לָהֶם אֱלֹהָי עַל גָּאֳלֵי הַכְּהֻנָּה וּבְרִית הַכְּהֻנָּה וְהַלְוִיִּם:

[דברי הימים ב פרק-כג-ד]  זֶה הַדָּבָר אֲשֶׁר תַּעֲשׂוּ הַשְּׁלִשִׁית מִכֶּם בָּאֵי הַשַּׁבָּת לַכֹּהֲנִים וְלַלְוִיִּם לְשֹׁעֲרֵי הַסִּפִּים:


ז-נ


[בראשית פרק-מט-יג]  זְבוּלֻן לְחוֹף יַמִּים יִשְׁכֹּן וְהוּא לְחוֹף אֳנִיּוֹת וְיַרְכָתוֹ עַל צִידֹן:

[ויקרא פרק-יד-ב]  זֹאת תִּהְיֶה תּוֹרַת הַמְּצֹרָע בְּיוֹם טָהֳרָתוֹ וְהוּבָא אֶל הַכֹּהֵן:

[במדבר פרק-ד-כח]  זֹאת עֲבֹדַת מִשְׁפְּחֹת בְּנֵי הַגֵּרְשֻׁנִּי בְּאֹהֶל מוֹעֵד וּמִשְׁמַרְתָּם בְּיַד אִיתָמָר בֶּן אַהֲרֹן הַכֹּהֵן:

[במדבר פרק-ד-לג]  זֹאת עֲבֹדַת מִשְׁפְּחֹת בְּנֵי מְרָרִי לְכָל עֲבֹדָתָם בְּאֹהֶל מוֹעֵד בְּיַד אִיתָמָר בֶּן אַהֲרֹן הַכֹּהֵן:

[יהושע פרק-יט-טז]  זֹאת נַחֲלַת בְּנֵי זְבוּלֻן לְמִשְׁפְּחוֹתָם הֶעָרִים הָאֵלֶּה וְחַצְרֵיהֶן:

[יהושע פרק-יט-כג]  זֹאת נַחֲלַת מַטֵּה בְנֵי יִשָּׂשכָר לְמִשְׁפְּחֹתָם הֶעָרִים וְחַצְרֵיהֶן:

[יהושע פרק-יט-לא]  זֹאת נַחֲלַת מַטֵּה בְנֵי אָשֵׁר לְמִשְׁפְּחֹתָם הֶעָרִים הָאֵלֶּה וְחַצְרֵיהֶן:

[יהושע פרק-יט-לט]  זֹאת נַחֲלַת מַטֵּה בְנֵי נַפְתָּלִי לְמִשְׁפְּחֹתָם הֶעָרִים וְחַצְרֵיהֶן:

[יהושע פרק-יט-מח]  זֹאת נַחֲלַת מַטֵּה בְנֵי דָן לְמִשְׁפְּחֹתָם הֶעָרִים הָאֵלֶּה וְחַצְרֵיהֶן:

[הושע פרק-ח-ה]  זָנַח עֶגְלֵךְ שֹׁמְרוֹן חָרָה אַפִּי בָּם עַד מָתַי לֹא יוּכְלוּ נִקָּיֹן:

[משלי פרק-כא-כד]  זֵד יָהִיר לֵץ שְׁמוֹ עוֹשֶׂה בְּעֶבְרַת זָדוֹן:


ז-ס


[שופטים פרק-א-ל]  זְבוּלֻן לֹא הוֹרִישׁ אֶת יוֹשְׁבֵי קִטְרוֹן וְאֶת יוֹשְׁבֵי נַהֲלֹל וַיֵּשֶׁב הַכְּנַעֲנִי בְּקִרְבּוֹ וַיִּהְיוּ לָמַס:


ז-פ


[תהילים פרק-צ-ה]  זְרַמְתָּם שֵׁנָה יִהְיוּ בַּבֹּקֶר כֶּחָצִיר יַחֲלֹף:


ז-צ


[ויקרא פרק-יא-מו]  זֹאת תּוֹרַת הַבְּהֵמָה וְהָעוֹף וְכֹל נֶפֶשׁ הַחַיָּה הָרֹמֶשֶׂת בַּמָּיִם וּלְכָל נֶפֶשׁ הַשֹּׁרֶצֶת עַל הָאָרֶץ:

[ישעיה פרק-יב-ה]  זַמְּרוּ יְהוָה כִּי גֵאוּת עָשָׂה (מידעת) מוּדַעַת זֹאת בְּכָל הָאָרֶץ:

[עובדיה פרק-א-ג]  זְדוֹן לִבְּךָ הִשִּׁיאֶךָ שֹׁכְנִי בְחַגְוֵי סֶּלַע מְרוֹם שִׁבְתּוֹ אֹמֵר בְּלִבּוֹ מִי יוֹרִדֵנִי אָרֶץ:

[משלי פרק-כד-ט]  זִמַּת אִוֶּלֶת חַטָּאת וְתוֹעֲבַת לְאָדָם לֵץ:

[איוב פרק-יח-יז]  זִכְרוֹ אָבַד מִנִּי אָרֶץ וְלֹא שֵׁם לוֹ עַל פְּנֵי חוּץ:


ז-ק


[ויקרא פרק-יד-נד]  זֹאת הַתּוֹרָה לְכָל נֶגַע הַצָּרַעַת וְלַנָּתֶק:

[תהילים פרק-קיב-ד]  זָרַח בַּחֹשֶׁךְ אוֹר לַיְשָׁרִים חַנּוּן וְרַחוּם וְצַדִּיק:


ז-ר


[בראשית פרק-יז-י]  זֹאת בְּרִיתִי אֲשֶׁר תִּשְׁמְרוּ בֵּינִי וּבֵינֵיכֶם וּבֵין זַרְעֲךָ אַחֲרֶיךָ הִמּוֹל לָכֶם כָּל זָכָר:

[ירמיה פרק-יג-כה]  זֶה גוֹרָלֵךְ מְנָת מִדַּיִךְ מֵאִתִּי נְאֻם יְהוָה אֲשֶׁר שָׁכַחַתְּ אוֹתִי וַתִּבְטְחִי בַּשָּׁקֶר:

[תהילים פרק-כב-לא]  זֶרַע יַעַבְדֶנּוּ יְסֻפַּר לַאדֹנָי לַדּוֹר:

[תהילים פרק-קה-ח]  זָכַר לְעוֹלָם בְּרִיתוֹ דָּבָר צִוָּה לְאֶלֶף דּוֹר:

[דברי הימים א פרק-טז-טו]  זִכְרוּ לְעוֹלָם בְּרִיתוֹ דָּבָר צִוָּה לְאֶלֶף דּוֹר:


ז-ש


[איכה פרק-ג-יט]  זְכָר עָנְיִי וּמְרוּדִי לַעֲנָה וָרֹאשׁ:


ז-ת


[יהושע פרק-א-יג]  זָכוֹר אֶת הַדָּבָר אֲשֶׁר צִוָּה אֶתְכֶם מֹשֶׁה עֶבֶד יְהוָה לֵאמֹר יְהוָה אֱלֹהֵיכֶם מֵנִיחַ לָכֶם וְנָתַן לָכֶם אֶת הָאָרֶץ הַזֹּאת:

[יהושע פרק-טו-כד]  זִיף וָטֶלֶם וּבְעָלוֹת:

[יחזקאל פרק-ט-ו]  זָקֵן בָּחוּר וּבְתוּלָה וְטַף וְנָשִׁים תַּהַרְגוּ לְמַשְׁחִית וְעַל כָּל אִישׁ אֲשֶׁר עָלָיו הַתָּו אַל תִּגַּשׁוּ וּמִמִּקְדָּשִׁי תָּחֵלּוּ וַיָּחֵלּוּ בָּאֲנָשִׁים הַזְּקֵנִים אֲשֶׁר לִפְנֵי הַבָּיִת:

[יחזקאל פרק-מג-יב]  זֹאת תּוֹרַת הַבָּיִת עַל רֹאשׁ הָהָר כָּל גְּבֻלוֹ סָבִיב סָבִיב קֹדֶשׁ קָדָשִׁים הִנֵּה זֹאת תּוֹרַת הַבָּיִת:

[צפניה פרק-ב-י]  זֹאת לָהֶם תַּחַת גְּאוֹנָם כִּי חֵרְפוּ וַיַּגְדִּלוּ עַל עַם יְהוָה צְבָאוֹת:

[זכריה פרק-יד-יט]  זֹאת תִּהְיֶה חַטַּאת מִצְרָיִם וְחַטַּאת כָּל הַגּוֹיִם אֲשֶׁר לֹא יַעֲלוּ לָחֹג אֶת חַג הַסֻּכּוֹת:

[תהילים פרק-מ-ז]  זֶבַח וּמִנְחָה לֹא חָפַצְתָּ אָזְנַיִם כָּרִיתָ לִּי עוֹלָה וַחֲטָאָה לֹא שָׁאָלְתָּ:

[תהילים פרק-קד-כה]  זֶה הַיָּם גָּדוֹל וּרְחַב יָדָיִם שָׁם רֶמֶשׂ וְאֵין מִסְפָּר חַיּוֹת קְטַנּוֹת עִם גְּדֹלוֹת:

[שיר השירים פרק-ז-ח]  זֹאת קוֹמָתֵךְ דָּמְתָה לְתָמָר וְשָׁדַיִךְ לְאַשְׁכֹּלוֹת:
