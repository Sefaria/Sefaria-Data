פרשת שופטים


{יח} שֹׁפְטִים וְשֹׁטְרִים תִּתֶּן לְךָ בְּכָל שְׁעָרֶיךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לִשְׁבָטֶיךָ וְשָׁפְטוּ אֶת הָעָם מִשְׁפַּט צֶדֶק:  מ  שֹׁפְטִים וְשֹׁטְרִים תִּתֶּן לְךָ בְּכָל שְׁעָרֶיךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לִשְׁבָטֶיךָ וְשָׁפְטוּ אֶת הָעָם מִשְׁפַּט צֶדֶק:  ת  דַיָנִין וּפֻרְעָנִין תְּמַנֵי לָךְ בְּכֹל קִרְוָיךְ דִי יְיָ אֱלָהָךְ יָהֵב לָךְ לְשִׁבְטָיךְ וִידוּנוּן יָת עַמָא דִין דִקְשׁוֹט:

{יט} לֹא תַטֶּה מִשְׁפָּט לֹא תַכִּיר פָּנִים וְלֹא תִקַּח שֹׁחַד כִּי הַשֹּׁחַד יְעַוֵּר עֵינֵי חֲכָמִים וִיסַלֵּף דִּבְרֵי צַדִּיקִם:  מ  לֹא תַטֶּה מִשְׁפָּט לֹא תַכִּיר פָּנִים וְלֹא תִקַּח שֹׁחַד כִּי הַשֹּׁחַד יְעַוֵּר עֵינֵי חֲכָמִים וִיסַלֵּף דִּבְרֵי צַדִּיקִם:  ת  לָא תַצְלֵי דִין לָא תִשְׁתְּמוֹדַע אַפִּין וְלָא תְקַבֵּל שֹׁחֲדָא אֲרֵי שֹׁחֲדָא מְעַוֵּר עֵינֵי חַכִּימִין וּמְקַלְקַל פִּתְגָמִין תְּרִיצִין:

{כ} צֶדֶק צֶדֶק תִּרְדֹּף לְמַעַן תִּחְיֶה וְיָרַשְׁתָּ אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:  מ  צֶדֶק צֶדֶק תִּרְדֹּף לְמַעַן תִּחְיֶה וְיָרַשְׁתָּ אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:  ת  קֻשְׁטָא קֻשְׁטָא תְּהֵי רָדִיף בְּדִיל דְתֵיחֵי וְתֵירַת יָת אַרְעָא דַיְיָ אֱלָהָךְ יָהֵב לָךְ: (ס)

{כא} לֹא תִטַּע לְךָ אֲשֵׁרָה כָּל עֵץ אֵצֶל מִזְבַּח יְהוָה אֱלֹהֶיךָ אֲשֶׁר תַּעֲשֶׂה לָּךְ:  מ  לֹא תִטַּע לְךָ אֲשֵׁרָה כָּל עֵץ אֵצֶל מִזְבַּח יְהוָה אֱלֹהֶיךָ אֲשֶׁר תַּעֲשֶׂה לָּךְ:  ת  לָא תִצוֹב לָךְ אֲשֵׁרַת כָּל אִילָן בִּסְטַר מַדְבְּחָא דַיְיָ אֱלָהָךְ דִי תַעְבֶּד לָךְ:

{כב} וְלֹא תָקִים לְךָ מַצֵּבָה אֲשֶׁר שָׂנֵא יְהוָה אֱלֹהֶיךָ:  מ  וְלֹא תָקִים לְךָ מַצֵּבָה אֲשֶׁר שָׂנֵא יְהוָה אֱלֹהֶיךָ:  ת  וְלָא תְקִים לָךְ קָמָא דִי סָנֵי יְיָ אֱלָהָךְ: (ס)




דברים פרק-יז

{א} לֹא תִזְבַּח לַיהוָה אֱלֹהֶיךָ שׁוֹר וָשֶׂה אֲשֶׁר יִהְיֶה בוֹ מוּם כֹּל דָּבָר רָע כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ הוּא:  מ  לֹא תִזְבַּח לַיהוָה אֱלֹהֶיךָ שׁוֹר וָשֶׂה אֲשֶׁר יִהְיֶה בוֹ מוּם כֹּל דָּבָר רָע כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ הוּא:  ת  לָא תִכּוֹס קֳדָם יְיָ אֱלָהָךְ תּוֹר וְאִמָר דִי יְהֵי בֵהּ מוּמָא כֹּל מִדַעַם בִּישׁ אֲרֵי מְרָחָק קֳדָם יְיָ אֱלָהָךְ הוּא: (ס)

{ב} כִּי יִמָּצֵא בְקִרְבְּךָ בְּאַחַד שְׁעָרֶיךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ אִישׁ אוֹ אִשָּׁה אֲשֶׁר יַעֲשֶׂה אֶת הָרַע בְּעֵינֵי יְהוָה אֱלֹהֶיךָ לַעֲבֹר בְּרִיתוֹ:  מ  כִּי יִמָּצֵא בְקִרְבְּךָ בְּאַחַד שְׁעָרֶיךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ אִישׁ אוֹ אִשָּׁה אֲשֶׁר יַעֲשֶׂה אֶת הָרַע בְּעֵינֵי יְהוָה אֱלֹהֶיךָ לַעֲבֹר בְּרִיתוֹ:  ת  אֲרֵי יִשְׁתְּכַח בֵּינָךְ בַּחֲדָא מִקִרְוָיךְ דַיְיָ אֱלָהָךְ יָהֵב לָךְ גְבַר אוֹ אִתְּתָא דִי יַעְבֵּד יָת דְבִישׁ קֳדָם יְיָ אֱלָהָךְ לְמֶעְבַּר עַל קְיָמֵהּ:

{ג} וַיֵּלֶךְ וַיַּעֲבֹד אֱלֹהִים אֲחֵרִים וַיִּשְׁתַּחוּ לָהֶם וְלַשֶּׁמֶשׁ אוֹ לַיָּרֵחַ אוֹ לְכָל צְבָא הַשָּׁמַיִם אֲשֶׁר לֹא צִוִּיתִי:  מ  וַיֵּלֶךְ וַיַּעֲבֹד אֱלֹהִים אֲחֵרִים וַיִּשְׁתַּחוּ לָהֶם וְלַשֶּׁמֶשׁ אוֹ לַיָּרֵחַ אוֹ לְכָל צְבָא הַשָּׁמַיִם אֲשֶׁר לֹא צִוִּיתִי:  ת  וַאֲזַל וּפְלַח לְטַעֲוַת עַמְמַיָא וּסְגִיד לְהוֹן וּלְשִׁמְשָׁא אוֹ לְסִיהֲרָא אוֹ לְכָל חֵילֵי שְׁמַיָא דִי לָא פַקֵדִית:

{ד} וְהֻגַּד לְךָ וְשָׁמָעְתָּ וְדָרַשְׁתָּ הֵיטֵב וְהִנֵּה אֱמֶת נָכוֹן הַדָּבָר נֶעֶשְׂתָה הַתּוֹעֵבָה הַזֹּאת בְּיִשְׂרָאֵל:  מ  וְהֻגַּד לְךָ וְשָׁמָעְתָּ וְדָרַשְׁתָּ הֵיטֵב וְהִנֵּה אֱמֶת נָכוֹן הַדָּבָר נֶעֶשְׂתָה הַתּוֹעֵבָה הַזֹּאת בְּיִשְׂרָאֵל:  ת  וְיִתְחַוֵי לָךְ וְתִשְׁמַע וְתִתְבַּע יָאוּת וְהָא קֻשְׁטָא כֵּוַן פִּתְגָמָא אִתְעֲבִידַת תּוֹעֶבְתָּא הָדָא בְּיִשְׂרָאֵל:

{ה} וְהוֹצֵאתָ אֶת הָאִישׁ הַהוּא אוֹ אֶת הָאִשָּׁה הַהִוא אֲשֶׁר עָשׂוּ אֶת הַדָּבָר הָרָע הַזֶּה אֶל שְׁעָרֶיךָ אֶת הָאִישׁ אוֹ אֶת הָאִשָּׁה וּסְקַלְתָּם בָּאֲבָנִים וָמֵתוּ:  מ  וְהוֹצֵאתָ אֶת הָאִישׁ הַהוּא אוֹ אֶת הָאִשָּׁה הַהִוא אֲשֶׁר עָשׂוּ אֶת הַדָּבָר הָרָע הַזֶּה אֶל שְׁעָרֶיךָ אֶת הָאִישׁ אוֹ אֶת הָאִשָּׁה וּסְקַלְתָּם בָּאֲבָנִים וָמֵתוּ:  ת  וְתַפֵּק יָת גַבְרָא הַהוּא אוֹ יָת אִתְּתָא הַהִיא דִי עֲבָדוּ יָת פִּתְגָמָא בִישָׁא הָדֵין לִתְרַע בֵּית דִינָךְ יָת גַבְרָא אוֹ יָת אִתְּתָא וְתִרְגְמִנוּן בְּאַבְנַיָא וִימוּתוּן:

{ו} עַל פִּי שְׁנַיִם עֵדִים אוֹ שְׁלֹשָׁה עֵדִים יוּמַת הַמֵּת לֹא יוּמַת עַל פִּי עֵד אֶחָד:  מ  עַל פִּי שְׁנַיִם עֵדִים אוֹ שְׁלֹשָׁה עֵדִים יוּמַת הַמֵּת לֹא יוּמַת עַל פִּי עֵד אֶחָד:  ת  עַל מֵימַר תְּרֵין סַהֲדִין אוֹ תְּלָתָא סַהֲדִין יִתְקְטִל דְחַיַב קְטוֹל לָא יִתְקְטֵל עַל מֵימַר סָהִיד חָד:

{ז} יַד הָעֵדִים תִּהְיֶה בּוֹ בָרִאשֹׁנָה לַהֲמִיתוֹ וְיַד כָּל הָעָם בָּאַחֲרֹנָה וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  מ  יַד הָעֵדִים תִּהְיֶה בּוֹ בָרִאשֹׁנָה לַהֲמִיתוֹ וְיַד כָּל הָעָם בָּאַחֲרֹנָה וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  ת  יְדָא דְסַהֲדַיָא תְּהֵי בֵהּ בְּקַדְמֵיתָא לְמִקְטְלֵהּ וִידָא דְכָל עַמָא בְּבַתְרֵיתָא וּתְפַלֵי עָבֵד דְבִישׁ מִבֵּינָךְ: (פ)

{ח} כִּי יִפָּלֵא מִמְּךָ דָבָר לַמִּשְׁפָּט בֵּין דָּם לְדָם בֵּין דִּין לְדִין וּבֵין נֶגַע לָנֶגַע דִּבְרֵי רִיבֹת בִּשְׁעָרֶיךָ וְקַמְתָּ וְעָלִיתָ אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ:  מ  כִּי יִפָּלֵא מִמְּךָ דָבָר לַמִּשְׁפָּט בֵּין דָּם לְדָם בֵּין דִּין לְדִין וּבֵין נֶגַע לָנֶגַע דִּבְרֵי רִיבֹת בִּשְׁעָרֶיךָ וְקַמְתָּ וְעָלִיתָ אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ:  ת  אֲרֵי יִתְכַסֵי מִנָךְ פִּתְגָמָא לְדִינָא בֵּין דְמָא לִדְמָא בֵּין דִינָא לְדִינָא וּבֵין מַכְתַּשׁ סְגִירוּ לְמַכְתַּשׁ סְגִירוּ פִּתְגָמֵי פְלֻגְתָּא דִינָא בְּקִרְוָיךְ וּתְקוּם וְתִסַק לְאַתְרָא דִי יִתִּרְעֵי יְיָ אֱלָהָךְ בֵּהּ:

{ט} וּבָאתָ אֶל הַכֹּהֲנִים הַלְוִיִּם וְאֶל הַשֹּׁפֵט אֲשֶׁר יִהְיֶה בַּיָּמִים הָהֵם וְדָרַשְׁתָּ וְהִגִּידוּ לְךָ אֵת דְּבַר הַמִּשְׁפָּט:  מ  וּבָאתָ אֶל הַכֹּהֲנִים הַלְוִיִּם וְאֶל הַשֹּׁפֵט אֲשֶׁר יִהְיֶה בַּיָּמִים הָהֵם וְדָרַשְׁתָּ וְהִגִּידוּ לְךָ אֵת דְּבַר הַמִּשְׁפָּט:  ת  וְתֵיתֵי לְוָת כָּהֲנַיָא לֵוָאֵי וּלְוָת דַיָנָא דִי יְהֵי בְּיוֹמַיָא הָאִנוּן וְתִתְבַּע וִיחַוּוּן לָךְ יָת פִּתְגָמָא דְדִינָא:

{י} וְעָשִׂיתָ עַל פִּי הַדָּבָר אֲשֶׁר יַגִּידוּ לְךָ מִן הַמָּקוֹם הַהוּא אֲשֶׁר יִבְחַר יְהוָה וְשָׁמַרְתָּ לַעֲשׂוֹת כְּכֹל אֲשֶׁר יוֹרוּךָ:  מ  וְעָשִׂיתָ עַל פִּי הַדָּבָר אֲשֶׁר יַגִּידוּ לְךָ מִן הַמָּקוֹם הַהוּא אֲשֶׁר יִבְחַר יְהוָה וְשָׁמַרְתָּ לַעֲשׂוֹת כְּכֹל אֲשֶׁר יוֹרוּךָ:  ת  וְתַעְבֵּד עַל מֵימַר פִּתְגָמָא דִי יְחַוּוּן לָךְ מִן אַתְרָא הַהוּא דִי יִתִּרְעֵי יְיָ וְתִטַר לְמֶעְבַּד כְּכֹל דִי יַלְפֻנָךְ:

{יא} עַל פִּי הַתּוֹרָה אֲשֶׁר יוֹרוּךָ וְעַל הַמִּשְׁפָּט אֲשֶׁר יֹאמְרוּ לְךָ תַּעֲשֶׂה לֹא תָסוּר מִן הַדָּבָר אֲשֶׁר יַגִּידוּ לְךָ יָמִין וּשְׂמֹאל:  מ  עַל פִּי הַתּוֹרָה אֲשֶׁר יוֹרוּךָ וְעַל הַמִּשְׁפָּט אֲשֶׁר יֹאמְרוּ לְךָ תַּעֲשֶׂה לֹא תָסוּר מִן הַדָּבָר אֲשֶׁר יַגִּידוּ לְךָ יָמִין וּשְׂמֹאל:  ת  עַל מֵימַר אוֹרַיְתָא דִי יַלְפֻנָךְ וְעַל דִינָא דִי יֵימְרוּן לָךְ תַּעְבֵּד לָא תִסְטֵּי מִן פִּתְגָמָא דִי יְחַוּוּן לָךְ יַמִינָא וּשְׂמָאלָא:

{יב} וְהָאִישׁ אֲשֶׁר יַעֲשֶׂה בְזָדוֹן לְבִלְתִּי שְׁמֹעַ אֶל הַכֹּהֵן הָעֹמֵד לְשָׁרֶת שָׁם אֶת יְהוָה אֱלֹהֶיךָ אוֹ אֶל הַשֹּׁפֵט וּמֵת הָאִישׁ הַהוּא וּבִעַרְתָּ הָרָע מִיִּשְׂרָאֵל:  מ  וְהָאִישׁ אֲשֶׁר יַעֲשֶׂה בְזָדוֹן לְבִלְתִּי שְׁמֹעַ אֶל הַכֹּהֵן הָעֹמֵד לְשָׁרֶת שָׁם אֶת יְהוָה אֱלֹהֶיךָ אוֹ אֶל הַשֹּׁפֵט וּמֵת הָאִישׁ הַהוּא וּבִעַרְתָּ הָרָע מִיִּשְׂרָאֵל:  ת  וּגְבַר דִי יַעְבֵּד בִּרְשַׁע בְּדִיל דְלָא לְקַבָּלָא מִן כַּהֲנָא דְקָאֵם לְשַׁמָשָׁא תַמָן קֳדָם יְיָ אֱלָהָךְ אוֹ מִן דַיָנָא וְיִתְקְטֵל גַבְרָא הַהוּא וּתְפַלֵי עָבֵד דְבִישׁ מִיִשְׂרָאֵל:

{יג} וְכָל הָעָם יִשְׁמְעוּ וְיִרָאוּ וְלֹא יְזִידוּן עוֹד:  מ  וְכָל הָעָם יִשְׁמְעוּ וְיִרָאוּ וְלֹא יְזִידוּן עוֹד:  ת  וְכָל עַמָא יִשְׁמְעוּן וְיִדְחֲלוּן וְלָא יַרְשִׁיעוּן עוֹד: (ס)

{יד}  שני  כִּי תָבֹא אֶל הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ וִירִשְׁתָּהּ וְיָשַׁבְתָּה בָּהּ וְאָמַרְתָּ אָשִׂימָה עָלַי מֶלֶךְ כְּכָל הַגּוֹיִם אֲשֶׁר סְבִיבֹתָי:  מ  כִּי תָבֹא אֶל הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ וִירִשְׁתָּהּ וְיָשַׁבְתָּה בָּהּ וְאָמַרְתָּ אָשִׂימָה עָלַי מֶלֶךְ כְּכָל הַגּוֹיִם אֲשֶׁר סְבִיבֹתָי:  ת  אֲרֵי תֵעוֹל לְאַרְעָא דַיְיָ אֱלָהָךְ יָהֵב לָךְ וְתֵירְתַהּ וְתֵיתֵב בַּהּ וְתֵימַר אֱמַנִי עֲלַי מַלְכָּא כְּכָל עַמְמַיָא דִי בְסַחֲרָנָי:

{טו} שׂוֹם תָּשִׂים עָלֶיךָ מֶלֶךְ אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ מִקֶּרֶב אַחֶיךָ תָּשִׂים עָלֶיךָ מֶלֶךְ לֹא תוּכַל לָתֵת עָלֶיךָ אִישׁ נָכְרִי אֲשֶׁר לֹא אָחִיךָ הוּא:  מ  שׂוֹם תָּשִׂים עָלֶיךָ מֶלֶךְ אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ מִקֶּרֶב אַחֶיךָ תָּשִׂים עָלֶיךָ מֶלֶךְ לֹא תוּכַל לָתֵת עָלֶיךָ אִישׁ נָכְרִי אֲשֶׁר לֹא אָחִיךָ הוּא:  ת  מַנָאָה תְמַנֵי עֲלָךְ מַלְכָּא דִי יִתִּרְעֵי יְיָ אֱלָהָךְ בֵּהּ מִגוֹ אַחָיךְ תְּמַנֵי עֲלָךְ מַלְכָּא לֵת לָךְ רְשׁוּ לְמַנָאָה עֲלָךְ גְבַר נוּכְרַי דִי לָא אֲחוּךְ הוּא:

{טז} רַק לֹא יַרְבֶּה לּוֹ סוּסִים וְלֹא יָשִׁיב אֶת הָעָם מִצְרַיְמָה לְמַעַן הַרְבּוֹת סוּס וַיהוָה אָמַר לָכֶם לֹא תֹסִפוּן לָשׁוּב בַּדֶּרֶךְ הַזֶּה עוֹד:  מ  רַק לֹא יַרְבֶּה לּוֹ סוּסִים וְלֹא יָשִׁיב אֶת הָעָם מִצְרַיְמָה לְמַעַן הַרְבּוֹת סוּס וַיהוָה אָמַר לָכֶם לֹא תֹסִפוּן לָשׁוּב בַּדֶּרֶךְ הַזֶּה עוֹד:  ת  לְחוֹד לָא יַסְגֵי לֵהּ סוּסָוָן וְלָא יָתֵב יָת עַמָא לְמִצְרָיִם בְּדִיל לְאַסְגָּאָה לֵהּ סוּסָוָן וַיְיָ אֲמַר לְכוֹן לָא תוֹסְפוּן לְמִתוּב בְּאֹרַח הָדֵין עוֹד:

{יז} וְלֹא יַרְבֶּה לּוֹ נָשִׁים וְלֹא יָסוּר לְבָבוֹ וְכֶסֶף וְזָהָב לֹא יַרְבֶּה לּוֹ מְאֹד:  מ  וְלֹא יַרְבֶּה לּוֹ נָשִׁים וְלֹא יָסוּר לְבָבוֹ וְכֶסֶף וְזָהָב לֹא יַרְבֶּה לּוֹ מְאֹד:  ת  וְלָא יַסְגֵי לֵהּ נְשִׁין וְלָא יִטְעֵי לִבֵּהּ וְכַסְפָּא וְדַהַבָא לָא יַסגֵי לֵהּ לַחֲדָא:

{יח} וְהָיָה כְשִׁבְתּוֹ עַל כִּסֵּא מַמְלַכְתּוֹ וְכָתַב לוֹ אֶת מִשְׁנֵה הַתּוֹרָה הַזֹּאת עַל סֵפֶר מִלִּפְנֵי הַכֹּהֲנִים הַלְוִיִּם:  מ  וְהָיָה כְשִׁבְתּוֹ עַל כִּסֵּא מַמְלַכְתּוֹ וְכָתַב לוֹ אֶת מִשְׁנֵה הַתּוֹרָה הַזֹּאת עַל סֵפֶר מִלִּפְנֵי הַכֹּהֲנִים הַלְוִיִּם:  ת  וִיהֵי כְמִתְּבֵהּ עַל כָּרְסָא מַלְכוּתֵהּ וְיִכְתּוֹב לֵהּ יָת פַּתְשֶׁגֶן אוֹרַיְתָא הָדָא עַל סִפְרָא מִן קֳדָם כָּהֲנַיָא לֵוָאֵי:

{יט} וְהָיְתָה עִמּוֹ וְקָרָא בוֹ כָּל יְמֵי חַיָּיו לְמַעַן יִלְמַד לְיִרְאָה אֶת יְהוָה אֱלֹהָיו לִשְׁמֹר אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת וְאֶת הַחֻקִּים הָאֵלֶּה לַעֲשֹׂתָם:  מ  וְהָיְתָה עִמּוֹ וְקָרָא בוֹ כָּל יְמֵי חַיָּיו לְמַעַן יִלְמַד לְיִרְאָה אֶת יְהוָה אֱלֹהָיו לִשְׁמֹר אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת וְאֶת הַחֻקִּים הָאֵלֶּה לַעֲשֹׂתָם:  ת  וּתְהֵי עִמֵהּ וִיהֵי קָרֵי בֵהּ כָּל יוֹמֵי חַיוֹהִי בְּדִיל דְיֵלַף לְמִדְחַל קֳדָם יְיָ אֱלָהֵהּ לְמִטַּר יָת כָּל פִּתְגָמֵי אוֹרַיְתָא הָדָא וְיָת קְיָמַיָא הָאִלֵין לְמֶעְבָּדְהוֹן:

{כ} לְבִלְתִּי רוּם לְבָבוֹ מֵאֶחָיו וּלְבִלְתִּי סוּר מִן הַמִּצְוָה יָמִין וּשְׂמֹאול לְמַעַן יַאֲרִיךְ יָמִים עַל מַמְלַכְתּוֹ הוּא וּבָנָיו בְּקֶרֶב יִשְׂרָאֵל:  מ  לְבִלְתִּי רוּם לְבָבוֹ מֵאֶחָיו וּלְבִלְתִּי סוּר מִן הַמִּצְוָה יָמִין וּשְׂמֹאול לְמַעַן יַאֲרִיךְ יָמִים עַל מַמְלַכְתּוֹ הוּא וּבָנָיו בְּקֶרֶב יִשְׂרָאֵל:  ת  בְּדִיל דְלָא יְרִים לִבֵּהּ מֵאֲחוֹהִי וּבְדִיל דְלָא יִסְטֵּי מִן תַּפְקֶדְתָּא יַמִינָא וּשְׂמָאלָא בְּדִיל דְיוֹרֵךְ יוֹמִין עַל מַלְכוּתֵהּ הוּא וּבְנוֹהִי בְּגוֹ יִשְׂרָאֵל: (ס)

{א}  שלישי  לֹא יִהְיֶה לַכֹּהֲנִים הַלְוִיִּם כָּל שֵׁבֶט לֵוִי חֵלֶק וְנַחֲלָה עִם יִשְׂרָאֵל אִשֵּׁי יְהוָה וְנַחֲלָתוֹ יֹאכֵלוּן:  מ  לֹא יִהְיֶה לַכֹּהֲנִים הַלְוִיִּם כָּל שֵׁבֶט לֵוִי חֵלֶק וְנַחֲלָה עִם יִשְׂרָאֵל אִשֵּׁי יְהוָה וְנַחֲלָתוֹ יֹאכֵלוּן:  ת  לָא יְהֵי לְכָהֲנַיָא לֵוָאֵי כָּל שִׁבְטָא דְלֵוִי חֳלָק וְאַחֲסָנָא עִם יִשְׂרָאֵל קֻרְבָּנַיָא דַיְיָ וְאַחֲסַנְתֵּהּ יֵיכְלוּן:

{ב} וְנַחֲלָה לֹא יִהְיֶה לּוֹ בְּקֶרֶב אֶחָיו יְהוָה הוּא נַחֲלָתוֹ כַּאֲשֶׁר דִּבֶּר לוֹ:  מ  וְנַחֲלָה לֹא יִהְיֶה לּוֹ בְּקֶרֶב אֶחָיו יְהוָה הוּא נַחֲלָתוֹ כַּאֲשֶׁר דִּבֶּר לוֹ:  ת  וְאַחֲסָנָא לָא יְהֵי לֵהּ בְּגוֹ אֲחוֹהִי מַתְּנָן דִיהַב לֵהּ יְיָ אִנוּן אַחֲסַנְתֵּהּ כְּמָא דִי מַלִיל לֵהּ: (ס)

{ג} וְזֶה יִהְיֶה מִשְׁפַּט הַכֹּהֲנִים מֵאֵת הָעָם מֵאֵת זֹבְחֵי הַזֶּבַח אִם שׁוֹר אִם שֶׂה וְנָתַן לַכֹּהֵן הַזְּרֹעַ וְהַלְּחָיַיִם וְהַקֵּבָה:  מ  וְזֶה יִהְיֶה מִשְׁפַּט הַכֹּהֲנִים מֵאֵת הָעָם מֵאֵת זֹבְחֵי הַזֶּבַח אִם שׁוֹר אִם שֶׂה וְנָתַן לַכֹּהֵן הַזְּרֹעַ וְהַלְּחָיַיִם וְהַקֵּבָה:  ת  וְדֵין יְהֵי דְחָזֵי לְכָהֲנַיָא מִן עַמָא מִן נָכְסֵי נִכְסָתָא אִם תּוֹר אִם אִמָר וְיִתֵּן לְכַהֲנָא דְרוֹעָא וְלוֹעָא וְקֻבְתָא:

{ד} רֵאשִׁית דְּגָנְךָ תִּירֹשְׁךָ וְיִצְהָרֶךָ וְרֵאשִׁית גֵּז צֹאנְךָ תִּתֶּן לּוֹ:  מ  רֵאשִׁית דְּגָנְךָ תִּירֹשְׁךָ וְיִצְהָרֶךָ וְרֵאשִׁית גֵּז צֹאנְךָ תִּתֶּן לּוֹ:  ת  רֵישׁ עִבוּרָךְ חַמְרָךְ וּמִשְׁחָךְ וְרֵישׁ גִזָא דְעָנָךְ תַּפְרֵשׁ לֵהּ:

{ה} כִּי בוֹ בָּחַר יְהוָה אֱלֹהֶיךָ מִכָּל שְׁבָטֶיךָ לַעֲמֹד לְשָׁרֵת בְּשֵׁם יְהוָה הוּא וּבָנָיו כָּל הַיָּמִים:  מ  כִּי בוֹ בָּחַר יְהוָה אֱלֹהֶיךָ מִכָּל שְׁבָטֶיךָ לַעֲמֹד לְשָׁרֵת בְּשֵׁם יְהוָה הוּא וּבָנָיו כָּל הַיָּמִים:  ת  אֲרֵי בֵהּ יִתִּרְעֵי יְיָ אֱלָהָךְ מִכָּל שִׁבְטָיךְ לְמֵקַם לְשַׁמָשָׁא בִשְׁמָא דַיְיָ הוּא וּבְנוֹהִי כָּל יוֹמַיָא: (ס)

{ו}  רביעי  וְכִי יָבֹא הַלֵּוִי מֵאַחַד שְׁעָרֶיךָ מִכָּל יִשְׂרָאֵל אֲשֶׁר הוּא גָּר שָׁם וּבָא בְּכָל אַוַּת נַפְשׁוֹ אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה:  מ  וְכִי יָבֹא הַלֵּוִי מֵאַחַד שְׁעָרֶיךָ מִכָּל יִשְׂרָאֵל אֲשֶׁר הוּא גָּר שָׁם וּבָא בְּכָל אַוַּת נַפְשׁוֹ אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה:  ת  וַאֲרֵי יֵיתֵי לֵוָאָה מֵחֲדָא מִקִרְוָיךְ מִכָּל יִשְׂרָאֵל דִי הוּא דָר תַּמָן וְיֵיתֵי בְּכָל רְעוּת נַפְשֵׁהּ לְאַתְרָא דִי יִתִּרְעֵי יְיָ:

{ז} וְשֵׁרֵת בְּשֵׁם יְהוָה אֱלֹהָיו כְּכָל אֶחָיו הַלְוִיִּם הָעֹמְדִים שָׁם לִפְנֵי יְהוָה:  מ  וְשֵׁרֵת בְּשֵׁם יְהוָה אֱלֹהָיו כְּכָל אֶחָיו הַלְוִיִּם הָעֹמְדִים שָׁם לִפְנֵי יְהוָה:  ת  וִישַׁמַשׁ בִּשְׁמָא דַיְיָ אֱלָהֵהּ כְּכָל אֲחוֹהִי לֵוָאֵי דִמְשַׁמְשִׁין תַּמָן (בִּצְלוֹ) קֳדָם יְיָ:

{ח} חֵלֶק כְּחֵלֶק יֹאכֵלוּ לְבַד מִמְכָּרָיו עַל הָאָבוֹת:  מ  חֵלֶק כְּחֵלֶק יֹאכֵלוּ לְבַד מִמְכָּרָיו עַל הָאָבוֹת:  ת  חֳלָק כָּחֳלָק יֵיכְלוּן בַּר מִמַטַרְתָּא דְיֵיתֵי בְּשַׁבַּתָּא דְכֵן אַתְקִינוּ אֲבָהָתָא: (ס)

{ט} כִּי אַתָּה בָּא אֶל הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ לֹא תִלְמַד לַעֲשׂוֹת כְּתוֹעֲבֹת הַגּוֹיִם הָהֵם:  מ  כִּי אַתָּה בָּא אֶל הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ לֹא תִלְמַד לַעֲשׂוֹת כְּתוֹעֲבֹת הַגּוֹיִם הָהֵם:  ת  אֲרֵי אַתְּ עָלֵל לְאַרְעָא דַיְיָ אֱלָהָךְ יָהֵב לָךְ לָא תֵילַף לְמֶעְבַּד כְּתֹעֲבַת עַמְמַיָא הָאִנוּן:

{י} לֹא יִמָּצֵא בְךָ מַעֲבִיר בְּנוֹ וּבִתּוֹ בָּאֵשׁ קֹסֵם קְסָמִים מְעוֹנֵן וּמְנַחֵשׁ וּמְכַשֵּׁף:  מ  לֹא יִמָּצֵא בְךָ מַעֲבִיר בְּנוֹ וּבִתּוֹ בָּאֵשׁ קֹסֵם קְסָמִים מְעוֹנֵן וּמְנַחֵשׁ וּמְכַשֵּׁף:  ת  לָא יִשְׁתְּכַח בָּךְ מַעֲבַר בְּרֵהּ וּבְרַתֵּהּ בְּנוּרָא קָסֵם קֻסְמִין מְעָנֵן וּמְנַחֵשׁ וְחָרָשׁ:

{יא} וְחֹבֵר חָבֶר וְשֹׁאֵל אוֹב וְיִדְּעֹנִי וְדֹרֵשׁ אֶל הַמֵּתִים:  מ  וְחֹבֵר חָבֶר וְשֹׁאֵל אוֹב וְיִדְּעֹנִי וְדֹרֵשׁ אֶל הַמֵּתִים:  ת  וְרָטִין רְטָן וְשָׁאֵל בִּדִין וּדְכוּרוּ וְתָבַע מִן מֵתַיָא:

{יב} כִּי תוֹעֲבַת יְהוָה כָּל עֹשֵׂה אֵלֶּה וּבִגְלַל הַתּוֹעֵבֹת הָאֵלֶּה יְהוָה אֱלֹהֶיךָ מוֹרִישׁ אוֹתָם מִפָּנֶיךָ:  מ  כִּי תוֹעֲבַת יְהוָה כָּל עֹשֵׂה אֵלֶּה וּבִגְלַל הַתּוֹעֵבֹת הָאֵלֶּה יְהוָה אֱלֹהֶיךָ מוֹרִישׁ אוֹתָם מִפָּנֶיךָ:  ת  אֲרֵי מְרָחָק יְיָ כָּל דְיַעְבֵּד אִלֵין וּבְדִיל תּוֹעֲבָתָא הָאִלֵין יְיָ אֱלָהָךְ מְתָרַךְ יָתְהוֹן מִקֳדָמָךְ:

{יג} תָּמִים תִּהְיֶה עִם יְהוָה אֱלֹהֶיךָ:  מ  תָּמִים תִּהְיֶה עִם יְהוָה אֱלֹהֶיךָ:  ת  שְׁלִים תְּהֵי בְּדַחַלְתָּא דַיְיָ אֱלָהָךְ:

{יד}  חמישי  כִּי הַגּוֹיִם הָאֵלֶּה אֲשֶׁר אַתָּה יוֹרֵשׁ אוֹתָם אֶל מְעֹנְנִים וְאֶל קֹסְמִים יִשְׁמָעוּ וְאַתָּה לֹא כֵן נָתַן לְךָ יְהוָה אֱלֹהֶיךָ:  מ  כִּי הַגּוֹיִם הָאֵלֶּה אֲשֶׁר אַתָּה יוֹרֵשׁ אוֹתָם אֶל מְעֹנְנִים וְאֶל קֹסְמִים יִשְׁמָעוּ וְאַתָּה לֹא כֵן נָתַן לְךָ יְהוָה אֱלֹהֶיךָ:  ת  אֲרֵי עַמְמַיָא הָאִלֵין דִי אַתְּ יָרֵת יָתְהוֹן מִן מְעָנְנַיָא וּמִן קַסָמַיָא שָׁמְעִין וְאַתְּ לָא כֵן יְהַב לָךְ יְיָ אֱלָהָךְ:

{טו} נָבִיא מִקִּרְבְּךָ מֵאַחֶיךָ כָּמֹנִי יָקִים לְךָ יְהוָה אֱלֹהֶיךָ אֵלָיו תִּשְׁמָעוּן:  מ  נָבִיא מִקִּרְבְּךָ מֵאַחֶיךָ כָּמֹנִי יָקִים לְךָ יְהוָה אֱלֹהֶיךָ אֵלָיו תִּשְׁמָעוּן:  ת  נְבִיָא מִבֵּינָךְ מֵאַחָיךְ כְּוָתִי יְקִים לָךְ יְיָ אֱלָהָךְ מִנֵהּ תְּקַבְּלוּן:

{טז} כְּכֹל אֲשֶׁר שָׁאַלְתָּ מֵעִם יְהוָה אֱלֹהֶיךָ בְּחֹרֵב בְּיוֹם הַקָּהָל לֵאמֹר לֹא אֹסֵף לִשְׁמֹעַ אֶת קוֹל יְהוָה אֱלֹהָי וְאֶת הָאֵשׁ הַגְּדֹלָה הַזֹּאת לֹא אֶרְאֶה עוֹד וְלֹא אָמוּת:  מ  כְּכֹל אֲשֶׁר שָׁאַלְתָּ מֵעִם יְהוָה אֱלֹהֶיךָ בְּחֹרֵב בְּיוֹם הַקָּהָל לֵאמֹר לֹא אֹסֵף לִשְׁמֹעַ אֶת קוֹל יְהוָה אֱלֹהָי וְאֶת הָאֵשׁ הַגְּדֹלָה הַזֹּאת לֹא אֶרְאֶה עוֹד וְלֹא אָמוּת:  ת  כְּכֹל דִי שְׁאֶלְתָּא מִקֳדָם יְיָ אֱלָהָךְ בְּחֹרֵב בְּיוֹמָא דִקְהָלָא לְמֵימָר לָא אוֹסֵף לְמִשְׁמָע יָת קָל מֵימְרָא דַיְיָ אֱלָהַי וְיָת אֶשָׁתָא רַבְּתָא הָדָא לָא אֶחֱזֵי עוֹד וְלָא אֵמוּת:

{יז} וַיֹּאמֶר יְהוָה אֵלָי הֵיטִיבוּ אֲשֶׁר דִּבֵּרוּ:  מ  וַיֹּאמֶר יְהוָה אֵלָי הֵיטִיבוּ אֲשֶׁר דִּבֵּרוּ:  ת  וַאֲמַר יְיָ לִי אוֹטִיבוּ דִי מַלִילוּ:

{יח} נָבִיא אָקִים לָהֶם מִקֶּרֶב אֲחֵיהֶם כָּמוֹךָ וְנָתַתִּי דְבָרַי בְּפִיו וְדִבֶּר אֲלֵיהֶם אֵת כָּל אֲשֶׁר אֲצַוֶּנּוּ:  מ  נָבִיא אָקִים לָהֶם מִקֶּרֶב אֲחֵיהֶם כָּמוֹךָ וְנָתַתִּי דְבָרַי בְּפִיו וְדִבֶּר אֲלֵיהֶם אֵת כָּל אֲשֶׁר אֲצַוֶּנּוּ:  ת  נְבִיָא אֲקִים לְהוֹן מִגוֹ אֲחֵיהוֹן כְּוָתָךְ וְאֶתֵּן פִּתְגָמֵי נְבוּאָתִי בְּפוּמֵהּ וִימַלֵל עִמְהוֹן יָת כָּל דִי אֲפַקְדִנֵהּ:

{יט} וְהָיָה הָאִישׁ אֲשֶׁר לֹא יִשְׁמַע אֶל דְּבָרַי אֲשֶׁר יְדַבֵּר בִּשְׁמִי אָנֹכִי אֶדְרֹשׁ מֵעִמּוֹ:  מ  וְהָיָה הָאִישׁ אֲשֶׁר לֹא יִשְׁמַע אֶל דְּבָרַי אֲשֶׁר יְדַבֵּר בִּשְׁמִי אָנֹכִי אֶדְרֹשׁ מֵעִמּוֹ:  ת  וִיהֵי גַבְרָא דִי לָא יְקַבֵּל לְפִתְגָמַי דִי יְמַלֵל בִּשְׁמִי מֵימְרִי יִתְבַּע מִנֵהּ:

{כ} אַךְ הַנָּבִיא אֲשֶׁר יָזִיד לְדַבֵּר דָּבָר בִּשְׁמִי אֵת אֲשֶׁר לֹא צִוִּיתִיו לְדַבֵּר וַאֲשֶׁר יְדַבֵּר בְּשֵׁם אֱלֹהִים אֲחֵרִים וּמֵת הַנָּבִיא הַהוּא:  מ  אַךְ הַנָּבִיא אֲשֶׁר יָזִיד לְדַבֵּר דָּבָר בִּשְׁמִי אֵת אֲשֶׁר לֹא צִוִּיתִיו לְדַבֵּר וַאֲשֶׁר יְדַבֵּר בְּשֵׁם אֱלֹהִים אֲחֵרִים וּמֵת הַנָּבִיא הַהוּא:  ת  בְּרַם נְבִיָא דִי יַרְשַׁע לְמַלָלָא פִתְגָמָא בִּשְׁמִי יָת דִי לָא פַקֵדִתֵּה לְמַלָלָא וְדִי יְמַלֵיל בְּשׁוּם טַעֲוַת עַמְמַיָא וְיִתְקְטֵל נְבִיָא הַהוּא:

{כא} וְכִי תֹאמַר בִּלְבָבֶךָ אֵיכָה נֵדַע אֶת הַדָּבָר אֲשֶׁר לֹא דִבְּרוֹ יְהוָה:  מ  וְכִי תֹאמַר בִּלְבָבֶךָ אֵיכָה נֵדַע אֶת הַדָּבָר אֲשֶׁר לֹא דִבְּרוֹ יְהוָה:  ת  וַאֲרֵי תֵימַר בְּלִבָּךְ אֶכְדֵן נִדַע יָת פִּתְגָמָא דִי לָא מַלְלֵהּ יְיָ:

{כב} אֲשֶׁר יְדַבֵּר הַנָּבִיא בְּשֵׁם יְהוָה וְלֹא יִהְיֶה הַדָּבָר וְלֹא יָבוֹא הוּא הַדָּבָר אֲשֶׁר לֹא דִבְּרוֹ יְהוָה בְּזָדוֹן דִּבְּרוֹ הַנָּבִיא לֹא תָגוּר מִמֶּנּוּ:  מ  אֲשֶׁר יְדַבֵּר הַנָּבִיא בְּשֵׁם יְהוָה וְלֹא יִהְיֶה הַדָּבָר וְלֹא יָבוֹא הוּא הַדָּבָר אֲשֶׁר לֹא דִבְּרוֹ יְהוָה בְּזָדוֹן דִּבְּרוֹ הַנָּבִיא לֹא תָגוּר מִמֶּנּוּ:  ת  דִי יְמַלֵל נְבִיָא בִּשְׁמָא דַיְיָ וְלָא יְהֵי פִתְגָמָא וְלָא יִתְקַיַם הוּא פִתְגָמָא דִי לָא מַלְלֵהּ יְיָ בִּרְשַׁע מַלְלֵהּ נְבִיָא לָא תִדְחַלּ מִנֵהּ: (ס)




דברים פרק-יט

{א} כִּי יַכְרִית יְהוָה אֱלֹהֶיךָ אֶת הַגּוֹיִם אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ אֶת אַרְצָם וִירִשְׁתָּם וְיָשַׁבְתָּ בְעָרֵיהֶם וּבְבָתֵּיהֶם:  מ  כִּי יַכְרִית יְהוָה אֱלֹהֶיךָ אֶת הַגּוֹיִם אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ אֶת אַרְצָם וִירִשְׁתָּם וְיָשַׁבְתָּ בְעָרֵיהֶם וּבְבָתֵּיהֶם:  ת  אֲרֵי יְשֵׁצֵי יְיָ אֱלָהָךְ יָת עַמְמַיָא דִי יְיָ אֱלָהָךְ יָהֵב לָךְ יָת אַרְעֲהוֹן וְתֵרְתִנוּן וְתֵיתֵב בְּקִרְוֵיהוֹן וּבְבָתֵּיהוֹן:

{ב} שָׁלוֹשׁ עָרִים תַּבְדִּיל לָךְ בְּתוֹךְ אַרְצְךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לְרִשְׁתָּהּ:  מ  שָׁלוֹשׁ עָרִים תַּבְדִּיל לָךְ בְּתוֹךְ אַרְצְךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לְרִשְׁתָּהּ:  ת  תְּלַת קִרְוִין תַּפְרֵשׁ לָךְ בְּגוֹ אַרְעָךְ דִי יְיָ אֱלָהָךְ יָהֵב לָךְ לְמֵירְתַהּ:

{ג} תָּכִין לְךָ הַדֶּרֶךְ וְשִׁלַּשְׁתָּ אֶת גְּבוּל אַרְצְךָ אֲשֶׁר יַנְחִילְךָ יְהוָה אֱלֹהֶיךָ וְהָיָה לָנוּס שָׁמָּה כָּל רֹצֵחַ:  מ  תָּכִין לְךָ הַדֶּרֶךְ וְשִׁלַּשְׁתָּ אֶת גְּבוּל אַרְצְךָ אֲשֶׁר יַנְחִילְךָ יְהוָה אֱלֹהֶיךָ וְהָיָה לָנוּס שָׁמָּה כָּל רֹצֵחַ:  ת  תְּתַקֵן לָךְ אָרְחָא וּתְתַלַת יָת תְּחוּם אַרְעָךְ דִי יַחְסְנִנָךְ יְיָ אֱלָהָךְ וִיהֵי לְמֵעִירוֹק תַּמָן כָּל קָטוֹלָא:

{ד} וְזֶה דְּבַר הָרֹצֵחַ אֲשֶׁר יָנוּס שָׁמָּה וָחָי אֲשֶׁר יַכֶּה אֶת רֵעֵהוּ בִּבְלִי דַעַת וְהוּא לֹא שֹׂנֵא לוֹ מִתְּמֹל שִׁלְשֹׁם:  מ  וְזֶה דְּבַר הָרֹצֵחַ אֲשֶׁר יָנוּס שָׁמָּה וָחָי אֲשֶׁר יַכֶּה אֶת רֵעֵהוּ בִּבְלִי דַעַת וְהוּא לֹא שֹׂנֵא לוֹ מִתְּמֹל שִׁלְשֹׁם:  ת  וְדֵין פִּתְגַם קָטוֹלָא דְיֵעִרוֹק תַּמָן וְיִתְקַיַם דִי יִקְטוֹל יָת חַבְרֵהּ בְּלָא מַנְדְעֵהּ וְהוּא לָא סָנֵי לֵהּ מֵאִתְמַלֵי וּמִדְקָמוֹהִי:

{ה} וַאֲשֶׁר יָבֹא אֶת רֵעֵהוּ בַיַּעַר לַחְטֹב עֵצִים וְנִדְּחָה יָדוֹ בַגַּרְזֶן לִכְרֹת הָעֵץ וְנָשַׁל הַבַּרְזֶל מִן הָעֵץ וּמָצָא אֶת רֵעֵהוּ וָמֵת הוּא יָנוּס אֶל אַחַת הֶעָרִים הָאֵלֶּה וָחָי:  מ  וַאֲשֶׁר יָבֹא אֶת רֵעֵהוּ בַיַּעַר לַחְטֹב עֵצִים וְנִדְּחָה יָדוֹ בַגַּרְזֶן לִכְרֹת הָעֵץ וְנָשַׁל הַבַּרְזֶל מִן הָעֵץ וּמָצָא אֶת רֵעֵהוּ וָמֵת הוּא יָנוּס אֶל אַחַת הֶעָרִים הָאֵלֶּה וָחָי:  ת  וְדִי יֵעוֹל עִם חַבְרֵהּ בְּחֻרְשָׁא לְמִקַץ אָעִין וְתִתְמְרֵיג יְדֵהּ בְּפַרְזְלָא לְמִקַץ אָעָא וְיִשְׁתְּלֵף פַּרְזְלָא מִן אָעָא וְיַשְׁכַּח יָת חַבְרֵהּ וִימוּת הוּא יֵעִרוֹק לַחֲדָא מִן קִרְוַיָא הָאִלֵין וְיִתְקַיַם:

{ו} פֶּן יִרְדֹּף גֹּאֵל הַדָּם אַחֲרֵי הָרֹצֵחַ כִּי יֵחַם לְבָבוֹ וְהִשִּׂיגוֹ כִּי יִרְבֶּה הַדֶּרֶךְ וְהִכָּהוּ נָפֶשׁ וְלוֹ אֵין מִשְׁפַּט מָוֶת כִּי לֹא שֹׂנֵא הוּא לוֹ מִתְּמוֹל שִׁלְשׁוֹם:  מ  פֶּן יִרְדֹּף גֹּאֵל הַדָּם אַחֲרֵי הָרֹצֵחַ כִּי יֵחַם לְבָבוֹ וְהִשִּׂיגוֹ כִּי יִרְבֶּה הַדֶּרֶךְ וְהִכָּהוּ נָפֶשׁ וְלוֹ אֵין מִשְׁפַּט מָוֶת כִּי לֹא שֹׂנֵא הוּא לוֹ מִתְּמוֹל שִׁלְשׁוֹם:  ת  דִילְמָא יִרְדוֹף גָאֵל דְמָא בָּתַר קָטוֹלָא אֲרֵי יֵחַם לִבֵּה וְיַדְבְּקִנֵהּ אֲרֵי תִסְגֵי אָרְחָא וְיִקְטְלִנֵהּ נְפַשׁ וְלֵהּ לֵית חוֹבַת דִין דִקְטוֹל אֲרֵי לָא סָנֵי הוּא לֵהּ מִתְּמַלֵי וּמִדְקָמוֹהִי:

{ז} עַל כֵּן אָנֹכִי מְצַוְּךָ לֵאמֹר שָׁלֹשׁ עָרִים תַּבְדִּיל לָךְ:  מ  עַל כֵּן אָנֹכִי מְצַוְּךָ לֵאמֹר שָׁלֹשׁ עָרִים תַּבְדִּיל לָךְ:  ת  עַל כֵּן אֲנָא מְפַקְדָךְ לְמֵימָר תְּלַת קִרְוִין תַּפְרֵשׁ לָךְ:

{ח} וְאִם יַרְחִיב יְהוָה אֱלֹהֶיךָ אֶת גְּבֻלְךָ כַּאֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ וְנָתַן לְךָ אֶת כָּל הָאָרֶץ אֲשֶׁר דִּבֶּר לָתֵת לַאֲבֹתֶיךָ:  מ  וְאִם יַרְחִיב יְהוָה אֱלֹהֶיךָ אֶת גְּבֻלְךָ כַּאֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ וְנָתַן לְךָ אֶת כָּל הָאָרֶץ אֲשֶׁר דִּבֶּר לָתֵת לַאֲבֹתֶיךָ:  ת  וְאִם יַפְתֵּי יְיָ אֱלָהָךְ יָת תְּחוּמָךְ כְּמָא דִי קַיִים לַאֲבָהָתָךְ וְיִתֶּן לָךְ יָת כָּל אַרְעָא דִי מַלִיל לְמִתַּן לַאֲבָהָתָךְ:

{ט} כִּי תִשְׁמֹר אֶת כָּל הַמִּצְוָה הַזֹּאת לַעֲשֹׂתָהּ אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לְאַהֲבָה אֶת יְהוָה אֱלֹהֶיךָ וְלָלֶכֶת בִּדְרָכָיו כָּל הַיָּמִים וְיָסַפְתָּ לְךָ עוֹד שָׁלֹשׁ עָרִים עַל הַשָּׁלֹשׁ הָאֵלֶּה:  מ  כִּי תִשְׁמֹר אֶת כָּל הַמִּצְוָה הַזֹּאת לַעֲשֹׂתָהּ אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לְאַהֲבָה אֶת יְהוָה אֱלֹהֶיךָ וְלָלֶכֶת בִּדְרָכָיו כָּל הַיָּמִים וְיָסַפְתָּ לְךָ עוֹד שָׁלֹשׁ עָרִים עַל הַשָּׁלֹשׁ הָאֵלֶּה:  ת  אֲרֵי תִטַר יָת כָּל תַּפְקֶדְתָּא הָדָא לְמֶעְבְּדַהּ דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין לְמִרְחַם יָת יְיָ אֱלָהָךְ וְלִמְהַךְ בְּאָרְחָן דְתַקְנָן קֳדָמוֹהִי כָּל יוֹמַיָא וְתוֹסֵף לָךְ עוֹד תְּלַת קִרְוִין עַל תְּלַת אִלֵין:

{י} וְלֹא יִשָּׁפֵךְ דָּם נָקִי בְּקֶרֶב אַרְצְךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה וְהָיָה עָלֶיךָ דָּמִים:  מ  וְלֹא יִשָּׁפֵךְ דָּם נָקִי בְּקֶרֶב אַרְצְךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה וְהָיָה עָלֶיךָ דָּמִים:  ת  וְלָא יִשְׁתְּפֵךְ דָם זַכַּי בְּגוֹ אַרְעָךְ דִי יְיָ אֱלָהָךְ יָהֵב לָךְ אַחֲסָנָא וִיהֵי עֲלָךְ חוֹבַת דִין דִקְטוֹל: (פ)

{יא} וְכִי יִהְיֶה אִישׁ שֹׂנֵא לְרֵעֵהוּ וְאָרַב לוֹ וְקָם עָלָיו וְהִכָּהוּ נֶפֶשׁ וָמֵת וְנָס אֶל אַחַת הֶעָרִים הָאֵל:  מ  וְכִי יִהְיֶה אִישׁ שֹׂנֵא לְרֵעֵהוּ וְאָרַב לוֹ וְקָם עָלָיו וְהִכָּהוּ נֶפֶשׁ וָמֵת וְנָס אֶל אַחַת הֶעָרִים הָאֵל:  ת  וַאֲרֵי יְהֵי גְבַר סָנֵי לְחַבְרֵהּ וְיִכְמַן לֵהּ וִיקוּם עֲלוֹהִי וְיִמְחִנֵהּ נְפָשׁ וִימוּת וְיֵעִרוֹק לַחֲדָא מִן קִרְוַיָא הָאִלֵין:

{יב} וְשָׁלְחוּ זִקְנֵי עִירוֹ וְלָקְחוּ אֹתוֹ מִשָּׁם וְנָתְנוּ אֹתוֹ בְּיַד גֹּאֵל הַדָּם וָמֵת:  מ  וְשָׁלְחוּ זִקְנֵי עִירוֹ וְלָקְחוּ אֹתוֹ מִשָּׁם וְנָתְנוּ אֹתוֹ בְּיַד גֹּאֵל הַדָּם וָמֵת:  ת  וִישַׁלְחוּן סָבֵי קַרְתֵּהּ וְיִדְבְּרוּן יָתֵהּ מִתַּמָן וְיִתְּנוּן יָתֵהּ בְּיַד גָאֵל דְמָא וִימוּת:

{יג} לֹא תָחוֹס עֵינְךָ עָלָיו וּבִעַרְתָּ דַם הַנָּקִי מִיִּשְׂרָאֵל וְטוֹב לָךְ:  מ  לֹא תָחוֹס עֵינְךָ עָלָיו וּבִעַרְתָּ דַם הַנָּקִי מִיִּשְׂרָאֵל וְטוֹב לָךְ:  ת  לָא תְחוּס עֵינָךְ עֲלוֹהִי וּתְפַלֵי אַשְׁדִי דָם זַכַּי מִיִשְׂרָאֵל וְיֵיטַב לָךְ: (ס)

{יד}  שישי  לֹא תַסִּיג גְּבוּל רֵעֲךָ אֲשֶׁר גָּבְלוּ רִאשֹׁנִים בְּנַחֲלָתְךָ אֲשֶׁר תִּנְחַל בָּאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לְרִשְׁתָּהּ:  מ  לֹא תַסִּיג גְּבוּל רֵעֲךָ אֲשֶׁר גָּבְלוּ רִאשֹׁנִים בְּנַחֲלָתְךָ אֲשֶׁר תִּנְחַל בָּאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לְרִשְׁתָּהּ:  ת  לָא תַשְׁנֵי תְּחוּמָא דְחַבְרָךְ דִי תְחִימוּ קַדְמָאֵי בְּאַחֲסַנְתָּךְ דִי תַחֲסֵן בְּאַרְעָא דִי יְיָ אֱלָהָךְ יָהֵב לָךְ לְמֵירְתַהּ: (ס)

{טו} לֹא יָקוּם עֵד אֶחָד בְּאִישׁ לְכָל עָוֹן וּלְכָל חַטָּאת בְּכָל חֵטְא אֲשֶׁר יֶחֱטָא עַל פִּי שְׁנֵי עֵדִים אוֹ עַל פִּי שְׁלֹשָׁה עֵדִים יָקוּם דָּבָר:  מ  לֹא יָקוּם עֵד אֶחָד בְּאִישׁ לְכָל עָוֹן וּלְכָל חַטָּאת בְּכָל חֵטְא אֲשֶׁר יֶחֱטָא עַל פִּי שְׁנֵי עֵדִים אוֹ עַל פִּי שְׁלֹשָׁה עֵדִים יָקוּם דָּבָר:  ת  לָא יְקוּם סָהִיד חַד בִּגְבַר לְכָל עַוָיָן וּלְכָל חוֹבִין בְּכָל חֵט דִי יֶחְטֵי עַל מֵימַר תְּרֵין סָהֲדִין אוֹ עַל מֵימַר תְּלָתָא סָהֲדִין יִתְקַיַם פִּתְגָמָא:

{טז} כִּי יָקוּם עֵד חָמָס בְּאִישׁ לַעֲנוֹת בּוֹ סָרָה:  מ  כִּי יָקוּם עֵד חָמָס בְּאִישׁ לַעֲנוֹת בּוֹ סָרָה:  ת  אֲרֵי יְקוּם סָהִיד שְׁקַר בִּגְבָר לְאַסְהָדָא בֵהּ סַטְּיָא:

{יז} וְעָמְדוּ שְׁנֵי הָאֲנָשִׁים אֲשֶׁר לָהֶם הָרִיב לִפְנֵי יְהוָה לִפְנֵי הַכֹּהֲנִים וְהַשֹּׁפְטִים אֲשֶׁר יִהְיוּ בַּיָּמִים הָהֵם:  מ  וְעָמְדוּ שְׁנֵי הָאֲנָשִׁים אֲשֶׁר לָהֶם הָרִיב לִפְנֵי יְהוָה לִפְנֵי הַכֹּהֲנִים וְהַשֹּׁפְטִים אֲשֶׁר יִהְיוּ בַּיָּמִים הָהֵם:  ת  וִיקוּמוּן תְרֵין גֻבְרִין דִי לְהוֹן דִינָא קֳדָם יְיָ קֳדָם כָּהֲנַיָא וְדַיָנַיָא דִי יְהוֹן בְּיוֹמַיָא הָאִנוּן:

{יח} וְדָרְשׁוּ הַשֹּׁפְטִים הֵיטֵב וְהִנֵּה עֵד שֶׁקֶר הָעֵד שֶׁקֶר עָנָה בְאָחִיו:  מ  וְדָרְשׁוּ הַשֹּׁפְטִים הֵיטֵב וְהִנֵּה עֵד שֶׁקֶר הָעֵד שֶׁקֶר עָנָה בְאָחִיו:  ת  וְיִתְבְּעוּן דַיָנַיָא יָאוּת וְהָא סָהִיד שְׁקַר סָהֲדָא שִׁקְרָא אַסְהַד בַּאֲחוֹהִי:

{יט} וַעֲשִׂיתֶם לוֹ כַּאֲשֶׁר זָמַם לַעֲשׂוֹת לְאָחִיו וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  מ  וַעֲשִׂיתֶם לוֹ כַּאֲשֶׁר זָמַם לַעֲשׂוֹת לְאָחִיו וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  ת  וְתַעְבְּדוּן לֵהּ כְּמָא דִי חַשִׁיב לְמֶעְבַּד לַאֲחוֹהִי וּתְפַלֵי עָבֵד דְבִישׁ מִבֵּינָךְ:

{כ} וְהַנִּשְׁאָרִים יִשְׁמְעוּ וְיִרָאוּ וְלֹא יֹסִפוּ לַעֲשׂוֹת עוֹד כַּדָּבָר הָרָע הַזֶּה בְּקִרְבֶּךָ:  מ  וְהַנִּשְׁאָרִים יִשְׁמְעוּ וְיִרָאוּ וְלֹא יֹסִפוּ לַעֲשׂוֹת עוֹד כַּדָּבָר הָרָע הַזֶּה בְּקִרְבֶּךָ:  ת  וּדְיִשְׁתַּאֲרוּן יִשְׁמְעוּן וְיִדְחֲלוּן וְלָא יוֹסְפוּן לְמֶעְבַּד עוֹד כְּפִתְגָמָא בִישָׁא הָדֵין בֵּינָךְ:

{כא} וְלֹא תָחוֹס עֵינֶךָ נֶפֶשׁ בְּנֶפֶשׁ עַיִן בְּעַיִן שֵׁן בְּשֵׁן יָד בְּיָד רֶגֶל בְּרָגֶל:  מ  וְלֹא תָחוֹס עֵינֶךָ נֶפֶשׁ בְּנֶפֶשׁ עַיִן בְּעַיִן שֵׁן בְּשֵׁן יָד בְּיָד רֶגֶל בְּרָגֶל:  ת  וְלָא תְחוּס עֵינָךְ נַפְשָׁא חֳלַף נַפְשָׁא עֵינָא חֳלַף עֵינָא שִׁנָא חֳלַף שִׁנָא יְדָא חֳלַף יְדָא רַגְלָא חֳלַף רַגְלָא: (ס)




דברים פרק-כ

{א} כִּי תֵצֵא לַמִּלְחָמָה עַל אֹיְבֶיךָ וְרָאִיתָ סוּס וָרֶכֶב עַם רַב מִמְּךָ לֹא תִירָא מֵהֶם כִּי יְהוָה אֱלֹהֶיךָ עִמָּךְ הַמַּעַלְךָ מֵאֶרֶץ מִצְרָיִם:  מ  כִּי תֵצֵא לַמִּלְחָמָה עַל אֹיְבֶיךָ וְרָאִיתָ סוּס וָרֶכֶב עַם רַב מִמְּךָ לֹא תִירָא מֵהֶם כִּי יְהוָה אֱלֹהֶיךָ עִמָּךְ הַמַּעַלְךָ מֵאֶרֶץ מִצְרָיִם:  ת  אֲרֵי תִפּוֹק לַאֲגָחָא קְרָבָא עַל בַּעֲלֵי דְבָבָךְ וְתֶחֱזֵי סוּסָוָן וּרְתִכִּין עַם סַגִי מִנָךְ לָא תִדְחַל מִנְהוֹן אֲרֵי יְיָ אֱלָהָךְ מֵימְרֵהּ בְּסַעְדָךְ דְאַסְקָךְ מֵאַרְעָא דְמִצְרָיִם:

{ב} וְהָיָה כְּקָרָבְכֶם אֶל הַמִּלְחָמָה וְנִגַּשׁ הַכֹּהֵן וְדִבֶּר אֶל הָעָם:  מ  וְהָיָה כְּקָרָבְכֶם אֶל הַמִּלְחָמָה וְנִגַּשׁ הַכֹּהֵן וְדִבֶּר אֶל הָעָם:  ת  וִיהֵי כְּמִקְרָבְכוֹן לַאֲגָחָא קְרָבָא וְיִתְקְרִיב כַּהֲנָא וִימַלֵל עִם עַמָא:

{ג} וְאָמַר אֲלֵהֶם שְׁמַע יִשְׂרָאֵל אַתֶּם קְרֵבִים הַיּוֹם לַמִּלְחָמָה עַל אֹיְבֵיכֶם אַל יֵרַךְ לְבַבְכֶם אַל תִּירְאוּ וְאַל תַּחְפְּזוּ וְאַל תַּעַרְצוּ מִפְּנֵיהֶם:  מ  וְאָמַר אֲלֵהֶם שְׁמַע יִשְׂרָאֵל אַתֶּם קְרֵבִים הַיּוֹם לַמִּלְחָמָה עַל אֹיְבֵיכֶם אַל יֵרַךְ לְבַבְכֶם אַל תִּירְאוּ וְאַל תַּחְפְּזוּ וְאַל תַּעַרְצוּ מִפְּנֵיהֶם:  ת  וְיֵימַר לְהוֹן שְׁמַע יִשְׂרָאֵל אַתּוּן קְרִיבִין יוֹמָא דֵין לַאֲגָחָא קְרָבָא עַל בַּעֲלֵי דְבָבֵיכוֹן לָא יְזוּעַ לִבְּכוֹן לָא תִדְחֲלוּן וְלָא תִתְבַּהֲתוּן וְלָא תִתָּבְרוּן מִקֳדָמֵיהוֹן:

{ד} כִּי יְהוָה אֱלֹהֵיכֶם הַהֹלֵךְ עִמָּכֶם לְהִלָּחֵם לָכֶם עִם אֹיְבֵיכֶם לְהוֹשִׁיעַ אֶתְכֶם:  מ  כִּי יְהוָה אֱלֹהֵיכֶם הַהֹלֵךְ עִמָּכֶם לְהִלָּחֵם לָכֶם עִם אֹיְבֵיכֶם לְהוֹשִׁיעַ אֶתְכֶם:  ת  אֲרֵי יְיָ אֱלָהֲכוֹן דִמְדַבַּר קֳדָמֵיכוֹן לַאֲגָחָא לְכוֹן קְרָב עִם בַּעֲלֵי דְבָבֵיכוֹן לְמִפְרַק יָתְכוֹן:

{ה} וְדִבְּרוּ הַשֹּׁטְרִים אֶל הָעָם לֵאמֹר מִי הָאִישׁ אֲשֶׁר בָּנָה בַיִת חָדָשׁ וְלֹא חֲנָכוֹ יֵלֵךְ וְיָשֹׁב לְבֵיתוֹ פֶּן יָמוּת בַּמִּלְחָמָה וְאִישׁ אַחֵר יַחְנְכֶנּוּ:  מ  וְדִבְּרוּ הַשֹּׁטְרִים אֶל הָעָם לֵאמֹר מִי הָאִישׁ אֲשֶׁר בָּנָה בַיִת חָדָשׁ וְלֹא חֲנָכוֹ יֵלֵךְ וְיָשֹׁב לְבֵיתוֹ פֶּן יָמוּת בַּמִּלְחָמָה וְאִישׁ אַחֵר יַחְנְכֶנּוּ:  ת  וִימַלְלוּן סָרְכַיָא קֳדָם עַמָא לְמֵימָר מָן גַבְרָא דִי בְנָא בֵיתָא חַדְתָּא וְלָא חָנְכֵהּ יְהַךְ וִיתוּב לְבֵיתֵהּ דִילְמָא יְמוּת בַּאֲגָחָא קְרָבָא וּגְבַר אָחָרָן יַחְנְכִנֵהּ:

{ו} וּמִי הָאִישׁ אֲשֶׁר נָטַע כֶּרֶם וְלֹא חִלְּלוֹ יֵלֵךְ וְיָשֹׁב לְבֵיתוֹ פֶּן יָמוּת בַּמִּלְחָמָה וְאִישׁ אַחֵר יְחַלְּלֶנּוּ:  מ  וּמִי הָאִישׁ אֲשֶׁר נָטַע כֶּרֶם וְלֹא חִלְּלוֹ יֵלֵךְ וְיָשֹׁב לְבֵיתוֹ פֶּן יָמוּת בַּמִּלְחָמָה וְאִישׁ אַחֵר יְחַלְּלֶנּוּ:  ת  וּמָן גַבְרָא דִי נְצִיב כַּרְמָא וְלָא אַחֲלֵהּ יְהַךְ וִיתוּב לְבֵיתֵהּ דִילְמָא יְמוּת בַּאֲגָחָא קְרָבָא וּגְבַר אָחָרָן יְחַלִנֵהּ:

{ז} וּמִי הָאִישׁ אֲשֶׁר אֵרַשׂ אִשָּׁה וְלֹא לְקָחָהּ יֵלֵךְ וְיָשֹׁב לְבֵיתוֹ פֶּן יָמוּת בַּמִּלְחָמָה וְאִישׁ אַחֵר יִקָּחֶנָּה:  מ  וּמִי הָאִישׁ אֲשֶׁר אֵרַשׂ אִשָּׁה וְלֹא לְקָחָהּ יֵלֵךְ וְיָשֹׁב לְבֵיתוֹ פֶּן יָמוּת בַּמִּלְחָמָה וְאִישׁ אַחֵר יִקָּחֶנָּה:  ת  וּמָן גַבְרָא דִי אֵרַס אִתְּתָא וְלָא נָסְבַהּ יְהַךְ וִיתוּב לְבֵיתֵהּ דִילְמָא יְמוּת בַּאֲגָחָא קְרָבָא וּגְבַר אָחֳרָן יִסְבִנַהּ:

{ח} וְיָסְפוּ הַשֹּׁטְרִים לְדַבֵּר אֶל הָעָם וְאָמְרוּ מִי הָאִישׁ הַיָּרֵא וְרַךְ הַלֵּבָב יֵלֵךְ וְיָשֹׁב לְבֵיתוֹ וְלֹא יִמַּס אֶת לְבַב אֶחָיו כִּלְבָבוֹ:  מ  וְיָסְפוּ הַשֹּׁטְרִים לְדַבֵּר אֶל הָעָם וְאָמְרוּ מִי הָאִישׁ הַיָּרֵא וְרַךְ הַלֵּבָב יֵלֵךְ וְיָשֹׁב לְבֵיתוֹ וְלֹא יִמַּס אֶת לְבַב אֶחָיו כִּלְבָבוֹ:  ת  וְיוֹסְפוּן סָרְכַיָא לְמַלָלָא עִם עַמָא וְיֵימְרוּן מָן גַבְרָא דְדָחֵל וּתְבִיר לִבָּא יְהַךְ וִיתוּב לְבֵיתֵהּ וְלָא יִתְּבַר יָת לִבָּא דַאַחוֹהִי כְּלִבֵּהּ:

{ט} וְהָיָה כְּכַלֹּת הַשֹּׁטְרִים לְדַבֵּר אֶל הָעָם וּפָקְדוּ שָׂרֵי צְבָאוֹת בְּרֹאשׁ הָעָם:  מ  וְהָיָה כְּכַלֹּת הַשֹּׁטְרִים לְדַבֵּר אֶל הָעָם וּפָקְדוּ שָׂרֵי צְבָאוֹת בְּרֹאשׁ הָעָם:  ת  וִיהֵי כַּד יְשֵׁצוּן סָרְכַיָא לְמַלָלָא עִם עַמָא וִימַנוּן רַבָּנֵי חֵילָא בְּרֵישׁ עַמָא: (ס)

{י}  שביעי  כִּי תִקְרַב אֶל עִיר לְהִלָּחֵם עָלֶיהָ וְקָרָאתָ אֵלֶיהָ לְשָׁלוֹם:  מ  כִּי תִקְרַב אֶל עִיר לְהִלָּחֵם עָלֶיהָ וְקָרָאתָ אֵלֶיהָ לְשָׁלוֹם:  ת  אֲרֵי תִקְרַב לְקַרְתָּא לַאֲגָחָא (קְרָבָא) עֲלַהּ וְתִקְרֵי לַהּ מִלִין דִשְׁלָם:

{יא} וְהָיָה אִם שָׁלוֹם תַּעַנְךָ וּפָתְחָה לָךְ וְהָיָה כָּל הָעָם הַנִּמְצָא בָהּ יִהְיוּ לְךָ לָמַס וַעֲבָדוּךָ:  מ  וְהָיָה אִם שָׁלוֹם תַּעַנְךָ וּפָתְחָה לָךְ וְהָיָה כָּל הָעָם הַנִּמְצָא בָהּ יִהְיוּ לְךָ לָמַס וַעֲבָדוּךָ:  ת  וִיהֵי אִם שְׁלַם תַּעֲנִנָךְ וְתִפְתַּח לָךְ וִיהֵי כָּל עַמָא דְיִשְׁתְּכַח בַּהּ יְהוֹן לָךְ מַסְקֵי מִסִין וְיִפְלְחֻנָךְ:

{יב} וְאִם לֹא תַשְׁלִים עִמָּךְ וְעָשְׂתָה עִמְּךָ מִלְחָמָה וְצַרְתָּ עָלֶיהָ:  מ  וְאִם לֹא תַשְׁלִים עִמָּךְ וְעָשְׂתָה עִמְּךָ מִלְחָמָה וְצַרְתָּ עָלֶיהָ:  ת  וְאִם לָא תַשְׁלֵם עִמָךְ וְתַעְבֵּד עִמָךְ קְרָב וּתְצוּר עֲלַהּ:

{יג} וּנְתָנָהּ יְהוָה אֱלֹהֶיךָ בְּיָדֶךָ וְהִכִּיתָ אֶת כָּל זְכוּרָהּ לְפִי חָרֶב:  מ  וּנְתָנָהּ יְהוָה אֱלֹהֶיךָ בְּיָדֶךָ וְהִכִּיתָ אֶת כָּל זְכוּרָהּ לְפִי חָרֶב:  ת  וְיִמְסְרִנַהּ יְיָ אֱלָהָךְ בִּידָךְ וְתִמְחֵי יָת כָּל דְכוּרַהּ לְפִתְגַם דְחָרֶב:

{יד} רַק הַנָּשִׁים וְהַטַּף וְהַבְּהֵמָה וְכֹל אֲשֶׁר יִהְיֶה בָעִיר כָּל שְׁלָלָהּ תָּבֹז לָךְ וְאָכַלְתָּ אֶת שְׁלַל אֹיְבֶיךָ אֲשֶׁר נָתַן יְהוָה אֱלֹהֶיךָ לָךְ:  מ  רַק הַנָּשִׁים וְהַטַּף וְהַבְּהֵמָה וְכֹל אֲשֶׁר יִהְיֶה בָעִיר כָּל שְׁלָלָהּ תָּבֹז לָךְ וְאָכַלְתָּ אֶת שְׁלַל אֹיְבֶיךָ אֲשֶׁר נָתַן יְהוָה אֱלֹהֶיךָ לָךְ:  ת  לְחוֹד נְשַׁיָא וְטַפְלָא וּבְעִירָא וְכֹל דִי יְהֵי בְקַרְתָּא כָּל עֲדָאַהּ תֵּבוֹז לָךְ וְתֵיכוּל יָת עֲדָאָה דְסַנְאָךְ דִי יְהַב יְיָ אֱלָהָךְ לָךְ:

{טו} כֵּן תַּעֲשֶׂה לְכָל הֶעָרִים הָרְחֹקֹת מִמְּךָ מְאֹד אֲשֶׁר לֹא מֵעָרֵי הַגּוֹיִם הָאֵלֶּה הֵנָּה:  מ  כֵּן תַּעֲשֶׂה לְכָל הֶעָרִים הָרְחֹקֹת מִמְּךָ מְאֹד אֲשֶׁר לֹא מֵעָרֵי הַגּוֹיִם הָאֵלֶּה הֵנָּה:  ת  כֵּן תַּעְבֵד לְכָל קִרְוַיָא דִי רְחִיקִין מִנָךְ לַחֲדָא דִי לָא מִקִרְוֵי עַמְמַיָא הָאִלֵין אִנוּן:

{טז} רַק מֵעָרֵי הָעַמִּים הָאֵלֶּה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה לֹא תְחַיֶּה כָּל נְשָׁמָה:  מ  רַק מֵעָרֵי הָעַמִּים הָאֵלֶּה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה לֹא תְחַיֶּה כָּל נְשָׁמָה:  ת  לְחוֹד מִקִרְוֵי עַמְמַיָא הָאִלֵין דִי יְיָ אֱלָהָךְ יָהֵב לָךְ אַחֲסָנָא לָא תְקַיַם כָּל נִשְׁמָתָא:

{יז} כִּי הַחֲרֵם תַּחֲרִימֵם הַחִתִּי וְהָאֱמֹרִי הַכְּנַעֲנִי וְהַפְּרִזִּי הַחִוִּי וְהַיְבוּסִי כַּאֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ:  מ  כִּי הַחֲרֵם תַּחֲרִימֵם הַחִתִּי וְהָאֱמֹרִי הַכְּנַעֲנִי וְהַפְּרִזִּי הַחִוִּי וְהַיְבוּסִי כַּאֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ:  ת  אֲרֵי גַמָרָא תְגַמְרִנוּן חִתָּאֵי וֶאֱמֹרָאֵי כְּנַעֲנָאֵי וּפְרִזָאֵי חִוָאֵי וִיבוּסָאֵי כְּמָא דִי פַקְדָךְ יְיָ אֱלָהָךְ:

{יח} לְמַעַן אֲשֶׁר לֹא יְלַמְּדוּ אֶתְכֶם לַעֲשׂוֹת כְּכֹל תּוֹעֲבֹתָם אֲשֶׁר עָשׂוּ לֵאלֹהֵיהֶם וַחֲטָאתֶם לַיהוָה אֱלֹהֵיכֶם:  מ  לְמַעַן אֲשֶׁר לֹא יְלַמְּדוּ אֶתְכֶם לַעֲשׂוֹת כְּכֹל תּוֹעֲבֹתָם אֲשֶׁר עָשׂוּ לֵאלֹהֵיהֶם וַחֲטָאתֶם לַיהוָה אֱלֹהֵיכֶם:  ת  בְּדִיל דִי לָא יַלְפוּן יָתְכוֹן לְמֶעְבַּד כְּכָל תּוֹעֲבָתְהוֹן דִי עֲבָדוּ לְטַעֲוָתְהוֹן וּתְחוֹבוּן קֳדָם יְיָ אֱלָהֲכוֹן: (ס)

{יט} כִּי תָצוּר אֶל עִיר יָמִים רַבִּים לְהִלָּחֵם עָלֶיהָ לְתָפְשָׂהּ לֹא תַשְׁחִית אֶת עֵצָהּ לִנְדֹּחַ עָלָיו גַּרְזֶן כִּי מִמֶּנּוּ תֹאכֵל וְאֹתוֹ לֹא תִכְרֹת כִּי הָאָדָם עֵץ הַשָּׂדֶה לָבֹא מִפָּנֶיךָ בַּמָּצוֹר:  מ  כִּי תָצוּר אֶל עִיר יָמִים רַבִּים לְהִלָּחֵם עָלֶיהָ לְתָפְשָׂהּ לֹא תַשְׁחִית אֶת עֵצָהּ לִנְדֹּחַ עָלָיו גַּרְזֶן כִּי מִמֶּנּוּ תֹאכֵל וְאֹתוֹ לֹא תִכְרֹת כִּי הָאָדָם עֵץ הַשָּׂדֶה לָבֹא מִפָּנֶיךָ בַּמָּצוֹר:  ת  אֲרֵי תְצוּר לְקַרְתָּא יוֹמִין סַגִיאִין לַאֲגָחָא עֲלַהּ לְמִכְבְּשַׁהּ לָא תְחַבֵּל יָת אִילָנַהּ לַאֲרָמָא עֲלוֹהִי פַּרְזְלָא אֲרֵי מִנֵה תֵּיכוּל וְיָתֵהּ לָא תְקוּץ אֲרֵי לָא כַאֲנָשָׁא אִילַן חַקְלָא לְמֵיעַל מְקָדָמָךְ בִּצְיָרָא:

{כ} רַק עֵץ אֲשֶׁר תֵּדַע כִּי לֹא עֵץ מַאֲכָל הוּא אֹתוֹ תַשְׁחִית וְכָרָתָּ וּבָנִיתָ מָצוֹר עַל הָעִיר אֲשֶׁר הִוא עֹשָׂה עִמְּךָ מִלְחָמָה עַד רִדְתָּהּ:  מ  רַק עֵץ אֲשֶׁר תֵּדַע כִּי לֹא עֵץ מַאֲכָל הוּא אֹתוֹ תַשְׁחִית וְכָרָתָּ וּבָנִיתָ מָצוֹר עַל הָעִיר אֲשֶׁר הִוא עֹשָׂה עִמְּךָ מִלְחָמָה עַד רִדְתָּהּ:  ת  לְחוֹד אִילָן דְתִדַע אֲרֵי לָא אִילַן דְמַאֲכָל הוּא יָתֵהּ תְּחַבַּל וּתְקוּץ וְתִבְנֵי כַרְקוֹמִין עַל קַרְתָּא דִי הִיא עָבְדָא עִמָךְ קְרָבָא עַד דְתִכְבְּשַׁהּ: (פ)




דברים פרק-כא

{א} כִּי יִמָּצֵא חָלָל בָּאֲדָמָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לְרִשְׁתָּהּ נֹפֵל בַּשָּׂדֶה לֹא נוֹדַע מִי הִכָּהוּ:  מ  כִּי יִמָּצֵא חָלָל בָּאֲדָמָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לְרִשְׁתָּהּ נֹפֵל בַּשָּׂדֶה לֹא נוֹדַע מִי הִכָּהוּ:  ת  אֲרֵי יִשְׁתְּכַח קְטִּילָא בְּאַרְעָא דִי יְיָ אֱלָהָךְ יָהֵב לָךְ לְמֵירְתַהּ רָמֵי בְּחַקְלָא לָא אִתְיְדַע מַן קָטְּלֵהּ:

{ב} וְיָצְאוּ זְקֵנֶיךָ וְשֹׁפְטֶיךָ וּמָדְדוּ אֶל הֶעָרִים אֲשֶׁר סְבִיבֹת הֶחָלָל:  מ  וְיָצְאוּ זְקֵנֶיךָ וְשֹׁפְטֶיךָ וּמָדְדוּ אֶל הֶעָרִים אֲשֶׁר סְבִיבֹת הֶחָלָל:  ת  וְיִפְּקוּן סָבָיךְ וְדַיָנָיךְ וְיִמְשְׁחוּן לְקִרְוַיָא דִי סַחֲרָנוּת קְטִילָא:

{ג} וְהָיָה הָעִיר הַקְּרֹבָה אֶל הֶחָלָל וְלָקְחוּ זִקְנֵי הָעִיר הַהִוא עֶגְלַת בָּקָר אֲשֶׁר לֹא עֻבַּד בָּהּ אֲשֶׁר לֹא מָשְׁכָה בְּעֹל:  מ  וְהָיָה הָעִיר הַקְּרֹבָה אֶל הֶחָלָל וְלָקְחוּ זִקְנֵי הָעִיר הַהִוא עֶגְלַת בָּקָר אֲשֶׁר לֹא עֻבַּד בָּהּ אֲשֶׁר לֹא מָשְׁכָה בְּעֹל:  ת  וּתְהֵי קַרְתָּא דִקְרִיבָא לִקְטִילָא וְיִדְבְּרוּן סָבֵי קַרְתָּא הַהִיא עֶגְלַת תּוֹרֵי דִי לָא אִתְפְּלַח בַּהּ דִי לָא נְגִידַת בְּנוּר:

{ד} וְהוֹרִדוּ זִקְנֵי הָעִיר הַהִוא אֶת הָעֶגְלָה אֶל נַחַל אֵיתָן אֲשֶׁר לֹא יֵעָבֵד בּוֹ וְלֹא יִזָּרֵעַ וְעָרְפוּ שָׁם אֶת הָעֶגְלָה בַּנָּחַל:  מ  וְהוֹרִדוּ זִקְנֵי הָעִיר הַהִוא אֶת הָעֶגְלָה אֶל נַחַל אֵיתָן אֲשֶׁר לֹא יֵעָבֵד בּוֹ וְלֹא יִזָּרֵעַ וְעָרְפוּ שָׁם אֶת הָעֶגְלָה בַּנָּחַל:  ת  וְיֶחֱתוּן סָבֵי קַרְתָּא הַהִיא יָת עֲגַלְתָּא לְנַחֲלָא בַיָר דִי לָא יִתְפְּלַח בֵּהּ וְלָא יִזְדְרַע וְיִנְקְפוּן תַּמָן יָת עֲגַלְתָּא בְּנַחֲלָא:

{ה} וְנִגְּשׁוּ הַכֹּהֲנִים בְּנֵי לֵוִי כִּי בָם בָּחַר יְהוָה אֱלֹהֶיךָ לְשָׁרְתוֹ וּלְבָרֵךְ בְּשֵׁם יְהוָה וְעַל פִּיהֶם יִהְיֶה כָּל רִיב וְכָל נָגַע:  מ  וְנִגְּשׁוּ הַכֹּהֲנִים בְּנֵי לֵוִי כִּי בָם בָּחַר יְהוָה אֱלֹהֶיךָ לְשָׁרְתוֹ וּלְבָרֵךְ בְּשֵׁם יְהוָה וְעַל פִּיהֶם יִהְיֶה כָּל רִיב וְכָל נָגַע:  ת  וְיִתְקָרְבוּן כָּהֲנַיָא בְּנֵי לֵוִי אֲרֵי בְהוֹן יִתִּרְעֵי יְיָ אֱלָהָךְ לְשַׁמָשׁוּתֵהּ וּלְבָרָכָא בִּשְׁמָא דַיְיָ וְעַל מֵימְרֵהוֹן יְהֵי כָּל דִין וְכָל מַכְתַּשׁ סְגִירוּ:

{ו} וְכֹל זִקְנֵי הָעִיר הַהִוא הַקְּרֹבִים אֶל הֶחָלָל יִרְחֲצוּ אֶת יְדֵיהֶם עַל הָעֶגְלָה הָעֲרוּפָה בַנָּחַל:  מ  וְכֹל זִקְנֵי הָעִיר הַהִוא הַקְּרֹבִים אֶל הֶחָלָל יִרְחֲצוּ אֶת יְדֵיהֶם עַל הָעֶגְלָה הָעֲרוּפָה בַנָּחַל:  ת  וְכֹל סָבֵי קַרְתָּא הַהִיא דִקְרִיבִין לִקְטִילָא יַסְחוּן יָת יְדֵיהוֹן עַל עֲגַלְתָּא דִנְקֶפְתָּא בְנַחֲלָא:

{ז}  מפטיר  וְעָנוּ וְאָמְרוּ יָדֵינוּ לֹא (שפכה) שָׁפְכוּ אֶת הַדָּם הַזֶּה וְעֵינֵינוּ לֹא רָאוּ:  מ  וְעָנוּ וְאָמְרוּ יָדֵינוּ לֹא (שפכה) שָׁפְכוּ אֶת הַדָּם הַזֶּה וְעֵינֵינוּ לֹא רָאוּ:  ת  וִיתִיבוּן וְיֵימְרוּן יְדָנָא לָא אַשְׁדוֹ יָת דְמָא הָדֵין וְעֵינָנָא לָא חֲזָאָה:

{ח} כַּפֵּר לְעַמְּךָ יִשְׂרָאֵל אֲשֶׁר פָּדִיתָ יְהוָה וְאַל תִּתֵּן דָּם נָקִי בְּקֶרֶב עַמְּךָ יִשְׂרָאֵל וְנִכַּפֵּר לָהֶם הַדָּם:  מ  כַּפֵּר לְעַמְּךָ יִשְׂרָאֵל אֲשֶׁר פָּדִיתָ יְהוָה וְאַל תִּתֵּן דָּם נָקִי בְּקֶרֶב עַמְּךָ יִשְׂרָאֵל וְנִכַּפֵּר לָהֶם הַדָּם:  ת  כָּהֲנַיָא יֵימְרוּן כַּפַּר לְעַמָךְ יִשְׂרָאֵל דִי פְרַקְתָּא יְיָ וְלָא תִתֵּן חוֹבַת דַם זַכַּי בְּגוֹ עַמָךְ יִשְׂרָאֵל וְיִתְכַּפַּר לְהוֹן עַל דְמָא:

{ט} וְאַתָּה תְּבַעֵר הַדָּם הַנָּקִי מִקִּרְבֶּךָ כִּי תַעֲשֶׂה הַיָּשָׁר בְּעֵינֵי יְהוָה:  מ  וְאַתָּה תְּבַעֵר הַדָּם הַנָּקִי מִקִּרְבֶּךָ כִּי תַעֲשֶׂה הַיָּשָׁר בְּעֵינֵי יְהוָה:  ת  וְאַתְּ תְּפַלֵי אַשְׁדִי דַם זַכַּי מִבֵּינָךְ אֲרֵי תַעְבֵּד דְכָשַׁר קֳדָם יְיָ: ססס: (ססס)
