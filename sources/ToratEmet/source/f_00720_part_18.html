צ

צ-א


[ויקרא פרק-יג-יא]  צָרַעַת נוֹשֶׁנֶת הִוא בְּעוֹר בְּשָׂרוֹ וְטִמְּאוֹ הַכֹּהֵן לֹא יַסְגִּרֶנּוּ כִּי טָמֵא הוּא:

[משלי פרק-כז-טז]  צֹפְנֶיהָ צָפַן רוּחַ וְשֶׁמֶן יְמִינוֹ יִקְרָא:


צ-ב


[ישעיה פרק-מח-כ]  צְאוּ מִבָּבֶל בִּרְחוּ מִכַּשְׂדִּים בְּקוֹל רִנָּה הַגִּידוּ הַשְׁמִיעוּ זֹאת הוֹצִיאוּהָ עַד קְצֵה הָאָרֶץ אִמְרוּ גָּאַל יְהוָה עַבְדּוֹ יַעֲקֹב:

[תהילים פרק-מ-יא]  צִדְקָתְךָ לֹא כִסִּיתִי בְּתוֹךְ לִבִּי אֱמוּנָתְךָ וּתְשׁוּעָתְךָ אָמָרְתִּי לֹא כִחַדְתִּי חַסְדְּךָ וַאֲמִתְּךָ לְקָהָל רָב:


צ-ד


[ויקרא פרק-כד-ב]  צַו אֶת בְּנֵי יִשְׂרָאֵל וְיִקְחוּ אֵלֶיךָ שֶׁמֶן זַיִת זָךְ כָּתִית לַמָּאוֹר לְהַעֲלֹת נֵר תָּמִיד:

[יהושע פרק-טו-לז]  צְנָן וַחֲדָשָׁה וּמִגְדַּל גָּד:

[ירמיה פרק-יב-א]  צַדִּיק אַתָּה יְהוָה כִּי אָרִיב אֵלֶיךָ אַךְ מִשְׁפָּטִים אֲדַבֵּר אוֹתָךְ מַדּוּעַ דֶּרֶךְ רְשָׁעִים צָלֵחָה שָׁלוּ כָּל בֹּגְדֵי בָגֶד:

[תהילים פרק-קיט-קלח]  צִוִּיתָ צֶדֶק עֵדֹתֶיךָ וֶאֱמוּנָה מְאֹד:


צ-ה


[בראשית פרק-ו-טז]  צֹהַר תַּעֲשֶׂה לַתֵּבָה וְאֶל אַמָּה תְּכַלֶנָּה מִלְמַעְלָה וּפֶתַח הַתֵּבָה בְּצִדָּהּ תָּשִׂים תַּחְתִּיִּם שְׁנִיִּם וּשְׁלִשִׁים תַּעֲשֶׂהָ:

[במדבר פרק-לד-ב]  צַו אֶת בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם כִּי אַתֶּם בָּאִים אֶל הָאָרֶץ כְּנָעַן זֹאת הָאָרֶץ אֲשֶׁר תִּפֹּל לָכֶם בְּנַחֲלָה אֶרֶץ כְּנַעַן לִגְבֻלֹתֶיהָ:

[שמואל ב פרק-כג-לז]  צֶלֶק הָעַמֹּנִי נַחְרַי הַבְּאֵרֹתִי (נשאי) נֹשֵׂא כְּלֵי יוֹאָב בֶּן צְרֻיָה:

[ישעיה פרק-א-כז]  צִיּוֹן בְּמִשְׁפָּט תִּפָּדֶה וְשָׁבֶיהָ בִּצְדָקָה:

[ירמיה פרק-נא-מה]  צְאוּ מִתּוֹכָהּ עַמִּי וּמַלְּטוּ אִישׁ אֶת נַפְשׁוֹ מֵחֲרוֹן אַף יְהוָה:

[תהילים פרק-לו-ז]  צִדְקָתְךָ כְּהַרְרֵי אֵל מִשְׁפָּטֶךָ תְּהוֹם רַבָּה אָדָם וּבְהֵמָה תוֹשִׁיעַ יְהוָה:

[תהילים פרק-לז-כט]  צַדִּיקִים יִירְשׁוּ אָרֶץ וְיִשְׁכְּנוּ לָעַד עָלֶיהָ:

[תהילים פרק-צב-יג]  צַדִּיק כַּתָּמָר יִפְרָח כְּאֶרֶז בַּלְּבָנוֹן יִשְׂגֶּה:

[תהילים פרק-קיט-קמ]  צְרוּפָה אִמְרָתְךָ מְאֹד וְעַבְדְּךָ אֲהֵבָהּ:

[תהילים פרק-קיט-קמד]  צֶדֶק עֵדְוֹתֶיךָ לְעוֹלָם הֲבִינֵנִי וְאֶחְיֶה:

[דברי הימים א פרק-יא-לט]  צֶלֶק הָעַמּוֹנִי נַחְרַי הַבֵּרֹתִי נֹשֵׂא כְּלֵי יוֹאָב בֶּן צְרוּיָה:


צ-ו


[ויקרא פרק-ו-ב]  צַו אֶת אַהֲרֹן וְאֶת בָּנָיו לֵאמֹר זֹאת תּוֹרַת הָעֹלָה הִוא הָעֹלָה עַל מוֹקְדָה עַל הַמִּזְבֵּחַ כָּל הַלַּיְלָה עַד הַבֹּקֶר וְאֵשׁ הַמִּזְבֵּחַ תּוּקַד בּוֹ:

[במדבר פרק-כח-ב]  צַו אֶת בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם אֶת קָרְבָּנִי לַחְמִי לְאִשַּׁי רֵיחַ נִיחֹחִי תִּשְׁמְרוּ לְהַקְרִיב לִי בְּמוֹעֲדוֹ:

[הושע פרק-ט-ח]  צֹפֶה אֶפְרַיִם עִם אֱלֹהָי נָבִיא פַּח יָקוֹשׁ עַל כָּל דְּרָכָיו מַשְׂטֵמָה בְּבֵית אֱלֹהָיו:

[הושע פרק-יג-יב]  צָרוּר עֲוֹן אֶפְרָיִם צְפוּנָה חַטָּאתוֹ:

[תהילים פרק-לז-לב]  צוֹפֶה רָשָׁע לַצַּדִּיק וּמְבַקֵּשׁ לַהֲמִיתוֹ:

[תהילים פרק-סח-כט]  צִוָּה אֱלֹהֶיךָ עֻזֶּךָ עוּזָּה אֱלֹהִים זוּ פָּעַלְתָּ לָּנוּ:

[תהילים פרק-פה-יד]  צֶדֶק לְפָנָיו יְהַלֵּךְ וְיָשֵׂם לְדֶרֶךְ פְּעָמָיו:

[תהילים פרק-פט-יג]  צָפוֹן וְיָמִין אַתָּה בְרָאתָם תָּבוֹר וְחֶרְמוֹן בְּשִׁמְךָ יְרַנֵּנוּ:

[תהילים פרק-קמה-יז]  צַדִּיק יְהוָה בְּכָל דְּרָכָיו וְחָסִיד בְּכָל מַעֲשָׂיו:

[משלי פרק-ז-כ]  צְרוֹר הַכֶּסֶף לָקַח בְּיָדוֹ לְיוֹם הַכֵּסֶא יָבֹא בֵיתוֹ:

[משלי פרק-יא-ו]  צִדְקַת יְשָׁרִים תַּצִּילֵם וּבְהַוַּת בֹּגְדִים יִלָּכֵדוּ:

[משלי פרק-יא-ח]  צַדִּיק מִצָּרָה נֶחֱלָץ וַיָּבֹא רָשָׁע תַּחְתָּיו:

[משלי פרק-יח-יז]  צַדִּיק הָרִאשׁוֹן בְּרִיבוֹ (יבא) וּבָא רֵעֵהוּ וַחֲקָרוֹ:

[איוב פרק-כד-יט]  צִיָּה גַם חֹם יִגְזְלוּ מֵימֵי שֶׁלֶג שְׁאוֹל חָטָאוּ:

[שיר השירים פרק-ג-יא]  צְאֶינָה וּרְאֶינָה בְּנוֹת צִיּוֹן בַּמֶּלֶךְ שְׁלֹמֹה בָּעֲטָרָה שֶׁעִטְּרָה לּוֹ אִמּוֹ בְּיוֹם חֲתֻנָּתוֹ וּבְיוֹם שִׂמְחַת לִבּוֹ:

[איכה פרק-ד-יח]  צָדוּ צְעָדֵינוּ מִלֶּכֶת בִּרְחֹבֹתֵינוּ קָרַב קִצֵּינוּ מָלְאוּ יָמֵינוּ כִּי בָא קִצֵּינוּ:

[דברי הימים א פרק-ו-לח]  צָדוֹק בְּנוֹ אֲחִימַעַץ בְּנוֹ:


צ-ח


[ירמיה פרק-נ-ה]  צִיּוֹן יִשְׁאָלוּ דֶּרֶךְ הֵנָּה פְנֵיהֶם בֹּאוּ וְנִלְווּ אֶל יְהוָה בְּרִית עוֹלָם לֹא תִשָּׁכֵחַ:


צ-י


[שמואל ב פרק-כג-כח]  צַלְמוֹן הָאֲחֹחִי מַהְרַי הַנְּטֹפָתִי:

[ישעיה פרק-ח-טז]  צוֹר תְּעוּדָה חֲתוֹם תּוֹרָה בְּלִמֻּדָי:

[תהילים פרק-ח-ח]  צֹנֶה וַאֲלָפִים כֻּלָּם וְגַם בַּהֲמוֹת שָׂדָי:

[תהילים פרק-כה-יז]  צָרוֹת לְבָבִי הִרְחִיבוּ מִמְּצוּקוֹתַי הוֹצִיאֵנִי:

[תהילים פרק-קיט-קלט]  צִמְּתַתְנִי קִנְאָתִי כִּי שָׁכְחוּ דְבָרֶיךָ צָרָי:

[תהילים פרק-קיט-קמא]  צָעִיר אָנֹכִי וְנִבְזֶה פִּקֻּדֶיךָ לֹא שָׁכָחְתִּי:

[תהילים פרק-קיט-קמג]  צַר וּמָצוֹק מְצָאוּנִי מִצְוֹתֶיךָ שַׁעֲשֻׁעָי:

[איוב פרק-כט-יד]  צֶדֶק לָבַשְׁתִּי וַיִּלְבָּשֵׁנִי כִּמְעִיל וְצָנִיף מִשְׁפָּטִי:

[איכה פרק-א-יח]  צַדִּיק הוּא יְהוָה כִּי פִיהוּ מָרִיתִי שִׁמְעוּ נָא כָל (עמים) הָעַמִּים וּרְאוּ מַכְאֹבִי בְּתוּלֹתַי וּבַחוּרַי הָלְכוּ בַשֶּׁבִי:

[איכה פרק-ג-נג]  צָמְתוּ בַבּוֹר חַיָּי וַיַּדּוּ אֶבֶן בִּי:

[איכה פרק-ג-נד]  צָפוּ מַיִם עַל רֹאשִׁי אָמַרְתִּי נִגְזָרְתִּי:


צ-כ


[בראשית פרק-ח-טז]  צֵא מִן הַתֵּבָה אַתָּה וְאִשְׁתְּךָ וּבָנֶיךָ וּנְשֵׁי בָנֶיךָ אִתָּךְ:

[דברים פרק-טז-כ]  צֶדֶק צֶדֶק תִּרְדֹּף לְמַעַן תִּחְיֶה וְיָרַשְׁתָּ אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:

[דברים פרק-לב-יח]  צוּר יְלָדְךָ תֶּשִׁי וַתִּשְׁכַּח אֵל מְחֹלְלֶךָ:

[ישעיה פרק-כב-יח]  צָנוֹף יִצְנָפְךָ צְנֵפָה כַּדּוּר אֶל אֶרֶץ רַחֲבַת יָדָיִם שָׁמָּה תָמוּת וְשָׁמָּה מַרְכְּבוֹת כְּבוֹדֶךָ קְלוֹן בֵּית אֲדֹנֶיךָ:

[תהילים פרק-פט-טו]  צֶדֶק וּמִשְׁפָּט מְכוֹן כִּסְאֶךָ חֶסֶד וֶאֱמֶת יְקַדְּמוּ פָנֶיךָ:

[תהילים פרק-קיט-קלז]  צַדִּיק אַתָּה יְהוָה וְיָשָׁר מִשְׁפָּטֶיךָ:

[איכה פרק-ב-יח]  צָעַק לִבָּם אֶל אֲדֹנָי חוֹמַת בַּת צִיּוֹן הוֹרִידִי כַנַּחַל דִּמְעָה יוֹמָם וָלַיְלָה אַל תִּתְּנִי פוּגַת לָךְ אַל תִּדֹּם בַּת עֵינֵךְ:


צ-ל


[ישעיה פרק-יב-ו]  צַהֲלִי וָרֹנִּי יוֹשֶׁבֶת צִיּוֹן כִּי גָדוֹל בְּקִרְבֵּךְ קְדוֹשׁ יִשְׂרָאֵל:

[משלי פרק-לא-כז]  צוֹפִיָּה הֲלִיכוֹת בֵּיתָהּ וְלֶחֶם עַצְלוּת לֹא תֹאכֵל:


צ-מ


[במדבר פרק-כה-יז]  צָרוֹר אֶת הַמִּדְיָנִים וְהִכִּיתֶם אוֹתָם:

[במדבר פרק-לה-ב]  צַו אֶת בְּנֵי יִשְׂרָאֵל וְנָתְנוּ לַלְוִיִּם מִנַּחֲלַת אֲחֻזָּתָם עָרִים לָשָׁבֶת וּמִגְרָשׁ לֶעָרִים סְבִיבֹתֵיהֶם תִּתְּנוּ לַלְוִיִּם:

[שמואל א פרק-ח-יז]  צֹאנְכֶם יַעְשֹׂר וְאַתֶּם תִּהְיוּ לוֹ לַעֲבָדִים:

[שמואל ב פרק-א-כו]  צַר לִי עָלֶיךָ אָחִי יְהוֹנָתָן נָעַמְתָּ לִּי מְאֹד נִפְלְאַתָה אַהֲבָתְךָ לִי מֵאַהֲבַת נָשִׁים:

[ישעיה פרק-נו-י]  (צפו) צֹפָיו עִוְרִים כֻּלָּם לֹא יָדָעוּ כֻּלָּם כְּלָבִים אִלְּמִים לֹא יוּכְלוּ לִנְבֹּחַ הֹזִים שֹׁכְבִים אֹהֲבֵי לָנוּם:

[ירמיה פרק-נ-ו]  צֹאן אֹבְדוֹת (היה) הָיוּ עַמִּי רֹעֵיהֶם הִתְעוּם הָרִים (שובבים) שׁוֹבְבוּם מֵהַר אֶל גִּבְעָה הָלָכוּ שָׁכְחוּ רִבְצָם:

[הושע פרק-ד-יט]  צָרַר רוּחַ אוֹתָהּ בִּכְנָפֶיהָ וְיֵבֹשׁוּ מִזִּבְחוֹתָם:

[תהילים פרק-ח-ט]  צִפּוֹר שָׁמַיִם וּדְגֵי הַיָּם עֹבֵר אָרְחוֹת יַמִּים:

[תהילים פרק-לד-יח]  צָעֲקוּ וַיהוָה שָׁמֵעַ וּמִכָּל צָרוֹתָם הִצִּילָם:

[תהילים פרק-מב-ג]  צָמְאָה נַפְשִׁי לֵאלֹהִים לְאֵל חָי מָתַי אָבוֹא וְאֵרָאֶה פְּנֵי אֱלֹהִים:

[תהילים פרק-קלב-טו]  צֵידָהּ בָּרֵךְ אֲבָרֵךְ אֶבְיוֹנֶיהָ אַשְׂבִּיעַ לָחֶם:

[משלי פרק-טז-כד]  צוּף דְּבַשׁ אִמְרֵי נֹעַם מָתוֹק לַנֶּפֶשׁ וּמַרְפֵּא לָעָצֶם:

[משלי פרק-כב-ה]  צִנִּים פַּחִים בְּדֶרֶךְ עִקֵּשׁ שׁוֹמֵר נַפְשׁוֹ יִרְחַק מֵהֶם:

[איוב פרק-כו-ח]  צֹרֵר מַיִם בְּעָבָיו וְלֹא נִבְקַע עָנָן תַּחְתָּם:

[איכה פרק-ג-נב]  צוֹד צָדוּנִי כַּצִּפּוֹר אֹיְבַי חִנָּם:


צ-נ


[יהושע פרק-ד-טז]  צַוֵּה אֶת הַכֹּהֲנִים נֹשְׂאֵי אֲרוֹן הָעֵדוּת וְיַעֲלוּ מִן הַיַּרְדֵּן:

[שיר השירים פרק-א-יג]  צְרוֹר הַמֹּר דּוֹדִי לִי בֵּין שָׁדַי יָלִין:


צ-ע


[משלי פרק-יא-ה]  צִדְקַת תָּמִים תְּיַשֵּׁר דַּרְכּוֹ וּבְרִשְׁעָתוֹ יִפֹּל רָשָׁע:


צ-צ


[ישעיה פרק-כד-יא]  צְוָחָה עַל הַיַּיִן בַּחוּצוֹת עָרְבָה כָּל שִׂמְחָה גָּלָה מְשׂוֹשׂ הָאָרֶץ:

[משלי פרק-י-ל]  צַדִּיק לְעוֹלָם בַּל יִמּוֹט וּרְשָׁעִים לֹא יִשְׁכְּנוּ אָרֶץ:


צ-ק


[שיר השירים פרק-ז-ה]  צַוָּארֵךְ כְּמִגְדַּל הַשֵּׁן עֵינַיִךְ בְּרֵכוֹת בְּחֶשְׁבּוֹן עַל שַׁעַר בַּת רַבִּים אַפֵּךְ כְּמִגְדַּל הַלְּבָנוֹן צוֹפֶה פְּנֵי דַמָּשֶׂק:


צ-ר


[דברים פרק-ג-ט]  צִידֹנִים יִקְרְאוּ לְחֶרְמוֹן שִׂרְיֹן וְהָאֱמֹרִי יִקְרְאוּ לוֹ שְׂנִיר:

[משלי פרק-יג-כה]  צַדִּיק אֹכֵל לְשֹׂבַע נַפְשׁוֹ וּבֶטֶן רְשָׁעִים תֶּחְסָר:


צ-ש


[במדבר פרק-ה-ב]  צַו אֶת בְּנֵי יִשְׂרָאֵל וִישַׁלְּחוּ מִן הַמַּחֲנֶה כָּל צָרוּעַ וְכָל זָב וְכֹל טָמֵא לָנָפֶשׁ:


צ-ת


[ישעיה פרק-י-ל]  צַהֲלִי קוֹלֵךְ בַּת גַּלִּים הַקְשִׁיבִי לַיְשָׁה עֲנִיָּה עֲנָתוֹת:

[תהילים פרק-קיט-קמב]  צִדְקָתְךָ צֶדֶק לְעוֹלָם וְתוֹרָתְךָ אֱמֶת:

[משלי פרק-יג-ו]  צְדָקָה תִּצֹּר תָּם דָּרֶךְ וְרִשְׁעָה תְּסַלֵּף חַטָּאת:

[משלי פרק-יד-לד]  צְדָקָה תְרוֹמֵם גּוֹי וְחֶסֶד לְאֻמִּים חַטָּאת:
