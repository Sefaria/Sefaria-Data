סימן צח - דין אסור שנתערב בהתר ואפן בטולו, ובו ט' סעיפים


א אִסוּר שֶׁנִּתְעָרֵב בְּהֶתֵּר מִין בְּשֶׁאֵינוֹ מִינוֹ, כְּגוֹן (א) חָלָב שֶׁנִּתְעָרֵב בְּבָשָׂר, יִטְעֲמֶנּוּ (ב) עוֹבֵד כּוֹכָבִים, אִם אוֹמֵר שֶׁאֵין בּוֹ טַעַם חָלָב אוֹ שֶׁאוֹמֵר שֶׁיֵּשׁ בּוֹ טַעַם אֶלָּא שֶׁהוּא פָּגוּם, מֻתָּר. וְהוּא שֶׁלֹּא יְהֵא סוֹפוֹ לְהַשְׁבִּיחַ. וְצָרִיךְ שֶׁלֹּא יֵדַע (ג) שֶׁסוֹמְכִין עָלָיו. וְאִם אֵין שָׁם עוֹבֵד כּוֹכָבִים לְטָעֲמוֹ, (ד) מִשַׁעֲרִינָן בְּס'. וְכֵן אִם הוּא מִין בְּמִינוֹ, כֵּיוָן דְּ יכָּא לְמֵיקַֹם אַטַּעֲמָא, מְשַׁעֲרִים בְּס'. וְאֵין נוֹהֲגִים עַכְשָׁו (ה) לִסְמֹךְ עַל אָעוֹבְדֵי כּוֹכָבִים, וּמְשַׁעֲרִינָן הַכֹּל בס' (בְּאָגוּר וּתְשׁוּבַת מהר''מ פַּדָּוואָה סִימָן ע''ט וּשְׁאָר אַחֲרוֹנִים) .


ב אִם נִתְעָרֵב מִין בְּמִינוֹ וְנִשְׁפָּךְ, בְּעִנְיָן שֶׁאֵין יְכוֹלִין לַעֲמֹד עָלָיו לְשַׁעֲרוֹ, אִם (ו) נוֹדָע שֶׁהָיָה רֻבּוֹ הֶתֵּר, מֻתָּר. וְאִם לֹא נוֹדָע שֶׁהָיָה רֻבּוֹ הֶתֵּר, אָסוּר. הגה: וּלְעִנְיַן מִין בְּמִינוֹ אַזְלִינָן בָּתַר (ז) שְׁמָא, אִם הוּא שָׁוֶה הָוֵי מִין בְּמִינוֹ. אֲבָל לֹא אַזְלִינָן בָּתַר טַעֲמָא אִם הוּא שָׁוֶה אוֹ לֹא (בֵּית יוֹסֵף בְּשֵׁם הָאָגוּר וְכֵן הוּא בְּהַגָּהוֹת ש''ד סי' ל''ט) . אֲבָל אִם (ח) נִתְעָרֵב בְּשֶׁאֵינוֹ מִינוֹ וְנִשְׁפָּךְ בְּעִנְיָן שֶׁאֵין יְכוֹלִין לַעֲמֹד עָלָיו לְשַׁעֲרוֹ, אֲפִלּוּ נוֹדָע שֶׁהָיָה רֻבּוֹ הֶתֵּר, אָסוּר. וְאִם נִתְעָרֵב בְּמִינוֹ וּבְשֶאֵינוֹ מִינוֹ וְנִשְׁפָּךְ בְּעִנְיָן שֶׁאֵין יְכוֹלִין לַעֲמֹד עָלָיו לְשַׁעֲרוֹ, וְנוֹדַע שֶׁהָיָה רֻבּוֹ הֶתֵּר מִמִּינוֹ, רוֹאִין אֶת שֶׁאֵינוֹ מִינוֹ כְּאִלּוּ אֵינוֹ, וְהַשְּׁאָר, מִינוֹ רָבֶה עָלָיו (ט) וּמְבַטְּלוֹ.


ג בַּמֶּה דְּבָרִים אֲמוּרִים, כְּשֶׁנִּשְׁפַּךְ. אֲבָל (י) אִסוּר שֶׁנִּתְעָרֵב בְּהֶתֵּר, וְהוּא לְפָנֵינוּ, וְאִי אֶפְשָׁר לַעֲמֹד עַל שִׁעוּרוֹ, אַף עַל פִּי שֶׁהוּא מֵאִסוּרִים שֶׁל דִּבְרֵיהֶם, אָסוּר.


ד אִסוּר שֶׁנִּתְבַּשֵׁל עִם הֶתֵּר, אֲפִלּוּ מַכִּירוֹ וְהוּא שָׁלֵם וּזְרָקוֹ, צָרִיךְ שִׁשִּׁים כְּנֶגֶד כָּל הָאִסוּר, מִפְּנֵי שֶׁאֵין אָנוּ יוֹדְעִים כַּמָּה יָצָא מִמֶּנּוּ. לְפִיכָךְ, הַמְּבַשֵּׁל בִּקְדֵרַת אִסוּר שֶׁהִיא בַּת יוֹמָא, אוֹ תּוֹחֵב כַּף שֶׁל אִסוּר בְּהֶתֵּר, צָרִיךְ ס' כְּנֶגֶד כָּל הַקְּדֵרָה וּכְנֶגֶד כָּל מַה שֶּׁתָּחַב מֵהַכַּף, שֶׁאֵין אָנוּ יוֹדְעִין כַּמָּה בָּלְעוּ, בֵּין שֶׁהֵם שֶׁל חֶרֶס אוֹ עֵץ אוֹ מַתֶּכֶת. הגה: וּבִלְבַד שֶׁבָּלְעוּ עַל יְדֵי רְתִיחַת אֵשׁ, שֶׁאָז הַבְּלִיעָה הוֹלֶכֶת (יא) בְּכָל הַכְּלִי, אֲבָל עַל יְדֵי רְתִיחַת מְלִיחָה אֵינוֹ נִבְלָע בַּכְּלִי רַק כְּדֵי קְלִפָּה, וְאֵין צָרִיךְ לְשַׁעֵר רַק כְּדֵי קְלִפָּה (מָרְדְּכַי פכ''ה וּבְאָרֹךְ כְּלָל כ''ד) . וְעַיֵּן לְעֵיל סִימָן ס''ט. וְיֵשׁ מִי שֶׁמַּחְמִיר בְּכַף שֶׁל מַתֶּכֶת לְהַצְרִיךְ ס' כְּנֶגֶד כֻּלּוֹ, אֲפִלּוּ לֹא הִכְנִיס אֶלָּא מִקְצָת, מִשּׁוּם דְּחַם מִקְּצָתוֹ חַם כֻּלּוֹ. הגה: וְנוֹהֲגִין כִּסְבָרָא רִאשׁוֹנָה. וְכָל אִסּוּר שֶׁמְּבַטְּלִים בְּשִׁשִּׁים, אִם מַכִּירוֹ צָרִיךְ לְהָסִירוֹ מִשָּׁם אַף עַל גַּב דִּכְבָר נִתְבַּטֵּל טַעֲמוֹ בְּשִׁשִּׁים. וְלָכֵן אִם נָפַל חֵלֶב לְתוֹךְ הַתַּבְשִׁיל וְנִתְבַּטֵּל טַעֲמוֹ בְּשִׁשִּׁים, צָרִיךְ לִתֵּן שָׁם מַיִם צוֹנְנִים, וְטֶבַע הַחֵלֶב לְהַקְפִּיא וְלָצוּף לְמַעְלָה עַל הַמַּיִם, (יב) וִיסִירֶנּוּ מִשָּׁם, דְּמֵאַחַר דְּאֶפְשָׁר לְהָסִירוֹ הֲוֵי כְּאִלּוּ מַכִּירוֹ וְצָרִיךְ לְהָסִירוֹ מִשָּׁם (הַגָּהוֹת מָרְדְּכַי בְּחֻלִּין וּבְאָרֹךְ כְּלָל כ''ג) . אִסּוּר שֶׁנִּתְבַּטֵּל בִּקְדֵרָה וֶהֱסִירוּהוּ מִשָּׁם, וְנָפַל לִקְדֵרָה אַחֶרֶת, צָרִיךְ לַחֲזֹר וּלְבַטְּלוֹ בְּשִׁשִּׁים נֶגֶד כֻּלּוֹ, וְכֵן לְעוֹלָם. אֲבָל אִם נָפַל לַקְּדֵרָה הָרִאשׁוֹנָה ב' פְּעָמִים, אֵין צָרִיךְ רַק שִׁשִּׁים (פַּעַם) אַחַת כְּנֶגְדּוֹ (בְּאָרֹךְ כְּלָל כ''ד דִּין ב') וְעַיֵּן לְעֵיל סִימָן צ''ד.


ה אִם יָדוּעַ כַּמָּה (יג) הוּא הָאִסוּר, כְּגוֹן כַּף חֲדָשָׁה אוֹ שֶׁאֵינָהּ בַּת יוֹמָא שֶׁנִּעֵר בָּהּ וּבָלְעָה (יד) כַּזַּיִת חָלָב, וְאַחַר כָּךְ נִעֵר בָּהּ קְדֵרָה שֶׁל בָּשָׂר, אֵין צָרִיךְ אֶלָּא ס' לְבַטֵּל הַכַּזַּיִת שֶׁבָּלְעָה. וְלֹא אָמְרִינָן גַּבֵּי כְּלִי חֲתִיכָה נַעֲשֵׂית נְבֵלָה, אֲפִלּוּ אִם נִעֲרוּ בּוֹ אִסּוּר אֲבָל כַּף יְשָׁנָה וּבַת יוֹמָא, מְשַׁעֲרִין בְּכֻלָּהּ דְּכָל מַה שֶּׁבָּלַע נַעֲשֵׂית אִסּוּר, וְלֹא יַדְעִינָן כַּמָּה בָּלַע. (שָׁם) . וְיֵשׁ מִי שֶׁאוֹמֵר שֶׁגַּם בְּזוֹ אֵין צָרִיךְ אֶלָּא ס' לְבַטֵּל הַכַּזַּיִת שֶׁבָּלַע. הגה: וְהַסְּבָרָא רִאשׁוֹנָה עִקָּר, כְּמוֹ שֶנִּתְבָּאֵר לְעֵיל גַּבֵּי טִפַּת חָלָב שֶׁנָּפְלָה עַל הַקְּדֵרָה. וְיֵשׁ שֶׁאֵינָן מְחַלְּקִין בֵּין כַּף יָשָׁן לְחָדָשׁ, רַק בֵּין כְּלִי (טו) חֶרֶס לִשְׁאָר כֵּלִים (מָרְדְּכַי פ' גִּיד הַנָּשֶׁה) וְאוֹמְרִים דְּבִכְלֵי חֶרֶס דְּאִי אֶפְשָׁר לְהַפְרִיד הָאִסּוּר עַל יְדֵי הַגְעָלָה אַמְרִינָן הַכְּלִי נַעֲשֶׂה נְבֵלָה, אֲבָל לֹא בִּשְׁאָר כֵּלִים, וְטוֹב לָחוּשׁ לְחֻמְרָא וְעַיִּןִ לְעֵיל סִימָן צ''ב.


ו כַּחֲצִי זַיִת שֶׁל (טז) אִסוּר שֶׁנִּתְעָרֵב בְּהֶתֵּר, צָרִיךְ שִׁשִּׁים חֲצָאֵי זֵיתֵי הֶתֵּר לְבַטְּלוֹ.


ז בֵּיצָה שֶׁיֵּשׁ בָּהּ אֶפְרוֹחַ אוֹ טִפַּת דָּם שֶׁנִּתִבַּשְּׁלָה עִם אֲחֵרוֹת, צָרִיךְ שִׁשִּׁים וְאַחַת לְבַטֵּל (יז) פְּלִיטָתָהּ.


ח כְּחָל מִתְבַּטֵּל בְּתִשְׁעָה וַחֲמִשִּׁים. הגה: כָּל (יח) הָאִסּוּרִים הַנּוֹהֲגִין בַּזְּמַן הַזֶּה כֻּלָּם מִתְבַּטְּלִים בְּשִׁשִּׁים, מִלְּבַד חָמֵץ בְּפֶסַח וְיַיִן נֶסֶךְ, כַּאֲשֶׁר נִתְבָּאֵר בְּהִלְכוֹתֵיהֶם (טוּר), וּבִלְבַד שֶׁהָאִסּוּר אֵינוֹ נוֹתֵן טַעַם בַּקְּדֵרָה, אֲבָל אִם נוֹתֵן טַעַם בְּאוֹתָהּ קְדֵרָה, וְהוּא אָסוּר מִצַּד עַצְמוֹ, אֲפִלּוּ בְּאֶלֶף לֹא בָּטֵל כָּל זְמַן שֶׁמַּרְגִּישִׁין טַעֲמוֹ. וְלָכֵן מֶלַח וְתַבְלִין מִדְּבָרִים דַּעֲבִידֵי לְטַעֲמָא, אִם אֲסוּרִים מֵחֲמַת (יט) עַצְמָן אֵינָן בְּטֵלִים בְּשִׁשִּׁים בְּאָרֹךְ כְּלָל כ''ה וְע''ל סוֹף סִימָן ק''ה.


ט קְדֵרָה שֶׁיֵּשׁ בָּהּ נ''ט זֵיתִים הֶתֵּר, וְנָפְלוּ בָּהּ שְׁתֵּי זֵיתִים, אֶחָד שֶׁל דָּם וְאֶחָד שֶׁל חֵלֶב, כָּל אֶחָד (כ) מִצְטָרֵף עִם הַנ''ט שֶׁל הֶתֵּר לְבַטֵּל חֲבֵרוֹ. וְכֵן כ''ט זִיתִים שֶׁל הֶתֵּר שֶׁנָּפַל בָּהֶם כַּזַּיִת חֵלֶב, וּבִקִדֵרָה אַחֶרֶת הָיוּ שְׁלֹשִׁים שֶׁל הֶתֵּר וְנָפַל לְתוֹכוֹ כַּזַּיִת שֶׁל דָּם, (כאֹ) וְנִתְעָרְבוּ בְּשׁוֹגֵג, מֻתָּר. וְכָל שֶׁכֵּן בְּב' זֵיתִים, אֶחָד שֶׁל גְּבִינָה וְאֶחָד שֶׁל בָּשָׂר, דְּכָל אֶחָד (כב) מְבַטֵּל חֲבֵרוֹ (בְּאָרֹךְ כְּלָל כ''ד) .



סימן צט - דין העצמות אם מצטרפין לבטל אסור, ושלא לבטל אסור לכתחלה, ובו ז' סעיפים


א חֲתִיכַת נְבֵלָה שֶׁיֵּשׁ בָּהּ (א) בָּשָׂר וַעֲצָמוֹת שֶׁנָּפְלָה לִקְדֵרַת הֶתֵּר, עַצְמוֹת הָאִסוּר מִצְטָרְפִים עִם הַהֶתֵּר לְבַטֵּל הָאִסוּר, וְאֵין צָרִיךְ לוֹמַר שֶׁעַצְמוֹת הַהֶתֵּר מִצְטָרְפִין עִם הַהֶתֵּר, אֲבָל הַמֹּחַ שֶׁבְּעַצְמוֹת אִסוּר מִצְטָרֵף עִם הָאִסוּר. (ב) וְגוּף הַקְּדֵרָה אֵינָהּ מִצְטָרֶפֶת, לֹא עִם הָאִסוּר וְלֹא עִם הַהֶתֵּר. הגה: וְיֵשׁ (ג) מַחְמִירִים שֶׁלֹּא לְצָרֵף עַצְמוֹת הָאִסּוּר עִם הַהֶתֵּר לְבַטֵּל (הַגָּהָה אַחַת בְּש''ד בְּשֵׁם אוֹר זָרוּעַ) וּבִמְקוֹם הֶפְסֵד יֵשׁ לִסְמֹךְ אַמְּקִלִּין וּמַתִּירִין, כִּי כֵן עִקָּר.


ב בַּמֶּה דְּבָרִים אֲמוּרִים שֶׁעַצְמוֹת הָאִסוּר מִצְטָרְפִין עִם הַהֶתֵּר, כְּשֶׁנָּפְלָה חֲתִיכַת הַנְּבֵלָה לִקְדֵרַת הֶתֵּר כְּשֶׁהָיְתָה חַיָּה, אֲבָל אִם נִתְבַּשְּׁלָה (תְּחִלָּה) וְאַחַר כָּךְ נָפְלָה לִקְדֵרַת הֶתֵּר, עֲצָמוֹת שֶׁבָּהּ מִצְטָרְפִים עִם הָאִסוּר, (ד) לְפִי שֶׁבָּלְעוּ מִבְּשָׂר הַנְּבֵלָה כְּשֶׁנִּתְבַּשְּׁלָה לְבַדָּהּ.


ג בִּשְׁאָר אִסוּרִים, חוּץ מִבָּשָׂר בְּחָלָב, חֲתִיכָה (ה) הַבְּלוּעָה מֵאִסוּר מִצְטָרֶפֶת לְבַטֵּל הָאִסוּר. הגה: וְאֵין נוֹהֲגִין כֵּן, כִּי קַיְמָא לָן בְּכָל אִסּוּרִים חֲתִיכָה נַעֲשֵׂית נְבֵלָה, כְּדִלְעֵיל סִימָן צ''ב.


ד מְשַׁעֲרִים בָּרֹטֶב וּבַקִּיפָה (פֵּרוּשׁ הַדַּק דַּק שֶׁל בָּשָׂר וְתַבְלִין הַמִּתְאַסֵף בְּשׁוּלֵי קְדֵרָה) וּבַחֲתִיכוֹת. וּמְשַׁעֲרִין הַהֶתֵּר וְהָאִסוּר כְּמוֹ שֶׁבָּא לְפָנֵינוּ, אַף עַל פִּי שֶׁהָיָה בַּהֶתֵּר יוֹתֵר מִתְּחִלָּה (ו) וְנִתְמַעֵט בְּבִשּׁוּלוֹ וְנִבְלַע בַּקְּדֵרָה. וְהָנֵי מִלֵּי בְּמִין בְּשֶׁאֵינוֹ מִינוֹ, אֲבָל אִם הוּא מִין בְּמִינוֹ מְשַׁעֲרִין גַּם בַּמֶּה (ז) שֶּׁבָּלְעָה הַקְּדֵרָה, וְעוֹמֵד בְּדָפְנֵי הַקְּדֵרָה. וּמְשַׁעֲרִים זֶה בְּאֹמֶד יָפֶה, וְרוֹאִין אוֹתוֹ כְּאִלּוּ הוּא בְּעַיִן, אֲבָל מַה שֶּׁכָּלָה וְאָבַד מֵחֲמַת הָאוּר אֵינוֹ מִצְטָרֵף, שֶׁזֶּה כָּלָה לְגַמְרֵי.


ה אֵין מְבַטְּלִין אִסוּר (ח) לְכַתְּחִלָּה. וַאֲפִלּוּ נָפַל לְתוֹךְ הֵתֶּר שֶׁאֵין בּוֹ שִׁעוּר לְבַטְּלוֹ, אֵין מוֹסִיפִין עָלָיו הֶתֵּר כְּדֵי לְבַטְּלוֹ. (ט) עָבַר וּבִטְּלוֹ, אוֹ שֶׁרִבָּה עָלָיו, אִם בְּשׁוֹגֵג, מֻתָּר. וְאִם בְּמֵזִיד, אָסוּר לַמְבַטֵּל עַצְמוֹ, אִם הוּא שֶׁלּוֹ, וְכֵן לְמִי שֶׁנִּתְבַּטֵּל (י) בִּשְׁבִילוֹ. (יא) וַאֲסוּרִים לְמָכְרוֹ גַּם כֵּן לְיִשְׂרָאֵל אַחֵר, שֶׁלֹּא יֵהָנוּ מִמַּה שֶּׁבִּטְּלוּ (אָרֹךְ כְּלָל כ''ד) . וְלִשְׁאָר כָּל אָדָם, מֻתָּר. הגה: וְדַוְקָא שֶׁנִּתְעָרֵב יָבֵשׁ בְּיָבֵשׁ, אוֹ אֲפִלּוּ לַח בְּלַח לְמָאן דְּאָמַר שֶׁאֵין אוֹמְרִים בּוֹ חֲתִיכָה נַעֲשֵׂית (יב) נְבֵלָה, כְּדִלְעֵיל סי' צ''ב. אֲבָל חֲתִיכָה שֶׁבָּלְעָה אִסּוּר, לֹא מְהַנֵּי שֶׁנִּתּוֹסֵף אַחַר כָּךְ הַהֶתֵּר, דְּהָא אָמְרִינָן בֵּהּ חֲתִיכָה נַעֲשֵׂית נְבֵלָה. (רַשְׁבָּ''א סי' תצ''ה עי' ס''ק י''ד) . וְיֵשׁ אוֹמְרִים דַּאֲפִלּוּ בְּמָקוֹם דְּלָא אַמְרִינָן חֲתִיכָה נַעֲשֵׂית נְבֵלָה לֹא מְהַנֵּי הַהֶתֵּר לְבַטֵּל, אֶלָּא אִם נִתּוֹסֵף קֹדֶם שֶׁנּוֹדַע הַתַּעֲרוֹבוֹת, אֲבָל אִם נוֹדַע הַתַּעֲרוֹבֶת קֹדֶם, לֹא מְהַנֵּי מַה שֶׁנִּתּוֹסֵף אַחַר כָּךְ. וּלְפִי זֶה, הָיָה צָרִיךְ הֶחָכָם הַמּוֹרֶה לְבַטֵּל אִסּוּר לַחֲקֹר אִם נִתּוֹסֵף הַהֶתֵּר לְאַחַר שֶׁנּוֹדַע (אָרֹךְ כ' ל''ז) . וְלֹא (יג) נָהֲגוּ כֵן.


ו אִסוּר שֶׁל דִּבְרֵיהֶם, אֵין מְעָרְבִין אוֹתוֹ בְּיָדַיִם כְּדֵי לְבַטְּלוֹ. וְאִם עָשָׂה כֵן, בְּמֵזִיד, אָסוּר. אֲבָל אִם נָפַל מֵעַצְמוֹ, וְאֵין בַּהֶתֵּר כְּדֵי לְבַטְּלוֹ, (יד) מַרְבֶּה עָלָיו וּמְבַטְּלוֹ. הגה: וְיֵשׁ אוֹמְרִים דְּאֵין לְבַטֵּל אִסּוּר דְּרַבָּנָן אוֹ לְהוֹסִיף עָלָיו, כְּמוֹ בְּאִסּוּר דְּאוֹרַיְיתָא. וְכֵן נוֹהֲגִין, וְאֵין (טו) לְשַׁנּוֹת. (טוּר (טז) בְּשֵׁם הָרֹא''שׁ וּבת''ה סִימָן צ''ד וּבְהַגָּהַת ש''ד וּרא''ה בב''ה ור''ן וּמָרְדְּכַי וַאֲגֻדָּה פ''ק דְּבֵיצָה) . אִסּוּר שֶׁנִּתְבַּטֵּל, כְּגוֹן שֶׁהָיָה ס' כְּנֶגְדּוֹ, וְנִתּוֹסֵף בּוֹ אַחַר כָּךְ מִן הָאִסּוּר הָרִאשׁוֹן, חוֹזֵר וְנֵעוֹר וְנֶאֱסָר, לָא שְׁנָא מִין בְּמִינוֹ לָא שְׁנָא מִין בְּשֶׁאֵינוֹ מִינוֹ, לָא שְׁנָא יָבֵשׁ לָא שְׁנָא לַח, לָא שְׁנָא נוֹדַע בֵּינְתַיִם אוֹ לֹא נוֹדַע בֵּינְתַיִם. (הַגָּהוֹת ש''ד סִימָן נ''ט) . כַּזַּיִת חָלָב שֶׁנָּפַל לַמַּיִם וְנִתְבַּטֵּל בְּס', וְאַחַר כָּךְ נָפַל מִן הַמַּיִם לִקְדֵרָה שֶׁל בָּשָׂר, מֻתָּר, אַף עַל פִּי שֶׁאֵין (יז) בַּבָּשָׂר ס' נֶגֶד הֶחָלָב, שֶׁהֲרֵי נִתְבַּטֵּל בַּמַּיִם (בְּאָרֹךְ) . וְכָל כַּיּוֹצֵא בְּזֶה.


ז אִם נִבְלַע אִסוּר מֻעָט לְתוֹךְ כְּלִי בָּשָׂר, אִם דַּרְכּוֹ שֶׁל אוֹתוֹ כְּלִי (יח) לְהִשְׁתַּמֵּשׁ בּוֹ בְּשֶׁפַע הֶתֵּר, מֻתָּר לְהִשְׁתַּמֵּשׁ בּוֹ לְכַתְּחִלָּה, כֵּיוָן שֶׁהָאִסוּר מֻעָט וְאִי אֶפְשָׁר לָבֹא לִידֵי נְתִינַת טַעַם, וּלְפִיכָךְ אִסוּר מַשֶּׁהוּ שֶׁנִּבְלַע בִּקְדֵרָה אוֹ בְּתוֹךְ קַנְקַנִּים וְכַיּוֹצֵא בָּהֶם, מֻתָּר לְהִשְׁתַּמֵּשׁ בּוֹ לְכַתְּחִלָּה, וַאֲפִלּוּ בְּבֶן יוֹמוֹ, לְפִי שֶׁאִי אֶפְשָׁר לָבֹא לִידֵי נְתִינַת טַעַם. אֲבָל אִם נִבְלַע בִּכְלִי שֶׁדַּרְכּוֹ לְהִשְׁתַּמֵּשׁ לְעִתִּים בְּדָבָר מֻעָט כִּקְעָרָה וְכַיּוֹצֵא בָּהּ, אָסוּר לְהִשְׁתַּמֵּשׁ אֲפִלּוּ בְּשֶׁפַע, גְּזֵרָה שֶׁמָּא יִשְׁתַּמֵּשׁ בָּהּ בְּדָבָר מֻעָט וְיָבֹא לִידֵי נְתִינַת טַעַם.



סימן ק - בריה אפלו באלף לא בטיל, ובו ד' סעיפים


א בְּרִיָּה, דְּהַיְנוּ כְּגוֹן נְמָלָה אוֹ עוֹף טָמֵא, וְגִיד הַנָּשֶׁה, וְאֵבֶר מִן הַחַי, וּבֵיצָה שֶׁיֵּשׁ בָּהּ (א) אֶפְרוֹחַ וְכַיּוֹצֵא בָהֶם, אֲפִלּוּ בְּאֶלֶף לֹא בָּטֵל. וְאֵין לוֹ דִּין בְּרִיָּה, אֶלָּא אִם כֵּן הוּא דָּבָר שֶׁהָיָה בּוֹ חַיּוּת, לַאֲפוּקֵי חִטָּה אַחַת שֶׁל אִסוּר. וְכֵן צָרִיךְ שֶׁיְּהֵא דָּבָר שֶׁאָסוּר מִתְּחִלַּת בְּרִיָּתוֹ, לַאֲפוּקֵי עוֹף טָהוֹר שֶׁנִּתְנַבֵּל וְשׁוֹר הַנִּסְקָל. וְכֵן צָרִיךְ שֶׁיִּהְיֶה דָּבָר שָׁלֵם, שֶׁאִם יֵחָלֵק אֵין שְׁמוֹ עָלָיו, לַאֲפוּקֵי חֵלֶב. וְכֵן צָרִיךְ. שֶׁיִּהְיֶה שָׁלֵם. הגה: וְעִקַּר גִּיד הַנָּשֶׁה אֵינוֹ אֶלָּא עַל הַכַּף בִּלְבַד, וְהוּא כְּרֹחַב ד' (ב) אֶצְבָּעוֹת, וְאִם הוּא שָׁלֵם מִקְרֵי בְּרִיָּה. (בְּאָרֹךְ כְּלָל כ''ח) .


ב דָּבָר שֶׁהוּא בְּרִיָּה שֶׁנִּתְבַּשֵּׁל עִם הַהֶתֵּר, אִם אֵינוֹ מַכִּירוֹ הַכֹּל אָסוּר, וְהָרֹטֶב בְּנוֹתֵן טַעַם. וְאִם מַכִּירוֹ, זוֹרְקוֹ. וְהָאֲחֵרִים וְהָרֹטֶב צָרִיךְ שֶׁיִּהְיֶה בָּהֶם ס' כְּנֶגְדּוֹ, חוּץ מִגִּיד הַנָּשֶׁה שֶׁאֵין טַעֲמוֹ אוֹסֵר, דְּאֵין בַּגִּידִים בְּנוֹתֵן טַעַם. אֲבָל שַׁמְנוֹ, אוֹסֵר, וְצָרִיךְ שִׁשִּׁים כְּנֶגְדּוֹ לְהַתִּיר הָרֹטֶב. לְפִיכָךְ יָרֵךְ שֶׁנִּתְבַּשֵּׁל עִם גִּידוֹ, אִם מַכִּירוֹ זוֹרְקוֹ, וְכָל הַשְּׁאָר מֻתָּר, אִם יֵשׁ בּוֹ ס' כְּנֶגֶד שַׁמְנוֹ. וְאִם אֵינוֹ מַכִּירוֹ, כָּל הַחֲתִיכוֹת אֲסוּרוֹת. וְאִם יֵשׁ בַּקְּדֵרָה ס' כְּנֶגֶד (ג) הַשֻּׁמָּן, מֻתָּר הָרֹטֶב; וְאִם לָאו, אָסוּר. (ד) וְאִם נִמּוֹחַ הַגִּיד וְאֵינוֹ נִכָּר, צָרִיךְ גַּם כֵּן ס' כְּנֶגְדּוֹ.


ג קְדֵרָה שֶׁל מָרָק שֶׁנָּפַל שָׁם בְּרִיָּה וְנֶאֶבְדָה, אָסוּר (ה) הַכֹּל.


ד יְרָקוֹת מְבֻשָּׁלוֹת שֶׁנִּמְצְאוּ בָּהֶם ג' תּוֹלָעִים, הַיְרָקוֹת (ו) אֲסוּרִים; אֲבָל מֵי הַשַּׁלְקוּת, מְסַנְּנָן וּמֻתָּרִים. וְכֵן הַבָּשָׂר, יִרְחֲצֶנּוּ וְיִבְדְּקֶנּוּ וּמֻתָּר.



סימן קא - דין חתיכה הראויה להתכבד, ובו ט' סעיפים


א חֲתִיכָה הָרְאוּיָה לְהִתְכַּבֵּד דִּינָהּ כִּבְרִיָּה, דַּאֲפִילוּ בְּאֶלֶף לֹא בְּטֵלָה. וַאֲפִלּוּ אִם הִיא אֲסוּרָה (א) בַּהֲנָאָה, כֵּיוָן שֶׁאִם תִּתְבַּטֵּל הָיְתָה מֻתֶּרֶת וּרְאוּיָה לְהִתְכַּבֵּד. הגה: וַאֲפִלּוּ אֵינָהּ אֲסוּרָה רַק מִדְּרַבָּנָן, אֵינָהּ בְּטֵלָה (בֵּית יוֹסֵף בְּשֵׁם הָרַמְבַּ''ם) . וְאִם הוּא סָפֵק אִם רְאוּיָה לְהִתְכַּבֵּד אוֹ לֹא, אַזְלִינָן (ב) לְקֻלָּא (אָרֹךְ כְּלָל כ''ה) אֲפִלּוּ הִיא אֲסוּרָה מִדְּאוֹרַיְתָא.


ב אֵין לָהּ דִּין חֲתִיכָה הָרְאוּיָה לְהִתְכַּבֵּד, אֶלָּא אִם כֵּן אִסוּרָהּ מֵחֲמַת עַצְמָהּ, כְּגוֹן נְבֵלָה וּבָשָׂר (ג) בְּחָלָב; אֲבָל אִם נֶאֶסְרָה מֵחֲמַת שֶׁקִּבְּלָה (ד) טַעַם מֵאִסוּר וְלֹא הָיָה בָּהּ ס' לְבַטְּלוֹ, אֲפִלּוּ לְמִי שֶׁסוֹבֵר חֲתִיכָה עַצְמָהּ נַעֲשֵׂית נְבֵלָה אֵין לָהּ דִּין חֲתִיכָה הָרְאוּיָה לְהִתְכַּבֵּד. הגה: וַאֲפִלּוּ חֲתִיכָה שֶׁלֹּא נִמְלְחָה, בְּטֵלָה, דְּאֵין אִסּוּרָהּ מֵחֲמַת עַצְמָהּ רַק מֵחֲמַת דָּם הַבָּלוּעַ בָּהּ (שָׁם וְת''ה סי' ק''ע) . כָּל חֲתִיכָה שֶׁלֹּא נֶאֶסְרָה רַק כְּדֵי (ה) קְלִפָּה, בְּטֵלָה, דַּהֲרֵי הַקְּלִפָּה אֵינָהּ רְאוּיָה לְהִתְכַּבֵּד. (סְבָרַת הַתּוֹסָפוֹת וּבְהַגָּהוֹת ש''ד) .


ג תַּרְנְגֹלֶת בְּנוֹצָתָהּ שֶׁנִּתְעָרְבָה בַּאֲחֵרוֹת, בְּטֵלָה, שֶׁהֲרֵי אֵינָהּ רְאוּיָה (ו) לְהִתְכַּבֵּד לִפְנֵי הָאוֹרְחִים כְּמוֹת שֶׁהִיא. וְאַף עַל פִּי שֶׁאַחַר שֶׁנִּתְעָרְבָה (ז) הֵסִירוּ הַנּוֹצָה. וְכֵן לֹא חֲשִׁיבָה רְאוּיָה לְהִתְכַּבֵּד אֶלָּא אִם כֵּן הִיא מְבֻשֶּׁלֶת. וְכֵן כֶּבֶשׂ (ח) שָׁלֵם אוֹ חֲתִיכָה גְּדוֹלָה יוֹתֵר מִדַּאי, לֹא חֲשִׁיבָה רְאוּיָה לְהִתְכַּבֵּד, שֶׁאֵין דֶּרֶךְ לִתֵּן לִפְנֵי הָאוֹרֵחַ כֶּבֶשׂ שָׁלֵם אוֹ חֲתִיכָה גְּדוֹלָה יוֹתֵר מִדַּאי. וְיֵשׁ חוֹלְקִים בְּכָל זֶה. הגה: וְכֵן נוֹהֲגִין, מִלְּבַד תַּרְנְגֹלֶת (ט) בְּנוֹצָתָהּ דְּמְחֻסֶּרֶת מַעֲשֶׂה גָּדוֹל, דְּנוֹהֲגִים בָּהּ שֶׁהִיא בְּטֵלָה (טוּר בְּשֵׁם י''א וּבְמָרְדֳּכַי וּבְש''ד) . וְכֵן רַגְלַיִם אוֹ רֹאשׁ שֶׁלֹּא נֶחְרְכוּ מְשַׁעֲרָן. אֲבָל אִם כְּבָר נֶחְרְכוּ, חֲשִׁיבֵי חֲתִיכָה הָרְאוּיָה לְהִתְכַּבֵּד, אֲפִלּוּ לֹא נִמְלְחוּ עֲדַיִן. וַאֲפִלּוּ כֶּבֶשׂ שָׁלֵם אֵינוֹ בָּטֵל (בהַגָּהוֹת ש''ד) . מִיהוּ רַגְלַיִם שֶׁל עוֹפוֹת, מִלְּבַד רַגְלֵי (י) אַוָּז, וְכֵן רַגְלֵי בְּהֵמָה דַּקָּה בִּמְדִינוֹת אֵלּוּ, לֹא חֲשִׁיבֵי כְּלָל, וּבְטֵלִים. רֹאשׁ שֶׁל עוֹפוֹת אֵינוֹ חָשׁוּב, וּבָטֵּל (אָרֹךְ כְּלָל כ''ה) .


ד שֻׁמַּן (יא) הַכַּנְתָּא אֵינוֹ רָאוּי לְהִתְכַּבֵּד. אֲבָל עוֹר (יב) שֻׁמַּן אַוָּז מִקְרֵי רָאוּי לְהִתְכַּבֵּד. וְאֵינוֹ בָּטֵל (שָׁם) .


ה קֻרְקְבָן וְכֵן (יג) שְׁאָר בְּנֵי מֵעַיִם, אֵינָם רְאוּיִים לְהִתְכַּבֵּד.


ו הָא דִּבְרִיָּה וַחֲתִיכָה הָרְאוּיָה לְהִתְכַּבֵּד לֹא בָּטְלֵי, דַּוְקָא בְּעוֹדָם שְׁלֵמִים, אֲבָל אִם נֶחְתְּכוּ (יד) אוֹ נִתְרַסְקוּ עַד שֶׁנֶּאֶבְדָה צוּרָתָן, בָּטְלֵי, דְּתוּ לֹא חֲשִיבֵי. וַאֲפִלּוּ אִם נִתְרַסְקוּ לְאַחַר שֶׁנִּתְעָרְבוּ עִם הַהֶתֵּר. וַאֲפִלּוּ לְאַחַר שֶׁנּוֹדַע הַתַּעֲרֹבֶת (הַגָּהַת ש''ד וְד''ע לַאֲפוּקֵי או''ה) . וְהוּא שֶׁנִּתְרַסְקוּ שֶׁלֹּא בְּמִתְכַּוֵּן לְבַטֵּל הָאִסוּר, אֲבָל אִם נִתְכַּוֵּן לְכָךְ, אָסוּר לַמְרַסֵק, אִם הוּא שֶׁלּוֹ, וְכֵן לְמִי שֶׁנִּתְרַסֵק בִּשְׁבִילוֹ. וְאֵין חִלּוּק בֵּין נִתְעָרְבוּ (טו) בְּמִינָם אוֹ שֶׁלֹּא בְּמִינָם (ד''ע וּמַשְׁמָעוּת הַפּוֹסְקִים וְלַאֲפוּקֵי או''ה) .


ז אִם נֶחְתְּכָה חֲתִיכָה אַחַת, אֵין תּוֹלִין לוֹמַר שֶׁל אִסוּר נֶחְתַּךְ, וּמִתְבַּטֵּל, וְיִהְיוּ כֻּלָּן (טז) מֻתָּרוֹת, אֶלָּא אוֹתָהּ שֶׁנֶּחְתְּכָה בִּלְבַד מֻתֶּרֶת מִמַּה נַּפְשָׁךְ, אִם הִיא שֶׁל אִסוּר, אַף כֻּלָּן מֻתָּרוֹת. וְאִם אֵינָהּ שֶׁל אִסוּר, הֲרֵי הִיא מֻתֶּרֶת. וַאֲפִלּוּ נֶחְתְּכוּ רֻבָּן אֵין (יז) תּוֹלִין לוֹמַר שֶׁהָאִסוּר מֵהָרֹב שֶׁנֶּחְתְּכוּ, אֶלָּא כָּל הַנֶּחְתָּכוֹת מֻתָּרוֹת וְהַשְּׁלֵמוֹת אֲסוּרוֹת.


ח קֻרְקְבָן שֶׁנִּמְצָא נָקוּב, וְנִתְעָרְבָה אוֹתָהּ תַּרְנְגֹלֶת עִם אֲחֵרוֹת, מְדַמִּין שֻׁמָּן שֶׁבַּקֻּרְקְבָן לְשֻׁמַּן הַתַּרְנְגֹלֶת שֶׁל מְקוֹם חִבּוּר הַקֻּרְקְבָן, וְאִם דּוֹמִים לְגַמְרֵי, מַכְשִׁירִים הָאֲחֵרוֹת. וְכֵן כָּל כַּיּוֹצֵא בָּזֶה.


ט רֹאשׁ כֶּבֶשׂ שֶׁנִּמְצָא טְרֵפָה, וְלֹא נוֹדַע מֵאֵיזֶה כֶּבֶשׂ הוּא, וְהִקִּיפוּ הָרֹאשׁ לְצַוָּארוֹ שֶׁל אֶחָד מֵהַכְּבָשִׂים וְנִמְצְאוּ (יח) הַחֲתִיכוֹת דּוֹמוֹת וּמְכֻוָּנוֹת יָפֶה, יֵשׁ לִסְמֹךְ עַל זֶה לְהַתִּיר הָאֲחֵרוֹת.



סימן קב - דין דבר שיש לו מתירין, ובו ד' סעיפים


א כָּל דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין, כְּגוֹן בֵּיצָה שֶׁנּוֹלְדָה בְּיוֹם טוֹב, שֶׁרְאוּיָה לְמָחָר, אִם (א) נִתְעָרְבָה בַּאֲחֵרוֹת, בֵּין שְׁלֵמָה בֵּין טְרוּפָה, אֵינָהּ בְּטֵלָה אֲפִלּוּ בְּאֶלֶף. וַאֲפִלּוּ סָפֵק אִם נוֹלְדָה בְּיוֹם טוֹב, וְנִתְעָרְבָה בַּאֲחֵרוֹת, אֲסוּרוֹת. וְאִם נִתְעָרְבָה (ב) בְּשֶׁאֵינָהּ מִינָהּ, בְּטֵלָה בס'. (טוּר וְש''ד ותשו' הָרֹא''שׁ וְר' יְרוּחָם וְהַגָּהוֹת ש''ד וְרש''ל וְרוֹקֵחַ וְכנ''ד סְמַ''ג וּסְמַ''ק בהל' י''ט וְכֵן כָּתַב או''ה כְּלָל כ''ו בְּשֵׁם סְמַ''ג וְת''ח כְּלָל מ''ג וְלֹא כְּבַ''ח וּסְמַ''ק סִימָן רי''ד וְכֵן מַשְׁמָע בַּאֲגֻדָּה וּמָרְדְּכַי רֵישׁ בֵּיצָה בְּשֵׁם ראבי''ה וְכֵן כָּתַב טא''ח סִימָן תקי''ג וְהָאָגוּר בִּשְׁמוֹ וְהאו''ה כְּלָל כ''ה בְּשֵׁם מָרְדְּכַי רֵישׁ בֵּיצָה בְּשֵׁם ראבי''ה וְלֹא כב''ח) . הגה: מִיהוּ אִם (ג) לִבְּנוּ בָּהּ מַאֲכָל, אוֹ נָתְנוּ בַּקְּדֵרָה לְתַקֵּן הַקְּדֵרָה, כְּגוֹן שֶׁמִּלְאוּהָ בְּתַרְנְגֹלֶת, אֵינָהּ בְּטֵלָה (הַגָּהַת ש''ד וּבְאָרֹךְ) וְעַיֵּן בא''ח סִימָן תקי''ג.


ב יֵשׁ מִי שֶׁאוֹמֵר שֶׁלֹּא אָמְרוּ דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין אֶלָּא כְּשֶׁהַמַּתִּיר (ד) עָתִיד לָבוֹא עַל כָּל פָּנִים, אוֹ אִם הַמַּתִּיר בְּיָדוֹ לַעֲשׂוֹתוֹ בְּלֹא הֶפְסֵד, אֲבָל דָּבָר שֶׁאֵינוֹ בְּיָדוֹ, וְאֵינוֹ וַדַּאי שֶׁבָּא הַמַּתִּיר, אֵינוֹ בְּדִין דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין. לְפִיכָךְ, בֵּיצָה שֶׁל (ה) סְפֵק טְרֵפָה שֶׁנִּתְעָרְבָה בַּאֲחֵרוֹת, אֵינוֹ בְּדִין דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין, לְפִי שֶׁאֵין הַמַּתִּיר בְּוַדַּאי וְאֵינוֹ בְּיָדוֹ.


ג כְּלִי שֶׁנֶּאֱסַר בִּבְלִיעַת אִסוּר, שֶׁנִּתְעָרֵב בַּאֲחֵרִים וְאֵינוֹ נִכָּר, בָּטֵל בְּרֹב וְאֵין דָּנִין אוֹתוֹ כְּדָבָר שֶׁיֵּשׁ לוֹ מַתִּירִין לְפִי שֶׁצָּרִיךְ לְהוֹצִיא עָלָיו (ו) הוֹצָאוֹת לְהַגְעִילוֹ, וְכָל כַּיּוֹצֵא בָּזֶה.


ד יֵשׁ מִי שֶׁאוֹמֵר דְּלֹא שַׁיָּךְ דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין הֵיכָא שֶׁהַמַּאֲכָל מִתְקַלְקֵל. הגה: הָא דְּדָבָר שֶׁיֵּשׁ לוֹ מַתִּירִין אֵינוֹ בָּטֵל, הַיְנוּ דַּוְקָא אִם הָאִסּוּר בְּעַיִן אוֹ שֶׁיֵּשׁ עֲדַיִן מַמָּשׁוּת הָאִסּוּר בַּתַּעֲרֹבֶת, אֲבָל (ז) טַעֲמוֹ, בָּטֵל (טוּר א''ח סי' תקי''ג וּבְאָרֹךְ וְהג''א סוֹף עֲבוֹדַת כּוֹכָבִים וּמהרא''י בְּשֵׁם מהרי''ח) . וְכֵן אִם אֵין אִסּוּרוֹ מֵחֲמַת (ח) עַצְמוֹ, בָּטֵל. וְלָכֵן חֲתִיכָה שֶׁלֹּא נִמְלְחָה תּוֹךְ ג' יָמִים, אַף עַל פִּי (ט) שֶׁיֵּשׁ אוֹמְרִים דְּמִקְרֵי דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין הוֹאִיל וּמֻתֶּרֶת לְצָלִי, אֲפִלּוּ הָכֵי בְּטֵלָה, דְּאֵין אִסּוּרָהּ אֶלָּא מֵחֲמַת דָּם (י) הַבָּלוּעַ בָּהּ (תּוֹרַת הַבַּיִת סִימָן ק''ע) . כָּל אִסּוּר שֶׁלֹּא הָיָה נִכָּר (יא) קֹדֶם שֶׁנִּתְעָרֵב, הָוֵי בָּטֵל אַף עַל פִּי שֶׁהוּא דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין (בְּאָרֹךְ כְּלָל כ''ה בְּשֵׁם סְמַ''ג וּמָרְדְּכַי פ''ק דְּשַׁבָּת) . מִי שֶׁנָּדַר מִדָּבָר אֶחָד וְנִתְעָרֵב אַחַר כָּךְ, לְדִידֵיהּ מִקְרֵי דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין (שָׁם וּבְמַיְמוֹנִי פ''ח דִּנְדָרִים ובר''ן פ''ק וְאו''ה וְהוּא ש''ס עָרוּךְ פ''ז דִּנְדָרִים דַּף נ''ט וִירוּשַׁלְמִי פ''ו דִּנְדָרִים וּמֻסְכָּם מִכָּל הַפּוֹסְקִים), דְּהָא אֶפְשָׁר לִשְׁאֹל עַל נִדְרוֹ. דָּבָר שֶׁיֵּשׁ לוֹ הֶתֵּר וְחוֹזֵר וְנֶאֱסָר, כְּגוֹן חָמֵץ בְּפֶסַח, לֹא מִקְרֵי דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין (מָרְדְּכַי פֶּרֶק כ''ש) . וְיֵשׁ (יב) חוֹלְקִין בְּזֶה. (רַמְבַּ''ם פט''ו מהמ''א) וְלֹא מִקְרֵי דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין אֶלָּא אִם כֵּן הֻתַּר לְמִי שֶׁנֶּאֱסָר, אֲבָל אִם נִשְׁאָר לְאֶחָד לְעוֹלָם אָסוּר, אַף עַל גַּב שֶׁמֻּתָּר לַאֲחֵרִים כְּגוֹן (יג) הַמְבַשֵּׁל בְּשַׁבָּת, לֹא מִקְרֵי דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין (בֵּית יוֹסֵף בְּשֵׁם רַבֵּנוּ יְרוּחָם בְּשֵׁם הַרמ''ה) .



סימן קג - דין נותן טעם לפגם, ובו ז' סעיפים


א כָּל דָּבָר שֶׁטַּעֲמוֹ פָּגוּם, אֵינוֹ אוֹסֵר תַּעֲרָבְתּוֹ. וַאֲפִלּוּ אֵין טַעֲמוֹ פָּגוּם מֵחֲמַת עַצְמוֹ, שֶׁבִּפְנֵי עַצְמוֹ הוּא מֻטְעָם וּמְשֻׁבָּח, אֶלָּא שֶׁפּוֹגֵם תַּעֲרָבְתּוֹ, מֻתָּר. הגה: מִיהוּ דְּבָרִים הַחֲשׁוּבִים כִּבְרִיָּה (א) אוֹ כַּיּוֹצֵא בָהּ, אִם אֵינָן פְּגוּמִים בְּעַצְמָן אַף עַל פִּי שֶׁפּוֹגְמִין הַתַּבְשִׁיל, אֵינָן (ב) בְּטֵלִים אֲפִלּוּ בְּאֶלֶף (וְכֵן כָּתַב הָאָרֹךְ כְּלָל ל''ב) .


ב פְּגָם זֶה אֵין צָרִיךְ שֶׁיִּפְגֹּם לְגַמְרֵי עַד שֶׁיְּהֵא קָץ לְאָכְלוֹ, אֶלָּא אֲפִלּוּ פּוֹגֵם (ג) קְצָת אֵינוֹ אוֹסֵר תַּעֲרָבְתּוֹ. וְיֵשׁ מִי שֶׁאוֹמֵר, דְּהַיְנוּ דַּוְקָ א כְּשֶׁנִּתְעָרֵב אִסוּר (ד) מֻעָט עִם הֶתֵּר (ה) מְרֻבֶּה, אֲבָל אִסוּר מְרֻבֶּה לְתוֹךְ הֶתֵּר מֻעָט, וַאֲפִלּוּ מֶחֱצָה עַל מֶחֱצָה, אֵין אוֹמְרִים נוֹתֵן טַעַם לִפְגָם מֻתָּר, עַד שֶׁיִּפְגֹּם לְגַמְרֵי שֶׁאֵינוֹ רָאוּי לְמַאֲכַל אָדָם. וְאִם אֵין שָׁם מַמָּשׁוּת שֶׁל אִסוּר, אֶלָּא טַעֲמוֹ בִּלְבַד, אֲפִלּוּ אִסוּר מְרֻבֶּה וְהֶתֵּר מֻעָט, מֻתָּר אִם פּוֹגֵם קְצָת. וְיֵשׁ מִי שֶׁחוֹכֵךְ (פי' מְקַוֶּה לְהַחְמִיר, וְאוֹסֵר) לוֹמַר שֶׁאִם הִגְדִּיל הָאִסוּר מִדָּתוֹ שֶׁל הֶתֵּר עַד שֶׁהוּא מַשְׁבִּיחַ יוֹתֵר כְּשֶׁאָכְלוֹ בְּגֹדֶל מִדָּתוֹ מִמַּה שֶּׁהוּא פּוֹגֵם בְּהֶפְסֵד טַעֲמוֹ, אָסוּר, עַד שֶׁיִּפָּסֵל מִלֶּאֱכֹל אָדָם. (ו) בַּמֶּה דְּבָרִים אֲמוּרִים, שֶׁפּוֹגֵם מִתְּחִלָּתוֹ וְעַד סוֹפוֹ, אֲבָל אִם (ז) הִשְבִּיחַ וּלְבַסוֹף פּוֹגֵם, אוֹ פּוֹגֵם וּלְבַסוֹף הִשְבִּיחַ, (ח) אָסוּר. הגה: יֵשׁ אוֹמְרִים אַף עַל גַּב דְּהָאִסּוּר (נוֹתֵן טַעַם) לִפְגָם וְהַמַּאֲכָל מֻתָּר, מִכָּל (ט) מָקוֹם הַקְּדֵרָה אֲסוּרָה, וְאִם בִּשְּׁלוּ בָּהּ אַחַר כָּךְ תּוֹךְ מֵעֵת לְעֵת תַּבְשִׁיל שֶׁהָאִסּוּר הָרִאשׁוֹן נוֹתֵן בּוֹ טַעַם לְשֶׁבַח, נֶאֱסָר הַתַּבְשִׁיל הַשֵּׁנִי אִם לֹא הָיָה בּוֹ ס' נֶגֶד הָאִסּוּר הָרִאשׁוֹן. אֲבָל אִם נִעֲרוּ הַתַּבְשִׁיל הָרִאשׁוֹן בְּכַף, וְתָחֲבוּ אַחַר כָּךְ הַכַּף לְתַבְשִׁיל שֵׁנִי שֶׁהוּא גַּם כֵּן פּוֹגֵם, לֹא נֶאֶסְרָה הַקְּדֵרָה. וְכֵן בְּדָבָר שֶׁאֵין לוֹ טַעַם כְּלָל, כְּגוֹן הַיּוֹרָה שֶׁמַּתִּיכִין בּוֹ הַדְּבַשׁ, אַף עַל פִּי שֶׁיֵּשׁ שָׁם רַגְלֵי הַדְּבוֹרִים, לֹא נֶאֶסְרָה הַיּוֹרָה וְכָל כַּיּוֹצֵא בָּזֶה. (שָׁם בָּאָרֹךְ) .


ג אֲפִלּוּ אֵין כֹּחַ בָּאִסוּר לְבַדּוֹ (י) לִפְגֹּם אֶלָּא עַל יְדֵי דָּבָר אַחַר שֶׁמְּסַיְּעוֹ, כְּגוֹן שֶׁנָּפַל אִסוּר לִקְדֵרָה שֶׁיֵּשׁ בָּהּ מֶלַח אוֹ תַּבְלִין מְדוּכִין (בַּטּוּר מְרֻבִּין) וְאִלְמָלֵא הַמֶּלַח וְהַתַּבְלִין שֶׁבָּהּ לֹא הָיָה כֹּחַ בָּאִסוּר לִפְגֹּם, אֲפִלּוּ הָכֵי מֻתָּר.


ד שֶׁמֶן וּדְבַשׁ שֶׁל עוֹבֵד כּוֹכָבִים, אַף עַל פִּי שֶׁהֵם מְבֻשָּׁלִים, מֻתָּרִים מִפְּנֵי שֶׁהַבָּשָׂר (יא) פּוֹגֵם אֶת הַשֶּׁמֶן וּמַסְרִיחוֹ, וְכֵן לִדְבַשׁ. הגה: וְיֵשׁ אוֹמְרִים דְּבָשָׂר אֵינוֹ פּוֹגֵם דְּבַשׁ עַצְמוֹ, רַק מַשְׁקֶה הַנַּעֲשֶׂה מִדְּבַשׁ (מהרי''ל בְּהַגָּהַת ש''ד סִימָן כ''ב וְעוֹד הַרְבֵּה פּוֹסְקִים עַיֵּן ס''ק י''ג) ; וּבְמָקוֹם שֶׁאֵין (יב) הֶפְסֵד גָּדוֹל, יֵשׁ לְהַחְמִיר. בָּשָׂר אוֹ חָלָב בְּיַיִן, הָוֵי לִפְגָם, וּמֻתָּר. (הַגָּהוֹת אֲשֵׁרִ''י פ''ב דע''ג) .


ה כָּל קְדֵרָה שֶׁאֵינָה בַּת יוֹמָא חֲשִׁיבָה טַעֲמָהּ לִפְגָם, וְאֵינָהּ אוֹסֶרֶת. וְנִקְרֵאת בַּת יוֹמָא כָּל זְמַן שֶׁלֹּא שָׁהֲתָה מֵעֵת לְעֵת אַחַר שֶׁנִּתְבַּשֵׁל בָּהּ הָאִסוּר, וְכֵיוָן שֶׁעָבַר עָלֶיהָ מֵעֵת לְעֵת אַחַר שֶׁנִּתְבַּשֵׁל בָּהּ הָאִסוּר אֵינָה נִקְרֵאת בַּת יוֹמָא. וְאִם בִּשֵּׁל בָּהּ כְּשֶׁאֵינָהּ בַּת יוֹמָא, הַתַּבְשִׁיל מֻתָּר דְּהָוֵי נוֹתֵן טַעַם לִפְגָם. וְהוּא שֶׁתִּהְיֶה (יג) מוּדַחַת שֶׁלֹּא יְהֵא שֻׁמָּן עַל פָּנֶיהָ, שֶׁאִם לֹא הֵדִיחָהּ, אוֹסֵר, וַהֲרֵי הִיא כַּחֲתִיכַת אִסוּר שֶׁלֹּא נִפְגְּמָה. וְיֵשׁ מַתִּירִין אֲפִלּוּ בִּשֵּׁל בָּהּ קֹדֶם שֶׁהֵדִיחָהּ. הגה: וְאִם יֵשׁ ס' נֶגֶד מַה שֶּׁדָּבוּק עָלָיו, לְכֻלֵּי (יד) עָלְמָא שָׁרֵי מֵאַחַר דְּהַקְּדֵרָה אֵינָהּ בַּת יוֹמָא. וְהָכֵי נָהוּג (או''ה) .


ו יֵשׁ מִי שֶׁאוֹמֵר, שֶׁאִם שָׂמוּ פִּלְפְּלִין (טו) בִּקְדֵרָה שֶׁל אִסוּר שֶׁאֵינָה בַּת יוֹמָא, הַכֹּל אָסוּר, דְּחוּרְפֵיהּ מְשַׁוְּיָא לֵהּ לְשֶׁבַח (הַכְרָעַת עַצְמוֹ בב''י) .


ז קְדֵרָה שֶׁהִיא בְּלוּעָה מִבָּשָׂר וְחָלָב שֶׁנִּתְבַּשְּׁלוּ בָּהּ בְּיַחַד אוֹ בָּזֶה אַחַר זֶה, וְקֹדֶם שֶׁעָבַר (טז) לַיְלָה אֶחָד הֻחַמּוּ (יז) בָּהּ מַיִם, חֲשִׁיבָה בַּת יוֹמָא עַד שֶׁתִּשְׁהֶה מֵעֵת לְעֵת מִשָּׁעָה שֶׁהוּחֲמוּ בָּהּ הַמַּיִם. הגה: אֲבָל אִם עָבְרָה לַיְלָה, מֻתָּר. וְכֵן אִם עָבְרָה לַיְלָה בֵּין בִּשּׁוּל בָּשָׂר לְחָלָב, אַף עַל גַּב דְּאַחַר כָּךְ הֻחַמּוּ הַמַּיִם מִיָּד, נָמֵי דִּינָא הָכֵי. וְהוּא הַדִּין (יח) בְּכָל הָאִסּוּרִים לְפִי מַאי דְּקַיְמָא לָן בְּכָל הָאִסּוּרִים חֲתִיכָה נַעֲשֵׂית נְבֵלָה (בֵּית יוֹסֵף וּסְבָרַת הָרַב) . וְעַיִּןִ לְעֵיל סִימָן צ''ב. וּבִמְקוֹם הֶפְסֵד יֵשׁ לְהַתִּיר בִּכְהַאי גַּוְנָא בִּשְׁאָר אִסּוּרִים בְּכָל עִנְיָן, רַק שֶׁיִּהְיֶה מֵעֵת לְעֵת מִזְּמַן בִּשּׁוּל הָאִסּוּר.



סימן קד - דין עכבר שנמצא ביין או בשכר, ובו ג' סעיפים


א עַכְבְּרָא (א) דְּדַבְּרָא נוֹתֵן טַעַם לְשֶׁבַח הוּא, שֶׁהֲרֵי עוֹלֶה עַל שֻׁלְחַן מְלָכִים. אֲבָל עַכְבְּרָא דְּמָתָא, מְסַפְּקָא לָן אִם מַשְׁבִּיחַ בְּשֵׁכָר וְחֹמֶץ, אוֹ אִם הוּא פּוֹגֵם. וּלְפִיכָךְ, אִם נָפַל לְשֵׁכָר אוֹ לְחֹמֶץ, בְּצוֹנֵן, וְהֵסִירוֹ שָׁלֵם, אִם לֹא שָׁהָה בְּתוֹכוֹ מֵעֵת לְעֵת, מֻתָּר. אֲבָל אִם הָיָה רוֹתֵחַ, אוֹ אֲפִלּוּ צוֹנֵן וְשָׁהָה בְּתוֹכוֹ מֵעֵת לְעֵת, בֵּין שֶׁהֵסִירוֹ שָׁלֵם, בֵּין שֶׁנֶּחְתַּךְ לַחֲתִיכוֹת דַּקּוֹת וְיָכוֹל לְסַנְּנוֹ בְּמִסְנֶנֶת בְּעִנְיָן שֶׁלֹּא יִשָּׁאֵר מִמֶּנּוּ בְּתוֹכוֹ כְּלוּם, בֵּין שֶׁנִּמּוֹחַ בְּתוֹכוֹ לְגַמְרֵי וְנַעֲשָׂה כֻּלּוֹ מַשְׁקֶה וְלֹא נִשְׁתַּיֵּר מִמֶּנּוּ שׁוּם מַמָּשׁוּת, נִתָּר עַל יְדֵי שֶׁיְּהֵא שִׁשִּׁים בַּהֵתֶּר כְּנֶגֶד הָעַכְבָּר. וְאִם נֶחְתַּךְ לַחֲתִיכוֹת דַּקּוֹת, וְהוּא בְּעִנְיָן שֶׁאֵינוֹ יָכוֹל לְסַנְּנוֹ כְּגוֹן שֶׁנִּתְעָרֵב הַשֵּׁכָר אוֹ הַחֹמֶץ בְּמַאֲכָל עָב, הַכֹּל אָסוּר וְאֵין שָׁם בִּטּוּל, דְּחַיְשִינָן שֶׁמָּא יִפְגַּע בְּמַמָּשׁוֹ שֶׁל אִסוּר וְלֹא יַרְגִּישׁ. הגה: וְדַוְקָא בְּשֶׁרֶץ יֵשׁ לָחוּשׁ אִם נִשְׁאַר שָׁם שֶׁלֹּא יוּכַל לְהוֹצִיאוֹ, אֲבָל בִּשְׁאָר (ב) אִסּוּרִין אֵין לָחוּשׁ (כָּךְ מַשְׁמָע בְּאָרֹךְ כְּלָל ל''ב) .


ב אִם נָפַל לְיַיִן וְשֶׁמֶן אוֹ לִשְׁאָר מַשְׁקִין, (ג) פּוֹגֵם בְּוַדַּאי וְאֵין צָרִיךְ ס' לְבַטֵּל פְּלִיטָתוֹ. הגה: וּלְפִי זֶה, אִם נָפַל לְשֻׁמָּן נָמֵי דִּינָא הָכֵי. וְיֵשׁ מַחֲמִירִין בְּשֻׁמָּן וַאֲפִלּוּ אִם הוּא קָשֶׁה לְפָנֵינוּ, אִם שָׁפְכוּ מִדֵּי יוֹם וָיוֹם שֻׁמָּן בַּקְּדֵרָה וְיֵשׁ (ד) לְסַפֵּק (ה) שֶׁמָּא הָעַכְבָּר הָיָה שָׁם כְּשֶׁעֵרוּ עָלָיו שֻׁמָּן רוֹתֵחַ, הַכֹּל אָסוּר, אֲפִלּוּ יֵשׁ ס' בַּשֻּׁמָּן נֶגֶד כָּל הָעַכְבָּר; וְאִם לֹא עֵרוּ עָלָיו, אוֹ שֶׁעֵרוּ עָלָיו וְיָדוּעַ שֶׁלֹּא הָיָה שָׁם הָעַכְבָּר כְּשֶׁעֵרוּ עָלָיו, וְהַשֻּׁמָּן בָּא לְפָנֵינוּ כְּשֶׁהוּא קָשֶׁה (תְּשׁוּבַת הָרֹא''שׁ וְהַגָּהַת מַיְמוֹנִי פט''ו וְהַגָּהוֹת מָרְדְּכַי דְּחֻלִּין וּבְהַגָּהַת ש''ד וּבְאָרֹךְ) וְכֵן נִמְצָא הָעַכְבָּר עָלָיו, סָגֵי לֵיהּ בִּנְטִילַת מָקוֹם וְלֹא מַחְזְקִינָן אִסּוּר שֶׁמָּא הָיָה הַשֻּׁמָּן רַךְ כְּשֶׁנָּפַל (ו) שָׁם וְכָבוּשׁ הָוֵי כִּמְבֻשָּׁל, דְּמֻתָּר מִכֹּחַ סְפֵק סְפֵיקָא, סָפֵק נָפַל שָׁם כְּשֶׁהָיָה קָשֶׁה, וְאִם תִּמְצָא לוֹמַר כְּשֶׁהָיָה רַךְ, שֶׁמָּא נִתְקַשָּׁה קֹדֶם שִׁעוּר כְּבִישָׁה. וּבְמָקוֹם שֶׁהַשֻּׁמָּן מָאוּס לֶאֱכֹל, אָסוּר לְהַדְלִיקוֹ גַּם כֵּן בְּבֵית הַכְּנֶסֶת, מִשּׁוּם הַקְרִיבֵהוּ נָא לְפֶחָתֶךָ (מַלְאָכִי א, ח) (ר''ן בְּשֵׁם גָּאוֹן וְרַשְׁבָּ''א פג''ה וְר' יְרוּחָם ואו''ה בְּשֵׁם מָרְדְּכַי פג''ה וְהַמְחַבֵּר בא''ח סי' קנ''ד סי''ב) וּבִמְקוֹם (ז) הֶפְסֵד גָּדוֹל יֵשׁ לִסְמֹךְ אַדִּבְרֵי הַמְקִלִּין.


ג דְּבָרִים הַמְּאוּסִים, שֶׁנַּפְשׁוֹ שֶׁל אָדָם קָצָה בָּהֶם, כִּנְמָלִים וּזְבוּבִים וְיַתּוּשִׁים שֶׁכָּל אָדָם בּוֹדֵל מֵהֶם לְמִאוּסָן, אֲפִלּוּ (ח) נִתְעָרְבוּ בְּתַבְשִׁיל וְנִמְחָה גּוּפָן לְתוֹכוֹ, אִם הַהֶתֵּר רָבֶה עָלָיו, מֻתָּרִים. וּמִכָּל מָקוֹם כָּל שֶׁאֶפְשָׁר לִבְדֹּק וּלְהַעֲבִיר בְּמִסְנֶנֶת, בּוֹדֵק וּמְסַנֵּן. מִיהוּ בַּחֲלָא וְשִׁכְרָא, יֵשׁ לָחוּשׁ כְּמוֹ בְּעַכְבָּר. (בֵּית יוֹסֵף לְדַעַת הָרַשְׁבָּ''א) .



סימן קה - דין אסור שנפל לתוך התר, ובו י''ד סעיפים


א אִסוּר שֶׁנִּשְׁרָה עִם הֶתֵּר מֵעֵת לְעֵת בְּצוֹנֵן, מִקְּרֵי (א) כָּבוּשׁ, וַהֲרֵי הוּא כִּמְבֻשָּׁל וְנֶאֱסָר כֻּלּוֹ. אֲבָל פָּחוֹת מִכָּאן, בַּהֲדָחָה סָגֵי. הגה: וְכָל מָקוֹם דְּאָמְרִינָן כָּבוּשׁ כִּמְבֻשָּׁל, אֲפִלּוּ מַה שֶּׁחוּץ לַכְּבִישָׁה אָסוּר, דְּעַל יְדֵי הַכְּבִישָׁה שֶׁלְּמַטָּה מְפַעְפֵּעַ לְמַעְלָה, כְּמוֹ בְּבִשּׁוּל. וְיֵשׁ (ב) מְקִלִּין בְּמַה שֶּׁבַּחוּץ (אָרֹךְ כְּלָל ל') . (ג) וְסָפֵק כָּבוּשׁ, אָסוּר, מִלְּבַד בְּבָשָׂר עִם חָלָב דְּאָזְלִינָן לְקֻלָּא, דְּמִן הַתּוֹרָה אֵינוֹ אָסוּר רַק בְּבִשּׁוּל מַמָּשׁ (שָׁם וְרָשָׁ''ל) . וְאִם הוּא כָּבוּשׁ בְּתוֹךְ צִיר אוֹ (ד) בְּתוֹךְ חֹמֶץ (ד''ע), אִם שָׁהָה כְּדֵי שֶׁיִּתְּנֶנּוּ עַל הָאוּר וְיַרְתִּיחַ וְיַתְחִיל לְהִתְבַּשֵּׁל, הֲרֵי הוּא כִּמְבֻשָּׁל; וּבְפָחוֹת מִשִּׁעוּר זֶה, לֹא נֶאֱסַר אֶלָּא כְּדֵי קְלִפָּה. וְעַיֵּן לְעֵיל סִימָן ע' מִדִּין בָּשָׂר שֶׁנָּפַל (ה) לְצִיר.


ב חֹם שֶׁל כְּלִי רִאשׁוֹן שֶׁהַיָּד סוֹלֶדֶת בּוֹ, מְבַשֵּׁל וְאוֹסֵר כֻּלּוֹ. אֲבָל חֹם שֶׁל כְּלִי שֵׁנִי אֵינוֹ מְבַשֵּׁל. וְיֵשׁ אוֹמְרִים שֶׁגַּם כֵּן אֵינוֹ מַפְלִיט וְאֵינוֹ מַבְלִיעַ. וְיֵשׁ אוֹמְרִים דְּמִכָּל מָקוֹם הוּא מַפְלִיט וּמַבְלִיעַ, וְאוֹסֵר כְּדֵי קְלִפָּה. וְרָאוּי לָחוּשׁ (ו) לִזָּהֵר בַּדָּבָר לְכַתְּחִלָּה (וְעַיֵּן לְעֵיל סי' ס''ח סְעִיף י''ג), אֲבָל בְּדִיעֲבַד מֻתָּר בְּלֹא קְלִפָּה, וּבַהֲדָחָה בְּעָלְמָא סָגֵי. וְעַיֵּן לְעֵיל סִימָן ס''ח וצ''ב וצ''ה נִתְבָּאֲרוּ דִּינֵי כְּלִי שֵׁנִי וְעֵרוּי.


ג נָפַל אִסוּר חַם לְתוֹךְ הֶתֵּר חַם דִּכְלִי רִאשׁוֹן, אוֹ אֲפִלּוּ אִסוּר צוֹנֵן לְתוֹךְ הֶתֵּר חַם, הַכֹּל אָסוּר, דְּתַתָּאָה גָּבַר עַל הָעֶלְיוֹן וּמְחַמְּמוֹ עַד שֶׁמַּפְלִיט בַּתַּחְתּוֹן. וְאֵין צָרִיךְ לוֹמַר דְּהֶתֵּר צוֹנֵן לְתוֹךְ אִסוּר חַם, שֶׁהַכֹּל אָסוּר. אֲבָל אִם הָעֶלְיוֹן חַם וְהַתַּחְתּוֹן צוֹנֵן, אֵינוֹ אוֹסֵר אֶלָּא כְּדֵי קְלִפָּה, אֲפִלּוּ אִם הָעֶלְיוֹן הַחַם אִסוּר. הגה: וְכָל זֶה לֹא מַיְרֵי אֶלָּא בְּחֹם כְּלִי רִאשׁוֹן, כְּגוֹן מִיָּד שֶׁהֵסִירוֹ מִן הָאֵשׁ (ז) מַנִּיחוֹ עִם הַהֶתֵּר. אֲבָל אִם כְּבָר מֻנָּח בִּכְלִי שֵׁנִי, וְאַחַר כָּךְ מַנִּיחַ הַהֶתֵּר אֶצְלוֹ אוֹ עָלָיו, אֵינוֹ אוֹסֵר כְּלָל, דִּכְלִי שֵׁנִי אֵינוֹ אוֹסֵר, כְּמוֹ שֶׁנִּתְבָּאֵר (כָּךְ מַשְׁמָע מִלְּשׁוֹן הָרַ''ן) . וְאִם הִנִּיחָהּ זֶה אֵצֶל זֶה, אִם שְׁנֵיהֶם חַמִּים מֵחֹם (ח) כְּלִי רִאשׁוֹן, הַכֹּל אָסוּר. וְאִם הָאֶחָד צוֹנֵן, הַהֶתֵּר צָרִיךְ (ט) קְלִיפָה בַּמָּקוֹם שֶׁנָּגַע (אָרֹךְ כְּלָל כ''ט ד''ב וְהַגָּהַת ש''ד) . אִסּוּר שֶׁהִנִּיחוּ בִּכְלִי הֶתֵּר אוֹ לְהֶפֶךְ, אַמְרִינָן בֵּיהּ גַּם כֵּן דִּין תַּתָּאָה (י) גָּבַר, כְּמוֹ בְּב' חֲתִיכוֹת (שָׁם כְּלָל ל''ו) . וְעַיֵּן לְעֵיל סִימָן צ''ד אִם חָתַךְ בָּשָׂר בְּסַכִּין חוֹלֶבֶת. אָסוּר (יא) לְעָרוֹת מִכְּלִי שֶׁיֵּשׁ בּוֹ שֻׁמָּן כָּשֵׁר לְנֵר דּוֹלֵק שֶׁיֵּשׁ בּוֹ חֵלֶב אוֹ שֻׁמַּן אִסּוּר (ג''ז שָׁם וּמָרְדְּכַי פכ''ה וְעַיֵּן בִּתְשׁוּבַת מַהֲרִי''ל וּבְפִסְקֵי מהרא''י סִימָן ק''ג), ובְּדִיעֲבַד אֵין לָחוּשׁ (אָרֹךְ) .


ד בַּמֶּה דְּבָרִים אֲמוּרִים, כְּשֶׁנָּפַל לְתוֹךְ הַתַּבְשִׁיל, מִפְּנֵי שֶׁהָרֹטֶב מוֹלִיךְ פְּלִיטַת הָאִסוּר וּמְעָרְבוֹ בְּכָל הַתַּבְשִׁיל. אֲבָל אִסוּר, בֵּין חַם בֵּין קַר, שֶׁנָּפַל עַל הַצְּלִי שֶׁאֵצֶל הָאֵשׁ, אֵינוֹ אוֹסֵר אֶלָּא כְּדֵי (יב) נְטִילָה, שֶׁהוּא כָּעֳבִי רֹחַב אֶצְבַּע. לְפִיכָךְ, יָרֵךְ שֶׁצְּלָאוֹ בְּגִידוֹ, אוֹ חֲתִיכַת אִסוּר שֶׁצְּלָאָהּ עִם חֲתִיכַת הֶתֵּר וְנוֹגֵעַ זֶה בָּזֶה, צָרִיךְ לְהָסִיר כְּדֵי נְטִילָה סְבִיב הַגִּיד, וְכֵן מִן הַחֲתִיכָה מָקוֹם שֶׁנָּגְעָה בַּחֲתִיכַת אִסוּר. וְכֵן אִם נָפַל אִסוּר עַל חֲתִיכָה שֶׁבַּקְּדֵרָה שֶׁהִיא חוּץ לָרֹטֶב, וְלֹא נִעֵר הַקְּדֵרָה וְלֹא כִּסָה אוֹתָהּ, אֵינוֹ אוֹסֵר אֶלָּא כְּדֵי נְטִילָה. אֲבָל אִם הִיא בָּרֹטֶב, לְרַשִׁ''י (כֻּלָּהּ ולר''י) אֲפִלּוּ מִקְצָתָהּ, אוֹ אֲפִלּוּ כֻּלָּהּ חוּץ לָרֹטֶב וְנִעֵר אוֹ כִּסָה הַקְּדֵרָה, הָרֹטֶב מְפַעְפֵּעַ הַטַּעַם וּמְעָרְבוֹ וְנִכְנָס בְּכֻלּוֹ.


ה בַּמֶּה דְּבָרִים אֲמוּרִים, שֶׁאֵין הַצָּלִי אוֹסֵר אֶלָּא כְּדֵי נְטִילָה, בְּיָרֵךְ עִם גִּידוֹ וְכַיּוֹצֵא בוֹ, דָּבָר כָּחוּשׁ שֶׁאֵין בּוֹ כֹּחַ לְפַעְפֵּעַ בְּכָל הַחֲתִיכָה; אֲבָל גְּדִי שָׁמֵן שֶׁצְּלָאוֹ בְּחֶלְבּוֹ, אִם אֵין בְּכָל הַגְּדִי שִׁשִּׁים כְּנֶגֶד כָּל הַחֵלֶב שֶׁבּוֹ, אָסוּר לֶאֱכֹל אֲפִלּוּ מֵרֹאשׁ אָזְנוֹ, שֶׁכֵּיוָן שֶׁהוּא שָׁמֵן מְפַעְפֵּעַ בְּכֻלּוֹ. אֲבָל אִם הוּא כָּחוּשׁ, אַף עַל פִּי שֶׁאֵין בּוֹ ס' כְּנֶגֶד כָּל הַחֵלֶב שֶׁבּוֹ, אֵינוֹ אוֹסֵר אֶלָּא כְּדֵי נְטִילָה, שֶׁהַחֵלֶב שֶׁל בְּהֵמָה כְּחוּשָׁה, כָּחוּשׁ הוּא בְּטִבְעוֹ, וְאֵינוֹ מְפַעְפֵּעַ. וַאֲפִלּוּ חֲתִיכַת הָאִסוּר כְּחוּשָׁה וַחֲתִיכַת הֶתֵּר שֶׁנִּצְלֵית עִמָּהּ (יג) שְׁמֵנָה, הָאִסוּר מְפַעְפֵּעַ בְּכֻלָּהּ. וְכָל דָּבָר שֶׁיֵּשׁ לוֹ פִּעְפּוּעַ שֶׁנָּפַל (יד) לְמָקוֹם יָדוּעַ מֵהַחֲתִיכָה בַּצָּלִי, אַף עַל פִּי שֶׁיֵּשׁ בַּחֲתִיכָה שִׁשִּׁים לְבַטֵּל הָאִסוּר, צָרִיךְ נְטִילַת מָקוֹם. (מָרְדְּכַי בְּשֵׁם כַּמָּה רַבְּוָתָא וּסְמַ''ק וְהַגה''א וְאו''ה כְּלָל י''ד וְאָגוּר בְּשֵׁם מַהֲרִי''ל) הגה: וְכָל מְבֻשָּׁל בְּלֹא רֹטֶב, אוֹ אָפוּי, דִּינוֹ כְּצָלִי בְּכָל דָּבָר (טוּר וּבֵית יוֹסֵף) . וְיֵשׁ אוֹמְרִים דְּאֵין אָנוּ בְּקִיאִין אֵיזֶה מִקְרֵי כָּחוּשׁ אוֹ שָׁמֵן, וְיֵשׁ לֶאֱסֹר בְּכָל עִנְיָן עַד דְּאִכָּא ס' (הַגָּהַת ש''ד וְהג''מ פט''ו ואו''ה וּסְמַ''ק וְאָגוּר בְּשֵׁם מַהֲרִי''ל) . וַאֲפִלּוּ אִכָּא ס', צָרִיךְ נְטִילַת מָקוֹם (אָרֹךְ), וְהָכֵי נָהוּג. וְדַוְקָא בְּאִסּוּר חֵלֶב אוֹ שְׁאָר אִסּוּר (טו) דְּשַׁיָּךְ בּוֹ שַׁמְנוּנִית, אֲבָל בְּאִסּוּר דְּלֹא שַׁיָּךְ בֵּיהּ שַׁמְנוּנִית, וְהוּא בְּוַדַּאי כָּחוּשׁ, אֵינוֹ אוֹסֵר רַק כְּדֵי נְטִילָה. (עי' ס''ק ט') (כָּךְ מַשְׁמָע בְּבֵּית יוֹסֵף) .


ו דָּבָר שֶׁיֵּשׁ לוֹ פִּעְפּוּעַ בְּצָלִי, אִם הָיָה אֶחָד חַם וְאֶחָד צוֹנֵן, הַתַּחְתּוֹן גּוֹבֵר. לְפִיכָךְ אִם נָפַל חַם עַל צוֹנֵן, אֵינוֹ אוֹסֵר אֶלָּא כְּדֵי קְלִפָּה. וְאִם נָפַל צוֹנֵן עַל חַם, אִם אֵין בַּהֶתֵּר ס' כְּנֶגֶד הָאִסוּר, הַכֹּל אָסוּר.


ז הָא דַּחֲתִיכַת אִסוּר אוֹסֶרֶת חֲבֶרְתָּהּ בִּנְגִיעָתָהּ, דַּוְקָא כְּשֶׁאִסוּרָהּ מֵחֲמַת עַצְמָהּ, כְּגוֹן נְבֵלָה אוֹ בָּשָׂר (טז) בְּחָלָב, אֲבָל אִם אֵין בָּהּ אִסוּר אֶלָּא מַה שֶּׁבָּלַע מִמָּקוֹם אַחֵר, אֵינָהּ אוֹסֶרֶת אַחֶרֶת הַנּוֹגַעַת בָּהּ, אֲפִלּוּ אִם נִצְלוּ (יז) יַחַד. בַּמֶּה (יח) דְּבָרִים אֲמוּרִים, כְּשֶׁבָּלְעָה אִסוּר שֶׁאֵינוֹ מְפַעְפֵּעַ, אֲבָל בָּלְעָה מִן הַדְּבָרִים הַמְּפַעְפְּעִים בְּטִבְעָן, כְּגוֹן שֻׁמָּן, אוֹסֶרֶת חֲבֶרְתָּהּ בֵּין בְּצָלִי בֵּין בְּנוֹגַעַת זוֹ בָּזוֹ חַם בְּחַם, אוֹ בְּתַחְתּוֹנָה חַמָּה וְעֶלְיוֹנָה צוֹנֶנֶת, שֶׁהָאִסוּר הַבָּלוּעַ בְּעַצְמָהּ מְפַעְפֵּעַ וְיוֹצֵא מֵחֲתִיכָה לַחֲתִיכָה (חִדּוּשֵׁי רַשְׁבָּ''א פכ''ה דַּף ק''ז ע''ג וּמהרא''י בְּשֵׁם אוֹר זָרוּעַ וּמהר''מ) . הגה: (יט) וְכָל זֶה בְּב' חֲתִיכוֹת, אֲבָל כְּלִי שֶׁבָּלַע אִסּוּר, אוֹסֵר הֶתֵּר שֶׁנּוֹגֵעַ בּוֹ אֲפִלּוּ בְּאִסּוּר שֶׁאֵינוֹ שָׁמֵן (הַגָּהוֹת ש''ד בְּשֵׁם ר''י מֵאִבְּרָא וְאו''ה כְּלָל ל''ו) .


ח כֻּלְיָא שֶׁצְּלָאוֹ בְּחֶלְבּוֹ אֵינוֹ אוֹסֵר אֶלָּא כְּדֵי קְלִפָּה, שֶׁהַקְּרוּם מַפְסִיק. הגה: וְיֵשׁ אוֹסְרִים, וְכֵן נוֹהֲגִין וְאֵין לְשַׁנּוֹת, וְדִינוֹ כִּשְׁאָר חֵלֶב הַנִּצְלֶה עִם בָּשָׂר, (כ) וְאִם נִתְבַּשֵּׁל כָּךְ הַכֻּלְיָא נַעֲשֵׂית נְבֵלָה וְצָרִיךְ שִׁשִּׁים נֶגֶד כֻּלָּהּ, וְהוּא הַדִּין בַּקְּרוּם שֶׁעַל הַיּוֹתֶרֶת (סמ''ק וְהַגָּהוֹת ש''ד וְהג''א) .


ט מָלִיחַ שֶׁאֵינוֹ נֶאֱכָל מֵחֲמַת מִלְחוֹ (כא), דִּינוֹ כְּרוֹתֵחַ וּמַפְלִיט לֶאֱסֹר כְּדֵי קְלִפָּה. וְאִם הוּא שֻׁמַּן הַגִּיד וּקְנוֹקָנוֹת שֶׁבּוֹ וְהַקְּרוּמוֹת, יֵשׁ לְהַחֲמִיר וּלְהַצְרִיכָם (כב) נְטִילַת מָקוֹם. וְאִם הוּא חֵלֶב מַמָּשׁ, צָרִיךְ שִׁשִּׁים לְבַטְּלוֹ. וְאִם הַבָּשָׂר שֶׁנִּמְלַח עִמּוֹ הוּא שָׁמֵן, אֲפִלּוּ בְּשֻׁמַּן (כג) הַגִּיד וּקְנוֹקָנוֹת שֶׁבּוֹ, צָרִיךְ ס' לְבַטְּלוֹ, וְלִטֹּל מִמָּקוֹם שֶׁנָּגַע כְּדֵי נְטִילַת מָקוֹם, וּלְפָחוֹת קְלִפָּה. וְכָל זֶה לְעִנְיַן חֲתִיכָה (כד) עַצְמָהּ שֶׁהָיָה הַחֵלֶב דָּבוּק בָּהּ. וּלְעִנְיַן שְׁאָר חֲתִיכוֹת שֶׁנִּמְלְחוּ יַחַד, אִם אֵין בְּכָל אַחַת מֵהֶם בִּפְנֵי עַצְמָהּ שִׁשִּׁים לְבַטֵּל הַחֵלֶב, אֲסוּרוֹת וְאֵין מִצְטָרְפִין בְּיַחַד לְבַטֵּל הַחֵלֶב, דְּאֵין חֵלֶב מְפַעְפֵּעַ מֵחֲתִיכָה לַחֲתִיכָה בְּלֹא רֹטֶב. וְאִם אֵין יָדוּעַ אִם נָגַע בְּכֻלָּן, כֻּלָּן אֲסוּרוֹת. וְאִם יָדוּעַ שֶׁלֹּא נָגַע אֶלָּא בְּאַחַת, וְאֵין יָדוּעַ אֵיזוֹ הִיא, כֻּלָּן מֻתָּרוֹת, דְּחַד בִּתְרֵי בָּטִיל, אֲפִלּוּ הֵן חֲתִיכוֹת הָרְאוּיוֹת (כה) לְהִתְכַּבֵּד. הגה: וְיֵשׁ אוֹמְרִים דְּכָל (כו) מְלִיחָה אֵינָהּ אוֹסֶרֶת רַק כְּדֵי קְלִפָּה (טוּר וּמָרְדְּכַי בְּשֵׁם ראבי''ה וְר' אֶפְרַיִם וּפוֹסְקִים וְרַ''ן וְרַשְׁבָּ''א בְּמ''ה וּרא''ה בְּב''ה דף צ''ב בְּשֵׁם רַמְבַּ''ן וְצ''ע אמ''ש בֵּית יוֹסֵף סִימָן צ''א וְת''ח רֵישׁ כְּלָל ל''ח ע''ש רַמְבַּ''ן לְהֵפֶךְ) . וּלְפִי שֶׁאֵין אָנוּ בְּקִיאִין בֵּין בָּשָׂר שָׁמֵן לְכָחוּשׁ, נוֹהֲגִין לְשַׁעֵר בְּכָל מְלִיחָה בְּס' כְּמוֹ בְּבִשּׁוּל (מהרי''ו וּבְאָרֹךְ וּבְהַגָּהַת ש''ד וּבְמָרְדְּכַי בְּשֵׁם אוֹר זָרוּעַ וְכָל הָאַחֲרוֹנִים), דְּאִם אִכָּא ס' בֵּין הַכֹּל בְּיַחַד נֶגֶד כָּל הַחֲתִיכָה שֶׁהַחֵלֶב דָּבוּק בָּהּ, אָז הַכֹּל שָׁרֵי מִלְּבַד אוֹתָהּ חֲתִיכָה שֶׁהָאִסּוּר דָּבוּק בָּהּ, מִיהוּ אוֹתָהּ הַחֲתִיכָה שֶׁהָאִסּוּר נָגַע בָּהּ צָרִיךְ קְלִיפָה מְעַט (רַשְׁבָּ''א) . וְאִם אֵינוֹ יוֹדֵעַ אֵיזֶה נָגַע, הַכֹּל שָׁרֵי (מָרְדְּכַי פֶּרֶק ג''ה וְש''ד) . וְאִם אֵין ס', הַכֹּל אָסוּר. וְאֵין לְשַׁנּוֹת הַמִּנְהָג. וְאַף עַל פִּי שֶׁיֵּשׁ בָּזֶה קְצָת (כז) קֻלָּא אִם הָיָה הָאִסּוּר שָׁמֵן וּמְפַעְפֵּעַ, מִכָּל מָקוֹם יֵשׁ לִסְמֹךְ בְּכִי הַאי גַּוְנָא אַדִּבְרֵי הַמְּקִלִּין וּסְבִירָא לְהוּ דְּאֵינוֹ אוֹסֵר בִּמְלִיחָה רַק כְּדֵי קְלִפָּה, כְּדֵי שֶׁלֹּא נִצְטָרֵךְ לְשַׁעֵר בֵּין אִסּוּר שָׁמֵן לְכָחוּשׁ, כִּי אֵין אָנוּ בְּקִיאִין. וְכָל זֶה בְּאִסּוּר שֶׁשַּׁיָּךְ בּוֹ שַמְנוּנִית, כְּגוֹן חֵלֶב אוֹ (כח) צִיר וְכִי הַאי גַּוְנָא, אֲבָל בְּאִסּוּר שֶׁאֵין בּוֹ שַמְנוּנִית כְּלָל, כְּגוֹן חָמֵץ (כט) בְּפֶסַח, לְכֻלֵּי עָלְמָא מְלִיחָה אֵינָהּ אוֹסֶרֶת רַק כְּדֵי קְלִיפָה (סְבָרַת הָרַב מִמַּשְׁמָעוּת פּוֹסְקִים הַנַּ''ל) . וְעַיִּןִ בְּאֹרַח חַיִּים הִלְכוֹת חָמֵץ כֵּיצַד נוֹהֲגִין. וְעַיֵּן לְעֵיל סִימָן ס''ד אִם הַקְּרוּמִים אוֹ חוּטִים שֶׁל חֵלֶב נִמְלְחוּ עִם בָּשָׂר. יֵשׁ אוֹמְרִים דִּבְהֶפְסֵד מְרֻבֶּה יֵשׁ לְהַתִּיר בִּמְלִיחָה עַל יְדֵי קְלִיפָה, אַף עַל גַּב שֶׁכְּבָר נָהֲגוּ לְשַׁעֵר כָּל מְלִיחָה בְּס', אֲפִלּוּ בְּאִסּוּר כָּחוּשׁ; וְיֵשׁ לִסְמֹךְ עַל זֶה בְּאִסּוּר כָּחוּשׁ, שֶׁאֵינוֹ אֶלָּא מִנְהָג לְשַׁעֵר בְּס' (לְפִי סְבָרַת הַרַמְבַּ''ם וְהַנִּמְשָׁכִים אַחֲרָיו וּמַהֲרַ''ם פַּדָּוואָה בִּתְשׁוּבָה לְהָרַב סִימָן צ'), אֲבָל לֹא בְּאִסּוּר שָׁמֵן שֶׁמִּדִּינָא צָרִיךְ ס'.


י בַּמֶּה דְּבָרִים אֲמוּרִים, כְּשֶׁהָאִסוּר וְהַהֶתֵּר שְׁנֵיהֶם מְלוּחִים. וַאֲפִלּוּ אִסוּר מָלִיחַ וְהַהֶתֵּר תָּפֵל. אֲבָל אִם הַהֶתֵּר מָלוּחַ וְהָאִסוּר תָּפֵל, אֵינוֹ צָרִיךְ אֶלָּא הֲדָחָה. וְעַיֵּן לְעֵיל סִימָן ע'. וְיֵשׁ אוֹסְרִין אִם נוֹגְעִין זֶה בָּזֶה, וְיֵשׁ לְהָקֵל בִּמְקוֹם הֶפְסֵד.


יא זֶה שֶׁאָמַרְנוּ בְּאִסוּר מָלִיחַ וְהֶתֵּר תָּפֵל, שֶׁאִם הוּא שָׁמֵן אוֹסֵר עַד שֶׁיְּהֵא בּוֹ שִׁשִּׁים לְבַטְּלוֹ, דַּוְקָא כְּשֶׁהָיָה הָאִסוּר הַמָּלִיחַ לְמַטָּה וְהַהֶתֵּר תָּפֵל לְמַעְלָה, מִשּׁוּם דְּתַתָּאָה גָּבַר. אֲבָל אִם הָיָה הַהֶתֵּר תָּפֵל לְמַטָּה, וְהָאִסוּר הַמָּלִיחַ לְמַעְלָה, אֵינוֹ אוֹסֵר אֶלָּא כְּדֵי קְלִפָּה. הגה: וְיֵשׁ (ל) חוֹלְקִין וְאוֹמְרִים דִּבְמָלִיחַ אֵין חִלּוּק אִם הוּא לְמַעְלָה אוֹ לְמַטָּה (תּוֹסָפוֹת וְהָרֹא''שׁ וּסְמַ''ק וְהַטּוּר וְהַגָּהוֹת מַיְמוֹנִי) . וְהָכֵי נָהוּג.


יב גְּבִינוֹת שֶׁנַּעֲשׂוּ בִּדְפוּסֵי עוֹבְדֵי כּוֹכָבִים, אַף עַל פִּי שֶׁנִּמְלְחוּ בְּתוֹכָן, מֻתָּרִים. הגה: וְהוּא הַדִּין אִם נִמְלַח הֶתֵּר בִּשְׁאָר כְּלִי (לא) אִסּוּר (תְּשׁוּבַת הָרַשְׁבָּ''א), וַאֲפִלּוּ אֵינוֹ מְנֻקָּב (ד''ע), דְּאֵין מָלִיחַ כְּרוֹתֵחַ כָּל כָּךְ לְהַפְלִיט מִן הַכְּלִי מַה שֶּׁבָּלַע בְּתוֹכוֹ. וְדַוְקָא בְּדִיעֲבַד, אֲבָל לְכַתְּחִלָּה, אָסוּר. (מָרְדְּכַי וְהַגָּהַת ש''ד) .


יג מֶלַח אוֹ תַּבְלִין שֶׁהֵם בִּקְעָרָה שֶׁל בָּשָׂר, מֻתָּר לִתְּנֵם (לב) בְּחָלָב. הגה: וְהוּא הַדִּין אִם הָיוּ בִּכְלִי אִסּוּר, דְּמֵאַחַר שֶׁהֵם (לג) יְבֵשִׁים אֵינָן בּוֹלְעִים מִן הַכְּלִי, רַק שֶׁהַכְּלִי הוּא נָקִי וְאֵין אִסּוּר דָּבוּק בּוֹ (הַגָּהוֹת ש''ד וְכֵן כָּתַב א''ח בְּשֵׁם הרא''ה) . וְהַמַּחְמִיר לְכַתְּחִלָּה, תָּבֹא עָלָיו בְּרָכָה (אָרֹךְ וּסְמַ''ג לְעֵיל סוֹף סִימָן צ''ה) .


יד (לד) מֶלַח הַבָּלוּעַ מִדָּם, כְּגוֹן מִמְּלִיחַת בָּשָׂר, וּנְתָנוּהוּ בִּקְדֵרָה, אוֹ שֶׁנָּתְנוּ בַּקְּדֵרָה בָּשָׂר מָלוּחַ בְּלֹא הֲדָחָה, אִם יֵשׁ ס' כְּנֶגֶד הַמֶּלַח, מֻתָּר. הגה: אַף עַל פִּי שֶׁהַמֶּלַח עֲדַיִן נוֹתֵן טַעַם בַּקְּדֵרָה, מֵאַחַר שֶׁאֵין הַמֶּלַח אָסוּר מֵחֲמַת עַצְמוֹ, אֶלָּא מֵחֲמַת דָּם שֶׁבְּתוֹכוֹ, דְּכָל מָקוֹם שֶׁאֵין הָאוֹסֵר יָכוֹל לֵילֵךְ שָׁם, אֵין הַנֶּאֱסָר אוֹסֵר יוֹתֵר מִן הָאוֹסְרוֹ. (וְעַיֵּן לְעֵיל סִימָן ס''ט בַּטּוּר וּבְהַגָּהוֹת שַׁ''ד וּבְש''ד) .



סימן קו - דין היאך מבטלין בששים, ובו ב' סעיפים


א חֲתִיכָה שֶׁבָּלְעָה אִסוּר, וְאֵין בָּהּ ס' לְבַטְּלוֹ, שֶׁנָּפְלָה לִקְדֵרָה, אֵינָהּ אוֹסֶרֶת אֶלָּא לְפִי חֶשְׁבּוֹן אִסוּר שֶׁבָּהּ, שֶׁאִם יֵשׁ בְּמַה שֶּׁבַּקְּדֵרָה מְצֹרָף עִם הַחֲתִיכָה עַצְמָהּ שִׁשִּׁים כְּנֶגֶד אִסוּר הַבָּלוּעַ בָּהּ, מֻתָּר מַה שֶּׁבַּקְּדֵרָה, אֲבָל הַחֲתִיכָה עַצְמָהּ אֲסוּרָה, לְפִי שֶׁאִסוּר שֶׁבָּהּ אֵינוֹ נִפְלָט מִמֶּנָּהּ לְגַמְרֵי (רַשְׁבָּ''א ותוס' פֶּרֶק ג''ה דַּף ק' וְרֹא''שׁ וּמָרְדְּכַי שָׁם וּסְמַ''ג דַּף נ''א ע''ג וסה''ת סי' נ''ב וְכ''ה בִּלְשׁוֹן ר' אֶפְרַיִם גּוּפֵיהּ שֶׁבְּחִדּוּשֵׁי רַשְׁבָּ''א דַּף קי''ח וְשֶׁבְּסוֹף סֵפֶר ראב''ן וְכֵן כָּתַב מַהַרְשַׁ''ל פֶּרֶק ג''ה סִי' נ''ז וְכָל הָאַחֲרוֹנִים) וְכֵן עִקָּר, וּדְלֹא כְּמוֹ שֶׁכָּתַב לְעֵיל סִימָן צ''ב דְּאַף אוֹתָהּ חֲתִיכָה מֻתֶּרֶת. מַה שֶּׁאֵין כֵּן בְּדָבָר הַנִּבְלָל וְנִמּוֹחַ, שֶׁאִם נָפַל דָּם וְכַיּוֹצֵא בוֹ לְתוֹךְ רֹטֶב שֶׁל הֶתֵּר וַאֲסָרוֹ מֵחֲמַת מִעוּטוֹ, וְאַחַר כָּךְ נִתְרַבֶּה הָרֹטֶב שֶׁל הֶתֵּר עַד שֶׁיֵּשׁ בֵּין כֻּלּוֹ ס' לְבַטֵּל הַדָּם כֻּלּוֹ, מֻתָּר, שֶׁהַכֹּל נִבְלַל וְנִתְעָרֵב. הגה: וּלְפִי מַה שֶּׁנּוֹהֲגִין לוֹמַר בְּכָל הָאִסּוּרִים חֲתִיכָה נַעֲשֵׂית נְבֵלָה, אֵין חִלּוּק בֵּין דָּבָר לַח לְדָבָר יָבֵשׁ אֶלָּא לְעִנְיָן זֶה דְּאִם הַנֶּאֱסָר כְּבָר הוּא דָּבָר יָבֵשׁ וְיֵשׁ ס' כְּנֶגְדוֹ, הַחֲתִיכָה הַנֶּאֱסֶרֶת תְּחִלָּה נִשְׁאֶרֶת בְּאִסּוּרָהּ וְצָרִיךְ לַהֲסִירָהּ מִשָּׁם אִם מַכִּירָהּ, וְאִם אֵינוֹ מַכִּירָהּ בְּטֵלָה אִם אֵינָהּ חֲתִיכָה הָרְאוּיָה (א) לְהִתְכַּבֵּד. וְאִם הוּא דָּבָר לַח, הַכֹּל מֻתָּר, מֵאַחַר דְּאִכָּא ס' נֶגֶד מַה שֶּׁנֶּאֶסְרָה תְּחִלָּה. וְעַיֵּן לְעֵיל סִימָן צ''ב וצ''ט.


ב חֲתִיכָה שֶׁיֵּשׁ בָּהּ חֵלֶב שֶׁנִּתְבַּשְּׁלָה בִּקְדֵרָה שֶׁיֵּשׁ בָּהּ ס' (ב) לְבַטֵּל הַחֵלֶב, צָרִיךְ לִזָּהֵר שֶׁלֹּא יָסִיר שׁוּם דָּבָר מֵהַקְּדֵרָה בְּעוֹד חֲתִיכַת הָאִסוּר בְּתוֹכָהּ, דְּחַיְשִינָן שֶׁמָּא תִּשָּׁאֵר בָּאַחֲרוֹנָה בְּשָׁעָה שֶׁאֵין בַּקְּדֵרָה ס' לְבַטֵּל הַחֵלֶב. וְגַם לֹא יוֹצִיאֶנָּה תְּחִלָּה, שֶׁחֵלֶב שֶׁבָּהּ יֶאֱסֹר אוֹתָהּ. וּמַה תַּקָּנָתָהּ, יַנִּיחֶנָּה עַד (ג) שֶׁתִּצְטַנֵּן הַקְּדֵרָה. הגה: וּלְפִי מַה דְּקַיְמָא לָן חֲתִיכָה נַעֲשֵׂית (ד) נְבֵלָה, מֵסִיר הַחֲתִיכָה הָאֲסוּרָה מִשָּׁם, וְהַשְּׁאָר מֻתָּר (ד''ע) .



סימן קז - דין המבשל ביצים, ודבר מאוס הנמצא בתבשיל, ובו ב' סעיפים


א הַמְּבַשֵּׁל בֵּיצִים הַרְבֵּה, בִּקְלִפָּתָן, לֹא יוֹצִיאֵם מֵהַמַּיִם שֶׁנִּתְבַּשְּׁלוּ בָּהֶם עַד שֶׁיִּצְטַנְּנוּ, אוֹ יִתֵּן עֲלֵיהֶם מַיִם צוֹנְנִים לְצַנְּנָם וְאַחַר כָּךְ יוֹצִיאֵם, מִשּׁוּם דְּחַיְשִינָן שֶׁמָּא יִמְצָא בְּאַחַת מֵהֶן אֶפְרוֹחַ, וְאִם הָיָה מוֹצִיא מֵהֶם קֹדֶם שֶׁיִּצְטַנְּנוּ שֶׁמָּא הָיְתָה נִשְׁאָר אוֹתָהּ שֶׁיֵּשׁ בָּהּ אֶפְרוֹחַ עִם הָאַחֲרוֹנוֹת וְהָיְתָה אוֹסַרְתָּן, לְפִי שֶׁלֹּא הָיָה נִשְׁאָר שָׁם שִׁשִּׁים לְבַטְּלָהּ. הגה: וְאִם לֹא עָשָׂה כֵן, אֶלָּא עֵרָה אוֹתָן (א) לִקְעָרָה וְנִמְצָא אַחַת מֵהֶן טְרֵפָה, יֵשׁ אוֹסְרִין הַכֹּל דְּחַיְשִינָן שֶׁמָּא הַטְּרֵפָה נִשְׁאַר לְבַסּוֹף, וְלֹא הָיָה שִׁשִּׁים בַּקְּדֵרָה לְבַטֵּל, וְנֶאֱסַר מַה שֶּׁבַּקְּדֵרָה וְחוֹזֵר וְאוֹסֵר כָּל מַה שֶּׁבַּקְּעָרָה. (בֵּית יוֹסֵף בְּשֵׁם סְמַ''ג וּבְאָרֹךְ) וְכֵן בְּדָגִים קְטַנִּים, שֶׁנִּמְצָא דָּג טָמֵא בִּקְעָרָה, וְלֹא עֵרָה כֻּלָּם בְּפַעַם אַחַת אֶל תּוֹךְ הַקְּעָרָה, דְּאָז יֵשׁ לָחוּשׁ שֶׁמָּא נִשְׁאַר הָאִסּוּר לְבַסּוֹף. וְיֵשׁ מַתִּירִין בְּכָל עִנְיָן (טוּר בְּשֵׁם י''א), דְּלֹא מַחְזְקִינָן אִסּוּר לוֹמַר דְּנִשְׁאַר הָאִסּוּר בְּלֹא ס' (בַּיִת יוֹסֵף בְּשֵׁם ר''ש) . וְכֵן (ב) עִקָּר. וַאֲפִלּוּ לְדַעַת הָאוֹסְרִים, אֵין לֶאֱסֹר (ג) הַכֵּלִים שֶׁנִּתְבַּשְּׁלוּ בּוֹ, דְּמַעֲמִידִין הַכְּלִי עַל חֶזְקָתוֹ. (בֵּית יוֹסֵף בְּשֵׁם סְמַ''ג וְהַגָּהוֹת מַיְמוֹנִי) .


ב (ד) זְבוּב, וְכַיּוֹצֵא בוֹ מִדְּבָרִים הַמְּאוּסִים שֶׁנַּפְשׁוֹ שֶׁל אָדָם קָצָה בָּהֶם, שֶׁנִּמְצָא בְּתַבְשִׁיל, זוֹרְקָן, וְהַתַּבְשִׁיל מֻתָּר, שֶׁאֵין פְּלִיטַת דְּבָרִים אֵלּוּ הַפְּגוּמִים אוֹסֶרֶת. (ועי' ס''ק ו') הגה: וְכֵן הַמִּנְהָג פָּשׁוּט. אַף עַל גַּב דְּיֵשׁ מַחְמִירִין, דִּבְרֵי הַמְקִלִּין עִקָּר. (בֵּית יוֹסֵף פָּסַק כְּרַשְׁבָּ''א וְרוֹקֵחַ וְטוּר סי' ק''ד) . וְאֵין לְשַׁנּוֹת הַמִּנְהָג. וְאִם לָקַח עִם הַכַּף דָּג טָמֵא מִן הַקְּדֵרָה, אוֹ שְׁאָר דָּבָר שֶׁאֵינוֹ פּוֹגֵם, אָסוּר (ה) לְהַחֲזִיר הַכַּף לַקְּדֵרָה. וְאִם הֶחֱזִיר, צָרִיךְ ס' מִן הַקְּדֵרָה נֶגֶד הָאִסּוּר, וְלֹא נֶגֶד (ו) כָּל הַכַּף. (ד''ע דְּלֹא אַמְרִינָן גַּבֵּי כְּלִי חֲתִיכָה (ז) נַעֲשֵׂית נְבֵלָה) . אֲבָל אִם הָיָה מְעַט מִן הַתַּבְשִׁיל עִם הָאִסּוּר בַּכַּף וְהֻחְזַר לַקְּדֵרָה, צָרִיךְ ס' גַּם נֶגֶד מְעַט הַתַּבְשִׁיל, דַּהֲרֵי נַעֲשֶׂה נְבֵלָה בַּכַּף (אָרֹךְ כְּלָל כ''ב) .



סימן קח - שלא לאפות התר ואסור בתנור אחד, ובו ז' סעיפים


א אֵין צוֹלִין בְּשַׂר כְּשֵׁרָה עִם בְּשַׂר נְבֵלָה אוֹ שֶׁל בְּהֵמָה טְמֵאָה (א) בְּתַנּוּר אֶחָד, וְאַף עַל פִּי שֶׁאֵין נוֹגְעִים זֶה בָּזֶה. וְאִם צְלָאָן, הֲרֵי זֶה (ב) מֻתָּר. (ג) וַאֲפִלּוּ הָיְתָה הָאֲסוּרָה שְׁמֵנָה הַרְבֵּה, וְהַמֻתֶּרֶת רָזָה. וְאִם הַתַּנּוּר גָּדוֹל שֶׁמַּחֲזִיק י''ב עֶשְׂרוֹנִים, וּפִיו פָּתוּחַ, מֻתָּר לִצְלוֹתָם בּוֹ, וּבִלְּבַד שֶׁלֹּא יִגְּעוּ זֶה בָּזֶה. וְאִם אֶחָד מֵהֶם מְכֻסֶה בִּקְעָרָה אוֹ בְּבָצֵק וְכַיּוֹצֵא בוֹ, מֻתָּר לִצְלוֹתָם, אֲפִלּוּ בְּתַנּוּר קָטָן וּפִיו סָתוּם. הגה: וְהוּא הַדִּין (ד) לְבָשָׂר עִם חָלָב נָמֵי דִּינָא הָכֵי. (טוּר סִימָן צ''ז) . וְנוֹהֲגִין לְהַחְמִיר לְכַתְּחִלָּה, אֲפִלּוּ בְּתַנּוּר גָּדוֹל; וּבְדִיעֲבַד, לְהָקֵל אֲפִלּוּ (ה) בְּתַנּוּר קָטָן. (אָרֹךְ כְּלָל ל''ט וּבִתְשׁוּבַת ר''י מִינְץ וּבְהַגָּהַת ש''ד וְאָגוּר בְּשֵׁם מַהֲרִי''ל וְטוּר בְּשֵׁם רַשְׁבָּ''א שֶׁכֵּן הַסְכָּמַת רֹב הַמּוֹרִים וְהוּא שִׁיטַת רַשִׁ''י וְה''ג וְרִי''ף וְרַמְבַּ''ם) . וְאִם אָפָה פַּת עִם בָּשָׂר, אָסוּר (ו) לְאָכְלוֹ עִם חָלָב, אִם יֵשׁ לוֹ פַּת אַחֵר (שָׁם) . וְכֵן אִם עוֹבֵד כּוֹכָבִים אָפָה פַּת עִם אִסּוּר, אָסוּר (ז) לִקְנוֹת אוֹתוֹ פַּת אִם יֵשׁ פַּת אַחֵר, דְּכָל זֶה מִקְרֵי לְכַתְּחִלָּה. אֲבָל אִם אֵין לוֹ פַּת אַחַר בְּרֶוַח, מֻתָּר בִּשְׁנֵיהֶם, דְּזֶה מִקְרֵי לְעִנְיָן זֶה דִּיעֲבַד. (אָגוּר בְּשֵׁם ר''י מוּלִין) . יֵשׁ אוֹמְרִים דְּאֵין מַתִּירִין רֵיחָא, אֲפִלּוּ בְּדִיעֲבַד, אֶלָּא אִם כֵּן הַתַּנּוּר פָּתוּחַ קְצָת מִן הַצַּד אוֹ לְמַעְלָה בְּמָקוֹם שֶׁהֶעָשָׁן יוֹצֵא (שָׁם בְּאָרֹךְ) . וּבִמְקוֹם הֶפְסֵד אֵין לְהַחְמִיר, בְּדִיעֲבַד (כִּי כֵן נִרְאֶה מֵהַפּוֹסְקִים וְכ''פ בֵּית יוֹסֵף), אֲפִלּוּ סָתוּם לְגַמְרֵי. וְאִם הָאִסּוּר דָּבָר (ח) חָרִיף, וְכָל שֶׁכֵּן אִם הַהֶתֵּר דָּבָר חָרִיף, רֵיחָא מִלְּתָא הִיא וַאֲפִלּוּ בְּדִיעֲבַד אָסוּר, אִם שְׁנֵיהֶם מְגֻלִּים. אֲבָל אִם אֶחָד מֵהֶם מְכֻסֶּה, אֲפִלּוּ בְּבָצֵק בְּעָלְמָא, (ט) מֻתָּר. (מָרְדְּכַי פ' ג''ה וְאו''ה) . אִם אָפוּ אוֹ צָלוּ אִסּוּר וְהֶתֵּר תַּחַת מַחֲבַת אַחַת מְגֻלִּין, אָסוּר (תּוֹסָפוֹת) אֲפִלּוּ בְּדִיעֲבַד (שָׁם בְּאָרֹךְ) . וְהוּא הַדִּין אִם אָפוּ בִּכְהַאי גַּוְנָא פַּת עִם בָּשָׂר, אָסוּר לְאָכְלוֹ (י) בְּחָלָב (ש''ד סִימָן ס' וּמהרא''י) . אֲבָל בָּזֶה אַחַר זֶה אֵין לָחוּשׁ (תְּשׁוּבַת מיי' סוֹף הִלְכוֹת מ''א וּתְשׁוּבַת ר''י וְאו''ה), אֶלָּא אִם כֵּן (יא) הִזִּיעַ הַמַּחֲבַת מִשְּׁנֵיהֶם, דְּאָז אָסוּר אֲפִלּוּ בָּזֶה אַחַר זֶה, אִם הָיוּ שְׁנֵיהֶם מְגֻלִּין, דְּהָוֵי כְּכִסּוּי שֶׁל קְדֵרָה. (ד''ע מִמַּשְׁמָעוּת הָרֹא''שׁ סִימָן צ''ג), כְּדִלְעֵיל סִימָן צ''ג. יֵשׁ אוֹמְרִים דְּכָל מָקוֹם דְּאָמְרִינַן רֵיחָא מִלְּתָא וְאוֹסֵר בְּדִיעֲבַד, הַיְנוּ דַּוְקָא דְּלֵיכָּא שִׁשִּׁים מִן הַהֶתֵּר נֶגֶד הָאִסּוּר. אֲבָל בִּדְאִכָּא שִׁשִּׁים מִן הַהֶתֵּר, אֲפִלּוּ בְּכָל מַה שֶּׁבַּתַּנּוּר, מְבַטֵּל הָאִסּוּר. (שָׁם בְּאָרֹךְ כְּלָל ל''ט די''ח) . (יב) וּלְצֹרֶךְ הֶפְסֵד יֵשׁ לִנְהֹג כֵּן. יֵשׁ אוֹמְרִים דְּאִסּוּר הָאוֹסֵר בְּמַשֶּׁהוּ, כְּגוֹן חָמֵץ בְּפֶסַח, רֵיחָא מִלְּתָא וְאוֹסֵר אֲפִלּוּ בְּדִיעֲבַד, אִם הַתַּנּוּר (יג) קָטָן וְהוּא סָתוּם, וְהָאִסּוּר וְהַהֶתֵּר מְגֻלִּין תּוֹךְ הַתַּנּוּר. (ד''מ בְּשֵׁם הַגָּהַת סמ''ק וּבְשֵׁם תּוֹסָפוֹת ע''ז דַּף ס''ו ע''ב) . וְיֵשׁ אוֹמְרִים שֶׁאֵין (יד) לְחַלֵּק. (ד''מ בְּשֵׁם מָרְדְּכַי) . וּבִמְקוֹם הֶפְסֵד יֵשׁ לִסְמֹךְ אַדִּבְרֵי הַמְּקִלִּין. וְעַיֵּן לְקַמָּן סוֹף סִימָן קי''ח אִם יֵשׁ לְהַחְמִיר לְכַתְּחִלָּה לִשְׁפּוֹת ב' קְדֵרוֹת, וְא' מֵהֶן שֶׁל אִסּוּר, עַל הַכִּירָה אוֹ לִצְלוֹת אִסּוּר אֵצֶל הֶתֵּר.


ב בַּמֶּה דְּבָרִים אֲמוּרִים, בְּצָלִי. אֲבָל אִם בָּא לְבַשְּׁלָם בִּקְדֵרָה, זֶה לְעַצְמוֹ וְזֶה לְעַצְמוֹ, אֲפִלּוּ בְּתַנּוּר קָטָן וּפִיו סָתוּם, מֻתָּר, וְאַף עַל פִּי שֶׁפִּי הַקְּדֵרוֹת (טו) מְגֻלֶּה. הגה: וְדַוְקָא שֶׁהַתַּנּוּר פָּתוּחַ קְצָת, אֲבָל אִם הוּא סָתוּם מִכָּל הַצְּדָדִים כְּדֶרֶךְ שֶׁמַּטְמִינִים הַחַמִּין לְצֹרֶךְ שַׁבָּת, אָסוּר. (מהרי''ו סִימָן מ''ד וְאו''ה וְהַגָּהַת ש''ד בְּשֵׁם מַהֲרָ''ם) . וַאֲפִלּוּ בְּדִיעֲבַד יֵשׁ מַחְמִירִין וְאוֹסְרִין, אִם הָאִסּוּר וְהַהֶתֵּר מְגֻלֶּה. (כָּךְ מַשְׁמָע בְּהַגָּהָת ש''ד) . וּבִמְקוֹם הֶפְסֵד (טז) מְרֻבֶּה יֵשׁ לְהָקֵל.


ג אִם יֵשׁ שַׁמְנוּנִית שֶׁל אִסוּר עַל הַמַּרְדֶּה שֶׁקּוֹרִין פאל'' א, אָסוּר לִתֵּן עָלֶיהָ הֶתֵּר כָּל הַיּוֹם. מִיהוּ כְּשֶׁאֵינָהּ בַּת יוֹמָא, מֻתָּר לְהִשְׁתַּמֵּשׁ בָּהּ, מִשּׁוּם דְּאִי (יז) אֶפְשָׁר בְּעִנְיָן אַחֵר. הגה: כָּל זְמַן שֶׁהִיא בַּת יוֹמָא לֹא מְהָנֵי בָּהּ הַגְעָלָה וְלֹא (יח) קְלִיפָה בִּכְלֵי אֻמָּנוּת (ת''ח סִימָן ק''ל וְהַגָּהַת ש''ד וּמָרְדְּכַי) .


ד פַּת חַמָּה שֶׁמֻּנָּח עַל גַּבֵּי חָבִית פְּתוּחָה שֶׁל (יט) יַיִן נֶסֶךְ, אֲסוּרָה. וְדַוְקָא אִם מֻנַּחַת נֶגֶד הַמְּגוּפָה. (אָרֹךְ כְּלָל ל''ט) . אֲבָל אִם הַפַּת צוֹנֶנֶת, אֲפִלּוּ אִם הֶחָבִית פְּתוּחָה, אוֹ פַּת (כ) חַמָּה וְחָבִית מְגוּפָה (פי' סְתוּמָה), מֻתָּר. וְאִם הָיָה פַּת שְׂעוֹרִים, אָסוּר אִם הַפַּת חַמָּה, אֲפִלּוּ חָבִית מְגוּפָה.


ה מֻתָּר לִשְׁאֹף בְּפִיו רֵיחַ יַיִן נֶסֶךְ דֶּרֶךְ נֶקֶב שֶׁבְּחָבִית, לֵידַע אִם הוּא טוֹב. הגה: אֲבָל אָסוּר (כא) לְטוֹעֲמוֹ, אַף עַל פִּי שֶׁאֵינוֹ בּוֹלְעוֹ (ריב''ש סִימָן רפ''ח) . וְאָסוּר לְזַלֵּף יַיִן נֶסֶךְ שֶׁאָסוּר (כב) בַּהֲנָאָה (בִּשְׁעָרִים בְּשֵׁם הַגָּהוֹת אֲשֵׁרִי סוֹף ע''ז) . אֲבָל מֻתָּר לְזַלֵּף סְתָם יֵינָם, דְּמֻתָּר בַּהֲנָאָה (אָרֹךְ כְּלָל ל''ט) .


ו שַׂק שֶׁל פִּלְפְּלִין וְזַנְגְּבִיל שֶׁמְּשִׂימִין בְּקַנְקַנֵּי יַיִן נֶסֶךְ, מֻתָּר לְהָרִיחַ בָּהֶם. אֲבָל (כג) לִבְשָׂמִים דְּהַבְדָּלָה, אָסוּר.


ז בְּשָׂמִים שֶׁל עֲבוֹדַת כּוֹכָבִים וְכִלְאֵי הַכֶּרֶם וְעָרְלָה, (כד) אָסוּר לְהָרִיחַ בָּהֶם.



סימן קט - דין יבש ביבש שנתערב, ובו ב' סעיפים


א חֲתִיכָה שֶׁאֵינָהּ רְאוּיָה לְהִתְכַּבֵּד, שֶׁנִּתְעָרְבָה בַּאֲחֵרוֹת מִין בְּמִינוֹ יָבֵשׁ (א) בְּיָבֵשׁ, דְּהַיְנוּ שֶׁאֵין נִבְלָל, וְהָאִסּוּר (ב) עוֹמֵד בְּעַצְמוֹ אֶלָּא שֶׁנִּתְעָרֵב וְאֵינוֹ מַכִּירוֹ (בֵּית יוֹסֵף בְּשֵׁם תְּשׁוּבַת הָרַשְׁבָּ''א וְהָרַמְבַּ''ן וְסה''ת) חַד בִּתְרֵי, בָּטִיל, וּמֻתָּר לְאָכְלָן אָדָם אֶחָד, כָּל אַחַת בִּפְנֵי (ג) עַצְמָהּ. אֲבָל לֹא יֹאכַל שְׁלָשְׁתָּם יַחַד. וְיֵשׁ מִי שֶׁאוֹסֵר לְאָכְלָם אָדָם אֶחָד א', אֲפִלּוּ זֶה אַחַר זֶה. הגה: וְכֵן יֵשׁ לִנְהֹג לְכַתְּחִלָּה. וְיֵשׁ מַחְמִירִין לְהַשְׁלִיךְ אֶחָד אוֹ לִתֵּן לְעוֹבֵד כּוֹכָבִים (הַגָּהוֹת ש''ד בְּשֵׁם מוהר''ם וְרַשִׁ''י בְּע''ז דַּף ע''ד ע''א) . וְאֵינוֹ אֶלָּא חֻמְרָא בְּעָלְמָא. וְכָל זֶה כְּשֶׁנִּתְעָרֵב בְּמִינוֹ, אֲבָל (ד) שֶׁלֹּא בְּמִינוֹ וְאֵין מַכִּירוֹ, אֲפִלּוּ יָבֵשׁ בְּיָבֵשׁ צָרִיךְ שִׁשִּׁים (טוּר בְּשֵׁם בַּעַל הַתְּרוּמָה וְעוֹד הַרְבֵּה פּוֹסְקִים) . וְאֵין חִלּוּק בְּכָל זֶה בֵּין אִם הָאִסּוּר (ה) מִדְּרַבָּנָן אוֹ מִדְּאוֹרַיְתָא (בֵּית יוֹסֵף בְּשֵׁם הַפּוֹסְקִים דְּלֹא כְּהַגָּהַת ש''ד) . וְע''ל סִימָן קכ''ב אִם נִתְעָרְבוּ כֵּלִים בְּיַחַד.


ב יָבֵשׁ בְּיָבֵשׁ שֶׁנִּתְבַּטֵּל חַד בִּתְרֵי, אִם בִּשְּׁלָן כֻּלָּן כְּאֶחָד, וַאֲפִלּוּ לֶאֱכֹל כָּל אֶחָד בִּפְנֵי עַצְמוֹ אָסוּר, אִם אֵין שָׁם שִׁשִּׁים, מִפְּנֵי שֶׁהָרֹטֶב בְּנוֹתֵן טַעַם וְנִבְלָע בַּחֲתִיכוֹת. וְאִם רוֹצֶה לְבַשְּׁלָן יַחַד וְאֵין שָׁם ס', מֻתָּר (ו) לְהַרְבּוֹת עֲלֵיהֶם כְּדֵי שִׁעוּר שִׁשִּׁים וּלְבַשְּׁלָן, וְאֵין בּוֹ (ז) מִשּׁוּם מְבַטֵּל אִסוּר. הגה: וְיֵשׁ אוֹמְרִים דְּאִם נוֹדַע הַתַּעֲרוֹבוֹת קֹדֶם שֶׁנִּתְבַּשְּׁלוּ בְּיַחַד, הַכֹּל שָׁרֵי, דְּאֵין חוֹזְרִין וְנֶאֱסָרִין מֵאַחַר שֶׁנִּתְבַּטֵּל בְּיָבֵשׁ (טוּר בְּשֵׁם הָרֹא''שׁ וְרַבֵּנוּ יְרוּחָם בְּשֵׁם הַתּוֹסָפוֹת וְכָתַב שֶׁכֵּן עִקָר) . וּבִמְקוֹם הֶפְסֵד יֵשׁ לִסְמֹךְ אַדִּבְרֵי הַמְּקִלִּין וּלְהַתִּיר.



סימן קי - דין ספק טרפות שארע בבשר, ובו י' סעיפים


א דָּבָר חָשׁוּב אוֹסֵר בְּמִינוֹ בְּכָל (א) שֶׁהוּא, וְהֵם ז' דְּבָרִים, וְאֵלּוּ הֵם: אֱגוֹזֵי פֶּרֶךְ וְרִמּוֹנֵי (ב) בָּדָן (ג) וְחָבִיוֹת סְתוּמוֹת וְחָלְפוֹת תְּרָדִין וְקִלְחֵי כְּרוּב וּדְלַעַת יְוָנִית וְכִכָּרוֹת שֶׁל בַּעַל הַבַּיִת. וְכֵן בַּעֲלֵי חַיִּים חֲשׁוּבִים הֵם וְאֵינָם בְּטֵלִים. אֲבָל שְׁאָר דְּבָרִים, אַף עַל פִּי שֶׁדַּרְכָּן (ד) לִמְנוֹת, הֲרֵי אֵלּוּ עוֹלִים בְּשִׁעוּרָן. הגה: וְיֵשׁ אוֹמְרִים דְּכָל דָּבָר שֶׁבְּמִנְיָן, דְּהַיְנוּ שֶׁדַּרְכּוֹ (ה) לִמְנוֹתוֹ תָּמִיד, אֵינוֹ בָּטֵל (טוּר בְּשֵׁם ר''י וּבֵית יוֹסֵף) . וְכֵן נוֹהֲגִין. הָא דְּדָבָר חָשׁוּב אֵינוֹ בָּטִיל, אֵינוֹ אֶלָּא מִדְּרַבָּנָן וְאַזְלִינָן (ו) בִּסְפֵקוֹ לְקֻלָּא. (אָרֹךְ כְּלָל כ''ה וְהָרַמְבַּ''ם פט''ז דְּמַאֲכָלוֹת אֲסוּרוֹת וְתוֹסָפוֹת דִּזְבָחִים סוֹף דַּף ע''ג וַאֲגֻדָּה שָׁם וְרֹא''שׁ פֶּרֶק ג''ה וְרַשְׁבָּ''א בת''ה ונ''י פ' הֶעָרֵל וְעוֹד הַרְבֵּה פּוֹסְקִים כְּמוֹ שֶׁיִּתְבָּאֵר בְּסי' זֶה בְּכַמָּה דּוּכְתֵי) . כָּל דָּבָר שֶׁהוּא חָשׁוּב אֵצֶל בְּנֵי מָקוֹם מֵהַמְּקוֹמוֹת, כְּגוֹן אֱגוֹזֵי פֶרֶךְ וְרִמּוֹנֵי בָּדָן בְּאֶרֶץ יִשְׂרָאֵל בְּאוֹתָם הַזְּמַנִּים, הוּא אוֹסֵר בְּכָל שֶׁהוּא, לְפִי חֲשִׁיבוּתוֹ בְּאוֹתוֹ מָקוֹם וּבְאוֹתוֹ זְמַן, וְלֹא הֻזְכְּרוּ אֵלּוּ, אֶלָּא לְפִי שֶׁהֵן אוֹסְרִים בְּכָל שֶׁהֵן בְּכָל מָקוֹם. וְהוּא הַדִּין בְּכָל כַּיּוֹצֵא בָהֶן, בִּשְׁאָר מְקוֹמוֹת.


ב בַּעֲלֵי חַיִּים שֶׁנִּתְעָרְבוּ בַּאֲחֵרִים, וְנִשְׁחֲטוּ, בָּטֵל חֲשִׁיבוּתָן וּבְטֵלִין; וְהוּא שֶׁנִּשְׁחֲטוּ (ז) בְּשׁוֹגֵג. וְדַוְקָא בַּעֲלֵי חַיִּים קְטַנִּים, שֶׁאֵינָם רְאוּיִם (ח) לְהִתְכַּבֵּד לְאַחַר שְׁחִיטָתָן (אָרֹךְ כ' כה) .


ג ט' חֲנֻיּוֹת מוֹכְרוֹת בְּשַׂר שְׁחוּטָה, וְאַחַת מוֹכֶרֶת בְּשַׂר נְבֵלָה, לָקַח מֵאַחַת מֵהֶן וְאֵין יָדוּעַ מֵאֵיזֶה מֵהֶן לָקַח, הֲרֵי זֶה אָסוּר, שֶׁכָּל (ט) קָבוּעַ כְּמֶחֱצָה עַל מֶחֱצָה דָּמִי. אֲבָל בָּשָׂר הַנִּמְצָא בַּשּׁוּק, אוֹ בְּיַד עוֹבֵד כּוֹכָבִים, (י) מֻתָּר, כֵּיוָן שֶׁרֹב הַחֲנֻיּוֹת מוֹכְרוֹת בְּשַׂר שְׁחוּטָה, דְּכָל דְּפָרִישׁ מֵרֻבָּא פָּרִישׁ. זֶהוּ דִּין תּוֹרָה, אֲבָל חֲכָמִים (יא) אֲסָרוּהוּ, אַף עַל פִּי שֶׁכָּל הַשּׁוֹחֲטִים וְכָל הַמּוֹכְרִים יִשְׂרָאֵל. הגה: וְעַיֵּן לְעֵיל סִימָן ס''ג. וְהָא דְּאָמְרִינַן כָּל דְּפָרִישׁ מֵרֻבָּא פָּרִישׁ, הַיְנוּ שֶׁלֹּא פֵּרַשׁ (יב) לְפָנֵינוּ, אֲבָל אִם פֵּרַשׁ לְפָנֵינוּ אוֹ שֶׁרוֹאֶה כְּשֶׁהַעוֹבֵד כּוֹכָבִים לְקָחוֹ, הָוֵי כְּאִלּוּ לְקָחוֹ מִשָּׁם בְּיָדוֹ (טוּר בְּשֵׁם רַשְׁבָּ''א וְתוס' פֶּרֶק ג''ה דַּף צ''ה וְרַ''ן שָׁם וְהג''א בְּשֵׁם ר''י וְתוס' וַאֲגֻדָּה פ' הַתַּעֲרוֹבוֹת וְרֹא''שׁ בִּתְשׁוּבָה כְּלָל כ' סִימָן י''ז וּסְמַ''ג סוֹף דַּף נ''ב וּסְמַ''ק סִימָן רי''ד וְש''ד סִימָן מ''ו וְר' יְרוּחָם נט''ו והמ''מ פ''ה מהמ''א בְּשֵׁם הַמְפָרְשִׁים) .


ד רֹב חֲנֻיוֹת מוֹכְרוֹת בְּשַׂר שְׁחוּטָה, וּמִעוּט מוֹכְרוֹת בְּשַׂר נְבֵלָה, לָקַח מֵאַחַת מֵהֶן, וְאֵינוֹ יוֹדֵעַ מֵאֵיזֶה מֵהֶן לָקַח וְנִתְעָרְבָה בַּאֲחֵרוֹת וְאֵינָהּ נִכֶּרֶת, (יג) בְּטֵלָה בְּרֹב מִשּׁוּם סְפֵק סְפֵיקָא. הגה: וְיֵשׁ אוֹסְרִין (יד) בִּכְהַאי גַּוְנָא, מִשּׁוּם דְּכָל מָקוֹם דְּאִסּוּר בִּמְקוֹמוֹ הָוֵי כְּמֶחֱצָה עַל מֶחֱצָה, וְהוֹאִיל וְסָפֵק הָרִאשׁוֹן אָסוּר מִן הַתּוֹרָה וְאֵין כָּאן עוֹד סְפֵקָא לְהַתִּיר, רַק תַּעֲרֹבֶת, לֹא מִקְרֵי סְפֵק סְפֵיקָא (אָרֹךְ כְּלָל כ''ה) . וְהָכֵי נָהוּג.


ה מִי שֶׁלָּקַח בָּשָׂר מְהַמָּקוֹלִין (פִּי' בֵּית הַמִּטְבָּחַיִם) וַאֲפִלּוּ חֲתִיכָה הָרְאוּיָה לְהִתְכַּבֵּד, וְנִמְצֵאת טְרֵפָה בַּמָּקוֹלִין וְלֹא נוֹדְעוּ (טו) חֲתִיכוֹת הַטְּרֵפָה, וְאֵינוֹ יוֹדֵעַ מֵאֵיזוֹ לָקַח, כָּל מַה שֶּׁלָּקְחוּ מֵהַמָּקוֹלִין קֹדֶם שֶׁנִּמְצֵאת הַטְּרֵפָה, מֻתָּר, שֶׁלֹּא נָפַל הַסָפֵק בַּקָּבוּעַ אֶלָּא לְאַחַר שֶׁפֵּרַשׁ, וְכֵיוָן שֶׁהָרֹב כָּשֵׁר, מֻתָּר. אֲבָל לִקַּח מִכָּאן (טז) וַאֵילַךְ, אָסוּר, וַאֲפִלּוּ חֲתִיכָה שֶׁאֵינָהּ רְאוּיָה לְהִתְכַּבֵּד, שֶׁאֵין הַכֹּל בְּקִיאִין בָּזֶה, (יז) וְיִטְעוּ בֵּין רְאוּיָה לְהִתְכַּבֵּד לְשֶׁאֵינָהּ רְאוּיָה.


ו בַּעֲלֵי חַיִּים וּשְׁאָר דְּבָרִים הַחֲשׁוּבִים שֶׁנִּתְעָרְבוּ בְּהֶתֵּר, שֶׁאֵינָם בְּטֵלִים אֲפִלּוּ בְּאֶלֶף, אֲפִלּוּ אִם פֵּרֵשׁ אֶחָד מִן הָרֹב, אַחַר שֶׁנּוֹדַע הַתַּעֲרוֹבוֹת, אָסוּר. וְהוּא שֶׁהִפְרִישׁוֹ בְּמִתְכַּוֵּן, אֲבָל אִם פֵּרֵשׁ (יח) מִמֵּילָא, שָׁרֵי. הגה: וְהָא דְּאָסוּר אִם הִפְרִישׁוֹ בְּמִתְכַּוֵּן, הַיְנוּ שֶׁלֹּא פֵּרַשׁ רַק מְעַט וְנִשְׁאֲרוּ מִקְצָת הָאִסּוּרִים בִּמְקוֹמָם, דְּאָז חַיְשִינָן שֶׁמָּא יִקַּח מִן הַקָּבוּעַ גַּם כֵּן. אֲבָל אִם נִתְפָּרְשׁוּ כֻּלָּן בְּיַחַד, וְלֹא נִשְׁאֲרוּ בִּמְקוֹמָן, וּבִשְׁעַת עֲקִירָתָן פֵּרְשׁוּ קְצָתָן, אוֹתָן שֶׁפֵּרְשׁוּ אָז, מֻתָּרִים (דִּבְרֵי קַבָּלָה מֵחָמִיו הַגָּאוֹן מהר''ר שַׁכְנָא זַ''ל), רַק שְׁנַיִם הָאַחֲרוֹנִים אֲסוּרִים (טוּר סוֹף סִימָן נ''ז) . וְאִם נִתְעָרְבָה תַּרְנְגֹלֶת טְרֵפָה בִּכְשֵׁרוֹת, (יט) וְנִמְצָא בֵּיצָה בֵּינֵיהֶם, הַבֵּיצָה מֻתֶּרֶת אַף עַל גַּב דְּהַתַּרְנְגוֹלִים חֲשׁוּבִים וְלֹא מִתְבַּטְּלִין, לְגַבֵּי בֵּיצָה אָזְלִינָן בָּתַר רֻבָּא (אָרֹךְ כְּלָל כ''ו) .


ז דָּבָר שֶׁאֵינוֹ בָּטֵל מֵחֲמַת חֲשִׁיבוּתוֹ, כְּגוֹן בַּעֲלֵי חַיִּים וּבְרִיָּה, וַחֲתִיכָה הָרְאוּיָה לְהִתְכַּבֵּד, וְדָבָר שֶׁיֵּשׁ לוֹ מַתִּירִין, שֶׁנִּתְעָרֵב בָּאֲחֵרִים וְנֶאֱכַל אֶחָד מֵהֶם בְּשׁוֹגֵג, בֵּין שֶׁאֲכָלוֹ עַצְמוֹ בְּשׁוֹגֵג בֵּין שֶׁאֲכָלוּהוּ אֲחֵרִים בְּשׁוֹגֵג, (ת''ה סִימָן קע''ה) אוֹ נָפַל (כ) מֵעַצְמוֹ לַיָּם בְּעִנְיָן שֶׁנֶּאֱבַד מִן הָעוֹלָם, הֻתְּרוּ כָּל הָאֲחֵרִים, שֶׁאָנוּ תּוֹלִין לוֹמַר הָאִסוּר הָלַךְ לוֹ. וְדַוְקָא כְּשֶׁאוֹכֵל הַנִּשְׁאָרוֹת שְׁתַּיִם שְׁתַּיִם בְּיַחַד, דְּמִמַּה נַּפְשָׁךְ אִכָּא (כא) חֲדָא דְּהֶתֵּרָא, אֲבָל לְאָכְלָם אַחַת אַחַת, אָסוּר. הגה: וַאֲפִלּוּ לְאָכְלָם שְׁתַּיִם שְׁתַּיִם, אָסוּר לְאָדָם (כב) אֶחָד לֶאֱכֹל אֶת כֻּלָּם. וְאֲפִלּוּ (כג) ב' בְּנֵי אָדָם, אֵין לְאָכְלָם כֻּלָּם בְּבַת אַחַת (בֵּית יוֹסֵף לָדַעַת טוּר) .


ח דָּבָר שֶׁאֵינוֹ בָּטֵל מֵחֲמַת חֲשִׁיבוּתוֹ, שֶׁנִּתְעָרֵב בַּאֲחֵרִים וְנָפַל (כד) מֵהַתַּעֲרֹבֶת הַזֹּאת אֶחָד לִשְׁנַיִם אֲחֵרִים, וְנָפַל מִן הַשְּׁלֹשָׁה אֶחָד לִשְׁנַיִם אֲחֵרִים, הֲרֵי אֵלּוּ הָאֲחֵרִים מֻתָּרִים. שֶׁהֲרֵי הָאֶחָד שֶׁל תַּעֲרֹבֶת הָרִאשׁוֹנָה בָּטֵל בְּרֹב. וְאִם נָפַל אֶחָד מֵהַתַּעֲרֹבֶת הָרִאשׁוֹנָה לְאֶלֶף, כֻּלָּם אֲסוּרִים, וְלֹא נֶאֱמַר בָּטֵל בְּרֹב אֶלָּא לְהַתִּיר סְפֵק סְפֵקָן, שֶׁאִם יִפֹּל מִן הַתַּעֲרֹבֶת הַשְּׁנִיָּה לְמָקוֹם אַחֵר, אֵינוֹ אוֹסֵר. (לְשׁוֹן רַמְבַּ''ם פֶּרֶק ט''ז מהמ''א דִּין י') הגה: וּמִכָּל מָקוֹם אֵין לְאָדָם אֶחָד לֶאֱכֹל אֶת כֻּלָּן. וְדַוְקָא שֶׁהָאִסּוּר שֶׁנִּתְעָרֵב הוּא (כה) וַדַּאי אִסּוּר, אֲבָל סְפֵק אִסּוּר שֶׁנִּתְעָרֵב וְנָפַל מֵאוֹתָהּ תַּעֲרֹבֶת לְמָקוֹם אַחֵר, הַתַּעֲרֹבֶת הַשֵּׁנִית שָׁרֵי. (אָרֹךְ כְּלָל כ''ו) . יֵשׁ אוֹמְרִים דְּדָבָר שֶׁיֵּשׁ לוֹ (כו) מַתִּירִין אֵין לְהַתִּיר מִכֹּחַ סְפֵק סְפֵיקָא (שָׁם כְּלָל כ''ה), וְטוֹב לְהַחְמִיר אִם לֹא לְצֹרֶךְ, מֵאַחַר שֶׁיֵּשׁ לוֹ הֶתֵּר בְּלָאו הָכֵי.


ט סְפֵק טְרֵפָה שֶׁנִּתְעָרֵב בַּאֲחֵרוֹת, כֻּלָּן אֲסוּרוֹת עַד שֶׁיְּהֵא בְּהֶתֵּר כְּדֵי לְבַטֵּל הָאִסוּר אִם הוּא מִדְּבָרִים הַמִּתְבַּטְּלִים. שֶׁכֵּיוָן שֶׁסָפֵק הָרִאשׁוֹן הָיָה (כז) בְּגוּפוֹ, אֵין לְהַתִּירוֹ (כח) מִסְפֵק סְפֵיקָא. הגה: וְיֵשׁ אוֹמְרִים הַטַּעַם דְּאָסוּר הוֹאִיל וְהַסָּפֵק הוּא אִסּוּר דְּאוֹרַיְתָא וְלֹא נוּכַל לוֹמַר עוֹד סָפֵק, שֶׁאֵין כָּאן אִסּוּר רַק שֶׁנִּתְעָרֵב בַּאֲחֵרִים, לֹא מִקְרֵי סְפֵק סְפֵיקָא וְאָסוּר (שָׁם) . אֲבָל אִם הָיוּ ב' סְפֵקוֹת אִם הָיָה כָּאן אִסּוּר כְּלָל, (כט) וְנוֹדְעוּ ב' הַסְּפֵקוֹת בְּיַחַד, מַתִּירִים סְפֵק סְפֵיקָא בְּכָל מָקוֹם, אֲפִלּוּ בְּאִסּוּר דְּאוֹרַיְתָא וְגוּפוֹ שֶׁל אִסּוּר, וַאֲפִלּוּ הָיָה לוֹ חֶזְקַת אִסּוּר, כְּגוֹן עוֹף שֶׁבְּחֶזְקַת אִסּוּר עוֹמֵד וְנִשְׁבַּר אוֹ נִשְׁמַט גַּפּוֹ סָפֵק מֵחַיִּים אוֹ לְאַחַר שְׁחִיטָה, וְאִם תִּמְצָא לוֹמַר מֵחַיִּים, שֶׁמָּא לֹא נִקְּבָה הָרֵאָה, יֵשׁ לְהַתִּיר מִכֹּחַ סְפֵק סְפֵיקָא; אַף עַל פִּי שֶׁיֵּשׁ לְבָרֵר עַל יְדֵי בְּדִיקַת הָרֵאָה, אֵין לָחוּשׁ. (ל) וְעַיֵּן לְעֵיל סִימָן נ''ג.


י רֵאָה טְרֵפָה שֶׁנִּמְצֵאת בֵּין רֵאוֹת כְּשֵׁרוֹת, וְאָמַר הַבּוֹדֵק שֶׁהָיָה יוֹדֵעַ בְּאוֹתָהּ סִרְכָא, וּבִשְׁגָגָה נִתְעָרְבָה אוֹתָהּ רֵאָה טְרֵפָה בֵּין הַכְּשֵׁרוֹת, וְלֹא הָיְתָה בַּכְּבָשִׂים עִרְבּוּבְיָא, הַכְּבָשִׁים מֻתָּרִים.



סימן קיא - דין כלים טריפות שנתערבו בכשרות, ובו ז' סעיפים


א שְׁתֵּי קְדֵרוֹת אַחַת שֶׁל הֶתֵּר וְאַחַת שֶׁל אִסוּר, וּלְפָנָיו שְׁתֵּי חֲתִיכוֹת אַחַת שֶׁל הֶתֵּר וְאַחַת שֶׁל אִסוּר; אִם הַחֲתִיכָה הִיא מֵאִסוּר דְּרַבָּנָן, (א) כְּגוֹן שֻׁמָּנוֹ שֶׁל גִּיד, וְנָפְלוּ אֵלּוּ לְתוֹךְ אֵלּוּ, מֻתָּרִים, שֶׁאָנוּ תּוֹלִים לוֹמַר, הָאִסוּר נָפַל לְתוֹךְ הָאִסוּר, וְהַהֶתֵּר לְתוֹךְ שֶׁל הֶתֵּר. וַאֲפִלּוּ אִם הַהֶתֵּר שֶׁבַּקְּדֵרָה אֵינָהּ (ב) רַבָּה עַל שֶׁלְּפָנֶיהָ. וְכֵן הַדִּין אִם לֹא הָיָה כָּאן אֶלָּא קְדֵרָה אַחַת שֶׁל בְּשַׂר שְׁחוּטָה, וְנָפַל בָּהּ אַחַת מֵאֵלּוּ שְׁתֵּי (ג) הַחֲתִיכוֹת, וְאֵין יָדוּעַ אֵיזוֹ הִיא, אָנוּ תּוֹלִין דְּשֶׁל הֶתֵּר נָפְלָה, אֲפִלּוּ אֵין הַהֶתֵּר רַבֶּה עַל הָאִסוּר. אוֹ אִם לֹא הָיָה כָּאן אֶלָּא חֲתִיכָה אַחַת מֵאִסוּר (ד) דְּרַבָּנָן, וְיֵשׁ שְׁתֵּי קְדֵרוֹת, אַחַת שֶׁל הֶתֵּר וְאַחַת שֶׁל אִסוּר, וְאֵין יָדוּעַ לְאֵיזוֹ נָפְלָה, גַּם בָּזוֹ תּוֹלִין לְהָקֵל.


ב בַּמֶּה דְּבָרִים אֲמוּרִים, שֶׁתּוֹלִין בְּכָל אֵלּוּ לְהָקֵל, אֲפִלּוּ אֵין הַהֶתֵּר רַבֶּה עַל הָאִסוּר, שֶׁאֵין הָאִסוּר גַּם כֵּן רַבֶּה עַל הַהֶתֵּר. (ה) אֲבָל אִם הָאִסוּר רַבֶּה עַל הַהֶתֵּר, אֵין תּוֹלִין לְהָקֵל.


ג הָיָה הָאִסוּר שֶׁל תּוֹרָה, אֵין תּוֹלִין לְהָקֵל לוֹמַר לְתוֹךְ הָאִסוּר נָפַל, עַד שֶׁיְּהֵא הַהֶתֵּר רַבֶּה עַל הָאִסוּר, כְּדֵי (ו) שֶׁיִּתְבַּטֵּל בְּתוֹכוֹ מִדִּבְרֵי תּוֹרָה.


ד הָיָה הַהֶתֵּר רַבֶּה עַל הָאִסוּר, כְּדֵי שֶׁיִּתְבַּטֵּל בְּתוֹכוֹ מִדִּבְרֵי תוֹרָה, תּוֹלִין לְהָקֵל. כֵּיצַד, שְׁתֵּי קְדֵרוֹת אַחַת שֶׁל בְּשַׂר שְׁחוּטָה וְאַחַת שֶׁל נְבֵלָה, וְנָפְלָה חֲתִיכַת נְבֵלָה לְתוֹךְ אַחַת מֵהֶן, אִם הַהֶתֵּר שֶׁבִּקְדֵרַת הַהֶתֵּר רַבָּה קְצָת עַל חֲתִיכַת הַנְּבֵלָה, תּוֹלִין לְהָקֵל, לְפִי שֶׁדְּבַר תּוֹרָה מִין (ז) בְּמִינוֹ בָּטֵל בְּרֹב, אֶלָּא שֶׁחֲכָמִים הִצְרִיכוּ (ח) שִׁשִּׁים. הָיוּ שְׁתֵּי הַקְּדֵרוֹת מִמִּין אֶחָד, וְהָאִסוּר מִמִּין אַחֵר, אֵין תּוֹלִין לְהָקֵל עַד שֶׁיִּרְבֶּה הַהֶתֵּר שִׁשִּׁים עַל הָאִסוּר.


ה הָיוּ כָּאן שְׁתֵּי קְדֵרוֹת שֶׁל הֶתֵּר, וְלִפְנֵיהֶם שְׁתֵּי חֲתִיכוֹת, אַחַת שֶׁל הֶתֵּר וְאַחַת שֶׁל אִסוּר, וְנָפְלָה אַחַת לְתוֹךְ זוֹ וְאַחַת לְתוֹךְ זוֹ, (ט) אֲסוּרוֹת שְׁתֵּיהֶן, אֲפִלּוּ בְּאִסוּר דְּרַבָּנָן, אִם אֵין בְּשׁוּם אַחַת כְּדֵי לְבַטֵּל הָאִסוּר. אֲבָל אִם יֵשׁ בְּאַחַת מֵהֶן לְבַטֵּל הָאִסוּר, שְׁתֵּיהֶן (י) מֻתָּרוֹת.


ו הָיוּ כָּאן שְׁתֵּי קְדֵרוֹת שֶׁל הֶתֵּר, וְנָפַל אִסוּר לְתוֹךְ אַחַת מֵהֶן, וְיָדוּעַ לְאֵיזוֹ נָפַל, וְחָזַר וְנָפַל אִסוּר לְתוֹךְ אַחַת מֵהֶן, וְאֵין יָדוּעַ לְאֵיזוֹ נָפַל, אֲנִי אוֹמֵר לִמְקוֹם אִסוּר הָרִאשׁוֹן נָפַל גַּם (יא) הַשֵּׁנִי. אֲבָל אִם נָפַל אִסוּר לְתוֹךְ אַחַת מֵהֶן, וְאֵין יָדוּעַ לְאֵיזוֹ נָפַל, וְאַחַר כָּךְ נָפַל אִסוּר לְתוֹךְ אַחַת מֵהֶן וְיָדוּעַ לְאֵיזוֹ נָפַל, שְׁתֵּיהֶן (יב) אֲסוּרוֹת. הגה: וְדַוְקָא שֶׁאֵין ס' לְבַטֵּל הָאִסּוּר שֶׁנָּפַל שָׁם, אֲבָל אִם הָיָה ס' לְבַטֵּל הָאִסּוּר שֶׁנָּפַל שָׁם, הָוֵי כְּאִלּוּ לֹא נָפַל שָׁם אִסּוּר כְּלָל (בֵּית יוֹסֵף בְּשֵׁם הָרַשְׁבָּ''א) .


ז הָיָה כָּאן שְׁתֵּי קְדֵרוֹת שֶׁל הֶתֵּר, וְנָפַל אִסוּר לְתוֹךְ אַחַת מֵהֶן וְאֵין יָדוּעַ לְאֵיזוֹ נָפַל, וְאֵין בְּאַחַת כְּדֵי לְבַטֵּל הָאִסוּר, וְיֵשׁ בִּשְׁתֵּיהֶן כְּדֵי לְבַטְּלוֹ, שְׁתֵּיהֶן (יג) מִצְטָרְפוֹת לְבַטְּלוֹ. וְלֹא עוֹד, אֶלָּא אֲפִלּוּ אַחַת בַּבַּיִת וְאַחַת בָּעֲלִיָּה, מִצְטָרְפוֹת. וְהוּא הַדִּין אֲפִלּוּ לְמֵאָה. בַּמֶּה דְּבָרִים אֲמוּרִים, כְּשֶׁשְּׁתֵּיהֶן שֶׁל אָדָם אֶחָד, לְפִי שֶׁכָּל שֶׁהֵן שֶׁל אָדָם אֶחָד (יד) עָתִיד לְהִתְעָרֵב. אֲבָל אִם הֵם שֶׁל שְׁנֵי בְּנֵי אָדָם, אֵין מִצְטָרְפוֹת. הגה: וְיֵשׁ מַחְמִירִים דַּאֲפִילוּ בְּאָדָם אֶחָד אֵין לְהָקֵל. וּלְצֹרֶךְ הֶפְסֵד יֵשׁ (טו) לְהַתִּיר בְּאָדָם אֶחָד; וּמִיהוּ אֵין לְאָכְלוֹ אֶלָּא לְאַחַר (טז) שֶׁיְּעָרְבֶנּוּ יַחַד, דְּאָז כְּבַר נִתְבַּטֵּל הָאִסּוּר (בְּאָרֹךְ כְּלָל כ''ו בְּשֵׁם מהרי''ש) .
