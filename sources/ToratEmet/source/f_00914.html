פרק א - משנה א

אֵלּוּ דְבָרִים {א} שֶׁאֵין לָהֶם שִׁעוּר. הַפֵּאָה {ב}, וְהַבִּכּוּרִים, וְהָרֵאָיוֹן, וּגְמִילוּת חֲסָדִים, וְתַלְמוּד תּוֹרָה. אֵלּוּ דְבָרִים שֶׁאָדָם אוֹכֵל {ה} פֵּרוֹתֵיהֶן בָּעוֹלָם הַזֶּה וְהַקֶּרֶן {ו} קַיֶּמֶת לוֹ לָעוֹלָם הַבָּא. כִּבּוּד אָב וָאֵם {ז}, וּגְמִילוּת חֲסָדִים, וַהֲבָאַת שָׁלוֹם בֵּין אָדָם לַחֲבֵרוֹ וְתַלְמוּד תּוֹרָה {ח} כְּנֶגֶד כֻּלָּם:


פרק א - משנה ב

אֵין פּוֹחֲתִין לַפֵּאָה מִשִּׁשִּׁים, וְאַף עַל פִּי שֶׁאָמְרוּ אֵין לַפֵּאָה שִׁעוּר. הַכֹּל לְפִי גֹדֶל הַשָּׂדֶה, וּלְפִי רֹב הָעֲנִיִּים, וּלְפִי רֹב הָעֲנָוָה:


פרק א - משנה ג

נוֹתְנִין {י} פֵּאָה מִתְּחִלַּת הַשָּׂדֶה וּמֵאֶמְצָעָהּ. רַבִּי שִׁמְעוֹן אוֹמֵר, וּבִלְבַד שֶׁיִּתֵּן בַּסּוֹף כַּשִּׁעוּר. רַבִּי יְהוּדָה אוֹמֵר, אִם שִׁיֵּר קֶלַח אֶחָד, סוֹמֵךְ לוֹ מִשּׁוּם פֵּאָה. וְאִם לָאו, אֵינוֹ נוֹתֵן אֶלָּא מִשּׁוּם הֶפְקֵר:


פרק א - משנה ד

כְּלָל אָמְרוּ בַּפֵּאָה. כָּל שֶׁהוּא אֹכֶל, וְנִשְׁמָר, וְגִדּוּלָיו מִן הָאָרֶץ, וּלְקִיטָתוֹ כְאַחַת, וּמַכְנִיסוֹ לְקִיּוּם, חַיָּב בַּפֵּאָה. וְהַתְּבוּאָה {יח} וְהַקִּטְנִיּוֹת בַּכְּלָל הַזֶּה:


פרק א - משנה ה

וּבָאִילָן, הָאוֹג וְהֶחָרוּבִין וְאֱגוֹזִים וְהַשְּׁקֵדִים וְהַגְּפָנִים וְהָרִמּוֹנִים וְהַזֵּיתִים וְהַתְּמָרִים, חַיָּבִין בַּפֵּאָה:


פרק א - משנה ו

לְעוֹלָם הוּא נוֹתֵן מִשּׁוּם פֵּאָה וּפָטוּר מִן הַמַּעַשְׂרוֹת, עַד שֶׁיְּמַרֵח. וְנוֹתֵן מִשּׁוּם הֶפְקֵר וּפָטוּר מִן הַמַּעַשְׂרוֹת, עַד שֶׁיְּמַרֵח. וּמַאֲכִיל לַבְּהֵמָה וְלַחַיָּה וְלָעוֹפוֹת וּפָטוּר מִן הַמַּעַשְׂרוֹת, עַד שֶׁיְּמַרֵח. וְנוֹטֵל מִן הַגֹּרֶן וְזוֹרֵעַ וּפָטוּר מִן הַמַּעַשְׂרוֹת, עַד שֶׁיְּמַרֵח, דִּבְרֵי רַבִּי עֲקִיבָא. כֹּהֵן וְלֵוִי שֶׁלָּקְחוּ אֶת הַגֹּרֶן, הַמַּעַשְׂרוֹת שֶׁלָּהֶם, עַד שֶׁיְּמַרֵח. הַמַּקְדִּישׁ וּפוֹדֶה, חַיָּב בְּמַעַשְׂרוֹת, עַד שֶׁיְּמַרֵח הַגִּזְבָּר:


פרק ב

פרק ב - משנה א

וְאֵלּוּ מַפְסִיקִין לַפֵּאָה. הַנַּחַל, וְהַשְּׁלוּלִית {א}, וְדֶרֶךְ הַיָּחִיד, וְדֶרֶךְ הָרַבִּים, וּשְׁבִיל הָרַבִּים, וּשְׁבִיל הַיָּחִיד הַקָּבוּעַ {ב} בִּימוֹת הַחַמָּה וּבִימוֹת הַגְּשָׁמִים, וְהַבּוּר, וְהַנִּיר, וְזֶרַע אַחֵר. וְהַקּוֹצֵר לְשַׁחַת מַפְסִיק, דִּבְרֵי רַבִּי מֵאִיר. וַחֲכָמִים אוֹמְרִים, אֵינוֹ מַפְסִיק, אֶלָּא אִם כֵּן חָרָשׁ {ד}:


פרק ב - משנה ב

אַמַּת הַמַּיִם שֶׁאֵינָהּ יְכוֹלָה לְהִקָּצֵר כְּאַחַת, רַבִּי יְהוּדָה אוֹמֵר, מַפְסֶקֶת. וְכֹל הֶהָרִים אֲשֶׁר בַּמַּעְדֵּר יֵעָדֵרוּן, אַף עַל פִּי שֶׁאֵין הַבָּקָר יָכוֹל לַעֲבֹר בְּכֵלָיו, הוּא נוֹתֵּן פֵּאָה לַכֹּל:


פרק ב - משנה ג

הַכֹּל מַפְסִיק לִזְרָעִים, וְאֵינוֹ מַפְסִיק לְאִילָן {ו} אֶלָּא גָדֵר. וְאִם הָיָה שֵׂעָר {ז} כּוֹתֵשׁ, אֵינוֹ מַפְסִיק, אֶלָּא נוֹתֵן פֵּאָה לַכֹּל:


פרק ב - משנה ד

וְלֶחָרוּבִין {ח} כָּל הָרוֹאִין זֶה אֶת זֶה. אָמַר רַבָּן גַּמְלִיאֵל, נוֹהֲגִין הָיוּ בֵּית אַבָּא, נוֹתְנִין פֵּאָה אַחַת לַזֵּיתִים שֶׁהָיוּ לָהֶם בְּכָל רוּחַ, וְלֶחָרוּבִין כָּל הָרוֹאִין זֶה אֶת זֶה. רַבִּי אֱלִיעֶזֶר בְּרַבִּי צָדוֹק אוֹמֵר מִשְּׁמוֹ, אַף לֶחָרוּבִין שֶׁהָיוּ לָהֶם בְּכָל הָעִיר:


פרק ב - משנה ה

הַזּוֹרֵעַ אֶת שָׂדֵהוּ מִין אֶחָד, אַף עַל פִּי שֶׁהוּא עוֹשֵׂהוּ שְׁתֵּי גְרָנוֹת, נוֹתֵן פֵּאָה אֶחָת. זְרָעָהּ שְׁנֵי מִינִין, אַף עַל פִּי שֶׁעֲשָׂאָן גֹּרֶן אַחַת, נוֹתֵן שְׁתֵּי פֵאוֹת. הַזּוֹרֵעַ אֶת שָׂדֵהוּ שְׁנֵי מִינֵי חִטִין, עֲשָׂאָן גֹּרֶן אַחַת, נוֹתֵן פֵּאָה אַחַת שְׁתֵּי גְרָנוֹת, נוֹתֵן שְׁתֵּי פֵאוֹת:


פרק ב - משנה ו

מַעֲשֶׂה שֶׁזָּרַע רַבִּי שִׁמְעוֹן אִישׁ הַמִּצְפָּה לִפְנֵי רַבָּן גַּמְלִיאֵל, וְעָלוּ לְלִשְׁכַּת הַגָּזִית וְשָׁאֲלוּ. אָמַר נַחוּם הַלַּבְלָר, מְקֻבָּל אֲנִי מֵרַבִּי מְיָאשָׁא, שֶׁקִּבֵּל מֵאַבָּא, שֶׁקִּבֵּל מִן הַזּוּגוֹת, שֶׁקִּבְּלוּ מִן הַנְּבִיאִים, הֲלָכָה לְמשֶׁה מִסִּינַי, בְּזוֹרֵעַ אֶת שָׂדֵהוּ שְׁנֵי מִינֵי חִטִין, אִם עֲשָׂאָן גֹּרֶן אַחַת {י}, נוֹתֵן פֵּאָה אַחַת. שְׁתֵּי גְרָנוֹת, נוֹתֵן שְׁתֵּי פֵאוֹת:


פרק ב - משנה ז

שָׂדֶה שֶׁקְּצָרוּהָ גוֹיִם, קְצָרוּהָ לִסְטִים, קִרְסְמוּהָ נְמָלִים, שְׁבָרַתָּהּ הָרוּחַ אוֹ בְהֵמָה, פְּטוּרָה. קָצַר חֶצְיָהּ וְקָצְרוּ לִסְטִים חֶצְיָהּ, פְּטוּרָה, שֶׁחוֹבַת הַפֵּאָה בַּקָּמָה:


פרק ב - משנה ח

קְצָרוּהָ לִסְטִים חֶצְיָהּ וְקָצַר הוּא חֶצְיָהּ, נוֹתֵן פֵּאָה מִמַּה שֶּׁקָּצַר. קָצַר חֶצְיָהּ וּמָכַר חֶצְיָהּ, הַלּוֹקֵחַ נוֹתֵן פֵּאָה לַכֹּל. קָצַר חֶצְיָהּ וְהִקְדִּישׁ חֶצְיָהּ, הַפּוֹדֶה מִיַּד הַגִּזְבָּר, הוּא נוֹתֵן פֵּאָה לַכֹּל:


פרק ג

פרק ג - משנה א

מַלְבְּנוֹת הַתְּבוּאָה שֶׁבֵּין הַזֵּיתִים, בֵּית שַׁמַּאי אוֹמְרִים, פֵּאָה מִכָּל אֶחָד וְאֶחָד. בֵּית הִלֵּל אוֹמְרִים, מֵאֶחָד עַל הַכֹּל. וּמוֹדִים {א}, שֶׁאִם הָיוּ רָאשֵׁי שׁוּרוֹת {ב} מְעֹרָבִין, שֶׁהוּא נוֹתֵן פֵּאָה מֵאֶחָד עַל הַכֹּל:


פרק ג - משנה ב

הַמְנַמֵּר אֶת שָׂדֵהוּ וְשִׁיֵּר קְלָחִים לַחִים, רַבִּי עֲקִיבָא אוֹמֵר, נוֹתֵן פֵּאָה מִכָּל אֶחָד וְאֶחָד. וַחֲכָמִים אוֹמְרִים, מֵאֶחָד עַל הַכֹּל. וּמוֹדִים חֲכָמִים לְרַבִּי עֲקִיבָא בְּזוֹרֵעַ שֶׁבֶת אוֹ חַרְדָּל בִּשְׁלשָׁה מְקוֹמוֹת, שֶׁהוּא נוֹתֵן פֵּאָה מִכָּל אֶחָד וְאֶחָד {ה}:


פרק ג - משנה ג

הַמַּחֲלִיק בְּצָלִים לַחִים לַשּׁוּק וּמְקַיֵּם יְבֵשִׁים לַגֹּרֶן, נוֹתֵן {ו} פֵּאָה לְאֵלּוּ לְעַצְמָן וּלְאֵלּוּ לְעַצְמָן. וְכֵן בַּאֲפוּנִין {ז}, וְכֵן בְּכָרֶם. הַמֵּדֵל, נוֹתֵן מִן הַמְשֹׁאָר עַל מַה שֶּׁשִּׁיֵּר. וְהַמַּחֲלִיק מֵאַחַת יַד, נוֹתֵן מִן הַמְשֹׁאָר עַל הַכֹּל:


פרק ג - משנה ד

הָאִמָּהוֹת שֶׁל בְּצָלִים חַיָּבוֹת בְּפֵּאָה, וְרַבִּי יוֹסֵי פּוֹטֵר. מַלְבְּנוֹת הַבְּצָלִים שֶׁבֵּין הַיָּרָק, רַבִּי יוֹסֵי אוֹמֵר, פֵּאָה מִכָּל אֶחָד וְאֶחָד. וַחֲכָמִים אוֹמְרִים, מֵאֶחָד עַל הַכֹּל:


פרק ג - משנה ה

הָאַחִין שֶׁחָלְקוּ, נוֹתְנִין שְׁתֵּי פֵאוֹת. חָזְרוּ וְנִשְׁתַּתְּפוּ, נוֹתְנִין פֵּאָה אֶחָת. שְׁנַיִם שֶׁלָּקְחוּ אֶת הָאִילָן, נוֹתְנִין פֵּאָה אֶחָת. לָקַח זֶה צְפוֹנוֹ וְזֶה דְרוֹמוֹ, זֶה נוֹתֵן פֵּאָה לְעַצְמוֹ, וְזֶה נוֹתֵן פֵּאָה לְעַצְמוֹ. הַמּוֹכֵר קִלְחֵי אִילָן בְּתוֹךְ שָׂדֵהוּ, נוֹתֵן פֵּאָה מִכָּל אֶחָד וְאֶחָד. אָמַר רַבִּי יְהוּדָה, אֵימָתַי. בִּזְמַן שֶׁלֹּא שִׁיֵּר בַּעַל הַשָּׂדֶה. אֲבָל אִם שִׁיֵּר בַּעַל הַשָּׂדֶה, הוּא נוֹתֵן פֵּאָה לַכֹּל:


פרק ג - משנה ו

רַבִּי אֱלִיעֶזֶר אוֹמֵר, קַרְקַע בֵּית רֹבַע, חַיָּב בַּפֵּאָה. רַבִּי יְהוֹשֻׁעַ אוֹמֵר, הָעוֹשֶׂה סָאתַיִם {יג}. רַבִּי טַרְפוֹן אוֹמֵר, שִׁשָּׁה עַל שִׁשָּׁה טְפָחִים. רַבִּי יְהוּדָה בֶן בְּתֵירָה אוֹמֵר, כְּדֵי לִקְצֹר וְלִשְׁנוֹת. וַהֲלָכָה כִּדְבָרָיו. רַבִּי עֲקִיבָא אוֹמֵר, קַרְקַע כָּל שֶׁהוּא, חַיָּב בַּפֵּאָה, וּבַבִּכּוּרִים, וְלִכְתֹּב עָלָיו פְּרוֹזְבּוּל, וְלִקְנוֹת עִמּוֹ נְכָסִים שֶׁאֵין לָהֶם אַחֲרָיוּת בְּכֶסֶף וּבִשְׁטָר וּבַחֲזָקָה:


פרק ג - משנה ז

נָטַל מִקְצָת פֵּאָה וּזְרָקָהּ עַל הַשְּׁאָר, אֵין לוֹ בָה כְּלוּם. נָפַל לוֹ עָלֶיהָ, וּפֵרֵשׂ {י} טַלִּיתוֹ עָלֶיהָ, מַעֲבִירִין אוֹתוֹ הֵימֶנָּה. וְכֵן בְּלֶקֶט, וְכֵן בְּעֹמֶר {יא} הַשִּׁכְחָה:


פרק ד - משנה ד

פֵּאָה אֵין קוֹצְרִין אוֹתָהּ בְּמַגָּלוֹת {יב}, וְאֵין עוֹקְרִין אוֹתָהּ בְּקַרְדֻּמּוֹת, כְּדֵי שֶׁלֹּא יַכּוּ אִישׁ אֶת רֵעֵהוּ {יג}:


פרק ד - משנה ה

שָׁלשׁ אַבְעָיוֹת בַּיּוֹם, בַּשַׁחַר וּבַחֲצוֹת וּבַמִּנְחָה. רַבָּן גַּמְלִיאֵל אוֹמֵר, לֹא אָמְרוּ אֶלָּא כְּדֵי שֶׁלֹּא יִפְחֲתוּ. רַבִּי עֲקִיבָא אוֹמֵר, לֹא אָמְרוּ אֶלָּא כְּדֵי שֶׁלֹּא יוֹסִיפוּ. שֶׁל בֵּית נָמֵר הָיוּ מְלַקְּטִין עַל הַחֶבֶל, וְנוֹתְנִים פֵּאָה מִכָּל אוֹמָן וְאוֹמָן:


פרק ד - משנה ו

נָכְרִי שֶׁקָּצַר אֶת שָׂדֵהוּ וְאַחַר כָּךְ נִתְגַּיֵּר, פָּטוּר מִן הַלֶּקֶט {יז} וּמִן הַשִּׁכְחָה וּמִן הַפֵּאָה. רַבִּי יְהוּדָה מְחַיֵּב בְּשִׁכְחָה, שֶׁאֵין הַשִּׁכְחָה אֶלָּא בִשְׁעַת הָעִמּוּר:

הִקְדִּישׁ קָמָה וּפָדָה קָמָה, חַיָּב. עֳמָרִין וּפָדָה עֳמָרִין {יח}, חַיָּב. קָמָה וּפָדָה עֳמָרִין, פְּטוּרָה, שֶׁבִּשְׁעַת חוֹבָתָהּ הָיְתָה פְטוּרָה:


פרק ד - משנה ח

כַּיּוֹצֵא בוֹ, הַמַּקְּדִּישׁ פֵּרוֹתָיו עַד שֶׁלֹּא בָאוּ לְעוֹנַת הַמַּעַשְׂרוֹת, וּפְדָאָן, חַיָּבִין. מִשֶּׁבָּאוּ לְעוֹנַת הַמַּעַשְׂרוֹת, וּפְדָאָן, חַיָּבִין. הִקְדִּישָׁן עַד שֶׁלֹּא נִגְמְרוּ וּגְמָרָן הַגִּזְבָּר, וְאַחַר כָּךְ פְּדָאָן, פְּטוּרִים, שֶׁבִּשְׁעַת חוֹבָתָן הָיוּ פְטוּרִים:


פרק ד - משנה ט

מִי שֶׁלָּקַט אֶת הַפֵּאָה וְאָמַר הֲרֵי זוֹ לְאִישׁ פְּלוֹנִי עָנִי, רַבִּי אֱלִיעֶזֶר אוֹמֵר, זָכָה לוֹ. וַחֲכָמִים אוֹמְרִים, יִתְּנֶנָּה לְעָנִי שֶׁנִּמְצָא רִאשׁוֹן {כ}. הַלֶּקֶט וְהַשִּׁכְחָה וְהַפֵּאָה שֶׁל נָכְרִי חַיָּב בְּמַעַשְׂרוֹת, אֶלָּא אִם כֵּן הִפְקִיר:


פרק ד - משנה י

אֵיזֶהוּ לֶקֶט. הַנּוֹשֵׁר בִּשְׁעַת הַקְּצִירָה. קָצַר מְלֹא יָדוֹ, תָּלַשׁ מְלֹא קֻמְצוֹ, הִכָּהוּ קוֹץ וְנָפַל מִיָּדוֹ לָאָרֶץ, הֲרֵי הוּא שֶׁל בַּעַל הַבָּיִת. תּוֹךְ הַיָּד וְתוֹךְ הַמַּגָּל לָעֲנִיִּים. אַחַר הַיָּד וְאַחַר הַמַּגָּל, לְבַעַל הַבָּיִת. רֹאשׁ הַיָּד וְרֹאשׁ הַמַּגָּל, רַבִּי יִשְׁמָעֵאל אוֹמֵר, לָעֲנִיִּים. רַבִּי עֲקִיבָא אוֹמֵר, לְבַעַל הַבָּיִת:


פרק ד - משנה יא

פרק ה - משנה א

גָּדִישׁ שֶׁלֹּא לֻקַּט תַּחְתָּיו, כָּל הַנּוֹגֵעַ בָּאָרֶץ {א} הֲרֵי הוּא שֶׁל עֲנִיִּים. הָרוּחַ שֶׁפִּזְּרָה אֶת הָעֳמָרִים, אוֹמְדִים אוֹתָהּ כַּמָּה לֶקֶט הִיא רְאוּיָה לַעֲשׂוֹת, וְנוֹתֵן לָעֲנִיִּים {ג}. רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר, נוֹתֵן לָעֲנִיִּים בִּכְדֵי נְפִילָה:

פרק ה - משנה ד

פרק ה - משנה ה
