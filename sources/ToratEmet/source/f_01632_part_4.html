פרשת ראה


{כו} רְאֵה אָנֹכִי נֹתֵן לִפְנֵיכֶם הַיּוֹם בְּרָכָה וּקְלָלָה:  מ  רְאֵה אָנֹכִי נֹתֵן לִפְנֵיכֶם הַיּוֹם בְּרָכָה וּקְלָלָה:  ת  חֲזֵי דִי אֲנָא יָהֵב קֳדָמֵיכוֹן יוֹמָא דֵין בִּרְכָן וּלְוָטִין:

{כז} אֶת הַבְּרָכָה אֲשֶׁר תִּשְׁמְעוּ אֶל מִצְוֹת יְהוָה אֱלֹהֵיכֶם אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם הַיּוֹם:  מ  אֶת הַבְּרָכָה אֲשֶׁר תִּשְׁמְעוּ אֶל מִצְוֹת יְהוָה אֱלֹהֵיכֶם אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם הַיּוֹם:  ת  יָת בִּרְכָן דִי תְקַבְּלוּן לְפִקוּדַיָא דַיְיָ אֳלָהֲכוֹן דִי אֲנָא מְפַקֵד יָתְכוֹן יוֹמָא דֵין:

{כח} וְהַקְּלָלָה אִם לֹא תִשְׁמְעוּ אֶל מִצְוֹת יְהוָה אֱלֹהֵיכֶם וְסַרְתֶּם מִן הַדֶּרֶךְ אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם הַיּוֹם לָלֶכֶת אַחֲרֵי אֱלֹהִים אֲחֵרִים אֲשֶׁר לֹא יְדַעְתֶּם:  מ  וְהַקְּלָלָה אִם לֹא תִשְׁמְעוּ אֶל מִצְוֹת יְהוָה אֱלֹהֵיכֶם וְסַרְתֶּם מִן הַדֶּרֶךְ אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם הַיּוֹם לָלֶכֶת אַחֲרֵי אֱלֹהִים אֲחֵרִים אֲשֶׁר לֹא יְדַעְתֶּם:  ת  וּלְוָטִין אִם לָא תְקַבְּלוּן לְפִקוּדַיָא דַיְיָ אֱלָהֲכוֹן וְתִסְטוּן מִן אָרְחָא דִי אֲנָא מְפַקֵד יָתְכוֹן יוֹמָא דֵין לִמְהַךְ בָּתַר טַעֲוַת עַמְמַיָא דִי לָא יְדַעְתּוּן: (ס)

{כט} וְהָיָה כִּי יְבִיאֲךָ יְהוָה אֱלֹהֶיךָ אֶל הָאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ וְנָתַתָּה אֶת הַבְּרָכָה עַל הַר גְּרִזִים וְאֶת הַקְּלָלָה עַל הַר עֵיבָל:  מ  וְהָיָה כִּי יְבִיאֲךָ יְהוָה אֱלֹהֶיךָ אֶל הָאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ וְנָתַתָּה אֶת הַבְּרָכָה עַל הַר גְּרִזִים וְאֶת הַקְּלָלָה עַל הַר עֵיבָל:  ת  וִיהֵי אֲרֵי יָעֵלִנָךְ יְיָ אֱלָהָךְ לְאַרְעָא דִי אַתְּ עָלֵל תַּמָן לְמֵירְתַהּ וְתִתֵּן יָת מְבָרְכַיָא עַל טוּרָא דִגְרִזִין וְיָת מְלַטְטַיָא עַל טוּרָא דְעֵיבָל:

{ל} הֲלֹא הֵמָּה בְּעֵבֶר הַיַּרְדֵּן אַחֲרֵי דֶּרֶךְ מְבוֹא הַשֶּׁמֶשׁ בְּאֶרֶץ הַכְּנַעֲנִי הַיֹּשֵׁב בָּעֲרָבָה מוּל הַגִּלְגָּל אֵצֶל אֵלוֹנֵי מֹרֶה:  מ  הֲלֹא הֵמָּה בְּעֵבֶר הַיַּרְדֵּן אַחֲרֵי דֶּרֶךְ מְבוֹא הַשֶּׁמֶשׁ בְּאֶרֶץ הַכְּנַעֲנִי הַיֹּשֵׁב בָּעֲרָבָה מוּל הַגִּלְגָּל אֵצֶל אֵלוֹנֵי מֹרֶה:  ת  הֲלָא אִנוּן בְּעִבְרָא דְיַרְדְנָא אֲחוֹרֵי אֹרַח מַעֲלָנֵי שִׁמְשָׁא בְּאַרְעָא כְּנַעְנָאָה דְיָתֵב בְּמֵישְׁרָא לָקֳבֵל גִלְגְלָא בִּסְטַר מֵישְׁרֵי מֹרֶה:

{לא} כִּי אַתֶּם עֹבְרִים אֶת הַיַּרְדֵּן לָבֹא לָרֶשֶׁת אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֵיכֶם נֹתֵן לָכֶם וִירִשְׁתֶּם אֹתָהּ וִישַׁבְתֶּם בָּהּ:  מ  כִּי אַתֶּם עֹבְרִים אֶת הַיַּרְדֵּן לָבֹא לָרֶשֶׁת אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֵיכֶם נֹתֵן לָכֶם וִירִשְׁתֶּם אֹתָהּ וִישַׁבְתֶּם בָּהּ:  ת  אֲרֵי אַתּוּן עָבְרִין יָת יַרְדְנָא לְמֵעַל לְמֵירַת יָת אַרְעָא דַיְיָ אֱלָהֲכוֹן יָהֵב לְכוֹן וְתֵירְתוּן יָתַהּ וְתֵיתְבוּן בַּהּ:

{לב} וּשְׁמַרְתֶּם לַעֲשׂוֹת אֵת כָּל הַחֻקִּים וְאֶת הַמִּשְׁפָּטִים אֲשֶׁר אָנֹכִי נֹתֵן לִפְנֵיכֶם הַיּוֹם:  מ  וּשְׁמַרְתֶּם לַעֲשׂוֹת אֵת כָּל הַחֻקִּים וְאֶת הַמִּשְׁפָּטִים אֲשֶׁר אָנֹכִי נֹתֵן לִפְנֵיכֶם הַיּוֹם:  ת  וְתִטְרוּן לְמֶעְבַּד יָת כָּל קְיָמַיָא וְיָת דִינַיָא דִי אֲנָא יָהֵב קֳדָמֵיכוֹן יוֹמָא דֵין:




דברים פרק-יב

{א} אֵלֶּה הַחֻקִּים וְהַמִּשְׁפָּטִים אֲשֶׁר תִּשְׁמְרוּן לַעֲשׂוֹת בָּאָרֶץ אֲשֶׁר נָתַן יְהוָה אֱלֹהֵי אֲבֹתֶיךָ לְךָ לְרִשְׁתָּהּ כָּל הַיָּמִים אֲשֶׁר אַתֶּם חַיִּים עַל הָאֲדָמָה:  מ  אֵלֶּה הַחֻקִּים וְהַמִּשְׁפָּטִים אֲשֶׁר תִּשְׁמְרוּן לַעֲשׂוֹת בָּאָרֶץ אֲשֶׁר נָתַן יְהוָה אֱלֹהֵי אֲבֹתֶיךָ לְךָ לְרִשְׁתָּהּ כָּל הַיָּמִים אֲשֶׁר אַתֶּם חַיִּים עַל הָאֲדָמָה:  ת  אִלֵין קְיָמַיָא וְדִינַיָא דִי תִטְרוּן לְמֶעְבַּד בְּאַרְעָא דִיהַב יְיָ אֱלָהָא דַאֲבְהָתָךְ לָךְ לְמֵירְתַהּ כָּל יוֹמַיָא דִי אַתּוּן קַיָמִין עַל אַרְעָא:

{ב} אַבֵּד תְּאַבְּדוּן אֶת כָּל הַמְּקֹמוֹת אֲשֶׁר עָבְדוּ שָׁם הַגּוֹיִם אֲשֶׁר אַתֶּם יֹרְשִׁים אֹתָם אֶת אֱלֹהֵיהֶם עַל הֶהָרִים הָרָמִים וְעַל הַגְּבָעוֹת וְתַחַת כָּל עֵץ רַעֲנָן:  מ  אַבֵּד תְּאַבְּדוּן אֶת כָּל הַמְּקֹמוֹת אֲשֶׁר עָבְדוּ שָׁם הַגּוֹיִם אֲשֶׁר אַתֶּם יֹרְשִׁים אֹתָם אֶת אֱלֹהֵיהֶם עַל הֶהָרִים הָרָמִים וְעַל הַגְּבָעוֹת וְתַחַת כָּל עֵץ רַעֲנָן:  ת  אַבָּדָא תְאַבְּדוּן יָת כָּל אַתְרַיָא דִי פְלָחוּ תַמָן עַמְמַיָא דִי אַתּוּן יָרְתִין יָתְהוֹן יָת טַעֲוָתְהוֹן עַל טוּרַיָא רָמַיָא וְעַל רָמָתָא וּתְחוֹת כָּל אִילַן עַבּוֹף:

{ג} וְנִתַּצְתֶּם אֶת מִזְבּחֹתָם וְשִׁבַּרְתֶּם אֶת מַצֵּבֹתָם וַאֲשֵׁרֵיהֶם תִּשְׂרְפוּן בָּאֵשׁ וּפְסִילֵי אֱלֹהֵיהֶם תְּגַדֵּעוּן וְאִבַּדְתֶּם אֶת שְׁמָם מִן הַמָּקוֹם הַהוּא:  מ  וְנִתַּצְתֶּם אֶת מִזְבּחֹתָם וְשִׁבַּרְתֶּם אֶת מַצֵּבֹתָם וַאֲשֵׁרֵיהֶם תִּשְׂרְפוּן בָּאֵשׁ וּפְסִילֵי אֱלֹהֵיהֶם תְּגַדֵּעוּן וְאִבַּדְתֶּם אֶת שְׁמָם מִן הַמָּקוֹם הַהוּא:  ת  וּתְתָרְעוּן יָת אֱגוֹרֵיהוֹן וּתְתַבְּרוּן יָת קָמָתְהוֹן וַאֲשֵׁרֵיהוֹן תּוֹקְדוּן בְּנוּרָא וְצַלְמֵי טַעֲוָתְהוֹן תְּקוֹצְצוּן וְתוֹבְדוּן יָת שִׁמְהוֹן מִן אַ:תרָא הַהוּא:

{ד} לֹא תַעֲשׂוּן כֵּן לַיהוָה אֱלֹהֵיכֶם:  מ  לֹא תַעֲשׂוּן כֵּן לַיהוָה אֱלֹהֵיכֶם:  ת  לָא תַעְבְּדוּן כֵּן קֳדָם יְיָ אֱלָהָכוֹן:

{ה} כִּי אִם אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֵיכֶם מִכָּל שִׁבְטֵיכֶם לָשׂוּם אֶת שְׁמוֹ שָׁם לְשִׁכְנוֹ תִדְרְשׁוּ וּבָאתָ שָׁמָּה:  מ  כִּי אִם אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֵיכֶם מִכָּל שִׁבְטֵיכֶם לָשׂוּם אֶת שְׁמוֹ שָׁם לְשִׁכְנוֹ תִדְרְשׁוּ וּבָאתָ שָׁמָּה:  ת  אֱלָהֵן לְאַתְרָא דִי יִתִּרְעֵי יְיָ אֱלָהֲכוֹן מִכָּל שִׁבְטֵּיכוֹן לְאַשְׁרָאָה שְׁכִנְתֵּהּ תַּמָן לְבֵית שְׁכִנְתֵּהּ תִּתְבְּעוּן וְתֵיתוּן לְתַמָן:

{ו} וַהֲבֵאתֶם שָׁמָּה עֹלֹתֵיכֶם וְזִבְחֵיכֶם וְאֵת מַעְשְׂרֹתֵיכֶם וְאֵת תְּרוּמַת יֶדְכֶם וְנִדְרֵיכֶם וְנִדְבֹתֵיכֶם וּבְכֹרֹת בְּקַרְכֶם וְצֹאנְכֶם:  מ  וַהֲבֵאתֶם שָׁמָּה עֹלֹתֵיכֶם וְזִבְחֵיכֶם וְאֵת מַעְשְׂרֹתֵיכֶם וְאֵת תְּרוּמַת יֶדְכֶם וְנִדְרֵיכֶם וְנִדְבֹתֵיכֶם וּבְכֹרֹת בְּקַרְכֶם וְצֹאנְכֶם:  ת  וְתַיְתוּן לְתַמָן עֲלָוָתֵיכוֹן וְנִכְסַת קוּדְשֵׁיכוֹן וְיָת מַעְשְׂרָתֵיכוֹן וְיָת אַפְרָשׁוּת יֶדְכוֹן וְנִדְרֵיכוֹן וְנִדְבָתֵיכוֹן וּבְכוֹרֵי תוֹרְכוֹן וְעָנְכוֹן:

{ז} וַאֲכַלְתֶּם שָׁם לִפְנֵי יְהוָה אֱלֹהֵיכֶם וּשְׂמַחְתֶּם בְּכֹל מִשְׁלַח יֶדְכֶם אַתֶּם וּבָתֵּיכֶם אֲשֶׁר בֵּרַכְךָ יְהוָה אֱלֹהֶיךָ:  מ  וַאֲכַלְתֶּם שָׁם לִפְנֵי יְהוָה אֱלֹהֵיכֶם וּשְׂמַחְתֶּם בְּכֹל מִשְׁלַח יֶדְכֶם אַתֶּם וּבָתֵּיכֶם אֲשֶׁר בֵּרַכְךָ יְהוָה אֱלֹהֶיךָ:  ת  וְתֵיכְלוּן תַּמָן קֳדָם יְיָ אֱלָהֲכוֹן וְתֶחֱדוּן בְכֹל אוֹשָׁטוּת יֶדְכוֹן אַתּוּן וֶאֱנַשׁ בָּתֵּיכוֹן דִי בָרְכָךְ יְיָ אֱלָהָךְ:

{ח} לֹא תַעֲשׂוּן כְּכֹל אֲשֶׁר אֲנַחְנוּ עֹשִׂים פֹּה הַיּוֹם אִישׁ כָּל הַיָּשָׁר בְּעֵינָיו:  מ  לֹא תַעֲשׂוּן כְּכֹל אֲשֶׁר אֲנַחְנוּ עֹשִׂים פֹּה הַיּוֹם אִישׁ כָּל הַיָּשָׁר בְּעֵינָיו:  ת  לָא תַעְבְּדוּן כְּכֹל דִי אֲנַחְנָא עָבְדִין הָכָא יוֹמָא דֵין גְבַר כָּל מָן דְכָשַׁר קֳדָמוֹהִי:

{ט} כִּי לֹא בָּאתֶם עַד עָתָּה אֶל הַמְּנוּחָה וְאֶל הַנַּחֲלָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:  מ  כִּי לֹא בָּאתֶם עַד עָתָּה אֶל הַמְּנוּחָה וְאֶל הַנַּחֲלָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:  ת  אֲרֵי לָא אֲתֵיתוּן עַד כְּעָן לְבֵית נְיָחָא וּלְאַחֲסָנָא דַיְיָ אֱלָהָךְ יָהֵב לָךְ:

{י} וַעֲבַרְתֶּם אֶת הַיַּרְדֵּן וִישַׁבְתֶּם בָּאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֵיכֶם מַנְחִיל אֶתְכֶם וְהֵנִיחַ לָכֶם מִכָּל אֹיְבֵיכֶם מִסָּבִיב וִישַׁבְתֶּם בֶּטַח:  מ  וַעֲבַרְתֶּם אֶת הַיַּרְדֵּן וִישַׁבְתֶּם בָּאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֵיכֶם מַנְחִיל אֶתְכֶם וְהֵנִיחַ לָכֶם מִכָּל אֹיְבֵיכֶם מִסָּבִיב וִישַׁבְתֶּם בֶּטַח:  ת  וְתַעְבְּרוּן יָת יַרְדְנָא וְתֵיתְבוּן בְּאַרְעָא דַיְיָ אֱלָהֲכוֹן מַחְסֵן יָתְכוֹן וִינִיחַ לְכוֹן מִכָּל בַּעֲלֵי דְבָבֵיכוֹן מִסְחוֹר סְחוֹר וְתֵיתְבוּן לְרָחְצָן:

{יא}  שני  וְהָיָה הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֵיכֶם בּוֹ לְשַׁכֵּן שְׁמוֹ שָׁם שָׁמָּה תָבִיאוּ אֵת כָּל אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם עוֹלֹתֵיכֶם וְזִבְחֵיכֶם מַעְשְׂרֹתֵיכֶם וּתְרֻמַת יֶדְכֶם וְכֹל מִבְחַר נִדְרֵיכֶם אֲשֶׁר תִּדְּרוּ לַיהוָה:  מ  וְהָיָה הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֵיכֶם בּוֹ לְשַׁכֵּן שְׁמוֹ שָׁם שָׁמָּה תָבִיאוּ אֵת כָּל אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם עוֹלֹתֵיכֶם וְזִבְחֵיכֶם מַעְשְׂרֹתֵיכֶם וּתְרֻמַת יֶדְכֶם וְכֹל מִבְחַר נִדְרֵיכֶם אֲשֶׁר תִּדְּרוּ לַיהוָה:  ת  וִיהֵי אַתְרָא דִי יִתִּרְעֵי יְיָ אֱלָהֲכוֹן בֵּהּ לְאַשְׁרָאָה שְׁכִנְתֵּהּ תַּמָן לְתַמָן תַּיְתוּן יָת כָּל דִי אֲנָא מְפַקֵד יָתְכוֹן עֲלָוָתֵיכוֹן וְנִכְסַת קוּדְשֵׁיכוֹן מַעְשְׂרָתֵיכוֹן וְאַפְרָשׁוּת יֶדְכוֹן וְכֹל שְׁפַר נִדְרֵיכוֹן דִי תִדְרוּן קֳדָם יְיָ:

{יב} וּשְׂמַחְתֶּם לִפְנֵי יְהוָה אֱלֹהֵיכֶם אַתֶּם וּבְנֵיכֶם וּבְנֹתֵיכֶם וְעַבְדֵיכֶם וְאַמְהֹתֵיכֶם וְהַלֵּוִי אֲשֶׁר בְּשַׁעֲרֵיכֶם כִּי אֵין לוֹ חֵלֶק וְנַחֲלָה אִתְּכֶם:  מ  וּשְׂמַחְתֶּם לִפְנֵי יְהוָה אֱלֹהֵיכֶם אַתֶּם וּבְנֵיכֶם וּבְנֹתֵיכֶם וְעַבְדֵיכֶם וְאַמְהֹתֵיכֶם וְהַלֵּוִי אֲשֶׁר בְּשַׁעֲרֵיכֶם כִּי אֵין לוֹ חֵלֶק וְנַחֲלָה אִתְּכֶם:  ת  וְתֶחֱדוּן קֳדָם יְיָ אֱלָהֲכוֹן אַתּוּן וּבְנֵיכוֹן וּבְנָתֵיכוֹן וְעַבְדֵיכוֹן וְאַמְהָתֵיכוֹן וְלֵוָאָה דִי בְקִרְוֵיכוֹן אֲרֵי לֵית לֵהּ חֳלָק וְאַחֲסָנָא עִמְכוֹן:

{יג} הִשָּׁמֶר לְךָ פֶּן תַּעֲלֶה עֹלֹתֶיךָ בְּכָל מָקוֹם אֲשֶׁר תִּרְאֶה:  מ  הִשָּׁמֶר לְךָ פֶּן תַּעֲלֶה עֹלֹתֶיךָ בְּכָל מָקוֹם אֲשֶׁר תִּרְאֶה:  ת  אִסְתַּמַר לָךְ דִילְמָא תַסֵק עֲלָוָתָךְ בְּכָל אַתְרָא דִי תֶחֱזֵי:

{יד} כִּי אִם בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה בְּאַחַד שְׁבָטֶיךָ שָׁם תַּעֲלֶה עֹלֹתֶיךָ וְשָׁם תַּעֲשֶׂה כֹּל אֲשֶׁר אָנֹכִי מְצַוֶּךָּ:  מ  כִּי אִם בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה בְּאַחַד שְׁבָטֶיךָ שָׁם תַּעֲלֶה עֹלֹתֶיךָ וְשָׁם תַּעֲשֶׂה כֹּל אֲשֶׁר אָנֹכִי מְצַוֶּךָּ:  ת  אֶלָהֵן בְּאַתְרָא דִי יִתִּרְעֵי יְיָ בְּחַד מִן שִׁבְטָיךְ תַּמָן תַּסֵק עֲלָוָתָךְ וְתַמָן תַּעְבֵּד כֹּל דִי אֲנָא מְפַקְדָךְ:

{טו} רַק בְּכָל אַוַּת נַפְשְׁךָ תִּזְבַּח וְאָכַלְתָּ בָשָׂר כְּבִרְכַּת יְהוָה אֱלֹהֶיךָ אֲשֶׁר נָתַן לְךָ בְּכָל שְׁעָרֶיךָ הַטָּמֵא וְהַטָּהוֹר יֹאכְלֶנּוּ כַּצְּבִי וְכָאַיָּל:  מ  רַק בְּכָל אַוַּת נַפְשְׁךָ תִּזְבַּח וְאָכַלְתָּ בָשָׂר כְּבִרְכַּת יְהוָה אֱלֹהֶיךָ אֲשֶׁר נָתַן לְךָ בְּכָל שְׁעָרֶיךָ הַטָּמֵא וְהַטָּהוֹר יֹאכְלֶנּוּ כַּצְּבִי וְכָאַיָּל:  ת  לְחוֹד בְּכָל רְעוּת נַפְשָׁךְ תִּכּוֹס וְתֵיכוּל בִּשְׂרָא כְּבִרְכְּתָא דַיְיָ אֱלָהָךְ דִי יְהַב לָךְ בְּכָל קִרְוָיךְ מְסָאָבָא וְדַכְיָא יֵכְלֻנֵהּ כִּבְשַׂר טַבְיָא וְאַיְלָא:

{טז} רַק הַדָּם לֹא תֹאכֵלוּ עַל הָאָרֶץ תִּשְׁפְּכֶנּוּ כַּמָּיִם:  מ  רַק הַדָּם לֹא תֹאכֵלוּ עַל הָאָרֶץ תִּשְׁפְּכֶנּוּ כַּמָּיִם:  ת  לְחוֹד דְמָא לָא תֵיכְלוּן עַל אַרְעָא תִשְׁדִנֵהּ כְּמַיָא:

{יז} לֹא תוּכַל לֶאֱכֹל בִּשְׁעָרֶיךָ מַעְשַׂר דְּגָנְךָ וְתִירֹשְׁךָ וְיִצְהָרֶךָ וּבְכֹרֹת בְּקָרְךָ וְצֹאנֶךָ וְכָל נְדָרֶיךָ אֲשֶׁר תִּדֹּר וְנִדְבֹתֶיךָ וּתְרוּמַת יָדֶךָ:  מ  לֹא תוּכַל לֶאֱכֹל בִּשְׁעָרֶיךָ מַעְשַׂר דְּגָנְךָ וְתִירֹשְׁךָ וְיִצְהָרֶךָ וּבְכֹרֹת בְּקָרְךָ וְצֹאנֶךָ וְכָל נְדָרֶיךָ אֲשֶׁר תִּדֹּר וְנִדְבֹתֶיךָ וּתְרוּמַת יָדֶךָ:  ת  לֵית לָךְ רְשׁוּ לְמֵיכַל בְּקִרְוָיךְ מַעְשַׂר עִבוּרָךְ חַמְרָךְ וּמִשְׁחָךְ וּבְכוֹרֵי תוֹרָךְ וְעָנָךְ וְכָל נִדְרָךְ דִי תִדַר וְנִדְבָתָיךְ וְאַפְרָשׁוּת יְדָךְ:

{יח} כִּי אִם לִפְנֵי יְהוָה אֱלֹהֶיךָ תֹּאכְלֶנּוּ בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ אַתָּה וּבִנְךָ וּבִתֶּךָ וְעַבְדְּךָ וַאֲמָתֶךָ וְהַלֵּוִי אֲשֶׁר בִּשְׁעָרֶיךָ וְשָׂמַחְתָּ לִפְנֵי יְהוָה אֱלֹהֶיךָ בְּכֹל מִשְׁלַח יָדֶךָ:  מ  כִּי אִם לִפְנֵי יְהוָה אֱלֹהֶיךָ תֹּאכְלֶנּוּ בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ אַתָּה וּבִנְךָ וּבִתֶּךָ וְעַבְדְּךָ וַאֲמָתֶךָ וְהַלֵּוִי אֲשֶׁר בִּשְׁעָרֶיךָ וְשָׂמַחְתָּ לִפְנֵי יְהוָה אֱלֹהֶיךָ בְּכֹל מִשְׁלַח יָדֶךָ:  ת  אֱלָהֵן קֳדָם יְיָ אֱלָהָךְ תֵּיכְלִנֵהּ בְּאַתְרָא דִי יִתִּרְעֵי יְיָ אֱלָהָךְ בֵּהּ אַתְּ וּבְרָךְ וּבְרַתָּךְ וְעַבְדָךְ וְאַמְתָךְ וְלֵוָאָה דִי בְקִרְוָיךְ וְתֶחֱדֵי קֳדָם יְיָ אֱלָהָךְ בְּכָל אוֹשָׁטּוּת יְדָךְ:

{יט} הִשָּׁמֶר לְךָ פֶּן תַּעֲזֹב אֶת הַלֵּוִי כָּל יָמֶיךָ עַל אַדְמָתֶךָ:  מ  הִשָּׁמֶר לְךָ פֶּן תַּעֲזֹב אֶת הַלֵּוִי כָּל יָמֶיךָ עַל אַדְמָתֶךָ:  ת  אִסְתַּמַר לָךְ דִילְמָא תִשְׁבּוֹק יָת לֵוָאֵי כָּל יוֹמָיךְ עַל אַרְעָךְ: (ס)

{כ} כִּי יַרְחִיב יְהוָה אֱלֹהֶיךָ אֶת גְּבוּלְךָ כַּאֲשֶׁר דִּבֶּר לָךְ וְאָמַרְתָּ אֹכְלָה בָשָׂר כִּי תְאַוֶּה נַפְשְׁךָ לֶאֱכֹל בָּשָׂר בְּכָל אַוַּת נַפְשְׁךָ תֹּאכַל בָּשָׂר:  מ  כִּי יַרְחִיב יְהוָה אֱלֹהֶיךָ אֶת גְּבוּלְךָ כַּאֲשֶׁר דִּבֶּר לָךְ וְאָמַרְתָּ אֹכְלָה בָשָׂר כִּי תְאַוֶּה נַפְשְׁךָ לֶאֱכֹל בָּשָׂר בְּכָל אַוַּת נַפְשְׁךָ תֹּאכַל בָּשָׂר:  ת  אֲרֵי יַפְתֵּי יְיָ אֱלָהָךְ יָת תְּחוּמָךְ כְּמָא דִי מַלִיל לָךְ וְתֵימַר אֵיכוּל בִּשְׂרָא אֲרֵי תִתִּרְעֵי נַפְשָׁךְ לְמֵיכַל בִּשְׂרָא בְּכָל רְעוּת נַפְשָׁךְ תֵּיכוּל בִּשְׂרָא:

{כא} כִּי יִרְחַק מִמְּךָ הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ לָשׂוּם שְׁמוֹ שָׁם וְזָבַחְתָּ מִבְּקָרְךָ וּמִצֹּאנְךָ אֲשֶׁר נָתַן יְהוָה לְךָ כַּאֲשֶׁר צִוִּיתִךָ וְאָכַלְתָּ בִּשְׁעָרֶיךָ בְּכֹל אַוַּת נַפְשֶׁךָ:  מ  כִּי יִרְחַק מִמְּךָ הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ לָשׂוּם שְׁמוֹ שָׁם וְזָבַחְתָּ מִבְּקָרְךָ וּמִצֹּאנְךָ אֲשֶׁר נָתַן יְהוָה לְךָ כַּאֲשֶׁר צִוִּיתִךָ וְאָכַלְתָּ בִּשְׁעָרֶיךָ בְּכֹל אַוַּת נַפְשֶׁךָ:  ת  אֲרֵי יִתְרָחַק מִנָךְ אַתְרָא דִי יִתִּרְעֵי יְיָ אֱלָהָךְ לְאַשְׁרָאָה שְׁכִנְתֵּהּ תַּמָן וְתִכּוֹס מִתּוֹרָךְ וּמֵעָנָךְ דִי יְהַב יְיָ לָךְ כְּמָא דִי פַּקֵדִתָּךְ וְתֵיכוּל בְּקִרְוָיךְ בְּכֹל רְעוּת נַפְשָׁךְ:

{כב} אַךְ כַּאֲשֶׁר יֵאָכֵל אֶת הַצְּבִי וְאֶת הָאַיָּל כֵּן תֹּאכְלֶנּוּ הַטָּמֵא וְהַטָּהוֹר יַחְדָּו יֹאכְלֶנּוּ:  מ  אַךְ כַּאֲשֶׁר יֵאָכֵל אֶת הַצְּבִי וְאֶת הָאַיָּל כֵּן תֹּאכְלֶנּוּ הַטָּמֵא וְהַטָּהוֹר יַחְדָּו יֹאכְלֶנּוּ:  ת  בְּרַם כְּמָא דִי מִתְאֲכֵל בְּשַׂר טַבְיָא וְאַיְלָא כֵּן תֵּיכְלִנֵהּ מְסָאָבָא וְדַכְיָא כַּחֲדָא יֵכְלִנֵהּ:

{כג} רַק חֲזַק לְבִלְתִּי אֲכֹל הַדָּם כִּי הַדָּם הוּא הַנָּפֶשׁ וְלֹא תֹאכַל הַנֶּפֶשׁ עִם הַבָּשָׂר:  מ  רַק חֲזַק לְבִלְתִּי אֲכֹל הַדָּם כִּי הַדָּם הוּא הַנָּפֶשׁ וְלֹא תֹאכַל הַנֶּפֶשׁ עִם הַבָּשָׂר:  ת  לְחוֹד תְּקֵף בְּדִיל דְלָא לְמֵיכַל דְּמָא אֲרֵי דְמָא הוּא נַפְשָׁא וְלָא תֵיכוּל נַפְשָׁא עִם בִּשְׂרָא:

{כד} לֹא תֹּאכְלֶנּוּ עַל הָאָרֶץ תִּשְׁפְּכֶנּוּ כַּמָּיִם:  מ  לֹא תֹּאכְלֶנּוּ עַל הָאָרֶץ תִּשְׁפְּכֶנּוּ כַּמָּיִם:  ת  לָא תֵּיכְלִנֵהּ עַל אַרְעָא תִשְׁדִנֵהּ כְּמַיָא:

{כה} לֹא תֹּאכְלֶנּוּ לְמַעַן יִיטַב לְךָ וּלְבָנֶיךָ אַחֲרֶיךָ כִּי תַעֲשֶׂה הַיָּשָׁר בְּעֵינֵי יְהוָה:  מ  לֹא תֹּאכְלֶנּוּ לְמַעַן יִיטַב לְךָ וּלְבָנֶיךָ אַחֲרֶיךָ כִּי תַעֲשֶׂה הַיָּשָׁר בְּעֵינֵי יְהוָה:  ת  לָא תֵּיכְלִנֵהּ בְּדִיל דְיֵיטַב לָךְ וְלִבְנָיךְ בַּתְרָךְ אֲרֵי תַעְבֵּד דְכָשָׁר קֳדָם יְיָ:

{כו} רַק קָדָשֶׁיךָ אֲשֶׁר יִהְיוּ לְךָ וּנְדָרֶיךָ תִּשָּׂא וּבָאתָ אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה:  מ  רַק קָדָשֶׁיךָ אֲשֶׁר יִהְיוּ לְךָ וּנְדָרֶיךָ תִּשָּׂא וּבָאתָ אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה:  ת  לְחוֹד מַעְשַׂר קוּדְשָׁיךְ דִיהוֹן לָךְ וְנִדְרָךְ תִּטוֹל וְתֵיתֵי לְאַתְרָא דִי יִתִּרְעֵי יְיָ:

{כז} וְעָשִׂיתָ עֹלֹתֶיךָ הַבָּשָׂר וְהַדָּם עַל מִזְבַּח יְהוָה אֱלֹהֶיךָ וְדַם זְבָחֶיךָ יִשָּׁפֵךְ עַל מִזְבַּח יְהוָה אֱלֹהֶיךָ וְהַבָּשָׂר תֹּאכֵל:  מ  וְעָשִׂיתָ עֹלֹתֶיךָ הַבָּשָׂר וְהַדָּם עַל מִזְבַּח יְהוָה אֱלֹהֶיךָ וְדַם זְבָחֶיךָ יִשָּׁפֵךְ עַל מִזְבַּח יְהוָה אֱלֹהֶיךָ וְהַבָּשָׂר תֹּאכֵל:  ת  וְתַעְבֵּד עֲלָוָתָךְ בִּשְׂרָא וּדְמָא עַל מַדְבְּחָא דַיְיָ אֱלָהָךְ וְדַם נִכְסַת קוּדְשָׁיךְ יִתָּשַׁד עַל מַדְבְּחָא דַיְיָ אֱלָהָךְ וּבִשְׂרָא תֵּיכוּל:

{כח} שְׁמֹר וְשָׁמַעְתָּ אֵת כָּל הַדְּבָרִים הָאֵלֶּה אֲשֶׁר אָנֹכִי מְצַוֶּךָּ לְמַעַן יִיטַב לְךָ וּלְבָנֶיךָ אַחֲרֶיךָ עַד עוֹלָם כִּי תַעֲשֶׂה הַטּוֹב וְהַיָּשָׁר בְּעֵינֵי יְהוָה אֱלֹהֶיךָ:  מ  שְׁמֹר וְשָׁמַעְתָּ אֵת כָּל הַדְּבָרִים הָאֵלֶּה אֲשֶׁר אָנֹכִי מְצַוֶּךָּ לְמַעַן יִיטַב לְךָ וּלְבָנֶיךָ אַחֲרֶיךָ עַד עוֹלָם כִּי תַעֲשֶׂה הַטּוֹב וְהַיָּשָׁר בְּעֵינֵי יְהוָה אֱלֹהֶיךָ:  ת  טַר וּתְקַבֵּל יָת כָּל פִּתְגָמַיָא הָאִלֵין דִי אֲנָא מְפַקְדָךְ בְּדִיל דְיֵיטַב לָךְ וְלִבְנָיךְ בַּתְרָךְ עַד עֳלָם אֲרֵי תַעְבֵּד דְתָקֵן וּדְכָשַׁר קֳדָם יְיָ אֱלָהָךְ: (ס)

{כט}  שלישי  כִּי יַכְרִית יְהוָה אֱלֹהֶיךָ אֶת הַגּוֹיִם אֲשֶׁר אַתָּה בָא שָׁמָּה לָרֶשֶׁת אוֹתָם מִפָּנֶיךָ וְיָרַשְׁתָּ אֹתָם וְיָשַׁבְתָּ בְּאַרְצָם:  מ  כִּי יַכְרִית יְהוָה אֱלֹהֶיךָ אֶת הַגּוֹיִם אֲשֶׁר אַתָּה בָא שָׁמָּה לָרֶשֶׁת אוֹתָם מִפָּנֶיךָ וְיָרַשְׁתָּ אֹתָם וְיָשַׁבְתָּ בְּאַרְצָם:  ת  אֲרֵי יְשֵׁצֵי יְיָ אֱלָהָךְ יָת עַמְמַיָא דִי אַתְּ עָלֵל לְתַמָן לְתָרָכָא יָתְהוֹן מִקֳדָמָךְ וְתֵירַת יָתְהוֹן וְתֵיתַב בְּאַרְעֲהוֹן:

{ל} הִשָּׁמֶר לְךָ פֶּן תִּנָּקֵשׁ אַחֲרֵיהֶם אַחֲרֵי הִשָּׁמְדָם מִפָּנֶיךָ וּפֶן תִּדְרֹשׁ לֵאלֹהֵיהֶם לֵאמֹר אֵיכָה יַעַבְדוּ הַגּוֹיִם הָאֵלֶּה אֶת אֱלֹהֵיהֶם וְאֶעֱשֶׂה כֵּן גַּם אָנִי:  מ  הִשָּׁמֶר לְךָ פֶּן תִּנָּקֵשׁ אַחֲרֵיהֶם אַחֲרֵי הִשָּׁמְדָם מִפָּנֶיךָ וּפֶן תִּדְרֹשׁ לֵאלֹהֵיהֶם לֵאמֹר אֵיכָה יַעַבְדוּ הַגּוֹיִם הָאֵלֶּה אֶת אֱלֹהֵיהֶם וְאֶעֱשֶׂה כֵּן גַּם אָנִי:  ת  אִסְתַּמַר לָךְ דִילְמָא תִתָּקַל בַּתְרֵיהוֹן בָּתַר דְיִשְׁתֵּצוּן מִקֳדָמָךְ וְדִילְמָא תִתְבַּע לְטַעֲוָתְהוֹן לְמֵימַר אֶכְדֵין פָּלְחִין עַמְמַיָא הָאִלֵין יָת טַעֲוָתְהוֹן וְאֶעְבֵּד כֵּן אַף אֲנָא:

{לא} לֹא תַעֲשֶׂה כֵן לַיהוָה אֱלֹהֶיךָ כִּי כָּל תּוֹעֲבַת יְהוָה אֲשֶׁר שָׂנֵא עָשׂוּ לֵאלֹהֵיהֶם כִּי גַם אֶת בְּנֵיהֶם וְאֶת בְּנֹתֵיהֶם יִשְׂרְפוּ בָאֵשׁ לֵאלֹהֵיהֶם:  מ  לֹא תַעֲשֶׂה כֵן לַיהוָה אֱלֹהֶיךָ כִּי כָּל תּוֹעֲבַת יְהוָה אֲשֶׁר שָׂנֵא עָשׂוּ לֵאלֹהֵיהֶם כִּי גַם אֶת בְּנֵיהֶם וְאֶת בְּנֹתֵיהֶם יִשְׂרְפוּ בָאֵשׁ לֵאלֹהֵיהֶם:  ת  לָא תַעְבֵּד כֵּן קֳדָם יְיָ אֱלָהָךְ אֲרֵי כָּל דִמְרָחָק קֳדָם יְיָ דִי סָנֵי עֲבָדוּ לְטַעֲוָתְהוֹן אֲרֵי אַף יָת בְּנֵיהוֹן וְיָת בְּנָתֵיהוֹן מוֹקְדִין בְּנוּרָא לְטַּעֲוָתְהוֹן:




דברים פרק-יג

{א} אֵת כָּל הַדָּבָר אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם אֹתוֹ תִשְׁמְרוּ לַעֲשׂוֹת לֹא תֹסֵף עָלָיו וְלֹא תִגְרַע מִמֶּנּוּ:  מ  אֵת כָּל הַדָּבָר אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם אֹתוֹ תִשְׁמְרוּ לַעֲשׂוֹת לֹא תֹסֵף עָלָיו וְלֹא תִגְרַע מִמֶּנּוּ:  ת  יָת כָּל פִּתְגָמָא דִי אֲנָא מְפַקֵד יָתְכוֹן יָתֵהּ תִּטְרוּן לְמֶעְבָּד לָא תוֹסְפוּן עֲלוֹהִי וְלָא תִמְנְעוּן מִנֵהּ: (פ)

{ב} כִּי יָקוּם בְּקִרְבְּךָ נָבִיא אוֹ חֹלֵם חֲלוֹם וְנָתַן אֵלֶיךָ אוֹת אוֹ מוֹפֵת:  מ  כִּי יָקוּם בְּקִרְבְּךָ נָבִיא אוֹ חֹלֵם חֲלוֹם וְנָתַן אֵלֶיךָ אוֹת אוֹ מוֹפֵת:  ת  אֲרֵי יְקוּם בֵּינָךְ נְבִיָא אוֹ חָלֵם חֶלְמָא וְיִתֶּן לָךְ אָת אוֹ מוֹפְתָא:

{ג} וּבָא הָאוֹת וְהַמּוֹפֵת אֲשֶׁר דִּבֶּר אֵלֶיךָ לֵאמֹר נֵלְכָה אַחֲרֵי אֱלֹהִים אֲחֵרִים אֲשֶׁר לֹא יְדַעְתָּם וְנָעָבְדֵם:  מ  וּבָא הָאוֹת וְהַמּוֹפֵת אֲשֶׁר דִּבֶּר אֵלֶיךָ לֵאמֹר נֵלְכָה אַחֲרֵי אֱלֹהִים אֲחֵרִים אֲשֶׁר לֹא יְדַעְתָּם וְנָעָבְדֵם:  ת  וְיֵיתֵי אָתָא וּמוֹפְתָא דִי מַלִיל עִמָךְ לְמֵימָר נְהַךְ בָּתַר טַעֲוַת עַמְמַיָא דִי לָא יְדַעְתָּנוּן וְנִפְלְחִנוּן:

{ד} לֹא תִשְׁמַע אֶל דִּבְרֵי הַנָּבִיא הַהוּא אוֹ אֶל חוֹלֵם הַחֲלוֹם הַהוּא כִּי מְנַסֶּה יְהוָה אֱלֹהֵיכֶם אֶתְכֶם לָדַעַת הֲיִשְׁכֶם אֹהֲבִים אֶת יְהוָה אֱלֹהֵיכֶם בְּכָל לְבַבְכֶם וּבְכָל נַפְשְׁכֶם:  מ  לֹא תִשְׁמַע אֶל דִּבְרֵי הַנָּבִיא הַהוּא אוֹ אֶל חוֹלֵם הַחֲלוֹם הַהוּא כִּי מְנַסֶּה יְהוָה אֱלֹהֵיכֶם אֶתְכֶם לָדַעַת הֲיִשְׁכֶם אֹהֲבִים אֶת יְהוָה אֱלֹהֵיכֶם בְּכָל לְבַבְכֶם וּבְכָל נַפְשְׁכֶם:  ת  לָא תְקַבֵּל לְפִתְגָמֵי נְבִיָא הַהוּא אוֹ לְחָלֵם חֶלְמָא הַהוּא אֲרֵי מְנַסֵי יְיָ אֱלָהֲכוֹן יָתְכוֹן לְמִדַע הַאִיתֵיכוֹן רָחֲמִין יָת יְיָ אֱלָהֲכוֹן בְכָל לִבְּכוֹן וּבְכָל נַפְשְׁכוֹן:

{ה} אַחֲרֵי יְהוָה אֱלֹהֵיכֶם תֵּלֵכוּ וְאֹתוֹ תִירָאוּ וְאֶת מִצְוֹתָיו תִּשְׁמֹרוּ וּבְקֹלוֹ תִשְׁמָעוּ וְאֹתוֹ תַעֲבֹדוּ וּבוֹ תִדְבָּקוּן:  מ  אַחֲרֵי יְהוָה אֱלֹהֵיכֶם תֵּלֵכוּ וְאֹתוֹ תִירָאוּ וְאֶת מִצְוֹתָיו תִּשְׁמֹרוּ וּבְקֹלוֹ תִשְׁמָעוּ וְאֹתוֹ תַעֲבֹדוּ וּבוֹ תִדְבָּקוּן:  ת  בָּתַר דַחַלְתָּא דַיְיָ אֱלָהֲכוֹן תְהָכוּן וְיָתֵהּ תִּדְחֲלוּן וְיָת פִּקוּדוֹהִי תִטְּרוּן וּלְמֵימְרֵהּ תְּקַבְּלוּן וְקָדָמוֹהִי תִפְלְחוּן וּבְדַחַלְתֵּהּ תִּתְקָרְבוּן:

{ו} וְהַנָּבִיא הַהוּא אוֹ חֹלֵם הַחֲלוֹם הַהוּא יוּמָת כִּי דִבֶּר סָרָה עַל יְהוָה אֱלֹהֵיכֶם הַמּוֹצִיא אֶתְכֶם מֵאֶרֶץ מִצְרַיִם וְהַפֹּדְךָ מִבֵּית עֲבָדִים לְהַדִּיחֲךָ מִן הַדֶּרֶךְ אֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ לָלֶכֶת בָּהּ וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  מ  וְהַנָּבִיא הַהוּא אוֹ חֹלֵם הַחֲלוֹם הַהוּא יוּמָת כִּי דִבֶּר סָרָה עַל יְהוָה אֱלֹהֵיכֶם הַמּוֹצִיא אֶתְכֶם מֵאֶרֶץ מִצְרַיִם וְהַפֹּדְךָ מִבֵּית עֲבָדִים לְהַדִּיחֲךָ מִן הַדֶּרֶךְ אֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ לָלֶכֶת בָּהּ וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:  ת  וּנְבִיָא הַהוּא אוֹ חָלֵם חֶלְמָא הַהוּא יִתְקְטֵּל אֲרֵי מַלִיל סַטְיָא עַל יְיָ אֱלָהֲכוֹן דִי אַפֵּיק יָתְכוֹן מֵאַרְעָא דְמִצְרַיִם וּדְפָרְקָךְ מִבֵּית עַבְדוּתָא לְאַטְעֲיוּתָךְ מִן אָרְחָא דִי פַקְדָךְ יְיָ אֱלָהָךְ לִמְהַךְ בַּהּ וּתְפַלֵי עָבֵד דְבִישׁ מִבֵּינָךְ: (פ)

{ז} כִּי יְסִיתְךָ אָחִיךָ בֶן אִמֶּךָ אוֹ בִנְךָ אוֹ בִתְּךָ אוֹ אֵשֶׁת חֵיקֶךָ אוֹ רֵעֲךָ אֲשֶׁר כְּנַפְשְׁךָ בַּסֵּתֶר לֵאמֹר נֵלְכָה וְנַעַבְדָה אֱלֹהִים אֲחֵרִים אֲשֶׁר לֹא יָדַעְתָּ אַתָּה וַאֲבֹתֶיךָ:  מ  כִּי יְסִיתְךָ אָחִיךָ בֶן אִמֶּךָ אוֹ בִנְךָ אוֹ בִתְּךָ אוֹ אֵשֶׁת חֵיקֶךָ אוֹ רֵעֲךָ אֲשֶׁר כְּנַפְשְׁךָ בַּסֵּתֶר לֵאמֹר נֵלְכָה וְנַעַבְדָה אֱלֹהִים אֲחֵרִים אֲשֶׁר לֹא יָדַעְתָּ אַתָּה וַאֲבֹתֶיךָ:  ת  אֲרֵי יְמַלְכִנָךְ אֲחוּךְ בַּר אִמָךְ אוֹ בְרָךְ אוֹ בְרַתָּךְ אוֹ אִתַּת קְיָמָךְ אוֹ חַבְרָךְ דִי כְנַפְשָׁךְ בְּסִתְרָא לְמֵימַר נְהַךְ וְנִפְלַח לְטַעֲוַת עַמְמַיָא דִי לָא יְדַעְתְּ אַתְּ וַאֲבָהָתָךְ:

{ח} מֵאֱלֹהֵי הָעַמִּים אֲשֶׁר סְבִיבֹתֵיכֶם הַקְּרֹבִים אֵלֶיךָ אוֹ הָרְחֹקִים מִמֶּךָּ מִקְצֵה הָאָרֶץ וְעַד קְצֵה הָאָרֶץ:  מ  מֵאֱלֹהֵי הָעַמִּים אֲשֶׁר סְבִיבֹתֵיכֶם הַקְּרֹבִים אֵלֶיךָ אוֹ הָרְחֹקִים מִמֶּךָּ מִקְצֵה הָאָרֶץ וְעַד קְצֵה הָאָרֶץ:  ת  מִטַעֲוַת עַמְמַיָא דִי בְסַחֲרָנֵיכוֹן דִקְרִיבִין לָךְ אוֹ דִרְחִיקִין מִנָךְ מִסְיָפֵי אַרְעָא וְעַד סְיָפֵי אַרְעָא:

{ט} לֹא תֹאבֶה לוֹ וְלֹא תִשְׁמַע אֵלָיו וְלֹא תָחוֹס עֵינְךָ עָלָיו וְלֹא תַחְמֹל וְלֹא תְכַסֶּה עָלָיו:  מ  לֹא תֹאבֶה לוֹ וְלֹא תִשְׁמַע אֵלָיו וְלֹא תָחוֹס עֵינְךָ עָלָיו וְלֹא תַחְמֹל וְלֹא תְכַסֶּה עָלָיו:  ת  לָא תֵיבֵי לֵהּ וְלָא תְקַבֵּל מִנֵהּ וְלָא תְחוּס עֵינָךְ עֲלוֹהִי וְלָא תְרַחַם וְלָא תְכַסֵי עֲלוֹהִי:

{י} כִּי הָרֹג תַּהַרְגֶנּוּ יָדְךָ תִּהְיֶה בּוֹ בָרִאשׁוֹנָה לַהֲמִיתוֹ וְיַד כָּל הָעָם בָּאַחֲרֹנָה:  מ  כִּי הָרֹג תַּהַרְגֶנּוּ יָדְךָ תִּהְיֶה בּוֹ בָרִאשׁוֹנָה לַהֲמִיתוֹ וְיַד כָּל הָעָם בָּאַחֲרֹנָה:  ת  אֲרֵי מִקְטַל תִּקְטְּלִנֵהּ יְדָךְ תְּהֵי בֵהּ בְּקַדְמֵיתָא לְמִקְטְלֵהּ וִידָא דְכָל עַמָא בְבַתְרֵיתָא:

{יא} וּסְקַלְתּוֹ בָאֲבָנִים וָמֵת כִּי בִקֵּשׁ לְהַדִּיחֲךָ מֵעַל יְהוָה אֱלֹהֶיךָ הַמּוֹצִיאֲךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים:  מ  וּסְקַלְתּוֹ בָאֲבָנִים וָמֵת כִּי בִקֵּשׁ לְהַדִּיחֲךָ מֵעַל יְהוָה אֱלֹהֶיךָ הַמּוֹצִיאֲךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים:  ת  וְתִרְגְמִנֵהּ בְּאַבְנַיָא וִימוּת אֲרֵי בָעָא לְאַטְעֲיוּתָךְ מִדַחַלְתָּא דַיְיָ אֱלָהָךְ דִי אַפְּקָךְ מֵאַרְעָא דְמִצְרַיִם מִבֵּית עַבְדוּתָא:

{יב} וְכָל יִשְׂרָאֵל יִשְׁמְעוּ וְיִרָאוּן וְלֹא יוֹסִפוּ לַעֲשׂוֹת כַּדָּבָר הָרָע הַזֶּה בְּקִרְבֶּךָ:  מ  וְכָל יִשְׂרָאֵל יִשְׁמְעוּ וְיִרָאוּן וְלֹא יוֹסִפוּ לַעֲשׂוֹת כַּדָּבָר הָרָע הַזֶּה בְּקִרְבֶּךָ:  ת  וְכָל יִשְׂרָאֵל יִשְׁמְעוּן וְיִדְחֲלוּן וְלָא יוֹסְפוּן לְמֶעְבַּד כְּפִתְגָמָא בִישָׁא הָדֵין בֵּינָךְ: (פ)

{יג} כִּי תִשְׁמַע בְּאַחַת עָרֶיךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לָשֶׁבֶת שָׁם לֵאמֹר:  מ  כִּי תִשְׁמַע בְּאַחַת עָרֶיךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לָשֶׁבֶת שָׁם לֵאמֹר:  ת  אֲרֵי תִשְׁמַע בְּחַד מִקִרְוָיךְ דִי יְיָ אֱלָהָךְ יָהֵב לָךְ לְמִתַּב תַּמָן לְמֵימָר:

{יד} יָצְאוּ אֲנָשִׁים בְּנֵי בְלִיַּעַל מִקִּרְבֶּךָ וַיַּדִּיחוּ אֶת יֹשְׁבֵי עִירָם לֵאמֹר נֵלְכָה וְנַעַבְדָה אֱלֹהִים אֲחֵרִים אֲשֶׁר לֹא יְדַעְתֶּם:  מ  יָצְאוּ אֲנָשִׁים בְּנֵי בְלִיַּעַל מִקִּרְבֶּךָ וַיַּדִּיחוּ אֶת יֹשְׁבֵי עִירָם לֵאמֹר נֵלְכָה וְנַעַבְדָה אֱלֹהִים אֲחֵרִים אֲשֶׁר לֹא יְדַעְתֶּם:  ת  נְפָקוּ גֻבְרִין בְּנֵי רִשְׁעָא מִבֵּינָךְ וְאַטְעָיוּ יָת יָתְבֵי קַרְתְּהוֹן לְמֵימָר נְהַךְ וְנִפְלַח לְטַעֲוַת עַמְמַיָא דִי לָא יְדַעְתּוּן:

{טו} וְדָרַשְׁתָּ וְחָקַרְתָּ וְשָׁאַלְתָּ הֵיטֵב וְהִנֵּה אֱמֶת נָכוֹן הַדָּבָר נֶעֶשְׂתָה הַתּוֹעֵבָה הַזֹּאת בְּקִרְבֶּךָ:  מ  וְדָרַשְׁתָּ וְחָקַרְתָּ וְשָׁאַלְתָּ הֵיטֵב וְהִנֵּה אֱמֶת נָכוֹן הַדָּבָר נֶעֶשְׂתָה הַתּוֹעֵבָה הַזֹּאת בְּקִרְבֶּךָ:  ת  וְתִתְבַּע וְתִבְדוֹק וְתִשְׁאַל יָאוּת וְהָא קֻשְׁטָּא כֵּוַן פִּתְגָמָא אִתְעֲבִידָא תּוֹעֶבְתָּא הָדָא בֵּינָךְ:

{טז} הַכֵּה תַכֶּה אֶת יֹשְׁבֵי הָעִיר (ההוא) הַהִיא לְפִי חָרֶב הַחֲרֵם אֹתָהּ וְאֶת כָּל אֲשֶׁר בָּהּ וְאֶת בְּהֶמְתָּהּ לְפִי חָרֶב:  מ  הַכֵּה תַכֶּה אֶת יֹשְׁבֵי הָעִיר (ההוא) הַהִיא לְפִי חָרֶב הַחֲרֵם אֹתָהּ וְאֶת כָּל אֲשֶׁר בָּהּ וְאֶת בְּהֶמְתָּהּ לְפִי חָרֶב:  ת  מִמְחָא תִמְחֵי יָת יָתְבֵי קַרְתָּא הַהִיא לְפִתְגַם דְחָרֶב גַמַר יָתַהּ וְיָת כָּל דִי בַהּ וְיָת בְּעִירַהּ לְפִתְגַם דְחָרֶב:

{יז} וְאֶת כָּל שְׁלָלָהּ תִּקְבֹּץ אֶל תּוֹךְ רְחֹבָהּ וְשָׂרַפְתָּ בָאֵשׁ אֶת הָעִיר וְאֶת כָּל שְׁלָלָהּ כָּלִיל לַיהוָה אֱלֹהֶיךָ וְהָיְתָה תֵּל עוֹלָם לֹא תִבָּנֶה עוֹד:  מ  וְאֶת כָּל שְׁלָלָהּ תִּקְבֹּץ אֶל תּוֹךְ רְחֹבָהּ וְשָׂרַפְתָּ בָאֵשׁ אֶת הָעִיר וְאֶת כָּל שְׁלָלָהּ כָּלִיל לַיהוָה אֱלֹהֶיךָ וְהָיְתָה תֵּל עוֹלָם לֹא תִבָּנֶה עוֹד:  ת  וְיָת כָּל עֲדָאַהּ תִּכְנוֹשׁ לְגוֹ פְתָיַהּ וְתוֹקֵד בְּנוּרָא יָת קַרְתָּא וְיָת כָּל עֲדָאַהּ גְמִיר קֳדָם יְיָ אֱלָהָךְ וּתְהֵי תֵּל חָרִיב לְעָלָם לָא תִתְבְּנֵי עוֹד:

{יח} וְלֹא יִדְבַּק בְּיָדְךָ מְאוּמָה מִן הַחֵרֶם לְמַעַן יָשׁוּב יְהוָה מֵחֲרוֹן אַפּוֹ וְנָתַן לְךָ רַחֲמִים וְרִחַמְךָ וְהִרְבֶּךָ כַּאֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ:  מ  וְלֹא יִדְבַּק בְּיָדְךָ מְאוּמָה מִן הַחֵרֶם לְמַעַן יָשׁוּב יְהוָה מֵחֲרוֹן אַפּוֹ וְנָתַן לְךָ רַחֲמִים וְרִחַמְךָ וְהִרְבֶּךָ כַּאֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ:  ת  וְלָא יִדְבַּק בִּידָךְ מִדַעַם מִן חֶרְמָא בְּדִיל דִיתוּב יְיָ מִתְּקוֹף רוּגְזֵהּ וִיהַב לָךְ רַחֲמִין וִירַחַם עֲלָךְ וְיַסְגִנָךְ כְּמָא דִי קַיִים לַאֲבָהָתָךְ:

{יט} כִּי תִשְׁמַע בְּקוֹל יְהוָה אֱלֹהֶיךָ לִשְׁמֹר אֶת כָּל מִצְוֹתָיו אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לַעֲשׂוֹת הַיָּשָׁר בְּעֵינֵי יְהוָה אֱלֹהֶיךָ:  מ  כִּי תִשְׁמַע בְּקוֹל יְהוָה אֱלֹהֶיךָ לִשְׁמֹר אֶת כָּל מִצְוֹתָיו אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לַעֲשׂוֹת הַיָּשָׁר בְּעֵינֵי יְהוָה אֱלֹהֶיךָ:  ת  אֲרֵי תְקַבֵּל לְמֵימְרָא דַיְיָ אֱלָהָךְ לְמִטַר יָת כָּל פִּקוּדוֹהִי דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין לְמֶעְבַּד דְכָשָׁר קֳדָם יְיָ אֱלָהָךְ: (פ)




דברים פרק-יד

{א}  רביעי  בָּנִים אַתֶּם לַיהוָה אֱלֹהֵיכֶם לֹא תִתְגֹּדְדוּ וְלֹא תָשִׂימוּ קָרְחָה בֵּין עֵינֵיכֶם לָמֵת:  מ  בָּנִים אַתֶּם לַיהוָה אֱלֹהֵיכֶם לֹא תִתְגֹּדְדוּ וְלֹא תָשִׂימוּ קָרְחָה בֵּין עֵינֵיכֶם לָמֵת:  ת  בְּנִין אַתּוּן קֳדָם יְיָ אֱלָהֲכוֹן לָא תִתְחַמְמוּן וְלָא תְשַׁווּן מְרָט בֵּין עֵינֵיכוֹן עַל מִית:

{ב} כִּי עַם קָדוֹשׁ אַתָּה לַיהוָה אֱלֹהֶיךָ וּבְךָ בָּחַר יְהוָה לִהְיוֹת לוֹ לְעַם סְגֻלָּה מִכֹּל הָעַמִּים אֲשֶׁר עַל פְּנֵי הָאֲדָמָה:  מ  כִּי עַם קָדוֹשׁ אַתָּה לַיהוָה אֱלֹהֶיךָ וּבְךָ בָּחַר יְהוָה לִהְיוֹת לוֹ לְעַם סְגֻלָּה מִכֹּל הָעַמִּים אֲשֶׁר עַל פְּנֵי הָאֲדָמָה:  ת  אֲרֵי עַם קַדִישׁ אַתְּ קֳדָם יְיָ אֱלָהָךְ וּבָךְ אִתִּרְעֵי יְיָ לְמֶהֱוֵי לֵהּ לְעַם חַבִּיב מִכָּל עַמְמַיָא דִי עַל אַפֵּי אַרְעָא: (ס)

{ג} לֹא תֹאכַל כָּל תּוֹעֵבָה:  מ  לֹא תֹאכַל כָּל תּוֹעֵבָה:  ת  לָא תֵיכוּל כָּל דִמְרָחָק:

{ד} זֹאת הַבְּהֵמָה אֲשֶׁר תֹּאכֵלוּ שׁוֹר שֵׂה כְשָׂבִים וְשֵׂה עִזִּים:  מ  זֹאת הַבְּהֵמָה אֲשֶׁר תֹּאכֵלוּ שׁוֹר שֵׂה כְשָׂבִים וְשֵׂה עִזִּים:  ת  דֵין בְּעִירָא דְתֵיכְלוּן תּוֹרִין אִמְרִין דִרְחֵלִין וְגָדְיָן דְעִזִין:

{ה} אַיָּל וּצְבִי וְיַחְמוּר וְאַקּוֹ וְדִישֹׁן וּתְאוֹ וָזָמֶר:  מ  אַיָּל וּצְבִי וְיַחְמוּר וְאַקּוֹ וְדִישֹׁן וּתְאוֹ וָזָמֶר:  ת  אַיָלָא וְטַבְיָא וְיַחְמוּרָא וְיַעְלָא וְרֵימָא וְתוֹרְבָּלָא וְדִיצָא:

{ו} וְכָל בְּהֵמָה מַפְרֶסֶת פַּרְסָה וְשֹׁסַעַת שֶׁסַע שְׁתֵּי פְרָסוֹת מַעֲלַת גֵּרָה בַּבְּהֵמָה אֹתָהּ תֹּאכֵלוּ:  מ  וְכָל בְּהֵמָה מַפְרֶסֶת פַּרְסָה וְשֹׁסַעַת שֶׁסַע שְׁתֵּי פְרָסוֹת מַעֲלַת גֵּרָה בַּבְּהֵמָה אֹתָהּ תֹּאכֵלוּ:  ת  וְכָל בְּעִירָא דִסְדִיקָא פַרְסָתָא וּמַטִלְפָא טִלְפִין תַּרְתֵּין פַּרְסָתָא מַסְקָא פִשְׁרָא בִּבְעִירָא יָתַהּ תֵּיכְלוּן:

{ז} אַךְ אֶת זֶה לֹא תֹאכְלוּ מִמַּעֲלֵי הַגֵּרָה וּמִמַּפְרִיסֵי הַפַּרְסָה הַשְּׁסוּעָה אֶת הַגָּמָל וְאֶת הָאַרְנֶבֶת וְאֶת הַשָּׁפָן כִּי מַעֲלֵה גֵרָה הֵמָּה וּפַרְסָה לֹא הִפְרִיסוּ טְמֵאִים הֵם לָכֶם:  מ  אַךְ אֶת זֶה לֹא תֹאכְלוּ מִמַּעֲלֵי הַגֵּרָה וּמִמַּפְרִיסֵי הַפַּרְסָה הַשְּׁסוּעָה אֶת הַגָּמָל וְאֶת הָאַרְנֶבֶת וְאֶת הַשָּׁפָן כִּי מַעֲלֵה גֵרָה הֵמָּה וּפַרְסָה לֹא הִפְרִיסוּ טְמֵאִים הֵם לָכֶם:  ת  בְּרַם יָת דֵין לָא תֵיכְלוּן מִמַסְקֵי פִשְׁרָא וּמִמַסְדִיקֵי פַרְסָתָא מַטִלְפֵי טִלְפַיָא יָת גַמְלָא וְיָת אַרְנְבָא וְיָת טַּפְזָא אֲרֵי מַסְקֵי פִּשְׁרָא אִנוּן וּפַרְסָתְהוֹן לָא סְדִיקָא מְסָאֲבִין אִנוּן לְכוֹן:

{ח} וְאֶת הַחֲזִיר כִּי מַפְרִיס פַּרְסָה הוּא וְלֹא גֵרָה טָמֵא הוּא לָכֶם מִבְּשָׂרָם לֹא תֹאכֵלוּ וּבְנִבְלָתָם לֹא תִגָּעוּ:  מ  וְאֶת הַחֲזִיר כִּי מַפְרִיס פַּרְסָה הוּא וְלֹא גֵרָה טָמֵא הוּא לָכֶם מִבְּשָׂרָם לֹא תֹאכֵלוּ וּבְנִבְלָתָם לֹא תִגָּעוּ:  ת  וְיָת חֲזִירָא אֲרֵי מַסְדִיק פַּרְסָתָא הוּא וְלָא פָשַׁר מְסָאָב הוּא לְכוֹן מִבִּסְרְהוֹן לָא תֵיכְלוּן וּבִנְבֵלְתְּהוֹן לָא תִקְרְבוּן: (ס)

{ט} אֶת זֶה תֹּאכְלוּ מִכֹּל אֲשֶׁר בַּמָּיִם כֹּל אֲשֶׁר לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת תֹּאכֵלוּ:  מ  אֶת זֶה תֹּאכְלוּ מִכֹּל אֲשֶׁר בַּמָּיִם כֹּל אֲשֶׁר לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת תֹּאכֵלוּ:  ת  יָת דֵין תֵּיכְלוּן מִכֹּל דִי בְמַיָא כֹּל דִי לֵהּ צִיצִין וְקַלְפִין תֵּיכְלוּן:

{י} וְכֹל אֲשֶׁר אֵין לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת לֹא תֹאכֵלוּ טָמֵא הוּא לָכֶם:  מ  וְכֹל אֲשֶׁר אֵין לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת לֹא תֹאכֵלוּ טָמֵא הוּא לָכֶם:  ת  וְכֹל דִי לֵית לֵהּ צִיצִין וְקַלְפִין לָא תֵיכְלוּן מְסָאָב הוּא לְכוֹן: (ס)

{יא} כָּל צִפּוֹר טְהֹרָה תֹּאכֵלוּ:  מ  כָּל צִפּוֹר טְהֹרָה תֹּאכֵלוּ:  ת  כָּל צִפַּר דַכְיָא תֵּיכְלוּן:

{יב} וְזֶה אֲשֶׁר לֹא תֹאכְלוּ מֵהֶם הַנֶּשֶׁר וְהַפֶּרֶס וְהָעָזְנִיָּה:  מ  וְזֶה אֲשֶׁר לֹא תֹאכְלוּ מֵהֶם הַנֶּשֶׁר וְהַפֶּרֶס וְהָעָזְנִיָּה:  ת  וְדֵין דִי לָא תֵיכְלוּן מִנְהוֹן נִשְׁרָא וְעָר וְעַזְיָא:

{יג} וְהָרָאָה וְאֶת הָאַיָּה וְהַדַּיָּה לְמִינָהּ:  מ  וְהָרָאָה וְאֶת הָאַיָּה וְהַדַּיָּה לְמִינָהּ:  ת  וּבַת כַּנְפָא וְטָרָפִיתָא וְדַיְתָא לִזְנַהּ:

{יד} וְאֵת כָּל עֹרֵב לְמִינוֹ:  מ  וְאֵת כָּל עֹרֵב לְמִינוֹ:  ת  וְיָת כָּל עֹרְבָא לִזְנֵהּ:

{טו} וְאֵת בַּת הַיַּעֲנָה וְאֶת הַתַּחְמָס וְאֶת הַשָּׁחַף וְאֶת הַנֵּץ לְמִינֵהוּ:  מ  וְאֵת בַּת הַיַּעֲנָה וְאֶת הַתַּחְמָס וְאֶת הַשָּׁחַף וְאֶת הַנֵּץ לְמִינֵהוּ:  ת  וְיָת בַּת נַעֲמִיתָא וְצִיצָא וְצִפַּר שַׁחְפָּא וְנָצָא לִזְנוֹהִי:

{טז} אֶת הַכּוֹס וְאֶת הַיַּנְשׁוּף וְהַתִּנְשָׁמֶת:  מ  אֶת הַכּוֹס וְאֶת הַיַּנְשׁוּף וְהַתִּנְשָׁמֶת:  ת  וְקַדְיָא וְקִפוֹפָא וּבַוְתָא:

{יז} וְהַקָּאָת וְאֶת הָרָחָמָה וְאֶת הַשָּׁלָךְ:  מ  וְהַקָּאָת וְאֶת הָרָחָמָה וְאֶת הַשָּׁלָךְ:  ת  וְקָתָא וִירַקְרֵקָא וְשָׁלֵי נוּנָא:

{יח} וְהַחֲסִידָה וְהָאֲנָפָה לְמִינָהּ וְהַדּוּכִיפַת וְהָעֲטַלֵּף:  מ  וְהַחֲסִידָה וְהָאֲנָפָה לְמִינָהּ וְהַדּוּכִיפַת וְהָעֲטַלֵּף:  ת  וְחַוָרִיתָא וְאִבּוּ לִזְנַהּ וְנַגַר טוּרָא וַעֲטַּלֵפָא:

{יט} וְכֹל שֶׁרֶץ הָעוֹף טָמֵא הוּא לָכֶם לֹא יֵאָכֵלוּ:  מ  וְכֹל שֶׁרֶץ הָעוֹף טָמֵא הוּא לָכֶם לֹא יֵאָכֵלוּ:  ת  וְכֹל רִחֲשָׁא דְעוֹפָא מְסָאָב הוּא לְכוֹן לָא יִתְאָכְלוּן:

{כ} כָּל עוֹף טָהוֹר תֹּאכֵלוּ:  מ  כָּל עוֹף טָהוֹר תֹּאכֵלוּ:  ת  כָּל עוֹפָא דְכֵי תֵּיכְלוּן:

{כא} לֹא תֹאכְלוּ כָל נְבֵלָה לַגֵּר אֲשֶׁר בִּשְׁעָרֶיךָ תִּתְּנֶנָּה וַאֲכָלָהּ אוֹ מָכֹר לְנָכְרִי כִּי עַם קָדוֹשׁ אַתָּה לַיהוָה אֱלֹהֶיךָ לֹא תְבַשֵּׁל גְּדִי בַּחֲלֵב אִמּוֹ:  מ  לֹא תֹאכְלוּ כָל נְבֵלָה לַגֵּר אֲשֶׁר בִּשְׁעָרֶיךָ תִּתְּנֶנָּה וַאֲכָלָהּ אוֹ מָכֹר לְנָכְרִי כִּי עַם קָדוֹשׁ אַתָּה לַיהוָה אֱלֹהֶיךָ לֹא תְבַשֵּׁל גְּדִי בַּחֲלֵב אִמּוֹ:  ת  לָא תֵיכְלוּן כָּל נְבִילָא לְתוֹתָב עָרֵל דִי בְקִרְוָיךְ תִּתְּנִנַהּ וְיֵיכְלִנַהּ אוֹ תְזַבְּנִנַהּ לְבַר עַמְמִין אֲרֵי עַם קַדִישׁ אַתְּ קֳדָם יְיָ אֱלָהָךְ לָא תֵיכוּל בְּשַׂר בַּחֲלָב: (ס)

{כב}  חמישי  עַשֵּׂר תְּעַשֵּׂר אֵת כָּל תְּבוּאַת זַרְעֶךָ הַיֹּצֵא הַשָּׂדֶה שָׁנָה שָׁנָה:  מ  עַשֵּׂר תְּעַשֵּׂר אֵת כָּל תְּבוּאַת זַרְעֶךָ הַיֹּצֵא הַשָּׂדֶה שָׁנָה שָׁנָה:  ת  עַשְׂרָא תְעַשַׂר יָת כָּל עֲלָלַת זַרְעָךְ דְיִפּוֹק חַקְלָא שַׁתָּא בְשַׁתָּא:

{כג} וְאָכַלְתָּ לִפְנֵי יְהוָה אֱלֹהֶיךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר לְשַׁכֵּן שְׁמוֹ שָׁם מַעְשַׂר דְּגָנְךָ תִּירֹשְׁךָ וְיִצְהָרֶךָ וּבְכֹרֹת בְּקָרְךָ וְצֹאנֶךָ לְמַעַן תִּלְמַד לְיִרְאָה אֶת יְהוָה אֱלֹהֶיךָ כָּל הַיָּמִים:  מ  וְאָכַלְתָּ לִפְנֵי יְהוָה אֱלֹהֶיךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר לְשַׁכֵּן שְׁמוֹ שָׁם מַעְשַׂר דְּגָנְךָ תִּירֹשְׁךָ וְיִצְהָרֶךָ וּבְכֹרֹת בְּקָרְךָ וְצֹאנֶךָ לְמַעַן תִּלְמַד לְיִרְאָה אֶת יְהוָה אֱלֹהֶיךָ כָּל הַיָּמִים:  ת  וְתֵיכוּל קֳדָם יְיָ אֱלָהָךְ בְּאַתְרָא דִי יִתִּרְעֵי לְאַשְׁרָאָה שְׁכִנְתֵּהּ תַּמָן מַעְשַׂר עִבוּרָךְ חַמְרָךְ וּמִשְׁחָךְ וּבְכוֹרֵי תוֹרָךְ וְעָנָךְ בְּדִיל דְתֵילַף לְמִדְחַל קֳדָם יְיָ אֱלָהָךְ כָּל יוֹמַיָא:

{כד} וְכִי יִרְבֶּה מִמְּךָ הַדֶּרֶךְ כִּי לֹא תוּכַל שְׂאֵתוֹ כִּי יִרְחַק מִמְּךָ הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ לָשׂוּם שְׁמוֹ שָׁם כִּי יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ:  מ  וְכִי יִרְבֶּה מִמְּךָ הַדֶּרֶךְ כִּי לֹא תוּכַל שְׂאֵתוֹ כִּי יִרְחַק מִמְּךָ הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ לָשׂוּם שְׁמוֹ שָׁם כִּי יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ:  ת  וַאֲרֵי יִסְגֵי מִנָךְ אָרְחָא אֲרֵי לָא תִכּוּל לְמִטְלֵהּ אֲרֵי יִתְרָחַק מִנָךְ אַתְרָא דִי יִתִּרְעֵי יְיָ אֱלָהָךְ לְאַשְׁרָאָה שְׁכִנְתֵּהּ תַּמָן אֲרֵי יְבָרֵכִנָךְ יְיָ אֱלָהָךְ:

{כה} וְנָתַתָּה בַּכָּסֶף וְצַרְתָּ הַכֶּסֶף בְּיָדְךָ וְהָלַכְתָּ אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ:  מ  וְנָתַתָּה בַּכָּסֶף וְצַרְתָּ הַכֶּסֶף בְּיָדְךָ וְהָלַכְתָּ אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ:  ת  וְתִתֵּן בְּכַסְפָּא וּתְצוּר כַּסְפָּא בִּידָךְ וּתְהַךְ לְאַתְרָא דִי יִתִּרְעֵי יְיָ אֱלָהָךְ בֵּהּ:

{כו} וְנָתַתָּה הַכֶּסֶף בְּכֹל אֲשֶׁר תְּאַוֶּה נַפְשְׁךָ בַּבָּקָר וּבַצֹּאן וּבַיַּיִן וּבַשֵּׁכָר וּבְכֹל אֲשֶׁר תִּשְׁאָלְךָ נַפְשֶׁךָ וְאָכַלְתָּ שָּׁם לִפְנֵי יְהוָה אֱלֹהֶיךָ וְשָׂמַחְתָּ אַתָּה וּבֵיתֶךָ:  מ  וְנָתַתָּה הַכֶּסֶף בְּכֹל אֲשֶׁר תְּאַוֶּה נַפְשְׁךָ בַּבָּקָר וּבַצֹּאן וּבַיַּיִן וּבַשֵּׁכָר וּבְכֹל אֲשֶׁר תִּשְׁאָלְךָ נַפְשֶׁךָ וְאָכַלְתָּ שָּׁם לִפְנֵי יְהוָה אֱלֹהֶיךָ וְשָׂמַחְתָּ אַתָּה וּבֵיתֶךָ:  ת  וְתִתֵּן כַּסְפָּא בְּכֹל דִי יִתִּרְעֵי נַפְשָׁךְ בְּתוֹרֵי וּבְעָנָא וּבַחֲמַר חֲדַת וְעַתִּיק וּבְכֹל דִי תִשְׁאֲלִנָךְ נַפְשָׁךְ וְתֵיכוּל תַּמָן קֳדָם יְיָ אֱלָהָךְ וְתֶחֱדֵי אַתְּ וֶאֱנַשׁ בֵּיתָךְ:

{כז} וְהַלֵּוִי אֲשֶׁר בִּשְׁעָרֶיךָ לֹא תַעַזְבֶנּוּ כִּי אֵין לוֹ חֵלֶק וְנַחֲלָה עִמָּךְ:  מ  וְהַלֵּוִי אֲשֶׁר בִּשְׁעָרֶיךָ לֹא תַעַזְבֶנּוּ כִּי אֵין לוֹ חֵלֶק וְנַחֲלָה עִמָּךְ:  ת  וְלֵוָאָה דִי בְקִרְוָיךְ לָא תִשְׁבְּקִנֵהּ אֲרֵי לֵית לֵהּ חֳלָק וְאַחֲסָנָא עִמָךְ: (ס)

{כח} מִקְצֵה שָׁלֹשׁ שָׁנִים תּוֹצִיא אֶת כָּל מַעְשַׂר תְּבוּאָתְךָ בַּשָּׁנָה הַהִוא וְהִנַּחְתָּ בִּשְׁעָרֶיךָ:  מ  מִקְצֵה שָׁלֹשׁ שָׁנִים תּוֹצִיא אֶת כָּל מַעְשַׂר תְּבוּאָתְךָ בַּשָּׁנָה הַהִוא וְהִנַּחְתָּ בִּשְׁעָרֶיךָ:  ת  מִסוֹף תְּלַת שְׁנִין תַּפֵק יָת כָּל מַעְשַׂר עֲלַלְתָּךְ בְּשַׁתָּא הַהִיא וְתַצְנַע בְּקִרְוָיךְ:

{כט} וּבָא הַלֵּוִי כִּי אֵין לוֹ חֵלֶק וְנַחֲלָה עִמָּךְ וְהַגֵּר וְהַיָּתוֹם וְהָאַלְמָנָה אֲשֶׁר בִּשְׁעָרֶיךָ וְאָכְלוּ וְשָׂבֵעוּ לְמַעַן יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכָל מַעֲשֵׂה יָדְךָ אֲשֶׁר תַּעֲשֶׂה:  מ  וּבָא הַלֵּוִי כִּי אֵין לוֹ חֵלֶק וְנַחֲלָה עִמָּךְ וְהַגֵּר וְהַיָּתוֹם וְהָאַלְמָנָה אֲשֶׁר בִּשְׁעָרֶיךָ וְאָכְלוּ וְשָׂבֵעוּ לְמַעַן יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכָל מַעֲשֵׂה יָדְךָ אֲשֶׁר תַּעֲשֶׂה:  ת  וְיֵיתֵי לֵוָאָה אֲרֵי לֵית לֵהּ חֳלָק וְאַחֲסָנָא עִמָךְ וְגִיוֹרָא וְיִתַּמָא וְאַרְמְלָא דִי בְקִרְוָיךְ וְיֵיכְלוּן וְיִשְׂבְּעוּן בְּדִיל דִי יְבָרֵכִנָךְ יְיָ אֱלָהָךְ בְּכָל עוֹבָדֵי יְדָךְ דִי תַעְבֵּד: (ס)




דברים פרק-טו

{א}  שישי  מִקֵּץ שֶׁבַע שָׁנִים תַּעֲשֶׂה שְׁמִטָּה:  מ  מִקֵּץ שֶׁבַע שָׁנִים תַּעֲשֶׂה שְׁמִטָּה:  ת  מִסוֹף שְׁבַע שְׁנִין תַּעְבֵּד שְׁמִטְתָא:

{ב} וְזֶה דְּבַר הַשְּׁמִטָּה שָׁמוֹט כָּל בַּעַל מַשֵּׁה יָדוֹ אֲשֶׁר יַשֶּׁה בְּרֵעֵהוּ לֹא יִגֹּשׂ אֶת רֵעֵהוּ וְאֶת אָחִיו כִּי קָרָא שְׁמִטָּה לַיהוָה:  מ  וְזֶה דְּבַר הַשְּׁמִטָּה שָׁמוֹט כָּל בַּעַל מַשֵּׁה יָדוֹ אֲשֶׁר יַשֶּׁה בְּרֵעֵהוּ לֹא יִגֹּשׂ אֶת רֵעֵהוּ וְאֶת אָחִיו כִּי קָרָא שְׁמִטָּה לַיהוָה:  ת  וְדֵין פִּתְגַם שְׁמִטְתָא דְתַשְׁמֵט כָּל גְבַר מַרֵי רְשׁוּ דִי יַרְשֵׁי בְּחַבְרֵהּ לָא יִתְבַּע מִן חַבְרֵהּ וּמִן אֲחוּהִי אֲרֵי קְרָא שְׁמִטְתָא קֳדָם יְיָ:

{ג} אֶת הַנָּכְרִי תִּגֹּשׂ וַאֲשֶׁר יִהְיֶה לְךָ אֶת אָחִיךָ תַּשְׁמֵט יָדֶךָ:  מ  אֶת הַנָּכְרִי תִּגֹּשׂ וַאֲשֶׁר יִהְיֶה לְךָ אֶת אָחִיךָ תַּשְׁמֵט יָדֶךָ:  ת  מִן בַּר עַמְמִין תִּתְבַּע וְדִי יְהֵי לָךְ עִם אָחוּךְ תַּשְׁמֵט יְדָךְ:

{ד} אֶפֶס כִּי לֹא יִהְיֶה בְּךָ אֶבְיוֹן כִּי בָרֵךְ יְבָרֶכְךָ יְהוָה בָּאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה לְרִשְׁתָּהּ:  מ  אֶפֶס כִּי לֹא יִהְיֶה בְּךָ אֶבְיוֹן כִּי בָרֵךְ יְבָרֶכְךָ יְהוָה בָּאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה לְרִשְׁתָּהּ:  ת  לְחוֹד אֲרֵי לָא יְהֵי בָךְ מִסְכֵּנָא אֲרֵי בָרָכָא יְבָרֵכִנָךְ יְיָ בְּאַרְעָא דִי יְיָ אֱלָהָךְ יָהֵב לָךְ אַחֲסָנָא לְמֵירְתַהּ:

{ה} רַק אִם שָׁמוֹעַ תִּשְׁמַע בְּקוֹל יְהוָה אֱלֹהֶיךָ לִשְׁמֹר לַעֲשׂוֹת אֶת כָּל הַמִּצְוָה הַזֹּאת אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם:  מ  רַק אִם שָׁמוֹעַ תִּשְׁמַע בְּקוֹל יְהוָה אֱלֹהֶיךָ לִשְׁמֹר לַעֲשׂוֹת אֶת כָּל הַמִּצְוָה הַזֹּאת אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם:  ת  לְחוֹד אִם קַבָּלָא תְקַבֵּל לְמֵימְרָא דַיְיָ אֱלָהָךְ לְמִטַר לְמֶעְבַּד יָת כָּל תַּפְקֶדְתָּא הָדָא דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין:

{ו} כִּי יְהוָה אֱלֹהֶיךָ בֵּרַכְךָ כַּאֲשֶׁר דִּבֶּר לָךְ וְהַעֲבַטְתָּ גּוֹיִם רַבִּים וְאַתָּה לֹא תַעֲבֹט וּמָשַׁלְתָּ בְּגוֹיִם רַבִּים וּבְךָ לֹא יִמְשֹׁלוּ:  מ  כִּי יְהוָה אֱלֹהֶיךָ בֵּרַכְךָ כַּאֲשֶׁר דִּבֶּר לָךְ וְהַעֲבַטְתָּ גּוֹיִם רַבִּים וְאַתָּה לֹא תַעֲבֹט וּמָשַׁלְתָּ בְּגוֹיִם רַבִּים וּבְךָ לֹא יִמְשֹׁלוּ:  ת  אֲרֵי יְיָ אֱלָהָךְ בָּרְכָךְ כְּמָא דִי מַלִיל לָךְ וְתוֹזֵף לְעַמְמִין סַגִיאִין וְאַתְּ לָא תְזוּף וְתִשְׁלוֹט בְּעַמְמִין סַגִיאִין וּבָךְ לָא יִשְׁלְטוּן: (ס)

{ז} כִּי יִהְיֶה בְךָ אֶבְיוֹן מֵאַחַד אַחֶיךָ בְּאַחַד שְׁעָרֶיךָ בְּאַרְצְךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ לֹא תְאַמֵּץ אֶת לְבָבְךָ וְלֹא תִקְפֹּץ אֶת יָדְךָ מֵאָחִיךָ הָאֶבְיוֹן:  מ  כִּי יִהְיֶה בְךָ אֶבְיוֹן מֵאַחַד אַחֶיךָ בְּאַחַד שְׁעָרֶיךָ בְּאַרְצְךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ לֹא תְאַמֵּץ אֶת לְבָבְךָ וְלֹא תִקְפֹּץ אֶת יָדְךָ מֵאָחִיךָ הָאֶבְיוֹן:  ת  אֲרֵי יְהֵי בָךְ מִסְכֵּנָא מֵחַד מֵאַחָיךְ בַּחֲדָא מִקִרְוָיךְ בְּאַרְעָךְ דַיְיָ אֱלָהָךְ יָהֵב לָךְ לָא תַתְקֵף יָת לִבָּךְ וְלָא תִקְפֹּץ יָת יְדָךְ מֵאֲחוּךְ מִסְכֵּנָא:

{ח} כִּי פָתֹחַ תִּפְתַּח אֶת יָדְךָ לוֹ וְהַעֲבֵט תַּעֲבִיטֶנּוּ דֵּי מַחְסֹרוֹ אֲשֶׁר יֶחְסַר לוֹ:  מ  כִּי פָתֹחַ תִּפְתַּח אֶת יָדְךָ לוֹ וְהַעֲבֵט תַּעֲבִיטֶנּוּ דֵּי מַחְסֹרוֹ אֲשֶׁר יֶחְסַר לוֹ:  ת  אֶלָא מִפְתַּח תִּפְתַּח יָת יְדָךְ לֵהּ וְאוֹזָפָא תּוֹזְפִנֵהּ כְּמִסַת חֶסְרוֹנֵהּ דְיַחֲסִיר לֵהּ:

{ט} הִשָּׁמֶר לְךָ פֶּן יִהְיֶה דָבָר עִם לְבָבְךָ בְלִיַּעַל לֵאמֹר קָרְבָה שְׁנַת הַשֶּׁבַע שְׁנַת הַשְּׁמִטָּה וְרָעָה עֵינְךָ בְּאָחִיךָ הָאֶבְיוֹן וְלֹא תִתֵּן לוֹ וְקָרָא עָלֶיךָ אֶל יְהוָה וְהָיָה בְךָ חֵטְא:  מ  הִשָּׁמֶר לְךָ פֶּן יִהְיֶה דָבָר עִם לְבָבְךָ בְלִיַּעַל לֵאמֹר קָרְבָה שְׁנַת הַשֶּׁבַע שְׁנַת הַשְּׁמִטָּה וְרָעָה עֵינְךָ בְּאָחִיךָ הָאֶבְיוֹן וְלֹא תִתֵּן לוֹ וְקָרָא עָלֶיךָ אֶל יְהוָה וְהָיָה בְךָ חֵטְא:  ת  אִסְתַּמַר לָךְ דִילְמָא יְהֵי פִתְגָם עִם לִבָּךְ בִּרְשַׁע לְמֵימַר קְרִיבָא שַׁתָּא שְׁבִיעֲתָא שַׁתָּא דִשְׁמִטְתָא וְתַבְאֵשׁ עֵינָךְ בְּאָחוּךְ מִסְכֵּנָא וְלָא תִתֵּן לֵהּ וְיִקְרֵי עֲלָךְ קֳדָם יְיָ וִיהֵי בָךְ חוֹבָא:

{י} נָתוֹן תִּתֵּן לוֹ וְלֹא יֵרַע לְבָבְךָ בְּתִתְּךָ לוֹ כִּי בִּגְלַל הַדָּבָר הַזֶּה יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכָל מַעֲשֶׂךָ וּבְכֹל מִשְׁלַח יָדֶךָ:  מ  נָתוֹן תִּתֵּן לוֹ וְלֹא יֵרַע לְבָבְךָ בְּתִתְּךָ לוֹ כִּי בִּגְלַל הַדָּבָר הַזֶּה יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכָל מַעֲשֶׂךָ וּבְכֹל מִשְׁלַח יָדֶךָ:  ת  מִתַּן תִתֵּן לֵהּ וְלָא יַבְאֵשׁ לִבָּךְ בְּמִתְּנָךְ לֵהּ אֲרֵי בְּדִיל פִּתְגָמָא הָדֵין יְבָרֵכִנָךְ יְיָ אֱלָהָךְ בְּכָל עוֹבָדָךְ וּבְכֹל אוֹשָׁטוּת יְדָךְ:

{יא} כִּי לֹא יֶחְדַּל אֶבְיוֹן מִקֶּרֶב הָאָרֶץ עַל כֵּן אָנֹכִי מְצַוְּךָ לֵאמֹר פָּתֹחַ תִּפְתַּח אֶת יָדְךָ לְאָחִיךָ לַעֲנִיֶּךָ וּלְאֶבְיֹנְךָ בְּאַרְצֶךָ:  מ  כִּי לֹא יֶחְדַּל אֶבְיוֹן מִקֶּרֶב הָאָרֶץ עַל כֵּן אָנֹכִי מְצַוְּךָ לֵאמֹר פָּתֹחַ תִּפְתַּח אֶת יָדְךָ לְאָחִיךָ לַעֲנִיֶּךָ וּלְאֶבְיֹנְךָ בְּאַרְצֶךָ:  ת  אֲרֵי לָא יִפְסוֹק מִסְכֵּנָא מִגוֹ אַרְעָא עַל כֵּן אֲנָא מְפַקְדָךְ לְמֵימַר מִפְתַּח תִּפְתַּח יָת יְדָךְ לַאֲחוּךְ לְעַנְיָךְ וּלְמִסְכֵּנָךְ בְּאַרְעָךְ: (ס)

{יב} כִּי יִמָּכֵר לְךָ אָחִיךָ הָעִבְרִי אוֹ הָעִבְרִיָּה וַעֲבָדְךָ שֵׁשׁ שָׁנִים וּבַשָּׁנָה הַשְּׁבִיעִת תְּשַׁלְּחֶנּוּ חָפְשִׁי מֵעִמָּךְ:  מ  כִּי יִמָּכֵר לְךָ אָחִיךָ הָעִבְרִי אוֹ הָעִבְרִיָּה וַעֲבָדְךָ שֵׁשׁ שָׁנִים וּבַשָּׁנָה הַשְּׁבִיעִת תְּשַׁלְּחֶנּוּ חָפְשִׁי מֵעִמָּךְ:  ת  אֲרֵי יִזְדַבַּן לָךְ אֲחוּךְ בַּר יִשְׂרָאֵל אוֹ בַּת יִשְׂרָאֵל וְיִפְלְחִנָךְ שִׁית שְׁנִין וּבְשַׁתָּא שְׁבִיעֲתָא תִּפְטְרִנֵהּ בַּר חוֹרִין מֵעִמָךְ:

{יג} וְכִי תְשַׁלְּחֶנּוּ חָפְשִׁי מֵעִמָּךְ לֹא תְשַׁלְּחֶנּוּ רֵיקָם:  מ  וְכִי תְשַׁלְּחֶנּוּ חָפְשִׁי מֵעִמָּךְ לֹא תְשַׁלְּחֶנּוּ רֵיקָם:  ת  וַאֲרֵי תִפְטְרִנֵה בַּר חוֹרִין מֵעִמָךְ לָא תִפְטְּרִנֵה רֵיקָן:

{יד} הַעֲנֵיק תַּעֲנִיק לוֹ מִצֹּאנְךָ וּמִגָּרְנְךָ וּמִיִּקְבֶךָ אֲשֶׁר בֵּרַכְךָ יְהוָה אֱלֹהֶיךָ תִּתֶּן לוֹ:  מ  הַעֲנֵיק תַּעֲנִיק לוֹ מִצֹּאנְךָ וּמִגָּרְנְךָ וּמִיִּקְבֶךָ אֲשֶׁר בֵּרַכְךָ יְהוָה אֱלֹהֶיךָ תִּתֶּן לוֹ:  ת  אַפְרָשָׁא תַפְרֵשׁ לֵהּ מֵעָנָךְ וּמֵאִדְרָךְ וּמִמַעֲצַרְתָּךְ דִי בָרְכָךְ יְיָ אֱלָהָךְ תִּתֶּן לֵהּ:

{טו} וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרַיִם וַיִּפְדְּךָ יְהוָה אֱלֹהֶיךָ עַל כֵּן אָנֹכִי מְצַוְּךָ אֶת הַדָּבָר הַזֶּה הַיּוֹם:  מ  וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרַיִם וַיִּפְדְּךָ יְהוָה אֱלֹהֶיךָ עַל כֵּן אָנֹכִי מְצַוְּךָ אֶת הַדָּבָר הַזֶּה הַיּוֹם:  ת  וְתִדְכַּר אֲרֵי עַבְדָא הֲוֵיתָא בְּאַרְעָא דְמִצְרַיִם וּפָרְקָךְ יְיָ אֱלָהָךְ עַל כֵּן אֲנָא מְפַקְדָךְ יָת פִּתְגָמָא הָדֵין יוֹמָא דֵין:

{טז} וְהָיָה כִּי יֹאמַר אֵלֶיךָ לֹא אֵצֵא מֵעִמָּךְ כִּי אֲהֵבְךָ וְאֶת בֵּיתֶךָ כִּי טוֹב לוֹ עִמָּךְ:  מ  וְהָיָה כִּי יֹאמַר אֵלֶיךָ לֹא אֵצֵא מֵעִמָּךְ כִּי אֲהֵבְךָ וְאֶת בֵּיתֶךָ כִּי טוֹב לוֹ עִמָּךְ:  ת  וִיהֵי אֲרֵי יֵימַר לָךְ לָא אֶפוֹק מֵעִמָךְ אֲרֵי רָחֲמָךְ וְיָת אֱנַשׁ בֵּיתָךְ אֲרֵי טַב לֵהּ עִמָךְ:

{יז} וְלָקַחְתָּ אֶת הַמַּרְצֵעַ וְנָתַתָּה בְאָזְנוֹ וּבַדֶּלֶת וְהָיָה לְךָ עֶבֶד עוֹלָם וְאַף לַאֲמָתְךָ תַּעֲשֶׂה כֵּן:  מ  וְלָקַחְתָּ אֶת הַמַּרְצֵעַ וְנָתַתָּה בְאָזְנוֹ וּבַדֶּלֶת וְהָיָה לְךָ עֶבֶד עוֹלָם וְאַף לַאֲמָתְךָ תַּעֲשֶׂה כֵּן:  ת  וְתִסַב יָת מַרְצְעָא וְתִתֵּן בְּאֻדְנֵהּ וּבְדָשָׁא וִיהֵי לָךְ עִבֵד פָּלַח לְעָלָם וְאַף לְאַמְתָךְ תַּעְבֶּד כֵּן:

{יח} לֹא יִקְשֶׁה בְעֵינֶךָ בְּשַׁלֵּחֲךָ אֹתוֹ חָפְשִׁי מֵעִמָּךְ כִּי מִשְׁנֶה שְׂכַר שָׂכִיר עֲבָדְךָ שֵׁשׁ שָׁנִים וּבֵרַכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל אֲשֶׁר תַּעֲשֶׂה:  מ  לֹא יִקְשֶׁה בְעֵינֶךָ בְּשַׁלֵּחֲךָ אֹתוֹ חָפְשִׁי מֵעִמָּךְ כִּי מִשְׁנֶה שְׂכַר שָׂכִיר עֲבָדְךָ שֵׁשׁ שָׁנִים וּבֵרַכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל אֲשֶׁר תַּעֲשֶׂה:  ת  לָא יִקְשֵׁי בְעֵינָךְ בְּמִפְטְרָךְ יָתֵהּ בַּר חוֹרִין מֵעִמָךְ אֲרֵי עַל חַד תְּרֵין כַּאֲגַר אֲגִירָא פָּלְחָךְ שִׁית שְׁנִין וִיבָרֵכִנָךְ יְיָ אֱלָהָךְ בְּכֹל דִי תַעְבֵּד: (ס)

{יט}  שביעי  כָּל הַבְּכוֹר אֲשֶׁר יִוָּלֵד בִּבְקָרְךָ וּבְצֹאנְךָ הַזָּכָר תַּקְדִּישׁ לַיהוָה אֱלֹהֶיךָ לֹא תַעֲבֹד בִּבְכֹר שׁוֹרֶךָ וְלֹא תָגֹז בְּכוֹר צֹאנֶךָ:  מ  כָּל הַבְּכוֹר אֲשֶׁר יִוָּלֵד בִּבְקָרְךָ וּבְצֹאנְךָ הַזָּכָר תַּקְדִּישׁ לַיהוָה אֱלֹהֶיךָ לֹא תַעֲבֹד בִּבְכֹר שׁוֹרֶךָ וְלֹא תָגֹז בְּכוֹר צֹאנֶךָ:  ת  כָּל בּוּכְרָא דִי יִתְיְלִיד בְּתוֹרָךְ וּבְעָנָךְ דִכְרִין תַּקְדֵשׁ קָדָם יְיָ אֱלָהָךְ לָא תִפְלַח בְּבוּכְרָא דְתוֹרָךְ וְלָא תֵגוֹז בּוּכְרָא דְעָנָךְ:

{כ} לִפְנֵי יְהוָה אֱלֹהֶיךָ תֹאכֲלֶנּוּ שָׁנָה בְשָׁנָה בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אַתָּה וּבֵיתֶךָ:  מ  לִפְנֵי יְהוָה אֱלֹהֶיךָ תֹאכֲלֶנּוּ שָׁנָה בְשָׁנָה בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אַתָּה וּבֵיתֶךָ:  ת  קֳדָם יְיָ אֱלָהָךְ תֵּיכְלִנֵהּ שַׁתָּא בְשַׁתָּא בְּאַתְרָא דִי יִתִּרְעֵי יְיָ אַתְּ וֶאֱנַשׁ בֵּיתָךְ:

{כא} וְכִי יִהְיֶה בוֹ מוּם פִּסֵּחַ אוֹ עִוֵּר כֹּל מוּם רָע לֹא תִזְבָּחֶנּוּ לַיהוָה אֱלֹהֶיךָ:  מ  וְכִי יִהְיֶה בוֹ מוּם פִּסֵּחַ אוֹ עִוֵּר כֹּל מוּם רָע לֹא תִזְבָּחֶנּוּ לַיהוָה אֱלֹהֶיךָ:  ת  וַאֲרֵי יְהֵי בֵהּ מוּמָא חֲגִיר אוֹ עֲוִיר כֹּל מוּם בִּישׁ לָא תִכְּסִנֵהּ קֳדָם יְיָ אֱלָהָךְ:

{כב} בִּשְׁעָרֶיךָ תֹּאכֲלֶנּוּ הַטָּמֵא וְהַטָּהוֹר יַחְדָּו כַּצְּבִי וְכָאַיָּל:  מ  בִּשְׁעָרֶיךָ תֹּאכֲלֶנּוּ הַטָּמֵא וְהַטָּהוֹר יַחְדָּו כַּצְּבִי וְכָאַיָּל:  ת  בְּקִרְוָיךְ תֵּיכְלִנֵהּ מְסָאָבָא וְדַכְיָא כַּחֲדָא כִּבְסַר טַבְיָא וְאַיְלָא:

{כג} רַק אֶת דָּמוֹ לֹא תֹאכֵל עַל הָאָרֶץ תִּשְׁפְּכֶנּוּ כַּמָּיִם:  מ  רַק אֶת דָּמוֹ לֹא תֹאכֵל עַל הָאָרֶץ תִּשְׁפְּכֶנּוּ כַּמָּיִם:  ת  לְחוֹד יָת דְמֵהּ לָא תֵיכוּל עַל אַרְעָא תִשְׁדִנֵהּ כְּמַיָא: (ס)




דברים פרק-טז

{א} שָׁמוֹר אֶת חֹדֶשׁ הָאָבִיב וְעָשִׂיתָ פֶּסַח לַיהוָה אֱלֹהֶיךָ כִּי בְּחֹדֶשׁ הָאָבִיב הוֹצִיאֲךָ יְהוָה אֱלֹהֶיךָ מִמִּצְרַיִם לָיְלָה:  מ  שָׁמוֹר אֶת חֹדֶשׁ הָאָבִיב וְעָשִׂיתָ פֶּסַח לַיהוָה אֱלֹהֶיךָ כִּי בְּחֹדֶשׁ הָאָבִיב הוֹצִיאֲךָ יְהוָה אֱלֹהֶיךָ מִמִּצְרַיִם לָיְלָה:  ת  טַר יָת יַרְחָא דַאֲבִיבָא וְתַעְבֵּד פִּסְחָא קֳדָם יְיָ אֱלָהָךְ אֲרֵי בְּיַרְחָא דַאֲבִיבָא אַפְּקָךְ יְיָ אֱלָהָךְ מִמִצְרַיִם וַעֲבַד לָךְ נִסִין בְּלֵילְיָא:

{ב} וְזָבַחְתָּ פֶּסַח לַיהוָה אֱלֹהֶיךָ צֹאן וּבָקָר בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה לְשַׁכֵּן שְׁמוֹ שָׁם:  מ  וְזָבַחְתָּ פֶּסַח לַיהוָה אֱלֹהֶיךָ צֹאן וּבָקָר בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה לְשַׁכֵּן שְׁמוֹ שָׁם:  ת  וְתִכּוֹס פִּסְחָא קֳדָם יְיָ אֱלָהָךְ מִן בְּנֵי עָנָא וְנִכְסַת קוּדְשַׁיָא מִן תּוֹרֵי בְּאַתְרָא דִי יִתִּרְעֵי יְיָ לְאַשְׁרָאָה שְׁכִנְתֵּה תַּמָן:

{ג} לֹא תֹאכַל עָלָיו חָמֵץ שִׁבְעַת יָמִים תֹּאכַל עָלָיו מַצּוֹת לֶחֶם עֹנִי כִּי בְחִפָּזוֹן יָצָאתָ מֵאֶרֶץ מִצְרַיִם לְמַעַן תִּזְכֹּר אֶת יוֹם צֵאתְךָ מֵאֶרֶץ מִצְרַיִם כֹּל יְמֵי חַיֶּיךָ:  מ  לֹא תֹאכַל עָלָיו חָמֵץ שִׁבְעַת יָמִים תֹּאכַל עָלָיו מַצּוֹת לֶחֶם עֹנִי כִּי בְחִפָּזוֹן יָצָאתָ מֵאֶרֶץ מִצְרַיִם לְמַעַן תִּזְכֹּר אֶת יוֹם צֵאתְךָ מֵאֶרֶץ מִצְרַיִם כֹּל יְמֵי חַיֶּיךָ:  ת  לָא תֵיכוּל עֲלוֹהִי חֲמִיעַ שִׁבְעָא יוֹמִין תֵּיכוּל עֲלוֹהִי פַטִיר לְחֵם עֹנִי אֲרֵי בִּבְהִילוּ נְפַקְתָּא מֵאַרְעָא דְמִצְרַיִם בְּדִיל דְתִדְכַּר יָת יוֹמָא מִפְּקָךְ מֵאַרְעָא דְמִצְרַיִם כָּל יוֹמֵי חַיָיךְ:

{ד} וְלֹא יֵרָאֶה לְךָ שְׂאֹר בְּכָל גְּבֻלְךָ שִׁבְעַת יָמִים וְלֹא יָלִין מִן הַבָּשָׂר אֲשֶׁר תִּזְבַּח בָּעֶרֶב בַּיּוֹם הָרִאשׁוֹן לַבֹּקֶר:  מ  וְלֹא יֵרָאֶה לְךָ שְׂאֹר בְּכָל גְּבֻלְךָ שִׁבְעַת יָמִים וְלֹא יָלִין מִן הַבָּשָׂר אֲשֶׁר תִּזְבַּח בָּעֶרֶב בַּיּוֹם הָרִאשׁוֹן לַבֹּקֶר:  ת  וְלָא יִתְחֲזֵי לָךְ חֲמִר בְּכָל תְּחוּמָךְ שִׁבְעָא יוֹמִין וְלָא יְבִית מִן בִּסְרָא דִי תִכּוֹס בְּרַמְשָׁא בְּיוֹמָא קַדְמָאָה לְצַפְרָא:

{ה} לֹא תוּכַל לִזְבֹּחַ אֶת הַפָּסַח בְּאַחַד שְׁעָרֶיךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:  מ  לֹא תוּכַל לִזְבֹּחַ אֶת הַפָּסַח בְּאַחַד שְׁעָרֶיךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:  ת  לֵית לָךְ רְשׁוּ לְמִכֹּס יָת פִּסְחָא בַּחֲדָא מִקִרְוָיךְ דַיְיָ אֱלָהָךְ יָהֵב לָךְ:

{ו} כִּי אִם אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ לְשַׁכֵּן שְׁמוֹ שָׁם תִּזְבַּח אֶת הַפֶּסַח בָּעָרֶב כְּבוֹא הַשֶּׁמֶשׁ מוֹעֵד צֵאתְךָ מִמִּצְרָיִם:  מ  כִּי אִם אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ לְשַׁכֵּן שְׁמוֹ שָׁם תִּזְבַּח אֶת הַפֶּסַח בָּעָרֶב כְּבוֹא הַשֶּׁמֶשׁ מוֹעֵד צֵאתְךָ מִמִּצְרָיִם:  ת  אֱלָהֵן לְאַתְרָא דִי יִתִּרְעֵי יְיָ אֱלָהָךְ לְאַשְׁרָאָה שְׁכִנְתֵּהּ תַּמָן תִּכּוֹס יָת פִּסְחָא בְּרַמְשָׁא כְּמֵעַל שִׁמְשָׁא זְמַן מִפְקָךְ מִמִצְרָיִם:

{ז} וּבִשַּׁלְתָּ וְאָכַלְתָּ בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ וּפָנִיתָ בַבֹּקֶר וְהָלַכְתָּ לְאֹהָלֶיךָ:  מ  וּבִשַּׁלְתָּ וְאָכַלְתָּ בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ וּפָנִיתָ בַבֹּקֶר וְהָלַכְתָּ לְאֹהָלֶיךָ:  ת  וּתְבַשֵׁל וְתֵיכוּל בְּאַתְרָא דִי יִתִּרְעֵי יְיָ אֱלָהָךְ בֵּהּ וְתִתְפְּנֵי בְצַפְרָא וּתְהַךְ לְקִרְוָיךְ:

{ח} שֵׁשֶׁת יָמִים תֹּאכַל מַצּוֹת וּבַיּוֹם הַשְּׁבִיעִי עֲצֶרֶת לַיהוָה אֱלֹהֶיךָ לֹא תַעֲשֶׂה מְלָאכָה:  מ  שֵׁשֶׁת יָמִים תֹּאכַל מַצּוֹת וּבַיּוֹם הַשְּׁבִיעִי עֲצֶרֶת לַיהוָה אֱלֹהֶיךָ לֹא תַעֲשֶׂה מְלָאכָה:  ת  שִׁתָּא יוֹמִין תֵּיכוּל פַּטִירָא וּבְיוֹמָא שְׁבִיעָאָה כְּנֵשׁ קֳדָם יְיָ אֱלָהָךְ לָא תַעְבֵּד עִבִידָא: (ס)

{ט} שִׁבְעָה שָׁבֻעֹת תִּסְפָּר לָךְ מֵהָחֵל חֶרְמֵשׁ בַּקָּמָה תָּחֵל לִסְפֹּר שִׁבְעָה שָׁבֻעוֹת:  מ  שִׁבְעָה שָׁבֻעֹת תִּסְפָּר לָךְ מֵהָחֵל חֶרְמֵשׁ בַּקָּמָה תָּחֵל לִסְפֹּר שִׁבְעָה שָׁבֻעוֹת:  ת  שִׁבְעָא שָׁבוּעִין תִּמְנֵי לָךְ מִדְשַׁרְיוּת מַגְלָא בַּחֲצַד עוּמְרָא דַאֲרָמוּתָא תִּשְׁרֵי לְמִמְנֵּי שִׁבְעָא שָׁבוּעִין:

{י} וְעָשִׂיתָ חַג שָׁבֻעוֹת לַיהוָה אֱלֹהֶיךָ מִסַּת נִדְבַת יָדְךָ אֲשֶׁר תִּתֵּן כַּאֲשֶׁר יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ:  מ  וְעָשִׂיתָ חַג שָׁבֻעוֹת לַיהוָה אֱלֹהֶיךָ מִסַּת נִדְבַת יָדְךָ אֲשֶׁר תִּתֵּן כַּאֲשֶׁר יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ:  ת  וְתַעְבֵּד חַגָא דְשָׁבוּעַיָא קֳדָם יְיָ אֱלָהָךְ מִסַת נִדְבַת יְדָךְ דִי תִתֵּן כְּמָא דִי יְבָרֵכִנָךְ יְיָ אֱלָהָךְ:

{יא} וְשָׂמַחְתָּ לִפְנֵי יְהוָה אֱלֹהֶיךָ אַתָּה וּבִנְךָ וּבִתֶּךָ וְעַבְדְּךָ וַאֲמָתֶךָ וְהַלֵּוִי אֲשֶׁר בִּשְׁעָרֶיךָ וְהַגֵּר וְהַיָּתוֹם וְהָאַלְמָנָה אֲשֶׁר בְּקִרְבֶּךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ לְשַׁכֵּן שְׁמוֹ שָׁם:  מ  וְשָׂמַחְתָּ לִפְנֵי יְהוָה אֱלֹהֶיךָ אַתָּה וּבִנְךָ וּבִתֶּךָ וְעַבְדְּךָ וַאֲמָתֶךָ וְהַלֵּוִי אֲשֶׁר בִּשְׁעָרֶיךָ וְהַגֵּר וְהַיָּתוֹם וְהָאַלְמָנָה אֲשֶׁר בְּקִרְבֶּךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ לְשַׁכֵּן שְׁמוֹ שָׁם:  ת  וְתֶחֱדֵי קֳדָם יְיָ אֱלָהָךְ אַתְּ וּבְרָךְ וּבְרַתָּךְ וְעַבְדָךְ וְאַמְתָךְ וְלֵוָאָה דִי בְקִרְוָיךְ וְגִיוֹרָא וְיִתַּמָא וְאַרְמְלָא דִי בֵינָךְ בְּאַתְרָא דִי יִתִּרְעֵי יְיָ אֱלָהָךְ לְאַשְׁרָאָה שְׁכִנְתֵּהּ תַּמָן:

{יב} וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּמִצְרָיִם וְשָׁמַרְתָּ וְעָשִׂיתָ אֶת הַחֻקִּים הָאֵלֶּה:  מ  וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּמִצְרָיִם וְשָׁמַרְתָּ וְעָשִׂיתָ אֶת הַחֻקִּים הָאֵלֶּה:  ת  וְתִדְכַּר אֲרֵי עַבְדָא הֲוֵיתָא בְּמִצְרָיִם וְתִטַר וְתַעְבֵּד יָת קְיָמַיָא הָאִלֵין: (פ)

{יג}  מפטיר  חַג הַסֻּכֹּת תַּעֲשֶׂה לְךָ שִׁבְעַת יָמִים בְּאָסְפְּךָ מִגָּרְנְךָ וּמִיִּקְבֶךָ:  מ  חַג הַסֻּכֹּת תַּעֲשֶׂה לְךָ שִׁבְעַת יָמִים בְּאָסְפְּךָ מִגָּרְנְךָ וּמִיִּקְבֶךָ:  ת  חַגָא דִמְטָלַיָא תַּעְבֵּד לָךְ שִׁבְעָא יוֹמִין בְּמִכְנְשָׁךְ מֵאִדְרָךְ וּמִמַעְצַרְתָּךְ:

{יד} וְשָׂמַחְתָּ בְּחַגֶּךָ אַתָּה וּבִנְךָ וּבִתֶּךָ וְעַבְדְּךָ וַאֲמָתֶךָ וְהַלֵּוִי וְהַגֵּר וְהַיָּתוֹם וְהָאַלְמָנָה אֲשֶׁר בִּשְׁעָרֶיךָ:  מ  וְשָׂמַחְתָּ בְּחַגֶּךָ אַתָּה וּבִנְךָ וּבִתֶּךָ וְעַבְדְּךָ וַאֲמָתֶךָ וְהַלֵּוִי וְהַגֵּר וְהַיָּתוֹם וְהָאַלְמָנָה אֲשֶׁר בִּשְׁעָרֶיךָ:  ת  וְתֶחֱדֵי בְּחַגָךְ אַתְּ וּבְרָךְ וּבְרַתָּךְ וְעַבְדָךְ וְאַמְתָךְ וְלֵוָאָה וְגִיוֹרָא וְיִתַּמָא וְאַרְמְלָא דִי בְקִרְוָיךְ:

{טו} שִׁבְעַת יָמִים תָּחֹג לַיהוָה אֱלֹהֶיךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה כִּי יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל תְּבוּאָתְךָ וּבְכֹל מַעֲשֵׂה יָדֶיךָ וְהָיִיתָ אַךְ שָׂמֵחַ:  מ  שִׁבְעַת יָמִים תָּחֹג לַיהוָה אֱלֹהֶיךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה כִּי יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל תְּבוּאָתְךָ וּבְכֹל מַעֲשֵׂה יָדֶיךָ וְהָיִיתָ אַךְ שָׂמֵחַ:  ת  שִׁבְעָא יוֹמִין תֵּחוֹג קָדָם יְיָ אֱלָהָךְ בְּאַתְרָא דִי יִתִּרְעֵי יְיָ אֲרֵי יְבָרֵכִנָךְ יְיָ אֱלָהָךְ בְּכָל עֲלַלְתָּךְ וּבְכָל עוֹבָדֵי יְדָיךְ וּתְהֵי בְּרַם חָדֵי:

{טז} שָׁלוֹשׁ פְּעָמִים בַּשָּׁנָה יֵרָאֶה כָל זְכוּרְךָ אֶת פְּנֵי יְהוָה אֱלֹהֶיךָ בַּמָּקוֹם אֲשֶׁר יִבְחָר בְּחַג הַמַּצּוֹת וּבְחַג הַשָּׁבֻעוֹת וּבְחַג הַסֻּכּוֹת וְלֹא יֵרָאֶה אֶת פְּנֵי יְהוָה רֵיקָם:  מ  שָׁלוֹשׁ פְּעָמִים בַּשָּׁנָה יֵרָאֶה כָל זְכוּרְךָ אֶת פְּנֵי יְהוָה אֱלֹהֶיךָ בַּמָּקוֹם אֲשֶׁר יִבְחָר בְּחַג הַמַּצּוֹת וּבְחַג הַשָּׁבֻעוֹת וּבְחַג הַסֻּכּוֹת וְלֹא יֵרָאֶה אֶת פְּנֵי יְהוָה רֵיקָם:  ת  תְּלַת זִמְנִין בְּשַׁתָּא יִתְחֲזֵי כָל דְכוּרָךְ קֳדָם יְיָ אֱלָהָךְ בְּאַתְרָא דִי יִתִּרְעֵי בְּחַגָא דְפַטִירַיָא וּבְחַגָא דְשָׁבוּעַיָא וּבְחַגָא דִמְטָּלַיָא וְלָא יִתְחֲזֵי קֳדָם יְיָ רֵיקָנוּן:

{יז} אִישׁ כְּמַתְּנַת יָדוֹ כְּבִרְכַּת יְהוָה אֱלֹהֶיךָ אֲשֶׁר נָתַן לָךְ:  מ  אִישׁ כְּמַתְּנַת יָדוֹ כְּבִרְכַּת יְהוָה אֱלֹהֶיךָ אֲשֶׁר נָתַן לָךְ:  ת  גְבַר כְּמַתְּנַת יְדֵהּ כְּבִרְכָּתָא דַיְיָ אֱלָהָךְ דִי יְהַב לָךְ: ססס: (ססס)
