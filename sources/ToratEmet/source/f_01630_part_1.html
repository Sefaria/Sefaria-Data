פרשת ויקרא



ויקרא פרק-א

{א} וַיִּקְרָא אֶל מֹשֶׁה וַיְדַבֵּר יְהוָה אֵלָיו מֵאֹהֶל מוֹעֵד לֵאמֹר:  מ  וַיִּקְרָא אֶל מֹשֶׁה וַיְדַבֵּר יְהוָה אֵלָיו מֵאֹהֶל מוֹעֵד לֵאמֹר:  ת  וּקְרָא לְמֹשֶׁה וּמַלֵיל יְיָ עִמֵיהּ מִמַשְׁכַּן זִמְנָא לְמֵימָר:

{ב} דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם אָדָם כִּי יַקְרִיב מִכֶּם קָרְבָּן לַיהוָה מִן הַבְּהֵמָה מִן הַבָּקָר וּמִן הַצֹּאן תַּקְרִיבוּ אֶת קָרְבַּנְכֶם:  מ  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם אָדָם כִּי יַקְרִיב מִכֶּם קָרְבָּן לַיהוָה מִן הַבְּהֵמָה מִן הַבָּקָר וּמִן הַצֹּאן תַּקְרִיבוּ אֶת קָרְבַּנְכֶם:  ת  מַלֵיל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן אֱנַשׁ אֲרֵי יְקָרֵב מִנְכוֹן קוּרְבָּנָא קֳדָם יְיָ מִן בְּעִירָא מִן תּוֹרֵי וּמִן עָנָא תְּקָרְבוּן יָת קוּרְבַּנְכוֹן:

{ג} אִם עֹלָה קָרְבָּנוֹ מִן הַבָּקָר זָכָר תָּמִים יַקְרִיבֶנּוּ אֶל פֶּתַח אֹהֶל מוֹעֵד יַקְרִיב אֹתוֹ לִרְצֹנוֹ לִפְנֵי יְהוָה:  מ  אִם עֹלָה קָרְבָּנוֹ מִן הַבָּקָר זָכָר תָּמִים יַקְרִיבֶנּוּ אֶל פֶּתַח אֹהֶל מוֹעֵד יַקְרִיב אֹתוֹ לִרְצֹנוֹ לִפְנֵי יְהוָה:  ת  אִם עֲלָתָא קוּרְבָּנֵיהּ מִן תּוֹרֵי דְכַר שְׁלִים יְקָרְבִנֵהּ לִתְרַע מַשְׁכַּן זִמְנָא יְקָרֵיב יָתֵיהּ לְרַעֲוָא לֵיהּ קֳדָם יְיָ:

{ד} וְסָמַךְ יָדוֹ עַל רֹאשׁ הָעֹלָה וְנִרְצָה לוֹ לְכַפֵּר עָלָיו:  מ  וְסָמַךְ יָדוֹ עַל רֹאשׁ הָעֹלָה וְנִרְצָה לוֹ לְכַפֵּר עָלָיו:  ת  וְיִסְמוֹךְ יְדֵיהּ עַל רֵישׁ עֲלָתָא וְיִתְרְעֵי לֵיהּ לְכַפָּרָא עֲלוֹהִי:

{ה} וְשָׁחַט אֶת בֶּן הַבָּקָר לִפְנֵי יְהוָה וְהִקְרִיבוּ בְּנֵי אַהֲרֹן הַכֹּהֲנִים אֶת הַדָּם וְזָרְקוּ אֶת הַדָּם עַל הַמִּזְבֵּחַ סָבִיב אֲשֶׁר פֶּתַח אֹהֶל מוֹעֵד:  מ  וְשָׁחַט אֶת בֶּן הַבָּקָר לִפְנֵי יְהוָה וְהִקְרִיבוּ בְּנֵי אַהֲרֹן הַכֹּהֲנִים אֶת הַדָּם וְזָרְקוּ אֶת הַדָּם עַל הַמִּזְבֵּחַ סָבִיב אֲשֶׁר פֶּתַח אֹהֶל מוֹעֵד:  ת  וְיִכּוֹס יָת בַּר תּוֹרֵי קֳדָם יְיָ וִיקָרְבוּן בְּנֵי אַהֲרֹן כָּהֲנַיָא יָת דְמָא וְיִזְרְקוּן יָת דְמָא עַל מַדְבְּחָא סְחוֹר סְחוֹר דִי בִתְרַע מַשְׁכַּן זִמְנָא:

{ו} וְהִפְשִׁיט אֶת הָעֹלָה וְנִתַּח אֹתָהּ לִנְתָחֶיהָ:  מ  וְהִפְשִׁיט אֶת הָעֹלָה וְנִתַּח אֹתָהּ לִנְתָחֶיהָ:  ת  וְיַשְׁלַח יָת עֲלָתָא וִיפַלֵג יָתַהּ לְאֶבְרָהָא:

{ז} וְנָתְנוּ בְּנֵי אַהֲרֹן הַכֹּהֵן אֵשׁ עַל הַמִּזְבֵּחַ וְעָרְכוּ עֵצִים עַל הָאֵשׁ:  מ  וְנָתְנוּ בְּנֵי אַהֲרֹן הַכֹּהֵן אֵשׁ עַל הַמִּזְבֵּחַ וְעָרְכוּ עֵצִים עַל הָאֵשׁ:  ת  וְיִתְּנוּן בְּנֵי אַהֲרֹן כַּהֲנָא אֶשָׁתָא עַל מַדְבְּחָא וִיסַדְרוּן אָעַיָא עַל אֶשָׁתָא:

{ח} וְעָרְכוּ בְּנֵי אַהֲרֹן הַכֹּהֲנִים אֵת הַנְּתָחִים אֶת הָרֹאשׁ וְאֶת הַפָּדֶר עַל הָעֵצִים אֲשֶׁר עַל הָאֵשׁ אֲשֶׁר עַל הַמִּזְבֵּחַ:  מ  וְעָרְכוּ בְּנֵי אַהֲרֹן הַכֹּהֲנִים אֵת הַנְּתָחִים אֶת הָרֹאשׁ וְאֶת הַפָּדֶר עַל הָעֵצִים אֲשֶׁר עַל הָאֵשׁ אֲשֶׁר עַל הַמִּזְבֵּחַ:  ת  וִיסַדְרוּן בְּנֵי אַהֲרֹן כָּהֲנַיָא יָת אֶבְרַיָא יָת רֵישָׁא וְיָת תַּרְבָּא עַל אָעַיָא דִי עַל אֶשָׁתָא דִי עַל מַדְבְּחָא:

{ט} וְקִרְבּוֹ וּכְרָעָיו יִרְחַץ בַּמָּיִם וְהִקְטִיר הַכֹּהֵן אֶת הַכֹּל הַמִּזְבֵּחָה עֹלָה אִשֵּׁה רֵיחַ נִיחוֹחַ לַיהוָה:  מ  וְקִרְבּוֹ וּכְרָעָיו יִרְחַץ בַּמָּיִם וְהִקְטִיר הַכֹּהֵן אֶת הַכֹּל הַמִּזְבֵּחָה עֹלָה אִשֵּׁה רֵיחַ נִיחוֹחַ לַיהוָה:  ת  וְגַוֵיהּ וּכְרָעוֹהִי יְחַלֵל בְּמַיָא וְיַסֵק כַּהֲנָא יָת כּוֹלָא לְמַדְבְּחָא עֲלָתָא קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ: (ס)

{י} וְאִם מִן הַצֹּאן קָרְבָּנוֹ מִן הַכְּשָׂבִים אוֹ מִן הָעִזִּים לְעֹלָה זָכָר תָּמִים יַקְרִיבֶנּוּ:  מ  וְאִם מִן הַצֹּאן קָרְבָּנוֹ מִן הַכְּשָׂבִים אוֹ מִן הָעִזִּים לְעֹלָה זָכָר תָּמִים יַקְרִיבֶנּוּ:  ת  וְאִם מִן עָנָא קוּרְבָּנֵיהּ מִן אִמְרַיָא אוֹ מִן בְּנֵי עִזַיָא לַעֲלָתָא דְכַר שְׁלִים יְקָרְבִינֵיהּ:

{יא} וְשָׁחַט אֹתוֹ עַל יֶרֶךְ הַמִּזְבֵּחַ צָפֹנָה לִפְנֵי יְהוָה וְזָרְקוּ בְּנֵי אַהֲרֹן הַכֹּהֲנִים אֶת דָּמוֹ עַל הַמִּזְבֵּחַ סָבִיב:  מ  וְשָׁחַט אֹתוֹ עַל יֶרֶךְ הַמִּזְבֵּחַ צָפֹנָה לִפְנֵי יְהוָה וְזָרְקוּ בְּנֵי אַהֲרֹן הַכֹּהֲנִים אֶת דָּמוֹ עַל הַמִּזְבֵּחַ סָבִיב:  ת  וְיִכּוֹס יָתֵיהּ עַל צִדָא דְמַדְבְּחָא צִפּוּנָא קֳדָם יְיָ וְיִזְרְקוּן בְּנֵי אַהֲרֹן כָּהֲנַיָא יָת דְמֵיהּ עַל מַדְבְּחָא סְחוֹר סְחוֹר:

{יב} וְנִתַּח אֹתוֹ לִנְתָחָיו וְאֶת רֹאשׁוֹ וְאֶת פִּדְרוֹ וְעָרַךְ הַכֹּהֵן אֹתָם עַל הָעֵצִים אֲשֶׁר עַל הָאֵשׁ אֲשֶׁר עַל הַמִּזְבֵּחַ:  מ  וְנִתַּח אֹתוֹ לִנְתָחָיו וְאֶת רֹאשׁוֹ וְאֶת פִּדְרוֹ וְעָרַךְ הַכֹּהֵן אֹתָם עַל הָעֵצִים אֲשֶׁר עַל הָאֵשׁ אֲשֶׁר עַל הַמִּזְבֵּחַ:  ת  וִיפַלֵג יָתֵיהּ לְאֶבְרוֹהִי וְיָת רֵישֵׁיהּ וְיָת תַּרְבֵּיהּ וְיַסְדַר כַּהֲנָא יָתְהוֹן עַל אָעַיָא דִי עַל אֶשָׁתָא דִי עַל מַדְבְּחָא:

{יג} וְהַקֶּרֶב וְהַכְּרָעַיִם יִרְחַץ בַּמָּיִם וְהִקְרִיב הַכֹּהֵן אֶת הַכֹּל וְהִקְטִיר הַמִּזְבֵּחָה עֹלָה הוּא אִשֵּׁה רֵיחַ נִיחֹחַ לַיהוָה:  מ  וְהַקֶּרֶב וְהַכְּרָעַיִם יִרְחַץ בַּמָּיִם וְהִקְרִיב הַכֹּהֵן אֶת הַכֹּל וְהִקְטִיר הַמִּזְבֵּחָה עֹלָה הוּא אִשֵּׁה רֵיחַ נִיחֹחַ לַיהוָה:  ת  וְגַוָא וּכְרָעַיָא יְחַלֵל בְּמַיָא וִיקָרֵב כַּהֲנָא יָת כּוֹלָא וְיַסֵק לְמַדְבְּחָא עֲלָתָא הוּא קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קָדָם יְיָ: (פ)

{יד}  שני  וְאִם מִן הָעוֹף עֹלָה קָרְבָּנוֹ לַיהוָה וְהִקְרִיב מִן הַתֹּרִים אוֹ מִן בְּנֵי הַיּוֹנָה אֶת קָרְבָּנוֹ:  מ  וְאִם מִן הָעוֹף עֹלָה קָרְבָּנוֹ לַיהוָה וְהִקְרִיב מִן הַתֹּרִים אוֹ מִן בְּנֵי הַיּוֹנָה אֶת קָרְבָּנוֹ:  ת  וְאִם מִן עוֹפָא עֲלָתָא קוּרְבָּנֵיהּ קֳדָם יְיָ וִיקָרֵב מִן שַׁפְנִינַיָא אוֹ מִן בְּנֵי יוֹנָה יָת קוּרְבָּנֵיהּ:

{טו} וְהִקְרִיבוֹ הַכֹּהֵן אֶל הַמִּזְבֵּחַ וּמָלַק אֶת רֹאשׁוֹ וְהִקְטִיר הַמִּזְבֵּחָה וְנִמְצָה דָמוֹ עַל קִיר הַמִּזְבֵּחַ:  מ  וְהִקְרִיבוֹ הַכֹּהֵן אֶל הַמִּזְבֵּחַ וּמָלַק אֶת רֹאשׁוֹ וְהִקְטִיר הַמִּזְבֵּחָה וְנִמְצָה דָמוֹ עַל קִיר הַמִּזְבֵּחַ:  ת  וִיקָרְבִינֵהּ כַּהֲנָא לְמַדְבְּחָא וְיִמְלוֹק יָת רֵישֵׁהּ וְיַסֵק לְמַדְבְּחָא וְיִתְמְצֵי דְמֵיהּ עַל כּוֹתֶל מַדְבְּחָא:

{טז} וְהֵסִיר אֶת מֻרְאָתוֹ בְּנֹצָתָהּ וְהִשְׁלִיךְ אֹתָהּ אֵצֶל הַמִּזְבֵּחַ קֵדְמָה אֶל מְקוֹם הַדָּשֶׁן:  מ  וְהֵסִיר אֶת מֻרְאָתוֹ בְּנֹצָתָהּ וְהִשְׁלִיךְ אֹתָהּ אֵצֶל הַמִּזְבֵּחַ קֵדְמָה אֶל מְקוֹם הַדָּשֶׁן:  ת  וְיַעֲדֵי יָת זְפָקֵיהּ בְּאוּכְלֵיהּ וְיִרְמֵי יָתַהּ לִסְטַר מַדְבְּחָא קִידוּמָא לַאֲתַר דְמוֹשְׁדִין קִטְמָא:

{יז} וְשִׁסַּע אֹתוֹ בִכְנָפָיו לֹא יַבְדִּיל וְהִקְטִיר אֹתוֹ הַכֹּהֵן הַמִּזְבֵּחָה עַל הָעֵצִים אֲשֶׁר עַל הָאֵשׁ עֹלָה הוּא אִשֵּׁה רֵיחַ נִיחֹחַ לַיהוָה:  מ  וְשִׁסַּע אֹתוֹ בִכְנָפָיו לֹא יַבְדִּיל וְהִקְטִיר אֹתוֹ הַכֹּהֵן הַמִּזְבֵּחָה עַל הָעֵצִים אֲשֶׁר עַל הָאֵשׁ עֹלָה הוּא אִשֵּׁה רֵיחַ נִיחֹחַ לַיהוָה:  ת  וִיפָרֵיק יָתֵיהּ בְּגַדְפוֹהִי לָא יַפְרֵישׁ וְיַסֵק יָתֵיהּ כַּהֲנָא לְמַדְבְּחָא עַל אָעַיָא דִי עַל אֶשָׁתָא עֲלָתָא הוּא קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ: (ס)




ויקרא פרק-ב

{א} וְנֶפֶשׁ כִּי תַקְרִיב קָרְבַּן מִנְחָה לַיהוָה סֹלֶת יִהְיֶה קָרְבָּנוֹ וְיָצַק עָלֶיהָ שֶׁמֶן וְנָתַן עָלֶיהָ לְבֹנָה:  מ  וְנֶפֶשׁ כִּי תַקְרִיב קָרְבַּן מִנְחָה לַיהוָה סֹלֶת יִהְיֶה קָרְבָּנוֹ וְיָצַק עָלֶיהָ שֶׁמֶן וְנָתַן עָלֶיהָ לְבֹנָה:  ת  וֶאֱנַשׁ אֲרֵי יְקָרֵב קוּרְבַּן מִנְחָתָא קֳדָם יְיָ סוּלְתָּא יְהֵי קוּרְבָּנֵיהּ וִירִיק עֲלָהּ מִשְׁחָא וְיִתֵּן עֲלָהּ לְבוֹנְתָּא:

{ב} וֶהֱבִיאָהּ אֶל בְּנֵי אַהֲרֹן הַכֹּהֲנִים וְקָמַץ מִשָּׁם מְלֹא קֻמְצוֹ מִסָּלְתָּהּ וּמִשַּׁמְנָהּ עַל כָּל לְבֹנָתָהּ וְהִקְטִיר הַכֹּהֵן אֶת אַזְכָּרָתָהּ הַמִּזְבֵּחָה אִשֵּׁה רֵיחַ נִיחֹחַ לַיהוָה:  מ  וֶהֱבִיאָהּ אֶל בְּנֵי אַהֲרֹן הַכֹּהֲנִים וְקָמַץ מִשָּׁם מְלֹא קֻמְצוֹ מִסָּלְתָּהּ וּמִשַּׁמְנָהּ עַל כָּל לְבֹנָתָהּ וְהִקְטִיר הַכֹּהֵן אֶת אַזְכָּרָתָהּ הַמִּזְבֵּחָה אִשֵּׁה רֵיחַ נִיחֹחַ לַיהוָה:  ת  וְיַיְתִינָהּ לְוַת בְּנֵי אַהֲרֹן כַּהֲנַיָא וְיִקְמוֹץ מִתַּמָן מְלֵי קוּמְצְיהּ מִסִלְתָּהּ וּמִמִשְׁחָהּ עַל כָּל לְבוֹנְתָּהּ וְיַסֵק כַּהֲנָא יָת אַדְכַּרְתָּהּ לְמַדְבְּחָא קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ:

{ג} וְהַנּוֹתֶרֶת מִן הַמִּנְחָה לְאַהֲרֹן וּלְבָנָיו קֹדֶשׁ קָדָשִׁים מֵאִשֵּׁי יְהוָה:  מ  וְהַנּוֹתֶרֶת מִן הַמִּנְחָה לְאַהֲרֹן וּלְבָנָיו קֹדֶשׁ קָדָשִׁים מֵאִשֵּׁי יְהוָה:  ת  וּדְאִשְׁתָּאַר מִן מִנְחָתָא לְאַהֲרֹן וְלִבְנוֹהִי קוֹדֶשׁ קוּדְשִׁין מִקוּרְבָּנַיָא דַיְיָ: (ס)

{ד} וְכִי תַקְרִב קָרְבַּן מִנְחָה מַאֲפֵה תַנּוּר סֹלֶת חַלּוֹת מַצֹּת בְּלוּלֹת בַּשֶּׁמֶן וּרְקִיקֵי מַצּוֹת מְשֻׁחִים בַּשָּׁמֶן:  מ  וְכִי תַקְרִב קָרְבַּן מִנְחָה מַאֲפֵה תַנּוּר סֹלֶת חַלּוֹת מַצֹּת בְּלוּלֹת בַּשֶּׁמֶן וּרְקִיקֵי מַצּוֹת מְשֻׁחִים בַּשָּׁמֶן:  ת  וַאֲרֵי תְקָרֵיב קוּרְבַּן מִנְחָתָא מַאֲפֵה תַנוּרָא סוּלְתָּא גְרִיצָן פַּטִירָן דְפִילָן בִּמְשָׁח וְאִסְפּוּגִין פַטִירִין דִמְשִׁיחִין בִּמְשָׁח: (ס)

{ה} וְאִם מִנְחָה עַל הַמַּחֲבַת קָרְבָּנֶךָ סֹלֶת בְּלוּלָה בַשֶּׁמֶן מַצָּה תִהְיֶה:  מ  וְאִם מִנְחָה עַל הַמַּחֲבַת קָרְבָּנֶךָ סֹלֶת בְּלוּלָה בַשֶּׁמֶן מַצָּה תִהְיֶה:  ת  וְאִם מִנְחָתָא עַל מַסְרֵיתָא קוּרְבָּנָךְ סוּלְתָּא דְפִילָא בִמְשַׁח פַּטִיר תְּהֵי:

{ו} פָּתוֹת אֹתָהּ פִּתִּים וְיָצַקְתָּ עָלֶיהָ שָׁמֶן מִנְחָה הִוא:  מ  פָּתוֹת אֹתָהּ פִּתִּים וְיָצַקְתָּ עָלֶיהָ שָׁמֶן מִנְחָה הִוא:  ת  בְּצַע יָתָהּ בִּיצוּעִין וּתְרִיק עֲלָהּ מִשְׁחָא מִנְחָתָא הִיא: (ס)

{ז}  שלישי  וְאִם מִנְחַת מַרְחֶשֶׁת קָרְבָּנֶךָ סֹלֶת בַּשֶּׁמֶן תֵּעָשֶׂה:  מ  וְאִם מִנְחַת מַרְחֶשֶׁת קָרְבָּנֶךָ סֹלֶת בַּשֶּׁמֶן תֵּעָשֶׂה:  ת  וְאִם מִנְחָתָא דְרַדְתָּא קוּרְבָּנָךְ סוּלְתָּא בִּמְשַׁח תִּתְעֲבֵד:

{ח} וְהֵבֵאתָ אֶת הַמִּנְחָה אֲשֶׁר יֵעָשֶׂה מֵאֵלֶּה לַיהוָה וְהִקְרִיבָהּ אֶל הַכֹּהֵן וְהִגִּישָׁהּ אֶל הַמִּזְבֵּחַ:  מ  וְהֵבֵאתָ אֶת הַמִּנְחָה אֲשֶׁר יֵעָשֶׂה מֵאֵלֶּה לַיהוָה וְהִקְרִיבָהּ אֶל הַכֹּהֵן וְהִגִּישָׁהּ אֶל הַמִּזְבֵּחַ:  ת  וְתַיְתֵי יַת מִנְחָתָא דִי יִתְעֲבֵיד מֵאִלֵין קֳדָם יְיָ וִיקַרְבִינָהּ לְוַת כַּהֲנָא וִיקַרְבִינָהּ לְמַדְבְּחָא:

{ט} וְהֵרִים הַכֹּהֵן מִן הַמִּנְחָה אֶת אַזְכָּרָתָהּ וְהִקְטִיר הַמִּזְבֵּחָה אִשֵּׁה רֵיחַ נִיחֹחַ לַיהוָה:  מ  וְהֵרִים הַכֹּהֵן מִן הַמִּנְחָה אֶת אַזְכָּרָתָהּ וְהִקְטִיר הַמִּזְבֵּחָה אִשֵּׁה רֵיחַ נִיחֹחַ לַיהוָה:  ת  וְיַפְרֵישׁ כַּהֲנָא מִן מִנְחָתָא יָת אַדְכַּרְתָּהּ וְיַסֵק לְמַדְבְּחָא קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ:

{י} וְהַנּוֹתֶרֶת מִן הַמִּנְחָה לְאַהֲרֹן וּלְבָנָיו קֹדֶשׁ קָדָשִׁים מֵאִשֵּׁי יְהוָה:  מ  וְהַנּוֹתֶרֶת מִן הַמִּנְחָה לְאַהֲרֹן וּלְבָנָיו קֹדֶשׁ קָדָשִׁים מֵאִשֵּׁי יְהוָה:  ת  וּדְאִשְׁתָּאַר מִן מִנְחָתָא לְאַהֲרֹן וְלִּבְנוֹהִי קוֹדֶשׁ קוּדְשִׁין מִקוּרְבָּנַיָא דַיְיָ:

{יא} כָּל הַמִּנְחָה אֲשֶׁר תַּקְרִיבוּ לַיהוָה לֹא תֵעָשֶׂה חָמֵץ כִּי כָל שְׂאֹר וְכָל דְּבַשׁ לֹא תַקְטִירוּ מִמֶּנּוּ אִשֶּׁה לַיהוָה:  מ  כָּל הַמִּנְחָה אֲשֶׁר תַּקְרִיבוּ לַיהוָה לֹא תֵעָשֶׂה חָמֵץ כִּי כָל שְׂאֹר וְכָל דְּבַשׁ לֹא תַקְטִירוּ מִמֶּנּוּ אִשֶּׁה לַיהוָה:  ת  כָּל מִנְחָתָא דִי תְקַרְבוּן קֳדָם יְיָ לָא תִתְעֲבֵיד חֲמִיעַ אֲרֵי כָּל חֲמִיר וְכָל דְבַשׁ לָא תַסְקוּן מִנֵיהּ קוּרְבָּנָא קֳדָם יְיָ:

{יב} קָרְבַּן רֵאשִׁית תַּקְרִיבוּ אֹתָם לַיהוָה וְאֶל הַמִּזְבֵּחַ לֹא יַעֲלוּ לְרֵיחַ נִיחֹחַ:  מ  קָרְבַּן רֵאשִׁית תַּקְרִיבוּ אֹתָם לַיהוָה וְאֶל הַמִּזְבֵּחַ לֹא יַעֲלוּ לְרֵיחַ נִיחֹחַ:  ת  קוּרְבַּן קַדְמַי תְּקָרְבוּן יַתְהוֹן קֳדָם יְיָ וּלְמַדְבְּחָא לָא יִתַּסְקוּן לְאִתְקַבָּלָא בְּרַעֲוָא:

{יג} וְכָל קָרְבַּן מִנְחָתְךָ בַּמֶּלַח תִּמְלָח וְלֹא תַשְׁבִּית מֶלַח בְּרִית אֱלֹהֶיךָ מֵעַל מִנְחָתֶךָ עַל כָּל קָרְבָּנְךָ תַּקְרִיב מֶלַח:  מ  וְכָל קָרְבַּן מִנְחָתְךָ בַּמֶּלַח תִּמְלָח וְלֹא תַשְׁבִּית מֶלַח בְּרִית אֱלֹהֶיךָ מֵעַל מִנְחָתֶךָ עַל כָּל קָרְבָּנְךָ תַּקְרִיב מֶלַח:  ת  וְכָל קוּרְבַּן מִנְחָתָךְ בְּמִלְחָא תִּמְלַח וְלָא תְבַטֵל מְלַח קְיַם אֱלָהָךְ מֵעַל מִנְחָתָךְ עַל כָּל קוּרְבָּנָךְ תְּקָרֵב מִלְחָא: (ס)

{יד} וְאִם תַּקְרִיב מִנְחַת בִּכּוּרִים לַיהוָה אָבִיב קָלוּי בָּאֵשׁ גֶּרֶשׂ כַּרְמֶל תַּקְרִיב אֵת מִנְחַת בִּכּוּרֶיךָ:  מ  וְאִם תַּקְרִיב מִנְחַת בִּכּוּרִים לַיהוָה אָבִיב קָלוּי בָּאֵשׁ גֶּרֶשׂ כַּרְמֶל תַּקְרִיב אֵת מִנְחַת בִּכּוּרֶיךָ:  ת  וְאִם תְּקָרֵב מִנְחַת בִּכּוּרִין קֳדָם יְיָ אָבִיב קְלֵי בְנוּר פֵּירוּכִין רַכִּיכָן תְּקָרֵיב יַת מִנְחַת בִּכּוּרָךְ:

{טו} וְנָתַתָּ עָלֶיהָ שֶׁמֶן וְשַׂמְתָּ עָלֶיהָ לְבֹנָה מִנְחָה הִוא:  מ  וְנָתַתָּ עָלֶיהָ שֶׁמֶן וְשַׂמְתָּ עָלֶיהָ לְבֹנָה מִנְחָה הִוא:  ת  וְתִתֵּן עֲלַהּ מִשְׁחָא וּתְשַׁוֵי עֲלַהּ לְבוֹנְתָּא מִנְחָתָא הִיא:

{טז} וְהִקְטִיר הַכֹּהֵן אֶת אַזְכָּרָתָהּ מִגִּרְשָׂהּ וּמִשַּׁמְנָהּ עַל כָּל לְבֹנָתָהּ אִשֶּׁה לַיהוָה:  מ  וְהִקְטִיר הַכֹּהֵן אֶת אַזְכָּרָתָהּ מִגִּרְשָׂהּ וּמִשַּׁמְנָהּ עַל כָּל לְבֹנָתָהּ אִשֶּׁה לַיהוָה:  ת  וְיַסֵק כַּהֲנָא יַת אַדְכַּרְתַּהּ מִפֵּרוּכַהּ וּמִמִשְׁחַהּ עַל כָּל לְבוֹנְתָּהּ קוּרְבָּנָא קֳדָם יְיָ: (ס)




ויקרא פרק-ג

{א}  רביעי  וְאִם זֶבַח שְׁלָמִים קָרְבָּנוֹ אִם מִן הַבָּקָר הוּא מַקְרִיב אִם זָכָר אִם נְקֵבָה תָּמִים יַקְרִיבֶנּוּ לִפְנֵי יְהוָה:  מ  וְאִם זֶבַח שְׁלָמִים קָרְבָּנוֹ אִם מִן הַבָּקָר הוּא מַקְרִיב אִם זָכָר אִם נְקֵבָה תָּמִים יַקְרִיבֶנּוּ לִפְנֵי יְהוָה:  ת  וְאִם נִכְסַת קוּדְשַׁיָא קוּרְבָּנֵיהּ אִם מִן תּוֹרֵי הוּא מְקָרֵב אִם דְכַר אִם נוּקְבָא שְׁלִים יְקָרְבִינֵיהּ קֳדָם יְיָ:

{ב} וְסָמַךְ יָדוֹ עַל רֹאשׁ קָרְבָּנוֹ וּשְׁחָטוֹ פֶּתַח אֹהֶל מוֹעֵד וְזָרְקוּ בְּנֵי אַהֲרֹן הַכֹּהֲנִים אֶת הַדָּם עַל הַמִּזְבֵּחַ סָבִיב:  מ  וְסָמַךְ יָדוֹ עַל רֹאשׁ קָרְבָּנוֹ וּשְׁחָטוֹ פֶּתַח אֹהֶל מוֹעֵד וְזָרְקוּ בְּנֵי אַהֲרֹן הַכֹּהֲנִים אֶת הַדָּם עַל הַמִּזְבֵּחַ סָבִיב:  ת  וְיִסְמוֹךְ יְדֵיהּ עַל רֵישׁ קוּרְבָּנֵיהּ וְיִכְסִינֵיהּ בִּתְרַע מַשְׁכַּן זִמְנָא וְיִזְרְקוּן בְּנֵי אַהֲרֹן כַּהֲנַיָא יַת דְמָא עַל מַדְבְּחָא סְחוֹר סְחוֹר:

{ג} וְהִקְרִיב מִזֶּבַח הַשְּׁלָמִים אִשֶּׁה לַיהוָה אֶת הַחֵלֶב הַמְכַסֶּה אֶת הַקֶּרֶב וְאֵת כָּל הַחֵלֶב אֲשֶׁר עַל הַקֶּרֶב:  מ  וְהִקְרִיב מִזֶּבַח הַשְּׁלָמִים אִשֶּׁה לַיהוָה אֶת הַחֵלֶב הַמְכַסֶּה אֶת הַקֶּרֶב וְאֵת כָּל הַחֵלֶב אֲשֶׁר עַל הַקֶּרֶב:  ת  וִיקָרֵב מִנִכְסַת קוּדְשַׁיָא קוּרְבָּנָא קֳדָם יְיָ יַת תַּרְבָּא דְחָפֵי יַת גַוָא וְיַת כָּל תַּרְבָּא דִי עַל גַוָא:

{ד} וְאֵת שְׁתֵּי הַכְּלָיֹת וְאֶת הַחֵלֶב אֲשֶׁר עֲלֵהֶן אֲשֶׁר עַל הַכְּסָלִים וְאֶת הַיֹּתֶרֶת עַל הַכָּבֵד עַל הַכְּלָיוֹת יְסִירֶנָּה:  מ  וְאֵת שְׁתֵּי הַכְּלָיֹת וְאֶת הַחֵלֶב אֲשֶׁר עֲלֵהֶן אֲשֶׁר עַל הַכְּסָלִים וְאֶת הַיֹּתֶרֶת עַל הַכָּבֵד עַל הַכְּלָיוֹת יְסִירֶנָּה:  ת  וְיַת תַּרְתֵּין כֻּלְיָן וְיַת תַּרְבָּא דִי עֲלֵיהֶן דִי עַל גִסְסַיָא וְיָת חַצְרָא דְעַל כַּבְדָא עַל כּוּלְיָתָא יַעֲדִינָהּ:

{ה} וְהִקְטִירוּ אֹתוֹ בְנֵי אַהֲרֹן הַמִּזְבֵּחָה עַל הָעֹלָה אֲשֶׁר עַל הָעֵצִים אֲשֶׁר עַל הָאֵשׁ אִשֵּׁה רֵיחַ נִיחֹחַ לַיהוָה:  מ  וְהִקְטִירוּ אֹתוֹ בְנֵי אַהֲרֹן הַמִּזְבֵּחָה עַל הָעֹלָה אֲשֶׁר עַל הָעֵצִים אֲשֶׁר עַל הָאֵשׁ אִשֵּׁה רֵיחַ נִיחֹחַ לַיהוָה:  ת  וְיִסְקוּן יָתֵיהּ בְּנֵי אַהֲרֹן לְמַדְבְּחָא עַל עֲלָתָא דִי עַל אָעַיָא דִי עַל אֶשָׁתָא קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ: (פ)

{ו} וְאִם מִן הַצֹּאן קָרְבָּנוֹ לְזֶבַח שְׁלָמִים לַיהוָה זָכָר אוֹ נְקֵבָה תָּמִים יַקְרִיבֶנּוּ:  מ  וְאִם מִן הַצֹּאן קָרְבָּנוֹ לְזֶבַח שְׁלָמִים לַיהוָה זָכָר אוֹ נְקֵבָה תָּמִים יַקְרִיבֶנּוּ:  ת  וְאִם מִן עָנָא קוּרְבָּנֵיהּ לְנִכְסַת קוּדְשַׁיָא קֳדָם יְיָ דְכַר אוֹ נוּקְבָא שְׁלִים יְקָרְבִינֵיהּ:

{ז} אִם כֶּשֶׂב הוּא מַקְרִיב אֶת קָרְבָּנוֹ וְהִקְרִיב אֹתוֹ לִפְנֵי יְהוָה:  מ  אִם כֶּשֶׂב הוּא מַקְרִיב אֶת קָרְבָּנוֹ וְהִקְרִיב אֹתוֹ לִפְנֵי יְהוָה:  ת  אִם אִימַר הוּא מְקָרֵב יַת קוּרְבָּנֵיהּ וִיקָרֵיב יָתֵי קֳדָם יְיָ:

{ח} וְסָמַךְ אֶת יָדוֹ עַל רֹאשׁ קָרְבָּנוֹ וְשָׁחַט אֹתוֹ לִפְנֵי אֹהֶל מוֹעֵד וְזָרְקוּ בְּנֵי אַהֲרֹן אֶת דָּמוֹ עַל הַמִּזְבֵּחַ סָבִיב:  מ  וְסָמַךְ אֶת יָדוֹ עַל רֹאשׁ קָרְבָּנוֹ וְשָׁחַט אֹתוֹ לִפְנֵי אֹהֶל מוֹעֵד וְזָרְקוּ בְּנֵי אַהֲרֹן אֶת דָּמוֹ עַל הַמִּזְבֵּחַ סָבִיב:  ת  וְיִסְמוֹךְ יַת יְדֵיהּ עַל רֵישׁ קוּרְבָּנֵיהּ וְיִכּוֹס יָתֵיהּ קֳדָם מַשְׁכַּן זִמְנָא וְיִזְרְקוּן בְּנֵי אַהֲרֹן יַת דְמֵיהּ עַל מַדְבְּחָא סְחוֹר םְחוֹר:

{ט} וְהִקְרִיב מִזֶּבַח הַשְּׁלָמִים אִשֶּׁה לַיהוָה חֶלְבּוֹ הָאַלְיָה תְמִימָה לְעֻמַּת הֶעָצֶה יְסִירֶנָּה וְאֶת הַחֵלֶב הַמְכַסֶּה אֶת הַקֶּרֶב וְאֵת כָּל הַחֵלֶב אֲשֶׁר עַל הַקֶּרֶב:  מ  וְהִקְרִיב מִזֶּבַח הַשְּׁלָמִים אִשֶּׁה לַיהוָה חֶלְבּוֹ הָאַלְיָה תְמִימָה לְעֻמַּת הֶעָצֶה יְסִירֶנָּה וְאֶת הַחֵלֶב הַמְכַסֶּה אֶת הַקֶּרֶב וְאֵת כָּל הַחֵלֶב אֲשֶׁר עַל הַקֶּרֶב:  ת  וִיקָרֵב מִנִכְסַת קוּדְשַׁיָא קֻרְבָּנָא קֳדָם יְיָ תַּרְבֵּיהּ אֲלִיתָא שְׁלִמְתָּא לָקֳבֵל שֵׁזַרְתָּא יֶעְדִנַהּ וְיָת תַּרְבָּא דְחָפֵי יָת גַוָא וְיָת כָּל תַּרְבָּא דִי עַל גַוָא:

{י} וְאֵת שְׁתֵּי הַכְּלָיֹת וְאֶת הַחֵלֶב אֲשֶׁר עֲלֵהֶן אֲשֶׁר עַל הַכְּסָלִים וְאֶת הַיֹּתֶרֶת עַל הַכָּבֵד עַל הַכְּלָיֹת יְסִירֶנָּה:  מ  וְאֵת שְׁתֵּי הַכְּלָיֹת וְאֶת הַחֵלֶב אֲשֶׁר עֲלֵהֶן אֲשֶׁר עַל הַכְּסָלִים וְאֶת הַיֹּתֶרֶת עַל הַכָּבֵד עַל הַכְּלָיֹת יְסִירֶנָּה:  ת  וְיָת תַּרְתֵּין כָּלְיָן וְיָת תַּרְבָּא דִי עֲלֵיהֶן דִי עַל גִסְסַיָא וְיָת חַצְרָא דְעַל כַּבְדָא עַל כָּלְיֵתָא יֶעְדִנַהּ:

{יא} וְהִקְטִירוֹ הַכֹּהֵן הַמִּזְבֵּחָה לֶחֶם אִשֶּׁה לַיהוָה:  מ  וְהִקְטִירוֹ הַכֹּהֵן הַמִּזְבֵּחָה לֶחֶם אִשֶּׁה לַיהוָה:  ת  וְיַסְקִנֵהּ כַּהֲנָא לְמַדְבְּחָא לְחֵם קֻרְבָּנָא קֳדָם יְיָ: (פ)

{יב} וְאִם עֵז קָרְבָּנוֹ וְהִקְרִיבוֹ לִפְנֵי יְהוָה:  מ  וְאִם עֵז קָרְבָּנוֹ וְהִקְרִיבוֹ לִפְנֵי יְהוָה:  ת  וְאִם מִן בְּנֵי עִזַיָא קֻרְבָּנֵהּ וִיקָרְבִנֵהּ קֳדָם יְיָ:

{יג} וְסָמַךְ אֶת יָדוֹ עַל רֹאשׁוֹ וְשָׁחַט אֹתוֹ לִפְנֵי אֹהֶל מוֹעֵד וְזָרְקוּ בְּנֵי אַהֲרֹן אֶת דָּמוֹ עַל הַמִּזְבֵּחַ סָבִיב:  מ  וְסָמַךְ אֶת יָדוֹ עַל רֹאשׁוֹ וְשָׁחַט אֹתוֹ לִפְנֵי אֹהֶל מוֹעֵד וְזָרְקוּ בְּנֵי אַהֲרֹן אֶת דָּמוֹ עַל הַמִּזְבֵּחַ סָבִיב:  ת  וְיִסְמוֹךְ יָת יְדֵהּ עַל רֵישֵׁיהּ וְיִכּוֹס יָתֵהּ קֳדָם מַשְׁכַּן זִמְנָא וְיִזְרְקוּן בְּנֵי אַהֲרֹן יָת דְמֵהּ עַל מַדְבְּחָא סְחוֹר סְחוֹר:

{יד} וְהִקְרִיב מִמֶּנּוּ קָרְבָּנוֹ אִשֶּׁה לַיהוָה אֶת הַחֵלֶב הַמְכַסֶּה אֶת הַקֶּרֶב וְאֵת כָּל הַחֵלֶב אֲשֶׁר עַל הַקֶּרֶב:  מ  וְהִקְרִיב מִמֶּנּוּ קָרְבָּנוֹ אִשֶּׁה לַיהוָה אֶת הַחֵלֶב הַמְכַסֶּה אֶת הַקֶּרֶב וְאֵת כָּל הַחֵלֶב אֲשֶׁר עַל הַקֶּרֶב:  ת  וִיקָרֵב מִנֵהּ קֻרְבָּנֵהּ קֻרְבָּן קֳדָם יְיָ יָת תַּרְבָּא דְחָפֵי יָת גַוָא וְיָת כָּל תַּרְבָּא דִי עַל גַוָא:

{טו} וְאֵת שְׁתֵּי הַכְּלָיֹת וְאֶת הַחֵלֶב אֲשֶׁר עֲלֵהֶן אֲשֶׁר עַל הַכְּסָלִים וְאֶת הַיֹּתֶרֶת עַל הַכָּבֵד עַל הַכְּלָיֹת יְסִירֶנָּה:  מ  וְאֵת שְׁתֵּי הַכְּלָיֹת וְאֶת הַחֵלֶב אֲשֶׁר עֲלֵהֶן אֲשֶׁר עַל הַכְּסָלִים וְאֶת הַיֹּתֶרֶת עַל הַכָּבֵד עַל הַכְּלָיֹת יְסִירֶנָּה:  ת  וְיָת תַּרְתֵּין כָּלְיָן וְיָת תַּרְבָּא דִי עֲלֵיהֶן דִי עַל גִסְסַיָא וְיָת חַצְרָא דִי עַל כַּבְדָא עַל כָּלְיֵתָא יֶעְדִנַהּ:

{טז} וְהִקְטִירָם הַכֹּהֵן הַמִּזְבֵּחָה לֶחֶם אִשֶּׁה לְרֵיחַ נִיחֹחַ כָּל חֵלֶב לַיהוָה:  מ  וְהִקְטִירָם הַכֹּהֵן הַמִּזְבֵּחָה לֶחֶם אִשֶּׁה לְרֵיחַ נִיחֹחַ כָּל חֵלֶב לַיהוָה:  ת  וְיַסְקִנוּן כַּהֲנָא לְמַדְבְּחָא לְחֵם קֻרְבָּנָא לְאִתְקַבָּלָא בְּרַעֲוָא כָּל תַּרְבָּא קֳדָם יְיָ:

{יז} חֻקַּת עוֹלָם לְדֹרֹתֵיכֶם בְּכֹל מוֹשְׁבֹתֵיכֶם כָּל חֵלֶב וְכָל דָּם לֹא תֹאכֵלוּ:  מ  חֻקַּת עוֹלָם לְדֹרֹתֵיכֶם בְּכֹל מוֹשְׁבֹתֵיכֶם כָּל חֵלֶב וְכָל דָּם לֹא תֹאכֵלוּ:  ת  קְיָם עָלָם לְדָרָתֵיכוֹן בְּכָל מוֹתְבָנֵיכוֹן כָּל תַּרְבָּא וְכָל דְמָא לָא תֵיכְלוּן: (פ)




ויקרא פרק-ד

{א}  חמישי  וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה לֵּאמֹר:  מ  וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה לֵּאמֹר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר נֶפֶשׁ כִּי תֶחֱטָא בִשְׁגָגָה מִכֹּל מִצְוֹת יְהוָה אֲשֶׁר לֹא תֵעָשֶׂינָה וְעָשָׂה מֵאַחַת מֵהֵנָּה:  מ  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר נֶפֶשׁ כִּי תֶחֱטָא בִשְׁגָגָה מִכֹּל מִצְוֹת יְהוָה אֲשֶׁר לֹא תֵעָשֶׂינָה וְעָשָׂה מֵאַחַת מֵהֵנָּה:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל לְמֵימָר אֱנַשׁ אֲרֵי יֵחוֹב בְּשָׁלוּ מִכָּל פִּקוּדַיָא דַיְיָ דִי לָא כַשְׁרִין לְאִתְעֲבָדָא וְיַעְבֵּד מִן חָד מִ נְה וֹן:

{ג} אִם הַכֹּהֵן הַמָּשִׁיחַ יֶחֱטָא לְאַשְׁמַת הָעָם וְהִקְרִיב עַל חַטָּאתוֹ אֲשֶׁר חָטָא פַּר בֶּן בָּקָר תָּמִים לַיהוָה לְחַטָּאת:  מ  אִם הַכֹּהֵן הַמָּשִׁיחַ יֶחֱטָא לְאַשְׁמַת הָעָם וְהִקְרִיב עַל חַטָּאתוֹ אֲשֶׁר חָטָא פַּר בֶּן בָּקָר תָּמִים לַיהוָה לְחַטָּאת:  ת  אִם כַּהֲנָא רַבָּא יֵיחוֹב לְחוֹבַת עַמָא וִיקָרֵב עַל חוֹבְתֵהּ דִי חָב תּוֹר בַּר תּוֹרֵי שְׁלִים קֳדָם יְיָ לְחַטָאתָא:

{ד} וְהֵבִיא אֶת הַפָּר אֶל פֶּתַח אֹהֶל מוֹעֵד לִפְנֵי יְהוָה וְסָמַךְ אֶת יָדוֹ עַל רֹאשׁ הַפָּר וְשָׁחַט אֶת הַפָּר לִפְנֵי יְהוָה:  מ  וְהֵבִיא אֶת הַפָּר אֶל פֶּתַח אֹהֶל מוֹעֵד לִפְנֵי יְהוָה וְסָמַךְ אֶת יָדוֹ עַל רֹאשׁ הַפָּר וְשָׁחַט אֶת הַפָּר לִפְנֵי יְהוָה:  ת  וְיַיְתִי יָת תּוֹרָא לִתְרַע מַשְׁכַּן זִמְנָא לָקֳדָם יְיָ וְיִסְמוֹךְ יָת יְדֵיהּ עַל רֵישׁ תּוֹרָא וְיִכּוֹס יָת תּוֹרָא קֳדָם יְיָ:

{ה} וְלָקַח הַכֹּהֵן הַמָּשִׁיחַ מִדַּם הַפָּר וְהֵבִיא אֹתוֹ אֶל אֹהֶל מוֹעֵד:  מ  וְלָקַח הַכֹּהֵן הַמָּשִׁיחַ מִדַּם הַפָּר וְהֵבִיא אֹתוֹ אֶל אֹהֶל מוֹעֵד:  ת  וְיִסַב כַּהֲנָא רַבָּא מִדְמָא דְתוֹרָא וְיָעֵל יָתֵהּ לְמַשְׁכַּן זִמְנָא:

{ו} וְטָבַל הַכֹּהֵן אֶת אֶצְבָּעוֹ בַּדָּם וְהִזָּה מִן הַדָּם שֶׁבַע פְּעָמִים לִפְנֵי יְהוָה אֶת פְּנֵי פָּרֹכֶת הַקֹּדֶשׁ:  מ  וְטָבַל הַכֹּהֵן אֶת אֶצְבָּעוֹ בַּדָּם וְהִזָּה מִן הַדָּם שֶׁבַע פְּעָמִים לִפְנֵי יְהוָה אֶת פְּנֵי פָּרֹכֶת הַקֹּדֶשׁ:  ת  וְיִטְבּוֹל כַּהֲנָא יָת אֶצְבְּעֵהּ בִּדְמָא וְיַדִי מִן דְמָא שְׁבַע זִמְנִין קֳדָם יְיָ קֳדָם פָּרוּכְתָּא דְקוּדְשָׁא:

{ז} וְנָתַן הַכֹּהֵן מִן הַדָּם עַל קַרְנוֹת מִזְבַּח קְטֹרֶת הַסַּמִּים לִפְנֵי יְהוָה אֲשֶׁר בְּאֹהֶל מוֹעֵד וְאֵת כָּל דַּם הַפָּר יִשְׁפֹּךְ אֶל יְסוֹד מִזְבַּח הָעֹלָה אֲשֶׁר פֶּתַח אֹהֶל מוֹעֵד:  מ  וְנָתַן הַכֹּהֵן מִן הַדָּם עַל קַרְנוֹת מִזְבַּח קְטֹרֶת הַסַּמִּים לִפְנֵי יְהוָה אֲשֶׁר בְּאֹהֶל מוֹעֵד וְאֵת כָּל דַּם הַפָּר יִשְׁפֹּךְ אֶל יְסוֹד מִזְבַּח הָעֹלָה אֲשֶׁר פֶּתַח אֹהֶל מוֹעֵד:  ת  וְיִתֵּן כַּהֲנָא מִן דְמָא עַל קַרְנַת מַדְבַּח קְטוֹרֶת בּוּסְמַיָא קֳדָם יְיָ דִי בְּמַשְׁכַּן זִמְנָא וְיָת כָּל דְמָא דְתוֹרָא יֵישׁוּד לִיסוֹדָא דְּמַדְבְּחָא דַעֲלָתָא דִי בִתְרַע מַשְׁכַּן זִ מְנָא:

{ח} וְאֶת כָּל חֵלֶב פַּר הַחַטָּאת יָרִים מִמֶּנּוּ אֶת הַחֵלֶב הַמְכַסֶּה עַל הַקֶּרֶב וְאֵת כָּל הַחֵלֶב אֲשֶׁר עַל הַקֶּרֶב:  מ  וְאֶת כָּל חֵלֶב פַּר הַחַטָּאת יָרִים מִמֶּנּוּ אֶת הַחֵלֶב הַמְכַסֶּה עַל הַקֶּרֶב וְאֵת כָּל הַחֵלֶב אֲשֶׁר עַל הַקֶּרֶב:  ת  וְיָת כָּל תְּרַב תּוֹרָא דְחַטָאתָא יַפְרֵשׁ מִנֵהּ יָת תַּרְבָּא דְחָפֵי עַל גַוָא וְיָת כָּל תַּרְבָּא דִי עַל גַוָא:

{ט} וְאֵת שְׁתֵּי הַכְּלָיֹת וְאֶת הַחֵלֶב אֲשֶׁר עֲלֵיהֶן אֲשֶׁר עַל הַכְּסָלִים וְאֶת הַיֹּתֶרֶת עַל הַכָּבֵד עַל הַכְּלָיוֹת יְסִירֶנָּה:  מ  וְאֵת שְׁתֵּי הַכְּלָיֹת וְאֶת הַחֵלֶב אֲשֶׁר עֲלֵיהֶן אֲשֶׁר עַל הַכְּסָלִים וְאֶת הַיֹּתֶרֶת עַל הַכָּבֵד עַל הַכְּלָיוֹת יְסִירֶנָּה:  ת  וְיָת תַּרְתֵּין כָּלְיָן וְיָת תַּרְבָּא דִי עֲלֵיהֶן דִי עַל גִסְסַיָא וְיָת חִצְרָא דְעַל כַּבְדָא עַל כָּלְיֵתָא יֶעְדִנַהּ:

{י} כַּאֲשֶׁר יוּרַם מִשּׁוֹר זֶבַח הַשְּׁלָמִים וְהִקְטִירָם הַכֹּהֵן עַל מִזְבַּח הָעֹלָה:  מ  כַּאֲשֶׁר יוּרַם מִשּׁוֹר זֶבַח הַשְּׁלָמִים וְהִקְטִירָם הַכֹּהֵן עַל מִזְבַּח הָעֹלָה:  ת  כְּמָא דִי מִתַּפְרֵשׁ מִתּוֹר נִכְסַּת קוּדְשַׁיָא וְיַסֵקִנוּן כַּהֲנָא עַל מַדְבְּחָא דַעֲלָתָא:

{יא} וְאֶת עוֹר הַפָּר וְאֶת כָּל בְּשָׂרוֹ עַל רֹאשׁוֹ וְעַל כְּרָעָיו וְקִרְבּוֹ וּפִרְשׁוֹ:  מ  וְאֶת עוֹר הַפָּר וְאֶת כָּל בְּשָׂרוֹ עַל רֹאשׁוֹ וְעַל כְּרָעָיו וְקִרְבּוֹ וּפִרְשׁוֹ:  ת  וְיָת מְשַׁךְ תּוֹרָא וְיָת כָּל בִּסְרֵהּ עַל רֵישֵׁיהּ וְעַל כְּרָעוֹהִי וְגַוֵהּ וְאוּכְלֵהּ:

{יב} וְהוֹצִיא אֶת כָּל הַפָּר אֶל מִחוּץ לַמַּחֲנֶה אֶל מָקוֹם טָהוֹר אֶל שֶׁפֶךְ הַדֶּשֶׁן וְשָׂרַף אֹתוֹ עַל עֵצִים בָּאֵשׁ עַל שֶׁפֶךְ הַדֶּשֶׁן יִשָּׂרֵף:  מ  וְהוֹצִיא אֶת כָּל הַפָּר אֶל מִחוּץ לַמַּחֲנֶה אֶל מָקוֹם טָהוֹר אֶל שֶׁפֶךְ הַדֶּשֶׁן וְשָׂרַף אֹתוֹ עַל עֵצִים בָּאֵשׁ עַל שֶׁפֶךְ הַדֶּשֶׁן יִשָּׂרֵף:  ת  וְיַפֵּק יָת כָּל תּוֹרָא מִבָּרָא לְמַשְׁרִיתָא לַאֲתַר דְכֵי לַאֲתַר בֵּית מֵישַׁד קִטְמָא וְיוֹקִיד יָתֵהּ עַל אָעַיָא בְּאֶשָׁתָא עַל אֲתַר בֵּית מֵישַׁד קִטְמָא יִתּוֹקָד: (פ)

{יג} וְאִם כָּל עֲדַת יִשְׂרָאֵל יִשְׁגּוּ וְנֶעְלַם דָּבָר מֵעֵינֵי הַקָּהָל וְעָשׂוּ אַחַת מִכָּל מִצְוֹת יְהוָה אֲשֶׁר לֹא תֵעָשֶׂינָה וְאָשֵׁמוּ:  מ  וְאִם כָּל עֲדַת יִשְׂרָאֵל יִשְׁגּוּ וְנֶעְלַם דָּבָר מֵעֵינֵי הַקָּהָל וְעָשׂוּ אַחַת מִכָּל מִצְוֹת יְהוָה אֲשֶׁר לֹא תֵעָשֶׂינָה וְאָשֵׁמוּ:  ת  וְאִם כָּל כְּנִשְׁתָּא דְיִשְׂרָאֵל יִשְׁתְּלוּן וִיהֵי מְכַסָא פִּתְגָמָא מֵעֵינֵי קְהָלָא וְיַעְבְדוּן חָד מִכָּל פִּקוּדַיָא דַיְיָ דִי לָא כָשְׁרִין לְאִתְעֲבָדָא וִיחוֹבוּן:

{יד} וְנוֹדְעָה הַחַטָּאת אֲשֶׁר חָטְאוּ עָלֶיהָ וְהִקְרִיבוּ הַקָּהָל פַּר בֶּן בָּקָר לְחַטָּאת וְהֵבִיאוּ אֹתוֹ לִפְנֵי אֹהֶל מוֹעֵד:  מ  וְנוֹדְעָה הַחַטָּאת אֲשֶׁר חָטְאוּ עָלֶיהָ וְהִקְרִיבוּ הַקָּהָל פַּר בֶּן בָּקָר לְחַטָּאת וְהֵבִיאוּ אֹתוֹ לִפְנֵי אֹהֶל מוֹעֵד:  ת  וְתִתְיְדַע חוֹבְתָא דִי חָבוּ עֲלַהּ וִיקַרְבוּן קְהָלָא תּוֹר בַּר תּוֹרֵי לְחַטָּאתָא וְיַיְתוּן יָתֵהּ לָקֳדָם מַשְׁכַּן זִמְנָא:

{טו} וְסָמְכוּ זִקְנֵי הָעֵדָה אֶת יְדֵיהֶם עַל רֹאשׁ הַפָּר לִפְנֵי יְהוָה וְשָׁחַט אֶת הַפָּר לִפְנֵי יְהוָה:  מ  וְסָמְכוּ זִקְנֵי הָעֵדָה אֶת יְדֵיהֶם עַל רֹאשׁ הַפָּר לִפְנֵי יְהוָה וְשָׁחַט אֶת הַפָּר לִפְנֵי יְהוָה:  ת  וְיִסְמְכוּן סָבֵי כְנִשְׁתָּא יָת יְדֵיהוֹן עַל רֵישׁ תּוֹרָא קֳדָם יְיָ וְיִכּוֹס יָת תּוֹרָא קֳדָם יְיָ:

{טז} וְהֵבִיא הַכֹּהֵן הַמָּשִׁיחַ מִדַּם הַפָּר אֶל אֹהֶל מוֹעֵד:  מ  וְהֵבִיא הַכֹּהֵן הַמָּשִׁיחַ מִדַּם הַפָּר אֶל אֹהֶל מוֹעֵד:  ת  וְיָעֵל כַּהֲנָא רַבָּא מִדְמָא דְתוֹרָא לְמַשְׁכַּן זִמְנָא:

{יז} וְטָבַל הַכֹּהֵן אֶצְבָּעוֹ מִן הַדָּם וְהִזָּה שֶׁבַע פְּעָמִים לִפְנֵי יְהוָה אֵת פְּנֵי הַפָּרֹכֶת:  מ  וְטָבַל הַכֹּהֵן אֶצְבָּעוֹ מִן הַדָּם וְהִזָּה שֶׁבַע פְּעָמִים לִפְנֵי יְהוָה אֵת פְּנֵי הַפָּרֹכֶת:  ת  וְיִטְבּוֹל כַּהֲנָא אִצְבְּעֵהּ מִן דְמָא וְיַדִי שְׁבַע זִמְנִין קֳדָם יְיָ יָת קֳדָם פָּרֻכְתָּא:

{יח} וּמִן הַדָּם יִתֵּן עַל קַרְנֹת הַמִּזְבֵּחַ אֲשֶׁר לִפְנֵי יְהוָה אֲשֶׁר בְּאֹהֶל מוֹעֵד וְאֵת כָּל הַדָּם יִשְׁפֹּךְ אֶל יְסוֹד מִזְבַּח הָעֹלָה אֲשֶׁר פֶּתַח אֹהֶל מוֹעֵד:  מ  וּמִן הַדָּם יִתֵּן עַל קַרְנֹת הַמִּזְבֵּחַ אֲשֶׁר לִפְנֵי יְהוָה אֲשֶׁר בְּאֹהֶל מוֹעֵד וְאֵת כָּל הַדָּם יִשְׁפֹּךְ אֶל יְסוֹד מִזְבַּח הָעֹלָה אֲשֶׁר פֶּתַח אֹהֶל מוֹעֵד:  ת  וּמִן דְמָא יִתֵּן עַל קַרְנַת מַדְבְּחָא דִי קֳדָם יְיָ דִי בְּמַשְׁכַּן זִמְנָא וְיָת כָּל דְמָא יֵשׁוֹד לִיסוֹדָא דְמַדְבְּחָא דַעֲלָתָא דִי בִתְרַע מַשְׁכַּן זִמְנָא:

{יט} וְאֵת כָּל חֶלְבּוֹ יָרִים מִמֶּנּוּ וְהִקְטִיר הַמִּזְבֵּחָה:  מ  וְאֵת כָּל חֶלְבּוֹ יָרִים מִמֶּנּוּ וְהִקְטִיר הַמִּזְבֵּחָה:  ת  וְיָת כָּל תַּרְבֵּהּ יַפְרֵשׁ מִנֵהּ וְיַסֵק לְמַדְבְּחָא:

{כ} וְעָשָׂה לַפָּר כַּאֲשֶׁר עָשָׂה לְפַר הַחַטָּאת כֵּן יַעֲשֶׂה לּוֹ וְכִפֶּר עֲלֵהֶם הַכֹּהֵן וְנִסְלַח לָהֶם:  מ  וְעָשָׂה לַפָּר כַּאֲשֶׁר עָשָׂה לְפַר הַחַטָּאת כֵּן יַעֲשֶׂה לּוֹ וְכִפֶּר עֲלֵהֶם הַכֹּהֵן וְנִסְלַח לָהֶם:  ת  וְיַעְבֵּד לְתוֹרָא כְּמָא דִי עֲבַד לְתוֹרָא דְחַטָאתָא כֵּן יַעְבֵּד לֵהּ וִיכַפֵּר עֲלֵיהוֹן כַּהֲנָא וְיִשְׁתְּבֵק לְהוֹן:

{כא} וְהוֹצִיא אֶת הַפָּר אֶל מִחוּץ לַמַּחֲנֶה וְשָׂרַף אֹתוֹ כַּאֲשֶׁר שָׂרַף אֵת הַפָּר הָרִאשׁוֹן חַטַּאת הַקָּהָל הוּא:  מ  וְהוֹצִיא אֶת הַפָּר אֶל מִחוּץ לַמַּחֲנֶה וְשָׂרַף אֹתוֹ כַּאֲשֶׁר שָׂרַף אֵת הַפָּר הָרִאשׁוֹן חַטַּאת הַקָּהָל הוּא:  ת  וְיַפֵּק יָת תּוֹרָא לְמִבָּרָא לְמַשְׁרִיתָא וְאוֹקִיד יָתֵהּ כְּמָא דִי אוֹקִיד יָת תּוֹרָא קַדְמָאָה חַטַאת קְהָלָא הוּא: (פ)

{כב} אֲשֶׁר נָשִׂיא יֶחֱטָא וְעָשָׂה אַחַת מִכָּל מִצְוֹת יְהוָה אֱלֹהָיו אֲשֶׁר לֹא תֵעָשֶׂינָה בִּשְׁגָגָה וְאָשֵׁם:  מ  אֲשֶׁר נָשִׂיא יֶחֱטָא וְעָשָׂה אַחַת מִכָּל מִצְוֹת יְהוָה אֱלֹהָיו אֲשֶׁר לֹא תֵעָשֶׂינָה בִּשְׁגָגָה וְאָשֵׁם:  ת  אִם רַבָּא יֵחוֹב וְיַעְבֵּד חַד מִכָּל פִּקוּדַיָא דַיְיָ אֱלָהֵיהּ דְלָא כַשְׁרִין לְאִתְעֲבָדָא בְּשָׁלוּ וְיֵחוֹב:

{כג} אוֹ הוֹדַע אֵלָיו חַטָּאתוֹ אֲשֶׁר חָטָא בָּהּ וְהֵבִיא אֶת קָרְבָּנוֹ שְׂעִיר עִזִּים זָכָר תָּמִים:  מ  אוֹ הוֹדַע אֵלָיו חַטָּאתוֹ אֲשֶׁר חָטָא בָּהּ וְהֵבִיא אֶת קָרְבָּנוֹ שְׂעִיר עִזִּים זָכָר תָּמִים:  ת  אוֹ אִתְיְדַע לֵהּ חוֹבְתֵהּ דִי חַב בַּהּ וְיַיְתִי יָת קֻרְבָּנֵהּ צְפִיר בַּר עִזִין דְכַר שְׁלִים:

{כד} וְסָמַךְ יָדוֹ עַל רֹאשׁ הַשָּׂעִיר וְשָׁחַט אֹתוֹ בִּמְקוֹם אֲשֶׁר יִשְׁחַט אֶת הָעֹלָה לִפְנֵי יְהוָה חַטָּאת הוּא:  מ  וְסָמַךְ יָדוֹ עַל רֹאשׁ הַשָּׂעִיר וְשָׁחַט אֹתוֹ בִּמְקוֹם אֲשֶׁר יִשְׁחַט אֶת הָעֹלָה לִפְנֵי יְהוָה חַטָּאת הוּא:  ת  וְיִסְמוֹךְ יְדֵהּ עַל רֵישָׁא דִצְפִירָא וְיִכּוֹס יָתֵהּ בַּאֲתַר דְיִכּוֹס יָת עֲלָתָא לָקֳדָם יְיָ חַטָאתָא הוּא:

{כה} וְלָקַח הַכֹּהֵן מִדַּם הַחַטָּאת בְּאֶצְבָּעוֹ וְנָתַן עַל קַרְנֹת מִזְבַּח הָעֹלָה וְאֶת דָּמוֹ יִשְׁפֹּךְ אֶל יְסוֹד מִזְבַּח הָעֹלָה:  מ  וְלָקַח הַכֹּהֵן מִדַּם הַחַטָּאת בְּאֶצְבָּעוֹ וְנָתַן עַל קַרְנֹת מִזְבַּח הָעֹלָה וְאֶת דָּמוֹ יִשְׁפֹּךְ אֶל יְסוֹד מִזְבַּח הָעֹלָה:  ת  וְיִסַב כַּהֲנָא מִדְמָא דְחַטָּאתָא בְּאֶצְבְּעֵהּ וְיִתֵּן עַל קַרְנַת מַדְבְּחָא דַעֲלָתָא וְיָת דְמֵהּ יֵשׁוֹד לִיסוֹדָא לְמַדְבְּחָא דַעֲלָתָא:

{א} וְנֶפֶשׁ כִּי תֶחֱטָא וְשָׁמְעָה קוֹל אָלָה וְהוּא עֵד אוֹ רָאָה אוֹ יָדָע אִם לוֹא יַגִּיד וְנָשָׂא עֲוֹנוֹ:  מ  וְנֶפֶשׁ כִּי תֶחֱטָא וְשָׁמְעָה קוֹל אָלָה וְהוּא עֵד אוֹ רָאָה אוֹ יָדָע אִם לוֹא יַגִּיד וְנָשָׂא עֲוֹנוֹ:  ת  וֶאֵנַשׁ אֲרֵי יֵחוֹב וְיִשְׁמַע קַל מוֹמֵי וְהוּא סָהִיד אוֹ חֲזָא אוֹ יְדָע אִם לָא יְחַוִי וִיקַבֵּל חוֹבֵהּ:

{ב} אוֹ נֶפֶשׁ אֲשֶׁר תִּגַּע בְּכָל דָּבָר טָמֵא אוֹ בְנִבְלַת חַיָּה טְמֵאָה אוֹ בְּנִבְלַת בְּהֵמָה טְמֵאָה אוֹ בְּנִבְלַת שֶׁרֶץ טָמֵא וְנֶעְלַם מִמֶּנּוּ וְהוּא טָמֵא וְאָשֵׁם:  מ  אוֹ נֶפֶשׁ אֲשֶׁר תִּגַּע בְּכָל דָּבָר טָמֵא אוֹ בְנִבְלַת חַיָּה טְמֵאָה אוֹ בְּנִבְלַת בְּהֵמָה טְמֵאָה אוֹ בְּנִבְלַת שֶׁרֶץ טָמֵא וְנֶעְלַם מִמֶּנּוּ וְהוּא טָמֵא וְאָשֵׁם:  ת  אוֹ אֱנַשׁ דִי יִקְרַב בְּכָל מִדַעַם מְסָאָב אוֹ בְּנִבְלַת חַיְתָא מְסָאָבָא אוֹ בְּנִבְלַת בְּעִירָא מְסָאָבָא אוֹ בְּנִבְלַת רִחֲשָׁא מְסָאָב וִיהֵי מְכַסָא מִנֵהּ וְהוּא מְסָאָב וְחָב:

{ג} אוֹ כִי יִגַּע בְּטֻמְאַת אָדָם לְכֹל טֻמְאָתוֹ אֲשֶׁר יִטְמָא בָּהּ וְנֶעְלַם מִמֶּנּוּ וְהוּא יָדַע וְאָשֵׁם:  מ  אוֹ כִי יִגַּע בְּטֻמְאַת אָדָם לְכֹל טֻמְאָתוֹ אֲשֶׁר יִטְמָא בָּהּ וְנֶעְלַם מִמֶּנּוּ וְהוּא יָדַע וְאָשֵׁם:  ת  אוֹ אֲרֵי יִקְרַב בְּסוֹאֲבַת אֱנָשָׁא לְכָל סוֹבְתֵהּ דִי יִסְתָּאַב בַּהּ וִיהֵי מְכַסָא מִנֵהּ וְהוּא יְדַע וְחָב:

{ד} אוֹ נֶפֶשׁ כִּי תִשָּׁבַע לְבַטֵּא בִשְׂפָתַיִם לְהָרַע אוֹ לְהֵיטִיב לְכֹל אֲשֶׁר יְבַטֵּא הָאָדָם בִּשְׁבֻעָה וְנֶעְלַם מִמֶּנּוּ וְהוּא יָדַע וְאָשֵׁם לְאַחַת מֵאֵלֶּה:  מ  אוֹ נֶפֶשׁ כִּי תִשָּׁבַע לְבַטֵּא בִשְׂפָתַיִם לְהָרַע אוֹ לְהֵיטִיב לְכֹל אֲשֶׁר יְבַטֵּא הָאָדָם בִּשְׁבֻעָה וְנֶעְלַם מִמֶּנּוּ וְהוּא יָדַע וְאָשֵׁם לְאַחַת מֵאֵלֶּה:  ת  אוֹ אֱנַשׁ אֲרֵי יְקַיֵם לְפָרָשָׁא בְסִפְוָן לְאַבְאָשָׁא אוֹ לְאוֹטָבָא לְכֹל דִי יְפָרֵשׁ אֱנָשָׁא בְּקִיוּם וִיהֵי מְכַסָא מִנֵהּ וְהוּא יָדַע וְחָב לַחֲדָא מֵאִלֵין:

{ה} וְהָיָה כִי יֶאְשַׁם לְאַחַת מֵאֵלֶּה וְהִתְוַדָּה אֲשֶׁר חָטָא עָלֶיהָ:  מ  וְהָיָה כִי יֶאְשַׁם לְאַחַת מֵאֵלֶּה וְהִתְוַדָּה אֲשֶׁר חָטָא עָלֶיהָ:  ת  וִיהֵי אֲרֵי יֵחוֹב לַחֲדָא מֵאִלֵין וִיוַדִי דִי חָב עֲלַהּ:

{ו} וְהֵבִיא אֶת אֲשָׁמוֹ לַיהוָה עַל חַטָּאתוֹ אֲשֶׁר חָטָא נְקֵבָה מִן הַצֹּאן כִּשְׂבָּה אוֹ שְׂעִירַת עִזִּים לְחַטָּאת וְכִפֶּר עָלָיו הַכֹּהֵן מֵחַטָּאתוֹ:  מ  וְהֵבִיא אֶת אֲשָׁמוֹ לַיהוָה עַל חַטָּאתוֹ אֲשֶׁר חָטָא נְקֵבָה מִן הַצֹּאן כִּשְׂבָּה אוֹ שְׂעִירַת עִזִּים לְחַטָּאת וְכִפֶּר עָלָיו הַכֹּהֵן מֵחַטָּאתוֹ:  ת  וְיַיְתִי יָת אַשְׁמֵהּ קֳדָם יְיָ עַל חוֹבְתֵהּ דִי חָב נְקוּבְתָא מִן עָנָא אִמְרָתָא אוֹ צְפִירַת עִזִין לְחַטָאתָא וִיכַפֵּר עֲלוֹהִי כַהֲנָא מֵחוֹבְתֵהּ:

{ז} וְאִם לֹא תַגִּיע יָדוֹ דֵּי שֶׂה וְהֵבִיא אֶת אֲשָׁמוֹ אֲשֶׁר חָטָא שְׁתֵּי תֹרִים אוֹ שְׁנֵי בְנֵי יוֹנָה לַיהוָה אֶחָד לְחַטָּאת וְאֶחָד לְעֹלָה:  מ  וְאִם לֹא תַגִּיע יָדוֹ דֵּי שֶׂה וְהֵבִיא אֶת אֲשָׁמוֹ אֲשֶׁר חָטָא שְׁתֵּי תֹרִים אוֹ שְׁנֵי בְנֵי יוֹנָה לַיהוָה אֶחָד לְחַטָּאת וְאֶחָד לְעֹלָה:  ת  וְאִם לָא תִמְטֵי יְדֵהּ כְּמִסַת שֵׂיתָא וְיַיְתִי יָת חוֹבְתֵהּ דִי חָב תַּרְתֵּין שַׁפְנִינִין אוֹ תְרֵין בְּנֵי יוֹנָה קֳדָם יְיָ חַד לְחַטָאתָא וְחַד לַעֲלָתָא:

{ח} וְהֵבִיא אֹתָם אֶל הַכֹּהֵן וְהִקְרִיב אֶת אֲשֶׁר לַחַטָּאת רִאשׁוֹנָה וּמָלַק אֶת רֹאשׁוֹ מִמּוּל עָרְפּוֹ וְלֹא יַבְדִּיל:  מ  וְהֵבִיא אֹתָם אֶל הַכֹּהֵן וְהִקְרִיב אֶת אֲשֶׁר לַחַטָּאת רִאשׁוֹנָה וּמָלַק אֶת רֹאשׁוֹ מִמּוּל עָרְפּוֹ וְלֹא יַבְדִּיל:  ת  וְיַיְתִי יָתְהוֹן לְוָת כַּהֲנָא וִיקָרֵב יָת דִי לְחַטָאתָא קַדְמֵיתָא וְיִמְלוֹק יָת רֵישֵׁהּ מִלְקָבֵל קְדָלֵהּ וְלָא יַפְרֵשׁ:

{ט} וְהִזָּה מִדַּם הַחַטָּאת עַל קִיר הַמִּזְבֵּחַ וְהַנִּשְׁאָר בַּדָּם יִמָּצֵה אֶל יְסוֹד הַמִּזְבֵּחַ חַטָּאת הוּא:  מ  וְהִזָּה מִדַּם הַחַטָּאת עַל קִיר הַמִּזְבֵּחַ וְהַנִּשְׁאָר בַּדָּם יִמָּצֵה אֶל יְסוֹד הַמִּזְבֵּחַ חַטָּאת הוּא:  ת  וְיַדִי מִדְמָא דְחַטָּאתָא עַל כֹּתֶל מַדְבְּחָא וּדְאִשְׁתָּאַר בִּדְמָא יִתִּמְצֵי לִיסוֹדָא דְמַדְבְּחָא חַטָאתָא הוּא:

{י} וְאֶת הַשֵּׁנִי יַעֲשֶׂה עֹלָה כַּמִּשְׁפָּט וְכִפֶּר עָלָיו הַכֹּהֵן מֵחַטָּאתוֹ אֲשֶׁר חָטָא וְנִסְלַח לוֹ:  מ  וְאֶת הַשֵּׁנִי יַעֲשֶׂה עֹלָה כַּמִּשְׁפָּט וְכִפֶּר עָלָיו הַכֹּהֵן מֵחַטָּאתוֹ אֲשֶׁר חָטָא וְנִסְלַח לוֹ:  ת  וְיָת תִּנְיָנָא יַעְבֵּד עֲלָתָא כִּדְחָזֵי וִיכַפֵּר עֲלוֹהִי כַהֲנָא מֵחוֹבְתֵהּ דִי חָב וְיִשְׁתְּבֵק לֵהּ: (ס)

{יא}  שביעי  וְאִם לֹא תַשִּׂיג יָדוֹ לִשְׁתֵּי תֹרִים אוֹ לִשְׁנֵי בְנֵי יוֹנָה וְהֵבִיא אֶת קָרְבָּנוֹ אֲשֶׁר חָטָא עֲשִׂירִת הָאֵפָה סֹלֶת לְחַטָּאת לֹא יָשִׂים עָלֶיהָ שֶׁמֶן וְלֹא יִתֵּן עָלֶיהָ לְבֹנָה כִּי חַטָּאת הִיא:  מ  וְאִם לֹא תַשִּׂיג יָדוֹ לִשְׁתֵּי תֹרִים אוֹ לִשְׁנֵי בְנֵי יוֹנָה וְהֵבִיא אֶת קָרְבָּנוֹ אֲשֶׁר חָטָא עֲשִׂירִת הָאֵפָה סֹלֶת לְחַטָּאת לֹא יָשִׂים עָלֶיהָ שֶׁמֶן וְלֹא יִתֵּן עָלֶיהָ לְבֹנָה כִּי חַטָּאת הִיא:  ת  וְאִם לָא תַדְבֵּק יְדֵהּ לְתַרְתֵּין שַׁפְנִינִין אוֹ לִתְרֵין בְּנֵי יוֹנָה וְיַיְתִי יָת קֻרְבָּנֵהּ דִי חָב חַד מִן עַסְרָא בִתְלַת סְאִין סֻלְתָּא לְחַטָּאתָא לָא יְשַׁוִי עֲלַהּ מִשְׁחָא וְלָא יִתֵּן עֲלַהּ לְבֻנְתָּא אֲרֵי חַטָאתָא הִיא:

{יב} וֶהֱבִיאָהּ אֶל הַכֹּהֵן וְקָמַץ הַכֹּהֵן מִמֶּנָּה מְלוֹא קֻמְצוֹ אֶת אַזְכָּרָתָה וְהִקְטִיר הַמִּזְבֵּחָה עַל אִשֵּׁי יְהוָה חַטָּאת הִוא:  מ  וֶהֱבִיאָהּ אֶל הַכֹּהֵן וְקָמַץ הַכֹּהֵן מִמֶּנָּה מְלוֹא קֻמְצוֹ אֶת אַזְכָּרָתָה וְהִקְטִיר הַמִּזְבֵּחָה עַל אִשֵּׁי יְהוָה חַטָּאת הִוא:  ת  וְיַיְתִנַהּ לְוַת כַּהֲנָא וְיִקְמוֹץ כַּהֲנָא מִנַהּ מְלֵי קֻמְצֵהּ יָת אִדְכַּרְתַּהּ וְיַסֵק לְמַדְבְּחָא עַל קֻרְבָּנַיָא דַיְיָ חַטָאתָא הִיא:

{יג} וְכִפֶּר עָלָיו הַכֹּהֵן עַל חַטָּאתוֹ אֲשֶׁר חָטָא מֵאַחַת מֵאֵלֶּה וְנִסְלַח לוֹ וְהָיְתָה לַכֹּהֵן כַּמִּנְחָה:  מ  וְכִפֶּר עָלָיו הַכֹּהֵן עַל חַטָּאתוֹ אֲשֶׁר חָטָא מֵאַחַת מֵאֵלֶּה וְנִסְלַח לוֹ וְהָיְתָה לַכֹּהֵן כַּמִּנְחָה:  ת  וִיכַפֵּר עֲלוֹהִי כַהֲנָא עַל חוֹבְתֵהּ דִי חָב מֵחֲדָא מֵאִלֵין וְיִשְׁתְּבֵק לֵהּ וּתְהֵי לְכַהֲנָא כְּמִנְחָתָא: (ס)

{יד} וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה לֵּאמֹר:  מ  וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה לֵּאמֹר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{טו} נֶפֶשׁ כִּי תִמְעֹל מַעַל וְחָטְאָה בִּשְׁגָגָה מִקָּדְשֵׁי יְהוָה וְהֵבִיא אֶת אֲשָׁמוֹ לַיהוָה אַיִל תָּמִים מִן הַצֹּאן בְּעֶרְכְּךָ כֶּסֶף שְׁקָלִים בְּשֶׁקֶל הַקֹּדֶשׁ לְאָשָׁם:  מ  נֶפֶשׁ כִּי תִמְעֹל מַעַל וְחָטְאָה בִּשְׁגָגָה מִקָּדְשֵׁי יְהוָה וְהֵבִיא אֶת אֲשָׁמוֹ לַיהוָה אַיִל תָּמִים מִן הַצֹּאן בְּעֶרְכְּךָ כֶּסֶף שְׁקָלִים בְּשֶׁקֶל הַקֹּדֶשׁ לְאָשָׁם:  ת  אֱנַשׁ אֲרֵי יְשַׁקַר שְׁקַר וְיֵחוֹב בְּשָׁלוּ מִקוּדְשַׁיָא דַיְיָ וְיַיְתִי יָת אַשְׁמֵהּ קֳדָם יְיָ דְכַר שְׁלִים מִן עָנָא בְּפֻרְסָנֵהּ כְּסַף סִלְעִין בְּסִלְעֵי קוּדְשָׁא לַאֲשָׁמָא:

{טז} וְאֵת אֲשֶׁר חָטָא מִן הַקֹּדֶשׁ יְשַׁלֵּם וְאֶת חֲמִישִׁתוֹ יוֹסֵף עָלָיו וְנָתַן אֹתוֹ לַכֹּהֵן וְהַכֹּהֵן יְכַפֵּר עָלָיו בְּאֵיל הָאָשָׁם וְנִסְלַח לוֹ:  מ  וְאֵת אֲשֶׁר חָטָא מִן הַקֹּדֶשׁ יְשַׁלֵּם וְאֶת חֲמִישִׁתוֹ יוֹסֵף עָלָיו וְנָתַן אֹתוֹ לַכֹּהֵן וְהַכֹּהֵן יְכַפֵּר עָלָיו בְּאֵיל הָאָשָׁם וְנִסְלַח לוֹ:  ת  וְיָת דִי חָב מִן קוּדְשָׁא יְשַׁלֵם וְיָת חֻמְשֵׁהּ יוֹסֵף עֲלוֹהִי וְיִתֵּן יָתֵהּ לְכַהֲנָא וְכַהֲנָא יְכַפֵּר עֲלוֹהִי בְּדִכְרָא דַאֲשָׁמָא וְיִשְׁתְּבֵק לֵהּ: (פ)

{יז} וְאִם נֶפֶשׁ כִּי תֶחֱטָא וְעָשְׂתָה אַחַת מִכָּל מִצְוֹת יְהוָה אֲשֶׁר לֹא תֵעָשֶׂינָה וְלֹא יָדַע וְאָשֵׁם וְנָשָׂא עֲוֹנוֹ:  מ  וְאִם נֶפֶשׁ כִּי תֶחֱטָא וְעָשְׂתָה אַחַת מִכָּל מִצְוֹת יְהוָה אֲשֶׁר לֹא תֵעָשֶׂינָה וְלֹא יָדַע וְאָשֵׁם וְנָשָׂא עֲוֹנוֹ:  ת  וְאִם אֱנַשׁ אֲרֵי יֵחוֹב וְיַעְבֵּד חֲדָא מִכָּל פִּקוּדַיָא דַיְיָ דִי לָא כָשְׁרִין לְאִתְעֲבָדָא וְלָא יָדַע וְיֵחוֹב וִיקַבֵּל חוֹבֵהּ:

{יח} וְהֵבִיא אַיִל תָּמִים מִן הַצֹּאן בְּעֶרְכְּךָ לְאָשָׁם אֶל הַכֹּהֵן וְכִפֶּר עָלָיו הַכֹּהֵן עַל שִׁגְגָתוֹ אֲשֶׁר שָׁגָג וְהוּא לֹא יָדַע וְנִסְלַח לוֹ:  מ  וְהֵבִיא אַיִל תָּמִים מִן הַצֹּאן בְּעֶרְכְּךָ לְאָשָׁם אֶל הַכֹּהֵן וְכִפֶּר עָלָיו הַכֹּהֵן עַל שִׁגְגָתוֹ אֲשֶׁר שָׁגָג וְהוּא לֹא יָדַע וְנִסְלַח לוֹ:  ת  וְיַיְתִי דְכַר שְׁלִים מִן עָנָא בְּפֻרְסָנֵהּ לַאֲשָׁמָא לְוָת כַּהֲנָא וִיכַפֵּר עֲלוֹהִי כַהֲנָא עַל שָׁלוּתֵהּ דְאִשְׁתְּלִי וְהוּא לָא יָדַע וְיִשְׁתְּבֵק לֵהּ:

{יט} אָשָׁם הוּא אָשֹׁם אָשַׁם לַיהוָה:  מ  אָשָׁם הוּא אָשֹׁם אָשַׁם לַיהוָה:  ת  אֲשָׁמָא הוּא עַל חוֹבְתֵהּ דְהוּא חָב אֲשָׁמָא יְקָרֵב קֳדָם יְיָ: (פ)

{כ} וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה לֵּאמֹר:  מ  וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה לֵּאמֹר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{כא} נֶפֶשׁ כִּי תֶחֱטָא וּמָעֲלָה מַעַל בַּיהוָה וְכִחֵשׁ בַּעֲמִיתוֹ בְּפִקָּדוֹן אוֹ בִתְשׂוּמֶת יָד אוֹ בְגָזֵל אוֹ עָשַׁק אֶת עֲמִיתוֹ:  מ  נֶפֶשׁ כִּי תֶחֱטָא וּמָעֲלָה מַעַל בַּיהוָה וְכִחֵשׁ בַּעֲמִיתוֹ בְּפִקָּדוֹן אוֹ בִתְשׂוּמֶת יָד אוֹ בְגָזֵל אוֹ עָשַׁק אֶת עֲמִיתוֹ:  ת  אֱנַשׁ אֲרֵי יֵחוֹב וִישַׁקַר שְׁקַר קֳדָם יְיָ וִיכַדֵב בְּחַבְרֵהּ בְּפִקְדוֹנָא אוֹ בְשֻׁתָּפוּת יְדָא אוֹ בִגְזֵלָא אוֹ אַנִיס יָת חַבְרֵהּ:

{כב} אוֹ מָצָא אֲבֵדָה וְכִחֶשׁ בָּהּ וְנִשְׁבַּע עַל שָׁקֶר עַל אַחַת מִכֹּל אֲשֶׁר יַעֲשֶׂה הָאָדָם לַחֲטֹא בָהֵנָּה:  מ  אוֹ מָצָא אֲבֵדָה וְכִחֶשׁ בָּהּ וְנִשְׁבַּע עַל שָׁקֶר עַל אַחַת מִכֹּל אֲשֶׁר יַעֲשֶׂה הָאָדָם לַחֲטֹא בָהֵנָּה:  ת  אוֹ אַשְׁכַּח אֲבֶדְתָּא וִיכַדֵב בַּהּ וְאִשְׁתְּבַע עַל שִׁקְרָא עַל חֲדָא מִכֹּל דִי יַעְבֵּד אֱנָשָׁא לְמֵיחַב בְּהֵן:

{כג} וְהָיָה כִּי יֶחֱטָא וְאָשֵׁם וְהֵשִׁיב אֶת הַגְּזֵלָה אֲשֶׁר גָּזָל אוֹ אֶת הָעֹשֶׁק אֲשֶׁר עָשָׁק אוֹ אֶת הַפִּקָּדוֹן אֲשֶׁר הָפְקַד אִתּוֹ אוֹ אֶת הָאֲבֵדָה אֲשֶׁר מָצָא:  מ  וְהָיָה כִּי יֶחֱטָא וְאָשֵׁם וְהֵשִׁיב אֶת הַגְּזֵלָה אֲשֶׁר גָּזָל אוֹ אֶת הָעֹשֶׁק אֲשֶׁר עָשָׁק אוֹ אֶת הַפִּקָּדוֹן אֲשֶׁר הָפְקַד אִתּוֹ אוֹ אֶת הָאֲבֵדָה אֲשֶׁר מָצָא:  ת  וִיהֵי אֲרֵי יֶחְטֵי וְיֵחוֹב וְיָתֵב יָת גְזֵלָא דִי גְזַל אוֹ יָת עֻשְׁקָא דִי עֲשַׁק אוֹ יָת פִּקְדוֹנָא דִי אִתְפָּקַד לְוָתֵהּ אוֹ יָת אֲבֶדְתָּא דִי אַשְׁכַּח:

{כד}  מפטיר  אוֹ מִכֹּל אֲשֶׁר יִשָּׁבַע עָלָיו לַשֶּׁקֶר וְשִׁלַּם אֹתוֹ בְּרֹאשׁוֹ וַחֲמִשִׁתָיו יֹסֵף עָלָיו לַאֲשֶׁר הוּא לוֹ יִתְּנֶנּוּ בְּיוֹם אַשְׁמָתוֹ:  מ  אוֹ מִכֹּל אֲשֶׁר יִשָּׁבַע עָלָיו לַשֶּׁקֶר וְשִׁלַּם אֹתוֹ בְּרֹאשׁוֹ וַחֲמִשִׁתָיו יֹסֵף עָלָיו לַאֲשֶׁר הוּא לוֹ יִתְּנֶנּוּ בְּיוֹם אַשְׁמָתוֹ:  ת  אוֹ מִכֹּל דִי יִשְׁתְּבַע עֲלוֹהִי לְשִׁקְרָא וִישַׁלֵם יָתֵהּ בְּרֵישֵׁיהּ וְחֻמְשֵׁהּ יוֹסֵף עֲלוֹהִי לְדִי הוּא דִילֵהּ יִתְּנִנֵהּ בְּיוֹמָא דְחוֹבְתֵהּ:

{כה} וְאֶת אֲשָׁמוֹ יָבִיא לַיהוָה אַיִל תָּמִים מִן הַצֹּאן בְּעֶרְכְּךָ לְאָשָׁם אֶל הַכֹּהֵן:  מ  וְאֶת אֲשָׁמוֹ יָבִיא לַיהוָה אַיִל תָּמִים מִן הַצֹּאן בְּעֶרְכְּךָ לְאָשָׁם אֶל הַכֹּהֵן:  ת  וְיָת אַשָׁמֵיהּ יַיְתִי לָקֳדָם יְיָ דְכַר שְׁלִים מִן עָנָא בְּפֻרְסָנֵהּ לַאֲשָׁמָא לְוָת כַּהֲנָא:

{כו} וְכִפֶּר עָלָיו הַכֹּהֵן לִפְנֵי יְהוָה וְנִסְלַח לוֹ עַל אַחַת מִכֹּל אֲשֶׁר יַעֲשֶׂה לְאַשְׁמָה בָהּ:  מ  וְכִפֶּר עָלָיו הַכֹּהֵן לִפְנֵי יְהוָה וְנִסְלַח לוֹ עַל אַחַת מִכֹּל אֲשֶׁר יַעֲשֶׂה לְאַשְׁמָה בָהּ:  ת  וִיכַפֵּר עֲלוֹהִי כַהֲנָא קֳדָם יְיָ וְיִשְׁתְּבֵק לֵהּ עַל חֲדָא מִכֹּל דִי יַעְבֵּד לְמֵיחַב בַהּ: (פפפ)
