פרשת ויצא


{י} וַיֵּצֵא יַעֲקֹב מִבְּאֵר שָׁבַע וַיֵּלֶךְ חָרָנָה:  מ  וַיֵּצֵא יַעֲקֹב מִבְּאֵר שָׁבַע וַיֵּלֶךְ חָרָנָה:  ת  וּנְפַק יַעֲקֹב מִבְּאֵרָא דְשָׁבַע וַאֲזַל לְחָרָן:

{יא} וַיִּפְגַּע בַּמָּקוֹם וַיָּלֶן שָׁם כִּי בָא הַשֶּׁמֶשׁ וַיִּקַּח מֵאַבְנֵי הַמָּקוֹם וַיָּשֶׂם מְרַאֲשֹׁתָיו וַיִּשְׁכַּב בַּמָּקוֹם הַהוּא:  מ  וַיִּפְגַּע בַּמָּקוֹם וַיָּלֶן שָׁם כִּי בָא הַשֶּׁמֶשׁ וַיִּקַּח מֵאַבְנֵי הַמָּקוֹם וַיָּשֶׂם מְרַאֲשֹׁתָיו וַיִּשְׁכַּב בַּמָּקוֹם הַהוּא:  ת  וַעֲרַע בְּאַתְרָא וּבָת תַּמָּן אֲרֵי עָל שִׁמְשָׁא וּנְסִיב מֵאַבְנֵי אַתְרָא וְשַׁוִּי אִסָדוֹהִי וּשְׁכִיב בְּאַתְרָא הַהוּא:

{יב} וַיַּחֲלֹם וְהִנֵּה סֻלָּם מֻצָּב אַרְצָה וְרֹאשׁוֹ מַגִּיעַ הַשָּׁמָיְמָה וְהִנֵּה מַלְאֲכֵי אֱלֹהִים עֹלִים וְיֹרְדִים בּוֹ:  מ  וַיַּחֲלֹם וְהִנֵּה סֻלָּם מֻצָּב אַרְצָה וְרֹאשׁוֹ מַגִּיעַ הַשָּׁמָיְמָה וְהִנֵּה מַלְאֲכֵי אֱלֹהִים עֹלִים וְיֹרְדִים בּוֹ:  ת  וַחֲלַם וְהָא סֻלְמָא נָעִיץ בְּאַרְעָא וְרֵישֵׁהּ מָטֵי עַד צֵית שְׁמַיָּא וְהָא מַלְאָכַיָּא דַּיְיָ סָלְקִין וְנָחֲתִין בֵּהּ:

{יג} וְהִנֵּה יְהוָה נִצָּב עָלָיו וַיֹּאמַר אֲנִי יְהוָה אֱלֹהֵי אַבְרָהָם אָבִיךָ וֵאלֹהֵי יִצְחָק הָאָרֶץ אֲשֶׁר אַתָּה שֹׁכֵב עָלֶיהָ לְךָ אֶתְּנֶנָּה וּלְזַרְעֶךָ:  מ  וְהִנֵּה יְהוָה נִצָּב עָלָיו וַיֹּאמַר אֲנִי יְהוָה אֱלֹהֵי אַבְרָהָם אָבִיךָ וֵאלֹהֵי יִצְחָק הָאָרֶץ אֲשֶׁר אַתָּה שֹׁכֵב עָלֶיהָ לְךָ אֶתְּנֶנָּה וּלְזַרְעֶךָ:  ת  וְהָא יְקָרָא דַיְיָ מְעַתַּד עִלָווֹהִי וַאֲמַר אֲנָא יְיָ אֱלָהֵהּ דְּאַבְרָהָם אָבוּךְ וֵאלָהֵהּ דְּיִצְחָק אַרְעָא דִּי אַתְּ שְׁכִיב עֲלַהּ לָךְ אֶתְּנִנַּהּ וְלִבְנָיךְ:

{יד} וְהָיָה זַרְעֲךָ כַּעֲפַר הָאָרֶץ וּפָרַצְתָּ יָמָּה וָקֵדְמָה וְצָפֹנָה וָנֶגְבָּה וְנִבְרְכוּ בְךָ כָּל מִשְׁפְּחֹת הָאֲדָמָה וּבְזַרְעֶךָ:  מ  וְהָיָה זַרְעֲךָ כַּעֲפַר הָאָרֶץ וּפָרַצְתָּ יָמָּה וָקֵדְמָה וְצָפֹנָה וָנֶגְבָּה וְנִבְרְכוּ בְךָ כָּל מִשְׁפְּחֹת הָאֲדָמָה וּבְזַרְעֶךָ:  ת  וִיהוֹן בְּנָיךְ סַגִּיאִין כְּעַפְרָא דְאַרְעָא וְתִתְקֵף לְמַעַרְבָא וּלְמַדִינְחָא וּלְצִפּוּנָא וּלְדָרוֹמָא וְיִתְבָּרְכוּן בְּדִילָךְ כָּל זַרְעֲיַת אַרְעָא וּבְדִיל בְּנָיךְ:

{טו} וְהִנֵּה אָנֹכִי עִמָּךְ וּשְׁמַרְתִּיךָ בְּכֹל אֲשֶׁר תֵּלֵךְ וַהֲשִׁבֹתִיךָ אֶל הָאֲדָמָה הַזֹּאת כִּי לֹא אֶעֱזָבְךָ עַד אֲשֶׁר אִם עָשִׂיתִי אֵת אֲשֶׁר דִּבַּרְתִּי לָךְ:  מ  וְהִנֵּה אָנֹכִי עִמָּךְ וּשְׁמַרְתִּיךָ בְּכֹל אֲשֶׁר תֵּלֵךְ וַהֲשִׁבֹתִיךָ אֶל הָאֲדָמָה הַזֹּאת כִּי לֹא אֶעֱזָבְךָ עַד אֲשֶׁר אִם עָשִׂיתִי אֵת אֲשֶׁר דִּבַּרְתִּי לָךְ:  ת  וְהָא מֵימְרִי בְּסַעְדָךְ וְאֶטְרִנָךְ בְּכָל אֲתַר דִּי תְהַךְ וַאֲתֵיבִנָּךְ לְאַרְעָא הָדָא אֲרֵי לָא אֶשְׁבְּקִנָּךְ עַד דִּי אֶעְבֵּיד יָת דִּי מַלֵּלִית לָךְ:

{טז} וַיִּיקַץ יַעֲקֹב מִשְּׁנָתוֹ וַיֹּאמֶר אָכֵן יֵשׁ יְהוָה בַּמָּקוֹם הַזֶּה וְאָנֹכִי לֹא יָדָעְתִּי:  מ  וַיִּיקַץ יַעֲקֹב מִשְּׁנָתוֹ וַיֹּאמֶר אָכֵן יֵשׁ יְהוָה בַּמָּקוֹם הַזֶּה וְאָנֹכִי לֹא יָדָעְתִּי:  ת  וְאִתְּעַר יַעֲקֹב מִשִּׁנְתֵּהּ וַאֲמַר בְּקוּשְׁטָּא (אִית) יְקָרָא דַיְיָ שָׁרֵי בְּאַתְרָא הָדֵין וַאֲנָא לָא הֲוֵיתִי יְדָע:

{יז} וַיִּירָא וַיֹּאמַר מַה נּוֹרָא הַמָּקוֹם הַזֶּה אֵין זֶה כִּי אִם בֵּית אֱלֹהִים וְזֶה שַׁעַר הַשָּׁמָיִם:  מ  וַיִּירָא וַיֹּאמַר מַה נּוֹרָא הַמָּקוֹם הַזֶּה אֵין זֶה כִּי אִם בֵּית אֱלֹהִים וְזֶה שַׁעַר הַשָּׁמָיִם:  ת  וּדְחִיל וַאֲמַר מָה דְחִילוּ אַתְרָא הָדֵין לֵית דֵּין אֲתַר הֶדְיוֹט אֶלָהֵין אֲתַר דְּרַעֲוָא בֵהּ מִן קֳדָם יְיָ וְדֵין תְּרַע קֳבֵל שְׁמַיָּא:

{יח} וַיַּשְׁכֵּם יַעֲקֹב בַּבֹּקֶר וַיִּקַּח אֶת הָאֶבֶן אֲשֶׁר שָׂם מְרַאֲשֹׁתָיו וַיָּשֶׂם אֹתָהּ מַצֵּבָה וַיִּצֹק שֶׁמֶן עַל רֹאשָׁהּ:  מ  וַיַּשְׁכֵּם יַעֲקֹב בַּבֹּקֶר וַיִּקַּח אֶת הָאֶבֶן אֲשֶׁר שָׂם מְרַאֲשֹׁתָיו וַיָּשֶׂם אֹתָהּ מַצֵּבָה וַיִּצֹק שֶׁמֶן עַל רֹאשָׁהּ:  ת  וְאַקְדֵים יַעֲקֹב בְּצַפְרָא וּנְסִיב יָת אַבְנָא דִי שַׁוִּי אִסָדוֹהִי וְשַׁוִּי יָתַהּ קָמָא וַאֲרִיק מִשְׁחָא עַל רֵישֵׁהּ:

{יט} וַיִּקְרָא אֶת שֵׁם הַמָּקוֹם הַהוּא בֵּית אֵל וְאוּלָם לוּז שֵׁם הָעִיר לָרִאשֹׁנָה:  מ  וַיִּקְרָא אֶת שֵׁם הַמָּקוֹם הַהוּא בֵּית אֵל וְאוּלָם לוּז שֵׁם הָעִיר לָרִאשֹׁנָה:  ת  וּקְרָא יָת שְׁמָא דְאַתְרָא הַהוּא בֵּית אֵל וּבְרַם לוּז שְׁמָא דְקַרְתָּא בְּקַדְמֵיתָא:

{כ} וַיִּדַּר יַעֲקֹב נֶדֶר לֵאמֹר אִם יִהְיֶה אֱלֹהִים עִמָּדִי וּשְׁמָרַנִי בַּדֶּרֶךְ הַזֶּה אֲשֶׁר אָנֹכִי הוֹלֵךְ וְנָתַן לִי לֶחֶם לֶאֱכֹל וּבֶגֶד לִלְבֹּשׁ:  מ  וַיִּדַּר יַעֲקֹב נֶדֶר לֵאמֹר אִם יִהְיֶה אֱלֹהִים עִמָּדִי וּשְׁמָרַנִי בַּדֶּרֶךְ הַזֶּה אֲשֶׁר אָנֹכִי הוֹלֵךְ וְנָתַן לִי לֶחֶם לֶאֱכֹל וּבֶגֶד לִלְבֹּשׁ:  ת  וְקַיִּים יַעֲקֹב קְיָם לְמֵימָר אִם יְהֵא מֵימְרָא דַּיְיָ בְּסַעְדִי וְיִטְרִינַנִּי בְּאָרְחָא הָדֵין דִּי אֲנָא אָזֵל וְיִתֶּן לִי לַחֲמָא (נ''י לְחֵם) לְמֵיכַל וּכְסוּ לְמִלְבָּשׁ:

{כא} וְשַׁבְתִּי בְשָׁלוֹם אֶל בֵּית אָבִי וְהָיָה יְהוָה לִי לֵאלֹהִים:  מ  וְשַׁבְתִּי בְשָׁלוֹם אֶל בֵּית אָבִי וְהָיָה יְהוָה לִי לֵאלֹהִים:  ת  וְאֵתוּב בִּשְׁלַם לְבֵית אַבָּא וִיהֵא מֵימְרָא דַיְיָ לִי לֵאלָהָא:

{כב} וְהָאֶבֶן הַזֹּאת אֲשֶׁר שַׂמְתִּי מַצֵּבָה יִהְיֶה בֵּית אֱלֹהִים וְכֹל אֲשֶׁר תִּתֶּן לִי עַשֵּׂר אֲעַשְּׂרֶנּוּ לָךְ:  מ  וְהָאֶבֶן הַזֹּאת אֲשֶׁר שַׂמְתִּי מַצֵּבָה יִהְיֶה בֵּית אֱלֹהִים וְכֹל אֲשֶׁר תִּתֶּן לִי עַשֵּׂר אֲעַשְּׂרֶנּוּ לָךְ:  ת  וְאַבְנָא הָדָא דִי שַׁוֵּיתִי קָמָא תְּהֵי דִי אֵהֵי פָלַח עֲלַהּ קֳדָם יְיָ וְכֹל דִּי תִתֶּן לִי חַד מִן עַשְׂרָא אַפְרְשִׁנֵּהּ קֳדָמָךְ:




בראשית פרק-כט

{א}  שני  וַיִּשָּׂא יַעֲקֹב רַגְלָיו וַיֵּלֶךְ אַרְצָה בְנֵי קֶדֶם:  מ  וַיִּשָּׂא יַעֲקֹב רַגְלָיו וַיֵּלֶךְ אַרְצָה בְנֵי קֶדֶם:  ת  וּנְטַל יַעֲקֹב רַגְלוֹהִי (נ''י רִיגְלוֹהִי) וַאֲזַל לַאֲרַע בְּנֵי מַדִינְחָא:

{ב} וַיַּרְא וְהִנֵּה בְאֵר בַּשָּׂדֶה וְהִנֵּה שָׁם שְׁלֹשָׁה עֶדְרֵי צֹאן רֹבְצִים עָלֶיהָ כִּי מִן הַבְּאֵר הַהִוא יַשְׁקוּ הָעֲדָרִים וְהָאֶבֶן גְּדֹלָה עַל פִּי הַבְּאֵר:  מ  וַיַּרְא וְהִנֵּה בְאֵר בַּשָּׂדֶה וְהִנֵּה שָׁם שְׁלֹשָׁה עֶדְרֵי צֹאן רֹבְצִים עָלֶיהָ כִּי מִן הַבְּאֵר הַהִוא יַשְׁקוּ הָעֲדָרִים וְהָאֶבֶן גְּדֹלָה עַל פִּי הַבְּאֵר:  ת  וַחֲזָא וְהָא בֵירָא בְחַקְלָא וְהָא תַמָּן תְּלָתָא עֶדְרִין דְּעָן רְבֵעִין עֲלַהּ אֲרֵי מִן בֵּירָא הַהִיא מַשְׁקָן עֲדָרַיָּא וְאַבְנָא רַבְּתָא עַל פּוּמָא דְבֵירָא:

{ג} וְנֶאֶסְפוּ שָׁמָּה כָל הָעֲדָרִים וְגָלֲלוּ אֶת הָאֶבֶן מֵעַל פִּי הַבְּאֵר וְהִשְׁקוּ אֶת הַצֹּאן וְהֵשִׁיבוּ אֶת הָאֶבֶן עַל פִּי הַבְּאֵר לִמְקֹמָהּ:  מ  וְנֶאֶסְפוּ שָׁמָּה כָל הָעֲדָרִים וְגָלֲלוּ אֶת הָאֶבֶן מֵעַל פִּי הַבְּאֵר וְהִשְׁקוּ אֶת הַצֹּאן וְהֵשִׁיבוּ אֶת הָאֶבֶן עַל פִּי הַבְּאֵר לִמְקֹמָהּ:  ת  וּמִתְכַּנְשִׁין לְתַמָּן כָּל עֲדָרַיָּא וּמְגַנְדְּרִין יָת אַבְנָא מֵעַל פּוּמָא דְבֵירָא וּמַשְׁקָן יָת עָנָא וּמְתִיבִין יָת אַבְנָא עַל פּוּמָא דְבֵירָא לְאַתְרַהּ:

{ד} וַיֹּאמֶר לָהֶם יַעֲקֹב אַחַי מֵאַיִן אַתֶּם וַיֹּאמְרוּ מֵחָרָן אֲנָחְנוּ:  מ  וַיֹּאמֶר לָהֶם יַעֲקֹב אַחַי מֵאַיִן אַתֶּם וַיֹּאמְרוּ מֵחָרָן אֲנָחְנוּ:  ת  וַאֲמַר לְהוֹן יַעֲקֹב אַחַי מְנָן אַתּוּן וַאֲמָרוּ מֵחָרָן אֲנָחְנָא:

{ה} וַיֹּאמֶר לָהֶם הַיְדַעְתֶּם אֶת לָבָן בֶּן נָחוֹר וַיֹּאמְרוּ יָדָעְנוּ:  מ  וַיֹּאמֶר לָהֶם הַיְדַעְתֶּם אֶת לָבָן בֶּן נָחוֹר וַיֹּאמְרוּ יָדָעְנוּ:  ת  וַאֲמַר לְהוֹן הַיְדַעְתּוּן יָת לָבָן בַּר נָחוֹר וַאֲמָרוּ יְדַעְנָן:

{ו} וַיֹּאמֶר לָהֶם הֲשָׁלוֹם לוֹ וַיֹּאמְרוּ שָׁלוֹם וְהִנֵּה רָחֵל בִּתּוֹ בָּאָה עִם הַצֹּאן:  מ  וַיֹּאמֶר לָהֶם הֲשָׁלוֹם לוֹ וַיֹּאמְרוּ שָׁלוֹם וְהִנֵּה רָחֵל בִּתּוֹ בָּאָה עִם הַצֹּאן:  ת  וַאֲמַר לְהוֹן הַשְׁלַם לֵהּ וַאֲמָרוּ שְׁלַם וְהָא רָחֵל בְּרַתֵּהּ אַתְיָא עִם עָנָא:

{ז} וַיֹּאמֶר הֵן עוֹד הַיּוֹם גָּדוֹל לֹא עֵת הֵאָסֵף הַמִּקְנֶה הַשְׁקוּ הַצֹּאן וּלְכוּ רְעוּ:  מ  וַיֹּאמֶר הֵן עוֹד הַיּוֹם גָּדוֹל לֹא עֵת הֵאָסֵף הַמִּקְנֶה הַשְׁקוּ הַצֹּאן וּלְכוּ רְעוּ:  ת  וַאֲמַר הָא עוֹד יוֹמָא סַגִי לָא עִדַן לְמִכְנַשׁ בְּעִיר אַשְׁקוּ עָנָא וְאִזִילוּ רְעוּ:

{ח} וַיֹּאמְרוּ לֹא נוּכַל עַד אֲשֶׁר יֵאָסְפוּ כָּל הָעֲדָרִים וְגָלֲלוּ אֶת הָאֶבֶן מֵעַל פִּי הַבְּאֵר וְהִשְׁקִינוּ הַצֹּאן:  מ  וַיֹּאמְרוּ לֹא נוּכַל עַד אֲשֶׁר יֵאָסְפוּ כָּל הָעֲדָרִים וְגָלֲלוּ אֶת הָאֶבֶן מֵעַל פִּי הַבְּאֵר וְהִשְׁקִינוּ הַצֹּאן:  ת  וַאֲמָרוּ לָא נִכּוּל עַד דִי יִתְכַּנְשִׁין כָּל עֲדָרַיָּא וִיגַנְדְרוּן יָת אַבְנָא מֵעַל פוּמָא דְבֵירָא וְנַשְׁקֵי עָנָא:

{ט} עוֹדֶנּוּ מְדַבֵּר עִמָּם וְרָחֵל בָּאָה עִם הַצֹּאן אֲשֶׁר לְאָבִיהָ כִּי רֹעָה הִוא:  מ  עוֹדֶנּוּ מְדַבֵּר עִמָּם וְרָחֵל בָּאָה עִם הַצֹּאן אֲשֶׁר לְאָבִיהָ כִּי רֹעָה הִוא:  ת  עַד דְהוּא מְמַלֵל עִמְהוֹן וְרָחֵל אָתַת עִם עָנָא דִי לַאֲבוּהָ אֲרֵי רָעִיתָא הִיא:

{י} וַיְהִי כַּאֲשֶׁר רָאָה יַעֲקֹב אֶת רָחֵל בַּת לָבָן אֲחִי אִמּוֹ וְאֶת צֹאן לָבָן אֲחִי אִמּוֹ וַיִּגַּשׁ יַעֲקֹב וַיָּגֶל אֶת הָאֶבֶן מֵעַל פִּי הַבְּאֵר וַיַּשְׁקְ אֶת צֹאן לָבָן אֲחִי אִמּוֹ:  מ  וַיְהִי כַּאֲשֶׁר רָאָה יַעֲקֹב אֶת רָחֵל בַּת לָבָן אֲחִי אִמּוֹ וְאֶת צֹאן לָבָן אֲחִי אִמּוֹ וַיִּגַּשׁ יַעֲקֹב וַיָּגֶל אֶת הָאֶבֶן מֵעַל פִּי הַבְּאֵר וַיַּשְׁקְ אֶת צֹאן לָבָן אֲחִי אִמּוֹ:  ת  וַהֲוָה כַּד חֲזָא יַעֲקֹב יָת רָחֵל בַּת לָבָן אֲחוּהָא דְאִמֵהּ וְיָת עָנָא דְלָבָן אֲחוּהָא דְאִמֵהּ וּקְרֵב יַעֲקֹב וּגַנְדַר יָת אַבְנָא מֵעַל פּוּמָא דְבֵירָא וְאַשְׁקֵי יָת עָנָא דְלָבָן אֲחוּהָא דְאִמֵהּ:

{יא} וַיִּשַּׁק יַעֲקֹב לְרָחֵל וַיִּשָּׂא אֶת קֹלוֹ וַיֵּבְךְּ:  מ  וַיִּשַּׁק יַעֲקֹב לְרָחֵל וַיִּשָּׂא אֶת קֹלוֹ וַיֵּבְךְּ:  ת  וּנְשַׁק יַעֲקֹב לְרָחֵל וַאֲרֵם יָת קָלֵהּ וּבְכָא:

{יב} וַיַּגֵּד יַעֲקֹב לְרָחֵל כִּי אֲחִי אָבִיהָ הוּא וְכִי בֶן רִבְקָה הוּא וַתָּרָץ וַתַּגֵּד לְאָבִיהָ:  מ  וַיַּגֵּד יַעֲקֹב לְרָחֵל כִּי אֲחִי אָבִיהָ הוּא וְכִי בֶן רִבְקָה הוּא וַתָּרָץ וַתַּגֵּד לְאָבִיהָ:  ת  וְחַוִּי יַעֲקֹב לְרָחֵל אֲרֵי בַר אֲחַת אָבוּהָא הוּא וַאֲרֵי בַר רִבְקָה הוּא וּרְהָטַת וְחַוִּיאַת לְאָבוּהָא:

{יג} וַיְהִי כִשְׁמֹעַ לָבָן אֶת שֵׁמַע יַעֲקֹב בֶּן אֲחֹתוֹ וַיָּרָץ לִקְרָאתוֹ וַיְחַבֶּק לוֹ וַיְנַשֶּׁק לוֹ וַיְבִיאֵהוּ אֶל בֵּיתוֹ וַיְסַפֵּר לְלָבָן אֵת כָּל הַדְּבָרִים הָאֵלֶּה:  מ  וַיְהִי כִשְׁמֹעַ לָבָן אֶת שֵׁמַע יַעֲקֹב בֶּן אֲחֹתוֹ וַיָּרָץ לִקְרָאתוֹ וַיְחַבֶּק לוֹ וַיְנַשֶּׁק לוֹ וַיְבִיאֵהוּ אֶל בֵּיתוֹ וַיְסַפֵּר לְלָבָן אֵת כָּל הַדְּבָרִים הָאֵלֶּה:  ת  וַהֲוָה כַּד שְׁמַע לָבָן יָת שֵׁמַע יַעֲקֹב בַּר אֲחָתֵהּ וּרְהַט לְקַדָמוּתֵהּ וְגַפֵּף לֵהּ וְנַשִׁיק לֵהּ וְאַעֲלֵהּ לְבֵיתֵהּ וְאִשְׁתָּעִי לְלָבָן יָת כָּל פִּתְגָמַיָּא הָאִלֵּין:

{יד} וַיֹּאמֶר לוֹ לָבָן אַךְ עַצְמִי וּבְשָׂרִי אָתָּה וַיֵּשֶׁב עִמּוֹ חֹדֶשׁ יָמִים:  מ  וַיֹּאמֶר לוֹ לָבָן אַךְ עַצְמִי וּבְשָׂרִי אָתָּה וַיֵּשֶׁב עִמּוֹ חֹדֶשׁ יָמִים:  ת  וַאֲמַר לֵהּ לָבָן בְּרַם קָרִיבִי וּבִשְׂרִי אָתְּ וִיתֵיב עִמֵּהּ יְרַח יוֹמִין:

{טו} וַיֹּאמֶר לָבָן לְיַעֲקֹב הֲכִי אָחִי אַתָּה וַעֲבַדְתַּנִי חִנָּם הַגִּידָה לִּי מַה מַּשְׂכֻּרְתֶּךָ:  מ  וַיֹּאמֶר לָבָן לְיַעֲקֹב הֲכִי אָחִי אַתָּה וַעֲבַדְתַּנִי חִנָּם הַגִּידָה לִּי מַה מַּשְׂכֻּרְתֶּךָ:  ת  וַאֲמַר לָבָן לְיַעֲקֹב הֲמִדְּאָחִי אַתְּ וְתִפְלְחִנַּנִּי מַגָּן חַוִּי לִי מָה אַגְרָךְ:

{טז} וּלְלָבָן שְׁתֵּי בָנוֹת שֵׁם הַגְּדֹלָה לֵאָה וְשֵׁם הַקְּטַנָּה רָחֵל:  מ  וּלְלָבָן שְׁתֵּי בָנוֹת שֵׁם הַגְּדֹלָה לֵאָה וְשֵׁם הַקְּטַנָּה רָחֵל:  ת  וּלְלָבָן תַּרְתֵּין בְּנָן שׁוּם רַבְּתָא לֵאָה וְשׁוּם זְעֶרְתָּא רָחֵל:

{יז} וְעֵינֵי לֵאָה רַכּוֹת וְרָחֵל הָיְתָה יְפַת תֹּאַר וִיפַת מַרְאֶה:  מ  וְעֵינֵי לֵאָה רַכּוֹת וְרָחֵל הָיְתָה יְפַת תֹּאַר וִיפַת מַרְאֶה:  ת  וְעֵינֵי לֵאָה יָאֲיָן וְרָחֵל הֲוַת שַׁפִּירָא בְרֵיוָא וְיָאֲיָא בְחֶזְוָא:

{יח}  שלישי  וַיֶּאֱהַב יַעֲקֹב אֶת רָחֵל וַיֹּאמֶר אֶעֱבָדְךָ שֶׁבַע שָׁנִים בְּרָחֵל בִּתְּךָ הַקְּטַנָּה:  מ  וַיֶּאֱהַב יַעֲקֹב אֶת רָחֵל וַיֹּאמֶר אֶעֱבָדְךָ שֶׁבַע שָׁנִים בְּרָחֵל בִּתְּךָ הַקְּטַנָּה:  ת  וּרְחֵם יַעֲקֹב יָת רָחֵל וַאֲמַר אֶפְלְחִנָּךְ שְׁבַע שְׁנִין בְּרָחֵל בְּרַתָּךְ זְעֶרְתָּא:

{יט} וַיֹּאמֶר לָבָן טוֹב תִּתִּי אֹתָהּ לָךְ מִתִּתִּי אֹתָהּ לְאִישׁ אַחֵר שְׁבָה עִמָּדִי:  מ  וַיֹּאמֶר לָבָן טוֹב תִּתִּי אֹתָהּ לָךְ מִתִּתִּי אֹתָהּ לְאִישׁ אַחֵר שְׁבָה עִמָּדִי:  ת  וַאֲמַר לָבָן טַב דְּאֶתֵּן יָתַהּ לָךְ מִדְּאֶתֵּן יָתַהּ לִגְבַר אָחֳרָן תִּיב עִמִּי:

{כ} וַיַּעֲבֹד יַעֲקֹב בְּרָחֵל שֶׁבַע שָׁנִים וַיִּהְיוּ בְעֵינָיו כְּיָמִים אֲחָדִים בְּאַהֲבָתוֹ אֹתָהּ:  מ  וַיַּעֲבֹד יַעֲקֹב בְּרָחֵל שֶׁבַע שָׁנִים וַיִּהְיוּ בְעֵינָיו כְּיָמִים אֲחָדִים בְּאַהֲבָתוֹ אֹתָהּ:  ת  וּפְלַח יַעֲקֹב בְּרָחֵל שְׁבַע שְׁנִין וַהֲווֹ בְעֵינוֹהִי כְּיוֹמִין זְעִירִין כַּד רְחֵים יָתַהּ:

{כא} וַיֹּאמֶר יַעֲקֹב אֶל לָבָן הָבָה אֶת אִשְׁתִּי כִּי מָלְאוּ יָמָי וְאָבוֹאָה אֵלֶיהָ:  מ  וַיֹּאמֶר יַעֲקֹב אֶל לָבָן הָבָה אֶת אִשְׁתִּי כִּי מָלְאוּ יָמָי וְאָבוֹאָה אֵלֶיהָ:  ת  וַאֲמַר יַעֲקֹב לְלָבָן הַב יָת אִתְּתִי אֲרֵי אַשְׁלֵמִית יוֹמֵי פָלְחָנִי וְאֵעוֹל לְוָתַהּ:

{כב} וַיֶּאֱסֹף לָבָן אֶת כָּל אַנְשֵׁי הַמָּקוֹם וַיַּעַשׂ מִשְׁתֶּה:  מ  וַיֶּאֱסֹף לָבָן אֶת כָּל אַנְשֵׁי הַמָּקוֹם וַיַּעַשׂ מִשְׁתֶּה:  ת  וּכְנַשׁ לָבָן יָת כָּל אֲנָשֵׁי אַתְרָא וַעֲבַד מִשְׁתְּיָא:

{כג} וַיְהִי בָעֶרֶב וַיִּקַּח אֶת לֵאָה בִתּוֹ וַיָּבֵא אֹתָהּ אֵלָיו וַיָּבֹא אֵלֶיהָ:  מ  וַיְהִי בָעֶרֶב וַיִּקַּח אֶת לֵאָה בִתּוֹ וַיָּבֵא אֹתָהּ אֵלָיו וַיָּבֹא אֵלֶיהָ:  ת  וַהֲוָה בְרַמְשָׁא וּדְבַר יָת לֵאָה בְרַתֵּהּ וְאָעֵיל יָתַהּ לְוָתֵהּ וְעַל לְוָתַהּ:

{כד} וַיִּתֵּן לָבָן לָהּ אֶת זִלְפָּה שִׁפְחָתוֹ לְלֵאָה בִתּוֹ שִׁפְחָה:  מ  וַיִּתֵּן לָבָן לָהּ אֶת זִלְפָּה שִׁפְחָתוֹ לְלֵאָה בִתּוֹ שִׁפְחָה:  ת  וִיהַב לָבָן לַהּ יָת זִלְפָּה אַמְתֵהּ לְלֵאָה בְרַתֵּהּ לְאַמְהוּ:

{כה} וַיְהִי בַבֹּקֶר וְהִנֵּה הִוא לֵאָה וַיֹּאמֶר אֶל לָבָן מַה זֹּאת עָשִׂיתָ לִּי הֲלֹא בְרָחֵל עָבַדְתִּי עִמָּךְ וְלָמָּה רִמִּיתָנִי:  מ  וַיְהִי בַבֹּקֶר וְהִנֵּה הִוא לֵאָה וַיֹּאמֶר אֶל לָבָן מַה זֹּאת עָשִׂיתָ לִּי הֲלֹא בְרָחֵל עָבַדְתִּי עִמָּךְ וְלָמָּה רִמִּיתָנִי:  ת  וַהֲוָה בְצַפְרָא וְהָא הִיא לֵאָה וַאֲמַר לְלָבָן מָה דָּא עֲבַדְתָּ לִּי הֲלָא בְרָחֵל פְּלָחִית עִמָּךְ וּלְמָא שַׁקַּרְתָּ בִּי:

{כו} וַיֹּאמֶר לָבָן לֹא יֵעָשֶׂה כֵן בִּמְקוֹמֵנוּ לָתֵת הַצְּעִירָה לִפְנֵי הַבְּכִירָה:  מ  וַיֹּאמֶר לָבָן לֹא יֵעָשֶׂה כֵן בִּמְקוֹמֵנוּ לָתֵת הַצְּעִירָה לִפְנֵי הַבְּכִירָה:  ת  וַאֲמַר לָבָן לָא יִתְעֲבֵד כְּדֵין בְּאַתְרָנָא לְמִתַּן זְעֶרְתָּא קֳדָם רַבְּתָא:

{כז} מַלֵּא שְׁבֻעַ זֹאת וְנִתְּנָה לְךָ גַּם אֶת זֹאת בַּעֲבֹדָה אֲשֶׁר תַּעֲבֹד עִמָּדִי עוֹד שֶׁבַע שָׁנִים אֲחֵרוֹת:  מ  מַלֵּא שְׁבֻעַ זֹאת וְנִתְּנָה לְךָ גַּם אֶת זֹאת בַּעֲבֹדָה אֲשֶׁר תַּעֲבֹד עִמָּדִי עוֹד שֶׁבַע שָׁנִים אֲחֵרוֹת:  ת  אַשְׁלִים שְׁבוּעֲתָא דְדָא וְנִתֵּן לָךְ אַף יָת דָּא בְּפָלְחָנָא דִּי תִפְלַח עִמִּי עוֹד שְׁבַע שְׁנִין אָחֳרָנִין:

{כח} וַיַּעַשׂ יַעֲקֹב כֵּן וַיְמַלֵּא שְׁבֻעַ זֹאת וַיִּתֶּן לוֹ אֶת רָחֵל בִּתּוֹ לוֹ לְאִשָּׁה:  מ  וַיַּעַשׂ יַעֲקֹב כֵּן וַיְמַלֵּא שְׁבֻעַ זֹאת וַיִּתֶּן לוֹ אֶת רָחֵל בִּתּוֹ לוֹ לְאִשָּׁה:  ת  וַעֲבַד יַעֲקֹב כֵּן וְאַשְׁלֵים שְׁבוּעֲתָא דְדָא וִיהַב לֵהּ יָת רָחֵל בְּרַתֵּהּ לֵהּ לְאִנְתּוּ:

{כט} וַיִּתֵּן לָבָן לְרָחֵל בִּתּוֹ אֶת בִּלְהָה שִׁפְחָתוֹ לָהּ לְשִׁפְחָה:  מ  וַיִּתֵּן לָבָן לְרָחֵל בִּתּוֹ אֶת בִּלְהָה שִׁפְחָתוֹ לָהּ לְשִׁפְחָה:  ת  וִיהַב לָבָן לְרָחֵל בְּרַתֵּהּ יָת בִּלְהָה אַמְתֵהּ לַהּ לְאַמְהוּ:

{ל} וַיָּבֹא גַּם אֶל רָחֵל וַיֶּאֱהַב גַּם אֶת רָחֵל מִלֵּאָה וַיַּעֲבֹד עִמּוֹ עוֹד שֶׁבַע שָׁנִים אֲחֵרוֹת:  מ  וַיָּבֹא גַּם אֶל רָחֵל וַיֶּאֱהַב גַּם אֶת רָחֵל מִלֵּאָה וַיַּעֲבֹד עִמּוֹ עוֹד שֶׁבַע שָׁנִים אֲחֵרוֹת:  ת  וְעַל אַף לְּוַת רָחֵל וּרְחֵים אַף יָת רָחֵל מִלֵּאָה וּפְלַח עִמֵּהּ עוֹד שְׁבַע שְׁנִין אָחֳרָנִין:

{לא} וַיַּרְא יְהוָה כִּי שְׂנוּאָה לֵאָה וַיִּפְתַּח אֶת רַחְמָהּ וְרָחֵל עֲקָרָה:  מ  וַיַּרְא יְהוָה כִּי שְׂנוּאָה לֵאָה וַיִּפְתַּח אֶת רַחְמָהּ וְרָחֵל עֲקָרָה:  ת  וַחֲזָא יְיָ אֲרֵי שְׂנִיאֲתָא לֵאָה וִיהַב לַהּ עִדּוּי וְרָחֵל עַקְרָא:

{לב} וַתַּהַר לֵאָה וַתֵּלֶד בֵּן וַתִּקְרָא שְׁמוֹ רְאוּבֵן כִּי אָמְרָה כִּי רָאָה יְהוָה בְּעָנְיִי כִּי עַתָּה יֶאֱהָבַנִי אִישִׁי:  מ  וַתַּהַר לֵאָה וַתֵּלֶד בֵּן וַתִּקְרָא שְׁמוֹ רְאוּבֵן כִּי אָמְרָה כִּי רָאָה יְהוָה בְּעָנְיִי כִּי עַתָּה יֶאֱהָבַנִי אִישִׁי:  ת  וְעַדִּיאַת לֵּאָה וִילִידַת בָּר וּקְרַת שְׁמֵהּ רְאוּבֵן אֲרֵי אֲמֶרֶת אֲרֵי גְלִי קֳדָם יְיָ עֻלְבּוֹנִי אֲרֵי כְעַן יְרַחֲמִנַנִי בַעֲלִי:

{לג} וַתַּהַר עוֹד וַתֵּלֶד בֵּן וַתֹּאמֶר כִּי שָׁמַע יְהוָה כִּי שְׂנוּאָה אָנֹכִי וַיִּתֶּן לִי גַּם אֶת זֶה וַתִּקְרָא שְׁמוֹ שִׁמְעוֹן:  מ  וַתַּהַר עוֹד וַתֵּלֶד בֵּן וַתֹּאמֶר כִּי שָׁמַע יְהוָה כִּי שְׂנוּאָה אָנֹכִי וַיִּתֶּן לִי גַּם אֶת זֶה וַתִּקְרָא שְׁמוֹ שִׁמְעוֹן:  ת  וְעַדִּיאַת עוֹד וִילִידַת בָּר וַאֲמֶרֶת אֲרֵי שְׁמִיעַ קֳדָם יְיָ אֲרֵי שְׂנִיאֲתָא אֲנָא וִיהַב לִי אַף יָת דֵּין וּקְרַת שְׁמֵהּ שִׁמְעוֹן:

{לד} וַתַּהַר עוֹד וַתֵּלֶד בֵּן וַתֹּאמֶר עַתָּה הַפַּעַם יִלָּוֶה אִישִׁי אֵלַי כִּי יָלַדְתִּי לוֹ שְׁלֹשָׁה בָנִים עַל כֵּן קָרָא שְׁמוֹ לֵוִי:  מ  וַתַּהַר עוֹד וַתֵּלֶד בֵּן וַתֹּאמֶר עַתָּה הַפַּעַם יִלָּוֶה אִישִׁי אֵלַי כִּי יָלַדְתִּי לוֹ שְׁלֹשָׁה בָנִים עַל כֵּן קָרָא שְׁמוֹ לֵוִי:  ת  וְעַדִּיאַת עוֹד וִילִידַת בָּר וַאֲמֶרֶת הָדָא זִמְנָא יִתְחַבַּר לִי בַעֲלִי אֲרֵי יְלֵידִית לֵהּ תְּלָתָא בְנִין עַל כֵּן קְרָא שְׁמֵהּ לֵוִי:

{לה} וַתַּהַר עוֹד וַתֵּלֶד בֵּן וַתֹּאמֶר הַפַּעַם אוֹדֶה אֶת יְהוָה עַל כֵּן קָרְאָה שְׁמוֹ יְהוּדָה וַתַּעֲמֹד מִלֶּדֶת:  מ  וַתַּהַר עוֹד וַתֵּלֶד בֵּן וַתֹּאמֶר הַפַּעַם אוֹדֶה אֶת יְהוָה עַל כֵּן קָרְאָה שְׁמוֹ יְהוּדָה וַתַּעֲמֹד מִלֶּדֶת:  ת  וְעַדִּיאַת עוֹד וִילִידַת בָּר וַאֲמֶרֶת הָדָא זִמְנָא אוֹדֶה קֳדָם יְיָ עַל כֵּן קְרַת שְׁמֵהּ יְהוּדָה וְקָמַת מִלְּמֵילַד:




בראשית פרק-ל

{א} וַתֵּרֶא רָחֵל כִּי לֹא יָלְדָה לְיַעֲקֹב וַתְּקַנֵּא רָחֵל בַּאֲחֹתָהּ וַתֹּאמֶר אֶל יַעֲקֹב הָבָה לִּי בָנִים וְאִם אַיִן מֵתָה אָנֹכִי:  מ  וַתֵּרֶא רָחֵל כִּי לֹא יָלְדָה לְיַעֲקֹב וַתְּקַנֵּא רָחֵל בַּאֲחֹתָהּ וַתֹּאמֶר אֶל יַעֲקֹב הָבָה לִּי בָנִים וְאִם אַיִן מֵתָה אָנֹכִי:  ת  וַחֲזַת רָחֵל אֲרֵי לָא יְלִידַת לְיַעֲקֹב וְקַנִּיאַת רָחֵל בַּאֲחָתַהּ וַאֲמֶרֶת לְיַעֲקֹב הַב לִי בְנִין וְאִם לָא מֵיתָא אֲנָא:

{ב} וַיִּחַר אַף יַעֲקֹב בְּרָחֵל וַיֹּאמֶר הֲתַחַת אֱלֹהִים אָנֹכִי אֲשֶׁר מָנַע מִמֵּךְ פְּרִי בָטֶן:  מ  וַיִּחַר אַף יַעֲקֹב בְּרָחֵל וַיֹּאמֶר הֲתַחַת אֱלֹהִים אָנֹכִי אֲשֶׁר מָנַע מִמֵּךְ פְּרִי בָטֶן:  ת  וּתְקֵיף רוּגְזָא דְיַעֲקֹב בְּרָחֵל וַאֲמַר הָא מִנִּי אַתְּ בַּעְיָא הֲלָא מִן קֳדָם יְיָ תִּבְעִין דִּי מְנַע מִנִּיךְ וַלְדָא דִּמְעִין:

{ג} וַתֹּאמֶר הִנֵּה אֲמָתִי בִלְהָה בֹּא אֵלֶיהָ וְתֵלֵד עַל בִּרְכַּי וְאִבָּנֶה גַם אָנֹכִי מִמֶּנָּה:  מ  וַתֹּאמֶר הִנֵּה אֲמָתִי בִלְהָה בֹּא אֵלֶיהָ וְתֵלֵד עַל בִּרְכַּי וְאִבָּנֶה גַם אָנֹכִי מִמֶּנָּה:  ת  וַאֲמֶרֶת הָא אַמְתִי בִלְהָה עוּל לְוָתַהּ וּתְלִיד וַאֲנָא אֱרַבֵּי וְאִתְבְּנֵי אַף אֲנָא מִנַּהּ:

{ד} וַתִּתֶּן לוֹ אֶת בִּלְהָה שִׁפְחָתָהּ לְאִשָּׁה וַיָּבֹא אֵלֶיהָ יַעֲקֹב:  מ  וַתִּתֶּן לוֹ אֶת בִּלְהָה שִׁפְחָתָהּ לְאִשָּׁה וַיָּבֹא אֵלֶיהָ יַעֲקֹב:  ת  וִיהָבַת לֵהּ יָת בִּלְהָה אַמְתַהּ לְאִנְתּוּ וְעַל לְוָתַהּ יַעֲקֹב:

{ה} וַתַּהַר בִּלְהָה וַתֵּלֶד לְיַעֲקֹב בֵּן:  מ  וַתַּהַר בִּלְהָה וַתֵּלֶד לְיַעֲקֹב בֵּן:  ת  וְעַדִּיאַת בִּלְהָה וִילִידַת לְיַעֲקֹב בַּר:

{ו} וַתֹּאמֶר רָחֵל דָּנַנִּי אֱלֹהִים וְגַם שָׁמַע בְּקֹלִי וַיִּתֶּן לִי בֵּן עַל כֵּן קָרְאָה שְׁמוֹ דָּן:  מ  וַתֹּאמֶר רָחֵל דָּנַנִּי אֱלֹהִים וְגַם שָׁמַע בְּקֹלִי וַיִּתֶּן לִי בֵּן עַל כֵּן קָרְאָה שְׁמוֹ דָּן:  ת  וַאֲמֶרֶת רָחֵל דְּנַנִּי יְיָ וְאַף קַבִּיל צְלוֹתִי וִיהַב לִי בַּר עַל כֵּן קְרַת שְׁמֵהּ דָּן:

{ז} וַתַּהַר עוֹד וַתֵּלֶד בִּלְהָה שִׁפְחַת רָחֵל בֵּן שֵׁנִי לְיַעֲקֹב:  מ  וַתַּהַר עוֹד וַתֵּלֶד בִּלְהָה שִׁפְחַת רָחֵל בֵּן שֵׁנִי לְיַעֲקֹב:  ת  וְעַדִּיאַת עוֹד וִילִידַת בִּלְהָה אַמְתָא דְרָחֵל בָּר תִּנְיָן לְיַעֲקֹב:

{ח} וַתֹּאמֶר רָחֵל נַפְתּוּלֵי אֱלֹהִים נִפְתַּלְתִּי עִם אֲחֹתִי גַּם יָכֹלְתִּי וַתִּקְרָא שְׁמוֹ נַפְתָּלִי:  מ  וַתֹּאמֶר רָחֵל נַפְתּוּלֵי אֱלֹהִים נִפְתַּלְתִּי עִם אֲחֹתִי גַּם יָכֹלְתִּי וַתִּקְרָא שְׁמוֹ נַפְתָּלִי:  ת  וַאֲמֶרֶת רָחֵל קַבִּיל יְיָ בָּעוּתִי בְּאִתְחַנָּנוּתִי בִּצְלוֹתִי חֲמֵידַת דִּיהֵי לִי וְלַד כַּאֲחָתִי אַף אִתְיְהִיב לִי וּקְרַת שְׁמֵהּ נַפְתָּלִי:

{ט} וַתֵּרֶא לֵאָה כִּי עָמְדָה מִלֶּדֶת וַתִּקַּח אֶת זִלְפָּה שִׁפְחָתָהּ וַתִּתֵּן אֹתָהּ לְיַעֲקֹב לְאִשָּׁה:  מ  וַתֵּרֶא לֵאָה כִּי עָמְדָה מִלֶּדֶת וַתִּקַּח אֶת זִלְפָּה שִׁפְחָתָהּ וַתִּתֵּן אֹתָהּ לְיַעֲקֹב לְאִשָּׁה:  ת  וַחֲזַת לֵּאָה אֲרֵי קָמַת מִלְּמֵילַד וּדְבָרַת יָת זִלְפָּה אַמְתַהּ וִיהָבַת יָתַהּ לְיַעֲקֹב לְאִנְתּוּ:

{י} וַתֵּלֶד זִלְפָּה שִׁפְחַת לֵאָה לְיַעֲקֹב בֵּן:  מ  וַתֵּלֶד זִלְפָּה שִׁפְחַת לֵאָה לְיַעֲקֹב בֵּן:  ת  וִילִידַת זִלְפָּה אַמְתָא דְלֵאָה לְיַעֲקֹב בָּר:

{יא} וַתֹּאמֶר לֵאָה (בגד) בָּא גָד וַתִּקְרָא אֶת שְׁמוֹ גָּד:  מ  וַתֹּאמֶר לֵאָה (בגד) בָּא גָד וַתִּקְרָא אֶת שְׁמוֹ גָּד:  ת  וַאֲמֶרֶת לֵאָה אֲתָא גָד וּקְרַת יָת שְׁמֵהּ גָּד:

{יב} וַתֵּלֶד זִלְפָּה שִׁפְחַת לֵאָה בֵּן שֵׁנִי לְיַעֲקֹב:  מ  וַתֵּלֶד זִלְפָּה שִׁפְחַת לֵאָה בֵּן שֵׁנִי לְיַעֲקֹב:  ת  וִילִידַת זִלְפָּה אַמְתָא דְלֵאָה בָּר תִּנְיָן לְיַעֲקֹב:

{יג} וַתֹּאמֶר לֵאָה בְּאָשְׁרִי כִּי אִשְּׁרוּנִי בָּנוֹת וַתִּקְרָא אֶת שְׁמוֹ אָשֵׁר:  מ  וַתֹּאמֶר לֵאָה בְּאָשְׁרִי כִּי אִשְּׁרוּנִי בָּנוֹת וַתִּקְרָא אֶת שְׁמוֹ אָשֵׁר:  ת  וַאֲמֶרֶת לֵאָה תֻּשְׁבַּחְתָּא הֲוַת לִי אֲרֵי בְכֵן יְשַׁבְּחֻנַּנִּי נְשַׁיָּא וּקְרַת יָת שְׁמֵהּ אָשֵׁר:

{יד}  רביעי  וַיֵּלֶךְ רְאוּבֵן בִּימֵי קְצִיר חִטִּים וַיִּמְצָא דוּדָאִים בַּשָּׂדֶה וַיָּבֵא אֹתָם אֶל לֵאָה אִמּוֹ וַתֹּאמֶר רָחֵל אֶל לֵאָה תְּנִי נָא לִי מִדּוּדָאֵי בְּנֵךְ:  מ  וַיֵּלֶךְ רְאוּבֵן בִּימֵי קְצִיר חִטִּים וַיִּמְצָא דוּדָאִים בַּשָּׂדֶה וַיָּבֵא אֹתָם אֶל לֵאָה אִמּוֹ וַתֹּאמֶר רָחֵל אֶל לֵאָה תְּנִי נָא לִי מִדּוּדָאֵי בְּנֵךְ:  ת  וַאֲזַל רְאוּבֵן בְּיוֹמֵי חֲצַד חִטִּין וְאַשְׁכַּח יַבְרוּחִין בְּחַקְלָא וְאַיְתִי יָתְהוֹן לְלֵאָה אִמֵּהּ וַאֲמֶרֶת רָחֵל לְלֵאָה הָבִי כְעַן לִי מִיַּבְרוּחֵי דִּבְרִיךְ:

{טו} וַתֹּאמֶר לָהּ הַמְעַט קַחְתֵּךְ אֶת אִישִׁי וְלָקַחַת גַּם אֶת דּוּדָאֵי בְּנִי וַתֹּאמֶר רָחֵל לָכֵן יִשְׁכַּב עִמָּךְ הַלַּיְלָה תַּחַת דּוּדָאֵי בְנֵךְ:  מ  וַתֹּאמֶר לָהּ הַמְעַט קַחְתֵּךְ אֶת אִישִׁי וְלָקַחַת גַּם אֶת דּוּדָאֵי בְּנִי וַתֹּאמֶר רָחֵל לָכֵן יִשְׁכַּב עִמָּךְ הַלַּיְלָה תַּחַת דּוּדָאֵי בְנֵךְ:  ת  וַאֲמֶרֶת לַהּ הַזְעֵיר דִּדְבַרְתְּ יָת בַּעְלִי וְתִסְּבִין אַף יָת יַבְרוּחֵי דִּבְרִי וַאֲמֶרֶת רָחֵל בְּכֵן יִשְׁכּוּב עִמָּךְ בְּלֵילְיָא חֳלַף יַבְרוּחֵי דִּבְרִיךְ:

{טז} וַיָּבֹא יַעֲקֹב מִן הַשָּׂדֶה בָּעֶרֶב וַתֵּצֵא לֵאָה לִקְרָאתוֹ וַתֹּאמֶר אֵלַי תָּבוֹא כִּי שָׂכֹר שְׂכַרְתִּיךָ בְּדוּדָאֵי בְּנִי וַיִּשְׁכַּב עִמָּהּ בַּלַּיְלָה הוּא:  מ  וַיָּבֹא יַעֲקֹב מִן הַשָּׂדֶה בָּעֶרֶב וַתֵּצֵא לֵאָה לִקְרָאתוֹ וַתֹּאמֶר אֵלַי תָּבוֹא כִּי שָׂכֹר שְׂכַרְתִּיךָ בְּדוּדָאֵי בְּנִי וַיִּשְׁכַּב עִמָּהּ בַּלַּיְלָה הוּא:  ת  וַאֲתָא (נ''י וְעַל) יַעֲקֹב מִן חַקְלָא בְּרַמְשָׁא וּנְפָקַת לֵאָה לְקַדָּמוּתֵהּ וַאֲמֶרֶת לְוָתִי תֵיעוֹל אֲרֵי מֵיגַר אֲגַרְתִּיךְ בְּיַבְרוּחֵי דִּבְרִי וּשְׁכִיב עִמַּהּ בְּלֵילְיָא הוּא:

{יז} וַיִּשְׁמַע אֱלֹהִים אֶל לֵאָה וַתַּהַר וַתֵּלֶד לְיַעֲקֹב בֵּן חֲמִישִׁי:  מ  וַיִּשְׁמַע אֱלֹהִים אֶל לֵאָה וַתַּהַר וַתֵּלֶד לְיַעֲקֹב בֵּן חֲמִישִׁי:  ת  וְקַבִּיל יְיָ צְלוֹתַהּ דְּלֵאָה וְעַדִּיאַת וִילִידַת לְיַעֲקֹב בַּר חֲמִישָׁאֵי:

{יח} וַתֹּאמֶר לֵאָה נָתַן אֱלֹהִים שְׂכָרִי אֲשֶׁר נָתַתִּי שִׁפְחָתִי לְאִישִׁי וַתִּקְרָא שְׁמוֹ יִשָּׂשכָר:  מ  וַתֹּאמֶר לֵאָה נָתַן אֱלֹהִים שְׂכָרִי אֲשֶׁר נָתַתִּי שִׁפְחָתִי לְאִישִׁי וַתִּקְרָא שְׁמוֹ יִשָּׂשכָר:  ת  וַאֲמֶרֶת לֵאָה יְהַב יְיָ אַגְרִי דִּי יְהָבִית אַמְתִי לְבַעְלִי וּקְרַת שְׁמֵהּ יִשָּׂשׂכָר:

{יט} וַתַּהַר עוֹד לֵאָה וַתֵּלֶד בֵּן שִׁשִּׁי לְּיַעֲקֹב:  מ  וַתַּהַר עוֹד לֵאָה וַתֵּלֶד בֵּן שִׁשִּׁי לְּיַעֲקֹב:  ת  וְעַדִּיאַת עוֹד לֵאָה וִילִידַת בַּר שְׁתִיתָאֵי לְיַעֲקֹב:

{כ} וַתֹּאמֶר לֵאָה זְבָדַנִי אֱלֹהִים אֹתִי זֵבֶד טוֹב הַפַּעַם יִזְבְּלֵנִי אִישִׁי כִּי יָלַדְתִּי לוֹ שִׁשָּׁה בָנִים וַתִּקְרָא אֶת שְׁמוֹ זְבֻלוּן:  מ  וַתֹּאמֶר לֵאָה זְבָדַנִי אֱלֹהִים אֹתִי זֵבֶד טוֹב הַפַּעַם יִזְבְּלֵנִי אִישִׁי כִּי יָלַדְתִּי לוֹ שִׁשָּׁה בָנִים וַתִּקְרָא אֶת שְׁמוֹ זְבֻלוּן:  ת  וַאֲמֶרֶת לֵאָה יְהַב יְיָ יָתֵהּ לִי חָלַק טַב הָדָא זִמְנָא יְהֵי מְדוֹרֵהּ דְּבַעְלִי לְוָתִי אֲרֵי יְלֵידִית לֵהּ שִׁתָּא בְנִין וּקְרַת יָת שְׁמֵהּ זְבֻלוּן:

{כא} וְאַחַר יָלְדָה בַּת וַתִּקְרָא אֶת שְׁמָהּ דִּינָה:  מ  וְאַחַר יָלְדָה בַּת וַתִּקְרָא אֶת שְׁמָהּ דִּינָה:  ת  וּבָתַר כֵּן יְלִידַת בְּרַתָּא וּקְרַת יָת שְׁמַהּ דִּינָה:

{כב} וַיִּזְכֹּר אֱלֹהִים אֶת רָחֵל וַיִּשְׁמַע אֵלֶיהָ אֱלֹהִים וַיִּפְתַּח אֶת רַחְמָהּ:  מ  וַיִּזְכֹּר אֱלֹהִים אֶת רָחֵל וַיִּשְׁמַע אֵלֶיהָ אֱלֹהִים וַיִּפְתַּח אֶת רַחְמָהּ:  ת  וְעַל דּוּכְרָנָא דְרָחֵל קֳדָם יְיָ וְקַבִּיל צְלוֹתַהּ יְיָ וִיהַב לַהּ עִדּוּי:

{כג} וַתַּהַר וַתֵּלֶד בֵּן וַתֹּאמֶר אָסַף אֱלֹהִים אֶת חֶרְפָּתִי:  מ  וַתַּהַר וַתֵּלֶד בֵּן וַתֹּאמֶר אָסַף אֱלֹהִים אֶת חֶרְפָּתִי:  ת  וְעַדִּיאַת וִילִידַת בָּר וַאֲמֶרֶת כְּנַשׁ יְיָ יָת חִסּוּדִי:

{כד} וַתִּקְרָא אֶת שְׁמוֹ יוֹסֵף לֵאמֹר יֹסֵף יְהוָה לִי בֵּן אַחֵר:  מ  וַתִּקְרָא אֶת שְׁמוֹ יוֹסֵף לֵאמֹר יֹסֵף יְהוָה לִי בֵּן אַחֵר:  ת  וּקְרַת יָת שְׁמֵהּ יוֹסֵף לְמֵימָר יוֹסֵף יְיָ לִי בַּר אָחֳרָן:

{כה} וַיְהִי כַּאֲשֶׁר יָלְדָה רָחֵל אֶת יוֹסֵף וַיֹּאמֶר יַעֲקֹב אֶל לָבָן שַׁלְּחֵנִי וְאֵלְכָה אֶל מְקוֹמִי וּלְאַרְצִי:  מ  וַיְהִי כַּאֲשֶׁר יָלְדָה רָחֵל אֶת יוֹסֵף וַיֹּאמֶר יַעֲקֹב אֶל לָבָן שַׁלְּחֵנִי וְאֵלְכָה אֶל מְקוֹמִי וּלְאַרְצִי:  ת  וַהֲוָה כַּד יְלִידַת רָחֵל יָת יוֹסֵף וַאֲמַר יַעֲקֹב לְלָבָן שַׁלְּחַנִּי וְאֵיהַךְ לְאַתְרִי וּלְאַרְעִי:

{כו} תְּנָה אֶת נָשַׁי וְאֶת יְלָדַי אֲשֶׁר עָבַדְתִּי אֹתְךָ בָּהֵן וְאֵלֵכָה כִּי אַתָּה יָדַעְתָּ אֶת עֲבֹדָתִי אֲשֶׁר עֲבַדְתִּיךָ:  מ  תְּנָה אֶת נָשַׁי וְאֶת יְלָדַי אֲשֶׁר עָבַדְתִּי אֹתְךָ בָּהֵן וְאֵלֵכָה כִּי אַתָּה יָדַעְתָּ אֶת עֲבֹדָתִי אֲשֶׁר עֲבַדְתִּיךָ:  ת  הַב יָת נְשַׁי וְיָת בְּנַי דִּי פְלָחִית יָתָךְ בְּהֵן וְאֵזִיל אֲרֵי אַתְּ יְדַעְתָּ יָת פָּלְחָנִי דִּי פְלָחִתָּךְ:

{כז} וַיֹּאמֶר אֵלָיו לָבָן אִם נָא מָצָאתִי חֵן בְּעֵינֶיךָ נִחַשְׁתִּי וַיְבָרְכֵנִי יְהוָה בִּגְלָלֶךָ:  מ  וַיֹּאמֶר אֵלָיו לָבָן אִם נָא מָצָאתִי חֵן בְּעֵינֶיךָ נִחַשְׁתִּי וַיְבָרְכֵנִי יְהוָה בִּגְלָלֶךָ:  ת  וַאֲמַר לֵהּ לָבָן אִם כְּעַן אַשְׁכָּחִית רַחֲמִין קֳדָמָךְ נַסֵּיתִי וּבָרְכַנִי יְיָ בְּדִילָךְ:

{כח}  חמישי  וַיֹּאמַר נָקְבָה שְׂכָרְךָ עָלַי וְאֶתֵּנָה:  מ  וַיֹּאמַר נָקְבָה שְׂכָרְךָ עָלַי וְאֶתֵּנָה:  ת  וַאֲמָר פָּרֵישׁ אַגְרָךְ עֲלַי וְאֶתֵּן:

{כט} וַיֹּאמֶר אֵלָיו אַתָּה יָדַעְתָּ אֵת אֲשֶׁר עֲבַדְתִּיךָ וְאֵת אֲשֶׁר הָיָה מִקְנְךָ אִתִּי:  מ  וַיֹּאמֶר אֵלָיו אַתָּה יָדַעְתָּ אֵת אֲשֶׁר עֲבַדְתִּיךָ וְאֵת אֲשֶׁר הָיָה מִקְנְךָ אִתִּי:  ת  וַאֲמַר לֵהּ אַתְּ יְדַעְתָּ יָת דִּי פְלָחִתָּךְ וְיָת דַּהֲוָה בְעִירָךְ עִמִּי:

{ל} כִּי מְעַט אֲשֶׁר הָיָה לְךָ לְפָנַי וַיִּפְרֹץ לָרֹב וַיְבָרֶךְ יְהוָה אֹתְךָ לְרַגְלִי וְעַתָּה מָתַי אֶעֱשֶׂה גַם אָנֹכִי לְבֵיתִי:  מ  כִּי מְעַט אֲשֶׁר הָיָה לְךָ לְפָנַי וַיִּפְרֹץ לָרֹב וַיְבָרֶךְ יְהוָה אֹתְךָ לְרַגְלִי וְעַתָּה מָתַי אֶעֱשֶׂה גַם אָנֹכִי לְבֵיתִי:  ת  אֲרֵי זְעֵיר דִּי הֲוָה לָךְ קֳדָמַי וּתְקֵיף לְמִסְגֵּי וּבְרִיךְ יְיָ יָתָךְ בְּדִילִי וּכְעַן אֵימָתַי אֶעְבֵּיד אַף אֲנָא לְבֵיתִי:

{לא} וַיֹּאמֶר מָה אֶתֶּן לָךְ וַיֹּאמֶר יַעֲקֹב לֹא תִתֶּן לִי מְאוּמָה אִם תַּעֲשֶׂה לִּי הַדָּבָר הַזֶּה אָשׁוּבָה אֶרְעֶה צֹאנְךָ אֶשְׁמֹר:  מ  וַיֹּאמֶר מָה אֶתֶּן לָךְ וַיֹּאמֶר יַעֲקֹב לֹא תִתֶּן לִי מְאוּמָה אִם תַּעֲשֶׂה לִּי הַדָּבָר הַזֶּה אָשׁוּבָה אֶרְעֶה צֹאנְךָ אֶשְׁמֹר:  ת  וַאֲמַר מָה אֶתֵּן לָךְ וַאֲמַר יַעֲקֹב לָא תִתֶּן לִי מִדַּעַם אִם תַּעְבֶּד לִי פִּתְגָּמָא הָדֵין אֱתוּב אֶרְעֵי עָנָךְ אֶטָּר:

{לב} אֶעֱבֹר בְּכָל צֹאנְךָ הַיּוֹם הָסֵר מִשָּׁם כָּל שֶׂה נָקֹד וְטָלוּא וְכָל שֶׂה חוּם בַּכְּשָׂבִים וְטָלוּא וְנָקֹד בָּעִזִּים וְהָיָה שְׂכָרִי:  מ  אֶעֱבֹר בְּכָל צֹאנְךָ הַיּוֹם הָסֵר מִשָּׁם כָּל שֶׂה נָקֹד וְטָלוּא וְכָל שֶׂה חוּם בַּכְּשָׂבִים וְטָלוּא וְנָקֹד בָּעִזִּים וְהָיָה שְׂכָרִי:  ת  אֶעְבַּר בְּכָל עָנָךְ יוֹמָא דֵּין הַעֲדֵי (נ''י אַעְדִּי) מִתַּמָּן כָּל אִמָּר נְמוֹר וּרְקוֹעַ וְכָל אִמָּר שְׁחוּם בְּאִמְרַיָא וּרְקוֹעַ וּנְמוֹר בְּעִזַיָא וִיהֵי אַגְרִי:

{לג} וְעָנְתָה בִּי צִדְקָתִי בְּיוֹם מָחָר כִּי תָבוֹא עַל שְׂכָרִי לְפָנֶיךָ כֹּל אֲשֶׁר אֵינֶנּוּ נָקֹד וְטָלוּא בָּעִזִּים וְחוּם בַּכְּשָׂבִים גָּנוּב הוּא אִתִּי:  מ  וְעָנְתָה בִּי צִדְקָתִי בְּיוֹם מָחָר כִּי תָבוֹא עַל שְׂכָרִי לְפָנֶיךָ כֹּל אֲשֶׁר אֵינֶנּוּ נָקֹד וְטָלוּא בָּעִזִּים וְחוּם בַּכְּשָׂבִים גָּנוּב הוּא אִתִּי:  ת  וְתַסְהֵד בִּי זָכוּתִי בְּיוֹם דִּמְחָר אֲרֵי תֵיעוֹל עַל אַגְרִי קֳדָמָךְ כֹּל דִּי לֵיתוֹהִי נְמוֹר וּרְקוֹעַ בְּעִזַּיָּא וּשְׁחוּם בְּאִמְרַיָא גְּנוּבָא הוּא עִמִּי:

{לד} וַיֹּאמֶר לָבָן הֵן לוּ יְהִי כִדְבָרֶךָ:  מ  וַיֹּאמֶר לָבָן הֵן לוּ יְהִי כִדְבָרֶךָ:  ת  וַאֲמַר לָבָן בְּרַם לְוַי יְהֵי כְפִתְגָּמָךְ:

{לה} וַיָּסַר בַּיּוֹם הַהוּא אֶת הַתְּיָשִׁים הָעֲקֻדִּים וְהַטְּלֻאִים וְאֵת כָּל הָעִזִּים הַנְּקֻדּוֹת וְהַטְּלֻאֹת כֹּל אֲשֶׁר לָבָן בּוֹ וְכָל חוּם בַּכְּשָׂבִים וַיִּתֵּן בְּיַד בָּנָיו:  מ  וַיָּסַר בַּיּוֹם הַהוּא אֶת הַתְּיָשִׁים הָעֲקֻדִּים וְהַטְּלֻאִים וְאֵת כָּל הָעִזִּים הַנְּקֻדּוֹת וְהַטְּלֻאֹת כֹּל אֲשֶׁר לָבָן בּוֹ וְכָל חוּם בַּכְּשָׂבִים וַיִּתֵּן בְּיַד בָּנָיו:  ת  וְאַעְדִּי בְיוֹמָא הַהוּא יָת תְּיָשַׁיָא רְגוֹלַיָא וּרְקוֹעַיָא וְיָת כָּל עִזַּיָא נְמוֹרְתָא וּרְקוֹעֲתָא כֹּל דִּי חִוָּר בֵּהּ וְכָל דִּשְׁחוּם בְּאִמְּרַיָּא וִיהַב בְּיַד בְּנוֹהִי:

{לו} וַיָּשֶׂם דֶּרֶךְ שְׁלֹשֶׁת יָמִים בֵּינוֹ וּבֵין יַעֲקֹב וְיַעֲקֹב רֹעֶה אֶת צֹאן לָבָן הַנּוֹתָרֹת:  מ  וַיָּשֶׂם דֶּרֶךְ שְׁלֹשֶׁת יָמִים בֵּינוֹ וּבֵין יַעֲקֹב וְיַעֲקֹב רֹעֶה אֶת צֹאן לָבָן הַנּוֹתָרֹת:  ת  וְשַׁוִּי מַהֲלַךְ תְּלָתָא יוֹמִין בֵּינוֹהִי וּבֵין יַעֲקֹב וְיַעֲקֹב רָעֵי יָת עָנָא דְלָבָן דְּאִשְׁתָּאֳרָן:

{לז} וַיִּקַּח לוֹ יַעֲקֹב מַקַּל לִבְנֶה לַח וְלוּז וְעֶרְמוֹן וַיְפַצֵּל בָּהֵן פְּצָלוֹת לְבָנוֹת מַחְשֹׂף הַלָּבָן אֲשֶׁר עַל הַמַּקְלוֹת:  מ  וַיִּקַּח לוֹ יַעֲקֹב מַקַּל לִבְנֶה לַח וְלוּז וְעֶרְמוֹן וַיְפַצֵּל בָּהֵן פְּצָלוֹת לְבָנוֹת מַחְשֹׂף הַלָּבָן אֲשֶׁר עַל הַמַּקְלוֹת:  ת  וּנְסִיב לֵהּ יַעֲקֹב חוּטְרִין דִּלְבָן רַטִּיבִין וִּדְלוּז וְדִדְלוּף וְקַלִּיף בְּהוֹן קִלְפִין חִוָּרִין קִלּוּף חִוָּר דִּי עַל חוּטְרַיָּא:

{לח} וַיַּצֵּג אֶת הַמַּקְלוֹת אֲשֶׁר פִּצֵּל בָּרֳהָטִים בְּשִׁקֲתוֹת הַמָּיִם אֲשֶׁר תָּבֹאןָ הַצֹּאן לִשְׁתּוֹת לְנֹכַח הַצֹּאן וַיֵּחַמְנָה בְּבֹאָן לִשְׁתּוֹת:  מ  וַיַּצֵּג אֶת הַמַּקְלוֹת אֲשֶׁר פִּצֵּל בָּרֳהָטִים בְּשִׁקֲתוֹת הַמָּיִם אֲשֶׁר תָּבֹאןָ הַצֹּאן לִשְׁתּוֹת לְנֹכַח הַצֹּאן וַיֵּחַמְנָה בְּבֹאָן לִשְׁתּוֹת:  ת  וְדָעִיץ יָת חוּטְרַיָּא דִּי קַלִּיף בִּרְהָטַיָּא אֲתַר בֵּית שַׁקְיָא דְמַיָּא אֲתַר דְּאָתָן (נ''י דְּאָתְיָן) עָנָא לְמִשְׁתֵּי לְקִבְלֵיהוֹן דְּעָנָא וּמִתְיַחֲמָן בְּמֵיתֵיהוֹן לְמִשְׁתֵּי:

{לט} וַיֶּחֱמוּ הַצֹּאן אֶל הַמַּקְלוֹת וַתֵּלַדְןָ הַצֹּאן עֲקֻדִּים נְקֻדִּים וּטְלֻאִים:  מ  וַיֶּחֱמוּ הַצֹּאן אֶל הַמַּקְלוֹת וַתֵּלַדְןָ הַצֹּאן עֲקֻדִּים נְקֻדִּים וּטְלֻאִים:  ת  וְאִתְיַחֲמָא עָנָא בְּחוּטְרַיָּא וִילִידָא עָנָא רְגוֹלִין נְמוֹרִין וּרְקוֹעִין:

{מ} וְהַכְּשָׂבִים הִפְרִיד יַעֲקֹב וַיִּתֵּן פְּנֵי הַצֹּאן אֶל עָקֹד וְכָל חוּם בְּצֹאן לָבָן וַיָּשֶׁת לוֹ עֲדָרִים לְבַדּוֹ וְלֹא שָׁתָם עַל צֹאן לָבָן:  מ  וְהַכְּשָׂבִים הִפְרִיד יַעֲקֹב וַיִּתֵּן פְּנֵי הַצֹּאן אֶל עָקֹד וְכָל חוּם בְּצֹאן לָבָן וַיָּשֶׁת לוֹ עֲדָרִים לְבַדּוֹ וְלֹא שָׁתָם עַל צֹאן לָבָן:  ת  וְאִמְּרַיָּא אַפְרֵשׁ יַעֲקֹב וִיהַב בְּרֵישׁ עָנָא כָּל דִּרְגוֹל וְכָל דִּשְׁחוּם בְּעָנָא דְלָבָן וְשַׁוִּי לֵהּ עֶדְרִין בִּלְחוֹדוֹהִי וְלָא עֲרֵבִנּוּן עִם עָנָא דְלָבָן:

{מא} וְהָיָה בְּכָל יַחֵם הַצֹּאן הַמְקֻשָּׁרוֹת וְשָׂם יַעֲקֹב אֶת הַמַּקְלוֹת לְעֵינֵי הַצֹּאן בָּרֳהָטִים לְיַחְמֵנָּה בַּמַּקְלוֹת:  מ  וְהָיָה בְּכָל יַחֵם הַצֹּאן הַמְקֻשָּׁרוֹת וְשָׂם יַעֲקֹב אֶת הַמַּקְלוֹת לְעֵינֵי הַצֹּאן בָּרֳהָטִים לְיַחְמֵנָּה בַּמַּקְלוֹת:  ת  וַהֲוֵי בְּכָל עִדַּן דְּמִתְיַחֲמָן עָנָא מְבַכַּרְתָּא וּמְשַׁוִּי יַעֲקֹב יָת חוּטְרַיָּא לְעֵינֵי עָנָא בִּרְהָטַיָּא לְיַחָמוּתְהוֹן בְּחוּטְרַיָּא:

{מב} וּבְהַעֲטִיף הַצֹּאן לֹא יָשִׂים וְהָיָה הָעֲטֻפִים לְלָבָן וְהַקְּשֻׁרִים לְיַעֲקֹב:  מ  וּבְהַעֲטִיף הַצֹּאן לֹא יָשִׂים וְהָיָה הָעֲטֻפִים לְלָבָן וְהַקְּשֻׁרִים לְיַעֲקֹב:  ת  וּבְלַקִּישׁוּת עָנָא לָא מְשַׁוֵּי וַהֲווֹ לַקִּישָׁא לְלָבָן וּבַכִּירַיָּא לְיַעֲקֹב:

{מג} וַיִּפְרֹץ הָאִישׁ מְאֹד מְאֹד וַיְהִי לוֹ צֹאן רַבּוֹת וּשְׁפָחוֹת וַעֲבָדִים וּגְמַלִּים וַחֲמֹרִים:  מ  וַיִּפְרֹץ הָאִישׁ מְאֹד מְאֹד וַיְהִי לוֹ צֹאן רַבּוֹת וּשְׁפָחוֹת וַעֲבָדִים וּגְמַלִּים וַחֲמֹרִים:  ת  וּתְקֵיף גַּבְרָא לַחֲדָא לַחֲדָא וַהֲווֹ לֵהּ עָן סַגִּיאָן וְאַמְהָן וְעַבְדִּין וְגַמְלִין וַחֲמָרִין:




בראשית פרק-לא

{א} וַיִּשְׁמַע אֶת דִּבְרֵי בְנֵי לָבָן לֵאמֹר לָקַח יַעֲקֹב אֵת כָּל אֲשֶׁר לְאָבִינוּ וּמֵאֲשֶׁר לְאָבִינוּ עָשָׂה אֵת כָּל הַכָּבֹד הַזֶּה:  מ  וַיִּשְׁמַע אֶת דִּבְרֵי בְנֵי לָבָן לֵאמֹר לָקַח יַעֲקֹב אֵת כָּל אֲשֶׁר לְאָבִינוּ וּמֵאֲשֶׁר לְאָבִינוּ עָשָׂה אֵת כָּל הַכָּבֹד הַזֶּה:  ת  וּשְׁמַע יָת פִתְגָּמֵי בְנֵי לָבָן דְּאָמְרִין נְסִיב יַעֲקֹב יָת כָּל דִּי לְאָבוּנָא וּמִדִּי לְאָבוּנָא קְנָא יָת כָּל נִכְסַיָּא הָדֵין:

{ב} וַיַּרְא יַעֲקֹב אֶת פְּנֵי לָבָן וְהִנֵּה אֵינֶנּוּ עִמּוֹ כִּתְמוֹל שִׁלְשׁוֹם:  מ  וַיַּרְא יַעֲקֹב אֶת פְּנֵי לָבָן וְהִנֵּה אֵינֶנּוּ עִמּוֹ כִּתְמוֹל שִׁלְשׁוֹם:  ת  וַחֲזָא יַעֲקֹב יָת סְבַר אַפֵּי לָבָן וְהָא לֵיתוֹהִי עִמֵּהּ כְּמֵאִתְמַלִּי וּמִדְּקַמּוֹהִי:

{ג} וַיֹּאמֶר יְהוָה אֶל יַעֲקֹב שׁוּב אֶל אֶרֶץ אֲבוֹתֶיךָ וּלְמוֹלַדְתֶּךָ וְאֶהְיֶה עִמָּךְ:  מ  וַיֹּאמֶר יְהוָה אֶל יַעֲקֹב שׁוּב אֶל אֶרֶץ אֲבוֹתֶיךָ וּלְמוֹלַדְתֶּךָ וְאֶהְיֶה עִמָּךְ:  ת  וַאֲמַר יְיָ לְיַעֲקֹב תּוּב לְאַרְעָא דַּאֲבָהָתָךְ וּלְיַלָּדֻתָךְ וִיהֵי מֵימְרִי בְּסַעְדָּךְ:

{ד} וַיִּשְׁלַח יַעֲקֹב וַיִּקְרָא לְרָחֵל וּלְלֵאָה הַשָּׂדֶה אֶל צֹאנוֹ:  מ  וַיִּשְׁלַח יַעֲקֹב וַיִּקְרָא לְרָחֵל וּלְלֵאָה הַשָּׂדֶה אֶל צֹאנוֹ:  ת  וּשְׁלַח יַעֲקֹב וּקְרָא לְרָחֵל וּלְלֵאָה לְחַקְלָא לְוַת עָנֵהּ:

{ה} וַיֹּאמֶר לָהֶן רֹאֶה אָנֹכִי אֶת פְּנֵי אֲבִיכֶן כִּי אֵינֶנּוּ אֵלַי כִּתְמֹל שִׁלְשֹׁם וֵאלֹהֵי אָבִי הָיָה עִמָּדִי:  מ  וַיֹּאמֶר לָהֶן רֹאֶה אָנֹכִי אֶת פְּנֵי אֲבִיכֶן כִּי אֵינֶנּוּ אֵלַי כִּתְמֹל שִׁלְשֹׁם וֵאלֹהֵי אָבִי הָיָה עִמָּדִי:  ת  וַאֲמַר לְהֵן חָזֵי אֲנָא יָת סְבַר אַפֵּי אֲבוּכֶן אֲרֵי לֵיתוֹהִי עִמִּי כְּמֵאִתְמַלִּי וּמִדְּקַמֹּהִי וֵאלָהֵהּ דְּאַבָּא הֲוָה בְּסַעְדִּי:

{ו} וְאַתֵּנָה יְדַעְתֶּן כִּי בְּכָל כֹּחִי עָבַדְתִּי אֶת אֲבִיכֶן:  מ  וְאַתֵּנָה יְדַעְתֶּן כִּי בְּכָל כֹּחִי עָבַדְתִּי אֶת אֲבִיכֶן:  ת  וְאַתִּין יְדַעְתִּין אֲרֵי בְּכָל חֵילִי פְּלָחִית יָת אֲבוּכֶן:

{ז} וַאֲבִיכֶן הֵתֶל בִּי וְהֶחֱלִף אֶת מַשְׂכֻּרְתִּי עֲשֶׂרֶת מֹנִים וְלֹא נְתָנוֹ אֱלֹהִים לְהָרַע עִמָּדִי:  מ  וַאֲבִיכֶן הֵתֶל בִּי וְהֶחֱלִף אֶת מַשְׂכֻּרְתִּי עֲשֶׂרֶת מֹנִים וְלֹא נְתָנוֹ אֱלֹהִים לְהָרַע עִמָּדִי:  ת  וַאֲבוּכֶן שְׁקַּר בִּי וְאַשְׁנִי יָת אַגְרִי עֲשַׂר זִמְנִין וְלָא שָׁבְקֵהּ יְיָ לְאַבְאָשָׁא עִמִּי:

{ח} אִם כֹּה יֹאמַר נְקֻדִּים יִהְיֶה שְׂכָרֶךָ וְיָלְדוּ כָל הַצֹּאן נְקֻדִּים וְאִם כֹּה יֹאמַר עֲקֻדִּים יִהְיֶה שְׂכָרֶךָ וְיָלְדוּ כָל הַצֹּאן עֲקֻדִּים:  מ  אִם כֹּה יֹאמַר נְקֻדִּים יִהְיֶה שְׂכָרֶךָ וְיָלְדוּ כָל הַצֹּאן נְקֻדִּים וְאִם כֹּה יֹאמַר עֲקֻדִּים יִהְיֶה שְׂכָרֶךָ וְיָלְדוּ כָל הַצֹּאן עֲקֻדִּים:  ת  אִם כְּדֵין הֲוָה אֲמַר נְמוֹרִין יְהֵא אַגְרָךְ וִילִידָן כָּל עָנָא נְמוֹרִין וְאִם כְּדֵין הֲוָה אֲמַר רְגוֹלִין יְהֵא אַגְרָךְ וִילִידָן כָּל עָנָא רְגוֹלִין:

{ט} וַיַּצֵּל אֱלֹהִים אֶת מִקְנֵה אֲבִיכֶם וַיִּתֶּן לִי:  מ  וַיַּצֵּל אֱלֹהִים אֶת מִקְנֵה אֲבִיכֶם וַיִּתֶּן לִי:  ת  וְאַפְרֵשׁ יְיָ יָת גֵּיתֵי דַּאֲבוּכוֹן וִיהַב לִי:

{י} וַיְהִי בְּעֵת יַחֵם הַצֹּאן וָאֶשָּׂא עֵינַי וָאֵרֶא בַּחֲלוֹם וְהִנֵּה הָעַתֻּדִים הָעֹלִים עַל הַצֹּאן עֲקֻדִּים נְקֻדִּים וּבְרֻדִּים:  מ  וַיְהִי בְּעֵת יַחֵם הַצֹּאן וָאֶשָּׂא עֵינַי וָאֵרֶא בַּחֲלוֹם וְהִנֵּה הָעַתֻּדִים הָעֹלִים עַל הַצֹּאן עֲקֻדִּים נְקֻדִּים וּבְרֻדִּים:  ת  וַהֲוָה בְּעִדַּן דְּאִתְיַחֲמָא עָנָא וּזְקָפִית עַיְנַי וַחֲזֵית בְּחֶלְמָא וְהָא תְיָשַׁיָּא דְּסָלְקִין עַל עָנָא רְגוֹלִין נְמוֹרִין וּפַצִיחִין:

{יא} וַיֹּאמֶר אֵלַי מַלְאַךְ הָאֱלֹהִים בַּחֲלוֹם יַעֲקֹב וָאֹמַר הִנֵּנִי:  מ  וַיֹּאמֶר אֵלַי מַלְאַךְ הָאֱלֹהִים בַּחֲלוֹם יַעֲקֹב וָאֹמַר הִנֵּנִי:  ת  וַאֲמַר לִי מַלְאָכָא דַּיְיָ בְּחֶלְמָא יַעֲקֹב וַאֲמָרִית הָא אֲנָא:

{יב} וַיֹּאמֶר שָׂא נָא עֵינֶיךָ וּרְאֵה כָּל הָעַתֻּדִים הָעֹלִים עַל הַצֹּאן עֲקֻדִּים נְקֻדִּים וּבְרֻדִּים כִּי רָאִיתִי אֵת כָּל אֲשֶׁר לָבָן עֹשֶׂה לָּךְ:  מ  וַיֹּאמֶר שָׂא נָא עֵינֶיךָ וּרְאֵה כָּל הָעַתֻּדִים הָעֹלִים עַל הַצֹּאן עֲקֻדִּים נְקֻדִּים וּבְרֻדִּים כִּי רָאִיתִי אֵת כָּל אֲשֶׁר לָבָן עֹשֶׂה לָּךְ:  ת  וַאֲמַר זְקוֹף כְּעַן עַינָיךְ וַחֲזֵי כָּל תְּיָשַׁיָּא דְּסָלְקִין עַל עָנָא רְגוֹלִין נְמוֹרִין וּפַצִיחִין אֲרֵי גְלֵי קֳדָמַי יָת כָּל דִּי לָבָן עָבֵד לָךְ:

{יג} אָנֹכִי הָאֵל בֵּית אֵל אֲשֶׁר מָשַׁחְתָּ שָּׁם מַצֵּבָה אֲשֶׁר נָדַרְתָּ לִּי שָׁם נֶדֶר עַתָּה קוּם צֵא מִן הָאָרֶץ הַזֹּאת וְשׁוּב אֶל אֶרֶץ מוֹלַדְתֶּךָ:  מ  אָנֹכִי הָאֵל בֵּית אֵל אֲשֶׁר מָשַׁחְתָּ שָּׁם מַצֵּבָה אֲשֶׁר נָדַרְתָּ לִּי שָׁם נֶדֶר עַתָּה קוּם צֵא מִן הָאָרֶץ הַזֹּאת וְשׁוּב אֶל אֶרֶץ מוֹלַדְתֶּךָ:  ת  אֲנָא אֱלָהָא דְּאִתְגְּלֵיתִי עֲלָךְ בְּבֵית אֵל דִּי מְשַׁחְתָּא תַמָּן קָמָא דִּי קַיִּמְתָּא קֳדָמַי תַּמָּן קְיָם כְּעַן קוּם פּוּק מִן אַרְעָא הָדָא וְתוּב לְאַרְעָא יַלָּדוּתָךְ:

{יד} וַתַּעַן רָחֵל וְלֵאָה וַתֹּאמַרְנָה לוֹ הַעוֹד לָנוּ חֵלֶק וְנַחֲלָה בְּבֵית אָבִינוּ:  מ  וַתַּעַן רָחֵל וְלֵאָה וַתֹּאמַרְנָה לוֹ הַעוֹד לָנוּ חֵלֶק וְנַחֲלָה בְּבֵית אָבִינוּ:  ת  וַאֲתִיבַת רָחֵל וְלֵאָה וַאֲמָרַן לֵהּ הַעוֹד (כְּעַן) לָנָא חֳלָק וְאַחֲסָנָא בְּבֵית אֲבוּנָא:

{טו} הֲלוֹא נָכְרִיּוֹת נֶחְשַׁבְנוּ לוֹ כִּי מְכָרָנוּ וַיֹּאכַל גַּם אָכוֹל אֶת כַּסְפֵּנוּ:  מ  הֲלוֹא נָכְרִיּוֹת נֶחְשַׁבְנוּ לוֹ כִּי מְכָרָנוּ וַיֹּאכַל גַּם אָכוֹל אֶת כַּסְפֵּנוּ:  ת  הֲלָא נוּכְרָאָן אִתְחֲשַׁבְנָא לֵהּ אֲרֵי זַבְּנָנָא וַאֲכַל אַף מֵיכַל יָת כַּסְפָּנָא:

{טז} כִּי כָל הָעֹשֶׁר אֲשֶׁר הִצִּיל אֱלֹהִים מֵאָבִינוּ לָנוּ הוּא וּלְבָנֵינוּ וְעַתָּה כֹּל אֲשֶׁר אָמַר אֱלֹהִים אֵלֶיךָ עֲשֵׂה:  מ  כִּי כָל הָעֹשֶׁר אֲשֶׁר הִצִּיל אֱלֹהִים מֵאָבִינוּ לָנוּ הוּא וּלְבָנֵינוּ וְעַתָּה כֹּל אֲשֶׁר אָמַר אֱלֹהִים אֵלֶיךָ עֲשֵׂה:  ת  אֲרֵי כָל עוּתְרָא דִּי אַפְרֵשׁ יְיָ מֵאֲבוּנָא דִּילָנָא הוּא וְדִבְּנָנָא וּכְעַן כֹּל דִּי אֲמַר יְיָ לָךְ עִבֵד:

{יז}  שישי  וַיָּקָם יַעֲקֹב וַיִּשָּׂא אֶת בָּנָיו וְאֶת נָשָׁיו עַל הַגְּמַלִּים:  מ  וַיָּקָם יַעֲקֹב וַיִּשָּׂא אֶת בָּנָיו וְאֶת נָשָׁיו עַל הַגְּמַלִּים:  ת  וְקָם יַעֲקֹב וּנְטַל יָת בְּנוֹהִי וְיָת נְשׁוֹהִי עַל גַּמְלַיָּא:

{יח} וַיִּנְהַג אֶת כָּל מִקְנֵהוּ וְאֶת כָּל רְכֻשׁוֹ אֲשֶׁר רָכָשׁ מִקְנֵה קִנְיָנוֹ אֲשֶׁר רָכַשׁ בְּפַדַּן אֲרָם לָבוֹא אֶל יִצְחָק אָבִיו אַרְצָה כְּנָעַן:  מ  וַיִּנְהַג אֶת כָּל מִקְנֵהוּ וְאֶת כָּל רְכֻשׁוֹ אֲשֶׁר רָכָשׁ מִקְנֵה קִנְיָנוֹ אֲשֶׁר רָכַשׁ בְּפַדַּן אֲרָם לָבוֹא אֶל יִצְחָק אָבִיו אַרְצָה כְּנָעַן:  ת  וּדְבַר יָת כָּל גֵּיתוֹהִי וְיָת כָּל קִנְיָנֵהּ דִּקְנָא גֵּיתֵי קִנְיָנֵהּ דִּי קְנָא בְּפַדַּן אֲרָם לְמֵיתֵי לְוַת יִצְחָק אֲבוּהִי לְאַרְעָא כְנָעַן:

{יט} וְלָבָן הָלַךְ לִגְזֹז אֶת צֹאנוֹ וַתִּגְנֹב רָחֵל אֶת הַתְּרָפִים אֲשֶׁר לְאָבִיהָ:  מ  וְלָבָן הָלַךְ לִגְזֹז אֶת צֹאנוֹ וַתִּגְנֹב רָחֵל אֶת הַתְּרָפִים אֲשֶׁר לְאָבִיהָ:  ת  וְלָבָן אֲזַל לְמִגַּז יָת עָנֵהּ וּנְסִּבַת רָחֵל יָת צַלְמָנַיָּא דִּי לַאֲבוּהָא:

{כ} וַיִּגְנֹב יַעֲקֹב אֶת לֵב לָבָן הָאֲרַמִּי עַל בְּלִי הִגִּיד לוֹ כִּי בֹרֵחַ הוּא:  מ  וַיִּגְנֹב יַעֲקֹב אֶת לֵב לָבָן הָאֲרַמִּי עַל בְּלִי הִגִּיד לוֹ כִּי בֹרֵחַ הוּא:  ת  וְכַסִּי יַעֲקֹב מִן לִבָּא דְלָבָן אֲרַמָּאָה עַל דְּלָא חַוִּי לֵהּ אֲרֵי אָזֵל הוּא:

{כא} וַיִּבְרַח הוּא וְכָל אֲשֶׁר לוֹ וַיָּקָם וַיַּעֲבֹר אֶת הַנָּהָר וַיָּשֶׂם אֶת פָּנָיו הַר הַגִּלְעָד:  מ  וַיִּבְרַח הוּא וְכָל אֲשֶׁר לוֹ וַיָּקָם וַיַּעֲבֹר אֶת הַנָּהָר וַיָּשֶׂם אֶת פָּנָיו הַר הַגִּלְעָד:  ת  וַאֲזַלּ הוּא וְכָל דִּי לֵהּ וְקָם וַעֲבַר יָת פְּרָת וְשַׁוִּי יָת אַפּוֹהִי לְטוּרָא דְגִלְעָד:

{כב} וַיֻּגַּד לְלָבָן בַּיּוֹם הַשְּׁלִישִׁי כִּי בָרַח יַעֲקֹב:  מ  וַיֻּגַּד לְלָבָן בַּיּוֹם הַשְּׁלִישִׁי כִּי בָרַח יַעֲקֹב:  ת  וְאִתְחַוָא לְלָבָן בְּיוֹמָא תְלִיתָאָה אֲרֵי אֲזַל יַעֲקֹב:

{כג} וַיִּקַּח אֶת אֶחָיו עִמּוֹ וַיִּרְדֹּף אַחֲרָיו דֶּרֶךְ שִׁבְעַת יָמִים וַיַּדְבֵּק אֹתוֹ בְּהַר הַגִּלְעָד:  מ  וַיִּקַּח אֶת אֶחָיו עִמּוֹ וַיִּרְדֹּף אַחֲרָיו דֶּרֶךְ שִׁבְעַת יָמִים וַיַּדְבֵּק אֹתוֹ בְּהַר הַגִּלְעָד:  ת  וּדְבַר יָת אַחוּהִי עִמֵּהּ וּרְדַף בַּתְרוֹהִי מַהֲלַךְ שִׁבְעַת יוֹמִין וְאַדְבֵּק יָתֵהּ בְּטוּרָא דְגִלְעָד:

{כד} וַיָּבֹא אֱלֹהִים אֶל לָבָן הָאֲרַמִּי בַּחֲלֹם הַלָּיְלָה וַיֹּאמֶר לוֹ הִשָּׁמֶר לְךָ פֶּן תְּדַבֵּר עִם יַעֲקֹב מִטּוֹב עַד רָע:  מ  וַיָּבֹא אֱלֹהִים אֶל לָבָן הָאֲרַמִּי בַּחֲלֹם הַלָּיְלָה וַיֹּאמֶר לוֹ הִשָּׁמֶר לְךָ פֶּן תְּדַבֵּר עִם יַעֲקֹב מִטּוֹב עַד רָע:  ת  וַאֲתָא מֵימָר מִן קֳדָם יְיָ לְוַת לָבָן אֲרַמָּאָה בְּחֵילְמָא דְּלֵילְיָא וַאֲמַר לֵהּ אִסְתַּמַּר לָךְ דִּילְמָא תְמַלֵּיל עִם יַעֲקֹב מִטַּב עַד בִּישׁ:

{כה} וַיַּשֵּׂג לָבָן אֶת יַעֲקֹב וְיַעֲקֹב תָּקַע אֶת אָהֳלוֹ בָּהָר וְלָבָן תָּקַע אֶת אֶחָיו בְּהַר הַגִּלְעָד:  מ  וַיַּשֵּׂג לָבָן אֶת יַעֲקֹב וְיַעֲקֹב תָּקַע אֶת אָהֳלוֹ בָּהָר וְלָבָן תָּקַע אֶת אֶחָיו בְּהַר הַגִּלְעָד:  ת  וְאַדְבֵּיק לָבָן יָת יַעֲקֹב וְיַעֲקֹב פְּרַס יָת מַשְׁכְּנֵהּ בְּטוּרָא וְלָבָן אַשְׁרֵי יָת אֲחוֹהִי בְּטוּרָא דְגִלְעָד:

{כו} וַיֹּאמֶר לָבָן לְיַעֲקֹב מֶה עָשִׂיתָ וַתִּגְנֹב אֶת לְבָבִי וַתְּנַהֵג אֶת בְּנֹתַי כִּשְׁבֻיוֹת חָרֶב:  מ  וַיֹּאמֶר לָבָן לְיַעֲקֹב מֶה עָשִׂיתָ וַתִּגְנֹב אֶת לְבָבִי וַתְּנַהֵג אֶת בְּנֹתַי כִּשְׁבֻיוֹת חָרֶב:  ת  וַאֲמַר לָבָן לְיַעֲקֹב מָה עֲבַדְתָּ וְכַסֵּיתָא מִנִּי וּדְבַרְתָּא יָת בְּנָתַי כְּשַׁבְיוֹת חַרְבָּא:

{כז} לָמָּה נַחְבֵּאתָ לִבְרֹחַ וַתִּגְנֹב אֹתִי וְלֹא הִגַּדְתָּ לִּי וָאֲשַׁלֵּחֲךָ בְּשִׂמְחָה וּבְשִׁרִים בְּתֹף וּבְכִנּוֹר:  מ  לָמָּה נַחְבֵּאתָ לִבְרֹחַ וַתִּגְנֹב אֹתִי וְלֹא הִגַּדְתָּ לִּי וָאֲשַׁלֵּחֲךָ בְּשִׂמְחָה וּבְשִׁרִים בְּתֹף וּבְכִנּוֹר:  ת  לְמָה אִטְמַרְתָּ לְמֵיזַל וְכַסֵּיתָא מִנִּי וְלָא חַוֵּיתָא לִי וְשַׁלָּחִתָּךְ פּוֹן בְּחֶדְוָא וּבְתוּשְׁבְּחָן בְּתוּפִּין וּבְכִנָּרִין:

{כח} וְלֹא נְטַשְׁתַּנִי לְנַשֵּׁק לְבָנַי וְלִבְנֹתָי עַתָּה הִסְכַּלְתָּ עֲשׂוֹ:  מ  וְלֹא נְטַשְׁתַּנִי לְנַשֵּׁק לְבָנַי וְלִבְנֹתָי עַתָּה הִסְכַּלְתָּ עֲשׂוֹ:  ת  וְלָא שְׁבַקְתַּנִּי לְנַשָּׁקָא לִבְנַי וְלִבְנָתָי כְּעַן אַסְכֶּלְתָּא לְמֶעְבָּד:

{כט} יֶשׁ לְאֵל יָדִי לַעֲשׂוֹת עִמָּכֶם רָע וֵאלֹהֵי אֲבִיכֶם אֶמֶשׁ אָמַר אֵלַי לֵאמֹר הִשָּׁמֶר לְךָ מִדַּבֵּר עִם יַעֲקֹב מִטּוֹב עַד רָע:  מ  יֶשׁ לְאֵל יָדִי לַעֲשׂוֹת עִמָּכֶם רָע וֵאלֹהֵי אֲבִיכֶם אֶמֶשׁ אָמַר אֵלַי לֵאמֹר הִשָּׁמֶר לְךָ מִדַּבֵּר עִם יַעֲקֹב מִטּוֹב עַד רָע:  ת  אִית חֵילָא בִידִי לְמֶעְבַּד עִמְּכוֹן בִּישׁ וֵאלָהָא דַּאֲבוּכוֹן בְּרַמְשָׁא אֲמַר לִי לְמֵימַר אִסְתַּמַּר לָּךְ מִלְּמַלָּלָא עִם יַעֲקֹב מִטַּב עַד בִּישׁ:

{ל} וְעַתָּה הָלֹךְ הָלַכְתָּ כִּי נִכְסֹף נִכְסַפְתָּה לְבֵית אָבִיךָ לָמָּה גָנַבְתָּ אֶת אֱלֹהָי:  מ  וְעַתָּה הָלֹךְ הָלַכְתָּ כִּי נִכְסֹף נִכְסַפְתָּה לְבֵית אָבִיךָ לָמָּה גָנַבְתָּ אֶת אֱלֹהָי:  ת  וּכְעַן מֵיזַל אֲזַלְתָּ אֲרֵי חַמָּדָא חַמֵדְתָּא לְבֵית אָבוּךְ לְמָה נְסִבְתָּא יָת דַּחַלְתִּי:

{לא} וַיַּעַן יַעֲקֹב וַיֹּאמֶר לְלָבָן כִּי יָרֵאתִי כִּי אָמַרְתִּי פֶּן תִּגְזֹל אֶת בְּנוֹתֶיךָ מֵעִמִּי:  מ  וַיַּעַן יַעֲקֹב וַיֹּאמֶר לְלָבָן כִּי יָרֵאתִי כִּי אָמַרְתִּי פֶּן תִּגְזֹל אֶת בְּנוֹתֶיךָ מֵעִמִּי:  ת  וַאֲתֵיב יַעֲקֹב וַאֲמַר לְלָבָן אֲרֵי דְחֵילִית אֲרֵי אֲמָרִית דִּילְמָא תָנֵיס יָת בְּנָתָךְ מִנִּי:

{לב} עִם אֲשֶׁר תִּמְצָא אֶת אֱלֹהֶיךָ לֹא יִחְיֶה נֶגֶד אַחֵינוּ הַכֶּר לְךָ מָה עִמָּדִי וְקַח לָךְ וְלֹא יָדַע יַעֲקֹב כִּי רָחֵל גְּנָבָתַם:  מ  עִם אֲשֶׁר תִּמְצָא אֶת אֱלֹהֶיךָ לֹא יִחְיֶה נֶגֶד אַחֵינוּ הַכֶּר לְךָ מָה עִמָּדִי וְקַח לָךְ וְלֹא יָדַע יַעֲקֹב כִּי רָחֵל גְּנָבָתַם:  ת  עִם (נ''י אֲתַר) דִּי תַשְׁכַּח יָת דַּחַלְתָּךְ לָא יִתְקַיַּם קֳדָם אֲחָנָא אִשְׁתְּמוֹדַע לָךְ מָה דְעִמִּי וְסַב לָךְ וְלָא יְדַע יַעֲקֹב אֲרֵי רָחֵל נְסִיבָתְהוֹן:

{לג} וַיָּבֹא לָבָן בְּאֹהֶל יַעֲקֹב וּבְאֹהֶל לֵאָה וּבְאֹהֶל שְׁתֵּי הָאֲמָהֹת וְלֹא מָצָא וַיֵּצֵא מֵאֹהֶל לֵאָה וַיָּבֹא בְּאֹהֶל רָחֵל:  מ  וַיָּבֹא לָבָן בְּאֹהֶל יַעֲקֹב וּבְאֹהֶל לֵאָה וּבְאֹהֶל שְׁתֵּי הָאֲמָהֹת וְלֹא מָצָא וַיֵּצֵא מֵאֹהֶל לֵאָה וַיָּבֹא בְּאֹהֶל רָחֵל:  ת  וְעַל לָבָן בְּמַשְׁכְּנָא דְיַעֲקֹב וּבְמַשְׁכְּנָא דְלֵאָה וּבְמַשְׁכְּנָא דְּתַרְתֵּין לְחֵינָתָא וְלָא אַשְׁכַּח וּנְפַק מִמַּשְׁכְּנָא דְלֵאָה וְעַל בְּמַשְׁכְּנָא דְרָחֵל:

{לד} וְרָחֵל לָקְחָה אֶת הַתְּרָפִים וַתְּשִׂמֵם בְּכַר הַגָּמָל וַתֵּשֶׁב עֲלֵיהֶם וַיְמַשֵּׁשׁ לָבָן אֶת כָּל הָאֹהֶל וְלֹא מָצָא:  מ  וְרָחֵל לָקְחָה אֶת הַתְּרָפִים וַתְּשִׂמֵם בְּכַר הַגָּמָל וַתֵּשֶׁב עֲלֵיהֶם וַיְמַשֵּׁשׁ לָבָן אֶת כָּל הָאֹהֶל וְלֹא מָצָא:  ת  וְרָחֵל נְסִיבַת יָת צַלְמָנַיָּא וְשַׁוִּתִּנּוּן בַּעֲבִיטָא דְגַמְלָא וִיתִיבַת עֲלֵיהוֹן וּמַשִּׁישׁ לָבָן יָת כָּל מַשְׁכְּנָא וְלָא אַשְׁכָּח:

{לה} וַתֹּאמֶר אֶל אָבִיהָ אַל יִחַר בְּעֵינֵי אֲדֹנִי כִּי לוֹא אוּכַל לָקוּם מִפָּנֶיךָ כִּי דֶרֶךְ נָשִׁים לִי וַיְחַפֵּשׂ וְלֹא מָצָא אֶת הַתְּרָפִים:  מ  וַתֹּאמֶר אֶל אָבִיהָ אַל יִחַר בְּעֵינֵי אֲדֹנִי כִּי לוֹא אוּכַל לָקוּם מִפָּנֶיךָ כִּי דֶרֶךְ נָשִׁים לִי וַיְחַפֵּשׂ וְלֹא מָצָא אֶת הַתְּרָפִים:  ת  וַאֲמֶרֶת לַאֲבוּהָא לָא יִתְקֵף בְּעֵינֵי רִבּוֹנִי אֲרֵי לָא אִכּוּל לְמֵיקַם מִן קֳדָמָךְ אֲרֵי אֹרַח נְשִׁין לִי וּבְלַשׁ וְלָא אַשְׁכַּח יָת צַלְמָנַיָּא:

{לו} וַיִּחַר לְיַעֲקֹב וַיָּרֶב בְּלָבָן וַיַּעַן יַעֲקֹב וַיֹּאמֶר לְלָבָן מַה פִּשְׁעִי מַה חַטָּאתִי כִּי דָלַקְתָּ אַחֲרָי:  מ  וַיִּחַר לְיַעֲקֹב וַיָּרֶב בְּלָבָן וַיַּעַן יַעֲקֹב וַיֹּאמֶר לְלָבָן מַה פִּשְׁעִי מַה חַטָּאתִי כִּי דָלַקְתָּ אַחֲרָי:  ת  וּתְקֵיף לְיַעֲקֹב וּנְצָא עִם לָבָן וַאֲתֵיב יַעֲקֹב וַאֲמַר לְלָבָן מַה חוֹבִי מַה סוּרְחָנִי אֲרֵי רְדַפְתָּא בַּתְרָי:

{לז} כִּי מִשַּׁשְׁתָּ אֶת כָּל כֵּלַי מַה מָּצָאתָ מִכֹּל כְּלֵי בֵיתֶךָ שִׂים כֹּה נֶגֶד אַחַי וְאַחֶיךָ וְיוֹכִיחוּ בֵּין שְׁנֵינוּ:  מ  כִּי מִשַּׁשְׁתָּ אֶת כָּל כֵּלַי מַה מָּצָאתָ מִכֹּל כְּלֵי בֵיתֶךָ שִׂים כֹּה נֶגֶד אַחַי וְאַחֶיךָ וְיוֹכִיחוּ בֵּין שְׁנֵינוּ:  ת  אֲרֵי מַשִּׁשְׁתָּא יָת כָּל מָנַי מָה אַשְׁכַּחְתָּא מִכֹּל מָנֵי בֵיתָךְ שַׁוִּי הָכָא קֳדָם אַחַי וְאַחָיךְ וְיוֹכִיחוּן בֵּין תַּרְוָנָא:

{לח} זֶה עֶשְׂרִים שָׁנָה אָנֹכִי עִמָּךְ רְחֵלֶיךָ וְעִזֶּיךָ לֹא שִׁכֵּלוּ וְאֵילֵי צֹאנְךָ לֹא אָכָלְתִּי:  מ  זֶה עֶשְׂרִים שָׁנָה אָנֹכִי עִמָּךְ רְחֵלֶיךָ וְעִזֶּיךָ לֹא שִׁכֵּלוּ וְאֵילֵי צֹאנְךָ לֹא אָכָלְתִּי:  ת  דְּנַן עֶסְרִין שְׁנִין אֲנָא עִמָּךְ רְחֵלָיךְ וְעִזָּיךְ לָא אַתְכִּילוּ וְדִכְרֵי עָנָךְ לָא אֲכָלִית:

{לט} טְרֵפָה לֹא הֵבֵאתִי אֵלֶיךָ אָנֹכִי אֲחַטֶּנָּה מִיָּדִי תְּבַקְשֶׁנָּה גְּנֻבְתִי יוֹם וּגְנֻבְתִי לָיְלָה:  מ  טְרֵפָה לֹא הֵבֵאתִי אֵלֶיךָ אָנֹכִי אֲחַטֶּנָּה מִיָּדִי תְּבַקְשֶׁנָּה גְּנֻבְתִי יוֹם וּגְנֻבְתִי לָיְלָה:  ת  דִּתְבִירָא לָא אַיְתֵיתִי לְוָתָךְ דַּהֲוַת שָׁגְיָא מִמִּנְיָנָא מִנִּי אַתְּ בָּעֵי לַהּ נְטָרִית בִּימָמָא וּנְטָּרִית בְּלֵילְיָא:

{מ} הָיִיתִי בַיּוֹם אֲכָלַנִי חֹרֶב וְקֶרַח בַּלָּיְלָה וַתִּדַּד שְׁנָתִי מֵעֵינָי:  מ  הָיִיתִי בַיּוֹם אֲכָלַנִי חֹרֶב וְקֶרַח בַּלָּיְלָה וַתִּדַּד שְׁנָתִי מֵעֵינָי:  ת  הֲוֵיתִי בִימָמָא אֲכַלְנִי שַׁרְבָא וּגְלִידָא (הֲוָה) נָחֵית עֲלַי בְּלֵילְיָא וּנְדַד שִׁנְתִּי מֵעֵינָי:

{מא} זֶה לִּי עֶשְׂרִים שָׁנָה בְּבֵיתֶךָ עֲבַדְתִּיךָ אַרְבַּע עֶשְׂרֵה שָׁנָה בִּשְׁתֵּי בְנֹתֶיךָ וְשֵׁשׁ שָׁנִים בְּצֹאנֶךָ וַתַּחֲלֵף אֶת מַשְׂכֻּרְתִּי עֲשֶׂרֶת מֹנִים:  מ  זֶה לִּי עֶשְׂרִים שָׁנָה בְּבֵיתֶךָ עֲבַדְתִּיךָ אַרְבַּע עֶשְׂרֵה שָׁנָה בִּשְׁתֵּי בְנֹתֶיךָ וְשֵׁשׁ שָׁנִים בְּצֹאנֶךָ וַתַּחֲלֵף אֶת מַשְׂכֻּרְתִּי עֲשֶׂרֶת מֹנִים:  ת  דְּנַן לִי עֶשְׂרִין שְׁנִין בְּבֵיתָךְ פְּלָחִתָּךְ אַרְבַּע עֶשְׂרֵי שְׁנִין בְּתַרְתֵּין בְּנָתָךְ וְשִׁתְּ שְׁנִין בְּעָנָךְ וְאַשְׁנֵיתָא יָת אַגְרִי עֲשַׂר זִמְנִין:

{מב} לוּלֵי אֱלֹהֵי אָבִי אֱלֹהֵי אַבְרָהָם וּפַחַד יִצְחָק הָיָה לִי כִּי עַתָּה רֵיקָם שִׁלַּחְתָּנִי אֶת עָנְיִי וְאֶת יְגִיעַ כַּפַּי רָאָה אֱלֹהִים וַיּוֹכַח אָמֶשׁ:  מ  לוּלֵי אֱלֹהֵי אָבִי אֱלֹהֵי אַבְרָהָם וּפַחַד יִצְחָק הָיָה לִי כִּי עַתָּה רֵיקָם שִׁלַּחְתָּנִי אֶת עָנְיִי וְאֶת יְגִיעַ כַּפַּי רָאָה אֱלֹהִים וַיּוֹכַח אָמֶשׁ:  ת  אִילוּלָא פוֹן אֱלָהֵהּ דְּאַבָּא אֱלָהֵהּ דְּאַבְרָהָם וּדְדָחִיל לֵהּ יִצְחָק הֲוָה בְּסַעְדִּי אֲרֵי כְעַן רֵיקָן שִׁלַּחְתָּנִי יָת עַמְלִי וְיָת לֵיאוּת יְדַי גְלֵי קֳדָם יְיָ וְאוֹכַח בְּרַמְשָׁא:

{מג}  שביעי  וַיַּעַן לָבָן וַיֹּאמֶר אֶל יַעֲקֹב הַבָּנוֹת בְּנֹתַי וְהַבָּנִים בָּנַי וְהַצֹּאן צֹאנִי וְכֹל אֲשֶׁר אַתָּה רֹאֶה לִי הוּא וְלִבְנֹתַי מָה אֶעֱשֶׂה לָאֵלֶּה הַיּוֹם אוֹ לִבְנֵיהֶן אֲשֶׁר יָלָדוּ:  מ  וַיַּעַן לָבָן וַיֹּאמֶר אֶל יַעֲקֹב הַבָּנוֹת בְּנֹתַי וְהַבָּנִים בָּנַי וְהַצֹּאן צֹאנִי וְכֹל אֲשֶׁר אַתָּה רֹאֶה לִי הוּא וְלִבְנֹתַי מָה אֶעֱשֶׂה לָאֵלֶּה הַיּוֹם אוֹ לִבְנֵיהֶן אֲשֶׁר יָלָדוּ:  ת  וַאֲתֵיב לָבָן וַאֲמַר לְיַעֲקֹב בְּנָתָא בְּנָתַי וּבְנַיָּא בְנַי וְעָנָא עָנִי וְכֹל דִּי אַתְּ חָזֵי דִּילִי הוּא וְלִבְנָתַי מָה אֶעְבֵּיד לְאִלֵּין יוֹמָא דֵין אוֹ לִבְנֵיהֶן דִּילִידָן:

{מד} וְעַתָּה לְכָה נִכְרְתָה בְרִית אֲנִי וָאָתָּה וְהָיָה לְעֵד בֵּינִי וּבֵינֶךָ:  מ  וְעַתָּה לְכָה נִכְרְתָה בְרִית אֲנִי וָאָתָּה וְהָיָה לְעֵד בֵּינִי וּבֵינֶךָ:  ת  וּכְעַן אֱתָא נִגְזַר קְיָם אֲנָא וְאָתְּ וִיהֵי לְסָהִיד בֵּינִי וּבֵינָךְ:

{מה} וַיִּקַּח יַעֲקֹב אָבֶן וַיְרִימֶהָ מַצֵּבָה:  מ  וַיִּקַּח יַעֲקֹב אָבֶן וַיְרִימֶהָ מַצֵּבָה:  ת  וּנְסִיב יַעֲקֹב אַבְנָא וּזְקָפַהּ קָמָא:

{מו} וַיֹּאמֶר יַעֲקֹב לְאֶחָיו לִקְטוּ אֲבָנִים וַיִּקְחוּ אֲבָנִים וַיַּעֲשׂוּ גָל וַיֹּאכְלוּ שָׁם עַל הַגָּל:  מ  וַיֹּאמֶר יַעֲקֹב לְאֶחָיו לִקְטוּ אֲבָנִים וַיִּקְחוּ אֲבָנִים וַיַּעֲשׂוּ גָל וַיֹּאכְלוּ שָׁם עַל הַגָּל:  ת  וַאֲמַר יַעֲקֹב לַאֲחוֹהִי לְקוּטוּ אַבְנִין וּנְסִיבוּ אַבְנִין וַעֲבָדוּ דְּגוֹרָא וַאֲכָלוּ תַמָּן עַל דְּגוֹרָא:

{מז} וַיִּקְרָא לוֹ לָבָן יְגַר שָׂהֲדוּתָא וְיַעֲקֹב קָרָא לוֹ גַּלְעֵד:  מ  וַיִּקְרָא לוֹ לָבָן יְגַר שָׂהֲדוּתָא וְיַעֲקֹב קָרָא לוֹ גַּלְעֵד:  ת  וּקְרָא לֵהּ לָבָן יְגַר שָׂהֲדוּתָא וְיַעֲקֹב קְרָא לֵהּ גַּלְעֵד:

{מח} וַיֹּאמֶר לָבָן הַגַּל הַזֶּה עֵד בֵּינִי וּבֵינְךָ הַיּוֹם עַל כֵּן קָרָא שְׁמוֹ גַּלְעֵד:  מ  וַיֹּאמֶר לָבָן הַגַּל הַזֶּה עֵד בֵּינִי וּבֵינְךָ הַיּוֹם עַל כֵּן קָרָא שְׁמוֹ גַּלְעֵד:  ת  וַאֲמַר לָבָן דְּגוֹרָא הָדֵין סָהִיד בֵּינִי וּבֵינָךְ יוֹמָא דֵּין עַל כֵּן קְרָא שְׁמֵהּ גַּלְעֵד:

{מט} וְהַמִּצְפָּה אֲשֶׁר אָמַר יִצֶף יְהוָה בֵּינִי וּבֵינֶךָ כִּי נִסָּתֵר אִישׁ מֵרֵעֵהוּ:  מ  וְהַמִּצְפָּה אֲשֶׁר אָמַר יִצֶף יְהוָה בֵּינִי וּבֵינֶךָ כִּי נִסָּתֵר אִישׁ מֵרֵעֵהוּ:  ת  וְסָכוּתָא דִּי אֲמַר יִסֶּךְ מֵימְרָא דַּיְיָ בֵּינִי וּבֵינָךְ אֲרֵי נִתכַּסִּי גּבַר מֵחַבְרֵהּ:

{נ} אִם תְּעַנֶּה אֶת בְּנֹתַי וְאִם תִּקַּח נָשִׁים עַל בְּנֹתַי אֵין אִישׁ עִמָּנוּ רְאֵה אֱלֹהִים עֵד בֵּינִי וּבֵינֶךָ:  מ  אִם תְּעַנֶּה אֶת בְּנֹתַי וְאִם תִּקַּח נָשִׁים עַל בְּנֹתַי אֵין אִישׁ עִמָּנוּ רְאֵה אֱלֹהִים עֵד בֵּינִי וּבֵינֶךָ:  ת  אִם תְּעַנֵּי יָת בְּנָתַי וְאִם תִּסַּב נְשִׁין עַל בְּנָתַי לֵית אֱנַשׁ עִמָּנָא חָזֵי מֵימְרָא דַּיְיָ סָהִיד בֵּינִי וּבֵינָךְ:

{נא} וַיֹּאמֶר לָבָן לְיַעֲקֹב הִנֵּה הַגַּל הַזֶּה וְהִנֵּה הַמַצֵּבָה אֲשֶׁר יָרִיתִי בֵּינִי וּבֵינֶךָ:  מ  וַיֹּאמֶר לָבָן לְיַעֲקֹב הִנֵּה הַגַּל הַזֶּה וְהִנֵּה הַמַצֵּבָה אֲשֶׁר יָרִיתִי בֵּינִי וּבֵינֶךָ:  ת  וַאֲמַר לָבָן לְיַעֲקֹב הָא דְּגוֹרָא הָדֵין וְהָא קַמְתָא דִּי אֲקֵימִית בֵּינִי וּבֵינָךְ:

{נב} עֵד הַגַּל הַזֶּה וְעֵדָה הַמַּצֵּבָה אִם אָנִי לֹא אֶעֱבֹר אֵלֶיךָ אֶת הַגַּל הַזֶּה וְאִם אַתָּה לֹא תַעֲבֹר אֵלַי אֶת הַגַּל הַזֶּה וְאֶת הַמַּצֵּבָה הַזֹּאת לְרָעָה:  מ  עֵד הַגַּל הַזֶּה וְעֵדָה הַמַּצֵּבָה אִם אָנִי לֹא אֶעֱבֹר אֵלֶיךָ אֶת הַגַּל הַזֶּה וְאִם אַתָּה לֹא תַעֲבֹר אֵלַי אֶת הַגַּל הַזֶּה וְאֶת הַמַּצֵּבָה הַזֹּאת לְרָעָה:  ת  סָהִיד דְּגוֹרָא הָדֵין וְסָהֲדָא קָמָא אִם אֲנָא לָא אֶעְבַּר לְוָתָךְ יָת דְּגוֹרָא הָדֵין וְאִם אַתְּ לָא תֵעִבַּר לְוָתִי יָת דְּגוֹרָא הָדֵין וְיָת קַמְתָא הָדָא לְבִישׁוּ:

{נג} אֱלֹהֵי אַבְרָהָם וֵאלֹהֵי נָחוֹר יִשְׁפְּטוּ בֵינֵינוּ אֱלֹהֵי אֲבִיהֶם וַיִּשָּׁבַע יַעֲקֹב בְּפַחַד אָבִיו יִצְחָק:  מ  אֱלֹהֵי אַבְרָהָם וֵאלֹהֵי נָחוֹר יִשְׁפְּטוּ בֵינֵינוּ אֱלֹהֵי אֲבִיהֶם וַיִּשָּׁבַע יַעֲקֹב בְּפַחַד אָבִיו יִצְחָק:  ת  אֱלָהֵהּ דְּאַבְרָהָם וֵאֱלָהֵהּ דְּנָחוֹר יְדוּנוּן בֵּינָנָא אֱלָהֵהּ דַּאֲבוּהוֹן וְקַיֵּים יַעֲקֹב בִּדְדָחִיל לֵהּ אֲבוּהִי יִצְחָק:

{נד} וַיִּזְבַּח יַעֲקֹב זֶבַח בָּהָר וַיִּקְרָא לְאֶחָיו לֶאֱכָל לָחֶם וַיֹּאכְלוּ לֶחֶם וַיָּלִינוּ בָּהָר:  מ  וַיִּזְבַּח יַעֲקֹב זֶבַח בָּהָר וַיִּקְרָא לְאֶחָיו לֶאֱכָל לָחֶם וַיֹּאכְלוּ לֶחֶם וַיָּלִינוּ בָּהָר:  ת  וּנְכֵס יַעֲקֹב נִכְסָתָא בְּטּוּרָא וּקְרָא לַאֲחוֹהִי לְמֵיכַל לַחְמָא וַאֲכָלוּ לַחְמָא וּבִיתוּ בְּטוּרָא:




בראשית פרק-לב

{א}  מפטיר  וַיַּשְׁכֵּם לָבָן בַּבֹּקֶר וַיְנַשֵּׁק לְבָנָיו וְלִבְנוֹתָיו וַיְבָרֶךְ אֶתְהֶם וַיֵּלֶךְ וַיָּשָׁב לָבָן לִמְקֹמוֹ:  מ  וַיַּשְׁכֵּם לָבָן בַּבֹּקֶר וַיְנַשֵּׁק לְבָנָיו וְלִבְנוֹתָיו וַיְבָרֶךְ אֶתְהֶם וַיֵּלֶךְ וַיָּשָׁב לָבָן לִמְקֹמוֹ:  ת  וְאַקְדֵּם לָבָן בְּצַפְרָא וּנְשַׁק לִבְנוֹהִי וְלִבְנָתֵהּ וּבָרִיךְ יָתְהֵן וַאֲזַל וְתַב לָבָן לְאַתְרֵהּ:

{ב} וְיַעֲקֹב הָלַךְ לְדַרְכּוֹ וַיִּפְגְּעוּ בוֹ מַלְאֲכֵי אֱלֹהִים:  מ  וְיַעֲקֹב הָלַךְ לְדַרְכּוֹ וַיִּפְגְּעוּ בוֹ מַלְאֲכֵי אֱלֹהִים:  ת  וְיַעֲקֹב אֲזָל לְאָרְחֵהּ וַעֲרָעוּ בֵהּ מַלְאָכַיָּא דַּיְיָ:

{ג} וַיֹּאמֶר יַעֲקֹב כַּאֲשֶׁר רָאָם מַחֲנֵה אֱלֹהִים זֶה וַיִּקְרָא שֵׁם הַמָּקוֹם הַהוּא מַחֲנָיִם:  מ  וַיֹּאמֶר יַעֲקֹב כַּאֲשֶׁר רָאָם מַחֲנֵה אֱלֹהִים זֶה וַיִּקְרָא שֵׁם הַמָּקוֹם הַהוּא מַחֲנָיִם:  ת  וַאֲמַר יַעֲקֹב כַּד חֲזָנוּן מַשְׁרִיתָא מִן קֳדָם יְיָ דֵּין וּקְרָא שְׁמָא דְאַתְרָא הַהוּא מַחֲנָיִם: (פפפ)
