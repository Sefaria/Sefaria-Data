פרשת ויחי


{כח} וַיְחִי יַעֲקֹב בְּאֶרֶץ מִצְרַיִם שְׁבַע עֶשְׂרֵה שָׁנָה וַיְהִי יְמֵי יַעֲקֹב שְׁנֵי חַיָּיו שֶׁבַע שָׁנִים וְאַרְבָּעִים וּמְאַת שָׁנָה:  מ  וַיְחִי יַעֲקֹב בְּאֶרֶץ מִצְרַיִם שְׁבַע עֶשְׂרֵה שָׁנָה וַיְהִי יְמֵי יַעֲקֹב שְׁנֵי חַיָּיו שֶׁבַע שָׁנִים וְאַרְבָּעִים וּמְאַת שָׁנָה:  ת  וַחֲיָא יַעֲקֹב בְּאַרְעָא דְמִצְרַיִם שְׁבַע עֶשְׂרֵי שְׁנִין וַהֲווֹ יוֹמֵי יַעֲקֹב שְׁנֵי חַיוֹהִי מְאָה וְאַרְבְּעִין וּשְׁבַע שְׁנִין:

{כט} וַיִּקְרְבוּ יְמֵי יִשְׂרָאֵל לָמוּת וַיִּקְרָא לִבְנוֹ לְיוֹסֵף וַיֹּאמֶר לוֹ אִם נָא מָצָאתִי חֵן בְּעֵינֶיךָ שִׂים נָא יָדְךָ תַּחַת יְרֵכִי וְעָשִׂיתָ עִמָּדִי חֶסֶד וֶאֱמֶת אַל נָא תִקְבְּרֵנִי בְּמִצְרָיִם:  מ  וַיִּקְרְבוּ יְמֵי יִשְׂרָאֵל לָמוּת וַיִּקְרָא לִבְנוֹ לְיוֹסֵף וַיֹּאמֶר לוֹ אִם נָא מָצָאתִי חֵן בְּעֵינֶיךָ שִׂים נָא יָדְךָ תַּחַת יְרֵכִי וְעָשִׂיתָ עִמָּדִי חֶסֶד וֶאֱמֶת אַל נָא תִקְבְּרֵנִי בְּמִצְרָיִם:  ת  וּקְרִיבוּ יוֹמֵי יִשְׂרָאֵל לִמְמַת וּקְרָא לִבְרֵהּ לְיוֹסֵף וַאֲמַר לֵהּ אִם כְּעַן אַשְׁכָּחִית רַחֲמִין בְּעֵינָיךְ שַׁוִּי כְעַן יְדָךְ תְּחוֹת יַרְכִּי וְתַעְבֵּד עִמִּי טִיבוּ וּקְשׁוֹט לָא כְעַן תִּקְבְּרִנַּנִי בְּמִצְרַיִם:

{ל} וְשָׁכַבְתִּי עִם אֲבֹתַי וּנְשָׂאתַנִי מִמִּצְרַיִם וּקְבַרְתַּנִי בִּקְבֻרָתָם וַיֹּאמַר אָנֹכִי אֶעֱשֶׂה כִדְבָרֶךָ:  מ  וְשָׁכַבְתִּי עִם אֲבֹתַי וּנְשָׂאתַנִי מִמִּצְרַיִם וּקְבַרְתַּנִי בִּקְבֻרָתָם וַיֹּאמַר אָנֹכִי אֶעֱשֶׂה כִדְבָרֶךָ:  ת  וְאִשְׁכּוּב עִם אֲבָהָתַי וְתִטְּלִנַנִּי מִמִּצְרַיִם וְתִקְבְּרִנַנִּי בִּקְבֻרְתְּהוֹן וַאֲמַר אֲנָא אֶעְבַּד כְּפִתְגָּמָךְ:

{לא} וַיֹּאמֶר הִשָּׁבְעָה לִי וַיִּשָּׁבַע לוֹ וַיִּשְׁתַּחוּ יִשְׂרָאֵל עַל רֹאשׁ הַמִּטָּה:  מ  וַיֹּאמֶר הִשָּׁבְעָה לִי וַיִּשָּׁבַע לוֹ וַיִּשְׁתַּחוּ יִשְׂרָאֵל עַל רֹאשׁ הַמִּטָּה:  ת  וַאֲמַר קַיֵּם לִי וְקַיִים לֵהּ וּסְגִיד יִשְׂרָאֵל עַל רֵישׁ עַרְסָא: (פ)




בראשית פרק-מח

{א} וַיְהִי אַחֲרֵי הַדְּבָרִים הָאֵלֶּה וַיֹּאמֶר לְיוֹסֵף הִנֵּה אָבִיךָ חֹלֶה וַיִּקַּח אֶת שְׁנֵי בָנָיו עִמּוֹ אֶת מְנַשֶּׁה וְאֶת אֶפְרָיִם:  מ  וַיְהִי אַחֲרֵי הַדְּבָרִים הָאֵלֶּה וַיֹּאמֶר לְיוֹסֵף הִנֵּה אָבִיךָ חֹלֶה וַיִּקַּח אֶת שְׁנֵי בָנָיו עִמּוֹ אֶת מְנַשֶּׁה וְאֶת אֶפְרָיִם:  ת  וַהֲוָה בָּתַר פִּתְגָּמַיָּא הָאִלֵּין וַאֲמַר לְיוֹסֵף הָא אֲבוּךְ שְׁכִיב מְרָע וּדְבַר יָת תְּרֵין בְּנוֹהִי עִמֵּהּ יָת מְנַשֶּׁה וְיָת אֶפְרָיִם:

{ב} וַיַּגֵּד לְיַעֲקֹב וַיֹּאמֶר הִנֵּה בִּנְךָ יוֹסֵף בָּא אֵלֶיךָ וַיִּתְחַזֵּק יִשְׂרָאֵל וַיֵּשֶׁב עַל הַמִּטָּה:  מ  וַיַּגֵּד לְיַעֲקֹב וַיֹּאמֶר הִנֵּה בִּנְךָ יוֹסֵף בָּא אֵלֶיךָ וַיִּתְחַזֵּק יִשְׂרָאֵל וַיֵּשֶׁב עַל הַמִּטָּה:  ת  וְחַוִּי לְיַעֲקֹב וַאֲמַר הָא בְּרָךְ יוֹסֵף אֲתָא לְוָתָךְ וְאִתַּקַף יִשְׂרָאֵל וִיתֵיב עַל עַרְסָא:

{ג} וַיֹּאמֶר יַעֲקֹב אֶל יוֹסֵף אֵל שַׁדַּי נִרְאָה אֵלַי בְּלוּז בְּאֶרֶץ כְּנָעַן וַיְבָרֶךְ אֹתִי:  מ  וַיֹּאמֶר יַעֲקֹב אֶל יוֹסֵף אֵל שַׁדַּי נִרְאָה אֵלַי בְּלוּז בְּאֶרֶץ כְּנָעַן וַיְבָרֶךְ אֹתִי:  ת  וַאֲמַר יַעֲקֹב לְיוֹסֵף אֵל שַׁדַּי אִתְגְּלִי לִי בְלוּז בְּאַרְעָא דִכְנָעַן וּבָרִיךְ יָתִי:

{ד} וַיֹּאמֶר אֵלַי הִנְנִי מַפְרְךָ וְהִרְבִּיתִךָ וּנְתַתִּיךָ לִקְהַל עַמִּים וְנָתַתִּי אֶת הָאָרֶץ הַזֹּאת לְזַרְעֲךָ אַחֲרֶיךָ אֲחֻזַּת עוֹלָם:  מ  וַיֹּאמֶר אֵלַי הִנְנִי מַפְרְךָ וְהִרְבִּיתִךָ וּנְתַתִּיךָ לִקְהַל עַמִּים וְנָתַתִּי אֶת הָאָרֶץ הַזֹּאת לְזַרְעֲךָ אַחֲרֶיךָ אֲחֻזַּת עוֹלָם:  ת  וַאֲמַר לִי הָא אֲנָא מַפְשִׁינָךְ וְאַסְגִּנָּךְ וְאֶתְּנִנָּךְ לְכִנְשַׁת שִׁבְטִין וְאֶתֵּן יָת אַרְעָא הָדָא לִבְנָיךְ בַּתְרָךְ אַחֲסָנַת עֲלָם:

{ה} וְעַתָּה שְׁנֵי בָנֶיךָ הַנּוֹלָדִים לְךָ בְּאֶרֶץ מִצְרַיִם עַד בֹּאִי אֵלֶיךָ מִצְרַיְמָה לִי הֵם אֶפְרַיִם וּמְנַשֶּׁה כִּרְאוּבֵן וְשִׁמְעוֹן יִהְיוּ לִי:  מ  וְעַתָּה שְׁנֵי בָנֶיךָ הַנּוֹלָדִים לְךָ בְּאֶרֶץ מִצְרַיִם עַד בֹּאִי אֵלֶיךָ מִצְרַיְמָה לִי הֵם אֶפְרַיִם וּמְנַשֶּׁה כִּרְאוּבֵן וְשִׁמְעוֹן יִהְיוּ לִי:  ת  וּכְעַן תְּרֵין בְּנָיךְ דְּאִתְיְלִידוּ לָךְ בְּאַרְעָא דְמִצְרַיִם עַד מֵיתִי לְוָתָךְ לְמִצְרַיִם דִּילִי אִנּוּן אֶפְרַיִם וּמְנַשֶּׁה כִּרְאוּבֵן וְשִׁמְעוֹן יְהוֹן קֳדָמָי:

{ו} וּמוֹלַדְתְּךָ אֲשֶׁר הוֹלַדְתָּ אַחֲרֵיהֶם לְךָ יִהְיוּ עַל שֵׁם אֲחֵיהֶם יִקָּרְאוּ בְּנַחֲלָתָם:  מ  וּמוֹלַדְתְּךָ אֲשֶׁר הוֹלַדְתָּ אַחֲרֵיהֶם לְךָ יִהְיוּ עַל שֵׁם אֲחֵיהֶם יִקָּרְאוּ בְּנַחֲלָתָם:  ת  וּבְנָיךְ דִּי תוֹלִיד בַּתְרֵיהוֹן דִּילָךְ יְהוֹן עַל שׁוּם אֲחוּהוֹן יִתְקְרוּן בְּאַחֲסַנְתְּהוֹן:

{ז} וַאֲנִי בְּבֹאִי מִפַּדָּן מֵתָה עָלַי רָחֵל בְּאֶרֶץ כְּנַעַן בַּדֶּרֶךְ בְּעוֹד כִּבְרַת אֶרֶץ לָבֹא אֶפְרָתָה וָאֶקְבְּרֶהָ שָּׁם בְּדֶרֶךְ אֶפְרָת הִוא בֵּית לָחֶם:  מ  וַאֲנִי בְּבֹאִי מִפַּדָּן מֵתָה עָלַי רָחֵל בְּאֶרֶץ כְּנַעַן בַּדֶּרֶךְ בְּעוֹד כִּבְרַת אֶרֶץ לָבֹא אֶפְרָתָה וָאֶקְבְּרֶהָ שָּׁם בְּדֶרֶךְ אֶפְרָת הִוא בֵּית לָחֶם:  ת  וַאֲנָא בְּמֵיתִי מִפַּדָן מִיתַת עָלַי רָחֵל בְּאַרְעָא דִּכְנַעַן בְּאָרְחָא בְּעוֹד כְּרוּבָא דְאַרְעָא לְמֵיעַל לְאֶפְרָת וּקְבָרִתַּהּ תַּמָּן בְּאֹרַח אֶפְרָת הִיא בֵּית לָחֶם:

{ח} וַיַּרְא יִשְׂרָאֵל אֶת בְּנֵי יוֹסֵף וַיֹּאמֶר מִי אֵלֶּה:  מ  וַיַּרְא יִשְׂרָאֵל אֶת בְּנֵי יוֹסֵף וַיֹּאמֶר מִי אֵלֶּה:  ת  וַחֲזָא יִשְׂרָאֵל יָת בְּנֵי יוֹסֵף וַאֲמַר מַן אִלֵּין:

{ט} וַיֹּאמֶר יוֹסֵף אֶל אָבִיו בָּנַי הֵם אֲשֶׁר נָתַן לִי אֱלֹהִים בָּזֶה וַיֹּאמַר קָחֶם נָא אֵלַי וַאֲבָרְכֵם:  מ  וַיֹּאמֶר יוֹסֵף אֶל אָבִיו בָּנַי הֵם אֲשֶׁר נָתַן לִי אֱלֹהִים בָּזֶה וַיֹּאמַר קָחֶם נָא אֵלַי וַאֲבָרְכֵם:  ת  וַאֲמַר יוֹסֵף לַאֲבוּהִי בְּנַי אִנּוּן דִּי יְהַב לִי יְיָ הָכָא וַאֲמַר קָרֵבִנּוּן כְּעַן לְוָתִי וְאֵבָרֵכִנּוּן:

{י}  שני  וְעֵינֵי יִשְׂרָאֵל כָּבְדוּ מִזֹּקֶן לֹא יוּכַל לִרְאוֹת וַיַּגֵּשׁ אֹתָם אֵלָיו וַיִּשַּׁק לָהֶם וַיְחַבֵּק לָהֶם:  מ  וְעֵינֵי יִשְׂרָאֵל כָּבְדוּ מִזֹּקֶן לֹא יוּכַל לִרְאוֹת וַיַּגֵּשׁ אֹתָם אֵלָיו וַיִּשַּׁק לָהֶם וַיְחַבֵּק לָהֶם:  ת  וְעֵינֵי יִשְׂרָאֵל יְקָרָן מִסֵּיבוּ לָא יִכּוּל לְמֶחֱזֵי וְקָרִיב יָתְהוֹן לְוָתֵהּ וְנַשִּׁיק לְהוֹן וְגַפִּיף לְהוֹן:

{יא} וַיֹּאמֶר יִשְׂרָאֵל אֶל יוֹסֵף רְאֹה פָנֶיךָ לֹא פִלָּלְתִּי וְהִנֵּה הֶרְאָה אֹתִי אֱלֹהִים גַּם אֶת זַרְעֶךָ:  מ  וַיֹּאמֶר יִשְׂרָאֵל אֶל יוֹסֵף רְאֹה פָנֶיךָ לֹא פִלָּלְתִּי וְהִנֵּה הֶרְאָה אֹתִי אֱלֹהִים גַּם אֶת זַרְעֶךָ:  ת  וַאֲמַר יִשְׂרָאֵל לְיוֹסֵף לְמֶחֱזֵי אַפָיךְ לָא סְבָרִית וְהָא אַחֲזִי יָתִי יְיָ אַף יָת בְּנָיךְ:

{יב} וַיּוֹצֵא יוֹסֵף אֹתָם מֵעִם בִּרְכָּיו וַיִּשְׁתַּחוּ לְאַפָּיו אָרְצָה:  מ  וַיּוֹצֵא יוֹסֵף אֹתָם מֵעִם בִּרְכָּיו וַיִּשְׁתַּחוּ לְאַפָּיו אָרְצָה:  ת  וְאַפֵּיק יוֹסֵף יָתְהוֹן מִן קֳדָמוֹהִי וּסְגִיד עַל אַפּוֹהִי עַל אַרְעָא:

{יג} וַיִּקַּח יוֹסֵף אֶת שְׁנֵיהֶם אֶת אֶפְרַיִם בִּימִינוֹ מִשְּׂמֹאל יִשְׂרָאֵל וְאֶת מְנַשֶּׁה בִשְׂמֹאלוֹ מִימִין יִשְׂרָאֵל וַיַּגֵּשׁ אֵלָיו:  מ  וַיִּקַּח יוֹסֵף אֶת שְׁנֵיהֶם אֶת אֶפְרַיִם בִּימִינוֹ מִשְּׂמֹאל יִשְׂרָאֵל וְאֶת מְנַשֶּׁה בִשְׂמֹאלוֹ מִימִין יִשְׂרָאֵל וַיַּגֵּשׁ אֵלָיו:  ת  וּדְבַר יוֹסֵף יָת תַּרְוֵיהוֹן יָת אֶפְרַיִם בִּימִינֵהּ מִשְׂמָאלָא דְיִשְׂרָאֵל וְיָת מְנַשֶּׁה בִשְׂמָאלֵהּ מִימִינָא דְיִשְׂרָאֵל וְקָרֵיב לְוָתֵהּ:

{יד} וַיִּשְׁלַח יִשְׂרָאֵל אֶת יְמִינוֹ וַיָּשֶׁת עַל רֹאשׁ אֶפְרַיִם וְהוּא הַצָּעִיר וְאֶת שְׂמֹאלוֹ עַל רֹאשׁ מְנַשֶּׁה שִׂכֵּל אֶת יָדָיו כִּי מְנַשֶּׁה הַבְּכוֹר:  מ  וַיִּשְׁלַח יִשְׂרָאֵל אֶת יְמִינוֹ וַיָּשֶׁת עַל רֹאשׁ אֶפְרַיִם וְהוּא הַצָּעִיר וְאֶת שְׂמֹאלוֹ עַל רֹאשׁ מְנַשֶּׁה שִׂכֵּל אֶת יָדָיו כִּי מְנַשֶּׁה הַבְּכוֹר:  ת  וְאוֹשִׁיט יִשְׂרָאֵל יָת יְמִינֵהּ וְשַׁוִּי עַל רֵישָׁא דְאֶפְרַיִם וְהוּא זְעֵירָא וְיָת שְׂמָאלֵהּ עַל רֵישָׁא דִּמְנַשֶּׁה אַחְכִּמִינּוּן לִידוֹהִי אֲרֵי מְנַשֶּׁה בּוּכְרָא:

{טו} וַיְבָרֶךְ אֶת יוֹסֵף וַיֹּאמַר הָאֱלֹהִים אֲשֶׁר הִתְהַלְּכוּ אֲבֹתַי לְפָנָיו אַבְרָהָם וְיִצְחָק הָאֱלֹהִים הָרֹעֶה אֹתִי מֵעוֹדִי עַד הַיּוֹם הַזֶּה:  מ  וַיְבָרֶךְ אֶת יוֹסֵף וַיֹּאמַר הָאֱלֹהִים אֲשֶׁר הִתְהַלְּכוּ אֲבֹתַי לְפָנָיו אַבְרָהָם וְיִצְחָק הָאֱלֹהִים הָרֹעֶה אֹתִי מֵעוֹדִי עַד הַיּוֹם הַזֶּה:  ת  וּבְרִיךְ יָת יוֹסֵף וַאֲמָר יְיָ דִי פְלָחוּ אֲבָהָתַי קֳדָמוֹהִי אַבְרָהָם וְיִצְחָק יְיָ דְזַן יָתִי מִדְאִיתָנִי עַד יוֹמָא הָדֵין:

{טז} הַמַּלְאָךְ הַגֹּאֵל אֹתִי מִכָּל רָע יְבָרֵךְ אֶת הַנְּעָרִים וְיִקָּרֵא בָהֶם שְׁמִי וְשֵׁם אֲבֹתַי אַבְרָהָם וְיִצְחָק וְיִדְגּוּ לָרֹב בְּקֶרֶב הָאָרֶץ:  מ  הַמַּלְאָךְ הַגֹּאֵל אֹתִי מִכָּל רָע יְבָרֵךְ אֶת הַנְּעָרִים וְיִקָּרֵא בָהֶם שְׁמִי וְשֵׁם אֲבֹתַי אַבְרָהָם וְיִצְחָק וְיִדְגּוּ לָרֹב בְּקֶרֶב הָאָרֶץ:  ת  מַלְאָכָא דִי פְרַק יָתִי מִכָּל בִּישָׁא יְבָרֵךְ יָת עוּלֵמַיָּא וְיִתְקְרֵי בְהוֹן שְׁמִי וְשׁוּם אֲבָהָתַי אַבְרָהָם וְיִצְחָק וּכְנוּנֵי יַמָּא יִסְגוּן בְּגוֹ בְנֵי אֱנָשָׁא עַל אַרְעָא:

{יז}  שלישי  וַיַּרְא יוֹסֵף כִּי יָשִׁית אָבִיו יַד יְמִינוֹ עַל רֹאשׁ אֶפְרַיִם וַיֵּרַע בְּעֵינָיו וַיִּתְמֹךְ יַד אָבִיו לְהָסִיר אֹתָהּ מֵעַל רֹאשׁ אֶפְרַיִם עַל רֹאשׁ מְנַשֶּׁה:  מ  וַיַּרְא יוֹסֵף כִּי יָשִׁית אָבִיו יַד יְמִינוֹ עַל רֹאשׁ אֶפְרַיִם וַיֵּרַע בְּעֵינָיו וַיִּתְמֹךְ יַד אָבִיו לְהָסִיר אֹתָהּ מֵעַל רֹאשׁ אֶפְרַיִם עַל רֹאשׁ מְנַשֶּׁה:  ת  וַחֲזָא יוֹסֵף אֲרֵי שַׁוִּי אָבוּהִי יַד יְמִינֵהּ עַל רֵישָׁא דְאֶפְרַיִם וּבְאֵישׁ בְּעֵינוֹהִי וְסָעֵד יְדָא דְאָבוּהִי לְאַעֲדָאָה יָתַהּ מֵעַל רֵישָׁא דְאֶפְרַיִם לְאַנָחוּתַהּ עַל רֵישָׁא דִמְנַשֶּׁה:

{יח} וַיֹּאמֶר יוֹסֵף אֶל אָבִיו לֹא כֵן אָבִי כִּי זֶה הַבְּכֹר שִׂים יְמִינְךָ עַל רֹאשׁוֹ:  מ  וַיֹּאמֶר יוֹסֵף אֶל אָבִיו לֹא כֵן אָבִי כִּי זֶה הַבְּכֹר שִׂים יְמִינְךָ עַל רֹאשׁוֹ:  ת  וַאֲמַר יוֹסֵף לְאָבוּהִי לָא כֵן אַבָּא אֲרֵי דֵין בּוּכְרָא שַׁו יְמִינָךְ עַל רֵישֵׁהּ:

{יט} וַיְמָאֵן אָבִיו וַיֹּאמֶר יָדַעְתִּי בְנִי יָדַעְתִּי גַּם הוּא יִהְיֶה לְּעָם וְגַם הוּא יִגְדָּל וְאוּלָם אָחִיו הַקָּטֹן יִגְדַּל מִמֶּנּוּ וְזַרְעוֹ יִהְיֶה מְלֹא הַגּוֹיִם:  מ  וַיְמָאֵן אָבִיו וַיֹּאמֶר יָדַעְתִּי בְנִי יָדַעְתִּי גַּם הוּא יִהְיֶה לְּעָם וְגַם הוּא יִגְדָּל וְאוּלָם אָחִיו הַקָּטֹן יִגְדַּל מִמֶּנּוּ וְזַרְעוֹ יִהְיֶה מְלֹא הַגּוֹיִם:  ת  וְסָרֵיב אָבוּהִי וַאֲמַר יְדַעְנָא בְרִי יְדַעְנָא אַף הוּא יְהֵי לְעַמָּא וְאַף הוּא יִסְגֵּי וּבְרַם אָחוּהִי זְעִירָא יִסְגֵּי מִנֵּהּ וּבְנוֹהִי יְהוֹן שַׁלִּיטִין בְּעַמְמַיָּא:

{כ} וַיְבָרְכֵם בַּיּוֹם הַהוּא לֵאמוֹר בְּךָ יְבָרֵךְ יִשְׂרָאֵל לֵאמֹר יְשִׂמְךָ אֱלֹהִים כְּאֶפְרַיִם וְכִמְנַשֶּׁה וַיָּשֶׂם אֶת אֶפְרַיִם לִפְנֵי מְנַשֶּׁה:  מ  וַיְבָרְכֵם בַּיּוֹם הַהוּא לֵאמוֹר בְּךָ יְבָרֵךְ יִשְׂרָאֵל לֵאמֹר יְשִׂמְךָ אֱלֹהִים כְּאֶפְרַיִם וְכִמְנַשֶּׁה וַיָּשֶׂם אֶת אֶפְרַיִם לִפְנֵי מְנַשֶּׁה:  ת  וּבְרֵיכִנּוּן בְּיוֹמָא הַהוּא לְמֵימַר בָּךְ יְבָרֵךְ יִשְׂרָאֵל לְמֵימַר יְשַׁוִּינָךְ יְיָ כְּאֶפְרַיִם וְכִמְנַשֶּׁה וְשַׁוִּי יָת אֶפְרַיִם קֳדָם מְנַשֶּׁה:

{כא} וַיֹּאמֶר יִשְׂרָאֵל אֶל יוֹסֵף הִנֵּה אָנֹכִי מֵת וְהָיָה אֱלֹהִים עִמָּכֶם וְהֵשִׁיב אֶתְכֶם אֶל אֶרֶץ אֲבֹתֵיכֶם:  מ  וַיֹּאמֶר יִשְׂרָאֵל אֶל יוֹסֵף הִנֵּה אָנֹכִי מֵת וְהָיָה אֱלֹהִים עִמָּכֶם וְהֵשִׁיב אֶתְכֶם אֶל אֶרֶץ אֲבֹתֵיכֶם:  ת  וַאֲמַר יִשְׂרָאֵל לְיוֹסֵף הָא אֲנָא מָאִית וִיהֵי מֵימְרָא דַיְיָ בְּסַעְדְכוֹן וְיָתֵיב יָתְכוֹן לְאַרְעָא דַאֲבָהַתְכוֹן:

{כב} וַאֲנִי נָתַתִּי לְךָ שְׁכֶם אַחַד עַל אַחֶיךָ אֲשֶׁר לָקַחְתִּי מִיַּד הָאֱמֹרִי בְּחַרְבִּי וּבְקַשְׁתִּי:  מ  וַאֲנִי נָתַתִּי לְךָ שְׁכֶם אַחַד עַל אַחֶיךָ אֲשֶׁר לָקַחְתִּי מִיַּד הָאֱמֹרִי בְּחַרְבִּי וּבְקַשְׁתִּי:  ת  וַאֲנָא יְהָבִית לָךְ חוּלַק חַד יַתִּיר עַל אַחָיךְ דִּי נְסֵיבִית מִידָא דֶאֱמוֹרָאָה בִּצְלוֹתִי וּבְבָעוּתִי: (פ)




בראשית פרק-מט

{א}  רביעי  וַיִּקְרָא יַעֲקֹב אֶל בָּנָיו וַיֹּאמֶר הֵאָסְפוּ וְאַגִּידָה לָכֶם אֵת אֲשֶׁר יִקְרָא אֶתְכֶם בְּאַחֲרִית הַיָּמִים:  מ  וַיִּקְרָא יַעֲקֹב אֶל בָּנָיו וַיֹּאמֶר הֵאָסְפוּ וְאַגִּידָה לָכֶם אֵת אֲשֶׁר יִקְרָא אֶתְכֶם בְּאַחֲרִית הַיָּמִים:  ת  וּקְרָא יַעֲקֹב לִבְנוֹהִי וַאֲמַר אִתְכַּנָשׁוּ וַאֲחַוֵּי לְכוֹן יָת דִּי יְעָרַע יָתְכוֹן בְּסוֹף יוֹמַיָּא:

{ב} הִקָּבְצוּ וְשִׁמְעוּ בְּנֵי יַעֲקֹב וְשִׁמְעוּ אֶל יִשְׂרָאֵל אֲבִיכֶם:  מ  הִקָּבְצוּ וְשִׁמְעוּ בְּנֵי יַעֲקֹב וְשִׁמְעוּ אֶל יִשְׂרָאֵל אֲבִיכֶם:  ת  אִתְכַּנָּשׁוּ וּשְׁמָעוּ בְּנֵי יַעֲקֹב וְקַבִּילוּ אוּלְפַן מִן יִשְׂרָאֵל אֲבוּכוֹן:

{ג} רְאוּבֵן בְּכֹרִי אַתָּה כֹּחִי וְרֵאשִׁית אוֹנִי יֶתֶר שְׂאֵת וְיֶתֶר עָז:  מ  רְאוּבֵן בְּכֹרִי אַתָּה כֹּחִי וְרֵאשִׁית אוֹנִי יֶתֶר שְׂאֵת וְיֶתֶר עָז:  ת  רְאוּבֵן בּוּכְרִי אַתְּ חֵילִי וְרֵישׁ תָּקְפִּי לָךְ הֲוָה חָזֵי לְמִסַּב תְּלָתָא חוּלָקִין בְּכֵירוּתָא כְּהֻנְתָּא וּמַלְכוּתָא:

{ד} פַּחַז כַּמַּיִם אַל תּוֹתַר כִּי עָלִיתָ מִשְׁכְּבֵי אָבִיךָ אָז חִלַּלְתָּ יְצוּעִי עָלָה:  מ  פַּחַז כַּמַּיִם אַל תּוֹתַר כִּי עָלִיתָ מִשְׁכְּבֵי אָבִיךָ אָז חִלַּלְתָּ יְצוּעִי עָלָה:  ת  עַל דַאֲזַלְתָּא לָקֳבֵל אַפָּיךְ הָא כְמַיָּא בְּרַם לָא אַהֲנֵיתָא חוֹלָק יַתִּיר לָא תִסַּב אֲרֵי סְלֶקְתָּא בֵּית מִשְׁכָּבֵי אָבוּךְ בְּכֵן אֲחֶלְתָּא לְשִׁוּוּיִי בְּרִי סְלֶקְתָּא: (פ)

{ה} שִׁמְעוֹן וְלֵוִי אַחִים כְּלֵי חָמָס מְכֵרֹתֵיהֶם:  מ  שִׁמְעוֹן וְלֵוִי אַחִים כְּלֵי חָמָס מְכֵרֹתֵיהֶם:  ת  שִׁמְעוֹן וְלֵוִי אַחִין גַּבְרִין גִּבָּרִין בְּאַרַע תּוֹתָבוּתְהוֹן עֲבָדוּ גְּבוּרָא:

{ו} בְּסֹדָם אַל תָּבֹא נַפְשִׁי בִּקְהָלָם אַל תֵּחַד כְּבֹדִי כִּי בְאַפָּם הָרְגוּ אִישׁ וּבִרְצֹנָם עִקְּרוּ שׁוֹר:  מ  בְּסֹדָם אַל תָּבֹא נַפְשִׁי בִּקְהָלָם אַל תֵּחַד כְּבֹדִי כִּי בְאַפָּם הָרְגוּ אִישׁ וּבִרְצֹנָם עִקְּרוּ שׁוֹר:  ת  בְּרָזְהוֹן לָא הֲוַת נַפְשִׁי בְּאִתְכַּנְשֵׁהוֹן לִמְהַךְ לָא נְחָתִית מִן יְקָרִי אֲרֵי בְרָגְזְהוֹן קְטָלוּ קְטוֹל וּבִרְעוּתְהוֹן תַּרָעוּ שׁוּר סַנְּאָה:

{ז} אָרוּר אַפָּם כִּי עָז וְעֶבְרָתָם כִּי קָשָׁתָה אֲחַלְּקֵם בְּיַעֲקֹב וַאֲפִיצֵם בְּיִשְׂרָאֵל:  מ  אָרוּר אַפָּם כִּי עָז וְעֶבְרָתָם כִּי קָשָׁתָה אֲחַלְּקֵם בְּיַעֲקֹב וַאֲפִיצֵם בְּיִשְׂרָאֵל:  ת  לִיט רָגְזְהוֹן אֲרֵי תַקִיף וְחֵמַתְהוֹן אֲרֵי קַשְׁיָא אֲפַלְגִנוּן בְּיַעֲקֹב וְאֵבַדָרִנוּן בְּיִשְׂרָאֵל: (פ)

{ח} יְהוּדָה אַתָּה יוֹדוּךָ אַחֶיךָ יָדְךָ בְּעֹרֶף אֹיְבֶיךָ יִשְׁתַּחֲוּוּ לְךָ בְּנֵי אָבִיךָ:  מ  יְהוּדָה אַתָּה יוֹדוּךָ אַחֶיךָ יָדְךָ בְּעֹרֶף אֹיְבֶיךָ יִשְׁתַּחֲוּוּ לְךָ בְּנֵי אָבִיךָ:  ת  יְהוּדָה אַתְּ אוֹדֵיתָא וְלָא בְהֵיתְתָא בָּךְ יוֹדוּן אַחָיךְ יְדָךְ תִּתְקֵף עַל בַּעֲלֵי דְבָבָךְ יִתְבַּדְרוּן סַנְאָךְ יְהוֹן מְחַזְרִין קְדַּל קֳדָמָךְ וִיְהוֹן מְקַדְמִין לְמִשְׁאַל בִּשְׁלָמָךְ בְּנֵי אָבוּךְ:

{ט} גּוּר אַרְיֵה יְהוּדָה מִטֶּרֶף בְּנִי עָלִיתָ כָּרַע רָבַץ כְּאַרְיֵה וּכְלָבִיא מִי יְקִימֶנּוּ:  מ  גּוּר אַרְיֵה יְהוּדָה מִטֶּרֶף בְּנִי עָלִיתָ כָּרַע רָבַץ כְּאַרְיֵה וּכְלָבִיא מִי יְקִימֶנּוּ:  ת  שִׁלְטוֹן יְהֵי בְשֵׁרוּיָא וּבְסוֹפָא יִתְרַבָּא מַלְכָּא מִדְבֵית יְהוּדָה אֲרֵי מִדִין קַטְלָא בְּרִי נַפְשָׁךְ סְלֶקְתָּא יְנוּחַ יִשְׁרֵי בִתְקוֹף כְּאַרְיָא וּכְלֵיתָא וְלֵית מַלְכוּ דִתְזַעְזְעִנֵהּ:

{י} לֹא יָסוּר שֵׁבֶט מִיהוּדָה וּמְחֹקֵק מִבֵּין רַגְלָיו עַד כִּי יָבֹא (שילה) שִׁילוֹ וְלוֹ יִקְּהַת עַמִּים:  מ  לֹא יָסוּר שֵׁבֶט מִיהוּדָה וּמְחֹקֵק מִבֵּין רַגְלָיו עַד כִּי יָבֹא (שילה) שִׁילוֹ וְלוֹ יִקְּהַת עַמִּים:  ת  לָא יַעֲדֵי עָבֵד שָׁלְטַן מִדְבֵית יְהוּדָה וְסַפְרָא מִבְּנֵי בְנוֹהִי עַד עָלְמָא עַד דְיֵיתֵי מְשִׁיחָא דְדִילֵהּ הִיא מַלְכוּתָא וְלֵהּ יִשְׁתַּמְעוּן עַמְמַיָּא:

{יא} אֹסְרִי לַגֶּפֶן (עירה) עִירוֹ וְלַשֹּׂרֵקָה בְּנִי אֲתֹנוֹ כִּבֵּס בַּיַּיִן לְבֻשׁוֹ וּבְדַם עֲנָבִים (סותה) סוּתוֹ:  מ  אֹסְרִי לַגֶּפֶן (עירה) עִירוֹ וְלַשֹּׂרֵקָה בְּנִי אֲתֹנוֹ כִּבֵּס בַּיַּיִן לְבֻשׁוֹ וּבְדַם עֲנָבִים (סותה) סוּתוֹ:  ת  יַסְחַר יִשְׂרָאֵל לְקַרְתֵּהּ עַמָּא יִבְנוּן הֵיכְלֵהּ יְהוֹן צַדִּיקַיָּא סְחוֹר סְחוֹר לֵהּ וְעָבְדֵי אוֹרַיְתָא בְּאוּלְפַן עִמֵּהּ יְהֵי אַרְגְוָן טַב לְבוּשׁוֹהִי כְּסוּתֵהּ מֵילָא מֵילָא צְבַע זְהוֹרִי וְצִבְעוֹנִין:

{יב} חַכְלִילִי עֵינַיִם מִיָּיִן וּלְבֶן שִׁנַּיִם מֵחָלָב:  מ  חַכְלִילִי עֵינַיִם מִיָּיִן וּלְבֶן שִׁנַּיִם מֵחָלָב:  ת  ??? יְסַמְקוּן טּוּרוֹהִי בְּכַרְמוֹהִי יְטוּפוּן נַעֲווֹהִי בַּחֲמָר יְחַוְרָן בִּקְעֲתֵהּ בְּעִיבוּר וּבְעֶדְרֵי עָנָא: (פ)

{יג} זְבוּלֻן לְחוֹף יַמִּים יִשְׁכֹּן וְהוּא לְחוֹף אֳנִיּוֹת וְיַרְכָתוֹ עַל צִידֹן:  מ  זְבוּלֻן לְחוֹף יַמִּים יִשְׁכֹּן וְהוּא לְחוֹף אֳנִיּוֹת וְיַרְכָתוֹ עַל צִידֹן:  ת  זְבוּלֻן עַל סְפַר יַמְמַיָּא יִשְׁרֵי וְהוּא יְכַבֵּשׁ מְחוֹזִין בִּסְפִינַן וְטוּב יַמָּא יֵיכוּל וּתְחוּמֵהּ יְהֵי מָטֵי עַל צִידוֹן: (פ)

{יד} יִשָּׂשכָר חֲמֹר גָּרֶם רֹבֵץ בֵּין הַמִּשְׁפְּתָיִם:  מ  יִשָּׂשכָר חֲמֹר גָּרֶם רֹבֵץ בֵּין הַמִּשְׁפְּתָיִם:  ת  יִשָּׂשׂכָר עַתִּיר בְּנִכְסִין וְאַחֲסַנְתֵּהּ בֵּין תְּחוּמַיָּא:

{טו} וַיַּרְא מְנֻחָה כִּי טוֹב וְאֶת הָאָרֶץ כִּי נָעֵמָה וַיֵּט שִׁכְמוֹ לִסְבֹּל וַיְהִי לְמַס עֹבֵד:  מ  וַיַּרְא מְנֻחָה כִּי טוֹב וְאֶת הָאָרֶץ כִּי נָעֵמָה וַיֵּט שִׁכְמוֹ לִסְבֹּל וַיְהִי לְמַס עֹבֵד:  ת  וַחֲזָא חוּלָקָא אֲרֵי טַב וְיָת אַרְעָא אֲרֵי מְעַבְּדָא פֵירִין וִיכַבֵּשׁ מָחוֹזֵי עַמְמַיָּא וִישֵׁיצֵי יָת דַּיָרֵיהוֹן וּדְאִשְׁתַּאֲרוּן בְּהוֹן יְהוֹן לֵהּ פַּלְחִין וּמַסְּקֵי מַסִּין: (ס)

{טז} דָּן יָדִין עַמּוֹ כְּאַחַד שִׁבְטֵי יִשְׂרָאֵל:  מ  דָּן יָדִין עַמּוֹ כְּאַחַד שִׁבְטֵי יִשְׂרָאֵל:  ת  מִדְּבֵית דָּן יִתְבְּחַר וִיקוּם גַּבְרָא בְּיוֹמוֹהִי יִתְפְּרֵק עַמֵּהּ וּבִשְׁנוֹהִי יְנוּחוּן כַּחֲדָא שִׁבְטַּיָּא דְיִשְׂרָאֵל:

{יז} יְהִי דָן נָחָשׁ עֲלֵי דֶרֶךְ שְׁפִיפֹן עֲלֵי אֹרַח הַנֹּשֵׁךְ עִקְּבֵי סוּס וַיִּפֹּל רֹכְבוֹ אָחוֹר:  מ  יְהִי דָן נָחָשׁ עֲלֵי דֶרֶךְ שְׁפִיפֹן עֲלֵי אֹרַח הַנֹּשֵׁךְ עִקְּבֵי סוּס וַיִּפֹּל רֹכְבוֹ אָחוֹר:  ת  יְהֵי גַבְרָא דְיִתְבְּחַר וִיקוּם מִדְּבֵית דָּן אֵימְתֵהּ תִּתְרְמֵי עַל עַמְמַיָּא וּמְחָתֵהּ תִּתְקֵף בִּפְלִשְׁתָּאֵי כְּחִיוֵי חוּרְמָן יִשְׁרֵי עַל אָרְחָא וּכְפִתְנָא יִכְמוֹן עַל שְׁבִילָא יְקַטֵּיל גִּבָּרֵי מַשִׁרְיַת פְּלִשְׁתָּאֵי פָּרָשִׁין עִם רַגְלָאִין יְעַקֵר סוּסָוָן וּרְתִכִּין וִימַגֵּר רוֹכְבֵיהוֹן לַאֲחוֹרָא:

{יח} לִישׁוּעָתְךָ קִוִּיתִי יְהוָה:  מ  לִישׁוּעָתְךָ קִוִּיתִי יְהוָה:  ת  לְפוּרְקָנָךְ סַבָּרִית יְיָ: (ס)

{יט}  חמישי  גָּד גְּדוּד יְגוּדֶנּוּ וְהוּא יָגֻד עָקֵב:  מ  גָּד גְּדוּד יְגוּדֶנּוּ וְהוּא יָגֻד עָקֵב:  ת  מִדְבֵית גָד מַשִׁרְיַת מְזַיְנִין כַּד יַעַבְרוּן יָת יַרְדְנָא קֳדָם אֲחֵיהוֹן לִקְרָבָא וּבְנִכְסִין סַגִּיאִין יְתוּבוּן לְאַרְעֲהוֹן: (ס)

{כ} מֵאָשֵׁר שְׁמֵנָה לַחְמוֹ וְהוּא יִתֵּן מַעֲדַנֵּי מֶלֶךְ:  מ  מֵאָשֵׁר שְׁמֵנָה לַחְמוֹ וְהוּא יִתֵּן מַעֲדַנֵּי מֶלֶךְ:  ת  דְאָשֵׁר טָבָא אַרְעֵהּ וְהוּא מְרַבְיָא בְּתַפְנוּקֵי מַלְכִין: (ס)

{כא} נַפְתָּלִי אַיָּלָה שְׁלֻחָה הַנֹּתֵן אִמְרֵי שָׁפֶר:  מ  נַפְתָּלִי אַיָּלָה שְׁלֻחָה הַנֹּתֵן אִמְרֵי שָׁפֶר:  ת  נַפְתָּלִי בְּאַרַע טָבָא יִתְרְמֵי עַדְבֵהּ וְאַחְסַנְתֵּהּ תְּהֵי מְעַבְּדָא פֵירִין יְהוֹן מוֹדַן וּמְבָרְכִין עֲלֵיהוֹן: (ס)

{כב} בֵּן פֹּרָת יוֹסֵף בֵּן פֹּרָת עֲלֵי עָיִן בָּנוֹת צָעֲדָה עֲלֵי שׁוּר:  מ  בֵּן פֹּרָת יוֹסֵף בֵּן פֹּרָת עֲלֵי עָיִן בָּנוֹת צָעֲדָה עֲלֵי שׁוּר:  ת  בְּרִי דְיִסְגֵּי יוֹסֵף בְּרִי דְיִתְבָּרַךְ כְּגֹפֶן דִּנְצִב עַל עֵינָא דְמַיָּא תְּרֵין שִׁבְטִין יִפְּקוּן מִבְּנוֹהִי יְקַבְּלוּן חוּלָקָא וְאַחֲסַנְתָּא:

{כג} וַיְמָרֲרֻהוּ וָרֹבּוּ וַיִּשְׂטְמֻהוּ בַּעֲלֵי חִצִּים:  מ  וַיְמָרֲרֻהוּ וָרֹבּוּ וַיִּשְׂטְמֻהוּ בַּעֲלֵי חִצִּים:  ת  וְאִתְמַרְרוּ יָתֵהּ וְנַקְמוֹהִי וְאָעִיקוּ לֵהּ גֻבְרִין גִּבָּרִין בַּעֲלֵי פַלְגוּתָא:

{כד} וַתֵּשֶׁב בְּאֵיתָן קַשְׁתּוֹ וַיָּפֹזּוּ זְרֹעֵי יָדָיו מִידֵי אֲבִיר יַעֲקֹב מִשָּׁם רֹעֶה אֶבֶן יִשְׂרָאֵל:  מ  וַתֵּשֶׁב בְּאֵיתָן קַשְׁתּוֹ וַיָּפֹזּוּ זְרֹעֵי יָדָיו מִידֵי אֲבִיר יַעֲקֹב מִשָּׁם רֹעֶה אֶבֶן יִשְׂרָאֵל:  ת  וְתָבַת בְּהוֹן נְבִיאוּתֵהּ עַל דְּקַיֵּם אוֹרַיְתָא בְּסִתְרָא וְשַׁוִּי בְתוּקְפָא רוֹחֲצָנֵהּ בְּכֵן יִתְרְמָא דְהַב עַל דְּרָעוֹהִי וְאַחֲסִין מַלְכוּתָא וּתְקֵיף דָּא הֲוַת לֵהּ מִן קֳדָם אֵל תַּקִיפָא דְיַעֲקֹב דִּי בְמֵימְרֵהּ זַן אֲבָהָן וּבְנִין זַרְעָא דְיִשְׂרָאֵל:

{כה} מֵאֵל אָבִיךָ וְיַעְזְרֶךָּ וְאֵת שַׁדַּי וִיבָרְכֶךָּ בִּרְכֹת שָׁמַיִם מֵעָל בִּרְכֹת תְּהוֹם רֹבֶצֶת תָּחַת בִּרְכֹת שָׁדַיִם וָרָחַם:  מ  מֵאֵל אָבִיךָ וְיַעְזְרֶךָּ וְאֵת שַׁדַּי וִיבָרְכֶךָּ בִּרְכֹת שָׁמַיִם מֵעָל בִּרְכֹת תְּהוֹם רֹבֶצֶת תָּחַת בִּרְכֹת שָׁדַיִם וָרָחַם:  ת  מֵימַר אֱלָהָא דְאָבוּךְ יְהֵי בְסַעְדָךְ וְיָת שַׁדַּי וִיבָרְכִנָּךְ בִּרְכַן דְּנַחֲתָן מִטַּלָּא דִשְׁמַיָּא מִלְעֵלָא בִּרְכַן דְנַגְדַן מִמַעֲמַקֵי אַרְעָא מִלְרָע בִּרְכְּתָא דְאָבוּךְ וּדְאִמָּךְ:

{כו} בִּרְכֹת אָבִיךָ גָּבְרוּ עַל בִּרְכֹת הוֹרַי עַד תַּאֲוַת גִּבְעֹת עוֹלָם תִּהְיֶין לְרֹאשׁ יוֹסֵף וּלְקָדְקֹד נְזִיר אֶחָיו:  מ  בִּרְכֹת אָבִיךָ גָּבְרוּ עַל בִּרְכֹת הוֹרַי עַד תַּאֲוַת גִּבְעֹת עוֹלָם תִּהְיֶין לְרֹאשׁ יוֹסֵף וּלְקָדְקֹד נְזִיר אֶחָיו:  ת  בִּרְכְּתָא דְאָבוּךְ יִתּוֹסְפָן עַל בִּרְכְּתָא דִילִי בָּרִיכוּ אֲבָהָתַי דַחֲמִידוּ לְהֶן רַבְרְבַיָּא דְמִן עָלְמָא יֶהֶוְיַן כָּל אִלֵּין לְרֵישָׁא דְיוֹסֵף וּלְגַבְרָא פָּרִישָׁא דַאֲחוֹהִי: (פ)

{כז}  שישי  בִּנְיָמִין זְאֵב יִטְרָף בַּבֹּקֶר יֹאכַל עַד וְלָעֶרֶב יְחַלֵּק שָׁלָל:  מ  בִּנְיָמִין זְאֵב יִטְרָף בַּבֹּקֶר יֹאכַל עַד וְלָעֶרֶב יְחַלֵּק שָׁלָל:  ת  בִּנְיָמִין בְּאַרְעֵהּ תִּשְׁרֵי שְׁכִנְתָּא וּבְאַחֲסַנְתֵּהּ יִתְבְּנֵי מַקְדְשָׁא בְּצַפְרָא וּבְפַנְיָא יְהוֹן מְקָרְבִין כַּהֲנַיָּא קָרְבָּנָא וּבְעִדַּן רַמְשָׁא יְהוֹן מְפַלְגִין מוֹתַר חוּלְקְהוֹן מִשְׁאַר קָדְשַׁיָּא:

{כח} כָּל אֵלֶּה שִׁבְטֵי יִשְׂרָאֵל שְׁנֵים עָשָׂר וְזֹאת אֲשֶׁר דִּבֶּר לָהֶם אֲבִיהֶם וַיְבָרֶךְ אוֹתָם אִישׁ אֲשֶׁר כְּבִרְכָתוֹ בֵּרַךְ אֹתָם:  מ  כָּל אֵלֶּה שִׁבְטֵי יִשְׂרָאֵל שְׁנֵים עָשָׂר וְזֹאת אֲשֶׁר דִּבֶּר לָהֶם אֲבִיהֶם וַיְבָרֶךְ אוֹתָם אִישׁ אֲשֶׁר כְּבִרְכָתוֹ בֵּרַךְ אֹתָם:  ת  כָּל אִלֵּין שִׁבְטַיָּא דְיִשְׂרָאֵל תְּרֵין עֲסַר וְדָא דְמַלֵּיל לְהוֹן אֲבוּהוֹן וּבָרִיךְ יָתְהוֹן גְבַר דִּי כְבִרְכְּתֵהּ בָּרֵךְ יָתְהוֹן:

{כט} וַיְצַו אוֹתָם וַיֹּאמֶר אֲלֵהֶם אֲנִי נֶאֱסָף אֶל עַמִּי קִבְרוּ אֹתִי אֶל אֲבֹתָי אֶל הַמְּעָרָה אֲשֶׁר בִּשְׂדֵה עֶפְרוֹן הַחִתִּי:  מ  וַיְצַו אוֹתָם וַיֹּאמֶר אֲלֵהֶם אֲנִי נֶאֱסָף אֶל עַמִּי קִבְרוּ אֹתִי אֶל אֲבֹתָי אֶל הַמְּעָרָה אֲשֶׁר בִּשְׂדֵה עֶפְרוֹן הַחִתִּי:  ת  וּפַקֵד יָתְהוֹן וַאֲמַר לְהוֹן אֲנָא מִתְכְּנֵשׁ לְעַמִּי קְבָרוּ יָתִי לְוַת אֲבָהָתִי בִּמְעַרְתָּא דִי בַחֲקַל עֶפְרוֹן חִתָּאָה:

{ל} בַּמְּעָרָה אֲשֶׁר בִּשְׂדֵה הַמַּכְפֵּלָה אֲשֶׁר עַל פְּנֵי מַמְרֵא בְּאֶרֶץ כְּנָעַן אֲשֶׁר קָנָה אַבְרָהָם אֶת הַשָּׂדֶה מֵאֵת עֶפְרֹן הַחִתִּי לַאֲחֻזַּת קָבֶר:  מ  בַּמְּעָרָה אֲשֶׁר בִּשְׂדֵה הַמַּכְפֵּלָה אֲשֶׁר עַל פְּנֵי מַמְרֵא בְּאֶרֶץ כְּנָעַן אֲשֶׁר קָנָה אַבְרָהָם אֶת הַשָּׂדֶה מֵאֵת עֶפְרֹן הַחִתִּי לַאֲחֻזַּת קָבֶר:  ת  בִּמְעַרְתָּא דִי בַחֲקַל כָּפֶלְתָּא דִי עַל אַפֵּי מַמְרֵא בְּאַרְעָא דִכְנָעַן דִי זְבַן אַבְרָהָם יָת חַקְלָא מִן עֶפְרֹן חִתָּאָה לְאַחֲסָנַת קְבוּרָא:

{לא} שָׁמָּה קָבְרוּ אֶת אַבְרָהָם וְאֵת שָׂרָה אִשְׁתּוֹ שָׁמָּה קָבְרוּ אֶת יִצְחָק וְאֵת רִבְקָה אִשְׁתּוֹ וְשָׁמָּה קָבַרְתִּי אֶת לֵאָה:  מ  שָׁמָּה קָבְרוּ אֶת אַבְרָהָם וְאֵת שָׂרָה אִשְׁתּוֹ שָׁמָּה קָבְרוּ אֶת יִצְחָק וְאֵת רִבְקָה אִשְׁתּוֹ וְשָׁמָּה קָבַרְתִּי אֶת לֵאָה:  ת  תַּמָּן קְבָרוּ יָת אַבְרָהָם וְיָת שָׂרָה אִתְּתֵהּ תַּמָּן קְבָרוּ יָת יִצְחָק וְיָת רִבְקָה אִתְּתֵהּ וְתַמָּן קְבָרִית יָת לֵאָה:

{לב} מִקְנֵה הַשָּׂדֶה וְהַמְּעָרָה אֲשֶׁר בּוֹ מֵאֵת בְּנֵי חֵת:  מ  מִקְנֵה הַשָּׂדֶה וְהַמְּעָרָה אֲשֶׁר בּוֹ מֵאֵת בְּנֵי חֵת:  ת  זְבִינֵי חַקְלָא וּמְעַרְתָּא דִי בֵהּ מִן בְּנֵי חִתָּאָהּ:

{לג} וַיְכַל יַעֲקֹב לְצַוֹּת אֶת בָּנָיו וַיֶּאֱסֹף רַגְלָיו אֶל הַמִּטָּה וַיִּגְוַע וַיֵּאָסֶף אֶל עַמָּיו:  מ  וַיְכַל יַעֲקֹב לְצַוֹּת אֶת בָּנָיו וַיֶּאֱסֹף רַגְלָיו אֶל הַמִּטָּה וַיִּגְוַע וַיֵּאָסֶף אֶל עַמָּיו:  ת  וְשֵׁיצֵי יַעֲקֹב לְפַקָּדָא יָת בְּנוֹהִי וּכְנַשׁ רַגְלוֹהִי לְעַרְסָא וְאִתְנְגִיד וְאִתְכְּנִישׁ לְעַמֵּהּ:




בראשית פרק-נ

{א} וַיִּפֹּל יוֹסֵף עַל פְּנֵי אָבִיו וַיֵּבְךְּ עָלָיו וַיִּשַּׁק לוֹ:  מ  וַיִּפֹּל יוֹסֵף עַל פְּנֵי אָבִיו וַיֵּבְךְּ עָלָיו וַיִּשַּׁק לוֹ:  ת  וּנְפַל יוֹסֵף עַל אַפֵּי אָבוּהִי וּבְכָא עֲלוֹהִי וּנְשַׁק לֵהּ:

{ב} וַיְצַו יוֹסֵף אֶת עֲבָדָיו אֶת הָרֹפְאִים לַחֲנֹט אֶת אָבִיו וַיַּחַנְטוּ הָרֹפְאִים אֶת יִשְׂרָאֵל:  מ  וַיְצַו יוֹסֵף אֶת עֲבָדָיו אֶת הָרֹפְאִים לַחֲנֹט אֶת אָבִיו וַיַּחַנְטוּ הָרֹפְאִים אֶת יִשְׂרָאֵל:  ת  וּפַקֵּיד יוֹסֵף יָת עַבְדוֹהִי יָת אַסְוָתָא לְמֶחְנַט יָת אָבוּהִי וְחָנְטוּ אַסְוָתָא יָת יִשְׂרָאֵל:

{ג} וַיִּמְלְאוּ לוֹ אַרְבָּעִים יוֹם כִּי כֵּן יִמְלְאוּ יְמֵי הַחֲנֻטִים וַיִּבְכּוּ אֹתוֹ מִצְרַיִם שִׁבְעִים יוֹם:  מ  וַיִּמְלְאוּ לוֹ אַרְבָּעִים יוֹם כִּי כֵּן יִמְלְאוּ יְמֵי הַחֲנֻטִים וַיִּבְכּוּ אֹתוֹ מִצְרַיִם שִׁבְעִים יוֹם:  ת  וּשְׁלִימוּ לֵהּ אַרְבְּעִין יוֹמִין אֲרֵי כֵּן שְׁלִימוּ יוֹמֵי חֲנִיטַיָּא וּבְכוֹ יָתֵהּ מִצְרָאֵי שַׁבְעִין יוֹמִין:

{ד} וַיַּעַבְרוּ יְמֵי בְכִיתוֹ וַיְדַבֵּר יוֹסֵף אֶל בֵּית פַּרְעֹה לֵאמֹר אִם נָא מָצָאתִי חֵן בְּעֵינֵיכֶם דַּבְּרוּ נָא בְּאָזְנֵי פַרְעֹה לֵאמֹר:  מ  וַיַּעַבְרוּ יְמֵי בְכִיתוֹ וַיְדַבֵּר יוֹסֵף אֶל בֵּית פַּרְעֹה לֵאמֹר אִם נָא מָצָאתִי חֵן בְּעֵינֵיכֶם דַּבְּרוּ נָא בְּאָזְנֵי פַרְעֹה לֵאמֹר:  ת  וַעֲבָרוּ יוֹמֵי בְכִיתֵהּ וּמַלִיל יוֹסֵף עִם בֵּית פַּרְעֹה לְמֵימָר אִם כְּעַן אַשְׁכָּחִית רַחֲמִין בְּעֵינֵיכוֹן מַלִּילוּ כְעַן קֳדָם פַּרְעֹה לְמֵימָר:

{ה} אָבִי הִשְׁבִּיעַנִי לֵאמֹר הִנֵּה אָנֹכִי מֵת בְּקִבְרִי אֲשֶׁר כָּרִיתִי לִי בְּאֶרֶץ כְּנַעַן שָׁמָּה תִּקְבְּרֵנִי וְעַתָּה אֶעֱלֶה נָּא וְאֶקְבְּרָה אֶת אָבִי וְאָשׁוּבָה:  מ  אָבִי הִשְׁבִּיעַנִי לֵאמֹר הִנֵּה אָנֹכִי מֵת בְּקִבְרִי אֲשֶׁר כָּרִיתִי לִי בְּאֶרֶץ כְּנַעַן שָׁמָּה תִּקְבְּרֵנִי וְעַתָּה אֶעֱלֶה נָּא וְאֶקְבְּרָה אֶת אָבִי וְאָשׁוּבָה:  ת  אַבָּא קַיֵּים עָלַי לְמֵימָר הָא אֲנָא מָאִית בְּקִבְרִי דְאַתְקֵנִית לִי בְּאַרְעָא דִכְנָעַן תַּמָּן תִּקְבְּרִנָנִי וּכְעַן אֶסַּק כְּעַן וְאֶקְבַּר יָת אַבָּא וְאֵ יתוּב:

{ו} וַיֹּאמֶר פַּרְעֹה עֲלֵה וּקְבֹר אֶת אָבִיךָ כַּאֲשֶׁר הִשְׁבִּיעֶךָ:  מ  וַיֹּאמֶר פַּרְעֹה עֲלֵה וּקְבֹר אֶת אָבִיךָ כַּאֲשֶׁר הִשְׁבִּיעֶךָ:  ת  וַאֲמַר פַּרְעֹה סַק וּקְבוֹר יָת אֲבוּךְ כְּמָא דְקַיֵּים עֲלָךְ:

{ז} וַיַּעַל יוֹסֵף לִקְבֹּר אֶת אָבִיו וַיַּעֲלוּ אִתּוֹ כָּל עַבְדֵי פַרְעֹה זִקְנֵי בֵיתוֹ וְכֹל זִקְנֵי אֶרֶץ מִצְרָיִם:  מ  וַיַּעַל יוֹסֵף לִקְבֹּר אֶת אָבִיו וַיַּעֲלוּ אִתּוֹ כָּל עַבְדֵי פַרְעֹה זִקְנֵי בֵיתוֹ וְכֹל זִקְנֵי אֶרֶץ מִצְרָיִם:  ת  וּסְלֵיק יוֹסֵף לְמִקְבַּר יָת אָבוּהִי וּסְלִיקוּ עִמֵּהּ כָּל עַבְדֵי פַרְעֹה סָבֵי בֵיתֵהּ וְכֹל סָבֵי אַרְעָא דְמִצְרָיִם:

{ח} וְכֹל בֵּית יוֹסֵף וְאֶחָיו וּבֵית אָבִיו רַק טַפָּם וְצֹאנָם וּבְקָרָם עָזְבוּ בְּאֶרֶץ גֹּשֶׁן:  מ  וְכֹל בֵּית יוֹסֵף וְאֶחָיו וּבֵית אָבִיו רַק טַפָּם וְצֹאנָם וּבְקָרָם עָזְבוּ בְּאֶרֶץ גֹּשֶׁן:  ת  וְכֹל בֵּית יוֹסֵף וַאֲחוֹהִי וּבֵית אָבוּהִי לְחוֹד טַפְלְהוֹן וְעָנְהוֹן וְתוֹרֵיהוֹן שְׁבָקוּ בְּאַרְעָא דְגשֶׁן:

{ט} וַיַּעַל עִמּוֹ גַּם רֶכֶב גַּם פָּרָשִׁים וַיְהִי הַמַּחֲנֶה כָּבֵד מְאֹד:  מ  וַיַּעַל עִמּוֹ גַּם רֶכֶב גַּם פָּרָשִׁים וַיְהִי הַמַּחֲנֶה כָּבֵד מְאֹד:  ת  וּסְלִיקוּ עִמֵּהּ אַף רְתִכִּין אַף פָּרָשִׁין וַהֲוָה מַשְׁרִיתָא סַגִּיאָה לַחֲדָא:

{י} וַיָּבֹאוּ עַד גֹּרֶן הָאָטָד אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן וַיִּסְפְּדוּ שָׁם מִסְפֵּד גָּדוֹל וְכָבֵד מְאֹד וַיַּעַשׂ לְאָבִיו אֵבֶל שִׁבְעַת יָמִים:  מ  וַיָּבֹאוּ עַד גֹּרֶן הָאָטָד אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן וַיִּסְפְּדוּ שָׁם מִסְפֵּד גָּדוֹל וְכָבֵד מְאֹד וַיַּעַשׂ לְאָבִיו אֵבֶל שִׁבְעַת יָמִים:  ת  וַאֲתוֹ עַד בֵּית אִדְרֵי דְאָטָד דִּי בְּעִבְרָא דְיַרְדְנָא וּסְפָדוּ תַמָּן מִסְפֵּד רַב וְתַקִּיף לַחֲדָא וְעָבַד לַאֲבוּהִי אֶבְלָא שַׁבְעָא יוֹמִין:

{יא} וַיַּרְא יוֹשֵׁב הָאָרֶץ הַכְּנַעֲנִי אֶת הָאֵבֶל בְּגֹרֶן הָאָטָד וַיֹּאמְרוּ אֵבֶל כָּבֵד זֶה לְמִצְרָיִם עַל כֵּן קָרָא שְׁמָהּ אָבֵל מִצְרַיִם אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן:  מ  וַיַּרְא יוֹשֵׁב הָאָרֶץ הַכְּנַעֲנִי אֶת הָאֵבֶל בְּגֹרֶן הָאָטָד וַיֹּאמְרוּ אֵבֶל כָּבֵד זֶה לְמִצְרָיִם עַל כֵּן קָרָא שְׁמָהּ אָבֵל מִצְרַיִם אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן:  ת  וַחֲזָא יָתֵב אַרְעָא כְנַעֲנָאָה יָת אֶבְלָא בְּבֵית אִדְרֵי דְאָטָד וַאֲמָרוּ אָבֵל תַּקִּיף דֵּין לְמִצְרָאֵי עַל כֵּן קְרָא שְׁמַהּ אָבֵל מִצְרַיִם דִּי בְּעִבְרָא דְיַרְדְנָא:

{יב} וַיַּעֲשׂוּ בָנָיו לוֹ כֵּן כַּאֲשֶׁר צִוָּם:  מ  וַיַּעֲשׂוּ בָנָיו לוֹ כֵּן כַּאֲשֶׁר צִוָּם:  ת  וַעֲבָדוּ בְּנוֹהִי לֵהּ כֵּן כְּמָא דִי פַּקֵדִנּוּן:

{יג} וַיִּשְׂאוּ אֹתוֹ בָנָיו אַרְצָה כְּנַעַן וַיִּקְבְּרוּ אֹתוֹ בִּמְעָרַת שְׂדֵה הַמַּכְפֵּלָה אֲשֶׁר קָנָה אַבְרָהָם אֶת הַשָּׂדֶה לַאֲחֻזַּת קֶבֶר מֵאֵת עֶפְרֹן הַחִתִּי עַל פְּנֵי מַמְרֵא:  מ  וַיִּשְׂאוּ אֹתוֹ בָנָיו אַרְצָה כְּנַעַן וַיִּקְבְּרוּ אֹתוֹ בִּמְעָרַת שְׂדֵה הַמַּכְפֵּלָה אֲשֶׁר קָנָה אַבְרָהָם אֶת הַשָּׂדֶה לַאֲחֻזַּת קֶבֶר מֵאֵת עֶפְרֹן הַחִתִּי עַל פְּנֵי מַמְרֵא:  ת  וּנְטָלוּ יָתֵהּ בְּנוֹהִי לְאַרְעָא דִכְנַעַן וּקְבָרוּ יָתֵהּ בִּמְעָרַת חֲקַל כָּפֶלְתָּא דִי זְבַן אַבְרָהָם יָת חַקְלָא לְאַחֲסָנַת קְבוּרָא מִן עֶפְרֹן חִתָּאָה עַל אַפֵּי מַמְרֵא:

{יד} וַיָּשָׁב יוֹסֵף מִצְרַיְמָה הוּא וְאֶחָיו וְכָל הָעֹלִים אִתּוֹ לִקְבֹּר אֶת אָבִיו אַחֲרֵי קָבְרוֹ אֶת אָבִיו:  מ  וַיָּשָׁב יוֹסֵף מִצְרַיְמָה הוּא וְאֶחָיו וְכָל הָעֹלִים אִתּוֹ לִקְבֹּר אֶת אָבִיו אַחֲרֵי קָבְרוֹ אֶת אָבִיו:  ת  וְתַב יוֹסֵף לְמִצְרַיִם הוּא וַאֲחוֹהִי וְכָל דִּסְלִיקוּ עִמֵּהּ לְמִקְבַּר יָת אָבוּהִי בָּתַר דִּקְבַר יָת אָבוּהִי:

{טו} וַיִּרְאוּ אֲחֵי יוֹסֵף כִּי מֵת אֲבִיהֶם וַיֹּאמְרוּ לוּ יִשְׂטְמֵנוּ יוֹסֵף וְהָשֵׁב יָשִׁיב לָנוּ אֵת כָּל הָרָעָה אֲשֶׁר גָּמַלְנוּ אֹתוֹ:  מ  וַיִּרְאוּ אֲחֵי יוֹסֵף כִּי מֵת אֲבִיהֶם וַיֹּאמְרוּ לוּ יִשְׂטְמֵנוּ יוֹסֵף וְהָשֵׁב יָשִׁיב לָנוּ אֵת כָּל הָרָעָה אֲשֶׁר גָּמַלְנוּ אֹתוֹ:  ת  וַחֲזוֹ אֲחֵי יוֹסֵף אֲרֵי מִית אֲבוּהוֹן וַאֲמָרוּ דִּלְמָא יִטַּר לָנָא דְבָבוּ יוֹסֵף וְאֲתָבָא יָתֵיב לָנָא יָת כָּל בִּישְׁתָא דִי גְמַלְנָא יָתֵהּ:

{טז} וַיְצַוּוּ אֶל יוֹסֵף לֵאמֹר אָבִיךָ צִוָּה לִפְנֵי מוֹתוֹ לֵאמֹר:  מ  וַיְצַוּוּ אֶל יוֹסֵף לֵאמֹר אָבִיךָ צִוָּה לִפְנֵי מוֹתוֹ לֵאמֹר:  ת  וּפַקִידוּ לְוַת יוֹסֵף לְמֵימָר אָבוּךְ פַּקֵּד קֳדָם מוֹתֵהּ לְמֵימָר:

{יז} כֹּה תֹאמְרוּ לְיוֹסֵף אָנָּא שָׂא נָא פֶּשַׁע אַחֶיךָ וְחַטָּאתָם כִּי רָעָה גְמָלוּךָ וְעַתָּה שָׂא נָא לְפֶשַׁע עַבְדֵי אֱלֹהֵי אָבִיךָ וַיֵּבְךְּ יוֹסֵף בְּדַבְּרָם אֵלָיו:  מ  כֹּה תֹאמְרוּ לְיוֹסֵף אָנָּא שָׂא נָא פֶּשַׁע אַחֶיךָ וְחַטָּאתָם כִּי רָעָה גְמָלוּךָ וְעַתָּה שָׂא נָא לְפֶשַׁע עַבְדֵי אֱלֹהֵי אָבִיךָ וַיֵּבְךְּ יוֹסֵף בְּדַבְּרָם אֵלָיו:  ת  כְּדֵין תֵּימְרוּן לְיוֹסֵף בְּבָעוּ שְׁבוֹק כְּעַן לְחוֹבָא אַחָיךְ וְלַחֲטָאֵהוֹן אֲרֵי בִישְׁתָא גְמָלוּךְ וּכְעַן שְׁבוֹק כְּעַן לְחוֹבָא עַבְדֵי אֱלָהָא דְאָבוּךְ וּבְכָא יוֹסֵף בְּמַלָלוּתְהוֹן עִמֵּהּ:

{יח} וַיֵּלְכוּ גַּם אֶחָיו וַיִּפְּלוּ לְפָנָיו וַיֹּאמְרוּ הִנֶּנּוּ לְךָ לַעֲבָדִים:  מ  וַיֵּלְכוּ גַּם אֶחָיו וַיִּפְּלוּ לְפָנָיו וַיֹּאמְרוּ הִנֶּנּוּ לְךָ לַעֲבָדִים:  ת  וַאֲזָלוּ אַף אֲחוֹהִי וּנְפָלוּ קֳדָמוֹהִי וַאֲמָרוּ הָא אֲנַחְנָא לָךְ לְעַבְדִין:

{יט} וַיֹּאמֶר אֲלֵהֶם יוֹסֵף אַל תִּירָאוּ כִּי הֲתַחַת אֱלֹהִים אָנִי:  מ  וַיֹּאמֶר אֲלֵהֶם יוֹסֵף אַל תִּירָאוּ כִּי הֲתַחַת אֱלֹהִים אָנִי:  ת  וַאֲמַר לְהוֹן יוֹסֵף לָא תִדַחֲלוּן אֲרֵי דַחֲלָא דַּיְיָ אֲנָא:

{כ} וְאַתֶּם חֲשַׁבְתֶּם עָלַי רָעָה אֱלֹהִים חֲשָׁבָהּ לְטֹבָה לְמַעַן עֲשֹׂה כַּיּוֹם הַזֶּה לְהַחֲיֹת עַם רָב:  מ  וְאַתֶּם חֲשַׁבְתֶּם עָלַי רָעָה אֱלֹהִים חֲשָׁבָהּ לְטֹבָה לְמַעַן עֲשֹׂה כַּיּוֹם הַזֶּה לְהַחֲיֹת עַם רָב:  ת  וְאַתּוּן חֲשַׁבְתּוּן עֲלַי בִּישָׁא מִן קֳדָם יְיָ אִתְחַשְׁבַת לְטָבָא בְּדִיל לְמֶעְבַּד כְּיוֹמָא הָדֵין לְקַיָּמָא עַם סַגִּי:

{כא}  שביעי  וְעַתָּה אַל תִּירָאוּ אָנֹכִי אֲכַלְכֵּל אֶתְכֶם וְאֶת טַפְּכֶם וַיְנַחֵם אוֹתָם וַיְדַבֵּר עַל לִבָּם:  מ  וְעַתָּה אַל תִּירָאוּ אָנֹכִי אֲכַלְכֵּל אֶתְכֶם וְאֶת טַפְּכֶם וַיְנַחֵם אוֹתָם וַיְדַבֵּר עַל לִבָּם:  ת  וּכְעַן לָא תִדַחֲלוּן אֲנָא אֵזוּן יָתְכוֹן וְיָת טַפְלְכוֹן וְנַחֵם יָתְהוֹן וּמַלֵּיל תַּנְחוּמִין עַל לִבְּהוֹן:

{כב} וַיֵּשֶׁב יוֹסֵף בְּמִצְרַיִם הוּא וּבֵית אָבִיו וַיְחִי יוֹסֵף מֵאָה וָעֶשֶׂר שָׁנִים:  מ  וַיֵּשֶׁב יוֹסֵף בְּמִצְרַיִם הוּא וּבֵית אָבִיו וַיְחִי יוֹסֵף מֵאָה וָעֶשֶׂר שָׁנִים:  ת  וִיתֵיב יוֹסֵף בְּמִצְרַיִם הוּא וּבֵית אָבוּהִי וַחֲיָא יוֹסֵף מְאָה וַעֲסַר שְׁנִין:

{כג}  מפטיר  וַיַּרְא יוֹסֵף לְאֶפְרַיִם בְּנֵי שִׁלֵּשִׁים גַּם בְּנֵי מָכִיר בֶּן מְנַשֶּׁה יֻלְּדוּ עַל בִּרְכֵּי יוֹסֵף:  מ  וַיַּרְא יוֹסֵף לְאֶפְרַיִם בְּנֵי שִׁלֵּשִׁים גַּם בְּנֵי מָכִיר בֶּן מְנַשֶּׁה יֻלְּדוּ עַל בִּרְכֵּי יוֹסֵף:  ת  וַחֲזָא יוֹסֵף לְאֶפְרַיִם בְּנִין תְּלִיתָאִין אַף בְּנֵי מָכִיר בַּר מְנַשֶּׁה אִתְיְלִידוּ וְרַבִּי יוֹסֵף:

{כד} וַיֹּאמֶר יוֹסֵף אֶל אֶחָיו אָנֹכִי מֵת וֵאלֹהִים פָּקֹד יִפְקֹד אֶתְכֶם וְהֶעֱלָה אֶתְכֶם מִן הָאָרֶץ הַזֹּאת אֶל הָאָרֶץ אֲשֶׁר נִשְׁבַּע לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב:  מ  וַיֹּאמֶר יוֹסֵף אֶל אֶחָיו אָנֹכִי מֵת וֵאלֹהִים פָּקֹד יִפְקֹד אֶתְכֶם וְהֶעֱלָה אֶתְכֶם מִן הָאָרֶץ הַזֹּאת אֶל הָאָרֶץ אֲשֶׁר נִשְׁבַּע לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב:  ת  וַאֲמַר יוֹסֵף לַאֲחוֹהִי אֲנָא מָאִית וַיְיָ מִדְכַּר דְּכִיר יָתְכוֹן וְיִסַק יָתְכוֹן מִן אַרְעָא הָדָא לְאַרְעָא דִי קַיִים לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב:

{כה} וַיַּשְׁבַּע יוֹסֵף אֶת בְּנֵי יִשְׂרָאֵל לֵאמֹר פָּקֹד יִפְקֹד אֱלֹהִים אֶתְכֶם וְהַעֲלִתֶם אֶת עַצְמֹתַי מִזֶּה:  מ  וַיַּשְׁבַּע יוֹסֵף אֶת בְּנֵי יִשְׂרָאֵל לֵאמֹר פָּקֹד יִפְקֹד אֱלֹהִים אֶתְכֶם וְהַעֲלִתֶם אֶת עַצְמֹתַי מִזֶּה:  ת  וְאוֹמֵי יוֹסֵף יָת בְּנֵי יִשְׂרָאֵל לְמֵימָר מִדְכַּר דְּכִיר יְיָ יָתְכוֹן וְתַסְקוּן יָת גַּרְמַי מִכָּא:

{כו} וַיָּמָת יוֹסֵף בֶּן מֵאָה וָעֶשֶׂר שָׁנִים וַיַּחַנְטוּ אֹתוֹ וַיִּישֶׂם בָּאָרוֹן בְּמִצְרָיִם:  מ  וַיָּמָת יוֹסֵף בֶּן מֵאָה וָעֶשֶׂר שָׁנִים וַיַּחַנְטוּ אֹתוֹ וַיִּישֶׂם בָּאָרוֹן בְּמִצְרָיִם:  ת  וּמִית יוֹסֵף בַּר מְאָה וַעֲסַר שְׁנִין וַחֲנָטוּ יָתֵהּ וְשָׂמוּהִי בָאֳרוֹנָא בְּמִצְרָיִם: (חזק)
