פרשת חיי שרה



בראשית פרק-כג

{א} וַיִּהְיוּ חַיֵּי שָׂרָה מֵאָה שָׁנָה וְעֶשְׂרִים שָׁנָה וְשֶׁבַע שָׁנִים שְׁנֵי חַיֵּי שָׂרָה:  מ  וַיִּהְיוּ חַיֵּי שָׂרָה מֵאָה שָׁנָה וְעֶשְׂרִים שָׁנָה וְשֶׁבַע שָׁנִים שְׁנֵי חַיֵּי שָׂרָה:  ת  וַהֲווֹ חַיֵּי שָׂרָה מְאָה וְעֶשְׂרִין וּשְׁבַע שְׁנִין שְׁנֵי חַיֵּי שָׂרָה:

{ב} וַתָּמָת שָׂרָה בְּקִרְיַת אַרְבַּע הִוא חֶבְרוֹן בְּאֶרֶץ כְּנָעַן וַיָּבֹא אַבְרָהָם לִסְפֹּד לְשָׂרָה וְלִבְכֹּתָהּ:  מ  וַתָּמָת שָׂרָה בְּקִרְיַת אַרְבַּע הִוא חֶבְרוֹן בְּאֶרֶץ כְּנָעַן וַיָּבֹא אַבְרָהָם לִסְפֹּד לְשָׂרָה וְלִבְכֹּתָהּ:  ת  וּמִיתַת שָׂרָה בְּקִרְיַת אַרְבַּע הִיא חֶבְרוֹן בְּאַרְעָא דִּכְנָעַן וַאֲתָא אַבְרָהָם לְמִסְפְּדָה לְשָׂרָה וּלְמִבְכַּהּ:

{ג} וַיָּקָם אַבְרָהָם מֵעַל פְּנֵי מֵתוֹ וַיְדַבֵּר אֶל בְּנֵי חֵת לֵאמֹר:  מ  וַיָּקָם אַבְרָהָם מֵעַל פְּנֵי מֵתוֹ וַיְדַבֵּר אֶל בְּנֵי חֵת לֵאמֹר:  ת  וְקָם אַבְרָהָם מֵעַל אַפֵּי מִיתֵהּ וּמַלֵּל עִם בְּנֵי חִתָּאָה לְמֵימָר:

{ד} גֵּר וְתוֹשָׁב אָנֹכִי עִמָּכֶם תְּנוּ לִי אֲחֻזַּת קֶבֶר עִמָּכֶם וְאֶקְבְּרָה מֵתִי מִלְּפָנָי:  מ  גֵּר וְתוֹשָׁב אָנֹכִי עִמָּכֶם תְּנוּ לִי אֲחֻזַּת קֶבֶר עִמָּכֶם וְאֶקְבְּרָה מֵתִי מִלְּפָנָי:  ת  דַּיָּר וְתוֹתַב אֲנָא עִמְּכוֹן הָבוּ לִי אַחֲסָנַת קְבוּרָא עִמְּכוֹן וְאִקְבַּר מִיתִי מִן קֳדָמָי:

{ה} וַיַּעֲנוּ בְנֵי חֵת אֶת אַבְרָהָם לֵאמֹר לוֹ:  מ  וַיַּעֲנוּ בְנֵי חֵת אֶת אַבְרָהָם לֵאמֹר לוֹ:  ת  וַאֲתִיבוּ בְנֵי חִתָּאָה יָת אַבְרָהָם לְמֵימָר לֵהּ:

{ו} שְׁמָעֵנוּ אֲדֹנִי נְשִׂיא אֱלֹהִים אַתָּה בְּתוֹכֵנוּ בְּמִבְחַר קְבָרֵינוּ קְבֹר אֶת מֵתֶךָ אִישׁ מִמֶּנּוּ אֶת קִבְרוֹ לֹא יִכְלֶה מִמְּךָ מִקְּבֹר מֵתֶךָ:  מ  שְׁמָעֵנוּ אֲדֹנִי נְשִׂיא אֱלֹהִים אַתָּה בְּתוֹכֵנוּ בְּמִבְחַר קְבָרֵינוּ קְבֹר אֶת מֵתֶךָ אִישׁ מִמֶּנּוּ אֶת קִבְרוֹ לֹא יִכְלֶה מִמְּךָ מִקְּבֹר מֵתֶךָ:  ת  קַבֵּל מִנָּנָא רִבּוֹנָנָא רַב קֳדָם יְיָ אַתְּ בֵּינָנָא בִּשְׁפַר קִבְרָנָא קְבַר יָת מִיתָךְ אֱנַשׁ מִנָּנָא יָת קִבְרֵהּ לָא יִמְנַע מִנָּךְ מִלְמִקְבַּר מִיתָךְ:

{ז} וַיָּקָם אַבְרָהָם וַיִּשְׁתַּחוּ לְעַם הָאָרֶץ לִבְנֵי חֵת:  מ  וַיָּקָם אַבְרָהָם וַיִּשְׁתַּחוּ לְעַם הָאָרֶץ לִבְנֵי חֵת:  ת  וְקָם אַבְרָהָם וּסְגִיד לְעַמָּא דְאַרְעָא לִבְנֵי חִתָּאָה:

{ח} וַיְדַבֵּר אִתָּם לֵאמֹר אִם יֵשׁ אֶת נַפְשְׁכֶם לִקְבֹּר אֶת מֵתִי מִלְּפָנַי שְׁמָעוּנִי וּפִגְעוּ לִי בְּעֶפְרוֹן בֶּן צֹחַר:  מ  וַיְדַבֵּר אִתָּם לֵאמֹר אִם יֵשׁ אֶת נַפְשְׁכֶם לִקְבֹּר אֶת מֵתִי מִלְּפָנַי שְׁמָעוּנִי וּפִגְעוּ לִי בְּעֶפְרוֹן בֶּן צֹחַר:  ת  וּמַלֵּל עִמְּהוֹן לְמֵימָר אִם אִית רַעֲוָא בְנַפְשְׁכוֹן לְמִקְבַּר יָת מִיתִי מִן קֳדָמַי קַבִּילוּ מִנִּי וּבְעוּ לִי מִן עֶפְרוֹן בַּר צוֹחַר:

{ט} וְיִתֶּן לִי אֶת מְעָרַת הַמַּכְפֵּלָה אֲשֶׁר לוֹ אֲשֶׁר בִּקְצֵה שָׂדֵהוּ בְּכֶסֶף מָלֵא יִתְּנֶנָּה לִי בְּתוֹכְכֶם לַאֲחֻזַּת קָבֶר:  מ  וְיִתֶּן לִי אֶת מְעָרַת הַמַּכְפֵּלָה אֲשֶׁר לוֹ אֲשֶׁר בִּקְצֵה שָׂדֵהוּ בְּכֶסֶף מָלֵא יִתְּנֶנָּה לִי בְּתוֹכְכֶם לַאֲחֻזַּת קָבֶר:  ת  וְיִתֶּן לִי יָת מְעָרַת כָּפֶלְתָּא דִּי לֵהּ דִּי בִּסְטַר חַקְלֵהּ בְּכַסְפָּא שְׁלִים יִתְּנִינַהּ לִי בֵּינֵיכוֹן לְאַחֲסָנַת קְבוּרָא:

{י} וְעֶפְרוֹן יֹשֵׁב בְּתוֹךְ בְּנֵי חֵת וַיַּעַן עֶפְרוֹן הַחִתִּי אֶת אַבְרָהָם בְּאָזְנֵי בְנֵי חֵת לְכֹל בָּאֵי שַׁעַר עִירוֹ לֵאמֹר:  מ  וְעֶפְרוֹן יֹשֵׁב בְּתוֹךְ בְּנֵי חֵת וַיַּעַן עֶפְרוֹן הַחִתִּי אֶת אַבְרָהָם בְּאָזְנֵי בְנֵי חֵת לְכֹל בָּאֵי שַׁעַר עִירוֹ לֵאמֹר:  ת  וְעֶפְרוֹן יָתֵב בְּגוֹ בְּנֵי חִתָּאָה וְאָתֵיב עֶפְרוֹן חִתָּאָה יָת אַבְרָהָם קֳדָם בְּנֵי חִתָּאָה לְכֹל עָלֵי תְּרַע קַרְתֵּהּ לְמֵימָר:

{יא} לֹא אֲדֹנִי שְׁמָעֵנִי הַשָּׂדֶה נָתַתִּי לָךְ וְהַמְּעָרָה אֲשֶׁר בּוֹ לְךָ נְתַתִּיהָ לְעֵינֵי בְנֵי עַמִּי נְתַתִּיהָ לָּךְ קְבֹר מֵתֶךָ:  מ  לֹא אֲדֹנִי שְׁמָעֵנִי הַשָּׂדֶה נָתַתִּי לָךְ וְהַמְּעָרָה אֲשֶׁר בּוֹ לְךָ נְתַתִּיהָ לְעֵינֵי בְנֵי עַמִּי נְתַתִּיהָ לָּךְ קְבֹר מֵתֶךָ:  ת  לָא רִבּוֹנִי קַבֵּיל מִנִּי חַקְלָא יְהָבִית לָךְ וּמְעַרְתָּא דִּי בֵהּ לָךְ יְהָבִיתַהּ לְעֵינֵי בְנֵי עַמִּי יְהָבִיתַהּ לָךְ קְבַר מִיתָךְ:

{יב} וַיִּשְׁתַּחוּ אַבְרָהָם לִפְנֵי עַם הָאָרֶץ:  מ  וַיִּשְׁתַּחוּ אַבְרָהָם לִפְנֵי עַם הָאָרֶץ:  ת  וּסְגִיד אַבְרָהָם קֳדָם עַמָּא דְאַרְעָא:

{יג} וַיְדַבֵּר אֶל עֶפְרוֹן בְּאָזְנֵי עַם הָאָרֶץ לֵאמֹר אַךְ אִם אַתָּה לוּ שְׁמָעֵנִי נָתַתִּי כֶּסֶף הַשָּׂדֶה קַח מִמֶּנִּי וְאֶקְבְּרָה אֶת מֵתִי שָׁמָּה:  מ  וַיְדַבֵּר אֶל עֶפְרוֹן בְּאָזְנֵי עַם הָאָרֶץ לֵאמֹר אַךְ אִם אַתָּה לוּ שְׁמָעֵנִי נָתַתִּי כֶּסֶף הַשָּׂדֶה קַח מִמֶּנִּי וְאֶקְבְּרָה אֶת מֵתִי שָׁמָּה:  ת  וּמַלֵּיל עִם עֶפְרוֹן קֳדָם עַמָּא דְאַרְעָא לְמֵימָר בְּרַם אִם אַתְּ עָבֵד לִי טִיבוּ קַבֵּל מִנִּי אֶתֵּן כַּסְפָּא דְמֵי חַקְלָא סַב מִנִּי וְאֶקְבַּר יָת מִיתִי תַּמָּן:

{יד} וַיַּעַן עֶפְרוֹן אֶת אַבְרָהָם לֵאמֹר לוֹ:  מ  וַיַּעַן עֶפְרוֹן אֶת אַבְרָהָם לֵאמֹר לוֹ:  ת  וְאָתֵיב עֶפְרוֹן יָת אַבְרָהָם לְמֵימָר לֵהּ:

{טו} אֲדֹנִי שְׁמָעֵנִי אֶרֶץ אַרְבַּע מֵאֹת שֶׁקֶל כֶּסֶף בֵּינִי וּבֵינְךָ מַה הִוא וְאֶת מֵתְךָ קְבֹר:  מ  אֲדֹנִי שְׁמָעֵנִי אֶרֶץ אַרְבַּע מֵאֹת שֶׁקֶל כֶּסֶף בֵּינִי וּבֵינְךָ מַה הִוא וְאֶת מֵתְךָ קְבֹר:  ת  רִבּוֹנִי קַבֵּל מִנִּי אַרְעָא שַׁוְיָא אַרְבַּע מְאָה סִלְעִין דִּכְסַף בֵּינָא וּבֵינָךְ מַה הִיא וְיָת מִיתָךְ קְבַר:

{טז} וַיִּשְׁמַע אַבְרָהָם אֶל עֶפְרוֹן וַיִּשְׁקֹל אַבְרָהָם לְעֶפְרֹן אֶת הַכֶּסֶף אֲשֶׁר דִּבֶּר בְּאָזְנֵי בְנֵי חֵת אַרְבַּע מֵאוֹת שֶׁקֶל כֶּסֶף עֹבֵר לַסֹּחֵר:  מ  וַיִּשְׁמַע אַבְרָהָם אֶל עֶפְרוֹן וַיִּשְׁקֹל אַבְרָהָם לְעֶפְרֹן אֶת הַכֶּסֶף אֲשֶׁר דִּבֶּר בְּאָזְנֵי בְנֵי חֵת אַרְבַּע מֵאוֹת שֶׁקֶל כֶּסֶף עֹבֵר לַסֹּחֵר:  ת  וְקַבֵּל אַבְרָהָם מִן עֶפְרוֹן וּתְקַל אַבְרָהָם לְעֶפְרוֹן יָת כַּסְפָּא דְּמַלֵּיל קֳדָם בְּנֵי חִתָּאָה אַרְבַּע מְאָה סִלְעִין דִּכְסַף מִתְקַבֵּל סְחוֹרָא (נ''י דְּמִתְקַבֵּל סְחוֹרָתָא) בְּכָל מְדִינְתָּא:

{יז}  שני  וַיָּקָם שְׂדֵה עֶפְרוֹן אֲשֶׁר בַּמַּכְפֵּלָה אֲשֶׁר לִפְנֵי מַמְרֵא הַשָּׂדֶה וְהַמְּעָרָה אֲשֶׁר בּוֹ וְכָל הָעֵץ אֲשֶׁר בַּשָּׂדֶה אֲשֶׁר בְּכָל גְּבֻלוֹ סָבִיב:  מ  וַיָּקָם שְׂדֵה עֶפְרוֹן אֲשֶׁר בַּמַּכְפֵּלָה אֲשֶׁר לִפְנֵי מַמְרֵא הַשָּׂדֶה וְהַמְּעָרָה אֲשֶׁר בּוֹ וְכָל הָעֵץ אֲשֶׁר בַּשָּׂדֶה אֲשֶׁר בְּכָל גְּבֻלוֹ סָבִיב:  ת  וְקָם חֲקַל עֶפְרוֹן דִּי בְּכָפֶלְתָּא דִּי קֳדָם מַמְרֵא חַקְלָא וּמְעַרְתָּא דִּי בֵהּ וְכָל אִילָנֵי דִּי בְחַקְלָא דִּי בְכָל תְּחוּמֵהּ סְחוֹר סְחוֹר:

{יח} לְאַבְרָהָם לְמִקְנָה לְעֵינֵי בְנֵי חֵת בְּכֹל בָּאֵי שַׁעַר עִירוֹ:  מ  לְאַבְרָהָם לְמִקְנָה לְעֵינֵי בְנֵי חֵת בְּכֹל בָּאֵי שַׁעַר עִירוֹ:  ת  לְאַבְרָהָם לִזְבִינוֹהִי לְעֵינֵי בְּנֵי חִתָּאָה בְּכֹל עָלֵי תְּרַע קַרְתֵּהּ:

{יט} וְאַחֲרֵי כֵן קָבַר אַבְרָהָם אֶת שָׂרָה אִשְׁתּוֹ אֶל מְעָרַת שְׂדֵה הַמַּכְפֵּלָה עַל פְּנֵי מַמְרֵא הִוא חֶבְרוֹן בְּאֶרֶץ כְּנָעַן:  מ  וְאַחֲרֵי כֵן קָבַר אַבְרָהָם אֶת שָׂרָה אִשְׁתּוֹ אֶל מְעָרַת שְׂדֵה הַמַּכְפֵּלָה עַל פְּנֵי מַמְרֵא הִוא חֶבְרוֹן בְּאֶרֶץ כְּנָעַן:  ת  וּבָתַר כֵּן קְבַר אַבְרָהָם יָת שָׂרָה אִתְּתֵהּ בִּמְעָרַת חֲקַל כָּפֶלְתָּא עַל אַפֵּי מַמְרֵא הִיא חֶבְרוֹן בְּאַרְעָא דִּכְנָעַן:

{כ} וַיָּקָם הַשָּׂדֶה וְהַמְּעָרָה אֲשֶׁר בּוֹ לְאַבְרָהָם לַאֲחֻזַּת קָבֶר מֵאֵת בְּנֵי חֵת:  מ  וַיָּקָם הַשָּׂדֶה וְהַמְּעָרָה אֲשֶׁר בּוֹ לְאַבְרָהָם לַאֲחֻזַּת קָבֶר מֵאֵת בְּנֵי חֵת:  ת  וְקָם חַקְלָא וּמְעַרְתָּא דִּי בֵהּ לְאַבְרָהָם לְאַחֲסָנַת קְבוּרָא מִן בְּנֵי חִתָּאָה: (ס)




בראשית פרק-כד

{א} וְאַבְרָהָם זָקֵן בָּא בַּיָּמִים וַיהוָה בֵּרַךְ אֶת אַבְרָהָם בַּכֹּל:  מ  וְאַבְרָהָם זָקֵן בָּא בַּיָּמִים וַיהוָה בֵּרַךְ אֶת אַבְרָהָם בַּכֹּל:  ת  וְאַבְרָהָם סִיב עָל בְּיוֹמִין וַיְיָ בְּרִיךְ יָת אַבְרָהָם בְּכֹלָא:

{ב} וַיֹּאמֶר אַבְרָהָם אֶל עַבְדּוֹ זְקַן בֵּיתוֹ הַמֹּשֵׁל בְּכָל אֲשֶׁר לוֹ שִׂים נָא יָדְךָ תַּחַת יְרֵכִי:  מ  וַיֹּאמֶר אַבְרָהָם אֶל עַבְדּוֹ זְקַן בֵּיתוֹ הַמֹּשֵׁל בְּכָל אֲשֶׁר לוֹ שִׂים נָא יָדְךָ תַּחַת יְרֵכִי:  ת  וַאֲמַר אַבְרָהָם לְעַבְדֵהּ סָבָא דְבֵיתֵהּ דְּשַׁלִּיט בְּכָל דִי לֵהּ שַׁוִּי כְעַן יְדָךְ תְּחוֹת יַרְכִּי:

{ג} וְאַשְׁבִּיעֲךָ בַּיהוָה אֱלֹהֵי הַשָּׁמַיִם וֵאלֹהֵי הָאָרֶץ אֲשֶׁר לֹא תִקַּח אִשָּׁה לִבְנִי מִבְּנוֹת הַכְּנַעֲנִי אֲשֶׁר אָנֹכִי יוֹשֵׁב בְּקִרְבּוֹ:  מ  וְאַשְׁבִּיעֲךָ בַּיהוָה אֱלֹהֵי הַשָּׁמַיִם וֵאלֹהֵי הָאָרֶץ אֲשֶׁר לֹא תִקַּח אִשָּׁה לִבְנִי מִבְּנוֹת הַכְּנַעֲנִי אֲשֶׁר אָנֹכִי יוֹשֵׁב בְּקִרְבּוֹ:  ת  וַאֲקַיֵּם עֲלָךְ בְּמֵימְרָא דַּיְיָ אֱלָהָא דִּשְׁמַיָּא וֵאלָהָא דְאַרְעָא דְּלָא תִסַּב אִתְּתָא לִבְרִי מִבְּנַת כְּנַעֲנָאֵי דִּי אֲנָא יָתֵב בֵּינֵיהוֹן:

{ד} כִּי אֶל אַרְצִי וְאֶל מוֹלַדְתִּי תֵּלֵךְ וְלָקַחְתָּ אִשָּׁה לִבְנִי לְיִצְחָק:  מ  כִּי אֶל אַרְצִי וְאֶל מוֹלַדְתִּי תֵּלֵךְ וְלָקַחְתָּ אִשָּׁה לִבְנִי לְיִצְחָק:  ת  אֱלָהֵין לְאַרְעִי וּלְיַלָּדוּתִי תֵּיזִיל וְתִסַּב אִתְּתָא לִבְרִי לְיִצְחָק:

{ה} וַיֹּאמֶר אֵלָיו הָעֶבֶד אוּלַי לֹא תֹאבֶה הָאִשָּׁה לָלֶכֶת אַחֲרַי אֶל הָאָרֶץ הַזֹּאת הֶהָשֵׁב אָשִׁיב אֶת בִּנְךָ אֶל הָאָרֶץ אֲשֶׁר יָצָאתָ מִשָּׁם:  מ  וַיֹּאמֶר אֵלָיו הָעֶבֶד אוּלַי לֹא תֹאבֶה הָאִשָּׁה לָלֶכֶת אַחֲרַי אֶל הָאָרֶץ הַזֹּאת הֶהָשֵׁב אָשִׁיב אֶת בִּנְךָ אֶל הָאָרֶץ אֲשֶׁר יָצָאתָ מִשָּׁם:  ת  וַאֲמַר לֵהּ עַבְדָא מָאִים לָא תֵיבֵי אִתְּתָא לְמֵיתֵי בַתְרַי לְאַרְעָא הָדָא הַאֲתָבָא אָתֵיב יָת בְּרָךְ לְאַרְעָא דִּי נְפַקְתָּא מִתַּמָּן:

{ו} וַיֹּאמֶר אֵלָיו אַבְרָהָם הִשָּׁמֶר לְךָ פֶּן תָּשִׁיב אֶת בְּנִי שָׁמָּה:  מ  וַיֹּאמֶר אֵלָיו אַבְרָהָם הִשָּׁמֶר לְךָ פֶּן תָּשִׁיב אֶת בְּנִי שָׁמָּה:  ת  וַאֲמַר לֵהּ אַבְרָהָם אִסְתַּמַּר לָךְ דִּילְמָא תָתֵיב יָת בְּרִי לְתַמָּן:

{ז} יְהוָה אֱלֹהֵי הַשָּׁמַיִם אֲשֶׁר לְקָחַנִי מִבֵּית אָבִי וּמֵאֶרֶץ מוֹלַדְתִּי וַאֲשֶׁר דִּבֶּר לִי וַאֲשֶׁר נִשְׁבַּע לִי לֵאמֹר לְזַרְעֲךָ אֶתֵּן אֶת הָאָרֶץ הַזֹּאת הוּא יִשְׁלַח מַלְאָכוֹ לְפָנֶיךָ וְלָקַחְתָּ אִשָּׁה לִבְנִי מִשָּׁם:  מ  יְהוָה אֱלֹהֵי הַשָּׁמַיִם אֲשֶׁר לְקָחַנִי מִבֵּית אָבִי וּמֵאֶרֶץ מוֹלַדְתִּי וַאֲשֶׁר דִּבֶּר לִי וַאֲשֶׁר נִשְׁבַּע לִי לֵאמֹר לְזַרְעֲךָ אֶתֵּן אֶת הָאָרֶץ הַזֹּאת הוּא יִשְׁלַח מַלְאָכוֹ לְפָנֶיךָ וְלָקַחְתָּ אִשָּׁה לִבְנִי מִשָּׁם:  ת  יְיָ אֱלָהָא דִּשְׁמַיָּא דִּי דַּבְּרַנִי מִבֵּית אַבָּא וּמֵאֲרַע יַלְדוּתִי וְדִי מַלֵּל לִי וְדִי קַיֵּם לִי לְמֵימַר לִבְרָךְ אֶתֵּן יָת אַרְעָא הָדָא הוּא יִשְׁלַח מַלְאָכֵהּ קֳדָמָךְ וְתִסַּב אִתְּתָא לִבְרִי מִתַּמָּן:

{ח} וְאִם לֹא תֹאבֶה הָאִשָּׁה לָלֶכֶת אַחֲרֶיךָ וְנִקִּיתָ מִשְּׁבֻעָתִי זֹאת רַק אֶת בְּנִי לֹא תָשֵׁב שָׁמָּה:  מ  וְאִם לֹא תֹאבֶה הָאִשָּׁה לָלֶכֶת אַחֲרֶיךָ וְנִקִּיתָ מִשְּׁבֻעָתִי זֹאת רַק אֶת בְּנִי לֹא תָשֵׁב שָׁמָּה:  ת  וְאִם לָא תֵיבֵי אִתְּתָא לְמֵיתֵי בַתְרָךְ וּתְהֵי זַכָּאָה מִמּוֹמָתִי דָּא לְחוֹד יָת בְּרִי לָא תָתֵב לְתַמָּן:

{ט} וַיָּשֶׂם הָעֶבֶד אֶת יָדוֹ תַּחַת יֶרֶךְ אַבְרָהָם אֲדֹנָיו וַיִּשָּׁבַע לוֹ עַל הַדָּבָר הַזֶּה:  מ  וַיָּשֶׂם הָעֶבֶד אֶת יָדוֹ תַּחַת יֶרֶךְ אַבְרָהָם אֲדֹנָיו וַיִּשָּׁבַע לוֹ עַל הַדָּבָר הַזֶּה:  ת  וְשַׁוִּי עַבְדָא יָת יְדֵהּ תְּחוֹת יַרְכָא דְאַבְרָהָם רִבּוֹנֵהּ וְקַיֵּם לֵהּ עַל פִּתְגָּמָא הָדֵין:

{י}  שלישי  וַיִּקַּח הָעֶבֶד עֲשָׂרָה גְמַלִּים מִגְּמַלֵּי אֲדֹנָיו וַיֵּלֶךְ וְכָל טוּב אֲדֹנָיו בְּיָדוֹ וַיָּקָם וַיֵּלֶךְ אֶל אֲרַם נַהֲרַיִם אֶל עִיר נָחוֹר:  מ  וַיִּקַּח הָעֶבֶד עֲשָׂרָה גְמַלִּים מִגְּמַלֵּי אֲדֹנָיו וַיֵּלֶךְ וְכָל טוּב אֲדֹנָיו בְּיָדוֹ וַיָּקָם וַיֵּלֶךְ אֶל אֲרַם נַהֲרַיִם אֶל עִיר נָחוֹר:  ת  וּדְבַר עַבְדָא עַשְׂרָא גַּמְלִין מִגַּמְלֵי רִבּוֹנֵהּ וַאֲזַל וְכָל שְׁפַר רִבּוֹנֵהּ בִּידֵהּ וְקָם וַאֲזַל לַאֲרָם דִּי עַל פְּרַת לְקַרְתָּא דְנָחוֹר:

{יא} וַיַּבְרֵךְ הַגְּמַלִּים מִחוּץ לָעִיר אֶל בְּאֵר הַמָּיִם לְעֵת עֶרֶב לְעֵת צֵאת הַשֹּׁאֲבֹת:  מ  וַיַּבְרֵךְ הַגְּמַלִּים מִחוּץ לָעִיר אֶל בְּאֵר הַמָּיִם לְעֵת עֶרֶב לְעֵת צֵאת הַשֹּׁאֲבֹת:  ת  וְאַשְׁרֵי גַּמְלַיָּא מִבָּרָא לְקַרְתָּא לִבְאֵרָא דְמַיָּא לְעִדַּן רַמְשָׁא לְעִדַּן דְּנַפְקַן מַלְיָתָא:

{יב} וַיֹּאמַר יְהוָה אֱלֹהֵי אֲדֹנִי אַבְרָהָם הַקְרֵה נָא לְפָנַי הַיּוֹם וַעֲשֵׂה חֶסֶד עִם אֲדֹנִי אַבְרָהָם:  מ  וַיֹּאמַר יְהוָה אֱלֹהֵי אֲדֹנִי אַבְרָהָם הַקְרֵה נָא לְפָנַי הַיּוֹם וַעֲשֵׂה חֶסֶד עִם אֲדֹנִי אַבְרָהָם:  ת  וַאֲמַר יְיָ אֱלָהֵהּ דְּרִבּוֹנִי אַבְרָהָם זַמִּין כְּעַן קֳדָמַי יוֹמָא דֵּין וְעִבַד טִיבוּ עִם רִבּוֹנִי אַבְרָהָם:

{יג} הִנֵּה אָנֹכִי נִצָּב עַל עֵין הַמָּיִם וּבְנוֹת אַנְשֵׁי הָעִיר יֹצְאֹת לִשְׁאֹב מָיִם:  מ  הִנֵּה אָנֹכִי נִצָּב עַל עֵין הַמָּיִם וּבְנוֹת אַנְשֵׁי הָעִיר יֹצְאֹת לִשְׁאֹב מָיִם:  ת  הָא אֲנָא קָאֵם עַל עֵינָא דְמַיָּא וּבְנַת אֱנָשֵׁי קַרְתָּא נַפְקַן לְמִמְלֵי מַיָּא:

{יד} וְהָיָה הַנַּעֲרָ אֲשֶׁר אֹמַר אֵלֶיהָ הַטִּי נָא כַדֵּךְ וְאֶשְׁתֶּה וְאָמְרָה שְׁתֵה וְגַם גְּמַלֶּיךָ אַשְׁקֶה אֹתָהּ הֹכַחְתָּ לְעַבְדְּךָ לְיִצְחָק וּבָהּ אֵדַע כִּי עָשִׂיתָ חֶסֶד עִם אֲדֹנִי:  מ  וְהָיָה הַנַּעֲרָ אֲשֶׁר אֹמַר אֵלֶיהָ הַטִּי נָא כַדֵּךְ וְאֶשְׁתֶּה וְאָמְרָה שְׁתֵה וְגַם גְּמַלֶּיךָ אַשְׁקֶה אֹתָהּ הֹכַחְתָּ לְעַבְדְּךָ לְיִצְחָק וּבָהּ אֵדַע כִּי עָשִׂיתָ חֶסֶד עִם אֲדֹנִי:  ת  וּתְהֵי עוּלֶמְתָּא דְּאֵימַר לַהּ אַרְכִינִי כְעַן קוּלְתִיךְ וְאֶשְׁתִּי וְתֵימַר אֵשְׁתְּ וְאַף גַּמְלָךְ אַשְׁקֵי יָתַהּ זַמִּנְתָּא לְעבְדָּךְ לְיִצְחָק וּבַהּ אִידַע אֲרֵי עֲבַדְתָּ טִיבוּ עִם רִבּוֹנִי:

{טו} וַיְהִי הוּא טֶרֶם כִּלָּה לְדַבֵּר וְהִנֵּה רִבְקָה יֹצֵאת אֲשֶׁר יֻלְּדָה לִבְתוּאֵל בֶּן מִלְכָּה אֵשֶׁת נָחוֹר אֲחִי אַבְרָהָם וְכַדָּהּ עַל שִׁכְמָהּ:  מ  וַיְהִי הוּא טֶרֶם כִּלָּה לְדַבֵּר וְהִנֵּה רִבְקָה יֹצֵאת אֲשֶׁר יֻלְּדָה לִבְתוּאֵל בֶּן מִלְכָּה אֵשֶׁת נָחוֹר אֲחִי אַבְרָהָם וְכַדָּהּ עַל שִׁכְמָהּ:  ת  וַהֲוָה הוּא עַד לָא שֵׁיצֵי לְמַלָּלָא וְהָא רִבְקָה נְפָקַת דְּאִתְיְלִידַת לִבְתוּאֵל בַּר מִלְכָּה אִתַּת נָחוֹר אֲחוּהִי דְאַבְרָהָם וְקוּלְתַהּ עַל כַּתְפַּהּ:

{טז} וְהַנַּעֲרָ טֹבַת מַרְאֶה מְאֹד בְּתוּלָה וְאִישׁ לֹא יְדָעָהּ וַתֵּרֶד הָעַיְנָה וַתְּמַלֵּא כַדָּהּ וַתָּעַל:  מ  וְהַנַּעֲרָ טֹבַת מַרְאֶה מְאֹד בְּתוּלָה וְאִישׁ לֹא יְדָעָהּ וַתֵּרֶד הָעַיְנָה וַתְּמַלֵּא כַדָּהּ וַתָּעַל:  ת  וְעוּלֶמְתָּא שַׁפִּירַת חֵיזוּ (נ''י שַׁפִּירָא לְמֶחֱזֵי) לַחֲדָא בְּתֻלְתָּא וּגְבַר לָא יַדְעָהּ וּנְחָתַת לְעֵינָא וּמְלַת קוּלְתַהּ וּסְלֵקָת:

{יז} וַיָּרָץ הָעֶבֶד לִקְרָאתָהּ וַיֹּאמֶר הַגְמִיאִינִי נָא מְעַט מַיִם מִכַּדֵּךְ:  מ  וַיָּרָץ הָעֶבֶד לִקְרָאתָהּ וַיֹּאמֶר הַגְמִיאִינִי נָא מְעַט מַיִם מִכַּדֵּךְ:  ת  וּרְהַט עַבְדָּא לִקְדָמוּתַהּ וַאֲמַר אַשְׁקִינִי (נ''י אַטְעֵמְנִי) כְעַן זְעֵיר מַיָּא מִקּוּלְתִיךְ:

{יח} וַתֹּאמֶר שְׁתֵה אֲדֹנִי וַתְּמַהֵר וַתֹּרֶד כַּדָּהּ עַל יָדָהּ וַתַּשְׁקֵהוּ:  מ  וַתֹּאמֶר שְׁתֵה אֲדֹנִי וַתְּמַהֵר וַתֹּרֶד כַּדָּהּ עַל יָדָהּ וַתַּשְׁקֵהוּ:  ת  וַאֲמֶרֶת אֵשְׁתְּ רִבּוֹנִי וְאוֹחִיאַת וַאֲחֵיתַת קוּלְתַהּ עַל יְדַהּ וְאַשְׁקְתֵהּ:

{יט} וַתְּכַל לְהַשְׁקֹתוֹ וַתֹּאמֶר גַּם לִגְמַלֶּיךָ אֶשְׁאָב עַד אִם כִּלּוּ לִשְׁתֹּת:  מ  וַתְּכַל לְהַשְׁקֹתוֹ וַתֹּאמֶר גַּם לִגְמַלֶּיךָ אֶשְׁאָב עַד אִם כִּלּוּ לִשְׁתֹּת:  ת  וְשֵׁיצִיאַת לְאַשְׁקָיוּתֵהּ וַאֲמֶרֶת אַף לְגַמְלָךְ אַמְלֵי עַד דִּי סַפְקוּן לְמִשְׁתֵּי:

{כ} וַתְּמַהֵר וַתְּעַר כַּדָּהּ אֶל הַשֹּׁקֶת וַתָּרָץ עוֹד אֶל הַבְּאֵר לִשְׁאֹב וַתִּשְׁאַב לְכָל גְּמַלָּיו:  מ  וַתְּמַהֵר וַתְּעַר כַּדָּהּ אֶל הַשֹּׁקֶת וַתָּרָץ עוֹד אֶל הַבְּאֵר לִשְׁאֹב וַתִּשְׁאַב לְכָל גְּמַלָּיו:  ת  וְאוֹחִיאַת וּנְפָצַת קוּלְתַהּ לְבֵית שַׁקְיָא וּרְהָטַת עוֹד לְבֵירָא לְמִמְלֵי וּמְלַת לְכָל גַּמְלוֹהִי:

{כא} וְהָאִישׁ מִשְׁתָּאֵה לָהּ מַחֲרִישׁ לָדַעַת הַהִצְלִיחַ יְהוָה דַּרְכּוֹ אִם לֹא:  מ  וְהָאִישׁ מִשְׁתָּאֵה לָהּ מַחֲרִישׁ לָדַעַת הַהִצְלִיחַ יְהוָה דַּרְכּוֹ אִם לֹא:  ת  וְגַבְרָא שָׁהֵי בַהּ מִסְתַּכֵּל שָׁתִיק לְמֵידַע הַאַצְלַח יְיָ אָרְחֵהּ אִם לָא:

{כב} וַיְהִי כַּאֲשֶׁר כִּלּוּ הַגְּמַלִּים לִשְׁתּוֹת וַיִּקַּח הָאִישׁ נֶזֶם זָהָב בֶּקַע מִשְׁקָלוֹ וּשְׁנֵי צְמִידִים עַל יָדֶיהָ עֲשָׂרָה זָהָב מִשְׁקָלָם:  מ  וַיְהִי כַּאֲשֶׁר כִּלּוּ הַגְּמַלִּים לִשְׁתּוֹת וַיִּקַּח הָאִישׁ נֶזֶם זָהָב בֶּקַע מִשְׁקָלוֹ וּשְׁנֵי צְמִידִים עַל יָדֶיהָ עֲשָׂרָה זָהָב מִשְׁקָלָם:  ת  וַהֲוָה כַּד סַפִּיקוּ גַּמְלַיָּא לְמִשְׁתֵּי וּנְסֵיב גַּבְרָא קָדָשָׁא דְדַהֲבָא תִּקְלָא מַתְקְלֵהּ וּתְרֵין שֵׁירִין עַל יְדָהָא מַתְקַל עֲשַׂר סִלְעִין דְּדַהֲבָא מַתְקַלְהוֹן:

{כג} וַיֹּאמֶר בַּת מִי אַתְּ הַגִּידִי נָא לִי הֲיֵשׁ בֵּית אָבִיךְ מָקוֹם לָנוּ לָלִין:  מ  וַיֹּאמֶר בַּת מִי אַתְּ הַגִּידִי נָא לִי הֲיֵשׁ בֵּית אָבִיךְ מָקוֹם לָנוּ לָלִין:  ת  וַאֲמַר בַּת מָן אַתְּ חַוִּי כְעַן לִי הַאִית בֵּית אָבוּךְ אָתַר כָּשַׁר לָנָא לִמְבָת:

{כד} וַתֹּאמֶר אֵלָיו בַּת בְּתוּאֵל אָנֹכִי בֶּן מִלְכָּה אֲשֶׁר יָלְדָה לְנָחוֹר:  מ  וַתֹּאמֶר אֵלָיו בַּת בְּתוּאֵל אָנֹכִי בֶּן מִלְכָּה אֲשֶׁר יָלְדָה לְנָחוֹר:  ת  וַאֲמֶרֶת לֵהּ בַּת בְּתוּאֵל אֲנָא בַּר מִלְכָּה דִּילֵידַת לְנָחוֹר:

{כה} וַתֹּאמֶר אֵלָיו גַּם תֶּבֶן גַּם מִסְפּוֹא רַב עִמָּנוּ גַּם מָקוֹם לָלוּן:  מ  וַתֹּאמֶר אֵלָיו גַּם תֶּבֶן גַּם מִסְפּוֹא רַב עִמָּנוּ גַּם מָקוֹם לָלוּן:  ת  וַאֲמֶרֶת לֵהּ אַף תִּבְנָא אַף כִּסָּתָא סַגִּי עִמָּנָא אַף אָתַר כָּשַׁר לִמְבָת:

{כו} וַיִּקֹּד הָאִישׁ וַיִּשְׁתַּחוּ לַיהוָה:  מ  וַיִּקֹּד הָאִישׁ וַיִּשְׁתַּחוּ לַיהוָה:  ת  וּכְרַע גַּבְרָא וּסְגִיד קֳדָם יְיָ:

{כז}  רביעי  וַיֹּאמֶר בָּרוּךְ יְהוָה אֱלֹהֵי אֲדֹנִי אַבְרָהָם אֲשֶׁר לֹא עָזַב חַסְדּוֹ וַאֲמִתּוֹ מֵעִם אֲדֹנִי אָנֹכִי בַּדֶּרֶךְ נָחַנִי יְהוָה בֵּית אֲחֵי אֲדֹנִי:  מ  וַיֹּאמֶר בָּרוּךְ יְהוָה אֱלֹהֵי אֲדֹנִי אַבְרָהָם אֲשֶׁר לֹא עָזַב חַסְדּוֹ וַאֲמִתּוֹ מֵעִם אֲדֹנִי אָנֹכִי בַּדֶּרֶךְ נָחַנִי יְהוָה בֵּית אֲחֵי אֲדֹנִי:  ת  וַאֲמַר בְּרִיךְ יְיָ אֱלָהָא דְרִבּוֹנִי אַבְרָהָם דִּי לָא מְנַע טִיבוּתֵהּ וְקוּשְׁטֵהּ מִן רִבּוֹנִי אֲנָא בְּאוֹרַח תַּקְנָא דַּבְּרַנִּי יְיָ בֵּית אֲחוֹהִי דְרִבּוֹנִי:

{כח} וַתָּרָץ הַנַּעֲרָ וַתַּגֵּד לְבֵית אִמָּהּ כַּדְּבָרִים הָאֵלֶּה:  מ  וַתָּרָץ הַנַּעֲרָ וַתַּגֵּד לְבֵית אִמָּהּ כַּדְּבָרִים הָאֵלֶּה:  ת  וּרְהָטַת עוּלֶמְתָּא וְחַוִּיאַת לְבֵית אִמַּהּ כְּפִתְגָּמַיָּא הָאִלֵּין:

{כט} וּלְרִבְקָה אָח וּשְׁמוֹ לָבָן וַיָּרָץ לָבָן אֶל הָאִישׁ הַחוּצָה אֶל הָעָיִן:  מ  וּלְרִבְקָה אָח וּשְׁמוֹ לָבָן וַיָּרָץ לָבָן אֶל הָאִישׁ הַחוּצָה אֶל הָעָיִן:  ת  וּלְרִבְקָה אָחָא וּשְׁמֵהּ לָבָן וּרְהַט לָבָן לְוַת גַּבְרָא לְבָרָא לְוַת עֵינָא:

{ל} וַיְהִי כִּרְאֹת אֶת הַנֶּזֶם וְאֶת הַצְּמִדִים עַל יְדֵי אֲחֹתוֹ וּכְשָׁמְעוֹ אֶת דִּבְרֵי רִבְקָה אֲחֹתוֹ לֵאמֹר כֹּה דִבֶּר אֵלַי הָאִישׁ וַיָּבֹא אֶל הָאִישׁ וְהִנֵּה עֹמֵד עַל הַגְּמַלִּים עַל הָעָיִן:  מ  וַיְהִי כִּרְאֹת אֶת הַנֶּזֶם וְאֶת הַצְּמִדִים עַל יְדֵי אֲחֹתוֹ וּכְשָׁמְעוֹ אֶת דִּבְרֵי רִבְקָה אֲחֹתוֹ לֵאמֹר כֹּה דִבֶּר אֵלַי הָאִישׁ וַיָּבֹא אֶל הָאִישׁ וְהִנֵּה עֹמֵד עַל הַגְּמַלִּים עַל הָעָיִן:  ת  וַהֲוָה כַּד חָזָא יָת קָדָשָׁא וְיָת שֵׁירַיָּא עַל יְדֵי אֲחָתֵהּ וְכַד שְׁמַע יָת פִּתְגָּמֵי רִבְקָה אֲחָתֵהּ לְמֵימַר כְּדֵין מַלֵּיל עִמִּי גַּבְרָא וַאֲתָא לְוַת גַּבְרָא וְהָא קָאֵם עִלַּוֵּי גַּמְלַיָּא עַל עֵינָא:

{לא} וַיֹּאמֶר בּוֹא בְּרוּךְ יְהוָה לָמָּה תַעֲמֹד בַּחוּץ וְאָנֹכִי פִּנִּיתִי הַבַּיִת וּמָקוֹם לַגְּמַלִּים:  מ  וַיֹּאמֶר בּוֹא בְּרוּךְ יְהוָה לָמָּה תַעֲמֹד בַּחוּץ וְאָנֹכִי פִּנִּיתִי הַבַּיִת וּמָקוֹם לַגְּמַלִּים:  ת  וַאֲמַר עוּל בְּרִיכָא דַּיְיָ לְמָא אַתְּ קָאֵם בְּבָרָא וַאֲנָא פַּנֵּיתִי בֵיתָא וַאֲתַר כָּשַׁר לְגַמְלַיָּא:

{לב} וַיָּבֹא הָאִישׁ הַבַּיְתָה וַיְפַתַּח הַגְּמַלִּים וַיִּתֵּן תֶּבֶן וּמִסְפּוֹא לַגְּמַלִּים וּמַיִם לִרְחֹץ רַגְלָיו וְרַגְלֵי הָאֲנָשִׁים אֲשֶׁר אִתּוֹ:  מ  וַיָּבֹא הָאִישׁ הַבַּיְתָה וַיְפַתַּח הַגְּמַלִּים וַיִּתֵּן תֶּבֶן וּמִסְפּוֹא לַגְּמַלִּים וּמַיִם לִרְחֹץ רַגְלָיו וְרַגְלֵי הָאֲנָשִׁים אֲשֶׁר אִתּוֹ:  ת  וְעַל גַּבְרָא לְבֵיתָא וּשְׁרָא גַּמְלַיָּא וִיהַב תִּבְנָא וְכִסָּתָא לְגַמְלַיָּא וּמַיָּא לְאַסְחָאָה רַגְלוֹהִי וְרַגְלֵי גֻּבְרַיָּא דְּעִמֵּהּ:

{לג} (ויישם) וַיּוּשַׂם לְפָנָיו לֶאֱכֹל וַיֹּאמֶר לֹא אֹכַל עַד אִם דִּבַּרְתִּי דְּבָרָי וַיֹּאמֶר דַּבֵּר:  מ  (ויישם) וַיּוּשַׂם לְפָנָיו לֶאֱכֹל וַיֹּאמֶר לֹא אֹכַל עַד אִם דִּבַּרְתִּי דְּבָרָי וַיֹּאמֶר דַּבֵּר:  ת  וְשַׁוִּיאוּ קֳדָמוֹהִי לְמֵיכָל וַאֲמַר לָא אֵיכוּל עַד דַּאֲמַלֵּל פִּתְגָּמָי וַאֲמַר מַלֵּיל:

{לד} וַיֹּאמַר עֶבֶד אַבְרָהָם אָנֹכִי:  מ  וַיֹּאמַר עֶבֶד אַבְרָהָם אָנֹכִי:  ת  וַאֲמַר עַבְדָּא דְאַבְרָהָם אֲנָא:

{לה} וַיהוָה בֵּרַךְ אֶת אֲדֹנִי מְאֹד וַיִּגְדָּל וַיִּתֶּן לוֹ צֹאן וּבָקָר וְכֶסֶף וְזָהָב וַעֲבָדִם וּשְׁפָחֹת וּגְמַלִּים וַחֲמֹרִים:  מ  וַיהוָה בֵּרַךְ אֶת אֲדֹנִי מְאֹד וַיִּגְדָּל וַיִּתֶּן לוֹ צֹאן וּבָקָר וְכֶסֶף וְזָהָב וַעֲבָדִם וּשְׁפָחֹת וּגְמַלִּים וַחֲמֹרִים:  ת  וַיְיָ בְּרִיךְ יָת רִבּוֹנִי לַחֲדָא וּרְבָא וִיהַב לֵהּ עָאן וְתוֹרִין וּכְסַף וּדְהַב וְעַבְדִּין וְאַמְהָן וְגַמְלִין וְחַמָרִין:

{לו} וַתֵּלֶד שָׂרָה אֵשֶׁת אֲדֹנִי בֵן לַאדֹנִי אַחֲרֵי זִקְנָתָהּ וַיִּתֶּן לּוֹ אֶת כָּל אֲשֶׁר לוֹ:  מ  וַתֵּלֶד שָׂרָה אֵשֶׁת אֲדֹנִי בֵן לַאדֹנִי אַחֲרֵי זִקְנָתָהּ וַיִּתֶּן לּוֹ אֶת כָּל אֲשֶׁר לוֹ:  ת  וִילֵידַת שָׂרָה אִתַּת רִבּוֹנִי בַר לְרִבּוֹנִי בָּתַר דְּסֵיבָת וִיהַב לֵהּ יָת כָּל דִּילֵהּ:

{לז} וַיַּשְׁבִּעֵנִי אֲדֹנִי לֵאמֹר לֹא תִקַּח אִשָּׁה לִבְנִי מִבְּנוֹת הַכְּנַעֲנִי אֲשֶׁר אָנֹכִי יֹשֵׁב בְּאַרְצוֹ:  מ  וַיַּשְׁבִּעֵנִי אֲדֹנִי לֵאמֹר לֹא תִקַּח אִשָּׁה לִבְנִי מִבְּנוֹת הַכְּנַעֲנִי אֲשֶׁר אָנֹכִי יֹשֵׁב בְּאַרְצוֹ:  ת  וְקַיֵּם עָלַי רִבּוֹנִי לְמֵימָר לָא תִסַבּ אִתְּתָא לִבְרִי מִבְּנַת כְּנַעֲנָאֵי דִי אֲנָא יָתֵב בְּאַרְעֵהּ:

{לח} אִם לֹא אֶל בֵּית אָבִי תֵּלֵךְ וְאֶל מִשְׁפַּחְתִּי וְלָקַחְתָּ אִשָּׁה לִבְנִי:  מ  אִם לֹא אֶל בֵּית אָבִי תֵּלֵךְ וְאֶל מִשְׁפַּחְתִּי וְלָקַחְתָּ אִשָּׁה לִבְנִי:  ת  אֶלָהֵן לְבֵית אַבָּא תֵּזִיל וּלְזַרְעִיתִי וְתִסַּב אִתְּתָא לִבְרִי:

{לט} וָאֹמַר אֶל אֲדֹנִי אֻלַי לֹא תֵלֵךְ הָאִשָּׁה אַחֲרָי:  מ  וָאֹמַר אֶל אֲדֹנִי אֻלַי לֹא תֵלֵךְ הָאִשָּׁה אַחֲרָי:  ת  וַאֲמָרִית לְרִבּוֹנִי מָאִים לָא תֵיתֵי אִתְּתָא בַּתְרָי:

{מ} וַיֹּאמֶר אֵלָי יְהוָה אֲשֶׁר הִתְהַלַּכְתִּי לְפָנָיו יִשְׁלַח מַלְאָכוֹ אִתָּךְ וְהִצְלִיחַ דַּרְכֶּךָ וְלָקַחְתָּ אִשָּׁה לִבְנִי מִמִּשְׁפַּחְתִּי וּמִבֵּית אָבִי:  מ  וַיֹּאמֶר אֵלָי יְהוָה אֲשֶׁר הִתְהַלַּכְתִּי לְפָנָיו יִשְׁלַח מַלְאָכוֹ אִתָּךְ וְהִצְלִיחַ דַּרְכֶּךָ וְלָקַחְתָּ אִשָּׁה לִבְנִי מִמִּשְׁפַּחְתִּי וּמִבֵּית אָבִי:  ת  וַאֲמַר לִי יְיָ דִּי פְלָחִית קֳדָמוֹהִי יִשְׁלַח מַלְאָכֵהּ עִמָּךְ וְיַצְלַח אָרְחָךְ וְתִסַּב אִתְּתָא לִבְרִי מִזְּרְעִיתִי וּמִבֵּית אַבָּא:

{מא} אָז תִּנָּקֶה מֵאָלָתִי כִּי תָבוֹא אֶל מִשְׁפַּחְתִּי וְאִם לֹא יִתְּנוּ לָךְ וְהָיִיתָ נָקִי מֵאָלָתִי:  מ  אָז תִּנָּקֶה מֵאָלָתִי כִּי תָבוֹא אֶל מִשְׁפַּחְתִּי וְאִם לֹא יִתְּנוּ לָךְ וְהָיִיתָ נָקִי מֵאָלָתִי:  ת  בְּכֵן תְּהֵי זַכַּי (נ''י זַכָּא) מִמּוּמָתִי אֲרֵי תְהַךְ לְזַרְעִיתִי וְאִם לָא יִתְּנוּן לָךְ וּתְהֵי זַכַּי מִמּוּמָתִי:

{מב} וָאָבֹא הַיּוֹם אֶל הָעָיִן וָאֹמַר יְהוָה אֱלֹהֵי אֲדֹנִי אַבְרָהָם אִם יֶשְׁךָ נָּא מַצְלִיחַ דַּרְכִּי אֲשֶׁר אָנֹכִי הֹלֵךְ עָלֶיהָ:  מ  וָאָבֹא הַיּוֹם אֶל הָעָיִן וָאֹמַר יְהוָה אֱלֹהֵי אֲדֹנִי אַבְרָהָם אִם יֶשְׁךָ נָּא מַצְלִיחַ דַּרְכִּי אֲשֶׁר אָנֹכִי הֹלֵךְ עָלֶיהָ:  ת  וְאָתִית (נ''י וַאֲתֵיתִי) יוֹמָא דֵּין לְעֵינָא וַאֲמָרִית יְיָ אֱלָהָא דְּרִבּוֹנִי אַבְרָהָם אִם אִית כְּעַן רַעֲוָא קֳדָמָךְ לְאַצְלָחָא אָרְחִי דִּי אֲנָא אָזֵיל עֲלַהּ:

{מג} הִנֵּה אָנֹכִי נִצָּב עַל עֵין הַמָּיִם וְהָיָה הָעַלְמָה הַיֹּצֵאת לִשְׁאֹב וְאָמַרְתִּי אֵלֶיהָ הַשְׁקִינִי נָא מְעַט מַיִם מִכַּדֵּךְ:  מ  הִנֵּה אָנֹכִי נִצָּב עַל עֵין הַמָּיִם וְהָיָה הָעַלְמָה הַיֹּצֵאת לִשְׁאֹב וְאָמַרְתִּי אֵלֶיהָ הַשְׁקִינִי נָא מְעַט מַיִם מִכַּדֵּךְ:  ת  הָא אֲנָא קָאֵם עַל עֵינָא דְמַיָּא וִיהֵי עוּלֶמְתָּא דְּתִפּוֹק לְמִמְלֵי וְאֵימַר לַהּ אַשְׁקִינִי כְעַן זְעֵיר מַיָּא מִקּוּלְתִיךְ:

{מד} וְאָמְרָה אֵלַי גַּם אַתָּה שְׁתֵה וְגַם לִגְמַלֶּיךָ אֶשְׁאָב הִוא הָאִשָּׁה אֲשֶׁר הֹכִיחַ יְהוָה לְבֶן אֲדֹנִי:  מ  וְאָמְרָה אֵלַי גַּם אַתָּה שְׁתֵה וְגַם לִגְמַלֶּיךָ אֶשְׁאָב הִוא הָאִשָּׁה אֲשֶׁר הֹכִיחַ יְהוָה לְבֶן אֲדֹנִי:  ת  וְתֵימַר לִי אַף אַתְּ אֵשְׁתְּ וְאַף לְגַמְלָיךְ אַמְלִי הִיא אִתְּתָא דְּזַמִּין יְיָ לְבַר רִבּוֹנִי:

{מה} אֲנִי טֶרֶם אֲכַלֶּה לְדַבֵּר אֶל לִבִּי וְהִנֵּה רִבְקָה יֹצֵאת וְכַדָּהּ עַל שִׁכְמָהּ וַתֵּרֶד הָעַיְנָה וַתִּשְׁאָב וָאֹמַר אֵלֶיהָ הַשְׁקִינִי נָא:  מ  אֲנִי טֶרֶם אֲכַלֶּה לְדַבֵּר אֶל לִבִּי וְהִנֵּה רִבְקָה יֹצֵאת וְכַדָּהּ עַל שִׁכְמָהּ וַתֵּרֶד הָעַיְנָה וַתִּשְׁאָב וָאֹמַר אֵלֶיהָ הַשְׁקִינִי נָא:  ת  אֲנָא עַד לָא שֵׁיצִיתִי לְמַלָּלָא בְלִבִּי וְהָא רִבְקָה נְפָקַת וְקוּלְתַהּ עַל כַּתְפַּהּ וּנְחָתַת לְעֵינָא וּמְלָת וַאֲמָרִית לַהּ אַשְׁקִינִי כְעָן:

{מו} וַתְּמַהֵר וַתּוֹרֶד כַּדָּהּ מֵעָלֶיהָ וַתֹּאמֶר שְׁתֵה וְגַם גְּמַלֶּיךָ אַשְׁקֶה וָאֵשְׁתְּ וְגַם הַגְּמַלִּים הִשְׁקָתָה:  מ  וַתְּמַהֵר וַתּוֹרֶד כַּדָּהּ מֵעָלֶיהָ וַתֹּאמֶר שְׁתֵה וְגַם גְּמַלֶּיךָ אַשְׁקֶה וָאֵשְׁתְּ וְגַם הַגְּמַלִּים הִשְׁקָתָה:  ת  וְאוֹחִיאַת וַאֲחִיתַת קוּלְתַהּ מִנַּהּ וַאֲמֶרֶת אֵשְׁתְּ וְאַף גַּמְלָךְ אַשְׁקִי וּשְׁתֵיתִי וְאַף גַּמְלַיָּא אַשְׁקִיאַת:

{מז} וָאֶשְׁאַל אֹתָהּ וָאֹמַר בַּת מִי אַתְּ וַתֹּאמֶר בַּת בְּתוּאֵל בֶּן נָחוֹר אֲשֶׁר יָלְדָה לּוֹ מִלְכָּה וָאָשִׂם הַנֶּזֶם עַל אַפָּהּ וְהַצְּמִידִים עַל יָדֶיהָ:  מ  וָאֶשְׁאַל אֹתָהּ וָאֹמַר בַּת מִי אַתְּ וַתֹּאמֶר בַּת בְּתוּאֵל בֶּן נָחוֹר אֲשֶׁר יָלְדָה לּוֹ מִלְכָּה וָאָשִׂם הַנֶּזֶם עַל אַפָּהּ וְהַצְּמִידִים עַל יָדֶיהָ:  ת  וּשְׁאֵילִית יָתַהּ וַאֲמָרִית בַּת מָן אַתְּ וַאֲמֶרֶת בַּת בְּתוּאֵל בַּר נָחוֹר דִּילֵידַת לֵהּ מִלְכָּה וְשַׁוֵּיתִי קָדָשָׁא עַל אַפַּהּ וְשִׁירַיָּא עַל יְדָהָא:

{מח} וָאֶקֹּד וָאֶשְׁתַּחֲוֶה לַיהוָה וָאֲבָרֵךְ אֶת יְהוָה אֱלֹהֵי אֲדֹנִי אַבְרָהָם אֲשֶׁר הִנְחַנִי בְּדֶרֶךְ אֱמֶת לָקַחַת אֶת בַּת אֲחִי אֲדֹנִי לִבְנוֹ:  מ  וָאֶקֹּד וָאֶשְׁתַּחֲוֶה לַיהוָה וָאֲבָרֵךְ אֶת יְהוָה אֱלֹהֵי אֲדֹנִי אַבְרָהָם אֲשֶׁר הִנְחַנִי בְּדֶרֶךְ אֱמֶת לָקַחַת אֶת בַּת אֲחִי אֲדֹנִי לִבְנוֹ:  ת  וּכְרָעִית וּסְגָדִית קֳדָם יְיָ וּבָרֵכִית יָת יְיָ אֱלָהֵהּ דְּרִבּוֹנִי אַבְרָהָם דְּדַבְּרַנִּי בְּאוֹרַח קְשׁוֹט לְמִיסַב יָת בַּת אֲחוּהִי דְּרִבּוֹנִי לִבְרֵהּ:

{מט} וְעַתָּה אִם יֶשְׁכֶם עֹשִׂים חֶסֶד וֶאֱמֶת אֶת אֲדֹנִי הַגִּידוּ לִי וְאִם לֹא הַגִּידוּ לִי וְאֶפְנֶה עַל יָמִין אוֹ עַל שְׂמֹאל:  מ  וְעַתָּה אִם יֶשְׁכֶם עֹשִׂים חֶסֶד וֶאֱמֶת אֶת אֲדֹנִי הַגִּידוּ לִי וְאִם לֹא הַגִּידוּ לִי וְאֶפְנֶה עַל יָמִין אוֹ עַל שְׂמֹאל:  ת  וּכְעַן אִם אִיתֵיכוֹן עָבְדִין טִיבוּ וּקְשׁוֹט עִם רִבּוֹנִי חַוּוֹ לִי וְאִם לָא חַוּוֹ לִי וְאִתְפְּנֵי עַל יַמִּינָא אוֹ עַל שְׂמָאלָא:

{נ} וַיַּעַן לָבָן וּבְתוּאֵל וַיֹּאמְרוּ מֵיְהוָה יָצָא הַדָּבָר לֹא נוּכַל דַּבֵּר אֵלֶיךָ רַע אוֹ טוֹב:  מ  וַיַּעַן לָבָן וּבְתוּאֵל וַיֹּאמְרוּ מֵיְהוָה יָצָא הַדָּבָר לֹא נוּכַל דַּבֵּר אֵלֶיךָ רַע אוֹ טוֹב:  ת  וְאָתֵיב לָבָן וּבְתוּאֵל וַאֲמָרוּ מִן קֳדָם יְיָ נְפַק פִּתְגָּמָא לֵית אֲנַחְנָא יָכְלִין לְמַלָּלָא עִמָּךְ בִּישׁ אוֹ טָב:

{נא} הִנֵּה רִבְקָה לְפָנֶיךָ קַח וָלֵךְ וּתְהִי אִשָּׁה לְבֶן אֲדֹנֶיךָ כַּאֲשֶׁר דִּבֶּר יְהוָה:  מ  הִנֵּה רִבְקָה לְפָנֶיךָ קַח וָלֵךְ וּתְהִי אִשָּׁה לְבֶן אֲדֹנֶיךָ כַּאֲשֶׁר דִּבֶּר יְהוָה:  ת  הָא רִבְקָה קֳדָמָךְ דְּבַר וְאֵיזִיל וּתְהֵי אִתְּתָא לְבַר רִבּוֹנָךְ כְּמָא דִּי מַלֵּל יְיָ:

{נב} וַיְהִי כַּאֲשֶׁר שָׁמַע עֶבֶד אַבְרָהָם אֶת דִּבְרֵיהֶם וַיִּשְׁתַּחוּ אַרְצָה לַיהוָה:  מ  וַיְהִי כַּאֲשֶׁר שָׁמַע עֶבֶד אַבְרָהָם אֶת דִּבְרֵיהֶם וַיִּשְׁתַּחוּ אַרְצָה לַיהוָה:  ת  וַהֲוָה כַּד שְׁמַע עַבְדָּא דְאַבְרָהָם יָת פִּתְגָּמֵיהוֹן וּסְגֵד עַל אַרְעָא קֳדָם יְיָ:

{נג}  חמישי  וַיּוֹצֵא הָעֶבֶד כְּלֵי כֶסֶף וּכְלֵי זָהָב וּבְגָדִים וַיִּתֵּן לְרִבְקָה וּמִגְדָּנֹת נָתַן לְאָחִיהָ וּלְאִמָּהּ:  מ  וַיּוֹצֵא הָעֶבֶד כְּלֵי כֶסֶף וּכְלֵי זָהָב וּבְגָדִים וַיִּתֵּן לְרִבְקָה וּמִגְדָּנֹת נָתַן לְאָחִיהָ וּלְאִמָּהּ:  ת  וְאַפֵּק עַבְדָּא מָנִין דִּכְסַף וּמָנִין דִּדְהַב וּלְבוּשִׁין וִיהַב לְרִבְקָה וּמִגְדָּנִין יְהַב לַאֲחוּהָא וּלְאִמַּהּ:

{נד} וַיֹּאכְלוּ וַיִּשְׁתּוּ הוּא וְהָאֲנָשִׁים אֲשֶׁר עִמּוֹ וַיָּלִינוּ וַיָּקוּמוּ בַבֹּקֶר וַיֹּאמֶר שַׁלְּחֻנִי לַאדֹנִי:  מ  וַיֹּאכְלוּ וַיִּשְׁתּוּ הוּא וְהָאֲנָשִׁים אֲשֶׁר עִמּוֹ וַיָּלִינוּ וַיָּקוּמוּ בַבֹּקֶר וַיֹּאמֶר שַׁלְּחֻנִי לַאדֹנִי:  ת  וַאֲכָלוּ וּשְׁתִיאוּ הוּא וְגוּבְרַיָּא דִּי עִמֵּהּ וּבָתוּ וְקָמוּ בְצַפְרָא וַאֲמַר שַׁלְּחוּנִי לְוַת רִבּוֹנִי:

{נה} וַיֹּאמֶר אָחִיהָ וְאִמָּהּ תֵּשֵׁב הַנַּעֲרָ אִתָּנוּ יָמִים אוֹ עָשׂוֹר אַחַר תֵּלֵךְ:  מ  וַיֹּאמֶר אָחִיהָ וְאִמָּהּ תֵּשֵׁב הַנַּעֲרָ אִתָּנוּ יָמִים אוֹ עָשׂוֹר אַחַר תֵּלֵךְ:  ת  וַאֲמַר אֲחוּהָ וְאִמַּהּ תָּתֵיב עוּלֶמְתָּא עִמָּנָא עִדַּן בְּעִדַּן אוֹ עַשְׂרָא יַרְחִין בָּתַר כֵּן תֵּיזִיל:

{נו} וַיֹּאמֶר אֲלֵהֶם אַל תְּאַחֲרוּ אֹתִי וַיהוָה הִצְלִיחַ דַּרְכִּי שַׁלְּחוּנִי וְאֵלְכָה לַאדֹנִי:  מ  וַיֹּאמֶר אֲלֵהֶם אַל תְּאַחֲרוּ אֹתִי וַיהוָה הִצְלִיחַ דַּרְכִּי שַׁלְּחוּנִי וְאֵלְכָה לַאדֹנִי:  ת  וַאֲמַר לְהוֹן לָא תְאַחֲרוּן יָתִי וַיְיָ אַצְלַח אָרְחִי שַׁלְּחוּנִי וְאֵיהַךְ לְוַת רִבּוֹנִי:

{נז} וַיֹּאמְרוּ נִקְרָא לַנַּעֲרָ וְנִשְׁאֲלָה אֶת פִּיהָ:  מ  וַיֹּאמְרוּ נִקְרָא לַנַּעֲרָ וְנִשְׁאֲלָה אֶת פִּיהָ:  ת  וַאֲמַרוּ נִקְרֵי לְעוּלֶמְתָּא וְנִשְׁמָע מַה דְּהִיא אָמְרָה:

{נח} וַיִּקְרְאוּ לְרִבְקָה וַיֹּאמְרוּ אֵלֶיהָ הֲתֵלְכִי עִם הָאִישׁ הַזֶּה וַתֹּאמֶר אֵלֵךְ:  מ  וַיִּקְרְאוּ לְרִבְקָה וַיֹּאמְרוּ אֵלֶיהָ הֲתֵלְכִי עִם הָאִישׁ הַזֶּה וַתֹּאמֶר אֵלֵךְ:  ת  וּקְרוֹ לְרִבְקָה וַאֲמַרוּ לַהּ הֲתֵיזְלִי עִם גַּבְרָא הָדֵין וַאֲמֶרֶת אֵיזִיל:

{נט} וַיְשַׁלְּחוּ אֶת רִבְקָה אֲחֹתָם וְאֶת מֵנִקְתָּהּ וְאֶת עֶבֶד אַבְרָהָם וְאֶת אֲנָשָׁיו:  מ  וַיְשַׁלְּחוּ אֶת רִבְקָה אֲחֹתָם וְאֶת מֵנִקְתָּהּ וְאֶת עֶבֶד אַבְרָהָם וְאֶת אֲנָשָׁיו:  ת  וְשַׁלָּחוּ יָת רִבְקָה אֲחַתְהוֹן וְיָת מֵנִקְתַּהּ וְיָת עַבְדָּא דְאַבְרָהָם וְיָת גַּבְרוֹהִי:

{ס} וַיְבָרְכוּ אֶת רִבְקָה וַיֹּאמְרוּ לָהּ אֲחֹתֵנוּ אַתְּ הֲיִי לְאַלְפֵי רְבָבָה וְיִירַשׁ זַרְעֵךְ אֵת שַׁעַר שֹׂנְאָיו:  מ  וַיְבָרְכוּ אֶת רִבְקָה וַיֹּאמְרוּ לָהּ אֲחֹתֵנוּ אַתְּ הֲיִי לְאַלְפֵי רְבָבָה וְיִירַשׁ זַרְעֵךְ אֵת שַׁעַר שֹׂנְאָיו:  ת  וּבָרִיכוּ יָת רִבְקָה וַאֲמַרוּ לַהּ אֲחָתָנָא אַתְּ הֲוֵי לְאַלְפִין וּלְרִבְוָן וְיִרְתוּן בְּנַיְכִי יָת קִרְוֵי סַנְאֵיהוֹן:

{סא} וַתָּקָם רִבְקָה וְנַעֲרֹתֶיהָ וַתִּרְכַּבְנָה עַל הַגְּמַלִּים וַתֵּלַכְנָה אַחֲרֵי הָאִישׁ וַיִּקַּח הָעֶבֶד אֶת רִבְקָה וַיֵּלַךְ:  מ  וַתָּקָם רִבְקָה וְנַעֲרֹתֶיהָ וַתִּרְכַּבְנָה עַל הַגְּמַלִּים וַתֵּלַכְנָה אַחֲרֵי הָאִישׁ וַיִּקַּח הָעֶבֶד אֶת רִבְקָה וַיֵּלַךְ:  ת  וְקָמַת רִבְקָה וְעוּלֶמְתָּהָא וּרְכִיבָא עַל גַּמְלַיָּא וְאָזְלָּא בָּתַר גַּבְרָא וּדְבַר עַבְדָּא יָת רִבְקָה וַאֲזָל:

{סב} וְיִצְחָק בָּא מִבּוֹא בְּאֵר לַחַי רֹאִי וְהוּא יוֹשֵׁב בְּאֶרֶץ הַנֶּגֶב:  מ  וְיִצְחָק בָּא מִבּוֹא בְּאֵר לַחַי רֹאִי וְהוּא יוֹשֵׁב בְּאֶרֶץ הַנֶּגֶב:  ת  וְיִצְחָק אָתָא מִמֵּתוֹהִי (נ''י עַל בְּמֵיתוֹהִי) מִבֵּירָא דְמַלְאָךְ קַיָּמָא אִתְחֲזֵי עֲלַהּ וְהוּא יָתֵב בַּאֲרַע דָּרוֹמָא:

{סג} וַיֵּצֵא יִצְחָק לָשׂוּחַ בַּשָּׂדֶה לִפְנוֹת עָרֶב וַיִּשָּׂא עֵינָיו וַיַּרְא וְהִנֵּה גְמַלִּים בָּאִים:  מ  וַיֵּצֵא יִצְחָק לָשׂוּחַ בַּשָּׂדֶה לִפְנוֹת עָרֶב וַיִּשָּׂא עֵינָיו וַיַּרְא וְהִנֵּה גְמַלִּים בָּאִים:  ת  וּנְפַק יִצְחָק לְצַלָּאָה בְחַקְלָא לְמִפְנֵי רַמְשָׁא וּזְקַף עֵינוֹהִי וַחֲזָא וְהָא גַּמְלַיָּא אָתָן:

{סד} וַתִּשָּׂא רִבְקָה אֶת עֵינֶיהָ וַתֵּרֶא אֶת יִצְחָק וַתִּפֹּל מֵעַל הַגָּמָל:  מ  וַתִּשָּׂא רִבְקָה אֶת עֵינֶיהָ וַתֵּרֶא אֶת יִצְחָק וַתִּפֹּל מֵעַל הַגָּמָל:  ת  וּזְקָפַת רִבְקָה יָת עֵינָהָא וַחֲזַת יָת יִצְחָק וְאִתְרְכִינַת מֵעַל גַּמְלָא:

{סה} וַתֹּאמֶר אֶל הָעֶבֶד מִי הָאִישׁ הַלָּזֶה הַהֹלֵךְ בַּשָּׂדֶה לִקְרָאתֵנוּ וַיֹּאמֶר הָעֶבֶד הוּא אֲדֹנִי וַתִּקַּח הַצָּעִיף וַתִּתְכָּס:  מ  וַתֹּאמֶר אֶל הָעֶבֶד מִי הָאִישׁ הַלָּזֶה הַהֹלֵךְ בַּשָּׂדֶה לִקְרָאתֵנוּ וַיֹּאמֶר הָעֶבֶד הוּא אֲדֹנִי וַתִּקַּח הַצָּעִיף וַתִּתְכָּס:  ת  וַאֲמֶרֶת לְעַבְדָּא מָן גַּבְרָא דֵיכִי דִּמְהַלֵּךְ בְּחַקְלָא לְקַדָּמוּתָנָא וַאֲמַר עַבְדָּא הוּא רִבּוֹנִי וּנְסֵיבַת עֵיפָא וְאִתְכַּסִּיאַת:

{סו} וַיְסַפֵּר הָעֶבֶד לְיִצְחָק אֵת כָּל הַדְּבָרִים אֲשֶׁר עָשָׂה:  מ  וַיְסַפֵּר הָעֶבֶד לְיִצְחָק אֵת כָּל הַדְּבָרִים אֲשֶׁר עָשָׂה:  ת  וְאִשְׁתָּעֵי עַבְדָּא לְיִצְחָק יָת כָּל פִּתְגָּמַיָּא דִּי עֲבָד:

{סז} וַיְבִאֶהָ יִצְחָק הָאֹהֱלָה שָׂרָה אִמּוֹ וַיִּקַּח אֶת רִבְקָה וַתְּהִי לוֹ לְאִשָּׁה וַיֶּאֱהָבֶהָ וַיִּנָּחֵם יִצְחָק אַחֲרֵי אִמּוֹ:  מ  וַיְבִאֶהָ יִצְחָק הָאֹהֱלָה שָׂרָה אִמּוֹ וַיִּקַּח אֶת רִבְקָה וַתְּהִי לוֹ לְאִשָּׁה וַיֶּאֱהָבֶהָ וַיִּנָּחֵם יִצְחָק אַחֲרֵי אִמּוֹ:  ת  וְאַעֲלַהּ יִצְחָק לְמַשְׁכְּנָא וַחֲזָא וְהָא תַקְּנִין עוֹבָדָהָא כְּעוֹבָדֵי שָׂרָה אִמֵּהּ וּנְסֵיב יָת רִבְקָה וַהֲוַת לֵהּ לְאִנְּתּוּ וּרְחִימַהּ וְאִתְנְחֵם יִצְחָק בָּתַר דְּמִיתַת אִמֵּהּ: (פ)




בראשית פרק-כה

{א}  שישי  וַיֹּסֶף אַבְרָהָם וַיִּקַּח אִשָּׁה וּשְׁמָהּ קְטוּרָה:  מ  וַיֹּסֶף אַבְרָהָם וַיִּקַּח אִשָּׁה וּשְׁמָהּ קְטוּרָה:  ת  וְאוֹסֵיף אַבְרָהָם וּנְסֵיב אִתְּתָא וּשְׁמַהּ קְטּוּרָה:

{ב} וַתֵּלֶד לוֹ אֶת זִמְרָן וְאֶת יָקְשָׁן וְאֶת מְדָן וְאֶת מִדְיָן וְאֶת יִשְׁבָּק וְאֶת שׁוּחַ:  מ  וַתֵּלֶד לוֹ אֶת זִמְרָן וְאֶת יָקְשָׁן וְאֶת מְדָן וְאֶת מִדְיָן וְאֶת יִשְׁבָּק וְאֶת שׁוּחַ:  ת  וִילֵידַת לֵהּ יָת זִמְרָן וְיָת יָקְשָׁן וְיָת מְדָן וְיָת מִדְיָן וְיָת יִשְׁבָּק וְיָת שׁוּחַ:

{ג} וְיָקְשָׁן יָלַד אֶת שְׁבָא וְאֶת דְּדָן וּבְנֵי דְדָן הָיוּ אַשּׁוּרִם וּלְטוּשִׁים וּלְאֻמִּים:  מ  וְיָקְשָׁן יָלַד אֶת שְׁבָא וְאֶת דְּדָן וּבְנֵי דְדָן הָיוּ אַשּׁוּרִם וּלְטוּשִׁים וּלְאֻמִּים:  ת  וְיָקְשָׁן אוֹלִיד יָת שְׁבָא וְיָת דְּדָן וּבְנֵי דְדָן הֲווֹ לְמַשִּׁירְיָן וְלִשְׁכוּנִין וּלְּנַגְוָן:

{ד} וּבְנֵי מִדְיָן עֵיפָה וָעֵפֶר וַחֲנֹךְ וַאֲבִידָע וְאֶלְדָּעָה כָּל אֵלֶּה בְּנֵי קְטוּרָה:  מ  וּבְנֵי מִדְיָן עֵיפָה וָעֵפֶר וַחֲנֹךְ וַאֲבִידָע וְאֶלְדָּעָה כָּל אֵלֶּה בְּנֵי קְטוּרָה:  ת  וּבְנֵי מִדְיָן עֵיפָה וָעֵפֶר וַחֲנוֹךְ וַאֲבִידָע וְאֶלְדָּעָה כָּל אִלֵּין בְּנֵי קְטוּרָה:

{ה} וַיִּתֵּן אַבְרָהָם אֶת כָּל אֲשֶׁר לוֹ לְיִצְחָק:  מ  וַיִּתֵּן אַבְרָהָם אֶת כָּל אֲשֶׁר לוֹ לְיִצְחָק:  ת  וִיהַב אַבְרָהָם יָת כָּל דִּילֵהּ לְיִצְחָק:

{ו} וְלִבְנֵי הַפִּילַגְשִׁים אֲשֶׁר לְאַבְרָהָם נָתַן אַבְרָהָם מַתָּנֹת וַיְשַׁלְּחֵם מֵעַל יִצְחָק בְּנוֹ בְּעוֹדֶנּוּ חַי קֵדְמָה אֶל אֶרֶץ קֶדֶם:  מ  וְלִבְנֵי הַפִּילַגְשִׁים אֲשֶׁר לְאַבְרָהָם נָתַן אַבְרָהָם מַתָּנֹת וַיְשַׁלְּחֵם מֵעַל יִצְחָק בְּנוֹ בְּעוֹדֶנּוּ חַי קֵדְמָה אֶל אֶרֶץ קֶדֶם:  ת  וְלִבְנֵי לְחֵינָתָא דִּי לְאַבְרָהָם יְהַב אַבְרָהָם מַתְּנָן וְשַׁלְחִנּוּן מֵעַל יִצְחָק בְּרֵהּ בְּעוֹד דְּהוּא קַיָּם קִידוּמָא לַאֲרַע מַדִינְחָא:

{ז} וְאֵלֶּה יְמֵי שְׁנֵי חַיֵּי אַבְרָהָם אֲשֶׁר חָי מְאַת שָׁנָה וְשִׁבְעִים שָׁנָה וְחָמֵשׁ שָׁנִים:  מ  וְאֵלֶּה יְמֵי שְׁנֵי חַיֵּי אַבְרָהָם אֲשֶׁר חָי מְאַת שָׁנָה וְשִׁבְעִים שָׁנָה וְחָמֵשׁ שָׁנִים:  ת  וְאִלֵּין יוֹמֵי שְׁנֵי חַיֵּי אַבְרָהָם דַּחֲיָא מְאָה וְשַׁבְעִין וְחָמֵשׁ שְׁנִין:

{ח} וַיִּגְוַע וַיָּמָת אַבְרָהָם בְּשֵׂיבָה טוֹבָה זָקֵן וְשָׂבֵעַ וַיֵּאָסֶף אֶל עַמָּיו:  מ  וַיִּגְוַע וַיָּמָת אַבְרָהָם בְּשֵׂיבָה טוֹבָה זָקֵן וְשָׂבֵעַ וַיֵּאָסֶף אֶל עַמָּיו:  ת  וְאִתְנְגִיד וּמִית אַבְרָהָם בְּסִיבוּ טָבָא סִיב וּשְׂבַע יוֹמִין וְאִתְכְּנִישׁ לְעַמֵּהּ:

{ט} וַיִּקְבְּרוּ אֹתוֹ יִצְחָק וְיִשְׁמָעֵאל בָּנָיו אֶל מְעָרַת הַמַּכְפֵּלָה אֶל שְׂדֵה עֶפְרֹן בֶּן צֹחַר הַחִתִּי אֲשֶׁר עַל פְּנֵי מַמְרֵא:  מ  וַיִּקְבְּרוּ אֹתוֹ יִצְחָק וְיִשְׁמָעֵאל בָּנָיו אֶל מְעָרַת הַמַּכְפֵּלָה אֶל שְׂדֵה עֶפְרֹן בֶּן צֹחַר הַחִתִּי אֲשֶׁר עַל פְּנֵי מַמְרֵא:  ת  וּקְבָרוּ יָתֵהּ יִצְחָק וְיִשְׁמָעֵאל בְּנוֹהִי בִּמְעָרַת כָּפֶלְתָּא בַּחֲקַל עֶפְרוֹן בַּר צֹחַר חִתָּאָה דִּי עַל אַפֵּי מַמְרֵא:

{י} הַשָּׂדֶה אֲשֶׁר קָנָה אַבְרָהָם מֵאֵת בְּנֵי חֵת שָׁמָּה קֻבַּר אַבְרָהָם וְשָׂרָה אִשְׁתּוֹ:  מ  הַשָּׂדֶה אֲשֶׁר קָנָה אַבְרָהָם מֵאֵת בְּנֵי חֵת שָׁמָּה קֻבַּר אַבְרָהָם וְשָׂרָה אִשְׁתּוֹ:  ת  חַקְלָא דִּי זְבַן אַבְרָהָם מִן בְּנֵי חִתָּאָה תַּמָּן אִתְקְבַר אַבְרָהָם וְשָׂרָה אִתְּתֵהּ:

{יא} וַיְהִי אַחֲרֵי מוֹת אַבְרָהָם וַיְבָרֶךְ אֱלֹהִים אֶת יִצְחָק בְּנוֹ וַיֵּשֶׁב יִצְחָק עִם בְּאֵר לַחַי רֹאִי:  מ  וַיְהִי אַחֲרֵי מוֹת אַבְרָהָם וַיְבָרֶךְ אֱלֹהִים אֶת יִצְחָק בְּנוֹ וַיֵּשֶׁב יִצְחָק עִם בְּאֵר לַחַי רֹאִי:  ת  וַהֲוָה בָּתַר דְּמִית אַבְרָהָם וּבָרֵךְ יְיָ יָת יִצְחָק בְּרֵהּ וִיתֵב יִצְחָק עִם בֵּירָא דְמַלְאָךְ קַיָּמָא אִתְחֲזֵי עֲלַהּ: (פ)

{יב}  שביעי  וְאֵלֶּה תֹּלְדֹת יִשְׁמָעֵאל בֶּן אַבְרָהָם אֲשֶׁר יָלְדָה הָגָר הַמִּצְרִית שִׁפְחַת שָׂרָה לְאַבְרָהָם:  מ  וְאֵלֶּה תֹּלְדֹת יִשְׁמָעֵאל בֶּן אַבְרָהָם אֲשֶׁר יָלְדָה הָגָר הַמִּצְרִית שִׁפְחַת שָׂרָה לְאַבְרָהָם:  ת  וְאִלֵּין תּוּלְדַת יִשְׁמָעֵאל בַּר אַבְרָהָם דִּי יְלֵידַת הָגָר מִצְרֵתָא אַמְתָא דְשָׂרָה לְאַבְרָהָם:

{יג} וְאֵלֶּה שְׁמוֹת בְּנֵי יִשְׁמָעֵאל בִּשְׁמֹתָם לְתוֹלְדֹתָם בְּכֹר יִשְׁמָעֵאל נְבָיֹת וְקֵדָר וְאַדְבְּאֵל וּמִבְשָׂם:  מ  וְאֵלֶּה שְׁמוֹת בְּנֵי יִשְׁמָעֵאל בִּשְׁמֹתָם לְתוֹלְדֹתָם בְּכֹר יִשְׁמָעֵאל נְבָיֹת וְקֵדָר וְאַדְבְּאֵל וּמִבְשָׂם:  ת  וְאִלֵּין שְׁמָהַת בְּנֵי יִשְׁמָעֵאל בִּשְׁמָהַתְהוֹן לְתוּלְדַתְהוֹן בּוּכְרָא דְיִשְׁמָעֵאל נְבָיוֹת וְקֵדָר וְאַדְבְּאֵל וּמִבְשָׂם:

{יד} וּמִשְׁמָע וְדוּמָה וּמַשָּׂא:  מ  וּמִשְׁמָע וְדוּמָה וּמַשָּׂא:  ת  וּמִשְׁמָע וְדוּמָה וּמַשָּׂא:

{טו} חֲדַד וְתֵימָא יְטוּר נָפִישׁ וָקֵדְמָה:  מ  חֲדַד וְתֵימָא יְטוּר נָפִישׁ וָקֵדְמָה:  ת  חֲדַד וְתֵימָא יְטוּר נָפִישׁ וָקֵדְמָה:

{טז}  מפטיר  אֵלֶּה הֵם בְּנֵי יִשְׁמָעֵאל וְאֵלֶּה שְׁמֹתָם בְּחַצְרֵיהֶם וּבְטִירֹתָם שְׁנֵים עָשָׂר נְשִׂיאִם לְאֻמֹּתָם:  מ  אֵלֶּה הֵם בְּנֵי יִשְׁמָעֵאל וְאֵלֶּה שְׁמֹתָם בְּחַצְרֵיהֶם וּבְטִירֹתָם שְׁנֵים עָשָׂר נְשִׂיאִם לְאֻמֹּתָם:  ת  אִלֵּין אִנּוּן בְּנֵי יִשְׁמָעֵאל וְאִלֵּין שְׁמָהַתְהוֹן בְּפַצְחֵיהוֹן וּבִכְרַכֵּיהוֹן תְּרֵין עֲסַר רַבְרְבִין לְאֻמֵּיהוֹן:

{יז} וְאֵלֶּה שְׁנֵי חַיֵּי יִשְׁמָעֵאל מְאַת שָׁנָה וּשְׁלֹשִׁים שָׁנָה וְשֶׁבַע שָׁנִים וַיִּגְוַע וַיָּמָת וַיֵּאָסֶף אֶל עַמָּיו:  מ  וְאֵלֶּה שְׁנֵי חַיֵּי יִשְׁמָעֵאל מְאַת שָׁנָה וּשְׁלֹשִׁים שָׁנָה וְשֶׁבַע שָׁנִים וַיִּגְוַע וַיָּמָת וַיֵּאָסֶף אֶל עַמָּיו:  ת  וְאִלֵּין שְׁנֵי חַיֵּי יִשְׁמָעֵאל מְאָה וּתְלָתִין וּשְׁבַע שְׁנִין וְאִתְנְגִיד וּמִית וְאִתְכְּנִישׁ לְעַמֵּהּ:

{יח} וַיִּשְׁכְּנוּ מֵחֲוִילָה עַד שׁוּר אֲשֶׁר עַל פְּנֵי מִצְרַיִם בֹּאֲכָה אַשּׁוּרָה עַל פְּנֵי כָל אֶחָיו נָפָל:  מ  וַיִּשְׁכְּנוּ מֵחֲוִילָה עַד שׁוּר אֲשֶׁר עַל פְּנֵי מִצְרַיִם בֹּאֲכָה אַשּׁוּרָה עַל פְּנֵי כָל אֶחָיו נָפָל:  ת  וּשְׁרוֹ מֵחֲוִילָה עַד חַגְרָא דִּי עַל אַפֵּי מִצְרַיִם מָטֵי לְאָתוּר עַל אַפֵּי כָל אֲחוֹהִי שְׁרָא: (פפפ)
