פרשת בלק


{ב} וַיַּ֥רְא בָּלָ֖ק בֶּן-צִפּ֑וֹר אֵ֛ת כָּל-אֲשֶׁר-עָשָׂ֥ה יִשְׂרָאֵ֖ל לָֽאֱמֹרִֽי:  מ  וַיַּ֥רְא בָּלָ֖ק בֶּן-צִפּ֑וֹר אֵ֛ת כָּל-אֲשֶׁר-עָשָׂ֥ה יִשְׂרָאֵ֖ל לָֽאֱמֹרִֽי:  ת  וַחֲזָא בָּלָק בַּר צִפּוֹר יָת כָּל דִי עֲבַד יִשְׂרָאֵל לֶאֱמוֹרָאֵי:

{ג} וַיָּ֨גָר מוֹאָ֜ב מִפְּנֵ֥י הָעָ֛ם מְאֹ֖ד כִּ֣י רַב-ה֑וּא וַיָּ֣קָץ מוֹאָ֔ב מִפְּנֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וַיָּ֨גָר מוֹאָ֜ב מִפְּנֵ֥י הָעָ֛ם מְאֹ֖ד כִּ֣י רַב-ה֑וּא וַיָּ֣קָץ מוֹאָ֔ב מִפְּנֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וּדְחִיל מוֹאֲבָאָה מִן קֳדָם עַמָא לַחֲדָא אֲרֵי סַגִי הוּא וַעֲקַת לְמוֹאֲבָאָה מִן קֳדָם בְּנֵי יִשְׂרָאֵל:

{ד} וַיֹּ֨אמֶר מוֹאָ֜ב אֶל-זִקְנֵ֣י מִדְיָ֗ן עַתָּ֞ה יְלַחֲכ֤וּ הַקָּהָל֙ אֶת-כָּל-סְבִ֣יבֹתֵ֔ינוּ כִּלְחֹ֣ךְ הַשּׁ֔וֹר אֵ֖ת יֶ֣רֶק הַשָּׂדֶ֑ה וּבָלָ֧ק בֶּן-צִפּ֛וֹר מֶ֥לֶךְ לְמוֹאָ֖ב בָּעֵ֥ת הַהִֽוא:  מ  וַיֹּ֨אמֶר מוֹאָ֜ב אֶל-זִקְנֵ֣י מִדְיָ֗ן עַתָּ֞ה יְלַחֲכ֤וּ הַקָּהָל֙ אֶת-כָּל-סְבִ֣יבֹתֵ֔ינוּ כִּלְחֹ֣ךְ הַשּׁ֔וֹר אֵ֖ת יֶ֣רֶק הַשָּׂדֶ֑ה וּבָלָ֧ק בֶּן-צִפּ֛וֹר מֶ֥לֶךְ לְמוֹאָ֖ב בָּעֵ֥ת הַהִֽוא:  ת  וַאֲמַר מוֹאָב לְסָבֵי מִדְיָן כְּעַן יְשֵׁיצוּן קְהָלָא יָת כָּל סַחֲרָנָנָא כְּמָא דִמְלַחֵיךְ תּוֹרָא יָת יְרוֹקָא דְחַקְלָא וּבָלָק בַּר צִפּוֹר מַלְכָּא לְמוֹאָב בְּעִידָנָה הַהִיא:

{ה} וַיִּשְׁלַ֨ח מַלְאָכִ֜ים אֶל-בִּלְעָ֣ם בֶּן-בְּע֗וֹר פְּ֠תוֹרָה אֲשֶׁ֧ר עַל-הַנָּהָ֛ר אֶ֥רֶץ בְּנֵי-עַמּ֖וֹ לִקְרֹא-ל֑וֹ לֵאמֹ֗ר הִ֠נֵּה עַ֣ם יָצָ֤א מִמִּצְרַ֙יִם֙ הִנֵּ֤ה כִסָּה֙ אֶת-עֵ֣ין הָאָ֔רֶץ וְה֥וּא יֹשֵׁ֖ב מִמֻּלִֽי:  מ  וַיִּשְׁלַ֨ח מַלְאָכִ֜ים אֶל-בִּלְעָ֣ם בֶּן-בְּע֗וֹר פְּ֠תוֹרָה אֲשֶׁ֧ר עַל-הַנָּהָ֛ר אֶ֥רֶץ בְּנֵי-עַמּ֖וֹ לִקְרֹא-ל֑וֹ לֵאמֹ֗ר הִ֠נֵּה עַ֣ם יָצָ֤א מִמִּצְרַ֙יִם֙ הִנֵּ֤ה כִסָּה֙ אֶת-עֵ֣ין הָאָ֔רֶץ וְה֥וּא יֹשֵׁ֖ב מִמֻּלִֽי:  ת  וּשְׁלַח אִזְגַדִין לְוָת בִּלְעָם בַּר בְּעוֹר לִפְּתוֹר אֲרָם דְעַל פְּרָת אֲרַע בְּנֵי עַמֵיהּ לְמִקְרֵי לֵיהּ לְמֵימָר הָא עַמָא נְפַק מִמִצְרַיִם הָא חֲפָא יַת עֵין שִׁמְשָׁא דְאַרְעָא וְהוּא שָׁרֵי מִלְקִבְלִי:

{ו} וְעַתָּה֩ לְכָה-נָּ֨א אָֽרָה-לִּ֜י אֶת-הָעָ֣ם הַזֶּ֗ה כִּֽי-עָצ֥וּם הוּא֙ מִמֶּ֔נִּי אוּלַ֤י אוּכַל֙ נַכֶּה-בּ֔וֹ וַאֲגָרְשֶׁ֖נּוּ מִן-הָאָ֑רֶץ כִּ֣י יָדַ֗עְתִּי אֵ֤ת אֲשֶׁר-תְּבָרֵךְ֙ מְבֹרָ֔ךְ וַאֲשֶׁ֥ר תָּאֹ֖ר יוּאָֽר:  מ  וְעַתָּה֩ לְכָה-נָּ֨א אָֽרָה-לִּ֜י אֶת-הָעָ֣ם הַזֶּ֗ה כִּֽי-עָצ֥וּם הוּא֙ מִמֶּ֔נִּי אוּלַ֤י אוּכַל֙ נַכֶּה-בּ֔וֹ וַאֲגָרְשֶׁ֖נּוּ מִן-הָאָ֑רֶץ כִּ֣י יָדַ֗עְתִּי אֵ֤ת אֲשֶׁר-תְּבָרֵךְ֙ מְבֹרָ֔ךְ וַאֲשֶׁ֥ר תָּאֹ֖ר יוּאָֽר:  ת  וּכְעַן אִיתָא כְּעַן לוֹט לִי יָת עַמָא הָדֵין אֲרֵי תַקִיף הוּא מִנִי מָאִים אֵכוּל לְאַגָחָא בֵּיהּ קְרָב וַאֲתָרֵכִינֵיהּ מִן אַרְעָא אֲרֵי יְדַעֲנָא יָת דִי תְבָרַךְ מְבָרָךְ וְדִי תְלוֹט לִיט:

{ז} וַיֵּ֨לְכ֜וּ זִקְנֵ֤י מוֹאָב֙ וְזִקְנֵ֣י מִדְיָ֔ן וּקְסָמִ֖ים בְּיָדָ֑ם וַיָּבֹ֙אוּ֙ אֶל-בִּלְעָ֔ם וַיְדַבְּר֥וּ אֵלָ֖יו דִּבְרֵ֥י בָלָֽק:  מ  וַיֵּ֨לְכ֜וּ זִקְנֵ֤י מוֹאָב֙ וְזִקְנֵ֣י מִדְיָ֔ן וּקְסָמִ֖ים בְּיָדָ֑ם וַיָּבֹ֙אוּ֙ אֶל-בִּלְעָ֔ם וַיְדַבְּר֥וּ אֵלָ֖יו דִּבְרֵ֥י בָלָֽק:  ת  וַאֲזָלוּ סָבֵי מוֹאָב וְסָבֵי מִדְיָן וְקִסְמַיָא בִּידֵיהוֹן וַאֲתוֹ לְוַת בִּלְעָם וּמַלִילוּ עִמֵיהּ פִּתְגָמֵי בָּלָק:

{ח} וַיֹּ֣אמֶר אֲלֵיהֶ֗ם לִ֤ינוּ פֹה֙ הַלַּ֔יְלָה וַהֲשִׁבֹתִ֤י אֶתְכֶם֙ דָּבָ֔ר כַּאֲשֶׁ֛ר יְדַבֵּ֥ר יְהוָ֖ה אֵלָ֑י וַיֵּשְׁב֥וּ שָׂרֵֽי-מוֹאָ֖ב עִם-בִּלְעָֽם:  מ  וַיֹּ֣אמֶר אֲלֵיהֶ֗ם לִ֤ינוּ פֹה֙ הַלַּ֔יְלָה וַהֲשִׁבֹתִ֤י אֶתְכֶם֙ דָּבָ֔ר כַּאֲשֶׁ֛ר יְדַבֵּ֥ר יְהוָ֖ה אֵלָ֑י וַיֵּשְׁב֥וּ שָׂרֵֽי-מוֹאָ֖ב עִם-בִּלְעָֽם:  ת  וַאֲמַר לְהוֹן בִּיתוּ הָכָא בְּלֵילְיָא וְאָתֵב יַתְכוֹן פִּתְגָמָא כְּמָא דִי יְמַלֵיל יְיָ עִמִי וְאוֹרִיכוּ רַבְרְבֵי מוֹאָב עִם בִּלְעָם:

{ט} וַיָּבֹ֥א אֱלֹהִ֖ים אֶל-בִּלְעָ֑ם וַיֹּ֕אמֶר מִ֛י הָאֲנָשִׁ֥ים הָאֵ֖לֶּה עִמָּֽךְ:  מ  וַיָּבֹ֥א אֱלֹהִ֖ים אֶל-בִּלְעָ֑ם וַיֹּ֕אמֶר מִ֛י הָאֲנָשִׁ֥ים הָאֵ֖לֶּה עִמָּֽךְ:  ת  וַאֲתָא מֵימַר מִן קֳדָם יְיָ לְוַת בִּלְעָם וַאֲמַר מָן גוּבְרַיָא הָאִלֵין דְעִמָךְ:

{י} וַיֹּ֥אמֶר בִּלְעָ֖ם אֶל-הָאֱלֹהִ֑ים בָּלָ֧ק בֶּן-צִפֹּ֛ר מֶ֥לֶךְ מוֹאָ֖ב שָׁלַ֥ח אֵלָֽי:  מ  וַיֹּ֥אמֶר בִּלְעָ֖ם אֶל-הָאֱלֹהִ֑ים בָּלָ֧ק בֶּן-צִפֹּ֛ר מֶ֥לֶךְ מוֹאָ֖ב שָׁלַ֥ח אֵלָֽי:  ת  וַאֲמַר בִּלְעָם קֳדָם יְיָ בָּלָק בַּר צִפּוֹר מַלְכָּא דְמוֹאָב שְׁלַח לְוָתִי:

{יא} הִנֵּ֤ה הָעָם֙ הַיֹּצֵ֣א מִמִּצְרַ֔יִם וַיְכַ֖ס אֶת-עֵ֣ין הָאָ֑רֶץ עַתָּ֗ה לְכָ֤ה קָֽבָה-לִּי֙ אֹת֔וֹ אוּלַ֥י אוּכַ֛ל לְהִלָּ֥חֶם בּ֖וֹ וְגֵרַשְׁתִּֽיו:  מ  הִנֵּ֤ה הָעָם֙ הַיֹּצֵ֣א מִמִּצְרַ֔יִם וַיְכַ֖ס אֶת-עֵ֣ין הָאָ֑רֶץ עַתָּ֗ה לְכָ֤ה קָֽבָה-לִּי֙ אֹת֔וֹ אוּלַ֥י אוּכַ֛ל לְהִלָּ֥חֶם בּ֖וֹ וְגֵרַשְׁתִּֽיו:  ת  הָא עַמָא דִי נְפַק מִמִצְרַיִם וַחֲפָא יָת עֵין שִׁמְשָׁא דְאַרְעָא כְּעַן אִיתָא לוֹט לִי יָתֵיהּ מָאִים אֵיכוּל לְאַגָחָא בֵּיהּ קְרָב וְאֲתָרֵיכִינֵיהּ:

{יב} וַיֹּ֤אמֶר אֱלֹהִים֙ אֶל-בִּלְעָ֔ם לֹ֥א תֵלֵ֖ךְ עִמָּהֶ֑ם לֹ֤א תָאֹר֙ אֶת-הָעָ֔ם כִּ֥י בָר֖וּךְ הֽוּא:  מ  וַיֹּ֤אמֶר אֱלֹהִים֙ אֶל-בִּלְעָ֔ם לֹ֥א תֵלֵ֖ךְ עִמָּהֶ֑ם לֹ֤א תָאֹר֙ אֶת-הָעָ֔ם כִּ֥י בָר֖וּךְ הֽוּא:  ת  וַאֲמַר יְיָ לְבִלְעָם לָא תֵיזִיל עִמְהוֹן לָא תְלוֹט יָת עַמָא אֲרֵי בְרִיךְ הוּא:

{יג}  שני - במחוברות חמישי  וַיָּ֤קָם בִּלְעָם֙ בַּבֹּ֔קֶר וַיֹּ֙אמֶר֙ אֶל-שָׂרֵ֣י בָלָ֔ק לְכ֖וּ אֶֽל-אַרְצְכֶ֑ם כִּ֚י מֵאֵ֣ן יְהוָ֔ה לְתִתִּ֖י לַהֲלֹ֥ךְ עִמָּכֶֽם:  מ  וַיָּ֤קָם בִּלְעָם֙ בַּבֹּ֔קֶר וַיֹּ֙אמֶר֙ אֶל-שָׂרֵ֣י בָלָ֔ק לְכ֖וּ אֶֽל-אַרְצְכֶ֑ם כִּ֚י מֵאֵ֣ן יְהוָ֔ה לְתִתִּ֖י לַהֲלֹ֥ךְ עִמָּכֶֽם:  ת  וְקָם בִּלְעָם בְּצַפְרָא וַאֲמַר לְרַבְרְבֵי בָלָק אִיזִילוּ לְאַרְעֲכוֹן אֲרֵי לֵית רַעֲוָא קֳדָם יְיָ לְמִשְׁבְּקִי לְמֵיזַל עִמְכוֹן:

{יד} וַיָּק֙וּמוּ֙ שָׂרֵ֣י מוֹאָ֔ב וַיָּבֹ֖אוּ אֶל-בָּלָ֑ק וַיֹּ֣אמְר֔וּ מֵאֵ֥ן בִּלְעָ֖ם הֲלֹ֥ךְ עִמָּֽנוּ:  מ  וַיָּק֙וּמוּ֙ שָׂרֵ֣י מוֹאָ֔ב וַיָּבֹ֖אוּ אֶל-בָּלָ֑ק וַיֹּ֣אמְר֔וּ מֵאֵ֥ן בִּלְעָ֖ם הֲלֹ֥ךְ עִמָּֽנוּ:  ת  וְקָמוּ רַבְרְבֵי מוֹאָב וַאֲתוֹ לְוָת בָּלָק וַאֲמָרוּ סָרֵיב בִּלְעָם לְמֵיתֵי עִמָנָא:

{טו} וַיֹּ֥סֶף ע֖וֹד בָּלָ֑ק שְׁלֹ֣חַ שָׂרִ֔ים רַבִּ֥ים וְנִכְבָּדִ֖ים מֵאֵֽלֶּה:  מ  וַיֹּ֥סֶף ע֖וֹד בָּלָ֑ק שְׁלֹ֣חַ שָׂרִ֔ים רַבִּ֥ים וְנִכְבָּדִ֖ים מֵאֵֽלֶּה:  ת  וְאוֹסִיף עוֹד בָּלָק לְמִשְׁלַח רַבְרְבִין סַגִיאִין וְיַקִירִין מֵאִלֵין:

{טז} וַיָּבֹ֖אוּ אֶל-בִּלְעָ֑ם וַיֹּ֣אמְרוּ ל֗וֹ כֹּ֤ה אָמַר֙ בָּלָ֣ק בֶּן-צִפּ֔וֹר אַל-נָ֥א תִמָּנַ֖ע מֵהֲלֹ֥ךְ אֵלָֽי:  מ  וַיָּבֹ֖אוּ אֶל-בִּלְעָ֑ם וַיֹּ֣אמְרוּ ל֗וֹ כֹּ֤ה אָמַר֙ בָּלָ֣ק בֶּן-צִפּ֔וֹר אַל-נָ֥א תִמָּנַ֖ע מֵהֲלֹ֥ךְ אֵלָֽי:  ת  וַאֲתוֹ לְוָת בִּלְעָם וַאֲמָרוּ לֵיהּ כִּדְנַן אֲמַר בָּלָק בַּר צִפּוֹר לָא כְעַן תִּתְמְנַע מִלְמֵיתֵי לְוָתִי:

{יז} כִּֽי-כַבֵּ֤ד אֲכַבֶּדְךָ֙ מְאֹ֔ד וְכֹ֛ל אֲשֶׁר-תֹּאמַ֥ר אֵלַ֖י אֶֽעֱשֶׂ֑ה וּלְכָה-נָּא֙ קָֽבָה-לִּ֔י אֵ֖ת הָעָ֥ם הַזֶּֽה:  מ  כִּֽי-כַבֵּ֤ד אֲכַבֶּדְךָ֙ מְאֹ֔ד וְכֹ֛ל אֲשֶׁר-תֹּאמַ֥ר אֵלַ֖י אֶֽעֱשֶׂ֑ה וּלְכָה-נָּא֙ קָֽבָה-לִּ֔י אֵ֖ת הָעָ֥ם הַזֶּֽה:  ת  אֲרֵי יְקָרָא אֲיַקְרִינָךְ לַחֲדָא וְכֹל דִי תֵימַר לִי אַעְבֵּד וְאִיתָא כְעַן לוֹט לִי יַת עַמָא הָדֵין:

{יח} וַיַּ֣עַן בִּלְעָ֗ם וַיֹּ֙אמֶר֙ אֶל-עַבְדֵ֣י בָלָ֔ק אִם-יִתֶּן-לִ֥י בָלָ֛ק מְלֹ֥א בֵית֖וֹ כֶּ֣סֶף וְזָהָ֑ב לֹ֣א אוּכַ֗ל לַעֲבֹר֙ אֶת-פִּי֙ יְהוָ֣ה אֱלֹהָ֔י לַעֲשׂ֥וֹת קְטַנָּ֖ה א֥וֹ גְדוֹלָֽה:  מ  וַיַּ֣עַן בִּלְעָ֗ם וַיֹּ֙אמֶר֙ אֶל-עַבְדֵ֣י בָלָ֔ק אִם-יִתֶּן-לִ֥י בָלָ֛ק מְלֹ֥א בֵית֖וֹ כֶּ֣סֶף וְזָהָ֑ב לֹ֣א אוּכַ֗ל לַעֲבֹר֙ אֶת-פִּי֙ יְהוָ֣ה אֱלֹהָ֔י לַעֲשׂ֥וֹת קְטַנָּ֖ה א֥וֹ גְדוֹלָֽה:  ת  וַאֲתֵיב בִּלְעָם וַאֲמַר לְעַבְדֵי בָלָק אִם יִתֶּן לִי בָלָק מְלֵי בֵיתֵיהּ כְּסַף וּדְהַב לֵית לִי רְשׁוּ לְמֶעֱבַּר עַל גְזֵירַת מֵימְרָא דַיְיָ אֱלָהָי לְמֶעְבַּד זְעֵרְתָּא אוֹ רַבְּתָא:

{יט} וְעַתָּ֗ה שְׁב֨וּ נָ֥א בָזֶ֛ה גַּם-אַתֶּ֖ם הַלָּ֑יְלָה וְאֵ֣דְעָ֔ה מַה-יֹּסֵ֥ף יְהוָ֖ה דַּבֵּ֥ר עִמִּֽי:  מ  וְעַתָּ֗ה שְׁב֨וּ נָ֥א בָזֶ֛ה גַּם-אַתֶּ֖ם הַלָּ֑יְלָה וְאֵ֣דְעָ֔ה מַה-יֹּסֵ֥ף יְהוָ֖ה דַּבֵּ֥ר עִמִּֽי:  ת  וּכְעַן אוֹרִיכוּ כְעַן הָכָא אַף אַתּוּן בְּלֵילְיָא וְאִדַע מָה יוֹסֵף יְיָ לְמַלָלָא עִמִי:

{כ} וַיָּבֹ֨א אֱלֹהִ֥ים | אֶל-בִּלְעָם֮ לַיְלָה֒ וַיֹּ֣אמֶר ל֗וֹ אִם-לִקְרֹ֤א לְךָ֙ בָּ֣אוּ הָאֲנָשִׁ֔ים ק֖וּם לֵ֣ךְ אִתָּ֑ם וְאַ֗ךְ אֶת-הַדָּבָ֛ר אֲשֶׁר-אֲדַבֵּ֥ר אֵלֶ֖יךָ אֹת֥וֹ תַעֲשֶֽׂה:  מ  וַיָּבֹ֨א אֱלֹהִ֥ים | אֶל-בִּלְעָם֮ לַיְלָה֒ וַיֹּ֣אמֶר ל֗וֹ אִם-לִקְרֹ֤א לְךָ֙ בָּ֣אוּ הָאֲנָשִׁ֔ים ק֖וּם לֵ֣ךְ אִתָּ֑ם וְאַ֗ךְ אֶת-הַדָּבָ֛ר אֲשֶׁר-אֲדַבֵּ֥ר אֵלֶ֖יךָ אֹת֥וֹ תַעֲשֶֽׂה:  ת  וַאֲתָּא מֵימַר מִן קֳדָם יְיָ לְבִלְעָם בְּלֵילְיָא וַאֲמַר לֵיהּ אִם לְמִקְרֵי לָךְ אֲתוֹ גֻבְרַיָא קוּם אִיזֵיל עִמְהוֹן וּבְרַם יַת פִּתְגָמָא דִי אֲמַלֵיל עִמָךְ יָתֵיהּ תַּעְבֵיד:

{כא}  שלישי  וַיָּ֤קָם בִּלְעָם֙ בַּבֹּ֔קֶר וַֽיַּחֲבֹ֖שׁ אֶת-אֲתֹנ֑וֹ וַיֵּ֖לֶךְ עִם-שָׂרֵ֥י מוֹאָֽב:  מ  וַיָּ֤קָם בִּלְעָם֙ בַּבֹּ֔קֶר וַֽיַּחֲבֹ֖שׁ אֶת-אֲתֹנ֑וֹ וַיֵּ֖לֶךְ עִם-שָׂרֵ֥י מוֹאָֽב:  ת  וְקָם בִּלְעָם בְּצַפְרָא וְזָרֵיז יַת אֲתָנֵיהּ וַאֲזַל עִם רַבְרְבֵי מוֹאָב:

{כב} וַיִּֽחַר-אַ֣ף אֱלֹהִים֮ כִּֽי-הוֹלֵ֣ךְ הוּא֒ וַיִּתְיַצֵּ֞ב מַלְאַ֧ךְ יְהוָ֛ה בַּדֶּ֖רֶךְ לְשָׂטָ֣ן ל֑וֹ וְהוּא֙ רֹכֵ֣ב עַל-אֲתֹנ֔וֹ וּשְׁנֵ֥י נְעָרָ֖יו עִמּֽוֹ:  מ  וַיִּֽחַר-אַ֣ף אֱלֹהִים֮ כִּֽי-הוֹלֵ֣ךְ הוּא֒ וַיִּתְיַצֵּ֞ב מַלְאַ֧ךְ יְהוָ֛ה בַּדֶּ֖רֶךְ לְשָׂטָ֣ן ל֑וֹ וְהוּא֙ רֹכֵ֣ב עַל-אֲתֹנ֔וֹ וּשְׁנֵ֥י נְעָרָ֖יו עִמּֽוֹ:  ת  וּתְקֵיף רָגְזָא דַיְיָ אֲרֵי אָזֵיל הוּא וְאִתְעַתַּד מַלְאָכָא דַיְיָ בְּאוֹרְחָא לְשָׂטָן לֵהּ וְהוּא רָכִיב עַל אַתָּנֵיהּ וּתְרֵין עוּלֵימוֹהִי עִמֵהּ:

{כג} וַתֵּ֣רֶא הָאָתוֹן֩ אֶת-מַלְאַ֨ךְ יְהוָ֜ה נִצָּ֣ב בַּדֶּ֗רֶךְ וְחַרְבּ֤וֹ שְׁלוּפָה֙ בְּיָד֔וֹ וַתֵּ֤ט הָֽאָתוֹן֙ מִן-הַדֶּ֔רֶךְ וַתֵּ֖לֶךְ בַּשָּׂדֶ֑ה וַיַּ֤ךְ בִּלְעָם֙ אֶת-הָ֣אָת֔וֹן לְהַטֹּתָ֖הּ הַדָּֽרֶךְ:  מ  וַתֵּ֣רֶא הָאָתוֹן֩ אֶת-מַלְאַ֨ךְ יְהוָ֜ה נִצָּ֣ב בַּדֶּ֗רֶךְ וְחַרְבּ֤וֹ שְׁלוּפָה֙ בְּיָד֔וֹ וַתֵּ֤ט הָֽאָתוֹן֙ מִן-הַדֶּ֔רֶךְ וַתֵּ֖לֶךְ בַּשָּׂדֶ֑ה וַיַּ֤ךְ בִּלְעָם֙ אֶת-הָ֣אָת֔וֹן לְהַטֹּתָ֖הּ הַדָּֽרֶךְ:  ת  וַחֲזַת אֲתָּנָא יַת מַלְאָכָא דַיְיָ מְעַתַּד בְּאָרְחָא וְחַרְבֵּיהּ שְׁלִיפָא בִּידֵיהּ וּסְטַת אֲתָּנָא מִן אוֹרְחָא וַאֲזָלַת בְּחַקְלָא וּמְחָא בִּלְעָם יַת אֲתָּנָא לְאַסְטָיוּתַהּ לְאָרְחָא:

{כד} וַֽיַּעֲמֹד֙ מַלְאַ֣ךְ יְהוָ֔ה בְּמִשְׁע֖וֹל הַכְּרָמִ֑ים גָּדֵ֥ר מִזֶּ֖ה וְגָדֵ֥ר מִזֶּֽה:  מ  וַֽיַּעֲמֹד֙ מַלְאַ֣ךְ יְהוָ֔ה בְּמִשְׁע֖וֹל הַכְּרָמִ֑ים גָּדֵ֥ר מִזֶּ֖ה וְגָדֵ֥ר מִזֶּֽה:  ת  וְקָם מַלְאָכָא דַיְיָ בִּשְׁבִיל כַּרְמַיָא אַתְרָא דְגָדֵירָא מִכָּא וְגָדֵירָא מִכָּא:

{כה} וַתֵּ֨רֶא הָאָת֜וֹן אֶת-מַלְאַ֣ךְ יְהוָ֗ה וַתִּלָּחֵץ֙ אֶל-הַקִּ֔יר וַתִּלְחַ֛ץ אֶת-רֶ֥גֶל בִּלְעָ֖ם אֶל-הַקִּ֑יר וַיֹּ֖סֶף לְהַכֹּתָֽהּ:  מ  וַתֵּ֨רֶא הָאָת֜וֹן אֶת-מַלְאַ֣ךְ יְהוָ֗ה וַתִּלָּחֵץ֙ אֶל-הַקִּ֔יר וַתִּלְחַ֛ץ אֶת-רֶ֥גֶל בִּלְעָ֖ם אֶל-הַקִּ֑יר וַיֹּ֖סֶף לְהַכֹּתָֽהּ:  ת  וַחֲזַת אֲתָּנָא יַת מַלְאָכָא דַיְיָ וְאִדְחֵיקַת לְכָתְלָא וּדְחֵיקַת יַת רִגְלָא דְבִלְעָם לְכָתְלָא וְאוֹסִיף לְמִמְחַהּ:

{כו} וַיּ֥וֹסֶף מַלְאַךְ-יְהוָ֖ה עֲב֑וֹר וַֽיַּעֲמֹד֙ בְּמָק֣וֹם צָ֔ר אֲשֶׁ֛ר אֵֽין-דֶּ֥רֶךְ לִנְט֖וֹת יָמִ֥ין וּשְׂמֹֽאול:  מ  וַיּ֥וֹסֶף מַלְאַךְ-יְהוָ֖ה עֲב֑וֹר וַֽיַּעֲמֹד֙ בְּמָק֣וֹם צָ֔ר אֲשֶׁ֛ר אֵֽין-דֶּ֥רֶךְ לִנְט֖וֹת יָמִ֥ין וּשְׂמֹֽאול:  ת  וְאוֹסֵף מַלְאָכָא דַיְיָ לְמֶעְבָּר וְקָם בַּאֲתַר עָק דִי לֵית אֹרַח לְמִסְטֵי לְיַמִינָא וְלִשְׂמָאלָא:

{כז} וַתֵּ֤רֶא הָֽאָתוֹן֙ אֶת-מַלְאַ֣ךְ יְהוָ֔ה וַתִּרְבַּ֖ץ תַּ֣חַת בִּלְעָ֑ם וַיִּֽחַר-אַ֣ף בִּלְעָ֔ם וַיַּ֥ךְ אֶת-הָאָת֖וֹן בַּמַּקֵּֽל:  מ  וַתֵּ֤רֶא הָֽאָתוֹן֙ אֶת-מַלְאַ֣ךְ יְהוָ֔ה וַתִּרְבַּ֖ץ תַּ֣חַת בִּלְעָ֑ם וַיִּֽחַר-אַ֣ף בִּלְעָ֔ם וַיַּ֥ךְ אֶת-הָאָת֖וֹן בַּמַּקֵּֽל:  ת  וַחֲזַת אֲתָּנָא יַת מַלְאָכָא דַיְיָ וּרְבָעַת תְּחוֹת בִּלְעָם וּתְקֵיף רוּגְזָא דְבִלְעָם וּמְחָא יַת אֲתָּנָא בְּחֻטְרָא:

{כח} וַיִּפְתַּ֥ח יְהוָ֖ה אֶת-פִּ֣י הָאָת֑וֹן וַתֹּ֤אמֶר לְבִלְעָם֙ מֶה-עָשִׂ֣יתִֽי לְךָ֔ כִּ֣י הִכִּיתַ֔נִי זֶ֖ה שָׁלֹ֥שׁ רְגָלִֽים:  מ  וַיִּפְתַּ֥ח יְהוָ֖ה אֶת-פִּ֣י הָאָת֑וֹן וַתֹּ֤אמֶר לְבִלְעָם֙ מֶה-עָשִׂ֣יתִֽי לְךָ֔ כִּ֣י הִכִּיתַ֔נִי זֶ֖ה שָׁלֹ֥שׁ רְגָלִֽים:  ת  וּפְתַח יְיָ יַת פּוּמָא דַאֲתָּנָא וַאֲמָרַת לְבִּלְעָם מָה עֲבָדִית לָךְ אֲרֵי מְחִיתַנִי דְנַן תְּלַת זִמְנִין:

{כט} וַיֹּ֤אמֶר בִּלְעָם֙ לָֽאָת֔וֹן כִּ֥י הִתְעַלַּ֖לְתְּ בִּ֑י ל֤וּ יֶשׁ-חֶ֙רֶב֙ בְּיָדִ֔י כִּ֥י עַתָּ֖ה הֲרַגְתִּֽיךְ:  מ  וַיֹּ֤אמֶר בִּלְעָם֙ לָֽאָת֔וֹן כִּ֥י הִתְעַלַּ֖לְתְּ בִּ֑י ל֤וּ יֶשׁ-חֶ֙רֶב֙ בְּיָדִ֔י כִּ֥י עַתָּ֖ה הֲרַגְתִּֽיךְ:  ת  וַאֲמַר בִּלְעָם לַאֲתָּנָא אֲרֵי חַיִיכְתְּ בִּי אִלוּפוֹן אִית חַרְבָּא בִּידִי אֲרֵי כְעָן קְטַלְתִּיךְ:

{ל} וַתֹּ֨אמֶר הָאָת֜וֹן אֶל-בִּלְעָ֗ם הֲלוֹא֩ אָנֹכִ֨י אֲתֹֽנְךָ֜ אֲשֶׁר-רָכַ֣בְתָּ עָלַ֗י מֵעֽוֹדְךָ֙ עַד-הַיּ֣וֹם הַזֶּ֔ה הַֽהַסְכֵּ֣ן הִסְכַּ֔נְתִּי לַעֲשׂ֥וֹת לְךָ֖ כֹּ֑ה וַיֹּ֖אמֶר לֹֽא:  מ  וַתֹּ֨אמֶר הָאָת֜וֹן אֶל-בִּלְעָ֗ם הֲלוֹא֩ אָנֹכִ֨י אֲתֹֽנְךָ֜ אֲשֶׁר-רָכַ֣בְתָּ עָלַ֗י מֵעֽוֹדְךָ֙ עַד-הַיּ֣וֹם הַזֶּ֔ה הַֽהַסְכֵּ֣ן הִסְכַּ֔נְתִּי לַעֲשׂ֥וֹת לְךָ֖ כֹּ֑ה וַיֹּ֖אמֶר לֹֽא:  ת  וַאֲמָרַת אֲתָּנָא לְבִלְעָם הֲלָא אֲנָא אֲתָּנָךְ דְרָכַבְתְּ עֲלַי מִדְאִיתָךְ עַד יוֹמָא הָדֵין הֲמֵילַף אַלִפְנָא לְמֶעֱבַּד לָךְ כְּדֵין וַאֲמַר לָא:

{לא} וַיְגַ֣ל יְהוָה֮ אֶת-עֵינֵ֣י בִלְעָם֒ וַיַּ֞רְא אֶת-מַלְאַ֤ךְ יְהוָה֙ נִצָּ֣ב בַּדֶּ֔רֶךְ וְחַרְבּ֥וֹ שְׁלֻפָ֖ה בְּיָד֑וֹ וַיִּקֹּ֥ד וַיִּשְׁתַּ֖חוּ לְאַפָּֽיו:  מ  וַיְגַ֣ל יְהוָה֮ אֶת-עֵינֵ֣י בִלְעָם֒ וַיַּ֞רְא אֶת-מַלְאַ֤ךְ יְהוָה֙ נִצָּ֣ב בַּדֶּ֔רֶךְ וְחַרְבּ֥וֹ שְׁלֻפָ֖ה בְּיָד֑וֹ וַיִּקֹּ֥ד וַיִּשְׁתַּ֖חוּ לְאַפָּֽיו:  ת  וּגְלָא יְיָ יַת עֵינֵי בִלְעָם וַחֲזָא יַת מַלְאָכָא דַיְיָ מְעַתַּד בְּאָרְחָא וְחַרְבֵּיהּ שְׁלִיפָא בִידֵיהּ וּכְרַע וּסְגִיד לְאַפּוֹהִי:

{לב} וַיֹּ֤אמֶר אֵלָיו֙ מַלְאַ֣ךְ יְהוָ֔ה עַל-מָ֗ה הִכִּ֙יתָ֙ אֶת-אֲתֹ֣נְךָ֔ זֶ֖ה שָׁל֣וֹשׁ רְגָלִ֑ים הִנֵּ֤ה אָנֹכִי֙ יָצָ֣אתִי לְשָׂטָ֔ן כִּֽי-יָרַ֥ט הַדֶּ֖רֶךְ לְנֶגְדִּֽי:  מ  וַיֹּ֤אמֶר אֵלָיו֙ מַלְאַ֣ךְ יְהוָ֔ה עַל-מָ֗ה הִכִּ֙יתָ֙ אֶת-אֲתֹ֣נְךָ֔ זֶ֖ה שָׁל֣וֹשׁ רְגָלִ֑ים הִנֵּ֤ה אָנֹכִי֙ יָצָ֣אתִי לְשָׂטָ֔ן כִּֽי-יָרַ֥ט הַדֶּ֖רֶךְ לְנֶגְדִּֽי:  ת  וַאֲמַר לֵיהּ מַלְאָכָא דַיְיָ עַל מָה מָחִיתָא יַת אֲתָּנָךְ דְנַן תְּלַת זִמְנִין הָא אֲנָא נְפָקִית לְמִשְׂטַן אֲרֵי גְלֵי קֳדָמַי דְאַתְּ רָעֵי לְמֵיזַל בְּאוֹרְחָא לְקִבְּלִי:

{לג} וַתִּרְאַ֙נִי֙ הָֽאָת֔וֹן וַתֵּ֣ט לְפָנַ֔י זֶ֖ה שָׁלֹ֣שׁ רְגָלִ֑ים אוּלַי֙ נָטְתָ֣ה מִפָּנַ֔י כִּ֥י עַתָּ֛ה גַּם-אֹתְכָ֥ה הָרַ֖גְתִּי וְאוֹתָ֥הּ הֶחֱיֵֽיתִי:  מ  וַתִּרְאַ֙נִי֙ הָֽאָת֔וֹן וַתֵּ֣ט לְפָנַ֔י זֶ֖ה שָׁלֹ֣שׁ רְגָלִ֑ים אוּלַי֙ נָטְתָ֣ה מִפָּנַ֔י כִּ֥י עַתָּ֛ה גַּם-אֹתְכָ֥ה הָרַ֖גְתִּי וְאוֹתָ֥הּ הֶחֱיֵֽיתִי:  ת  וַחֲזַתְנִי אֲתָּנָא וּסְטַת מִן קֳדָמַי דְנַן תְּלַת זִמְנִין אִילוּפוֹן לָא סְטַּת מִן קֳדָמַי אֲרֵי כְעַן אַף יָתָךְ קְטָלִית וְיָתַהּ קַיֵימִית:

{לד} וַיֹּ֨אמֶר בִּלְעָ֜ם אֶל-מַלְאַ֤ךְ יְהוָה֙ חָטָ֔אתִי כִּ֚י לֹ֣א יָדַ֔עְתִּי כִּ֥י אַתָּ֛ה נִצָּ֥ב לִקְרָאתִ֖י בַּדָּ֑רֶךְ וְעַתָּ֛ה אִם-רַ֥ע בְּעֵינֶ֖יךָ אָשׁ֥וּבָה לִּֽי:  מ  וַיֹּ֨אמֶר בִּלְעָ֜ם אֶל-מַלְאַ֤ךְ יְהוָה֙ חָטָ֔אתִי כִּ֚י לֹ֣א יָדַ֔עְתִּי כִּ֥י אַתָּ֛ה נִצָּ֥ב לִקְרָאתִ֖י בַּדָּ֑רֶךְ וְעַתָּ֛ה אִם-רַ֥ע בְּעֵינֶ֖יךָ אָשׁ֥וּבָה לִּֽי:  ת  וַאֲמַר בִּלְעָם לְמַלְאָכָא דַיְיָ חָבִית אֲרֵי לָא יְדָעִית אֲרֵי אַתְּ מְעַתֵּד לָקֳדָמוּתִי בְּאָרְחָא וּכְעַן אִם בִּישׁ בְּעֵינָךְ אֵיתוּב לִי:

{לה} וַיֹּאמֶר֩ מַלְאַ֨ךְ יְהוָ֜ה אֶל-בִּלְעָ֗ם לֵ֚ךְ עִם-הָ֣אֲנָשִׁ֔ים וְאֶ֗פֶס אֶת-הַדָּבָ֛ר אֲשֶׁר-אֲדַבֵּ֥ר אֵלֶ֖יךָ אֹת֣וֹ תְדַבֵּ֑ר וַיֵּ֥לֶךְ בִּלְעָ֖ם עִם-שָׂרֵ֥י בָלָֽק:  מ  וַיֹּאמֶר֩ מַלְאַ֨ךְ יְהוָ֜ה אֶל-בִּלְעָ֗ם לֵ֚ךְ עִם-הָ֣אֲנָשִׁ֔ים וְאֶ֗פֶס אֶת-הַדָּבָ֛ר אֲשֶׁר-אֲדַבֵּ֥ר אֵלֶ֖יךָ אֹת֣וֹ תְדַבֵּ֑ר וַיֵּ֥לֶךְ בִּלְעָ֖ם עִם-שָׂרֵ֥י בָלָֽק:  ת  וַאֲמַר מַלְאָכָא דַיְיָ לְבִלְעָם אִיזֵיל עִם גוּבְרַיָא וּלְחוֹד יַת פִּתְגָמָא דִי אֲמַלֵיל עִמָךְ יָתֵיהּ תְּמַלֵיל וַאֲזַל בִּלְעָם עִם רַבְרְבֵי בָּלָק:

{לו} וַיִּשְׁמַ֥ע בָּלָ֖ק כִּ֣י בָ֣א בִלְעָ֑ם וַיֵּצֵ֨א לִקְרָאת֜וֹ אֶל-עִ֣יר מוֹאָ֗ב אֲשֶׁר֙ עַל-גְּב֣וּל אַרְנֹ֔ן אֲשֶׁ֖ר בִּקְצֵ֥ה הַגְּבֽוּל:  מ  וַיִּשְׁמַ֥ע בָּלָ֖ק כִּ֣י בָ֣א בִלְעָ֑ם וַיֵּצֵ֨א לִקְרָאת֜וֹ אֶל-עִ֣יר מוֹאָ֗ב אֲשֶׁר֙ עַל-גְּב֣וּל אַרְנֹ֔ן אֲשֶׁ֖ר בִּקְצֵ֥ה הַגְּבֽוּל:  ת  וּשְׁמַע בָּלָק אֲרֵי אָתָא בִלְעָם וּנְפַק לָקֳדָמוּתֵה לְקַרְתָּא דְמוֹאָב דִי עַל תְּחוּם אַרְנוֹן דִי בִּסְטַר תְּחוּמָא:

{לז} וַיֹּ֨אמֶר בָּלָ֜ק אֶל-בִּלְעָ֗ם הֲלֹא֩ שָׁלֹ֨חַ שָׁלַ֤חְתִּי אֵלֶ֙יךָ֙ לִקְרֹא-לָ֔ךְ לָ֥מָּה לֹא-הָלַ֖כְתָּ אֵלָ֑י הַֽאֻמְנָ֔ם לֹ֥א אוּכַ֖ל כַּבְּדֶֽךָ:  מ  וַיֹּ֨אמֶר בָּלָ֜ק אֶל-בִּלְעָ֗ם הֲלֹא֩ שָׁלֹ֨חַ שָׁלַ֤חְתִּי אֵלֶ֙יךָ֙ לִקְרֹא-לָ֔ךְ לָ֥מָּה לֹא-הָלַ֖כְתָּ אֵלָ֑י הַֽאֻמְנָ֔ם לֹ֥א אוּכַ֖ל כַּבְּדֶֽךָ:  ת  וַאֲמַר בָּלָק לְבִלְעָם הֲלָא מִשְׁלַח שַׁלְחִית לְוָתָךְ לְמִקְרֵי לָךְ לְמָא לָא אֲתֵיתָא לְוָתִי הַבְקוּשְׁטָא הֲוֵיתָא אָמַר לֵית אֲנָא יָכִיל לְיַקְרוּתָךְ:

{לח} וַיֹּ֨אמֶר בִּלְעָ֜ם אֶל-בָּלָ֗ק הִֽנֵּה-בָ֙אתִי֙ אֵלֶ֔יךָ עַתָּ֕ה הֲיָכ֥וֹל אוּכַ֖ל דַּבֵּ֣ר מְא֑וּמָה הַדָּבָ֗ר אֲשֶׁ֨ר יָשִׂ֧ים אֱלֹהִ֛ים בְּפִ֖י אֹת֥וֹ אֲדַבֵּֽר:  מ  וַיֹּ֨אמֶר בִּלְעָ֜ם אֶל-בָּלָ֗ק הִֽנֵּה-בָ֙אתִי֙ אֵלֶ֔יךָ עַתָּ֕ה הֲיָכ֥וֹל אוּכַ֖ל דַּבֵּ֣ר מְא֑וּמָה הַדָּבָ֗ר אֲשֶׁ֨ר יָשִׂ֧ים אֱלֹהִ֛ים בְּפִ֖י אֹת֥וֹ אֲדַבֵּֽר:  ת  וַאֲמַר בִּלְעָם לְבָלָק הָא אֲתֵיתִי לְוָתָךְ כְּעַן הֲמֵיכַל יְכִלְנָא לְמַלָלָא מִדַעַם פִּתְגָמָא דִי יְשַׁוֵי יְיָ בְּפוּמִי יָתֵיהּ אֲמַלֵיל:

{לט}  רביעי - במחוברות ששי  וַיֵּ֥לֶךְ בִּלְעָ֖ם עִם-בָּלָ֑ק וַיָּבֹ֖אוּ קִרְיַ֥ת חֻצֽוֹת:  מ  וַיֵּ֥לֶךְ בִּלְעָ֖ם עִם-בָּלָ֑ק וַיָּבֹ֖אוּ קִרְיַ֥ת חֻצֽוֹת:  ת  וַאֲזַל בִּלְעָם עִם בָּלָק וַאֲתוֹ לְקִרְיַת מְחוֹזוֹהִי:

{מ} וַיִּזְבַּ֥ח בָּלָ֖ק בָּקָ֣ר וָצֹ֑אן וַיְשַׁלַּ֣ח לְבִלְעָ֔ם וְלַשָּׂרִ֖ים אֲשֶׁ֥ר אִתּֽוֹ:  מ  וַיִּזְבַּ֥ח בָּלָ֖ק בָּקָ֣ר וָצֹ֑אן וַיְשַׁלַּ֣ח לְבִלְעָ֔ם וְלַשָּׂרִ֖ים אֲשֶׁ֥ר אִתּֽוֹ:  ת  וּנְכֵיס בָּלָק תּוֹרִין וְעָאן וּשְׁלַח לְבִלְעָם וּלְרַבְרְבַיָא דִי עִמֵיהּ:

{מא} וַיְהִ֣י בַבֹּ֔קֶר וַיִּקַּ֤ח בָּלָק֙ אֶת-בִּלְעָ֔ם וַֽיַּעֲלֵ֖הוּ בָּמ֣וֹת בָּ֑עַל וַיַּ֥רְא מִשָּׁ֖ם קְצֵ֥ה הָעָֽם:  מ  וַיְהִ֣י בַבֹּ֔קֶר וַיִּקַּ֤ח בָּלָק֙ אֶת-בִּלְעָ֔ם וַֽיַּעֲלֵ֖הוּ בָּמ֣וֹת בָּ֑עַל וַיַּ֥רְא מִשָּׁ֖ם קְצֵ֥ה הָעָֽם:  ת  וַהֲוָה בְּצַפְרָא וּדְבַר בָּלָק יַת בִּלְעָם וְאַסְקֵיהּ לְרָמַת דַחַלְתֵּיהּ וַחֲזָא מִתַּמָן קְצַת מִן עַמָא:




במדבר פרק-כג

{א} וַיֹּ֤אמֶר בִּלְעָם֙ אֶל-בָּלָ֔ק בְּנֵה-לִ֥י בָזֶ֖ה שִׁבְעָ֣ה מִזְבְּחֹ֑ת וְהָכֵ֥ן לִי֙ בָּזֶ֔ה שִׁבְעָ֥ה פָרִ֖ים וְשִׁבְעָ֥ה אֵילִֽים:  מ  וַיֹּ֤אמֶר בִּלְעָם֙ אֶל-בָּלָ֔ק בְּנֵה-לִ֥י בָזֶ֖ה שִׁבְעָ֣ה מִזְבְּחֹ֑ת וְהָכֵ֥ן לִי֙ בָּזֶ֔ה שִׁבְעָ֥ה פָרִ֖ים וְשִׁבְעָ֥ה אֵילִֽים:  ת  וַאֲמַר בִּלְעָם לְבָלָק בְּנֵה לִי הָכָא שַׁבְעָא מַדְבְּחִין וְאַתְקִין לִי הָכָא שַׁבְעָא תוֹרִין וְשַׁבְעָא דִכְרִין:

{ב} וַיַּ֣עַשׂ בָּלָ֔ק כַּאֲשֶׁ֖ר דִּבֶּ֣ר בִּלְעָ֑ם וַיַּ֨עַל בָּלָ֧ק וּבִלְעָ֛ם פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ:  מ  וַיַּ֣עַשׂ בָּלָ֔ק כַּאֲשֶׁ֖ר דִּבֶּ֣ר בִּלְעָ֑ם וַיַּ֨עַל בָּלָ֧ק וּבִלְעָ֛ם פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ:  ת  וַעֲבַד בָּלָק כְּמָא דִי מַלֵיל בִּלְעָם וְאַסֵיק בָּלָק וּבִלְעָם תּוֹר וּדְכַר עַל כָּל מַדְבְּחָא:

{ג} וַיֹּ֨אמֶר בִּלְעָ֜ם לְבָלָ֗ק הִתְיַצֵּב֮ עַל-עֹלָתֶךָ֒ וְאֵֽלְכָ֗ה אוּלַ֞י יִקָּרֵ֤ה יְהוָה֙ לִקְרָאתִ֔י וּדְבַ֥ר מַה-יַּרְאֵ֖נִי וְהִגַּ֣דְתִּי לָ֑ךְ וַיֵּ֖לֶךְ שֶֽׁפִי:  מ  וַיֹּ֨אמֶר בִּלְעָ֜ם לְבָלָ֗ק הִתְיַצֵּב֮ עַל-עֹלָתֶךָ֒ וְאֵֽלְכָ֗ה אוּלַ֞י יִקָּרֵ֤ה יְהוָה֙ לִקְרָאתִ֔י וּדְבַ֥ר מַה-יַּרְאֵ֖נִי וְהִגַּ֣דְתִּי לָ֑ךְ וַיֵּ֖לֶךְ שֶֽׁפִי:  ת  וַאֲמַר בִּלְעָם לְבָלָק אִתְעַתֵּד עַל עֲלָתָךְ וְאֵיהַךְ מָאִים יְעָרַע מֵימַר מִן קֳדָם יְיָ לָקֳדָמוּתִי וּפִתְגָמָא דְיַחֲזִנַנִי וְאֵחַוֵי לָךְ וַאֲזַל יְחִידִי:

{ד} וַיִּקָּ֥ר אֱלֹהִ֖ים אֶל-בִּלְעָ֑ם וַיֹּ֣אמֶר אֵלָ֗יו אֶת-שִׁבְעַ֤ת הַֽמִּזְבְּחֹת֙ עָרַ֔כְתִּי וָאַ֛עַל פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ:  מ  וַיִּקָּ֥ר אֱלֹהִ֖ים אֶל-בִּלְעָ֑ם וַיֹּ֣אמֶר אֵלָ֗יו אֶת-שִׁבְעַ֤ת הַֽמִּזְבְּחֹת֙ עָרַ֔כְתִּי וָאַ֛עַל פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ:  ת  וְעָרַע מֵימַר מִן קֳדָם יְיָ לְוַת בִּלְעָם וַאֲמַר לֵיהּ יַת שַׁבְעָא מַדְבְּחִין סְדָרִית וְאַסֵקִית תּוֹר וּדְכַר עַל כָּל מַדְבְּחָא:

{ה} וַיָּ֧שֶׂם יְהוָ֛ה דָּבָ֖ר בְּפִ֣י בִלְעָ֑ם וַיֹּ֛אמֶר שׁ֥וּב אֶל-בָּלָ֖ק וְכֹ֥ה תְדַבֵּֽר:  מ  וַיָּ֧שֶׂם יְהוָ֛ה דָּבָ֖ר בְּפִ֣י בִלְעָ֑ם וַיֹּ֛אמֶר שׁ֥וּב אֶל-בָּלָ֖ק וְכֹ֥ה תְדַבֵּֽר:  ת  וְשַׁוִי יְיָ פִּתְגָמָא בְּפוּמָא דְבִלְעָם וַאֲמַר תּוּב לְוַת בָּלָק וּכְדֵין תְּמַלֵיל:

{ו} וַיָּ֣שָׁב אֵלָ֔יו וְהִנֵּ֥ה נִצָּ֖ב עַל-עֹלָת֑וֹ ה֖וּא וְכָל-שָׂרֵ֥י מוֹאָֽב:  מ  וַיָּ֣שָׁב אֵלָ֔יו וְהִנֵּ֥ה נִצָּ֖ב עַל-עֹלָת֑וֹ ה֖וּא וְכָל-שָׂרֵ֥י מוֹאָֽב:  ת  וְתַב לְוָתֵיהּ וְהָא מְעַתֵּד עַל עֲלָתֵיהּ הוּא וְכָל רַבְרְבֵי מוֹאָב:

{ז} וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר מִן-אֲ֠רָם יַנְחֵ֨נִי בָלָ֤ק מֶֽלֶךְ-מוֹאָב֙ מֵֽהַרְרֵי-קֶ֔דֶם לְכָה֙ אָֽרָה-לִּ֣י יַעֲקֹ֔ב וּלְכָ֖ה זֹעֲמָ֥ה יִשְׂרָאֵֽל:  מ  וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר מִן-אֲ֠רָם יַנְחֵ֨נִי בָלָ֤ק מֶֽלֶךְ-מוֹאָב֙ מֵֽהַרְרֵי-קֶ֔דֶם לְכָה֙ אָֽרָה-לִּ֣י יַעֲקֹ֔ב וּלְכָ֖ה זֹעֲמָ֥ה יִשְׂרָאֵֽל:  ת  וּנְטַל מַתְלֵיהּ וַאֲמַר מִן אֲרָם דַבְּרַנִי בָלָק מַלְכָּא דְמוֹאָב מִטוּרֵי מַדִנְחָא אִיתָא לוֹטּ לִי יַעֲקֹב וְאִיתָא תָּרֵיךְ לִי יִשְׂרָאֵל:

{ח} מָ֣ה אֶקֹּ֔ב לֹ֥א קַבֹּ֖ה אֵ֑ל וּמָ֣ה אֶזְעֹ֔ם לֹ֥א זָעַ֖ם יְהוָֽה:  מ  מָ֣ה אֶקֹּ֔ב לֹ֥א קַבֹּ֖ה אֵ֑ל וּמָ֣ה אֶזְעֹ֔ם לֹ֥א זָעַ֖ם יְהוָֽה:  ת  מָה אֲלוֹטֵיהּ דְלָא לַטְיֵהּ אֵל וּמָה אֲתָרְכֵיהּ דְלָא תָרְכֵיהּ יְיָ:

{ט} כִּֽי-מֵרֹ֤אשׁ צֻרִים֙ אֶרְאֶ֔נּוּ וּמִגְּבָע֖וֹת אֲשׁוּרֶ֑נּוּ הֶן-עָם֙ לְבָדָ֣ד יִשְׁכֹּ֔ן וּבַגּוֹיִ֖ם לֹ֥א יִתְחַשָּֽׁב:  מ  כִּֽי-מֵרֹ֤אשׁ צֻרִים֙ אֶרְאֶ֔נּוּ וּמִגְּבָע֖וֹת אֲשׁוּרֶ֑נּוּ הֶן-עָם֙ לְבָדָ֣ד יִשְׁכֹּ֔ן וּבַגּוֹיִ֖ם לֹ֥א יִתְחַשָּֽׁב:  ת  אֲרֵי מֵרֵישׁ טוּרַיָא חֲזִיתֵיהּ וּמֵרָמָתָא סְכִיתֵיהּ הָא עַמָא בִּלְחוֹדֵיהוֹן עֲתִידִין דְיַחְסְנוּן עַלְמָא וּבְעַמְמַיָא לָא יִתְדְנוּן גְמִירָא:

{י} מִ֤י מָנָה֙ עֲפַ֣ר יַעֲקֹ֔ב וּמִסְפָּ֖ר אֶת-רֹ֣בַע יִשְׂרָאֵ֑ל תָּמֹ֤ת נַפְשִׁי֙ מ֣וֹת יְשָׁרִ֔ים וּתְהִ֥י אַחֲרִיתִ֖י כָּמֹֽהוּ:  מ  מִ֤י מָנָה֙ עֲפַ֣ר יַעֲקֹ֔ב וּמִסְפָּ֖ר אֶת-רֹ֣בַע יִשְׂרָאֵ֑ל תָּמֹ֤ת נַפְשִׁי֙ מ֣וֹת יְשָׁרִ֔ים וּתְהִ֥י אַחֲרִיתִ֖י כָּמֹֽהוּ:  ת  מָן יֵיכוֹל לְמִמְנֵי דַעַדְקַיָא דְבֵית יַעֲקֹב דְאָמִיר עֲלֵיהוֹן יִסְגוּן כְּעַפְרָא דְאַרְעָא אוֹ חֲדָא מֵאַרְבַּע מַשִׁרְיָתָא דְיִשְׂרָאֵל תְּמוּת נַפְשִׁי מוֹתָא דְקַשִׁיטוֹהִי וּתְהֵי סוֹפִי כְּוַתְהוֹן:

{יא} וַיֹּ֤אמֶר בָּלָק֙ אֶל-בִּלְעָ֔ם מֶ֥ה עָשִׂ֖יתָ לִ֑י לָקֹ֤ב אֹיְבַי֙ לְקַחְתִּ֔יךָ וְהִנֵּ֖ה בֵּרַ֥כְתָּ בָרֵֽךְ:  מ  וַיֹּ֤אמֶר בָּלָק֙ אֶל-בִּלְעָ֔ם מֶ֥ה עָשִׂ֖יתָ לִ֑י לָקֹ֤ב אֹיְבַי֙ לְקַחְתִּ֔יךָ וְהִנֵּ֖ה בֵּרַ֥כְתָּ בָרֵֽךְ:  ת  וַאֲמַר בָּלָק לְבִלְעָם מֶה עֲבַדְתָּא לִי לְמֵילַט סַנְאַי דְבַרְתִּיךְ וְהָא בְּרָכָא מְבָרְכַת לְהוֹן:

{יב} וַיַּ֖עַן וַיֹּאמַ֑ר הֲלֹ֗א אֵת֩ אֲשֶׁ֨ר יָשִׂ֤ים יְהוָה֙ בְּפִ֔י אֹת֥וֹ אֶשְׁמֹ֖ר לְדַבֵּֽר:  מ  וַיַּ֖עַן וַיֹּאמַ֑ר הֲלֹ֗א אֵת֩ אֲשֶׁ֨ר יָשִׂ֤ים יְהוָה֙ בְּפִ֔י אֹת֥וֹ אֶשְׁמֹ֖ר לְדַבֵּֽר:  ת  וַאֲתֵיב וַאֲמַר הֲלָא יָת דִי יְשַׁוֵי יְיָ בְּפוּמִי יָתֵיהּ אֶטַר לְמַלָלָא:

{יג}  חמישי  וַיֹּ֨אמֶר אֵלָ֜יו בָּלָ֗ק (לך)-לְכָה-נָּ֨א אִתִּ֜י אֶל-מָק֤וֹם אַחֵר֙ אֲשֶׁ֣ר תִּרְאֶ֣נּוּ מִשָּׁ֔ם אֶ֚פֶס קָצֵ֣הוּ תִרְאֶ֔ה וְכֻלּ֖וֹ לֹ֣א תִרְאֶ֑ה וְקָבְנוֹ-לִ֖י מִשָּֽׁם:  מ  וַיֹּ֨אמֶר אֵלָ֜יו בָּלָ֗ק (לך)-לְכָה-נָּ֨א אִתִּ֜י אֶל-מָק֤וֹם אַחֵר֙ אֲשֶׁ֣ר תִּרְאֶ֣נּוּ מִשָּׁ֔ם אֶ֚פֶס קָצֵ֣הוּ תִרְאֶ֔ה וְכֻלּ֖וֹ לֹ֣א תִרְאֶ֑ה וְקָבְנוֹ-לִ֖י מִשָּֽׁם:  ת  וַאֲמַר לֵיהּ בָּלָק אִיתָא כְעַן עִמִי לַאֲתַר אוֹחֲרָן דְתֶחֱזִנֵיהּ מִתַּמָן לְחוֹד קְצָתֵיהּ תֶּחֱזֵי וְכוּלֵיהּ לָא תֶחֱזֵי וּתְלוּטֵיהּ לִי מִתַּמָן:

{יד} וַיִּקָּחֵ֙הוּ֙ שְׂדֵ֣ה צֹפִ֔ים אֶל-רֹ֖אשׁ הַפִּסְגָּ֑ה וַיִּ֙בֶן֙ שִׁבְעָ֣ה מִזְבְּחֹ֔ת וַיַּ֛עַל פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ:  מ  וַיִּקָּחֵ֙הוּ֙ שְׂדֵ֣ה צֹפִ֔ים אֶל-רֹ֖אשׁ הַפִּסְגָּ֑ה וַיִּ֙בֶן֙ שִׁבְעָ֣ה מִזְבְּחֹ֔ת וַיַּ֛עַל פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ:  ת  וְדַבְּרֵיהּ לַחֲקַל סְכוּתָא לְרֵישׁ רָמָתָא וּבְנָא שַׁבְעָא מַדְבְּחִין וְאַסֵיק תּוֹר וּדְכַר עַל כָּל מַדְבְּחָא:

{טו} וַיֹּ֙אמֶר֙ אֶל-בָּלָ֔ק הִתְיַצֵּ֥ב כֹּ֖ה עַל-עֹלָתֶ֑ךָ וְאָנֹכִ֖י אִקָּ֥רֶה כֹּֽה:  מ  וַיֹּ֙אמֶר֙ אֶל-בָּלָ֔ק הִתְיַצֵּ֥ב כֹּ֖ה עַל-עֹלָתֶ֑ךָ וְאָנֹכִ֖י אִקָּ֥רֶה כֹּֽה:  ת  וַאֲמַר לְבָלָק אִתְעַתֵּד הָכָא עַל עֲלָתָךְ וַאֲנָא אִתְמְטֵי עַד כָּא:

{טז} וַיִּקָּ֤ר יְהוָה֙ אֶל-בִּלְעָ֔ם וַיָּ֥שֶׂם דָּבָ֖ר בְּפִ֑יו וַיֹּ֛אמֶר שׁ֥וּב אֶל-בָּלָ֖ק וְכֹ֥ה תְדַבֵּֽר:  מ  וַיִּקָּ֤ר יְהוָה֙ אֶל-בִּלְעָ֔ם וַיָּ֥שֶׂם דָּבָ֖ר בְּפִ֑יו וַיֹּ֛אמֶר שׁ֥וּב אֶל-בָּלָ֖ק וְכֹ֥ה תְדַבֵּֽר:  ת  וְעָרַע מֵימַר מִן קֳדָם יְיָ לְבִלְעָם וְשַׁוִי פִתְגָמָא בְּפוּמֵיהּ וַאֲמַר תּוּב לְוָת בָּלָק וּכְדֵין תְּמַלֵיל:

{יז} וַיָּבֹ֣א אֵלָ֗יו וְהִנּ֤וֹ נִצָּב֙ עַל-עֹ֣לָת֔וֹ וְשָׂרֵ֥י מוֹאָ֖ב אִתּ֑וֹ וַיֹּ֤אמֶר לוֹ֙ בָּלָ֔ק מַה-דִּבֶּ֖ר יְהוָֽה:  מ  וַיָּבֹ֣א אֵלָ֗יו וְהִנּ֤וֹ נִצָּב֙ עַל-עֹ֣לָת֔וֹ וְשָׂרֵ֥י מוֹאָ֖ב אִתּ֑וֹ וַיֹּ֤אמֶר לוֹ֙ בָּלָ֔ק מַה-דִּבֶּ֖ר יְהוָֽה:  ת  וַאֲתָא לְוָתֵיהּ וְהוּא מְעַתַּד עַל עֲלָתֵיהּ וְרַבְרְבֵי מוֹאָב עִמֵיהּ וַאֲמַר לֵיהּ בָּלָק מָה מַלִיל יְיָ:

{יח} וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר ק֤וּם בָּלָק֙ וּֽשֲׁמָ֔ע הַאֲזִ֥ינָה עָדַ֖י בְּנ֥וֹ צִפֹּֽר:  מ  וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר ק֤וּם בָּלָק֙ וּֽשֲׁמָ֔ע הַאֲזִ֥ינָה עָדַ֖י בְּנ֥וֹ צִפֹּֽר:  ת  וּנְטַל מַתְלֵיהּ וַאֲמַר קוּם בָּלָק וּשְׁמַע אָצִית לְמֵימְרִי בַּר צִפֹּר:

{יט} לֹ֣א אִ֥ישׁ אֵל֙ וִֽיכַזֵּ֔ב וּבֶן-אָדָ֖ם וְיִתְנֶחָ֑ם הַה֤וּא אָמַר֙ וְלֹ֣א יַעֲשֶׂ֔ה וְדִבֶּ֖ר וְלֹ֥א יְקִימֶֽנָּה:  מ  לֹ֣א אִ֥ישׁ אֵל֙ וִֽיכַזֵּ֔ב וּבֶן-אָדָ֖ם וְיִתְנֶחָ֑ם הַה֤וּא אָמַר֙ וְלֹ֣א יַעֲשֶׂ֔ה וְדִבֶּ֖ר וְלֹ֥א יְקִימֶֽנָּה:  ת  לָא כְמִלֵי בְּנֵי אֱנָשָׁא מֵימַר אֱלָהָא בְּנֵי אֱנָשָׁא אָמְרִין וּמְכַדְבִין וְאַף לָא כְעוֹבָדֵי בְּנֵי בִּשְׂרָא דְאִינוּן גָזְרִין לְמֶעְבַּד וְתַיְבִין וּמִתְמַלְכִין דְהוּא אָמַר וְעָבֵיד וְכָל מֵימְרֵהּ מִתְקַיָם:

{כ} הִנֵּ֥ה בָרֵ֖ךְ לָקָ֑חְתִּי וּבֵרֵ֖ךְ וְלֹ֥א אֲשִׁיבֶֽנָּה:  מ  הִנֵּ֥ה בָרֵ֖ךְ לָקָ֑חְתִּי וּבֵרֵ֖ךְ וְלֹ֥א אֲשִׁיבֶֽנָּה:  ת  הָא בִּרְכָן קַבֵּילִית וַאֲבָרֵכִינֵיהּ לְיִשְׂרָאֵל וְלָא אָתֵב בִּרְכְּתִי מִינֵיה:

{כא} לֹֽא-הִבִּ֥יט אָ֙וֶן֙ בְּיַעֲקֹ֔ב וְלֹא-רָאָ֥ה עָמָ֖ל בְּיִשְׂרָאֵ֑ל יְהוָ֤ה אֱלֹהָיו֙ עִמּ֔וֹ וּתְרוּעַ֥ת מֶ֖לֶךְ בּֽוֹ:  מ  לֹֽא-הִבִּ֥יט אָ֙וֶן֙ בְּיַעֲקֹ֔ב וְלֹא-רָאָ֥ה עָמָ֖ל בְּיִשְׂרָאֵ֑ל יְהוָ֤ה אֱלֹהָיו֙ עִמּ֔וֹ וּתְרוּעַ֥ת מֶ֖לֶךְ בּֽוֹ:  ת  אִסְתַּכָּלִית לֵית פָּלְחֵי גִלוּלִין בִּדְבֵית יַעֲקֹב וְאַף לָא עָבְדֵי לֵיאוּת שְׁקַר בְּיִשְׂרָאֵל מֵימְרָא דַיְיָ אֱלָהָהוֹן בְּסַעְדְהוֹן וּשְׁכִינַת מַלְכְּהוֹן בֵּינֵיהוֹן:

{כב} אֵ֖ל מוֹצִיאָ֣ם מִמִּצְרָ֑יִם כְּתוֹעֲפֹ֥ת רְאֵ֖ם לֽוֹ:  מ  אֵ֖ל מוֹצִיאָ֣ם מִמִּצְרָ֑יִם כְּתוֹעֲפֹ֥ת רְאֵ֖ם לֽוֹ:  ת  אֱלָהָא דְאַפֵּקִנוּן מִמִצְרָיִם תּוּקְפָא וְרוּמָא דִילֵיהּ:

{כג} כִּ֤י לֹא-נַ֙חַשׁ֙ בְּיַעֲקֹ֔ב וְלֹא-קֶ֖סֶם בְּיִשְׂרָאֵ֑ל כָּעֵ֗ת יֵאָמֵ֤ר לְיַעֲקֹב֙ וּלְיִשְׂרָאֵ֔ל מַה-פָּ֖עַל אֵֽל:  מ  כִּ֤י לֹא-נַ֙חַשׁ֙ בְּיַעֲקֹ֔ב וְלֹא-קֶ֖סֶם בְּיִשְׂרָאֵ֑ל כָּעֵ֗ת יֵאָמֵ֤ר לְיַעֲקֹב֙ וּלְיִשְׂרָאֵ֔ל מַה-פָּ֖עַל אֵֽל:  ת  אֲרֵי לָא נַחֲשַׁיָא צָבָן דְיִיטַב לִדְבֵית יַעֲקֹב וְאַף לָא קְסָמַיָא רְעַן בִּרְבוּת בֵּית יִשְׂרָאֵל כְּעִדַן יִתְאַמַר לְיַעֲקֹב וּלְיִשְׂרָאֵל מָה עֲבַד אֱלָהָא:

{כד} הֶן-עָם֙ כְּלָבִ֣יא יָק֔וּם וְכַאֲרִ֖י יִתְנַשָּׂ֑א לֹ֤א יִשְׁכַּב֙ עַד-יֹ֣אכַל טֶ֔רֶף וְדַם-חֲלָלִ֖ים יִשְׁתֶּֽה:  מ  הֶן-עָם֙ כְּלָבִ֣יא יָק֔וּם וְכַאֲרִ֖י יִתְנַשָּׂ֑א לֹ֤א יִשְׁכַּב֙ עַד-יֹ֣אכַל טֶ֔רֶף וְדַם-חֲלָלִ֖ים יִשְׁתֶּֽה:  ת  הָא עַמָא כְּלֵיתָא שָׁרֵי וּכְאַרְיָא יִתְנַטֵל לָא יִשְׁרֵי בְאַרְעֵיהּ עַד דְיִקְטוֹל קְטוֹל וְנִכְסֵי עֲמְמַיָא יֵירָת:

{כה} וַיֹּ֤אמֶר בָּלָק֙ אֶל-בִּלְעָ֔ם גַּם-קֹ֖ב לֹ֣א תִקֳּבֶ֑נּוּ גַּם-בָּרֵ֖ךְ לֹ֥א תְבָרֲכֶֽנּוּ:  מ  וַיֹּ֤אמֶר בָּלָק֙ אֶל-בִּלְעָ֔ם גַּם-קֹ֖ב לֹ֣א תִקֳּבֶ֑נּוּ גַּם-בָּרֵ֖ךְ לֹ֥א תְבָרֲכֶֽנּוּ:  ת  וַאֲמַר בָּלָק לְבִלְעָם אַף מֵילַט לָא תְלוּטִּינוּן אַף בְּרָכָא לָא תְבָרֵכִינוּן:

{כו} וַיַּ֣עַן בִּלְעָ֔ם וַיֹּ֖אמֶר אֶל-בָּלָ֑ק הֲלֹ֗א דִּבַּ֤רְתִּי אֵלֶ֙יךָ֙ לֵאמֹ֔ר כֹּ֛ל אֲשֶׁר-יְדַבֵּ֥ר יְהוָ֖ה אֹת֥וֹ אֶֽעֱשֶֽׂה:  מ  וַיַּ֣עַן בִּלְעָ֔ם וַיֹּ֖אמֶר אֶל-בָּלָ֑ק הֲלֹ֗א דִּבַּ֤רְתִּי אֵלֶ֙יךָ֙ לֵאמֹ֔ר כֹּ֛ל אֲשֶׁר-יְדַבֵּ֥ר יְהוָ֖ה אֹת֥וֹ אֶֽעֱשֶֽׂה:  ת  וַאֲתֵיב בִּלְעָם וַאֲמַר לְבָלָק הֲלָא מַלֵילִית עִמָךְ לְמֵימַר כֹּל דִי יְמַלֵל יְיָ יָתֵיהּ אַעֲבֵיד:

{כז}  שישי - במחוברות שביעי  וַיֹּ֤אמֶר בָּלָק֙ אֶל-בִּלְעָ֔ם לְכָה-נָּא֙ אֶקָּ֣חֲךָ֔ אֶל-מָק֖וֹם אַחֵ֑ר אוּלַ֤י יִישַׁר֙ בְּעֵינֵ֣י הָאֱלֹהִ֔ים וְקַבֹּ֥תוֹ לִ֖י מִשָּֽׁם:  מ  וַיֹּ֤אמֶר בָּלָק֙ אֶל-בִּלְעָ֔ם לְכָה-נָּא֙ אֶקָּ֣חֲךָ֔ אֶל-מָק֖וֹם אַחֵ֑ר אוּלַ֤י יִישַׁר֙ בְּעֵינֵ֣י הָאֱלֹהִ֔ים וְקַבֹּ֥תוֹ לִ֖י מִשָּֽׁם:  ת  וַאֲמַר בָּלָק לְבִלְעָם אִיתָא כְּעַן אֲדַבְּרִינָךְ לַאֲתַר אָחֳרָן מָאִים יְהֵי רַעֲוָא מִן קֳדָם יְיָ וּתְלוּטֵיהּ לִי מִתַּמָן:

{כח} וַיִּקַּ֥ח בָּלָ֖ק אֶת-בִּלְעָ֑ם רֹ֣אשׁ הַפְּע֔וֹר הַנִּשְׁקָ֖ף עַל-פְּנֵ֥י הַיְשִׁימֹֽן:  מ  וַיִּקַּ֥ח בָּלָ֖ק אֶת-בִּלְעָ֑ם רֹ֣אשׁ הַפְּע֔וֹר הַנִּשְׁקָ֖ף עַל-פְּנֵ֥י הַיְשִׁימֹֽן:  ת  וּדְבַר בָּלָק יַת בִּלְעָם רֵישׁ רָמָתָא דְמִסְתַּכְּיָא עַל אַפֵּי בֵּית יְשִׁימוֹן:

{כט} וַיֹּ֤אמֶר בִּלְעָם֙ אֶל-בָּלָ֔ק בְּנֵה-לִ֥י בָזֶ֖ה שִׁבְעָ֣ה מִזְבְּחֹ֑ת וְהָכֵ֥ן לִי֙ בָּזֶ֔ה שִׁבְעָ֥ה פָרִ֖ים וְשִׁבְעָ֥ה אֵילִֽים:  מ  וַיֹּ֤אמֶר בִּלְעָם֙ אֶל-בָּלָ֔ק בְּנֵה-לִ֥י בָזֶ֖ה שִׁבְעָ֣ה מִזְבְּחֹ֑ת וְהָכֵ֥ן לִי֙ בָּזֶ֔ה שִׁבְעָ֥ה פָרִ֖ים וְשִׁבְעָ֥ה אֵילִֽים:  ת  וַאֲמַר בִּלְעָם לְבָלָק בְּנֵה לִי הָכָא שַׁבְעָא מַדְבְּחִין וְאַתְקִין לִי הָכָא שַׁבְעָא תוֹרִין וְשַׁבְעָא דִכְרִין:

{ל} וַיַּ֣עַשׂ בָּלָ֔ק כַּאֲשֶׁ֖ר אָמַ֣ר בִּלְעָ֑ם וַיַּ֛עַל פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ:  מ  וַיַּ֣עַשׂ בָּלָ֔ק כַּאֲשֶׁ֖ר אָמַ֣ר בִּלְעָ֑ם וַיַּ֛עַל פָּ֥ר וָאַ֖יִל בַּמִּזְבֵּֽחַ:  ת  וַעֲבַד בָּלָק כְּמָא דִי אֲמַר בִּלְעָם וְאַסֵיק תּוֹר וּדְכַר עַל כָּל מַדְבְּחָא:




במדבר פרק-כד

{א} וַיַּ֣רְא בִּלְעָ֗ם כִּ֣י ט֞וֹב בְּעֵינֵ֤י יְהוָה֙ לְבָרֵ֣ךְ אֶת-יִשְׂרָאֵ֔ל וְלֹא-הָלַ֥ךְ כְּפַֽעַם-בְּפַ֖עַם לִקְרַ֣את נְחָשִׁ֑ים וַיָּ֥שֶׁת אֶל-הַמִּדְבָּ֖ר פָּנָֽיו:  מ  וַיַּ֣רְא בִּלְעָ֗ם כִּ֣י ט֞וֹב בְּעֵינֵ֤י יְהוָה֙ לְבָרֵ֣ךְ אֶת-יִשְׂרָאֵ֔ל וְלֹא-הָלַ֥ךְ כְּפַֽעַם-בְּפַ֖עַם לִקְרַ֣את נְחָשִׁ֑ים וַיָּ֥שֶׁת אֶל-הַמִּדְבָּ֖ר פָּנָֽיו:  ת  וַחֲזָא בִלְעָם אֲרֵי תַקִין קֳדָם יְיָ לְבָרָכָא יַת יִשְׂרָאֵל וְלָא הַלַךְ כִּזְמַן בִּזְמַן אֱלָהֵין לָקֳדָמוּת נְחָשַׁיָא וְשַׁוִי לָקָבֵיל עֶגְלָא דַעֲבָדוּ יִשְׂרָאֵל בְּמַדְבְּרָא אַפּוֹהִי:

{ב} וַיִּשָּׂ֨א בִלְעָ֜ם אֶת-עֵינָ֗יו וַיַּרְא֙ אֶת-יִשְׂרָאֵ֔ל שֹׁכֵ֖ן לִשְׁבָטָ֑יו וַתְּהִ֥י עָלָ֖יו ר֥וּחַ אֱלֹהִֽים:  מ  וַיִּשָּׂ֨א בִלְעָ֜ם אֶת-עֵינָ֗יו וַיַּרְא֙ אֶת-יִשְׂרָאֵ֔ל שֹׁכֵ֖ן לִשְׁבָטָ֑יו וַתְּהִ֥י עָלָ֖יו ר֥וּחַ אֱלֹהִֽים:  ת  וּזְקֵף בִּלְעָם יָת עֵינוֹהִי וַחֲזָא יָת יִשְׂרָאֵל שָׁרָן לְשִׁבְטוֹהִי וּשְׁרַת עֲלוֹהִי רוּחַ נְבוּאָה מִן קֳדָם יְיָ:

{ג} וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר נְאֻ֤ם בִּלְעָם֙ בְּנ֣וֹ בְעֹ֔ר וּנְאֻ֥ם הַגֶּ֖בֶר שְׁתֻ֥ם הָעָֽיִן:  מ  וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר נְאֻ֤ם בִּלְעָם֙ בְּנ֣וֹ בְעֹ֔ר וּנְאֻ֥ם הַגֶּ֖בֶר שְׁתֻ֥ם הָעָֽיִן:  ת  וּנְטַל מַתְלֵיהּ וַאֲמַר אֵימַר בִּלְעָם בְּרֵיהּ בְעוֹר וְאֵימַר גַבְרָא דְשַׁפִּיר חָזֵי:

{ד} נְאֻ֕ם שֹׁמֵ֖עַ אִמְרֵי-אֵ֑ל אֲשֶׁ֨ר מַחֲזֵ֤ה שַׁדַּי֙ יֶֽחֱזֶ֔ה נֹפֵ֖ל וּגְל֥וּי עֵינָֽיִם:  מ  נְאֻ֕ם שֹׁמֵ֖עַ אִמְרֵי-אֵ֑ל אֲשֶׁ֨ר מַחֲזֵ֤ה שַׁדַּי֙ יֶֽחֱזֶ֔ה נֹפֵ֖ל וּגְל֥וּי עֵינָֽיִם:  ת  אֵימַר דִשְׁמַע מֵימַר מִן קֳדָם אֵל וְחֵזוּ מִן קֳדָם שַׁדַי חָזֵי שְׁכִיב וּמִתְגְלֵי לֵיהּ:

{ה} מַה-טֹּ֥בוּ אֹהָלֶ֖יךָ יַעֲקֹ֑ב מִשְׁכְּנֹתֶ֖יךָ יִשְׂרָאֵֽל:  מ  מַה-טֹּ֥בוּ אֹהָלֶ֖יךָ יַעֲקֹ֑ב מִשְׁכְּנֹתֶ֖יךָ יִשְׂרָאֵֽל:  ת  מָה טָבָא מַשְׁכְּנָךְ יַעֲקֹב בֵּית מֵישְׁרָךְ יִשְׂרָאֵל:

{ו} כִּנְחָלִ֣ים נִטָּ֔יוּ כְּגַנֹּ֖ת עֲלֵ֣י נָהָ֑ר כַּאֲהָלִים֙ נָטַ֣ע יְהוָ֔ה כַּאֲרָזִ֖ים עֲלֵי-מָֽיִם:  מ  כִּנְחָלִ֣ים נִטָּ֔יוּ כְּגַנֹּ֖ת עֲלֵ֣י נָהָ֑ר כַּאֲהָלִים֙ נָטַ֣ע יְהוָ֔ה כַּאֲרָזִ֖ים עֲלֵי-מָֽיִם:  ת  כְּנַחֲלִין דִמְדַבְּרִין כְּגִינַת שַׁקְיָא דְעַל פְּרָת כְּבוּסְמַיָא דִנְצִיב יְיָ כְּאַרְזִין דִנְצִבִין עַל מַיָא:

{ז} יִֽזַּל-מַ֙יִם֙ מִדָּ֣לְיָ֔ו וְזַרְע֖וֹ בְּמַ֣יִם רַבִּ֑ים וְיָרֹ֤ם מֵֽאֲגַג֙ מַלְכּ֔וֹ וְתִנַּשֵּׂ֖א מַלְכֻתֽוֹ:  מ  יִֽזַּל-מַ֙יִם֙ מִדָּ֣לְיָ֔ו וְזַרְע֖וֹ בְּמַ֣יִם רַבִּ֑ים וְיָרֹ֤ם מֵֽאֲגַג֙ מַלְכּ֔וֹ וְתִנַּשֵּׂ֖א מַלְכֻתֽוֹ:  ת  יִסְגֵי מַלְכָּא דְיִתְרַבָּא מִבְּנוֹהִי וְיִשְׁלוֹט בְּעַמְמִין סַגִיאִין וְיִתְקוֹף מֵאֲגַג מַלְכֵּיהּ וְתִתְנַטֵל מַלְכוּתֵיהּ:

{ח} אֵ֚ל מוֹצִיא֣וֹ מִמִּצְרַ֔יִם כְּתוֹעֲפֹ֥ת רְאֵ֖ם ל֑וֹ יֹאכַ֞ל גּוֹיִ֣ם צָרָ֗יו וְעַצְמֹתֵיהֶ֛ם יְגָרֵ֖ם וְחִצָּ֥יו יִמְחָֽץ:  מ  אֵ֚ל מוֹצִיא֣וֹ מִמִּצְרַ֔יִם כְּתוֹעֲפֹ֥ת רְאֵ֖ם ל֑וֹ יֹאכַ֞ל גּוֹיִ֣ם צָרָ֗יו וְעַצְמֹתֵיהֶ֛ם יְגָרֵ֖ם וְחִצָּ֥יו יִמְחָֽץ:  ת  אֱלָהָא דְאַפֵּקִינוּן מִמִצְרַיִם תּוּקְפָא וְרוּמָא דִילֵיהּ יֵיכְלוּן בְּנֵי יִשְׂרָאֵל נִכְסֵי עַמְמַיָא סַנְאֵיהוֹן וּבְבִזַת מַלְכֵּיהוֹן יִתְפַּנְקוּן וְאַרְעֲתְהוֹן יַחְסְנוּן:

{ט} כָּרַ֨ע שָׁכַ֧ב כַּאֲרִ֛י וּכְלָבִ֖יא מִ֣י יְקִימֶ֑נּוּ מְבָרֲכֶ֣יךָ בָר֔וּךְ וְאֹרְרֶ֖יךָ אָרֽוּר:  מ  כָּרַ֨ע שָׁכַ֧ב כַּאֲרִ֛י וּכְלָבִ֖יא מִ֣י יְקִימֶ֑נּוּ מְבָרֲכֶ֣יךָ בָר֔וּךְ וְאֹרְרֶ֖יךָ אָרֽוּר:  ת  יְנוּחַ יִשְׁרֵי בִתְקוֹף כְּאַרְיָא וּכְלֵתָא לֵית מַלְכוּ דְתִזַעֲזְעִינֵיהּ בְּרִיכָךְ יְהוֹן בְּרִיכִין וְלִיטָךְ יְהוֹן לִיטִין:

{י} וַיִּֽחַר-אַ֤ף בָּלָק֙ אֶל-בִּלְעָ֔ם וַיִּסְפֹּ֖ק אֶת-כַּפָּ֑יו וַיֹּ֨אמֶר בָּלָ֜ק אֶל-בִּלְעָ֗ם לָקֹ֤ב אֹֽיְבַי֙ קְרָאתִ֔יךָ וְהִנֵּה֙ בֵּרַ֣כְתָּ בָרֵ֔ךְ זֶ֖ה שָׁלֹ֥שׁ פְּעָמִֽים:  מ  וַיִּֽחַר-אַ֤ף בָּלָק֙ אֶל-בִּלְעָ֔ם וַיִּסְפֹּ֖ק אֶת-כַּפָּ֑יו וַיֹּ֨אמֶר בָּלָ֜ק אֶל-בִּלְעָ֗ם לָקֹ֤ב אֹֽיְבַי֙ קְרָאתִ֔יךָ וְהִנֵּה֙ בֵּרַ֣כְתָּ בָרֵ֔ךְ זֶ֖ה שָׁלֹ֥שׁ פְּעָמִֽים:  ת  וּתְקֵיף רוּגְזָא דְבָלָק בְּבִלְעָם וּשְׁקָפִינוּן לִידוֹהִי וַאֲמַר בָּלָק לְבִלְעָם לְמֵילַט סַנְאַי קְרִיתָךְ וְהָא בָּרָכָא מְבָרְכַת לְהוֹן דְנַן תְּלַת זִמְנִין:

{יא} וְעַתָּ֖ה בְּרַח-לְךָ֣ אֶל-מְקוֹמֶ֑ךָ אָמַ֙רְתִּי֙ כַּבֵּ֣ד אֲכַבֶּדְךָ֔ וְהִנֵּ֛ה מְנָעֲךָ֥ יְהוָ֖ה מִכָּבֽוֹד:  מ  וְעַתָּ֖ה בְּרַח-לְךָ֣ אֶל-מְקוֹמֶ֑ךָ אָמַ֙רְתִּי֙ כַּבֵּ֣ד אֲכַבֶּדְךָ֔ וְהִנֵּ֛ה מְנָעֲךָ֥ יְהוָ֖ה מִכָּבֽוֹד:  ת  וּכְעַן אִזֵיל לָךְ לְאַתְרָךְ אֲמָרִית יְקָרָא אֲיַקְרִינָךְ וְהָא מְנָעָךְ יְיָ מִן יְקָר:

{יב} וַיֹּ֥אמֶר בִּלְעָ֖ם אֶל-בָּלָ֑ק הֲלֹ֗א גַּ֧ם אֶל-מַלְאָכֶ֛יךָ אֲשֶׁר-שָׁלַ֥חְתָּ אֵלַ֖י דִּבַּ֥רְתִּי לֵאמֹֽר:  מ  וַיֹּ֥אמֶר בִּלְעָ֖ם אֶל-בָּלָ֑ק הֲלֹ֗א גַּ֧ם אֶל-מַלְאָכֶ֛יךָ אֲשֶׁר-שָׁלַ֥חְתָּ אֵלַ֖י דִּבַּ֥רְתִּי לֵאמֹֽר:  ת  וַאֲמַר בִּלְעָם לְבָלָּק הֲלָא אַף לְאִזְגַדָךְ דִי שַׁלְחַת לְוָתִי מַלֵילִית לְמֵימָר:

{יג} אִם-יִתֶּן-לִ֨י בָלָ֜ק מְלֹ֣א בֵיתוֹ֮ כֶּ֣סֶף וְזָהָב֒ לֹ֣א אוּכַ֗ל לַעֲבֹר֙ אֶת-פִּ֣י יְהוָ֔ה לַעֲשׂ֥וֹת טוֹבָ֛ה א֥וֹ רָעָ֖ה מִלִּבִּ֑י אֲשֶׁר-יְדַבֵּ֥ר יְהוָ֖ה אֹת֥וֹ אֲדַבֵּֽר:  מ  אִם-יִתֶּן-לִ֨י בָלָ֜ק מְלֹ֣א בֵיתוֹ֮ כֶּ֣סֶף וְזָהָב֒ לֹ֣א אוּכַ֗ל לַעֲבֹר֙ אֶת-פִּ֣י יְהוָ֔ה לַעֲשׂ֥וֹת טוֹבָ֛ה א֥וֹ רָעָ֖ה מִלִּבִּ֑י אֲשֶׁר-יְדַבֵּ֥ר יְהוָ֖ה אֹת֥וֹ אֲדַבֵּֽר:  ת  אִם יִתֶּן לִי בָלָק מְלֵי בֵיתֵיהּ כְּסַף וּדְהַב לֵית לִי רְשׁוּ לְמֶעֱבַּר עַל גְזֵירַת מֵימְרָא דַיְיָ לְמֶעֱבַּד טַבְתָּא אוֹ בִּישְׁתָּא מֵרְעוּתִי דִי יְמַלֵל יְיָ יָתֵיהּ אֲמַלֵיל:

{יד}  שביעי  וְעַתָּ֕ה הִנְנִ֥י הוֹלֵ֖ךְ לְעַמִּ֑י לְכָה֙ אִיעָ֣צְךָ֔ אֲשֶׁ֨ר יַעֲשֶׂ֜ה הָעָ֥ם הַזֶּ֛ה לְעַמְּךָ֖ בְּאַחֲרִ֥ית הַיָּמִֽים:  מ  וְעַתָּ֕ה הִנְנִ֥י הוֹלֵ֖ךְ לְעַמִּ֑י לְכָה֙ אִיעָ֣צְךָ֔ אֲשֶׁ֨ר יַעֲשֶׂ֜ה הָעָ֥ם הַזֶּ֛ה לְעַמְּךָ֖ בְּאַחֲרִ֥ית הַיָּמִֽים:  ת  וּכְעַן הָא אֲנָא אָזֵיל לְעַמִי אִיתָא אֲמַלְכִינָךְ מָה דְתַּעְבֵיד וַאֲחַוֵי לָךְ מָא דְיַעֲבֵד עַמָא הָדֵין לְעַמָךְ בְּסוֹף יוֹמַיָא:

{טו} וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר נְאֻ֤ם בִּלְעָם֙ בְּנ֣וֹ בְעֹ֔ר וּנְאֻ֥ם הַגֶּ֖בֶר שְׁתֻ֥ם הָעָֽיִן:  מ  וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר נְאֻ֤ם בִּלְעָם֙ בְּנ֣וֹ בְעֹ֔ר וּנְאֻ֥ם הַגֶּ֖בֶר שְׁתֻ֥ם הָעָֽיִן:  ת  וּנְטַל מַתְלֵיהּ וַאֲמַר אֵימַר בִּלְעָם בְּרֵיהּ בְעוֹר וְאֵימַר גַבְרָא דְשַׁפִּיר חָזֵי:

{טז} נְאֻ֗ם שֹׁמֵ֙עַ֙ אִמְרֵי-אֵ֔ל וְיֹדֵ֖עַ דַּ֣עַת עֶלְי֑וֹן מַחֲזֵ֤ה שַׁדַּי֙ יֶֽחֱזֶ֔ה נֹפֵ֖ל וּגְל֥וּי עֵינָֽיִם:  מ  נְאֻ֗ם שֹׁמֵ֙עַ֙ אִמְרֵי-אֵ֔ל וְיֹדֵ֖עַ דַּ֣עַת עֶלְי֑וֹן מַחֲזֵ֤ה שַׁדַּי֙ יֶֽחֱזֶ֔ה נֹפֵ֖ל וּגְל֥וּי עֵינָֽיִם:  ת  אֵימַר דִשְׁמַע מֵימַר מִן קֳדָם אֵל וִידַע מַדַע מִן קֳדָם עִילָאָה חֵיזוּ מִן קֳדָם שַׁדַי חָזֵי שְׁכִיב וּמִתְגְלֵי לֵיהּ:

{יז} אֶרְאֶ֙נּוּ֙ וְלֹ֣א עַתָּ֔ה אֲשׁוּרֶ֖נּוּ וְלֹ֣א קָר֑וֹב דָּרַ֨ךְ כּוֹכָ֜ב מִֽיַּעֲקֹ֗ב וְקָ֥ם שֵׁ֙בֶט֙ מִיִּשְׂרָאֵ֔ל וּמָחַץ֙ פַּאֲתֵ֣י מוֹאָ֔ב וְקַרְקַ֖ר כָּל-בְּנֵי-שֵֽׁת:  מ  אֶרְאֶ֙נּוּ֙ וְלֹ֣א עַתָּ֔ה אֲשׁוּרֶ֖נּוּ וְלֹ֣א קָר֑וֹב דָּרַ֨ךְ כּוֹכָ֜ב מִֽיַּעֲקֹ֗ב וְקָ֥ם שֵׁ֙בֶט֙ מִיִּשְׂרָאֵ֔ל וּמָחַץ֙ פַּאֲתֵ֣י מוֹאָ֔ב וְקַרְקַ֖ר כָּל-בְּנֵי-שֵֽׁת:  ת  חֲזִיתֵיהּ וְלָא כְעַן סְכִיתֵיהּ וְלָא אִיתוֹהִי קָרִיב כַּד יְקוּם מַלְכָּא מִיַעֲקֹב וְיִתְרַבָּא מְשִׁיחָא מִיִשְׂרָאֵל וְיִקְטּוֹל רַבְרְבֵי מוֹאָב וְיִשְׁלוֹט בְּכָל בְּנֵי אֱנָשָׁא:

{יח} וְהָיָ֨ה אֱד֜וֹם יְרֵשָׁ֗ה וְהָיָ֧ה יְרֵשָׁ֛ה שֵׂעִ֖יר אֹיְבָ֑יו וְיִשְׂרָאֵ֖ל עֹ֥שֶׂה חָֽיִל:  מ  וְהָיָ֨ה אֱד֜וֹם יְרֵשָׁ֗ה וְהָיָ֧ה יְרֵשָׁ֛ה שֵׂעִ֖יר אֹיְבָ֑יו וְיִשְׂרָאֵ֖ל עֹ֥שֶׂה חָֽיִל:  ת  וִיהֵי אֱדוֹם יְרוּתָּא וִיהֵי יְרוּתָּא שֵׂעִיר לְבַעֲלֵי דְבָבוֹהִי וְיִשְׂרָאֵל יַצְלַח בְּנִכְסִין:

{יט} וְיֵ֖רְדְּ מִֽיַּעֲקֹ֑ב וְהֶֽאֱבִ֥יד שָׂרִ֖יד מֵעִֽיר:  מ  וְיֵ֖רְדְּ מִֽיַּעֲקֹ֑ב וְהֶֽאֱבִ֥יד שָׂרִ֖יד מֵעִֽיר:  ת  וְיֵיחוּת חָד מִדְבֵית יַעֲקֹב וְיוֹבִיד מְשֵׁיזִיב מִקִרְיַת עַמְמַיָא:

{כ} וַיַּרְא֙ אֶת-עֲמָלֵ֔ק וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר רֵאשִׁ֤ית גּוֹיִם֙ עֲמָלֵ֔ק וְאַחֲרִית֖וֹ עֲדֵ֥י אֹבֵֽד:  מ  וַיַּרְא֙ אֶת-עֲמָלֵ֔ק וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר רֵאשִׁ֤ית גּוֹיִם֙ עֲמָלֵ֔ק וְאַחֲרִית֖וֹ עֲדֵ֥י אֹבֵֽד:  ת  וַחֲזָא יַת עֲמַלְקָאָה וּנְטַל מַתְלֵיהּ וַאַמַר רֵישׁ קְרָבַיָא דְיִשְׂרָאֵל הֲוָה עֲמַלְקָאָה וְסוֹפֵיהּ לְעָלְמָא יֵיבָד:

{כא} וַיַּרְא֙ אֶת-הַקֵּינִ֔י וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר אֵיתָן֙ מֽוֹשָׁבֶ֔ךָ וְשִׂ֥ים בַּסֶּ֖לַע קִנֶּֽךָ:  מ  וַיַּרְא֙ אֶת-הַקֵּינִ֔י וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר אֵיתָן֙ מֽוֹשָׁבֶ֔ךָ וְשִׂ֥ים בַּסֶּ֖לַע קִנֶּֽךָ:  ת  וַחֲזָא יַת שַׁלְמָאָה וּנְטַל מַתְלֵיהּ וַאֲמַר תַּקִיף בֵּית מוֹתְבָךְ וְשַׁוֵי בִכְרַךְ תַּקִּיף מְדוֹרָךְ:

{כב} כִּ֥י אִם-יִהְיֶ֖ה לְבָ֣עֵֽר קָ֑יִן עַד-מָ֖ה אַשּׁ֥וּר תִּשְׁבֶּֽךָּ:  מ  כִּ֥י אִם-יִהְיֶ֖ה לְבָ֣עֵֽר קָ֑יִן עַד-מָ֖ה אַשּׁ֥וּר תִּשְׁבֶּֽךָּ:  ת  אֲרֵי אִם יְהֵי לְשֵׁיצָאָה שַׁלְמָאָה עַד מָה אֲתוּרָאָה יִשְׁבִּינָךְ:

{כג} וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר א֕וֹי מִ֥י יִחְיֶ֖ה מִשֻּׂמ֥וֹ אֵֽל:  מ  וַיִּשָּׂ֥א מְשָׁל֖וֹ וַיֹּאמַ֑ר א֕וֹי מִ֥י יִחְיֶ֖ה מִשֻּׂמ֥וֹ אֵֽל:  ת  וּנְטַל מַתְלֵיהּ וַאֲמַר וַי לְחַיָיבַיָא דְיֵחוֹן כַּד יַעֲבֵּד אֱלָהָא יָת אִלֵין:

{כד} וְצִים֙ מִיַּ֣ד כִּתִּ֔ים וְעִנּ֥וּ אַשּׁ֖וּר וְעִנּוּ-עֵ֑בֶר וְגַם-ה֖וּא עֲדֵ֥י אֹבֵֽד:  מ  וְצִים֙ מִיַּ֣ד כִּתִּ֔ים וְעִנּ֥וּ אַשּׁ֖וּר וְעִנּוּ-עֵ֑בֶר וְגַם-ה֖וּא עֲדֵ֥י אֹבֵֽד:  ת  וְסִיעַן יִצְטַרְחַן מֵרוֹמָאֵי וִיעֲנוּן לְאָתוּר וְיִשְׁתַּעְבְּדוּן לְעֵבַר פְּרָת וְאַף אִינוּן לְעָלְמָא יֵיבְדוּן:

{כה} וַיָּ֣קָם בִּלְעָ֔ם וַיֵּ֖לֶךְ וַיָּ֣שָׁב לִמְקֹמ֑וֹ וְגַם-בָּלָ֖ק הָלַ֥ךְ לְדַרְכּֽוֹ:  מ  וַיָּ֣קָם בִּלְעָ֔ם וַיֵּ֖לֶךְ וַיָּ֣שָׁב לִמְקֹמ֑וֹ וְגַם-בָּלָ֖ק הָלַ֥ךְ לְדַרְכּֽוֹ:  ת  וְקָם בִּלְעָם וַאֲזַל וְתָב לְאַתְרֵיהּ וְאַף בָּלָק אֲזַל לְאוֹרְחֵיהּ: (פ)




במדבר פרק-כה

{א} וַיֵּ֥שֶׁב יִשְׂרָאֵ֖ל בַּשִּׁטִּ֑ים וַיָּ֣חֶל הָעָ֔ם לִזְנ֖וֹת אֶל-בְּנ֥וֹת מוֹאָֽב:  מ  וַיֵּ֥שֶׁב יִשְׂרָאֵ֖ל בַּשִּׁטִּ֑ים וַיָּ֣חֶל הָעָ֔ם לִזְנ֖וֹת אֶל-בְּנ֥וֹת מוֹאָֽב:  ת  וִיתֵיב יִשְׂרָאֵל בְּשִׁיטִין וּשְׁרֵי עַמָא לְמִטְּעֵי בָּתַר בְּנַת מוֹאָב:

{ב} וַתִּקְרֶ֣אןָ לָעָ֔ם לְזִבְחֵ֖י אֱלֹהֵיהֶ֑ן וַיֹּ֣אכַל הָעָ֔ם וַיִּֽשְׁתַּחֲוּ֖וּ לֵֽאלֹהֵיהֶֽן:  מ  וַתִּקְרֶ֣אןָ לָעָ֔ם לְזִבְחֵ֖י אֱלֹהֵיהֶ֑ן וַיֹּ֣אכַל הָעָ֔ם וַיִּֽשְׁתַּחֲוּ֖וּ לֵֽאלֹהֵיהֶֽן:  ת  וּקְרָאָן לְעַמָא לְדִבְחֵי טַעֲוַתְהוֹן וַאֲכַל עַמָא וּסְגִידוּ לְטַעֲוַתְהוֹן:

{ג} וַיִּצָּ֥מֶד יִשְׂרָאֵ֖ל לְבַ֣עַל פְּע֑וֹר וַיִּֽחַר-אַ֥ף יְהוָ֖ה בְּיִשְׂרָאֵֽל:  מ  וַיִּצָּ֥מֶד יִשְׂרָאֵ֖ל לְבַ֣עַל פְּע֑וֹר וַיִּֽחַר-אַ֥ף יְהוָ֖ה בְּיִשְׂרָאֵֽל:  ת  וְאִתְחַבַּר יִשְׂרָאֵל לְבַעֲלָא פְעוֹר וּתְקֵף רוּגְזָא דַיְיָ בְּיִשְׂרָאֵל:

{ד} וַיֹּ֨אמֶר יְהוָ֜ה אֶל-מֹשֶׁ֗ה קַ֚ח אֶת-כָּל-רָאשֵׁ֣י הָעָ֔ם וְהוֹקַ֥ע אוֹתָ֛ם לַיהוָ֖ה נֶ֣גֶד הַשָּׁ֑מֶשׁ וְיָשֹׁ֛ב חֲר֥וֹן אַף-יְהוָ֖ה מִיִּשְׂרָאֵֽל:  מ  וַיֹּ֨אמֶר יְהוָ֜ה אֶל-מֹשֶׁ֗ה קַ֚ח אֶת-כָּל-רָאשֵׁ֣י הָעָ֔ם וְהוֹקַ֥ע אוֹתָ֛ם לַיהוָ֖ה נֶ֣גֶד הַשָּׁ֑מֶשׁ וְיָשֹׁ֛ב חֲר֥וֹן אַף-יְהוָ֖ה מִיִּשְׂרָאֵֽל:  ת  וַאֲמַר יְיָ לְמשֶׁה דְבַר יַת כָּל רֵישֵׁי עַמָא וְדוּן וּקְטוֹל דְחַיָב קְטוֹל קֳדָם יְיָ לָקֳבֵל שִׁמְשָׁא וִיתוּב תְּקוֹף רוּגְזָא דַיְיָ מִיִשְׂרָאֵל:

{ה} וַיֹּ֣אמֶר מֹשֶׁ֔ה אֶל-שֹׁפְטֵ֖י יִשְׂרָאֵ֑ל הִרְגוּ֙ אִ֣ישׁ אֲנָשָׁ֔יו הַנִּצְמָדִ֖ים לְבַ֥עַל פְּעֽוֹר:  מ  וַיֹּ֣אמֶר מֹשֶׁ֔ה אֶל-שֹׁפְטֵ֖י יִשְׂרָאֵ֑ל הִרְגוּ֙ אִ֣ישׁ אֲנָשָׁ֔יו הַנִּצְמָדִ֖ים לְבַ֥עַל פְּעֽוֹר:  ת  וַאֲמַר משֶׁה לְדַיָנֵי יִשְׂרָאֵל קְטִילוּ גְבַר גַבְרוֹהִי דְאִתְחַבָּרוּ לְבַעֲלָא פְעוֹר:

{ו} וְהִנֵּ֡ה אִישׁ֩ מִבְּנֵ֨י יִשְׂרָאֵ֜ל בָּ֗א וַיַּקְרֵ֤ב אֶל-אֶחָיו֙ אֶת-הַמִּדְיָנִ֔ית לְעֵינֵ֣י מֹשֶׁ֔ה וּלְעֵינֵ֖י כָּל-עֲדַ֣ת בְּנֵי-יִשְׂרָאֵ֑ל וְהֵ֣מָּה בֹכִ֔ים פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  מ  וְהִנֵּ֡ה אִישׁ֩ מִבְּנֵ֨י יִשְׂרָאֵ֜ל בָּ֗א וַיַּקְרֵ֤ב אֶל-אֶחָיו֙ אֶת-הַמִּדְיָנִ֔ית לְעֵינֵ֣י מֹשֶׁ֔ה וּלְעֵינֵ֖י כָּל-עֲדַ֣ת בְּנֵי-יִשְׂרָאֵ֑ל וְהֵ֣מָּה בֹכִ֔ים פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  ת  וְהָא גַבְרָא מִבְּנֵי יִשְׂרָאֵל אֲתָא וְקָרֵב לְוָת אֲחוֹהִי יָת מִדְיָנֵיתָא לְעֵינֵי משֶׁה וּלְעֵינֵי כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל וְאִינוּן בָּכָן בִּתְרַע מַשְׁכַּן זִמְנָא:

{ז}  מפטיר  וַיַּ֗רְא פִּֽינְחָס֙ בֶּן-אֶלְעָזָ֔ר בֶּֽן-אַהֲרֹ֖ן הַכֹּהֵ֑ן וַיָּ֙קָם֙ מִתּ֣וֹךְ הָֽעֵדָ֔ה וַיִּקַּ֥ח רֹ֖מַח בְּיָדֽוֹ:  מ  וַיַּ֗רְא פִּֽינְחָס֙ בֶּן-אֶלְעָזָ֔ר בֶּֽן-אַהֲרֹ֖ן הַכֹּהֵ֑ן וַיָּ֙קָם֙ מִתּ֣וֹךְ הָֽעֵדָ֔ה וַיִּקַּ֥ח רֹ֖מַח בְּיָדֽוֹ:  ת  וַחֲזָא פִּינְחָס בַּר אֶלְעָזָר בַּר אַהֲרֹן כַּהֲנָא וְקָם מִגוֹ כְנִשְׁתָּא וּנְסֵיב רוֹמְחָא בִידֵיהּ:

{ח} וַ֠יָּבֹא אַחַ֨ר אִֽישׁ-יִשְׂרָאֵ֜ל אֶל-הַקֻּבָּ֗ה וַיִּדְקֹר֙ אֶת-שְׁנֵיהֶ֔ם אֵ֚ת אִ֣ישׁ יִשְׂרָאֵ֔ל וְאֶת-הָאִשָּׁ֖ה אֶל-קֳבָתָ֑הּ וַתֵּֽעָצַר֙ הַמַּגֵּפָ֔ה מֵעַ֖ל בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וַ֠יָּבֹא אַחַ֨ר אִֽישׁ-יִשְׂרָאֵ֜ל אֶל-הַקֻּבָּ֗ה וַיִּדְקֹר֙ אֶת-שְׁנֵיהֶ֔ם אֵ֚ת אִ֣ישׁ יִשְׂרָאֵ֔ל וְאֶת-הָאִשָּׁ֖ה אֶל-קֳבָתָ֑הּ וַתֵּֽעָצַר֙ הַמַּגֵּפָ֔ה מֵעַ֖ל בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וְעַל בָּתַר גַבְרָא בַּר יִשְׂרָאֵל לְקוּבָּתָא וּבְזַע יַת תַּרְוֵיהוֹן יַת גַבְרָא בַּר יִשְׂרָאֵל וְיַת אִתְּתָא לִמְעָהָא וְאִתְכְּלִי מוֹתָנָא מֵעַל בְּנֵי יִשְׂרָאֵל:

{ט} וַיִּהְי֕וּ הַמֵּתִ֖ים בַּמַּגֵּפָ֑ה אַרְבָּעָ֥ה וְעֶשְׂרִ֖ים אָֽלֶף:  מ  וַיִּהְי֕וּ הַמֵּתִ֖ים בַּמַּגֵּפָ֑ה אַרְבָּעָ֥ה וְעֶשְׂרִ֖ים אָֽלֶף:  ת  וַהֲווֹ דְמִיתוּ בְּמוֹתָנָא עַשְׂרִין וְאַרְבְּעָה אַלְפִין: (פפפ)
