פרשת תצוה


{כ} וְאַתָּ֞ה תְּצַוֶּ֣ה | אֶת-בְּנֵ֣י יִשְׂרָאֵ֗ל וְיִקְח֨וּ אֵלֶ֜יךָ שֶׁ֣מֶן זַ֥יִת זָ֛ךְ כָּתִ֖ית לַמָּא֑וֹר לְהַעֲלֹ֥ת נֵ֖ר תָּמִֽיד:  מ  וְאַתָּ֞ה תְּצַוֶּ֣ה | אֶת-בְּנֵ֣י יִשְׂרָאֵ֗ל וְיִקְח֨וּ אֵלֶ֜יךָ שֶׁ֣מֶן זַ֥יִת זָ֛ךְ כָּתִ֖ית לַמָּא֑וֹר לְהַעֲלֹ֥ת נֵ֖ר תָּמִֽיד:  ת  וְאַתְּ תְּפַקֵד יָת בְּנֵי יִשְׂרָאֵל וְיִסְבוּן לָךְ מִשְׁחָא זֵיתָא דַכְיָא כָּתִישָׁא לְאַנְהָרָא לְאַדְלָקָא בוֹצִינַיָא תְּדִירָא:

{כא} בְּאֹ֣הֶל מוֹעֵד֩ מִח֨וּץ לַפָּרֹ֜כֶת אֲשֶׁ֣ר עַל-הָעֵדֻ֗ת יַעֲרֹךְ֩ אֹת֨וֹ אַהֲרֹ֧ן וּבָנָ֛יו מֵעֶ֥רֶב עַד-בֹּ֖קֶר לִפְנֵ֣י יְהוָ֑ה חֻקַּ֤ת עוֹלָם֙ לְדֹ֣רֹתָ֔ם מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  מ  בְּאֹ֣הֶל מוֹעֵד֩ מִח֨וּץ לַפָּרֹ֜כֶת אֲשֶׁ֣ר עַל-הָעֵדֻ֗ת יַעֲרֹךְ֩ אֹת֨וֹ אַהֲרֹ֧ן וּבָנָ֛יו מֵעֶ֥רֶב עַד-בֹּ֖קֶר לִפְנֵ֣י יְהוָ֑ה חֻקַּ֤ת עוֹלָם֙ לְדֹ֣רֹתָ֔ם מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  ת  בְּמַשְׁכַּן זִמְנָא מִבָּרָא לְפָרֻכְתָּא דְעַל סַהֲדוּתָא יְסַדַר יָתֵהּ אַהֲרֹן וּבְנוֹהִי מֵרַמְשָׁא עַד צַפְרָא קָדָם יְיָ קְיָם עֲלָם לְדָרֵיהוֹן מִן בְּנֵי יִשְׂרָאֵל: (ס)




שמות פרק-כח

{א} וְאַתָּ֡ה הַקְרֵ֣ב אֵלֶיךָ֩ אֶת-אַהֲרֹ֨ן אָחִ֜יךָ וְאֶת-בָּנָ֣יו אִתּ֔וֹ מִתּ֛וֹךְ בְּנֵ֥י יִשְׂרָאֵ֖ל לְכַהֲנוֹ-לִ֑י אַהֲרֹ֕ן נָדָ֧ב וַאֲבִיה֛וּא אֶלְעָזָ֥ר וְאִיתָמָ֖ר בְּנֵ֥י אַהֲרֹֽן:  מ  וְאַתָּ֡ה הַקְרֵ֣ב אֵלֶיךָ֩ אֶת-אַהֲרֹ֨ן אָחִ֜יךָ וְאֶת-בָּנָ֣יו אִתּ֔וֹ מִתּ֛וֹךְ בְּנֵ֥י יִשְׂרָאֵ֖ל לְכַהֲנוֹ-לִ֑י אַהֲרֹ֕ן נָדָ֧ב וַאֲבִיה֛וּא אֶלְעָזָ֥ר וְאִיתָמָ֖ר בְּנֵ֥י אַהֲרֹֽן:  ת  וְאַתְּ קָרֵב לְוָתָךְ יָת אַהֲרֹן אָחוּךְ וְיָת בְּנוֹהִי עִמֵהּ מִגוֹ בְּנֵי יִשְׂרָאֵל לְשַׁמָשָׁא קֳדָמַי אַהֲרֹן נָדָב וַאֲבִיהוּא אֶלְעָזָר וְאִיתָמָר בְּנֵי אַהֲרֹן:

{ב} וְעָשִׂ֥יתָ בִגְדֵי-קֹ֖דֶשׁ לְאַהֲרֹ֣ן אָחִ֑יךָ לְכָב֖וֹד וּלְתִפְאָֽרֶת:  מ  וְעָשִׂ֥יתָ בִגְדֵי-קֹ֖דֶשׁ לְאַהֲרֹ֣ן אָחִ֑יךָ לְכָב֖וֹד וּלְתִפְאָֽרֶת:  ת  וְתַעְבֵּד לְבוּשֵׁי קוּדְשָׁא לְאַהֲרֹן אָחוּךְ לִיקָר וּלְתֻשְׁבְּחָן:

{ג} וְאַתָּ֗ה תְּדַבֵּר֙ אֶל-כָּל-חַכְמֵי-לֵ֔ב אֲשֶׁ֥ר מִלֵּאתִ֖יו ר֣וּחַ חָכְמָ֑ה וְעָשׂ֞וּ אֶת-בִּגְדֵ֧י אַהֲרֹ֛ן לְקַדְּשׁ֖וֹ לְכַהֲנוֹ-לִֽי:  מ  וְאַתָּ֗ה תְּדַבֵּר֙ אֶל-כָּל-חַכְמֵי-לֵ֔ב אֲשֶׁ֥ר מִלֵּאתִ֖יו ר֣וּחַ חָכְמָ֑ה וְעָשׂ֞וּ אֶת-בִּגְדֵ֧י אַהֲרֹ֛ן לְקַדְּשׁ֖וֹ לְכַהֲנוֹ-לִֽי:  ת  וְאַתְּ תְּמַלֵל עִם כָּל חַכִּימֵי לִבָּא דְאַשְׁלֵמִית עִמֵהּ רוּחַ חָכְמְתָא וְיַעְבְּדוּן יָת לְבוּשֵׁי אַהֲרֹן לְקַדָשׁוּתֵהּ לְשַׁמָשָׁא קֳדָמַי:

{ד} וְאֵ֨לֶּה הַבְּגָדִ֜ים אֲשֶׁ֣ר יַעֲשׂ֗וּ חֹ֤שֶׁן וְאֵפוֹד֙ וּמְעִ֔יל וּכְתֹ֥נֶת תַּשְׁבֵּ֖ץ מִצְנֶ֣פֶת וְאַבְנֵ֑ט וְעָשׂ֨וּ בִגְדֵי-קֹ֜דֶשׁ לְאַהֲרֹ֥ן אָחִ֛יךָ וּלְבָנָ֖יו לְכַהֲנוֹ-לִֽי:  מ  וְאֵ֨לֶּה הַבְּגָדִ֜ים אֲשֶׁ֣ר יַעֲשׂ֗וּ חֹ֤שֶׁן וְאֵפוֹד֙ וּמְעִ֔יל וּכְתֹ֥נֶת תַּשְׁבֵּ֖ץ מִצְנֶ֣פֶת וְאַבְנֵ֑ט וְעָשׂ֨וּ בִגְדֵי-קֹ֜דֶשׁ לְאַהֲרֹ֥ן אָחִ֛יךָ וּלְבָנָ֖יו לְכַהֲנוֹ-לִֽי:  ת  וְאִלֵין לְבוּשַׁיָא דִי יַעְבְּדוּן חוּשְׁנָא וְאֵפוֹדָא וּמְעִילָא וְכִתּוּנָא מְרַמְצָא מִצְנֶפְתָּא וְהֶמְיָנָא וְיַעְבְּדוּן לְבוּשֵׁי קוּדְשָׁא לְאַהֲרֹן אָחוּךְ וְלִבְנוֹהִי לְשַׁמָשָׁא קֳדָמַי:

{ה} וְהֵם֙ יִקְח֣וּ אֶת-הַזָּהָ֔ב וְאֶת-הַתְּכֵ֖לֶת וְאֶת-הָֽאַרְגָּמָ֑ן וְאֶת-תּוֹלַ֥עַת הַשָּׁנִ֖י וְאֶת-הַשֵּֽׁשׁ:  מ  וְהֵם֙ יִקְח֣וּ אֶת-הַזָּהָ֔ב וְאֶת-הַתְּכֵ֖לֶת וְאֶת-הָֽאַרְגָּמָ֑ן וְאֶת-תּוֹלַ֥עַת הַשָּׁנִ֖י וְאֶת-הַשֵּֽׁשׁ:  ת  וְאִנוּן יִסְבוּן יָת דַהֲבָא וְיָת תִּכְלָא וְיָת אַרְגְוָנָא וְיָת צְבַע זְהוֹרִי וְיָת בּוּצָא: (פ)

{ו} וְעָשׂ֖וּ אֶת-הָאֵפֹ֑ד זָ֠הָב תְּכֵ֨לֶת וְאַרְגָּמָ֜ן תּוֹלַ֧עַת שָׁנִ֛י וְשֵׁ֥שׁ מָשְׁזָ֖ר מַעֲשֵׂ֥ה חֹשֵֽׁב:  מ  וְעָשׂ֖וּ אֶת-הָאֵפֹ֑ד זָ֠הָב תְּכֵ֨לֶת וְאַרְגָּמָ֜ן תּוֹלַ֧עַת שָׁנִ֛י וְשֵׁ֥שׁ מָשְׁזָ֖ר מַעֲשֵׂ֥ה חֹשֵֽׁב:  ת  וְיַעְבְּדוּן יָת אֵפוֹדָא דַהֲבָא תִּכְלָא וְאַרְגְוָנָא צְבַע זְהוֹרִי וּבוּץ שְׁזִיר עוֹבַד אֳמָן:

{ז} שְׁתֵּ֧י כְתֵפֹ֣ת חֹֽבְרֹ֗ת יִֽהְיֶה-לּ֛וֹ אֶל-שְׁנֵ֥י קְצוֹתָ֖יו וְחֻבָּֽר:  מ  שְׁתֵּ֧י כְתֵפֹ֣ת חֹֽבְרֹ֗ת יִֽהְיֶה-לּ֛וֹ אֶל-שְׁנֵ֥י קְצוֹתָ֖יו וְחֻבָּֽר:  ת  תַּרְתֵּין כַּתְפִּין מְלַפְּפָן יְהוֹן לֵהּ עַל תְּרֵין סִטְרוֹהִי וְיִתְלָפַף:

{ח} וְחֵ֤שֶׁב אֲפֻדָּתוֹ֙ אֲשֶׁ֣ר עָלָ֔יו כְּמַעֲשֵׂ֖הוּ מִמֶּ֣נּוּ יִהְיֶ֑ה זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מָשְׁזָֽר:  מ  וְחֵ֤שֶׁב אֲפֻדָּתוֹ֙ אֲשֶׁ֣ר עָלָ֔יו כְּמַעֲשֵׂ֖הוּ מִמֶּ֣נּוּ יִהְיֶ֑ה זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מָשְׁזָֽר:  ת  וְהֶמְיַן תִּקוּנֵי דִי עֲלוֹהִי כְּעוֹבָדוֹהִי מִנֵהּ יְהֵי דַהֲבָא תִּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר:

{ט} וְלָ֣קַחְתָּ֔ אֶת-שְׁתֵּ֖י אַבְנֵי-שֹׁ֑הַם וּפִתַּחְתָּ֣ עֲלֵיהֶ֔ם שְׁמ֖וֹת בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וְלָ֣קַחְתָּ֔ אֶת-שְׁתֵּ֖י אַבְנֵי-שֹׁ֑הַם וּפִתַּחְתָּ֣ עֲלֵיהֶ֔ם שְׁמ֖וֹת בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וְתִסַב יָת תַּרְתֵּין אַבְנֵי בוּרְלָא וְתִגְלוֹף עֲלֵיהֶן שְׁמָהַת בְּנֵי יִשְׂרָאֵל:

{י} שִׁשָּׁה֙ מִשְּׁמֹתָ֔ם עַ֖ל הָאֶ֣בֶן הָאֶחָ֑ת וְאֶת-שְׁמ֞וֹת הַשִּׁשָּׁ֧ה הַנּוֹתָרִ֛ים עַל-הָאֶ֥בֶן הַשֵּׁנִ֖ית כְּתוֹלְדֹתָֽם:  מ  שִׁשָּׁה֙ מִשְּׁמֹתָ֔ם עַ֖ל הָאֶ֣בֶן הָאֶחָ֑ת וְאֶת-שְׁמ֞וֹת הַשִּׁשָּׁ֧ה הַנּוֹתָרִ֛ים עַל-הָאֶ֥בֶן הַשֵּׁנִ֖ית כְּתוֹלְדֹתָֽם:  ת  שִׁתָּא מִשְׁמָהָתְהוֹן עַל אַבְנָא חֲדָא וְיָת שְׁמָהַת שִׁתָּא דְאִשְׁתָּאָרוּ עַל אַבְנָא תִנְיֵתָא כְּתוּלְדָתְהוֹן:

{יא} מַעֲשֵׂ֣ה חָרַשׁ֮ אֶבֶן֒ פִּתּוּחֵ֣י חֹתָ֗ם תְּפַתַּח֙ אֶת-שְׁתֵּ֣י הָאֲבָנִ֔ים עַל-שְׁמֹ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל מֻסַבֹּ֛ת מִשְׁבְּצ֥וֹת זָהָ֖ב תַּעֲשֶׂ֥ה אֹתָֽם:  מ  מַעֲשֵׂ֣ה חָרַשׁ֮ אֶבֶן֒ פִּתּוּחֵ֣י חֹתָ֗ם תְּפַתַּח֙ אֶת-שְׁתֵּ֣י הָאֲבָנִ֔ים עַל-שְׁמֹ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל מֻסַבֹּ֛ת מִשְׁבְּצ֥וֹת זָהָ֖ב תַּעֲשֶׂ֥ה אֹתָֽם:  ת  עוֹבַד אֳמָן אֶבֶן טָבָא כְּתַב מְפָרַשׁ כִּגְלֹף דְעִזְקָא תִגְלוֹף יָת תַּרְתֵּין אַבְנַיָא עַל שְׁמָהַת בְּנֵי יִשְׂרָאֵל מְשַׁקְעָן מְרַמְצָן דִדְהַב תַּעְבֵּד יָתְהוֹן:

{יב} וְשַׂמְתָּ֞ אֶת-שְׁתֵּ֣י הָאֲבָנִ֗ים עַ֚ל כִּתְפֹ֣ת הָֽאֵפֹ֔ד אַבְנֵ֥י זִכָּרֹ֖ן לִבְנֵ֣י יִשְׂרָאֵ֑ל וְנָשָׂא֩ אַהֲרֹ֨ן אֶת-שְׁמוֹתָ֜ם לִפְנֵ֧י יְהוָ֛ה עַל-שְׁתֵּ֥י כְתֵפָ֖יו לְזִכָּרֹֽן:  מ  וְשַׂמְתָּ֞ אֶת-שְׁתֵּ֣י הָאֲבָנִ֗ים עַ֚ל כִּתְפֹ֣ת הָֽאֵפֹ֔ד אַבְנֵ֥י זִכָּרֹ֖ן לִבְנֵ֣י יִשְׂרָאֵ֑ל וְנָשָׂא֩ אַהֲרֹ֨ן אֶת-שְׁמוֹתָ֜ם לִפְנֵ֧י יְהוָ֛ה עַל-שְׁתֵּ֥י כְתֵפָ֖יו לְזִכָּרֹֽן:  ת  וּתְשַׁוִּי יָת תַּרְתֵּין אַבְנַיָא עַל כִּתְפֵּי אֵפוֹדָא אַבְנֵי דָכְרָנָא לִבְנֵי יִשְׂרָאֵל וְיִטּוֹל אַהֲרֹן יָת שְׁמָהָתְהוֹן קֳדָם יְיָ עַל תְּרֵין כִּתְפּוֹהִי לְדָכְרָנָא: (ס)

{יג}  שני  וְעָשִׂ֥יתָ מִשְׁבְּצֹ֖ת זָהָֽב:  מ  וְעָשִׂ֥יתָ מִשְׁבְּצֹ֖ת זָהָֽב:  ת  וְתַעְבֵּד מְרַמְצָן דִדְהַב:

{יד} וּשְׁתֵּ֤י שַׁרְשְׁרֹת֙ זָהָ֣ב טָה֔וֹר מִגְבָּלֹ֛ת תַּעֲשֶׂ֥ה אֹתָ֖ם מַעֲשֵׂ֣ה עֲבֹ֑ת וְנָתַתָּ֛ה אֶת-שַׁרְשְׁרֹ֥ת הָעֲבֹתֹ֖ת עַל-הַֽמִּשְׁבְּצֹֽת:  מ  וּשְׁתֵּ֤י שַׁרְשְׁרֹת֙ זָהָ֣ב טָה֔וֹר מִגְבָּלֹ֛ת תַּעֲשֶׂ֥ה אֹתָ֖ם מַעֲשֵׂ֣ה עֲבֹ֑ת וְנָתַתָּ֛ה אֶת-שַׁרְשְׁרֹ֥ת הָעֲבֹתֹ֖ת עַל-הַֽמִּשְׁבְּצֹֽת:  ת  וְתַרְתֵּין תִּכִּין דְהַב דְכֵי מְתַחְמָן תַּעְבֵד יָתְהוֹן עוֹבַד גְדִילוּ וְתִתֵּן יָת תִּכַּיָא גְדִילָתָא עַל מְרַמְצָתָא: (ס)

{טו} וְעָשִׂ֜יתָ חֹ֤שֶׁן מִשְׁפָּט֙ מַעֲשֵׂ֣ה חֹשֵׁ֔ב כְּמַעֲשֵׂ֥ה אֵפֹ֖ד תַּעֲשֶׂ֑נּוּ זָ֠הָב תְּכֵ֨לֶת וְאַרְגָּמָ֜ן וְתוֹלַ֧עַת שָׁנִ֛י וְשֵׁ֥שׁ מָשְׁזָ֖ר תַּעֲשֶׂ֥ה אֹתֽוֹ:  מ  וְעָשִׂ֜יתָ חֹ֤שֶׁן מִשְׁפָּט֙ מַעֲשֵׂ֣ה חֹשֵׁ֔ב כְּמַעֲשֵׂ֥ה אֵפֹ֖ד תַּעֲשֶׂ֑נּוּ זָ֠הָב תְּכֵ֨לֶת וְאַרְגָּמָ֜ן וְתוֹלַ֧עַת שָׁנִ֛י וְשֵׁ֥שׁ מָשְׁזָ֖ר תַּעֲשֶׂ֥ה אֹתֽוֹ:  ת  וְתַעְבֵּד חשֶׁן דִינָא עוֹבַד אֳמָן כְּעוֹבַד אֵפוֹדָא תַּעְבְּדִנֵהּ דַהֲבָא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר תַּעְבֵּד יָתֵהּ:

{טז} רָב֥וּעַ יִֽהְיֶ֖ה כָּפ֑וּל זֶ֥רֶת אָרְכּ֖וֹ וְזֶ֥רֶת רָחְבּֽוֹ:  מ  רָב֥וּעַ יִֽהְיֶ֖ה כָּפ֑וּל זֶ֥רֶת אָרְכּ֖וֹ וְזֶ֥רֶת רָחְבּֽוֹ:  ת  מְרַבַּע יְהֵי עֵיף זֶרְתָּא אֻרְכֵּה וְזֶרְתָּא פְתָיֵהּ:

{יז} וּמִלֵּאתָ֥ בוֹ֙ מִלֻּ֣אַת אֶ֔בֶן אַרְבָּעָ֖ה טוּרִ֣ים אָ֑בֶן ט֗וּר אֹ֤דֶם פִּטְדָה֙ וּבָרֶ֔קֶת הַטּ֖וּר הָאֶחָֽד:  מ  וּמִלֵּאתָ֥ בוֹ֙ מִלֻּ֣אַת אֶ֔בֶן אַרְבָּעָ֖ה טוּרִ֣ים אָ֑בֶן ט֗וּר אֹ֤דֶם פִּטְדָה֙ וּבָרֶ֔קֶת הַטּ֖וּר הָאֶחָֽד:  ת  וְתַשְׁלֵם בֵּה אַשְׁלָמוּת אַבְנָא אַרְבְּעָא סִדְרִין אֶבֶן טָּבָא סִדְרָא קַדְמָאָה סָמְקָן יַרְקָן וּבַרְקָן סִדְרָא חָד:

{יח} וְהַטּ֖וּר הַשֵּׁנִ֑י נֹ֥פֶךְ סַפִּ֖יר וְיָהֲלֹֽם:  מ  וְהַטּ֖וּר הַשֵּׁנִ֑י נֹ֥פֶךְ סַפִּ֖יר וְיָהֲלֹֽם:  ת  וְסִדְרָא תִּנְיָנָא אִזְמָרַגְדִין וְשַׁבְזִיז וְסִבְהֲלוֹם:

{יט} וְהַטּ֖וּר הַשְּׁלִישִׁ֑י לֶ֥שֶׁם שְׁב֖וֹ וְאַחְלָֽמָה:  מ  וְהַטּ֖וּר הַשְּׁלִישִׁ֑י לֶ֥שֶׁם שְׁב֖וֹ וְאַחְלָֽמָה:  ת  וְסִדְרָא תְּלִיתָאָה קַנְכִּירֵי טַרְקְיָא וְעֵין עֶגְלָא:

{כ} וְהַטּוּר֙ הָרְבִיעִ֔י תַּרְשִׁ֥ישׁ וְשֹׁ֖הַם וְיָשְׁפֵ֑ה מְשֻׁבָּצִ֥ים זָהָ֛ב יִהְי֖וּ בְּמִלּוּאֹתָֽם:  מ  וְהַטּוּר֙ הָרְבִיעִ֔י תַּרְשִׁ֥ישׁ וְשֹׁ֖הַם וְיָשְׁפֵ֑ה מְשֻׁבָּצִ֥ים זָהָ֛ב יִהְי֖וּ בְּמִלּוּאֹתָֽם:  ת  וְסִדְרֵא רְבִיעָאָה כְּרוּם יַמָא וּבוּרְלָא וּפַנְטֵרֵי מְרַמְצָן דִדְהַב יְהוֹן בְּאַשְׁלָמוּתְהוֹן:

{כא} וְ֠הָאֲבָנִים תִּֽהְיֶ֜יןָ עַל-שְׁמֹ֧ת בְּנֵֽי-יִשְׂרָאֵ֛ל שְׁתֵּ֥ים עֶשְׂרֵ֖ה עַל-שְׁמֹתָ֑ם פִּתּוּחֵ֤י חוֹתָם֙ אִ֣ישׁ עַל-שְׁמ֔וֹ תִּֽהְיֶ֕יןָ לִשְׁנֵ֥י עָשָׂ֖ר שָֽׁבֶט:  מ  וְ֠הָאֲבָנִים תִּֽהְיֶ֜יןָ עַל-שְׁמֹ֧ת בְּנֵֽי-יִשְׂרָאֵ֛ל שְׁתֵּ֥ים עֶשְׂרֵ֖ה עַל-שְׁמֹתָ֑ם פִּתּוּחֵ֤י חוֹתָם֙ אִ֣ישׁ עַל-שְׁמ֔וֹ תִּֽהְיֶ֕יןָ לִשְׁנֵ֥י עָשָׂ֖ר שָֽׁבֶט:  ת  וְאַבְנַיָא יְהֶוְיַן עַל שְׁמָהַת בְּנֵי יִשְׂרָאֵל תַּרְתֵּי עֶסְרֵי עַל שְׁמָהַתְהוֹן כְּתַב מְפָרַשׁ כִּגְלוֹף דְעִזְקָא גְבַר עַל שְׁמֵהּ תְּהֶוְיָן לִתְרֵין עֲסַר שִׁבְטִין:

{כב} וְעָשִׂ֧יתָ עַל-הַחֹ֛שֶׁן שַֽׁרְשֹׁ֥ת גַּבְלֻ֖ת מַעֲשֵׂ֣ה עֲבֹ֑ת זָהָ֖ב טָהֽוֹר:  מ  וְעָשִׂ֧יתָ עַל-הַחֹ֛שֶׁן שַֽׁרְשֹׁ֥ת גַּבְלֻ֖ת מַעֲשֵׂ֣ה עֲבֹ֑ת זָהָ֖ב טָהֽוֹר:  ת  וְתַעְבֵּד עַל חוּשְׁנָא תִּכִּין מְתַחְמָן עוֹבַד גְדִילוּ דְהַב דְכֵי:

{כג} וְעָשִׂ֙יתָ֙ עַל-הַחֹ֔שֶׁן שְׁתֵּ֖י טַבְּע֣וֹת זָהָ֑ב וְנָתַתָּ֗ אֶת-שְׁתֵּי֙ הַטַּבָּע֔וֹת עַל-שְׁנֵ֖י קְצ֥וֹת הַחֹֽשֶׁן:  מ  וְעָשִׂ֙יתָ֙ עַל-הַחֹ֔שֶׁן שְׁתֵּ֖י טַבְּע֣וֹת זָהָ֑ב וְנָתַתָּ֗ אֶת-שְׁתֵּי֙ הַטַּבָּע֔וֹת עַל-שְׁנֵ֖י קְצ֥וֹת הַחֹֽשֶׁן:  ת  וְתַעְבֵּד עַל חוּשְׁנָא תַּרְתֵּין עִזְקָן דִדְהַב וְתִתֵּן יָת תַּרְתֵּין עִזְקָתָא עַל תְּרֵין סִטְרֵי חוּשְׁנָא:

{כד} וְנָתַתָּ֗ה אֶת-שְׁתֵּי֙ עֲבֹתֹ֣ת הַזָּהָ֔ב עַל-שְׁתֵּ֖י הַטַּבָּעֹ֑ת אֶל-קְצ֖וֹת הַחֹֽשֶׁן:  מ  וְנָתַתָּ֗ה אֶת-שְׁתֵּי֙ עֲבֹתֹ֣ת הַזָּהָ֔ב עַל-שְׁתֵּ֖י הַטַּבָּעֹ֑ת אֶל-קְצ֖וֹת הַחֹֽשֶׁן:  ת  וְתִתֵּן יָת תַּרְתֵּין גְדִילָן דִדְהַב עַל תַּרְתֵּין עִזְקָתָא לְסִטְרֵי חוּשְׁנָא:

{כה} וְאֵ֨ת שְׁתֵּ֤י קְצוֹת֙ שְׁתֵּ֣י הָעֲבֹתֹ֔ת תִּתֵּ֖ן עַל-שְׁתֵּ֣י הַֽמִּשְׁבְּצ֑וֹת וְנָתַתָּ֛ה עַל-כִּתְפ֥וֹת הָאֵפֹ֖ד אֶל-מ֥וּל פָּנָֽיו:  מ  וְאֵ֨ת שְׁתֵּ֤י קְצוֹת֙ שְׁתֵּ֣י הָעֲבֹתֹ֔ת תִּתֵּ֖ן עַל-שְׁתֵּ֣י הַֽמִּשְׁבְּצ֑וֹת וְנָתַתָּ֛ה עַל-כִּתְפ֥וֹת הָאֵפֹ֖ד אֶל-מ֥וּל פָּנָֽיו:  ת  וְיָת תַּרְתֵּין גְדִילָן דְעַל תַּרְתֵּין סִטְרוֹהִי תִּתֵּן עַל תַּרְתֵּין מְרַמְצָתָא וְתִתֵּן עַל כִּתְפֵּי אֵפוֹדָא לָקֳבֵל אַפּוֹהִי:

{כו} וְעָשִׂ֗יתָ שְׁתֵּי֙ טַבְּע֣וֹת זָהָ֔ב וְשַׂמְתָּ֣ אֹתָ֔ם עַל-שְׁנֵ֖י קְצ֣וֹת הַחֹ֑שֶׁן עַל-שְׂפָת֕וֹ אֲשֶׁ֛ר אֶל-עֵ֥בֶר הָאֵפֹ֖ד בָּֽיְתָה:  מ  וְעָשִׂ֗יתָ שְׁתֵּי֙ טַבְּע֣וֹת זָהָ֔ב וְשַׂמְתָּ֣ אֹתָ֔ם עַל-שְׁנֵ֖י קְצ֣וֹת הַחֹ֑שֶׁן עַל-שְׂפָת֕וֹ אֲשֶׁ֛ר אֶל-עֵ֥בֶר הָאֵפֹ֖ד בָּֽיְתָה:  ת  וְתַעְבֵּד תַּרְתֵּין עִזְקָן דִדְהַב וּתְשַׁוִּי יָתְהוֹן עַל תְּרֵין סִטְרֵי חוּשְׁנָא עַל סִפְתֵהּ דִי לְעִבְרָא דְאֵפוֹדָא מִלְגָו:

{כז} וְעָשִׂיתָ֮ שְׁתֵּ֣י טַבְּע֣וֹת זָהָב֒ וְנָתַתָּ֣ה אֹתָ֡ם עַל-שְׁתֵּי֩ כִתְפ֨וֹת הָאֵפ֤וֹד מִלְּמַ֙טָּה֙ מִמּ֣וּל פָּנָ֔יו לְעֻמַּ֖ת מֶחְבַּרְתּ֑וֹ מִמַּ֕עַל לְחֵ֖שֶׁב הָאֵפֽוֹד:  מ  וְעָשִׂיתָ֮ שְׁתֵּ֣י טַבְּע֣וֹת זָהָב֒ וְנָתַתָּ֣ה אֹתָ֡ם עַל-שְׁתֵּי֩ כִתְפ֨וֹת הָאֵפ֤וֹד מִלְּמַ֙טָּה֙ מִמּ֣וּל פָּנָ֔יו לְעֻמַּ֖ת מֶחְבַּרְתּ֑וֹ מִמַּ֕עַל לְחֵ֖שֶׁב הָאֵפֽוֹד:  ת  וְתַעְבֵּד תַּרְתֵּין עִזְקָן דִדְהַב וְתִתֵּן יָתְהוֹן עַל תַּרְתֵּין כִּתְפֵּי אֵפוֹדָא מִלְרַע מִלְקֳבֵל אַפּוֹהִי לָקֳבֵל בֵּית לוֹפֵי מֵעִלָוֵי לְהֶמְיַן אֵפוֹדָא:

{כח} וְיִרְכְּס֣וּ אֶת-הַ֠חֹשֶׁן (מטבעתו) מִֽטַּבְּעֹתָ֞יו אֶל-טַבְּעֹ֤ת הָאֵפֹד֙ בִּפְתִ֣יל תְּכֵ֔לֶת לִֽהְי֖וֹת עַל-חֵ֣שֶׁב הָאֵפ֑וֹד וְלֹֽא-יִזַּ֣ח הַחֹ֔שֶׁן מֵעַ֖ל הָאֵפֽוֹד:  מ  וְיִרְכְּס֣וּ אֶת-הַ֠חֹשֶׁן (מטבעתו) מִֽטַּבְּעֹתָ֞יו אֶל-טַבְּעֹ֤ת הָאֵפֹד֙ בִּפְתִ֣יל תְּכֵ֔לֶת לִֽהְי֖וֹת עַל-חֵ֣שֶׁב הָאֵפ֑וֹד וְלֹֽא-יִזַּ֣ח הַחֹ֔שֶׁן מֵעַ֖ל הָאֵפֽוֹד:  ת  וְיֶחְדוּן יָת חוּשְׁנָא מֵעִזְקְתֵהּ לְעִזְקְתָא דְאֵפוֹדָא בְּחוּטָא דִתְכֶלְתָּא לְמֶהֱוֵי עַל הֶמְיַן אֵפוֹדָא וְלָא יִתְפָּרֵק חוּשְׁנָא מֵעִלָוֵי אֵפוֹדָא:

{כט} וְנָשָׂ֣א אַ֠הֲרֹן אֶת-שְׁמ֨וֹת בְּנֵֽי-יִשְׂרָאֵ֜ל בְּחֹ֧שֶׁן הַמִּשְׁפָּ֛ט עַל-לִבּ֖וֹ בְּבֹא֣וֹ אֶל-הַקֹּ֑דֶשׁ לְזִכָּרֹ֥ן לִפְנֵֽי-יְהוָ֖ה תָּמִֽיד:  מ  וְנָשָׂ֣א אַ֠הֲרֹן אֶת-שְׁמ֨וֹת בְּנֵֽי-יִשְׂרָאֵ֜ל בְּחֹ֧שֶׁן הַמִּשְׁפָּ֛ט עַל-לִבּ֖וֹ בְּבֹא֣וֹ אֶל-הַקֹּ֑דֶשׁ לְזִכָּרֹ֥ן לִפְנֵֽי-יְהוָ֖ה תָּמִֽיד:  ת  וְיִטוֹל אַהֲרֹן יָת שְׁמָהַת בְּנֵי יִשְׂרָאֵל בְּחשֶׁן דִינָא עַל לִבֵּהּ בְּמֵעֲלֵהּ לְקוּדְשָׁא לְדוּכְרָנָא קֳדָם יְיָ תְּדִירָא:

{ל} וְנָתַתָּ֞ אֶל-חֹ֣שֶׁן הַמִּשְׁפָּ֗ט אֶת-הָאוּרִים֙ וְאֶת-הַתֻּמִּ֔ים וְהָיוּ֙ עַל-לֵ֣ב אַהֲרֹ֔ן בְּבֹא֖וֹ לִפְנֵ֣י יְהוָ֑ה וְנָשָׂ֣א אַ֠הֲרֹן אֶת-מִשְׁפַּ֨ט בְּנֵי-יִשְׂרָאֵ֧ל עַל-לִבּ֛וֹ לִפְנֵ֥י יְהוָ֖ה תָּמִֽיד:  מ  וְנָתַתָּ֞ אֶל-חֹ֣שֶׁן הַמִּשְׁפָּ֗ט אֶת-הָאוּרִים֙ וְאֶת-הַתֻּמִּ֔ים וְהָיוּ֙ עַל-לֵ֣ב אַהֲרֹ֔ן בְּבֹא֖וֹ לִפְנֵ֣י יְהוָ֑ה וְנָשָׂ֣א אַ֠הֲרֹן אֶת-מִשְׁפַּ֨ט בְּנֵי-יִשְׂרָאֵ֧ל עַל-לִבּ֛וֹ לִפְנֵ֥י יְהוָ֖ה תָּמִֽיד:  ת  וְתִתֵּן בְּחשֶׁן דִינָא יָת אוּרַיָא וְיָת תֻּמַיָא וִיהוֹן עַל לִבָּא דְאַהֲרֹן בְּמֵעֲלֵהּ קֳדָם יְיָ וְיִטוֹל אַהֲרֹן יָת דִינָא דִבְנֵי יִשְׂרָאֵל עַל לִבֵּהּ קֳדָם יְיָ תְּדִירָא: (ס)

{לא}  שלישי  וְעָשִׂ֛יתָ אֶת-מְעִ֥יל הָאֵפ֖וֹד כְּלִ֥יל תְּכֵֽלֶת:  מ  וְעָשִׂ֛יתָ אֶת-מְעִ֥יל הָאֵפ֖וֹד כְּלִ֥יל תְּכֵֽלֶת:  ת  וְתַעְבֵּד יָת מְעִילָא דְאֵפוֹדָא גְמִיר תִּכְלָא:

{לב} וְהָיָ֥ה פִֽי-רֹאשׁ֖וֹ בְּתוֹכ֑וֹ שָׂפָ֡ה יִֽהְיֶה֩ לְפִ֨יו סָבִ֜יב מַעֲשֵׂ֣ה אֹרֵ֗ג כְּפִ֥י תַחְרָ֛א יִֽהְיֶה-לּ֖וֹ לֹ֥א יִקָּרֵֽעַ:  מ  וְהָיָ֥ה פִֽי-רֹאשׁ֖וֹ בְּתוֹכ֑וֹ שָׂפָ֡ה יִֽהְיֶה֩ לְפִ֨יו סָבִ֜יב מַעֲשֵׂ֣ה אֹרֵ֗ג כְּפִ֥י תַחְרָ֛א יִֽהְיֶה-לּ֖וֹ לֹ֥א יִקָּרֵֽעַ:  ת  וִיהֵי פוּם רֵישֵׁהּ כָּפִיל לְגַוֵּהּ תּוֹרָא יְהֵא מַקַף לְפוּמֵהּ סְחוֹר סְחוֹר עוֹבַד מָחֵי כְּפוּם שִׂרְיָן יְהֵי לֵהּ לָא יִתְבְּזָע:

{לג} וְעָשִׂ֣יתָ עַל-שׁוּלָ֗יו רִמֹּנֵי֙ תְּכֵ֤לֶת וְאַרְגָּמָן֙ וְתוֹלַ֣עַת שָׁנִ֔י עַל-שׁוּלָ֖יו סָבִ֑יב וּפַעֲמֹנֵ֥י זָהָ֛ב בְּתוֹכָ֖ם סָבִֽיב:  מ  וְעָשִׂ֣יתָ עַל-שׁוּלָ֗יו רִמֹּנֵי֙ תְּכֵ֤לֶת וְאַרְגָּמָן֙ וְתוֹלַ֣עַת שָׁנִ֔י עַל-שׁוּלָ֖יו סָבִ֑יב וּפַעֲמֹנֵ֥י זָהָ֛ב בְּתוֹכָ֖ם סָבִֽיב:  ת  וְתַעְבֵּד עַל שִׁיפוֹלוֹהִי רִמוֹנֵי תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי עַל שִׁיפוֹלוֹהִי סְחוֹר סְחוֹר וְזַגִין דְדַהֲבָא בֵּינֵיהוֹן סְחוֹר סְחוֹר:

{לד} פַּעֲמֹ֤ן זָהָב֙ וְרִמּ֔וֹן פַּֽעֲמֹ֥ן זָהָ֖ב וְרִמּ֑וֹן עַל-שׁוּלֵ֥י הַמְּעִ֖יל סָבִֽיב:  מ  פַּעֲמֹ֤ן זָהָב֙ וְרִמּ֔וֹן פַּֽעֲמֹ֥ן זָהָ֖ב וְרִמּ֑וֹן עַל-שׁוּלֵ֥י הַמְּעִ֖יל סָבִֽיב:  ת  זַגָא דְדַהֲבָא וְרִמוֹנָא זַגָא דְדַהֲבָא וְרִמוֹנָא עַל שִׁיפוֹלֵי מְעִילָא סְחוֹר סְחוֹר:

{לה} וְהָיָ֥ה עַֽל-אַהֲרֹ֖ן לְשָׁרֵ֑ת וְנִשְׁמַ֣ע ק֠וֹלוֹ בְּבֹא֨וֹ אֶל-הַקֹּ֜דֶשׁ לִפְנֵ֧י יְהוָ֛ה וּבְצֵאת֖וֹ וְלֹ֥א יָמֽוּת:  מ  וְהָיָ֥ה עַֽל-אַהֲרֹ֖ן לְשָׁרֵ֑ת וְנִשְׁמַ֣ע ק֠וֹלוֹ בְּבֹא֨וֹ אֶל-הַקֹּ֜דֶשׁ לִפְנֵ֧י יְהוָ֛ה וּבְצֵאת֖וֹ וְלֹ֥א יָמֽוּת:  ת  וִיהֵי עַל אַהֲרֹן לְשַׁמָשָׁא וְיִשְׁתְּמַע קָלֵהּ בְּמֵעֲלֵהּ לְקוּדְשָׁא קֳדָם יְיָ וּבְמִפְקֵהּ וְלָא יְמוּת: (ס)

{לו} וְעָשִׂ֥יתָ צִּ֖יץ זָהָ֣ב טָה֑וֹר וּפִתַּחְתָּ֤ עָלָיו֙ פִּתּוּחֵ֣י חֹתָ֔ם קֹ֖דֶשׁ לַֽיהוָֽה:  מ  וְעָשִׂ֥יתָ צִּ֖יץ זָהָ֣ב טָה֑וֹר וּפִתַּחְתָּ֤ עָלָיו֙ פִּתּוּחֵ֣י חֹתָ֔ם קֹ֖דֶשׁ לַֽיהוָֽה:  ת  וְתַעְבֵּד צִיצָא דִדְהַב דְכֵי וְתִגְלוֹף עֲלוֹהִי כְּתַב מִפָרַשׁ קֹדֶשׁ לַיָי:

{לז} וְשַׂמְתָּ֤ אֹתוֹ֙ עַל-פְּתִ֣יל תְּכֵ֔לֶת וְהָיָ֖ה עַל-הַמִּצְנָ֑פֶת אֶל-מ֥וּל פְּנֵֽי-הַמִּצְנֶ֖פֶת יִהְיֶֽה:  מ  וְשַׂמְתָּ֤ אֹתוֹ֙ עַל-פְּתִ֣יל תְּכֵ֔לֶת וְהָיָ֖ה עַל-הַמִּצְנָ֑פֶת אֶל-מ֥וּל פְּנֵֽי-הַמִּצְנֶ֖פֶת יִהְיֶֽה:  ת  וּתְשַׁוִי יָתֵהּ עַל חוּטָא דִתְכֶלְתָּא וִיהֵי עַל מִצְנֶפְתָּא לָקֳבֵל אַפֵּי מִצְנֶפְתָּא יְהֵי:

{לח} וְהָיָה֮ עַל-מֵ֣צַח אַהֲרֹן֒ וְנָשָׂ֨א אַהֲרֹ֜ן אֶת-עֲוֹ֣ן הַקֳּדָשִׁ֗ים אֲשֶׁ֤ר יַקְדִּ֙ישׁוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לְכָֽל-מַתְּנֹ֖ת קָדְשֵׁיהֶ֑ם וְהָיָ֤ה עַל-מִצְחוֹ֙ תָּמִ֔יד לְרָצ֥וֹן לָהֶ֖ם לִפְנֵ֥י יְהוָֽה:  מ  וְהָיָה֮ עַל-מֵ֣צַח אַהֲרֹן֒ וְנָשָׂ֨א אַהֲרֹ֜ן אֶת-עֲוֹ֣ן הַקֳּדָשִׁ֗ים אֲשֶׁ֤ר יַקְדִּ֙ישׁוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לְכָֽל-מַתְּנֹ֖ת קָדְשֵׁיהֶ֑ם וְהָיָ֤ה עַל-מִצְחוֹ֙ תָּמִ֔יד לְרָצ֥וֹן לָהֶ֖ם לִפְנֵ֥י יְהוָֽה:  ת  וִיהֵי עַל בֵּית עֵינוֹהִי דְאַהֲרֹן וְיִטּוֹל אַהֲרֹן יָת עֲוָיַת קוּדְשַׁיָא דִי יְקַדְשׁוּן בְּנֵי יִשְׂרָאֵל לְכָל מַתְּנַּת קוּדְשֵׁיהוֹן וִיהֵי עַל בֵּית עֵינוֹהִי תְּדִירָא לְרַעֲוָא לְהוֹן קֳדָם יְיָ:

{לט} וְשִׁבַּצְתָּ֙ הַכְּתֹ֣נֶת שֵׁ֔שׁ וְעָשִׂ֖יתָ מִצְנֶ֣פֶת שֵׁ֑שׁ וְאַבְנֵ֥ט תַּעֲשֶׂ֖ה מַעֲשֵׂ֥ה רֹקֵֽם:  מ  וְשִׁבַּצְתָּ֙ הַכְּתֹ֣נֶת שֵׁ֔שׁ וְעָשִׂ֖יתָ מִצְנֶ֣פֶת שֵׁ֑שׁ וְאַבְנֵ֥ט תַּעֲשֶׂ֖ה מַעֲשֵׂ֥ה רֹקֵֽם:  ת  וּתְרַמֵץ כִּתּוּנָא בוּצָא וְתַעְבֵּד מִצְנֶפְתָּא דְבוּצָא וְהֶמְיָנָא תַּעְבֵּד עוֹבַד צַיָר:

{מ} וְלִבְנֵ֤י אַהֲרֹן֙ תַּעֲשֶׂ֣ה כֻתֳּנֹ֔ת וְעָשִׂ֥יתָ לָהֶ֖ם אַבְנֵטִ֑ים וּמִגְבָּעוֹת֙ תַּעֲשֶׂ֣ה לָהֶ֔ם לְכָב֖וֹד וּלְתִפְאָֽרֶת:  מ  וְלִבְנֵ֤י אַהֲרֹן֙ תַּעֲשֶׂ֣ה כֻתֳּנֹ֔ת וְעָשִׂ֥יתָ לָהֶ֖ם אַבְנֵטִ֑ים וּמִגְבָּעוֹת֙ תַּעֲשֶׂ֣ה לָהֶ֔ם לְכָב֖וֹד וּלְתִפְאָֽרֶת:  ת  וְלִבְנֵי אַהֲרֹן תַּעְבֵּד כִּתּוּנִין וְתַעְבֵּד לְהוֹן הֶמְיָנִין וְכוֹבָעִין תַּעְבֵּד לְהוֹן לִיקָר וּלְתֻשְׁבְּחָן:

{מא} וְהִלְבַּשְׁתָּ֤ אֹתָם֙ אֶת-אַהֲרֹ֣ן אָחִ֔יךָ וְאֶת-בָּנָ֖יו אִתּ֑וֹ וּמָשַׁחְתָּ֨ אֹתָ֜ם וּמִלֵּאתָ֧ אֶת-יָדָ֛ם וְקִדַּשְׁתָּ֥ אֹתָ֖ם וְכִהֲנ֥וּ לִֽי:  מ  וְהִלְבַּשְׁתָּ֤ אֹתָם֙ אֶת-אַהֲרֹ֣ן אָחִ֔יךָ וְאֶת-בָּנָ֖יו אִתּ֑וֹ וּמָשַׁחְתָּ֨ אֹתָ֜ם וּמִלֵּאתָ֧ אֶת-יָדָ֛ם וְקִדַּשְׁתָּ֥ אֹתָ֖ם וְכִהֲנ֥וּ לִֽי:  ת  וְתַלְבֵּש יָתְהוֹן יָת אַהֲרֹן אָחוּךְ וְיָת בְּנוֹהִי עִמֵהּ וּתְרַבֵּי יָתְהוֹן וּתְקָרֵב יָת קֻרְבָּנְהוֹן וּתְקַדֵשׁ יָתְהוֹן וִישַׁמְשׁוּן קֳדָמַי:

{מב} וַעֲשֵׂ֤ה לָהֶם֙ מִכְנְסֵי-בָ֔ד לְכַסּ֖וֹת בְּשַׂ֣ר עֶרְוָ֑ה מִמָּתְנַ֥יִם וְעַד-יְרֵכַ֖יִם יִהְיֽוּ:  מ  וַעֲשֵׂ֤ה לָהֶם֙ מִכְנְסֵי-בָ֔ד לְכַסּ֖וֹת בְּשַׂ֣ר עֶרְוָ֑ה מִמָּתְנַ֥יִם וְעַד-יְרֵכַ֖יִם יִהְיֽוּ:  ת  וְעִבֵד לְהוֹן מִכְנְסִין דְבוּץ לְכַסָאָה בְשַׂר עֶרְיְתָא מֵחַרְצָן וְעַד יַרְכָּן יְהוֹן:

{מג} וְהָיוּ֩ עַל-אַהֲרֹ֨ן וְעַל-בָּנָ֜יו בְּבֹאָ֣ם | אֶל-אֹ֣הֶל מוֹעֵ֗ד א֣וֹ בְגִשְׁתָּ֤ם אֶל-הַמִּזְבֵּ֙חַ֙ לְשָׁרֵ֣ת בַּקֹּ֔דֶשׁ וְלֹא-יִשְׂא֥וּ עָוֹ֖ן וָמֵ֑תוּ חֻקַּ֥ת עוֹלָ֛ם ל֖וֹ וּלְזַרְע֥וֹ אַחֲרָֽיו:  מ  וְהָיוּ֩ עַל-אַהֲרֹ֨ן וְעַל-בָּנָ֜יו בְּבֹאָ֣ם | אֶל-אֹ֣הֶל מוֹעֵ֗ד א֣וֹ בְגִשְׁתָּ֤ם אֶל-הַמִּזְבֵּ֙חַ֙ לְשָׁרֵ֣ת בַּקֹּ֔דֶשׁ וְלֹא-יִשְׂא֥וּ עָוֹ֖ן וָמֵ֑תוּ חֻקַּ֥ת עוֹלָ֛ם ל֖וֹ וּלְזַרְע֥וֹ אַחֲרָֽיו:  ת  וִיהוֹן עַל אַהֲרֹן וְעַל בְּנוֹהִי בְּמֵעַלְהוֹן לְּמַשְׁכַּן זִמְנָא אוֹ בְמִקְרָבְהוֹן לְמַדְבְּחָא לְשַׁמָשָׁא בְקוּדְשָׁא וְלָא יְקַבְּלוּן חוֹבָא וִימוּתוּן קְיָם עָלָם לֵהּ וְלִבְנוֹהִי בַתְרוֹהִי: (ס)




שמות פרק-כט

{א}  רביעי  וְזֶ֨ה הַדָּבָ֜ר אֲשֶֽׁר-תַּעֲשֶׂ֥ה לָהֶ֛ם לְקַדֵּ֥שׁ אֹתָ֖ם לְכַהֵ֣ן לִ֑י לְ֠קַח פַּ֣ר אֶחָ֧ד בֶּן-בָּקָ֛ר וְאֵילִ֥ם שְׁנַ֖יִם תְּמִימִֽם:  מ  וְזֶ֨ה הַדָּבָ֜ר אֲשֶֽׁר-תַּעֲשֶׂ֥ה לָהֶ֛ם לְקַדֵּ֥שׁ אֹתָ֖ם לְכַהֵ֣ן לִ֑י לְ֠קַח פַּ֣ר אֶחָ֧ד בֶּן-בָּקָ֛ר וְאֵילִ֥ם שְׁנַ֖יִם תְּמִימִֽם:  ת  וְדֵין פִּתְגָמָא דִי תַעְבֵּד לְהוֹן לְקַדָשָׁא יָתְהוֹן לְשַׁמָשָׁא קֳדָמַי סַב תּוֹר חָד בַּר תּוֹרֵי וְדִכְרִין תְּרֵין שַׁלְמִין:

{ב} וְלֶ֣חֶם מַצּ֗וֹת וְחַלֹּ֤ת מַצֹּת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן סֹ֥לֶת חִטִּ֖ים תַּעֲשֶׂ֥ה אֹתָֽם:  מ  וְלֶ֣חֶם מַצּ֗וֹת וְחַלֹּ֤ת מַצֹּת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן סֹ֥לֶת חִטִּ֖ים תַּעֲשֶׂ֥ה אֹתָֽם:  ת  וּלְחֵם פַּטִיר וּגְרִיצִין פַּטִּירִין דְפִילָן בִּמְשַׁח וְאֶסְפּוֹגִין פַּטִּירִין דִמְשִׁיחִין בִּמְשָׁח סֻלְתָּא חִטִין תַּעְבֵּד יָתְהוֹן:

{ג} וְנָתַתָּ֤ אוֹתָם֙ עַל-סַ֣ל אֶחָ֔ד וְהִקְרַבְתָּ֥ אֹתָ֖ם בַּסָּ֑ל וְאֶ֨ת-הַפָּ֔ר וְאֵ֖ת שְׁנֵ֥י הָאֵילִֽם:  מ  וְנָתַתָּ֤ אוֹתָם֙ עַל-סַ֣ל אֶחָ֔ד וְהִקְרַבְתָּ֥ אֹתָ֖ם בַּסָּ֑ל וְאֶ֨ת-הַפָּ֔ר וְאֵ֖ת שְׁנֵ֥י הָאֵילִֽם:  ת  וְתִתֵּן יָתְהוֹן עַל סַלָא חַד וּתְקָרֵב יָתְהוֹן בְּסַלָא וְיָת תּוֹרָא וְיָת תְּרֵין דִכְרִין:

{ד} וְאֶת-אַהֲרֹ֤ן וְאֶת-בָּנָיו֙ תַּקְרִ֔יב אֶל-פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְרָחַצְתָּ֥ אֹתָ֖ם בַּמָּֽיִם:  מ  וְאֶת-אַהֲרֹ֤ן וְאֶת-בָּנָיו֙ תַּקְרִ֔יב אֶל-פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְרָחַצְתָּ֥ אֹתָ֖ם בַּמָּֽיִם:  ת  וְיָת אַהֲרֹן וְיָת בְּנוֹהִי תְּקָרֵב לִתְרַע מַשְׁכַּן זִמְנָא וְתַסְחֵי יָתְהוֹן בְּמַיָא:

{ה} וְלָקַחְתָּ֣ אֶת-הַבְּגָדִ֗ים וְהִלְבַּשְׁתָּ֤ אֶֽת-אַהֲרֹן֙ אֶת-הַכֻּתֹּ֔נֶת וְאֵת֙ מְעִ֣יל הָאֵפֹ֔ד וְאֶת-הָאֵפֹ֖ד וְאֶת-הַחֹ֑שֶׁן וְאָפַדְתָּ֣ ל֔וֹ בְּחֵ֖שֶׁב הָאֵפֹֽד:  מ  וְלָקַחְתָּ֣ אֶת-הַבְּגָדִ֗ים וְהִלְבַּשְׁתָּ֤ אֶֽת-אַהֲרֹן֙ אֶת-הַכֻּתֹּ֔נֶת וְאֵת֙ מְעִ֣יל הָאֵפֹ֔ד וְאֶת-הָאֵפֹ֖ד וְאֶת-הַחֹ֑שֶׁן וְאָפַדְתָּ֣ ל֔וֹ בְּחֵ֖שֶׁב הָאֵפֹֽד:  ת  וְתִסַב יָת לְבוּשַׁיָא וְתַלְבֵּשׁ יָת אַהֲרֹן יָת כִּתּוּנָא וְיָת מְעִיל אֵפוֹדָא וְיָת אֵפוֹדָא וְיָת חוּשְׁנָא וְתַתְקֵן לֵהּ בְּהֶמְיַן אֵפוֹדָא:

{ו} וְשַׂמְתָּ֥ הַמִּצְנֶ֖פֶת עַל-רֹאשׁ֑וֹ וְנָתַתָּ֛ אֶת-נֵ֥זֶר הַקֹּ֖דֶשׁ עַל-הַמִּצְנָֽפֶת:  מ  וְשַׂמְתָּ֥ הַמִּצְנֶ֖פֶת עַל-רֹאשׁ֑וֹ וְנָתַתָּ֛ אֶת-נֵ֥זֶר הַקֹּ֖דֶשׁ עַל-הַמִּצְנָֽפֶת:  ת  וּתְשַׁוִי מִצְנֶפְתָּא עַל רֵישֵׁהּ וְתִתֵּן יָת כְּלִילָא דְקוּדְשָׁא עַל מִצְנֶפְתָּא:

{ז} וְלָֽקַחְתָּ֙ אֶת-שֶׁ֣מֶן הַמִּשְׁחָ֔ה וְיָצַקְתָּ֖ עַל-רֹאשׁ֑וֹ וּמָשַׁחְתָּ֖ אֹתֽוֹ:  מ  וְלָֽקַחְתָּ֙ אֶת-שֶׁ֣מֶן הַמִּשְׁחָ֔ה וְיָצַקְתָּ֖ עַל-רֹאשׁ֑וֹ וּמָשַׁחְתָּ֖ אֹתֽוֹ:  ת  וְתִסַב יָת מִשְׁחָא דִרְבוּתָא וּתְרִיק עַל רֵישֵׁהּ וּתְרַבֵּהּ יָתֵהּ:

{ח} וְאֶת-בָּנָ֖יו תַּקְרִ֑יב וְהִלְבַּשְׁתָּ֖ם כֻּתֳּנֹֽת:  מ  וְאֶת-בָּנָ֖יו תַּקְרִ֑יב וְהִלְבַּשְׁתָּ֖ם כֻּתֳּנֹֽת:  ת  וְיָת בְּנוֹהִי תְּקָרֵב וְתַלְבֵּשִׁנוּן כִּתּוּנִין:

{ט} וְחָגַרְתָּ֩ אֹתָ֨ם אַבְנֵ֜ט אַהֲרֹ֣ן וּבָנָ֗יו וְחָבַשְׁתָּ֤ לָהֶם֙ מִגְבָּעֹ֔ת וְהָיְתָ֥ה לָהֶ֛ם כְּהֻנָּ֖ה לְחֻקַּ֣ת עוֹלָ֑ם וּמִלֵּאתָ֥ יַֽד-אַהֲרֹ֖ן וְיַד-בָּנָֽיו:  מ  וְחָגַרְתָּ֩ אֹתָ֨ם אַבְנֵ֜ט אַהֲרֹ֣ן וּבָנָ֗יו וְחָבַשְׁתָּ֤ לָהֶם֙ מִגְבָּעֹ֔ת וְהָיְתָ֥ה לָהֶ֛ם כְּהֻנָּ֖ה לְחֻקַּ֣ת עוֹלָ֑ם וּמִלֵּאתָ֥ יַֽד-אַהֲרֹ֖ן וְיַד-בָּנָֽיו:  ת  וּתְזָרֵז יָתְהוֹן הֶמְיָנִין אַהֲרֹן וּבְנוֹהִי וְתַתְקֵן לְהוֹן כּוֹבָעִין וּתְהֵי לְהוֹן כְּהֻנְתָּא לִקְיַם עָלָם וּתְקָרֵב קֻרְבָּנָא דְאַהֲרֹן וְקֻרְבָּנָא דִבְנוֹהִי:

{י} וְהִקְרַבְתָּ֙ אֶת-הַפָּ֔ר לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְסָמַ֨ךְ אַהֲרֹ֧ן וּבָנָ֛יו אֶת-יְדֵיהֶ֖ם עַל-רֹ֥אשׁ הַפָּֽר:  מ  וְהִקְרַבְתָּ֙ אֶת-הַפָּ֔ר לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְסָמַ֨ךְ אַהֲרֹ֧ן וּבָנָ֛יו אֶת-יְדֵיהֶ֖ם עַל-רֹ֥אשׁ הַפָּֽר:  ת  וּתְקָרֵב יָת תּוֹרָא לִקֳדָם מַשְׁכַּן זִמְנָא וְיִסְמוֹךְ אַהֲרֹן וּבְנוֹהִי יָת יְדֵיהוֹן עַל רֵישׁ תּוֹרָא:

{יא} וְשָׁחַטְתָּ֥ אֶת-הַפָּ֖ר לִפְנֵ֣י יְהוָ֑ה פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  מ  וְשָׁחַטְתָּ֥ אֶת-הַפָּ֖ר לִפְנֵ֣י יְהוָ֑ה פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  ת  וְתִכּוֹס יָת תּוֹרָא קֳדָם יְיָ בִּתְרַע מַשְׁכַּן זִמְנָא:

{יב} וְלָֽקַחְתָּ֙ מִדַּ֣ם הַפָּ֔ר וְנָתַתָּ֛ה עַל-קַרְנֹ֥ת הַמִּזְבֵּ֖חַ בְּאֶצְבָּעֶ֑ךָ וְאֶת-כָּל-הַדָּ֣ם תִּשְׁפֹּ֔ךְ אֶל-יְס֖וֹד הַמִּזְבֵּֽחַ:  מ  וְלָֽקַחְתָּ֙ מִדַּ֣ם הַפָּ֔ר וְנָתַתָּ֛ה עַל-קַרְנֹ֥ת הַמִּזְבֵּ֖חַ בְּאֶצְבָּעֶ֑ךָ וְאֶת-כָּל-הַדָּ֣ם תִּשְׁפֹּ֔ךְ אֶל-יְס֖וֹד הַמִּזְבֵּֽחַ:  ת  וְתִסַב מִדְמָא דְתוֹרָא וְתִתֵּן עַל קַרְנַת מַדְבְּחָא בְּאֶצְבְּעָךְ וְיָת כָּל דְמָא תֵישׁוֹד לִיסוֹדָא דְמַדְבְּחָא:

{יג} וְלָֽקַחְתָּ֗ אֶֽת-כָּל-הַחֵלֶב֮ הַֽמְכַסֶּ֣ה אֶת-הַקֶּרֶב֒ וְאֵ֗ת הַיֹּתֶ֙רֶת֙ עַל-הַכָּבֵ֔ד וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת-הַחֵ֖לֶב אֲשֶׁ֣ר עֲלֵיהֶ֑ן וְהִקְטַרְתָּ֖ הַמִּזְבֵּֽחָה:  מ  וְלָֽקַחְתָּ֗ אֶֽת-כָּל-הַחֵלֶב֮ הַֽמְכַסֶּ֣ה אֶת-הַקֶּרֶב֒ וְאֵ֗ת הַיֹּתֶ֙רֶת֙ עַל-הַכָּבֵ֔ד וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת-הַחֵ֖לֶב אֲשֶׁ֣ר עֲלֵיהֶ֑ן וְהִקְטַרְתָּ֖ הַמִּזְבֵּֽחָה:  ת  וְתִסַב יָת כָּל תַּרְבָּא דְחָפֵי יָת גַוָא וְיָת חִצְרָא דִעַל כַּבְדָא וְיָת תַּרְתֵּין כָּלְיָן וְיָת תַּרְבָּא דִי עֲלֵיהֶן וְתַסֵיק לְמַדְבְּחָא:

{יד} וְאֶת-בְּשַׂ֤ר הַפָּר֙ וְאֶת-עֹר֣וֹ וְאֶת-פִּרְשׁ֔וֹ תִּשְׂרֹ֣ף בָּאֵ֔שׁ מִח֖וּץ לַֽמַּחֲנֶ֑ה חַטָּ֖את הֽוּא:  מ  וְאֶת-בְּשַׂ֤ר הַפָּר֙ וְאֶת-עֹר֣וֹ וְאֶת-פִּרְשׁ֔וֹ תִּשְׂרֹ֣ף בָּאֵ֔שׁ מִח֖וּץ לַֽמַּחֲנֶ֑ה חַטָּ֖את הֽוּא:  ת  וְיָת בִּסְרָא דְתוֹרָא וְיָת מַשְׁכֵהּ וְיָת אוּכְלֵהּ תּוֹקִיד בְּנוּרָא מִבָּרָא לְמַשְׁרִיתָא חַטָאתָא הוּא:

{טו} וְאֶת-הָאַ֥יִל הָאֶחָ֖ד תִּקָּ֑ח וְסָ֨מְכ֜וּ אַהֲרֹ֧ן וּבָנָ֛יו אֶת-יְדֵיהֶ֖ם עַל-רֹ֥אשׁ הָאָֽיִל:  מ  וְאֶת-הָאַ֥יִל הָאֶחָ֖ד תִּקָּ֑ח וְסָ֨מְכ֜וּ אַהֲרֹ֧ן וּבָנָ֛יו אֶת-יְדֵיהֶ֖ם עַל-רֹ֥אשׁ הָאָֽיִל:  ת  וְיָת דִכְרָא חַד תִּסַב וְיִסְמְכוּן אַהֲרֹן וּבְנוֹהִי יָת יְדֵיהוֹן עַל רֵישׁ דִכְרָא:

{טז} וְשָׁחַטְתָּ֖ אֶת-הָאָ֑יִל וְלָֽקַחְתָּ֙ אֶת-דָּמ֔וֹ וְזָרַקְתָּ֥ עַל-הַמִּזְבֵּ֖חַ סָבִֽיב:  מ  וְשָׁחַטְתָּ֖ אֶת-הָאָ֑יִל וְלָֽקַחְתָּ֙ אֶת-דָּמ֔וֹ וְזָרַקְתָּ֥ עַל-הַמִּזְבֵּ֖חַ סָבִֽיב:  ת  וְתִכּוֹס יָת דִכְרָא וְתִסַב יָת דְמֵהּ וְתִזְרוֹק עַל מַדְבְּחָא סְחוֹר סְחוֹר:

{יז} וְאֶ֨ת-הָאַ֔יִל תְּנַתֵּ֖חַ לִנְתָחָ֑יו וְרָחַצְתָּ֤ קִרְבּוֹ֙ וּכְרָעָ֔יו וְנָתַתָּ֥ עַל-נְתָחָ֖יו וְעַל-רֹאשֽׁוֹ:  מ  וְאֶ֨ת-הָאַ֔יִל תְּנַתֵּ֖חַ לִנְתָחָ֑יו וְרָחַצְתָּ֤ קִרְבּוֹ֙ וּכְרָעָ֔יו וְנָתַתָּ֥ עַל-נְתָחָ֖יו וְעַל-רֹאשֽׁוֹ:  ת  וְיָת דִכְרָא תְּפַלֵג לְאֵבָרוֹהִי וּתְחַלִיל גַּוֵהּ וּכְרָעוֹהִי וְתִתֵּן עַל אֵבָרוֹהִי וְעַל רֵישֵׁהּ:

{יח} וְהִקְטַרְתָּ֤ אֶת-כָּל-הָאַ֙יִל֙ הַמִּזְבֵּ֔חָה עֹלָ֥ה ה֖וּא לַֽיהוָ֑ה רֵ֣יחַ נִיח֔וֹחַ אִשֶּׁ֥ה לַיהוָ֖ה הֽוּא:  מ  וְהִקְטַרְתָּ֤ אֶת-כָּל-הָאַ֙יִל֙ הַמִּזְבֵּ֔חָה עֹלָ֥ה ה֖וּא לַֽיהוָ֑ה רֵ֣יחַ נִיח֔וֹחַ אִשֶּׁ֥ה לַיהוָ֖ה הֽוּא:  ת  וְתַסֵיק יָת כָּל דִכְרָא לְמַדְבְּחָא עֲלָתָא הוּא קֳדָם יְיָ לְאִתְקַבָּלָא בְרַעֲוָא קֻרְבָּנָא קֳדָם יְיָ הוּא:

{יט}  חמישי  וְלָ֣קַחְתָּ֔ אֵ֖ת הָאַ֣יִל הַשֵּׁנִ֑י וְסָמַ֨ךְ אַהֲרֹ֧ן וּבָנָ֛יו אֶת-יְדֵיהֶ֖ם עַל-רֹ֥אשׁ הָאָֽיִל:  מ  וְלָ֣קַחְתָּ֔ אֵ֖ת הָאַ֣יִל הַשֵּׁנִ֑י וְסָמַ֨ךְ אַהֲרֹ֧ן וּבָנָ֛יו אֶת-יְדֵיהֶ֖ם עַל-רֹ֥אשׁ הָאָֽיִל:  ת  וְתִסַב יָת דִכְרָא תִנְיָנָא וְיִסְמוֹךְ אַהֲרֹן וּבְנוֹהִי יָת יְדֵיהוֹן עַל רֵישׁ דִכְרָא:

{כ} וְשָׁחַטְתָּ֣ אֶת-הָאַ֗יִל וְלָקַחְתָּ֤ מִדָּמוֹ֙ וְנָֽתַתָּ֡ה עַל-תְּנוּךְ֩ אֹ֨זֶן אַהֲרֹ֜ן וְעַל-תְּנ֨וּךְ אֹ֤זֶן בָּנָיו֙ הַיְמָנִ֔ית וְעַל-בֹּ֤הֶן יָדָם֙ הַיְמָנִ֔ית וְעַל-בֹּ֥הֶן רַגְלָ֖ם הַיְמָנִ֑ית וְזָרַקְתָּ֧ אֶת-הַדָּ֛ם עַל-הַמִּזְבֵּ֖חַ סָבִֽיב:  מ  וְשָׁחַטְתָּ֣ אֶת-הָאַ֗יִל וְלָקַחְתָּ֤ מִדָּמוֹ֙ וְנָֽתַתָּ֡ה עַל-תְּנוּךְ֩ אֹ֨זֶן אַהֲרֹ֜ן וְעַל-תְּנ֨וּךְ אֹ֤זֶן בָּנָיו֙ הַיְמָנִ֔ית וְעַל-בֹּ֤הֶן יָדָם֙ הַיְמָנִ֔ית וְעַל-בֹּ֥הֶן רַגְלָ֖ם הַיְמָנִ֑ית וְזָרַקְתָּ֧ אֶת-הַדָּ֛ם עַל-הַמִּזְבֵּ֖חַ סָבִֽיב:  ת  וְתִכּוֹס יָת דִכְרָא וְתִסַב מִדְמֵהּ וְתִתֵּן עַל רוּם אוּדְנָא דְאַהֲרֹן וְעַל רוּם אוּדְנָא דִבְנוֹהִי דְיַמִינָא וְעַל אִלְיוֹן יְדֵיהוֹן דְיַמִינָא וְעַל אִלְיוֹן רַגְלֵיהוֹן דְיַמִינָא וְתִזְרוֹק יָת דְמָא עַל מַדְבְּחָא סְחוֹר סְחוֹר:

{כא} וְלָקַחְתָּ֞ מִן-הַדָּ֨ם אֲשֶׁ֥ר עַֽל-הַמִּזְבֵּחַ֮ וּמִשֶּׁ֣מֶן הַמִּשְׁחָה֒ וְהִזֵּיתָ֤ עַֽל-אַהֲרֹן֙ וְעַל-בְּגָדָ֔יו וְעַל-בָּנָ֛יו וְעַל-בִּגְדֵ֥י בָנָ֖יו אִתּ֑וֹ וְקָדַ֥שׁ הוּא֙ וּבְגָדָ֔יו וּבָנָ֛יו וּבִגְדֵ֥י בָנָ֖יו אִתּֽוֹ:  מ  וְלָקַחְתָּ֞ מִן-הַדָּ֨ם אֲשֶׁ֥ר עַֽל-הַמִּזְבֵּחַ֮ וּמִשֶּׁ֣מֶן הַמִּשְׁחָה֒ וְהִזֵּיתָ֤ עַֽל-אַהֲרֹן֙ וְעַל-בְּגָדָ֔יו וְעַל-בָּנָ֛יו וְעַל-בִּגְדֵ֥י בָנָ֖יו אִתּ֑וֹ וְקָדַ֥שׁ הוּא֙ וּבְגָדָ֔יו וּבָנָ֛יו וּבִגְדֵ֥י בָנָ֖יו אִתּֽוֹ:  ת  וְתִסַב מִן דְמָא דִי עַל מַדְבְּחָא וּמִמִשְׁחָא דִרְבוּתָא וְתַדֵי עַל אַהֲרֹן וְעַל לְבוּשׁוֹהִי וְעַל בְּנוֹהִי וְעַל לְבוּשֵׁי בְנוֹהִי עִמֵהּ וְיִתְקַדַשׁ הוּא וּלְבוּשׁוֹהִי וּבְנוֹהִי וּלְבוּשֵׁי בְנוֹהִי עִמֵהּ:

{כב} וְלָקַחְתָּ֣ מִן-הָ֠אַיִל הַחֵ֨לֶב וְהָֽאַלְיָ֜ה וְאֶת-הַחֵ֣לֶב | הַֽמְכַסֶּ֣ה אֶת-הַקֶּ֗רֶב וְאֵ֨ת יֹתֶ֤רֶת הַכָּבֵד֙ וְאֵ֣ת | שְׁתֵּ֣י הַכְּלָיֹ֗ת וְאֶת-הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן וְאֵ֖ת שׁ֣וֹק הַיָּמִ֑ין כִּ֛י אֵ֥יל מִלֻּאִ֖ים הֽוּא:  מ  וְלָקַחְתָּ֣ מִן-הָ֠אַיִל הַחֵ֨לֶב וְהָֽאַלְיָ֜ה וְאֶת-הַחֵ֣לֶב | הַֽמְכַסֶּ֣ה אֶת-הַקֶּ֗רֶב וְאֵ֨ת יֹתֶ֤רֶת הַכָּבֵד֙ וְאֵ֣ת | שְׁתֵּ֣י הַכְּלָיֹ֗ת וְאֶת-הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן וְאֵ֖ת שׁ֣וֹק הַיָּמִ֑ין כִּ֛י אֵ֥יל מִלֻּאִ֖ים הֽוּא:  ת  וְתִסַב מִן דִכְרָא תַּרְבָּא וְאַלִיתָא וְיָת תַּרְבָּא דְחָפֵי יָת גַוָא וְיָת חֲצַר כַּבְדָא וְיָת תַּרְתֵּן כָּלְיָן וְיָת תַּרְבָּא דִי עֲלֵיהֶן וְיָת שׁוֹקָא דְיַמִינָא אֲרֵי דְכַר קֻרְבָּנָא הוּא:

{כג} וְכִכַּ֨ר לֶ֜חֶם אַחַ֗ת וַֽחַלַּ֨ת לֶ֥חֶם שֶׁ֛מֶן אַחַ֖ת וְרָקִ֣יק אֶחָ֑ד מִסַּל֙ הַמַּצּ֔וֹת אֲשֶׁ֖ר לִפְנֵ֥י יְהוָֽה:  מ  וְכִכַּ֨ר לֶ֜חֶם אַחַ֗ת וַֽחַלַּ֨ת לֶ֥חֶם שֶׁ֛מֶן אַחַ֖ת וְרָקִ֣יק אֶחָ֑ד מִסַּל֙ הַמַּצּ֔וֹת אֲשֶׁ֖ר לִפְנֵ֥י יְהוָֽה:  ת  וּפִתָּא דִלְחֵם חַד וּגְרִצְתָּא דִלְחֵם מְשַׁח חֲדָא וְאֶסְפּוֹג חָד מִסַלָא דְפַטִירַיָא דִי קֳדָם יְיָ:

{כד} וְשַׂמְתָּ֣ הַכֹּ֔ל עַ֚ל כַּפֵּ֣י אַהֲרֹ֔ן וְעַ֖ל כַּפֵּ֣י בָנָ֑יו וְהֵנַפְתָּ֥ אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֽה:  מ  וְשַׂמְתָּ֣ הַכֹּ֔ל עַ֚ל כַּפֵּ֣י אַהֲרֹ֔ן וְעַ֖ל כַּפֵּ֣י בָנָ֑יו וְהֵנַפְתָּ֥ אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֽה:  ת  וּתְשַׁוֵי כֹלָא עַל יְדָא דְאַהֲרֹן וְעַל יְדָא דִבְנוֹהִי וּתְרִים יָתְהוֹן אֲרָמוּתָא קֳדָם יְיָ:

{כה} וְלָקַחְתָּ֤ אֹתָם֙ מִיָּדָ֔ם וְהִקְטַרְתָּ֥ הַמִּזְבֵּ֖חָה עַל-הָעֹלָ֑ה לְרֵ֤יחַ נִיח֙וֹחַ֙ לִפְנֵ֣י יְהוָ֔ה אִשֶּׁ֥ה ה֖וּא לַיהוָֽה:  מ  וְלָקַחְתָּ֤ אֹתָם֙ מִיָּדָ֔ם וְהִקְטַרְתָּ֥ הַמִּזְבֵּ֖חָה עַל-הָעֹלָ֑ה לְרֵ֤יחַ נִיח֙וֹחַ֙ לִפְנֵ֣י יְהוָ֔ה אִשֶּׁ֥ה ה֖וּא לַיהוָֽה:  ת  וְתִסַב יָתְהוֹן מִידֵיהוֹן וְתַסֵיק לְמַדְבְּחָא עַל עֲלָתָא לְאִתְקַבָּלָא בְרַעֲוָא קֳדָם יְיָ קֻרְבָּנָא הוּא קֳדָם יְיָ:

{כו} וְלָקַחְתָּ֣ אֶת-הֶֽחָזֶ֗ה מֵאֵ֤יל הַמִּלֻּאִים֙ אֲשֶׁ֣ר לְאַהֲרֹ֔ן וְהֵנַפְתָּ֥ אֹת֛וֹ תְּנוּפָ֖ה לִפְנֵ֣י יְהוָ֑ה וְהָיָ֥ה לְךָ֖ לְמָנָֽה:  מ  וְלָקַחְתָּ֣ אֶת-הֶֽחָזֶ֗ה מֵאֵ֤יל הַמִּלֻּאִים֙ אֲשֶׁ֣ר לְאַהֲרֹ֔ן וְהֵנַפְתָּ֥ אֹת֛וֹ תְּנוּפָ֖ה לִפְנֵ֣י יְהוָ֑ה וְהָיָ֥ה לְךָ֖ לְמָנָֽה:  ת  וְתִסַב יָת חֶדְיָא מִדְכַר קֻרְבָּנַיָא דִי לְאַהֲרֹן וּתְרִים יָתֵהּ אֲרָמוּתָא קֳדָם יְיָ וִיהֵי לָךְ לָחֳלָק:

{כז} וְקִדַּשְׁתָּ֞ אֵ֣ת | חֲזֵ֣ה הַתְּנוּפָ֗ה וְאֵת֙ שׁ֣וֹק הַתְּרוּמָ֔ה אֲשֶׁ֥ר הוּנַ֖ף וַאֲשֶׁ֣ר הוּרָ֑ם מֵאֵיל֙ הַמִּלֻּאִ֔ים מֵאֲשֶׁ֥ר לְאַהֲרֹ֖ן וּמֵאֲשֶׁ֥ר לְבָנָֽיו:  מ  וְקִדַּשְׁתָּ֞ אֵ֣ת | חֲזֵ֣ה הַתְּנוּפָ֗ה וְאֵת֙ שׁ֣וֹק הַתְּרוּמָ֔ה אֲשֶׁ֥ר הוּנַ֖ף וַאֲשֶׁ֣ר הוּרָ֑ם מֵאֵיל֙ הַמִּלֻּאִ֔ים מֵאֲשֶׁ֥ר לְאַהֲרֹ֖ן וּמֵאֲשֶׁ֥ר לְבָנָֽיו:  ת  וּתְקַדֵשׁ יָת חֶדְיָא דַאֲרָמוּתָא וְיָת שׁוֹקָא דְאַפְרָשׁוּתָא דִי אִתָּרַם וְדִי אִתַּפְרָשׁ מִדְכַר קֻרְבָּנַיָא מִדִי לְאַהֲרֹן וּמִדִי לִבְנוֹהִי:

{כח} וְהָיָה֩ לְאַהֲרֹ֨ן וּלְבָנָ֜יו לְחָק-עוֹלָ֗ם מֵאֵת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כִּ֥י תְרוּמָ֖ה ה֑וּא וּתְרוּמָ֞ה יִהְיֶ֨ה מֵאֵ֤ת בְּנֵֽי-יִשְׂרָאֵל֙ מִזִּבְחֵ֣י שַׁלְמֵיהֶ֔ם תְּרוּמָתָ֖ם לַיהוָֽה:  מ  וְהָיָה֩ לְאַהֲרֹ֨ן וּלְבָנָ֜יו לְחָק-עוֹלָ֗ם מֵאֵת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כִּ֥י תְרוּמָ֖ה ה֑וּא וּתְרוּמָ֞ה יִהְיֶ֨ה מֵאֵ֤ת בְּנֵֽי-יִשְׂרָאֵל֙ מִזִּבְחֵ֣י שַׁלְמֵיהֶ֔ם תְּרוּמָתָ֖ם לַיהוָֽה:  ת  וִיהֵי לְאַהֲרֹן וְלִבְנוֹהִי לִקְיַם עָלָם מִן בְּנֵי יִשְׂרָאֵל אֲרֵי אַפְרָשׁוּתָא הוּא וְאַפְרָשׁוּתָא יְהֵי מִן בְּנֵי יִשְׂרָאֵל מִנִכְסַת קוּדְשֵׁיהוֹן אַפְרָשׁוּתְהוֹן קֳדָם יְיָ:

{כט} וּבִגְדֵ֤י הַקֹּ֙דֶשׁ֙ אֲשֶׁ֣ר לְאַהֲרֹ֔ן יִהְי֥וּ לְבָנָ֖יו אַחֲרָ֑יו לְמָשְׁחָ֣ה בָהֶ֔ם וּלְמַלֵּא-בָ֖ם אֶת-יָדָֽם:  מ  וּבִגְדֵ֤י הַקֹּ֙דֶשׁ֙ אֲשֶׁ֣ר לְאַהֲרֹ֔ן יִהְי֥וּ לְבָנָ֖יו אַחֲרָ֑יו לְמָשְׁחָ֣ה בָהֶ֔ם וּלְמַלֵּא-בָ֖ם אֶת-יָדָֽם:  ת  וּלְבוּשֵׁי קוּדְשָׁא דִי לְאַהֲרֹן יְהוֹן לִבְנוֹהִי בַּתְרוֹהִי לְדַכָּאָה בְהוֹן וּלְקָרָבָא בְהוֹן יָת קֻרְבָּנְהוֹן:

{ל} שִׁבְעַ֣ת יָמִ֗ים יִלְבָּשָׁ֧ם הַכֹּהֵ֛ן תַּחְתָּ֖יו מִבָּנָ֑יו אֲשֶׁ֥ר יָבֹ֛א אֶל-אֹ֥הֶל מוֹעֵ֖ד לְשָׁרֵ֥ת בַּקֹּֽדֶשׁ:  מ  שִׁבְעַ֣ת יָמִ֗ים יִלְבָּשָׁ֧ם הַכֹּהֵ֛ן תַּחְתָּ֖יו מִבָּנָ֑יו אֲשֶׁ֥ר יָבֹ֛א אֶל-אֹ֥הֶל מוֹעֵ֖ד לְשָׁרֵ֥ת בַּקֹּֽדֶשׁ:  ת  שִׁבְעַת יוֹמִין יַלְבֵּשִׁנוּן כַּהֲנָא תְּחוֹתוֹהִי מִבְּנוֹהִי דִי יֵעוֹל לְמַשְׁכַּן זִמְנָא לְשַׁמָשָׁא בְקוּדְשָׁא:

{לא} וְאֵ֛ת אֵ֥יל הַמִּלֻּאִ֖ים תִּקָּ֑ח וּבִשַּׁלְתָּ֥ אֶת-בְּשָׂר֖וֹ בְּמָקֹ֥ם קָדֹֽשׁ:  מ  וְאֵ֛ת אֵ֥יל הַמִּלֻּאִ֖ים תִּקָּ֑ח וּבִשַּׁלְתָּ֥ אֶת-בְּשָׂר֖וֹ בְּמָקֹ֥ם קָדֹֽשׁ:  ת  וְיָת דְכַר קֻרְבָּנַיָא תִּסָב וּתְבַשֵׁל יָת בִּסְרֵהּ בַּאֲתַר קַדִישׁ:

{לב} וְאָכַ֨ל אַהֲרֹ֤ן וּבָנָיו֙ אֶת-בְּשַׂ֣ר הָאַ֔יִל וְאֶת-הַלֶּ֖חֶם אֲשֶׁ֣ר בַּסָּ֑ל פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  מ  וְאָכַ֨ל אַהֲרֹ֤ן וּבָנָיו֙ אֶת-בְּשַׂ֣ר הָאַ֔יִל וְאֶת-הַלֶּ֖חֶם אֲשֶׁ֣ר בַּסָּ֑ל פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  ת  וְיֵכוּל אַהֲרֹן וּבְנוֹהִי יָת בְּסַר דִכְרָא וְיָת לַחְמָא דִי בְסַלָא לִתְרַע מַשְׁכַּן זִמְנָא:

{לג} וְאָכְל֤וּ אֹתָם֙ אֲשֶׁ֣ר כֻּפַּ֣ר בָּהֶ֔ם לְמַלֵּ֥א אֶת-יָדָ֖ם לְקַדֵּ֣שׁ אֹתָ֑ם וְזָ֥ר לֹא-יֹאכַ֖ל כִּי-קֹ֥דֶשׁ הֵֽם:  מ  וְאָכְל֤וּ אֹתָם֙ אֲשֶׁ֣ר כֻּפַּ֣ר בָּהֶ֔ם לְמַלֵּ֥א אֶת-יָדָ֖ם לְקַדֵּ֣שׁ אֹתָ֑ם וְזָ֥ר לֹא-יֹאכַ֖ל כִּי-קֹ֥דֶשׁ הֵֽם:  ת  וְיֵכְלוּן יָתְהוֹן דִי יִתְכַּפַּר בְּהוֹן לְקָרָבָא יָת קֻרְבָּנְהוֹן לְקַדָשָׁא יָתְהוֹן וְחִלוֹנַי לָא יֵיכוּל אֲרֵי קוּדְשָׁא אִנוּן:

{לד} וְֽאִם-יִוָּתֵ֞ר מִבְּשַׂ֧ר הַמִּלֻּאִ֛ים וּמִן-הַלֶּ֖חֶם עַד-הַבֹּ֑קֶר וְשָׂרַפְתָּ֤ אֶת-הַנּוֹתָר֙ בָּאֵ֔שׁ לֹ֥א יֵאָכֵ֖ל כִּי-קֹ֥דֶשׁ הֽוּא:  מ  וְֽאִם-יִוָּתֵ֞ר מִבְּשַׂ֧ר הַמִּלֻּאִ֛ים וּמִן-הַלֶּ֖חֶם עַד-הַבֹּ֑קֶר וְשָׂרַפְתָּ֤ אֶת-הַנּוֹתָר֙ בָּאֵ֔שׁ לֹ֥א יֵאָכֵ֖ל כִּי-קֹ֥דֶשׁ הֽוּא:  ת  וְאִם יִשְׁתָּאַר מִבְּסַר קֻרְבָּנַיָא וּמִן לַחְמָא עַד צַפְרָא וְתוֹקִיד יָת דְאִשְׁתָּאַר בְּנוּרָא לָא יִתְאֲכֵל אֲרֵי קוּדְשָׁא הוּא:

{לה} וְעָשִׂ֜יתָ לְאַהֲרֹ֤ן וּלְבָנָיו֙ כָּ֔כָה כְּכֹ֥ל אֲשֶׁר-צִוִּ֖יתִי אֹתָ֑כָה שִׁבְעַ֥ת יָמִ֖ים תְּמַלֵּ֥א יָדָֽם:  מ  וְעָשִׂ֜יתָ לְאַהֲרֹ֤ן וּלְבָנָיו֙ כָּ֔כָה כְּכֹ֥ל אֲשֶׁר-צִוִּ֖יתִי אֹתָ֑כָה שִׁבְעַ֥ת יָמִ֖ים תְּמַלֵּ֥א יָדָֽם:  ת  וְתַעְבֵּד לְאַהֲרֹן וְלִבְנוֹהִי כְּדֵין כְּכֹל דִי פַקֵּדִית יָתָךְ שִׁבְעַת יוֹמִין תְּקָרֵב קֻרְבָּנְהוֹן:

{לו} וּפַ֨ר חַטָּ֜את תַּעֲשֶׂ֤ה לַיּוֹם֙ עַל-הַכִּפֻּרִ֔ים וְחִטֵּאתָ֙ עַל-הַמִּזְבֵּ֔חַ בְּכַפֶּרְךָ֖ עָלָ֑יו וּמָֽשַׁחְתָּ֥ אֹת֖וֹ לְקַדְּשֽׁוֹ:  מ  וּפַ֨ר חַטָּ֜את תַּעֲשֶׂ֤ה לַיּוֹם֙ עַל-הַכִּפֻּרִ֔ים וְחִטֵּאתָ֙ עַל-הַמִּזְבֵּ֔חַ בְּכַפֶּרְךָ֖ עָלָ֑יו וּמָֽשַׁחְתָּ֥ אֹת֖וֹ לְקַדְּשֽׁוֹ:  ת  וְתוֹרָא דְחַטָאתָא תַּעְבֵּד לְיוֹמָא עַל כִּפּוּרַיָא וּתְדַכֵּי עַל מַדְבְּחָא בְּכַפָּרוּתָךְ עֲלוֹהִי וּתְרַבֵּי יָתֵהּ לְקַדָשׁוּתֵהּ:

{לז} שִׁבְעַ֣ת יָמִ֗ים תְּכַפֵּר֙ עַל-הַמִּזְבֵּ֔חַ וְקִדַּשְׁתָּ֖ אֹת֑וֹ וְהָיָ֤ה הַמִּזְבֵּ֙חַ֙ קֹ֣דֶשׁ קָֽדָשִׁ֔ים כָּל-הַנֹּגֵ֥עַ בַּמִּזְבֵּ֖חַ יִקְדָּֽשׁ:  מ  שִׁבְעַ֣ת יָמִ֗ים תְּכַפֵּר֙ עַל-הַמִּזְבֵּ֔חַ וְקִדַּשְׁתָּ֖ אֹת֑וֹ וְהָיָ֤ה הַמִּזְבֵּ֙חַ֙ קֹ֣דֶשׁ קָֽדָשִׁ֔ים כָּל-הַנֹּגֵ֥עַ בַּמִּזְבֵּ֖חַ יִקְדָּֽשׁ:  ת  שִׁבְעַת יוֹמִין תְּכַפֵּר עַל מַדְבְּחָא וּתְקַדֵשׁ יָתֵהּ וִיהֵי מַדְבְּחָא קֹדֶשׁ קוּדְשַׁיָא כָּל דְיִקְרַב בְּמַדְבְּחָא יִתְקַדָשׁ: (ס)

{לח}  שישי  וְזֶ֕ה אֲשֶׁ֥ר תַּעֲשֶׂ֖ה עַל-הַמִּזְבֵּ֑חַ כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה שְׁנַ֥יִם לַיּ֖וֹם תָּמִֽיד:  מ  וְזֶ֕ה אֲשֶׁ֥ר תַּעֲשֶׂ֖ה עַל-הַמִּזְבֵּ֑חַ כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה שְׁנַ֥יִם לַיּ֖וֹם תָּמִֽיד:  ת  וְדֵין דִי תַעְבֵּד עַל מַדְבְּחָא אִמְרִין בְּנֵי שְׁנָא תְּרֵין לְיוֹמָא תְּדִירָא:

{לט} אֶת-הַכֶּ֥בֶשׂ הָאֶחָ֖ד תַּעֲשֶׂ֣ה בַבֹּ֑קֶר וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּעֲשֶׂ֖ה בֵּ֥ין הָעַרְבָּֽיִם:  מ  אֶת-הַכֶּ֥בֶשׂ הָאֶחָ֖ד תַּעֲשֶׂ֣ה בַבֹּ֑קֶר וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּעֲשֶׂ֖ה בֵּ֥ין הָעַרְבָּֽיִם:  ת  יָת אִמְרָא חַד תַּעְבֵּד בְּצַפְרָא וְיָת אִמְרָא תִנְיָנָא תַּעְבֵּד בֵּין שִׁמְשַׁיָא:

{מ} וְעִשָּׂרֹ֨ן סֹ֜לֶת בָּל֨וּל בְּשֶׁ֤מֶן כָּתִית֙ רֶ֣בַע הַהִ֔ין וְנֵ֕סֶךְ רְבִעִ֥ית הַהִ֖ין יָ֑יִן לַכֶּ֖בֶשׂ הָאֶחָֽד:  מ  וְעִשָּׂרֹ֨ן סֹ֜לֶת בָּל֨וּל בְּשֶׁ֤מֶן כָּתִית֙ רֶ֣בַע הַהִ֔ין וְנֵ֕סֶךְ רְבִעִ֥ית הַהִ֖ין יָ֑יִן לַכֶּ֖בֶשׂ הָאֶחָֽד:  ת  וְעִסְרוֹנָא סֻלְתָּא דְפִילָא בִמְשַׁח כָּתִישָׁא רַבְעוּת הִינָא וְנִסְכָּא רַבְעוּת הִינָא חַמְרָא לְאִמְרָא חָד:

{מא} וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּעֲשֶׂ֖ה בֵּ֣ין הָעַרְבָּ֑יִם כְּמִנְחַ֨ת הַבֹּ֤קֶר וּכְנִסְכָּהּ֙ תַּֽעֲשֶׂה-לָּ֔הּ לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַיהוָֽה:  מ  וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּעֲשֶׂ֖ה בֵּ֣ין הָעַרְבָּ֑יִם כְּמִנְחַ֨ת הַבֹּ֤קֶר וּכְנִסְכָּהּ֙ תַּֽעֲשֶׂה-לָּ֔הּ לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַיהוָֽה:  ת  וְיָת אִמְרָא תִנְיָנָא תַּעְבֵּד בֵּין שִׁמְשַׁיָא כְּמִנְחַת צַפְרָא וּכְנִסְכַּהּ תַּעְבֵּד לַהּ לְאִתְקַּבָּלָא בְרַעֲוָא קֻרְבָּנָא קֳדָם יְיָ:

{מב} עֹלַ֤ת תָּמִיד֙ לְדֹרֹ֣תֵיכֶ֔ם פֶּ֥תַח אֹֽהֶל-מוֹעֵ֖ד לִפְנֵ֣י יְהוָ֑ה אֲשֶׁ֨ר אִוָּעֵ֤ד לָכֶם֙ שָׁ֔מָּה לְדַבֵּ֥ר אֵלֶ֖יךָ שָֽׁם:  מ  עֹלַ֤ת תָּמִיד֙ לְדֹרֹ֣תֵיכֶ֔ם פֶּ֥תַח אֹֽהֶל-מוֹעֵ֖ד לִפְנֵ֣י יְהוָ֑ה אֲשֶׁ֨ר אִוָּעֵ֤ד לָכֶם֙ שָׁ֔מָּה לְדַבֵּ֥ר אֵלֶ֖יךָ שָֽׁם:  ת  עֲלָתָא תְדִירָא לְדָרֵיכוֹן בִּתְרַע מַשְׁכַּן זִמְנָא קֳדָם יְיָ דִאֱזַמֵן מֵימְרִי לְכוֹן תַּמָן לְמַלָלָא עִמָךְ תַּמָן:

{מג} וְנֹעַדְתִּ֥י שָׁ֖מָּה לִבְנֵ֣י יִשְׂרָאֵ֑ל וְנִקְדַּ֖שׁ בִּכְבֹדִֽי:  מ  וְנֹעַדְתִּ֥י שָׁ֖מָּה לִבְנֵ֣י יִשְׂרָאֵ֑ל וְנִקְדַּ֖שׁ בִּכְבֹדִֽי:  ת  וֶאֱזַמֵן מֵימְרִי תַמָן לִבְנֵי יִשְׂרָאֵל וְיִתְקַדַשׁ בִּיקָרִי:

{מד} וְקִדַּשְׁתִּ֛י אֶת-אֹ֥הֶל מוֹעֵ֖ד וְאֶת-הַמִּזְבֵּ֑חַ וְאֶת-אַהֲרֹ֧ן וְאֶת-בָּנָ֛יו אֲקַדֵּ֖שׁ לְכַהֵ֥ן לִֽי:  מ  וְקִדַּשְׁתִּ֛י אֶת-אֹ֥הֶל מוֹעֵ֖ד וְאֶת-הַמִּזְבֵּ֑חַ וְאֶת-אַהֲרֹ֧ן וְאֶת-בָּנָ֛יו אֲקַדֵּ֖שׁ לְכַהֵ֥ן לִֽי:  ת  וֶאֱקַדֵשׁ יָת מַשְׁכַּן זִמְנָא וְיָת מַדְבְּחָא וְיָת אַהֲרֹן וְיָת בְּנוֹהִי אֱקַדֵשׁ לְשַׁמָשָׁא קֳדָמַי:

{מה} וְשָׁ֣כַנְתִּ֔י בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל וְהָיִ֥יתִי לָהֶ֖ם לֵאלֹהִֽים:  מ  וְשָׁ֣כַנְתִּ֔י בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל וְהָיִ֥יתִי לָהֶ֖ם לֵאלֹהִֽים:  ת  וְאַשְׁרֵי שְׁכִנְתִּי בְּגוֹ בְּנֵי יִשְׂרָאֵל וְאֶהֱוֵי לְהוֹן לֶאֱלָהּ:

{מו} וְיָדְע֗וּ כִּ֣י אֲנִ֤י יְהוָה֙ אֱלֹ֣הֵיהֶ֔ם אֲשֶׁ֨ר הוֹצֵ֧אתִי אֹתָ֛ם מֵאֶ֥רֶץ מִצְרַ֖יִם לְשָׁכְנִ֣י בְתוֹכָ֑ם אֲנִ֖י יְהוָ֥ה אֱלֹהֵיהֶֽם:  מ  וְיָדְע֗וּ כִּ֣י אֲנִ֤י יְהוָה֙ אֱלֹ֣הֵיהֶ֔ם אֲשֶׁ֨ר הוֹצֵ֧אתִי אֹתָ֛ם מֵאֶ֥רֶץ מִצְרַ֖יִם לְשָׁכְנִ֣י בְתוֹכָ֑ם אֲנִ֖י יְהוָ֥ה אֱלֹהֵיהֶֽם:  ת  וְיִדְעוּן אֲרֵי אֲנָא יְיָ אֱלָהֲהוֹן דִי אַפֵּקִית יָתְהוֹן מֵאַרְעָא דְמִצְרַיִם לְאַשְׁרָאָה שְׁכִנְתִּי בֵּינֵיהוֹן אֲנָא יְיָ אֱלָהֲהוֹן: (פ)




שמות פרק-ל

{א}  שביעי  וְעָשִׂ֥יתָ מִזְבֵּ֖חַ מִקְטַ֣ר קְטֹ֑רֶת עֲצֵ֥י שִׁטִּ֖ים תַּעֲשֶׂ֥ה אֹתֽוֹ:  מ  וְעָשִׂ֥יתָ מִזְבֵּ֖חַ מִקְטַ֣ר קְטֹ֑רֶת עֲצֵ֥י שִׁטִּ֖ים תַּעֲשֶׂ֥ה אֹתֽוֹ:  ת  וְתַעְבֵד מַדְבְּחָא לְאַקְטָרָא עֲלוֹהִי קְטוֹרֶת בּוּסְמַיָא דְאָעֵי שִׁטִין תַּעְבֵּד יָתֵהּ:

{ב} אַמָּ֨ה אָרְכּ֜וֹ וְאַמָּ֤ה רָחְבּוֹ֙ רָב֣וּעַ יִהְיֶ֔ה וְאַמָּתַ֖יִם קֹמָת֑וֹ מִמֶּ֖נּוּ קַרְנֹתָֽיו:  מ  אַמָּ֨ה אָרְכּ֜וֹ וְאַמָּ֤ה רָחְבּוֹ֙ רָב֣וּעַ יִהְיֶ֔ה וְאַמָּתַ֖יִם קֹמָת֑וֹ מִמֶּ֖נּוּ קַרְנֹתָֽיו:  ת  אַמְתָא אֻרְכֵּהּ וְאַמְתָא פְתָיֵהּ מְרַבַּע יְהֵי וְתַרְתֵּין אַמִין רוּמֵהּ מִנֵהּ קַרְנוֹהִי:

{ג} וְצִפִּיתָ֨ אֹת֜וֹ זָהָ֣ב טָה֗וֹר אֶת-גַּגּ֧וֹ וְאֶת-קִירֹתָ֛יו סָבִ֖יב וְאֶת-קַרְנֹתָ֑יו וְעָשִׂ֥יתָ לּ֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב:  מ  וְצִפִּיתָ֨ אֹת֜וֹ זָהָ֣ב טָה֗וֹר אֶת-גַּגּ֧וֹ וְאֶת-קִירֹתָ֛יו סָבִ֖יב וְאֶת-קַרְנֹתָ֑יו וְעָשִׂ֥יתָ לּ֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב:  ת  וְתַחֲפֵי יָתֵהּ דְהַב דְכֵי יָת אִגָרֵהּ וְיָת כָּתְלוֹהִי סְחוֹר סְחוֹר וְיָת קַרְנוֹהִי וְתַעְבֵּד לֵהּ דֵיר דִדְהב סְחוֹר סְחוֹר:

{ד} וּשְׁתֵּי֩ טַבְּעֹ֨ת זָהָ֜ב תַּֽעֲשֶׂה-לּ֣וֹ | מִתַּ֣חַת לְזֵר֗וֹ עַ֚ל שְׁתֵּ֣י צַלְעֹתָ֔יו תַּעֲשֶׂ֖ה עַל-שְׁנֵ֣י צִדָּ֑יו וְהָיָה֙ לְבָתִּ֣ים לְבַדִּ֔ים לָשֵׂ֥את אֹת֖וֹ בָּהֵֽמָּה:  מ  וּשְׁתֵּי֩ טַבְּעֹ֨ת זָהָ֜ב תַּֽעֲשֶׂה-לּ֣וֹ | מִתַּ֣חַת לְזֵר֗וֹ עַ֚ל שְׁתֵּ֣י צַלְעֹתָ֔יו תַּעֲשֶׂ֖ה עַל-שְׁנֵ֣י צִדָּ֑יו וְהָיָה֙ לְבָתִּ֣ים לְבַדִּ֔ים לָשֵׂ֥את אֹת֖וֹ בָּהֵֽמָּה:  ת  וְתַרְתֵּין עִזְקַן דִדְהַב תַּעְבֶּד לֵהּ מִלְרַע לְדֵירֵהּ עַל תַּרְתֵּין זִוְיָתֵהּ תַּעְבֵּד עַל תְּרֵין סִטְרוֹהִי וִיהֵי לְאַתְרָא לַאֲרִיחַיָא לְמִטַל יָתֵהּ בְּהוֹן:

{ה} וְעָשִׂ֥יתָ אֶת-הַבַּדִּ֖ים עֲצֵ֣י שִׁטִּ֑ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָֽב:  מ  וְעָשִׂ֥יתָ אֶת-הַבַּדִּ֖ים עֲצֵ֣י שִׁטִּ֑ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָֽב:  ת  וְתַעְבֵּד יָת אֲרִיחַיָא דְאָעֵי שִׁטִין וְתַחֲפֵי יָתְהוֹן דַהֲבָא:

{ו} וְנָתַתָּ֤ה אֹתוֹ֙ לִפְנֵ֣י הַפָּרֹ֔כֶת אֲשֶׁ֖ר עַל-אֲרֹ֣ן הָעֵדֻ֑ת לִפְנֵ֣י הַכַּפֹּ֗רֶת אֲשֶׁר֙ עַל-הָ֣עֵדֻ֔ת אֲשֶׁ֛ר אִוָּעֵ֥ד לְךָ֖ שָֽׁמָּה:  מ  וְנָתַתָּ֤ה אֹתוֹ֙ לִפְנֵ֣י הַפָּרֹ֔כֶת אֲשֶׁ֖ר עַל-אֲרֹ֣ן הָעֵדֻ֑ת לִפְנֵ֣י הַכַּפֹּ֗רֶת אֲשֶׁר֙ עַל-הָ֣עֵדֻ֔ת אֲשֶׁ֛ר אִוָּעֵ֥ד לְךָ֖ שָֽׁמָּה:  ת  וְתִתֵּן יָתֵה קֳדָם פָּרֻכְתָּא דִי עַל אֲרוֹנָא דְסַהֲדוּתָא לִקֳדָם כַּפֻּרְתָּא דִי עַל סַהֲדוּתָא דִי אֲזַמֵן מֵימְרִי לָךְ תַּמָן:

{ז} וְהִקְטִ֥יר עָלָ֛יו אַהֲרֹ֖ן קְטֹ֣רֶת סַמִּ֑ים בַּבֹּ֣קֶר בַּבֹּ֗קֶר בְּהֵיטִיב֛וֹ אֶת-הַנֵּרֹ֖ת יַקְטִירֶֽנָּה:  מ  וְהִקְטִ֥יר עָלָ֛יו אַהֲרֹ֖ן קְטֹ֣רֶת סַמִּ֑ים בַּבֹּ֣קֶר בַּבֹּ֗קֶר בְּהֵיטִיב֛וֹ אֶת-הַנֵּרֹ֖ת יַקְטִירֶֽנָּה:  ת  וְיַקְטַר עֲלוֹהִי אַהֲרֹן קְטוֹרֶת בּוּסְמִין בִּצְפַר בִּצְפַר בְּאַתְקָנוּתֵהּ יָת בּוֹצִינַיָא יַסֵקִנַהּ:

{ח}  מפטיר  וּבְהַעֲלֹ֨ת אַהֲרֹ֧ן אֶת-הַנֵּרֹ֛ת בֵּ֥ין הָעֲרְבַּ֖יִם יַקְטִירֶ֑נָּה קְטֹ֧רֶת תָּמִ֛יד לִפְנֵ֥י יְהוָ֖ה לְדֹרֹתֵיכֶֽם:  מ  וּבְהַעֲלֹ֨ת אַהֲרֹ֧ן אֶת-הַנֵּרֹ֛ת בֵּ֥ין הָעֲרְבַּ֖יִם יַקְטִירֶ֑נָּה קְטֹ֧רֶת תָּמִ֛יד לִפְנֵ֥י יְהוָ֖ה לְדֹרֹתֵיכֶֽם:  ת  וּבְאַדְלָקוּת אַהֲרֹן יָת בּוֹצִינַיָא בֵּין שִׁמְשַׁיָא יַסֵקִנַהּ קְטוֹרֶת בּוּסְמַיָא תְדִירָא קֳדָם יְיָ לְדָרֵיכוֹן:

{ט} לֹא-תַעֲל֥וּ עָלָ֛יו קְטֹ֥רֶת זָרָ֖ה וְעֹלָ֣ה וּמִנְחָ֑ה וְנֵ֕סֶךְ לֹ֥א תִסְּכ֖וּ עָלָֽיו:  מ  לֹא-תַעֲל֥וּ עָלָ֛יו קְטֹ֥רֶת זָרָ֖ה וְעֹלָ֣ה וּמִנְחָ֑ה וְנֵ֕סֶךְ לֹ֥א תִסְּכ֖וּ עָלָֽיו:  ת  לָא תַסְקוּן עֲלוֹהִי קְטוֹרֶת בּוּסְמִין נוּכְרָאִין וַעֲלָתָא וּמִנְחָתָא וְנִסְכָּא לָא תְנַסְכוּן עֲלוֹהִי:

{י} וְכִפֶּ֤ר אַהֲרֹן֙ עַל-קַרְנֹתָ֔יו אַחַ֖ת בַּשָּׁנָ֑ה מִדַּ֞ם חַטַּ֣את הַכִּפֻּרִ֗ים אַחַ֤ת בַּשָּׁנָה֙ יְכַפֵּ֤ר עָלָיו֙ לְדֹרֹ֣תֵיכֶ֔ם קֹֽדֶשׁ-קָֽדָשִׁ֥ים ה֖וּא לַיהוָֽה:  מ  וְכִפֶּ֤ר אַהֲרֹן֙ עַל-קַרְנֹתָ֔יו אַחַ֖ת בַּשָּׁנָ֑ה מִדַּ֞ם חַטַּ֣את הַכִּפֻּרִ֗ים אַחַ֤ת בַּשָּׁנָה֙ יְכַפֵּ֤ר עָלָיו֙ לְדֹרֹ֣תֵיכֶ֔ם קֹֽדֶשׁ-קָֽדָשִׁ֥ים ה֖וּא לַיהוָֽה:  ת  וִיכַפֵר אַהֲרֹן עַל קַרְנוֹהִי חֲדָא בְּשַׁתָּא מִדְמָא חַטַאת כִּפּוּרַיָא חֲדָא בְשַׁתָּא יְכַפֵּר עֲלוֹהִי לְדָרֵיכוֹן קֹדֶשׁ קוּדְשִׁין הוּא קֳדָם יְיָ: (פפפ)
