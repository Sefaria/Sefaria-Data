פרשת תזריע




ויקרא פרק-יב

{א} וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} דַּבֵּ֞ר אֶל-בְּנֵ֤י יִשְׂרָאֵל֙ לֵאמֹ֔ר אִשָּׁה֙ כִּ֣י תַזְרִ֔יעַ וְיָלְדָ֖ה זָכָ֑ר וְטָֽמְאָה֙ שִׁבְעַ֣ת יָמִ֔ים כִּימֵ֛י נִדַּ֥ת דְּוֹתָ֖הּ תִּטְמָֽא:  מ  דַּבֵּ֞ר אֶל-בְּנֵ֤י יִשְׂרָאֵל֙ לֵאמֹ֔ר אִשָּׁה֙ כִּ֣י תַזְרִ֔יעַ וְיָלְדָ֖ה זָכָ֑ר וְטָֽמְאָה֙ שִׁבְעַ֣ת יָמִ֔ים כִּימֵ֛י נִדַּ֥ת דְּוֹתָ֖הּ תִּטְמָֽא:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל לְמֵימָר אִתְּתָא אֲרֵי תְעַדִי וּתְלִיד דְכָר וּתְהֵי מְסָאָבָא שִׁבְעַת יוֹמִין כְּיוֹמֵי רִחוּק סוֹבְתַהּ תְּהֵי מְסָאָבָא:

{ג} וּבַיּ֖וֹם הַשְּׁמִינִ֑י יִמּ֖וֹל בְּשַׂ֥ר עָרְלָתֽוֹ:  מ  וּבַיּ֖וֹם הַשְּׁמִינִ֑י יִמּ֖וֹל בְּשַׂ֥ר עָרְלָתֽוֹ:  ת  וּבְיוֹמָא תְּמִינָאָה יִגְזַר בִּסְרָא דְעָרְלְתֵהּ:

{ד} וּשְׁלֹשִׁ֥ים יוֹם֙ וּשְׁלֹ֣שֶׁת יָמִ֔ים תֵּשֵׁ֖ב בִּדְמֵ֣י טָהֳרָ֑ה בְּכָל-קֹ֣דֶשׁ לֹֽא-תִגָּ֗ע וְאֶל-הַמִּקְדָּשׁ֙ לֹ֣א תָבֹ֔א עַד-מְלֹ֖את יְמֵ֥י טָהֳרָֽהּ:  מ  וּשְׁלֹשִׁ֥ים יוֹם֙ וּשְׁלֹ֣שֶׁת יָמִ֔ים תֵּשֵׁ֖ב בִּדְמֵ֣י טָהֳרָ֑ה בְּכָל-קֹ֣דֶשׁ לֹֽא-תִגָּ֗ע וְאֶל-הַמִּקְדָּשׁ֙ לֹ֣א תָבֹ֔א עַד-מְלֹ֖את יְמֵ֥י טָהֳרָֽהּ:  ת  וּתְלָתִין וּתְלָתָא יוֹמִין תֵּיתֵב בְּדַם דְכוּ בְּכָל קוּדְשָׁא לָא תִקְרַב וּלְמִקְדְשָׁא לָא תֵעוֹל עַד מִשְׁלַם יוֹמֵי דָכוּתַהּ:

{ה} וְאִם-נְקֵבָ֣ה תֵלֵ֔ד וְטָמְאָ֥ה שְׁבֻעַ֖יִם כְּנִדָּתָ֑הּ וְשִׁשִּׁ֥ים יוֹם֙ וְשֵׁ֣שֶׁת יָמִ֔ים תֵּשֵׁ֖ב עַל-דְּמֵ֥י טָהֳרָֽה:  מ  וְאִם-נְקֵבָ֣ה תֵלֵ֔ד וְטָמְאָ֥ה שְׁבֻעַ֖יִם כְּנִדָּתָ֑הּ וְשִׁשִּׁ֥ים יוֹם֙ וְשֵׁ֣שֶׁת יָמִ֔ים תֵּשֵׁ֖ב עַל-דְּמֵ֥י טָהֳרָֽה:  ת  וְאִם נְקוּבְתָא תְלִיד וּתְהֵי מְסָאָבָא אַרְבְּעָה עֲסַר כְּרִחוּקַהּ וְשִׁתִּין וְשִׁתָּא יוֹמִין תֵּיתֵב עַל דַם דְכוּ:

{ו} וּבִמְלֹ֣את | יְמֵ֣י טָהֳרָ֗הּ לְבֵן֮ א֣וֹ לְבַת֒ תָּבִ֞יא כֶּ֤בֶשׂ בֶּן-שְׁנָתוֹ֙ לְעֹלָ֔ה וּבֶן-יוֹנָ֥ה אוֹ-תֹ֖ר לְחַטָּ֑את אֶל-פֶּ֥תַח אֹֽהֶל-מוֹעֵ֖ד אֶל-הַכֹּהֵֽן:  מ  וּבִמְלֹ֣את | יְמֵ֣י טָהֳרָ֗הּ לְבֵן֮ א֣וֹ לְבַת֒ תָּבִ֞יא כֶּ֤בֶשׂ בֶּן-שְׁנָתוֹ֙ לְעֹלָ֔ה וּבֶן-יוֹנָ֥ה אוֹ-תֹ֖ר לְחַטָּ֑את אֶל-פֶּ֥תַח אֹֽהֶל-מוֹעֵ֖ד אֶל-הַכֹּהֵֽן:  ת  וּבְמִשְׁלַם יוֹמֵי דָכוּתַהּ לִבְרָא אוֹ לִבְרַתָּא תַּיְתִי אִמָר בַּר שַׁתֵּהּ לַעֲלָתָא וּבַר יוֹנָה אוֹ שַׁפְנִינָא לְחַטָאתָא לִתְרַע מַשְׁכַּן זִמְנָא לְוַת כַּהֲנָא:

{ז} וְהִקְרִיב֞וֹ לִפְנֵ֤י יְהוָה֙ וְכִפֶּ֣ר עָלֶ֔יהָ וְטָהֲרָ֖ה מִמְּקֹ֣ר דָּמֶ֑יהָ זֹ֤את תּוֹרַת֙ הַיֹּלֶ֔דֶת לַזָּכָ֖ר א֥וֹ לַנְּקֵבָֽה:  מ  וְהִקְרִיב֞וֹ לִפְנֵ֤י יְהוָה֙ וְכִפֶּ֣ר עָלֶ֔יהָ וְטָהֲרָ֖ה מִמְּקֹ֣ר דָּמֶ֑יהָ זֹ֤את תּוֹרַת֙ הַיֹּלֶ֔דֶת לַזָּכָ֖ר א֥וֹ לַנְּקֵבָֽה:  ת  וִיקָרְבִנֵהּ קֳדָם יְיָ וִיכַפֵּר עֲלַהּ וְתִדְכֵּי מִסוֹאֲבַת דְמָהָא דָא אוֹרַיְתָא דְיַלֶדְתָּא לְדוּכְרָא אוֹ לִנְקֻבְתָא:

{ח} וְאִם-לֹ֨א תִמְצָ֣א יָדָהּ֮ דֵּ֣י שֶׂה֒ וְלָקְחָ֣ה שְׁתֵּֽי-תֹרִ֗ים א֤וֹ שְׁנֵי֙ בְּנֵ֣י יוֹנָ֔ה אֶחָ֥ד לְעֹלָ֖ה וְאֶחָ֣ד לְחַטָּ֑את וְכִפֶּ֥ר עָלֶ֛יהָ הַכֹּהֵ֖ן וְטָהֵֽרָה:  מ  וְאִם-לֹ֨א תִמְצָ֣א יָדָהּ֮ דֵּ֣י שֶׂה֒ וְלָקְחָ֣ה שְׁתֵּֽי-תֹרִ֗ים א֤וֹ שְׁנֵי֙ בְּנֵ֣י יוֹנָ֔ה אֶחָ֥ד לְעֹלָ֖ה וְאֶחָ֣ד לְחַטָּ֑את וְכִפֶּ֥ר עָלֶ֛יהָ הַכֹּהֵ֖ן וְטָהֵֽרָה:  ת  וְאִם לָא תַשְׁכַּח יְדַהּ כְּמִסַת אִמְרָא וְתִסַב תַּרְתֵּין שַׁפְנִינִין אוֹ תְרֵין בְּנֵי יוֹנָה חַד לַעֲלָתָא וְחַד לְחַטָאתָא וִיכַפֵּר עֲלַהּ כַּהֲנָא וְתִדְכֵּי: (פ)




ויקרא פרק-יג

{א} וַיְדַבֵּ֣ר יְהוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  מ  וַיְדַבֵּ֣ר יְהוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה וְעִם אַהֲרֹן לְמֵימָר:

{ב} אָדָ֗ם כִּֽי-יִהְיֶ֤ה בְעוֹר-בְּשָׂרוֹ֙ שְׂאֵ֤ת אֽוֹ-סַפַּ֙חַת֙ א֣וֹ בַהֶ֔רֶת וְהָיָ֥ה בְעוֹר-בְּשָׂר֖וֹ לְנֶ֣גַע צָרָ֑עַת וְהוּבָא֙ אֶל-אַהֲרֹ֣ן הַכֹּהֵ֔ן א֛וֹ אֶל-אַחַ֥ד מִבָּנָ֖יו הַכֹּהֲנִֽים:  מ  אָדָ֗ם כִּֽי-יִהְיֶ֤ה בְעוֹר-בְּשָׂרוֹ֙ שְׂאֵ֤ת אֽוֹ-סַפַּ֙חַת֙ א֣וֹ בַהֶ֔רֶת וְהָיָ֥ה בְעוֹר-בְּשָׂר֖וֹ לְנֶ֣גַע צָרָ֑עַת וְהוּבָא֙ אֶל-אַהֲרֹ֣ן הַכֹּהֵ֔ן א֛וֹ אֶל-אַחַ֥ד מִבָּנָ֖יו הַכֹּהֲנִֽים:  ת  אֱנַשׁ אֲרֵי יְהֵי בִמְשַׁךְ בִּסְרֵהּ עַמְקָא אוֹ עַדְיָא אוֹ בַהֲרָא וִיהֵי בִמְשַׁךְ בִּסְרֵהּ לְמַכְתַּשׁ סְגִירוּ וְיִתֵּתֵי לְוַת אַהֲרֹן כַּהֲנָא אוֹ לְוַת חַד מִבְּנוֹהִי כַּהֲנַיָא:

{ג} וְרָאָ֣ה הַכֹּהֵ֣ן אֶת-הַנֶּ֣גַע בְּעֽוֹר-הַ֠בָּשָׂר וְשֵׂעָ֨ר בַּנֶּ֜גַע הָפַ֣ךְ | לָבָ֗ן וּמַרְאֵ֤ה הַנֶּ֙גַע֙ עָמֹק֙ מֵע֣וֹר בְּשָׂר֔וֹ נֶ֥גַע צָרַ֖עַת ה֑וּא וְרָאָ֥הוּ הַכֹּהֵ֖ן וְטִמֵּ֥א אֹתֽוֹ:  מ  וְרָאָ֣ה הַכֹּהֵ֣ן אֶת-הַנֶּ֣גַע בְּעֽוֹר-הַ֠בָּשָׂר וְשֵׂעָ֨ר בַּנֶּ֜גַע הָפַ֣ךְ | לָבָ֗ן וּמַרְאֵ֤ה הַנֶּ֙גַע֙ עָמֹק֙ מֵע֣וֹר בְּשָׂר֔וֹ נֶ֥גַע צָרַ֖עַת ה֑וּא וְרָאָ֥הוּ הַכֹּהֵ֖ן וְטִמֵּ֥א אֹתֽוֹ:  ת  וְיֶחֱזֵי כַהֲנָא יָת מַכְתָּשָׁא בִמְשַׁךְ בִּסְרָא וְשֵׂעָר בְּמַכְתָּשָׁא אִתְהַפִיךְ לְמֶחֱוָר וּמֶחֱזֵי מַכְתָּשָׁא עַמִיק בִּמְשַׁךְ בִּסְרֵהּ מַכְתַּשׁ סְגִירוּתָא הוּא וְיֶחֱזִנֵהּ כַּהֲנָא וִיסָאֵב יָתֵהּ:

{ד} וְאִם-בַּהֶרֶת֩ לְבָנָ֨ה הִ֜וא בְּע֣וֹר בְּשָׂר֗וֹ וְעָמֹק֙ אֵין-מַרְאֶ֣הָ מִן-הָע֔וֹר וּשְׂעָרָ֖ה לֹא-הָפַ֣ךְ לָבָ֑ן וְהִסְגִּ֧יר הַכֹּהֵ֛ן אֶת-הַנֶּ֖גַע שִׁבְעַ֥ת יָמִֽים:  מ  וְאִם-בַּהֶרֶת֩ לְבָנָ֨ה הִ֜וא בְּע֣וֹר בְּשָׂר֗וֹ וְעָמֹק֙ אֵין-מַרְאֶ֣הָ מִן-הָע֔וֹר וּשְׂעָרָ֖ה לֹא-הָפַ֣ךְ לָבָ֑ן וְהִסְגִּ֧יר הַכֹּהֵ֛ן אֶת-הַנֶּ֖גַע שִׁבְעַ֥ת יָמִֽים:  ת  וְאִם בַּהֲרָא חַוְרָא הִיא בִּמְשַׁךְ בִּסְרֵהּ וְעַמִיק לֵית מֶחֱזָהָא מִן מַשְׁכָא וּשְׂעָרָא לָא אִתְהֲפִיךְ לְמֶחֱוָר וְיַסְגַר כַּהֲנָא יָת מַכְתָּשָׁא שִׁבְעָא יוֹמִין:

{ה} וְרָאָ֣הוּ הַכֹּהֵן֮ בַּיּ֣וֹם הַשְּׁבִיעִי֒ וְהִנֵּ֤ה הַנֶּ֙גַע֙ עָמַ֣ד בְּעֵינָ֔יו לֹֽא-פָשָׂ֥ה הַנֶּ֖גַע בָּע֑וֹר וְהִסְגִּיר֧וֹ הַכֹּהֵ֛ן שִׁבְעַ֥ת יָמִ֖ים שֵׁנִֽית:  מ  וְרָאָ֣הוּ הַכֹּהֵן֮ בַּיּ֣וֹם הַשְּׁבִיעִי֒ וְהִנֵּ֤ה הַנֶּ֙גַע֙ עָמַ֣ד בְּעֵינָ֔יו לֹֽא-פָשָׂ֥ה הַנֶּ֖גַע בָּע֑וֹר וְהִסְגִּיר֧וֹ הַכֹּהֵ֛ן שִׁבְעַ֥ת יָמִ֖ים שֵׁנִֽית:  ת  וְיֶחֱזִנֵהּ כַּהֲנָא בְּיוֹמָא שְׁבִיעָאָה וְהָא מַכְתָּשָׁא קָם כַּד הֲוָה לָא אוֹסֵף מַכְתָּשָׁא בְּמַשְׁכָּא וְיַסְגְרִנֵהּ כַּהֲנָא שִׁבְעָא יוֹמִין תִּנְיָנוּת:

{ו}  שני  וְרָאָה֩ הַכֹּהֵ֨ן אֹת֜וֹ בַּיּ֣וֹם הַשְּׁבִיעִי֮ שֵׁנִית֒ וְהִנֵּה֙ כֵּהָ֣ה הַנֶּ֔גַע וְלֹא-פָשָׂ֥ה הַנֶּ֖גַע בָּע֑וֹר וְטִהֲר֤וֹ הַכֹּהֵן֙ מִסְפַּ֣חַת הִ֔יא וְכִבֶּ֥ס בְּגָדָ֖יו וְטָהֵֽר:  מ  וְרָאָה֩ הַכֹּהֵ֨ן אֹת֜וֹ בַּיּ֣וֹם הַשְּׁבִיעִי֮ שֵׁנִית֒ וְהִנֵּה֙ כֵּהָ֣ה הַנֶּ֔גַע וְלֹא-פָשָׂ֥ה הַנֶּ֖גַע בָּע֑וֹר וְטִהֲר֤וֹ הַכֹּהֵן֙ מִסְפַּ֣חַת הִ֔יא וְכִבֶּ֥ס בְּגָדָ֖יו וְטָהֵֽר:  ת  וְיֶחֱזֵי כַהֲנָא יָתֵהּ בְּיוֹמָא שְׁבִיעָאָה תִּנְיָנוּת וְהָא עַמְיָא מַכְתָּשָׁא וְלָא אוֹסֵף מַכְתָּשָׁא בְּמַשְׁכָּא וִידַכִּנֵהּ כַּהֲנָא עֲדִיתָא הִיא וִיצַבַּע לְבוּשׁוֹהִי וְיִדְכֵּי:

{ז} וְאִם-פָּשֹׂ֨ה תִפְשֶׂ֤ה הַמִּסְפַּ֙חַת֙ בָּע֔וֹר אַחֲרֵ֧י הֵרָאֹת֛וֹ אֶל-הַכֹּהֵ֖ן לְטָהֳרָת֑וֹ וְנִרְאָ֥ה שֵׁנִ֖ית אֶל-הַכֹּהֵֽן:  מ  וְאִם-פָּשֹׂ֨ה תִפְשֶׂ֤ה הַמִּסְפַּ֙חַת֙ בָּע֔וֹר אַחֲרֵ֧י הֵרָאֹת֛וֹ אֶל-הַכֹּהֵ֖ן לְטָהֳרָת֑וֹ וְנִרְאָ֥ה שֵׁנִ֖ית אֶל-הַכֹּהֵֽן:  ת  וְאִם אוֹסָפָא תוֹסֵף עֲדִיתָא בְּמַשְׁכָּא בָּתַר דְאִתְּחֲזִי לְכַהֲנָא לְדַכּוּתֵהּ וְיִתַּחֲזִי תִנְיָנוּת לְכַהֲנָא:

{ח} וְרָאָה֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה פָּשְׂתָ֥ה הַמִּסְפַּ֖חַת בָּע֑וֹר וְטִמְּא֥וֹ הַכֹּהֵ֖ן צָרַ֥עַת הִֽוא:  מ  וְרָאָה֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה פָּשְׂתָ֥ה הַמִּסְפַּ֖חַת בָּע֑וֹר וְטִמְּא֥וֹ הַכֹּהֵ֖ן צָרַ֥עַת הִֽוא:  ת  וְיֶחֱזֵי כַהֲנָא וְהָא אוֹסֵפַת עֲדִיתָא בְּמַשְׁכָּא וִיסָאֵבִנֵהּ כַּהֲנָא סגִירוּתָא הִיא: (פ)

{ט} נֶ֣גַע צָרַ֔עַת כִּ֥י תִהְיֶ֖ה בְּאָדָ֑ם וְהוּבָ֖א אֶל-הַכֹּהֵֽן:  מ  נֶ֣גַע צָרַ֔עַת כִּ֥י תִהְיֶ֖ה בְּאָדָ֑ם וְהוּבָ֖א אֶל-הַכֹּהֵֽן:  ת  מַכְתַּשׁ סְגִירוּ אֲרֵי תְהֵי בֶּאֱנָשָׁא ויִתֵּתֵי לְוָת כַּהֲנָא:

{י} וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֤ה שְׂאֵת-לְבָנָה֙ בָּע֔וֹר וְהִ֕יא הָפְכָ֖ה שֵׂעָ֣ר לָבָ֑ן וּמִֽחְיַ֛ת בָּשָׂ֥ר חַ֖י בַּשְׂאֵֽת:  מ  וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֤ה שְׂאֵת-לְבָנָה֙ בָּע֔וֹר וְהִ֕יא הָפְכָ֖ה שֵׂעָ֣ר לָבָ֑ן וּמִֽחְיַ֛ת בָּשָׂ֥ר חַ֖י בַּשְׂאֵֽת:  ת  וְיֶחֱזֵי כֲהֲנָא וְהָא עַמְקָא חַוְרָא בְּמַשְׁכָּא וְהִיא הֲפָכַת שֵׂעָר לְחִוָר וְרֹשֶׁם בִּסְרָא חַיְתָא בְּעַמִקְתָּא:

{יא} צָרַ֨עַת נוֹשֶׁ֤נֶת הִוא֙ בְּע֣וֹר בְּשָׂר֔וֹ וְטִמְּא֖וֹ הַכֹּהֵ֑ן לֹ֣א יַסְגִּרֶ֔נּוּ כִּ֥י טָמֵ֖א הֽוּא:  מ  צָרַ֨עַת נוֹשֶׁ֤נֶת הִוא֙ בְּע֣וֹר בְּשָׂר֔וֹ וְטִמְּא֖וֹ הַכֹּהֵ֑ן לֹ֣א יַסְגִּרֶ֔נּוּ כִּ֥י טָמֵ֖א הֽוּא:  ת  סְגִירוּת עַתִּיקָא הִיא בִּמְשַׁךְ בִּסְרֵהּ וִיסָאֵבִנֵהּ כַּהֲנָא לָא יַסְגְרִנֵהּ אֲרֵי:מסָאָב הוּא:

{יב} וְאִם-פָּר֨וֹחַ תִּפְרַ֤ח הַצָּרַ֙עַת֙ בָּע֔וֹר וְכִסְּתָ֣ה הַצָּרַ֗עַת אֵ֚ת כָּל-ע֣וֹר הַנֶּ֔גַע מֵרֹאשׁ֖וֹ וְעַד-רַגְלָ֑יו לְכָל-מַרְאֵ֖ה עֵינֵ֥י הַכֹּהֵֽן:  מ  וְאִם-פָּר֨וֹחַ תִּפְרַ֤ח הַצָּרַ֙עַת֙ בָּע֔וֹר וְכִסְּתָ֣ה הַצָּרַ֗עַת אֵ֚ת כָּל-ע֣וֹר הַנֶּ֔גַע מֵרֹאשׁ֖וֹ וְעַד-רַגְלָ֑יו לְכָל-מַרְאֵ֖ה עֵינֵ֥י הַכֹּהֵֽן:  ת  וְאִם מִסְגָא תִסְגֵי סְגִירוּתָא בְּמַשְׁכָּא וְתַחְפֵי סְגִירוּתָא יָת כָּל מְשַׁךְ מַכְתָּשָׁא מֵרֵישֵׁהּ וְעַד רַגְלוֹהִי לְכָל חֵיזוּ עֵינֵי כַהֲנָא:

{יג} וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֨ה כִסְּתָ֤ה הַצָּרַ֙עַת֙ אֶת-כָּל-בְּשָׂר֔וֹ וְטִהַ֖ר אֶת-הַנָּ֑גַע כֻּלּ֛וֹ הָפַ֥ךְ לָבָ֖ן טָה֥וֹר הֽוּא:  מ  וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֨ה כִסְּתָ֤ה הַצָּרַ֙עַת֙ אֶת-כָּל-בְּשָׂר֔וֹ וְטִהַ֖ר אֶת-הַנָּ֑גַע כֻּלּ֛וֹ הָפַ֥ךְ לָבָ֖ן טָה֥וֹר הֽוּא:  ת  וְיֶחֱזֵי כַהֲנָא וְהָא חֲפַת סְגִירוּתָא יָת כָּל בִּסְרֵהּ וִידַכִּי יָת מַכְתָּשָׁא כֻּלֵהּ אִתְהֲפִיךְ לְמֶחֱוַר דְכֵי הוּא:

{יד} וּבְי֨וֹם הֵרָא֥וֹת בּ֛וֹ בָּשָׂ֥ר חַ֖י יִטְמָֽא:  מ  וּבְי֨וֹם הֵרָא֥וֹת בּ֛וֹ בָּשָׂ֥ר חַ֖י יִטְמָֽא:  ת  וּבְיוֹמָא דְאִתַּחֲזִי בֵהּ בִּסְרָא חַיָא יְהֵי מְסָאָב:

{טו} וְרָאָ֧ה הַכֹּהֵ֛ן אֶת-הַבָּשָׂ֥ר הַחַ֖י וְטִמְּא֑וֹ הַבָּשָׂ֥ר הַחַ֛י טָמֵ֥א ה֖וּא צָרַ֥עַת הֽוּא:  מ  וְרָאָ֧ה הַכֹּהֵ֛ן אֶת-הַבָּשָׂ֥ר הַחַ֖י וְטִמְּא֑וֹ הַבָּשָׂ֥ר הַחַ֛י טָמֵ֥א ה֖וּא צָרַ֥עַת הֽוּא:  ת  וְיֶחֱזֵי כַהֲנָא יָת בִּסְרָא חַיָא וִיסָאֵבִנֵהּ בִּסְרָא חַיָא מְסָאָב הוּא סְגִירוּת הוּא:

{טז} א֣וֹ כִ֥י יָשׁ֛וּב הַבָּשָׂ֥ר הַחַ֖י וְנֶהְפַּ֣ךְ לְלָבָ֑ן וּבָ֖א אֶל-הַכֹּהֵֽן:  מ  א֣וֹ כִ֥י יָשׁ֛וּב הַבָּשָׂ֥ר הַחַ֖י וְנֶהְפַּ֣ךְ לְלָבָ֑ן וּבָ֖א אֶל-הַכֹּהֵֽן:  ת  אוֹ אֲרֵי יְתוּב בִּסְרָא חַיָא וְיִתְהֲפִיךְ לְמֶחֱוָר וְיִתֵּתֵי לְוַת כַּהֲנָא:

{יז} וְרָאָ֙הוּ֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה נֶהְפַּ֥ךְ הַנֶּ֖גַע לְלָבָ֑ן וְטִהַ֧ר הַכֹּהֵ֛ן אֶת-הַנֶּ֖גַע טָה֥וֹר הֽוּא:  מ  וְרָאָ֙הוּ֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה נֶהְפַּ֥ךְ הַנֶּ֖גַע לְלָבָ֑ן וְטִהַ֧ר הַכֹּהֵ֛ן אֶת-הַנֶּ֖גַע טָה֥וֹר הֽוּא:  ת  וְיֶחֱזִנֵהּ כַּהֲנָא וְהָא אִתְהֲפִיךְ מַכְתָּשָׁא לְמֶחֱוָר וִידַכֵּי כַהֲנָא יָת מַכְתָּשָׁא דְכֵי הוּא: (פ)

{יח}  שלישי  וּבָשָׂ֕ר כִּֽי-יִהְיֶ֥ה בֽוֹ-בְעֹר֖וֹ שְׁחִ֑ין וְנִרְפָּֽא:  מ  וּבָשָׂ֕ר כִּֽי-יִהְיֶ֥ה בֽוֹ-בְעֹר֖וֹ שְׁחִ֑ין וְנִרְפָּֽא:  ת  וּבִסְרָא אֲרֵי יְהֵי בֵהּ בְּמַשְּׁכֵּהּ שִׁחֲנָא וְיִתַּסֵי:

{יט} וְהָיָ֞ה בִּמְק֤וֹם הַשְּׁחִין֙ שְׂאֵ֣ת לְבָנָ֔ה א֥וֹ בַהֶ֖רֶת לְבָנָ֣ה אֲדַמְדָּ֑מֶת וְנִרְאָ֖ה אֶל-הַכֹּהֵֽן:  מ  וְהָיָ֞ה בִּמְק֤וֹם הַשְּׁחִין֙ שְׂאֵ֣ת לְבָנָ֔ה א֥וֹ בַהֶ֖רֶת לְבָנָ֣ה אֲדַמְדָּ֑מֶת וְנִרְאָ֖ה אֶל-הַכֹּהֵֽן:  ת  וִיהֵי בַאֲתַר שִׁחֲנָא עַמְקָא חַוְרָא אוֹ בַהֲרָא חַוְרָא סַמְקָא וְיִתַּחֲזִי לְכַהֲנָא:

{כ} וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֤ה מַרְאֶ֙הָ֙ שָׁפָ֣ל מִן-הָע֔וֹר וּשְׂעָרָ֖הּ הָפַ֣ךְ לָבָ֑ן וְטִמְּא֧וֹ הַכֹּהֵ֛ן נֶֽגַע-צָרַ֥עַת הִ֖וא בַּשְּׁחִ֥ין פָּרָֽחָה:  מ  וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֤ה מַרְאֶ֙הָ֙ שָׁפָ֣ל מִן-הָע֔וֹר וּשְׂעָרָ֖הּ הָפַ֣ךְ לָבָ֑ן וְטִמְּא֧וֹ הַכֹּהֵ֛ן נֶֽגַע-צָרַ֥עַת הִ֖וא בַּשְּׁחִ֥ין פָּרָֽחָה:  ת  וְיֶחֱזֵי כַהֲנָא וְהָא מֶחֱזָהָא מַכִּיךְ מִן מַשְׁכָּא וּשְׂעָרַהּ אִתְהַפִיךְ לְמֶחֱוָר וִיסָאֵבִינֵהּ כַּהֲנָא מַכְתַּשׁ סְגִירוּתָא הִיא בְּשִׁחַנָא סְגִיאַת:

{כא} וְאִ֣ם | יִרְאֶ֣נָּה הַכֹּהֵ֗ן וְהִנֵּ֤ה אֵֽין-בָּהּ֙ שֵׂעָ֣ר לָבָ֔ן וּשְׁפָלָ֥ה אֵינֶ֛נָּה מִן-הָע֖וֹר וְהִ֣יא כֵהָ֑ה וְהִסְגִּיר֥וֹ הַכֹּהֵ֖ן שִׁבְעַ֥ת יָמִֽים:  מ  וְאִ֣ם | יִרְאֶ֣נָּה הַכֹּהֵ֗ן וְהִנֵּ֤ה אֵֽין-בָּהּ֙ שֵׂעָ֣ר לָבָ֔ן וּשְׁפָלָ֥ה אֵינֶ֛נָּה מִן-הָע֖וֹר וְהִ֣יא כֵהָ֑ה וְהִסְגִּיר֥וֹ הַכֹּהֵ֖ן שִׁבְעַ֥ת יָמִֽים:  ת  וְאִם יֶחֱזִנַהּ כַּהֲנָא וְהָא לֵית בַּהּ שֵׂעָר חִוָר וּמַכִּיכָא לֵיתָהָא מִן מַשְׁכָּא וְהִיא עָמַיָא וְיַסְגְרִנֵהּ כַּהֲנָא שִׁבְעָא יוֹמִין:

{כב} וְאִם-פָּשֹׂ֥ה תִפְשֶׂ֖ה בָּע֑וֹר וְטִמֵּ֧א הַכֹּהֵ֛ן אֹת֖וֹ נֶ֥גַע הִֽוא:  מ  וְאִם-פָּשֹׂ֥ה תִפְשֶׂ֖ה בָּע֑וֹר וְטִמֵּ֧א הַכֹּהֵ֛ן אֹת֖וֹ נֶ֥גַע הִֽוא:  ת  וְאִם אוֹסָפָא תוֹסֵף בְּמַשְׁכָּא וִיסָאֵב כַּהֲנָא יָתֵהּ מַכְתָּשָׁא הִיא:

{כג} וְאִם-תַּחְתֶּ֜יהָ תַּעֲמֹ֤ד הַבַּהֶ֙רֶת֙ לֹ֣א פָשָׂ֔תָה צָרֶ֥בֶת הַשְּׁחִ֖ין הִ֑וא וְטִהֲר֖וֹ הַכֹּהֵֽן:  מ  וְאִם-תַּחְתֶּ֜יהָ תַּעֲמֹ֤ד הַבַּהֶ֙רֶת֙ לֹ֣א פָשָׂ֔תָה צָרֶ֥בֶת הַשְּׁחִ֖ין הִ֑וא וְטִהֲר֖וֹ הַכֹּהֵֽן:  ת  וְאִם בְּאַתְרָהָא קָמַת בֶּהַרְתָּא לָא אוֹסֵפַת רֹשֶׁם שִׁחֲנָא הִיא וִידַכִּנֵהּ כַּהֲנָא: (ס)

{כד}  רביעי - שני במחוברין  א֣וֹ בָשָׂ֔ר כִּֽי-יִהְיֶ֥ה בְעֹר֖וֹ מִכְוַת-אֵ֑שׁ וְֽהָיְתָ֞ה מִֽחְיַ֣ת הַמִּכְוָ֗ה בַּהֶ֛רֶת לְבָנָ֥ה אֲדַמְדֶּ֖מֶת א֥וֹ לְבָנָֽה:  מ  א֣וֹ בָשָׂ֔ר כִּֽי-יִהְיֶ֥ה בְעֹר֖וֹ מִכְוַת-אֵ֑שׁ וְֽהָיְתָ֞ה מִֽחְיַ֣ת הַמִּכְוָ֗ה בַּהֶ֛רֶת לְבָנָ֥ה אֲדַמְדֶּ֖מֶת א֥וֹ לְבָנָֽה:  ת  אוֹ בִסְרָא אֲרֵי יְהֵי בְמַשְׁכֵּהּ כְּוָאָה דְנוּר וּתְהֵי רֹשֶׁם כְּוָאָה בַּהֲרָא חַוְרָא סֲמקָא אוֹ חַוְרָא:

{כה} וְרָאָ֣ה אֹתָ֣הּ הַכֹּהֵ֡ן וְהִנֵּ֣ה נֶהְפַּךְ֩ שֵׂעָ֨ר לָבָ֜ן בַּבַּהֶ֗רֶת וּמַרְאֶ֙הָ֙ עָמֹ֣ק מִן-הָע֔וֹר צָרַ֣עַת הִ֔וא בַּמִּכְוָ֖ה פָּרָ֑חָה וְטִמֵּ֤א אֹתוֹ֙ הַכֹּהֵ֔ן נֶ֥גַע צָרַ֖עַת הִֽוא:  מ  וְרָאָ֣ה אֹתָ֣הּ הַכֹּהֵ֡ן וְהִנֵּ֣ה נֶהְפַּךְ֩ שֵׂעָ֨ר לָבָ֜ן בַּבַּהֶ֗רֶת וּמַרְאֶ֙הָ֙ עָמֹ֣ק מִן-הָע֔וֹר צָרַ֣עַת הִ֔וא בַּמִּכְוָ֖ה פָּרָ֑חָה וְטִמֵּ֤א אֹתוֹ֙ הַכֹּהֵ֔ן נֶ֥גַע צָרַ֖עַת הִֽוא:  ת  וְיֶחֱזֵי יָתַהּ כַּהֲנָא וְהָא אִתְהֲפִיךְ שֵׂעָר חִוָר בְּבַהֶרְתָּא וּמֶחֱזָהָא עַמִיק מִן מַשְׁכָּא סְגִירוּתָא הִיא בִּכְוָּאָה סְגִיאָה וִיסָאֵב יָתֵהּ כַּהֲנָא מַכְתַּשׁ סְגִירוּתָא הִיא:

{כו} וְאִ֣ם | יִרְאֶ֣נָּה הַכֹּהֵ֗ן וְהִנֵּ֤ה אֵֽין-בַּבֶּהֶ֙רֶת֙ שֵׂעָ֣ר לָבָ֔ן וּשְׁפָלָ֥ה אֵינֶ֛נָּה מִן-הָע֖וֹר וְהִ֣וא כֵהָ֑ה וְהִסְגִּיר֥וֹ הַכֹּהֵ֖ן שִׁבְעַ֥ת יָמִֽים:  מ  וְאִ֣ם | יִרְאֶ֣נָּה הַכֹּהֵ֗ן וְהִנֵּ֤ה אֵֽין-בַּבֶּהֶ֙רֶת֙ שֵׂעָ֣ר לָבָ֔ן וּשְׁפָלָ֥ה אֵינֶ֛נָּה מִן-הָע֖וֹר וְהִ֣וא כֵהָ֑ה וְהִסְגִּיר֥וֹ הַכֹּהֵ֖ן שִׁבְעַ֥ת יָמִֽים:  ת  וְאִם יֶחֱזִנַהּ כַּהֲנָא וְהָא לֵית בְּבַהֶרְתָּא שֵׂעָר חִוָר וּמַכִּיכָא לֵיתָהָא מִן מַשְׁכָּא וְהִיא עָמְיָא וְיַסְגְרִנֵהּ כַּהֲנָא שִׁבְעָא יוֹמִין:

{כז} וְרָאָ֥הוּ הַכֹּהֵ֖ן בַּיּ֣וֹם הַשְּׁבִיעִ֑י אִם-פָּשֹׂ֤ה תִפְשֶׂה֙ בָּע֔וֹר וְטִמֵּ֤א הַכֹּהֵן֙ אֹת֔וֹ נֶ֥גַע צָרַ֖עַת הִֽוא:  מ  וְרָאָ֥הוּ הַכֹּהֵ֖ן בַּיּ֣וֹם הַשְּׁבִיעִ֑י אִם-פָּשֹׂ֤ה תִפְשֶׂה֙ בָּע֔וֹר וְטִמֵּ֤א הַכֹּהֵן֙ אֹת֔וֹ נֶ֥גַע צָרַ֖עַת הִֽוא:  ת  וְיֶחֱזִנֵהּ כַּהֲנָא בְּיוֹמָא שְׁבִיעָאָה אִם אוֹסָפָא תוֹסֵף בְּמַשְׁכָּא וִיסָאֵב כַּהֲנָא יָתֵהּ מַכְתַּשׁ סְגִירוּתָא הִיא:

{כח} וְאִם-תַּחְתֶּיהָ֩ תַעֲמֹ֨ד הַבַּהֶ֜רֶת לֹא-פָשְׂתָ֤ה בָעוֹר֙ וְהִ֣וא כֵהָ֔ה שְׂאֵ֥ת הַמִּכְוָ֖ה הִ֑וא וְטִֽהֲרוֹ֙ הַכֹּהֵ֔ן כִּֽי-צָרֶ֥בֶת הַמִּכְוָ֖ה הִֽוא:  מ  וְאִם-תַּחְתֶּיהָ֩ תַעֲמֹ֨ד הַבַּהֶ֜רֶת לֹא-פָשְׂתָ֤ה בָעוֹר֙ וְהִ֣וא כֵהָ֔ה שְׂאֵ֥ת הַמִּכְוָ֖ה הִ֑וא וְטִֽהֲרוֹ֙ הַכֹּהֵ֔ן כִּֽי-צָרֶ֥בֶת הַמִּכְוָ֖ה הִֽוא:  ת  וְאִם בְּאַתְרָהָא קָמַת בַּהֶרְתָּא לָא אוֹסֵפַת בְּמַשְׁכָּא וְהִיא עָמְיָא עוֹמֶק כְּוָאָה הִיא וִידַכִּנֵהּ כַּהֲנָא אֲרֵי רֹשֶׁם כְּוָאָה הִיא: (פ)

{כט}  חמישי  וְאִישׁ֙ א֣וֹ אִשָּׁ֔ה כִּֽי-יִהְיֶ֥ה ב֖וֹ נָ֑גַע בְּרֹ֖אשׁ א֥וֹ בְזָקָֽן:  מ  וְאִישׁ֙ א֣וֹ אִשָּׁ֔ה כִּֽי-יִהְיֶ֥ה ב֖וֹ נָ֑גַע בְּרֹ֖אשׁ א֥וֹ בְזָקָֽן:  ת  וּגְבַר אוֹ אִתְּתָא אֲרֵי יְהֵא בֵהּ מַכְתָּשָׁא בְּרֵישׁ אוֹ בִּדְקָן:

{ל} וְרָאָ֨ה הַכֹּהֵ֜ן אֶת-הַנֶּ֗גַע וְהִנֵּ֤ה מַרְאֵ֙הוּ֙ עָמֹ֣ק מִן-הָע֔וֹר וּב֛וֹ שֵׂעָ֥ר צָהֹ֖ב דָּ֑ק וְטִמֵּ֨א אֹת֤וֹ הַכֹּהֵן֙ נֶ֣תֶק ה֔וּא צָרַ֧עַת הָרֹ֛אשׁ א֥וֹ הַזָּקָ֖ן הֽוּא:  מ  וְרָאָ֨ה הַכֹּהֵ֜ן אֶת-הַנֶּ֗גַע וְהִנֵּ֤ה מַרְאֵ֙הוּ֙ עָמֹ֣ק מִן-הָע֔וֹר וּב֛וֹ שֵׂעָ֥ר צָהֹ֖ב דָּ֑ק וְטִמֵּ֨א אֹת֤וֹ הַכֹּהֵן֙ נֶ֣תֶק ה֔וּא צָרַ֧עַת הָרֹ֛אשׁ א֥וֹ הַזָּקָ֖ן הֽוּא:  ת  וְיֶחֱזֵי כַהֲנָא יָת מַכְתָּשָׁא וְהָא מֶחֱזוֹהִי עַמִיק מִן מַשְׁכָּא וּבֵהּ שֵׂעָר סֻמָק דַעְדָק וִיסָאֵב יָתֵהּ כַּהֲנָא נִתְקָא הוּא סְגִירוּת רֵישָׁא אוֹ דִקְנָא הוּא:

{לא} וְכִֽי-יִרְאֶ֨ה הַכֹּהֵ֜ן אֶת-נֶ֣גַע הַנֶּ֗תֶק וְהִנֵּ֤ה אֵין-מַרְאֵ֙הוּ֙ עָמֹ֣ק מִן-הָע֔וֹר וְשֵׂעָ֥ר שָׁחֹ֖ר אֵ֣ין בּ֑וֹ וְהִסְגִּ֧יר הַכֹּהֵ֛ן אֶת-נֶ֥גַע הַנֶּ֖תֶק שִׁבְעַ֥ת יָמִֽים:  מ  וְכִֽי-יִרְאֶ֨ה הַכֹּהֵ֜ן אֶת-נֶ֣גַע הַנֶּ֗תֶק וְהִנֵּ֤ה אֵין-מַרְאֵ֙הוּ֙ עָמֹ֣ק מִן-הָע֔וֹר וְשֵׂעָ֥ר שָׁחֹ֖ר אֵ֣ין בּ֑וֹ וְהִסְגִּ֧יר הַכֹּהֵ֛ן אֶת-נֶ֥גַע הַנֶּ֖תֶק שִׁבְעַ֥ת יָמִֽים:  ת  וַאֲרֵי יֶחֱזֵי כַהֲנָא יָת מַכְתַּשׁ נִתְקָא וְהָא לֵית מֶחֱזוֹהִי עַמִיק מִן מַשְּׁכָּא וְשֵׂעָר אֻכָם לֵית בֵּיהּ וְיַסְגַר כַּהֲנָא יָת מַכְתַּשׁ נִתְקָא שִׁבְעַת יוֹמִין:

{לב} וְרָאָ֨ה הַכֹּהֵ֣ן אֶת-הַנֶּגַע֮ בַּיּ֣וֹם הַשְּׁבִיעִי֒ וְהִנֵּה֙ לֹא-פָשָׂ֣ה הַנֶּ֔תֶק וְלֹא-הָ֥יָה ב֖וֹ שֵׂעָ֣ר צָהֹ֑ב וּמַרְאֵ֣ה הַנֶּ֔תֶק אֵ֥ין עָמֹ֖ק מִן-הָעֽוֹר:  מ  וְרָאָ֨ה הַכֹּהֵ֣ן אֶת-הַנֶּגַע֮ בַּיּ֣וֹם הַשְּׁבִיעִי֒ וְהִנֵּה֙ לֹא-פָשָׂ֣ה הַנֶּ֔תֶק וְלֹא-הָ֥יָה ב֖וֹ שֵׂעָ֣ר צָהֹ֑ב וּמַרְאֵ֣ה הַנֶּ֔תֶק אֵ֥ין עָמֹ֖ק מִן-הָעֽוֹר:  ת  וְיֶחֱזֵי כַהֲנָא יָת מַכְתָּשָׁא בְּיוֹמָא שְׁבִיעָאָה וְהָא לָא אוֹסֵף נִתְקָא וְלָא הֲוָה בֵהּ שֵׂעָר סֻמָק וּמֶחֱזֵי נִתְקָא לֵית עַמִיק מִן מַשְׁכָּא:

{לג} וְהִ֨תְגַּלָּ֔ח וְאֶת-הַנֶּ֖תֶק לֹ֣א יְגַלֵּ֑חַ וְהִסְגִּ֨יר הַכֹּהֵ֧ן אֶת-הַנֶּ֛תֶק שִׁבְעַ֥ת יָמִ֖ים שֵׁנִֽית:  מ  וְהִ֨תְגַּלָּ֔ח וְאֶת-הַנֶּ֖תֶק לֹ֣א יְגַלֵּ֑חַ וְהִסְגִּ֨יר הַכֹּהֵ֧ן אֶת-הַנֶּ֛תֶק שִׁבְעַ֥ת יָמִ֖ים שֵׁנִֽית:  ת  וִיגַלַח סָחֳרָנֵי נִתְקָא וּדְעִם נִתְקָא לָא יְגַלַח וְיַסְגַר כַּהֲנָא יָת נִתְקָא שִׁבְעָא יוֹמִין תִּנְיָנוּת:

{לד} וְרָאָה֩ הַכֹּהֵ֨ן אֶת-הַנֶּ֜תֶק בַּיּ֣וֹם הַשְּׁבִיעִ֗י וְ֠הִנֵּה לֹא-פָשָׂ֤ה הַנֶּ֙תֶק֙ בָּע֔וֹר וּמַרְאֵ֕הוּ אֵינֶ֥נּוּ עָמֹ֖ק מִן-הָע֑וֹר וְטִהַ֤ר אֹתוֹ֙ הַכֹּהֵ֔ן וְכִבֶּ֥ס בְּגָדָ֖יו וְטָהֵֽר:  מ  וְרָאָה֩ הַכֹּהֵ֨ן אֶת-הַנֶּ֜תֶק בַּיּ֣וֹם הַשְּׁבִיעִ֗י וְ֠הִנֵּה לֹא-פָשָׂ֤ה הַנֶּ֙תֶק֙ בָּע֔וֹר וּמַרְאֵ֕הוּ אֵינֶ֥נּוּ עָמֹ֖ק מִן-הָע֑וֹר וְטִהַ֤ר אֹתוֹ֙ הַכֹּהֵ֔ן וְכִבֶּ֥ס בְּגָדָ֖יו וְטָהֵֽר:  ת  וְיֶחֱזֵי כַהֲנָא יָת נִתְקָא בְּיוֹמָא שְׁבִיעָאָה וְהָא לָא אוֹסֵף נִתְקָא בְּמַשְׁכָּא וּמֶחֱזוֹהִי לֵיתוֹהִי עַמִיק מִן מַשְׁכָּא וִידַכֵּי יָתֵהּ כַּהֲנָא וִיצַבַּע לְבוּשׁוֹהִי וְיִדְכֵּי:

{לה} וְאִם-פָּשֹׂ֥ה יִפְשֶׂ֛ה הַנֶּ֖תֶק בָּע֑וֹר אַחֲרֵ֖י טָהֳרָתֽוֹ:  מ  וְאִם-פָּשֹׂ֥ה יִפְשֶׂ֛ה הַנֶּ֖תֶק בָּע֑וֹר אַחֲרֵ֖י טָהֳרָתֽוֹ:  ת  וְאִם אוֹסָפָא יוֹסֵף נִתְקָא בְּמַשְׁכָּא בָּתַר דָכוּתֵהּ:

{לו} וְרָאָ֙הוּ֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה פָּשָׂ֥ה הַנֶּ֖תֶק בָּע֑וֹר לֹֽא-יְבַקֵּ֧ר הַכֹּהֵ֛ן לַשֵּׂעָ֥ר הַצָּהֹ֖ב טָמֵ֥א הֽוּא:  מ  וְרָאָ֙הוּ֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה פָּשָׂ֥ה הַנֶּ֖תֶק בָּע֑וֹר לֹֽא-יְבַקֵּ֧ר הַכֹּהֵ֛ן לַשֵּׂעָ֥ר הַצָּהֹ֖ב טָמֵ֥א הֽוּא:  ת  וְיֶחֱזִנֵהּ כַּהֲנָא וְהָא אוֹסֵף נִתְקָא בְּמַשְׁכָּא לָא יְבַקֵר כַּהֲנָא לְשֵׂעָר סֻמָק מְסָאָב הוּא:

{לז} וְאִם-בְּעֵינָיו֩ עָמַ֨ד הַנֶּ֜תֶק וְשֵׂעָ֨ר שָׁחֹ֧ר צָֽמַח-בּ֛וֹ נִרְפָּ֥א הַנֶּ֖תֶק טָה֣וֹר ה֑וּא וְטִהֲר֖וֹ הַכֹּהֵֽן:  מ  וְאִם-בְּעֵינָיו֩ עָמַ֨ד הַנֶּ֜תֶק וְשֵׂעָ֨ר שָׁחֹ֧ר צָֽמַח-בּ֛וֹ נִרְפָּ֥א הַנֶּ֖תֶק טָה֣וֹר ה֑וּא וְטִהֲר֖וֹ הַכֹּהֵֽן:  ת  וְאִם כַּד הֲוָה קָם נִתְקָא וְשֵׂעָר אֻכָּם צְמַח בֵּהּ אִתַּסִי נִתְקָא דְכֵי הוּא וִידַכִּנֵהּ כַּהֲנָא: (ס)

{לח} וְאִישׁ֙ אֽוֹ-אִשָּׁ֔ה כִּֽי-יִהְיֶ֥ה בְעוֹר-בְּשָׂרָ֖ם בֶּהָרֹ֑ת בֶּהָרֹ֖ת לְבָנֹֽת:  מ  וְאִישׁ֙ אֽוֹ-אִשָּׁ֔ה כִּֽי-יִהְיֶ֥ה בְעוֹר-בְּשָׂרָ֖ם בֶּהָרֹ֑ת בֶּהָרֹ֖ת לְבָנֹֽת:  ת  וגְבַר אוֹ אִתְּתָא אֲרֵי יְהֵי בִמְשַׁךְ בִּסְרְהוֹן בַּהֲרָן בַּהֲרָן חַוְרָן:

{לט} וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֧ה בְעוֹר-בְּשָׂרָ֛ם בֶּהָרֹ֖ת כֵּה֣וֹת לְבָנֹ֑ת בֹּ֥הַק ה֛וּא פָּרַ֥ח בָּע֖וֹר טָה֥וֹר הֽוּא:  מ  וְרָאָ֣ה הַכֹּהֵ֗ן וְהִנֵּ֧ה בְעוֹר-בְּשָׂרָ֛ם בֶּהָרֹ֖ת כֵּה֣וֹת לְבָנֹ֑ת בֹּ֥הַק ה֛וּא פָּרַ֥ח בָּע֖וֹר טָה֥וֹר הֽוּא:  ת  וְיֶחֱזֵי כַהֲנָא וְהָא בִמְשַׁךְ בִּסְרְהוֹן בַּהֲרָן עָמְיָן חַוְרָן בָּהֲקָא הוּא סָגֵי בְמַשְׁכָּא דְכֵי הוּא: (ס)

{מ}  שישי - שני במחוברין  וְאִ֕ישׁ כִּ֥י יִמָּרֵ֖ט רֹאשׁ֑וֹ קֵרֵ֥חַ ה֖וּא טָה֥וֹר הֽוּא:  מ  וְאִ֕ישׁ כִּ֥י יִמָּרֵ֖ט רֹאשׁ֑וֹ קֵרֵ֥חַ ה֖וּא טָה֥וֹר הֽוּא:  ת  וּגְבַר אֲרֵי יַתַּר שְׂעַר רֵישֵׁהּ קְרַח הוּא דְכֵי הוּא:

{מא} וְאִם֙ מִפְּאַ֣ת פָּנָ֔יו יִמָּרֵ֖ט רֹאשׁ֑וֹ גִּבֵּ֥חַ ה֖וּא טָה֥וֹר הֽוּא:  מ  וְאִם֙ מִפְּאַ֣ת פָּנָ֔יו יִמָּרֵ֖ט רֹאשׁ֑וֹ גִּבֵּ֥חַ ה֖וּא טָה֥וֹר הֽוּא:  ת  וְאִם מִלָקֳבֵל אַפּוֹהִי יַתַּר שְׂעַר רֵישֵׁהּ גְלוֹשׁ הוּא דְכֵי הוּא:

{מב} וְכִֽי-יִהְיֶ֤ה בַקָּרַ֙חַת֙ א֣וֹ בַגַּבַּ֔חַת נֶ֖גַע לָבָ֣ן אֲדַמְדָּ֑ם צָרַ֤עַת פֹּרַ֙חַת֙ הִ֔וא בְּקָרַחְתּ֖וֹ א֥וֹ בְגַבַּחְתּֽוֹ:  מ  וְכִֽי-יִהְיֶ֤ה בַקָּרַ֙חַת֙ א֣וֹ בַגַּבַּ֔חַת נֶ֖גַע לָבָ֣ן אֲדַמְדָּ֑ם צָרַ֤עַת פֹּרַ֙חַת֙ הִ֔וא בְּקָרַחְתּ֖וֹ א֥וֹ בְגַבַּחְתּֽוֹ:  ת  וַאֲרֵי יְהֵי בְקָרָחוּתֵהּ אוֹ בִגְלוֹשׁוּתֵהּ מַכְתַּשׁ חִוָר סַמוֹק סְגִירוּת סָגְיָא הִיא בְּקָרָחוּתֵהּ אוֹ בִגְלוֹשּׁוּתֵהּ:

{מג} וְרָאָ֨ה אֹת֜וֹ הַכֹּהֵ֗ן וְהִנֵּ֤ה שְׂאֵת-הַנֶּ֙גַע֙ לְבָנָ֣ה אֲדַמְדֶּ֔מֶת בְּקָרַחְתּ֖וֹ א֣וֹ בְגַבַּחְתּ֑וֹ כְּמַרְאֵ֥ה צָרַ֖עַת ע֥וֹר בָּשָֽׂר:  מ  וְרָאָ֨ה אֹת֜וֹ הַכֹּהֵ֗ן וְהִנֵּ֤ה שְׂאֵת-הַנֶּ֙גַע֙ לְבָנָ֣ה אֲדַמְדֶּ֔מֶת בְּקָרַחְתּ֖וֹ א֣וֹ בְגַבַּחְתּ֑וֹ כְּמַרְאֵ֥ה צָרַ֖עַת ע֥וֹר בָּשָֽׂר:  ת  וְיֶחֱזֵי יָתֵהּ כַּהֲנָא וְהָא עַמִיק מַכְתָּשָׁא חַוְרָא סַמְקָא בְּקָרָחוּתֵהּ אוֹ בִגְלוֹשׁוּתֵהּ כְּמֶחֱזֵי סְגִירוּתָא מְשַׁךְ בִּסְרָא:

{מד} אִישׁ-צָר֥וּעַ ה֖וּא טָמֵ֣א ה֑וּא טַמֵּ֧א יְטַמְּאֶ֛נּוּ הַכֹּהֵ֖ן בְּרֹאשׁ֥וֹ נִגְעֽוֹ:  מ  אִישׁ-צָר֥וּעַ ה֖וּא טָמֵ֣א ה֑וּא טַמֵּ֧א יְטַמְּאֶ֛נּוּ הַכֹּהֵ֖ן בְּרֹאשׁ֥וֹ נִגְעֽוֹ:  ת  גְבַר סְגִיר הוּא מְסָאָב הוּא סָאֳבָא יְסָאֵבִנֵהּ כַּהֲנָא בְּרֵישֵּׁהּ מַכְתָּשֵׁהּ:

{מה} וְהַצָּר֜וּעַ אֲשֶׁר-בּ֣וֹ הַנֶּ֗גַע בְּגָדָ֞יו יִהְי֤וּ פְרֻמִים֙ וְרֹאשׁוֹ֙ יִהְיֶ֣ה פָר֔וּעַ וְעַל-שָׂפָ֖ם יַעְטֶ֑ה וְטָמֵ֥א | טָמֵ֖א יִקְרָֽא:  מ  וְהַצָּר֜וּעַ אֲשֶׁר-בּ֣וֹ הַנֶּ֗גַע בְּגָדָ֞יו יִהְי֤וּ פְרֻמִים֙ וְרֹאשׁוֹ֙ יִהְיֶ֣ה פָר֔וּעַ וְעַל-שָׂפָ֖ם יַעְטֶ֑ה וְטָמֵ֥א | טָמֵ֖א יִקְרָֽא:  ת  וּסְגִירָא דִי בֵהּ מַכְתָּשָׁא לְבוּשׁוֹהִי יְהוֹן מְבַזְעִין וְרֵישֵׁהּ יְהֵי פְרִיעַ וְעַל שָׂפָם כַּאֲבֵלָא יִתְעַטָף וְלָא תִסְתַּאֲבוּן וְלָא תִסְתַּאֲבוּן יִקְרֵי:

{מו} כָּל-יְמֵ֞י אֲשֶׁ֨ר הַנֶּ֥גַע בּ֛וֹ יִטְמָ֖א טָמֵ֣א ה֑וּא בָּדָ֣ד יֵשֵׁ֔ב מִח֥וּץ לַֽמַּחֲנֶ֖ה מוֹשָׁבֽוֹ:  מ  כָּל-יְמֵ֞י אֲשֶׁ֨ר הַנֶּ֥גַע בּ֛וֹ יִטְמָ֖א טָמֵ֣א ה֑וּא בָּדָ֣ד יֵשֵׁ֔ב מִח֥וּץ לַֽמַּחֲנֶ֖ה מוֹשָׁבֽוֹ:  ת  כָּל יוֹמֵי דִי מַכְתָּשָּׁא בֵהּ יְהֵי מְסָאָב מְסָאָב הוּא בִּלְחוֹדוֹהִי יֵיתֵב מִבָּרָא לְמַשְׁרִיתָא מוֹתְבֵהּ: (ס)

{מז} וְהַבֶּ֕גֶד כִּֽי-יִהְיֶ֥ה ב֖וֹ נֶ֣גַע צָרָ֑עַת בְּבֶ֣גֶד צֶ֔מֶר א֖וֹ בְּבֶ֥גֶד פִּשְׁתִּֽים:  מ  וְהַבֶּ֕גֶד כִּֽי-יִהְיֶ֥ה ב֖וֹ נֶ֣גַע צָרָ֑עַת בְּבֶ֣גֶד צֶ֔מֶר א֖וֹ בְּבֶ֥גֶד פִּשְׁתִּֽים:  ת  וּלְבוּשָׁא אֲרֵי יְהֵי בֵהּ מַכְתַּשׁ סְגִירוּ בִּלְבֻשׁ עֲמַר אוֹ בִּלְבוּשׁ כִּתָּן:

{מח} א֤וֹ בִֽשְׁתִי֙ א֣וֹ בְעֵ֔רֶב לַפִּשְׁתִּ֖ים וְלַצָּ֑מֶר א֣וֹ בְע֔וֹר א֖וֹ בְּכָל-מְלֶ֥אכֶת עֽוֹר:  מ  א֤וֹ בִֽשְׁתִי֙ א֣וֹ בְעֵ֔רֶב לַפִּשְׁתִּ֖ים וְלַצָּ֑מֶר א֣וֹ בְע֔וֹר א֖וֹ בְּכָל-מְלֶ֥אכֶת עֽוֹר:  ת  אוֹ בְשִׁתְיָא אוֹ בְעַרְבָא לְכִתָּנָא וּלְעַמְרָא אוֹ בְמַשְׁכָּא אוֹ בְּכָל עֲבִידַת מְשָׁךְ:

{מט} וְהָיָ֨ה הַנֶּ֜גַע יְרַקְרַ֣ק | א֣וֹ אֲדַמְדָּ֗ם בַּבֶּגֶד֩ א֨וֹ בָע֜וֹר אֽוֹ-בַשְּׁתִ֤י אוֹ-בָעֵ֙רֶב֙ א֣וֹ בְכָל-כְּלִי-ע֔וֹר נֶ֥גַע צָרַ֖עַת ה֑וּא וְהָרְאָ֖ה אֶת-הַכֹּהֵֽן:  מ  וְהָיָ֨ה הַנֶּ֜גַע יְרַקְרַ֣ק | א֣וֹ אֲדַמְדָּ֗ם בַּבֶּגֶד֩ א֨וֹ בָע֜וֹר אֽוֹ-בַשְּׁתִ֤י אוֹ-בָעֵ֙רֶב֙ א֣וֹ בְכָל-כְּלִי-ע֔וֹר נֶ֥גַע צָרַ֖עַת ה֑וּא וְהָרְאָ֖ה אֶת-הַכֹּהֵֽן:  ת  וִיהֵי מַכְתָּשָׁא יָרוֹק אוֹ סַמוֹק בִּלְבוּשָׁא אוֹ בְמַשְׁכָּא אוֹ בְשִׁתְיָא אוֹ בְעַרְבָא אוֹ בְכָל מַאן דִמְשַׁךְ מַכְתַּשׁ סְגִירוּתָא הוּא וְיִתַּחֲזֵי לְכַהֲנָא:

{נ} וְרָאָ֥ה הַכֹּהֵ֖ן אֶת-הַנָּ֑גַע וְהִסְגִּ֥יר אֶת-הַנֶּ֖גַע שִׁבְעַ֥ת יָמִֽים:  מ  וְרָאָ֥ה הַכֹּהֵ֖ן אֶת-הַנָּ֑גַע וְהִסְגִּ֥יר אֶת-הַנֶּ֖גַע שִׁבְעַ֥ת יָמִֽים:  ת  וְיֶחֱזֵי כַהֲנָא יָת מַכְתָּשָׁא וְיַסְגַר יָת מַכְתָּשָׁא שִׁבְעַת יוֹמִין:

{נא} וְרָאָ֨ה אֶת-הַנֶּ֜גַע בַּיּ֣וֹם הַשְּׁבִיעִ֗י כִּֽי-פָשָׂ֤ה הַנֶּ֙גַע֙ בַּ֠בֶּגֶד אֽוֹ-בַשְּׁתִ֤י אֽוֹ-בָעֵ֙רֶב֙ א֣וֹ בָע֔וֹר לְכֹ֛ל אֲשֶׁר-יֵעָשֶׂ֥ה הָע֖וֹר לִמְלָאכָ֑ה צָרַ֧עַת מַמְאֶ֛רֶת הַנֶּ֖גַע טָמֵ֥א הֽוּא:  מ  וְרָאָ֨ה אֶת-הַנֶּ֜גַע בַּיּ֣וֹם הַשְּׁבִיעִ֗י כִּֽי-פָשָׂ֤ה הַנֶּ֙גַע֙ בַּ֠בֶּגֶד אֽוֹ-בַשְּׁתִ֤י אֽוֹ-בָעֵ֙רֶב֙ א֣וֹ בָע֔וֹר לְכֹ֛ל אֲשֶׁר-יֵעָשֶׂ֥ה הָע֖וֹר לִמְלָאכָ֑ה צָרַ֧עַת מַמְאֶ֛רֶת הַנֶּ֖גַע טָמֵ֥א הֽוּא:  ת  וְיֶחֱזֵי יָת מַכְתָּשָׁא בְּיוֹמָא שְׁבִיעָאָה אֲרֵי אוֹסֵף מַכְתָּשָּׁא בִּלְבוּשָׁא אוֹ בְשִׁתְיָא אוֹ בְעַרְבָא אוֹ בְמַשְׁכָּא לְכֹל דִי יִתְעֲבֵד מַשְׁכָּא לְעִבִידָא סְגִירוּת מְחַסְרָא מַכְתָּשָׁא מְסָאָב הוּא:

{נב} וְשָׂרַ֨ף אֶת-הַבֶּ֜גֶד א֥וֹ אֶֽת-הַשְּׁתִ֣י | א֣וֹ אֶת-הָעֵ֗רֶב בַּצֶּ֙מֶר֙ א֣וֹ בַפִּשְׁתִּ֔ים א֚וֹ אֶת-כָּל-כְּלִ֣י הָע֔וֹר אֲשֶׁר-יִהְיֶ֥ה ב֖וֹ הַנָּ֑גַע כִּֽי-צָרַ֤עַת מַמְאֶ֙רֶת֙ הִ֔וא בָּאֵ֖שׁ תִּשָּׂרֵֽף:  מ  וְשָׂרַ֨ף אֶת-הַבֶּ֜גֶד א֥וֹ אֶֽת-הַשְּׁתִ֣י | א֣וֹ אֶת-הָעֵ֗רֶב בַּצֶּ֙מֶר֙ א֣וֹ בַפִּשְׁתִּ֔ים א֚וֹ אֶת-כָּל-כְּלִ֣י הָע֔וֹר אֲשֶׁר-יִהְיֶ֥ה ב֖וֹ הַנָּ֑גַע כִּֽי-צָרַ֤עַת מַמְאֶ֙רֶת֙ הִ֔וא בָּאֵ֖שׁ תִּשָּׂרֵֽף:  ת  וְיוֹקִיד יָת לְבוּשָׁא אוֹ יָת שִׁתְיָא אוֹ יָת עַרְבָא בְעַמְרָא אוֹ בְכִתָּנָא אוֹ יָת כָּל מָאן דִמְשַׁךְ דִי יְהֵי בֵהּ מַכְתָּשָׁא אֲרֵי סְגִירוּת מְחַסְרָא הִיא בְּנוּרָא תִּתּוֹקָד:

{נג} וְאִם֮ יִרְאֶ֣ה הַכֹּהֵן֒ וְהִנֵּה֙ לֹא-פָשָׂ֣ה הַנֶּ֔גַע בַּבֶּ֕גֶד א֥וֹ בַשְּׁתִ֖י א֣וֹ בָעֵ֑רֶב א֖וֹ בְּכָל-כְּלִי-עֽוֹר:  מ  וְאִם֮ יִרְאֶ֣ה הַכֹּהֵן֒ וְהִנֵּה֙ לֹא-פָשָׂ֣ה הַנֶּ֔גַע בַּבֶּ֕גֶד א֥וֹ בַשְּׁתִ֖י א֣וֹ בָעֵ֑רֶב א֖וֹ בְּכָל-כְּלִי-עֽוֹר:  ת  וְאִם יֶחֱזֵי כַהֲנָא וְהָא לָא אוֹסֵף מַכְתָּשָׁא בִּלְבוּשָׁא אוֹ בְשִׁתְיָא אוֹ בְעַרְבָא אוֹ בְּכָל מָאן דִמְשָׁךְ:

{נד} וְצִוָּה֙ הַכֹּהֵ֔ן וְכִ֨בְּס֔וּ אֵ֥ת אֲשֶׁר-בּ֖וֹ הַנָּ֑גַע וְהִסְגִּיר֥וֹ שִׁבְעַת-יָמִ֖ים שֵׁנִֽית:  מ  וְצִוָּה֙ הַכֹּהֵ֔ן וְכִ֨בְּס֔וּ אֵ֥ת אֲשֶׁר-בּ֖וֹ הַנָּ֑גַע וְהִסְגִּיר֥וֹ שִׁבְעַת-יָמִ֖ים שֵׁנִֽית:  ת  וִיפַקֵד כַּהֲנָא וִיחַוְרוּן יָת דִי בֵהּ מַכְתָּשָׁא וְיַסְגְרִנֵהּ שִׁבְעַת יוֹמִין תִּנְיָנוּת:

{נה}  שביעי - רביעי במחוברין  וְרָאָ֨ה הַכֹּהֵ֜ן אַחֲרֵ֣י | הֻכַּבֵּ֣ס אֶת-הַנֶּ֗גַע וְ֠הִנֵּה לֹֽא-הָפַ֨ךְ הַנֶּ֤גַע אֶת-עֵינוֹ֙ וְהַנֶּ֣גַע לֹֽא-פָשָׂ֔ה טָמֵ֣א ה֔וּא בָּאֵ֖שׁ תִּשְׂרְפֶ֑נּוּ פְּחֶ֣תֶת הִ֔וא בְּקָרַחְתּ֖וֹ א֥וֹ בְגַבַּחְתּֽוֹ:  מ  וְרָאָ֨ה הַכֹּהֵ֜ן אַחֲרֵ֣י | הֻכַּבֵּ֣ס אֶת-הַנֶּ֗גַע וְ֠הִנֵּה לֹֽא-הָפַ֨ךְ הַנֶּ֤גַע אֶת-עֵינוֹ֙ וְהַנֶּ֣גַע לֹֽא-פָשָׂ֔ה טָמֵ֣א ה֔וּא בָּאֵ֖שׁ תִּשְׂרְפֶ֑נּוּ פְּחֶ֣תֶת הִ֔וא בְּקָרַחְתּ֖וֹ א֥וֹ בְגַבַּחְתּֽוֹ:  ת  וְיֶחֱזֵי כַהֲנָא בָּתַר דְחַוָרוּ יָת מַכְתָּשָׁא וְהָא לָא שְׁנָא מַכְתָּשָׁא מִן כַּד הֲוָה וּמַכְתָּשָׁא לָא אוֹסֵף מְסָאָב הוּא בְּנוּרָא תּוֹקְדִנֵהּ תַּבְרָא הִיא בִּשְׁחִיקוּתֵהּ אוֹ בחַדָתוּתֵהּ:

{נו} וְאִם֮ רָאָ֣ה הַכֹּהֵן֒ וְהִנֵּה֙ כֵּהָ֣ה הַנֶּ֔גַע אַחֲרֵ֖י הֻכַּבֵּ֣ס אֹת֑וֹ וְקָרַ֣ע אֹת֗וֹ מִן-הַבֶּ֙גֶד֙ א֣וֹ מִן-הָע֔וֹר א֥וֹ מִן-הַשְּׁתִ֖י א֥וֹ מִן-הָעֵֽרֶב:  מ  וְאִם֮ רָאָ֣ה הַכֹּהֵן֒ וְהִנֵּה֙ כֵּהָ֣ה הַנֶּ֔גַע אַחֲרֵ֖י הֻכַּבֵּ֣ס אֹת֑וֹ וְקָרַ֣ע אֹת֗וֹ מִן-הַבֶּ֙גֶד֙ א֣וֹ מִן-הָע֔וֹר א֥וֹ מִן-הַשְּׁתִ֖י א֥וֹ מִן-הָעֵֽרֶב:  ת  וְאִם חֲזָא:כַהֲנָא וְהָא עָמְיָא מַכְתָּשָׁא בָּתַר דְחַוָרוּ יָתֵהּ וּבְזַע יָתֵהּ מִן לְבוּשָׁא אוֹ מִן מַשְׁכָּא אוֹ מִן שִׁתְיָא אוֹ מִן עַרְבָא:

{נז}  מפטיר  וְאִם-תֵּרָאֶ֨ה ע֜וֹד בַּ֠בֶּגֶד אֽוֹ-בַשְּׁתִ֤י אֽוֹ-בָעֵ֙רֶב֙ א֣וֹ בְכָל-כְּלִי-ע֔וֹר פֹּרַ֖חַת הִ֑וא בָּאֵ֣שׁ תִּשְׂרְפֶ֔נּוּ אֵ֥ת אֲשֶׁר-בּ֖וֹ הַנָּֽגַע:  מ  וְאִם-תֵּרָאֶ֨ה ע֜וֹד בַּ֠בֶּגֶד אֽוֹ-בַשְּׁתִ֤י אֽוֹ-בָעֵ֙רֶב֙ א֣וֹ בְכָל-כְּלִי-ע֔וֹר פֹּרַ֖חַת הִ֑וא בָּאֵ֣שׁ תִּשְׂרְפֶ֔נּוּ אֵ֥ת אֲשֶׁר-בּ֖וֹ הַנָּֽגַע:  ת  וְאִם תִּתְחֲזֵי עוֹד בִּלְבוּשָׁא אוֹ בְשִׁתְיָא אוֹ בְעַרְבָא אוֹ בְכָל מָאן דִמְשַׁךְ סָגְיָא הִיא בְּנוּרָא תוֹקְדִנֵהּ יָת דִי בֵהּ מַכְתָּשָׁא:

{נח} וְהַבֶּ֡גֶד אֽוֹ-הַשְּׁתִ֨י אוֹ-הָעֵ֜רֶב אֽוֹ-כָל-כְּלִ֤י הָעוֹר֙ אֲשֶׁ֣ר תְּכַבֵּ֔ס וְסָ֥ר מֵהֶ֖ם הַנָּ֑גַע וְכֻבַּ֥ס שֵׁנִ֖ית וְטָהֵֽר:  מ  וְהַבֶּ֡גֶד אֽוֹ-הַשְּׁתִ֨י אוֹ-הָעֵ֜רֶב אֽוֹ-כָל-כְּלִ֤י הָעוֹר֙ אֲשֶׁ֣ר תְּכַבֵּ֔ס וְסָ֥ר מֵהֶ֖ם הַנָּ֑גַע וְכֻבַּ֥ס שֵׁנִ֖ית וְטָהֵֽר:  ת  וּלְבוּשָׁא אוֹ שַׁתְיָא אוֹ עַרְבָא אוֹ כָל מָאן דִמְשָׁךְ דִי יִתְחַוַר וְיֶעְדֵי מִנְהוֹן מַכְתָּשָׁא וְיִצְטַּבַּע תִּנְיָנוּת וְיִדְכֵּי:

{נט} זֹ֠את תּוֹרַ֨ת נֶֽגַע-צָרַ֜עַת בֶּ֥גֶד הַצֶּ֣מֶר | א֣וֹ הַפִּשְׁתִּ֗ים א֤וֹ הַשְּׁתִי֙ א֣וֹ הָעֵ֔רֶב א֖וֹ כָּל-כְּלִי-ע֑וֹר לְטַהֲר֖וֹ א֥וֹ לְטַמְּאֽוֹ:  מ  זֹ֠את תּוֹרַ֨ת נֶֽגַע-צָרַ֜עַת בֶּ֥גֶד הַצֶּ֣מֶר | א֣וֹ הַפִּשְׁתִּ֗ים א֤וֹ הַשְּׁתִי֙ א֣וֹ הָעֵ֔רֶב א֖וֹ כָּל-כְּלִי-ע֑וֹר לְטַהֲר֖וֹ א֥וֹ לְטַמְּאֽוֹ:  ת  דָא אוֹרַיְתָא דְמַכְתַּשׁ סְגִירוּ בִּלְבוּשׁ עֲמַר אוֹ כִתָּנָא אוֹ שִׁתְיָא אוֹ עַרְבָא אוֹ כָּל מָאן דִמְשָׁךְ לְדַכָּיוּתֵהּ אוֹ לְסַאֲבוּתֵהּ: (פפפ)
