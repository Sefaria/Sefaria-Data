פרשת תרומה




שמות פרק-כה

{א} וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{ב} דַּבֵּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל וְיִקְחוּ-לִ֖י תְּרוּמָ֑ה מֵאֵ֤ת כָּל-אִישׁ֙ אֲשֶׁ֣ר יִדְּבֶ֣נּוּ לִבּ֔וֹ תִּקְח֖וּ אֶת-תְּרוּמָתִֽי:  מ  דַּבֵּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל וְיִקְחוּ-לִ֖י תְּרוּמָ֑ה מֵאֵ֤ת כָּל-אִישׁ֙ אֲשֶׁ֣ר יִדְּבֶ֣נּוּ לִבּ֔וֹ תִּקְח֖וּ אֶת-תְּרוּמָתִֽי:  ת  מַלֵיל עִם בְּנֵי יִשְׂרָאֵל וְיַפְרְשׁוּן קֳדָמַי אַפְרָשׁוּתָא מִן כָּל גְבַר דִי יִתִּרְעֵי לִבֵּהּ תִּסְבוּן יָת אַפְרָשׁוּתִי:

{ג} וְזֹאת֙ הַתְּרוּמָ֔ה אֲשֶׁ֥ר תִּקְח֖וּ מֵאִתָּ֑ם זָהָ֥ב וָכֶ֖סֶף וּנְחֹֽשֶׁת:  מ  וְזֹאת֙ הַתְּרוּמָ֔ה אֲשֶׁ֥ר תִּקְח֖וּ מֵאִתָּ֑ם זָהָ֥ב וָכֶ֖סֶף וּנְחֹֽשֶׁת:  ת  וְדָא אַפְרָשׁוּתָא דִתִסְבוּן מִנְהוֹן דַהֲבָא וְכַסְפָּא וּנְחָשָׁא:

{ד} וּתְכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ וְעִזִּֽים:  מ  וּתְכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ וְעִזִּֽים:  ת  וְתִכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ וּמֵעַזֵי:

{ה} וְעֹרֹ֨ת אֵילִ֧ם מְאָדָּמִ֛ים וְעֹרֹ֥ת תְּחָשִׁ֖ים וַעֲצֵ֥י שִׁטִּֽים:  מ  וְעֹרֹ֨ת אֵילִ֧ם מְאָדָּמִ֛ים וְעֹרֹ֥ת תְּחָשִׁ֖ים וַעֲצֵ֥י שִׁטִּֽים:  ת  וּמַשְׁכֵי דְדִכְרֵי מְסַמְקֵי וּמַשְׁכֵי סַסְגוֹנָא וְאָעֵי שִׁטִין:

{ו} שֶׁ֖מֶן לַמָּאֹ֑ר בְּשָׂמִים֙ לְשֶׁ֣מֶן הַמִּשְׁחָ֔ה וְלִקְטֹ֖רֶת הַסַּמִּֽים:  מ  שֶׁ֖מֶן לַמָּאֹ֑ר בְּשָׂמִים֙ לְשֶׁ֣מֶן הַמִּשְׁחָ֔ה וְלִקְטֹ֖רֶת הַסַּמִּֽים:  ת  מִשְׁחָא לְאַנְהָרוּתָא בּוּסְמַיָא לְמִשְׁחָא דִרְבוּתָא וְלִקְטוֹרֶת בּוּסְמַיָא:

{ז} אַבְנֵי-שֹׁ֕הַם וְאַבְנֵ֖י מִלֻּאִ֑ים לָאֵפֹ֖ד וְלַחֹֽשֶׁן:  מ  אַבְנֵי-שֹׁ֕הַם וְאַבְנֵ֖י מִלֻּאִ֑ים לָאֵפֹ֖ד וְלַחֹֽשֶׁן:  ת  אַבְנֵי בוּרְלָא וְאַבְנֵי אַשְׁלָמוּתָא לְשַׁקָעָא בְאֵפוֹדָא וּבְחוּשְׁנָא:

{ח} וְעָ֥שׂוּ לִ֖י מִקְדָּ֑שׁ וְשָׁכַנְתִּ֖י בְּתוֹכָֽם:  מ  וְעָ֥שׂוּ לִ֖י מִקְדָּ֑שׁ וְשָׁכַנְתִּ֖י בְּתוֹכָֽם:  ת  וְיַעְבְּדוּן קֳדָמַי מִקְדְשָׁא וְאַשְׁרֵי שְׁכִנְתִּי בֵּינֵיהוֹן:

{ט} כְּכֹ֗ל אֲשֶׁ֤ר אֲנִי֙ מַרְאֶ֣ה אוֹתְךָ֔ אֵ֚ת תַּבְנִ֣ית הַמִּשְׁכָּ֔ן וְאֵ֖ת תַּבְנִ֣ית כָּל-כֵּלָ֑יו וְכֵ֖ן תַּעֲשֽׂוּ:  מ  כְּכֹ֗ל אֲשֶׁ֤ר אֲנִי֙ מַרְאֶ֣ה אוֹתְךָ֔ אֵ֚ת תַּבְנִ֣ית הַמִּשְׁכָּ֔ן וְאֵ֖ת תַּבְנִ֣ית כָּל-כֵּלָ֑יו וְכֵ֖ן תַּעֲשֽׂוּ:  ת  כְּכֹל דִי אֲנָא מַחֲזֵי יָתָךְ יָת דְמוּת מַשְׁכְּנָא וְיָת דְמוּת כָּל מָנוֹהִי וְכֵן תַּעְבְּדוּן: (ס)

{י} וְעָשׂ֥וּ אֲר֖וֹן עֲצֵ֣י שִׁטִּ֑ים אַמָּתַ֨יִם וָחֵ֜צִי אָרְכּ֗וֹ וְאַמָּ֤ה וָחֵ֙צִי֙ רָחְבּ֔וֹ וְאַמָּ֥ה וָחֵ֖צִי קֹמָתֽוֹ:  מ  וְעָשׂ֥וּ אֲר֖וֹן עֲצֵ֣י שִׁטִּ֑ים אַמָּתַ֨יִם וָחֵ֜צִי אָרְכּ֗וֹ וְאַמָּ֤ה וָחֵ֙צִי֙ רָחְבּ֔וֹ וְאַמָּ֥ה וָחֵ֖צִי קֹמָתֽוֹ:  ת  וְיַעְבְּדוּן אֲרוֹנָא דְאָעֵי שִׁטִּין תַּרְתֵּין אַמִין וּפַלְגָא אֻרְכֵּהּ וְאַמְתָא וּפַלְגָא פְּתָיֵהּ וְאַמְתָא וּפַלְגָא רוּמֵה:

{יא} וְצִפִּיתָ֤ אֹתוֹ֙ זָהָ֣ב טָה֔וֹר מִבַּ֥יִת וּמִח֖וּץ תְּצַפֶּ֑נּוּ וְעָשִׂ֧יתָ עָלָ֛יו זֵ֥ר זָהָ֖ב סָבִֽיב:  מ  וְצִפִּיתָ֤ אֹתוֹ֙ זָהָ֣ב טָה֔וֹר מִבַּ֥יִת וּמִח֖וּץ תְּצַפֶּ֑נּוּ וְעָשִׂ֧יתָ עָלָ֛יו זֵ֥ר זָהָ֖ב סָבִֽיב:  ת  וְתֶחֱפֵי יָתֵהּ דְהַב דְכֵי מִגָו וּמִבָּרָא תַּחֲפִנֵהּ וְתַעְבֵּד עֲלוֹהִי זֵיר דִדְהַב סְחוֹר סְחוֹר:

{יב} וְיָצַ֣קְתָּ לּ֗וֹ אַרְבַּע֙ טַבְּעֹ֣ת זָהָ֔ב וְנָ֣תַתָּ֔ה עַ֖ל אַרְבַּ֣ע פַּעֲמֹתָ֑יו וּשְׁתֵּ֣י טַבָּעֹ֗ת עַל-צַלְעוֹ֙ הָֽאֶחָ֔ת וּשְׁתֵּי֙ טַבָּעֹ֔ת עַל-צַלְע֖וֹ הַשֵּׁנִֽית:  מ  וְיָצַ֣קְתָּ לּ֗וֹ אַרְבַּע֙ טַבְּעֹ֣ת זָהָ֔ב וְנָ֣תַתָּ֔ה עַ֖ל אַרְבַּ֣ע פַּעֲמֹתָ֑יו וּשְׁתֵּ֣י טַבָּעֹ֗ת עַל-צַלְעוֹ֙ הָֽאֶחָ֔ת וּשְׁתֵּי֙ טַבָּעֹ֔ת עַל-צַלְע֖וֹ הַשֵּׁנִֽית:  ת  וְתַתִּיךְ לֵהּ אַרְבַּע עִזְקָן דִדְהַב וְתִתֵּן עַל אַרְבַּע זִוְיָתֵהּ וְתַרְתֵּין עִזְקָן עַל סִטְרֵהּ חַד וְתַרְתֵּן עִזְקָן עַל סִטְרֵהּ תִּנְיֵתָא:

{יג} וְעָשִׂ֥יתָ בַדֵּ֖י עֲצֵ֣י שִׁטִּ֑ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָֽב:  מ  וְעָשִׂ֥יתָ בַדֵּ֖י עֲצֵ֣י שִׁטִּ֑ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָֽב:  ת  וְתַעְבֵּד אֲרִיחֵי דְאָעֵי שִׁטִּין וְתַחֲפֵי יָתְהוֹן דַהֲבָא:

{יד} וְהֵבֵאתָ֤ אֶת-הַבַּדִּים֙ בַּטַּבָּעֹ֔ת עַ֖ל צַלְעֹ֣ת הָאָרֹ֑ן לָשֵׂ֥את אֶת-הָאָרֹ֖ן בָּהֶֽם:  מ  וְהֵבֵאתָ֤ אֶת-הַבַּדִּים֙ בַּטַּבָּעֹ֔ת עַ֖ל צַלְעֹ֣ת הָאָרֹ֑ן לָשֵׂ֥את אֶת-הָאָרֹ֖ן בָּהֶֽם:  ת  וְתָעֵל יָת אֲרִיחֲיָא בְּעִזְקָתָא עַל סִטְרֵי אֲרוֹנָא לְמִטַל יָת אֲרוֹנָא בְּהוֹן:

{טו} בְּטַבְּעֹת֙ הָאָרֹ֔ן יִהְי֖וּ הַבַּדִּ֑ים לֹ֥א יָסֻ֖רוּ מִמֶּֽנּוּ:  מ  בְּטַבְּעֹת֙ הָאָרֹ֔ן יִהְי֖וּ הַבַּדִּ֑ים לֹ֥א יָסֻ֖רוּ מִמֶּֽנּוּ:  ת  בְּעִזְקְתָא דַאֲרוֹנָא יְהוֹן אֲרִיחַיָא לָא יְעִדוּן מִנֵהּ:

{טז} וְנָתַתָּ֖ אֶל-הָאָרֹ֑ן אֵ֚ת הָעֵדֻ֔ת אֲשֶׁ֥ר אֶתֵּ֖ן אֵלֶֽיךָ:  מ  וְנָתַתָּ֖ אֶל-הָאָרֹ֑ן אֵ֚ת הָעֵדֻ֔ת אֲשֶׁ֥ר אֶתֵּ֖ן אֵלֶֽיךָ:  ת  וְתִתֵּן בַּאֲרוֹנָא יָת סַהֲדוּתָא דִי אֶתֵּן לָךְ:

{יז}  שני  וְעָשִׂ֥יתָ כַפֹּ֖רֶת זָהָ֣ב טָה֑וֹר אַמָּתַ֤יִם וָחֵ֙צִי֙ אָרְכָּ֔הּ וְאַמָּ֥ה וָחֵ֖צִי רָחְבָּֽהּ:  מ  וְעָשִׂ֥יתָ כַפֹּ֖רֶת זָהָ֣ב טָה֑וֹר אַמָּתַ֤יִם וָחֵ֙צִי֙ אָרְכָּ֔הּ וְאַמָּ֥ה וָחֵ֖צִי רָחְבָּֽהּ:  ת  וְתַעְבֵּד כַּפֻּרְתָּא דִדְהַב דְכֵי תַּרְתֵּין אַמִין וּפַלְגָא אֻרְכַּהּ וְאַמְתָא וּפַלְגָא פְּתָיַהּ:

{יח} וְעָשִׂ֛יתָ שְׁנַ֥יִם כְּרֻבִ֖ים זָהָ֑ב מִקְשָׁה֙ תַּעֲשֶׂ֣ה אֹתָ֔ם מִשְּׁנֵ֖י קְצ֥וֹת הַכַּפֹּֽרֶת:  מ  וְעָשִׂ֛יתָ שְׁנַ֥יִם כְּרֻבִ֖ים זָהָ֑ב מִקְשָׁה֙ תַּעֲשֶׂ֣ה אֹתָ֔ם מִשְּׁנֵ֖י קְצ֥וֹת הַכַּפֹּֽרֶת:  ת  וְתַעְבֵּד תְּרֵין כְּרוּבִין דִדְהָב נְגִיד תַּעְבֵּד יָתְהוֹן מִתְּרֵין סִטְרֵי כַפֻּרְתָּא:

{יט} וַ֠עֲשֵׂה כְּר֨וּב אֶחָ֤ד מִקָּצָה֙ מִזֶּ֔ה וּכְרוּב-אֶחָ֥ד מִקָּצָ֖ה מִזֶּ֑ה מִן-הַכַּפֹּ֛רֶת תַּעֲשׂ֥וּ אֶת-הַכְּרֻבִ֖ים עַל-שְׁנֵ֥י קְצוֹתָֽיו:  מ  וַ֠עֲשֵׂה כְּר֨וּב אֶחָ֤ד מִקָּצָה֙ מִזֶּ֔ה וּכְרוּב-אֶחָ֥ד מִקָּצָ֖ה מִזֶּ֑ה מִן-הַכַּפֹּ֛רֶת תַּעֲשׂ֥וּ אֶת-הַכְּרֻבִ֖ים עַל-שְׁנֵ֥י קְצוֹתָֽיו:  ת  וְעִבֵּד כְּרוּבָא חַד מִסִטְרָא מִכָּא וּכְרוּבָא חַד מִסִטְרָא מִכָּא מִן כַּפֻּרְתָּא תַּעְבְּדוּן יָת כְּרוּבַיָא עַל תְּרֵין סִטְרוֹהִי:

{כ} וְהָי֣וּ הַכְּרֻבִים֩ פֹּרְשֵׂ֨י כְנָפַ֜יִם לְמַ֗עְלָה סֹכְכִ֤ים בְּכַנְפֵיהֶם֙ עַל-הַכַּפֹּ֔רֶת וּפְנֵיהֶ֖ם אִ֣ישׁ אֶל-אָחִ֑יו אֶל-הַכַּפֹּ֔רֶת יִהְי֖וּ פְּנֵ֥י הַכְּרֻבִֽים:  מ  וְהָי֣וּ הַכְּרֻבִים֩ פֹּרְשֵׂ֨י כְנָפַ֜יִם לְמַ֗עְלָה סֹכְכִ֤ים בְּכַנְפֵיהֶם֙ עַל-הַכַּפֹּ֔רֶת וּפְנֵיהֶ֖ם אִ֣ישׁ אֶל-אָחִ֑יו אֶל-הַכַּפֹּ֔רֶת יִהְי֖וּ פְּנֵ֥י הַכְּרֻבִֽים:  ת  וִיהוֹן כְּרוּבַיָא פְּרִיסָן גַדְפֵיהוֹן לְעֵלָא מְטַלִין בְּגַדְפֵהוֹן עַל כַּפֻּרְתָּא וְאַפֵּיהוֹן חַד לָקֳבֵל חַד לָקֳבֵל כַּפֻּרְתָּא יְהוֹן אַפֵי כְרוּבַיָא:

{כא} וְנָתַתָּ֧ אֶת-הַכַּפֹּ֛רֶת עַל-הָאָרֹ֖ן מִלְמָ֑עְלָה וְאֶל-הָ֣אָרֹ֔ן תִּתֵּן֙ אֶת-הָ֣עֵדֻ֔ת אֲשֶׁ֥ר אֶתֵּ֖ן אֵלֶֽיךָ:  מ  וְנָתַתָּ֧ אֶת-הַכַּפֹּ֛רֶת עַל-הָאָרֹ֖ן מִלְמָ֑עְלָה וְאֶל-הָ֣אָרֹ֔ן תִּתֵּן֙ אֶת-הָ֣עֵדֻ֔ת אֲשֶׁ֥ר אֶתֵּ֖ן אֵלֶֽיךָ:  ת  וְתִתֵּן יָת כַּפֻּרְתָּא עַל אֲרוֹנָא מִלְעֵלָא וּבַאֲרוֹנָא תִּתֵּן יָת סַהֲדוּתָא דִי אֶתֵּן לָךְ:

{כב} וְנוֹעַדְתִּ֣י לְךָ֮ שָׁם֒ וְדִבַּרְתִּ֨י אִתְּךָ֜ מֵעַ֣ל הַכַּפֹּ֗רֶת מִבֵּין֙ שְׁנֵ֣י הַכְּרֻבִ֔ים אֲשֶׁ֖ר עַל-אֲרֹ֣ן הָעֵדֻ֑ת אֵ֣ת כָּל-אֲשֶׁ֧ר אֲצַוֶּ֛ה אוֹתְךָ֖ אֶל-בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וְנוֹעַדְתִּ֣י לְךָ֮ שָׁם֒ וְדִבַּרְתִּ֨י אִתְּךָ֜ מֵעַ֣ל הַכַּפֹּ֗רֶת מִבֵּין֙ שְׁנֵ֣י הַכְּרֻבִ֔ים אֲשֶׁ֖ר עַל-אֲרֹ֣ן הָעֵדֻ֑ת אֵ֣ת כָּל-אֲשֶׁ֧ר אֲצַוֶּ֛ה אוֹתְךָ֖ אֶל-בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וֶאֱזַמֵן מֵימְרִי לָךְ תַּמָן וֶאֱמַלֵל עִמָךְ מֵעִלָוֵי כַפֻּרְתָּא מִבֵּין תְּרֵין כְּרוּבַיָא דִי עַל אֲרוֹנָא דְסַהֲדוּתָא יָת כָּל דִי אֲפַקֵד יָתָךְ לְוָת בְּנֵי יִשְׂרָאֵל: (פ)

{כג} וְעָשִׂ֥יתָ שֻׁלְחָ֖ן עֲצֵ֣י שִׁטִּ֑ים אַמָּתַ֤יִם אָרְכּוֹ֙ וְאַמָּ֣ה רָחְבּ֔וֹ וְאַמָּ֥ה וָחֵ֖צִי קֹמָתֽוֹ:  מ  וְעָשִׂ֥יתָ שֻׁלְחָ֖ן עֲצֵ֣י שִׁטִּ֑ים אַמָּתַ֤יִם אָרְכּוֹ֙ וְאַמָּ֣ה רָחְבּ֔וֹ וְאַמָּ֥ה וָחֵ֖צִי קֹמָתֽוֹ:  ת  וְתַעְבֵּד פָּתוֹרָא דְאָעֵי שִׁטִּין תַּרְתֵּין אַמִין אֻרְכֵּהּ וְאַמְתָא פְתָיֵהּ וְאַמְתָא וּפַלְגָא רוּמֵהּ:

{כד} וְצִפִּיתָ֥ אֹת֖וֹ זָהָ֣ב טָה֑וֹר וְעָשִׂ֥יתָ לּ֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב:  מ  וְצִפִּיתָ֥ אֹת֖וֹ זָהָ֣ב טָה֑וֹר וְעָשִׂ֥יתָ לּ֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב:  ת  וְתַחֲפֵי יָתֵהּ דְהַב דְכֵי וְתַעְבֵּד לֵהּ זֵיר דִדְהַב סְחוֹר סְחוֹר:

{כה} וְעָשִׂ֨יתָ לּ֥וֹ מִסְגֶּ֛רֶת טֹ֖פַח סָבִ֑יב וְעָשִׂ֧יתָ זֵר-זָהָ֛ב לְמִסְגַּרְתּ֖וֹ סָבִֽיב:  מ  וְעָשִׂ֨יתָ לּ֥וֹ מִסְגֶּ֛רֶת טֹ֖פַח סָבִ֑יב וְעָשִׂ֧יתָ זֵר-זָהָ֛ב לְמִסְגַּרְתּ֖וֹ סָבִֽיב:  ת  וְתַעְבֵּד לֵהּ גְדַנְפָא רוּמֵהּ פֻּשְׁכָּא סְחוֹר סְחוֹר וְתַעְבֵּד זֵיר דִדְהַב לִגְדַנְפֵהּ סְחוֹר סְחוֹר:

{כו} וְעָשִׂ֣יתָ לּ֔וֹ אַרְבַּ֖ע טַבְּעֹ֣ת זָהָ֑ב וְנָתַתָּ֙ אֶת-הַטַּבָּעֹ֔ת עַ֚ל אַרְבַּ֣ע הַפֵּאֹ֔ת אֲשֶׁ֖ר לְאַרְבַּ֥ע רַגְלָֽיו:  מ  וְעָשִׂ֣יתָ לּ֔וֹ אַרְבַּ֖ע טַבְּעֹ֣ת זָהָ֑ב וְנָתַתָּ֙ אֶת-הַטַּבָּעֹ֔ת עַ֚ל אַרְבַּ֣ע הַפֵּאֹ֔ת אֲשֶׁ֖ר לְאַרְבַּ֥ע רַגְלָֽיו:  ת  וְתַעְבֵּד לֵהּ אַרְבַּע עִזְקָן דִדְהָב וְתִתֵּן יָת עִזְקָתָא עַל אַרְבַּע זִיוְיָתָא דִי לְאַרְבַּע רַגְלוֹהִי:

{כז} לְעֻמַּת֙ הַמִּסְגֶּ֔רֶת תִּהְיֶ֖יןָ הַטַּבָּעֹ֑ת לְבָתִּ֣ים לְבַדִּ֔ים לָשֵׂ֖את אֶת-הַשֻּׁלְחָֽן:  מ  לְעֻמַּת֙ הַמִּסְגֶּ֔רֶת תִּהְיֶ֖יןָ הַטַּבָּעֹ֑ת לְבָתִּ֣ים לְבַדִּ֔ים לָשֵׂ֖את אֶת-הַשֻּׁלְחָֽן:  ת  לָקֳבֵל גְדַנְפָא יְהֶוְיָן עִזְקָתָא לְאַתְרָא לַאֲרִיחַיָא לְמִטַל יָת פָּתוֹרָא:

{כח} וְעָשִׂ֤יתָ אֶת-הַבַּדִּים֙ עֲצֵ֣י שִׁטִּ֔ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָ֑ב וְנִשָּׂא-בָ֖ם אֶת-הַשֻּׁלְחָֽן:  מ  וְעָשִׂ֤יתָ אֶת-הַבַּדִּים֙ עֲצֵ֣י שִׁטִּ֔ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָ֑ב וְנִשָּׂא-בָ֖ם אֶת-הַשֻּׁלְחָֽן:  ת  וְתַעְבֵּד יָת אֲרִיחַיָא דְאָעֵי שִׁיטִין וְתַחֲפֵי יָתְהוֹן דַהֲבָא וִיהוֹן נָטְלִין בְּהוֹן יָת פָּתוֹרָא:

{כט} וְעָשִׂ֨יתָ קְּעָרֹתָ֜יו וְכַפֹּתָ֗יו וּקְשׂוֹתָיו֙ וּמְנַקִּיֹּתָ֔יו אֲשֶׁ֥ר יֻסַּ֖ךְ בָּהֵ֑ן זָהָ֥ב טָה֖וֹר תַּעֲשֶׂ֥ה אֹתָֽם:  מ  וְעָשִׂ֨יתָ קְּעָרֹתָ֜יו וְכַפֹּתָ֗יו וּקְשׂוֹתָיו֙ וּמְנַקִּיֹּתָ֔יו אֲשֶׁ֥ר יֻסַּ֖ךְ בָּהֵ֑ן זָהָ֥ב טָה֖וֹר תַּעֲשֶׂ֥ה אֹתָֽם:  ת  וְתַעְבֵּד מָגִיסוֹהִי וּבָזִכּוֹהִי וְקַסְוָתֵהּ וּמְכִילָתֵהּ דְיִתְנַסַךְ בְּהֵן דְהַב דְכֵי תַּעְבֵּד יָתְהוֹן:

{ל} וְנָתַתָּ֧ עַֽל-הַשֻּׁלְחָ֛ן לֶ֥חֶם פָּנִ֖ים לְפָנַ֥י תָּמִֽיד:  מ  וְנָתַתָּ֧ עַֽל-הַשֻּׁלְחָ֛ן לֶ֥חֶם פָּנִ֖ים לְפָנַ֥י תָּמִֽיד:  ת  וְתִתֵּן עַל פָּתוֹרָא לְחֵם אַפַּיָא קֳדָמַי תְּדִירָא: (פ)

{לא}  שלישי  וְעָשִׂ֥יתָ מְנֹרַ֖ת זָהָ֣ב טָה֑וֹר מִקְשָׁ֞ה תֵּעָשֶׂ֤ה הַמְּנוֹרָה֙ יְרֵכָ֣הּ וְקָנָ֔הּ גְּבִיעֶ֛יהָ כַּפְתֹּרֶ֥יהָ וּפְרָחֶ֖יהָ מִמֶּ֥נָּה יִהְיֽוּ:  מ  וְעָשִׂ֥יתָ מְנֹרַ֖ת זָהָ֣ב טָה֑וֹר מִקְשָׁ֞ה תֵּעָשֶׂ֤ה הַמְּנוֹרָה֙ יְרֵכָ֣הּ וְקָנָ֔הּ גְּבִיעֶ֛יהָ כַּפְתֹּרֶ֥יהָ וּפְרָחֶ֖יהָ מִמֶּ֥נָּה יִהְיֽוּ:  ת  וְתַעְבֵּד מְנַרְתָּא דִדְהַב דְכֵי נְגִיד תִּתְעֲבֵד מְנַרְתָּא שִׁדַהּ וּקְנַהּ כַּלִידָהָא חֵזוּרָהָא וְשׁוֹשַׁנָהָא מִנַהּ יְהוֹן:

{לב} וְשִׁשָּׁ֣ה קָנִ֔ים יֹצְאִ֖ים מִצִּדֶּ֑יהָ שְׁלֹשָׁ֣ה | קְנֵ֣י מְנֹרָ֗ה מִצִּדָּהּ֙ הָאֶחָ֔ד וּשְׁלֹשָׁה֙ קְנֵ֣י מְנֹרָ֔ה מִצִּדָּ֖הּ הַשֵּׁנִֽי:  מ  וְשִׁשָּׁ֣ה קָנִ֔ים יֹצְאִ֖ים מִצִּדֶּ֑יהָ שְׁלֹשָׁ֣ה | קְנֵ֣י מְנֹרָ֗ה מִצִּדָּהּ֙ הָאֶחָ֔ד וּשְׁלֹשָׁה֙ קְנֵ֣י מְנֹרָ֔ה מִצִּדָּ֖הּ הַשֵּׁנִֽי:  ת  וְשִׁתָּא קְנִין נָפְקִין מִסִטְרָהָא תְּלָתָא קְנֵי מְנַרְתָּא מִסִטְרַהּ חַד וּתְלָתָא קְנֵי מְנַרְתָּא מִסִטְרַהּ תִּנְיָנָא:

{לג} שְׁלֹשָׁ֣ה גְ֠בִעִים מְֽשֻׁקָּדִ֞ים בַּקָּנֶ֣ה הָאֶחָד֮ כַּפְתֹּ֣ר וָפֶרַח֒ וּשְׁלֹשָׁ֣ה גְבִעִ֗ים מְשֻׁקָּדִ֛ים בַּקָּנֶ֥ה הָאֶחָ֖ד כַּפְתֹּ֣ר וָפָ֑רַח כֵּ֚ן לְשֵׁ֣שֶׁת הַקָּנִ֔ים הַיֹּצְאִ֖ים מִן-הַמְּנֹרָֽה:  מ  שְׁלֹשָׁ֣ה גְ֠בִעִים מְֽשֻׁקָּדִ֞ים בַּקָּנֶ֣ה הָאֶחָד֮ כַּפְתֹּ֣ר וָפֶרַח֒ וּשְׁלֹשָׁ֣ה גְבִעִ֗ים מְשֻׁקָּדִ֛ים בַּקָּנֶ֥ה הָאֶחָ֖ד כַּפְתֹּ֣ר וָפָ֑רַח כֵּ֚ן לְשֵׁ֣שֶׁת הַקָּנִ֔ים הַיֹּצְאִ֖ים מִן-הַמְּנֹרָֽה:  ת  תְּלָתָא כַלִידִין מְצַיְרִין בְּקַנְיָא חַד חֵזוּר וְשׁוֹשַׁן וּתְלָתָא כַלִידִין מְצַיְרִין בְּקַנְיָא חַד חֵזוּר וְשׁוֹשָׁן כֵּן לְשִׁתָּא קְנִין דְנָפְקִין מִן מְנַרְתָּא:

{לד} וּבַמְּנֹרָ֖ה אַרְבָּעָ֣ה גְבִעִ֑ים מְשֻׁקָּדִ֔ים כַּפְתֹּרֶ֖יהָ וּפְרָחֶֽיהָ:  מ  וּבַמְּנֹרָ֖ה אַרְבָּעָ֣ה גְבִעִ֑ים מְשֻׁקָּדִ֔ים כַּפְתֹּרֶ֖יהָ וּפְרָחֶֽיהָ:  ת  וּבִמְנַרְתָּא אַרְבְּעָא כַלִידִין מְצַיְרִין חֵזוּרָהָא וְשׁוֹשַׁנָהָא:

{לה} וְכַפְתֹּ֡ר תַּחַת֩ שְׁנֵ֨י הַקָּנִ֜ים מִמֶּ֗נָּה וְכַפְתֹּר֙ תַּ֣חַת שְׁנֵ֤י הַקָּנִים֙ מִמֶּ֔נָּה וְכַפְתֹּ֕ר תַּחַת-שְׁנֵ֥י הַקָּנִ֖ים מִמֶּ֑נָּה לְשֵׁ֙שֶׁת֙ הַקָּנִ֔ים הַיֹּצְאִ֖ים מִן-הַמְּנֹרָֽה:  מ  וְכַפְתֹּ֡ר תַּחַת֩ שְׁנֵ֨י הַקָּנִ֜ים מִמֶּ֗נָּה וְכַפְתֹּר֙ תַּ֣חַת שְׁנֵ֤י הַקָּנִים֙ מִמֶּ֔נָּה וְכַפְתֹּ֕ר תַּחַת-שְׁנֵ֥י הַקָּנִ֖ים מִמֶּ֑נָּה לְשֵׁ֙שֶׁת֙ הַקָּנִ֔ים הַיֹּצְאִ֖ים מִן-הַמְּנֹרָֽה:  ת  וְחֵזוּר תְּחוֹת תְּרֵין קְנִין מִנַהּ וְחֵזוּר תְּחוֹת תְּרֵין קְנִין מִנַהּ וְחֵזוּר תְּחוֹת תְּרֵין קְנִין מִנַהּ לְשִׁתָּא קְנִין דְנָפְקִין מִן מְנַרְתָּא:

{לו} כַּפְתֹּרֵיהֶ֥ם וּקְנֹתָ֖ם מִמֶּ֣נָּה יִהְי֑וּ כֻּלָּ֛הּ מִקְשָׁ֥ה אַחַ֖ת זָהָ֥ב טָהֽוֹר:  מ  כַּפְתֹּרֵיהֶ֥ם וּקְנֹתָ֖ם מִמֶּ֣נָּה יִהְי֑וּ כֻּלָּ֛הּ מִקְשָׁ֥ה אַחַ֖ת זָהָ֥ב טָהֽוֹר:  ת  חֵזוּרֵיהוֹן וּקְנֵיהוֹן מִנַהּ יְהוֹן כֻּלַהּ נְגִידָא חֲדָא דִדְהַב דְכֵי:

{לז} וְעָשִׂ֥יתָ אֶת-נֵרֹתֶ֖יהָ שִׁבְעָ֑ה וְהֶֽעֱלָה֙ אֶת-נֵ֣רֹתֶ֔יהָ וְהֵאִ֖יר עַל-עֵ֥בֶר פָּנֶֽיהָ:  מ  וְעָשִׂ֥יתָ אֶת-נֵרֹתֶ֖יהָ שִׁבְעָ֑ה וְהֶֽעֱלָה֙ אֶת-נֵ֣רֹתֶ֔יהָ וְהֵאִ֖יר עַל-עֵ֥בֶר פָּנֶֽיהָ:  ת  וְתַעְבֵּד יָת בּוֹצִינָהָא שִׁבְעָא וְתַדְלֵק יָת בּוֹצִינָהָא וִיהוֹן מְנַהֲרִין עַל קֳבֵל אַפָּהָא:

{לח} וּמַלְקָחֶ֥יהָ וּמַחְתֹּתֶ֖יהָ זָהָ֥ב טָהֽוֹר:  מ  וּמַלְקָחֶ֥יהָ וּמַחְתֹּתֶ֖יהָ זָהָ֥ב טָהֽוֹר:  ת  וְצִבְיְתָהָא וּמַחְתְּיָתָהָא דִדְהַב דְכֵי:

{לט} כִּכָּ֛ר זָהָ֥ב טָה֖וֹר יַעֲשֶׂ֣ה אֹתָ֑הּ אֵ֥ת כָּל-הַכֵּלִ֖ים הָאֵֽלֶּה:  מ  כִּכָּ֛ר זָהָ֥ב טָה֖וֹר יַעֲשֶׂ֣ה אֹתָ֑הּ אֵ֥ת כָּל-הַכֵּלִ֖ים הָאֵֽלֶּה:  ת  כִּכְּרָא דִדְהַב דְכֵי יַעְבֵּד יָתַהּ יָת כָּל מָנַיָא הָאִלֵין:

{מ} וּרְאֵ֖ה וַעֲשֵׂ֑ה בְּתַ֨בְנִיתָ֔ם אֲשֶׁר-אַתָּ֥ה מָרְאֶ֖ה בָּהָֽר:  מ  וּרְאֵ֖ה וַעֲשֵׂ֑ה בְּתַ֨בְנִיתָ֔ם אֲשֶׁר-אַתָּ֥ה מָרְאֶ֖ה בָּהָֽר:  ת  וַחֲזֵי וַעֲבֵיד כִּדְמוּתְהוֹן דִאַתְּ מִתְחֲזִי בְּטוּרָא: (ס)




שמות פרק-כו

{א} וְאֶת-הַמִּשְׁכָּ֥ן תַּעֲשֶׂ֖ה עֶ֣שֶׂר יְרִיעֹ֑ת שֵׁ֣שׁ מָשְׁזָ֗ר וּתְכֵ֤לֶת וְאַרְגָּמָן֙ וְתֹלַ֣עַת שָׁנִ֔י כְּרֻבִ֛ים מַעֲשֵׂ֥ה חֹשֵׁ֖ב תַּעֲשֶׂ֥ה אֹתָֽם:  מ  וְאֶת-הַמִּשְׁכָּ֥ן תַּעֲשֶׂ֖ה עֶ֣שֶׂר יְרִיעֹ֑ת שֵׁ֣שׁ מָשְׁזָ֗ר וּתְכֵ֤לֶת וְאַרְגָּמָן֙ וְתֹלַ֣עַת שָׁנִ֔י כְּרֻבִ֛ים מַעֲשֵׂ֥ה חֹשֵׁ֖ב תַּעֲשֶׂ֥ה אֹתָֽם:  ת  וְיָת מַשְׁכְּנָא תַּעְבֵּד עֲשַׂר יְרִיעָן דְבוּץ שְׁזִיר וְתִכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי צוּרַת כְּרוּבִין עוֹבַד אֳמָן תַּעְבֵּד יָתְהוֹן:

{ב} אֹ֣רֶךְ | הַיְרִיעָ֣ה הָֽאַחַ֗ת שְׁמֹנֶ֤ה וְעֶשְׂרִים֙ בָּֽאַמָּ֔ה וְרֹ֙חַב֙ אַרְבַּ֣ע בָּאַמָּ֔ה הַיְרִיעָ֖ה הָאֶחָ֑ת מִדָּ֥ה אַחַ֖ת לְכָל-הַיְרִיעֹֽת:  מ  אֹ֣רֶךְ | הַיְרִיעָ֣ה הָֽאַחַ֗ת שְׁמֹנֶ֤ה וְעֶשְׂרִים֙ בָּֽאַמָּ֔ה וְרֹ֙חַב֙ אַרְבַּ֣ע בָּאַמָּ֔ה הַיְרִיעָ֖ה הָאֶחָ֑ת מִדָּ֥ה אַחַ֖ת לְכָל-הַיְרִיעֹֽת:  ת  אֻרְכָּא דִירִיעֲתָא חֲדָא עַשְׂרִין וְתַמְנֵי אַמִין וּפוּתְיָא אַרְבַּע אַמִין דִירִיעֲתָא חֲדָא מְשַׁחְתָּא חֲדָא לְכָל יְרִיעֲתָא:

{ג} חֲמֵ֣שׁ הַיְרִיעֹ֗ת תִּֽהְיֶ֙יןָ֙ חֹֽבְרֹ֔ת אִשָּׁ֖ה אֶל-אֲחֹתָ֑הּ וְחָמֵ֤שׁ יְרִיעֹת֙ חֹֽבְרֹ֔ת אִשָּׁ֖ה אֶל-אֲחֹתָֽהּ:  מ  חֲמֵ֣שׁ הַיְרִיעֹ֗ת תִּֽהְיֶ֙יןָ֙ חֹֽבְרֹ֔ת אִשָּׁ֖ה אֶל-אֲחֹתָ֑הּ וְחָמֵ֤שׁ יְרִיעֹת֙ חֹֽבְרֹ֔ת אִשָּׁ֖ה אֶל-אֲחֹתָֽהּ:  ת  חֲמֵשׁ יְרִיעָן יְהֶוְיָן מְלַפְּפָן חֲדָא עִם חֲדָא וַחֲמֵשׁ יְרִיעָן מְלַפְּפָן חֲדָא עִם חֲדָא:

{ד} וְעָשִׂ֜יתָ לֻֽלְאֹ֣ת תְּכֵ֗לֶת עַ֣ל שְׂפַ֤ת הַיְרִיעָה֙ הָאֶחָ֔ת מִקָּצָ֖ה בַּחֹבָ֑רֶת וְכֵ֤ן תַּעֲשֶׂה֙ בִּשְׂפַ֣ת הַיְרִיעָ֔ה הַקִּ֣יצוֹנָ֔ה בַּמַּחְבֶּ֖רֶת הַשֵּׁנִֽית:  מ  וְעָשִׂ֜יתָ לֻֽלְאֹ֣ת תְּכֵ֗לֶת עַ֣ל שְׂפַ֤ת הַיְרִיעָה֙ הָאֶחָ֔ת מִקָּצָ֖ה בַּחֹבָ֑רֶת וְכֵ֤ן תַּעֲשֶׂה֙ בִּשְׂפַ֣ת הַיְרִיעָ֔ה הַקִּ֣יצוֹנָ֔ה בַּמַּחְבֶּ֖רֶת הַשֵּׁנִֽית:  ת  וְתַעְבֵּד עֲנוּבִין דְתִכְלָא עַל שִׂפְתָא דִירִיעֲתָא חֲדָא מִסִטְרָא בֵּית לוֹפֵי וְכֵן תַּעְבֵּד בְּשִׂפְתָא דִירִיעֲתָא בְּסִטְרָא דְבֵית לוֹפֵי תִּנְיֵתָא:

{ה} חֲמִשִּׁ֣ים לֻֽלָאֹ֗ת תַּעֲשֶׂה֮ בַּיְרִיעָ֣ה הָאֶחָת֒ וַחֲמִשִּׁ֣ים לֻֽלָאֹ֗ת תַּעֲשֶׂה֙ בִּקְצֵ֣ה הַיְרִיעָ֔ה אֲשֶׁ֖ר בַּמַּחְבֶּ֣רֶת הַשֵּׁנִ֑ית מַקְבִּילֹת֙ הַלֻּ֣לָאֹ֔ת אִשָּׁ֖ה אֶל-אֲחֹתָֽהּ:  מ  חֲמִשִּׁ֣ים לֻֽלָאֹ֗ת תַּעֲשֶׂה֮ בַּיְרִיעָ֣ה הָאֶחָת֒ וַחֲמִשִּׁ֣ים לֻֽלָאֹ֗ת תַּעֲשֶׂה֙ בִּקְצֵ֣ה הַיְרִיעָ֔ה אֲשֶׁ֖ר בַּמַּחְבֶּ֣רֶת הַשֵּׁנִ֑ית מַקְבִּילֹת֙ הַלֻּ֣לָאֹ֔ת אִשָּׁ֖ה אֶל-אֲחֹתָֽהּ:  ת  חַמְשִׁין עֲנוּבִין תַּעְבֵּד בִּירִיעֲתָא חֲדָא וְחַמְשִׁין עֲנוּבִין תַּעְבֵּד בְּסִטְרָא דִירִיעֲתָא דְבֵית לוֹפֵי תִּנְיָנָא מְכַוְנָן עֲנוּבַיָא חֲדָא לָקֳבֵל חֲדָא:

{ו} וְעָשִׂ֕יתָ חֲמִשִּׁ֖ים קַרְסֵ֣י זָהָ֑ב וְחִבַּרְתָּ֙ אֶת-הַיְרִיעֹ֜ת אִשָּׁ֤ה אֶל-אֲחֹתָהּ֙ בַּקְּרָסִ֔ים וְהָיָ֥ה הַמִּשְׁכָּ֖ן אֶחָֽד:  מ  וְעָשִׂ֕יתָ חֲמִשִּׁ֖ים קַרְסֵ֣י זָהָ֑ב וְחִבַּרְתָּ֙ אֶת-הַיְרִיעֹ֜ת אִשָּׁ֤ה אֶל-אֲחֹתָהּ֙ בַּקְּרָסִ֔ים וְהָיָ֥ה הַמִּשְׁכָּ֖ן אֶחָֽד:  ת  וְתַעְבֵּד חַמְשִׁין פּוּרְפִין דִדְהָב וּתְלָפֵף יָת יְרִיעֲתָא חֲדָא עִם חֲדָא בְּפּוּרְפַיָא וִיהֵי מַשְׁכְּנָא חָד:

{ז} וְעָשִׂ֙יתָ֙ יְרִיעֹ֣ת עִזִּ֔ים לְאֹ֖הֶל עַל-הַמִּשְׁכָּ֑ן עַשְׁתֵּי-עֶשְׂרֵ֥ה יְרִיעֹ֖ת תַּעֲשֶׂ֥ה אֹתָֽם:  מ  וְעָשִׂ֙יתָ֙ יְרִיעֹ֣ת עִזִּ֔ים לְאֹ֖הֶל עַל-הַמִּשְׁכָּ֑ן עַשְׁתֵּי-עֶשְׂרֵ֥ה יְרִיעֹ֖ת תַּעֲשֶׂ֥ה אֹתָֽם:  ת  וְתַעְבֵּד יְרִיעָן מְעַזֵי לִפְרָסָא עַל מַשְׁכְּנָא חֲדָא עֶסְרֵי יְרִיעָן תַּעְבֵּד יָתְהוֹן:

{ח} אֹ֣רֶךְ | הַיְרִיעָ֣ה הָֽאַחַ֗ת שְׁלֹשִׁים֙ בָּֽאַמָּ֔ה וְרֹ֙חַב֙ אַרְבַּ֣ע בָּאַמָּ֔ה הַיְרִיעָ֖ה הָאֶחָ֑ת מִדָּ֣ה אַחַ֔ת לְעַשְׁתֵּ֥י עֶשְׂרֵ֖ה יְרִיעֹֽת:  מ  אֹ֣רֶךְ | הַיְרִיעָ֣ה הָֽאַחַ֗ת שְׁלֹשִׁים֙ בָּֽאַמָּ֔ה וְרֹ֙חַב֙ אַרְבַּ֣ע בָּאַמָּ֔ה הַיְרִיעָ֖ה הָאֶחָ֑ת מִדָּ֣ה אַחַ֔ת לְעַשְׁתֵּ֥י עֶשְׂרֵ֖ה יְרִיעֹֽת:  ת  אֻרְכָּא דִירִיעֲתָא חֲדָא תְּלָתִין אַמִין וּפוּתְיָא אַרְבַּע אַמִין דִירִיעֲתָא חֲדָא מְשַׁחְתָּא חֲדָא לַחֲדָא עֶסְרֵי יְרִיעָן:

{ט} וְחִבַּרְתָּ֞ אֶת-חֲמֵ֤שׁ הַיְרִיעֹת֙ לְבָ֔ד וְאֶת-שֵׁ֥שׁ הַיְרִיעֹ֖ת לְבָ֑ד וְכָפַלְתָּ֙ אֶת-הַיְרִיעָ֣ה הַשִּׁשִּׁ֔ית אֶל-מ֖וּל פְּנֵ֥י הָאֹֽהֶל:  מ  וְחִבַּרְתָּ֞ אֶת-חֲמֵ֤שׁ הַיְרִיעֹת֙ לְבָ֔ד וְאֶת-שֵׁ֥שׁ הַיְרִיעֹ֖ת לְבָ֑ד וְכָפַלְתָּ֙ אֶת-הַיְרִיעָ֣ה הַשִּׁשִּׁ֔ית אֶל-מ֖וּל פְּנֵ֥י הָאֹֽהֶל:  ת  וּתְלָפֵף יָת חֲמֵשׁ יְרִיעָן לְחוֹד וְיָת שִׁית יְרִיעָן לְחוֹד וְתֵיעוּף יָת יְרִיעֲתָא שְׁתִיתֵיתָא לָקֳבֵל אַפֵּי מַשְׁכְּנָא:

{י} וְעָשִׂ֜יתָ חֲמִשִּׁ֣ים לֻֽלָאֹ֗ת עַ֣ל שְׂפַ֤ת הַיְרִיעָה֙ הָֽאֶחָ֔ת הַקִּיצֹנָ֖ה בַּחֹבָ֑רֶת וַחֲמִשִּׁ֣ים לֻֽלָאֹ֗ת עַ֚ל שְׂפַ֣ת הַיְרִיעָ֔ה הַחֹבֶ֖רֶת הַשֵּׁנִֽית:  מ  וְעָשִׂ֜יתָ חֲמִשִּׁ֣ים לֻֽלָאֹ֗ת עַ֣ל שְׂפַ֤ת הַיְרִיעָה֙ הָֽאֶחָ֔ת הַקִּיצֹנָ֖ה בַּחֹבָ֑רֶת וַחֲמִשִּׁ֣ים לֻֽלָאֹ֗ת עַ֚ל שְׂפַ֣ת הַיְרִיעָ֔ה הַחֹבֶ֖רֶת הַשֵּׁנִֽית:  ת  וְתַעְבֵּד חַמְשִׁין עֲנוּבִין עַל שִׂפְתָא דִירִיעֲתָא חֲדָא בְּסִטְּרָא בֵּית לוֹפֵי וְחַמְשִׁין עֲנוּבִין עַל שִׂפְתָא דִירִיעֲתָא דְבֵית לוֹפֵי תִּנְיֵתָא:

{יא} וְעָשִׂ֛יתָ קַרְסֵ֥י נְחֹ֖שֶׁת חֲמִשִּׁ֑ים וְהֵבֵאתָ֤ אֶת-הַקְּרָסִים֙ בַּלֻּ֣לָאֹ֔ת וְחִבַּרְתָּ֥ אֶת-הָאֹ֖הֶל וְהָיָ֥ה אֶחָֽד:  מ  וְעָשִׂ֛יתָ קַרְסֵ֥י נְחֹ֖שֶׁת חֲמִשִּׁ֑ים וְהֵבֵאתָ֤ אֶת-הַקְּרָסִים֙ בַּלֻּ֣לָאֹ֔ת וְחִבַּרְתָּ֥ אֶת-הָאֹ֖הֶל וְהָיָ֥ה אֶחָֽד:  ת  וְתַעְבֵּד פּוּרְפִין דִנְחָשָׁא חַמְשִׁין וְתָעֵיל יָת פּוּרְפַיָא בַּעֲנוּבַיָא וּתְלָפֵף יָת מַשְׁכְּנָא וִיהֵי חָד:

{יב} וְסֶ֙רַח֙ הָעֹדֵ֔ף בִּירִיעֹ֖ת הָאֹ֑הֶל חֲצִ֤י הַיְרִיעָה֙ הָעֹדֶ֔פֶת תִּסְרַ֕ח עַ֖ל אֲחֹרֵ֥י הַמִּשְׁכָּֽן:  מ  וְסֶ֙רַח֙ הָעֹדֵ֔ף בִּירִיעֹ֖ת הָאֹ֑הֶל חֲצִ֤י הַיְרִיעָה֙ הָעֹדֶ֔פֶת תִּסְרַ֕ח עַ֖ל אֲחֹרֵ֥י הַמִּשְׁכָּֽן:  ת  וְסִרְחָא דְיַתִּירָא בִּירִיעַת מַשְׁכְּנָא פַּלְגוּת יְרִיעֲתָא דְיַתִּירָא תִּסְרַח עַל אֲחוֹרֵי מַשְׁכְּנָא:

{יג} וְהָאַמָּ֨ה מִזֶּ֜ה וְהָאַמָּ֤ה מִזֶּה֙ בָּעֹדֵ֔ף בְּאֹ֖רֶךְ יְרִיעֹ֣ת הָאֹ֑הֶל יִהְיֶ֨ה סָר֜וּחַ עַל-צִדֵּ֧י הַמִּשְׁכָּ֛ן מִזֶּ֥ה וּמִזֶּ֖ה לְכַסֹּתֽוֹ:  מ  וְהָאַמָּ֨ה מִזֶּ֜ה וְהָאַמָּ֤ה מִזֶּה֙ בָּעֹדֵ֔ף בְּאֹ֖רֶךְ יְרִיעֹ֣ת הָאֹ֑הֶל יִהְיֶ֨ה סָר֜וּחַ עַל-צִדֵּ֧י הַמִּשְׁכָּ֛ן מִזֶּ֥ה וּמִזֶּ֖ה לְכַסֹּתֽוֹ:  ת  וְאַמְתָא מִכָּא וְאַמְתָא מִכָּא בִּדְיַתִּירָא בְּאֻרְכָּא יְרִיעֲתָא דְמַשְׁכְּנָא יְהֵי סְרִיחַ עַל סִטְרֵי מַשְׁכְּנָא מִכָּא וּמִכָּא לְכַסָיוּתֵהּ:

{יד} וְעָשִׂ֤יתָ מִכְסֶה֙ לָאֹ֔הֶל עֹרֹ֥ת אֵילִ֖ם מְאָדָּמִ֑ים וּמִכְסֵ֛ה עֹרֹ֥ת תְּחָשִׁ֖ים מִלְמָֽעְלָה:  מ  וְעָשִׂ֤יתָ מִכְסֶה֙ לָאֹ֔הֶל עֹרֹ֥ת אֵילִ֖ם מְאָדָּמִ֑ים וּמִכְסֵ֛ה עֹרֹ֥ת תְּחָשִׁ֖ים מִלְמָֽעְלָה:  ת  וְתַעְבֵּד חוֹפָאָה לִפְרָסָא מַשְׁכֵי דְדִכְרֵי מְסַמְקֵי וְחוֹפָאָה מַשְׁכֵי דְסַסְגוֹנָא מִלְעֵלָא: (פ)

{טו}  רביעי  וְעָשִׂ֥יתָ אֶת-הַקְּרָשִׁ֖ים לַמִּשְׁכָּ֑ן עֲצֵ֥י שִׁטִּ֖ים עֹמְדִֽים:  מ  וְעָשִׂ֥יתָ אֶת-הַקְּרָשִׁ֖ים לַמִּשְׁכָּ֑ן עֲצֵ֥י שִׁטִּ֖ים עֹמְדִֽים:  ת  וְתַעְבֵּד יָת דַפַּיָא לְמַשְׁכְּנָא דְאָעֵי שִטִין קָיְמִין:

{טז} עֶ֥שֶׂר אַמּ֖וֹת אֹ֣רֶךְ הַקָּ֑רֶשׁ וְאַמָּה֙ וַחֲצִ֣י הָֽאַמָּ֔ה רֹ֖חַב הַקֶּ֥רֶשׁ הָאֶחָֽד:  מ  עֶ֥שֶׂר אַמּ֖וֹת אֹ֣רֶךְ הַקָּ֑רֶשׁ וְאַמָּה֙ וַחֲצִ֣י הָֽאַמָּ֔ה רֹ֖חַב הַקֶּ֥רֶשׁ הָאֶחָֽד:  ת  עֲשַׂר אַמִין אֻרְכָּא דְדַפָּא וְאַמְתָא וּפַלְגוּת אַמְתָא פּוּתְיָא דְדַפָּא חָד:

{יז} שְׁתֵּ֣י יָד֗וֹת לַקֶּ֙רֶשׁ֙ הָאֶחָ֔ד מְשֻׁלָּבֹ֔ת אִשָּׁ֖ה אֶל-אֲחֹתָ֑הּ כֵּ֣ן תַּעֲשֶׂ֔ה לְכֹ֖ל קַרְשֵׁ֥י הַמִּשְׁכָּֽן:  מ  שְׁתֵּ֣י יָד֗וֹת לַקֶּ֙רֶשׁ֙ הָאֶחָ֔ד מְשֻׁלָּבֹ֔ת אִשָּׁ֖ה אֶל-אֲחֹתָ֑הּ כֵּ֣ן תַּעֲשֶׂ֔ה לְכֹ֖ל קַרְשֵׁ֥י הַמִּשְׁכָּֽן:  ת  תַּרְתֵּין צִירִין לְדַפָּא חַד מְשַׁלְבִין חֲדָא לָקֳבֵל חֲדָא כֵּן תַּעְבֵּד לְכֹל דַפֵּי מַשְׁכְּנָא:

{יח} וְעָשִׂ֥יתָ אֶת-הַקְּרָשִׁ֖ים לַמִּשְׁכָּ֑ן עֶשְׂרִ֣ים קֶ֔רֶשׁ לִפְאַ֖ת נֶ֥גְבָּה תֵימָֽנָה:  מ  וְעָשִׂ֥יתָ אֶת-הַקְּרָשִׁ֖ים לַמִּשְׁכָּ֑ן עֶשְׂרִ֣ים קֶ֔רֶשׁ לִפְאַ֖ת נֶ֥גְבָּה תֵימָֽנָה:  ת  וְתַעְבֵּד יָת דַפַּיָא לְמַשְׁכְּנָא עַסְרִין דַפִּין לְרוּחַ עֵבַר דְרוֹמָא:

{יט} וְאַרְבָּעִים֙ אַדְנֵי-כֶ֔סֶף תַּעֲשֶׂ֕ה תַּ֖חַת עֶשְׂרִ֣ים הַקָּ֑רֶשׁ שְׁנֵ֨י אֲדָנִ֜ים תַּֽחַת-הַקֶּ֤רֶשׁ הָאֶחָד֙ לִשְׁתֵּ֣י יְדֹתָ֔יו וּשְׁנֵ֧י אֲדָנִ֛ים תַּֽחַת-הַקֶּ֥רֶשׁ הָאֶחָ֖ד לִשְׁתֵּ֥י יְדֹתָֽיו:  מ  וְאַרְבָּעִים֙ אַדְנֵי-כֶ֔סֶף תַּעֲשֶׂ֕ה תַּ֖חַת עֶשְׂרִ֣ים הַקָּ֑רֶשׁ שְׁנֵ֨י אֲדָנִ֜ים תַּֽחַת-הַקֶּ֤רֶשׁ הָאֶחָד֙ לִשְׁתֵּ֣י יְדֹתָ֔יו וּשְׁנֵ֧י אֲדָנִ֛ים תַּֽחַת-הַקֶּ֥רֶשׁ הָאֶחָ֖ד לִשְׁתֵּ֥י יְדֹתָֽיו:  ת  וְאַרְבְּעִין סַמְכִין דִכְסַף תַּעְבֵּד תְּחוֹת עַסְרִין דַפִּין תְּרֵין סַמְכִין תְּחוֹת דַפָּא חַד לִתְרֵין צִירוֹהִי וּתְרֵין סַמְכִין תְּחוֹת דַפָּא חַד לִתְרֵין צִירוֹהִי:

{כ} וּלְצֶ֧לַע הַמִּשְׁכָּ֛ן הַשֵּׁנִ֖ית לִפְאַ֣ת צָפ֑וֹן עֶשְׂרִ֖ים קָֽרֶשׁ:  מ  וּלְצֶ֧לַע הַמִּשְׁכָּ֛ן הַשֵּׁנִ֖ית לִפְאַ֣ת צָפ֑וֹן עֶשְׂרִ֖ים קָֽרֶשׁ:  ת  וְלִסְטַר מַשְׁכְּנָא תִּנְיֵתָא לְרוּחַ צִפּוּנָא עֶסְרִין דַפִּין:

{כא} וְאַרְבָּעִ֥ים אַדְנֵיהֶ֖ם כָּ֑סֶף שְׁנֵ֣י אֲדָנִ֗ים תַּ֚חַת הַקֶּ֣רֶשׁ הָֽאֶחָ֔ד וּשְׁנֵ֣י אֲדָנִ֔ים תַּ֖חַת הַקֶּ֥רֶשׁ הָאֶחָֽד:  מ  וְאַרְבָּעִ֥ים אַדְנֵיהֶ֖ם כָּ֑סֶף שְׁנֵ֣י אֲדָנִ֗ים תַּ֚חַת הַקֶּ֣רֶשׁ הָֽאֶחָ֔ד וּשְׁנֵ֣י אֲדָנִ֔ים תַּ֖חַת הַקֶּ֥רֶשׁ הָאֶחָֽד:  ת  וְאַרְבְּעִין סַמְכֵיהוֹן דִכְסַף תְּרֵין סַמְכִין תְּחוֹת דַפָּא חַד וּתְרֵין סַמְכִין תְּחוֹת דַפָּא חָד:

{כב} וּֽלְיַרְכְּתֵ֥י הַמִּשְׁכָּ֖ן יָ֑מָּה תַּעֲשֶׂ֖ה שִׁשָּׁ֥ה קְרָשִֽׁים:  מ  וּֽלְיַרְכְּתֵ֥י הַמִּשְׁכָּ֖ן יָ֑מָּה תַּעֲשֶׂ֖ה שִׁשָּׁ֥ה קְרָשִֽׁים:  ת  וְלִסְיָפֵי מַשְׁכְּנָא מַעַרְבָא תַּעְבֵּד שִׁתָּא דַפִּין:

{כג} וּשְׁנֵ֤י קְרָשִׁים֙ תַּעֲשֶׂ֔ה לִמְקֻצְעֹ֖ת הַמִּשְׁכָּ֑ן בַּיַּרְכָתָֽיִם:  מ  וּשְׁנֵ֤י קְרָשִׁים֙ תַּעֲשֶׂ֔ה לִמְקֻצְעֹ֖ת הַמִּשְׁכָּ֑ן בַּיַּרְכָתָֽיִם:  ת  וּתְרֵין דַפִּין תַּעְבֵּד לְזָוְיַת מַשְׁכְּנָא בְּסוֹפֵיהוֹן:

{כד} וְיִֽהְי֣וּ תֹֽאֲמִים֮ מִלְּמַטָּה֒ וְיַחְדָּ֗ו יִהְי֤וּ תַמִּים֙ עַל-רֹאשׁ֔וֹ אֶל-הַטַּבַּ֖עַת הָאֶחָ֑ת כֵּ֚ן יִהְיֶ֣ה לִשְׁנֵיהֶ֔ם לִשְׁנֵ֥י הַמִּקְצֹעֹ֖ת יִהְיֽוּ:  מ  וְיִֽהְי֣וּ תֹֽאֲמִים֮ מִלְּמַטָּה֒ וְיַחְדָּ֗ו יִהְי֤וּ תַמִּים֙ עַל-רֹאשׁ֔וֹ אֶל-הַטַּבַּ֖עַת הָאֶחָ֑ת כֵּ֚ן יִהְיֶ֣ה לִשְׁנֵיהֶ֔ם לִשְׁנֵ֥י הַמִּקְצֹעֹ֖ת יִהְיֽוּ:  ת  וִיהוֹן מְכַוְנִין מִלְרַע וְכַחֲדָא יְהוֹן מְכַוְנִין עַל רֵישֵׁהּ לְעִזְקְתָא חֲדָא כֵּן יְהֵי לְתַרְוֵיהוֹן לִתְרֵין זָוְיָן יְהוֹן:

{כה} וְהָיוּ֙ שְׁמֹנָ֣ה קְרָשִׁ֔ים וְאַדְנֵיהֶ֣ם כֶּ֔סֶף שִׁשָּׁ֥ה עָשָׂ֖ר אֲדָנִ֑ים שְׁנֵ֣י אֲדָנִ֗ים תַּ֚חַת הַקֶּ֣רֶשׁ הָאֶחָ֔ד וּשְׁנֵ֣י אֲדָנִ֔ים תַּ֖חַת הַקֶּ֥רֶשׁ הָאֶחָֽד:  מ  וְהָיוּ֙ שְׁמֹנָ֣ה קְרָשִׁ֔ים וְאַדְנֵיהֶ֣ם כֶּ֔סֶף שִׁשָּׁ֥ה עָשָׂ֖ר אֲדָנִ֑ים שְׁנֵ֣י אֲדָנִ֗ים תַּ֚חַת הַקֶּ֣רֶשׁ הָאֶחָ֔ד וּשְׁנֵ֣י אֲדָנִ֔ים תַּ֖חַת הַקֶּ֥רֶשׁ הָאֶחָֽד:  ת  וִיהוֹן תְּמַנְיָא דַפִּין וְסַמְכֵיהוֹן דִכְסַף שִׁתָּא עֲשַׂר סַמְכִין תְּרֵין סַמְכִין תְּחוֹת דַפָּא חַד וּתְרֵין סַמְכִין תְּחוֹת דַפָּא חָד:

{כו} וְעָשִׂ֥יתָ בְרִיחִ֖ם עֲצֵ֣י שִׁטִּ֑ים חֲמִשָּׁ֕ה לְקַרְשֵׁ֥י צֶֽלַע-הַמִּשְׁכָּ֖ן הָאֶחָֽד:  מ  וְעָשִׂ֥יתָ בְרִיחִ֖ם עֲצֵ֣י שִׁטִּ֑ים חֲמִשָּׁ֕ה לְקַרְשֵׁ֥י צֶֽלַע-הַמִּשְׁכָּ֖ן הָאֶחָֽד:  ת  וְתַעְבֵּד עַבְּרֵי דְאָעֵי שִׁטִין חַמְשָׁא לְדַפֵּי סְטַּר מַשְׁכְּנָא חָד:

{כז} וַחֲמִשָּׁ֣ה בְרִיחִ֔ם לְקַרְשֵׁ֥י צֶֽלַע-הַמִּשְׁכָּ֖ן הַשֵּׁנִ֑ית וַחֲמִשָּׁ֣ה בְרִיחִ֗ם לְקַרְשֵׁי֙ צֶ֣לַע הַמִּשְׁכָּ֔ן לַיַּרְכָתַ֖יִם יָֽמָּה:  מ  וַחֲמִשָּׁ֣ה בְרִיחִ֔ם לְקַרְשֵׁ֥י צֶֽלַע-הַמִּשְׁכָּ֖ן הַשֵּׁנִ֑ית וַחֲמִשָּׁ֣ה בְרִיחִ֗ם לְקַרְשֵׁי֙ צֶ֣לַע הַמִּשְׁכָּ֔ן לַיַּרְכָתַ֖יִם יָֽמָּה:  ת  וְחַמְשָׁא עַבְּרִין לְדַפֵּי סְטַר מַשְׁכְּנָא תִּנְיֵתָא וְחַמְשָׁא עַבְּרִין לְדַפֵּי סְטַּר מַשְׁכְּנָא לְסוֹפֵיהוֹן מַעַרְבָא:

{כח} וְהַבְּרִ֥יחַ הַתִּיכֹ֖ן בְּת֣וֹךְ הַקְּרָשִׁ֑ים מַבְרִ֕חַ מִן-הַקָּצֶ֖ה אֶל-הַקָּצֶֽה:  מ  וְהַבְּרִ֥יחַ הַתִּיכֹ֖ן בְּת֣וֹךְ הַקְּרָשִׁ֑ים מַבְרִ֕חַ מִן-הַקָּצֶ֖ה אֶל-הַקָּצֶֽה:  ת  וְעַבְּרָא מְצִיעָאָה בְּגוֹ דַפַּיָא מַעְבַר מִן סְיָפֵי לִסְיָפֵי:

{כט} וְֽאֶת-הַקְּרָשִׁ֞ים תְּצַפֶּ֣ה זָהָ֗ב וְאֶת-טַבְּעֹֽתֵיהֶם֙ תַּעֲשֶׂ֣ה זָהָ֔ב בָּתִּ֖ים לַבְּרִיחִ֑ם וְצִפִּיתָ֥ אֶת-הַבְּרִיחִ֖ם זָהָֽב:  מ  וְֽאֶת-הַקְּרָשִׁ֞ים תְּצַפֶּ֣ה זָהָ֗ב וְאֶת-טַבְּעֹֽתֵיהֶם֙ תַּעֲשֶׂ֣ה זָהָ֔ב בָּתִּ֖ים לַבְּרִיחִ֑ם וְצִפִּיתָ֥ אֶת-הַבְּרִיחִ֖ם זָהָֽב:  ת  וְיָת דַפַּיָא תַּחְפֵּי דַהֲבָא וְיָת עִזְקָתְהוֹן תַּעְבֵּד דַהֲבָא אַתְרָא לְעַבְרַיָא וְתַחְפֵּי יָת עַבְּרַיָא דַהֲבָא:

{ל} וַהֲקֵמֹתָ֖ אֶת-הַמִּשְׁכָּ֑ן כְּמִ֨שְׁפָּט֔וֹ אֲשֶׁ֥ר הָרְאֵ֖יתָ בָּהָֽר:  מ  וַהֲקֵמֹתָ֖ אֶת-הַמִּשְׁכָּ֑ן כְּמִ֨שְׁפָּט֔וֹ אֲשֶׁ֥ר הָרְאֵ֖יתָ בָּהָֽר:  ת  וּתְקִים יָת מַשְׁכְּנָא כְּהִלְכְתֵהּ דִי אִתְחֲזֵיתָא בְּטוּרָא: (ס)

{לא}  חמישי  וְעָשִׂ֣יתָ פָרֹ֗כֶת תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מָשְׁזָ֑ר מַעֲשֵׂ֥ה חֹשֵׁ֛ב יַעֲשֶׂ֥ה אֹתָ֖הּ כְּרֻבִֽים:  מ  וְעָשִׂ֣יתָ פָרֹ֗כֶת תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מָשְׁזָ֑ר מַעֲשֵׂ֥ה חֹשֵׁ֛ב יַעֲשֶׂ֥ה אֹתָ֖הּ כְּרֻבִֽים:  ת  וְתַעְבֵּד פָּרֻכְתָּא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר עוֹבַד אֳמָן יַעְבֵּד יָתַהּ צוּרַת כְּרוּבִין:

{לב} וְנָתַתָּ֣ה אֹתָ֗הּ עַל-אַרְבָּעָה֙ עַמּוּדֵ֣י שִׁטִּ֔ים מְצֻפִּ֣ים זָהָ֔ב וָוֵיהֶ֖ם זָהָ֑ב עַל-אַרְבָּעָ֖ה אַדְנֵי-כָֽסֶף:  מ  וְנָתַתָּ֣ה אֹתָ֗הּ עַל-אַרְבָּעָה֙ עַמּוּדֵ֣י שִׁטִּ֔ים מְצֻפִּ֣ים זָהָ֔ב וָוֵיהֶ֖ם זָהָ֑ב עַל-אַרְבָּעָ֖ה אַדְנֵי-כָֽסֶף:  ת  וְתִתֵּן יָתַהּ עַל אַרְבְּעָא עַמוּדֵי שִׁטִין מְחַפְּיָן דַהֲבָא וָוֵיהוֹן דַהֲבָא עַל אַרְבְּעָא סַמְכֵי דִכְסָף:

{לג} וְנָתַתָּ֣ה אֶת-הַפָּרֹכֶת֮ תַּ֣חַת הַקְּרָסִים֒ וְהֵבֵאתָ֥ שָׁ֙מָּה֙ מִבֵּ֣ית לַפָּרֹ֔כֶת אֵ֖ת אֲר֣וֹן הָעֵד֑וּת וְהִבְדִּילָ֤ה הַפָּרֹ֙כֶת֙ לָכֶ֔ם בֵּ֣ין הַקֹּ֔דֶשׁ וּבֵ֖ין קֹ֥דֶשׁ הַקֳּדָשִֽׁים:  מ  וְנָתַתָּ֣ה אֶת-הַפָּרֹכֶת֮ תַּ֣חַת הַקְּרָסִים֒ וְהֵבֵאתָ֥ שָׁ֙מָּה֙ מִבֵּ֣ית לַפָּרֹ֔כֶת אֵ֖ת אֲר֣וֹן הָעֵד֑וּת וְהִבְדִּילָ֤ה הַפָּרֹ֙כֶת֙ לָכֶ֔ם בֵּ֣ין הַקֹּ֔דֶשׁ וּבֵ֖ין קֹ֥דֶשׁ הַקֳּדָשִֽׁים:  ת  וְתִתֵּן יָת פָּרֻכְתָּא תְּחוֹת פּוּרְפַיָא וְתָעֵיל תַּמָן מִגָּו לְפָרֻכְתָּא יָת אֲרוֹנָא דְסַהֲדוּתָא וְתַפְרֵשׁ פָּרֻכְתָּא לְכוֹן בֵּין קוּדְשָׁא וּבֵין קֹדֶשׁ קוּדְשַׁיָא:

{לד} וְנָתַתָּ֙ אֶת-הַכַּפֹּ֔רֶת עַ֖ל אֲר֣וֹן הָעֵדֻ֑ת בְּקֹ֖דֶשׁ הַקֳּדָשִֽׁים:  מ  וְנָתַתָּ֙ אֶת-הַכַּפֹּ֔רֶת עַ֖ל אֲר֣וֹן הָעֵדֻ֑ת בְּקֹ֖דֶשׁ הַקֳּדָשִֽׁים:  ת  וְתִתֵּן יָת כַּפֻּרְתָּא עַל אֲרוֹנָא דְסַהֲדוּתָא בְּקֹדֶשׁ קוּדְשַׁיָא:

{לה} וְשַׂמְתָּ֤ אֶת-הַשֻּׁלְחָן֙ מִח֣וּץ לַפָּרֹ֔כֶת וְאֶת-הַמְּנֹרָה֙ נֹ֣כַח הַשֻּׁלְחָ֔ן עַ֛ל צֶ֥לַע הַמִּשְׁכָּ֖ן תֵּימָ֑נָה וְהַ֨שֻּׁלְחָ֔ן תִּתֵּ֖ן עַל-צֶ֥לַע צָפֽוֹן:  מ  וְשַׂמְתָּ֤ אֶת-הַשֻּׁלְחָן֙ מִח֣וּץ לַפָּרֹ֔כֶת וְאֶת-הַמְּנֹרָה֙ נֹ֣כַח הַשֻּׁלְחָ֔ן עַ֛ל צֶ֥לַע הַמִּשְׁכָּ֖ן תֵּימָ֑נָה וְהַ֨שֻּׁלְחָ֔ן תִּתֵּ֖ן עַל-צֶ֥לַע צָפֽוֹן:  ת  וּתְשַׁוִי יָת פָּתוֹרָא מִבָּרָא לְפָרֻכְתָּא וְיָת מְנַרְתָּא לָקֳבֵל פָּתוֹרָא עַל צִדָּא דְמַשְּׁכְּנָא דָרוֹמָא וּפָתוֹרָא תִתֵּן עַל סְטַר צִפּוּנָא:

{לו} וְעָשִׂ֤יתָ מָסָךְ֙ לְפֶ֣תַח הָאֹ֔הֶל תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מָשְׁזָ֑ר מַעֲשֵׂ֖ה רֹקֵֽם:  מ  וְעָשִׂ֤יתָ מָסָךְ֙ לְפֶ֣תַח הָאֹ֔הֶל תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מָשְׁזָ֑ר מַעֲשֵׂ֖ה רֹקֵֽם:  ת  וְתַעְבֵּד פְּרָסָא לִתְרַע מַשְׁכְּנָא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר עוֹבַד צַיָר:

{לז} וְעָשִׂ֣יתָ לַמָּסָ֗ךְ חֲמִשָּׁה֙ עַמּוּדֵ֣י שִׁטִּ֔ים וְצִפִּיתָ֤ אֹתָם֙ זָהָ֔ב וָוֵיהֶ֖ם זָהָ֑ב וְיָצַקְתָּ֣ לָהֶ֔ם חֲמִשָּׁ֖ה אַדְנֵ֥י נְחֹֽשֶׁת:  מ  וְעָשִׂ֣יתָ לַמָּסָ֗ךְ חֲמִשָּׁה֙ עַמּוּדֵ֣י שִׁטִּ֔ים וְצִפִּיתָ֤ אֹתָם֙ זָהָ֔ב וָוֵיהֶ֖ם זָהָ֑ב וְיָצַקְתָּ֣ לָהֶ֔ם חֲמִשָּׁ֖ה אַדְנֵ֥י נְחֹֽשֶׁת:  ת  וְתַעְבֵּד לִפְרָסָא חַמְשָׁא עַמוּדֵי שִׁטִין וְתַחֲפֵי יָתְהוֹן דַהֲבָא וָוֵיהוֹן דַהֲבָא וְתַתִּיךְ לְהוֹן חַמְשָׁא סַמְכֵי נְחָשָׁא: (ס)




שמות פרק-כז

{א}  שישי  וְעָשִׂ֥יתָ אֶת-הַמִּזְבֵּ֖חַ עֲצֵ֣י שִׁטִּ֑ים חָמֵשׁ֩ אַמּ֨וֹת אֹ֜רֶךְ וְחָמֵ֧שׁ אַמּ֣וֹת רֹ֗חַב רָב֤וּעַ יִהְיֶה֙ הַמִּזְבֵּ֔חַ וְשָׁלֹ֥שׁ אַמּ֖וֹת קֹמָתֽוֹ:  מ  וְעָשִׂ֥יתָ אֶת-הַמִּזְבֵּ֖חַ עֲצֵ֣י שִׁטִּ֑ים חָמֵשׁ֩ אַמּ֨וֹת אֹ֜רֶךְ וְחָמֵ֧שׁ אַמּ֣וֹת רֹ֗חַב רָב֤וּעַ יִהְיֶה֙ הַמִּזְבֵּ֔חַ וְשָׁלֹ֥שׁ אַמּ֖וֹת קֹמָתֽוֹ:  ת  וְתַעְבֵּד יָת מַדְבְּחָא דְאָעֵי שִׁטִין חֲמֵשׁ אַמִין אֻרְכָּא וַחֲמֵשׁ אַמִין פּוּתְיָא מְרַבַּע יְהֵי מַדְבְּחָא וּתְלַת אַמִין רוּמֵהּ:

{ב} וְעָשִׂ֣יתָ קַרְנֹתָ֗יו עַ֚ל אַרְבַּ֣ע פִּנֹּתָ֔יו מִמֶּ֖נּוּ תִּהְיֶ֣יןָ קַרְנֹתָ֑יו וְצִפִּיתָ֥ אֹת֖וֹ נְחֹֽשֶׁת:  מ  וְעָשִׂ֣יתָ קַרְנֹתָ֗יו עַ֚ל אַרְבַּ֣ע פִּנֹּתָ֔יו מִמֶּ֖נּוּ תִּהְיֶ֣יןָ קַרְנֹתָ֑יו וְצִפִּיתָ֥ אֹת֖וֹ נְחֹֽשֶׁת:  ת  וְתַעְבֵּד קַרְנוֹהִי עַל אַרְבַּע זִוְיָתֵהּ מִנֵהּ תְּהֶוְיָן קַרְנוֹהִי וְתַחֲפֵי יָתֵהּ נְחָשָׁא:

{ג} וְעָשִׂ֤יתָ סִּֽירֹתָיו֙ לְדַשְּׁנ֔וֹ וְיָעָיו֙ וּמִזְרְקֹתָ֔יו וּמִזְלְגֹתָ֖יו וּמַחְתֹּתָ֑יו לְכָל-כֵּלָ֖יו תַּעֲשֶׂ֥ה נְחֹֽשֶׁת:  מ  וְעָשִׂ֤יתָ סִּֽירֹתָיו֙ לְדַשְּׁנ֔וֹ וְיָעָיו֙ וּמִזְרְקֹתָ֔יו וּמִזְלְגֹתָ֖יו וּמַחְתֹּתָ֑יו לְכָל-כֵּלָ֖יו תַּעֲשֶׂ֥ה נְחֹֽשֶׁת:  ת  וְתַעְבֵּד פְּסַכְתְּרָוָתֵהּ לְמִסְפֵי קִטְּמֵיהּ וּמַגְרוֹפְיָתֵהּ וּמִזְרְקָתֵהּ וְצִנוּרְיָתֵהּ וּמַחְתְּיָתֵהּ לְכָל מָנוֹהִי תַּעְבֵּד נְחָשָׁא:

{ד} וְעָשִׂ֤יתָ לּוֹ֙ מִכְבָּ֔ר מַעֲשֵׂ֖ה רֶ֣שֶׁת נְחֹ֑שֶׁת וְעָשִׂ֣יתָ עַל-הָרֶ֗שֶׁת אַרְבַּע֙ טַבְּעֹ֣ת נְחֹ֔שֶׁת עַ֖ל אַרְבַּ֥ע קְצוֹתָֽיו:  מ  וְעָשִׂ֤יתָ לּוֹ֙ מִכְבָּ֔ר מַעֲשֵׂ֖ה רֶ֣שֶׁת נְחֹ֑שֶׁת וְעָשִׂ֣יתָ עַל-הָרֶ֗שֶׁת אַרְבַּע֙ טַבְּעֹ֣ת נְחֹ֔שֶׁת עַ֖ל אַרְבַּ֥ע קְצוֹתָֽיו:  ת  וְתַעְבֵּד לֵהּ סְרָדָא עוֹבַד מְצַדְתָּא דִנְחָשָׁא וְתַעְבֵּד עַל מְצַדְתָּא אַרְבַּע עִזְקָן דִנְחָשָׁא עַל אַרְבַּע סִטְּרוֹהִי:

{ה} וְנָתַתָּ֣ה אֹתָ֗הּ תַּ֛חַת כַּרְכֹּ֥ב הַמִּזְבֵּ֖חַ מִלְּמָ֑טָּה וְהָיְתָ֣ה הָרֶ֔שֶׁת עַ֖ד חֲצִ֥י הַמִּזְבֵּֽחַ:  מ  וְנָתַתָּ֣ה אֹתָ֗הּ תַּ֛חַת כַּרְכֹּ֥ב הַמִּזְבֵּ֖חַ מִלְּמָ֑טָּה וְהָיְתָ֣ה הָרֶ֔שֶׁת עַ֖ד חֲצִ֥י הַמִּזְבֵּֽחַ:  ת  וְתִתֵּן יָתַהּ תְּחוֹת סוֹבָבֵי מַדְבְּחָא מִלְרָע וּתְהֵי מְצַדְתָּא עַד פַּלְגוּת מַדְבְּחָא:

{ז} וְהוּבָ֥א אֶת-בַּדָּ֖יו בַּטַּבָּעֹ֑ת וְהָי֣וּ הַבַּדִּ֗ים עַל-שְׁתֵּ֛י צַלְעֹ֥ת הַמִּזְבֵּ֖חַ בִּשְׂאֵ֥ת אֹתֽוֹ:  מ  וְהוּבָ֥א אֶת-בַּדָּ֖יו בַּטַּבָּעֹ֑ת וְהָי֣וּ הַבַּדִּ֗ים עַל-שְׁתֵּ֛י צַלְעֹ֥ת הַמִּזְבֵּ֖חַ בִּשְׂאֵ֥ת אֹתֽוֹ:  ת  וְיָעֵל יָת אֲרִיחוֹהִי בְּעִזְקָתָא וִיהוֹן אֲרִיחַיָא עַל תְּרֵין סִטְרֵי מַדְבְּחָא בְּמִטַל יָתֵהּ:

{ח} נְב֥וּב לֻחֹ֖ת תַּעֲשֶׂ֣ה אֹת֑וֹ כַּאֲשֶׁ֨ר הֶרְאָ֥ה אֹתְךָ֛ בָּהָ֖ר כֵּ֥ן יַעֲשֽׂוּ:  מ  נְב֥וּב לֻחֹ֖ת תַּעֲשֶׂ֣ה אֹת֑וֹ כַּאֲשֶׁ֨ר הֶרְאָ֥ה אֹתְךָ֛ בָּהָ֖ר כֵּ֥ן יַעֲשֽׂוּ:  ת  חֲלִיל לוּחֵי תַּעְבֵּד יָתֵהּ כְּמָא דִי אַחֲזִי יָתָךְ בְּטוּרָא כֵּן יַעְבְּדוּן: (ס)
