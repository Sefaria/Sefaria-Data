הלכות ביצים



סימן פו - סימני ביצים ודיניהם, ובו י' סעיפים


א סִימָנֵי בֵּיצִים לְהַכִּיר שֶׁהֵם מֵעוֹף טָהוֹר; אִם שְׁנֵי רָאשֶׁיהָ כַּדִּין אוֹ חַדִּין, אוֹ שֶׁהָיָה חֶלְמוֹן מִבַּחוּץ וְחֶלְבּוֹן מִבִּפְנִים, בְּיָדוּעַ שֶׁהוּא בֵּיצַת עוֹף טָמֵא. וַאֲפִלּוּ אִם יֹאמַר (א) הַמּוֹכֵר שֶׁהֵם מֵעוֹף טָהוֹר, אֵין סוֹמְכִין עָלָיו. וְאִם רֹאשׁוֹ אֶחָד כַּד וְעָגֹל וְרֹאשׁוֹ הַשֵּׁנִי חַד וְעָגֹל וְחֶלְבּוֹן מִבַּחוּץ וְחֶלְמוֹן מִבִּפְנִים, אֶפְשָׁר שֶׁהִיא בֵּיצַת עוֹף טָמֵא וְאֶפְשָׁר שֶׁהִיא בֵּיצַת עוֹף טָהוֹר. לְפִיכָךְ, שׁוֹאֵל לְצַיָּד יִשְׂרָאֵל, אוֹ עוֹבֵד כּוֹכָבִים, (טוּר וּבֵית יוֹסֵף בְּשֵׁם ה''ה שֶׁכֵּן כָּתַב בְּשֵׁם הריב''ח) הַמּוֹכֵר, אִם אָמַר לוֹ: שֶׁל עוֹף פְּלוֹנִי הוּא, וְאָנוּ מַכִּירִים שֶׁאוֹתוֹ עוֹף טָהוֹר הוּא, (ב) סוֹמֵךְ עָלָיו. וְאִם אָמַר: שֶׁל עוֹף טָהוֹר, וְלֹא אָמַר שְׁמוֹ, (ג) אֵינוֹ סוֹמֵךְ עָלָיו. לְפִיכָךְ אֵין לוֹקְחִין בֵּיצִים מִן הָעוֹבְדֵי כּוֹכָבִים, אֶלָּא אִם כֵּן הָיָה מַכִּיר אוֹתָם וְיֵשׁ לוֹ בָּהֶם טְבִיעוּת עַיִן שֶׁהֵם בֵּיצֵי עוֹף פְּלוֹנִי טָהוֹר; וְאֵין חוֹשְׁשִׁין לָהֶם שֶׁמָּא הֵם בֵּיצֵי (ד) טְרֵפָה אוֹ נְבֵלָה.


ב עַכְשָׁו נוֹהֲגִים לִקַּח בֵּיצִים, בִּסְתָם, מִכָּל אָדָם, לְפִי שֶׁאֵין בֵּיצֵי עוֹף טָמֵא מָצוּי בֵּינֵינוּ. הגה: וְדַוְקָא בַּבֵּיצִים הַשְּׁכִיחִים, דְּהַיְנוּ בֵּיצַת תַּרְנְגוֹלִים אוֹ אַוָּזוֹת וְכַיּוֹצֵא בָּהֶן, אֲבָל אֵין לוֹקְחִים בֵּיצִים מֵעוֹפוֹת שֶׁאֵינָם שְׁכִיחִים יוֹתֵר מֵעוֹפוֹת טְמֵאִים, עַד שֶׁיִּרְאֶה (ה) בַּסִּימָנִים, כְּמוֹ שֶׁנִּתְבָּאֵר. (אָרֹךְ כְּלָל מ''ב) .


ג בֵּיצַת (ו) נְבֵלָה וּטְרֵפָה, אֲפִלּוּ נִגְמְרָה, שֶׁכַּיּוֹצֵא בָּהּ נִמְכֶּרֶת בַּשּׁוּק, אֲסוּרָה. וַאֲפִלּוּ (ז) נִתְעָרְבָה בְּאֶלֶף, כֻּלָּן אֲסוּרוֹת. אֲבָל סָפֵק אִם הִיא בֵּיצַת נְבֵלָה אוֹ אִם הִיא בֵּיצַת שְׁחוּטָה, (ח) מֻתֶּרֶת, אֲפִלּוּ בְּלֹא תַּעֲרֹבֶת. אֲבָל סָפֵק אִם הִיא בֵּיצַת טְרֵפָה אוֹ בֵּיצַת שְׁחוּטָה, אֲסוּרָה. הגה: וְאִם נִשְׁחֲטָה הַתַּרְנְגֹלֶת וְנִמְצֵאת טְרֵפָה, דִּין בֵּיצִים כְּדִין חָלָב שֶׁנֶּחְלַב מִבְּהֵמָה (בֵּית יוֹסֵף) . וְעַיֵּן לְעֵיל סִימָן פ''א.


ד הִכָּה תַּרְנְגֹלֶת עַל זְנָבָהּ וְהִפִּילָה בֵּיצָתָהּ קֹדֶם שֶׁנִּגְמְרָה, מֻתֶּרֶת, וְאֵין כָּאן מִשּׁוּם אֵבֶר מִן הַחַי. וְהוּא שֶׁאֵינָהּ מְעֹרָה בְּגִידִין.


ה בֵּיצַת עוֹף טָמֵא שֶׁנִּתְבַּשְּׁלָה עִם אֲחֵרוֹת, אִם הִיא (ט) בִּקְלִפָּתָהּ, אֵינָהּ אוֹסֶרֶת. וְאִם הִיא קְלוּפָה, אוֹסֶרֶת, וְאִם הִיא נְקוּבָה, דִּינָה כִּקְלוּפָה (אָרֹךְ כְּלָל מ''ב וְהַגָּ''ה ש''ד) וְצָרִיךְ (י) ס'' א בֵּיצִים שֶׁל הֶתֵּר כְּנֶגְדָּהּ לְבַטְּלָהּ, וַאֲפִלּוּ אֵינָהּ שְׁלֵמָה רַק טְרוּפָה (בְּאָרֹךְ וּבֵית יוֹסֵף בְּשֵׁם הָרַשְׁבָּ''א) . וְאִם נִתְבַּשְּׁלָה בְּתַבְשִׁיל שֶׁל הֶתֵּר, אֵין צָרִיךְ אֶלָּא שִׁשִּׁים מִן הַתַּבְשִׁיל, לְבַטְּלָהּ. וּבֵיצָה שֶׁיֵּשׁ בָּהּ (יא) אֶפְרוֹחַ אוֹ דָּם, אוֹסֶרֶת הַמִּתְבַּשֵּׁל עִמָּהּ, אֲפִלּוּ אִם אֵינָהּ קְלוּפָה. הגה: וּקְלִפּוֹת הָאִסּוּר מִצְטָרְפִים גַּם עִם הַהֶתֵּר לְבַטֵּל הָאִסּוּר, וְכָל שֶׁכֵּן קְלִפּוֹת הַהֶתֵּר. (תא''ו נט''ו וירושלמי וש''ס עי' ס''ק י''ד) . וְע''ל סִימָן (יב) צ''ט.


ו בֵּיצַת טְרֵפָה וּנְבֵלָה, דִּינָה כְּדִין בֵּיצַת עוֹף טָמֵא, דְּכָל שֶׁאֵינָהּ קְלוּפָה וְנִתְבַּשְּׁלָה עִם אֲחֵרוֹת אֵינָהּ אוֹסֶרֶת, דְּמַיָּא דְּבֵיעֵי בְּעָלְמָא הוּא. הגה: וְיֵשׁ אוֹמְרִים דְּכָל בֵּיצִים דִּינָם שָׁוֶה וְאוֹסְרִים, אֲפִלּוּ אֵינָם קְלוּפִים (טוּר בְּשֵׁם בה''ת וְרֹב הַפּוֹסְקִים) . וְכֵן יֵשׁ לִנְהֹג בְּמָקוֹם שֶׁאֵין הֶפְסֵד מְרֻבֶּה, אֲבָל בְּהֶפְסֵד מְרֻבֶּה יֵשׁ לִסְמֹךְ אַמְּקִלִּין.


ז אֶפְרוֹחַ שֶׁנּוֹלַד מִבֵּיצַת נְבֵלָה וּטְרֵפָה, (יג) מֻתָּר. הגה: (יד) וְיֵשׁ אוֹמְרִים דְּלֹא נְהִיגִין כֵּן (בְּאָרֹךְ) . וּלְכַתְּחִילָה יִזָּהֵר.


ח בֵּיצַת עוֹף טָמֵא שֶׁהִתְחִיל הָעוֹף לְהִתְרַקֵּם בָּהּ, לוֹקִים עָלֶיהָ מִשּׁוּם שֶׁרֶץ הָעוֹף (וַיִּקְרָא יא, כ, כג) . וּבֵיצַת עוֹף טָהוֹר שֶׁהִתְחִיל הָעוֹף לְהִתְרַקֵּם בָּהּ, וַאֲכָלָהּ, הָיוּ מַכִּין אוֹתוֹ (טו) מַכַּת מַרְדוּת.


ט טְרֵפָה אֵינָהּ יוֹלֶדֶת. לְפִיכָךְ עוֹף שֶׁהוּא סְפֵק טְרֵפָה וְהֵטִיל בֵּיצִים, (טז) מַשְׁהִין אוֹתוֹ עַד שֶׁיָּטִיל כָּל בֵּיצָיו שֶׁטּוֹעֵן כְּבָר; אִם יִטְעֹן פַּעַם אַחֶרֶת, כְּשֵׁרָה. וְנֵדַע זֶה, כְּשֶׁשָּׁהֲתָה בֵּין סוֹף לֵדָה לִתְחִלַּת לֵדָה כ'' א יוֹם שְׁלֵמִים, כִּי כֵן זְמַן עִבּוּרָהּ. וְהוּא הַדִּין לְתַרְנְגֹלֶת שֶׁלֹּא יָלְדָה מֵעוֹלָם שֶׁנּוֹלַד בָּהּ (יז) סְפֵק טְרֵפוּת, שֶׁאִם יָלְדָה אַחַר כ'' א יוֹם מִזְּמַן שֶׁנּוֹלַד בָּהּ סְפֵק טְרֵפוּת, הִיא וְהַבֵּיצִים מֻתָּרִים (אֹרְחַת חַיִּים מְבִיאוֹ בֵּית יוֹסֵף בְּסוֹף סִימָן נ''ז) .


י אֵין מוֹכְרִין בֵּיצַת נְבֵלָה וּטְרֵפָה לְעוֹבֵד כּוֹכָבִים, שֶׁמָּא יַחֲזֹר וְיִמְכְּרֵם לְיִשְׂרָאֵל. כֵּיצַד יַעֲשֶׂה, יִטְרְפֶנָּה בִּקְעָרָה וְיִמְכְּרֶנָּה לְעוֹבֵד כּוֹכָבִים, שֶׁאָז וַדַּאי לֹא יִקְנֶה אוֹתָהּ יִשְׂרָאֵל מִמֶּנּוּ. (יח) בַּמֶּה דְּבָרִים אֲמוּרִים, שֶׁרוֹאִים אוֹתָהּ טְרוּפָה בְּיָדוֹ, שֶׁיֵּשׁ לָחוּשׁ שֶׁיִּשְׂרָאֵל מְכָרָהּ לוֹ מִפְּנֵי שֶׁהִיא טְרֵפָה. אֲבָל מֻתָּר לִקְנוֹת מִמֶּנּוּ פַּת שֶׁנִּלּוֹשׁ בְּבֵיצִים, בְּמָקוֹם (יט) שֶׁנּוֹהֲגִין לֶאֱכֹל פַּת שֶׁל עוֹבֵד כּוֹכָבִים, כֵּיוָן שֶׁאֵין רוֹאִים אוֹתָהּ טְרוּפָה בְּיָדוֹ. הגה: וְיֵשׁ אוֹמְרִים דְּעַכְשָׁיו דְּאֵין נִזְהָרִין מִפַּת שֶׁל עוֹבֵד כּוֹכָבִים, אֵין לִמְכֹּר לְעוֹבֵד כּוֹכָבִים בֵּיצָה אֲסוּרָה, אֲפִלּוּ טְרוּפָה, דְּחַיְשִׁינָן שֶׁמָּא יְעָרְבֶנָּה עִם פַּת וְיִמְכְּרֶנָּה לְיִשְׂרָאֵל (אָרֹךְ כְּלָל מ''ב וּמ''ד בְּשֵׁם סמ''ק) . וְהָכֵי נָהוּג בְּמָקוֹם דְּנוֹהֲגִין לַעֲשׂוֹת הַפַּת עִם בֵּיצִים וְנוֹהֲגִין בּוֹ הֶתֵּר לִקְנוֹתוֹ, אֲבָל בִּשְׁאָר מְקוֹמוֹת אֵין לָחוּשׁ.
