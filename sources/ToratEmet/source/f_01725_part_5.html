פרשת קרח




במדבר פרק-טז

{א} וַיִּקַּ֣ח קֹ֔רַח בֶּן-יִצְהָ֥ר בֶּן-קְהָ֖ת בֶּן-לֵוִ֑י וְדָתָ֨ן וַאֲבִירָ֜ם בְּנֵ֧י אֱלִיאָ֛ב וְא֥וֹן בֶּן-פֶּ֖לֶת בְּנֵ֥י רְאוּבֵֽן:  מ  וַיִּקַּ֣ח קֹ֔רַח בֶּן-יִצְהָ֥ר בֶּן-קְהָ֖ת בֶּן-לֵוִ֑י וְדָתָ֨ן וַאֲבִירָ֜ם בְּנֵ֧י אֱלִיאָ֛ב וְא֥וֹן בֶּן-פֶּ֖לֶת בְּנֵ֥י רְאוּבֵֽן:  ת  וְאִתְפְּלֵג קֹרַח בַּר יִצְהָר בַּר קְהָת בַּר לֵוִי וְדָתָן וַאֲבִירָם בְּנֵי אֱלִיאָב וְאוֹן בַּר פֶּלֶת בְּנֵי רְאוּבֵן:

{ב} וַיָּקֻ֙מוּ֙ לִפְנֵ֣י מֹשֶׁ֔ה וַאֲנָשִׁ֥ים מִבְּנֵֽי-יִשְׂרָאֵ֖ל חֲמִשִּׁ֣ים וּמָאתָ֑יִם נְשִׂיאֵ֥י עֵדָ֛ה קְרִאֵ֥י מוֹעֵ֖ד אַנְשֵׁי-שֵֽׁם:  מ  וַיָּקֻ֙מוּ֙ לִפְנֵ֣י מֹשֶׁ֔ה וַאֲנָשִׁ֥ים מִבְּנֵֽי-יִשְׂרָאֵ֖ל חֲמִשִּׁ֣ים וּמָאתָ֑יִם נְשִׂיאֵ֥י עֵדָ֛ה קְרִאֵ֥י מוֹעֵ֖ד אַנְשֵׁי-שֵֽׁם:  ת  וְקָמוּ לְאַפֵּי משֶׁה וְגֻבְרַיָא מִבְּנֵי יִשְׂרָאֵל מָאתָן וְחַמְשִׁין רַבְרְבֵי כְנִשְׁתָּא מְעַרְעֵי זְמַן אֱנָשִׁין דִשְׁמָא:

{ג} וַיִּֽקָּהֲל֞וּ עַל-מֹשֶׁ֣ה וְעַֽל-אַהֲרֹ֗ן וַיֹּאמְר֣וּ אֲלֵהֶם֮ רַב-לָכֶם֒ כִּ֤י כָל-הָֽעֵדָה֙ כֻּלָּ֣ם קְדֹשִׁ֔ים וּבְתוֹכָ֖ם יְהוָ֑ה וּמַדּ֥וּעַ תִּֽתְנַשְּׂא֖וּ עַל-קְהַ֥ל יְהוָֽה:  מ  וַיִּֽקָּהֲל֞וּ עַל-מֹשֶׁ֣ה וְעַֽל-אַהֲרֹ֗ן וַיֹּאמְר֣וּ אֲלֵהֶם֮ רַב-לָכֶם֒ כִּ֤י כָל-הָֽעֵדָה֙ כֻּלָּ֣ם קְדֹשִׁ֔ים וּבְתוֹכָ֖ם יְהוָ֑ה וּמַדּ֥וּעַ תִּֽתְנַשְּׂא֖וּ עַל-קְהַ֥ל יְהוָֽה:  ת  וְאִתְכְּנָשׁוּ עַל משֶׁה וְעַל אַהֲרֹן וַאֲמָרוּ לְהוֹן סַגִי לְכוֹן אֲרֵי כָל כְּנִשְׁתָּא כֻּלְהוֹן קַדִישִׁין וּבֵינֵיהוֹן שַׁרְיָא שְׁכִנְתָּא דַיְיָ וּמָא דֵין אַתּוּן מִתְרַבְרְבִין עַל קְהָלָא דַיְיָ:

{ד} וַיִּשְׁמַ֣ע מֹשֶׁ֔ה וַיִּפֹּ֖ל עַל-פָּנָֽיו:  מ  וַיִּשְׁמַ֣ע מֹשֶׁ֔ה וַיִּפֹּ֖ל עַל-פָּנָֽיו:  ת  וּשְׁמַע משֶׁה וּנְפַל עַל אַפּוֹהִי:

{ה} וַיְדַבֵּ֨ר אֶל-קֹ֜רַח וְאֶֽל-כָּל-עֲדָתוֹ֮ לֵאמֹר֒ בֹּ֠קֶר וְיֹדַ֨ע יְהוָ֧ה אֶת-אֲשֶׁר-ל֛וֹ וְאֶת-הַקָּד֖וֹשׁ וְהִקְרִ֣יב אֵלָ֑יו וְאֵ֛ת אֲשֶׁ֥ר יִבְחַר-בּ֖וֹ יַקְרִ֥יב אֵלָֽיו:  מ  וַיְדַבֵּ֨ר אֶל-קֹ֜רַח וְאֶֽל-כָּל-עֲדָתוֹ֮ לֵאמֹר֒ בֹּ֠קֶר וְיֹדַ֨ע יְהוָ֧ה אֶת-אֲשֶׁר-ל֛וֹ וְאֶת-הַקָּד֖וֹשׁ וְהִקְרִ֣יב אֵלָ֑יו וְאֵ֛ת אֲשֶׁ֥ר יִבְחַר-בּ֖וֹ יַקְרִ֥יב אֵלָֽיו:  ת  וּמַלִיל עִם קֹרַח וְעִם כָּל כְּנִשְׁתֵּהּ לְמֵימַר בְּצַפְרָא וִיהוֹדַע יְיָ יָת דְכָשַׁר לֵהּ וְיָת דְקַדִישׁ וִיקָרֵב לָקֳדָמוֹהִי וְיָת דִי יִתְרְעֵי בֵהּ יְקָרֵב לְשִׁמוּשֵׁהּ:

{ו} זֹ֖את עֲשׂ֑וּ קְחוּ-לָכֶ֣ם מַחְתּ֔וֹת קֹ֖רַח וְכָל-עֲדָתֽוֹ:  מ  זֹ֖את עֲשׂ֑וּ קְחוּ-לָכֶ֣ם מַחְתּ֔וֹת קֹ֖רַח וְכָל-עֲדָתֽוֹ:  ת  דָא עַבִידוּ סָבוּ לְכוֹן מַחְתְּיָן קֹרַח וְכָל כְּנִשְׁתֵּהּ:

{ז} וּתְנ֣וּ בָהֵ֣ן | אֵ֡שׁ וְשִׂימוּ֩ עֲלֵיהֶ֨ן קְטֹ֜רֶת לִפְנֵ֤י יְהוָה֙ מָחָ֔ר וְהָיָ֗ה הָאִ֛ישׁ אֲשֶׁר-יִבְחַ֥ר יְהוָ֖ה ה֣וּא הַקָּד֑וֹשׁ רַב-לָכֶ֖ם בְּנֵ֥י לֵוִֽי:  מ  וּתְנ֣וּ בָהֵ֣ן | אֵ֡שׁ וְשִׂימוּ֩ עֲלֵיהֶ֨ן קְטֹ֜רֶת לִפְנֵ֤י יְהוָה֙ מָחָ֔ר וְהָיָ֗ה הָאִ֛ישׁ אֲשֶׁר-יִבְחַ֥ר יְהוָ֖ה ה֣וּא הַקָּד֑וֹשׁ רַב-לָכֶ֖ם בְּנֵ֥י לֵוִֽי:  ת  וְהָבוּ בְהוֹן אֵשָׁתָא וְשַׁוִיאוּ עֲלֵיהוֹן קְטוֹרֶת בּוּסְמִין קֳדָם יְיָ מְחַר וִיהֵי גַבְרָא דְיִתְרְעֵי יְיָ הוּא קַדִישּׁ סַגִי לְכּוֹן בְּנֵי לֵוִי:

{ח} וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל-קֹ֑רַח שִׁמְעוּ-נָ֖א בְּנֵ֥י לֵוִֽי:  מ  וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל-קֹ֑רַח שִׁמְעוּ-נָ֖א בְּנֵ֥י לֵוִֽי:  ת  וַאֲמַר משֶׁה לְקֹרַח שְׁמָעוּ כְעַן בְּנֵי לֵוִי:

{ט} הַמְעַ֣ט מִכֶּ֗ם כִּֽי-הִבְדִּיל֩ אֱלֹהֵ֨י יִשְׂרָאֵ֤ל אֶתְכֶם֙ מֵעֲדַ֣ת יִשְׂרָאֵ֔ל לְהַקְרִ֥יב אֶתְכֶ֖ם אֵלָ֑יו לַעֲבֹ֗ד אֶת-עֲבֹדַת֙ מִשְׁכַּ֣ן יְהוָ֔ה וְלַעֲמֹ֛ד לִפְנֵ֥י הָעֵדָ֖ה לְשָׁרְתָֽם:  מ  הַמְעַ֣ט מִכֶּ֗ם כִּֽי-הִבְדִּיל֩ אֱלֹהֵ֨י יִשְׂרָאֵ֤ל אֶתְכֶם֙ מֵעֲדַ֣ת יִשְׂרָאֵ֔ל לְהַקְרִ֥יב אֶתְכֶ֖ם אֵלָ֑יו לַעֲבֹ֗ד אֶת-עֲבֹדַת֙ מִשְׁכַּ֣ן יְהוָ֔ה וְלַעֲמֹ֛ד לִפְנֵ֥י הָעֵדָ֖ה לְשָׁרְתָֽם:  ת  הַזְעֵר לְכוֹן אֲרֵי אַפְרֵשׁ אֱלָהָא דְיִשְׂרָאֵל יָתְכוֹן מִכְּנִשְׁתָּא דְיִשְׂרָאֵל לְקֳרָבָא יָתְכוֹן קֳדָמוֹהִי לְמִפְלַח יָת פּוּלְחַן מַשְׁכְּנָא דַיְיָ וּלְמֵקַם קֳדָם כְּנִשְׁתָּא לְשַׁמָשׁוּתְהוֹן:

{י} וַיַּקְרֵב֙ אֹֽתְךָ֔ וְאֶת-כָּל-אַחֶ֥יךָ בְנֵי-לֵוִ֖י אִתָּ֑ךְ וּבִקַּשְׁתֶּ֖ם גַּם-כְּהֻנָּֽה:  מ  וַיַּקְרֵב֙ אֹֽתְךָ֔ וְאֶת-כָּל-אַחֶ֥יךָ בְנֵי-לֵוִ֖י אִתָּ֑ךְ וּבִקַּשְׁתֶּ֖ם גַּם-כְּהֻנָּֽה:  ת  וְקָרֵב יָתָךְ וְיַת כָּל אֲחָךְ בְּנֵי לְוֵי עִמָךְ וּכְעַן אַתּוּן בָּעָן אַף כְּהוּנְתָא רַבְּתָא:

{יא} לָכֵ֗ן אַתָּה֙ וְכָל-עֲדָ֣תְךָ֔ הַנֹּעָדִ֖ים עַל-יְהוָ֑ה וְאַהֲרֹ֣ן מַה-ה֔וּא כִּ֥י (תלונו) תַלִּ֖ינוּ עָלָֽיו:  מ  לָכֵ֗ן אַתָּה֙ וְכָל-עֲדָ֣תְךָ֔ הַנֹּעָדִ֖ים עַל-יְהוָ֑ה וְאַהֲרֹ֣ן מַה-ה֔וּא כִּ֥י (תלונו) תַלִּ֖ינוּ עָלָֽיו:  ת  בְּכֵן אַתְּ וְכָל כְּנִשְׁתָּךְ דְאִזְדַמַנְתּוּן עַל יְיָ וְאַהֲרֹן מַה הוּא אֲרֵי אִתְרַעַמְתּוּן עֲלוֹהִי:

{יב} וַיִּשְׁלַ֣ח מֹשֶׁ֔ה לִקְרֹ֛א לְדָתָ֥ן וְלַאֲבִירָ֖ם בְּנֵ֣י אֱלִיאָ֑ב וַיֹּאמְר֖וּ לֹ֥א נַעֲלֶֽה:  מ  וַיִּשְׁלַ֣ח מֹשֶׁ֔ה לִקְרֹ֛א לְדָתָ֥ן וְלַאֲבִירָ֖ם בְּנֵ֣י אֱלִיאָ֑ב וַיֹּאמְר֖וּ לֹ֥א נַעֲלֶֽה:  ת  וּשְׁלַח משֶׁה לְמִקְרֵי לְדָתָן וְלַאֲבִירָם בְּנֵי אֱלִיאָב וַאֲמָרוּ לָא נִסָק:

{יג} הַמְעַ֗ט כִּ֤י הֶֽעֱלִיתָ֙נוּ֙ מֵאֶ֨רֶץ זָבַ֤ת חָלָב֙ וּדְבַ֔שׁ לַהֲמִיתֵ֖נוּ בַּמִּדְבָּ֑ר כִּֽי-תִשְׂתָּרֵ֥ר עָלֵ֖ינוּ גַּם-הִשְׂתָּרֵֽר:  מ  הַמְעַ֗ט כִּ֤י הֶֽעֱלִיתָ֙נוּ֙ מֵאֶ֨רֶץ זָבַ֤ת חָלָב֙ וּדְבַ֔שׁ לַהֲמִיתֵ֖נוּ בַּמִּדְבָּ֑ר כִּֽי-תִשְׂתָּרֵ֥ר עָלֵ֖ינוּ גַּם-הִשְׂתָּרֵֽר:  ת  הַזְעֵיר אֲרֵי אַסֵקְתָּנָא מֵאַרְעָא עָבְדָא חֲלַב וּדְבָשׁ לְקַטָלוּתָנָא בְּמַדְבְּרָא אֲרֵי אִתְרַבְרַבְתְּ עֲלָנָא אַף אִתְרַבְרָבָא: (ס)

{יד}  שני  אַ֡ף לֹ֣א אֶל-אֶרֶץ֩ זָבַ֨ת חָלָ֤ב וּדְבַשׁ֙ הֲבִ֣יאֹתָ֔נוּ וַתִּ֨תֶּן-לָ֔נוּ נַחֲלַ֖ת שָׂדֶ֣ה וָכָ֑רֶם הַעֵינֵ֞י הָאֲנָשִׁ֥ים הָהֵ֛ם תְּנַקֵּ֖ר לֹ֥א נַעֲלֶֽה:  מ  אַ֡ף לֹ֣א אֶל-אֶרֶץ֩ זָבַ֨ת חָלָ֤ב וּדְבַשׁ֙ הֲבִ֣יאֹתָ֔נוּ וַתִּ֨תֶּן-לָ֔נוּ נַחֲלַ֖ת שָׂדֶ֣ה וָכָ֑רֶם הַעֵינֵ֞י הָאֲנָשִׁ֥ים הָהֵ֛ם תְּנַקֵּ֖ר לֹ֥א נַעֲלֶֽה:  ת  בְּרַם לָא לְאַרְעָא עָבְדָא חֲלָב וּדְבָשׁ אַעֵלְתָּנָא וִיהַבְתְּ לָנָא אַחֲסָנַת חַקְלִין וְכַרְמִין הַעֵינֵי גֻבְרַיָא הָאִינוּן תְּשַׁלַח לְעַוָרָא לָא נִסָק:

{טו} וַיִּ֤חַר לְמֹשֶׁה֙ מְאֹ֔ד וַיֹּ֙אמֶר֙ אֶל-יְהוָ֔ה אַל-תֵּ֖פֶן אֶל-מִנְחָתָ֑ם לֹ֠א חֲמ֨וֹר אֶחָ֤ד מֵהֶם֙ נָשָׂ֔אתִי וְלֹ֥א הֲרֵעֹ֖תִי אֶת-אַחַ֥ד מֵהֶֽם:  מ  וַיִּ֤חַר לְמֹשֶׁה֙ מְאֹ֔ד וַיֹּ֙אמֶר֙ אֶל-יְהוָ֔ה אַל-תֵּ֖פֶן אֶל-מִנְחָתָ֑ם לֹ֠א חֲמ֨וֹר אֶחָ֤ד מֵהֶם֙ נָשָׂ֔אתִי וְלֹ֥א הֲרֵעֹ֖תִי אֶת-אַחַ֥ד מֵהֶֽם:  ת  וּתְקֵף לְמשֶׁה לַחֲדָא וַאֲמַר קֳדָם יְיָ לָא תְקַבֵּל בְּרַעֲוָא קוּרְבַּנְהוֹן לָא חֲמָרָא דְחַד מִנְהוֹן שְׁחָרִית וְלָא אַבְאֵישִׁית יַת חָד מִנְהוֹן:

{טז} וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל-קֹ֔רַח אַתָּה֙ וְכָל-עֲדָ֣תְךָ֔ הֱי֖וּ לִפְנֵ֣י יְהוָ֑ה אַתָּ֥ה וָהֵ֛ם וְאַהֲרֹ֖ן מָחָֽר:  מ  וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל-קֹ֔רַח אַתָּה֙ וְכָל-עֲדָ֣תְךָ֔ הֱי֖וּ לִפְנֵ֣י יְהוָ֑ה אַתָּ֥ה וָהֵ֛ם וְאַהֲרֹ֖ן מָחָֽר:  ת  וַאֲמַר משֶׁה לְקֹרַח אַתְּ וְכָל כְּנִשְׁתָּךְ הֱווֹ זְמִינִין לָקֳדָם יְיָ אַתְּ וְאִינוּן וְאַהֲרֹן מְחָר:

{יז} וּקְח֣וּ | אִ֣ישׁ מַחְתָּת֗וֹ וּנְתַתֶּ֤ם עֲלֵיהֶם֙ קְטֹ֔רֶת וְהִקְרַבְתֶּ֞ם לִפְנֵ֤י יְהוָה֙ אִ֣ישׁ מַחְתָּת֔וֹ חֲמִשִּׁ֥ים וּמָאתַ֖יִם מַחְתֹּ֑ת וְאַתָּ֥ה וְאַהֲרֹ֖ן אִ֥ישׁ מַחְתָּתֽוֹ:  מ  וּקְח֣וּ | אִ֣ישׁ מַחְתָּת֗וֹ וּנְתַתֶּ֤ם עֲלֵיהֶם֙ קְטֹ֔רֶת וְהִקְרַבְתֶּ֞ם לִפְנֵ֤י יְהוָה֙ אִ֣ישׁ מַחְתָּת֔וֹ חֲמִשִּׁ֥ים וּמָאתַ֖יִם מַחְתֹּ֑ת וְאַתָּ֥ה וְאַהֲרֹ֖ן אִ֥ישׁ מַחְתָּתֽוֹ:  ת  וּסְבוּ גְבַר מַחְתִּיתֵיהּ וּתְשַׁווּן עֲלֵיהוֹן קְטוֹרֶת בּוּסְמִין וְתִקָרְבוּן קֳדָם יְיָ גְבַר מַחְתִּיתֵיהּ מָאתָן וְחַמְשִׁין מַחְתְּיָן וְאַתְּ וְאַהֲרֹן גְבַר מַחְתִּיתֵהּ:

{יח} וַיִּקְח֞וּ אִ֣ישׁ מַחְתָּת֗וֹ וַיִּתְּנ֤וּ עֲלֵיהֶם֙ אֵ֔שׁ וַיָּשִׂ֥ימוּ עֲלֵיהֶ֖ם קְטֹ֑רֶת וַֽיַּעַמְד֗וּ פֶּ֛תַח אֹ֥הֶל מוֹעֵ֖ד וּמֹשֶׁ֥ה וְאַהֲרֹֽן:  מ  וַיִּקְח֞וּ אִ֣ישׁ מַחְתָּת֗וֹ וַיִּתְּנ֤וּ עֲלֵיהֶם֙ אֵ֔שׁ וַיָּשִׂ֥ימוּ עֲלֵיהֶ֖ם קְטֹ֑רֶת וַֽיַּעַמְד֗וּ פֶּ֛תַח אֹ֥הֶל מוֹעֵ֖ד וּמֹשֶׁ֥ה וְאַהֲרֹֽן:  ת  וּנְסִיבוּ גְבַר מַחְתִּיתֵהּ וִיהָבוּ עֲלֵיהוֹן אֵשָׁתָא וְשַׁוִיאוּ עֲלֵיהוֹן קְטֹרֶת בּוּסְמִין וְקָמוּ בִּתְרַע מַשְׁכַּן זִמְנָא וּמשֶׁה וְאַהֲרֹן:

{יט} וַיַּקְהֵ֨ל עֲלֵיהֶ֥ם קֹ֙רַח֙ אֶת-כָּל-הָ֣עֵדָ֔ה אֶל-פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וַיֵּרָ֥א כְבוֹד-יְהוָ֖ה אֶל-כָּל-הָעֵדָֽה:  מ  וַיַּקְהֵ֨ל עֲלֵיהֶ֥ם קֹ֙רַח֙ אֶת-כָּל-הָ֣עֵדָ֔ה אֶל-פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וַיֵּרָ֥א כְבוֹד-יְהוָ֖ה אֶל-כָּל-הָעֵדָֽה:  ת  וְאַכְנֵשׁ עֲלֵיהוֹן קֹרַח יָת כָּל כְּנִשְׁתָּא לִתְרַע מַשְׁכַּן זִמְנָא וְאִתְגְלֵי יְקָרָא דַיְיָ לְכָל כְּנִשְׁתָּא:

{כ}  שלישי  וַיְדַבֵּ֣ר יְהוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  מ  וַיְדַבֵּ֣ר יְהוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  ת  וּמַלֵיל יְיָ עִם משֶׁה וְעִם אַהֲרֹן לְמֵימָר:

{כא} הִבָּ֣דְל֔וּ מִתּ֖וֹךְ הָעֵדָ֣ה הַזֹּ֑את וַאַכַלֶּ֥ה אֹתָ֖ם כְּרָֽגַע:  מ  הִבָּ֣דְל֔וּ מִתּ֖וֹךְ הָעֵדָ֣ה הַזֹּ֑את וַאַכַלֶּ֥ה אֹתָ֖ם כְּרָֽגַע:  ת  אִתְפְּרָשׁוּ מִגוֹ כְּנִשְׁתָּא הָדָא וֶאֱשֵׁיצֵי יַתְהוֹן כְּשָׁעָה:

{כב} וַיִּפְּל֤וּ עַל-פְּנֵיהֶם֙ וַיֹּ֣אמְר֔וּ אֵ֕ל אֱלֹהֵ֥י הָרוּחֹ֖ת לְכָל-בָּשָׂ֑ר הָאִ֤ישׁ אֶחָד֙ יֶחֱטָ֔א וְעַ֥ל כָּל-הָעֵדָ֖ה תִּקְצֹֽף:  מ  וַיִּפְּל֤וּ עַל-פְּנֵיהֶם֙ וַיֹּ֣אמְר֔וּ אֵ֕ל אֱלֹהֵ֥י הָרוּחֹ֖ת לְכָל-בָּשָׂ֑ר הָאִ֤ישׁ אֶחָד֙ יֶחֱטָ֔א וְעַ֥ל כָּל-הָעֵדָ֖ה תִּקְצֹֽף:  ת  וּנְפָלוּ עַל אַפֵּיהוֹן וַאֲמָרוּ אֵל אֱלָהָא רוּחַיָא לְכָל בִּשְׂרָא גַבְרָא חָד יֵיחוֹב וְעַל כָּל כְּנִשְׁתָּא יְהֵי רוּגְזָא: (ס)

{כג} וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר:

{כד} דַּבֵּ֥ר אֶל-הָעֵדָ֖ה לֵאמֹ֑ר הֵֽעָלוּ֙ מִסָּבִ֔יב לְמִשְׁכַּן-קֹ֖רַח דָּתָ֥ן וַאֲבִירָֽם:  מ  דַּבֵּ֥ר אֶל-הָעֵדָ֖ה לֵאמֹ֑ר הֵֽעָלוּ֙ מִסָּבִ֔יב לְמִשְׁכַּן-קֹ֖רַח דָּתָ֥ן וַאֲבִירָֽם:  ת  מַלֵל עִם כְּנִשְׁתָּא לְמֵימָר אִסְתַּלָקוּ מִסְחוֹר סְחוֹר לְמַשְׁכְּנָא דְקֹרַח דָתָן וַאֲבִירָם:

{כה} וַיָּ֣קָם מֹשֶׁ֔ה וַיֵּ֖לֶךְ אֶל-דָּתָ֣ן וַאֲבִירָ֑ם וַיֵּלְכ֥וּ אַחֲרָ֖יו זִקְנֵ֥י יִשְׂרָאֵֽל:  מ  וַיָּ֣קָם מֹשֶׁ֔ה וַיֵּ֖לֶךְ אֶל-דָּתָ֣ן וַאֲבִירָ֑ם וַיֵּלְכ֥וּ אַחֲרָ֖יו זִקְנֵ֥י יִשְׂרָאֵֽל:  ת  וְקָם משֶׁה וַאֲזַל לְוַת דָתָן וַאֲבִירָם וַאֲזָלוּ בַּתְרוֹהִי סָבֵי יִשְׂרָאֵל:

{כו} וַיְדַבֵּ֨ר אֶל-הָעֵדָ֜ה לֵאמֹ֗ר ס֣וּרוּ נָ֡א מֵעַל֩ אָהֳלֵ֨י הָאֲנָשִׁ֤ים הָֽרְשָׁעִים֙ הָאֵ֔לֶּה וְאַֽל-תִּגְּע֖וּ בְּכָל-אֲשֶׁ֣ר לָהֶ֑ם פֶּן-תִּסָּפ֖וּ בְּכָל-חַטֹּאתָֽם:  מ  וַיְדַבֵּ֨ר אֶל-הָעֵדָ֜ה לֵאמֹ֗ר ס֣וּרוּ נָ֡א מֵעַל֩ אָהֳלֵ֨י הָאֲנָשִׁ֤ים הָֽרְשָׁעִים֙ הָאֵ֔לֶּה וְאַֽל-תִּגְּע֖וּ בְּכָל-אֲשֶׁ֣ר לָהֶ֑ם פֶּן-תִּסָּפ֖וּ בְּכָל-חַטֹּאתָֽם:  ת  וּמַלֵיל עִם כְּנִשְׁתָּא לְמֵימָר זוּרוּ כְעַן מֵעִלַוֵי מַשְׁכְּנֵי גֻבְרַיָא חַיָיבַיָא הָאִלֵין וְלָא תְקָרְבוּן בְּכָל דִי לְהוֹן דִלְמָא תִלְקוּן בְּכָל חוֹבֵיהוֹן:

{כז} וַיֵּעָל֗וּ מֵעַ֧ל מִשְׁכַּן-קֹ֛רֶח דָּתָ֥ן וַאֲבִירָ֖ם מִסָּבִ֑יב וְדָתָ֨ן וַאֲבִירָ֜ם יָצְא֣וּ נִצָּבִ֗ים פֶּ֚תַח אָֽהֳלֵיהֶ֔ם וּנְשֵׁיהֶ֥ם וּבְנֵיהֶ֖ם וְטַפָּֽם:  מ  וַיֵּעָל֗וּ מֵעַ֧ל מִשְׁכַּן-קֹ֛רֶח דָּתָ֥ן וַאֲבִירָ֖ם מִסָּבִ֑יב וְדָתָ֨ן וַאֲבִירָ֜ם יָצְא֣וּ נִצָּבִ֗ים פֶּ֚תַח אָֽהֳלֵיהֶ֔ם וּנְשֵׁיהֶ֥ם וּבְנֵיהֶ֖ם וְטַפָּֽם:  ת  וְאִסְתַּלָקוּ מֵעִלַוֵי מַשְׁכְּנָא דְקֹרַח דָתָן וַאֲבִירָם מִסְחוֹר סְחוֹר וְדָתָן וַאֲבִירָם נְפָקוּ קַיְימִין בִּתְרַע מַשְׁכְּנֵיהוֹן וּנְשֵׁיהוֹן וּבְנֵיהוֹן וְטַפְלְהוֹן:

{כח} וַיֹּאמֶר֮ מֹשֶׁה֒ בְּזֹאת֙ תֵּֽדְע֔וּן כִּֽי-יְהוָ֣ה שְׁלָחַ֔נִי לַעֲשׂ֕וֹת אֵ֥ת כָּל-הַֽמַּעֲשִׂ֖ים הָאֵ֑לֶּה כִּי-לֹ֖א מִלִּבִּֽי:  מ  וַיֹּאמֶר֮ מֹשֶׁה֒ בְּזֹאת֙ תֵּֽדְע֔וּן כִּֽי-יְהוָ֣ה שְׁלָחַ֔נִי לַעֲשׂ֕וֹת אֵ֥ת כָּל-הַֽמַּעֲשִׂ֖ים הָאֵ֑לֶּה כִּי-לֹ֖א מִלִּבִּֽי:  ת  וַאֲמַר משֶׁה בְּדָא תִדְעוּן אֲרֵי יְיָ שַׁלְחַנִי לְמֶעְבַּד יָת כָּל עוֹבָדַיָא הָאִלֵין אֲרֵי לָא מֵרְעוּתִי:

{כט} אִם-כְּמ֤וֹת כָּל-הָֽאָדָם֙ יְמֻת֣וּן אֵ֔לֶּה וּפְקֻדַּת֙ כָּל-הָ֣אָדָ֔ם יִפָּקֵ֖ד עֲלֵיהֶ֑ם לֹ֥א יְהוָ֖ה שְׁלָחָֽנִי:  מ  אִם-כְּמ֤וֹת כָּל-הָֽאָדָם֙ יְמֻת֣וּן אֵ֔לֶּה וּפְקֻדַּת֙ כָּל-הָ֣אָדָ֔ם יִפָּקֵ֖ד עֲלֵיהֶ֑ם לֹ֥א יְהוָ֖ה שְׁלָחָֽנִי:  ת  אִם כְּמוֹתָא דְכָל אֱנָשָׁא יְמוּתוּן אִלֵין וּסְעָרָא דְכָל אֱנָשָׁא יִסְתְּעַר עֲלֵיהוֹן לָא יְיָ שַׁלְחַנִי:

{ל} וְאִם-בְּרִיאָ֞ה יִבְרָ֣א יְהוָ֗ה וּפָצְתָ֨ה הָאֲדָמָ֤ה אֶת-פִּ֙יהָ֙ וּבָלְעָ֤ה אֹתָם֙ וְאֶת-כָּל-אֲשֶׁ֣ר לָהֶ֔ם וְיָרְד֥וּ חַיִּ֖ים שְׁאֹ֑לָה וִֽידַעְתֶּ֕ם כִּ֧י נִֽאֲצ֛וּ הָאֲנָשִׁ֥ים הָאֵ֖לֶּה אֶת-יְהוָֽה:  מ  וְאִם-בְּרִיאָ֞ה יִבְרָ֣א יְהוָ֗ה וּפָצְתָ֨ה הָאֲדָמָ֤ה אֶת-פִּ֙יהָ֙ וּבָלְעָ֤ה אֹתָם֙ וְאֶת-כָּל-אֲשֶׁ֣ר לָהֶ֔ם וְיָרְד֥וּ חַיִּ֖ים שְׁאֹ֑לָה וִֽידַעְתֶּ֕ם כִּ֧י נִֽאֲצ֛וּ הָאֲנָשִׁ֥ים הָאֵ֖לֶּה אֶת-יְהוָֽה:  ת  וְאִם בְּרִיאָה יִבְרֵי יְיָ וּפְתָחַת אַרְעָא יָת פּוּמַהּ וְתִבְלַע יַתְהוֹן וְיַת כָּל דִי לְהוֹן וְיֶחֱתוּן כַּד חַיִין לִשְׁאוֹל וְתִדְעוּן אֲרֵי אַרְגִיזוּ גוּבְרַיָא הָאִלֵין קֳדָם יְיָ:

{לא} וַיְהִי֙ כְּכַלֹּת֔וֹ לְדַבֵּ֕ר אֵ֥ת כָּל-הַדְּבָרִ֖ים הָאֵ֑לֶּה וַתִּבָּקַ֥ע הָאֲדָמָ֖ה אֲשֶׁ֥ר תַּחְתֵּיהֶֽם:  מ  וַיְהִי֙ כְּכַלֹּת֔וֹ לְדַבֵּ֕ר אֵ֥ת כָּל-הַדְּבָרִ֖ים הָאֵ֑לֶּה וַתִּבָּקַ֥ע הָאֲדָמָ֖ה אֲשֶׁ֥ר תַּחְתֵּיהֶֽם:  ת  וַהֲוָה כַּד שֵׁיצֵי לְמַלָלָא יָת כָּל פִּתְגָמַיָא הָאִלֵין וְאִתְבְּזָעַת אַרְעָא דִי תְּחוֹתֵיהוֹן:

{לב} וַתִּפְתַּ֤ח הָאָ֙רֶץ֙ אֶת-פִּ֔יהָ וַתִּבְלַ֥ע אֹתָ֖ם וְאֶת-בָּתֵּיהֶ֑ם וְאֵ֤ת כָּל-הָאָדָם֙ אֲשֶׁ֣ר לְקֹ֔רַח וְאֵ֖ת כָּל-הָרֲכֽוּשׁ:  מ  וַתִּפְתַּ֤ח הָאָ֙רֶץ֙ אֶת-פִּ֔יהָ וַתִּבְלַ֥ע אֹתָ֖ם וְאֶת-בָּתֵּיהֶ֑ם וְאֵ֤ת כָּל-הָאָדָם֙ אֲשֶׁ֣ר לְקֹ֔רַח וְאֵ֖ת כָּל-הָרֲכֽוּשׁ:  ת  וּפְתָחַת אַרְעָא יָת פּוּמָהּ וּבְלָעַת יַתְהוֹן וְיַת אֱנַשׁ בָּתֵּיהוֹן וְיַת כָּל אֱנָשָׁא דִי לְקֹרַח וְיַת כָּל קִנְיָנָא:

{לג} וַיֵּ֨רְד֜וּ הֵ֣ם וְכָל-אֲשֶׁ֥ר לָהֶ֛ם חַיִּ֖ים שְׁאֹ֑לָה וַתְּכַ֤ס עֲלֵיהֶם֙ הָאָ֔רֶץ וַיֹּאבְד֖וּ מִתּ֥וֹךְ הַקָּהָֽל:  מ  וַיֵּ֨רְד֜וּ הֵ֣ם וְכָל-אֲשֶׁ֥ר לָהֶ֛ם חַיִּ֖ים שְׁאֹ֑לָה וַתְּכַ֤ס עֲלֵיהֶם֙ הָאָ֔רֶץ וַיֹּאבְד֖וּ מִתּ֥וֹךְ הַקָּהָֽל:  ת  וּנְחָתוּ אִינוּן וְכָל דִי לְהוֹן כַּד חַיִין לִשְׁאוֹל וַחֲפַת עֲלֵיהוֹן אַרְעָא וַאֲבָדוּ מִגוֹ קְהָלָא:

{לד} וְכָל-יִשְׂרָאֵ֗ל אֲשֶׁ֛ר סְבִיבֹתֵיהֶ֖ם נָ֣סוּ לְקֹלָ֑ם כִּ֣י אָֽמְר֔וּ פֶּן-תִּבְלָעֵ֖נוּ הָאָֽרֶץ:  מ  וְכָל-יִשְׂרָאֵ֗ל אֲשֶׁ֛ר סְבִיבֹתֵיהֶ֖ם נָ֣סוּ לְקֹלָ֑ם כִּ֣י אָֽמְר֔וּ פֶּן-תִּבְלָעֵ֖נוּ הָאָֽרֶץ:  ת  וְכָל יִשְׂרָאֵל דִי בְסַחֲרָנֵיהוֹן עֲרָקוּ לְקַלְהוֹן אֲרֵי אֲמָרוּ דִלְמָא תִבְלְעִנָנָא אַרְעָא:

{לה} וְאֵ֥שׁ יָצְאָ֖ה מֵאֵ֣ת יְהוָ֑ה וַתֹּ֗אכַל אֵ֣ת הַחֲמִשִּׁ֤ים וּמָאתַ֙יִם֙ אִ֔ישׁ מַקְרִיבֵ֖י הַקְּטֹֽרֶת:  מ  וְאֵ֥שׁ יָצְאָ֖ה מֵאֵ֣ת יְהוָ֑ה וַתֹּ֗אכַל אֵ֣ת הַחֲמִשִּׁ֤ים וּמָאתַ֙יִם֙ אִ֔ישׁ מַקְרִיבֵ֖י הַקְּטֹֽרֶת:  ת  וְאֶשָׁתָא נְפָקַת מִן קֳדָם יְיָ וַאֲכָלַת יָת מָאתָן וְחַמְשִׁין גַבְרָא מְקַרְבֵי קְטוֹרֶת בּוּסְמַיָא: (ס)




במדבר פרק-יז

{א} וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר:

{ב} אֱמֹ֨ר אֶל-אֶלְעָזָ֜ר בֶּן-אַהֲרֹ֣ן הַכֹּהֵ֗ן וְיָרֵ֤ם אֶת-הַמַּחְתֹּת֙ מִבֵּ֣ין הַשְּׂרֵפָ֔ה וְאֶת-הָאֵ֖שׁ זְרֵה-הָ֑לְאָה כִּ֖י קָדֵֽשׁוּ:  מ  אֱמֹ֨ר אֶל-אֶלְעָזָ֜ר בֶּן-אַהֲרֹ֣ן הַכֹּהֵ֗ן וְיָרֵ֤ם אֶת-הַמַּחְתֹּת֙ מִבֵּ֣ין הַשְּׂרֵפָ֔ה וְאֶת-הָאֵ֖שׁ זְרֵה-הָ֑לְאָה כִּ֖י קָדֵֽשׁוּ:  ת  אֱמַר לְאֶלְעָזָר בַּר אַהֲרֹן כַּהֲנָא וְיַפְרֵשׁ יָת מַחְתְּיָתָא מִבֵּין יְקִידַיָא וְיָת אֶשָׁתָא יַרְחִיק לְהָלָא אֲרֵי אִתְקַדָשׁוּ:

{ג} אֵ֡ת מַחְתּוֹת֩ הַֽחַטָּאִ֨ים הָאֵ֜לֶּה בְּנַפְשֹׁתָ֗ם וְעָשׂ֨וּ אֹתָ֜ם רִקֻּעֵ֤י פַחִים֙ צִפּ֣וּי לַמִּזְבֵּ֔חַ כִּֽי-הִקְרִיבֻ֥ם לִפְנֵֽי-יְהוָ֖ה וַיִּקְדָּ֑שׁוּ וְיִֽהְי֥וּ לְא֖וֹת לִבְנֵ֥י יִשְׂרָאֵֽל:  מ  אֵ֡ת מַחְתּוֹת֩ הַֽחַטָּאִ֨ים הָאֵ֜לֶּה בְּנַפְשֹׁתָ֗ם וְעָשׂ֨וּ אֹתָ֜ם רִקֻּעֵ֤י פַחִים֙ צִפּ֣וּי לַמִּזְבֵּ֔חַ כִּֽי-הִקְרִיבֻ֥ם לִפְנֵֽי-יְהוָ֖ה וַיִּקְדָּ֑שׁוּ וְיִֽהְי֥וּ לְא֖וֹת לִבְנֵ֥י יִשְׂרָאֵֽל:  ת  יָת מַחְתְּיַת חַיָיבַיָא הָאִלֵין דְאִתְחַיְיבוּ בְּנַפְשָׁתֵיהוֹן וְיַעְבְּדוּן יַתְהוֹן טַסִין רְדִידִין חוֹפָאָה לְמַדְבְּחָא אֲרֵי קָרֵבֻנוּן קֳדָם יְיָ וְאִתְקַדָשׁוּ וִיהֶוְיָן לְאָת לִבְּנֵי יִשְׂרָאֵל:

{ד} וַיִּקַּ֞ח אֶלְעָזָ֣ר הַכֹּהֵ֗ן אֵ֚ת מַחְתּ֣וֹת הַנְּחֹ֔שֶׁת אֲשֶׁ֥ר הִקְרִ֖יבוּ הַשְּׂרֻפִ֑ים וַֽיְרַקְּע֖וּם צִפּ֥וּי לַמִּזְבֵּֽחַ:  מ  וַיִּקַּ֞ח אֶלְעָזָ֣ר הַכֹּהֵ֗ן אֵ֚ת מַחְתּ֣וֹת הַנְּחֹ֔שֶׁת אֲשֶׁ֥ר הִקְרִ֖יבוּ הַשְּׂרֻפִ֑ים וַֽיְרַקְּע֖וּם צִפּ֥וּי לַמִּזְבֵּֽחַ:  ת  וּנְסִיב אֶלְעָזָר כַּהֲנָא יָת מַחְתְּיָתָא דִנְחָשָׁא דְקָרִיבוּ יוֹקִידַיָא וְרַדִידֻנוּן חוֹפָאָה לְמַדְבְּחָא:

{ה} זִכָּר֞וֹן לִבְנֵ֣י יִשְׂרָאֵ֗ל לְ֠מַעַן אֲשֶׁ֨ר לֹֽא-יִקְרַ֜ב אִ֣ישׁ זָ֗ר אֲ֠שֶׁר לֹ֣א מִזֶּ֤רַע אַהֲרֹן֙ ה֔וּא לְהַקְטִ֥יר קְטֹ֖רֶת לִפְנֵ֣י יְהוָ֑ה וְלֹֽא-יִהְיֶ֤ה כְקֹ֙רַח֙ וְכַ֣עֲדָת֔וֹ כַּאֲשֶׁ֨ר דִּבֶּ֧ר יְהוָ֛ה בְּיַד-מֹשֶׁ֖ה לֽוֹ:  מ  זִכָּר֞וֹן לִבְנֵ֣י יִשְׂרָאֵ֗ל לְ֠מַעַן אֲשֶׁ֨ר לֹֽא-יִקְרַ֜ב אִ֣ישׁ זָ֗ר אֲ֠שֶׁר לֹ֣א מִזֶּ֤רַע אַהֲרֹן֙ ה֔וּא לְהַקְטִ֥יר קְטֹ֖רֶת לִפְנֵ֣י יְהוָ֑ה וְלֹֽא-יִהְיֶ֤ה כְקֹ֙רַח֙ וְכַ֣עֲדָת֔וֹ כַּאֲשֶׁ֨ר דִּבֶּ֧ר יְהוָ֛ה בְּיַד-מֹשֶׁ֖ה לֽוֹ:  ת  דָכְרָנָא לִבְנֵי יִשְׂרָאֵל בְּדִיל דִי לָא יִקְרַב גְבַר חִלוֹנַי דִי לָא מִזַרְעָא דְאַהֲרֹן הוּא לְאַסָקָא קְטוֹרֶת בּוּסְמִין קֳדָם יְיָ וְלָא יְהֵי כְקֹרַח וְכִכְנִשְׁתֵּיהּ כְּמָא דִי מַלֵיל יְיָ בִּידָא דְמשֶׁה לֵיהּ: (פ)

{ו} וַיִּלֹּ֜נוּ כָּל-עֲדַ֤ת בְּנֵֽי-יִשְׂרָאֵל֙ מִֽמָּחֳרָ֔ת עַל-מֹשֶׁ֥ה וְעַֽל-אַהֲרֹ֖ן לֵאמֹ֑ר אַתֶּ֥ם הֲמִתֶּ֖ם אֶת-עַ֥ם יְהוָֽה:  מ  וַיִּלֹּ֜נוּ כָּל-עֲדַ֤ת בְּנֵֽי-יִשְׂרָאֵל֙ מִֽמָּחֳרָ֔ת עַל-מֹשֶׁ֥ה וְעַֽל-אַהֲרֹ֖ן לֵאמֹ֑ר אַתֶּ֥ם הֲמִתֶּ֖ם אֶת-עַ֥ם יְהוָֽה:  ת  וְאִתְרַעֲמוּ כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל מִיוֹמָא דְבַתְרוֹהִי עַל משֶׁה וְעַל אַהֲרֹן לְמֵימָר אַתּוּן גְרַמְתּוּן דְמִית עַמָא דַיְיָ:

{ז} וַיְהִ֗י בְּהִקָּהֵ֤ל הָֽעֵדָה֙ עַל-מֹשֶׁ֣ה וְעַֽל-אַהֲרֹ֔ן וַיִּפְנוּ֙ אֶל-אֹ֣הֶל מוֹעֵ֔ד וְהִנֵּ֥ה כִסָּ֖הוּ הֶעָנָ֑ן וַיֵּרָ֖א כְּב֥וֹד יְהוָֽה:  מ  וַיְהִ֗י בְּהִקָּהֵ֤ל הָֽעֵדָה֙ עַל-מֹשֶׁ֣ה וְעַֽל-אַהֲרֹ֔ן וַיִּפְנוּ֙ אֶל-אֹ֣הֶל מוֹעֵ֔ד וְהִנֵּ֥ה כִסָּ֖הוּ הֶעָנָ֑ן וַיֵּרָ֖א כְּב֥וֹד יְהוָֽה:  ת  וַהֲוָה בְּאִתְכַּנָשׁוּת כְּנִשְׁתָּא עַל משֶׁה וְעַל אַהֲרֹן וְאִתְפְּנִיאוּ לְמַשְׁכַּן זִמְנָא וְהָא חֲפָהִי עֲנָנָא וְאִתְגְלִי יְקָרָא דַיְיָ:

{ח} וַיָּבֹ֤א מֹשֶׁה֙ וְאַהֲרֹ֔ן אֶל-פְּנֵ֖י אֹ֥הֶל מוֹעֵֽד:  מ  וַיָּבֹ֤א מֹשֶׁה֙ וְאַהֲרֹ֔ן אֶל-פְּנֵ֖י אֹ֥הֶל מוֹעֵֽד:  ת  וְעָאל משֶׁה וְאַהֲרֹן לָקֳדָם מַשְׁכַּן זִמְנָא: (ס)

{ט}  רביעי  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר:

{י} הֵרֹ֗מּוּ מִתּוֹךְ֙ הָעֵדָ֣ה הַזֹּ֔את וַאֲכַלֶּ֥ה אֹתָ֖ם כְּרָ֑גַע וַֽיִּפְּל֖וּ עַל-פְּנֵיהֶֽם:  מ  הֵרֹ֗מּוּ מִתּוֹךְ֙ הָעֵדָ֣ה הַזֹּ֔את וַאֲכַלֶּ֥ה אֹתָ֖ם כְּרָ֑גַע וַֽיִּפְּל֖וּ עַל-פְּנֵיהֶֽם:  ת  אִתְפְרָשׁוּ מִגוֹ כְּנִשְׁתָּא הָדָא וְאֱשֵׁיצֵי יָתְהוֹן כְּשָׁעָה וּנְפָלוּ עַל אַפֵּיהוֹן:

{יא} וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶֽל-אַהֲרֹ֗ן קַ֣ח אֶת-הַ֠מַּחְתָּה וְתֶן-עָלֶ֨יהָ אֵ֜שׁ מֵעַ֤ל הַמִּזְבֵּ֙חַ֙ וְשִׂ֣ים קְטֹ֔רֶת וְהוֹלֵ֧ךְ מְהֵרָ֛ה אֶל-הָעֵדָ֖ה וְכַפֵּ֣ר עֲלֵיהֶ֑ם כִּֽי-יָצָ֥א הַקֶּ֛צֶף מִלִּפְנֵ֥י יְהוָ֖ה הֵחֵ֥ל הַנָּֽגֶף:  מ  וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶֽל-אַהֲרֹ֗ן קַ֣ח אֶת-הַ֠מַּחְתָּה וְתֶן-עָלֶ֨יהָ אֵ֜שׁ מֵעַ֤ל הַמִּזְבֵּ֙חַ֙ וְשִׂ֣ים קְטֹ֔רֶת וְהוֹלֵ֧ךְ מְהֵרָ֛ה אֶל-הָעֵדָ֖ה וְכַפֵּ֣ר עֲלֵיהֶ֑ם כִּֽי-יָצָ֥א הַקֶּ֛צֶף מִלִּפְנֵ֥י יְהוָ֖ה הֵחֵ֥ל הַנָּֽגֶף:  ת  וַאֲמַר משֶׁה לְאַהֲרֹן סַב יָת מַחְתִּיתָא וְהַב עֲלַהּ אֶשָׁתָא מֵעִלָוֵי מַדְבְּחָא וְשַׁו קְטוֹרֶת בּוּסְמַיָא וְאוֹבִיל בִּפְרִיעַ לִכְנִשְׁתָּא וְכַפֵּר עֲלֵיהוֹן אֲרֵי נְפַק רָגְזָא מִן קֳדָם יְיָ שָׁרֵי מוֹתָנָא:

{יב} וַיִּקַּ֨ח אַהֲרֹ֜ן כַּאֲשֶׁ֣ר | דִּבֶּ֣ר מֹשֶׁ֗ה וַיָּ֙רָץ֙ אֶל-תּ֣וֹך הַקָּהָ֔ל וְהִנֵּ֛ה הֵחֵ֥ל הַנֶּ֖גֶף בָּעָ֑ם וַיִּתֵּן֙ אֶֽת-הַקְּטֹ֔רֶת וַיְכַפֵּ֖ר עַל-הָעָֽם:  מ  וַיִּקַּ֨ח אַהֲרֹ֜ן כַּאֲשֶׁ֣ר | דִּבֶּ֣ר מֹשֶׁ֗ה וַיָּ֙רָץ֙ אֶל-תּ֣וֹך הַקָּהָ֔ל וְהִנֵּ֛ה הֵחֵ֥ל הַנֶּ֖גֶף בָּעָ֑ם וַיִּתֵּן֙ אֶֽת-הַקְּטֹ֔רֶת וַיְכַפֵּ֖ר עַל-הָעָֽם:  ת  וּנְסֵיב אַהֲרֹן כְּמָא דְמַלֵיל משֶׁה וּרְהַט לְגוֹ קְהָלָא וְהָא שָׁרֵי מוֹתָנָא בְּעַמָא וִיהַב יָת קְטּוֹרֶת בּוּסְמַיָא וְכַפֵּר עַל עַמָא:

{יג} וַיַּעֲמֹ֥ד בֵּֽין-הַמֵּתִ֖ים וּבֵ֣ין הַֽחַיִּ֑ים וַתֵּעָצַ֖ר הַמַּגֵּפָֽה:  מ  וַיַּעֲמֹ֥ד בֵּֽין-הַמֵּתִ֖ים וּבֵ֣ין הַֽחַיִּ֑ים וַתֵּעָצַ֖ר הַמַּגֵּפָֽה:  ת  וְקָם בֵּין מֵתַיָא וּבֵין חַיַיָא וְאִתְכְּלִי מוֹתָנָא:

{יד} וַיִּהְי֗וּ הַמֵּתִים֙ בַּמַּגֵּפָ֔ה אַרְבָּעָ֥ה עָשָׂ֛ר אֶ֖לֶף וּשְׁבַ֣ע מֵא֑וֹת מִלְּבַ֥ד הַמֵּתִ֖ים עַל-דְּבַר-קֹֽרַח:  מ  וַיִּהְי֗וּ הַמֵּתִים֙ בַּמַּגֵּפָ֔ה אַרְבָּעָ֥ה עָשָׂ֛ר אֶ֖לֶף וּשְׁבַ֣ע מֵא֑וֹת מִלְּבַ֥ד הַמֵּתִ֖ים עַל-דְּבַר-קֹֽרַח:  ת  וַהֲווֹ דְמִיתוּ בְּמוֹתָנָא אַרְבַּעַת עֲשַׂר אַלְפִין וּשְׁבַע מְאָה בַּר מִדְמִיתוּ עַל פְּלֻגְתָּא דְקֹרַח:

{טו} וַיָּ֤שָׁב אַהֲרֹן֙ אֶל-מֹשֶׁ֔ה אֶל-פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְהַמַּגֵּפָ֖ה נֶעֱצָֽרָה:  מ  וַיָּ֤שָׁב אַהֲרֹן֙ אֶל-מֹשֶׁ֔ה אֶל-פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְהַמַּגֵּפָ֖ה נֶעֱצָֽרָה:  ת  וְתָב אַהֲרֹן לְמשֶׁה לִתְרַע מַשְׁכַּן זִמְנָא וּמוֹתָנָא אִתְכְּלִי: (פ)

{טז}  חמישי  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר:

{כ} וְהָיָ֗ה הָאִ֛ישׁ אֲשֶׁ֥ר אֶבְחַר-בּ֖וֹ מַטֵּ֣הוּ יִפְרָ֑ח וַהֲשִׁכֹּתִ֣י מֵֽעָלַ֗י אֶת-תְּלֻנּוֹת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁ֛ר הֵ֥ם מַלִּינִ֖ם עֲלֵיכֶֽם:  מ  וְהָיָ֗ה הָאִ֛ישׁ אֲשֶׁ֥ר אֶבְחַר-בּ֖וֹ מַטֵּ֣הוּ יִפְרָ֑ח וַהֲשִׁכֹּתִ֣י מֵֽעָלַ֗י אֶת-תְּלֻנּוֹת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁ֛ר הֵ֥ם מַלִּינִ֖ם עֲלֵיכֶֽם:  ת  וִיהֵי גַבְרָא דִי אִתְרְעֵי בֵּהּ חוּטְרֵיהּ יִנְעֵי וַאֲנִיחַ מִן קֳדָמַי יָת תּוּרְעֲמַת בְּנֵי יִשְׂרָאֵל דִי אִנוּן מִתְרַעֲמִין עֲלֵיכוֹן:

{כא} וַיְדַבֵּ֨ר מֹשֶׁ֜ה אֶל-בְּנֵ֣י יִשְׂרָאֵ֗ל וַיִּתְּנ֣וּ אֵלָ֣יו | כָּֽל-נְשִֽׂיאֵיהֶ֡ם מַטֶּה֩ לְנָשִׂ֨יא אֶחָ֜ד מַטֶּ֨ה לְנָשִׂ֤יא אֶחָד֙ לְבֵ֣ית אֲבֹתָ֔ם שְׁנֵ֥ים עָשָׂ֖ר מַטּ֑וֹת וּמַטֵּ֥ה אַהֲרֹ֖ן בְּת֥וֹךְ מַטּוֹתָֽם:  מ  וַיְדַבֵּ֨ר מֹשֶׁ֜ה אֶל-בְּנֵ֣י יִשְׂרָאֵ֗ל וַיִּתְּנ֣וּ אֵלָ֣יו | כָּֽל-נְשִֽׂיאֵיהֶ֡ם מַטֶּה֩ לְנָשִׂ֨יא אֶחָ֜ד מַטֶּ֨ה לְנָשִׂ֤יא אֶחָד֙ לְבֵ֣ית אֲבֹתָ֔ם שְׁנֵ֥ים עָשָׂ֖ר מַטּ֑וֹת וּמַטֵּ֥ה אַהֲרֹ֖ן בְּת֥וֹךְ מַטּוֹתָֽם:  ת  וּמַלֵיל משֶׁה עִם בְּנֵי יִשְׂרָאֵל וִיהָבוּ לֵיהּ כָּל רַבְרְבָנֵיהוֹן חוּטְרָא לְרַבָּא חָד חוּטְרָא לְרַבָּא חָד לְבֵית אֲבָהַתְהוֹן תְּרֵי עֲשַׂר חוּטְרִין וְחוּטְרָא דְאַהֲרֹן בְּגוֹ חוּטְרֵיהוֹן:

{כב} וַיַּנַּ֥ח מֹשֶׁ֛ה אֶת-הַמַּטֹּ֖ת לִפְנֵ֣י יְהוָ֑ה בְּאֹ֖הֶל הָעֵדֻֽת:  מ  וַיַּנַּ֥ח מֹשֶׁ֛ה אֶת-הַמַּטֹּ֖ת לִפְנֵ֣י יְהוָ֑ה בְּאֹ֖הֶל הָעֵדֻֽת:  ת  וְאַצְנַע משֶׁה יָת חוּטְּרַיָא קֳדָם יְיָ בְּמַשְׁכְּנָא דְסַהֲדוּתָא:

{כג} וַיְהִ֣י מִֽמָּחֳרָ֗ת וַיָּבֹ֤א מֹשֶׁה֙ אֶל-אֹ֣הֶל הָעֵד֔וּת וְהִנֵּ֛ה פָּרַ֥ח מַטֵּֽה-אַהֲרֹ֖ן לְבֵ֣ית לֵוִ֑י וַיֹּ֤צֵֽא פֶ֙רַח֙ וַיָּ֣צֵֽץ צִ֔יץ וַיִּגְמֹ֖ל שְׁקֵדִֽים:  מ  וַיְהִ֣י מִֽמָּחֳרָ֗ת וַיָּבֹ֤א מֹשֶׁה֙ אֶל-אֹ֣הֶל הָעֵד֔וּת וְהִנֵּ֛ה פָּרַ֥ח מַטֵּֽה-אַהֲרֹ֖ן לְבֵ֣ית לֵוִ֑י וַיֹּ֤צֵֽא פֶ֙רַח֙ וַיָּ֣צֵֽץ צִ֔יץ וַיִּגְמֹ֖ל שְׁקֵדִֽים:  ת  וַהֲוָה בְּיוֹמָא דְבַתְרוֹהִי וְעַאל משֶׁה לְמַשְׁכְּנָא דְסַהֲדוּתָא וְהָא נְעָא חוּטְרָא דְאַהֲרֹן לְבֵית לְוֵי וְאַפֵּק לַבְלְבִין וְאָנֵץ נֵץ וְכָפֵית שִׁיגְדִין:

{כד} וַיֹּצֵ֨א מֹשֶׁ֤ה אֶת-כָּל-הַמַּטֹּת֙ מִלִּפְנֵ֣י יְהוָ֔ה אֶֽל-כָּל-בְּנֵ֖י יִשְׂרָאֵ֑ל וַיִּרְא֥וּ וַיִּקְח֖וּ אִ֥ישׁ מַטֵּֽהוּ:  מ  וַיֹּצֵ֨א מֹשֶׁ֤ה אֶת-כָּל-הַמַּטֹּת֙ מִלִּפְנֵ֣י יְהוָ֔ה אֶֽל-כָּל-בְּנֵ֖י יִשְׂרָאֵ֑ל וַיִּרְא֥וּ וַיִּקְח֖וּ אִ֥ישׁ מַטֵּֽהוּ:  ת  וְאַפֵיק משֶׁה יָת כָּל חוּטְרַיָא מִן קֳדָם יְיָ לְוָת כָּל בְּנֵי יִשְׂרָאֵל וְאִשְׁתְּמוֹדָעוּ וּנְסִיבוּ גְבַר חוּטְרֵיהּ: (פ)

{כו} וַיַּ֖עַשׂ מֹשֶׁ֑ה כַּאֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֹת֖וֹ כֵּ֥ן עָשָֽׂה:  מ  וַיַּ֖עַשׂ מֹשֶׁ֑ה כַּאֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֹת֖וֹ כֵּ֥ן עָשָֽׂה:  ת  וַעֲבַד משֶׁה כְּמָא דִי פַקֵיד יְיָ יָתֵיה כֵּן עֲבָד: (פ)

במדבר פרק-יח

{א} וַיֹּ֤אמֶר יְהוָה֙ אֶֽל-אַהֲרֹ֔ן אַתָּ֗ה וּבָנֶ֤יךָ וּבֵית-אָבִ֙יךָ֙ אִתָּ֔ךְ תִּשְׂא֖וּ אֶת-עֲוֹ֣ן הַמִּקְדָּ֑שׁ וְאַתָּה֙ וּבָנֶ֣יךָ אִתָּ֔ךְ תִּשְׂא֖וּ אֶת-עֲוֹ֥ן כְּהֻנַּתְכֶֽם:  מ  וַיֹּ֤אמֶר יְהוָה֙ אֶֽל-אַהֲרֹ֔ן אַתָּ֗ה וּבָנֶ֤יךָ וּבֵית-אָבִ֙יךָ֙ אִתָּ֔ךְ תִּשְׂא֖וּ אֶת-עֲוֹ֣ן הַמִּקְדָּ֑שׁ וְאַתָּה֙ וּבָנֶ֣יךָ אִתָּ֔ךְ תִּשְׂא֖וּ אֶת-עֲוֹ֥ן כְּהֻנַּתְכֶֽם:  ת  וַאֲמַר יְיָ לְאַהֲרֹן אַתְּ וּבְנָךְ וּבֵית אֲבוּךְ עִמָךְ תְּסַלְחוּן עַל חוֹבֵי מַקְדְשָׁא וְאַתְּ וּבְנָךְ עִמָךְ תְּסַלְחוּן עַל חוֹבֵי כְּהוּנַתְכוֹן:

{ב} וְגַ֣ם אֶת-אַחֶיךָ֩ מַטֵּ֨ה לֵוִ֜י שֵׁ֤בֶט אָבִ֙יךָ֙ הַקְרֵ֣ב אִתָּ֔ךְ וְיִלָּו֥וּ עָלֶ֖יךָ וִֽישָׁרְת֑וּךָ וְאַתָּה֙ וּבָנֶ֣יךָ אִתָּ֔ךְ לִפְנֵ֖י אֹ֥הֶל הָעֵדֻֽת:  מ  וְגַ֣ם אֶת-אַחֶיךָ֩ מַטֵּ֨ה לֵוִ֜י שֵׁ֤בֶט אָבִ֙יךָ֙ הַקְרֵ֣ב אִתָּ֔ךְ וְיִלָּו֥וּ עָלֶ֖יךָ וִֽישָׁרְת֑וּךָ וְאַתָּה֙ וּבָנֶ֣יךָ אִתָּ֔ךְ לִפְנֵ֖י אֹ֥הֶל הָעֵדֻֽת:  ת  וְאַף יָת אֲחָיךְ שִׁבְטָא דְלֵוִי שִׁבְטָא דַאֲבוּךְ קָרֵיב לְוָתָךְ וְיִתּוֹסְפוּן עֲלָךְ וִישַׁמְשׁוּנָךְ וְאַתְּ וּבְנָךְ עִמָךְ קֳדָם מַשְׁכְּנָא דְסַהֲדוּתָא:

{ג} וְשָֽׁמְרוּ֙ מִֽשְׁמַרְתְּךָ֔ וּמִשְׁמֶ֖רֶת כָּל-הָאֹ֑הֶל אַךְ֩ אֶל-כְּלֵ֨י הַקֹּ֤דֶשׁ וְאֶל-הַמִּזְבֵּ֙חַ֙ לֹ֣א יִקְרָ֔בוּ וְלֹֽא-יָמֻ֥תוּ גַם-הֵ֖ם גַּם-אַתֶּֽם:  מ  וְשָֽׁמְרוּ֙ מִֽשְׁמַרְתְּךָ֔ וּמִשְׁמֶ֖רֶת כָּל-הָאֹ֑הֶל אַךְ֩ אֶל-כְּלֵ֨י הַקֹּ֤דֶשׁ וְאֶל-הַמִּזְבֵּ֙חַ֙ לֹ֣א יִקְרָ֔בוּ וְלֹֽא-יָמֻ֥תוּ גַם-הֵ֖ם גַּם-אַתֶּֽם:  ת  וְיִטְרוּן מַטַרְתָּךְ וּמַטְרַת כָּל מַשְׁכְּנָא בְּרַם לְמָנֵי קוּדְשָׁא וּלְמַדְבְּחָא לָא יִקְרְבוּן וְלָא יְמוּתוּן אַף אִינוּן אַף אַתּוּן:

{ד} וְנִלְו֣וּ עָלֶ֔יךָ וְשָֽׁמְר֗וּ אֶת-מִשְׁמֶ֙רֶת֙ אֹ֣הֶל מוֹעֵ֔ד לְכֹ֖ל עֲבֹדַ֣ת הָאֹ֑הֶל וְזָ֖ר לֹא-יִקְרַ֥ב אֲלֵיכֶֽם:  מ  וְנִלְו֣וּ עָלֶ֔יךָ וְשָֽׁמְר֗וּ אֶת-מִשְׁמֶ֙רֶת֙ אֹ֣הֶל מוֹעֵ֔ד לְכֹ֖ל עֲבֹדַ֣ת הָאֹ֑הֶל וְזָ֖ר לֹא-יִקְרַ֥ב אֲלֵיכֶֽם:  ת  וְיִתּוֹסְפוּן עֲלָךְ וְיִטְרוּן יָת מַטְרַת מַשְׁכַּן זִמְנָא לְכָל פּוּלְחַן מַשְׁכְּנָא וְחִילוֹנֵי לָא יִקְרַב לְוָתְכוֹן:

{ה} וּשְׁמַרְתֶּ֗ם אֵ֚ת מִשְׁמֶ֣רֶת הַקֹּ֔דֶשׁ וְאֵ֖ת מִשְׁמֶ֣רֶת הַמִּזְבֵּ֑חַ וְלֹֽא-יִהְיֶ֥ה ע֛וֹד קֶ֖צֶף עַל-בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וּשְׁמַרְתֶּ֗ם אֵ֚ת מִשְׁמֶ֣רֶת הַקֹּ֔דֶשׁ וְאֵ֖ת מִשְׁמֶ֣רֶת הַמִּזְבֵּ֑חַ וְלֹֽא-יִהְיֶ֥ה ע֛וֹד קֶ֖צֶף עַל-בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וְתִטְרוּן יָת מַטְרַת קוּדְשָׁא וְיָת מַטְרַת מַדְבְּחָא וְלָא יְהֵי עוֹד רוּגְזָא עַל בְּנֵי יִשְׂרָאֵל:

{ו} וַאֲנִ֗י הִנֵּ֤ה לָקַ֙חְתִּי֙ אֶת-אֲחֵיכֶ֣ם הַלְוִיִּ֔ם מִתּ֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל לָכֶ֞ם מַתָּנָ֤ה נְתֻנִים֙ לַֽיהוָ֔ה לַעֲבֹ֕ד אֶת-עֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד:  מ  וַאֲנִ֗י הִנֵּ֤ה לָקַ֙חְתִּי֙ אֶת-אֲחֵיכֶ֣ם הַלְוִיִּ֔ם מִתּ֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל לָכֶ֞ם מַתָּנָ֤ה נְתֻנִים֙ לַֽיהוָ֔ה לַעֲבֹ֕ד אֶת-עֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד:  ת  וַאֲנָא הָא קְרֵיבִית יָת אֲחֵיכוֹן לֵוָאֵי מִגוֹ בְּנֵי יִשְׂרָאֵל לְכוֹן מַתְּנָא יְהִיבִין קֳדָם יְיָ לְמִפְלַח יָת פּוּלְחַן מַשְׁכַּן זִמְנָא:

{ז} וְאַתָּ֣ה וּבָנֶ֣יךָ אִ֠תְּךָ תִּשְׁמְר֨וּ אֶת-כְּהֻנַּתְכֶ֜ם לְכָל-דְּבַ֧ר הַמִּזְבֵּ֛חַ וּלְמִבֵּ֥ית לַפָּרֹ֖כֶת וַעֲבַדְתֶּ֑ם עֲבֹדַ֣ת מַתָּנָ֗ה אֶתֵּן֙ אֶת-כְּהֻנַּתְכֶ֔ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת:  מ  וְאַתָּ֣ה וּבָנֶ֣יךָ אִ֠תְּךָ תִּשְׁמְר֨וּ אֶת-כְּהֻנַּתְכֶ֜ם לְכָל-דְּבַ֧ר הַמִּזְבֵּ֛חַ וּלְמִבֵּ֥ית לַפָּרֹ֖כֶת וַעֲבַדְתֶּ֑ם עֲבֹדַ֣ת מַתָּנָ֗ה אֶתֵּן֙ אֶת-כְּהֻנַּתְכֶ֔ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת:  ת  וְאַתְּ וּבְנָךְ עִמָךְ תִּטְרוּן יָת כְּהוּנַתְכוֹן לְכָל פִּתְגַם מַדְבְּחָא וּלְמִגָיו לְפָרוּכְתָּא וְתִפְלְחוּן פוּלְחַן מַתְּנָא אֵיהַב יָת כְּהוּנַתְכוֹן וְחִילוֹנֵי דְיִקְרַב יִתְקְטֵיל: (פ)

{ח} וַיְדַבֵּ֣ר יְהוָה֮ אֶֽל-אַהֲרֹן֒ וַאֲנִי֙ הִנֵּ֣ה נָתַ֣תִּֽי לְךָ֔ אֶת-מִשְׁמֶ֖רֶת תְּרוּמֹתָ֑י לְכָל-קָדְשֵׁ֣י בְנֵֽי-יִ֠שְׂרָאֵל לְךָ֙ נְתַתִּ֧ים לְמָשְׁחָ֛ה וּלְבָנֶ֖יךָ לְחָק-עוֹלָֽם:  מ  וַיְדַבֵּ֣ר יְהוָה֮ אֶֽל-אַהֲרֹן֒ וַאֲנִי֙ הִנֵּ֣ה נָתַ֣תִּֽי לְךָ֔ אֶת-מִשְׁמֶ֖רֶת תְּרוּמֹתָ֑י לְכָל-קָדְשֵׁ֣י בְנֵֽי-יִ֠שְׂרָאֵל לְךָ֙ נְתַתִּ֧ים לְמָשְׁחָ֛ה וּלְבָנֶ֖יךָ לְחָק-עוֹלָֽם:  ת  וּמַלֵיל יְיָ עִם אַהֲרֹן וַאֲנָא הָא יְהָבִית לָךְ יָת מַטְרַת אַפְרָשׁוּתָי לְכָל קוּדְשַׁיָא דִבְנֵי יִשְׂרָאֵל לָךְ יְהַבְתִּינוּן לִרְבוּ וְלִבְנָךְ לִקְיָם עֲלָם:

{ט} זֶֽה-יִהְיֶ֥ה לְךָ֛ מִקֹּ֥דֶשׁ הַקֳּדָשִׁ֖ים מִן-הָאֵ֑שׁ כָּל-קָ֠רְבָּנָם לְֽכָל-מִנְחָתָ֞ם וּלְכָל-חַטָּאתָ֗ם וּלְכָל-אֲשָׁמָם֙ אֲשֶׁ֣ר יָשִׁ֣יבוּ לִ֔י קֹ֣דֶשׁ קָֽדָשִׁ֥ים לְךָ֛ ה֖וּא וּלְבָנֶֽיךָ:  מ  זֶֽה-יִהְיֶ֥ה לְךָ֛ מִקֹּ֥דֶשׁ הַקֳּדָשִׁ֖ים מִן-הָאֵ֑שׁ כָּל-קָ֠רְבָּנָם לְֽכָל-מִנְחָתָ֞ם וּלְכָל-חַטָּאתָ֗ם וּלְכָל-אֲשָׁמָם֙ אֲשֶׁ֣ר יָשִׁ֣יבוּ לִ֔י קֹ֣דֶשׁ קָֽדָשִׁ֥ים לְךָ֛ ה֖וּא וּלְבָנֶֽיךָ:  ת  דֵין יְהֵי לָךְ מִקוֹדֶשׁ קוּדְשַׁיָא מוֹתַר מִן אֶשָׁתָא כָּל קוּרְבַּנְהוֹן לְכָל מִנְחַתְהוֹן וּלְכָל חַטְוַתְהוֹן וּלְכָל אֲשַׁמְהוֹן דִי יְתִיבוּן קֳדָמַי קוֹדֶשׁ קוּדְשִׁין דִילָךְ הִיא וְלִבְנָךְ:

{י} בְּקֹ֥דֶשׁ הַקֳּדָשִׁ֖ים תֹּאכֲלֶ֑נּוּ כָּל-זָכָר֙ יֹאכַ֣ל אֹת֔וֹ קֹ֖דֶשׁ יִֽהְיֶה-לָּֽךְ:  מ  בְּקֹ֥דֶשׁ הַקֳּדָשִׁ֖ים תֹּאכֲלֶ֑נּוּ כָּל-זָכָר֙ יֹאכַ֣ל אֹת֔וֹ קֹ֖דֶשׁ יִֽהְיֶה-לָּֽךְ:  ת  בְּקֹדֶשׁ קוּדְשִׁין תֵּיכְלוּנֵיהּ כָּל דְכוּרָא יֵיכוֹל יָתֵיהּ קוּדְשָׁא יְהֵי לָךְ:

{יא} וְזֶה-לְּךָ֞ תְּרוּמַ֣ת מַתָּנָ֗ם לְכָל-תְּנוּפֹת֮ בְּנֵ֣י יִשְׂרָאֵל֒ לְךָ֣ נְתַתִּ֗ים וּלְבָנֶ֧יךָ וְלִבְנֹתֶ֛יךָ אִתְּךָ֖ לְחָק-עוֹלָ֑ם כָּל-טָה֥וֹר בְּבֵיתְךָ֖ יֹאכַ֥ל אֹתֽוֹ:  מ  וְזֶה-לְּךָ֞ תְּרוּמַ֣ת מַתָּנָ֗ם לְכָל-תְּנוּפֹת֮ בְּנֵ֣י יִשְׂרָאֵל֒ לְךָ֣ נְתַתִּ֗ים וּלְבָנֶ֧יךָ וְלִבְנֹתֶ֛יךָ אִתְּךָ֖ לְחָק-עוֹלָ֑ם כָּל-טָה֥וֹר בְּבֵיתְךָ֖ יֹאכַ֥ל אֹתֽוֹ:  ת  וְדֵין לָךְ אַפְרָשׁוּת מַתְּנַתְהוֹן לְכָל אֲרָמוּת בְּנֵי יִשְׂרָאֵל לָךְ יְהַבְתִּינוּן וְלִבְנָיךְ וְלִבְנְתָיךְ עִמָךְ לִקְיַם עֲלָם כָּל דִדְכֵי בְּבֵיתָךְ יֵיכוֹל יָתֵיהּ:

{יב} כֹּ֚ל חֵ֣לֶב יִצְהָ֔ר וְכָל-חֵ֖לֶב תִּיר֣וֹשׁ וְדָגָ֑ן רֵאשִׁיתָ֛ם אֲשֶׁר-יִתְּנ֥וּ לַֽיהוָ֖ה לְךָ֥ נְתַתִּֽים:  מ  כֹּ֚ל חֵ֣לֶב יִצְהָ֔ר וְכָל-חֵ֖לֶב תִּיר֣וֹשׁ וְדָגָ֑ן רֵאשִׁיתָ֛ם אֲשֶׁר-יִתְּנ֥וּ לַֽיהוָ֖ה לְךָ֥ נְתַתִּֽים:  ת  כָּל טוּב מְשַׁח וְכָל טוּב חֲמַר וְעִבּוּר רֵאשִׁיתְהוֹן דִי יִתְּנוּן קֳדָם יְיָ לָךְ יְהַבְתִּינוּן:

{יג} בִּכּוּרֵ֞י כָּל-אֲשֶׁ֧ר בְּאַרְצָ֛ם אֲשֶׁר-יָבִ֥יאוּ לַיהוָ֖ה לְךָ֣ יִהְיֶ֑ה כָּל-טָה֥וֹר בְּבֵיתְךָ֖ יֹאכֲלֶֽנּוּ:  מ  בִּכּוּרֵ֞י כָּל-אֲשֶׁ֧ר בְּאַרְצָ֛ם אֲשֶׁר-יָבִ֥יאוּ לַיהוָ֖ה לְךָ֣ יִהְיֶ֑ה כָּל-טָה֥וֹר בְּבֵיתְךָ֖ יֹאכֲלֶֽנּוּ:  ת  בִּכּוּרֵי כָּל דִי בְאַרְעֲהוֹן דִי יַיְתוּן קֳדָם יְיָ דִילָךְ יְהֵי כָּל דִדְכֵי בְּבֵיתָךְ יֵיכְלִנֵיהּ:

{יד} כָּל-חֵ֥רֶם בְּיִשְׂרָאֵ֖ל לְךָ֥ יִהְיֶֽה:  מ  כָּל-חֵ֥רֶם בְּיִשְׂרָאֵ֖ל לְךָ֥ יִהְיֶֽה:  ת  כָּל חֶרְמָא בְּיִשְׂרָאֵל דִילָךְ יְהֵי:

{טו} כָּל-פֶּ֣טֶר רֶ֠חֶם לְֽכָל-בָּשָׂ֞ר אֲשֶׁר-יַקְרִ֧יבוּ לַֽיהוָ֛ה בָּאָדָ֥ם וּבַבְּהֵמָ֖ה יִֽהְיֶה-לָּ֑ךְ אַ֣ךְ | פָּדֹ֣ה תִפְדֶּ֗ה אֵ֚ת בְּכ֣וֹר הָֽאָדָ֔ם וְאֵ֛ת בְּכֽוֹר-הַבְּהֵמָ֥ה הַטְּמֵאָ֖ה תִּפְדֶּֽה:  מ  כָּל-פֶּ֣טֶר רֶ֠חֶם לְֽכָל-בָּשָׂ֞ר אֲשֶׁר-יַקְרִ֧יבוּ לַֽיהוָ֛ה בָּאָדָ֥ם וּבַבְּהֵמָ֖ה יִֽהְיֶה-לָּ֑ךְ אַ֣ךְ | פָּדֹ֣ה תִפְדֶּ֗ה אֵ֚ת בְּכ֣וֹר הָֽאָדָ֔ם וְאֵ֛ת בְּכֽוֹר-הַבְּהֵמָ֥ה הַטְּמֵאָ֖ה תִּפְדֶּֽה:  ת  כָּל פְּתַח וַלְדָא לְכָל בִּשְׂרָא דִי יְקָרְבוּן קֳדָם יְיָ בֶּאֱנָשָׁא וּבִבְעִירָא יְהֵי לָךְ בְּרַם מִפְרַק תִּפְרוֹק יָת בּוּכְרָא דֶאֱנָשָׁא וְיָת בּוּכְרָא דִבְעִירָא מְסַאֲבָא תִּפְרוֹק:

{טז} וּפְדוּיָו֙ מִבֶּן-חֹ֣דֶשׁ תִּפְדֶּ֔ה בְּעֶ֨רְכְּךָ֔ כֶּ֛סֶף חֲמֵ֥שֶׁת שְׁקָלִ֖ים בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ עֶשְׂרִ֥ים גֵּרָ֖ה הֽוּא:  מ  וּפְדוּיָו֙ מִבֶּן-חֹ֣דֶשׁ תִּפְדֶּ֔ה בְּעֶ֨רְכְּךָ֔ כֶּ֛סֶף חֲמֵ֥שֶׁת שְׁקָלִ֖ים בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ עֶשְׂרִ֥ים גֵּרָ֖ה הֽוּא:  ת  וּפוּרְקָנֵיהּ מִבַּר יַרְחָא תִּפְרוֹק בְּפוּרְסָנֵיהּ כְּסַף חַמְשָׁא סִלְעִין בְּסִלְעֵי קוּדְשָׁא עַשְׂרִין מָעִין הוּא:

{יז} אַ֣ךְ בְּֽכוֹר-שׁ֡וֹר אֽוֹ-בְכ֨וֹר כֶּ֜שֶׂב אֽוֹ-בְכ֥וֹר עֵ֛ז לֹ֥א תִפְדֶּ֖ה קֹ֣דֶשׁ הֵ֑ם אֶת-דָּמָ֞ם תִּזְרֹ֤ק עַל-הַמִּזְבֵּ֙חַ֙ וְאֶת-חֶלְבָּ֣ם תַּקְטִ֔יר אִשֶּׁ֛ה לְרֵ֥יחַ נִיחֹ֖חַ לַֽיהוָֽה:  מ  אַ֣ךְ בְּֽכוֹר-שׁ֡וֹר אֽוֹ-בְכ֨וֹר כֶּ֜שֶׂב אֽוֹ-בְכ֥וֹר עֵ֛ז לֹ֥א תִפְדֶּ֖ה קֹ֣דֶשׁ הֵ֑ם אֶת-דָּמָ֞ם תִּזְרֹ֤ק עַל-הַמִּזְבֵּ֙חַ֙ וְאֶת-חֶלְבָּ֣ם תַּקְטִ֔יר אִשֶּׁ֛ה לְרֵ֥יחַ נִיחֹ֖חַ לַֽיהוָֽה:  ת  בְּרַם בּוּכְרָא דְתוֹרָא אוֹ בּוּכְרָא דְאִמְרָא אוֹ בּוּכְרָא דְעִזָא לָא תִפְרוֹק קוּדְשָׁא אִינוּן יָת דִמְהוֹן תִּזְרוֹק עַל מַדְבְּחָא וְיָת תַּרְבְּהוֹן תַּסֵיק קוּרְבַּן לְאִתְקַבָּלָא בְּרַעֲוָא קֳדָם יְיָ:

{יח} וּבְשָׂרָ֖ם יִהְיֶה-לָּ֑ךְ כַּחֲזֵ֧ה הַתְּנוּפָ֛ה וּכְשׁ֥וֹק הַיָּמִ֖ין לְךָ֥ יִהְיֶֽה:  מ  וּבְשָׂרָ֖ם יִהְיֶה-לָּ֑ךְ כַּחֲזֵ֧ה הַתְּנוּפָ֛ה וּכְשׁ֥וֹק הַיָּמִ֖ין לְךָ֥ יִהְיֶֽה:  ת  וּבִשְׂרְהוֹן יְהֵי לָךְ כְּחַדְיָא דַאֲרָמוּתָא וּכְשַׁקָא דְיַמִינָא דִילָךְ יְהֵי:

{יט} כֹּ֣ל | תְּרוּמֹ֣ת הַקֳּדָשִׁ֗ים אֲשֶׁ֨ר יָרִ֥ימוּ בְנֵֽי-יִשְׂרָאֵל֮ לַֽיהוָה֒ נָתַ֣תִּֽי לְךָ֗ וּלְבָנֶ֧יךָ וְלִבְנֹתֶ֛יךָ אִתְּךָ֖ לְחָק-עוֹלָ֑ם בְּרִית֩ מֶ֨לַח עוֹלָ֥ם הִוא֙ לִפְנֵ֣י יְהוָ֔ה לְךָ֖ וּלְזַרְעֲךָ֥ אִתָּֽךְ:  מ  כֹּ֣ל | תְּרוּמֹ֣ת הַקֳּדָשִׁ֗ים אֲשֶׁ֨ר יָרִ֥ימוּ בְנֵֽי-יִשְׂרָאֵל֮ לַֽיהוָה֒ נָתַ֣תִּֽי לְךָ֗ וּלְבָנֶ֧יךָ וְלִבְנֹתֶ֛יךָ אִתְּךָ֖ לְחָק-עוֹלָ֑ם בְּרִית֩ מֶ֨לַח עוֹלָ֥ם הִוא֙ לִפְנֵ֣י יְהוָ֔ה לְךָ֖ וּלְזַרְעֲךָ֥ אִתָּֽךְ:  ת  כָּל אַפְרָשׁוּת קוּדְשַׁיָא דִי יִפְרְשׁוּן בְּנֵי יִשְׂרָאֵל קֳדָם יְיָ יְהָבִית לָךְ וְלִבְנָיךְ וְלִבְנָתָיךְ עִמָךְ לִקְיַם עֲלָם קְיַים מְלַח עֲלָם הִיא קֳדָם יְיָ לָךְ וְלִבְנָךְ עִמָךְ:

{כ} וַיֹּ֨אמֶר יְהוָ֜ה אֶֽל-אַהֲרֹ֗ן בְּאַרְצָם֙ לֹ֣א תִנְחָ֔ל וְחֵ֕לֶק לֹא-יִהְיֶ֥ה לְךָ֖ בְּתוֹכָ֑ם אֲנִ֤י חֶלְקְךָ֙ וְנַחֲלָ֣תְךָ֔ בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וַיֹּ֨אמֶר יְהוָ֜ה אֶֽל-אַהֲרֹ֗ן בְּאַרְצָם֙ לֹ֣א תִנְחָ֔ל וְחֵ֕לֶק לֹא-יִהְיֶ֥ה לְךָ֖ בְּתוֹכָ֑ם אֲנִ֤י חֶלְקְךָ֙ וְנַחֲלָ֣תְךָ֔ בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וַאֲמַר יְיָ לְאַהֲרֹן בְּאַרְעֲהוֹן לָא תַחֲסִין וְחוּלַק לָא יְהֵי לָךְ בֵּינֵיהוֹן מַתְּנָן דִי יְהָבִית לָךְ אִנוּן חוּלְקָךְ וְאַחֲסַנְתָּךְ בְּגוֹ בְּנֵי יִשְׂרָאֵל: (ס)

{כא}  שביעי  וְלִבְנֵ֣י לֵוִ֔י הִנֵּ֥ה נָתַ֛תִּי כָּל-מַֽעֲשֵׂ֥ר בְּיִשְׂרָאֵ֖ל לְנַחֲלָ֑ה חֵ֤לֶף עֲבֹֽדָתָם֙ אֲשֶׁר-הֵ֣ם עֹֽבְדִ֔ים אֶת-עֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד:  מ  וְלִבְנֵ֣י לֵוִ֔י הִנֵּ֥ה נָתַ֛תִּי כָּל-מַֽעֲשֵׂ֥ר בְּיִשְׂרָאֵ֖ל לְנַחֲלָ֑ה חֵ֤לֶף עֲבֹֽדָתָם֙ אֲשֶׁר-הֵ֣ם עֹֽבְדִ֔ים אֶת-עֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד:  ת  וְלִבְנֵי לֵוִי הָא יְהָבִית כָּל מַעַשְׂרָא בְּיִשְׂרָאֵל לְאַחֲסָנָא חֲלַף פּוּלְחַנְהוֹן דִי אִינוּן פָּלְחִין יָת פּוּלְחַן מַשְׁכַּן זִמְנָא:

{כב} וְלֹא-יִקְרְב֥וּ ע֛וֹד בְּנֵ֥י יִשְׂרָאֵ֖ל אֶל-אֹ֣הֶל מוֹעֵ֑ד לָשֵׂ֥את חֵ֖טְא לָמֽוּת:  מ  וְלֹא-יִקְרְב֥וּ ע֛וֹד בְּנֵ֥י יִשְׂרָאֵ֖ל אֶל-אֹ֣הֶל מוֹעֵ֑ד לָשֵׂ֥את חֵ֖טְא לָמֽוּת:  ת  וְלָא יִקְרְבוּן עוֹד בְּנֵי יִשְׂרָאֵל לְמַשְׁכַּן זִמְנָא לְקַבָּלָא חוֹבָא לִמְמָת:

{כג} וְעָבַ֨ד הַלֵּוִ֜י ה֗וּא אֶת-עֲבֹדַת֙ אֹ֣הֶל מוֹעֵ֔ד וְהֵ֖ם יִשְׂא֣וּ עֲוֹנָ֑ם חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם וּבְתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לֹ֥א יִנְחֲל֖וּ נַחֲלָֽה:  מ  וְעָבַ֨ד הַלֵּוִ֜י ה֗וּא אֶת-עֲבֹדַת֙ אֹ֣הֶל מוֹעֵ֔ד וְהֵ֖ם יִשְׂא֣וּ עֲוֹנָ֑ם חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם וּבְתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לֹ֥א יִנְחֲל֖וּ נַחֲלָֽה:  ת  וְיִפְלְחוּן לֵוָאֵי אִינוּן יָת פּוּלְחַן מַשְׁכַּן זִמְנָא וְאִינוּן יְקַבְּלוּן חוֹבֵיהוֹן קְיַם עֲלָם לְדָרֵיכוֹן וּבְגוֹ בְּנֵי יִשְׂרָאֵל לָא יַחְסְנוּן אַחְסָנָא:

{כד} כִּ֞י אֶת-מַעְשַׂ֣ר בְּנֵֽי-יִשְׂרָאֵ֗ל אֲשֶׁ֨ר יָרִ֤ימוּ לַֽיהוָה֙ תְּרוּמָ֔ה נָתַ֥תִּי לַלְוִיִּ֖ם לְנַחֲלָ֑ה עַל-כֵּן֙ אָמַ֣רְתִּי לָהֶ֔ם בְּתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לֹ֥א יִנְחֲל֖וּ נַחֲלָֽה:  מ  כִּ֞י אֶת-מַעְשַׂ֣ר בְּנֵֽי-יִשְׂרָאֵ֗ל אֲשֶׁ֨ר יָרִ֤ימוּ לַֽיהוָה֙ תְּרוּמָ֔ה נָתַ֥תִּי לַלְוִיִּ֖ם לְנַחֲלָ֑ה עַל-כֵּן֙ אָמַ֣רְתִּי לָהֶ֔ם בְּתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לֹ֥א יִנְחֲל֖וּ נַחֲלָֽה:  ת  אֲרֵי יָת מַעַשְׂרָא דִבְנֵי יִשְׂרָאֵל דִי יִפְרְשׁוּן קֳדָם יְיָ אַפְרָשׁוּתָא יְהָבִית לְלֵוָאֵי לְאַחֲסָנָא עַל כֵּן אֲמָרִית לְהוֹן בְּגוֹ בְּנֵי יִשְׂרָאֵל לָא יַחְסְנוּן אַחֲסָנָא: (פ)

{כה} וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר:

{כו} וְאֶל-הַלְוִיִּ֣ם תְּדַבֵּר֮ וְאָמַרְתָּ֣ אֲלֵהֶם֒ כִּֽי-תִ֠קְחוּ מֵאֵ֨ת בְּנֵֽי-יִשְׂרָאֵ֜ל אֶת-הַֽמַּעֲשֵׂ֗ר אֲשֶׁ֨ר נָתַ֧תִּי לָכֶ֛ם מֵאִתָּ֖ם בְּנַחֲלַתְכֶ֑ם וַהֲרֵמֹתֶ֤ם מִמֶּ֙נּוּ֙ תְּרוּמַ֣ת יְהוָ֔ה מַעֲשֵׂ֖ר מִן-הַֽמַּעֲשֵֽׂר:  מ  וְאֶל-הַלְוִיִּ֣ם תְּדַבֵּר֮ וְאָמַרְתָּ֣ אֲלֵהֶם֒ כִּֽי-תִ֠קְחוּ מֵאֵ֨ת בְּנֵֽי-יִשְׂרָאֵ֜ל אֶת-הַֽמַּעֲשֵׂ֗ר אֲשֶׁ֨ר נָתַ֧תִּי לָכֶ֛ם מֵאִתָּ֖ם בְּנַחֲלַתְכֶ֑ם וַהֲרֵמֹתֶ֤ם מִמֶּ֙נּוּ֙ תְּרוּמַ֣ת יְהוָ֔ה מַעֲשֵׂ֖ר מִן-הַֽמַּעֲשֵֽׂר:  ת  וּלְלֵוָאֵי תְּמַלֵיל וְתֵימַר לְהוֹן אֲרֵי תִסְבוּן מִן בְּנֵי יִשְׂרָאֵל יָת מַעַשְׂרָא דִי יְהָבִית לְכוֹן מִנְהוֹן בְּאַחֲסַנְתְּכוֹן וְתַפְרְשׁוּן מִנֵיהּ אַפְרָשׁוּתָא קֳדָם יְיָ מַעַשְׂרָא מִן מַעַשְׂרָא:

{כז} וְנֶחְשַׁ֥ב לָכֶ֖ם תְּרוּמַתְכֶ֑ם כַּדָּגָן֙ מִן-הַגֹּ֔רֶן וְכַֽמְלֵאָ֖ה מִן-הַיָּֽקֶב:  מ  וְנֶחְשַׁ֥ב לָכֶ֖ם תְּרוּמַתְכֶ֑ם כַּדָּגָן֙ מִן-הַגֹּ֔רֶן וְכַֽמְלֵאָ֖ה מִן-הַיָּֽקֶב:  ת  וְתִתְחַשֵׁב לְכוֹן אַפְרָשׁוּתְכוֹן כְּעִבּוּרָא מִן אִדְרָא וְכִמְלֵאָתָא מִן מַעְצַרְתָּא:

{כח} כֵּ֣ן תָּרִ֤ימוּ גַם-אַתֶּם֙ תְּרוּמַ֣ת יְהוָ֔ה מִכֹּל֙ מַעְשְׂרֹ֣תֵיכֶ֔ם אֲשֶׁ֣ר תִּקְח֔וּ מֵאֵ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל וּנְתַתֶּ֤ם מִמֶּ֙נּוּ֙ אֶת-תְּרוּמַ֣ת יְהוָ֔ה לְאַהֲרֹ֖ן הַכֹּהֵֽן:  מ  כֵּ֣ן תָּרִ֤ימוּ גַם-אַתֶּם֙ תְּרוּמַ֣ת יְהוָ֔ה מִכֹּל֙ מַעְשְׂרֹ֣תֵיכֶ֔ם אֲשֶׁ֣ר תִּקְח֔וּ מֵאֵ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל וּנְתַתֶּ֤ם מִמֶּ֙נּוּ֙ אֶת-תְּרוּמַ֣ת יְהוָ֔ה לְאַהֲרֹ֖ן הַכֹּהֵֽן:  ת  כֵּן תַּפְרְשׁוּן אַף אַתּוּן אַפְרָשׁוּתָא קֳדָם יְיָ מִכֹּל מַעַשְׂרָתֵיכוֹן דִי תִסְבוּן מִן בְּנֵי יִשְׂרָאֵל וְתִתְּנוּן מִנֵיהּ יָת אַפְרָשׁוּתָא קֳדָם יְיָ לְאַהֲרֹן כַּהֲנָא:

{כט} מִכֹּל֙ מַתְּנֹ֣תֵיכֶ֔ם תָּרִ֕ימוּ אֵ֖ת כָּל-תְּרוּמַ֣ת יְהוָ֑ה מִכָּל-חֶלְבּ֔וֹ אֶֽת-מִקְדְּשׁ֖וֹ מִמֶּֽנּוּ:  מ  מִכֹּל֙ מַתְּנֹ֣תֵיכֶ֔ם תָּרִ֕ימוּ אֵ֖ת כָּל-תְּרוּמַ֣ת יְהוָ֑ה מִכָּל-חֶלְבּ֔וֹ אֶֽת-מִקְדְּשׁ֖וֹ מִמֶּֽנּוּ:  ת  מִכֹּל מַתְּנָתֵכוֹן תִּפְרְשׁוּן יָת כָּל אַפְרָשׁוּתָא דַיְיָ מִכָּל שׁוּפְרֵיהּ יָת מַקְדְשֵׁיהּ מִנֵיהּ:

{ל}  מפטיר  וְאָמַרְתָּ֖ אֲלֵהֶ֑ם בַּהֲרִֽימְכֶ֤ם אֶת-חֶלְבּוֹ֙ מִמֶּ֔נּוּ וְנֶחְשַׁב֙ לַלְוִיִּ֔ם כִּתְבוּאַ֥ת גֹּ֖רֶן וְכִתְבוּאַ֥ת יָֽקֶב:  מ  וְאָמַרְתָּ֖ אֲלֵהֶ֑ם בַּהֲרִֽימְכֶ֤ם אֶת-חֶלְבּוֹ֙ מִמֶּ֔נּוּ וְנֶחְשַׁב֙ לַלְוִיִּ֔ם כִּתְבוּאַ֥ת גֹּ֖רֶן וְכִתְבוּאַ֥ת יָֽקֶב:  ת  וְתֵימַר לְהוֹן בְּאַפְרָשׁוּתְכוֹן יָת שׁוּפְרֵיהּ מִנֵיהּ וְיִתְחַשֵׁב לְלֵּוָאֵי כַּעֲלָלַת אִדְרָא וְכַעֲלָלַת מַעֲצַרְתָּא:

{לא} וַאֲכַלְתֶּ֤ם אֹתוֹ֙ בְּכָל-מָק֔וֹם אַתֶּ֖ם וּבֵֽיתְכֶ֑ם כִּֽי-שָׂכָ֥ר הוּא֙ לָכֶ֔ם חֵ֥לֶף עֲבֹֽדַתְכֶ֖ם בְּאֹ֥הֶל מוֹעֵֽד:  מ  וַאֲכַלְתֶּ֤ם אֹתוֹ֙ בְּכָל-מָק֔וֹם אַתֶּ֖ם וּבֵֽיתְכֶ֑ם כִּֽי-שָׂכָ֥ר הוּא֙ לָכֶ֔ם חֵ֥לֶף עֲבֹֽדַתְכֶ֖ם בְּאֹ֥הֶל מוֹעֵֽד:  ת  וְתֵיכְלוּן יָתֵיהּ בְּכָל אֲתַר אַתּוּן וֶאֱנַשׁ בָּתֵּיכוֹן אֲרֵי אַגְרָא הוּא לְכוֹן חֲלַף פָּלְחַנְכוֹן בְּמַשְׁכַּן זִמְנָא:

{לב} וְלֹֽא-תִשְׂא֤וּ עָלָיו֙ חֵ֔טְא בַּהֲרִֽימְכֶ֥ם אֶת-חֶלְבּ֖וֹ מִמֶּ֑נּוּ וְאֶת-קָדְשֵׁ֧י בְנֵי-יִשְׂרָאֵ֛ל לֹ֥א תְחַלְּל֖וּ וְלֹ֥א תָמֽוּתוּ:  מ  וְלֹֽא-תִשְׂא֤וּ עָלָיו֙ חֵ֔טְא בַּהֲרִֽימְכֶ֥ם אֶת-חֶלְבּ֖וֹ מִמֶּ֑נּוּ וְאֶת-קָדְשֵׁ֧י בְנֵי-יִשְׂרָאֵ֛ל לֹ֥א תְחַלְּל֖וּ וְלֹ֥א תָמֽוּתוּ:  ת  וְלָא תְקַבְּלוּן עֲלוֹהִי חוֹבָא בְּאַפְרָשׁוּתְכוֹן יָת שׁוּפְרֵיהּ מִנֵיהּ וְיָת קוּדְשַׁיָא דִבְנֵי יִשְׂרָאֵל לָא תְחַלְלוּן וְלָא תְמוּתוּן: (פפפ)
