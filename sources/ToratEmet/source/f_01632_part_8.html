פרשת נצבים


{ט} אַתֶּם נִצָּבִים הַיּוֹם כֻּלְּכֶם לִפְנֵי יְהוָה אֱלֹהֵיכֶם רָאשֵׁיכֶם שִׁבְטֵיכֶם זִקְנֵיכֶם וְשֹׁטְרֵיכֶם כֹּל אִישׁ יִשְׂרָאֵל:  מ  אַתֶּם נִצָּבִים הַיּוֹם כֻּלְּכֶם לִפְנֵי יְהוָה אֱלֹהֵיכֶם רָאשֵׁיכֶם שִׁבְטֵיכֶם זִקְנֵיכֶם וְשֹׁטְרֵיכֶם כֹּל אִישׁ יִשְׂרָאֵל:  ת  אַתּוּן קָיְמִין יוֹמָא דֵין כֻּלְכוֹן קֳדָם יְיָ אֱלָהֲכוֹן רֵישֵׁיכוֹן שִׁבְטֵיכוֹן סָבֵיכוֹן וְסָרְכֵיכוֹן כָּל אֱנַשׁ יִשְׂרָאֵל:

{י} טַפְּכֶם נְשֵׁיכֶם וְגֵרְךָ אֲשֶׁר בְּקֶרֶב מַחֲנֶיךָ מֵחֹטֵב עֵצֶיךָ עַד שֹׁאֵב מֵימֶיךָ:  מ  טַפְּכֶם נְשֵׁיכֶם וְגֵרְךָ אֲשֶׁר בְּקֶרֶב מַחֲנֶיךָ מֵחֹטֵב עֵצֶיךָ עַד שֹׁאֵב מֵימֶיךָ:  ת  טַפְלְכוֹן נְשֵׁיכוֹן וְגִיוֹרָךְ דִי בְּגוֹ מַשְׁרִיתָךְ מִלָקֵטּ אָעָיךְ עַד מְלֵי מַיָיךְ:

{יא} לְעָבְרְךָ בִּבְרִית יְהוָה אֱלֹהֶיךָ וּבְאָלָתוֹ אֲשֶׁר יְהוָה אֱלֹהֶיךָ כֹּרֵת עִמְּךָ הַיּוֹם:  מ  לְעָבְרְךָ בִּבְרִית יְהוָה אֱלֹהֶיךָ וּבְאָלָתוֹ אֲשֶׁר יְהוָה אֱלֹהֶיךָ כֹּרֵת עִמְּךָ הַיּוֹם:  ת  לְאָעָלוּתָךְ בִּקְיָמָא דַיְיָ אֱלָהָךְ וּבְמוֹמָתֵהּ דִי יְיָ אֱלָהָךְ גְזַר עִמָךְ יוֹמָא דֵין:

{יב}  שני  לְמַעַן הָקִים אֹתְךָ הַיּוֹם לוֹ לְעָם וְהוּא יִהְיֶה לְּךָ לֵאלֹהִים כַּאֲשֶׁר דִּבֶּר לָךְ וְכַאֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב:  מ  לְמַעַן הָקִים אֹתְךָ הַיּוֹם לוֹ לְעָם וְהוּא יִהְיֶה לְּךָ לֵאלֹהִים כַּאֲשֶׁר דִּבֶּר לָךְ וְכַאֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב:  ת  בְּדִיל לַאֲקָמָא יָתָךְ יוֹמָא דֵין קֳדָמוֹהִי לְעַמָא וְהוּא יְהֵי לָךְ לֶאֶלָהּ כְּמָא דִי מַלִיל לָךְ וּכְמָא דִי קַיִים לַאֲבָהָתָךְ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב:

{יג} וְלֹא אִתְּכֶם לְבַדְּכֶם אָנֹכִי כֹּרֵת אֶת הַבְּרִית הַזֹּאת וְאֶת הָאָלָה הַזֹּאת:  מ  וְלֹא אִתְּכֶם לְבַדְּכֶם אָנֹכִי כֹּרֵת אֶת הַבְּרִית הַזֹּאת וְאֶת הָאָלָה הַזֹּאת:  ת  וְלָא עִמְכוֹן בִּלְחוֹדֵיכוֹן אֲנָא גְזַר יָת קְיָמָא הָדָא וְיָת מוֹמָתָא הָדָא:

{יד} כִּי אֶת אֲשֶׁר יֶשְׁנוֹ פֹּה עִמָּנוּ עֹמֵד הַיּוֹם לִפְנֵי יְהוָה אֱלֹהֵינוּ וְאֵת אֲשֶׁר אֵינֶנּוּ פֹּה עִמָּנוּ הַיּוֹם:  מ  כִּי אֶת אֲשֶׁר יֶשְׁנוֹ פֹּה עִמָּנוּ עֹמֵד הַיּוֹם לִפְנֵי יְהוָה אֱלֹהֵינוּ וְאֵת אֲשֶׁר אֵינֶנּוּ פֹּה עִמָּנוּ הַיּוֹם:  ת  אֲרֵי יָת מָן דְאִיתוֹהִי הָכָא עִמָנָא קָאֵם יוֹמָא דֵין קֳדָם יְיָ אֱלָהָנָא וְיָת מָן דְלֵיתוֹהִי הָכָא עִמָנָא יוֹמָא דֵין:

{טו}  שלישי  כִּי אַתֶּם יְדַעְתֶּם אֵת אֲשֶׁר יָשַׁבְנוּ בְּאֶרֶץ מִצְרָיִם וְאֵת אֲשֶׁר עָבַרְנוּ בְּקֶרֶב הַגּוֹיִם אֲשֶׁר עֲבַרְתֶּם:  מ  כִּי אַתֶּם יְדַעְתֶּם אֵת אֲשֶׁר יָשַׁבְנוּ בְּאֶרֶץ מִצְרָיִם וְאֵת אֲשֶׁר עָבַרְנוּ בְּקֶרֶב הַגּוֹיִם אֲשֶׁר עֲבַרְתֶּם:  ת  אֲרֵי אַתּוּן יְדַעְתּוּן יָת דִיתֵבְנָא בְּאַרְעָא דְמִצְרָיִם וְיָת דִי עֲבַרְנָא בְּגוֹ עַמְמַיָא דִי עֲבַרְתּוּן:

{טז} וַתִּרְאוּ אֶת שִׁקּוּצֵיהֶם וְאֵת גִּלֻּלֵיהֶם עֵץ וָאֶבֶן כֶּסֶף וְזָהָב אֲשֶׁר עִמָּהֶם:  מ  וַתִּרְאוּ אֶת שִׁקּוּצֵיהֶם וְאֵת גִּלֻּלֵיהֶם עֵץ וָאֶבֶן כֶּסֶף וְזָהָב אֲשֶׁר עִמָּהֶם:  ת  וַחֲזֵיתוּן יָת שִׁקוּצֵיהוֹן וְיָת טַעֲוָתְהוֹן אָעָא וְאַבְנָא כַּסְפָּא וְדַהֲבָא דִי עִמְהוֹן:

{יז} פֶּן יֵשׁ בָּכֶם אִישׁ אוֹ אִשָּׁה אוֹ מִשְׁפָּחָה אוֹ שֵׁבֶט אֲשֶׁר לְבָבוֹ פֹנֶה הַיּוֹם מֵעִם יְהוָה אֱלֹהֵינוּ לָלֶכֶת לַעֲבֹד אֶת אֱלֹהֵי הַגּוֹיִם הָהֵם פֶּן יֵשׁ בָּכֶם שֹׁרֶשׁ פֹּרֶה רֹאשׁ וְלַעֲנָה:  מ  פֶּן יֵשׁ בָּכֶם אִישׁ אוֹ אִשָּׁה אוֹ מִשְׁפָּחָה אוֹ שֵׁבֶט אֲשֶׁר לְבָבוֹ פֹנֶה הַיּוֹם מֵעִם יְהוָה אֱלֹהֵינוּ לָלֶכֶת לַעֲבֹד אֶת אֱלֹהֵי הַגּוֹיִם הָהֵם פֶּן יֵשׁ בָּכֶם שֹׁרֶשׁ פֹּרֶה רֹאשׁ וְלַעֲנָה:  ת  דִילְמָא אִית בְּכוֹן גְבַר אוֹ אִתְּתָא אוֹ זַרְעִית אוֹ שִׁבְטָא דִי לִבֵּהּ פָּנֵי יוֹמָא דֵין מִדַחַלְתָּא דַייָ אֱלָהָנָא לִמְהַךְ לְמִפְלַח יָת טַעֲוַת עַמְמַיָא הָאִנוּן דִילְמָא אִית בְּכוֹן גְבַר מְהַרְהַר חֶטְאִין אוֹ זָדוֹן:

{יח} וְהָיָה בְּשָׁמְעוֹ אֶת דִּבְרֵי הָאָלָה הַזֹּאת וְהִתְבָּרֵךְ בִּלְבָבוֹ לֵאמֹר שָׁלוֹם יִהְיֶה לִּי כִּי בִּשְׁרִרוּת לִבִּי אֵלֵךְ לְמַעַן סְפוֹת הָרָוָה אֶת הַצְּמֵאָה:  מ  וְהָיָה בְּשָׁמְעוֹ אֶת דִּבְרֵי הָאָלָה הַזֹּאת וְהִתְבָּרֵךְ בִּלְבָבוֹ לֵאמֹר שָׁלוֹם יִהְיֶה לִּי כִּי בִּשְׁרִרוּת לִבִּי אֵלֵךְ לְמַעַן סְפוֹת הָרָוָה אֶת הַצְּמֵאָה:  ת  וִיהֵי בְּמִשְׁמְעֵהּ יָת פִּתְגָמֵי מוֹמָתָא הָדָא וִיחַשַׁב בְּלִבֵּהּ לְמֵימַר שְׁלָמָא יְהֵי לִי אֲרֵי בְּהַרְהוֹר לִבִּי אֲנָא אָזֵל בְּדִיל לְאוֹסָפָא לֵהּ חֶטְאֵי שָׁלוּתָא עַל זְדָנוּתָא:

{יט} לֹא יֹאבֶה יְהוָה סְלֹחַ לוֹ כִּי אָז יֶעְשַׁן אַף יְהוָה וְקִנְאָתוֹ בָּאִישׁ הַהוּא וְרָבְצָה בּוֹ כָּל הָאָלָה הַכְּתוּבָה בַּסֵּפֶר הַזֶּה וּמָחָה יְהוָה אֶת שְׁמוֹ מִתַּחַת הַשָּׁמָיִם:  מ  לֹא יֹאבֶה יְהוָה סְלֹחַ לוֹ כִּי אָז יֶעְשַׁן אַף יְהוָה וְקִנְאָתוֹ בָּאִישׁ הַהוּא וְרָבְצָה בּוֹ כָּל הָאָלָה הַכְּתוּבָה בַּסֵּפֶר הַזֶּה וּמָחָה יְהוָה אֶת שְׁמוֹ מִתַּחַת הַשָּׁמָיִם:  ת  לָא יֵיבֵי יְיָ לְמִשְׁבַּק לֵהּ אֲרֵי בְּכֵן יִתְקֵף רָגְזָא דַיְיָ וְחֶמְתֵהּ בְּגַבְרָא הַהוּא וְיִדְבְּקוּן בֵּהּ כָּל לְוָטַּיָא דִכְתִיבִין בְּסִפְרָא הָדֵין וְיִמְחֵי יְיָ יָת שְׁמֵהּ מִתְּחוֹת שְׁמַיָא:

{כ} וְהִבְדִּילוֹ יְהוָה לְרָעָה מִכֹּל שִׁבְטֵי יִשְׂרָאֵל כְּכֹל אָלוֹת הַבְּרִית הַכְּתוּבָה בְּסֵפֶר הַתּוֹרָה הַזֶּה:  מ  וְהִבְדִּילוֹ יְהוָה לְרָעָה מִכֹּל שִׁבְטֵי יִשְׂרָאֵל כְּכֹל אָלוֹת הַבְּרִית הַכְּתוּבָה בְּסֵפֶר הַתּוֹרָה הַזֶּה:  ת  וְיַפְרְשִׁנֵהּ יְיָ לְבִישָׁא מִכֹּל שִׁבְטַיָא דְיִשְׂרָאֵל כְּכֹל לְוָטֵי קְיָמָא דִכְתִיבִין בְּסִפְרָא דְאוֹרַיְתָא הָדֵין:

{כא} וְאָמַר הַדּוֹר הָאַחֲרוֹן בְּנֵיכֶם אֲשֶׁר יָקוּמוּ מֵאַחֲרֵיכֶם וְהַנָּכְרִי אֲשֶׁר יָבֹא מֵאֶרֶץ רְחוֹקָה וְרָאוּ אֶת מַכּוֹת הָאָרֶץ הַהִוא וְאֶת תַּחֲלֻאֶיהָ אֲשֶׁר חִלָּה יְהוָה בָּהּ:  מ  וְאָמַר הַדּוֹר הָאַחֲרוֹן בְּנֵיכֶם אֲשֶׁר יָקוּמוּ מֵאַחֲרֵיכֶם וְהַנָּכְרִי אֲשֶׁר יָבֹא מֵאֶרֶץ רְחוֹקָה וְרָאוּ אֶת מַכּוֹת הָאָרֶץ הַהִוא וְאֶת תַּחֲלֻאֶיהָ אֲשֶׁר חִלָּה יְהוָה בָּהּ:  ת  וְיֵימַר דָרָא בַתְרָאָה בְּנֵיכוֹן דִי יְקוּמוּן מִבַּתְרֵיכוֹן וּבַר עַמְמִין דִי יֵיתֵי מֵאַרְעָא רְחִיקָא וְיֶחֱזוּן יָת מְחָתָא דְאַרְעָא הַהִיא וְיָת מַרְעָהָא דִי אַמְרַע יְיָ בַּהּ:

{כב} גָּפְרִית וָמֶלַח שְׂרֵפָה כָל אַרְצָהּ לֹא תִזָּרַע וְלֹא תַצְמִחַ וְלֹא יַעֲלֶה בָהּ כָּל עֵשֶׂב כְּמַהְפֵּכַת סְדֹם וַעֲמֹרָה אַדְמָה (וצביים) וּצְבוֹיִם אֲשֶׁר הָפַךְ יְהוָה בְּאַפּוֹ וּבַחֲמָתוֹ:  מ  גָּפְרִית וָמֶלַח שְׂרֵפָה כָל אַרְצָהּ לֹא תִזָּרַע וְלֹא תַצְמִחַ וְלֹא יַעֲלֶה בָהּ כָּל עֵשֶׂב כְּמַהְפֵּכַת סְדֹם וַעֲמֹרָה אַדְמָה (וצביים) וּצְבוֹיִם אֲשֶׁר הָפַךְ יְהוָה בְּאַפּוֹ וּבַחֲמָתוֹ:  ת  גָפְרֵתָא וּמִלְחָא יְקִידַת כָּל אַרְעַהּ לָא תִזְרַע וְלָא תַצְמַח וְלָא יִסַק בַּהּ כָּל עֶסְבָּא כְּמַהְפֶּכְתָּא דִסְדֹם וַעֲמֹרָה אַדְמָה וּצְבוֹיִם דִי הֲפַךְ יְיָ בְּרָגְזֵהּ וּבְחֶמְתֵהּ:

{כג} וְאָמְרוּ כָּל הַגּוֹיִם עַל מֶה עָשָׂה יְהוָה כָּכָה לָאָרֶץ הַזֹּאת מֶה חֳרִי הָאַף הַגָּדוֹל הַזֶּה:  מ  וְאָמְרוּ כָּל הַגּוֹיִם עַל מֶה עָשָׂה יְהוָה כָּכָה לָאָרֶץ הַזֹּאת מֶה חֳרִי הָאַף הַגָּדוֹל הַזֶּה:  ת  וְיֵימְרוּן כָּל עַמְמַיָא עַל מֶה עֲבַד יְיָ כְּדֵין לְאַרְעָא הָדָא מָא תְקוֹף רָגְזָא רַבָּא הָדֵין:

{כד} וְאָמְרוּ עַל אֲשֶׁר עָזְבוּ אֶת בְּרִית יְהוָה אֱלֹהֵי אֲבֹתָם אֲשֶׁר כָּרַת עִמָּם בְּהוֹצִיאוֹ אֹתָם מֵאֶרֶץ מִצְרָיִם:  מ  וְאָמְרוּ עַל אֲשֶׁר עָזְבוּ אֶת בְּרִית יְהוָה אֱלֹהֵי אֲבֹתָם אֲשֶׁר כָּרַת עִמָּם בְּהוֹצִיאוֹ אֹתָם מֵאֶרֶץ מִצְרָיִם:  ת  וְיֵימְרוּן עַל דִשְׁבָקוּ יָת קְיָמָא דַיְיָ אֱלָהָא דַאֲבָהָתְהוֹן דִי גְזַר עִמְהוֹן בְּאַפָּקוּתֵהּ יָתְהוֹן מֵאַרְעָא דְמִצְרָיִם:

{כה} וַיֵּלְכוּ וַיַּעַבְדוּ אֱלֹהִים אֲחֵרִים וַיִּשְׁתַּחֲוּוּ לָהֶם אֱלֹהִים אֲשֶׁר לֹא יְדָעוּם וְלֹא חָלַק לָהֶם:  מ  וַיֵּלְכוּ וַיַּעַבְדוּ אֱלֹהִים אֲחֵרִים וַיִּשְׁתַּחֲוּוּ לָהֶם אֱלֹהִים אֲשֶׁר לֹא יְדָעוּם וְלֹא חָלַק לָהֶם:  ת  וַאֲזָלוּ וּפְלָחוּ לְטַעֲוַת עַמְמַיָא וּסְגִידוּ לְהוֹן דַחֲלָן דִי לָא יְדָעֻנוּן וְלָא אוֹטִיבָא לְהוֹן:

{כו} וַיִּחַר אַף יְהוָה בָּאָרֶץ הַהִוא לְהָבִיא עָלֶיהָ אֶת כָּל הַקְּלָלָה הַכְּתוּבָה בַּסֵּפֶר הַזֶּה:  מ  וַיִּחַר אַף יְהוָה בָּאָרֶץ הַהִוא לְהָבִיא עָלֶיהָ אֶת כָּל הַקְּלָלָה הַכְּתוּבָה בַּסֵּפֶר הַזֶּה:  ת  וּתְקֵף רָגְזָא דַיְיָ בְּאַרְעָא הַהִיא לְאַיְתָאָה עֲלַהּ יָת כָּל לְוָטַיָא דִכְתִיבִין בְּסִפְרָא הָדֵין:

{כז} וַיִּתְּשֵׁם יְהוָה מֵעַל אַדְמָתָם בְּאַף וּבְחֵמָה וּבְקֶצֶף גָּדוֹל וַיַּשְׁלִכֵם אֶל אֶרֶץ אַחֶרֶת כַּיּוֹם הַזֶּה:  מ  וַיִּתְּשֵׁם יְהוָה מֵעַל אַדְמָתָם בְּאַף וּבְחֵמָה וּבְקֶצֶף גָּדוֹל וַיַּשְׁלִכֵם אֶל אֶרֶץ אַחֶרֶת כַּיּוֹם הַזֶּה:  ת  וְטַּלְטְּלִנוּן יְיָ מֵעַל אַרְעֲהוֹן בִּרְגַז וּבְחֵמְתָא וּבִתְקוֹף רְגַז וְאַגְלִנוּן לְאַרְעָא אָחֳרִי כְּיוֹמָא הָדֵין:

{כח} הַנִּסְתָּרֹת לַיהוָה אֱלֹהֵינוּ וְהַנִּגְלֹת לָנוּ וּלְבָנֵינוּ עַד עוֹלָם לַעֲשׂוֹת אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת:  מ  הַנִּסְתָּרֹת לַיהוָה אֱלֹהֵינוּ וְהַנִּגְלֹת לָנוּ וּלְבָנֵינוּ עַד עוֹלָם לַעֲשׂוֹת אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת:  ת  דְמִטַמְרָן קֳדָם יְיָ אֱלָהָנָא וּדְמִגַלְיָן לָנָא וְלִבְנָנָא עַד עֳלָם לְמֶעְבַּד יָת כָּל פִּתְגָמֵי אוֹרַיְתָא הָדָא: (ס)




דברים פרק-ל

{א}  רביעי - במחוברות שני  וְהָיָה כִי יָבֹאוּ עָלֶיךָ כָּל הַדְּבָרִים הָאֵלֶּה הַבְּרָכָה וְהַקְּלָלָה אֲשֶׁר נָתַתִּי לְפָנֶיךָ וַהֲשֵׁבֹתָ אֶל לְבָבֶךָ בְּכָל הַגּוֹיִם אֲשֶׁר הִדִּיחֲךָ יְהוָה אֱלֹהֶיךָ שָׁמָּה:  מ  וְהָיָה כִי יָבֹאוּ עָלֶיךָ כָּל הַדְּבָרִים הָאֵלֶּה הַבְּרָכָה וְהַקְּלָלָה אֲשֶׁר נָתַתִּי לְפָנֶיךָ וַהֲשֵׁבֹתָ אֶל לְבָבֶךָ בְּכָל הַגּוֹיִם אֲשֶׁר הִדִּיחֲךָ יְהוָה אֱלֹהֶיךָ שָׁמָּה:  ת  וִיהֵי אֲרֵי יֵיתוּן עֲלָךְ כָּל פִּתְגָמַיָא הָאִלֵין בִּרְכָן וּלְוָטִין דִי יְהָבִית קֳדָמָךְ וּתְתוּב לְלִבָּךְ בְּכָל עַמְמַיָא דִי אַגְלָךְ יְיָ אֱלָהָךְ לְתַמָן:

{ב} וְשַׁבְתָּ עַד יְהוָה אֱלֹהֶיךָ וְשָׁמַעְתָּ בְקֹלוֹ כְּכֹל אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם אַתָּה וּבָנֶיךָ בְּכָל לְבָבְךָ וּבְכָל נַפְשֶׁךָ:  מ  וְשַׁבְתָּ עַד יְהוָה אֱלֹהֶיךָ וְשָׁמַעְתָּ בְקֹלוֹ כְּכֹל אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם אַתָּה וּבָנֶיךָ בְּכָל לְבָבְךָ וּבְכָל נַפְשֶׁךָ:  ת  וּתְתוּב לְדַחַלְתָּא דַיְיָ אֱלָהָךְ וּתְקַבֵּל לְמֵימְרֵהּ כְּכֹל דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין אַתְּ וּבְנָיךְ בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ:

{ג} וְשָׁב יְהוָה אֱלֹהֶיךָ אֶת שְׁבוּתְךָ וְרִחֲמֶךָ וְשָׁב וְקִבֶּצְךָ מִכָּל הָעַמִּים אֲשֶׁר הֱפִיצְךָ יְהוָה אֱלֹהֶיךָ שָׁמָּה:  מ  וְשָׁב יְהוָה אֱלֹהֶיךָ אֶת שְׁבוּתְךָ וְרִחֲמֶךָ וְשָׁב וְקִבֶּצְךָ מִכָּל הָעַמִּים אֲשֶׁר הֱפִיצְךָ יְהוָה אֱלֹהֶיךָ שָׁמָּה:  ת  וִיתוּב יְיָ אֱלָהָךְ יָת שְׁבִי גַלְוָתָךְ וִירַחַם עֲלָךְ וִיתוּב וְיִכְנְשִׁנָךְ מִכָּל עַמְמַיָא דִי בַדְרָךְ יְיָ אֱלָהָךְ לְתַמָן:

{ד} אִם יִהְיֶה נִדַּחֲךָ בִּקְצֵה הַשָּׁמָיִם מִשָּׁם יְקַבֶּצְךָ יְהוָה אֱלֹהֶיךָ וּמִשָּׁם יִקָּחֶךָ:  מ  אִם יִהְיֶה נִדַּחֲךָ בִּקְצֵה הַשָּׁמָיִם מִשָּׁם יְקַבֶּצְךָ יְהוָה אֱלֹהֶיךָ וּמִשָּׁם יִקָּחֶךָ:  ת  אִם יְהֵי גַלְוָתָךְ בִּסְיָפֵי שְׁמַיָא מִתַּמָן יִכְנְשִׁנָךְ יְיָ אֱלָהָךְ וּמִתַּמָן יְקָרְבִנָךְ:

{ה} וֶהֱבִיאֲךָ יְהוָה אֱלֹהֶיךָ אֶל הָאָרֶץ אֲשֶׁר יָרְשׁוּ אֲבֹתֶיךָ וִירִשְׁתָּהּ וְהֵיטִבְךָ וְהִרְבְּךָ מֵאֲבֹתֶיךָ:  מ  וֶהֱבִיאֲךָ יְהוָה אֱלֹהֶיךָ אֶל הָאָרֶץ אֲשֶׁר יָרְשׁוּ אֲבֹתֶיךָ וִירִשְׁתָּהּ וְהֵיטִבְךָ וְהִרְבְּךָ מֵאֲבֹתֶיךָ:  ת  וְיָעֵלִנָךְ יְיָ אֱלָהָךְ לְאַרְעָא דִירִיתוּ אֲבָהָתָךְ וְתֵרְתַהּ וְיוֹטֶב לָךְ וְיַסְגִנָךְ מֵאֲבְהָתָךְ:

{ו} וּמָל יְהוָה אֱלֹהֶיךָ אֶת לְבָבְךָ וְאֶת לְבַב זַרְעֶךָ לְאַהֲבָה אֶת יְהוָה אֱלֹהֶיךָ בְּכָל לְבָבְךָ וּבְכָל נַפְשְׁךָ לְמַעַן חַיֶּיךָ:  מ  וּמָל יְהוָה אֱלֹהֶיךָ אֶת לְבָבְךָ וְאֶת לְבַב זַרְעֶךָ לְאַהֲבָה אֶת יְהוָה אֱלֹהֶיךָ בְּכָל לְבָבְךָ וּבְכָל נַפְשְׁךָ לְמַעַן חַיֶּיךָ:  ת  וְיֶעְדֵי יְיָ אֱלָהָךְ יָת טַפְשׁוּת לִבָּךְ וְיָת טַּפְשׁוּת לִבָּא דִבְנָיךְ לְמִרְחַם יָת יְיָ אֱלָהָךְ בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ בְּדִיל חַיָיךְ:

{ז}  חמישי - במחוברות שלישי  וְנָתַן יְהוָה אֱלֹהֶיךָ אֵת כָּל הָאָלוֹת הָאֵלֶּה עַל אֹיְבֶיךָ וְעַל שֹׂנְאֶיךָ אֲשֶׁר רְדָפוּךָ:  מ  וְנָתַן יְהוָה אֱלֹהֶיךָ אֵת כָּל הָאָלוֹת הָאֵלֶּה עַל אֹיְבֶיךָ וְעַל שֹׂנְאֶיךָ אֲשֶׁר רְדָפוּךָ:  ת  וְיִתֵּן יְיָ אֱלָהָךְ יָת כָּל לְוָטַיָא הָאִלֵין עַל בַּעֲלֵי דְבָבָךְ וְעַל סָנְאָךְ דִי רְדָפוּךְ:

{ח} וְאַתָּה תָשׁוּב וְשָׁמַעְתָּ בְּקוֹל יְהוָה וְעָשִׂיתָ אֶת כָּל מִצְוֹתָיו אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם:  מ  וְאַתָּה תָשׁוּב וְשָׁמַעְתָּ בְּקוֹל יְהוָה וְעָשִׂיתָ אֶת כָּל מִצְוֹתָיו אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם:  ת  וְאַתְּ תְּתוּב וּתְקַבַּל לְמֵימְרָא דַיְיָ וְתַעְבֵּד יָת כָּל פִּקוּדוֹהִי דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין:

{ט} וְהוֹתִירְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל מַעֲשֵׂה יָדֶךָ בִּפְרִי בִטְנְךָ וּבִפְרִי בְהֶמְתְּךָ וּבִפְרִי אַדְמָתְךָ לְטוֹבָה כִּי יָשׁוּב יְהוָה לָשׂוּשׂ עָלֶיךָ לְטוֹב כַּאֲשֶׁר שָׂשׂ עַל אֲבֹתֶיךָ:  מ  וְהוֹתִירְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל מַעֲשֵׂה יָדֶךָ בִּפְרִי בִטְנְךָ וּבִפְרִי בְהֶמְתְּךָ וּבִפְרִי אַדְמָתְךָ לְטוֹבָה כִּי יָשׁוּב יְהוָה לָשׂוּשׂ עָלֶיךָ לְטוֹב כַּאֲשֶׁר שָׂשׂ עַל אֲבֹתֶיךָ:  ת  וְיוֹתְרִינָךְ יְיָ אֱלָהָךְ בְּכֹל עוֹבָדֵי יְדָךְ בְּוַלְדָא דִמְעָךְ וּבְוַלְדָא דִבְעִירָךְ וּבְאִבָּא דְאַרְעָךְ לְטָּבָא אֲרֵי יְתוּב יְיָ לְמֶחֱדֵי עֲלָךְ לְטָב כְּמָא דִי חֲדִי עַל אֲבָהָתָךְ:

{י} כִּי תִשְׁמַע בְּקוֹל יְהוָה אֱלֹהֶיךָ לִשְׁמֹר מִצְוֹתָיו וְחֻקֹּתָיו הַכְּתוּבָה בְּסֵפֶר הַתּוֹרָה הַזֶּה כִּי תָשׁוּב אֶל יְהוָה אֱלֹהֶיךָ בְּכָל לְבָבְךָ וּבְכָל נַפְשֶׁךָ:  מ  כִּי תִשְׁמַע בְּקוֹל יְהוָה אֱלֹהֶיךָ לִשְׁמֹר מִצְוֹתָיו וְחֻקֹּתָיו הַכְּתוּבָה בְּסֵפֶר הַתּוֹרָה הַזֶּה כִּי תָשׁוּב אֶל יְהוָה אֱלֹהֶיךָ בְּכָל לְבָבְךָ וּבְכָל נַפְשֶׁךָ:  ת  אֲרֵי תְקַבַּל בְּמֵימְרָא דַיְיָ אֱלָהָךְ לְמִטַר פִּקוּדוֹהִי וּקְיָמוֹהִי דִכְתִיבִין בְּסִפְרָא דְאוֹרַיְתָא הָדֵין אֲרֵי תְתוּב קֳדָם יְיָ אֱלָהָךְ בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ: (ס)

{יא}  שישי  כִּי הַמִּצְוָה הַזֹּאת אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לֹא נִפְלֵאת הִוא מִמְּךָ וְלֹא רְחֹקָה הִוא:  מ  כִּי הַמִּצְוָה הַזֹּאת אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לֹא נִפְלֵאת הִוא מִמְּךָ וְלֹא רְחֹקָה הִוא:  ת  אֲרֵי תַּפְקֶדְתָּא הָדָא דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין לָא מַפְרְשָׁא הִיא מִנָךְ וְלָא רְחִיקָא הִיא:

{יב} לֹא בַשָּׁמַיִם הִוא לֵאמֹר מִי יַעֲלֶה לָּנוּ הַשָּׁמַיְמָה וְיִקָּחֶהָ לָּנוּ וְיַשְׁמִעֵנוּ אֹתָהּ וְנַעֲשֶׂנָּה:  מ  לֹא בַשָּׁמַיִם הִוא לֵאמֹר מִי יַעֲלֶה לָּנוּ הַשָּׁמַיְמָה וְיִקָּחֶהָ לָּנוּ וְיַשְׁמִעֵנוּ אֹתָהּ וְנַעֲשֶׂנָּה:  ת  לָא בִשְׁמַיָא הִיא לְמֵימָר מָן יִסַק לָנָא לִשְׁמַיָא וְיִסְבַהּ לָנָא וְיַשְׁמְעִנָנָא יָתַהּ וְנַעְבְּדִנַהּ:

{יג} וְלֹא מֵעֵבֶר לַיָּם הִוא לֵאמֹר מִי יַעֲבָר לָנוּ אֶל עֵבֶר הַיָּם וְיִקָּחֶהָ לָּנוּ וְיַשְׁמִעֵנוּ אֹתָהּ וְנַעֲשֶׂנָּה:  מ  וְלֹא מֵעֵבֶר לַיָּם הִוא לֵאמֹר מִי יַעֲבָר לָנוּ אֶל עֵבֶר הַיָּם וְיִקָּחֶהָ לָּנוּ וְיַשְׁמִעֵנוּ אֹתָהּ וְנַעֲשֶׂנָּה:  ת  וְלָא מֵעִבְרָא לְיַמָא הִיא לְמֵימַר מָן יֵעִבַר לָנָא לְעִבְרָא דְיַמָא וְיִסְבַהּ לָנָא וְיַשְׁמְעִנָנָא יָתַהּ וְנַעְבְּדִנַהּ:

{יד} כִּי קָרוֹב אֵלֶיךָ הַדָּבָר מְאֹד בְּפִיךָ וּבִלְבָבְךָ לַעֲשֹׂתוֹ:  מ  כִּי קָרוֹב אֵלֶיךָ הַדָּבָר מְאֹד בְּפִיךָ וּבִלְבָבְךָ לַעֲשֹׂתוֹ:  ת  אֲרֵי קָרִיב לָךְ פִּתְגָמָא לַחֲדָא בְּפוּמָךְ וּבְלִבָּךְ לְמֶעְבְּדֵהּ: (ס)

{טו}  שביעי - במחוברות רביעי  רְאֵה נָתַתִּי לְפָנֶיךָ הַיּוֹם אֶת הַחַיִּים וְאֶת הַטּוֹב וְאֶת הַמָּוֶת וְאֶת הָרָע:  מ  רְאֵה נָתַתִּי לְפָנֶיךָ הַיּוֹם אֶת הַחַיִּים וְאֶת הַטּוֹב וְאֶת הַמָּוֶת וְאֶת הָרָע:  ת  חֲזֵי דִיהָבִית קֳדָמָךְ יוֹמָא דֵין יָת חַיֵי וְיָת טַבְתָא וְיָת מוֹתָא וְיָת בִּישָׁא:

{טז} אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לְאַהֲבָה אֶת יְהוָה אֱלֹהֶיךָ לָלֶכֶת בִּדְרָכָיו וְלִשְׁמֹר מִצְוֹתָיו וְחֻקֹּתָיו וּמִשְׁפָּטָיו וְחָיִיתָ וְרָבִיתָ וּבֵרַכְךָ יְהוָה אֱלֹהֶיךָ בָּאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ:  מ  אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לְאַהֲבָה אֶת יְהוָה אֱלֹהֶיךָ לָלֶכֶת בִּדְרָכָיו וְלִשְׁמֹר מִצְוֹתָיו וְחֻקֹּתָיו וּמִשְׁפָּטָיו וְחָיִיתָ וְרָבִיתָ וּבֵרַכְךָ יְהוָה אֱלֹהֶיךָ בָּאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ:  ת  דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין לְמִרְחַם יָת יְיָ אֱלָהָךְ לִמְהַךְ בְּאָרְחָן דְתָקְנָן קֳדָמוֹהִי וּלְמִטַר פִּקוּדוֹהִי וּקְיָמוֹהִי וְדִינוֹהִי וְתֵיחֵי וְתִסְגֵי וִיבָרֵכִנָךְ יְיָ אֱלָהָךְ בְּאַרְעָא דִי אַתְּ עָלֵל לְתַמָן לְמֵירְתַהּ:

{יז} וְאִם יִפְנֶה לְבָבְךָ וְלֹא תִשְׁמָע וְנִדַּחְתָּ וְהִשְׁתַּחֲוִיתָ לֵאלֹהִים אֲחֵרִים וַעֲבַדְתָּם:  מ  וְאִם יִפְנֶה לְבָבְךָ וְלֹא תִשְׁמָע וְנִדַּחְתָּ וְהִשְׁתַּחֲוִיתָ לֵאלֹהִים אֲחֵרִים וַעֲבַדְתָּם:  ת  וְאִם יִפְנֵי לִבָּךְ וְלָא תְקַבֵּל וְתִטְּעֵי וְתִסְגוּד לְטַּעֲוַת עַמְמַיָא וְתִפְלְחִנוּן:

{יח} הִגַּדְתִּי לָכֶם הַיּוֹם כִּי אָבֹד תֹּאבֵדוּן לֹא תַאֲרִיכֻן יָמִים עַל הָאֲדָמָה אֲשֶׁר אַתָּה עֹבֵר אֶת הַיַּרְדֵּן לָבֹא שָׁמָּה לְרִשְׁתָּהּ:  מ  הִגַּדְתִּי לָכֶם הַיּוֹם כִּי אָבֹד תֹּאבֵדוּן לֹא תַאֲרִיכֻן יָמִים עַל הָאֲדָמָה אֲשֶׁר אַתָּה עֹבֵר אֶת הַיַּרְדֵּן לָבֹא שָׁמָּה לְרִשְׁתָּהּ:  ת  חַוֵיתִי לְכוֹן יוֹמָא דֵין אֲרֵי מֵיבַד תֵּיבְדוּן לָא תוֹרְכוּן יוֹמִין עַל אַרְעָא דִי אַתְּ עָבֵר יָת יַרְדְנָא לְמֵעַל לְתַמָן לְמֵירְתַהּ:

{יט} הַעִידֹתִי בָכֶם הַיּוֹם אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ הַחַיִּים וְהַמָּוֶת נָתַתִּי לְפָנֶיךָ הַבְּרָכָה וְהַקְּלָלָה וּבָחַרְתָּ בַּחַיִּים לְמַעַן תִּחְיֶה אַתָּה וְזַרְעֶךָ:  מ  הַעִידֹתִי בָכֶם הַיּוֹם אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ הַחַיִּים וְהַמָּוֶת נָתַתִּי לְפָנֶיךָ הַבְּרָכָה וְהַקְּלָלָה וּבָחַרְתָּ בַּחַיִּים לְמַעַן תִּחְיֶה אַתָּה וְזַרְעֶךָ:  ת  אַסְהֵדִית בְּכוֹן יוֹמָא דֵין יָת שְׁמַיָא וְיָת אַרְעָא חַיֵי וּמוֹתָא יְהָבִית קֳדָמָךְ בִּרְכָן וּלְוָטִין וְתִתִּרְעֵי בְּחַיֵי בְּדִיל דְתֵיחֵי אַתְּ וּבְנָיךְ:

{כ} לְאַהֲבָה אֶת יְהוָה אֱלֹהֶיךָ לִשְׁמֹעַ בְּקֹלוֹ וּלְדָבְקָה בוֹ כִּי הוּא חַיֶּיךָ וְאֹרֶךְ יָמֶיךָ לָשֶׁבֶת עַל הָאֲדָמָה אֲשֶׁר נִשְׁבַּע יְהוָה לַאֲבֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לָתֵת לָהֶם:  מ  לְאַהֲבָה אֶת יְהוָה אֱלֹהֶיךָ לִשְׁמֹעַ בְּקֹלוֹ וּלְדָבְקָה בוֹ כִּי הוּא חַיֶּיךָ וְאֹרֶךְ יָמֶיךָ לָשֶׁבֶת עַל הָאֲדָמָה אֲשֶׁר נִשְׁבַּע יְהוָה לַאֲבֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לָתֵת לָהֶם:  ת  לְמִרְחַם יָת יְיָ אֱלָהָךְ לְקַבָּלָא לְמֵימְרֵהּ וּלְאִתְקְרָבָא לְדַחַלְתֵּהּ אֲרֵי הוּא חַיָיךְ וְאוֹרָכוּת יוֹמָיךְ לְמִתַּב עַל אַרְעָא דִי קַיִים יְיָ לַאֲבָהָתָךְ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לְמִתַּן לְהוֹן: (פפפ)
