{א} בַּגְּמָרָא יָלֵיף לֵיהּ מִקְּרָא: {ב} נָשִׁים. אַף עַל גַּב דִּתְנַן בְּמִשְׁנָה ז' פֶּרֶק קַמָּא דְּקִדּוּשִׁין כָּל מִצְוֹת לֹא תַעֲשֶׂה כוּ' אֶחָד אֲנָשִׁים וְאֶחָד נָשִׁים חַיָּבִים כוּ', אִצְטְרִיךְ לְמִתְנֵי הָכִי, דְּמַהוּ דְּתֵימָא הָנֵי מִלֵּי עֹנֶשׁ דְּשָׁוֶה בֵּין בְּיָחִיד בֵּין בְּצִבּוּר, אֲבָל הָכָא כֵּיוָן דְּעֹנֶשׁ שֶׁאֵינוֹ שָׁוֶה בַּכֹּל הוּא, דִּתְנַן אֵין הַצִּבּוּר וְהַשֻּׁתָּפִים עוֹשִׂין תְּמוּרָה, אִשָּׁה נַמִּי כִּי עֲבַדָה לֹא לָקְיָא, קָא מַשְׁמַע לָן, מִדִּכְתִיב וְאִם הָמֵר, וַהֲוָה מָצֵי לְמִכְתַּב אִם, אָתֵי וָי"ו לְרַבּוֹת הָאִשָּׁה. גְּמָרָא: {ג} הָאַרְבָּעִים. וְאִם תֹּאמַר וְאַמַּאי לֹא תָּנֵי שְׁמוֹנִים, דְּהָא כְּתִיב לֹא יַחֲלִיפֶנּוּ וְלֹא יָמִיר אֹתוֹ. וְיֵשׁ לוֹמַר דְּלֹא נָחֵית לְמִנְיַן מַלְקִיּוֹת אֶלָּא כְּלוֹמַר דְּלָקֵי עֲלֵיהּ. עוֹד יֵשׁ לוֹמַר דִּלְעוֹלָם לֹא לָקֵי אֶלָּא אַרְבָּעִים, וּתְרֵי לָאוֵי צְרִיכֵי, חַד בְּקָרְבָּן שֶׁלּוֹ וְחַד בְּשֶׁל חֲבֵרוֹ, וּכְגוֹן דְּאָמַר כָּל הָרוֹצֶה לְהָמִיר יָבֹא וְיָמִיר. תּוֹסָפוֹת: {ד} וְכֵן כָּתַב הָרַמְבַּ"ם. וּבַגְּמָרָא יָלְפִינַן לְנִשְׁבַּע וּמְקַלֵּל מִקְּרָאֵי. וּמֵימֵר, אָמַר רַבִּי יוֹחָנָן לֹא תִתְנֵי וּמֵימֵר מִשּׁוּם דִּבְדִבּוּרוֹ עֲבִיד מַעֲשֶׂה. פֵּרֵשׁ רַשִׁ"י שֶׁעָשָׂה מֵחֻלִּין קָדָשִׁים. וְאִם כֵּן קֻשְׁיַת הָרַ"ב לֹא קָשֶׁה מִידִי, וְכֵן הִקְשָׁה הַכֶּסֶף מִשְׁנֶה וְהַמַּגִּיד מִשְׁנֶה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} וְכֵן כֹּהֵן שֶׁהֵמִיר בִּבְכוֹר שֶׁנּוֹלַד לוֹ, לֹא בִּבְכוֹר שֶׁלָּקַח מִיִּשְׂרָאֵל, הֲרֵי זוֹ תְּמוּרָה. הָרַמְבַּ"ם:

{ח} רַשִׁ"י וְהָרַ"מ. וּלְפֵרוּשׁ זֶה אֵינוֹ דּוֹמֶה הַךְ לְפִי חֶשְׁבּוֹן לְאִינָךְ, דְּאִינְהוּ מַיְרֵי לְפִי חֶשְׁבּוֹן הַדָּבָר הַפּוֹסֵל. לְכָךְ נִרְאֶה פֵּרוּשׁ הַתּוֹסָפוֹת שֶׁפֵּרְשׁוּ, חֶשְׁבּוֹן הַלֻּגִּין שֶׁפּוֹסְלִין הַמִּקְוֶה, דְּהַיְנוּ שְׁלֹשָׁה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ט} וְכֵן כָּתַב הָרַמְבַּ"ם. וְתֵימַהּ, דְּהָא סָתַם לָן כְּוָתֵיהּ בְּסוֹף פֶּרֶק ג' דְּמִקְוָאוֹת [וְהָא דְּנָקַט בִּדְבָרָיו רַבִּי יוֹסֵי בֶּן חוֹנִי, דְּהָכִי קָתָנֵי לַהּ בְּהֶדְיָא בַּבְּרַיְתָא. הָרַ"שׁ]. וּבֶאֱמֶת הָרַמְבַּ"ם בְּחִבּוּרוֹ חָזַר בּוֹ:

{י} גְּמָרָא, מַאי חָזֵית דְּאָמַרְתְּ סֵיפֵיהּ דִּקְרָא דַּוְקָא, דִּלְמָא רֵישָׁא דִּקְרָא דַּוְקָא [וְאֶל כֶּלִי לְהָכִי אָתָא שֶׁתְּהֵא חִיּוּתוֹ בַּכְּלִי שֶׁלֹּא יְמַלְּאֵם בִּכְלִי אַחֵר וְיִעָרֵם לִכְלִי זֶה, אֶלָּא זֶה יִתֵּן לְתוֹךְ הַמַּעְיָן וְיִקַּח הַמַּיִם לְתוֹכוֹ. רַשִׁ"י] לֹא מָצֵית אָמַרְתְּ. מַאי מָצִינוּ בְּכָל מָקוֹם מַכְשִׁיר לְמַעְלָה [כְּגוֹן בְּסוֹטָה, עָפָר לְמַעְלָה שֶׁהוּא הַמַּכְשִׁיר אֶת הַמַּיִם לְבַדּוֹ וְהָוֵי לְמַעְלָה, דְּמִצְוָה לְהַקְדִּים הַמַּיִם כְּדִכְתִיב וּמִן הֶעָפָר וְגוֹ'], אַף כָּאן מַכְשִׁיר לְמַעְלָה: {יא} וּפְלִיאָה הוּא בְּעֵינַי, דְּהָתְנַן בְּמִשְׁנָה ג' פֶּרֶק י"ז דְּאֳהָלוֹת, הַחוֹרֵשׁ מִשָּׂדֶה שֶׁאָבַד הַקֶּבֶר בְּתוֹכָהּ אֵינוֹ עוֹשֶׂה בֵּית הַפְּרָס, וּפֵרֵשׁ הָרַ"ב דַּהֲוֵי לֵיהּ סְפֵק סְפֵקָא, שֶׁמָּא לֹא חָרַשׁ בִּמְקוֹם הַקֶּבֶר [וַאֲפִלּוּ חָרַשׁ כָּל הַשָּׂדֶה יֵשׁ לְכָל צַד סָפֵק דְּשֶׁמָּא הָיָה הַקֶּבֶר בַּצַּד הָאַחֵר], וְאִם תִּמְצָא לוֹמַר חָרַשׁ, שֶׁמָּא לֹא הוֹלִיךְ מִמֶּנָּה עֶצֶם. וְלִי נִרְאֶה לְפָרֵשׁ דְּמַאי דְּאָמַר בַּגְּמָרָא שָׁלֹשׁ שָׂדוֹת וּשְׁתֵּי מַעֲנוֹת, פֵּרוּשׁ, בֵּין כָּל הַשָּׁלֹשׁ שָׂדוֹת שְׁתֵּי מַעֲנוֹת וְהַקֶּבֶר הָיָה בְּאֶמְצָעוֹ שֶׁל אֶמְצָעִי. וּכְפֵרוּשׁ הָרַ"ב הָתָם רֵישׁ הַפֶּרֶק. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יב} וְכֵן פֵּרֵשׁ רַשִׁ"י, דְּהָא סְתָמָא קָאָמַר אֵינָהּ תְּרוּמָה. וְהַתּוֹסָפוֹת הִקְשׁוּ, דְּאֵין זֶה כְּמוֹ וְלֹא תְּמוּרָה אַחַר תְּמוּרָה, דְּהָתָם הָרִאשׁוֹן תְּמוּרָה וְהַשֵּׁנִי אֵינָהּ תְּמוּרָה. לָכֵן נִרְאֶה לְפָרֵשׁ הָכִי נַמִּי דְּהָרִאשׁוֹן תְּרוּמָה וְלֹא הַשֵּׁנִי: {יג} דְּפֶרֶק ג' דִּתְרוּמוֹת מִשְׁנָה ג' דְּקָאָמַר אֵין תְּרוּמַת שְׁנֵיהֶם תְּרוּמָה. וְהָא דְּאִיתָא הָתָם תְּרוּמַת שְׁנֵיהֶם תְּרוּמָה, כָּתַב הָרַ"שׁ, דְּחִלּוּף גִּרְסָאוֹת הֵן. וּלְפֵרוּשׁ הַתּוֹסָפוֹת דִּבְסָמוּךְ, צָרִיךְ לוֹמַר דְּגָרְסִינַן דְּתַנְיָא, וּמַיְרֵי בְּשֶׁנָּתְנוּ רְשׁוּת זֶה לָזֶה לִתְרֹם. וְהָתָם בִּתְרוּמוֹת מַיְרֵי בְּלֹא נָתְנוּ רְשׁוּת זֶה לָזֶה רְשׁוּת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יד} וְהוּא הַדִּין אִפְּכָא נַמִּי. וְעַיֵּן שָׁם בְּפֵרוּשׁ הָרַ"ב: {טו} כְּלוֹמַר, דְּאַף תְּרוּמַת הַשֵּׁנִי תְּרוּמָה, דְּאֵין לוֹמַר דִּתְרוּמַת הַשֵּׁנִי דַּוְקָא תְּרוּמָה וְלֹא הָרִאשׁוֹן, דְּמַאי אוּלְמֵיהּ דְּהַאי מֵהַאי. כֶּסֶף מִשְׁנֶה:

{טז} קָרְבְּנוֹת כוּ'. וְבַגְּמָרָא גָּרַס קָדְשֵׁי. וְכֵן בְּרַשִׁ"י וְהָרַ"ב]. וְיֵשׁ לְיַשֵּׁב גִּרְסַת הַסֵּפֶר, דְּהָא מוֹדוּ רַבָּנָן דְּקָרְבַּן ה' אִקְּרוּ כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב שֶׁהֶאֱרִיךְ דְּרַבִּי שִׁמְעוֹן וְרַבָּנָן לֹא פְּלִיגֵי אֶלָּא בְּמַשְׁמָעוּת דּוֹרְשִׁין וְלֹא בְּעִקַּר הַדִּין, דְּכִי אִמַּעֲטוּ קָרְבְּנוֹת צִבּוּר מִלָּאו, אִמַּעֲטוּ נַמִּי דְּאֵין מְמִירִין, דְּהָא אֵין מְמִירִין אֶלָּא בְּשֶׁלָּהֶם. וְדֹחַק לוֹמַר דְּמַיְרֵי כְּשֶׁאָמְרוּ הַצִּבּוּר כָּל הָרוֹצֶה לְהָמִיר יָמִיר: {יז} לְהָקִישׁ כוּ'. וּמַאן דְּפָלֵיג אַדְּרַבִּי שִׁמְעוֹן אִצְטְרִיכָא לֵיהּ בַּגְּמָרָא לִדְרָשָׁא אַחֲרִיתִי:

בְּנִקְבָּרִין אַף עַל גַּב דְּאִיכָּא נַמִּי לְמֵיחַשׁ לְהָכִי, מִכָּל מָקוֹם אִי אֶפְשָׁר לְשָׂרְפָן, דַּאֲפָרָן אָסוּר בַּהֲנָאָה. וְהוֹאִיל וְהַנִּשְׂרָפִין אֲפָרָן מֻתָּר אָתֵי לְמִתְהַנֵּי מִנֵּיהּ, וְאִי אֶפְשָׁר לוֹמַר שֶׁאֵפֶר אֵלּוּ יִקָּבֵר, לְפִי שֶׁאֵין בְּיַד כָּל אָדָם לְחַלֵּק בֵּין אֵפֶר לְאֵפֶר אֵיזֶהוּ בִּקְבוּרָה וְאֵיזֶהוּ בִּשְׂרֵפָה, וְאָתֵי לְמִטְעֵי. הִלְכָּךְ אָמְרוּ אוֹתָן שֶׁאֲפָרָן אָסוּר יִקְבְּרֶנּוּ בְּעַצְמָן. נִרְאֶה לִי: {יט} צָרִיךְ עִיּוּן טַעֲמָא מַאי. וְאוֹמֵר מוֹרִי הָרמ"ר דְּנִשְׂרָפִין כֵּיוָן שֶׁצִּוָּה הַכָּתוּב לְשָׂרְפָן אַחַר שֶׁעָשָׂה כְּאִלּוּ נַעֲשֵׂית מִצְוָתָן, וְאֵין לְךָ דָּבָר שֶׁנַּעֲשֵׂית מִצְוָתוֹ וּמוֹעֲלִין בּוֹ, הָכִי נַמִּי כֵּיוָן שֶׁנַּעֲשֵׂית מִצְוָתוֹ הָלַךְ אִסּוּרֵיהּ. אֲבָל הַנִּקְבָּרִים דְּלֹא הִטְעִין הַכָּתוּב לְשָׂרְפָן מָשׁוּךְ אִסּוּרַיְהוּ לְעוֹלָם. תּוֹסָפוֹת: {כ} וְכֵיוָן דְּטָעוּן גְּנִיזָה אָסוּר בַּהֲנָאָה. רַשִׁ"י: {כא} אֵינוֹ כוּ'. שֶׁאַף עַל פִּי שֶׁהוּא מַחֲמִיר בִּשְׂרֵפָתָהּ. הֲרֵי הֵקֵל בַּאֲפָרָן. שֶׁאֵפֶר הַנִּקְבָּרִים אָסוּר. הָרַמְבַּ"ם: {ט} תֵּימַהּ, דִּלְמַאי נַפְקָא מִנַּהּ מִתְחַשְּׁבִין בְּאַפֵּי נַפְשַׁיְהוּ כוּ'. אֲבָל הָרַמְבַּ"ם כָּתַב וּמְחַשְּׁבִין לָהּ מַה שֶּׁרָאוּי לָהֶם כוּ'. וּלְפִיכָךְ נִרְאֶה לִי לְהַגִּיהַּ בְּדִבְרֵי הָרַ"ב וְנִפְדִּין בְּשָׁוְיָן וְהַשְּׁאָר לְפִי חֶשְׁבּוֹן. וְעוֹד יֵשׁ לוֹמַר, דְּדַעְתּוֹ כְדַעַת הַתּוֹסָפוֹת דְּהַנָּפְקָא מִנַּהּ שֶׁלֹּא חָשְׁבִינַן לְהוּ הַמִּדְרוֹן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
