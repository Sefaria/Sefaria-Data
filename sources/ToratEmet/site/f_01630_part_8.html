פרשת אמור



ויקרא פרק-כא

{א} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה אֱמֹר אֶל הַכֹּֽהֲנִים בְּנֵי אַֽהֲרֹן וְאָֽמַרְתָּ אֲלֵהֶם לְנֶפֶשׁ לֹֽא יִטַּמָּא בְּעַמָּֽיו:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה אֱמֹר אֶל הַכֹּֽהֲנִים בְּנֵי אַֽהֲרֹן וְאָֽמַרְתָּ אֲלֵהֶם לְנֶפֶשׁ לֹֽא יִטַּמָּא בְּעַמָּֽיו:  ת  וַאֲמַר יְיָ לְמֹשֶׁה אֱמַר לְכַהֲנַיָא בְּנֵי אַהֲרֹן וְתֵימַר לְהוֹן עַל מִית לָא יִסְתָּאַב בְּעַמֵהּ:

{ב} כִּי אִם לִשְׁאֵרוֹ הַקָּרֹב אֵלָיו לְאִמּוֹ וּלְאָבִיו וְלִבְנוֹ וּלְבִתּוֹ וּלְאָחִֽיו:  מ  כִּי אִם לִשְׁאֵרוֹ הַקָּרֹב אֵלָיו לְאִמּוֹ וּלְאָבִיו וְלִבְנוֹ וּלְבִתּוֹ וּלְאָחִֽיו:  ת  אֱלָהֵן לְקָרִיבֵהּ דְקָרִיב לֵהּ לְאִמֵהּ וְלַאֲבוּהִי וְלִבְרֵהּ וְלִבְרַתֵּהּ וְלַאֲחוּהִי:

{ג} וְלַֽאֲחֹתוֹ הַבְּתוּלָה הַקְּרוֹבָה אֵלָיו אֲשֶׁר לֹא הָֽיְתָה לְאִישׁ לָהּ יִטַּמָּֽא:  מ  וְלַֽאֲחֹתוֹ הַבְּתוּלָה הַקְּרוֹבָה אֵלָיו אֲשֶׁר לֹא הָֽיְתָה לְאִישׁ לָהּ יִטַּמָּֽא:  ת  וְלַאֲחָתֵהּ בְּתֻלְתָּא דְקָרִיבָא לֵהּ דִי לָא הֲוַת לִגְבַר לַהּ יִסְתָּאָב:

{ד} לֹא יִטַּמָּא בַּעַל בְּעַמָּיו לְהֵחַלּֽוֹ:  מ  לֹא יִטַּמָּא בַּעַל בְּעַמָּיו לְהֵחַלּֽוֹ:  ת  לָא יִסְתָּאַב רַבָּא בְעַמֵהּ לְאַחֲלוּתֵהּ:

{ה} לֹֽא (יקרחה) יִקְרְחוּ קָרְחָה בְּרֹאשָׁם וּפְאַת זְקָנָם לֹא יְגַלֵּחוּ וּבִבְשָׂרָם לֹא יִשְׂרְטוּ שָׂרָֽטֶת:  מ  לֹֽא (יקרחה) יִקְרְחוּ קָרְחָה בְּרֹאשָׁם וּפְאַת זְקָנָם לֹא יְגַלֵּחוּ וּבִבְשָׂרָם לֹא יִשְׂרְטוּ שָׂרָֽטֶת:  ת  לָא יִמְרְטוּן מְרַט בְּרֵישְׁהוֹן וּפָאתָא דְדִקְנְהוֹן לָא יְגַלְחוּן וּבְבִשְׂרְהוֹן לָא יְחַבְּלוּן חִבּוּל:

{ו} קְדשִׁים יִֽהְיוּ לֵאלֹהֵיהֶם וְלֹא יְחַלְּלוּ שֵׁם אֱלֹֽהֵיהֶם כִּי אֶת אִשֵּׁי יְהֹוָה לֶחֶם אֱלֹֽהֵיהֶם הֵם מַקְרִיבִם וְהָיוּ קֹֽדֶשׁ:  מ  קְדשִׁים יִֽהְיוּ לֵאלֹהֵיהֶם וְלֹא יְחַלְּלוּ שֵׁם אֱלֹֽהֵיהֶם כִּי אֶת אִשֵּׁי יְהֹוָה לֶחֶם אֱלֹֽהֵיהֶם הֵם מַקְרִיבִם וְהָיוּ קֹֽדֶשׁ:  ת  קַדִישִׁין יְהוֹן קֳדָם אֱלָהֲהוֹן וְלָא יְחַלוּן שְׁמָא דֶאֱלָהֲהוֹן אֲרֵי יָת קֻרְבָּנַיָא דַיְיָ קֻרְבַּן אֱלָהֲהוֹן אִנוּן מַקְרִיבִין וִיהוֹן קַדִישִׁין:

{ז} אִשָּׁה זֹנָה וַֽחֲלָלָה לֹא יִקָּחוּ וְאִשָּׁה גְּרוּשָׁה מֵֽאִישָׁהּ לֹא יִקָּחוּ כִּֽי קָדֹשׁ הוּא לֵֽאלֹהָֽיו:  מ  אִשָּׁה זֹנָה וַֽחֲלָלָה לֹא יִקָּחוּ וְאִשָּׁה גְּרוּשָׁה מֵֽאִישָׁהּ לֹא יִקָּחוּ כִּֽי קָדֹשׁ הוּא לֵֽאלֹהָֽיו:  ת  אִתְּתָא מַטְעֲיָא וּמְחַלְלָא לָא יִסְבוּן וְאִתְּתָא דִמְתָרְכָא מִבַּעְלַהּ לָא יִסְבוּן אֲרֵי קַדִישׁ הוּא קֳדָם אֱלָהֵהּ:

{ח} וְקִדַּשְׁתּוֹ כִּֽי אֶת לֶחֶם אֱלֹהֶיךָ הוּא מַקְרִיב קָדשׁ יִֽהְיֶה לָּךְ כִּי קָדוֹשׁ אֲנִי יְהוָֹה מְקַדִּשְׁכֶֽם:  מ  וְקִדַּשְׁתּוֹ כִּֽי אֶת לֶחֶם אֱלֹהֶיךָ הוּא מַקְרִיב קָדשׁ יִֽהְיֶה לָּךְ כִּי קָדוֹשׁ אֲנִי יְהוָֹה מְקַדִּשְׁכֶֽם:  ת  וּתְקַדְשִׁנֵהּ אֲרֵי יָת קֻרְבַּן אֱלָהָךְ הוּא מְקָרֵב קַדִישׁ יְהֵא לָךְ אֲרֵי קַדִישׁ אֲנָא יְיָ מְקַדִשְׁכוֹן:

{ט} וּבַת אִישׁ כֹּהֵן כִּי תֵחֵל לִזְנוֹת אֶת אָבִיהָ הִיא מְחַלֶּלֶת בָּאֵשׁ תִּשָּׂרֵֽף:  מ  וּבַת אִישׁ כֹּהֵן כִּי תֵחֵל לִזְנוֹת אֶת אָבִיהָ הִיא מְחַלֶּלֶת בָּאֵשׁ תִּשָּׂרֵֽף:  ת  וּבַת גְבַר כָּהֵן אֲרֵי תִתְחַל לְמִטְעֵי מִקְדֻשַׁת אֲבוּהָ הִיא מִתַּחֲלָא בְּנוּרָא תִּתּוֹקָד: (ס)

{י} וְהַכֹּהֵן הַגָּדוֹל מֵֽאֶחָיו אֲֽשֶׁר יוּצַק עַל רֹאשׁוֹ שֶׁמֶן הַמִּשְׁחָה וּמִלֵּא אֶת יָדוֹ לִלְבֹּשׁ אֶת הַבְּגָדִים אֶת רֹאשׁוֹ לֹא יִפְרָע וּבְגָדָיו לֹא יִפְרֹֽם:  מ  וְהַכֹּהֵן הַגָּדוֹל מֵֽאֶחָיו אֲֽשֶׁר יוּצַק עַל רֹאשׁוֹ שֶׁמֶן הַמִּשְׁחָה וּמִלֵּא אֶת יָדוֹ לִלְבֹּשׁ אֶת הַבְּגָדִים אֶת רֹאשׁוֹ לֹא יִפְרָע וּבְגָדָיו לֹא יִפְרֹֽם:  ת  וְכַהֲנָא דְאִתְרַבָּא מֵאֲחוֹהִי דְיִתָּרִיק עַל רֵישֵׁהּ מִשְׁחָא דִרְבוּתָא וְדִי קָרִיב יָת קֻרְבָּנֵהּ לְמִלְבָּשׁ יָת לְבוּשַׁיָא יָת רֵישֵׁהּ לָא יְרַבֵּי פֵרוּעַ וּלְבוּשׁוֹהִי לָא יִבְזַע:

{יא} וְעַל כָּל נַפְשֹׁת מֵת לֹא יָבֹא לְאָבִיו וּלְאִמּוֹ לֹא יִטַּמָּֽא:  מ  וְעַל כָּל נַפְשֹׁת מֵת לֹא יָבֹא לְאָבִיו וּלְאִמּוֹ לֹא יִטַּמָּֽא:  ת  וְעַל כָּל נַפְשַׁת מֵתָא לָא יֵיעוֹל לַאֲבוּהִי וּלְאִמֵהּ לָא יִסְתָּאָב:

{יב} וּמִן הַמִּקְדָּשׁ לֹא יֵצֵא וְלֹא יְחַלֵּל אֵת מִקְדַּשׁ אֱלֹהָיו כִּי נֵזֶר שֶׁמֶן מִשְׁחַת אֱלֹהָיו עָלָיו אֲנִי יְהוָֹֽה:  מ  וּמִן הַמִּקְדָּשׁ לֹא יֵצֵא וְלֹא יְחַלֵּל אֵת מִקְדַּשׁ אֱלֹהָיו כִּי נֵזֶר שֶׁמֶן מִשְׁחַת אֱלֹהָיו עָלָיו אֲנִי יְהוָֹֽה:  ת  וּמִן מַקְדְשָׁא לָא יִפּוֹק וְלָא יַחֵל יָת מַקְדְשָׁא דֶאֱלָהֵהּ אֲרֵי כְּלִיל מְשַׁח רְבוּתָא דֶאֱלָהֵהּ עֲלוֹהִי אֲנָא יְיָ:

{יג} וְהוּא אִשָּׁה בִבְתוּלֶיהָ יִקָּֽח:  מ  וְהוּא אִשָּׁה בִבְתוּלֶיהָ יִקָּֽח:  ת  וְהוּא אִתְּתָא בִבְתֻלְתָּהָא יִסָב:

{יד} אַלְמָנָה וּגְרוּשָׁה וַֽחֲלָלָה זֹנָה אֶת אֵלֶּה לֹא יִקָּח כִּי אִם בְּתוּלָה מֵֽעַמָּיו יִקַּח אִשָּֽׁה:  מ  אַלְמָנָה וּגְרוּשָׁה וַֽחֲלָלָה זֹנָה אֶת אֵלֶּה לֹא יִקָּח כִּי אִם בְּתוּלָה מֵֽעַמָּיו יִקַּח אִשָּֽׁה:  ת  אַרְמְלָא וּמְתָרְכָא וְחַלִילָא מַטְעֲיָא יָת אִלֵין לָא יִסָב אֱלָהֵין בְּתֻלְתָּא מֵעַמֵהּ יִסַב אִתְּתָא:

{טו} וְלֹֽא יְחַלֵּל זַרְעוֹ בְּעַמָּיו כִּי אֲנִי יְהוָֹה מְקַדְּשֽׁוֹ:  מ  וְלֹֽא יְחַלֵּל זַרְעוֹ בְּעַמָּיו כִּי אֲנִי יְהוָֹה מְקַדְּשֽׁוֹ:  ת  וְלָא יַחֵל זַרְעֵהּ בְּעַמֵהּ אֲרֵי אֲנָא יְיָ מְקַדְשֵׁהּ: (ס)

{טז}  שני  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{יז} דַּבֵּר אֶֽל אַֽהֲרֹן לֵאמֹר אִישׁ מִֽזַּרְעֲךָ לְדֹֽרֹתָם אֲשֶׁר יִהְיֶה בוֹ מוּם לֹא יִקְרַב לְהַקְרִיב לֶחֶם אֱלֹהָֽיו:  מ  דַּבֵּר אֶֽל אַֽהֲרֹן לֵאמֹר אִישׁ מִֽזַּרְעֲךָ לְדֹֽרֹתָם אֲשֶׁר יִהְיֶה בוֹ מוּם לֹא יִקְרַב לְהַקְרִיב לֶחֶם אֱלֹהָֽיו:  ת  מַלֵל עִם אַהֲרֹן לְמֵימָר גְבַר מִבְּנָיִךְ לְדָרֵיהוֹן דִי יְהֵּי בֵהּ מוּמָא לָא יִקְרַב לְקָרָבָא קֻרְבָּנָא קֳדָם אֱלָהֵהּ:

{יח} כִּי כָל אִישׁ אֲשֶׁר בּוֹ מוּם לֹא יִקְרָב אִישׁ עִוֵּר אוֹ פִסֵּחַ אוֹ חָרֻם אוֹ שָׂרֽוּעַ:  מ  כִּי כָל אִישׁ אֲשֶׁר בּוֹ מוּם לֹא יִקְרָב אִישׁ עִוֵּר אוֹ פִסֵּחַ אוֹ חָרֻם אוֹ שָׂרֽוּעַ:  ת  אֲרֵי כָל גְבַר דִּי בֵהּ מוּמָא לָא יִקְרַב גְבַר עַוִיר אוֹ חֲגִיר אוֹ חָרִים אוֹ שָׂרִיע:

{יט} אוֹ אִישׁ אֲשֶׁר יִֽהְיֶה בוֹ שֶׁבֶר רָגֶל אוֹ שֶׁבֶר יָֽד:  מ  אוֹ אִישׁ אֲשֶׁר יִֽהְיֶה בוֹ שֶׁבֶר רָגֶל אוֹ שֶׁבֶר יָֽד:  ת  אוֹ גְבַר דִי יְהֵי בֵהּ תְּבַר רַגְלָא אוֹ תְּבַר יְדָא:

{כ} אֽוֹ גִבֵּן אוֹ דַק אוֹ תְּבַלֻּל בְּעֵינוֹ אוֹ גָרָב אוֹ יַלֶּפֶת אוֹ מְרוֹחַ אָֽשֶׁךְ:  מ  אֽוֹ גִבֵּן אוֹ דַק אוֹ תְּבַלֻּל בְּעֵינוֹ אוֹ גָרָב אוֹ יַלֶּפֶת אוֹ מְרוֹחַ אָֽשֶׁךְ:  ת  אוֹ גְבִין אוֹ דֻקָא אוֹ חִלִיז בְּעֵינֵהּ אוֹ גַרְבָן אוֹ חֲזָזָן אוֹ מְרִיס פַּחֲדִין:

{כא} כָּל אִישׁ אֲשֶׁר בּוֹ מוּם מִזֶּרַע אַֽהֲרֹן הַכֹּהֵן לֹא יִגַּשׁ לְהַקְרִיב אֶת אִשֵּׁי יְהוָֹה מוּם בּוֹ אֵת לֶחֶם אֱלֹהָיו לֹא יִגַּשׁ לְהַקְרִֽיב:  מ  כָּל אִישׁ אֲשֶׁר בּוֹ מוּם מִזֶּרַע אַֽהֲרֹן הַכֹּהֵן לֹא יִגַּשׁ לְהַקְרִיב אֶת אִשֵּׁי יְהוָֹה מוּם בּוֹ אֵת לֶחֶם אֱלֹהָיו לֹא יִגַּשׁ לְהַקְרִֽיב:  ת  כָּל גְבַר דִי בֵהּ מוּמָא מִזַרְעָא דְאַהֲרֹן כַּהֲנָא לָא יִקְרַב לְקָרָבָא יָת קֻרְבָּנַיָא דַיְיָ מוּמָא בֵהּ יָת קֻרְבַּן אֱלָהֵהּ לָא יִקְרַב לְקָרָבָא:

{כב} לֶחֶם אֱלֹהָיו מִקָּדְשֵׁי הַקֳּדָשִׁים וּמִן הַקֳּדָשִׁים יֹאכֵֽל:  מ  לֶחֶם אֱלֹהָיו מִקָּדְשֵׁי הַקֳּדָשִׁים וּמִן הַקֳּדָשִׁים יֹאכֵֽל:  ת  קֻרְבַּן אֱלָהֵהּ מִקֹדֶשׁ קוּדְשַׁיָא וּמִן קוּדְשַׁיָא יֵיכוּל:

{כג} אַךְ אֶל הַפָּרֹכֶת לֹא יָבֹא וְאֶל הַמִּזְבֵּחַ לֹא יִגַּשׁ כִּי מוּם בּוֹ וְלֹא יְחַלֵּל אֶת מִקְדָּשַׁי כִּי אֲנִי יְהוָֹה מְקַדְּשָֽׁם:  מ  אַךְ אֶל הַפָּרֹכֶת לֹא יָבֹא וְאֶל הַמִּזְבֵּחַ לֹא יִגַּשׁ כִּי מוּם בּוֹ וְלֹא יְחַלֵּל אֶת מִקְדָּשַׁי כִּי אֲנִי יְהוָֹה מְקַדְּשָֽׁם:  ת  בְּרַם לְפָרֻכְתָּא לָא יֵיעוֹל וּלְמַדְבְּחָא לָא יִקְרַב אֲרֵי מוּמָא בֵהּ וְלָא יַחֵל יָת מַקְדְשַׁי אֲרֵי אֲנָא יְיָ מְקַדִשְׁהוֹן:

{כד} וַיְדַבֵּר מֹשֶׁה אֶֽל אַֽהֲרֹן וְאֶל בָּנָיו וְאֶֽל כָּל בְּנֵי יִשְׂרָאֵֽל:  מ  וַיְדַבֵּר מֹשֶׁה אֶֽל אַֽהֲרֹן וְאֶל בָּנָיו וְאֶֽל כָּל בְּנֵי יִשְׂרָאֵֽל:  ת  וּמַלִיל מֹשֶׁה עִם אַהֲרֹן וְעִם בְּנוֹהִי וְעִם כָּל בְּנֵי יִשְׂרָאֵל: (פ)




ויקרא פרק-כב

{א} וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} דַּבֵּר אֶֽל אַֽהֲרֹן וְאֶל בָּנָיו וְיִנָּֽזְרוּ מִקָּדְשֵׁי בְנֵֽי יִשְׂרָאֵל וְלֹא יְחַלְּלוּ אֶת שֵׁם קָדְשִׁי אֲשֶׁר הֵם מַקְדִּשִׁים לִי אֲנִי יְהוָֹֽה:  מ  דַּבֵּר אֶֽל אַֽהֲרֹן וְאֶל בָּנָיו וְיִנָּֽזְרוּ מִקָּדְשֵׁי בְנֵֽי יִשְׂרָאֵל וְלֹא יְחַלְּלוּ אֶת שֵׁם קָדְשִׁי אֲשֶׁר הֵם מַקְדִּשִׁים לִי אֲנִי יְהוָֹֽה:  ת  מַלֵיל עִם אַהֲרֹן וְעִם בְּנוֹהִי וְיִפָּרְשׁוּן מִקוּדְשַׁיָא דִבְנֵי יִשְׂרָאֵל וְלָא יְחַלוּן יָת שְׁמָא דְקוּדְשִׁי דִי אִנוּן מְקַדְשִׁין קֳדָמַי אֲנָא יְיָ:

{ג} אֱמֹר אֲלֵהֶם לְדֹרֹתֵיכֶם כָּל אִישׁ אֲשֶׁר יִקְרַב מִכָּל זַֽרְעֲכֶם אֶל הַקֳּדָשִׁים אֲשֶׁר יַקְדִּישׁוּ בְנֵֽי יִשְׂרָאֵל לַֽיהֹוָה וְטֻמְאָתוֹ עָלָיו וְנִכְרְתָה הַנֶּפֶשׁ הַהִוא מִלְּפָנַי אֲנִי יְהוָֹֽה:  מ  אֱמֹר אֲלֵהֶם לְדֹרֹתֵיכֶם כָּל אִישׁ אֲשֶׁר יִקְרַב מִכָּל זַֽרְעֲכֶם אֶל הַקֳּדָשִׁים אֲשֶׁר יַקְדִּישׁוּ בְנֵֽי יִשְׂרָאֵל לַֽיהֹוָה וְטֻמְאָתוֹ עָלָיו וְנִכְרְתָה הַנֶּפֶשׁ הַהִוא מִלְּפָנַי אֲנִי יְהוָֹֽה:  ת  אֱמַר לְהוֹן לְדָרֵיכוֹן כָּל גְבַר דִי יִקְרַב מִכָּל בְּנֵיכוֹן לְקוּדְשַׁיָא דִי יְקַדְשׁוּן בְּנֵי יִשְׂרָאֵל קֳדָם יְיָ וְסֹאבְתֵיהּ עֲלוֹהִי וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהִיא מִן קֳדָמַי אֲנָא יְיָ:

{ד} אִישׁ אִישׁ מִזֶּרַע אַֽהֲרֹן וְהוּא צָרוּעַ אוֹ זָב בַּקֳּדָשִׁים לֹא יֹאכַל עַד אֲשֶׁר יִטְהָר וְהַנֹּגֵעַ בְּכָל טְמֵא נֶפֶשׁ אוֹ אִישׁ אֲשֶׁר תֵּצֵא מִמֶּנּוּ שִׁכְבַת זָֽרַע:  מ  אִישׁ אִישׁ מִזֶּרַע אַֽהֲרֹן וְהוּא צָרוּעַ אוֹ זָב בַּקֳּדָשִׁים לֹא יֹאכַל עַד אֲשֶׁר יִטְהָר וְהַנֹּגֵעַ בְּכָל טְמֵא נֶפֶשׁ אוֹ אִישׁ אֲשֶׁר תֵּצֵא מִמֶּנּוּ שִׁכְבַת זָֽרַע:  ת  גְבַר גְבַר מִזַרְעָא דְאַהֲרֹן וְהוּא סְגִיר אוֹ דָאִיב בְּקוּדְשַׁיָא לָא יֵיכוּל עַד דִי יִדְכֵּי וּדְיִקְרַב בְּכָל טְמֵא נַפְשָׁא אוֹ גְבַר דִי תִפּוֹק מִנֵהּ שִׁכְבַת זַרְעָא:

{ה} אוֹ אִישׁ אֲשֶׁר יִגַּע בְּכָל שֶׁרֶץ אֲשֶׁר יִטְמָא לוֹ אוֹ בְאָדָם אֲשֶׁר יִטְמָא לוֹ לְכֹל טֻמְאָתֽוֹ:  מ  אוֹ אִישׁ אֲשֶׁר יִגַּע בְּכָל שֶׁרֶץ אֲשֶׁר יִטְמָא לוֹ אוֹ בְאָדָם אֲשֶׁר יִטְמָא לוֹ לְכֹל טֻמְאָתֽוֹ:  ת  אוֹ גְבַר דִי יִקְרַב בְּכָל רִחֲשָׁא דִי יִסְתָּאַב לֵהּ אוֹ בַאֲנָשָׁא דִי יִסְתָּאַב לֵהּ לְכָל סֹאבְתֵיהּ:

{ו} נֶפֶשׁ אֲשֶׁר תִּֽגַּֽע בּוֹ וְטָֽמְאָה עַד הָעָרֶב וְלֹא יֹאכַל מִן הַקֳּדָשִׁים כִּי אִם רָחַץ בְּשָׂרוֹ בַּמָּֽיִם:  מ  נֶפֶשׁ אֲשֶׁר תִּֽגַּֽע בּוֹ וְטָֽמְאָה עַד הָעָרֶב וְלֹא יֹאכַל מִן הַקֳּדָשִׁים כִּי אִם רָחַץ בְּשָׂרוֹ בַּמָּֽיִם:  ת  אֱנַשׁ דִי יִקְרַב בֵּהּ וִיהֵי מְסָאָב עַד רַמְשָׁא וְלָא יֵיכוּל מִן קוּדְשַׁיָא אֱלָהֵן אַסְחֵי בִשׂרֵהּ בְּמַיָא:

{ז} וּבָא הַשֶּׁמֶשׁ וְטָהֵר וְאַחַר יֹאכַל מִן הַקֳּדָשִׁים כִּי לַחְמוֹ הֽוּא:  מ  וּבָא הַשֶּׁמֶשׁ וְטָהֵר וְאַחַר יֹאכַל מִן הַקֳּדָשִׁים כִּי לַחְמוֹ הֽוּא:  ת  וּבְמֵעַל שִׁמְשָׁא וְיִדְכֵּי וּבָתַר כֵּן יֵיכוּל מִן קוּדְשַׁיָא אֲרֵי לַחְמֵהּ הוּא:

{ח} נְבֵלָה וּטְרֵפָה לֹא יֹאכַל לְטָמְאָה בָהּ אֲנִי יְהוָֹֽה:  מ  נְבֵלָה וּטְרֵפָה לֹא יֹאכַל לְטָמְאָה בָהּ אֲנִי יְהוָֹֽה:  ת  נְבִילָא וּתְבִירָא לָא יֵיכוּל לְאִסְתָּאָבָא בָהּ אֲנָא יְיָ:

{ט} וְשָֽׁמְרוּ אֶת מִשְׁמַרְתִּי וְלֹֽא יִשְׂאוּ עָלָיו חֵטְא וּמֵתוּ בוֹ כִּי יְחַלְּלֻהוּ אֲנִי יְהוָֹה מְקַדְּשָֽׁם:  מ  וְשָֽׁמְרוּ אֶת מִשְׁמַרְתִּי וְלֹֽא יִשְׂאוּ עָלָיו חֵטְא וּמֵתוּ בוֹ כִּי יְחַלְּלֻהוּ אֲנִי יְהוָֹה מְקַדְּשָֽׁם:  ת  וְיִטְרוּן יָת מַטְּרַת מֵימְרִי וְלָא יְקַבְּלוּן עֲלוֹהִי חוֹבָא וִימוּתוּן בֵּהּ אֲרֵי יְחַלֻנֵהּ אֲנָא יְיָ מְקַדִשְׁהוֹן:

{י} וְכָל זָר לֹא יֹאכַל קֹדֶשׁ תּוֹשַׁב כֹּהֵן וְשָׂכִיר לֹא יֹאכַל קֹֽדֶשׁ:  מ  וְכָל זָר לֹא יֹאכַל קֹדֶשׁ תּוֹשַׁב כֹּהֵן וְשָׂכִיר לֹא יֹאכַל קֹֽדֶשׁ:  ת  וְכָל חִלוֹנַי לָא יֵיכוּל קוּדְשָׁא תּוֹתָבָא דְכַהֲנָא וַאֲגִירָא לָא יֵיכוּל קוּדְשָׁא:

{יא} וְכֹהֵן כִּֽי יִקְנֶה נֶפֶשׁ קִנְיַן כַּסְפּוֹ הוּא יֹאכַל בּוֹ וִילִיד בֵּיתוֹ הֵם יֹֽאכְלוּ בְלַחְמֽוֹ:  מ  וְכֹהֵן כִּֽי יִקְנֶה נֶפֶשׁ קִנְיַן כַּסְפּוֹ הוּא יֹאכַל בּוֹ וִילִיד בֵּיתוֹ הֵם יֹֽאכְלוּ בְלַחְמֽוֹ:  ת  וְכָהֵן אֲרֵי יִקְנֵי נְפַשׁ קִנְיַן כַּסְפֵּהּ הוּא יֵיכוּל בֵּהּ וִילִיד בֵּיתֵהּ אִנוּן יֵיכְלוּן בְּלַחְמֵהּ:

{יב} וּבַת כֹּהֵן כִּי תִֽהְיֶה לְאִישׁ זָר הִוא בִּתְרוּמַת הַקֳּדָשִׁים לֹא תֹאכֵֽל:  מ  וּבַת כֹּהֵן כִּי תִֽהְיֶה לְאִישׁ זָר הִוא בִּתְרוּמַת הַקֳּדָשִׁים לֹא תֹאכֵֽל:  ת  וּבַת כָּהֵן אֲרֵי תְהִי לִגְבַר חִלוֹנָי הִיא בְּאַפְרָשׁוּת קוּדְשַׁיָא לָא תֵיכוּל:

{יג} וּבַת כֹּהֵן כִּי תִֽהְיֶה אַלְמָנָה וּגְרוּשָׁה וְזֶרַע אֵין לָהּ וְשָׁבָה אֶל בֵּית אָבִיהָ כִּנְעוּרֶיהָ מִלֶּחֶם אָבִיהָ תֹּאכֵל וְכָל זָר לֹא יֹאכַל בּֽוֹ:  מ  וּבַת כֹּהֵן כִּי תִֽהְיֶה אַלְמָנָה וּגְרוּשָׁה וְזֶרַע אֵין לָהּ וְשָׁבָה אֶל בֵּית אָבִיהָ כִּנְעוּרֶיהָ מִלֶּחֶם אָבִיהָ תֹּאכֵל וְכָל זָר לֹא יֹאכַל בּֽוֹ:  ת  וּבַת כָּהֵן אֲרֵי תְהֵי אַרְמְלָא וּמְתָרְכָא וּבַר לֵית לַהּ וּתְתוּב לְבֵית אֲבוּהָא כְּרַבְיוּתָהָא מִלַחְמָא דַאֲבוּהָא תֵּיכוּל וְכָל חִלוֹנַי לָא יֵיכוּל בֵּהּ:

{יד} וְאִישׁ כִּֽי יֹאכַל קֹדֶשׁ בִּשְׁגָגָה וְיָסַף חֲמִֽשִׁיתוֹ עָלָיו וְנָתַן לַכֹּהֵן אֶת הַקֹּֽדֶשׁ:  מ  וְאִישׁ כִּֽי יֹאכַל קֹדֶשׁ בִּשְׁגָגָה וְיָסַף חֲמִֽשִׁיתוֹ עָלָיו וְנָתַן לַכֹּהֵן אֶת הַקֹּֽדֶשׁ:  ת  וּגְבַר אֲרֵי יֵיכוּל קוּדְשָׁא בְּשָׁלוּ וְיוֹסֵף חוּמְשֵׁהּ עֲלוֹהִי וְיִתֵּן לְכַהֲנָא יָת קוּדְשָׁא:

{טו} וְלֹא יְחַלְּלוּ אֶת קָדְשֵׁי בְּנֵי יִשְׂרָאֵל אֵת אֲשֶׁר יָרִימוּ לַֽיהוָֹֽה:  מ  וְלֹא יְחַלְּלוּ אֶת קָדְשֵׁי בְּנֵי יִשְׂרָאֵל אֵת אֲשֶׁר יָרִימוּ לַֽיהוָֹֽה:  ת  וְלָא יְחַלוּן יָת קוּדְשַׁיָא דִבְנֵי יִשְׂרָאֵל יָת דִי יַפְרְשׁוּן קֳדָם יְיָ:

{טז} וְהִשִּׂיאוּ אוֹתָם עֲוֹן אַשְׁמָה בְּאָכְלָם אֶת קָדְשֵׁיהֶם כִּי אֲנִי יְהוָֹה מְקַדְּשָֽׁם:  מ  וְהִשִּׂיאוּ אוֹתָם עֲוֹן אַשְׁמָה בְּאָכְלָם אֶת קָדְשֵׁיהֶם כִּי אֲנִי יְהוָֹה מְקַדְּשָֽׁם:  ת  וִיקַבְּלוּן עֲלֵיהוֹן עֲוָיָן וְחוֹבִין בְּמֵיכָלְהוֹן בְּסוֹאֲבָא יָת קוּדְשֵׁיהוֹן אֲרֵי אֲנָא יְיָ מְקַדִשְׁהוֹן: (פ)

{יז}  שלישי  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{יח} דַּבֵּר אֶֽל אַֽהֲרֹן וְאֶל בָּנָיו וְאֶל כָּל בְּנֵי יִשְׂרָאֵל וְאָֽמַרְתָּ אֲלֵהֶם אִישׁ אִישׁ מִבֵּית יִשְׂרָאֵל וּמִן הַגֵּר בְּיִשְׂרָאֵל אֲשֶׁר יַקְרִיב קָרְבָּנוֹ לְכָל נִדְרֵיהֶם וּלְכָל נִדְבוֹתָם אֲשֶׁר יַקְרִיבוּ לַֽיהוָֹה לְעֹלָֽה:  מ  דַּבֵּר אֶֽל אַֽהֲרֹן וְאֶל בָּנָיו וְאֶל כָּל בְּנֵי יִשְׂרָאֵל וְאָֽמַרְתָּ אֲלֵהֶם אִישׁ אִישׁ מִבֵּית יִשְׂרָאֵל וּמִן הַגֵּר בְּיִשְׂרָאֵל אֲשֶׁר יַקְרִיב קָרְבָּנוֹ לְכָל נִדְרֵיהֶם וּלְכָל נִדְבוֹתָם אֲשֶׁר יַקְרִיבוּ לַֽיהוָֹה לְעֹלָֽה:  ת  מַלֵל עִם אַהֲרֹן וְעִם בְּנוֹהִי וְעִם כָּל בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן גְבַר גְבַר מִבֵּית יִשְׂרָאֵל וּמִן גִיוֹרָא בְּיִשְׂרָאֵל דִי יְקָרֵב קֻרְבָּנֵהּ לְכָל נִדְרֵיהוֹן וּלְכָל נִדְבָתְהוֹן דִי יְקָרְבוּן קֳדָם יְיָ לַעֲלָתָא:

{יט} לִֽרְצֹֽנְכֶם תָּמִים זָכָר בַּבָּקָר בַּכְּשָׂבִים וּבָֽעִזִּֽים:  מ  לִֽרְצֹֽנְכֶם תָּמִים זָכָר בַּבָּקָר בַּכְּשָׂבִים וּבָֽעִזִּֽים:  ת  לְרַעֲוָא לְכוֹן שְׁלִים דְכוּרָא בְּתוֹרֵי בְּאִמְרַיָא וּבְעִזַיָא:

{כ} כֹּל אֲשֶׁר בּוֹ מוּם לֹא תַקְרִיבוּ כִּי לֹא לְרָצוֹן יִהְיֶה לָכֶֽם:  מ  כֹּל אֲשֶׁר בּוֹ מוּם לֹא תַקְרִיבוּ כִּי לֹא לְרָצוֹן יִהְיֶה לָכֶֽם:  ת  כֹּל דִי בֵהּ מוּמָא לָא תְקָרְבוּן אֲרֵי לָא לְרַעֲוָא יְהֵי לְכוֹן:

{כא} וְאִישׁ כִּֽי יַקְרִיב זֶֽבַח שְׁלָמִים לַֽיהֹוָה לְפַלֵּא נֶדֶר אוֹ לִנְדָבָה בַּבָּקָר אוֹ בַצֹּאן תָּמִים יִֽהְיֶה לְרָצוֹן כָּל מוּם לֹא יִֽהְיֶה בּֽוֹ:  מ  וְאִישׁ כִּֽי יַקְרִיב זֶֽבַח שְׁלָמִים לַֽיהֹוָה לְפַלֵּא נֶדֶר אוֹ לִנְדָבָה בַּבָּקָר אוֹ בַצֹּאן תָּמִים יִֽהְיֶה לְרָצוֹן כָּל מוּם לֹא יִֽהְיֶה בּֽוֹ:  ת  וּגְבַר אֲרֵי יְקָרֵב נִכְסַת קוּדְשַׁיָא קֳדָם יְיָ לְפָרָשָׁא נִדְרָא אוֹ לִנְדַבְתָּא בְּתוֹרֵי אוֹ בְעָנָא שְׁלִים יְהֵי לְרַעֲוָא כָּל מוּמָא לָא יְהֵי בֵהּ:

{כב} עַוֶּרֶת אוֹ שָׁבוּר אֽוֹ חָרוּץ אֽוֹ יַבֶּלֶת אוֹ גָרָב אוֹ יַלֶּפֶת לֹֽא תַקְרִיבוּ אֵלֶּה לַֽיהוָֹה וְאִשֶּׁה לֹֽא תִתְּנוּ מֵהֶם עַל הַמִּזְבֵּחַ לַֽיהוָֹֽה:  מ  עַוֶּרֶת אוֹ שָׁבוּר אֽוֹ חָרוּץ אֽוֹ יַבֶּלֶת אוֹ גָרָב אוֹ יַלֶּפֶת לֹֽא תַקְרִיבוּ אֵלֶּה לַֽיהוָֹה וְאִשֶּׁה לֹֽא תִתְּנוּ מֵהֶם עַל הַמִּזְבֵּחַ לַֽיהוָֹֽה:  ת  עַוִיר אוֹ תְבִיר אוֹ פְסִיק אוֹ יַבְּלָן אוֹ גַרְבָן אוֹ חֲזָזָן לָא תְקָרְבוּן אִלֵין קֳדָם יְיָ וְקֻרְבָּנָא לָא תִתְּנוּן מִנְהוֹן עַל מַדְבְּחָא קֳדָם יְיָ:

{כג} וְשׁוֹר וָשֶׂה שָׂרוּעַ וְקָלוּט נְדָבָה תַּֽעֲשֶׂה אֹתוֹ וּלְנֵדֶר לֹא יֵֽרָצֶֽה:  מ  וְשׁוֹר וָשֶׂה שָׂרוּעַ וְקָלוּט נְדָבָה תַּֽעֲשֶׂה אֹתוֹ וּלְנֵדֶר לֹא יֵֽרָצֶֽה:  ת  וְתוֹר וְאִמַר יַתִּיר וְחַסִיר נְדַבְתָּא תַּעְבֵּד יָתֵהּ וּלְנִדְרָא לָא יְהֵי לְרַעֲוָא:

{כד} וּמָעוּךְ וְכָתוּת וְנָתוּק וְכָרוּת לֹא תַקְרִיבוּ לַֽיהוָֹה וּֽבְאַרְצְכֶם לֹא תַֽעֲשֽׂוּ:  מ  וּמָעוּךְ וְכָתוּת וְנָתוּק וְכָרוּת לֹא תַקְרִיבוּ לַֽיהוָֹה וּֽבְאַרְצְכֶם לֹא תַֽעֲשֽׂוּ:  ת  וְדִי מְרִיס וְדִי רְסִיס וְדִי שְׁלִיף וְדִי גְזִיר לָא תְקָרְבוּן קֳדָם יְיָ וּבְאַרְעֲכוֹן לָא תַעְבְּדוּן:

{כה} וּמִיַּד בֶּן נֵכָר לֹא תַקְרִיבוּ אֶת לֶחֶם אֱלֹֽהֵיכֶם מִכָּל אֵלֶּה כִּי מָשְׁחָתָם בָּהֶם מוּם בָּם לֹא יֵֽרָצוּ לָכֶֽם:  מ  וּמִיַּד בֶּן נֵכָר לֹא תַקְרִיבוּ אֶת לֶחֶם אֱלֹֽהֵיכֶם מִכָּל אֵלֶּה כִּי מָשְׁחָתָם בָּהֶם מוּם בָּם לֹא יֵֽרָצוּ לָכֶֽם:  ת  וּמִן בַּר עַמְמִין לָא תְקַרְבוּן יָת קֻרְבַּן אֱלָהֲכוֹן מִכָּל אִלֵין אֲרֵי חִבּוּלְהוֹן בְּהוֹן מוּמָא בְהוֹן לָא לְרַעֲוָא יְהוֹן לְכוֹן: (ס)

{כו} וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{כז} שׁוֹר אוֹ כֶשֶׂב אוֹ עֵז כִּי יִוָּלֵד וְהָיָה שִׁבְעַת יָמִים תַּחַת אִמּוֹ וּמִיּוֹם הַשְּׁמִינִי וָהָלְאָה יֵֽרָצֶה לְקָרְבַּן אִשֶּׁה לַֽיהוָֹֽה:  מ  שׁוֹר אוֹ כֶשֶׂב אוֹ עֵז כִּי יִוָּלֵד וְהָיָה שִׁבְעַת יָמִים תַּחַת אִמּוֹ וּמִיּוֹם הַשְּׁמִינִי וָהָלְאָה יֵֽרָצֶה לְקָרְבַּן אִשֶּׁה לַֽיהוָֹֽה:  ת  תּוֹר אוֹ אִמָר אוֹ עִזָא אֲרֵי יִתְיְלִיד וִיהֵי שִׁבְעָא יוֹמִין בָּתַר אִמֵהּ וּמִיוֹמָא תְמִינָאָה וּלְהַלָא יִתִּרְעִי לְקָרָבָא קֻרְבָּנָא קֳדָם יְיָ:

{כח} וְשׁוֹר אוֹ שֶׂה אֹתוֹ וְאֶת בְּנוֹ לֹא תִשְׁחֲטוּ בְּיוֹם אֶחָֽד:  מ  וְשׁוֹר אוֹ שֶׂה אֹתוֹ וְאֶת בְּנוֹ לֹא תִשְׁחֲטוּ בְּיוֹם אֶחָֽד:  ת  וְתוֹרְתָּא אוֹ שֵׂיתָא לֵהּ וְלִבְרֵהּ לָא תִכְּסוּן בְּיוֹמָא חָד:

{כט} וְכִֽי תִזְבְּחוּ זֶֽבַח תּוֹדָה לַֽיהוָֹה לִֽרְצֹֽנְכֶם תִּזְבָּֽחוּ:  מ  וְכִֽי תִזְבְּחוּ זֶֽבַח תּוֹדָה לַֽיהוָֹה לִֽרְצֹֽנְכֶם תִּזְבָּֽחוּ:  ת  וַאֲרֵי תִּכְּסוּן נִכְסַת תּוֹדְתָא קֳדָם יְיָ לְרַעֲוָא לְכוֹן תִּכְּסֻנֵהּ:

{ל} בַּיּוֹם הַהוּא יֵֽאָכֵל לֹֽא תוֹתִירוּ מִמֶּנּוּ עַד בֹּקֶר אֲנִי יְהוָֹֽה:  מ  בַּיּוֹם הַהוּא יֵֽאָכֵל לֹֽא תוֹתִירוּ מִמֶּנּוּ עַד בֹּקֶר אֲנִי יְהוָֹֽה:  ת  בְּיוֹמָא הַהוּא יִתְאֲכֵל לָא תַשְׁאֲרוּן מִנֵהּ עַד צַפְרָא אֲנָא יְיָ:

{לא} וּשְׁמַרְתֶּם מִצְוֹתַי וַֽעֲשִׂיתֶם אֹתָם אֲנִי יְהוָֹֽה:  מ  וּשְׁמַרְתֶּם מִצְוֹתַי וַֽעֲשִׂיתֶם אֹתָם אֲנִי יְהוָֹֽה:  ת  וְתִטְרוּן פִּקוּדַי וְתַעֲבְּדוּן יָתְהוֹן אֲנָא יְיָ:

{לב} וְלֹא תְחַלְּלוּ אֶת שֵׁם קָדְשִׁי וְנִקְדַּשְׁתִּי בְּתוֹךְ בְּנֵי יִשְׂרָאֵל אֲנִי יְהוָֹה מְקַדִּשְׁכֶֽם:  מ  וְלֹא תְחַלְּלוּ אֶת שֵׁם קָדְשִׁי וְנִקְדַּשְׁתִּי בְּתוֹךְ בְּנֵי יִשְׂרָאֵל אֲנִי יְהוָֹה מְקַדִּשְׁכֶֽם:  ת  וְלָא תְחַלְלוּן יָת שְׁמָא דְקוּדְשַׁי וְאִתְקַדַשׁ בְּגוֹ בְּנֵי יִשְׂרָאֵל אֲנָא יְיָ מְקַדִשְּׁכוֹן:

{לג} הַמּוֹצִיא אֶתְכֶם מֵאֶרֶץ מִצְרַיִם לִהְיוֹת לָכֶם לֵֽאלֹהִים אֲנִי יְהוָֹֽה:  מ  הַמּוֹצִיא אֶתְכֶם מֵאֶרֶץ מִצְרַיִם לִהְיוֹת לָכֶם לֵֽאלֹהִים אֲנִי יְהוָֹֽה:  ת  דְאַפִּיק יָתְכוֹן מֵאַרְעָא דְמִצְרַיִם לְמֶהֱוֵי לְכוֹן לֵאלָהָא אֲנָא יְיָ: (פ)




ויקרא פרק-כג

{א}  רביעי  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָֽמַרְתָּ אֲלֵהֶם מֽוֹעֲדֵי יְהֹוָה אֲשֶׁר תִּקְרְאוּ אֹתָם מִקְרָאֵי קֹדֶשׁ אֵלֶּה הֵם מֽוֹעֲדָֽי:  מ  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָֽמַרְתָּ אֲלֵהֶם מֽוֹעֲדֵי יְהֹוָה אֲשֶׁר תִּקְרְאוּ אֹתָם מִקְרָאֵי קֹדֶשׁ אֵלֶּה הֵם מֽוֹעֲדָֽי:  ת  מַלֵיל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן מוֹעֲדַיָא דַיְיָ דִי תְעָרְעוּן יָתְהוֹן מְעָרְעֵי קַדִישׁ אִלֵין אִנוּן מוֹעֲדָי:

{ג} שֵׁשֶׁת יָמִים תֵּֽעָשֶׂה מְלָאכָה וּבַיּוֹם הַשְּׁבִיעִי שַׁבַּת שַׁבָּתוֹן מִקְרָא קֹדֶשׁ כָּל מְלָאכָה לֹא תַֽעֲשׂוּ שַׁבָּת הִוא לַֽיהֹוָה בְּכֹל מֽוֹשְׁבֹֽתֵיכֶֽם:  מ  שֵׁשֶׁת יָמִים תֵּֽעָשֶׂה מְלָאכָה וּבַיּוֹם הַשְּׁבִיעִי שַׁבַּת שַׁבָּתוֹן מִקְרָא קֹדֶשׁ כָּל מְלָאכָה לֹא תַֽעֲשׂוּ שַׁבָּת הִוא לַֽיהֹוָה בְּכֹל מֽוֹשְׁבֹֽתֵיכֶֽם:  ת  שִׁתָּא יוֹמִין תִּתְעֲבֵד עִבִידָא וּבְיוֹמָא שְׁבִיעָאָה שְׁבַת שְׁבָתָא מְעָרַע קַדִישׁ כָּל עִבִידָא לָא תַעֲבְּדוּן שַׁבְּתָא הִיא קֳדָם יְיָ בְּכֹל מוֹתְבָנֵיכוֹן: (פ)

{ד} אֵלֶּה מֽוֹעֲדֵי יְהֹוָה מִקְרָאֵי קֹדֶשׁ אֲשֶׁר תִּקְרְאוּ אֹתָם בְּמֽוֹעֲדָֽם:  מ  אֵלֶּה מֽוֹעֲדֵי יְהֹוָה מִקְרָאֵי קֹדֶשׁ אֲשֶׁר תִּקְרְאוּ אֹתָם בְּמֽוֹעֲדָֽם:  ת  אִלֵין מוֹעֲדַיָא דַיְיָ מְעָרְעֵי קַדִישׁ דִי תְעָרְעוּן יָתְהוֹן בְּזִמְנֵיהוֹן:

{ה} בַּחֹדֶשׁ הָֽרִאשׁוֹן בְּאַרְבָּעָה עָשָׂר לַחֹדֶשׁ בֵּין הָֽעַרְבָּיִם פֶּסַח לַֽיהוָֹֽה:  מ  בַּחֹדֶשׁ הָֽרִאשׁוֹן בְּאַרְבָּעָה עָשָׂר לַחֹדֶשׁ בֵּין הָֽעַרְבָּיִם פֶּסַח לַֽיהוָֹֽה:  ת  בְּיַרְחָא קַדְמָאָה בְּאַרְבְּעַת עַשְׂרָא לְיַרְחָא בֵּין שִׁמְשַׁיָא פִּסְחָא קֳדָם יְיָ:

{ו} וּבַֽחֲמִשָּׁה עָשָׂר יוֹם לַחֹדֶשׁ הַזֶּה חַג הַמַּצּוֹת לַֽיהוָֹה שִׁבְעַת יָמִים מַצּוֹת תֹּאכֵֽלוּ:  מ  וּבַֽחֲמִשָּׁה עָשָׂר יוֹם לַחֹדֶשׁ הַזֶּה חַג הַמַּצּוֹת לַֽיהוָֹה שִׁבְעַת יָמִים מַצּוֹת תֹּאכֵֽלוּ:  ת  וּבְחַמְשַׁת עַשְׂרָא יוֹמָא לְיַרְחָא הָדֵין חַגָא דְפַטִירַיָא קֳדָם יְיָ שִׁבְעָא יוֹמִין פַּטִיר תֵּיכְלוּן:

{ז} בַּיּוֹם הָֽרִאשׁוֹן מִקְרָא קֹדֶשׁ יִהְיֶה לָכֶם כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשֽׂוּ:  מ  בַּיּוֹם הָֽרִאשׁוֹן מִקְרָא קֹדֶשׁ יִהְיֶה לָכֶם כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשֽׂוּ:  ת  בְּיוֹמָא קַדְמָאָה מְעָרַע קַדִישׁ יְהֵי לְכוֹן כָּל עִבִידַת פָּלְחַן לָא תַעֲבְּדוּן:

{ח} וְהִקְרַבְתֶּם אִשֶּׁה לַֽיהוָֹה שִׁבְעַת יָמִים בַּיּוֹם הַשְּׁבִיעִי מִקְרָא קֹדֶשׁ כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשֽׂוּ:  מ  וְהִקְרַבְתֶּם אִשֶּׁה לַֽיהוָֹה שִׁבְעַת יָמִים בַּיּוֹם הַשְּׁבִיעִי מִקְרָא קֹדֶשׁ כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשֽׂוּ:  ת  וּתְקָרְבוּן קֻרְבָּנָא קֳדָם יְיָ שִׁבְעָא יוֹמִין בְּיוֹמָא שְׁבִיעָאָה מְעָרַע קַדִישׁ כָּל עִבִידַת פָּלְחָן לָא תַעֲבְּדוּן: (פ)

{ט} וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{י} דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָֽמַרְתָּ אֲלֵהֶם כִּֽי תָבֹאוּ אֶל הָאָרֶץ אֲשֶׁר אֲנִי נֹתֵן לָכֶם וּקְצַרְתֶּם אֶת קְצִירָהּ וַֽהֲבֵאתֶם אֶת עֹמֶר רֵאשִׁית קְצִֽירְכֶם אֶל הַכֹּהֵֽן:  מ  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָֽמַרְתָּ אֲלֵהֶם כִּֽי תָבֹאוּ אֶל הָאָרֶץ אֲשֶׁר אֲנִי נֹתֵן לָכֶם וּקְצַרְתֶּם אֶת קְצִירָהּ וַֽהֲבֵאתֶם אֶת עֹמֶר רֵאשִׁית קְצִֽירְכֶם אֶל הַכֹּהֵֽן:  ת  מַלֵיל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן אֲרֵי תֵעֲלוּן לְאַרְעָא דִי אֲנָא יָהֵב לְכוֹן וְתַחְצְדוּן יָת חֲצָדַהּ וְתַיְתוּן יָת עוּמְרָא רֵישׁ חֲצָדְכוֹן לְוָת כַּהֲנָא:

{יא} וְהֵנִיף אֶת הָעֹמֶר לִפְנֵי יְהוָֹה לִֽרְצֹֽנְכֶם מִֽמָּֽחֳרַת הַשַּׁבָּת יְנִיפֶנּוּ הַכֹּהֵֽן:  מ  וְהֵנִיף אֶת הָעֹמֶר לִפְנֵי יְהוָֹה לִֽרְצֹֽנְכֶם מִֽמָּֽחֳרַת הַשַּׁבָּת יְנִיפֶנּוּ הַכֹּהֵֽן:  ת  וִירִים יָת עוּמְרָא קֳדָם יְיָ לְרַעֲוָא לְכוֹן מִבָּתַר יוֹמָא טָבָא יְרִימִנֵהּ כַּהֲנָא:

{יב} וַֽעֲשִׂיתֶם בְּיוֹם הֲנִֽיפְכֶם אֶת הָעֹמֶר כֶּבֶשׂ תָּמִים בֶּן שְׁנָתוֹ לְעֹלָה לַֽיהוָֹֽה:  מ  וַֽעֲשִׂיתֶם בְּיוֹם הֲנִֽיפְכֶם אֶת הָעֹמֶר כֶּבֶשׂ תָּמִים בֶּן שְׁנָתוֹ לְעֹלָה לַֽיהוָֹֽה:  ת  וְתַעֲבְּדוּן בְּיוֹמָא דַאֲרָמוּתְכוֹן יָת עוּמְרָא אִמַר שְׁלִים בַּר שַׁתֵּהּ לַעֲלָתָא קֳדָם יְיָ:

{יג} וּמִנְחָתוֹ שְׁנֵי עֶשְׂרֹנִים סֹלֶת בְּלוּלָה בַשֶּׁמֶן אִשֶּׁה לַֽיהוָֹה רֵיחַ נִיחֹחַ וְנִסְכֹּה יַיִן רְבִיעִת הַהִֽין:  מ  וּמִנְחָתוֹ שְׁנֵי עֶשְׂרֹנִים סֹלֶת בְּלוּלָה בַשֶּׁמֶן אִשֶּׁה לַֽיהוָֹה רֵיחַ נִיחֹחַ וְנִסְכֹּה יַיִן רְבִיעִת הַהִֽין:  ת  וּמִנְחָתֵהּ תְּרֵין עֶשְׂרוֹנִין סֻלְתָּא דְפִילָא בִמְשַׁח קֻרְבָּנָא קֳדָם יְיָ לְאִתְקַבָּלָא בְרַעֲוָא וְנִסְכֵּהּ חַמְרָא רַבְעוּת הִינָא:

{יד} וְלֶחֶם וְקָלִי וְכַרְמֶל לֹא תֹֽאכְלוּ עַד עֶצֶם הַיּוֹם הַזֶּה עַד הֲבִיאֲכֶם אֶת קָרְבַּן אֱלֹֽהֵיכֶם חֻקַּת עוֹלָם לְדֹרֹתֵיכֶם בְּכֹל מֽשְׁבֹֽתֵיכֶֽם:  מ  וְלֶחֶם וְקָלִי וְכַרְמֶל לֹא תֹֽאכְלוּ עַד עֶצֶם הַיּוֹם הַזֶּה עַד הֲבִיאֲכֶם אֶת קָרְבַּן אֱלֹֽהֵיכֶם חֻקַּת עוֹלָם לְדֹרֹתֵיכֶם בְּכֹל מֽשְׁבֹֽתֵיכֶֽם:  ת  וּלְחֵם וּקְלִי וּפֵרוּכָן לָא תֵיכְלוּן עַד כְּרַן יוֹמָא הָדֵין עַד אַיְתוֹאֵיכוֹן יָת קֻרְבָּן אֱלָהָכוֹן קְיַם עָלָם לְדָרֵיכוֹן בְּכֹל מוֹתְבָנֵיכוֹן: (ס)

{טו} וּסְפַרְתֶּם לָכֶם מִמָּֽחֳרַת הַשַּׁבָּת מִיּוֹם הֲבִיאֲכֶם אֶת עֹמֶר הַתְּנוּפָה שֶׁבַע שַׁבָּתוֹת תְּמִימֹת תִּֽהְיֶֽינָה:  מ  וּסְפַרְתֶּם לָכֶם מִמָּֽחֳרַת הַשַּׁבָּת מִיּוֹם הֲבִיאֲכֶם אֶת עֹמֶר הַתְּנוּפָה שֶׁבַע שַׁבָּתוֹת תְּמִימֹת תִּֽהְיֶֽינָה:  ת  וְתִמְנוּן לְכוֹן מִבָּתַר יוֹמָא טָּבָא מִיוֹם אַיְתוֹאֵיכוֹן יָת עוּמְרָא דַאֲרָמוּתָא שִׁבְעָא שָׁבוּעִין שַּׁלְמִין יְהֶוְיָן (נ''א תְּהֶוְיָן):

{טז} עַד מִמָּֽחֳרַת הַשַּׁבָּת הַשְּׁבִיעִת תִּסְפְּרוּ חֲמִשִּׁים יוֹם וְהִקְרַבְתֶּם מִנְחָה חֲדָשָׁה לַֽיהוָֹֽה:  מ  עַד מִמָּֽחֳרַת הַשַּׁבָּת הַשְּׁבִיעִת תִּסְפְּרוּ חֲמִשִּׁים יוֹם וְהִקְרַבְתֶּם מִנְחָה חֲדָשָׁה לַֽיהוָֹֽה:  ת  עַד מִבָּתַר שְׁבוּעֲתָא שְׁבִיעֵתָא תִּמְנוּן חַמְשִׁין יוֹמִין וּתְקַרְבוּן מִנְחָתָא חֲדַתָּא קֳדָם יְיָ:

{יז} מִמּֽוֹשְׁבֹתֵיכֶם תָּבִיאִוּ לֶחֶם תְּנוּפָה שְׁתַּיִם שְׁנֵי עֶשְׂרֹנִים סֹלֶת תִּֽהְיֶינָה חָמֵץ תֵּֽאָפֶינָה בִּכּוּרִים לַֽיהוָֹֽה:  מ  מִמּֽוֹשְׁבֹתֵיכֶם תָּבִיאִוּ לֶחֶם תְּנוּפָה שְׁתַּיִם שְׁנֵי עֶשְׂרֹנִים סֹלֶת תִּֽהְיֶינָה חָמֵץ תֵּֽאָפֶינָה בִּכּוּרִים לַֽיהוָֹֽה:  ת  מִמוֹתְבָנֵיכוֹן תַּיְתוּן לְחֵם אֲרָמוּתָא תַּרְתֵּין (גְרִיצַן) תְּרֵין עֶשְׂרוֹנִין סֻלְתָּא יְהֶוְיָן חֲמִיעַ יִתְאַפְיָן בִּכּוּרִין קֳדָם יְיָ:

{יח} וְהִקְרַבְתֶּם עַל הַלֶּחֶם שִׁבְעַת כְּבָשִׂים תְּמִימִם בְּנֵי שָׁנָה וּפַר בֶּן בָּקָר אֶחָד וְאֵילִם שְׁנָיִם יִהְיוּ עֹלָה לַֽיהֹוָה וּמִנְחָתָם וְנִסְכֵּיהֶם אִשֵּׁה רֵֽיחַ נִיחֹחַ לַֽיהוָֹֽה:  מ  וְהִקְרַבְתֶּם עַל הַלֶּחֶם שִׁבְעַת כְּבָשִׂים תְּמִימִם בְּנֵי שָׁנָה וּפַר בֶּן בָּקָר אֶחָד וְאֵילִם שְׁנָיִם יִהְיוּ עֹלָה לַֽיהֹוָה וּמִנְחָתָם וְנִסְכֵּיהֶם אִשֵּׁה רֵֽיחַ נִיחֹחַ לַֽיהוָֹֽה:  ת  וּתְקָרְבוּן עַל לַחְמָא שִׁבְעָא אִמְרִין שַׁלְמִין בְּנֵי שְׁנָא וְתוֹר בַּר תּוֹרֵי חַד וְדִכְרִין תְּרֵין יְהוֹן עֲלָתָא קֳדָם יְיָ וּמִנְחָתְהוֹן וְנִסְכֵּיהוֹן קֻרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ:

{יט} וַֽעֲשִׂיתֶם שְׂעִיר עִזִּים אֶחָד לְחַטָּאת וּשְׁנֵי כְבָשִׂים בְּנֵי שָׁנָה לְזֶבַח שְׁלָמִֽים:  מ  וַֽעֲשִׂיתֶם שְׂעִיר עִזִּים אֶחָד לְחַטָּאת וּשְׁנֵי כְבָשִׂים בְּנֵי שָׁנָה לְזֶבַח שְׁלָמִֽים:  ת  וְתַעְבְּדוּן צְפִיר בַּר עִזֵי חַד לְחַטָאתָא וּתְרֵין אִמְרִין בְּנֵי שְׁנָא לְנִכְסַת קוּדְשַׁיָא:

{כ} וְהֵנִיף הַכֹּהֵן אֹתָם עַל לֶחֶם הַבִּכֻּרִים תְּנוּפָה לִפְנֵי יְהֹוָה עַל שְׁנֵי כְּבָשִׂים קֹדֶשׁ יִהְיוּ לַֽיהוָֹה לַכֹּהֵֽן:  מ  וְהֵנִיף הַכֹּהֵן אֹתָם עַל לֶחֶם הַבִּכֻּרִים תְּנוּפָה לִפְנֵי יְהֹוָה עַל שְׁנֵי כְּבָשִׂים קֹדֶשׁ יִהְיוּ לַֽיהוָֹה לַכֹּהֵֽן:  ת  וִירִים כַּהֲנָא יָתְהוֹן עַל לְחֵם בִּכּוּרַיָא אֲרָמוּתָא קֳדָם יְיָ עַל תְּרֵין אִמְרִין קוּדְשָׁא יְהוֹן קֳדָם יְיָ לְכַהֲנָא:

{כא} וּקְרָאתֶם בְּעֶצֶם הַיּוֹם הַזֶּה מִֽקְרָא קֹדֶשׁ יִהְיֶה לָכֶם כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשֹׂוּ חֻקַּת עוֹלָם בְּכָל מֽוֹשְׁבֹתֵיכֶם לְדֹרֹֽתֵיכֶֽם:  מ  וּקְרָאתֶם בְּעֶצֶם הַיּוֹם הַזֶּה מִֽקְרָא קֹדֶשׁ יִהְיֶה לָכֶם כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשֹׂוּ חֻקַּת עוֹלָם בְּכָל מֽוֹשְׁבֹתֵיכֶם לְדֹרֹֽתֵיכֶֽם:  ת  וּתְעָרְעוּן בִּכְרַן יוֹמָא הָדֵין מְעָרַע קַדִישׁ יְהֵי לְכוֹן כָּל עִבִידַת פָּלְחָן לָא תַעֲבְּדוּן קְיַם עָלָם בְּכָל מוֹתְבָנֵיכוֹן לְדָרֵיכוֹן:

{כב} וּֽבְקֻצְרְכֶם אֶת קְצִיר אַרְצְכֶם לֹֽא תְכַלֶּה פְּאַת שָֽׂדְךָ בְּקֻצְרֶךָ וְלֶקֶט קְצִֽירְךָ לֹא תְלַקֵּט לֶֽעָנִי וְלַגֵּר תַּֽעֲזֹב אֹתָם אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  מ  וּֽבְקֻצְרְכֶם אֶת קְצִיר אַרְצְכֶם לֹֽא תְכַלֶּה פְּאַת שָֽׂדְךָ בְּקֻצְרֶךָ וְלֶקֶט קְצִֽירְךָ לֹא תְלַקֵּט לֶֽעָנִי וְלַגֵּר תַּֽעֲזֹב אֹתָם אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  ת  וּבְמֶחֱצָדְכוֹן יָת חֲצָדָא דְאַרְעֲכוֹן לָא תְשֵׁיצֵי פָּאתָא דְחַקְלָךְ בְּמֶחֱצָדָךְ וּלְקָטָא דַחֲצָדָךְ לָא תְלַקֵט לְעַנְיֵי וּלְגִיוֹרֵי תִּשְׁבּוֹק יָתְהוֹן אֲנָא יְיָ אֱלָהֲכוֹן: (פ)

{כג}  חמישי  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{כד} דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר בַּחֹדֶשׁ הַשְּׁבִיעִי בְּאֶחָד לַחֹדֶשׁ יִהְיֶה לָכֶם שַׁבָּתוֹן זִכְרוֹן תְּרוּעָה מִקְרָא קֹֽדֶשׁ:  מ  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר בַּחֹדֶשׁ הַשְּׁבִיעִי בְּאֶחָד לַחֹדֶשׁ יִהְיֶה לָכֶם שַׁבָּתוֹן זִכְרוֹן תְּרוּעָה מִקְרָא קֹֽדֶשׁ:  ת  מַלֵיל עִם בְּנֵי יִשְׂרָאֵל לְמֵימָר בְּיַרְחָא שְׁבִיעָאָה בְּחַד לְיַרְחָא יְהֵי לְכוֹן נְיָחָא דוּכְרָן יַבָּבָא מְעָרַע קַדִישׁ:

{כה} כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשׂוּ וְהִקְרַבְתֶּם אִשֶּׁה לַֽיהוָֹֽה:  מ  כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשׂוּ וְהִקְרַבְתֶּם אִשֶּׁה לַֽיהוָֹֽה:  ת  כָּל עִבִידַת פָלְחָנָא לָא תַעֲבְּדוּן וּתְקָרְבוּן קֻרְבָּנָא קֳדָם יְיָ: (ס)

{כו} וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{כז} אַךְ בֶּֽעָשׂוֹר לַחֹדֶשׁ הַשְּׁבִיעִי הַזֶּה יוֹם הַכִּפֻּרִים הוּא מִֽקְרָא קֹדֶשׁ יִהְיֶה לָכֶם וְעִנִּיתֶם אֶת נַפְשֹֽׁתֵיכֶם וְהִקְרַבְתֶּם אִשֶּׁה לַֽיהוָֹֽה:  מ  אַךְ בֶּֽעָשׂוֹר לַחֹדֶשׁ הַשְּׁבִיעִי הַזֶּה יוֹם הַכִּפֻּרִים הוּא מִֽקְרָא קֹדֶשׁ יִהְיֶה לָכֶם וְעִנִּיתֶם אֶת נַפְשֹֽׁתֵיכֶם וְהִקְרַבְתֶּם אִשֶּׁה לַֽיהוָֹֽה:  ת  בְּרַם בְּעַשְׂרָא לְיַרְחָא שְׁבִיעָאָה הָדֵין יוֹמָא דְכִפּוּרַיָא הוּא מְעָרַע קַדִישׁ יְהֵי לְכוֹן וּתְעַנוּן יָת נַפְשָׁתֵיכוֹן וּתְקָרְבוּן קֻרְבָּנָא קֳדָם יְיָ:

{כח} וְכָל מְלָאכָה לֹא תַֽעֲשׂוּ בְּעֶצֶם הַיּוֹם הַזֶּה כִּי יוֹם כִּפֻּרִים הוּא לְכַפֵּר עֲלֵיכֶם לִפְנֵי יְהוָֹה אֱלֹֽהֵיכֶֽם:  מ  וְכָל מְלָאכָה לֹא תַֽעֲשׂוּ בְּעֶצֶם הַיּוֹם הַזֶּה כִּי יוֹם כִּפֻּרִים הוּא לְכַפֵּר עֲלֵיכֶם לִפְנֵי יְהוָֹה אֱלֹֽהֵיכֶֽם:  ת  וְכָל עִבִידָא לָא תַעֲבְּדוּן בִּכְרַן יוֹמָא הָדֵין אֲרֵי יוֹמָא דְכִפּוּרַיָא הוּא לְכַפָּרָא עֲלֵיכוֹן קֳדָם יְיָ אֱלָהֲכוֹן:

{כט} כִּי כָל הַנֶּפֶשׁ אֲשֶׁר לֹֽא תְעֻנֶּה בְּעֶצֶם הַיּוֹם הַזֶּה וְנִכְרְתָה מֵֽעַמֶּֽיהָ:  מ  כִּי כָל הַנֶּפֶשׁ אֲשֶׁר לֹֽא תְעֻנֶּה בְּעֶצֶם הַיּוֹם הַזֶּה וְנִכְרְתָה מֵֽעַמֶּֽיהָ:  ת  אֲרֵי כָל אֲנָשָׁא דִי לָא מִתְעַנֵי בִּכְרַן יוֹמָא הָדֵין וְיִשְׁתֵּיצֵי מֵעַמָהָא:

{ל} וְכָל הַנֶּפֶשׁ אֲשֶׁר תַּֽעֲשֶׂה כָּל מְלָאכָה בְּעֶצֶם הַיּוֹם הַזֶּה וְהַֽאֲבַדְתִּי אֶת הַנֶּפֶשׁ הַהִוא מִקֶּרֶב עַמָּֽהּ:  מ  וְכָל הַנֶּפֶשׁ אֲשֶׁר תַּֽעֲשֶׂה כָּל מְלָאכָה בְּעֶצֶם הַיּוֹם הַזֶּה וְהַֽאֲבַדְתִּי אֶת הַנֶּפֶשׁ הַהִוא מִקֶּרֶב עַמָּֽהּ:  ת  וְכָל אֱנַשׁ דִי תַעְבֵּד כָּל עִבִידָא בִּכְרַן יוֹמָא הָדֵין וְאוֹבֵד יָת אֲנָשָׁא הַהִיא מִגוֹ עַמָהּ:

{לא} כָּל מְלָאכָה לֹא תַֽעֲשׂוּ חֻקַּת עוֹלָם לְדֹרֹתֵיכֶם בְּכֹל מֽשְׁבֹֽתֵיכֶֽם:  מ  כָּל מְלָאכָה לֹא תַֽעֲשׂוּ חֻקַּת עוֹלָם לְדֹרֹתֵיכֶם בְּכֹל מֽשְׁבֹֽתֵיכֶֽם:  ת  כָּל עִבִידָא לָא תַעֲבְּדוּן קְיַם עָלָם לְדָרֵיכוֹן בְּכֹל מוֹתְבָנֵיכוֹן:

{לב} שַׁבַּת שַׁבָּתוֹן הוּא לָכֶם וְעִנִּיתֶם אֶת נַפְשֹֽׁתֵיכֶם בְּתִשְׁעָה לַחֹדֶשׁ בָּעֶרֶב מֵעֶרֶב עַד עֶרֶב תִּשְׁבְּתוּ שַֽׁבַּתְּכֶֽם:  מ  שַׁבַּת שַׁבָּתוֹן הוּא לָכֶם וְעִנִּיתֶם אֶת נַפְשֹֽׁתֵיכֶם בְּתִשְׁעָה לַחֹדֶשׁ בָּעֶרֶב מֵעֶרֶב עַד עֶרֶב תִּשְׁבְּתוּ שַֽׁבַּתְּכֶֽם:  ת  שְׁבַת שְׁבָתָא הוּא לְכוֹן וּתְעַנוּן יָת נַפְשָׁתֵיכוֹן בְּתִשְׁעָה לְיַרְחָא בְּרַמְשָׁא מֵרַמְשָׁא עַד רַמְשָׁא תְּנוּחוּן נְיַחֲכוֹן: (פ)

{לג}  שישי  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{לד} דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר בַּֽחֲמִשָּׁה עָשָׂר יוֹם לַחֹדֶשׁ הַשְּׁבִיעִי הַזֶּה חַג הַסֻּכּוֹת שִׁבְעַת יָמִים לַֽיהוָֹֽה:  מ  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר בַּֽחֲמִשָּׁה עָשָׂר יוֹם לַחֹדֶשׁ הַשְּׁבִיעִי הַזֶּה חַג הַסֻּכּוֹת שִׁבְעַת יָמִים לַֽיהוָֹֽה:  ת  מַלֵיל עִם בְּנֵי יִשְׂרָאֵל לְמֵימָר בְּחַמְשָׁא עַשְׂרָא יוֹמָא לְיַרְחָא שְׁבִיעָאָה הָדֵין חַגָא דִמְטַלַיָא שִׁבְעַת יוֹמִין קֳדָם יְיָ:

{לה} בַּיּוֹם הָֽרִאשׁוֹן מִקְרָא קֹדֶשׁ כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשֽׂוּ:  מ  בַּיּוֹם הָֽרִאשׁוֹן מִקְרָא קֹדֶשׁ כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשֽׂוּ:  ת  בְּיּוֹמָא קַדְמָאָה מְעָרַע קַדִישׁ כָּל עִבִידַת פָּלְחָן לָא תַעֲבְּדוּן:

{לו} שִׁבְעַת יָמִים תַּקְרִיבוּ אִשֶּׁה לַֽיהוָֹה בַּיּוֹם הַשְּׁמִינִי מִקְרָא קֹדֶשׁ יִהְיֶה לָכֶם וְהִקְרַבְתֶּם אִשֶּׁה לַֽיהוָֹה עֲצֶרֶת הִוא כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשֽׂוּ:  מ  שִׁבְעַת יָמִים תַּקְרִיבוּ אִשֶּׁה לַֽיהוָֹה בַּיּוֹם הַשְּׁמִינִי מִקְרָא קֹדֶשׁ יִהְיֶה לָכֶם וְהִקְרַבְתֶּם אִשֶּׁה לַֽיהוָֹה עֲצֶרֶת הִוא כָּל מְלֶאכֶת עֲבֹדָה לֹא תַֽעֲשֽׂוּ:  ת  שִׁבְעָא יוֹמִין תְּקָרְבוּן קֻרְבָּנָא קֳדָם יְיָ בְּיוֹמָא תְמִינָאָה מְעָרַע קַדִישׁ יְהֵי לְכוֹן וּתְקָרְבוּן קֻרְבָּנָא קֳדָם יְיָ כְּנֵשׁ תְּהוֹן כָּל עִבִידַת פָּלְחַן לָא תַעֲבְּדוּן:

{לז} אֵלֶּה מֽוֹעֲדֵי יְהֹוָה אֲשֶׁר תִּקְרְאוּ אֹתָם מִקְרָאֵי קֹדֶשׁ לְהַקְרִיב אִשֶּׁה לַֽיהֹוָה עֹלָה וּמִנְחָה זֶבַח וּנְסָכִים דְּבַר יוֹם בְּיוֹמֽוֹ:  מ  אֵלֶּה מֽוֹעֲדֵי יְהֹוָה אֲשֶׁר תִּקְרְאוּ אֹתָם מִקְרָאֵי קֹדֶשׁ לְהַקְרִיב אִשֶּׁה לַֽיהֹוָה עֹלָה וּמִנְחָה זֶבַח וּנְסָכִים דְּבַר יוֹם בְּיוֹמֽוֹ:  ת  אִלֵין מוֹעֲדַיָא דַיְיָ דִי תְעָרְעוּן יָתְהוֹן מְעָרְעֵי קַדִישׁ לְקָרָבָא קֻרְבָּנָא קֳדָם יְיָ עֲלָתָא וּמִנְחָתָא נִכְסַת קוּדְשִׁין וְנִסוּכִין פִּתְגַם יוֹם בְּיוֹמֵהּ:

{לח} מִלְּבַד שַׁבְּתֹת יְהוָֹה וּמִלְּבַד מַתְּנֽוֹתֵיכֶם וּמִלְּבַד כָּל נִדְרֵיכֶם וּמִלְּבַד כָּל נִדְבֹתֵיכֶם אֲשֶׁר תִּתְּנוּ לַֽיהוָֹֽה:  מ  מִלְּבַד שַׁבְּתֹת יְהוָֹה וּמִלְּבַד מַתְּנֽוֹתֵיכֶם וּמִלְּבַד כָּל נִדְרֵיכֶם וּמִלְּבַד כָּל נִדְבֹתֵיכֶם אֲשֶׁר תִּתְּנוּ לַֽיהוָֹֽה:  ת  בַּר שַׁבַּיָא דַיְיָ וּבַר מַתְּנָתֵיכוֹן וּבַר כָּל נִדְרֵיכוֹן וּבַר כָּל נִדְבָתֵיכוֹן דִי תִתְּנוּן קֳדָם יְיָ:

{לט} אַךְ בַּֽחֲמִשָּׁה עָשָׂר יוֹם לַחֹדֶשׁ הַשְּׁבִיעִי בְּאָסְפְּכֶם אֶת תְּבוּאַת הָאָרֶץ תָּחֹגּוּ אֶת חַג יְהוָֹה שִׁבְעַת יָמִים בַּיּוֹם הָֽרִאשׁוֹן שַׁבָּתוֹן וּבַיּוֹם הַשְּׁמִינִי שַׁבָּתֽוֹן:  מ  אַךְ בַּֽחֲמִשָּׁה עָשָׂר יוֹם לַחֹדֶשׁ הַשְּׁבִיעִי בְּאָסְפְּכֶם אֶת תְּבוּאַת הָאָרֶץ תָּחֹגּוּ אֶת חַג יְהוָֹה שִׁבְעַת יָמִים בַּיּוֹם הָֽרִאשׁוֹן שַׁבָּתוֹן וּבַיּוֹם הַשְּׁמִינִי שַׁבָּתֽוֹן:  ת  בְּרַם בְּחַמְשָׁא עֲשַׂר יוֹמָא לְיַרְחָא שְׁבִיעָאָה בְּמִכְנְשֵׁיכוֹן יָת עֲלַלַת אַרְעָא תְּחַגוּן יָת חַגָא קֳדָם יְיָ שִׁבְעָא יוֹמִין בְּיוֹמָא קַדְמָאָה נְיָחָא וּבְיוֹמָא תְמִינָאָה נְיָחָא:

{מ} וּלְקַחְתֶּם לָכֶם בַּיּוֹם הָֽרִאשׁוֹן פְּרִי עֵץ הָדָר כַּפֹּת תְּמָרִים וַֽעֲנַף עֵץ עָבֹת וְעַרְבֵי נָחַל וּשְׂמַחְתֶּם לִפְנֵי יְהוָֹה אֱלֹֽהֵיכֶם שִׁבְעַת יָמִֽים:  מ  וּלְקַחְתֶּם לָכֶם בַּיּוֹם הָֽרִאשׁוֹן פְּרִי עֵץ הָדָר כַּפֹּת תְּמָרִים וַֽעֲנַף עֵץ עָבֹת וְעַרְבֵי נָחַל וּשְׂמַחְתֶּם לִפְנֵי יְהוָֹה אֱלֹֽהֵיכֶם שִׁבְעַת יָמִֽים:  ת  וְתִסְבוּן לְכוֹן בְּיוֹמָא קַדְמָאָה פֵּרֵי אִילָנָא אֶתְרוֹגִין וְלֻלָבִין וַהֲדַסִין וְעַרְבִין דִנְחָל וְתֶחֱדוּן קֳדָם יְיָ אֱלָהֲכוֹן שַׁבְעַת יוֹמִין:

{מא} וְחַגֹּתֶם אֹתוֹ חַג לַֽיהֹוָה שִׁבְעַת יָמִים בַּשָּׁנָה חֻקַּת עוֹלָם לְדֹרֹתֵיכֶם בַּחֹדֶשׁ הַשְּׁבִיעִי תָּחֹגּוּ אֹתֽוֹ:  מ  וְחַגֹּתֶם אֹתוֹ חַג לַֽיהֹוָה שִׁבְעַת יָמִים בַּשָּׁנָה חֻקַּת עוֹלָם לְדֹרֹתֵיכֶם בַּחֹדֶשׁ הַשְּׁבִיעִי תָּחֹגּוּ אֹתֽוֹ:  ת  וּתְחַגוּן יָתֵהּ חַגָא קֳדָם יְיָ שַׁבְעַת יוֹמִין בְּשַׁתָּא קְיַם עָלָם לְדָרֵיכוֹן בְּיַרְחָא שְׁבִיעָאָה תְּחַגוּן יָתֵהּ:

{מב} בַּסֻּכֹּת תֵּֽשְׁבוּ שִׁבְעַת יָמִים כָּל הָֽאֶזְרָח בְּיִשְׂרָאֵל יֵֽשְׁבוּ בַּסֻּכֹּֽת:  מ  בַּסֻּכֹּת תֵּֽשְׁבוּ שִׁבְעַת יָמִים כָּל הָֽאֶזְרָח בְּיִשְׂרָאֵל יֵֽשְׁבוּ בַּסֻּכֹּֽת:  ת  בִּמְטַלַיָא תֵיתְבוּן שַׁבְעַת יוֹמִין כָּל יַצִיבָא בְּיִשְׂרָאֵל יֵתְבוּן בִּמְטַלַיָא:

{מג} לְמַעַן יֵֽדְעוּ דֹרֹֽתֵיכֶם כִּי בַסֻּכּוֹת הוֹשַׁבְתִּי אֶת בְּנֵי יִשְׂרָאֵל בְּהֽוֹצִיאִי אוֹתָם מֵאֶרֶץ מִצְרָיִם אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  מ  לְמַעַן יֵֽדְעוּ דֹרֹֽתֵיכֶם כִּי בַסֻּכּוֹת הוֹשַׁבְתִּי אֶת בְּנֵי יִשְׂרָאֵל בְּהֽוֹצִיאִי אוֹתָם מֵאֶרֶץ מִצְרָיִם אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  ת  בְּדִיל דְיִדְעוּן דָרֵיכוֹן אֲרֵי בִמְטַלָיוּת עֲנָנֵי אוֹתֵבִית יָת בְּנֵי יִשְׂרָאֵל בְּאַפָּקוּתִי יָתְהוֹן מֵאַרְעָא דְמִצְרַיִם אֲנָא יְיָ אֱלָהֲכוֹן:

{מד} וַיְדַבֵּר מֹשֶׁה אֶת מֹֽעֲדֵי יְהוָֹה אֶל בְּנֵי יִשְׂרָאֵֽל:  מ  וַיְדַבֵּר מֹשֶׁה אֶת מֹֽעֲדֵי יְהוָֹה אֶל בְּנֵי יִשְׂרָאֵֽל:  ת  וּמַלִיל מֹשֶׁה יָת סְדַר מוֹעֲדַיָא דַיְיָ וְאַלֵפִנוּן לִבְנֵי יִשְׂרָאֵל: (פ)




ויקרא פרק-כד

{א}  שביעי  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} צַו אֶת בְּנֵי יִשְׂרָאֵל וְיִקְחוּ אֵלֶיךָ שֶׁמֶן זַיִת זָךְ כָּתִית לַמָּאוֹר לְהַֽעֲלֹת נֵר תָּמִֽיד:  מ  צַו אֶת בְּנֵי יִשְׂרָאֵל וְיִקְחוּ אֵלֶיךָ שֶׁמֶן זַיִת זָךְ כָּתִית לַמָּאוֹר לְהַֽעֲלֹת נֵר תָּמִֽיד:  ת  פַּקֵד יָת בְּנֵי יִשְׂרָאֵל וְיִסְבוּן לָךְ מְשַׁח זֵיתָא דַכְיָא כָּתִישָׁא לְאַנְהָרָא לְאַדְלָקָא בוֹצִינַיָא תְּדִירָא:

{ג} מִחוּץ לְפָרֹכֶת הָֽעֵדֻת בְּאֹהֶל מוֹעֵד יַֽעֲרֹךְ אֹתוֹ אַֽהֲרֹן מֵעֶרֶב עַד בֹּקֶר לִפְנֵי יְהוָֹה תָּמִיד חֻקַּת עוֹלָם לְדֹרֹֽתֵיכֶֽם:  מ  מִחוּץ לְפָרֹכֶת הָֽעֵדֻת בְּאֹהֶל מוֹעֵד יַֽעֲרֹךְ אֹתוֹ אַֽהֲרֹן מֵעֶרֶב עַד בֹּקֶר לִפְנֵי יְהוָֹה תָּמִיד חֻקַּת עוֹלָם לְדֹרֹֽתֵיכֶֽם:  ת  מִבָּרָא לְפָרֻכְתָּא דְסַהֲדוּתָא בְּמַשְׁכַּן זִמְנָא יַסְדַר יָתֵהּ אַהֲרֹן מֵרַמְשָׁא עַד צַפְרָא קֳדָם יְיָ תְּדִירָא קְיָם עָלָם לְדָרֵיכוֹן:

{ד} עַל הַמְּנֹרָה הַטְּהֹרָה יַֽעֲרֹךְ אֶת הַנֵּרוֹת לִפְנֵי יְהוָֹה תָּמִֽיד:  מ  עַל הַמְּנֹרָה הַטְּהֹרָה יַֽעֲרֹךְ אֶת הַנֵּרוֹת לִפְנֵי יְהוָֹה תָּמִֽיד:  ת  עַל מְנַרְתָּא דָכִיתָא יַסְדַר יָת בּוֹצִינַיָא קֳדָם יְיָ תְּדִירָא: (פ)

{ה} וְלָֽקַחְתָּ סֹלֶת וְאָֽפִיתָ אֹתָהּ שְׁתֵּים עֶשְׂרֵה חַלּוֹת שְׁנֵי עֶשְׂרֹנִים יִֽהְיֶה הַֽחַלָּה הָֽאֶחָֽת:  מ  וְלָֽקַחְתָּ סֹלֶת וְאָֽפִיתָ אֹתָהּ שְׁתֵּים עֶשְׂרֵה חַלּוֹת שְׁנֵי עֶשְׂרֹנִים יִֽהְיֶה הַֽחַלָּה הָֽאֶחָֽת:  ת  וְתִסַב סֻלְתָּא וְתֵיפֵי יָתַהּ תַּרְתָּא עַשְׂרֵי גְרִיצָן תְּרֵין עֶשְׂרוֹנִין תְּהֵי גְרִצְתָּא חֲדָא:

{ו} וְשַׂמְתָּ אוֹתָם שְׁתַּיִם מַֽעֲרָכוֹת שֵׁשׁ הַֽמַּֽעֲרָכֶת עַל הַשֻּׁלְחָן הַטָּהֹר לִפְנֵי יְהוָֹֽה:  מ  וְשַׂמְתָּ אוֹתָם שְׁתַּיִם מַֽעֲרָכוֹת שֵׁשׁ הַֽמַּֽעֲרָכֶת עַל הַשֻּׁלְחָן הַטָּהֹר לִפְנֵי יְהוָֹֽה:  ת  וּתְשַׁוִי יָתְהוֹן תַּרְתֵּי סִדְרִין שִׁית סִדְרָא עַל פָּתוֹרָא דַכְיָא קֳדָם יְיָ:

{ז} וְנָֽתַתָּ עַל הַֽמַּֽעֲרֶכֶת לְבֹנָה זַכָּה וְהָֽיְתָה לַלֶּחֶם לְאַזְכָּרָה אִשֶּׁה לַֽיהוָֹֽה:  מ  וְנָֽתַתָּ עַל הַֽמַּֽעֲרֶכֶת לְבֹנָה זַכָּה וְהָֽיְתָה לַלֶּחֶם לְאַזְכָּרָה אִשֶּׁה לַֽיהוָֹֽה:  ת  וְתִתֵּן עַל סִדְרָא לְבֹנְתָּא דָכִיתָא וּתְהֵי לִלְחֵם לְאִדְכַּרְתָּא קֻרְבָּנָא קָדָם יְיָ:

{ח} בְּיוֹם הַשַּׁבָּת בְּיוֹם הַשַּׁבָּת יַֽעַרְכֶנּוּ לִפְנֵי יְהוָֹה תָּמִיד מֵאֵת בְּנֵֽי יִשְׂרָאֵל בְּרִית עוֹלָֽם:  מ  בְּיוֹם הַשַּׁבָּת בְּיוֹם הַשַּׁבָּת יַֽעַרְכֶנּוּ לִפְנֵי יְהוָֹה תָּמִיד מֵאֵת בְּנֵֽי יִשְׂרָאֵל בְּרִית עוֹלָֽם:  ת  בְּיוֹמָא דְשַׁבְּתָא בְּיוֹמָא דְשַׁבְּתָא יַסְדְרִנֵהּ קֳדָם יְיָ תְּדִירָא מִן קֳדָם בְּנֵי יִשְׂרָאֵל קְיַם עָלָם:

{ט} וְהָֽיְתָה לְאַֽהֲרֹן וּלְבָנָיו וַֽאֲכָלֻהוּ בְּמָקוֹם קָדֹשׁ כִּי קֹדֶשׁ קָֽדָשִׁים הוּא לוֹ מֵֽאִשֵּׁי יְהוָֹה חָק עוֹלָֽם:  מ  וְהָֽיְתָה לְאַֽהֲרֹן וּלְבָנָיו וַֽאֲכָלֻהוּ בְּמָקוֹם קָדֹשׁ כִּי קֹדֶשׁ קָֽדָשִׁים הוּא לוֹ מֵֽאִשֵּׁי יְהוָֹה חָק עוֹלָֽם:  ת  וּתְהֵי לְאַהֲרֹן וְלִבְנוֹהִי וְיֵכְלֻנֵהּ בַּאֲתַר קַדִישׁ אֲרֵי קֹדֶשׁ קוּדְשִׁין הוּא לֵהּ מִקֻרְבָּנַיָא דַיְיָ קְיַם עָלָם: (ס)

{י} וַיֵּצֵא בֶּן אִשָּׁה יִשְׂרְאֵלִית וְהוּא בֶּן אִישׁ מִצְרִי בְּתוֹךְ בְּנֵי יִשְׂרָאֵל וַיִּנָּצוּ בַּֽמַּֽחֲנֶה בֶּן הַיִּשְׂרְאֵלִית וְאִישׁ הַיִּשְׂרְאֵלִֽי:  מ  וַיֵּצֵא בֶּן אִשָּׁה יִשְׂרְאֵלִית וְהוּא בֶּן אִישׁ מִצְרִי בְּתוֹךְ בְּנֵי יִשְׂרָאֵל וַיִּנָּצוּ בַּֽמַּֽחֲנֶה בֶּן הַיִּשְׂרְאֵלִית וְאִישׁ הַיִּשְׂרְאֵלִֽי:  ת  וּנְפַק בַּר אִתְּתָא בַת יִשְׂרָאֵל וְהוּא בַּר גְבַר מִצְרָאָה בְּגוֹ בְּנֵי יִשְׂרָאֵל וּנְצוֹ בְּמַשְׁרִיתָא בַּר אִתְּתָא בַת יִשְׂרָאֵל וְגַבְרָא בַּר יִשְׂרָאֵל:

{יא} וַיִּקֹּב בֶּן הָֽאִשָּׁה הַיִּשְׂרְאֵלִית אֶת הַשֵּׁם וַיְקַלֵּל וַיָּבִיאוּ אֹתוֹ אֶל מֹשֶׁה וְשֵׁם אִמּוֹ שְׁלֹמִית בַּת דִּבְרִי לְמַטֵּה דָֽן:  מ  וַיִּקֹּב בֶּן הָֽאִשָּׁה הַיִּשְׂרְאֵלִית אֶת הַשֵּׁם וַיְקַלֵּל וַיָּבִיאוּ אֹתוֹ אֶל מֹשֶׁה וְשֵׁם אִמּוֹ שְׁלֹמִית בַּת דִּבְרִי לְמַטֵּה דָֽן:  ת  וּפָרֵשׁ בַּר אִתְּתָא בַת יִשְׂרָאֵל יָת שְׁמָא וְאַרְגֵז וְאַיְתִיאוּ יָתֵהּ לְוָת מֹשֶׁה וְשׁוּם אִמֵהּ שְׁלֹמִית בַּת דִבְרִי לְשִׁבְטָא דְדָן:

{יב} וַיַּנִּיחֻהוּ בַּמִּשְׁמָר לִפְרֹשׁ לָהֶם עַל פִּי יְהוָֹֽה:  מ  וַיַּנִּיחֻהוּ בַּמִּשְׁמָר לִפְרֹשׁ לָהֶם עַל פִּי יְהוָֹֽה:  ת  וְאַסְרוֹהִי בְּבֵית מַטְרָא עַד דְיִתְפָּרַשׁ לְהוֹן עַל גְזֵרַת מֵימְרָא דַיְיָ: (פ)

{יג} וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{יד} הוֹצֵא אֶת הַֽמְקַלֵּל אֶל מִחוּץ לַֽמַּֽחֲנֶה וְסָמְכוּ כָל הַשֹּֽׁמְעִים אֶת יְדֵיהֶם עַל רֹאשׁוֹ וְרָֽגְמוּ אֹתוֹ כָּל הָֽעֵדָֽה:  מ  הוֹצֵא אֶת הַֽמְקַלֵּל אֶל מִחוּץ לַֽמַּֽחֲנֶה וְסָמְכוּ כָל הַשֹּֽׁמְעִים אֶת יְדֵיהֶם עַל רֹאשׁוֹ וְרָֽגְמוּ אֹתוֹ כָּל הָֽעֵדָֽה:  ת  אַפֵּק יָת דְאַרְגֵז לְמִבָּרָא לְמַשְּׁרִיתָא וְיִסְמְכוּן כָּל דִי שְׁמָעוּ יָת יְדֵיהוֹן עַל רֵישֵׁהּ וְיִרְגְמוּן יָתֵהּ כָּל כְּנִשְּׁתָּא:

{טו} וְאֶל בְּנֵי יִשְׂרָאֵל תְּדַבֵּר לֵאמֹר אִישׁ אִישׁ כִּֽי יְקַלֵּל אֱלֹהָיו וְנָשָׂא חֶטְאֽוֹ:  מ  וְאֶל בְּנֵי יִשְׂרָאֵל תְּדַבֵּר לֵאמֹר אִישׁ אִישׁ כִּֽי יְקַלֵּל אֱלֹהָיו וְנָשָׂא חֶטְאֽוֹ:  ת  וְעִם בְּנֵי יִשְׂרָאֵל תְּמַלֵל לְמֵימָר גְבַר גְבַר דִי יַרְגֵז קֳדָם אֱלָהֵהּ וִיקַבֵּל חוֹבֵהּ:

{טז} וְנֹקֵב שֵׁם יְהוָֹה מוֹת יוּמָת רָגוֹם יִרְגְּמוּ בוֹ כָּל הָֽעֵדָה כַּגֵּר כָּֽאֶזְרָח בְּנָקְבוֹ שֵׁם יוּמָֽת:  מ  וְנֹקֵב שֵׁם יְהוָֹה מוֹת יוּמָת רָגוֹם יִרְגְּמוּ בוֹ כָּל הָֽעֵדָה כַּגֵּר כָּֽאֶזְרָח בְּנָקְבוֹ שֵׁם יוּמָֽת:  ת  וְדִי פָרֵשׁ שְׁמָא דַיְיָ אִתְקְטָּלָא יִתְקְטֵל מִרְגַם יִרְגְמוּן בֵּהּ כָּל כְּנִשְׁתָּא כְּגִיוֹרָא כְּיַצִיבָא בְּפָרָשׁוּתֵהּ שְׁמָא יִתְקְטֵל:

{יז} וְאִישׁ כִּי יַכֶּה כָּל נֶפֶשׁ אָדָם מוֹת יוּמָֽת:  מ  וְאִישׁ כִּי יַכֶּה כָּל נֶפֶשׁ אָדָם מוֹת יוּמָֽת:  ת  וּגְבַר אֲרֵי יִקְטוֹל כָּל נַפְשָׁא דַאֲנָשָׁא אִתְקְטָלָא יִתְקְטֵל:

{יח} וּמַכֵּה נֶֽפֶשׁ בְּהֵמָה יְשַׁלְּמֶנָּה נֶפֶשׁ תַּחַת נָֽפֶשׁ:  מ  וּמַכֵּה נֶֽפֶשׁ בְּהֵמָה יְשַׁלְּמֶנָּה נֶפֶשׁ תַּחַת נָֽפֶשׁ:  ת  וּדְיִקְטוֹל נְפַשׁ בְּעִירָא יְשַׁלְמִנַהּ נַפְשָׁא חֳלַף נַפְשָׁא:

{יט} וְאִישׁ כִּֽי יִתֵּן מוּם בַּֽעֲמִיתוֹ כַּֽאֲשֶׁר עָשָׂה כֵּן יֵעָשֶׂה לּֽוֹ:  מ  וְאִישׁ כִּֽי יִתֵּן מוּם בַּֽעֲמִיתוֹ כַּֽאֲשֶׁר עָשָׂה כֵּן יֵעָשֶׂה לּֽוֹ:  ת  וּגְבַר אֲרֵי יִתֵּן מוּמָא בְּחַבְרֵהּ כְּמָא דִי עֲבַד כֵּן יִתְעֲבֵד לֵיהּ:

{כ} שֶׁבֶר תַּחַת שֶׁבֶר עַיִן תַּחַת עַיִן שֵׁן תַּחַת שֵׁן כַּֽאֲשֶׁר יִתֵּן מוּם בָּֽאָדָם כֵּן יִנָּתֶן בּֽוֹ:  מ  שֶׁבֶר תַּחַת שֶׁבֶר עַיִן תַּחַת עַיִן שֵׁן תַּחַת שֵׁן כַּֽאֲשֶׁר יִתֵּן מוּם בָּֽאָדָם כֵּן יִנָּתֶן בּֽוֹ:  ת  תַּבְרָא חֳלַף תַּבְרָא עֵינָא חֳלַף עֵינָא שִׁנָא חֳלַף שִׁנָא כְּמָא דִי יִתֵּן מוּמָא בַּאֲנָשָׁא כֵּן יִתְיְהֵב בֵּהּ:

{כא}  מפטיר  וּמַכֵּה בְהֵמָה יְשַׁלְּמֶנָּה וּמַכֵּה אָדָם יוּמָֽת:  מ  וּמַכֵּה בְהֵמָה יְשַׁלְּמֶנָּה וּמַכֵּה אָדָם יוּמָֽת:  ת  וּדְיִקְטוֹל בְּעִירָא יְשַׁלְמִנַהּ וּדְיִקְטוֹל אֲנָשָׁא יִתְקְטֵל:

{כב} מִשְׁפַּט אֶחָד יִהְיֶה לָכֶם כַּגֵּר כָּֽאֶזְרָח יִֽהְיֶה כִּי אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  מ  מִשְׁפַּט אֶחָד יִהְיֶה לָכֶם כַּגֵּר כָּֽאֶזְרָח יִֽהְיֶה כִּי אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  ת  דִינָא חֲדָא יְהֵי לְכוֹן כְּגִיוֹרָא כְּיַצִיבָא יְהֵי אֲרֵי אֲנָא יְיָ אֱלָהֲכוֹן:

{כג} וַיְדַבֵּר מֹשֶׁה אֶל בְּנֵי יִשְׂרָאֵל וַיּוֹצִיאוּ אֶת הַֽמְקַלֵּל אֶל מִחוּץ לַֽמַּֽחֲנֶה וַיִּרְגְּמוּ אֹתוֹ אָבֶן וּבְנֵֽי יִשְׂרָאֵל עָשׂוּ כַּֽאֲשֶׁר צִוָּה יְהוָֹה אֶת מֹשֶֽׁה:  מ  וַיְדַבֵּר מֹשֶׁה אֶל בְּנֵי יִשְׂרָאֵל וַיּוֹצִיאוּ אֶת הַֽמְקַלֵּל אֶל מִחוּץ לַֽמַּֽחֲנֶה וַיִּרְגְּמוּ אֹתוֹ אָבֶן וּבְנֵֽי יִשְׂרָאֵל עָשׂוּ כַּֽאֲשֶׁר צִוָּה יְהוָֹה אֶת מֹשֶֽׁה:  ת  וּמַלִיל מֹשֶׁה עִם בְּנֵי יִשְׂרָאֵל וְאַפִּיקוּ יָת דְאַרְגֵז לְמִבָּרָא לְמַשְׁרִיתָא וּרְגָמוּ יָתֵהּ אַבְנָא וּבְנֵי יִשְׂרָאֵל עֲבָדוּ כְּמָא דִי פַּקֵיד יְיָ יָת מֹשֶׁה: (פפפ)
