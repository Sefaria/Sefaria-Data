דברים פרק-א

{א} אֵ֣לֶּה הַדְּבָרִ֗ים אֲשֶׁ֨ר דִּבֶּ֤ר מֹשֶׁה֙ אֶל-כָּל-יִשְׂרָאֵ֔ל בְּעֵ֖בֶר הַיַּרְדֵּ֑ן בַּמִּדְבָּ֡ר בָּֽעֲרָבָה֩ מ֨וֹל ס֜וּף בֵּין-פָּארָ֧ן וּבֵין-תֹּ֛פֶל וְלָבָ֑ן וַֽחֲצֵרֹ֖ת וְדִ֥י זָהָֽב:  מ  אֵ֣לֶּה הַדְּבָרִ֗ים אֲשֶׁ֨ר דִּבֶּ֤ר מֹשֶׁה֙ אֶל-כָּל-יִשְׂרָאֵ֔ל בְּעֵ֖בֶר הַיַּרְדֵּ֑ן בַּמִּדְבָּ֡ר בָּֽעֲרָבָה֩ מ֨וֹל ס֜וּף בֵּין-פָּארָ֧ן וּבֵין-תֹּ֛פֶל וְלָבָ֑ן וַֽחֲצֵרֹ֖ת וְדִ֥י זָהָֽב:  ת  אִלֵין פִּתְגָמַיָא דִי מַלִיל מֹשֶׁה עִם כָּל יִשְׂרָאֵל בְּעִבְרָא דְיַרְדְנָא אוֹכַח יָתְהוֹן עַל דְחָבוּ בְמַדְבְּרָא וְעַל דְאַרְגִיזוּ בְמֵישְׁרָא לָקֳבֵל יַם סוּף בְּפָארָן אִתַּפָּלוּ עַל מַנָא וּבַחֲצֵרוֹת אַרְגִיזוּ עַל בִּשְׂרָא וְעַל דַעֲבָדוּ עֵגַל דִדְהָב:

{ב} אַחַ֨ד עָשָׂ֥ר יוֹם֙ מֵֽחֹרֵ֔ב דֶּ֖רֶךְ הַר-שֵׂעִ֑יר עַ֖ד קָדֵ֥שׁ בַּרְנֵֽעַ:  מ  אַחַ֨ד עָשָׂ֥ר יוֹם֙ מֵֽחֹרֵ֔ב דֶּ֖רֶךְ הַר-שֵׂעִ֑יר עַ֖ד קָדֵ֥שׁ בַּרְנֵֽעַ:  ת  מַהֲלַךְ חָד עֲשַׂר יוֹמִין מֵחוֹרֵב אוֹרַח טּוּרָא דְשֵׂעִיר עַד רְקַם גֵיאָה:

{ג} וַֽיְהִי֙ בְּאַרְבָּעִ֣ים שָׁנָ֔ה בְּעַשְׁתֵּֽי-עָשָׂ֥ר חֹ֖דֶשׁ בְּאֶחָ֣ד לַחֹ֑דֶשׁ דִּבֶּ֤ר מֹשֶׁה֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל כְּ֠כֹל אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֹת֖וֹ אֲלֵהֶֽם:  מ  וַֽיְהִי֙ בְּאַרְבָּעִ֣ים שָׁנָ֔ה בְּעַשְׁתֵּֽי-עָשָׂ֥ר חֹ֖דֶשׁ בְּאֶחָ֣ד לַחֹ֑דֶשׁ דִּבֶּ֤ר מֹשֶׁה֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל כְּ֠כֹל אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֹת֖וֹ אֲלֵהֶֽם:  ת  וַהֲוָה בְּאַרְבְּעִין שְׁנִין בְּחַד עֲשַׂר יַרְחָא בְּחַד לְיַרְחָא מַלִיל מֹשֶׁה עִם בְּנֵי יִשְׂרָאֵל כְּכֹל דִי פַקִיד יְיָ יָתֵהּ לְוָתְהוֹן:

{ד} אַֽחֲרֵ֣י הַכֹּת֗וֹ אֵ֚ת סִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּ֑וֹן וְאֵ֗ת ע֚וֹג מֶ֣לֶךְ הַבָּשָׁ֔ן אֲשֶׁר-יוֹשֵׁ֥ב בְּעַשְׁתָּרֹ֖ת בְּאֶדְרֶֽעִי:  מ  אַֽחֲרֵ֣י הַכֹּת֗וֹ אֵ֚ת סִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּ֑וֹן וְאֵ֗ת ע֚וֹג מֶ֣לֶךְ הַבָּשָׁ֔ן אֲשֶׁר-יוֹשֵׁ֥ב בְּעַשְׁתָּרֹ֖ת בְּאֶדְרֶֽעִי:  ת  בָּתַר דִמְחָא יָת סִיחוֹן מַלְכָּא דֶאֱמוֹרָאָה דְיָתֵב בְּחֶשְׁבּוֹן וְיָת עוֹג מַלְכָּא דְמַתְנָן דְיָתֵב בְּעַשְׁתָּרוֹת בְּאֶדְרֶעִי:

{ה} בְּעֵ֥בֶר הַיַּרְדֵּ֖ן בְּאֶ֣רֶץ מוֹאָ֑ב הוֹאִ֣יל מֹשֶׁ֔ה בֵּאֵ֛ר אֶת-הַתּוֹרָ֥ה הַזֹּ֖את לֵאמֹֽר:  מ  בְּעֵ֥בֶר הַיַּרְדֵּ֖ן בְּאֶ֣רֶץ מוֹאָ֑ב הוֹאִ֣יל מֹשֶׁ֔ה בֵּאֵ֛ר אֶת-הַתּוֹרָ֥ה הַזֹּ֖את לֵאמֹֽר:  ת  בְּעִבְרָא דְיַרְדְנָא בְּאַרְעָא דְמוֹאָב שָׁרֵי מֹשֶׁה פָּרֵשׁ יָת אוּלְפַן אוֹרַיְתָא הָדָא לְמֵימָר:

{ו} יְהֹוָ֧ה אֱלֹהֵ֛ינוּ דִּבֶּ֥ר אֵלֵ֖ינוּ בְּחֹרֵ֣ב לֵאמֹ֑ר רַב-לָכֶ֥ם שֶׁ֖בֶת בָּהָ֥ר הַזֶּֽה:  מ  יְהֹוָ֧ה אֱלֹהֵ֛ינוּ דִּבֶּ֥ר אֵלֵ֖ינוּ בְּחֹרֵ֣ב לֵאמֹ֑ר רַב-לָכֶ֥ם שֶׁ֖בֶת בָּהָ֥ר הַזֶּֽה:  ת  יְיָ אֱלָהָנָא מַלִיל עִמָנָא בְּחוֹרֵב לְמֵימָר סַגִי לְכוֹן דִיתֶבְתּוּן בְּטוּרָא הָדֵין:

{ז} פְּנ֣וּ | וּסְע֣וּ לָכֶ֗ם וּבֹ֨אוּ הַ֥ר הָֽאֱמֹרִי֘ וְאֶל-כָּל-שְׁכֵנָיו֒ בָּֽעֲרָבָ֥ה בָהָ֛ר וּבַשְּׁפֵלָ֥ה וּבַנֶּ֖גֶב וּבְח֣וֹף הַיָּ֑ם אֶ֤רֶץ הַֽכְּנַֽעֲנִי֙ וְהַלְּבָנ֔וֹן עַד-הַנָּהָ֥ר הַגָּדֹ֖ל נְהַר-פְּרָֽת:  מ  פְּנ֣וּ | וּסְע֣וּ לָכֶ֗ם וּבֹ֨אוּ הַ֥ר הָֽאֱמֹרִי֘ וְאֶל-כָּל-שְׁכֵנָיו֒ בָּֽעֲרָבָ֥ה בָהָ֛ר וּבַשְּׁפֵלָ֥ה וּבַנֶּ֖גֶב וּבְח֣וֹף הַיָּ֑ם אֶ֤רֶץ הַֽכְּנַֽעֲנִי֙ וְהַלְּבָנ֔וֹן עַד-הַנָּהָ֥ר הַגָּדֹ֖ל נְהַר-פְּרָֽת:  ת  אִתְפְּנוּ וְטוּלוּ לְכוֹן וְעוּלוּ לְטּוּרָא דֶאֱמוֹרָאָה וּלְכָל מָגִירוֹהִי בְּמֵישְׁרַיָא בְטוּרָא וּבִשְׁפֶלְתָּא וּבִדְרוֹמָא וּבִסְפַר יַמָא אַרְעָא דִכְנַעֲנָאָה וְלִבְנָן עַד נַהֲרָא רַבָּא נַהֲרָא פְרָת:

{ח} רְאֵ֛ה נָתַ֥תִּי לִפְנֵיכֶ֖ם אֶת-הָאָ֑רֶץ בֹּ֚אוּ וּרְשׁ֣וּ אֶת-הָאָ֔רֶץ אֲשֶׁ֣ר נִשְׁבַּ֣ע יְהֹוָ֠ה לַֽאֲבֹ֨תֵיכֶ֜ם לְאַבְרָהָ֨ם לְיִצְחָ֤ק וּֽלְיַֽעֲקֹב֙ לָתֵ֣ת לָהֶ֔ם וּלְזַרְעָ֖ם אַֽחֲרֵיהֶֽם:  מ  רְאֵ֛ה נָתַ֥תִּי לִפְנֵיכֶ֖ם אֶת-הָאָ֑רֶץ בֹּ֚אוּ וּרְשׁ֣וּ אֶת-הָאָ֔רֶץ אֲשֶׁ֣ר נִשְׁבַּ֣ע יְהֹוָ֠ה לַֽאֲבֹ֨תֵיכֶ֜ם לְאַבְרָהָ֨ם לְיִצְחָ֤ק וּֽלְיַֽעֲקֹב֙ לָתֵ֣ת לָהֶ֔ם וּלְזַרְעָ֖ם אַֽחֲרֵיהֶֽם:  ת  חֲזֵי דִיהָבִית קֳדָמֵיכוֹן יָת אַרְעָא עוּלוּ וְאַחֲסִינוּ יָת אַרְעָא דִי קַיַם יְיָ לַאֲבָהָתְכוֹן לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לְמִתַּן לְהוֹן וְלִבְנֵיהוֹן בַּתְרֵיהוֹן:

{ט} וָֽאֹמַ֣ר אֲלֵכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר לֹֽא-אוּכַ֥ל לְבַדִּ֖י שְׂאֵ֥ת אֶתְכֶֽם:  מ  וָֽאֹמַ֣ר אֲלֵכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר לֹֽא-אוּכַ֥ל לְבַדִּ֖י שְׂאֵ֥ת אֶתְכֶֽם:  ת  וַאֲמָרִית לְכוֹן בְּעִדָנָא הַהִיא לְמֵימָר לֵית אֲנָא יָכִיל בִּלְחוֹדִי לְסוֹבָרָא יָתְכוֹן:

{י} יְהֹוָ֥ה אֱלֹֽהֵיכֶ֖ם הִרְבָּ֣ה אֶתְכֶ֑ם וְהִנְּכֶ֣ם הַיּ֔וֹם כְּכֽוֹכְבֵ֥י הַשָּׁמַ֖יִם לָרֹֽב:  מ  יְהֹוָ֥ה אֱלֹֽהֵיכֶ֖ם הִרְבָּ֣ה אֶתְכֶ֑ם וְהִנְּכֶ֣ם הַיּ֔וֹם כְּכֽוֹכְבֵ֥י הַשָּׁמַ֖יִם לָרֹֽב:  ת  יְיָ אֱלָהָכוֹן אַסְגֵי יָתְכוֹן וְהָא אִיתֵיכוֹן יוֹמָא דֵין כְּכוֹכְבֵי שְׁמַיָא לִסְגֵי:

{יא} יְהֹוָ֞ה אֱלֹהֵ֣י אֲבֽוֹתֵכֶ֗ם יֹסֵ֧ף עֲלֵיכֶ֛ם כָּכֶ֖ם אֶ֣לֶף פְּעָמִ֑ים וִֽיבָרֵ֣ךְ אֶתְכֶ֔ם כַּֽאֲשֶׁ֖ר דִּבֶּ֥ר לָכֶֽם:  מ  יְהֹוָ֞ה אֱלֹהֵ֣י אֲבֽוֹתֵכֶ֗ם יֹסֵ֧ף עֲלֵיכֶ֛ם כָּכֶ֖ם אֶ֣לֶף פְּעָמִ֑ים וִֽיבָרֵ֣ךְ אֶתְכֶ֔ם כַּֽאֲשֶׁ֖ר דִּבֶּ֥ר לָכֶֽם:  ת  יְיָ אֱלָהָא דַאֲבָהָתְכוֹן יוֹסֵף עֲלֵיכוֹן כְּוָתְכוֹן אֲלַף זִמְנִין וִיבָרֵךְ יָתְכוֹן כְּמָא דִי מַלִיל לְכוֹן:

{יב}  שני  אֵיכָ֥ה אֶשָּׂ֖א לְבַדִּ֑י טָרְחֲכֶ֥ם וּמַֽשַּֽׂאֲכֶ֖ם וְרִֽיבְכֶֽם:  מ  אֵיכָ֥ה אֶשָּׂ֖א לְבַדִּ֑י טָרְחֲכֶ֥ם וּמַֽשַּֽׂאֲכֶ֖ם וְרִֽיבְכֶֽם:  ת  אֶכְדֵין אֵסוֹבַר בִּלְחוֹדִי טָרְחֲכוֹן וְעִסְקֵיכוֹן וְדִינְכוֹן:

{יג} הָב֣וּ לָ֠כֶם אֲנָשִׁ֨ים חֲכָמִ֧ים וּנְבֹנִ֛ים וִֽידֻעִ֖ים לְשִׁבְטֵיכֶ֑ם וַֽאֲשִׂימֵ֖ם בְּרָֽאשֵׁיכֶֽם:  מ  הָב֣וּ לָ֠כֶם אֲנָשִׁ֨ים חֲכָמִ֧ים וּנְבֹנִ֛ים וִֽידֻעִ֖ים לְשִׁבְטֵיכֶ֑ם וַֽאֲשִׂימֵ֖ם בְּרָֽאשֵׁיכֶֽם:  ת  הָבוּ לְכוֹן גֻבְרִין חַכִּימִין וְסוּכְלְתָנוּן וּמַדְעָן לְשִׁבְטֵיכוֹן וֶאֱמַנִנוּן רֵישִׁין עֲלֵיכוֹן:

{יד} וַֽתַּֽעֲנ֖וּ אֹתִ֑י וַתֹּ֣אמְר֔וּ טוֹב-הַדָּבָ֥ר אֲשֶׁר-דִּבַּ֖רְתָּ לַֽעֲשֽׂוֹת:  מ  וַֽתַּֽעֲנ֖וּ אֹתִ֑י וַתֹּ֣אמְר֔וּ טוֹב-הַדָּבָ֥ר אֲשֶׁר-דִּבַּ֖רְתָּ לַֽעֲשֽׂוֹת:  ת  וַאֲתֶבְתּוּן יָתִי וַאֲמַרְתּוּן תַּקִין פִּתְגָמָא דִי מַלֶלְתָּא לְמֶעְבָּד:

{טו} וָֽאֶקַּ֞ח אֶת-רָאשֵׁ֣י שִׁבְטֵיכֶ֗ם אֲנָשִׁ֤ים חֲכָמִים֙ וִֽידֻעִ֔ים וָֽאֶתֵּ֥ן אוֹתָ֛ם רָאשִׁ֖ים עֲלֵיכֶ֑ם שָׂרֵ֨י אֲלָפִ֜ים וְשָׂרֵ֣י מֵא֗וֹת וְשָׂרֵ֤י חֲמִשִּׁים֙ וְשָׂרֵ֣י עֲשָׂרֹ֔ת וְשֹֽׁטְרִ֖ים לְשִׁבְטֵיכֶֽם:  מ  וָֽאֶקַּ֞ח אֶת-רָאשֵׁ֣י שִׁבְטֵיכֶ֗ם אֲנָשִׁ֤ים חֲכָמִים֙ וִֽידֻעִ֔ים וָֽאֶתֵּ֥ן אוֹתָ֛ם רָאשִׁ֖ים עֲלֵיכֶ֑ם שָׂרֵ֨י אֲלָפִ֜ים וְשָׂרֵ֣י מֵא֗וֹת וְשָׂרֵ֤י חֲמִשִּׁים֙ וְשָׂרֵ֣י עֲשָׂרֹ֔ת וְשֹֽׁטְרִ֖ים לְשִׁבְטֵיכֶֽם:  ת  וּדְבָרִית יָת רֵישֵׁי שִׁבְטֵיכוֹן גֻבְרִין חַכִּימִין וּמַדְעָן וּמַנֵתִי יָתְהוֹן רֵישִׁין עֲלֵיכוֹן רַבָּנֵי אַלְפִין וְרַבָּנֵי מָאוָתָא וְרַבָּנֵי חַמְשִׁין וְרַבָּנֵי עִשׂוֹרְיָתָא וְסָרְכִין לְשִׁבְטֵיכוֹן:

{טז} וָֽאֲצַוֶּה֙ אֶת-שֹׁ֣פְטֵיכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר שָׁמֹ֤עַ בֵּין-אֲחֵיכֶם֙ וּשְׁפַטְתֶּ֣ם צֶ֔דֶק בֵּין-אִ֥ישׁ וּבֵין-אָחִ֖יו וּבֵ֥ין גֵּרֽוֹ:  מ  וָֽאֲצַוֶּה֙ אֶת-שֹׁ֣פְטֵיכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר שָׁמֹ֤עַ בֵּין-אֲחֵיכֶם֙ וּשְׁפַטְתֶּ֣ם צֶ֔דֶק בֵּין-אִ֥ישׁ וּבֵין-אָחִ֖יו וּבֵ֥ין גֵּרֽוֹ:  ת  וּפַקֵדִית יָת דַיָנֵיכוֹן בְּעִדָנָא הַהִיא לְמֵימָר שְׁמָעוּ בֵּין אַחֵיכוֹן וּתְדוּנוּן קוּשְׁטָא בֵּין גַבְרָא וּבֵין אֲחוּהִי וּבֵין גִיוֹרֵהּ:

{יז} לֹֽא-תַכִּ֨ירוּ פָנִ֜ים בַּמִּשְׁפָּ֗ט כַּקָּטֹ֤ן כַּגָּדֹל֙ תִּשְׁמָע֔וּן לֹ֤א תָג֨וּרוּ֙ מִפְּנֵי-אִ֔ישׁ כִּ֥י הַמִּשְׁפָּ֖ט לֵֽאלֹהִ֣ים ה֑וּא וְהַדָּבָר֙ אֲשֶׁ֣ר יִקְשֶׁ֣ה מִכֶּ֔ם תַּקְרִב֥וּן אֵלַ֖י וּשְׁמַעְתִּֽיו:  מ  לֹֽא-תַכִּ֨ירוּ פָנִ֜ים בַּמִּשְׁפָּ֗ט כַּקָּטֹ֤ן כַּגָּדֹל֙ תִּשְׁמָע֔וּן לֹ֤א תָג֨וּרוּ֙ מִפְּנֵי-אִ֔ישׁ כִּ֥י הַמִּשְׁפָּ֖ט לֵֽאלֹהִ֣ים ה֑וּא וְהַדָּבָר֙ אֲשֶׁ֣ר יִקְשֶׁ֣ה מִכֶּ֔ם תַּקְרִב֥וּן אֵלַ֖י וּשְׁמַעְתִּֽיו:  ת  לָא תִשְׁתְּמוֹדְעוּן אַפִּין בְּדִינָא מִלֵי זְעֵירָא כְרַבָּא תִּשְׁמְעוּן לָא תִדְחֲלוּן מִן קֳדָם גַבְרָא אֲרֵי דִינָא דַיְיָ הוּא וּפִתְגָמָא דִי יִקְשֵׁי מִנְכוֹן תְּקָרְבוּן לְוָתִי וְאֶשְׁמְעִנֵהּ:

{יח} וָֽאֲצַוֶּ֥ה אֶתְכֶ֖ם בָּעֵ֣ת הַהִ֑וא אֵ֥ת כָּל-הַדְּבָרִ֖ים אֲשֶׁ֥ר תַּֽעֲשֽׂוּן:  מ  וָֽאֲצַוֶּ֥ה אֶתְכֶ֖ם בָּעֵ֣ת הַהִ֑וא אֵ֥ת כָּל-הַדְּבָרִ֖ים אֲשֶׁ֥ר תַּֽעֲשֽׂוּן:  ת  וּפַקֵדִית יָתְכוֹן בְּעִדָנָא הַהִיא יָת כָּל פִּתְגָמַיָא דִי תַעְבְּדוּן:

{יט} וַנִּסַּ֣ע מֵֽחֹרֵ֗ב וַנֵּ֡לֶךְ אֵ֣ת כָּל-הַמִּדְבָּ֣ר הַגָּדוֹל֩ וְהַנּוֹרָ֨א הַה֜וּא אֲשֶׁ֣ר רְאִיתֶ֗ם דֶּ֚רֶךְ הַ֣ר הָֽאֱמֹרִ֔י כַּֽאֲשֶׁ֥ר צִוָּ֛ה יְהֹוָ֥ה אֱלֹהֵ֖ינוּ אֹתָ֑נוּ וַנָּבֹ֕א עַ֖ד קָדֵ֥שׁ בַּרְנֵֽעַ:  מ  וַנִּסַּ֣ע מֵֽחֹרֵ֗ב וַנֵּ֡לֶךְ אֵ֣ת כָּל-הַמִּדְבָּ֣ר הַגָּדוֹל֩ וְהַנּוֹרָ֨א הַה֜וּא אֲשֶׁ֣ר רְאִיתֶ֗ם דֶּ֚רֶךְ הַ֣ר הָֽאֱמֹרִ֔י כַּֽאֲשֶׁ֥ר צִוָּ֛ה יְהֹוָ֥ה אֱלֹהֵ֖ינוּ אֹתָ֑נוּ וַנָּבֹ֕א עַ֖ד קָדֵ֥שׁ בַּרְנֵֽעַ:  ת  וּנְטַלְנָא מֵחוֹרֵב וְהַלִיכְנָא יָת כָּל מַדְבְּרָא רַבָּא וּדְחִילָא הַהוּא דִי חֲזֵיתוּן אוֹרַח טוּרָא דֶאֱמוֹרָאָה כְּמָא דִי פַקִיד יְיָ אֱלָהָנָא יָתָנָא וַאֲתֵינָא עַד רְקַם גֵיאָה:

{כ} וָֽאֹמַ֖ר אֲלֵכֶ֑ם בָּאתֶם֙ עַד-הַ֣ר הָֽאֱמֹרִ֔י אֲשֶׁר-יְהֹוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ:  מ  וָֽאֹמַ֖ר אֲלֵכֶ֑ם בָּאתֶם֙ עַד-הַ֣ר הָֽאֱמֹרִ֔י אֲשֶׁר-יְהֹוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ:  ת  וַאֲמָרִית לְכוֹן אֲתֵיתוּן עַד טוּרָא דֶאֱמוֹרָאָה דַיְיָ אֱלָהָנָא יָהֵיב לָנָא:

{כא} רְ֠אֵה נָתַ֨ן יְהֹוָ֧ה אֱלֹהֶ֛יךָ לְפָנֶ֖יךָ אֶת-הָאָ֑רֶץ עֲלֵ֣ה רֵ֗שׁ כַּֽאֲשֶׁר֩ דִּבֶּ֨ר יְהֹוָ֜ה אֱלֹהֵ֤י אֲבֹתֶ֨יךָ֙ לָ֔ךְ אַל-תִּירָ֖א וְאַל-תֵּחָֽת:  מ  רְ֠אֵה נָתַ֨ן יְהֹוָ֧ה אֱלֹהֶ֛יךָ לְפָנֶ֖יךָ אֶת-הָאָ֑רֶץ עֲלֵ֣ה רֵ֗שׁ כַּֽאֲשֶׁר֩ דִּבֶּ֨ר יְהֹוָ֜ה אֱלֹהֵ֤י אֲבֹתֶ֨יךָ֙ לָ֔ךְ אַל-תִּירָ֖א וְאַל-תֵּחָֽת:  ת  חֲזֵי יְהַב יְיָ אֱלָהָךְ קֳדָמָךְ יָת אַרְעָא סַק אַחֲסֵין כְּמָא דִי מַלִיל יְיָ אֱלָהָא דַאֲבָהָתָךְ לָךְ לָא תִדְחַל וְלָא תִתְּבָּר:

{כב}  שלישי  וַתִּקְרְב֣וּן אֵלַי֘ כֻּלְּכֶם֒ וַתֹּֽאמְר֗וּ נִשְׁלְחָ֤ה אֲנָשִׁים֙ לְפָנֵ֔ינוּ וְיַחְפְּרוּ-לָ֖נוּ אֶת-הָאָ֑רֶץ וְיָשִׁ֤בוּ אֹתָ֨נוּ֙ דָּבָ֔ר אֶת-הַדֶּ֨רֶךְ֙ אֲשֶׁ֣ר נַֽעֲלֶה-בָּ֔הּ וְאֵת֙ הֶֽעָרִ֔ים אֲשֶׁ֥ר נָבֹ֖א אֲלֵיהֶֽן:  מ  וַתִּקְרְב֣וּן אֵלַי֘ כֻּלְּכֶם֒ וַתֹּֽאמְר֗וּ נִשְׁלְחָ֤ה אֲנָשִׁים֙ לְפָנֵ֔ינוּ וְיַחְפְּרוּ-לָ֖נוּ אֶת-הָאָ֑רֶץ וְיָשִׁ֤בוּ אֹתָ֨נוּ֙ דָּבָ֔ר אֶת-הַדֶּ֨רֶךְ֙ אֲשֶׁ֣ר נַֽעֲלֶה-בָּ֔הּ וְאֵת֙ הֶֽעָרִ֔ים אֲשֶׁ֥ר נָבֹ֖א אֲלֵיהֶֽן:  ת  וּקְרֶבְתּוּן לְוָתִי כֻּלְכוֹן וַאֲמַרְתּוּן נִשְׁלַח גֻבְרִין קֳדָמָנָא וִיאַלְלוּן לָנָא יָת אַרְעָא וִיתִיבוּן יָתָנָא פִּתְגָמָא יָת אָרְחָא דִי נִסֵק בַּהּ וְיָת קִרְוַיָא דִי נֵעוֹל לְהֵן:

{כג} וַיִּיטַ֥ב בְּעֵינַ֖י הַדָּבָ֑ר וָֽאֶקַּ֤ח מִכֶּם֙ שְׁנֵ֣ים עָשָׂ֣ר אֲנָשִׁ֔ים אִ֥ישׁ אֶחָ֖ד לַשָּֽׁבֶט:  מ  וַיִּיטַ֥ב בְּעֵינַ֖י הַדָּבָ֑ר וָֽאֶקַּ֤ח מִכֶּם֙ שְׁנֵ֣ים עָשָׂ֣ר אֲנָשִׁ֔ים אִ֥ישׁ אֶחָ֖ד לַשָּֽׁבֶט:  ת  וּשְׁפַר בְּעֵינַי פִּתְגָמָא וּדְבָרִית מִנְכוֹן תְּרֵין עֲשַׂר גֻבְרִין גַבְרָא חַד לְשִׁבְטָא:

{כד} וַיִּפְנוּ֙ וַיַּֽעֲל֣וּ הָהָ֔רָה וַיָּבֹ֖אוּ עַד-נַ֣חַל אֶשְׁכֹּ֑ל וַֽיְרַגְּל֖וּ אֹתָֽהּ:  מ  וַיִּפְנוּ֙ וַיַּֽעֲל֣וּ הָהָ֔רָה וַיָּבֹ֖אוּ עַד-נַ֣חַל אֶשְׁכֹּ֑ל וַֽיְרַגְּל֖וּ אֹתָֽהּ:  ת  וְאִתְפְּנִיוּ וּסְלִיקוּ לְטוּרָא וַאֲתוֹ עַד נַחֲלָא דְאַתְכָּלָא וְאַלִילוּ יָתַהּ:

{כה} וַיִּקְח֤וּ בְיָדָם֙ מִפְּרִ֣י הָאָ֔רֶץ וַיּוֹרִ֖דוּ אֵלֵ֑ינוּ וַיָּשִׁ֨בוּ אֹתָ֤נוּ דָבָר֙ וַיֹּ֣אמְר֔וּ טוֹבָ֣ה הָאָ֔רֶץ אֲשֶׁר-יְהֹוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ:  מ  וַיִּקְח֤וּ בְיָדָם֙ מִפְּרִ֣י הָאָ֔רֶץ וַיּוֹרִ֖דוּ אֵלֵ֑ינוּ וַיָּשִׁ֨בוּ אֹתָ֤נוּ דָבָר֙ וַיֹּ֣אמְר֔וּ טוֹבָ֣ה הָאָ֔רֶץ אֲשֶׁר-יְהֹוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ:  ת  וּנְסִיבוּ בִידֵיהוֹן מֵאִבָּא דְאַרְעָא וְאַחִיתוּ לָנָא וַאֲתִיבוּ יָתָנָא פִּתְגָמָא וַאֲמָרוּ טָבָא אַרְעָא דַיְיָ אֱלָהָנָא יָהֵיב לָנָא:

{כו} וְלֹ֥א אֲבִיתֶ֖ם לַֽעֲלֹ֑ת וַתַּמְר֕וּ אֶת-פִּ֥י יְהֹוָ֖ה אֱלֹֽהֵיכֶֽם:  מ  וְלֹ֥א אֲבִיתֶ֖ם לַֽעֲלֹ֑ת וַתַּמְר֕וּ אֶת-פִּ֥י יְהֹוָ֖ה אֱלֹֽהֵיכֶֽם:  ת  וְלָא אֲבִיתוּן לְמִסַק וְסָרֶבְתּוּן עַל מֵימְרָא דַיְיָ אֱלָהֲכוֹן:

{כז} וַתֵּרָֽגְנ֤וּ בְאָֽהֳלֵיכֶם֙ וַתֹּ֣אמְר֔וּ בְּשִׂנְאַ֤ת יְהֹוָה֙ אֹתָ֔נוּ הֽוֹצִיאָ֖נוּ מֵאֶ֣רֶץ מִצְרָ֑יִם לָתֵ֥ת אֹתָ֛נוּ בְּיַ֥ד הָֽאֱמֹרִ֖י לְהַשְׁמִידֵֽנוּ:  מ  וַתֵּרָֽגְנ֤וּ בְאָֽהֳלֵיכֶם֙ וַתֹּ֣אמְר֔וּ בְּשִׂנְאַ֤ת יְהֹוָה֙ אֹתָ֔נוּ הֽוֹצִיאָ֖נוּ מֵאֶ֣רֶץ מִצְרָ֑יִם לָתֵ֥ת אֹתָ֛נוּ בְּיַ֥ד הָֽאֱמֹרִ֖י לְהַשְׁמִידֵֽנוּ:  ת  וְאִתְרַעַמְתּוּן בְּמַשְׁכְּנֵיכוֹן וַאֲמַרְתּוּן בִּדְסָנֵי יְיָ יָתָנָא אַפְּקָנָא מֵאַרְעָא דְמִצְרָיִם לְמִמְסַר יָתָנָא בִּידָא דֶאֱמוֹרָאָה לְשֵׁיצָיוּתָנָא:

{כח} אָנָ֣ה | אֲנַ֣חְנוּ עֹלִ֗ים אַחֵ֩ינוּ֩ הֵמַ֨סּוּ אֶת-לְבָבֵ֜נוּ לֵאמֹ֗ר עַ֣ם גָּד֤וֹל וָרָם֙ מִמֶּ֔נּוּ עָרִ֛ים גְּדֹלֹ֥ת וּבְצוּרֹ֖ת בַּשָּׁמָ֑יִם וְגַם-בְּנֵ֥י עֲנָקִ֖ים רָאִ֥ינוּ שָֽׁם:  מ  אָנָ֣ה | אֲנַ֣חְנוּ עֹלִ֗ים אַחֵ֩ינוּ֩ הֵמַ֨סּוּ אֶת-לְבָבֵ֜נוּ לֵאמֹ֗ר עַ֣ם גָּד֤וֹל וָרָם֙ מִמֶּ֔נּוּ עָרִ֛ים גְּדֹלֹ֥ת וּבְצוּרֹ֖ת בַּשָּׁמָ֑יִם וְגַם-בְּנֵ֥י עֲנָקִ֖ים רָאִ֥ינוּ שָֽׁם:  ת  לְאָן אֲנַחְנָא סָלְקִין אֲחָנָא תְבָרוּ יָת לִבָּנָא לְמֵימַר עַם רַב וְתַקִיף מִנָנָא קִרְוִין רַבְרְבָן וּכְרִיכָן עַד צֵית שְׁמַיָא וְאַף בְּנֵי גִבָּרָאֵי חֲזֵינָא תַמָן:

{כט} וָֽאֹמַ֖ר אֲלֵכֶ֑ם לֹא-תַֽעַרְצ֥וּן וְֽלֹא-תִֽירְא֖וּן מֵהֶֽם:  מ  וָֽאֹמַ֖ר אֲלֵכֶ֑ם לֹא-תַֽעַרְצ֥וּן וְֽלֹא-תִֽירְא֖וּן מֵהֶֽם:  ת  וַאֲמָרִית לְכוֹן לָא תִתָּבְרוּן וְלָא תִדְחֲלוּן מִנְהוֹן:

{ל} יְהֹוָ֤ה אֱלֹֽהֵיכֶם֙ הַֽהֹלֵ֣ךְ לִפְנֵיכֶ֔ם ה֖וּא יִלָּחֵ֣ם לָכֶ֑ם כְּ֠כֹל אֲשֶׁ֨ר עָשָׂ֧ה אִתְּכֶ֛ם בְּמִצְרַ֖יִם לְעֵֽינֵיכֶֽם:  מ  יְהֹוָ֤ה אֱלֹֽהֵיכֶם֙ הַֽהֹלֵ֣ךְ לִפְנֵיכֶ֔ם ה֖וּא יִלָּחֵ֣ם לָכֶ֑ם כְּ֠כֹל אֲשֶׁ֨ר עָשָׂ֧ה אִתְּכֶ֛ם בְּמִצְרַ֖יִם לְעֵֽינֵיכֶֽם:  ת  יְיָ אֱלָהֲכוֹן דִמְדַבַּר קֳדָמֵיכוֹן הוּא יְגִיחַ לְכוֹן כְּכֹל דִי עֲבַד עִמְכוֹן בְּמִצְרַיִם לְעֵינֵיכוֹן:

{לא} וּבַמִּדְבָּר֙ אֲשֶׁ֣ר רָאִ֔יתָ אֲשֶׁ֤ר נְשָֽׂאֲךָ֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ כַּֽאֲשֶׁ֥ר יִשָּׂא-אִ֖ישׁ אֶת-בְּנ֑וֹ בְּכָל-הַדֶּ֨רֶךְ֙ אֲשֶׁ֣ר הֲלַכְתֶּ֔ם עַד-בֹּֽאֲכֶ֖ם עַד-הַמָּק֥וֹם הַזֶּֽה:  מ  וּבַמִּדְבָּר֙ אֲשֶׁ֣ר רָאִ֔יתָ אֲשֶׁ֤ר נְשָֽׂאֲךָ֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ כַּֽאֲשֶׁ֥ר יִשָּׂא-אִ֖ישׁ אֶת-בְּנ֑וֹ בְּכָל-הַדֶּ֨רֶךְ֙ אֲשֶׁ֣ר הֲלַכְתֶּ֔ם עַד-בֹּֽאֲכֶ֖ם עַד-הַמָּק֥וֹם הַזֶּֽה:  ת  וּבְמַדְבְּרָא דִי חֲזֵיתָא דִי סוֹבְרָךְ יְיָ אֱלָהָךְ כְּמָא דִי מִסוֹבַר גַבְרָא יָת בְּרֵהּ בְּכָל אָרְחָא דִי הַלֶכְתּוּן עַד מֵיתֵיכוֹן עַד אַתְרָא הָדֵין:

{לב} וּבַדָּבָ֖ר הַזֶּ֑ה אֵֽינְכֶם֙ מַֽאֲמִינִ֔ם בַּֽיהֹוָ֖ה אֱלֹֽהֵיכֶֽם:  מ  וּבַדָּבָ֖ר הַזֶּ֑ה אֵֽינְכֶם֙ מַֽאֲמִינִ֔ם בַּֽיהֹוָ֖ה אֱלֹֽהֵיכֶֽם:  ת  וּבְפִתְגָמָא הָדֵין לֵיתֵיכוֹן מְהֵימְנִין בְּמֵימְרָא דַיְיָ אֱלָהֲכוֹן:

{לג} הַֽהֹלֵ֨ךְ לִפְנֵיכֶ֜ם בַּדֶּ֗רֶךְ לָת֥וּר לָכֶ֛ם מָק֖וֹם לַֽחֲנֹֽתְכֶ֑ם בָּאֵ֣שׁ | לַ֗יְלָה לַרְאֹֽתְכֶם֙ בַּדֶּ֨רֶךְ֙ אֲשֶׁ֣ר תֵּֽלְכוּ-בָ֔הּ וּבֶֽעָנָ֖ן יוֹמָֽם:  מ  הַֽהֹלֵ֨ךְ לִפְנֵיכֶ֜ם בַּדֶּ֗רֶךְ לָת֥וּר לָכֶ֛ם מָק֖וֹם לַֽחֲנֹֽתְכֶ֑ם בָּאֵ֣שׁ | לַ֗יְלָה לַרְאֹֽתְכֶם֙ בַּדֶּ֨רֶךְ֙ אֲשֶׁ֣ר תֵּֽלְכוּ-בָ֔הּ וּבֶֽעָנָ֖ן יוֹמָֽם:  ת  דִי מְדַבֵּר קֳדָמֵיכוֹן בְּאָרְחָא לְאַתְקְנָאָה לְכוֹן אֲתַר בֵּית מֵישְׁרִי לְאַשְׁרָיוּתְכוֹן בְּעַמוּדָא דְאֶשָׁתָא בְּלֵילְיָא לְאַחֲזָיוּתְכוֹן בְּאָרְחָא דִי תְהָכוּן בַּהּ וּבְעַמוּדָא דַעֲנָנָא בִּימָמָא:

{לד} וַיִּשְׁמַ֥ע יְהֹוָ֖ה אֶת-קוֹל דִּבְרֵיכֶ֑ם וַיִּקְצֹ֖ף וַיִּשָּׁבַ֥ע לֵאמֹֽר:  מ  וַיִּשְׁמַ֥ע יְהֹוָ֖ה אֶת-קוֹל דִּבְרֵיכֶ֑ם וַיִּקְצֹ֖ף וַיִּשָּׁבַ֥ע לֵאמֹֽר:  ת  וּשְׁמִיעַ קֳדָם יְיָ יָת קַל פִּתְגָמֵיכוֹן וּרְגֵז וְקַיַם לְמֵימָר:

{לה} אִם-יִרְאֶ֥ה אִישׁ֙ בָּֽאֲנָשִׁ֣ים הָאֵ֔לֶּה הַדּ֥וֹר הָרָ֖ע הַזֶּ֑ה אֵ֚ת הָאָ֣רֶץ הַטּוֹבָ֔ה אֲשֶׁ֣ר נִשְׁבַּ֔עְתִּי לָתֵ֖ת לַֽאֲבֹֽתֵיכֶֽם:  מ  אִם-יִרְאֶ֥ה אִישׁ֙ בָּֽאֲנָשִׁ֣ים הָאֵ֔לֶּה הַדּ֥וֹר הָרָ֖ע הַזֶּ֑ה אֵ֚ת הָאָ֣רֶץ הַטּוֹבָ֔ה אֲשֶׁ֣ר נִשְׁבַּ֔עְתִּי לָתֵ֖ת לַֽאֲבֹֽתֵיכֶֽם:  ת  אִם יֶחֱזֵי גְבַר בְּגֻבְרַיָא הָאִלֵין דָרָא בִישָׁא הָדֵין יָת אַרְעָא טַבְתָא דִי קַיֵמִית לְמִתַּן לַאֲבָהָתְכוֹן:

{לו} זֽוּלָתִ֞י כָּלֵ֤ב בֶּן-יְפֻנֶּה֙ הוּא יִרְאֶ֔נָּה וְלֽוֹ-אֶתֵּ֧ן אֶת-הָאָ֛רֶץ אֲשֶׁ֥ר דָּֽרַךְ-בָּ֖הּ וּלְבָנָ֑יו יַ֕עַן אֲשֶׁ֥ר מִלֵּ֖א אַֽחֲרֵ֥י יְהֹוָֽה:  מ  זֽוּלָתִ֞י כָּלֵ֤ב בֶּן-יְפֻנֶּה֙ הוּא יִרְאֶ֔נָּה וְלֽוֹ-אֶתֵּ֧ן אֶת-הָאָ֛רֶץ אֲשֶׁ֥ר דָּֽרַךְ-בָּ֖הּ וּלְבָנָ֑יו יַ֕עַן אֲשֶׁ֥ר מִלֵּ֖א אַֽחֲרֵ֥י יְהֹוָֽה:  ת  אֱלָהֵן כָּלֵב בַּר יְפֻנֶה הוּא יֶחְזִנַהּ וְלֵהּ אֶתֵּן יָת אַרְעָא דִי דְרַךְ בַּהּ וְלִבְנוֹהִי חֲלַף דִי אַשְׁלִים בָּתַר דַחַלְתָּא דַיְיָ:

{לז} גַּם-בִּי֙ הִתְאַנַּ֣ף יְהֹוָ֔ה בִּגְלַלְכֶ֖ם לֵאמֹ֑ר גַּם-אַתָּ֖ה לֹֽא-תָבֹ֥א שָֽׁם:  מ  גַּם-בִּי֙ הִתְאַנַּ֣ף יְהֹוָ֔ה בִּגְלַלְכֶ֖ם לֵאמֹ֑ר גַּם-אַתָּ֖ה לֹֽא-תָבֹ֥א שָֽׁם:  ת  אַף עֲלַי הֲוָה רְגַז מִן קֳדָם יְיָ בְּדִילְכוֹן לְמֵימָר אַף אַתְּ לָא תֵיעוֹל תַּמָן:

{לח} יְהוֹשֻׁ֤עַ בִּן-נוּן֙ הָֽעֹמֵ֣ד לְפָנֶ֔יךָ ה֖וּא יָ֣בֹא שָׁ֑מָּה אֹת֣וֹ חַזֵּ֔ק כִּי-ה֖וּא יַנְחִלֶ֥נָּה אֶת-יִשְׂרָאֵֽל:  מ  יְהוֹשֻׁ֤עַ בִּן-נוּן֙ הָֽעֹמֵ֣ד לְפָנֶ֔יךָ ה֖וּא יָ֣בֹא שָׁ֑מָּה אֹת֣וֹ חַזֵּ֔ק כִּי-ה֖וּא יַנְחִלֶ֥נָּה אֶת-יִשְׂרָאֵֽל:  ת  יְהוֹשֻׁעַ בַּר נוּן דְקָאֵים קֳדָמָךְ הוּא יֵיעוֹל תַּמָן יָתֵהּ תַּקֵף אֲרֵי הוּא יַחְסְנִנַהּ לְיִשְׂרָאֵל:

{לט}  רביעי  וְטַפְּכֶם֩ אֲשֶׁ֨ר אֲמַרְתֶּ֜ם לָבַ֣ז יִֽהְיֶ֗ה וּ֠בְנֵיכֶם אֲשֶׁ֨ר לֹא-יָֽדְע֤וּ הַיּוֹם֙ טוֹב וָרָ֔ע הֵ֖מָּה יָבֹ֣אוּ שָׁ֑מָּה וְלָהֶ֣ם אֶתְּנֶ֔נָּה וְהֵ֖ם יִֽירָשֽׁוּהָ:  מ  וְטַפְּכֶם֩ אֲשֶׁ֨ר אֲמַרְתֶּ֜ם לָבַ֣ז יִֽהְיֶ֗ה וּ֠בְנֵיכֶם אֲשֶׁ֨ר לֹא-יָֽדְע֤וּ הַיּוֹם֙ טוֹב וָרָ֔ע הֵ֖מָּה יָבֹ֣אוּ שָׁ֑מָּה וְלָהֶ֣ם אֶתְּנֶ֔נָּה וְהֵ֖ם יִֽירָשֽׁוּהָ:  ת  וְטַפְלְכוֹן דִי אֲמַרְתּוּן לְבִזָא יְהוֹן וּבְנֵיכוֹן דִי לָא יְדָעוּ יוֹמָא דֵין טַב וּבִישׁ אִנוּן יַעֲלוּן לְתַמָן וּלְהוֹן אֶתְּנִנַהּ וְאִנוּן יֵירְתֻנַהּ:

{מ} וְאַתֶּ֖ם פְּנ֣וּ לָכֶ֑ם וּסְע֥וּ הַמִּדְבָּ֖רָה דֶּ֥רֶךְ יַם-סֽוּף:  מ  וְאַתֶּ֖ם פְּנ֣וּ לָכֶ֑ם וּסְע֥וּ הַמִּדְבָּ֖רָה דֶּ֥רֶךְ יַם-סֽוּף:  ת  וְאַתּוּן אִתְפְּנִיוּ לְכוֹן וְטוּלוּ לְמַדְבְּרָא אוֹרַח יַמָא דְסוּף:

{מא} וַֽתַּֽעֲנ֣וּ | וַתֹּֽאמְר֣וּ אֵלַ֗י חָטָאנוּ֘ לַֽיהֹוָה֒ אֲנַ֤חְנוּ נַֽעֲלֶה֙ וְנִלְחַ֔מְנוּ כְּכֹ֥ל אֲשֶׁר-צִוָּ֖נוּ יְהֹוָ֣ה אֱלֹהֵ֑ינוּ וַֽתַּחְגְּר֗וּ אִ֚ישׁ אֶת-כְּלֵ֣י מִלְחַמְתּוֹ וַ֔תָּהִ֖ינוּ לַֽעֲלֹ֥ת הָהָֽרָה:  מ  וַֽתַּֽעֲנ֣וּ | וַתֹּֽאמְר֣וּ אֵלַ֗י חָטָאנוּ֘ לַֽיהֹוָה֒ אֲנַ֤חְנוּ נַֽעֲלֶה֙ וְנִלְחַ֔מְנוּ כְּכֹ֥ל אֲשֶׁר-צִוָּ֖נוּ יְהֹוָ֣ה אֱלֹהֵ֑ינוּ וַֽתַּחְגְּר֗וּ אִ֚ישׁ אֶת-כְּלֵ֣י מִלְחַמְתּוֹ וַ֔תָּהִ֖ינוּ לַֽעֲלֹ֥ת הָהָֽרָה:  ת  וַאֲתֶבְתּוּן וַאֲמַרְתּוּן לִי חַבְנָא קֳדָם יְיָ אֲנַחְנָא נִיסַק וּנְגִיחַ קְרָב כְּכֹל דִי פַקְדָנָא יְיָ אֱלָהָנָא וְזָרֶזְתּוּן גְבַר יָת מָנֵי קְרָבֵהּ וְשָׁרֵיתוּן לְמֵיסַק לְטּוּרָא:

{מב} וַיֹּ֨אמֶר יְהֹוָ֜ה אֵלַ֗י אֱמֹ֤ר לָהֶם֙ לֹ֤א תַֽעֲלוּ֙ וְלֹ֣א-תִֽלָּֽחֲמ֔וּ כִּ֥י אֵינֶ֖נִּי בְּקִרְבְּכֶ֑ם וְלֹא֙ תִּנָּֽגְפ֔וּ לִפְנֵ֖י אֹֽיְבֵיכֶֽם:  מ  וַיֹּ֨אמֶר יְהֹוָ֜ה אֵלַ֗י אֱמֹ֤ר לָהֶם֙ לֹ֤א תַֽעֲלוּ֙ וְלֹ֣א-תִֽלָּֽחֲמ֔וּ כִּ֥י אֵינֶ֖נִּי בְּקִרְבְּכֶ֑ם וְלֹא֙ תִּנָּֽגְפ֔וּ לִפְנֵ֖י אֹֽיְבֵיכֶֽם:  ת  וַאֲמַר יְיָ לִי אֱמַר לְהוֹן לָא תִסְקוּן וְלָא תְגִחוּן קְרָב אֲרֵי לֵית שְׁכִינְתִּי שָׁרְיָא בֵינֵיכוֹן וְלָא תִּתָּבְרוּן קֳדָם בַּעֲלֵי דְבָבֵיכוֹן:

{מג} וָֽאֲדַבֵּ֥ר אֲלֵיכֶ֖ם וְלֹ֣א שְׁמַעְתֶּ֑ם וַתַּמְרוּ֙ אֶת-פִּ֣י יְהֹוָ֔ה וַתָּזִ֖דוּ וַתַּֽעֲל֥וּ הָהָֽרָה:  מ  וָֽאֲדַבֵּ֥ר אֲלֵיכֶ֖ם וְלֹ֣א שְׁמַעְתֶּ֑ם וַתַּמְרוּ֙ אֶת-פִּ֣י יְהֹוָ֔ה וַתָּזִ֖דוּ וַתַּֽעֲל֥וּ הָהָֽרָה:  ת  וּמַלֵלִית עִמְכוֹן וְלָא קַבֶּלְתּוּן וְסָרֶבְתּוּן עַל גְזֵרַת מֵימְרָא דַיְיָ וְאַרְשַׁעְתּוּן וּסְלֶקְתּוּן לְטוּרָא:

{מד} וַיֵּצֵ֨א הָֽאֱמֹרִ֜י הַיּשֵׁ֨ב בָּהָ֤ר הַהוּא֙ לִקְרַאתְכֶ֔ם וַיִּרְדְּפ֣וּ אֶתְכֶ֔ם כַּֽאֲשֶׁ֥ר תַּֽעֲשֶׂ֖ינָה הַדְּבֹרִ֑ים וַיַּכְּת֥וּ אֶתְכֶ֛ם בְּשֵׂעִ֖יר עַד-חָרְמָֽה:  מ  וַיֵּצֵ֨א הָֽאֱמֹרִ֜י הַיּשֵׁ֨ב בָּהָ֤ר הַהוּא֙ לִקְרַאתְכֶ֔ם וַיִּרְדְּפ֣וּ אֶתְכֶ֔ם כַּֽאֲשֶׁ֥ר תַּֽעֲשֶׂ֖ינָה הַדְּבֹרִ֑ים וַיַּכְּת֥וּ אֶתְכֶ֛ם בְּשֵׂעִ֖יר עַד-חָרְמָֽה:  ת  וּנְפַק אֱמֹרָאָה דְיָתֵב בְּטוּרָא הַהוּא לְקַדָמוּתְכוֹן וּרְדָפוּ יָתְכוֹן כְּמָא דִי נַתְּזָן דַבַּרְיָתָא וּטְרָדוּ יָתְכוֹן בְּשֵׂעִיר עַד חָרְמָה:

{מה} וַתָּשֻׁ֥בוּ וַתִּבְכּ֖וּ לִפְנֵ֣י יְהֹוָ֑ה וְלֹֽא-שָׁמַ֤ע יְהֹוָה֙ בְּקֹ֣לְכֶ֔ם וְלֹ֥א הֶֽאֱזִ֖ין אֲלֵיכֶֽם:  מ  וַתָּשֻׁ֥בוּ וַתִּבְכּ֖וּ לִפְנֵ֣י יְהֹוָ֑ה וְלֹֽא-שָׁמַ֤ע יְהֹוָה֙ בְּקֹ֣לְכֶ֔ם וְלֹ֥א הֶֽאֱזִ֖ין אֲלֵיכֶֽם:  ת  וְתַבְתּוּן וּבְכִיתוּן קֳדָם יְיָ וְלָא קַבִּיל יְיָ צְלוֹתְכוֹן וְלָא אַצֵית לְמִלֵיכוֹן:

{מו} וַתֵּֽשְׁב֥וּ בְקָדֵ֖שׁ יָמִ֣ים רַבִּ֑ים כַּיָּמִ֖ים אֲשֶׁ֥ר יְשַׁבְתֶּֽם:  מ  וַתֵּֽשְׁב֥וּ בְקָדֵ֖שׁ יָמִ֣ים רַבִּ֑ים כַּיָּמִ֖ים אֲשֶׁ֥ר יְשַׁבְתֶּֽם:  ת  וִיתֶבְתּוּן בִּרְקָם יוֹמִין סַגִיאִין כְּיוֹמַיָא דִי יְתֶבְתּוּן:




דברים פרק-ב

{א} וַנֵּ֜פֶן וַנִּסַּ֤ע הַמִּדְבָּ֨רָה֙ דֶּ֣רֶךְ יַם-ס֔וּף כַּֽאֲשֶׁ֛ר דִּבֶּ֥ר יְהֹוָ֖ה אֵלָ֑י וַנָּ֥סָב אֶת-הַר-שֵׂעִ֖יר יָמִ֥ים רַבִּֽים:  מ  וַנֵּ֜פֶן וַנִּסַּ֤ע הַמִּדְבָּ֨רָה֙ דֶּ֣רֶךְ יַם-ס֔וּף כַּֽאֲשֶׁ֛ר דִּבֶּ֥ר יְהֹוָ֖ה אֵלָ֑י וַנָּ֥סָב אֶת-הַר-שֵׂעִ֖יר יָמִ֥ים רַבִּֽים:  ת  וְאִתְפְּנֵינָא וּנְטַלְנָא לְמַדְבְּרָא אוֹרַח יַמָא דְסוּף כְּמָא דִי מַלִיל יְיָ עִמִי וְאַקֵפְנָא יָת טוּרָא דְשֵׂעִיר יוֹמִין סַגִיאִין: (ס)

{ב}  חמישי  וַיֹּ֥אמֶר יְהֹוָ֖ה אֵלַ֥י לֵאמֹֽר:  מ  וַיֹּ֥אמֶר יְהֹוָ֖ה אֵלַ֥י לֵאמֹֽר:  ת  וַאֲמַר יְיָ לִי לְמֵימָר:

{ג} רַב-לָכֶ֕ם סֹ֖ב אֶת-הָהָ֣ר הַזֶּ֑ה פְּנ֥וּ לָכֶ֖ם צָפֹֽנָה:  מ  רַב-לָכֶ֕ם סֹ֖ב אֶת-הָהָ֣ר הַזֶּ֑ה פְּנ֥וּ לָכֶ֖ם צָפֹֽנָה:  ת  סַגִי לְכוֹן דְאַקֶפְתּוּן יָת טוּרָא הָדֵין אִתְפְּנִיוּ לְכוֹן צִפּוּנָא:

{ד} וְאֶת-הָעָם֘ צַ֣ו לֵאמֹר֒ אַתֶּ֣ם עֹֽבְרִ֗ים בִּגְבוּל֙ אֲחֵיכֶ֣ם בְּנֵֽי-עֵשָׂ֔ו הַיֹּֽשְׁבִ֖ים בְּשֵׂעִ֑יר וְיִֽירְא֣וּ מִכֶּ֔ם וְנִשְׁמַרְתֶּ֖ם מְאֹֽד:  מ  וְאֶת-הָעָם֘ צַ֣ו לֵאמֹר֒ אַתֶּ֣ם עֹֽבְרִ֗ים בִּגְבוּל֙ אֲחֵיכֶ֣ם בְּנֵֽי-עֵשָׂ֔ו הַיֹּֽשְׁבִ֖ים בְּשֵׂעִ֑יר וְיִֽירְא֣וּ מִכֶּ֔ם וְנִשְׁמַרְתֶּ֖ם מְאֹֽד:  ת  וְיָת עַמָא פַּקֵד לְמֵימַר אַתּוּן עָבְרִין בִּתְחוּם אֲחוּכוֹן בְּנֵי עֵשָׂו דְיָתְבִין בְּשֵׂעִיר וְיִדְחֲלוּן מִנְכוֹן וְתִסְתַּמְרוּן לַחֲדָא:

{ה} אַל-תִּתְגָּ֣רוּ בָ֔ם כִּ֠י לֹֽא-אֶתֵּ֤ן לָכֶם֙ מֵֽאַרְצָ֔ם עַ֖ד מִדְרַ֣ךְ כַּף-רָ֑גֶל כִּֽי-יְרֻשָּׁ֣ה לְעֵשָׂ֔ו נָתַ֖תִּי אֶת-הַ֥ר שֵׂעִֽיר:  מ  אַל-תִּתְגָּ֣רוּ בָ֔ם כִּ֠י לֹֽא-אֶתֵּ֤ן לָכֶם֙ מֵֽאַרְצָ֔ם עַ֖ד מִדְרַ֣ךְ כַּף-רָ֑גֶל כִּֽי-יְרֻשָּׁ֣ה לְעֵשָׂ֔ו נָתַ֖תִּי אֶת-הַ֥ר שֵׂעִֽיר:  ת  לָא תִתְגָרוּן בְּהוֹן אֲרֵי לָא אֶתֵּן לְכוֹן מֵאַרְעֲהוֹן עַד מִדְרַךְ פַּרְסַת רִגְלָא אֲרֵי יְרֻתָּא לְעֵשָׂו יְהָבִית יָת טוּרָא דְשֵׂעִיר:

{ו} אֹ֣כֶל תִּשְׁבְּר֧וּ מֵֽאִתָּ֛ם בַּכֶּ֖סֶף וַֽאֲכַלְתֶּ֑ם וְגַם-מַ֜יִם תִּכְר֧וּ מֵֽאִתָּ֛ם בַּכֶּ֖סֶף וּשְׁתִיתֶֽם:  מ  אֹ֣כֶל תִּשְׁבְּר֧וּ מֵֽאִתָּ֛ם בַּכֶּ֖סֶף וַֽאֲכַלְתֶּ֑ם וְגַם-מַ֜יִם תִּכְר֧וּ מֵֽאִתָּ֛ם בַּכֶּ֖סֶף וּשְׁתִיתֶֽם:  ת  עִבּוּרָא תִזְבְּנוּן מִנְהוֹן בְּכַסְפָּא וְתֵיכְלוּן וְאַף מַיָא תִּכְרוּן מִנְהוֹן בְּכַסְפָּא וְתִשְׁתּוּן:

{ז} כִּי֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ בֵּֽרַכְךָ֗ בְּכֹל֙ מַֽעֲשֵׂ֣ה יָדֶ֔ךָ יָדַ֣ע לֶכְתְּךָ֔ אֶת-הַמִּדְבָּ֥ר הַגָּדֹ֖ל הַזֶּ֑ה זֶ֣ה | אַרְבָּעִ֣ים שָׁנָ֗ה יְהֹוָ֤ה אֱלֹהֶ֨יךָ֙ עִמָּ֔ךְ לֹ֥א חָסַ֖רְתָּ דָּבָֽר:  מ  כִּי֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ בֵּֽרַכְךָ֗ בְּכֹל֙ מַֽעֲשֵׂ֣ה יָדֶ֔ךָ יָדַ֣ע לֶכְתְּךָ֔ אֶת-הַמִּדְבָּ֥ר הַגָּדֹ֖ל הַזֶּ֑ה זֶ֣ה | אַרְבָּעִ֣ים שָׁנָ֗ה יְהֹוָ֤ה אֱלֹהֶ֨יךָ֙ עִמָּ֔ךְ לֹ֥א חָסַ֖רְתָּ דָּבָֽר:  ת  אֲרֵי יְיָ אֱלָהָךְ בֵּרְכָךְ בְּכָל עֹבַד יְדָךְ סַפֶּק לָךְ צָרְכָּךְ בְּמֶהָכָךְ יָת מַדְבְּרָא רַבָּא הָדֵין דְנַן אַרְבְּעִין שְׁנִין מֵימְרָא דַיְיָ אֱלָהָךְ בְּסַעֲדָךְ לָא מְנַעְתָּא מִדָעַם:

{ח}  וַֽנַּֽעֲבֹ֞ר מֵאֵ֧ת אַחֵ֣ינוּ בְנֵֽי עֵשָׂ֗ו הַיּֽשְׁבִים֙ בְּשֵׂעִ֔יר מִדֶּ֨רֶךְ֙ הָֽעֲרָבָ֔ה מֵֽאֵילַ֖ת וּמֵֽעֶצְיֹ֣ן גָּ֑בֶר וַנֵּ֨פֶן֙ וַֽנַּֽעֲבֹ֔ר דֶּ֖רֶךְ מִדְבַּ֥ר מוֹאָֽב:  מ   וַֽנַּֽעֲבֹ֞ר מֵאֵ֧ת אַחֵ֣ינוּ בְנֵֽי עֵשָׂ֗ו הַיּֽשְׁבִים֙ בְּשֵׂעִ֔יר מִדֶּ֨רֶךְ֙ הָֽעֲרָבָ֔ה מֵֽאֵילַ֖ת וּמֵֽעֶצְיֹ֣ן גָּ֑בֶר וַנֵּ֨פֶן֙ וַֽנַּֽעֲבֹ֔ר דֶּ֖רֶךְ מִדְבַּ֥ר מוֹאָֽב:  ת   וַעֲבַרְנָא מִלְוַת אֲחוּנָא בְנֵי עֵשָׂו דְיָתְבִין בְּשֵׂעִיר מֵאוֹרַח מֵישְׁרָא מֵאֵילַת וּמֶעְצְיוֹן גָבֶר וְאִתְפְּנֵינָא וַעֲבַרְנָא אוֹרַח מַדְבְּרָא דְמוֹאָב:

{ט} וַיֹּ֨אמֶר יְהֹוָ֜ה אֵלַ֗י אַל-תָּ֨צַר֙ אֶת-מוֹאָ֔ב וְאַל-תִּתְגָּ֥ר בָּ֖ם מִלְחָמָ֑ה כִּ֠י לֹֽא-אֶתֵּ֨ן לְךָ֤ מֵֽאַרְצוֹ֙ יְרֻשָּׁ֔ה כִּ֣י לִבְנֵי-ל֔וֹט נָתַ֥תִּי אֶת-עָ֖ר יְרֻשָּֽׁה:  מ  וַיֹּ֨אמֶר יְהֹוָ֜ה אֵלַ֗י אַל-תָּ֨צַר֙ אֶת-מוֹאָ֔ב וְאַל-תִּתְגָּ֥ר בָּ֖ם מִלְחָמָ֑ה כִּ֠י לֹֽא-אֶתֵּ֨ן לְךָ֤ מֵֽאַרְצוֹ֙ יְרֻשָּׁ֔ה כִּ֣י לִבְנֵי-ל֔וֹט נָתַ֥תִּי אֶת-עָ֖ר יְרֻשָּֽׁה:  ת  וַאֲמַר יְיָ לִי לָא תְצוּר עַל מוֹאַבָאֵי וְלָא תִתְגָרֵי לְמֶעְבַּד עִמְהוֹן קְרָב אֲרֵי לָא אֶתֵּן לָךְ מֵאַרְעֵהּ יְרֻתָּא אֲרֵי לִבְנֵי לוֹט יְהָבִית יָת לְחָיָת יְרֻתָּא:

{י} הָֽאֵמִ֥ים לְפָנִ֖ים יָ֣שְׁבוּ בָ֑הּ עַ֣ם גָּד֥וֹל וְרַ֛ב וָרָ֖ם כַּֽעֲנָקִֽים:  מ  הָֽאֵמִ֥ים לְפָנִ֖ים יָ֣שְׁבוּ בָ֑הּ עַ֣ם גָּד֥וֹל וְרַ֛ב וָרָ֖ם כַּֽעֲנָקִֽים:  ת  אֵימְתָנֵי מִלְקַדְמִין יְתִיבוּ בַהּ עַם רַב וְסַגִי וְתַקִיף כְּגִבָּרַיָא:

{יא} רְפָאִ֛ים יֵחָֽשְׁב֥וּ אַף-הֵ֖ם כַּֽעֲנָקִ֑ים וְהַמֹּ֣אָבִ֔ים יִקְרְא֥וּ לָהֶ֖ם אֵמִֽים:  מ  רְפָאִ֛ים יֵחָֽשְׁב֥וּ אַף-הֵ֖ם כַּֽעֲנָקִ֑ים וְהַמֹּ֣אָבִ֔ים יִקְרְא֥וּ לָהֶ֖ם אֵמִֽים:  ת  גִבָּרִין מִתְחַשְׁבִין אַף אִנוּן כְּגִבָּרַיָא וּמוֹאַבָאֵי קְרַן לְהוֹן אֵימְתָנֵי:

{יב} וּבְשֵׂעִ֞יר יָֽשְׁב֣וּ הַֽחֹרִים֘ לְפָנִים֒ וּבְנֵ֧י עֵשָׂ֣ו יִֽירָשׁ֗וּם וַיַּשְׁמִידוּם֙ מִפְּנֵיהֶ֔ם וַיֵּֽשְׁב֖וּ תַּחְתָּ֑ם כַּֽאֲשֶׁ֧ר עָשָׂ֣ה יִשְׂרָאֵ֗ל לְאֶ֨רֶץ֙ יְרֻשָּׁת֔וֹ אֲשֶׁר-נָתַ֥ן יְהֹוָ֖ה לָהֶֽם:  מ  וּבְשֵׂעִ֞יר יָֽשְׁב֣וּ הַֽחֹרִים֘ לְפָנִים֒ וּבְנֵ֧י עֵשָׂ֣ו יִֽירָשׁ֗וּם וַיַּשְׁמִידוּם֙ מִפְּנֵיהֶ֔ם וַיֵּֽשְׁב֖וּ תַּחְתָּ֑ם כַּֽאֲשֶׁ֧ר עָשָׂ֣ה יִשְׂרָאֵ֗ל לְאֶ֨רֶץ֙ יְרֻשָּׁת֔וֹ אֲשֶׁר-נָתַ֥ן יְהֹוָ֖ה לָהֶֽם:  ת  וּבְשֵׂעִיר יְתִיבוּ חֹרָאֵי מִלְקַדְמִין וּבְנֵי עֵשָׂו תָּרֵכֻנוּן וְשֵׁצָיֻנוּן מִקֳדָמֵיהוֹן וִיתִיבוּ בְּאַתְרֵהוֹן כְּמָא דִי עֲבַד יִשְׂרָאֵל לְאַרְעָא יְרֻתָּתֵהּ דִיהַב יְיָ לְהוֹן:

{יג} עַתָּ֗ה קֻ֛מוּ וְעִבְר֥וּ לָכֶ֖ם אֶת-נַ֣חַל זָ֑רֶד וַֽנַּֽעֲבֹ֖ר אֶת-נַ֥חַל זָֽרֶד:  מ  עַתָּ֗ה קֻ֛מוּ וְעִבְר֥וּ לָכֶ֖ם אֶת-נַ֣חַל זָ֑רֶד וַֽנַּֽעֲבֹ֖ר אֶת-נַ֥חַל זָֽרֶד:  ת  כְּעַן קוּמוּ וְעִבָרוּ לְכוֹן יָת נַחֲלָא דְזָרֶד וַעֲבַרְנָא יָת נַחֲלָא דְזָרֶד:

{יד} וְהַיָּמִ֞ים אֲשֶׁר-הָלַ֣כְנוּ | מִקָּדֵ֣שׁ בַּרְנֵ֗עַ עַ֤ד אֲשֶׁר-עָבַ֨רְנוּ֙ אֶת-נַ֣חַל זֶ֔רֶד שְׁלשִׁ֥ים וּשְׁמֹנֶ֖ה שָׁנָ֑ה עַד-תֹּ֨ם כָּל-הַדּ֜וֹר אַנְשֵׁ֤י הַמִּלְחָמָה֙ מִקֶּ֣רֶב הַֽמַּֽחֲנֶ֔ה כַּֽאֲשֶׁ֛ר נִשְׁבַּ֥ע יְהֹוָ֖ה לָהֶֽם:  מ  וְהַיָּמִ֞ים אֲשֶׁר-הָלַ֣כְנוּ | מִקָּדֵ֣שׁ בַּרְנֵ֗עַ עַ֤ד אֲשֶׁר-עָבַ֨רְנוּ֙ אֶת-נַ֣חַל זֶ֔רֶד שְׁלשִׁ֥ים וּשְׁמֹנֶ֖ה שָׁנָ֑ה עַד-תֹּ֨ם כָּל-הַדּ֜וֹר אַנְשֵׁ֤י הַמִּלְחָמָה֙ מִקֶּ֣רֶב הַֽמַּֽחֲנֶ֔ה כַּֽאֲשֶׁ֛ר נִשְׁבַּ֥ע יְהֹוָ֖ה לָהֶֽם:  ת  וְיוֹמַיָא דִי הַלֶכְנָא מֵרְקַם גֵיאָה עַד דִי עֲבַרְנָא יָת נַחֲלָא דְזֶרֶד תְּלָתִין וְתַמְנֵי שְׁנִין עַד דְסַף כָּל דָרָא גֻבְרֵי מְגִיחֵי קְרָבָא מִגוֹ מַשְׁרִיתָא כְּמָא דִי קַיִים יְיָ לְהוֹן:

{טו} וְגַ֤ם יַד-יְהֹוָה֙ הָ֣יְתָה בָּ֔ם לְהֻמָּ֖ם מִקֶּ֣רֶב הַֽמַּֽחֲנֶ֑ה עַ֖ד תֻּמָּֽם:  מ  וְגַ֤ם יַד-יְהֹוָה֙ הָ֣יְתָה בָּ֔ם לְהֻמָּ֖ם מִקֶּ֣רֶב הַֽמַּֽחֲנֶ֑ה עַ֖ד תֻּמָּֽם:  ת  וְאַף מָחָא מִן קֳדָם יְיָ הֲוַת בְּהוֹן לְשֵׁצָיוּתְהוֹן מִגוֹ מַשְׁרִיתָא עַד דִשְׁלִימוּ:

{טז} וַיְהִ֨י כַֽאֲשֶׁר-תַּ֜מּוּ כָּל-אַנְשֵׁ֧י הַמִּלְחָמָ֛ה לָמ֖וּת מִקֶּ֥רֶב הָעָֽם:  מ  וַיְהִ֨י כַֽאֲשֶׁר-תַּ֜מּוּ כָּל-אַנְשֵׁ֧י הַמִּלְחָמָ֛ה לָמ֖וּת מִקֶּ֥רֶב הָעָֽם:  ת  וַהֲוָה כַּד שְׁלִימוּ כָּל גֻבְרֵי מְגִיחֵי קְרָבָא לִמְמַת מִגוֹ עַמָא: (ס)

{יז} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֵלַ֥י לֵאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֵלַ֥י לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִמִי לְמֵימָר:

{יח} אַתָּ֨ה עֹבֵ֥ר הַיּ֛וֹם אֶת-גְּב֥וּל מוֹאָ֖ב אֶת-עָֽר:  מ  אַתָּ֨ה עֹבֵ֥ר הַיּ֛וֹם אֶת-גְּב֥וּל מוֹאָ֖ב אֶת-עָֽר:  ת  אַתְּ עָבֵר יוֹמָא דֵין יָת תְּחוּם מוֹאָב יָת לְחָיָת:

{יט} וְקָֽרַבְתָּ֗ מ֚וּל בְּנֵ֣י עַמּ֔וֹן אַל-תְּצֻרֵ֖ם וְאַל-תִּתְגָּ֣ר בָּ֑ם כִּ֣י לֹֽא-אֶ֠תֵּן מֵאֶ֨רֶץ בְּנֵֽי-עַמּ֤וֹן לְךָ֙ יְרֻשָּׁ֔ה כִּ֥י לִבְנֵי-ל֖וֹט נְתַתִּ֥יהָ יְרֻשָּֽׁה:  מ  וְקָֽרַבְתָּ֗ מ֚וּל בְּנֵ֣י עַמּ֔וֹן אַל-תְּצֻרֵ֖ם וְאַל-תִּתְגָּ֣ר בָּ֑ם כִּ֣י לֹֽא-אֶ֠תֵּן מֵאֶ֨רֶץ בְּנֵֽי-עַמּ֤וֹן לְךָ֙ יְרֻשָּׁ֔ה כִּ֥י לִבְנֵי-ל֖וֹט נְתַתִּ֥יהָ יְרֻשָּֽׁה:  ת  וְתִקְרַב לָקֳבֵל בְּנֵי עַמוֹן לָא תְצוּר עֲלֵיהוֹן וְלָא תִתְגָרֵי לְמֶעְבַּד עִמְהוֹן קְרָב אֲרֵי לָא אֶתֵּן מֵאַרְעָא בְנֵי עַמוֹן לָךְ יְרֻתָּא אֲרֵי לִבְנֵי לוֹט יְהָבִיתַּהּ יְרֻתָּא:

{כ} אֶֽרֶץ-רְפָאִ֥ים תֵּֽחָשֵׁ֖ב אַף-הִ֑וא רְפָאִ֤ים יָֽשְׁבוּ-בָהּ֙ לְפָנִ֔ים וְהָ֣עַמֹּנִ֔ים יִקְרְא֥וּ לָהֶ֖ם זַמְזֻמִּֽים:  מ  אֶֽרֶץ-רְפָאִ֥ים תֵּֽחָשֵׁ֖ב אַף-הִ֑וא רְפָאִ֤ים יָֽשְׁבוּ-בָהּ֙ לְפָנִ֔ים וְהָ֣עַמֹּנִ֔ים יִקְרְא֥וּ לָהֶ֖ם זַמְזֻמִּֽים:  ת  אַרְעָא גִבָּרַיָא מִתְחַשְׁבָא אַף הִיא גִבָּרִין יָתִיבוּ בַהּ מִלְקַדְמִין וְעַמוֹנָאֵי קְרַן לְהוֹן חֻשְׁבָּנֵי:

{כא} עַ֣ם גָּד֥וֹל וְרַ֛ב וָרָ֖ם כַּֽעֲנָקִ֑ים וַיַּשְׁמִידֵ֤ם יְהֹוָה֙ מִפְּנֵיהֶ֔ם וַיִּֽירָשֻׁ֖ם וַיֵּֽשְׁב֥וּ תַחְתָּֽם:  מ  עַ֣ם גָּד֥וֹל וְרַ֛ב וָרָ֖ם כַּֽעֲנָקִ֑ים וַיַּשְׁמִידֵ֤ם יְהֹוָה֙ מִפְּנֵיהֶ֔ם וַיִּֽירָשֻׁ֖ם וַיֵּֽשְׁב֥וּ תַחְתָּֽם:  ת  עַם רַב וְסַגִי וְתַקִיף כְּגִבְּרַיָא וְשֵׁצִנוּן יְיָ מִקֳדָמֵ יהוֹן וְתָרֵכֻנוּן וִיתִיבוּ בְאַתְרֵיהוֹן:

{כב} כַּֽאֲשֶׁ֤ר עָשָׂה֙ לִבְנֵ֣י עֵשָׂ֔ו הַיֹּֽשְׁבִ֖ים בְּשֵׂעִ֑יר אֲשֶׁ֨ר הִשְׁמִ֤יד אֶת-הַֽחֹרִי֙ מִפְּנֵיהֶ֔ם וַיִּֽירָשֻׁם֙ וַיֵּֽשְׁב֣וּ תַחְתָּ֔ם עַ֖ד הַיּ֥וֹם הַזֶּֽה:  מ  כַּֽאֲשֶׁ֤ר עָשָׂה֙ לִבְנֵ֣י עֵשָׂ֔ו הַיֹּֽשְׁבִ֖ים בְּשֵׂעִ֑יר אֲשֶׁ֨ר הִשְׁמִ֤יד אֶת-הַֽחֹרִי֙ מִפְּנֵיהֶ֔ם וַיִּֽירָשֻׁם֙ וַיֵּֽשְׁב֣וּ תַחְתָּ֔ם עַ֖ד הַיּ֥וֹם הַזֶּֽה:  ת  כְּמָא דִי עֲבַד לִבְנֵי עֵשָׂו דְיָתְבוּן בְּשֵׂעִיר דִי שֵׁצֵי יָת חוֹרָאֵי מִקֳדָמֵיהוֹן וְתָרֵכִנוּן וִיתִיבוּ בְאַתְרֵיהוֹן עַד יוֹמָא הָדֵין:

{כג} וְהָֽעַוִּ֛ים הַיֹּֽשְׁבִ֥ים בַּֽחֲצֵרִ֖ים עַד-עַזָּ֑ה כַּפְתֹּרִים֙ הַיֹּֽצְאִ֣ים מִכַּפְתֹּ֔ר הִשְׁמִידֻ֖ם וַיֵּֽשְׁב֥וּ תַחְתָּֽם:  מ  וְהָֽעַוִּ֛ים הַיֹּֽשְׁבִ֥ים בַּֽחֲצֵרִ֖ים עַד-עַזָּ֑ה כַּפְתֹּרִים֙ הַיֹּֽצְאִ֣ים מִכַּפְתֹּ֔ר הִשְׁמִידֻ֖ם וַיֵּֽשְׁב֥וּ תַחְתָּֽם:  ת  וְעַוָאֵי דְיָתְבִין בִּדְפִיחַ עַד עַזָה קַפּוּטְּקָאֵי דִנְפָקוּ מִקַפּוּטְקַיָא שֵׁצָיוּנוּן וִיתִיבוּ בְאַתְרֵיהוֹן:

{כד} ק֣וּמוּ סְּע֗וּ וְעִבְרוּ֘ אֶת-נַ֣חַל אַרְנֹן֒ רְאֵ֣ה נָתַ֣תִּי בְ֠יָֽדְךָ אֶת-סִיחֹ֨ן מֶֽלֶךְ-חֶשְׁבּ֧וֹן הָֽאֱמֹרִ֛י וְאֶת-אַרְצ֖וֹ הָחֵ֣ל רָ֑שׁ וְהִתְגָּ֥ר בּ֖וֹ מִלְחָמָֽה:  מ  ק֣וּמוּ סְּע֗וּ וְעִבְרוּ֘ אֶת-נַ֣חַל אַרְנֹן֒ רְאֵ֣ה נָתַ֣תִּי בְ֠יָֽדְךָ אֶת-סִיחֹ֨ן מֶֽלֶךְ-חֶשְׁבּ֧וֹן הָֽאֱמֹרִ֛י וְאֶת-אַרְצ֖וֹ הָחֵ֣ל רָ֑שׁ וְהִתְגָּ֥ר בּ֖וֹ מִלְחָמָֽה:  ת  קוּמוּ טוּלוּ וְעִבָרוּ יָת נַחֲלָא דְאַרְנוֹן חֲזִי דִמְסָרִית בִּידָךְ יָת סִיחוֹן מַלְכָּא דְחֶשְׁבּוֹן אֱמוֹרָאָה וְיָת אַרְעֵהּ שְׁרֵי לְתָרָכוּתֵהּ וְאִתְגָרֵי לְמֶעְבַּד עִמֵהּ קְרָבָא:

{כה} הַיּ֣וֹם הַזֶּ֗ה אָחֵל֙ תֵּ֤ת פַּחְדְּךָ֙ וְיִרְאָ֣תְךָ֔ עַל-פְּנֵי֙ הָֽעַמִּ֔ים תַּ֖חַת כָּל-הַשָּׁמָ֑יִם אֲשֶׁ֤ר יִשְׁמְעוּן֙ שִׁמְעֲךָ֔ וְרָֽגְז֥וּ וְחָל֖וּ מִפָּנֶֽיךָ:  מ  הַיּ֣וֹם הַזֶּ֗ה אָחֵל֙ תֵּ֤ת פַּחְדְּךָ֙ וְיִרְאָ֣תְךָ֔ עַל-פְּנֵי֙ הָֽעַמִּ֔ים תַּ֖חַת כָּל-הַשָּׁמָ֑יִם אֲשֶׁ֤ר יִשְׁמְעוּן֙ שִׁמְעֲךָ֔ וְרָֽגְז֥וּ וְחָל֖וּ מִפָּנֶֽיךָ:  ת  יוֹמָא הָדֵין אֱשָׁרֵי לְמִתַּן זְעֻתָּךְ וְדַחַלְתָּךְ עַל אַפֵּי עַמְמַיָא דִי תְחוֹת כָּל שְׁמַיָא דִי יִשְׁמְעוּן שִׁמְעָךְ וִיזוּעוּן וְיִדְחֲלוּן מִן קֳדָמָךְ:

{כו} וָֽאֶשְׁלַ֤ח מַלְאָכִים֙ מִמִּדְבַּ֣ר קְדֵמ֔וֹת אֶל-סִיח֖וֹן מֶ֣לֶךְ חֶשְׁבּ֑וֹן דִּבְרֵ֥י שָׁל֖וֹם לֵאמֹֽר:  מ  וָֽאֶשְׁלַ֤ח מַלְאָכִים֙ מִמִּדְבַּ֣ר קְדֵמ֔וֹת אֶל-סִיח֖וֹן מֶ֣לֶךְ חֶשְׁבּ֑וֹן דִּבְרֵ֥י שָׁל֖וֹם לֵאמֹֽר:  ת  וּשְׁלָחִית אִזְגַדִין מִמַדְבְּרָא דִקְדֵמוֹת לְוָת סִיחוֹן מַלְכָּא דְחֶשְׁבּוֹן פִתְגָמֵי שְׁלָמָא לְמֵימָר:

{כז} אֶעְבְּרָ֣ה בְאַרְצֶ֔ךָ בַּדֶּ֥רֶךְ בַּדֶּ֖רֶךְ אֵלֵ֑ךְ לֹ֥א אָס֖וּר יָמִ֥ין וּשְׂמֹֽאול:  מ  אֶעְבְּרָ֣ה בְאַרְצֶ֔ךָ בַּדֶּ֥רֶךְ בַּדֶּ֖רֶךְ אֵלֵ֑ךְ לֹ֥א אָס֖וּר יָמִ֥ין וּשְׂמֹֽאול:  ת  אֵעִבַר בְּאַרְעָךְ בְּאָרְחָא בְּאָרְחָא אֵזֵל לָא אֶסְטֵי יַמִינָא וּשְׂמָאלָא:

{כח} אֹ֣כֶל בַּכֶּ֤סֶף תַּשְׁבִּרֵ֨נִי֙ וְאָכַ֔לְתִּי וּמַ֛יִם בַּכֶּ֥סֶף תִּתֶּן-לִ֖י וְשָׁתִ֑יתִי רַ֖ק אֶעְבְּרָ֥ה בְרַגְלָֽי:  מ  אֹ֣כֶל בַּכֶּ֤סֶף תַּשְׁבִּרֵ֨נִי֙ וְאָכַ֔לְתִּי וּמַ֛יִם בַּכֶּ֥סֶף תִּתֶּן-לִ֖י וְשָׁתִ֑יתִי רַ֖ק אֶעְבְּרָ֥ה בְרַגְלָֽי:  ת  עִבוּרָא בְּכַסְפָּא תְּזַבֶּן לִי וְאֵיכוּל וּמַיָא בְּכַסְפָּא תִּתֶּן לִי וְאֶשְׁתֵּי לְחוֹד אֵעִבַר בְּרַגְלָי:

{כט} כַּֽאֲשֶׁ֨ר עָֽשׂוּ-לִ֜י בְּנֵ֣י עֵשָׂ֗ו הַיֹּֽשְׁבִים֙ בְּשֵׂעִ֔יר וְהַמּ֣וֹאָבִ֔ים הַיֹּֽשְׁבִ֖ים בְּעָ֑ר עַ֤ד אֲשֶֽׁר-אֶֽעֱבֹר֙ אֶת-הַיַּרְדֵּ֔ן אֶל-הָאָ֕רֶץ אֲשֶׁר-יְהֹוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ:  מ  כַּֽאֲשֶׁ֨ר עָֽשׂוּ-לִ֜י בְּנֵ֣י עֵשָׂ֗ו הַיֹּֽשְׁבִים֙ בְּשֵׂעִ֔יר וְהַמּ֣וֹאָבִ֔ים הַיֹּֽשְׁבִ֖ים בְּעָ֑ר עַ֤ד אֲשֶֽׁר-אֶֽעֱבֹר֙ אֶת-הַיַּרְדֵּ֔ן אֶל-הָאָ֕רֶץ אֲשֶׁר-יְהֹוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ:  ת  כְּמָא דִי עֲבָדוּ לִי בְּנֵי עֵשָׂו דְיָתְבִין בְּשֵׂעִיר וּמוֹאָבָאֵי דְיָתְבִין בִּלְחָיָת עַד דְאֵעִבַר יָת יַרְדְנָא לְאַרְעָא דַיְיָ אֱלָהָנָא יָהֵב לָנָא:

{ל} וְלֹ֣א אָבָ֗ה סִיחֹן֙ מֶ֣לֶךְ חֶשְׁבּ֔וֹן הַֽעֲבִרֵ֖נוּ בּ֑וֹ כִּֽי-הִקְשָׁה֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ אֶת-רוּח֗וֹ וְאִמֵּץ֙ אֶת-לְבָב֔וֹ לְמַ֛עַן תִּתּ֥וֹ בְיָֽדְךָ֖ כַּיּ֥וֹם הַזֶּֽה:  מ  וְלֹ֣א אָבָ֗ה סִיחֹן֙ מֶ֣לֶךְ חֶשְׁבּ֔וֹן הַֽעֲבִרֵ֖נוּ בּ֑וֹ כִּֽי-הִקְשָׁה֩ יְהֹוָ֨ה אֱלֹהֶ֜יךָ אֶת-רוּח֗וֹ וְאִמֵּץ֙ אֶת-לְבָב֔וֹ לְמַ֛עַן תִּתּ֥וֹ בְיָֽדְךָ֖ כַּיּ֥וֹם הַזֶּֽה:  ת  וְלָא אָבֵי סִיחֹן מַלְכָּא דְחֶשְׁבּוֹן לְמִשְׁבְּקָנָא לְמֶעְבַּר בִּתְחוּמֵהּ אֲרֵי אַקְשִׁי יְיָ אֱלָהָךְ יָת רוּחֵהּ וְתַקִיף יָת לִבֵּהּ בְּדִיל לְמִמְסְרֵהּ בִּידָךְ כְּיוֹמָא הָדֵין: (ס)

{לא}  שישי  וַיֹּ֤אמֶר יְהֹוָה֙ אֵלַ֔י רְאֵ֗ה הַֽחִלֹּ֨תִי֙ תֵּ֣ת לְפָנֶ֔יךָ אֶת-סִיחֹ֖ן וְאֶת-אַרְצ֑וֹ הָחֵ֣ל רָ֔שׁ לָרֶ֖שֶׁת אֶת-אַרְצֽוֹ:  מ  וַיֹּ֤אמֶר יְהֹוָה֙ אֵלַ֔י רְאֵ֗ה הַֽחִלֹּ֨תִי֙ תֵּ֣ת לְפָנֶ֔יךָ אֶת-סִיחֹ֖ן וְאֶת-אַרְצ֑וֹ הָחֵ֣ל רָ֔שׁ לָרֶ֖שֶׁת אֶת-אַרְצֽוֹ:  ת  וַאֲמַר יְיָ לִי חֲזֵי שָׁרֵיתִי לְמִמְסַר קֳדָמָךְ יָת סִיחֹן וְיָת אַרְעֵהּ שְׁרֵי לְתָרָכוּתֵהּ לְמֵירַת יָת אַרְעֵהּ:

{לב} וַיֵּצֵא֩ סִיחֹ֨ן לִקְרָאתֵ֜נוּ ה֧וּא וְכָל-עַמּ֛וֹ לַמִּלְחָמָ֖ה יָֽהְצָה:  מ  וַיֵּצֵא֩ סִיחֹ֨ן לִקְרָאתֵ֜נוּ ה֧וּא וְכָל-עַמּ֛וֹ לַמִּלְחָמָ֖ה יָֽהְצָה:  ת  וּנְפַק סִיחֹן לְקַדָמוּתָנָא הוּא וְכָל עַמֵהּ לַאֲגָחָא קְרָבָא לְיָהַץ:

{לג} וַֽיִּתְּנֵ֛הוּ יְהֹוָ֥ה אֱלֹהֵ֖ינוּ לְפָנֵ֑ינוּ וַנַּ֥ךְ אֹת֛וֹ וְאֶת-(בנו) בָּנָ֖יו וְאֶת-כָּל-עַמּֽוֹ:  מ  וַֽיִּתְּנֵ֛הוּ יְהֹוָ֥ה אֱלֹהֵ֖ינוּ לְפָנֵ֑ינוּ וַנַּ֥ךְ אֹת֛וֹ וְאֶת-(בנו) בָּנָ֖יו וְאֶת-כָּל-עַמּֽוֹ:  ת  וּמָסְרֵהּ יְיָ אֱלָהָנָא קֳדָמָנָא וּמְחֵנָא יָתֵהּ וְיָת בְּנוֹהִי וְיָת כָּל עַמֵהּ:

{לד} וַנִּלְכֹּ֤ד אֶת-כָּל-עָרָיו֙ בָּעֵ֣ת הַהִ֔וא וַֽנַּֽחֲרֵם֙ אֶת-כָּל-עִ֣יר מְתִ֔ם וְהַנָּשִׁ֖ים וְהַטָּ֑ף לֹ֥א הִשְׁאַ֖רְנוּ שָׂרִֽיד:  מ  וַנִּלְכֹּ֤ד אֶת-כָּל-עָרָיו֙ בָּעֵ֣ת הַהִ֔וא וַֽנַּֽחֲרֵם֙ אֶת-כָּל-עִ֣יר מְתִ֔ם וְהַנָּשִׁ֖ים וְהַטָּ֑ף לֹ֥א הִשְׁאַ֖רְנוּ שָׂרִֽיד:  ת  וּכְבַשְׁנָא יָת כָּל קִרְווֹהִי בְּעִדָנָא הַהִיא וְגַמַרְנָא יָת כָּל קִרְוֵי גֻבְרַיָא וּנְשַׁיָא וְטַפְלָא לָא אַשְׁאַרְנָא מְשֵׁזִבִ:

{לה} רַ֥ק הַבְּהֵמָ֖ה בָּזַ֣זְנוּ לָ֑נוּ וּשְׁלַ֥ל הֶֽעָרִ֖ים אֲשֶׁ֥ר לָכָֽדְנוּ:  מ  רַ֥ק הַבְּהֵמָ֖ה בָּזַ֣זְנוּ לָ֑נוּ וּשְׁלַ֥ל הֶֽעָרִ֖ים אֲשֶׁ֥ר לָכָֽדְנוּ:  ת  לְחוֹד בְּעִירָא בַּזָנָא לָנָא וַעֲדִי קִרְוַיָא דִי כְבַשְׁנָא:

{לו} מֵֽעֲרֹעֵ֡ר אֲשֶׁר֩ עַל-שְׂפַת-נַ֨חַל אַרְנֹ֜ן וְהָעִ֨יר אֲשֶׁ֤ר בַּנַּ֨חַל֙ וְעַד-הַגִּלְעָ֔ד לֹ֤א הָֽיְתָה֙ קִרְיָ֔ה אֲשֶׁ֥ר שָֽׂגְבָ֖ה מִמֶּ֑נּוּ אֶת-הַכֹּ֕ל נָתַ֛ן יְהֹוָ֥ה אֱלֹהֵ֖ינוּ לְפָנֵֽינוּ:  מ  מֵֽעֲרֹעֵ֡ר אֲשֶׁר֩ עַל-שְׂפַת-נַ֨חַל אַרְנֹ֜ן וְהָעִ֨יר אֲשֶׁ֤ר בַּנַּ֨חַל֙ וְעַד-הַגִּלְעָ֔ד לֹ֤א הָֽיְתָה֙ קִרְיָ֔ה אֲשֶׁ֥ר שָֽׂגְבָ֖ה מִמֶּ֑נּוּ אֶת-הַכֹּ֕ל נָתַ֛ן יְהֹוָ֥ה אֱלֹהֵ֖ינוּ לְפָנֵֽינוּ:  ת  מֵעֲרֹעֵר דִי עַל כֵּיף נַחֲלָא דְאַרְנֹן וְקַרְתָּא דִי בְנַחֲלָא וְעַד גִלְעָד לָא הֲוַת קַרְתָּא דִי תְקֵפַת מִנָנָא יָת כֹּלָא מְסַר יְיָ אֱלָהָנָא קֳדָמָנָא:

{לז} רַ֛ק אֶל-אֶ֥רֶץ בְּנֵֽי-עַמּ֖וֹן לֹ֣א קָרָ֑בְתָּ כָּל-יַ֞ד נַ֤חַל יַבֹּק֙ וְעָרֵ֣י הָהָ֔ר וְכֹ֥ל אֲשֶׁר-צִוָּ֖ה יְהֹוָ֥ה אֱלֹהֵֽינוּ:  מ  רַ֛ק אֶל-אֶ֥רֶץ בְּנֵֽי-עַמּ֖וֹן לֹ֣א קָרָ֑בְתָּ כָּל-יַ֞ד נַ֤חַל יַבֹּק֙ וְעָרֵ֣י הָהָ֔ר וְכֹ֥ל אֲשֶׁר-צִוָּ֖ה יְהֹוָ֥ה אֱלֹהֵֽינוּ:  ת  לְחוֹד לְאַרְעָא בְנֵי עַמוֹן לָא קְרֶבְתָּא כָּל כֵּיף נַחֲלָא יוּבְקָא וְקִרְוֵי טוּרָא וְכֹל דִי פַקִיד יְיָ אֱלָהָנָא:




דברים פרק-ג

{א} וַנֵּ֣פֶן וַנַּ֔עַל דֶּ֖רֶךְ הַבָּשָׁ֑ן וַיֵּצֵ֣א עוֹג֩ מֶֽלֶךְ-הַבָּשָׁ֨ן לִקְרָאתֵ֜נוּ ה֧וּא וְכָל-עַמּ֛וֹ לַמִּלְחָמָ֖ה אֶדְרֶֽעִי:  מ  וַנֵּ֣פֶן וַנַּ֔עַל דֶּ֖רֶךְ הַבָּשָׁ֑ן וַיֵּצֵ֣א עוֹג֩ מֶֽלֶךְ-הַבָּשָׁ֨ן לִקְרָאתֵ֜נוּ ה֧וּא וְכָל-עַמּ֛וֹ לַמִּלְחָמָ֖ה אֶדְרֶֽעִי:  ת  וְאִתְפְּנֵינָא וּסְלֶקְנָא אֹרַח דְמַתְנָן וּנְפַק עוֹג מַלְכָּא דְמַתְנָן לְקַדָמוּתָנָא הוּא וְכָל עַמֵהּ לַאֲגָחָא קְרָבָא אֶדְרֶעִי:

{ב} וַיֹּ֨אמֶר יְהֹוָ֤ה אֵלַי֙ אַל-תִּירָ֣א אֹת֔וֹ כִּ֣י בְיָֽדְךָ֞ נָתַ֧תִּי אֹת֛וֹ וְאֶת-כָּל-עַמּ֖וֹ וְאֶת-אַרְצ֑וֹ וְעָשִׂ֣יתָ לּ֔וֹ כַּֽאֲשֶׁ֣ר עָשִׂ֗יתָ לְסִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּֽוֹן:  מ  וַיֹּ֨אמֶר יְהֹוָ֤ה אֵלַי֙ אַל-תִּירָ֣א אֹת֔וֹ כִּ֣י בְיָֽדְךָ֞ נָתַ֧תִּי אֹת֛וֹ וְאֶת-כָּל-עַמּ֖וֹ וְאֶת-אַרְצ֑וֹ וְעָשִׂ֣יתָ לּ֔וֹ כַּֽאֲשֶׁ֣ר עָשִׂ֗יתָ לְסִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּֽוֹן:  ת  וַאֲמַר יְיָ לִי לָא תִדְחַל מִנֵיהּ אֲרֵי בִידָךְ מְסָרִית יָתֵהּ וְיָת כָּל עַמֵהּ וְיָת אַרְעֵהּ ותַעְבֶּד לֵהּ כְּמָא דִי עֲבַדְתָּא לְסִיחֹן מַלְכָּא דֶאֱמֹרָאָה דִי יָתֵב בְּחֶשְׁבּוֹן:

{ג} וַיִּתֵּן֩ יְהֹוָ֨ה אֱלֹהֵ֜ינוּ בְּיָדֵ֗נוּ גַּ֛ם אֶת-ע֥וֹג מֶֽלֶךְ-הַבָּשָׁ֖ן וְאֶת-כָּל-עַמּ֑וֹ וַנַּכֵּ֕הוּ עַד-בִּלְתִּ֥י הִשְׁאִֽיר-ל֖וֹ שָׂרִֽיד:  מ  וַיִּתֵּן֩ יְהֹוָ֨ה אֱלֹהֵ֜ינוּ בְּיָדֵ֗נוּ גַּ֛ם אֶת-ע֥וֹג מֶֽלֶךְ-הַבָּשָׁ֖ן וְאֶת-כָּל-עַמּ֑וֹ וַנַּכֵּ֕הוּ עַד-בִּלְתִּ֥י הִשְׁאִֽיר-ל֖וֹ שָׂרִֽיד:  ת  וּמְסַר יְיָ אֱלָהָנָא בִּידָנָא אַף יָת עוֹג מַלְכָּא דְמַתְנָן וְיָת כֹּל עַמֵהּ וּמְחִנוֹהִי עַד דְלָא אִשְׁתָּאַר לֵהּ מְשֵׁזֵב:

{ד} וַנִּלְכֹּ֤ד אֶת-כָּל-עָרָיו֙ בָּעֵ֣ת הַהִ֔וא לֹ֤א הָֽיְתָה֙ קִרְיָ֔ה אֲשֶׁ֥ר לֹֽא-לָקַ֖חְנוּ מֵֽאִתָּ֑ם שִׁשִּׁ֥ים עִיר֙ כָּל-חֶ֣בֶל אַרְגֹּ֔ב מַמְלֶ֥כֶת ע֖וֹג בַּבָּשָֽׁן:  מ  וַנִּלְכֹּ֤ד אֶת-כָּל-עָרָיו֙ בָּעֵ֣ת הַהִ֔וא לֹ֤א הָֽיְתָה֙ קִרְיָ֔ה אֲשֶׁ֥ר לֹֽא-לָקַ֖חְנוּ מֵֽאִתָּ֑ם שִׁשִּׁ֥ים עִיר֙ כָּל-חֶ֣בֶל אַרְגֹּ֔ב מַמְלֶ֥כֶת ע֖וֹג בַּבָּשָֽׁן:  ת  וּכְבַשְׁנָא יָת כָּל קִרְווֹהִי בְּעִדָנָא הַהִיא לָא הֲוַת קַרְתָּא דִי לָא נְסִבְנָא מִנְהוֹן שִׁתִּין קִרְוִין כָּל בֵּית פָּלָךְ טְרָכוֹנָא מַלְכוּתָא דְעוֹג בְּמַתְנָן:

{ה} כָּל-אֵ֜לֶּה עָרִ֧ים בְּצֻרֹ֛ת חוֹמָ֥ה גְבֹהָ֖ה דְּלָתַ֣יִם וּבְרִ֑יחַ לְבַ֛ד מֵֽעָרֵ֥י הַפְּרָזִ֖י הַרְבֵּ֥ה מְאֹֽד:  מ  כָּל-אֵ֜לֶּה עָרִ֧ים בְּצֻרֹ֛ת חוֹמָ֥ה גְבֹהָ֖ה דְּלָתַ֣יִם וּבְרִ֑יחַ לְבַ֛ד מֵֽעָרֵ֥י הַפְּרָזִ֖י הַרְבֵּ֥ה מְאֹֽד:  ת  כָּל אִלֵין קִרְוִין כְּרִיכָן מַקְפָן שׁוּר רָם דִילְהֶן דָשִׁין וְעַבְּרִין בַּר מִקִרְוֵי פַּצְחַיָא סַגִי לַחֲדָא:

{ו} וַנַּֽחֲרֵ֣ם אוֹתָ֔ם כַּֽאֲשֶׁ֣ר עָשִׂ֔ינוּ לְסִיחֹ֖ן מֶ֣לֶךְ חֶשְׁבּ֑וֹן הַֽחֲרֵם֙ כָּל-עִ֣יר מְתִ֔ם הַנָּשִׁ֖ים וְהַטָּֽף:  מ  וַנַּֽחֲרֵ֣ם אוֹתָ֔ם כַּֽאֲשֶׁ֣ר עָשִׂ֔ינוּ לְסִיחֹ֖ן מֶ֣לֶךְ חֶשְׁבּ֑וֹן הַֽחֲרֵם֙ כָּל-עִ֣יר מְתִ֔ם הַנָּשִׁ֖ים וְהַטָּֽף:  ת  וְגַמַרְנָא יָתְהוֹן כְּמָא דִי עֲבַדְנָא לְסִיחוֹן מַלְכָּא דְחֶשְׁבּוֹן גַמַר כָּל קִרְוֵי גֻבְרַיָא נְשַׁיָא וְטַפְלָא:

{ז} וְכָל-הַבְּהֵמָ֛ה וּשְׁלַ֥ל הֶֽעָרִ֖ים בַּזּ֥וֹנוּ לָֽנוּ:  מ  וְכָל-הַבְּהֵמָ֛ה וּשְׁלַ֥ל הֶֽעָרִ֖ים בַּזּ֥וֹנוּ לָֽנוּ:  ת  וְכָל בְּעִירָא וַעֲדִי קִרְוַיָא בַּזָנָא לָנָא:

{ח} וַנִּקַּ֞ח בָּעֵ֤ת הַהִוא֙ אֶת-הָאָ֔רֶץ מִיַּ֗ד שְׁנֵי֙ מַלְכֵ֣י הָֽאֱמֹרִ֔י אֲשֶׁ֖ר בְּעֵ֣בֶר הַיַּרְדֵּ֑ן מִנַּ֥חַל אַרְנֹ֖ן עַד-הַ֥ר חֶרְמֽוֹן:  מ  וַנִּקַּ֞ח בָּעֵ֤ת הַהִוא֙ אֶת-הָאָ֔רֶץ מִיַּ֗ד שְׁנֵי֙ מַלְכֵ֣י הָֽאֱמֹרִ֔י אֲשֶׁ֖ר בְּעֵ֣בֶר הַיַּרְדֵּ֑ן מִנַּ֥חַל אַרְנֹ֖ן עַד-הַ֥ר חֶרְמֽוֹן:  ת  וּנְסִבְנָא בְּעִדָנָא הַהִיא יָת אַרְעָא מִיַד תְּרֵין מַלְכֵי אֱמֹרָאָה דִי בְּעִבְרָא דְיַרְדְנָא מִנַחֲלָא דְאַרְנֹן עַד טוּרָא דְחֶרְמוֹן:

{ט} צִֽידֹנִ֛ים יִקְרְא֥וּ לְחֶרְמ֖וֹן שִׂרְיֹ֑ן וְהָ֣אֱמֹרִ֔י יִקְרְאוּ-ל֖וֹ שְׂנִֽיר:  מ  צִֽידֹנִ֛ים יִקְרְא֥וּ לְחֶרְמ֖וֹן שִׂרְיֹ֑ן וְהָ֣אֱמֹרִ֔י יִקְרְאוּ-ל֖וֹ שְׂנִֽיר:  ת  צִידֹנָאֵי קְרָאָן לְחֶרְמוֹן סִרְיֹן וֶאֱמֹרָאֵי קְרָאָן לֵהּ טוּר תַּלְגָא:

{י} כֹּ֣ל | עָרֵ֣י הַמִּישֹׁ֗ר וְכָל-הַגִּלְעָד֙ וְכָל-הַבָּשָׁ֔ן עַד-סַֽלְכָ֖ה וְאֶדְרֶ֑עִי עָרֵ֛י מַמְלֶ֥כֶת ע֖וֹג בַּבָּשָֽׁן:  מ  כֹּ֣ל | עָרֵ֣י הַמִּישֹׁ֗ר וְכָל-הַגִּלְעָד֙ וְכָל-הַבָּשָׁ֔ן עַד-סַֽלְכָ֖ה וְאֶדְרֶ֑עִי עָרֵ֛י מַמְלֶ֥כֶת ע֖וֹג בַּבָּשָֽׁן:  ת  כֹּל קִרְוֵי מֵישְׁרָא וְכָל גִלְעָד וְכָל מַתְנָן עַד סַלְכָה וְאֶדְרֶעִי קִרְוֵי מַלְכוּתָא דְעוֹג בְּמַתְנָן:

{יא} כִּ֣י רַק-ע֞וֹג מֶ֣לֶךְ הַבָּשָׁ֗ן נִשְׁאַר֘ מִיֶּ֣תֶר הָֽרְפָאִים֒ הִנֵּ֤ה עַרְשׂוֹ֙ עֶ֣רֶשׂ בַּרְזֶ֔ל הֲלֹ֣ה הִ֔וא בְּרַבַּ֖ת בְּנֵ֣י עַמּ֑וֹן תֵּ֧שַׁע אַמּ֣וֹת אָרְכָּ֗הּ וְאַרְבַּ֥ע אַמּ֛וֹת רָחְבָּ֖הּ בְּאַמַּת-אִֽישׁ:  מ  כִּ֣י רַק-ע֞וֹג מֶ֣לֶךְ הַבָּשָׁ֗ן נִשְׁאַר֘ מִיֶּ֣תֶר הָֽרְפָאִים֒ הִנֵּ֤ה עַרְשׂוֹ֙ עֶ֣רֶשׂ בַּרְזֶ֔ל הֲלֹ֣ה הִ֔וא בְּרַבַּ֖ת בְּנֵ֣י עַמּ֑וֹן תֵּ֧שַׁע אַמּ֣וֹת אָרְכָּ֗הּ וְאַרְבַּ֥ע אַמּ֛וֹת רָחְבָּ֖הּ בְּאַמַּת-אִֽישׁ:  ת  אֲרֵי לְחוֹד עוֹג מַלְכָּא דְמַתְנָן אִשְׁתָּאַר מִשְׁאָר גִבָּרַיָא הָא עַרְסֵהּ עַרְסָא דְפַרְזְלָא הֲלָא הִיא בְּרַבַּת בְּנֵי עַמוֹן תְּשַׁע אַמִין אָרְכַּהּ וְאַרְבַּע אַמִין פְּתָיַהּ בְּאַמַת מְלַךְ:

{יב} וְאֶת-הָאָ֧רֶץ הַזֹּ֛את יָרַ֖שְׁנוּ בָּעֵ֣ת הַהִ֑וא מֵֽעֲרֹעֵ֞ר אֲשֶׁר-עַל-נַ֣חַל אַרְנֹ֗ן וַֽחֲצִ֤י הַר-הַגִּלְעָד֙ וְעָרָ֔יו נָתַ֕תִּי לָרֽאוּבֵנִ֖י וְלַגָּדִֽי:  מ  וְאֶת-הָאָ֧רֶץ הַזֹּ֛את יָרַ֖שְׁנוּ בָּעֵ֣ת הַהִ֑וא מֵֽעֲרֹעֵ֞ר אֲשֶׁר-עַל-נַ֣חַל אַרְנֹ֗ן וַֽחֲצִ֤י הַר-הַגִּלְעָד֙ וְעָרָ֔יו נָתַ֕תִּי לָרֽאוּבֵנִ֖י וְלַגָּדִֽי:  ת  וְיָת אַרְעָא הָדָא יְרִיתְנָא בְּעִדָנָא הַהִיא מֵעֲרֹעֵר דִי עַל נַחֲלָא דְאַרְנֹן וּפַלְגוּת טּוּרָא דְגִלְעָד וְקִרְווֹהִי יְהָבִית לְשִׁבְטָא דִרְאוּבֵן וּלְשִׁבְטָא דְגָד:

{יג} וְיֶ֨תֶר הַגִּלְעָ֤ד וְכָל-הַבָּשָׁן֙ מַמְלֶ֣כֶת ע֔וֹג נָתַ֕תִּי לַֽחֲצִ֖י שֵׁ֣בֶט הַֽמְנַשֶּׁ֑ה כֹּ֣ל חֶ֤בֶל הָֽאַרְגֹּב֙ לְכָל-הַבָּשָׁ֔ן הַה֥וּא יִקָּרֵ֖א אֶ֥רֶץ רְפָאִֽים:  מ  וְיֶ֨תֶר הַגִּלְעָ֤ד וְכָל-הַבָּשָׁן֙ מַמְלֶ֣כֶת ע֔וֹג נָתַ֕תִּי לַֽחֲצִ֖י שֵׁ֣בֶט הַֽמְנַשֶּׁ֑ה כֹּ֣ל חֶ֤בֶל הָֽאַרְגֹּב֙ לְכָל-הַבָּשָׁ֔ן הַה֥וּא יִקָּרֵ֖א אֶ֥רֶץ רְפָאִֽים:  ת  וּשְׁאָר גִלְעָד וְכָל מַתְנָן מַלְכוּתָא דְעוֹג יְהָבִית לְפַלְגוּת שִׁבְטָא דִמְנַשֶׁה כֹּל בֵּית פָּלַךְ טְרָכוֹנָא לְכָל מַתְנָן הַהוּא מִתְקְרֵי אַרְעָא גִבָּרַיָא:

{יד} יָאִ֣יר בֶּן-מְנַשֶּׁ֗ה לָקַח֙ אֶת-כָּל-חֶ֣בֶל אַרְגֹּ֔ב עַד-גְּב֥וּל הַגְּשׁוּרִ֖י וְהַמַּֽעֲכָתִ֑י וַיִּקְרָא֩ אֹתָ֨ם עַל-שְׁמ֤וֹ אֶת-הַבָּשָׁן֙ חַוֹּ֣ת יָאִ֔יר עַ֖ד הַיּ֥וֹם הַזֶּֽה:  מ  יָאִ֣יר בֶּן-מְנַשֶּׁ֗ה לָקַח֙ אֶת-כָּל-חֶ֣בֶל אַרְגֹּ֔ב עַד-גְּב֥וּל הַגְּשׁוּרִ֖י וְהַמַּֽעֲכָתִ֑י וַיִּקְרָא֩ אֹתָ֨ם עַל-שְׁמ֤וֹ אֶת-הַבָּשָׁן֙ חַוֹּ֣ת יָאִ֔יר עַ֖ד הַיּ֥וֹם הַזֶּֽה:  ת  יָאִיר בַּר מְנַשֶׁה נְסִיב יָת כָּל בֵּית פְּלַךְ טְרָכוֹנָא עַד תְּחוּם גְשׁוּרָאָה וְאַפְקֵירוֹס וּקְרָא יָתְהוֹן עַל שְׁמֵהּ יָת מַתְנָן כַּפְרָנֵי יָאִיר עַד יוֹמָא הָדֵין:

{טו}  שביעי  וּלְמָכִ֖יר נָתַ֥תִּי אֶת-הַגִּלְעָֽד:  מ  וּלְמָכִ֖יר נָתַ֥תִּי אֶת-הַגִּלְעָֽד:  ת  וּלְמָכִיר יְהָבִית יָת גִלְעָד:

{טז} וְלָרֽאוּבֵנִ֨י וְלַגָּדִ֜י נָתַ֤תִּי מִן-הַגִּלְעָד֙ וְעַד-נַ֣חַל אַרְנֹ֔ן תּ֥וֹךְ הַנַּ֖חַל וּגְבֻ֑ל וְעַד֙ יַבֹּ֣ק הַנַּ֔חַל גְּב֖וּל בְּנֵ֥י עַמּֽוֹן:  מ  וְלָרֽאוּבֵנִ֨י וְלַגָּדִ֜י נָתַ֤תִּי מִן-הַגִּלְעָד֙ וְעַד-נַ֣חַל אַרְנֹ֔ן תּ֥וֹךְ הַנַּ֖חַל וּגְבֻ֑ל וְעַד֙ יַבֹּ֣ק הַנַּ֔חַל גְּב֖וּל בְּנֵ֥י עַמּֽוֹן:  ת  וּלְשִׁבְטָא דִרְאוּבֵן וּלְשִׁבְטָּא דְגָד יְהָּבִית מִן גִלְעָד וְעַד נַחֲלָא דְאַרְנֹן גוֹ נַחֲלָא וּתְחוּם וְעַד יוּבְקָא דְנַחֲלָא תְּחוּם בְּנֵי עַמוֹן:

{יז} וְהָֽעֲרָבָ֖ה וְהַיַּרְדֵּ֣ן וּגְבֻ֑ל מִכִּנֶּ֗רֶת וְעַ֨ד יָ֤ם הָֽעֲרָבָה֙ יָ֣ם הַמֶּ֔לַח תַּ֛חַת אַשְׁדֹּ֥ת הַפִּסְגָּ֖ה מִזְרָֽחָה:  מ  וְהָֽעֲרָבָ֖ה וְהַיַּרְדֵּ֣ן וּגְבֻ֑ל מִכִּנֶּ֗רֶת וְעַ֨ד יָ֤ם הָֽעֲרָבָה֙ יָ֣ם הַמֶּ֔לַח תַּ֛חַת אַשְׁדֹּ֥ת הַפִּסְגָּ֖ה מִזְרָֽחָה:  ת  וּמֵישְׁרָא וְיַרְדְנָא וּתְחוּמֵהּ מִגֵנוֹסַר וְעַד יַמָא דְמֵישְׁרָא יַמָא דְמִלְחָא תְּחוֹת מַשְׁפַּךְ מְרָמָתָא מָדִינְחָא:

{יח} וָֽאֲצַ֣ו אֶתְכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר יְהֹוָ֣ה אֱלֹֽהֵיכֶ֗ם נָתַ֨ן לָכֶ֜ם אֶת-הָאָ֤רֶץ הַזֹּאת֙ לְרִשְׁתָּ֔הּ חֲלוּצִ֣ים תַּֽעַבְר֗וּ לִפְנֵ֛י אֲחֵיכֶ֥ם בְּנֵֽי-יִשְׂרָאֵ֖ל כָּל-בְּנֵי-חָֽיִל:  מ  וָֽאֲצַ֣ו אֶתְכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר יְהֹוָ֣ה אֱלֹֽהֵיכֶ֗ם נָתַ֨ן לָכֶ֜ם אֶת-הָאָ֤רֶץ הַזֹּאת֙ לְרִשְׁתָּ֔הּ חֲלוּצִ֣ים תַּֽעַבְר֗וּ לִפְנֵ֛י אֲחֵיכֶ֥ם בְּנֵֽי-יִשְׂרָאֵ֖ל כָּל-בְּנֵי-חָֽיִל:  ת  וּפַקֵדִית יָתְכוֹן בְּעִדָנָא הַהִיא לְמֵימָר יְיָ אֱלָהֲכוֹן יְהַב לְכוֹן יָת אַרְעָא הָדָא לְמֵירְתַהּ מְזָרְזִין תַּעְבְּרוּן קֳדָם אֲחֵיכוֹן בְּנֵי יִשְׂרָאֵל כָּל מְזָרְזֵי חֵילָא:

{יט} רַ֠ק נְשֵׁיכֶ֣ם וְטַפְּכֶם֘ וּמִקְנֵכֶם֒ יָדַ֕עְתִּי כִּֽי-מִקְנֶ֥ה רַ֖ב לָכֶ֑ם יֵֽשְׁבוּ֙ בְּעָ֣רֵיכֶ֔ם אֲשֶׁ֥ר נָתַ֖תִּי לָכֶֽם:  מ  רַ֠ק נְשֵׁיכֶ֣ם וְטַפְּכֶם֘ וּמִקְנֵכֶם֒ יָדַ֕עְתִּי כִּֽי-מִקְנֶ֥ה רַ֖ב לָכֶ֑ם יֵֽשְׁבוּ֙ בְּעָ֣רֵיכֶ֔ם אֲשֶׁ֥ר נָתַ֖תִּי לָכֶֽם:  ת  לְחוֹד נְשֵׁיכוֹן וְטַפְלְכוֹן וּבְעִירְכוֹן יְדַעְנָא אֲרֵי בְעִיר סַגִי לְכוֹן יֵתְבוּן בְּקִרְוֵיכוֹן דִי יְהָבִית לְכוֹן:
