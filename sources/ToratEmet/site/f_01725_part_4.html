פרשת שלח לך




במדבר פרק-יג

{א} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{ב} שְׁלַח-לְךָ֣ אֲנָשִׁ֗ים וְיָתֻ֨רוּ֙ אֶת-אֶ֣רֶץ כְּנַ֔עַן אֲשֶׁר-אֲנִ֥י נֹתֵ֖ן לִבְנֵ֣י יִשְׂרָאֵ֑ל אִ֣ישׁ אֶחָד֩ אִ֨ישׁ אֶחָ֜ד לְמַטֵּ֤ה אֲבֹתָיו֙ תִּשְׁלָ֔חוּ כֹּ֖ל נָשִׂ֥יא בָהֶֽם:  מ  שְׁלַח-לְךָ֣ אֲנָשִׁ֗ים וְיָתֻ֨רוּ֙ אֶת-אֶ֣רֶץ כְּנַ֔עַן אֲשֶׁר-אֲנִ֥י נֹתֵ֖ן לִבְנֵ֣י יִשְׂרָאֵ֑ל אִ֣ישׁ אֶחָד֩ אִ֨ישׁ אֶחָ֜ד לְמַטֵּ֤ה אֲבֹתָיו֙ תִּשְׁלָ֔חוּ כֹּ֖ל נָשִׂ֥יא בָהֶֽם:  ת  שְׁלַח לָךְ גֻבְרִין וִיאַלְלוּן יָת אַרְעָא דִכְנָעַן דִי אֲנִי יָהֵב לִבְנֵי יִשְׂרָאֵל גַבְרָא חַד גַבְרָא חַד לְשִׁבְטָא דַאֲבָהָתוֹהִי תְּשַׁלְחוּן כֹּל רַבָּא דִבְהוֹן:

{ג} וַיִּשְׁלַ֨ח אֹתָ֥ם מֹשֶׁ֛ה מִמִּדְבַּ֥ר פָּארָ֖ן עַל-פִּ֣י יְהֹוָ֑ה כֻּלָּ֣ם אֲנָשִׁ֔ים רָאשֵׁ֥י בְנֵֽי-יִשְׂרָאֵ֖ל הֵֽמָּה:  מ  וַיִּשְׁלַ֨ח אֹתָ֥ם מֹשֶׁ֛ה מִמִּדְבַּ֥ר פָּארָ֖ן עַל-פִּ֣י יְהֹוָ֑ה כֻּלָּ֣ם אֲנָשִׁ֔ים רָאשֵׁ֥י בְנֵֽי-יִשְׂרָאֵ֖ל הֵֽמָּה:  ת  וּשְׁלַח יָתְהוֹן משֶׁה מִמַדְבְּרָא דְפָארָן עַל מֵימְרָא דַיְיָ כֻּלְהוֹן גֻבְרִין רֵישֵׁי בְנֵי יִשְׂרָאֵל אִנוּן:

{ד} וְאֵ֖לֶּה שְׁמוֹתָ֑ם לְמַטֵּ֣ה רְאוּבֵ֔ן שַׁמּ֖וּעַ בֶּן-זַכּֽוּר:  מ  וְאֵ֖לֶּה שְׁמוֹתָ֑ם לְמַטֵּ֣ה רְאוּבֵ֔ן שַׁמּ֖וּעַ בֶּן-זַכּֽוּר:  ת  וְאִלֵין שְׁמָהַתְהוֹן לְשִׁבְטָא דִרְאוּבֵן שַׁמוּעַ בַּר זַכּוּר:

{ה} לְמַטֵּ֣ה שִׁמְע֔וֹן שָׁפָ֖ט בֶּן-חוֹרִֽי:  מ  לְמַטֵּ֣ה שִׁמְע֔וֹן שָׁפָ֖ט בֶּן-חוֹרִֽי:  ת  לְשִׁבְטָא דְשִׁמְעוֹן שָׁפָט בַּר חוֹרִי:

{ו} לְמַטֵּ֣ה יְהוּדָ֔ה כָּלֵ֖ב בֶּן-יְפֻנֶּֽה:  מ  לְמַטֵּ֣ה יְהוּדָ֔ה כָּלֵ֖ב בֶּן-יְפֻנֶּֽה:  ת  לְשִׁבְטָא דִיהוּדָה כָּלֵב בַּר יְפֻנֶה:

{ז} לְמַטֵּ֣ה יִשָּׂשכָ֔ר יִגְאָ֖ל בֶּן-יוֹסֵֽף:  מ  לְמַטֵּ֣ה יִשָּׂשכָ֔ר יִגְאָ֖ל בֶּן-יוֹסֵֽף:  ת  לְשִׁבְטָא דְיִשָׂשׂכָר יִגְאָל בַּר יוֹסֵף:

{ח} לְמַטֵּ֥ה אֶפְרָ֖יִם הוֹשֵׁ֥עַ בִּן-נֽוּן:  מ  לְמַטֵּ֥ה אֶפְרָ֖יִם הוֹשֵׁ֥עַ בִּן-נֽוּן:  ת  לְשִׁבְטָא דְאֶפְרַיִם הוֹשֵׁעַ בַּר נוּן:

{ט} לְמַטֵּ֣ה בִנְיָמִ֔ן פַּלְטִ֖י בֶּן-רָפֽוּא:  מ  לְמַטֵּ֣ה בִנְיָמִ֔ן פַּלְטִ֖י בֶּן-רָפֽוּא:  ת  לְשִׁבְטָא דְבִנְיָמִן פַּלְטִי בַּר רָפוּא:

{י} לְמַטֵּ֣ה זְבוּלֻ֔ן גַּדִּיאֵ֖ל בֶּן-סוֹדִֽי:  מ  לְמַטֵּ֣ה זְבוּלֻ֔ן גַּדִּיאֵ֖ל בֶּן-סוֹדִֽי:  ת  לְשִׁבְטָא דִזְבוּלֻן גַדִיאֵל בַּר סוֹדִי:

{יא} לְמַטֵּ֥ה יוֹסֵ֖ף לְמַטֵּ֣ה מְנַשֶּׁ֑ה גַּדִּ֖י בֶּן-סוּסִֽי:  מ  לְמַטֵּ֥ה יוֹסֵ֖ף לְמַטֵּ֣ה מְנַשֶּׁ֑ה גַּדִּ֖י בֶּן-סוּסִֽי:  ת  לְשִׁבְטָא דְיוֹסֵף לְשִׁבְטָא דִמְנַשֶׁה גַדִי בַּר סוּסִי:

{יב} לְמַטֵּ֣ה דָ֔ן עַמִּיאֵ֖ל בֶּן-גְּמַלִּֽי:  מ  לְמַטֵּ֣ה דָ֔ן עַמִּיאֵ֖ל בֶּן-גְּמַלִּֽי:  ת  לְשִׁבְטָא דְדָן עַמִיאֵל בַּר גְמַלִי:

{יג} לְמַטֵּ֣ה אָשֵׁ֔ר סְת֖וּר בֶּן-מִֽיכָאֵֽל:  מ  לְמַטֵּ֣ה אָשֵׁ֔ר סְת֖וּר בֶּן-מִֽיכָאֵֽל:  ת  לְשִׁבְטָא דְאָשֵׁר סְתוּר בַּר מִיכָאֵל:

{יד} לְמַטֵּ֣ה נַפְתָּלִ֔י נַחְבִּ֖י בֶּן-וָפְסִֽי:  מ  לְמַטֵּ֣ה נַפְתָּלִ֔י נַחְבִּ֖י בֶּן-וָפְסִֽי:  ת  לְשִׁבְטָא דְנַפְתָּלִי נַחְבִּי בַּר וָפְסִי:

{טו} לְמַטֵּ֣ה גָ֔ד גְּאוּאֵ֖ל בֶּן-מָכִֽי:  מ  לְמַטֵּ֣ה גָ֔ד גְּאוּאֵ֖ל בֶּן-מָכִֽי:  ת  לְשִׁבְטָא דְגָד גְאוּאֵל בַּר מָכִי:

{טז} אֵ֚לֶּה שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁר-שָׁלַ֥ח מֹשֶׁ֖ה לָת֣וּר אֶת-הָאָ֑רֶץ וַיִּקְרָ֥א מֹשֶׁ֛ה לְהוֹשֵׁ֥עַ בִּן-נ֖וּן יְהוֹשֻֽׁעַ:  מ  אֵ֚לֶּה שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁר-שָׁלַ֥ח מֹשֶׁ֖ה לָת֣וּר אֶת-הָאָ֑רֶץ וַיִּקְרָ֥א מֹשֶׁ֛ה לְהוֹשֵׁ֥עַ בִּן-נ֖וּן יְהוֹשֻֽׁעַ:  ת  אִלֵין שְׁמָהַת גֻבְרַיָא דִי שְׁלַח משֶׁה לְאַלָלָא יָת אַרְעָא וּקְרָא משֶׁה לְהוֹשֵׁעַ בַּר נוּן יְהוֹשֻׁעַ:

{יז} וַיִּשְׁלַ֤ח אֹתָם֙ מֹשֶׁ֔ה לָת֖וּר אֶת-אֶ֣רֶץ כְּנָ֑עַן וַיֹּ֣אמֶר אֲלֵהֶ֗ם עֲל֥וּ זֶה֙ בַּנֶּ֔גֶב וַֽעֲלִיתֶ֖ם אֶת-הָהָֽר:  מ  וַיִּשְׁלַ֤ח אֹתָם֙ מֹשֶׁ֔ה לָת֖וּר אֶת-אֶ֣רֶץ כְּנָ֑עַן וַיֹּ֣אמֶר אֲלֵהֶ֗ם עֲל֥וּ זֶה֙ בַּנֶּ֔גֶב וַֽעֲלִיתֶ֖ם אֶת-הָהָֽר:  ת  וּשְׁלַח יָתְהוֹן משֶׁה לְאַלָלָא יָת אַרְעָא דִכְנָעַן וַאֲמַר לְהוֹן סָקוּ דָא בִּדְרוֹמָא וְתִסְקוּן לְטוּרָא:

{יח} וּרְאִיתֶ֥ם אֶת-הָאָ֖רֶץ מַה-הִ֑וא וְאֶת-הָעָם֙ הַיֹּשֵׁ֣ב עָלֶ֔יהָ הֶֽחָזָ֥ק הוּא֙ הֲרָפֶ֔ה הַמְעַ֥ט ה֖וּא אִם-רָֽב:  מ  וּרְאִיתֶ֥ם אֶת-הָאָ֖רֶץ מַה-הִ֑וא וְאֶת-הָעָם֙ הַיֹּשֵׁ֣ב עָלֶ֔יהָ הֶֽחָזָ֥ק הוּא֙ הֲרָפֶ֔ה הַמְעַ֥ט ה֖וּא אִם-רָֽב:  ת  וְתֶחֱזוּן יָת אַרְעָא מָה הִיא וְיָת עַמָא דְיָתֵב עֲלַהּ הֲתַקִיף הוּא אִם חֲלָשׁ הַזְעֵר הוּא אִם סַגִי:

{יט} וּמָ֣ה הָאָ֗רֶץ אֲשֶׁר-הוּא֙ יֹשֵׁ֣ב בָּ֔הּ הֲטוֹבָ֥ה הִ֖וא אִם-רָעָ֑ה וּמָ֣ה הֶֽעָרִ֗ים אֲשֶׁר-הוּא֙ יוֹשֵׁ֣ב בָּהֵ֔נָּה הַבְּמַֽחֲנִ֖ים אִ֥ם בְּמִבְצָרִֽים:  מ  וּמָ֣ה הָאָ֗רֶץ אֲשֶׁר-הוּא֙ יֹשֵׁ֣ב בָּ֔הּ הֲטוֹבָ֥ה הִ֖וא אִם-רָעָ֑ה וּמָ֣ה הֶֽעָרִ֗ים אֲשֶׁר-הוּא֙ יוֹשֵׁ֣ב בָּהֵ֔נָּה הַבְּמַֽחֲנִ֖ים אִ֥ם בְּמִבְצָרִֽים:  ת  וּמָה אַרְעָא דִי הוּא יָתֵב בָּהּ הֲטָבָא הִיא אִם בִּישָׁא וּמָה קִרְוַיָא דִי הוּא יָתֵב בְהוֹן הַבְּפַצְחִין אִם בִּכְרַכִּין:

{כ} וּמָ֣ה הָ֠אָ֠רֶץ הַשְּׁמֵנָ֨ה הִ֜וא אִם-רָזָ֗ה הֲיֵֽשׁ-בָּ֥הּ עֵץ֙ אִם-אַ֔יִן וְהִ֨תְחַזַּקְתֶּ֔ם וּלְקַחְתֶּ֖ם מִפְּרִ֣י הָאָ֑רֶץ וְהַ֨יָּמִ֔ים יְמֵ֖י בִּכּוּרֵ֥י עֲנָבִֽים:  מ  וּמָ֣ה הָ֠אָ֠רֶץ הַשְּׁמֵנָ֨ה הִ֜וא אִם-רָזָ֗ה הֲיֵֽשׁ-בָּ֥הּ עֵץ֙ אִם-אַ֔יִן וְהִ֨תְחַזַּקְתֶּ֔ם וּלְקַחְתֶּ֖ם מִפְּרִ֣י הָאָ֑רֶץ וְהַ֨יָּמִ֔ים יְמֵ֖י בִּכּוּרֵ֥י עֲנָבִֽים:  ת  וּמָה אַרְעָא הֲעַתִּירָא הִיא אִם מִסְכֵּנָא הֲאִית בַּהּ אִילָנִין אִם לָא וְתִתְקְפוּן וְתִסְבוּן מֵאִבָּא דְאַרְעָא וְיוֹמַיָא יוֹמֵי בִּכּוּרֵי עִנְבִין:

{כא}  שני  וַֽיַּעֲל֖וּ וַיָּתֻ֣רוּ אֶת-הָאָ֑רֶץ מִמִּדְבַּר-צִ֥ן עַד-רְחֹ֖ב לְבֹ֥א חֲמָֽת:  מ  וַֽיַּעֲל֖וּ וַיָּתֻ֣רוּ אֶת-הָאָ֑רֶץ מִמִּדְבַּר-צִ֥ן עַד-רְחֹ֖ב לְבֹ֥א חֲמָֽת:  ת  וסְלִיקוּ וְאַלִילוּ יָת אַרְעָא מִמַדְבְּרָא דְצִין עַד רְחוֹב לִמְטֵי חֲמָת:

{כב} וַיַּֽעֲל֣וּ בַנֶּ֘גֶב֘ וַיָּבֹ֣א עַד-חֶבְרוֹן֒ וְשָׁ֤ם אֲחִימָן֙ שֵׁשַׁ֣י וְתַלְמַ֔י יְלִידֵ֖י הָֽעֲנָ֑ק וְחֶבְר֗וֹן שֶׁ֤בַע שָׁנִים֙ נִבְנְתָ֔ה לִפְנֵ֖י צֹ֥עַן מִצְרָֽיִם:  מ  וַיַּֽעֲל֣וּ בַנֶּ֘גֶב֘ וַיָּבֹ֣א עַד-חֶבְרוֹן֒ וְשָׁ֤ם אֲחִימָן֙ שֵׁשַׁ֣י וְתַלְמַ֔י יְלִידֵ֖י הָֽעֲנָ֑ק וְחֶבְר֗וֹן שֶׁ֤בַע שָׁנִים֙ נִבְנְתָ֔ה לִפְנֵ֖י צֹ֥עַן מִצְרָֽיִם:  ת  וּסְלִיקוּ בְדָרוֹמָא וַאֲתָא עַד חֶבְרוֹן וְתַמָן אֲחִימָן שֵׁשַׁי וְתַלְמַי בְּנֵי גִבָּרַיָא וְחֶבְרוֹן שְׁבַע שְׁנִין אִתְבְּנִיאַת קֳדָם טָנֵס דְמִצְרָיִם:

{כג} וַיָּבֹ֜אוּ עַד-נַ֣חַל אֶשְׁכֹּ֗ל וַיִּכְרְת֨וּ מִשָׁ֤ם זְמוֹרָה֙ וְאֶשְׁכּ֤וֹל עֲנָבִים֙ אֶחָ֔ד וַיִּשָּׂאֻ֥הוּ בַמּ֖וֹט בִּשְׁנָ֑יִם וּמִן-הָֽרִמֹּנִ֖ים וּמִן-הַתְּאֵנִֽים:  מ  וַיָּבֹ֜אוּ עַד-נַ֣חַל אֶשְׁכֹּ֗ל וַיִּכְרְת֨וּ מִשָׁ֤ם זְמוֹרָה֙ וְאֶשְׁכּ֤וֹל עֲנָבִים֙ אֶחָ֔ד וַיִּשָּׂאֻ֥הוּ בַמּ֖וֹט בִּשְׁנָ֑יִם וּמִן-הָֽרִמֹּנִ֖ים וּמִן-הַתְּאֵנִֽים:  ת  וַאֲתוֹ עַד נַחֲלָא דְאִתְכָּלָא וּקְצוֹ מִתַּמָן עוֹבַרְתָּא וְאַתְכַּל עִנְבִין חַד וּנְטָלוֹהִי בַאֲרִיחָא בִּתְרֵין וּמִן רִמוֹנַיָא וּמִן תְּאֵנַיָא:

{כד} לַמָּק֣וֹם הַה֔וּא קָרָ֖א נַ֣חַל אֶשְׁכּ֑וֹל עַ֚ל אֹד֣וֹת הָֽאֶשְׁכּ֔וֹל אֲשֶׁר-כָּֽרְת֥וּ מִשָּׁ֖ם בְּנֵ֥י יִשְׂרָאֵֽל:  מ  לַמָּק֣וֹם הַה֔וּא קָרָ֖א נַ֣חַל אֶשְׁכּ֑וֹל עַ֚ל אֹד֣וֹת הָֽאֶשְׁכּ֔וֹל אֲשֶׁר-כָּֽרְת֥וּ מִשָּׁ֖ם בְּנֵ֥י יִשְׂרָאֵֽל:  ת  לְאַתְרָא הַהוּא קְרָא נַחְלָא דְאִתְכָּלָא עַל עֵסַק אִתְכָּלָא דִקְצוֹ מִתַּמָן בְּנֵי יִשְׂרָאֵל:

{כה} וַיָּשֻׁ֖בוּ מִתּ֣וּר הָאָ֑רֶץ מִקֵּ֖ץ אַרְבָּעִ֥ים יֽוֹם:  מ  וַיָּשֻׁ֖בוּ מִתּ֣וּר הָאָ֑רֶץ מִקֵּ֖ץ אַרְבָּעִ֥ים יֽוֹם:  ת  וְתָבוּ מִלְאַלָלָא יָת אַרְעָא מִסוֹף אַרְבְּעִין יוֹמִין:

{כו} וַיֵּֽלְכ֡וּ וַיָּבֹ֩אוּ֩ אֶל-מֹשֶׁ֨ה וְאֶֽל-אַהֲרֹ֜ן וְאֶל-כָּל-עֲדַ֧ת בְּנֵֽי-יִשְׂרָאֵ֛ל אֶל-מִדְבַּ֥ר פָּארָ֖ן קָדֵ֑שָׁה וַיָּשִׁ֨יבוּ אֹתָ֤ם דָּבָר֙ וְאֶת-כָּל-הָ֣עֵדָ֔ה וַיַּרְא֖וּם אֶת-פְּרִ֥י הָאָֽרֶץ:  מ  וַיֵּֽלְכ֡וּ וַיָּבֹ֩אוּ֩ אֶל-מֹשֶׁ֨ה וְאֶֽל-אַהֲרֹ֜ן וְאֶל-כָּל-עֲדַ֧ת בְּנֵֽי-יִשְׂרָאֵ֛ל אֶל-מִדְבַּ֥ר פָּארָ֖ן קָדֵ֑שָׁה וַיָּשִׁ֨יבוּ אֹתָ֤ם דָּבָר֙ וְאֶת-כָּל-הָ֣עֵדָ֔ה וַיַּרְא֖וּם אֶת-פְּרִ֥י הָאָֽרֶץ:  ת  וַאֲזָלוּ וְאָתוּ לְוַת משֶׁה וּלְוַת אַהֲרֹן וּלְוַת כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל לְמַדְבְּרָא דְפָארָן לִרְקָם וַאֲתִיבוּ יָתְהוֹן פִּתְגָמָא וְיָת כָּל כְּנִשְׁתָּא וְאַחֲזֵיאֻנוּן יָת אִבָּא דְאַרְעָא:

{כז} וַיְסַפְּרוּ-לוֹ֙ וַיֹּ֣אמְר֔וּ בָּ֕אנוּ אֶל-הָאָ֖רֶץ אֲשֶׁ֣ר שְׁלַחְתָּ֑נוּ וְ֠גַ֠ם זָבַ֨ת חָלָ֥ב וּדְבַ֛שׁ הִ֖וא וְזֶה-פִּרְיָֽהּ:  מ  וַיְסַפְּרוּ-לוֹ֙ וַיֹּ֣אמְר֔וּ בָּ֕אנוּ אֶל-הָאָ֖רֶץ אֲשֶׁ֣ר שְׁלַחְתָּ֑נוּ וְ֠גַ֠ם זָבַ֨ת חָלָ֥ב וּדְבַ֛שׁ הִ֖וא וְזֶה-פִּרְיָֽהּ:  ת  וְאִשְׁתָּעִיאוּ לֵהּ וַאֲמָרוּ אֲתֵינָא לְאַרְעָא דִי שְׁלַחְתָּנָא וְאַף עַבְדָא חֲלַב וּדְבַשׁ הִיא וְדֵין אִבַּהּ:

{כח} אֶ֚פֶס כִּֽי-עַ֣ז הָעָ֔ם הַיֹּשֵׁ֖ב בָּאָ֑רֶץ וְהֶֽעָרִ֗ים בְּצֻ֤רוֹת גְּדֹלֹת֙ מְאֹ֔ד וְגַם-יְלִדֵ֥י הָֽעֲנָ֖ק רָאִ֥ינוּ שָֽׁם:  מ  אֶ֚פֶס כִּֽי-עַ֣ז הָעָ֔ם הַיֹּשֵׁ֖ב בָּאָ֑רֶץ וְהֶֽעָרִ֗ים בְּצֻ֤רוֹת גְּדֹלֹת֙ מְאֹ֔ד וְגַם-יְלִדֵ֥י הָֽעֲנָ֖ק רָאִ֥ינוּ שָֽׁם:  ת  לְחוֹד אֲרֵי תַקִיף עַמָא דְיָתֵב בְּאַרְעָא וְקִרְוַיָא כְּרִיכָן רַבְרְבִין לַחֲדָא וְאַף בְּנֵי גִבָּרַיָא חֲזֵינָא תַמָן:

{כט} עֲמָלֵ֥ק יוֹשֵׁ֖ב בְּאֶ֣רֶץ הַנֶּ֑גֶב וְ֠הַֽחִתִּ֠י וְהַיְבוּסִ֤י וְהָֽאֱמֹרִי֙ יוֹשֵׁ֣ב בָּהָ֔ר וְהַֽכְּנַעֲנִי֙ יוֹשֵׁ֣ב עַל-הַיָּ֔ם וְעַ֖ל יַ֥ד הַיַּרְדֵּֽן:  מ  עֲמָלֵ֥ק יוֹשֵׁ֖ב בְּאֶ֣רֶץ הַנֶּ֑גֶב וְ֠הַֽחִתִּ֠י וְהַיְבוּסִ֤י וְהָֽאֱמֹרִי֙ יוֹשֵׁ֣ב בָּהָ֔ר וְהַֽכְּנַעֲנִי֙ יוֹשֵׁ֣ב עַל-הַיָּ֔ם וְעַ֖ל יַ֥ד הַיַּרְדֵּֽן:  ת  עֲמַלְקָאָה יָתֵב בַּאֲרַע דָרוֹמָא וְחִתָּאָה וִיבוּסָאָה וֶאֱמוֹרָאָה יָתֵב בְּטוּרָא וּכְנַעֲנָאָה יָתֵב עַל יַמָא וְעַל כֵּיף יַרְדְנָא:

{ל} וַיַּ֧הַס כָּלֵ֛ב אֶת-הָעָ֖ם אֶל-מֹשֶׁ֑ה וַיֹּ֗אמֶר עָלֹ֤ה נַֽעֲלֶה֙ וְיָרַ֣שְׁנוּ אֹתָ֔הּ כִּֽי-יָכ֥וֹל נוּכַ֖ל לָֽהּ:  מ  וַיַּ֧הַס כָּלֵ֛ב אֶת-הָעָ֖ם אֶל-מֹשֶׁ֑ה וַיֹּ֗אמֶר עָלֹ֤ה נַֽעֲלֶה֙ וְיָרַ֣שְׁנוּ אֹתָ֔הּ כִּֽי-יָכ֥וֹל נוּכַ֖ל לָֽהּ:  ת  וַאֲצֵית כָּלֵב יָת עַמָא לְמשֶׁה וַאֲמַר מֵיסַק נִיסַק וְנִירַת יָתַהּ אֲרֵי מֵיכַל נֵיכוּל לָהּ:

{לא} וְהָ֨אֲנָשִׁ֜ים אֲשֶׁר-עָל֤וּ עִמּוֹ֙ אָֽמְר֔וּ לֹ֥א נוּכַ֖ל לַֽעֲל֣וֹת אֶל-הָעָ֑ם כִּֽי-חָזָ֥ק ה֖וּא מִמֶּֽנּוּ:  מ  וְהָ֨אֲנָשִׁ֜ים אֲשֶׁר-עָל֤וּ עִמּוֹ֙ אָֽמְר֔וּ לֹ֥א נוּכַ֖ל לַֽעֲל֣וֹת אֶל-הָעָ֑ם כִּֽי-חָזָ֥ק ה֖וּא מִמֶּֽנּוּ:  ת  וְגֻבְרַיָא דִי סְלִיקוּ עִמֵהּ אֲמָרוּ לָא נֵיכוֹל לְמֵיסַק לְוַת עַמָא אֲרֵי תַּקִיף הוּא מִנָנָא:

{לב} וַיֹּצִ֜יאוּ דִּבַּ֤ת הָאָ֨רֶץ֙ אֲשֶׁ֣ר תָּר֣וּ אֹתָ֔הּ אֶל-בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר הָאָ֡רֶץ אֲשֶׁר֩ עָבַ֨רְנוּ בָ֜הּ לָת֣וּר אֹתָ֗הּ אֶ֣רֶץ אֹכֶ֤לֶת יֽוֹשְׁבֶ֨יהָ֙ הִ֔וא וְכָל-הָעָ֛ם אֲשֶׁר-רָאִ֥ינוּ בְתוֹכָ֖הּ אַנְשֵׁ֥י מִדּֽוֹת:  מ  וַיֹּצִ֜יאוּ דִּבַּ֤ת הָאָ֨רֶץ֙ אֲשֶׁ֣ר תָּר֣וּ אֹתָ֔הּ אֶל-בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר הָאָ֡רֶץ אֲשֶׁר֩ עָבַ֨רְנוּ בָ֜הּ לָת֣וּר אֹתָ֗הּ אֶ֣רֶץ אֹכֶ֤לֶת יֽוֹשְׁבֶ֨יהָ֙ הִ֔וא וְכָל-הָעָ֛ם אֲשֶׁר-רָאִ֥ינוּ בְתוֹכָ֖הּ אַנְשֵׁ֥י מִדּֽוֹת:  ת  וְאַפִּיקוּ שׁוּם בִּישׁ עַל אַרְעָא דִי אַלִילוּ יָתָהּ לְוַת בְּנֵי יִשְׂרָאֵל לְמֵימָר אַרְעָא דִי עֲבַרְנָא בָהּ לְאַלָלָא יָתָהּ אַרְעָא מְקַטְלָא יַתְבָהָא הִיא וְכָל עַמָא דִי חֲזֵינָא בְגַוָהּ אֱנָשִׁין דְמִשְׁחָן:

{לג} וְשָׁ֣ם רָאִ֗ינוּ אֶת-הַנְּפִילִ֛ים בְּנֵ֥י עֲנָ֖ק מִן-הַנְּפִלִ֑ים וַנְּהִ֤י בְעֵינֵ֨ינוּ֙ כַּֽחֲגָבִ֔ים וְכֵ֥ן הָיִ֖ינוּ בְּעֵֽינֵיהֶֽם:  מ  וְשָׁ֣ם רָאִ֗ינוּ אֶת-הַנְּפִילִ֛ים בְּנֵ֥י עֲנָ֖ק מִן-הַנְּפִלִ֑ים וַנְּהִ֤י בְעֵינֵ֨ינוּ֙ כַּֽחֲגָבִ֔ים וְכֵ֥ן הָיִ֖ינוּ בְּעֵֽינֵיהֶֽם:  ת  וְתַמָן חֲזֵינָא יָת גִבָּרַיָא בְּנֵי עֲנָק מִן גִבָּרַיָא וַהֲוֵינָא בְעֵינֵי נַפְשָׁנָא כְּקַמְצִין וְכֵן הֲוֵינָא בְּעֵינֵיהוֹן:




במדבר פרק-יד

{א} וַתִּשָּׂא֙ כָּל-הָ֣עֵדָ֔ה וַֽיִּתְּנ֖וּ אֶת-קוֹלָ֑ם וַיִּבְכּ֥וּ הָעָ֖ם בַּלַּ֥יְלָה הַהֽוּא:  מ  וַתִּשָּׂא֙ כָּל-הָ֣עֵדָ֔ה וַֽיִּתְּנ֖וּ אֶת-קוֹלָ֑ם וַיִּבְכּ֥וּ הָעָ֖ם בַּלַּ֥יְלָה הַהֽוּא:  ת  וַאֲרֵימַת כָּל כְּנִשְׁתָּא וִיהָבוּ יָת קָלְהוֹן וּבְכוֹ עַמָא בְּלֵילְיָא הַהוּא:

{ב} וַיִּלֹּ֨נוּ֙ עַל-מֹשֶׁ֣ה וְעַֽל-אַהֲרֹ֔ן כֹּ֖ל בְּנֵ֣י יִשְׂרָאֵ֑ל וַיֹּֽאמְר֨וּ אֲלֵהֶ֜ם כָּל-הָֽעֵדָ֗ה לוּ-מַ֨תְנוּ֙ בְּאֶ֣רֶץ מִצְרַ֔יִם א֛וֹ בַּמִּדְבָּ֥ר הַזֶּ֖ה לוּ-מָֽתְנוּ:  מ  וַיִּלֹּ֨נוּ֙ עַל-מֹשֶׁ֣ה וְעַֽל-אַהֲרֹ֔ן כֹּ֖ל בְּנֵ֣י יִשְׂרָאֵ֑ל וַיֹּֽאמְר֨וּ אֲלֵהֶ֜ם כָּל-הָֽעֵדָ֗ה לוּ-מַ֨תְנוּ֙ בְּאֶ֣רֶץ מִצְרַ֔יִם א֛וֹ בַּמִּדְבָּ֥ר הַזֶּ֖ה לוּ-מָֽתְנוּ:  ת  וְאִתְרַעֲמוּ עַל משֶׁה וְעַל אַהֲרֹן כֹּל בְּנֵי יִשְׂרָאֵל וַאֲמָרוּ לְהוֹן כָּל כְּנִשְׁתָּא לְוַי דְמִיתְנָא בְּאַרְעָא דְמִצְרַיִם אוֹ בְּמַדְבְּרָא הָדֵין לְוַי דְמִיתְנָא:

{ג} וְלָמָ֣ה יְ֠הֹוָ֠ה מֵבִ֨יא אֹתָ֜נוּ אֶל-הָאָ֤רֶץ הַזֹּאת֙ לִנְפֹּ֣ל בַּחֶ֔רֶב נָשֵׁ֥ינוּ וְטַפֵּ֖נוּ יִֽהְי֣וּ לָבַ֑ז הֲל֧וֹא ט֦וֹב לָ֖נוּ שׁ֥וּב מִצְרָֽיְמָה:  מ  וְלָמָ֣ה יְ֠הֹוָ֠ה מֵבִ֨יא אֹתָ֜נוּ אֶל-הָאָ֤רֶץ הַזֹּאת֙ לִנְפֹּ֣ל בַּחֶ֔רֶב נָשֵׁ֥ינוּ וְטַפֵּ֖נוּ יִֽהְי֣וּ לָבַ֑ז הֲל֧וֹא ט֦וֹב לָ֖נוּ שׁ֥וּב מִצְרָֽיְמָה:  ת  וּלְמָא יְיָ מָעֵל יָתָנָא לְאַרְעָא הָדָא לְּמִנְפַּל בְּחַרְבָּא נְשָׁנָא וְטַפְלָנָא יְהוֹן לְבִזָא הֲלָא טַב לָנָא דִי נְתוּב לְמִצְרָיִם:

{ד} וַיֹּֽאמְר֖וּ אִ֣ישׁ אֶל-אָחִ֑יו נִתְּנָ֥ה רֹ֖אשׁ וְנָשׁ֥וּבָה מִצְרָֽיְמָה:  מ  וַיֹּֽאמְר֖וּ אִ֣ישׁ אֶל-אָחִ֑יו נִתְּנָ֥ה רֹ֖אשׁ וְנָשׁ֥וּבָה מִצְרָֽיְמָה:  ת  וַאֲמָרוּ גְבַר לַאֲחוֹהִי נְמַנֵי רֵישָׁא וּנְתוּב לְמִצְרָיִם:

{ה} וַיִּפֹּ֥ל מֹשֶׁ֛ה וְאַֽהֲרֹ֖ן עַל-פְּנֵיהֶ֑ם לִפְנֵ֕י כָּל-קְהַ֥ל עֲדַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וַיִּפֹּ֥ל מֹשֶׁ֛ה וְאַֽהֲרֹ֖ן עַל-פְּנֵיהֶ֑ם לִפְנֵ֕י כָּל-קְהַ֥ל עֲדַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וּנְפַל משֶׁה וְאַהֲרֹן עַל אַפֵּיהוֹן קֳדָם כָּל קְהַל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל:

{ו} וִֽיהוֹשֻׁ֣עַ בִּן-נ֗וּן וְכָלֵב֙ בֶּן-יְפֻנֶּ֔ה מִן-הַתָּרִ֖ים אֶת-הָאָ֑רֶץ קָֽרְע֖וּ בִּגְדֵיהֶֽם:  מ  וִֽיהוֹשֻׁ֣עַ בִּן-נ֗וּן וְכָלֵב֙ בֶּן-יְפֻנֶּ֔ה מִן-הַתָּרִ֖ים אֶת-הָאָ֑רֶץ קָֽרְע֖וּ בִּגְדֵיהֶֽם:  ת  וִיהוֹשֻׁעַ בַּר נוּן וְכָלֵב בַּר יְפֻנֶה מִן מְאַלְלֵי יָת אַרְעָא בְּזָעוּ לְבוּשֵׁיהוֹן:

{ז} וַיֹּ֣אמְר֔וּ אֶל-כָּל-עֲדַ֥ת בְּנֵֽי-יִשְׂרָאֵ֖ל לֵאמֹ֑ר הָאָ֗רֶץ אֲשֶׁ֨ר עָבַ֤רְנוּ בָהּ֙ לָת֣וּר אֹתָ֔הּ טוֹבָ֥ה הָאָ֖רֶץ מְאֹ֥ד מְאֹֽד:  מ  וַיֹּ֣אמְר֔וּ אֶל-כָּל-עֲדַ֥ת בְּנֵֽי-יִשְׂרָאֵ֖ל לֵאמֹ֑ר הָאָ֗רֶץ אֲשֶׁ֨ר עָבַ֤רְנוּ בָהּ֙ לָת֣וּר אֹתָ֔הּ טוֹבָ֥ה הָאָ֖רֶץ מְאֹ֥ד מְאֹֽד:  ת  וַאֲמָרוּ לְכָל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל לְמֵימָר אַרְעָא דִי עֲבַרְנָא בָהּ לְאַלָלָא יָתַהּ טָבָא אַרְעָא לַחֲדָא לַחֲדָא:

{ח}  שלישי  אִם-חָפֵ֥ץ בָּ֨נוּ֨ יְהֹוָ֔ה וְהֵבִ֤יא אֹתָ֨נוּ֙ אֶל-הָאָ֣רֶץ הַזֹּ֔את וּנְתָנָ֖הּ לָ֑נוּ אֶ֕רֶץ אֲשֶׁר-הִ֛וא זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ:  מ  אִם-חָפֵ֥ץ בָּ֨נוּ֨ יְהֹוָ֔ה וְהֵבִ֤יא אֹתָ֨נוּ֙ אֶל-הָאָ֣רֶץ הַזֹּ֔את וּנְתָנָ֖הּ לָ֑נוּ אֶ֕רֶץ אֲשֶׁר-הִ֛וא זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ:  ת  אִם רַעֲוָא בָנָא קֳדָם יְיָ וְיָעֵל יָתָנָא לְאַרְעָא הָדָא וְיִתְּנִנַהּ לָנָא אַרְעָא דִי הִיא עַבְדָא חֲלַב וּדְבָשׁ:

{ט} אַ֣ךְ בַּֽיהֹוָה֘ אַל-תִּמְרֹ֒דוּ֒ וְאַתֶּ֗ם אַל-תִּֽירְאוּ֙ אֶת-עַ֣ם הָאָ֔רֶץ כִּ֥י לַחְמֵ֖נוּ הֵ֑ם סָ֣ר צִלָּ֧ם מֵֽעֲלֵיהֶ֛ם וַֽיהֹוָ֥ה אִתָּ֖נוּ אַל-תִּֽירָאֻֽם:  מ  אַ֣ךְ בַּֽיהֹוָה֘ אַל-תִּמְרֹ֒דוּ֒ וְאַתֶּ֗ם אַל-תִּֽירְאוּ֙ אֶת-עַ֣ם הָאָ֔רֶץ כִּ֥י לַחְמֵ֖נוּ הֵ֑ם סָ֣ר צִלָּ֧ם מֵֽעֲלֵיהֶ֛ם וַֽיהֹוָ֥ה אִתָּ֖נוּ אַל-תִּֽירָאֻֽם:  ת  בְּרַם בְּמֵימְרָא דַיְיָ לָא תִמְרְדוּן וְאַתּוּן לָא תִדְחֲלוּן מִן עַמָא דְאַרְעָא אֲרֵי בִידָנָא מְסִירִין אִנוּן עֲדָא תָקְפְהוֹן מִנְהוֹן וּמֵימְרָא דַיְיָ בְּסַעְדָנָא לָא תִדְחֲלוּן מִנְהוֹן:

{י} וַיֹּֽאמְרוּ֙ כָּל-הָ֣עֵדָ֔ה לִרְגּ֥וֹם אֹתָ֖ם בָּֽאֲבָנִ֑ים וּכְב֣וֹד יְהֹוָ֗ה נִרְאָה֙ בְּאֹ֣הֶל מוֹעֵ֔ד אֶֽל-כָּל-בְּנֵ֖י יִשְׂרָאֵֽל:  מ  וַיֹּֽאמְרוּ֙ כָּל-הָ֣עֵדָ֔ה לִרְגּ֥וֹם אֹתָ֖ם בָּֽאֲבָנִ֑ים וּכְב֣וֹד יְהֹוָ֗ה נִרְאָה֙ בְּאֹ֣הֶל מוֹעֵ֔ד אֶֽל-כָּל-בְּנֵ֖י יִשְׂרָאֵֽל:  ת  וַאֲמָרוּ כָּל כְּנִשְׁתָּא לְמִרְגַם יַתְהוֹן בְּאַבְנַיָא וִיקָרָא דַיְיָ אִתְגְלֵי בְּמַשְׁכַּן זִמְנָא לְכָל בְּנֵי יִשְׂרָאֵל: (פ)

{יא} וַיֹּ֤אמֶר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה עַד-אָ֥נָה יְנַֽאֲצֻ֖נִי הָעָ֣ם הַזֶּ֑ה וְעַד-אָ֨נָה֙ לֹֽא-יַאֲמִ֣ינוּ בִ֔י בְּכֹל֙ הָֽאֹת֔וֹת אֲשֶׁ֥ר עָשִׂ֖יתִי בְּקִרְבּֽוֹ:  מ  וַיֹּ֤אמֶר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה עַד-אָ֥נָה יְנַֽאֲצֻ֖נִי הָעָ֣ם הַזֶּ֑ה וְעַד-אָ֨נָה֙ לֹֽא-יַאֲמִ֣ינוּ בִ֔י בְּכֹל֙ הָֽאֹת֔וֹת אֲשֶׁ֥ר עָשִׂ֖יתִי בְּקִרְבּֽוֹ:  ת  וַאֲמַר יְיָ לְמשֶׁה עַד אֵימָתַי יְהוֹן מְרַגְזִין קֳדָמַי עַמָא הָדֵין וְעַד אֵימָתַי לָא יְהֵימְנוּן בְּמֵימְרִי בְּכֹל אָתַיָא דִי עֲבָדִית בֵּינֵיהוֹן:

{יב} אַכֶּ֥נּוּ בַדֶּ֖בֶר וְאֽוֹרִשֶׁ֑נּוּ וְאֶֽעֱשֶׂה֙ אֹֽתְךָ֔ לְגוֹי-גָּ֥דוֹל וְעָצ֖וּם מִמֶּֽנּוּ:  מ  אַכֶּ֥נּוּ בַדֶּ֖בֶר וְאֽוֹרִשֶׁ֑נּוּ וְאֶֽעֱשֶׂה֙ אֹֽתְךָ֔ לְגוֹי-גָּ֥דוֹל וְעָצ֖וּם מִמֶּֽנּוּ:  ת  אֶמְחִנוּן בְּמוֹתָא וַאֲשֵׁצִנוּן וְאֶעְבֵּד יָתָךְ לְעַם רַב וְתַקִיף מִנְהוֹן:

{יג} וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל-יְהֹוָ֑ה וְשָֽׁמְע֣וּ מִצְרַ֔יִם כִּֽי-הֶעֱלִ֧יתָ בְכֹֽחֲךָ֛ אֶת-הָעָ֥ם הַזֶּ֖ה מִקִּרְבּֽוֹ:  מ  וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל-יְהֹוָ֑ה וְשָֽׁמְע֣וּ מִצְרַ֔יִם כִּֽי-הֶעֱלִ֧יתָ בְכֹֽחֲךָ֛ אֶת-הָעָ֥ם הַזֶּ֖ה מִקִּרְבּֽוֹ:  ת  וַאֲמַר משֶׁה קֳדָם יְיָ וְיִשְׁמְעוּן מִצְרָאֵי אֲרֵי אַסֶקְתָּא בְחֵילָךְ יָת עַמָא הָדֵין מִבֵּינֵיהוֹן:

{יד} וְאָֽמְר֗וּ אֶל-יוֹשֵׁב֘ הָאָ֣רֶץ הַזֹּאת֒ שָֽׁמְעוּ֙ כִּֽי-אַתָּ֣ה יְהֹוָ֔ה בְּקֶ֖רֶב הָעָ֣ם הַזֶּ֑ה אֲשֶׁר-עַ֨יִן בְּעַ֜יִן נִרְאָ֣ה | אַתָּ֣ה יְהֹוָ֗ה וַעֲנָֽנְךָ֙ עֹמֵ֣ד עֲלֵהֶ֔ם וּבְעַמֻּ֣ד עָנָ֗ן אַתָּ֨ה הֹלֵ֤ךְ לִפְנֵיהֶם֙ יוֹמָ֔ם וּבְעַמּ֥וּד אֵ֖שׁ לָֽיְלָה:  מ  וְאָֽמְר֗וּ אֶל-יוֹשֵׁב֘ הָאָ֣רֶץ הַזֹּאת֒ שָֽׁמְעוּ֙ כִּֽי-אַתָּ֣ה יְהֹוָ֔ה בְּקֶ֖רֶב הָעָ֣ם הַזֶּ֑ה אֲשֶׁר-עַ֨יִן בְּעַ֜יִן נִרְאָ֣ה | אַתָּ֣ה יְהֹוָ֗ה וַעֲנָֽנְךָ֙ עֹמֵ֣ד עֲלֵהֶ֔ם וּבְעַמֻּ֣ד עָנָ֗ן אַתָּ֨ה הֹלֵ֤ךְ לִפְנֵיהֶם֙ יוֹמָ֔ם וּבְעַמּ֥וּד אֵ֖שׁ לָֽיְלָה:  ת  וְיֵימְרוּן לְיָתֵב אַרְעָא הָדָא דִשְׁמָעוּ אֲרֵי אַתְּ יְיָ דִשְׁכִנְתָּךְ שַׁרְיַת בְּגוֹ עַמָא הָדֵין דִי בְעֵינֵיהוֹן חָזַן שְׁכִינַת יְקָרָא דַיְיָ וַעֲנָנָךְ מַטֵל עִלַוֵהוֹן וּבְעַמוּדָא דַעֲנָנָא אַתְּ מְדַבַּר קֳדָמֵיהוֹן בִּימָמָא וּבְעַמוּדָא דְאֵשָׁתָא בְּלֵילְיָא:

{טו} וְהֵֽמַתָּ֛ה אֶת-הָעָ֥ם הַזֶּ֖ה כְּאִ֣ישׁ אֶחָ֑ד וְאָֽמְרוּ֙ הַגּוֹיִ֔ם אֲשֶׁר-שָֽׁמְע֥וּ אֶֽת-שִׁמְעֲךָ֖ לֵאמֹֽר:  מ  וְהֵֽמַתָּ֛ה אֶת-הָעָ֥ם הַזֶּ֖ה כְּאִ֣ישׁ אֶחָ֑ד וְאָֽמְרוּ֙ הַגּוֹיִ֔ם אֲשֶׁר-שָֽׁמְע֥וּ אֶֽת-שִׁמְעֲךָ֖ לֵאמֹֽר:  ת  וּתְקַטֵּל יָת עַמָא הָדֵין כְּגַבְרָא חָד וְיֵימְרוּן עַמְמַיָא דִי שְׁמָעוּ יָת שְׁמַע גְבֻרְתָּךְ לְמֵימָר:

{טז} מִבִּלְתִּ֞י יְכֹ֣לֶת יְהֹוָ֗ה לְהָבִיא֙ אֶת-הָעָ֣ם הַזֶּ֔ה אֶל-הָאָ֖רֶץ אֲשֶׁר-נִשְׁבַּ֣ע לָהֶ֑ם וַיִּשְׁחָטֵ֖ם בַּמִּדְבָּֽר:  מ  מִבִּלְתִּ֞י יְכֹ֣לֶת יְהֹוָ֗ה לְהָבִיא֙ אֶת-הָעָ֣ם הַזֶּ֔ה אֶל-הָאָ֖רֶץ אֲשֶׁר-נִשְׁבַּ֣ע לָהֶ֑ם וַיִּשְׁחָטֵ֖ם בַּמִּדְבָּֽר:  ת  מִדְלֵית יוּכְלָא קֳדָם יְיָ לְאָעָלָא יָת עַמָא הָדֵין לְאַרְעָא דִי קַיֵם לְהוֹן וְקַטְלִנוּן בְּמַדְבְּרָא:

{יז} וְעַתָּ֕ה יִגְדַּל-נָ֖א כֹּ֣חַ אֲדֹנָ֑י כַּֽאֲשֶׁ֥ר דִּבַּ֖רְתָּ לֵאמֹֽר:  מ  וְעַתָּ֕ה יִגְדַּל-נָ֖א כֹּ֣חַ אֲדֹנָ֑י כַּֽאֲשֶׁ֥ר דִּבַּ֖רְתָּ לֵאמֹֽר:  ת  וּכְעַן סַגִי כְעַן חֵילָא (מִן) קֳדָם יְיָ כְּמָא דִי מַלֵילְתָּא לְמֵימָר:

{יח} יְהֹוָ֗ה אֶ֤רֶךְ אַפַּ֨יִם֙ וְרַב-חֶ֔סֶד נֹשֵׂ֥א עָוֹ֖ן וָפָ֑שַׁע וְנַקֵּה֙ לֹ֣א יְנַקֶּ֔ה פֹּקֵ֞ד עֲוֹ֤ן אָבוֹת֙ עַל-בָּנִ֔ים עַל-שִׁלֵּשִׁ֖ים וְעַל-רִבֵּעִֽים:  מ  יְהֹוָ֗ה אֶ֤רֶךְ אַפַּ֨יִם֙ וְרַב-חֶ֔סֶד נֹשֵׂ֥א עָוֹ֖ן וָפָ֑שַׁע וְנַקֵּה֙ לֹ֣א יְנַקֶּ֔ה פֹּקֵ֞ד עֲוֹ֤ן אָבוֹת֙ עַל-בָּנִ֔ים עַל-שִׁלֵּשִׁ֖ים וְעַל-רִבֵּעִֽים:  ת  יְיָ מַרְחֵק רְגַז וּמַסְגֵי לְמֶעְבַּד טַבְוָן (וּקְשׁוֹט) שָׁבֵק לַעֲוָיָן וְלִמְרוֹד וּלְחוֹבִין סַלַח לִדְתַיְיבִין לְאוֹרַיְתֵהּ וּדְלָא תַיְיבִין לָא מְזַכֵּי מַסְעַר חוֹבֵי אֲבָהָן עַל בְּנִין מְרָדִין עַל דַר תְּלִיתָאֵי וְעַל דַר רְבִיעָאֵי:

{יט} סְלַֽח-נָ֗א לַֽעֲוֹ֛ן הָעָ֥ם הַזֶּ֖ה כְּגֹ֣דֶל חַסְדֶּ֑ךָ וְכַֽאֲשֶׁ֤ר נָשָׂ֨אתָה֙ לָעָ֣ם הַזֶּ֔ה מִמִּצְרַ֖יִם וְעַד-הֵֽנָּה:  מ  סְלַֽח-נָ֗א לַֽעֲוֹ֛ן הָעָ֥ם הַזֶּ֖ה כְּגֹ֣דֶל חַסְדֶּ֑ךָ וְכַֽאֲשֶׁ֤ר נָשָׂ֨אתָה֙ לָעָ֣ם הַזֶּ֔ה מִמִּצְרַ֖יִם וְעַד-הֵֽנָּה:  ת  שְׁבוֹק כְּעַן לְחוֹבֵי עַמָא הָדֵין כִּסְגִיאוּת טַבְוָתָךְ וּכְמָא דִי שְׁבַקְתָּא לְעַמָא הָדֵין מִמִצְרַיִם וְעַד כְּעָן:

{כ} וַיֹּ֣אמֶר יְהֹוָ֔ה סָלַ֖חְתִּי כִּדְבָרֶֽךָ:  מ  וַיֹּ֣אמֶר יְהֹוָ֔ה סָלַ֖חְתִּי כִּדְבָרֶֽךָ:  ת  וַאֲמַר יְיָ שְׁבָקִית כְּפִתְגָמָךְ:

{כא} וְאוּלָ֖ם חַי-אָ֑נִי וְיִמָּלֵ֥א כְבוֹד-יְהֹוָ֖ה אֶת-כָּל-הָאָֽרֶץ:  מ  וְאוּלָ֖ם חַי-אָ֑נִי וְיִמָּלֵ֥א כְבוֹד-יְהֹוָ֖ה אֶת-כָּל-הָאָֽרֶץ:  ת  וּבְרַם קַיָים אֲנָא וּמַלְיָא יְקָרָא דַיְיָ יָת כָּל אַרְעָא:

{כב} כִּ֣י כָל-הָֽאֲנָשִׁ֗ים הָֽרֹאִ֤ים אֶת-כְּבֹדִי֙ וְאֶת-אֹ֣תֹתַ֔י אֲשֶׁר-עָשִׂ֥יתִי בְמִצְרַ֖יִם וּבַמִּדְבָּ֑ר וַיְנַסּ֣וּ אֹתִ֗י זֶ֚ה עֶ֣שֶׂר פְּעָמִ֔ים וְלֹ֥א שָֽׁמְע֖וּ בְּקוֹלִֽי:  מ  כִּ֣י כָל-הָֽאֲנָשִׁ֗ים הָֽרֹאִ֤ים אֶת-כְּבֹדִי֙ וְאֶת-אֹ֣תֹתַ֔י אֲשֶׁר-עָשִׂ֥יתִי בְמִצְרַ֖יִם וּבַמִּדְבָּ֑ר וַיְנַסּ֣וּ אֹתִ֗י זֶ֚ה עֶ֣שֶׂר פְּעָמִ֔ים וְלֹ֥א שָֽׁמְע֖וּ בְּקוֹלִֽי:  ת  אֲרֵי כָל גֻבְרַיָא דַחֲזוֹ יָת יְקָרִי וְיָת אַתְוָתַי דִי עֲבָדִית בְּמִצְרַיִם וּבְמַדְבְּרָא וְנַסִיאוּ קֳדָמַי דְנַן עֲשַׂר זִמְנִין וְלָא קַבִּילוּ בְּמֵימְרִי:

{כג} אִם-יִרְאוּ֙ אֶת-הָאָ֔רֶץ אֲשֶׁ֥ר נִשְׁבַּ֖עְתִּי לַֽאֲבֹתָ֑ם וְכָל-מְנַֽאֲצַ֖י לֹ֥א יִרְאֽוּהָ:  מ  אִם-יִרְאוּ֙ אֶת-הָאָ֔רֶץ אֲשֶׁ֥ר נִשְׁבַּ֖עְתִּי לַֽאֲבֹתָ֑ם וְכָל-מְנַֽאֲצַ֖י לֹ֥א יִרְאֽוּהָ:  ת  אִם יֶחֶזוּן יָת אַרְעָא דִי קַיֵמִית לַאֲבָהַתְהוֹן וְכָל דְאַרְגִיזוּ קֳדָמַי לָא יֶחֱזִנַהּ:

{כד} וְעַבְדִּ֣י כָלֵ֗ב עֵ֣קֶב הָֽיְתָ֞ה ר֤וּחַ אַחֶ֨רֶת֙ עִמּ֔וֹ וַיְמַלֵּ֖א אַֽחֲרָ֑י וַהֲבִֽיאֹתִ֗יו אֶל-הָאָ֨רֶץ֙ אֲשֶׁר-בָּ֣א שָׁ֔מָּה וְזַרְע֖וֹ יֽוֹרִשֶֽׁנָּה:  מ  וְעַבְדִּ֣י כָלֵ֗ב עֵ֣קֶב הָֽיְתָ֞ה ר֤וּחַ אַחֶ֨רֶת֙ עִמּ֔וֹ וַיְמַלֵּ֖א אַֽחֲרָ֑י וַהֲבִֽיאֹתִ֗יו אֶל-הָאָ֨רֶץ֙ אֲשֶׁר-בָּ֣א שָׁ֔מָּה וְזַרְע֖וֹ יֽוֹרִשֶֽׁנָּה:  ת  וְעַבְדִי כָלֵב חֲלַף דַהֲוָה רוּחַ אוֹחֲרִי עִמֵהּ וְאַשְׁלֵם בָּתַר דַחַלְתִּי וְאַעֵלִנֵהּ לְאַרְעָא דִי עַאל לְתַמָן וּבְנוֹהִי יְתַרְכִנַהּ:

{כה} וְהָֽעֲמָלֵקִ֥י וְהַֽכְּנַעֲנִ֖י יוֹשֵׁ֣ב בָּעֵ֑מֶק מָחָ֗ר פְּנ֨וּ וּסְע֥וּ לָכֶ֛ם הַמִּדְבָּ֖ר דֶּ֥רֶךְ יַם-סֽוּף:  מ  וְהָֽעֲמָלֵקִ֥י וְהַֽכְּנַעֲנִ֖י יוֹשֵׁ֣ב בָּעֵ֑מֶק מָחָ֗ר פְּנ֨וּ וּסְע֥וּ לָכֶ֛ם הַמִּדְבָּ֖ר דֶּ֥רֶךְ יַם-סֽוּף:  ת  וַעֲמַלְקָאָה וּכְנַעֲנָאָה יָתֵב בְּמֵישְׁרָא מְחַר אִתְפְּנוּ וְטוּלוּ לְכוֹן לְמַדְבְּרָא אוֹרַח יַמָא דְסוּף: (פ)

{כו}  רביעי  וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  מ  וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל-מֹשֶׁ֥ה וְאֶֽל-אַהֲרֹ֖ן לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה וְעִם אַהֲרֹן לְמֵימָר:

{כז} עַד-מָתַ֗י לָֽעֵדָ֤ה הָֽרָעָה֙ הַזֹּ֔את אֲשֶׁ֛ר הֵ֥מָּה מַלִּינִ֖ים עָלָ֑י אֶת-תְּלֻנּ֞וֹת בְּנֵ֣י יִשְׂרָאֵ֗ל אֲשֶׁ֨ר הֵ֧מָּה מַלִּינִ֛ים עָלַ֖י שָׁמָֽעְתִּי:  מ  עַד-מָתַ֗י לָֽעֵדָ֤ה הָֽרָעָה֙ הַזֹּ֔את אֲשֶׁ֛ר הֵ֥מָּה מַלִּינִ֖ים עָלָ֑י אֶת-תְּלֻנּ֞וֹת בְּנֵ֣י יִשְׂרָאֵ֗ל אֲשֶׁ֨ר הֵ֧מָּה מַלִּינִ֛ים עָלַ֖י שָׁמָֽעְתִּי:  ת  עַד אֵימָתַי לִכְנִשְׁתָּא בִישָׁתָא הָדָא דִי אִנוּן מִתְרַעֲמִין עָלָי יָת תֻּרְעֲמַת בְּנֵי יִשְׂרָאֵל דִי אִנוּן מִתְרַעֲמִין עֲלַי שְׁמִיעַ קֳדָמָי:

{כח} אֱמֹ֣ר אֲלֵהֶ֗ם חַי-אָ֨נִי֙ נְאֻם-יְהֹוָ֔ה אִם-לֹ֕א כַּֽאֲשֶׁ֥ר דִּבַּרְתֶּ֖ם בְּאָזְנָ֑י כֵּ֖ן אֶֽעֱשֶׂ֥ה לָכֶֽם:  מ  אֱמֹ֣ר אֲלֵהֶ֗ם חַי-אָ֨נִי֙ נְאֻם-יְהֹוָ֔ה אִם-לֹ֕א כַּֽאֲשֶׁ֥ר דִּבַּרְתֶּ֖ם בְּאָזְנָ֑י כֵּ֖ן אֶֽעֱשֶׂ֥ה לָכֶֽם:  ת  אֱמַר לְהוֹן קַיָם אֲנָא אֲמַר יְיָ אִם לָא כְּמָא דִי מַלֵילְתּוּן קֳדָמָי כֵּן אֶעְבֵּד לְכוֹן:

{כט} בַּמִּדְבָּ֣ר הַ֠זֶּ֠ה יִפְּל֨וּ פִגְרֵיכֶ֜ם וְכָל-פְּקֻֽדֵיכֶם֙ לְכָל-מִסְפַּרְכֶ֔ם מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה אֲשֶׁ֥ר הֲלִֽינֹתֶ֖ם עָלָֽי:  מ  בַּמִּדְבָּ֣ר הַ֠זֶּ֠ה יִפְּל֨וּ פִגְרֵיכֶ֜ם וְכָל-פְּקֻֽדֵיכֶם֙ לְכָל-מִסְפַּרְכֶ֔ם מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה אֲשֶׁ֥ר הֲלִֽינֹתֶ֖ם עָלָֽי:  ת  בְּמַדְבְּרָא הָדֵין יִפְּלוּן פִּגְרֵיכוֹן וְכָל מִנְיָנֵיכוֹן לְכָל חֻשְׁבַּנְכוֹן מִבַּר עַשְׂרִין שְׁנִין וּלְעֵלָא דִי אִתְרַעַמְתּוּן עֲלָי:

{ל} אִם-אַתֶּם֙ תָּבֹ֣אוּ אֶל-הָאָ֔רֶץ אֲשֶׁ֤ר נָשָׂ֨אתִי֙ אֶת-יָדִ֔י לְשַׁכֵּ֥ן אֶתְכֶ֖ם בָּ֑הּ כִּ֚י אִם-כָּלֵ֣ב בֶּן-יְפֻנֶּ֔ה וִֽיהוֹשֻׁ֖עַ בִּן-נֽוּן:  מ  אִם-אַתֶּם֙ תָּבֹ֣אוּ אֶל-הָאָ֔רֶץ אֲשֶׁ֤ר נָשָׂ֨אתִי֙ אֶת-יָדִ֔י לְשַׁכֵּ֥ן אֶתְכֶ֖ם בָּ֑הּ כִּ֚י אִם-כָּלֵ֣ב בֶּן-יְפֻנֶּ֔ה וִֽיהוֹשֻׁ֖עַ בִּן-נֽוּן:  ת  אִם אַתּוּן תֵּעֲלוּן לְאַרְעָא דִי קַיֵמִית בְמֵימְרִי לְאַשְׁרָאָה יַתְכוֹן בַּהּ אֱלָהֵן כָּלֵב בַּר יְפֻנֶה וִיהוֹשֻׁעַ בַּר נוּן:

{לא} וְטַ֨פְּכֶ֔ם אֲשֶׁ֥ר אֲמַרְתֶּ֖ם לָבַ֣ז יִֽהְיֶ֑ה וְהֵֽבֵיאתִ֣י אֹתָ֔ם וְיָֽדְעוּ֙ אֶת-הָאָ֔רֶץ אֲשֶׁ֥ר מְאַסְתֶּ֖ם בָּֽהּ:  מ  וְטַ֨פְּכֶ֔ם אֲשֶׁ֥ר אֲמַרְתֶּ֖ם לָבַ֣ז יִֽהְיֶ֑ה וְהֵֽבֵיאתִ֣י אֹתָ֔ם וְיָֽדְעוּ֙ אֶת-הָאָ֔רֶץ אֲשֶׁ֥ר מְאַסְתֶּ֖ם בָּֽהּ:  ת  וְטַפְלְכוֹן דִי אֲמַרְתּוּן לְבִזָא יְהֵי וְאָעֵל יַתְהוֹן וְיֵדְעוּן יָת אַרְעָא דִי קַצְתּוּן בַּהּ:

{לב} וּפִגְרֵיכֶ֖ם אַתֶּ֑ם יִפְּל֖וּ בַּמִּדְבָּ֥ר הַזֶּֽה:  מ  וּפִגְרֵיכֶ֖ם אַתֶּ֑ם יִפְּל֖וּ בַּמִּדְבָּ֥ר הַזֶּֽה:  ת  וּפִגְרֵיכוֹן דִי לְכוֹן יִפְּלוּן בְּמַדְבְּרָא הָדֵין:

{לג} וּ֠בְנֵיכֶ֠ם יִֽהְי֨וּ רֹעִ֤ים בַּמִּדְבָּר֙ אַרְבָּעִ֣ים שָׁנָ֔ה וְנָֽשְׂא֖וּ אֶת-זְנֽוּתֵיכֶ֑ם עַד-תֹּ֥ם פִּגְרֵיכֶ֖ם בַּמִּדְבָּֽר:  מ  וּ֠בְנֵיכֶ֠ם יִֽהְי֨וּ רֹעִ֤ים בַּמִּדְבָּר֙ אַרְבָּעִ֣ים שָׁנָ֔ה וְנָֽשְׂא֖וּ אֶת-זְנֽוּתֵיכֶ֑ם עַד-תֹּ֥ם פִּגְרֵיכֶ֖ם בַּמִּדְבָּֽר:  ת  וּבְנֵיכוֹן יְהוֹן מְאַחֲרִין בְּמַדְבְּרָא אַרְבְּעִין שְׁנִין וִיקַבְּלוּן יָת חוֹבֵיכוֹן עַד דִיסוּפוּן פִגְרֵיכוֹן בְּמַדְבְּרָא:

{לד} בְּמִסְפַּ֨ר הַיָּמִ֜ים אֲשֶׁר-תַּרְתֶּ֣ם אֶת-הָאָ֘רֶץ֘ אַרְבָּעִ֣ים יוֹם֒ יוֹם לַשָּׁנָ֞ה יוֹם לַשָּׁנָ֗ה תִּשְׂאוּ֙ אֶת-עֲוֹנֹ֣תֵיכֶ֔ם אַרְבָּעִ֖ים שָׁנָ֑ה וִֽידַעְתֶּ֖ם אֶת-תְּנֽוּאָתִֽי:  מ  בְּמִסְפַּ֨ר הַיָּמִ֜ים אֲשֶׁר-תַּרְתֶּ֣ם אֶת-הָאָ֘רֶץ֘ אַרְבָּעִ֣ים יוֹם֒ יוֹם לַשָּׁנָ֞ה יוֹם לַשָּׁנָ֗ה תִּשְׂאוּ֙ אֶת-עֲוֹנֹ֣תֵיכֶ֔ם אַרְבָּעִ֖ים שָׁנָ֑ה וִֽידַעְתֶּ֖ם אֶת-תְּנֽוּאָתִֽי:  ת  בְּמִנְיַן יוֹמַיָא דִי אַלֶלְתּוּן יָת אַרְעָא אַרְבְּעִין יוֹמִין יוֹמָא לְשַׁתָּא יוֹמָא לְשַׁתָּא תְּקַבְּלוּן יָת חוֹבֵיכוֹן אַרְבְּעִין שְׁנִין וְתִדְעוּן יָת דְאִתְרַעַמְתּוּן עֲלָי:

{לה} אֲנִ֣י יְהֹוָה֘ דִּבַּ֒רְתִּי֒ אִם-לֹ֣א | זֹ֣את אֶֽעֱשֶׂ֗ה לְכָל-הָֽעֵדָ֤ה הָֽרָעָה֙ הַזֹּ֔את הַנּֽוֹעָדִ֖ים עָלָ֑י בַּמִּדְבָּ֥ר הַזֶּ֛ה יִתַּ֖מּוּ וְשָׁ֥ם יָמֻֽתוּ:  מ  אֲנִ֣י יְהֹוָה֘ דִּבַּ֒רְתִּי֒ אִם-לֹ֣א | זֹ֣את אֶֽעֱשֶׂ֗ה לְכָל-הָֽעֵדָ֤ה הָֽרָעָה֙ הַזֹּ֔את הַנּֽוֹעָדִ֖ים עָלָ֑י בַּמִּדְבָּ֥ר הַזֶּ֛ה יִתַּ֖מּוּ וְשָׁ֥ם יָמֻֽתוּ:  ת  אֲנָא יְיָ גְזָרִית בְּמֵימְרִי אִם לָא דָא אֶעְבֵּד לְכָל כְּנִשְׁתָּא בִישְׁתָא הָדָא דְאִזְדַמְנוּ עֲלָי בְּמַדְבְּרָא הָדֵין יְסוּפוּן וְתַמָן יְמוּתוּן:

{לו} וְהָ֣אֲנָשִׁ֔ים אֲשֶׁר-שָׁלַ֥ח מֹשֶׁ֖ה לָת֣וּר אֶת-הָאָ֑רֶץ וַיָּשֻׁ֗בוּ (וילונו) וַיַּלִּ֤ינוּ עָלָיו֙ אֶת-כָּל-הָ֣עֵדָ֔ה לְהוֹצִ֥יא דִבָּ֖ה עַל-הָאָֽרֶץ:  מ  וְהָ֣אֲנָשִׁ֔ים אֲשֶׁר-שָׁלַ֥ח מֹשֶׁ֖ה לָת֣וּר אֶת-הָאָ֑רֶץ וַיָּשֻׁ֗בוּ (וילונו) וַיַּלִּ֤ינוּ עָלָיו֙ אֶת-כָּל-הָ֣עֵדָ֔ה לְהוֹצִ֥יא דִבָּ֖ה עַל-הָאָֽרֶץ:  ת  וְגֻבְרַיָא דִי שְׁלַח משֶׁה לְאַלָלָא יָת אַרְעָא וְתָבוּ וְאַרְעִימוּ עֲלוֹהִי יָת כָּל כְּנִשְׁתָּא לְאַפָּקָא שׁוּם בִּישׁ עַל אַרְעָא:

{לז} וַיָּמֻ֨תוּ֙ הָֽאֲנָשִׁ֔ים מֽוֹצִאֵ֥י דִבַּת-הָאָ֖רֶץ רָעָ֑ה בַּמַּגֵּפָ֖ה לִפְנֵ֥י יְהֹוָֽה:  מ  וַיָּמֻ֨תוּ֙ הָֽאֲנָשִׁ֔ים מֽוֹצִאֵ֥י דִבַּת-הָאָ֖רֶץ רָעָ֑ה בַּמַּגֵּפָ֖ה לִפְנֵ֥י יְהֹוָֽה:  ת  וּמִיתוּ גֻבְרַיָא דְאַפִּיקוּ שׁוּם בִּישׁ עַל אַרְעָא בְּמוֹתָנָא קֳדָם יְיָ:

{לח} וִֽיהוֹשֻׁ֣עַ בִּן-נ֔וּן וְכָלֵ֖ב בֶּן-יְפֻנֶּ֑ה חָיוּ֙ מִן-הָֽאֲנָשִׁ֣ים הָהֵ֔ם הַֽהֹלְכִ֖ים לָת֥וּר אֶת-הָאָֽרֶץ:  מ  וִֽיהוֹשֻׁ֣עַ בִּן-נ֔וּן וְכָלֵ֖ב בֶּן-יְפֻנֶּ֑ה חָיוּ֙ מִן-הָֽאֲנָשִׁ֣ים הָהֵ֔ם הַֽהֹלְכִ֖ים לָת֥וּר אֶת-הָאָֽרֶץ:  ת  וִיהוֹשֻׁעַ בַּר נוּן וְכָלֵב בַּר יְפֻנֶה אִתְקַיְימוּ מִן גֻבְרַיָא הָאִנוּן דַאֲזָלוּ לְאַלָלָא יָת אַרְעָא:

{לט} וַיְדַבֵּ֤ר מֹשֶׁה֙ אֶת-הַדְּבָרִ֣ים הָאֵ֔לֶּה אֶֽל-כָּל-בְּנֵ֖י יִשְׂרָאֵ֑ל וַיִּֽתְאַבְּל֥וּ הָעָ֖ם מְאֹֽד:  מ  וַיְדַבֵּ֤ר מֹשֶׁה֙ אֶת-הַדְּבָרִ֣ים הָאֵ֔לֶּה אֶֽל-כָּל-בְּנֵ֖י יִשְׂרָאֵ֑ל וַיִּֽתְאַבְּל֥וּ הָעָ֖ם מְאֹֽד:  ת  וּמַלִיל משֶׁה יָת פִּתְגָמַיָא הָאִלֵין לְכָל בְּנֵי יִשְׂרָאֵל וְאִתְאַבָּלוּ עַמָא לַחֲדָא:

{מ} וַיַּשְׁכִּ֣מוּ בַבֹּ֔קֶר וַיַּֽעֲל֥וּ אֶל-רֹאשׁ-הָהָ֖ר לֵאמֹ֑ר הִנֶּ֗נּוּ וְעָלִ֛ינוּ אֶל-הַמָּק֛וֹם אֲשֶׁר-אָמַ֥ר יְהֹוָ֖ה כִּ֥י חָטָֽאנוּ:  מ  וַיַּשְׁכִּ֣מוּ בַבֹּ֔קֶר וַיַּֽעֲל֥וּ אֶל-רֹאשׁ-הָהָ֖ר לֵאמֹ֑ר הִנֶּ֗נּוּ וְעָלִ֛ינוּ אֶל-הַמָּק֛וֹם אֲשֶׁר-אָמַ֥ר יְהֹוָ֖ה כִּ֥י חָטָֽאנוּ:  ת  וְאַקְדִימוּ בְצַפְרָא וּסְלִיקוּ לְרֵישׁ טוּרָא לְמֵימָר הָא אֲנַחְנָא סָלְקִין לְאַתְרָא דִי אֲמַר יְיָ אֲרֵי חַבְנָא:

{מא} וַיֹּ֣אמֶר מֹשֶׁ֔ה לָ֥מָּה זֶּ֛ה אַתֶּ֥ם עֹֽבְרִ֖ים אֶת-פִּ֣י יְהֹוָ֑ה וְהִ֖וא לֹ֥א תִצְלָֽח:  מ  וַיֹּ֣אמֶר מֹשֶׁ֔ה לָ֥מָּה זֶּ֛ה אַתֶּ֥ם עֹֽבְרִ֖ים אֶת-פִּ֣י יְהֹוָ֑ה וְהִ֖וא לֹ֥א תִצְלָֽח:  ת  וַאֲמַר משֶׁה לְמָא דְנַן אַתּוּן עָבְרִין עַל גְזֵרַת מֵימְרָא דַיְיָ וְהִיא לָא תַצְלָח:

{מב} אַֽל-תַּעֲל֔וּ כִּ֛י אֵ֥ין יְהֹוָ֖ה בְּקִרְבְּכֶ֑ם וְלֹא֙ תִּנָּ֣גְפ֔וּ לִפְנֵ֖י אֹֽיְבֵיכֶֽם:  מ  אַֽל-תַּעֲל֔וּ כִּ֛י אֵ֥ין יְהֹוָ֖ה בְּקִרְבְּכֶ֑ם וְלֹא֙ תִּנָּ֣גְפ֔וּ לִפְנֵ֖י אֹֽיְבֵיכֶֽם:  ת  לָא תִסְקוּן אֲרֵי לֵית שְׁכִנְתָּא דַיְיָ בֵּינֵיכוֹן וְלָא תִתַּבְּרוּן קֳדָם בַּעֲלֵי דְבָבֵיכוֹן:

{מג} כִּי֩ הָעֲמָֽלֵקִ֨י וְהַכְּנַֽעֲנִ֥י שָׁם֙ לִפְנֵיכֶ֔ם וּנְפַלְתֶּ֖ם בֶּחָ֑רֶב כִּֽי-עַל-כֵּ֤ן שַׁבְתֶּם֙ מֵאַֽחֲרֵ֣י יְהֹוָ֔ה וְלֹֽא-יִהְיֶ֥ה יְהֹוָ֖ה עִמָּכֶֽם:  מ  כִּי֩ הָעֲמָֽלֵקִ֨י וְהַכְּנַֽעֲנִ֥י שָׁם֙ לִפְנֵיכֶ֔ם וּנְפַלְתֶּ֖ם בֶּחָ֑רֶב כִּֽי-עַל-כֵּ֤ן שַׁבְתֶּם֙ מֵאַֽחֲרֵ֣י יְהֹוָ֔ה וְלֹֽא-יִהְיֶ֥ה יְהֹוָ֖ה עִמָּכֶֽם:  ת  אֲרֵי עֲמַלְקָאָה וּכְנַעֲנָאָה תַמָן קֳדָמֵיכוֹן וְתִפְּלוּן בְּחַרְבָּא אֲרֵי עַל כֵּן תַּבְתּוּן מִבָּתַר פּוּלְחָנָא דַיְיָ וְלָא יְהֵי מֵימְרָא דַיְיָ בְּסַעְדְכוֹן:

{מד} וַיַּעְפִּ֕לוּ לַֽעֲל֖וֹת אֶל-רֹ֣אשׁ הָהָ֑ר וַֽאֲר֤וֹן בְּרִית-יְהֹוָה֙ וּמֹשֶׁ֔ה לֹא-מָ֖שׁוּ מִקֶּ֥רֶב הַֽמַּחֲנֶֽה:  מ  וַיַּעְפִּ֕לוּ לַֽעֲל֖וֹת אֶל-רֹ֣אשׁ הָהָ֑ר וַֽאֲר֤וֹן בְּרִית-יְהֹוָה֙ וּמֹשֶׁ֔ה לֹא-מָ֖שׁוּ מִקֶּ֥רֶב הַֽמַּחֲנֶֽה:  ת  וְאַרְשָׁעוּ לְמִסַק לְרֵישׁ טוּרָא וַאֲרוֹן קְיָמָא דַיְיָ וּמשֶׁה לָא עֲדוֹ מִגוֹ מַשְׁרִיתָא:

{מה} וַיֵּ֤רֶד הָעֲמָֽלֵקִי֙ וְהַֽכְּנַעֲנִ֔י הַיֹּשֵׁ֖ב בָּהָ֣ר הַה֑וּא וַיַּכּ֥וּם וַֽיַּכְּת֖וּם עַד-הַֽחָרְמָֽה:  מ  וַיֵּ֤רֶד הָעֲמָֽלֵקִי֙ וְהַֽכְּנַעֲנִ֔י הַיֹּשֵׁ֖ב בָּהָ֣ר הַה֑וּא וַיַּכּ֥וּם וַֽיַּכְּת֖וּם עַד-הַֽחָרְמָֽה:  ת  וּנְחַת עֲמַלְקָאָה וּכְנַעֲנָאָה דְיָתֵב בְּטוּרָא הַהוּא וּמְחֻנוּן וּטְרָדֻנוּן עַד חָרְמָה: (פ)




במדבר פרק-טו

{א} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{ב} דַּבֵּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם כִּ֣י תָבֹ֗אוּ אֶל-אֶ֨רֶץ֙ מוֹשְׁבֹ֣תֵיכֶ֔ם אֲשֶׁ֥ר אֲנִ֖י נֹתֵ֥ן לָכֶֽם:  מ  דַּבֵּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם כִּ֣י תָבֹ֗אוּ אֶל-אֶ֨רֶץ֙ מוֹשְׁבֹ֣תֵיכֶ֔ם אֲשֶׁ֥ר אֲנִ֖י נֹתֵ֥ן לָכֶֽם:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן אֲרֵי תֵעֲלוּן לַאֲרַע מוֹתְבָנֵיכוֹן דִי אֲנָא יָהֵב לְכוֹן:

{ג} וַֽעֲשִׂיתֶ֨ם אִשֶּׁ֤ה לַֽיהֹוָה֙ עֹלָ֣ה אוֹ-זֶ֔בַח לְפַלֵּא-נֶ֨דֶר֙ אוֹ בִנְדָבָ֔ה א֖וֹ בְּמֹֽעֲדֵיכֶ֑ם לַֽעֲשׂ֞וֹת רֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה מִן-הַבָּקָ֖ר א֥וֹ מִן-הַצֹּֽאן:  מ  וַֽעֲשִׂיתֶ֨ם אִשֶּׁ֤ה לַֽיהֹוָה֙ עֹלָ֣ה אוֹ-זֶ֔בַח לְפַלֵּא-נֶ֨דֶר֙ אוֹ בִנְדָבָ֔ה א֖וֹ בְּמֹֽעֲדֵיכֶ֑ם לַֽעֲשׂ֞וֹת רֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה מִן-הַבָּקָ֖ר א֥וֹ מִן-הַצֹּֽאן:  ת  וְתַעְבְּדוּן קֻרְבָּנָא קָדָם יְיָ עֲלָתָא אוֹ נִכְסַת קוּדְשַׁיָא לְאַפְרָשָׁא נִדְרָא אוֹ בִנְדַבְתָּא אוֹ בְּמוֹעֲדֵיכוֹן לְמֶעְבַּד לְאִתְקַבָּלָא בְרַעֲוָא קֳדָם יְיָ מִן תּוֹרֵי אוֹ מִן עָנָא:

{ד} וְהִקְרִ֛יב הַמַּקְרִ֥יב קָרְבָּנ֖וֹ לַֽיהֹוָ֑ה מִנְחָה֙ סֹ֣לֶת עִשָּׂר֔וֹן בָּל֕וּל בִּרְבִעִ֥ית הַהִ֖ין שָֽׁמֶן:  מ  וְהִקְרִ֛יב הַמַּקְרִ֥יב קָרְבָּנ֖וֹ לַֽיהֹוָ֑ה מִנְחָה֙ סֹ֣לֶת עִשָּׂר֔וֹן בָּל֕וּל בִּרְבִעִ֥ית הַהִ֖ין שָֽׁמֶן:  ת  וִיקָרֵב דִמְקָרֵב קֻרְבָּנֵהּ קֳדָם יְיָ מִנְחָתָא סֻלְתָּא עַשְׂרוֹנָא דְפִילָא בְּרַבְעוּת הִינָא מִשְׁחָא:

{ה} וְיַ֤יִן לַנֶּ֨סֶךְ֙ רְבִיעִ֣ית הַהִ֔ין תַּֽעֲשֶׂ֥ה עַל-הָֽעֹלָ֖ה אוֹ לַזָּ֑בַח לַכֶּ֖בֶשׂ הָֽאֶחָֽד:  מ  וְיַ֤יִן לַנֶּ֨סֶךְ֙ רְבִיעִ֣ית הַהִ֔ין תַּֽעֲשֶׂ֥ה עַל-הָֽעֹלָ֖ה אוֹ לַזָּ֑בַח לַכֶּ֖בֶשׂ הָֽאֶחָֽד:  ת  וְחַמְרָא לְנִסְכָּא רַבְעוּת הִינָא תַּעְבֵּד עַל עֲלָתָא אוֹ לְנִכְסַת קוּדְשַׁיָא לְאִמְרָא חָד:

{ו} א֤וֹ לָאַ֨יִל֙ תַּֽעֲשֶׂ֣ה מִנְחָ֔ה סֹ֖לֶת שְׁנֵ֣י עֶשְׂרֹנִ֑ים בְּלוּלָ֥ה בַשֶּׁ֖מֶן שְׁלִשִׁ֥ית הַהִֽין:  מ  א֤וֹ לָאַ֨יִל֙ תַּֽעֲשֶׂ֣ה מִנְחָ֔ה סֹ֖לֶת שְׁנֵ֣י עֶשְׂרֹנִ֑ים בְּלוּלָ֥ה בַשֶּׁ֖מֶן שְׁלִשִׁ֥ית הַהִֽין:  ת  אוֹ לְדִכְרָא תַּעְבֵּד מִנְחָתָא סֻלְתָּא תְּרֵין עַשְׂרוֹנִין דְפִילָא בִמְשַׁח תַּלְתּוּת הִינָא:

{ז} וְיַ֥יִן לַנֶּ֖סֶךְ שְׁלִשִׁ֣ית הַהִ֑ין תַּקְרִ֥יב רֵֽיחַ-נִיחֹ֖חַ לַֽיהֹוָֽה:  מ  וְיַ֥יִן לַנֶּ֖סֶךְ שְׁלִשִׁ֣ית הַהִ֑ין תַּקְרִ֥יב רֵֽיחַ-נִיחֹ֖חַ לַֽיהֹוָֽה:  ת  וְחַמְרָא לְנִסְכָּא תַּלְתּוּת הִינָא תְּקָרֵב לְאִתְקַבָּלָא בְרַעֲוָא קֳדָם יְיָ:

{ח}  חמישי  וְכִֽי-תַעֲשֶׂ֥ה בֶן-בָּקָ֖ר עֹלָ֣ה אוֹ-זָ֑בַח לְפַלֵּא-נֶ֥דֶר אֽוֹ-שְׁלָמִ֖ים לַֽיהֹוָֽה:  מ  וְכִֽי-תַעֲשֶׂ֥ה בֶן-בָּקָ֖ר עֹלָ֣ה אוֹ-זָ֑בַח לְפַלֵּא-נֶ֥דֶר אֽוֹ-שְׁלָמִ֖ים לַֽיהֹוָֽה:  ת  וַאֲרֵי תַעְבֵּד בַּר תּוֹרֵי עֲלָתָא אוֹ נִכְסַת קוּדְשַׁיָא לְפָרָשָׁא נִדְרָא אוֹ נִכְסַת קוּדְשַׁיָא קֳדָם יְיָ:

{ט} וְהִקְרִ֤יב עַל-בֶּן-הַבָּקָר֙ מִנְחָ֔ה סֹ֖לֶת שְׁלֹשָׁ֣ה עֶשְׂרֹנִ֑ים בָּל֥וּל בַּשֶּׁ֖מֶן חֲצִ֥י הַהִֽין:  מ  וְהִקְרִ֤יב עַל-בֶּן-הַבָּקָר֙ מִנְחָ֔ה סֹ֖לֶת שְׁלֹשָׁ֣ה עֶשְׂרֹנִ֑ים בָּל֥וּל בַּשֶּׁ֖מֶן חֲצִ֥י הַהִֽין:  ת  וִיקָרֵב עַל בַּר תּוֹרֵי מִנְחָתָא סֻלְתָּא תְּלָתָא עַשְׂרוֹנִין דְפִילָא בִמְשַׁח פַּלְגוּת הִינָא:

{י} וְיַ֛יִן תַּקְרִ֥יב לַנֶּ֖סֶךְ חֲצִ֣י הַהִ֑ין אִשֵּׁ֥ה רֵֽיחַ-נִיחֹ֖חַ לַֽיהֹוָֽה:  מ  וְיַ֛יִן תַּקְרִ֥יב לַנֶּ֖סֶךְ חֲצִ֣י הַהִ֑ין אִשֵּׁ֥ה רֵֽיחַ-נִיחֹ֖חַ לַֽיהֹוָֽה:  ת  וְחַמְרָא תְּקָרֵב לְנִסְכָּא פַּלְגוּת הִינָא קֻרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ:

{יא} כָּ֣כָה יֵֽעָשֶׂ֗ה לַשּׁוֹר֙ הָֽאֶחָ֔ד א֖וֹ לָאַ֣יִל הָֽאֶחָ֑ד אֽוֹ-לַשֶּׂ֥ה בַכְּבָשִׂ֖ים א֥וֹ בָֽעִזִּֽים:  מ  כָּ֣כָה יֵֽעָשֶׂ֗ה לַשּׁוֹר֙ הָֽאֶחָ֔ד א֖וֹ לָאַ֣יִל הָֽאֶחָ֑ד אֽוֹ-לַשֶּׂ֥ה בַכְּבָשִׂ֖ים א֥וֹ בָֽעִזִּֽים:  ת  כְּדֵין יִתְעֲבֵיד לְתוֹרָא חַד אוֹ לְדִכְרָא חָד אוֹ לְאִמַר בְּאִמְרַיָא אוֹ בְעִזַיָא:

{יב} כַּמִּסְפָּ֖ר אֲשֶׁ֣ר תַּֽעֲשׂ֑וּ כָּ֛כָה תַּֽעֲשׂ֥וּ לָֽאֶחָ֖ד כְּמִסְפָּרָֽם:  מ  כַּמִּסְפָּ֖ר אֲשֶׁ֣ר תַּֽעֲשׂ֑וּ כָּ֛כָה תַּֽעֲשׂ֥וּ לָֽאֶחָ֖ד כְּמִסְפָּרָֽם:  ת  כְּמִנְיָן דִי תַעְבְּדוּן כְּדֵין תַּעְבְּדוּן לְחַד כְּמִנְיָנֵהוֹן:

{יג} כָּל-הָֽאֶזְרָ֥ח יַֽעֲשֶׂה-כָּ֖כָה אֶת-אֵ֑לֶּה לְהַקְרִ֛יב אִשֵּׁ֥ה רֵֽיחַ-נִיחֹ֖חַ לַֽיהֹוָֽה:  מ  כָּל-הָֽאֶזְרָ֥ח יַֽעֲשֶׂה-כָּ֖כָה אֶת-אֵ֑לֶּה לְהַקְרִ֛יב אִשֵּׁ֥ה רֵֽיחַ-נִיחֹ֖חַ לַֽיהֹוָֽה:  ת  כָּל יַצִיבָא יַעְבֵּד כְּדֵין יָת אִלֵין לְקָרָבָא קֻרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ:

{יד} וְכִֽי-יָגוּר֩ אִתְּכֶ֨ם גֵּ֜ר א֤וֹ אֲשֶֽׁר-בְּתֽוֹכְכֶם֙ לְדֹרֹ֣תֵיכֶ֔ם וְעָשָׂ֛ה אִשֵּׁ֥ה רֵֽיחַ-נִיחֹ֖חַ לַֽיהֹוָ֑ה כַּֽאֲשֶׁ֥ר תַּֽעֲשׂ֖וּ כֵּ֥ן יַֽעֲשֶֽׂה:  מ  וְכִֽי-יָגוּר֩ אִתְּכֶ֨ם גֵּ֜ר א֤וֹ אֲשֶֽׁר-בְּתֽוֹכְכֶם֙ לְדֹרֹ֣תֵיכֶ֔ם וְעָשָׂ֛ה אִשֵּׁ֥ה רֵֽיחַ-נִיחֹ֖חַ לַֽיהֹוָ֑ה כַּֽאֲשֶׁ֥ר תַּֽעֲשׂ֖וּ כֵּ֥ן יַֽעֲשֶֽׂה:  ת  וַאֲרֵי יִתְגַיַר עִמְכוֹן גִיוֹרָא אוֹ דִי בֵינֵיכוֹן לְדָרֵיכוֹן וְיַעְבֵּד קֻרְבַּן דְמִתְקַּבֵּל בְּרַעֲוָא קֳדָם יְיָ כְּמָא דִי תַעְבְּדוּן כֵּן יַעְבֵּד:

{טו} הַקָּהָ֕ל חֻקָּ֥ה אַחַ֛ת לָכֶ֖ם וְלַגֵּ֣ר הַגָּ֑ר חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם כָּכֶ֛ם כַּגֵּ֥ר יִֽהְיֶ֖ה לִפְנֵ֥י יְהֹוָֽה:  מ  הַקָּהָ֕ל חֻקָּ֥ה אַחַ֛ת לָכֶ֖ם וְלַגֵּ֣ר הַגָּ֑ר חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם כָּכֶ֛ם כַּגֵּ֥ר יִֽהְיֶ֖ה לִפְנֵ֥י יְהֹוָֽה:  ת  קְהָלָא קְיָמָא חַד לְכוֹן וּלְגִיוֹרָא דְיִתְגַיָר קְיַם עָלָם לְדָרֵיכוֹן כְּוַתְכוֹן כְּגִיוֹרָא יְהֵי קָדָם יְיָ:

{טז} תּוֹרָ֥ה אַחַ֛ת וּמִשְׁפָּ֥ט אֶחָ֖ד יִֽהְיֶ֣ה לָכֶ֑ם וְלַגֵּ֖ר הַגָּ֥ר אִתְּכֶֽם:  מ  תּוֹרָ֥ה אַחַ֛ת וּמִשְׁפָּ֥ט אֶחָ֖ד יִֽהְיֶ֣ה לָכֶ֑ם וְלַגֵּ֖ר הַגָּ֥ר אִתְּכֶֽם:  ת  אוֹרַיְתָא חֲדָא וְדִינָא חַד יְהֵי לְכוֹן וּלְגִיוֹרָא דְיִתְגַיַר עִמְכוֹן: (פ)

{יז}  שישי  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{יח} דַּבֵּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם בְּבֹֽאֲכֶם֙ אֶל-הָאָ֔רֶץ אֲשֶׁ֥ר אֲנִ֛י מֵבִ֥יא אֶתְכֶ֖ם שָֽׁמָּה:  מ  דַּבֵּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם בְּבֹֽאֲכֶם֙ אֶל-הָאָ֔רֶץ אֲשֶׁ֥ר אֲנִ֛י מֵבִ֥יא אֶתְכֶ֖ם שָֽׁמָּה:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן בְּמֵעַלְכוֹן לְאַרְעָא דִי אֲנָא מָעֵל יָתְכוֹן לְתַמָן:

{יט} וְהָיָ֕ה בַּֽאֲכָלְכֶ֖ם מִלֶּ֣חֶם הָאָ֑רֶץ תָּרִ֥ימוּ תְרוּמָ֖ה לַֽיהֹוָֽה:  מ  וְהָיָ֕ה בַּֽאֲכָלְכֶ֖ם מִלֶּ֣חֶם הָאָ֑רֶץ תָּרִ֥ימוּ תְרוּמָ֖ה לַֽיהֹוָֽה:  ת  וִיהֵי בְמֵיכַלְכוֹן מִלַחְמָא דְאַרְעָא תַּפְרְשׁוּן אַפְרָשׁוּתָא קֳדָם יְיָ:

{כ} רֵאשִׁית֙ עֲרִסֹ֣תֵכֶ֔ם חַלָּ֖ה תָּרִ֣ימוּ תְרוּמָ֑ה כִּתְרוּמַ֣ת גֹּ֔רֶן כֵּ֖ן תָּרִ֥ימוּ אֹתָֽהּ:  מ  רֵאשִׁית֙ עֲרִסֹ֣תֵכֶ֔ם חַלָּ֖ה תָּרִ֣ימוּ תְרוּמָ֑ה כִּתְרוּמַ֣ת גֹּ֔רֶן כֵּ֖ן תָּרִ֥ימוּ אֹתָֽהּ:  ת  רֵישׁ אַצְוַתְכוֹן חַלְתָא תַּפְרְשׁוּן אַפְרָשׁוּתָא כְּמָא דִי מַפְרְשִׁין מִן אִדְרָא כֵּן תַּפְרְשׁוּן יָתַהּ:

{כא} מֵֽרֵאשִׁית֙ עֲרִסֹ֣תֵיכֶ֔ם תִּתְּנ֥וּ לַֽיהֹוָ֖ה תְּרוּמָ֑ה לְדֹרֹ֖תֵיכֶֽם:  מ  מֵֽרֵאשִׁית֙ עֲרִסֹ֣תֵיכֶ֔ם תִּתְּנ֥וּ לַֽיהֹוָ֖ה תְּרוּמָ֑ה לְדֹרֹ֖תֵיכֶֽם:  ת  מֵרֵישׁ אַצְוַתְכוֹן תִּתְּנוּן קֳדָם יְיָ אַפְרָשׁוּתָא לְדָרֵיכוֹן: (ס)

{כב} וְכִ֣י תִשְׁגּ֔וּ וְלֹ֣א תַֽעֲשׂ֔וּ אֵ֥ת כָּל-הַמִּצְוֹ֖ת הָאֵ֑לֶּה אֲשֶׁר-דִּבֶּ֥ר יְהֹוָ֖ה אֶל-מֹשֶֽׁה:  מ  וְכִ֣י תִשְׁגּ֔וּ וְלֹ֣א תַֽעֲשׂ֔וּ אֵ֥ת כָּל-הַמִּצְוֹ֖ת הָאֵ֑לֶּה אֲשֶׁר-דִּבֶּ֥ר יְהֹוָ֖ה אֶל-מֹשֶֽׁה:  ת  וַאֲרֵי תִשְׁתְּלוּן וְלָא תַעְבְּדוּן יָת כָּל פִּקוֹדַיָא הָאִלֵין דִי מַלִיל יְיָ עִם משֶׁה:

{כג} אֵת֩ כָּל-אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֲלֵיכֶ֖ם בְּיַד-מֹשֶׁ֑ה מִן-הַיּ֞וֹם אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה וָהָ֖לְאָה לְדֹרֹֽתֵיכֶֽם:  מ  אֵת֩ כָּל-אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֲלֵיכֶ֖ם בְּיַד-מֹשֶׁ֑ה מִן-הַיּ֞וֹם אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה וָהָ֖לְאָה לְדֹרֹֽתֵיכֶֽם:  ת  יָת כָּל דִי פַקִיד יְיָ יָתְכוֹן בִּידָא דְמשֶׁה מִן יוֹמָא דִי פָקִיד יְיָ וּלְהָלָא לְדָרֵיכוֹן:

{כד} וְהָיָ֗ה אִ֣ם מֵֽעֵינֵ֣י הָֽעֵדָה֘ נֶֽעֶשְׂתָ֣ה לִשְׁגָגָה֒ וְעָשׂ֣וּ כָל-הָֽעֵדָ֡ה פַּ֣ר בֶּן-בָּקָר֩ אֶחָ֨ד לְעֹלָ֜ה לְרֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה וּמִנְחָת֥וֹ וְנִסְכּ֖וֹ כַּמִּשְׁפָּ֑ט וּשְׂעִיר-עִזִּ֥ים אֶחָ֖ד לְחַטָּֽת:  מ  וְהָיָ֗ה אִ֣ם מֵֽעֵינֵ֣י הָֽעֵדָה֘ נֶֽעֶשְׂתָ֣ה לִשְׁגָגָה֒ וְעָשׂ֣וּ כָל-הָֽעֵדָ֡ה פַּ֣ר בֶּן-בָּקָר֩ אֶחָ֨ד לְעֹלָ֜ה לְרֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה וּמִנְחָת֥וֹ וְנִסְכּ֖וֹ כַּמִּשְׁפָּ֑ט וּשְׂעִיר-עִזִּ֥ים אֶחָ֖ד לְחַטָּֽת:  ת  וִיהֵי אִם מֵעֵינֵי כְנִשְׁתָּא אִתְעֲבִדַת לְשָׁלוּ וְיַעְבְּדוּן כָּל כְּנִשְׁתָּא תּוֹר בַּר תּוֹרֵי חַד לַעֲלָתָא לְאִתְקַבָּלָא בְרַעֲוָא קֳדָם יְיָ וּמִנְחָתֵהּ וְנִסְכֵּהּ כִּדְחָזֵי וּצְפִיר בַּר עִזֵי חַד לְחַטָּאתָא:

{כה} וְכִפֶּ֣ר הַכֹּהֵ֗ן עַֽל-כָּל-עֲדַ֛ת בְּנֵ֥י יִשְׂרָאֵ֖ל וְנִסְלַ֣ח לָהֶ֑ם כִּֽי-שְׁגָגָ֣ה הִ֔וא וְהֵם֩ הֵבִ֨יאוּ אֶת-קָרְבָּנָ֜ם אִשֶּׁ֣ה לַֽיהֹוָ֗ה וְחַטָּאתָ֛ם לִפְנֵ֥י יְהֹוָ֖ה עַל-שִׁגְגָתָֽם:  מ  וְכִפֶּ֣ר הַכֹּהֵ֗ן עַֽל-כָּל-עֲדַ֛ת בְּנֵ֥י יִשְׂרָאֵ֖ל וְנִסְלַ֣ח לָהֶ֑ם כִּֽי-שְׁגָגָ֣ה הִ֔וא וְהֵם֩ הֵבִ֨יאוּ אֶת-קָרְבָּנָ֜ם אִשֶּׁ֣ה לַֽיהֹוָ֗ה וְחַטָּאתָ֛ם לִפְנֵ֥י יְהֹוָ֖ה עַל-שִׁגְגָתָֽם:  ת  וִיכַפַּר כַּהֲנָא עַל כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל וְיִשְׁתְּבֵק לְהוֹן אֲרֵי שָׁלוּתָא הִיא וְאִנוּן אַיְתִיאוּ יָת קֻרְבַּנְהוֹן קֻרְבָּנָא קֳדָם יְיָ וְחוֹבַתְהוֹן קֳדָם יְיָ עַל שָׁלוּתְהוֹן:

{כו} וְנִסְלַ֗ח לְכָל-עֲדַת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וְלַגֵּ֖ר הַגָּ֣ר בְּתוֹכָ֑ם כִּ֥י לְכָל-הָעָ֖ם בִּשְׁגָגָֽה:  מ  וְנִסְלַ֗ח לְכָל-עֲדַת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וְלַגֵּ֖ר הַגָּ֣ר בְּתוֹכָ֑ם כִּ֥י לְכָל-הָעָ֖ם בִּשְׁגָגָֽה:  ת  וְיִשְׁתְּבֵק לְכָל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל וּלְגִיוֹרָא דְיִתְגַיַר בֵּינֵיהוֹן אֲרֵי לְכָל עַמָא בְּשָׁלוּתָא: (ס)

{כז}  שביעי  וְאִם-נֶ֥פֶשׁ אַחַ֖ת תֶּֽחֱטָ֣א בִשְׁגָגָ֑ה וְהִקְרִ֛יבָה עֵ֥ז בַּת-שְׁנָתָ֖הּ לְחַטָּֽאת:  מ  וְאִם-נֶ֥פֶשׁ אַחַ֖ת תֶּֽחֱטָ֣א בִשְׁגָגָ֑ה וְהִקְרִ֛יבָה עֵ֥ז בַּת-שְׁנָתָ֖הּ לְחַטָּֽאת:  ת  וְאִם אֱנַשׁ חַד יֵחוֹב בְּשָׁלוּ וִיקָרֵב עִזָא בַּת שַׁתַּהּ לְחַטָאתָא:

{כח} וְכִפֶּ֣ר הַכֹּהֵ֗ן עַל-הַנֶּ֧פֶשׁ הַשֹּׁגֶ֛גֶת בְּחֶטְאָ֥ה בִשְׁגָגָ֖ה לִפְנֵ֣י יְהֹוָ֑ה לְכַפֵּ֥ר עָלָ֖יו וְנִסְלַ֥ח לֽוֹ:  מ  וְכִפֶּ֣ר הַכֹּהֵ֗ן עַל-הַנֶּ֧פֶשׁ הַשֹּׁגֶ֛גֶת בְּחֶטְאָ֥ה בִשְׁגָגָ֖ה לִפְנֵ֣י יְהֹוָ֑ה לְכַפֵּ֥ר עָלָ֖יו וְנִסְלַ֥ח לֽוֹ:  ת  וִיכַפַּר כַּהֲנָא עַל אֱנַשׁ דְאִשְׁתְּלֵי בְּמֵחֲבֵהּ בְּשָׁלוּ קֳדָם יְיָ לְכַפָּרָא עֲלוֹהִי וְיִשְׁתְּבֵק לֵהּ:

{כט} הָֽאֶזְרָח֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל וְלַגֵּ֖ר הַגָּ֣ר בְּתוֹכָ֑ם תּוֹרָ֤ה אַחַת֙ יִֽהְיֶ֣ה לָכֶ֔ם לָֽעֹשֶׂ֖ה בִּשְׁגָגָֽה:  מ  הָֽאֶזְרָח֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל וְלַגֵּ֖ר הַגָּ֣ר בְּתוֹכָ֑ם תּוֹרָ֤ה אַחַת֙ יִֽהְיֶ֣ה לָכֶ֔ם לָֽעֹשֶׂ֖ה בִּשְׁגָגָֽה:  ת  יַצִיבָא בִּבְנֵי יִשְׂרָאֵל וּלְגִיוֹרָא דְיִתְגַיַר בֵּינֵיהוֹן אוֹרַיְתָא חֲדָא יְהֵי לְכוֹן לִדְיַעְבֵּד בְּשָׁלוּ:

{ל} וְהַנֶּ֜פֶשׁ אֲשֶֽׁר-תַּעֲשֶׂ֣ה | בְּיָ֣ד רָמָ֗ה מִן-הָֽאֶזְרָח֙ וּמִן-הַגֵּ֔ר אֶת-יְהֹוָ֖ה הוּא מְגַדֵּ֑ף וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מִקֶּ֥רֶב עַמָּֽהּ:  מ  וְהַנֶּ֜פֶשׁ אֲשֶֽׁר-תַּעֲשֶׂ֣ה | בְּיָ֣ד רָמָ֗ה מִן-הָֽאֶזְרָח֙ וּמִן-הַגֵּ֔ר אֶת-יְהֹוָ֖ה הוּא מְגַדֵּ֑ף וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מִקֶּ֥רֶב עַמָּֽהּ:  ת  וֶאֱנַשׁ דְיַעְבֵּד בְּרֵישׁ גְלֵי מִן יַצִיבַיָא וּמִן גִיוֹרַיָא קֳדָם יְיָ הוּא מַרְגֵז וְיִשְׁתֵּצֵי אֲנָשָׁא הַהִיא מִגוֹ עַמֵהּ:

{לא} כִּ֤י דְבַר-יְהֹוָה֙ בָּזָ֔ה וְאֶת-מִצְוָת֖וֹ הֵפַ֑ר הִכָּרֵ֧ת | תִּכָּרֵ֛ת הַנֶּ֥פֶשׁ הַהִ֖וא עֲוֹנָ֥ה בָֽהּ:  מ  כִּ֤י דְבַר-יְהֹוָה֙ בָּזָ֔ה וְאֶת-מִצְוָת֖וֹ הֵפַ֑ר הִכָּרֵ֧ת | תִּכָּרֵ֛ת הַנֶּ֥פֶשׁ הַהִ֖וא עֲוֹנָ֥ה בָֽהּ:  ת  אֲרֵי עַל מֵימְרָא דַיְיָ בְּסַר וְיָת פִּקוּדוֹהִי אַשְׁנֵי אִשְׁתֵּצָאָה תִשְׁתֵּצֵי אֲנָשָׁא הַהִיא חוֹבֵיהּ בֵּיהּ: (פ)

{לב} וַיִּֽהְי֥וּ בְנֵֽי-יִשְׂרָאֵ֖ל בַּמִּדְבָּ֑ר וַֽיִּמְצְא֗וּ אִ֛ישׁ מְקֹשֵׁ֥שׁ עֵצִ֖ים בְּי֥וֹם הַשַּׁבָּֽת:  מ  וַיִּֽהְי֥וּ בְנֵֽי-יִשְׂרָאֵ֖ל בַּמִּדְבָּ֑ר וַֽיִּמְצְא֗וּ אִ֛ישׁ מְקֹשֵׁ֥שׁ עֵצִ֖ים בְּי֥וֹם הַשַּׁבָּֽת:  ת  וַהֲווֹ בְנֵי יִשְׂרָאֵל בְּמַדְבְּרָא וְאַשְׁכָּחוּ גַבְרָא כַּד מְגַבֵּב אָעִין בְּיוֹמָא דְשַׁבְּתָא:

{לג} וַיַּקְרִ֣יבוּ אֹת֔וֹ הַמֹּֽצְאִ֥ים אֹת֖וֹ מְקֹשֵׁ֣שׁ עֵצִ֑ים אֶל-מֹשֶׁה֙ וְאֶֽל-אַהֲרֹ֔ן וְאֶ֖ל כָּל-הָֽעֵדָֽה:  מ  וַיַּקְרִ֣יבוּ אֹת֔וֹ הַמֹּֽצְאִ֥ים אֹת֖וֹ מְקֹשֵׁ֣שׁ עֵצִ֑ים אֶל-מֹשֶׁה֙ וְאֶֽל-אַהֲרֹ֔ן וְאֶ֖ל כָּל-הָֽעֵדָֽה:  ת  וְקָרִיבוּ יָתֵהּ דְאַשְׁכַּחוּ יָתֵהּ כַּד מְגַבֵּב אָעִין לְוָת משֶׁה וּלְוָת אַהֲרֹן וּלְוָת כָּל כְּנִשְׁתָּא:

{לד} וַיַּנִּ֥יחוּ אֹת֖וֹ בַּמִּשְׁמָ֑ר כִּ֚י לֹ֣א פֹרַ֔שׁ מַה-יֵּֽעָשֶׂ֖ה לֽוֹ:  מ  וַיַּנִּ֥יחוּ אֹת֖וֹ בַּמִּשְׁמָ֑ר כִּ֚י לֹ֣א פֹרַ֔שׁ מַה-יֵּֽעָשֶׂ֖ה לֽוֹ:  ת  וַאֲסָרוּ יָתֵהּ בְּבֵית מַטְרָא אֲרֵי לָא אִתְפְּרֵשׁ לְהוֹן מָה דְיַעֲבְּדוּן לֵהּ: (ס)

{לה} וַיֹּ֤אמֶר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה מ֥וֹת יוּמַ֖ת הָאִ֑ישׁ רָג֨וֹם אֹת֤וֹ בָֽאֲבָנִים֙ כָּל-הָ֣עֵדָ֔ה מִח֖וּץ לַֽמַּחֲנֶֽה:  מ  וַיֹּ֤אמֶר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה מ֥וֹת יוּמַ֖ת הָאִ֑ישׁ רָג֨וֹם אֹת֤וֹ בָֽאֲבָנִים֙ כָּל-הָ֣עֵדָ֔ה מִח֖וּץ לַֽמַּחֲנֶֽה:  ת  וַאֲמַר יְיָ לְמשֶׁה אִתְקְטָלָא יִתְקְטֵל גַבְרָא רְגוּמוּ יָתֵיהּ בְּאַבְנַיָא כָּל כְּנִשְׁתָּא מִבָּרָא לְמַשְׁרִיתָא:

{לו} וַיֹּצִ֨יאוּ אֹת֜וֹ כָּל-הָֽעֵדָ֗ה אֶל-מִחוּץ֙ לַֽמַּחֲנֶ֔ה וַיִּרְגְּמ֥וּ אֹת֛וֹ בָּֽאֲבָנִ֖ים וַיָּמֹ֑ת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת-מֹשֶֽׁה:  מ  וַיֹּצִ֨יאוּ אֹת֜וֹ כָּל-הָֽעֵדָ֗ה אֶל-מִחוּץ֙ לַֽמַּחֲנֶ֔ה וַיִּרְגְּמ֥וּ אֹת֛וֹ בָּֽאֲבָנִ֖ים וַיָּמֹ֑ת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת-מֹשֶֽׁה:  ת  וְאַפִּיקוּ יָתֵיהּ כָּל כְּנִשְׁתָּא לְמִבָּרָא לְמַשְׁרִיתָא וּרְגָמוּ יָתֵהּ בְּאַבְנַיָא וּמִית כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (פ)

{לז}  מפטיר  וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וַאֲמַר יְיָ לְמשֶׁה לְמֵימָר:

{לח} דַּבֵּ֞ר אֶל-בְּנֵ֤י יִשְׂרָאֵל֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם וְעָשׂ֨וּ לָהֶ֥ם צִיצִ֛ת עַל-כַּנְפֵ֥י בִגְדֵיהֶ֖ם לְדֹֽרֹתָ֑ם וְנָֽתְנ֛וּ עַל-צִיצִ֥ת הַכָּנָ֖ף פְּתִ֥יל תְּכֵֽלֶת:  מ  דַּבֵּ֞ר אֶל-בְּנֵ֤י יִשְׂרָאֵל֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם וְעָשׂ֨וּ לָהֶ֥ם צִיצִ֛ת עַל-כַּנְפֵ֥י בִגְדֵיהֶ֖ם לְדֹֽרֹתָ֑ם וְנָֽתְנ֛וּ עַל-צִיצִ֥ת הַכָּנָ֖ף פְּתִ֥יל תְּכֵֽלֶת:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן וְיַעֲבְּדוּן לְהוֹן כְּרוּסְפְּדִין עַל כַּנְפֵי כְסוּתְהוֹן לְדָרֵיהוֹן וְיִתְּנוּן עַל כְּרוּסְפְּדָא דְכַנְפָא חוּטָא דִתְכֶלְתָּא:

{לט} וְהָיָ֣ה לָכֶם֘ לְצִיצִת֒ וּרְאִיתֶ֣ם אֹת֗וֹ וּזְכַרְתֶּם֙ אֶת-כָּל-מִצְוֹ֣ת יְהֹוָ֔ה וַֽעֲשִׂיתֶ֖ם אֹתָ֑ם וְלֹֽא-תָת֜וּרוּ אַֽחֲרֵ֤י לְבַבְכֶם֙ וְאַֽחֲרֵ֣י עֵֽינֵיכֶ֔ם אֲשֶׁר-אַתֶּ֥ם זֹנִ֖ים אַֽחֲרֵיהֶֽם:  מ  וְהָיָ֣ה לָכֶם֘ לְצִיצִת֒ וּרְאִיתֶ֣ם אֹת֗וֹ וּזְכַרְתֶּם֙ אֶת-כָּל-מִצְוֹ֣ת יְהֹוָ֔ה וַֽעֲשִׂיתֶ֖ם אֹתָ֑ם וְלֹֽא-תָת֜וּרוּ אַֽחֲרֵ֤י לְבַבְכֶם֙ וְאַֽחֲרֵ֣י עֵֽינֵיכֶ֔ם אֲשֶׁר-אַתֶּ֥ם זֹנִ֖ים אַֽחֲרֵיהֶֽם:  ת  וִיהוֹן לְכוֹן לִכְרוּסְפְּדִין וְתֶחֱזוּן יָתֵיהּ וְתִדְכְּרוּן יָת כָּל פִּקוּדַיָא דַיְיָ וְתַעְבְּדוּן יָתְהוֹן וְלָא תִטְעוּן בָּתַר הִרְהוּר לִבְּכוֹן וּבָתַר חֵיזוּ עֵינֵיכוֹן דִי אַתּוּן טָּעַן בַּתְרֵיהוֹן:

{מ} לְמַ֣עַן תִּזְכְּר֔וּ וַֽעֲשִׂיתֶ֖ם אֶת-כָּל-מִצְוֹתָ֑י וִֽהְיִיתֶ֥ם קְדֹשִׁ֖ים לֵֽאלֹהֵיכֶֽם:  מ  לְמַ֣עַן תִּזְכְּר֔וּ וַֽעֲשִׂיתֶ֖ם אֶת-כָּל-מִצְוֹתָ֑י וִֽהְיִיתֶ֥ם קְדֹשִׁ֖ים לֵֽאלֹהֵיכֶֽם:  ת  בְּדִיל דְתִדְכְּרוּן וְתַעְבְּדוּן יָת כָּל פִּקוּדָי וּתְהוֹן קַדִישִׁין קֳדָם אֱלָהָכוֹן:

{מא} אֲנִ֞י יְהֹוָ֣ה אֱלֹֽהֵיכֶ֗ם אֲשֶׁ֨ר הוֹצֵ֤אתִי אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לִֽהְי֥וֹת לָכֶ֖ם לֵֽאלֹהִ֑ים אֲנִ֖י יְהֹוָ֥ה אֱלֹֽהֵיכֶֽם:  מ  אֲנִ֞י יְהֹוָ֣ה אֱלֹֽהֵיכֶ֗ם אֲשֶׁ֨ר הוֹצֵ֤אתִי אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לִֽהְי֥וֹת לָכֶ֖ם לֵֽאלֹהִ֑ים אֲנִ֖י יְהֹוָ֥ה אֱלֹֽהֵיכֶֽם:  ת  אֲנָא יְיָ אֱלָהָכוֹן דִי אַפֵּקִית יָתְכוֹן מֵאַרְעָא דְמִצְרַיִם לְמֵהֲוֵי לְכוֹן לֶאֱלָהָא אֲנָא יְיָ אֱלָהָכוֹן: (פפפ)
