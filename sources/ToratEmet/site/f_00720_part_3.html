סגולה שלא ישכח שמו ליום הדין : יאמר קודם יהיו לרצון השני בתפילת העמידה פסוק מהתנ''ך המתחיל באות הראשונה של שמו ומסתיים באות האחרונה של שמו או פסוק ששמו מופיע בו, להלן פסוקי התנ"ך ממויינים לפי האות הראשונה והאחרונה שלהם.

ג

ג-א


[משלי פרק-יט-ב]  גַּם בְּלֹא דַעַת נֶפֶשׁ לֹא טוֹב וְאָץ בְּרַגְלַיִם חוֹטֵא:

[איוב פרק-יג-טז]  גַּם הוּא לִי לִישׁוּעָה כִּי לֹא לְפָנָיו חָנֵף יָבוֹא:

[קהלת פרק-ה-יח]  גַּם כָּל הָאָדָם אֲשֶׁר נָתַן לוֹ הָאֱלֹהִים עֹשֶׁר וּנְכָסִים וְהִשְׁלִיטוֹ לֶאֱכֹל מִמֶּנּוּ וְלָשֵׂאת אֶת חֶלְקוֹ וְלִשְׂמֹחַ בַּעֲמָלוֹ זֹה מַתַּת אֱלֹהִים הִיא:


ג-ב


[בראשית פרק-מט-יט]  גָּד גְּדוּד יְגוּדֶנּוּ וְהוּא יָגֻד עָקֵב:

[נחום פרק-ג-יא]  גַּם אַתְּ תִּשְׁכְּרִי תְּהִי נַעֲלָמָה גַּם אַתְּ תְּבַקְשִׁי מָעוֹז מֵאוֹיֵב:

[תהילים פרק-יט-יב]  גַּם עַבְדְּךָ נִזְהָר בָּהֶם בְּשָׁמְרָם עֵקֶב רָב:

[תהילים פרק-יט-יד]  גַּם מִזֵּדִים חֲשֹׂךְ עַבְדֶּךָ אַל יִמְשְׁלוּ בִי אָז אֵיתָם וְנִקֵּיתִי מִפֶּשַׁע רָב:

[תהילים פרק-מא-י]  גַּם אִישׁ שְׁלוֹמִי אֲשֶׁר בָּטַחְתִּי בוֹ אוֹכֵל לַחְמִי הִגְדִּיל עָלַי עָקֵב:

[משלי פרק-כ-יט]  גּוֹלֶה סּוֹד הוֹלֵךְ רָכִיל וּלְפֹתֶה שְׂפָתָיו לֹא תִתְעָרָב:

[משלי פרק-כד-כג]  גַּם אֵלֶּה לַחֲכָמִים הַכֵּר פָּנִים בְּמִשְׁפָּט בַּל טוֹב:

[איכה פרק-ב-ג]  גָּדַע בָּחֳרִי אַף כֹּל קֶרֶן יִשְׂרָאֵל הֵשִׁיב אָחוֹר יְמִינוֹ מִפְּנֵי אוֹיֵב וַיִּבְעַר בְּיַעֲקֹב כְּאֵשׁ לֶהָבָה אָכְלָה סָבִיב:


ג-ד


[ירמיה פרק-ב-טז]  גַּם בְּנֵי נֹף (ותחפנס) וְתַחְפַּנְחֵס יִרְעוּךְ קָדְקֹד:

[תהילים פרק-ט-ו]  גָּעַרְתָּ גוֹיִם אִבַּדְתָּ רָשָׁע שְׁמָם מָחִיתָ לְעוֹלָם וָעֶד:

[משלי פרק-כט-כג]  גַּאֲוַת אָדָם תַּשְׁפִּילֶנּוּ וּשְׁפַל רוּחַ יִתְמֹךְ כָּבוֹד:


ג-ה


[בראשית פרק-לב-טז]  גְּמַלִּים מֵינִיקוֹת וּבְנֵיהֶם שְׁלֹשִׁים פָּרוֹת אַרְבָּעִים וּפָרִים עֲשָׂרָה אֲתֹנֹת עֶשְׂרִים וַעְיָרִם עֲשָׂרָה:

[דברים פרק-כב-יב]  גְּדִלִים תַּעֲשֶׂה לָּךְ עַל אַרְבַּע כַּנְפוֹת כְּסוּתְךָ אֲשֶׁר תְּכַסֶּה בָּהּ:

[שמואל א פרק-יב-כג]  גַּם אָנֹכִי חָלִילָה לִּי מֵחֲטֹא לַיהוָה מֵחֲדֹל לְהִתְפַּלֵּל בַּעַדְכֶם וְהוֹרֵיתִי אֶתְכֶם בְּדֶרֶךְ הַטּוֹבָה וְהַיְשָׁרָה:

[ישעיה פרק-כח-כט]  גַּם זֹאת מֵעִם יְהוָה צְבָאוֹת יָצָאָה הִפְלִיא עֵצָה הִגְדִּיל תּוּשִׁיָּה:

[ישעיה פרק-מג-יג]  גַּם מִיּוֹם אֲנִי הוּא וְאֵין מִיָּדִי מַצִּיל אֶפְעַל וּמִי יְשִׁיבֶנָּה:

[ירמיה פרק-ב-לד]  גַּם בִּכְנָפַיִךְ נִמְצְאוּ דַּם נַפְשׁוֹת אֶבְיוֹנִים נְקִיִּים לֹא בַמַּחְתֶּרֶת מְצָאתִים כִּי עַל כָּל אֵלֶּה:

[ירמיה פרק-ח-ז]  גַּם חֲסִידָה בַשָּׁמַיִם יָדְעָה מוֹעֲדֶיהָ וְתֹר (וסוס) וְסִיס וְעָגוּר שָׁמְרוּ אֶת עֵת בֹּאָנָה וְעַמִּי לֹא יָדְעוּ אֵת מִשְׁפַּט יְהוָה:

[הושע פרק-ט-ב]  גֹּרֶן וָיֶקֶב לֹא יִרְעֵם וְתִירוֹשׁ יְכַחֶשׁ בָּהּ:

[צפניה פרק-ב-יב]  גַּם אַתֶּם כּוּשִׁים חַלְלֵי חַרְבִּי הֵמָּה:

[תהילים פרק-לז-ה]  גּוֹל עַל יְהוָה דַּרְכֶּךָ וּבְטַח עָלָיו וְהוּא יַעֲשֶׂה:

[תהילים פרק-נב-ז]  גַּם אֵל יִתָּצְךָ לָנֶצַח יַחְתְּךָ וְיִסָּחֲךָ מֵאֹהֶל וְשֵׁרֶשְׁךָ מֵאֶרֶץ חַיִּים סֶלָה:

[תהילים פרק-סח-י]  גֶּשֶׁם נְדָבוֹת תָּנִיף אֱלֹהִים נַחֲלָתְךָ וְנִלְאָה אַתָּה כוֹנַנְתָּהּ:

[תהילים פרק-עז-טז]  גָּאַלְתָּ בִּזְרוֹעַ עַמֶּךָ בְּנֵי יַעֲקֹב וְיוֹסֵף סֶלָה:

[תהילים פרק-פ-ט]  גֶּפֶן מִמִּצְרַיִם תַּסִּיעַ תְּגָרֵשׁ גּוֹיִם וַתִּטָּעֶהָ:

[תהילים פרק-פג-ט]  גַּם אַשּׁוּר נִלְוָה עִמָּם הָיוּ זְרוֹעַ לִבְנֵי לוֹט סֶלָה:

[תהילים פרק-פה-יג]  גַּם יְהוָה יִתֵּן הַטּוֹב וְאַרְצֵנוּ תִּתֵּן יְבוּלָהּ:

[תהילים פרק-קלט-יב]  גַּם חֹשֶׁךְ לֹא יַחְשִׁיךְ מִמֶּךָ וְלַיְלָה כַּיּוֹם יָאִיר כַּחֲשֵׁיכָה כָּאוֹרָה:

[משלי פרק-יד-יג]  גַּם בִּשְׂחוֹק יִכְאַב לֵב וְאַחֲרִיתָהּ שִׂמְחָה תוּגָה:

[משלי פרק-כה-א]  גַּם אֵלֶּה מִשְׁלֵי שְׁלֹמֹה אֲשֶׁר הֶעְתִּיקוּ אַנְשֵׁי חִזְקִיָּה מֶלֶךְ יְהוּדָה:

[משלי פרק-לא-יב]  גְּמָלַתְהוּ טוֹב וְלֹא רָע כֹּל יְמֵי חַיֶּיה:

[איוב פרק-יב-ג]  גַּם לִי לֵבָב כְּמוֹכֶם לֹא נֹפֵל אָנֹכִי מִכֶּם וְאֶת מִי אֵין כְּמוֹ אֵלֶּה:

[איכה פרק-ג-ט]  גָּדַר דְּרָכַי בְּגָזִית נְתִיבֹתַי עִוָּה:

[קהלת פרק-ו-ה]  גַּם שֶׁמֶשׁ לֹא רָאָה וְלֹא יָדָע נַחַת לָזֶה מִזֶּה:

[נחמיה פרק-ד-טז]  גַּם בָּעֵת הַהִיא אָמַרְתִּי לָעָם אִישׁ וְנַעֲרוֹ יָלִינוּ בְּתוֹךְ יְרוּשָׁלִָם וְהָיוּ לָנוּ הַלַּיְלָה מִשְׁמָר וְהַיּוֹם מְלָאכָה:

[נחמיה פרק-ה-יג]  גַּם חָצְנִי נָעַרְתִּי וָאֹמְרָה כָּכָה יְנַעֵר הָאֱלֹהִים אֶת כָּל הָאִישׁ אֲשֶׁר לֹא יָקִים אֶת הַדָּבָר הַזֶּה מִבֵּיתוֹ וּמִיגִיעוֹ וְכָכָה יִהְיֶה נָעוּר וָרֵק וַיֹּאמְרוּ כָל הַקָּהָל אָמֵן וַיְהַלְלוּ אֶת יְהוָה וַיַּעַשׂ הָעָם כַּדָּבָר הַזֶּה:

[דברי הימים ב פרק-כא-יא]  גַּם הוּא עָשָׂה בָמוֹת בְּהָרֵי יְהוּדָה וַיֶּזֶן אֶת יֹשְׁבֵי יְרוּשָׁלִַם וַיַּדַּח אֶת יְהוּדָה:

[דברי הימים ב פרק-ל-יב]  גַּם בִּיהוּדָה הָיְתָה יַד הָאֱלֹהִים לָתֵת לָהֶם לֵב אֶחָד לַעֲשׂוֹת מִצְוַת הַמֶּלֶךְ וְהַשָּׂרִים בִּדְבַר יְהוָה:


ג-ו


[בראשית פרק-מט-ט]  גּוּר אַרְיֵה יְהוּדָה מִטֶּרֶף בְּנִי עָלִיתָ כָּרַע רָבַץ כְּאַרְיֵה וּכְלָבִיא מִי יְקִימֶנּוּ:

[במדבר פרק-לה-יט]  גֹּאֵל הַדָּם הוּא יָמִית אֶת הָרֹצֵחַ בְּפִגְעוֹ בוֹ הוּא יְמִיתֶנּוּ:

[דברים פרק-כט-כב]  גָּפְרִית וָמֶלַח שְׂרֵפָה כָל אַרְצָהּ לֹא תִזָּרַע וְלֹא תַצְמִחַ וְלֹא יַעֲלֶה בָהּ כָּל עֵשֶׂב כְּמַהְפֵּכַת סְדֹם וַעֲמֹרָה אַדְמָה (וצביים) וּצְבוֹיִם אֲשֶׁר הָפַךְ יְהוָה בְּאַפּוֹ וּבַחֲמָתוֹ:

[מלכים ב פרק-יז-יט]  גַּם יְהוּדָה לֹא שָׁמַר אֶת מִצְוֹת יְהוָה אֱלֹהֵיהֶם וַיֵּלְכוּ בְּחֻקּוֹת יִשְׂרָאֵל אֲשֶׁר עָשׂוּ:

[ישעיה פרק-יד-ח]  גַּם בְּרוֹשִׁים שָׂמְחוּ לְךָ אַרְזֵי לְבָנוֹן מֵאָז שָׁכַבְתָּ לֹא יַעֲלֶה הַכֹּרֵת עָלֵינוּ:

[ישעיה פרק-סו-ד]  גַּם אֲנִי אֶבְחַר בְּתַעֲלֻלֵיהֶם וּמְגוּרֹתָם אָבִיא לָהֶם יַעַן קָרָאתִי וְאֵין עוֹנֶה דִּבַּרְתִּי וְלֹא שָׁמֵעוּ וַיַּעֲשׂוּ הָרַע בְּעֵינַי וּבַאֲשֶׁר לֹא חָפַצְתִּי בָּחָרוּ:

[ירמיה פרק-ז-כט]  גָּזִּי נִזְרֵךְ וְהַשְׁלִיכִי וּשְׂאִי עַל שְׁפָיִם קִינָה כִּי מָאַס יְהוָה וַיִּטֹּשׁ אֶת דּוֹר עֶבְרָתוֹ:

[ירמיה פרק-לב-יט]  גְּדֹל הָעֵצָה וְרַב הָעֲלִילִיָּה אֲשֶׁר עֵינֶיךָ פְקֻחוֹת עַל כָּל דַּרְכֵי בְּנֵי אָדָם לָתֵת לְאִישׁ כִּדְרָכָיו וְכִפְרִי מַעֲלָלָיו:

[הושע פרק-י-ו]  גַּם אוֹתוֹ לְאַשּׁוּר יוּבָל מִנְחָה לְמֶלֶךְ יָרֵב בָּשְׁנָה אֶפְרַיִם יִקָּח וְיֵבוֹשׁ יִשְׂרָאֵל מֵעֲצָתוֹ:

[זכריה פרק-ט-יא]  גַּם אַתְּ בְּדַם בְּרִיתֵךְ שִׁלַּחְתִּי אֲסִירַיִךְ מִבּוֹר אֵין מַיִם בּוֹ:

[תהילים פרק-כא-ו]  גָּדוֹל כְּבוֹדוֹ בִּישׁוּעָתֶךָ הוֹד וְהָדָר תְּשַׁוֶּה עָלָיו:

[תהילים פרק-כב-ט]  גֹּל אֶל יְהוָה יְפַלְּטֵהוּ יַצִּילֵהוּ כִּי חָפֵץ בּוֹ:

[תהילים פרק-לד-ד]  גַּדְּלוּ לַיהוָה אִתִּי וּנְרוֹמְמָה שְׁמוֹ יַחְדָּו:

[תהילים פרק-מח-ב]  גָּדוֹל יְהוָה וּמְהֻלָּל מְאֹד בְּעִיר אֱלֹהֵינוּ הַר קָדְשׁוֹ:

[תהילים פרק-סח-לא]  גְּעַר חַיַּת קָנֶה עֲדַת אַבִּירִים בְּעֶגְלֵי עַמִּים מִתְרַפֵּס בְּרַצֵּי כָסֶף בִּזַּר עַמִּים קְרָבוֹת יֶחְפָּצוּ:

[משלי פרק-א-יד]  גּוֹרָלְךָ תַּפִּיל בְּתוֹכֵנוּ כִּיס אֶחָד יִהְיֶה לְכֻלָּנוּ:

[משלי פרק-כ-יא]  גַּם בְּמַעֲלָלָיו יִתְנַכֶּר נָעַר אִם זַךְ וְאִם יָשָׁר פָּעֳלוֹ:

[משלי פרק-כג-כד]  (גול) גִּיל (יגול) יָגִיל אֲבִי צַדִּיק (יולד) וְיוֹלֵד חָכָם (וישמח) יִשְׂמַח בּוֹ:

[משלי פרק-כט-ה]  גֶּבֶר מַחֲלִיק עַל רֵעֵהוּ רֶשֶׁת פּוֹרֵשׂ עַל פְּעָמָיו:

[איוב פרק-יח-ה]  גַּם אוֹר רְשָׁעִים יִדְעָךְ וְלֹא יִגַּהּ שְׁבִיב אִשּׁוֹ:

[איוב פרק-כד-ב]  גְּבֻלוֹת יַשִּׂיגוּ עֵדֶר גָּזְלוּ וַיִּרְעוּ:


ג-ח


[משלי פרק-כד-ה]  גֶּבֶר חָכָם בַּעוֹז וְאִישׁ דַּעַת מְאַמֶּץ כֹּחַ:

[איוב פרק-ל-ב]  גַּם כֹּחַ יְדֵיהֶם לָמָּה לִּי עָלֵימוֹ אָבַד כָּלַח:


ג-י


[בראשית פרק-כג-ד]  גֵּר וְתוֹשָׁב אָנֹכִי עִמָּכֶם תְּנוּ לִי אֲחֻזַּת קֶבֶר עִמָּכֶם וְאֶקְבְּרָה מֵתִי מִלְּפָנָי:

[שמות פרק-יב-לב]  גַּם צֹאנְכֶם גַּם בְּקַרְכֶם קְחוּ כַּאֲשֶׁר דִּבַּרְתֶּם וָלֵכוּ וּבֵרַכְתֶּם גַּם אֹתִי:

[שמואל א פרק-ב-טו]  גַּם בְּטֶרֶם יַקְטִרוּן אֶת הַחֵלֶב וּבָא נַעַר הַכֹּהֵן וְאָמַר לָאִישׁ הַזֹּבֵחַ תְּנָה בָשָׂר לִצְלוֹת לַכֹּהֵן וְלֹא יִקַּח מִמְּךָ בָּשָׂר מְבֻשָּׁל כִּי אִם חָי:

[ירמיה פרק-לג-כא]  גַּם בְּרִיתִי תֻפַר אֶת דָּוִד עַבְדִּי מִהְיוֹת לוֹ בֵן מֹלֵךְ עַל כִּסְאוֹ וְאֶת הַלְוִיִּם הַכֹּהֲנִים מְשָׁרְתָי:

[הושע פרק-ו-יא]  גַּם יְהוּדָה שָׁת קָצִיר לָךְ בְּשׁוּבִי שְׁבוּת עַמִּי:

[תהילים פרק-כג-ד]  גַּם כִּי אֵלֵךְ בְּגֵיא צַלְמָוֶת לֹא אִירָא רָע כִּי אַתָּה עִמָּדִי שִׁבְטְךָ וּמִשְׁעַנְתֶּךָ הֵמָּה יְנַחֲמֻנִי:

[תהילים פרק-עא-כד]  גַּם לְשׁוֹנִי כָּל הַיּוֹם תֶּהְגֶּה צִדְקָתֶךָ כִּי בֹשׁוּ כִי חָפְרוּ מְבַקְשֵׁי רָעָתִי:

[תהילים פרק-פד-ד]  גַּם צִפּוֹר מָצְאָה בַיִת וּדְרוֹר קֵן לָהּ אֲשֶׁר שָׁתָה אֶפְרֹחֶיהָ אֶת מִזְבְּחוֹתֶיךָ יְהוָה צְבָאוֹת מַלְכִּי וֵאלֹהָי:

[תהילים פרק-קיט-כב]  גַּל מֵעָלַי חֶרְפָּה וָבוּז כִּי עֵדֹתֶיךָ נָצָרְתִּי:

[תהילים פרק-קיט-כד]  גַּם עֵדֹתֶיךָ שַׁעֲשֻׁעָי אַנְשֵׁי עֲצָתִי:

[משלי פרק-יא-יז]  גֹּמֵל נַפְשׁוֹ אִישׁ חָסֶד וְעֹכֵר שְׁאֵרוֹ אַכְזָרִי:

[איוב פרק-ז-יא]  גַּם אֲנִי לֹא אֶחֱשָׂךְ פִּי אֲדַבְּרָה בְּצַר רוּחִי אָשִׂיחָה בְּמַר נַפְשִׁי:

[איוב פרק-טז-ד]  גַּם אָנֹכִי כָּכֶם אֲדַבֵּרָה לוּ יֵשׁ נַפְשְׁכֶם תַּחַת נַפְשִׁי אַחְבִּירָה עֲלֵיכֶם בְּמִלִּים וְאָנִיעָה עֲלֵיכֶם בְּמוֹ רֹאשִׁי:

[איוב פרק-יט-יח]  גַּם עֲוִילִים מָאֲסוּ בִי אָקוּמָה וַיְדַבְּרוּ בִי:

[איוב פרק-כג-ב]  גַּם הַיּוֹם מְרִי שִׂחִי יָדִי כָּבְדָה עַל אַנְחָתִי:

[איכה פרק-ג-ז]  גָּדַר בַּעֲדִי וְלֹא אֵצֵא הִכְבִּיד נְחָשְׁתִּי:

[איכה פרק-ג-ח]  גַּם כִּי אֶזְעַק וַאֲשַׁוֵּעַ שָׂתַם תְּפִלָּתִי:

[קהלת פרק-ט-יג]  גַּם זֹה רָאִיתִי חָכְמָה תַּחַת הַשָּׁמֶשׁ וּגְדוֹלָה הִיא אֵלָי:

[נחמיה פרק-ה-יד]  גַּם מִיּוֹם אֲשֶׁר צִוָּה אֹתִי לִהְיוֹת פֶּחָם בְּאֶרֶץ יְהוּדָה מִשְּׁנַת עֶשְׂרִים וְעַד שְׁנַת שְׁלֹשִׁים וּשְׁתַּיִם לְאַרְתַּחְשַׁסְתְּא הַמֶּלֶךְ שָׁנִים שְׁתֵּים עֶשְׂרֵה אֲנִי וְאַחַי לֶחֶם הַפֶּחָה לֹא אָכַלְתִּי:

[נחמיה פרק-ו-יט]  גַּם טוֹבֹתָיו הָיוּ אֹמְרִים לְפָנַי וּדְבָרַי הָיוּ מוֹצִיאִים לוֹ אִגְּרוֹת שָׁלַח טוֹבִיָּה לְיָרְאֵנִי:


ג-כ


[בראשית פרק-כו-ג]  גּוּר בָּאָרֶץ הַזֹּאת וְאֶהְיֶה עִמְּךָ וַאֲבָרְכֶךָּ כִּי לְךָ וּלְזַרְעֲךָ אֶתֵּן אֶת כָּל הָאֲרָצֹת הָאֵל וַהֲקִמֹתִי אֶת הַשְּׁבֻעָה אֲשֶׁר נִשְׁבַּעְתִּי לְאַבְרָהָם אָבִיךָ:

[דברים פרק-כח-סא]  גַּם כָּל חֳלִי וְכָל מַכָּה אֲשֶׁר לֹא כָתוּב בְּסֵפֶר הַתּוֹרָה הַזֹּאת יַעְלֵם יְהוָה עָלֶיךָ עַד הִשָּׁמְדָךְ:

[ישעיה פרק-מח-ח]  גַּם לֹא שָׁמַעְתָּ גַּם לֹא יָדַעְתָּ גַּם מֵאָז לֹא פִתְּחָה אָזְנֶךָ כִּי יָדַעְתִּי בָּגוֹד תִּבְגּוֹד וּפֹשֵׁעַ מִבֶּטֶן קֹרָא לָךְ:

[יחזקאל פרק-טז-נב]  גַּם אַתְּ שְׂאִי כְלִמָּתֵךְ אֲשֶׁר פִּלַּלְתְּ לַאֲחוֹתֵךְ בְּחַטֹּאתַיִךְ אֲשֶׁר הִתְעַבְתְּ מֵהֵן תִּצְדַּקְנָה מִמֵּךְ וְגַם אַתְּ בּוֹשִׁי וּשְׂאִי כְלִמָּתֵךְ בְּצַדֶּקְתֵּךְ אַחְיוֹתֵךְ:

[יחזקאל פרק-כח-יז]  גָּבַהּ לִבְּךָ בְּיָפְיֶךָ שִׁחַתָּ חָכְמָתְךָ עַל יִפְעָתֶךָ עַל אֶרֶץ הִשְׁלַכְתִּיךָ לִפְנֵי מְלָכִים נְתַתִּיךָ לְרַאֲוָה בָךְ:

[יחזקאל פרק-לח-ו]  גֹּמֶר וְכָל אֲגַפֶּיהָ בֵּית תּוֹגַרְמָה יַרְכְּתֵי צָפוֹן וְאֶת כָּל אֲגַפָּיו עַמִּים רַבִּים אִתָּךְ:

[תהילים פרק-קיב-ב]  גִּבּוֹר בָּאָרֶץ יִהְיֶה זַרְעוֹ דּוֹר יְשָׁרִים יְבֹרָךְ:

[תהילים פרק-קיט-יז]  גְּמֹל עַל עַבְדְּךָ אֶחְיֶה וְאֶשְׁמְרָה דְבָרֶךָ:

[תהילים פרק-קיט-יח]  גַּל עֵינַי וְאַבִּיטָה נִפְלָאוֹת מִתּוֹרָתֶךָ:

[תהילים פרק-קיט-יט]  גֵּר אָנֹכִי בָאָרֶץ אַל תַּסְתֵּר מִמֶּנִּי מִצְוֹתֶיךָ:

[תהילים פרק-קיט-כא]  גָּעַרְתָּ זֵדִים אֲרוּרִים הַשֹּׁגִים מִמִּצְוֹתֶיךָ:

[תהילים פרק-קיט-כג]  גַּם יָשְׁבוּ שָׂרִים בִּי נִדְבָּרוּ עַבְדְּךָ יָשִׂיחַ בְּחֻקֶּיךָ:

[תהילים פרק-קלט-י]  גַּם שָׁם יָדְךָ תַנְחֵנִי וְתֹאחֲזֵנִי יְמִינֶךָ:

[משלי פרק-טז-ג]  גֹּל אֶל יְהוָה מַעֲשֶׂיךָ וְיִכֹּנוּ מַחְשְׁבֹתֶיךָ:

[קהלת פרק-ז-כא]  גַּם לְכָל הַדְּבָרִים אֲשֶׁר יְדַבֵּרוּ אַל תִּתֵּן לִבֶּךָ אֲשֶׁר לֹא תִשְׁמַע אֶת עַבְדְּךָ מְקַלְלֶךָ:


ג-ל


[שמואל ב פרק-ה-ב]  גַּם אֶתְמוֹל גַּם שִׁלְשׁוֹם בִּהְיוֹת שָׁאוּל מֶלֶךְ עָלֵינוּ אַתָּה (הייתה) הָיִיתָ (מוציא) הַמּוֹצִיא (והמבי) וְהַמֵּבִיא אֶת יִשְׂרָאֵל וַיֹּאמֶר יְהוָה לְךָ אַתָּה תִרְעֶה אֶת עַמִּי אֶת יִשְׂרָאֵל וְאַתָּה תִּהְיֶה לְנָגִיד עַל יִשְׂרָאֵל:

[ישעיה פרק-מז-ד]  גֹּאֲלֵנוּ יְהוָה צְבָאוֹת שְׁמוֹ קְדוֹשׁ יִשְׂרָאֵל:

[ירמיה פרק-נ-לד]  גֹּאֲלָם חָזָק יְהוָה צְבָאוֹת שְׁמוֹ רִיב יָרִיב אֶת רִיבָם לְמַעַן הִרְגִּיעַ אֶת הָאָרֶץ וְהִרְגִּיז לְיֹשְׁבֵי בָבֶל:

[נחום פרק-א-ד]  גּוֹעֵר בַּיָּם וַיַּבְּשֵׁהוּ וְכָל הַנְּהָרוֹת הֶחֱרִיב אֻמְלַל בָּשָׁן וְכַרְמֶל וּפֶרַח לְבָנוֹן אֻמְלָל:

[תהילים פרק-עא-כב]  גַּם אֲנִי אוֹדְךָ בִכְלִי נֶבֶל אֲמִתְּךָ אֱלֹהָי אֲזַמְּרָה לְךָ בְכִנּוֹר קְדוֹשׁ יִשְׂרָאֵל:

[איוב פרק-לא-כח]  גַּם הוּא עָוֹן פְּלִילִי כִּי כִחַשְׁתִּי לָאֵל מִמָּעַל:

[דברי הימים א פרק-יא-ב]  גַּם תְּמוֹל גַּם שִׁלְשׁוֹם גַּם בִּהְיוֹת שָׁאוּל מֶלֶךְ אַתָּה הַמּוֹצִיא וְהַמֵּבִיא אֶת יִשְׂרָאֵל וַיֹּאמֶר יְהוָה אֱלֹהֶיךָ לְךָ אַתָּה תִרְעֶה אֶת עַמִּי אֶת יִשְׂרָאֵל וְאַתָּה תִּהְיֶה נָגִיד עַל עַמִּי יִשְׂרָאֵל:

[דברי הימים ב פרק-כט-ז]  גַּם סָגְרוּ דַּלְתוֹת הָאוּלָם וַיְכַבּוּ אֶת הַנֵּרוֹת וּקְטֹרֶת לֹא הִקְטִירוּ וְעֹלָה לֹא הֶעֱלוּ בַקֹּדֶשׁ לֵאלֹהֵי יִשְׂרָאֵל:


ג-מ


[דברים פרק-א-לז]  גַּם בִּי הִתְאַנַּף יְהוָה בִּגְלַלְכֶם לֵאמֹר גַּם אַתָּה לֹא תָבֹא שָׁם:

[שמואל א פרק-יב-טז]  גַּם עַתָּה הִתְיַצְּבוּ וּרְאוּ אֶת הַדָּבָר הַגָּדוֹל הַזֶּה אֲשֶׁר יְהוָה עֹשֶׂה לְעֵינֵיכֶם:

[שמואל א פרק-יז-לו]  גַּם אֶת הָאֲרִי גַּם הַדּוֹב הִכָּה עַבְדֶּךָ וְהָיָה הַפְּלִשְׁתִּי הֶעָרֵל הַזֶּה כְּאַחַד מֵהֶם כִּי חֵרֵף מַעַרְכֹת אֱלֹהִים חַיִּים:

[ירמיה פרק-ב-לז]  גַּם מֵאֵת זֶה תֵּצְאִי וְיָדַיִךְ עַל רֹאשֵׁךְ כִּי מָאַס יְהוָֹה בְּמִבְטַחַיִךְ וְלֹא תַצְלִיחִי לָהֶם:

[ירמיה פרק-ז-ו]  גֵּר יָתוֹם וְאַלְמָנָה לֹא תַעֲשֹׁקוּ וְדָם נָקִי אַל תִּשְׁפְּכוּ בַּמָּקוֹם הַזֶּה וְאַחֲרֵי אֱלֹהִים אֲחֵרִים לֹא תֵלְכוּ לְרַע לָכֶם:

[ירמיה פרק-לג-כו]  גַּם זֶרַע יַעֲקוֹב וְדָוִד עַבְדִּי אֶמְאַס מִקַּחַת מִזַּרְעוֹ מֹשְׁלִים אֶל זֶרַע אַבְרָהָם יִשְׂחָק וְיַעֲקֹב כִּי (אשוב) אָשִׁיב אֶת שְׁבוּתָם וְרִחַמְתִּים:

[ירמיה פרק-מו-כא]  גַּם שְׂכִרֶיהָ בְקִרְבָּהּ כְּעֶגְלֵי מַרְבֵּק כִּי גַם הֵמָּה הִפְנוּ נָסוּ יַחְדָּיו לֹא עָמָדוּ כִּי יוֹם אֵידָם בָּא עֲלֵיהֶם עֵת פְּקֻדָּתָם:

[יחזקאל פרק-לא-יז]  גַּם הֵם אִתּוֹ יָרְדוּ שְׁאוֹלָה אֶל חַלְלֵי חָרֶב וּזְרֹעוֹ יָשְׁבוּ בְצִלּוֹ בְּתוֹךְ גּוֹיִם:

[הושע פרק-ו-ח]  גִּלְעָד קִרְיַת פֹּעֲלֵי אָוֶן עֲקֻבָּה מִדָּם:

[הושע פרק-ח-י]  גַּם כִּי יִתְנוּ בַגּוֹיִם עַתָּה אֲקַבְּצֵם וַיָּחֵלּוּ מְּעָט מִמַּשָּׂא מֶלֶךְ שָׂרִים:

[נחום פרק-ג-י]  גַּם הִיא לַגֹּלָה הָלְכָה בַשֶּׁבִי גַּם עֹלָלֶיהָ יְרֻטְּשׁוּ בְּרֹאשׁ כָּל חוּצוֹת וְעַל נִכְבַּדֶּיהָ יַדּוּ גוֹרָל וְכָל גְּדוֹלֶיהָ רֻתְּקוּ בַזִּקִּים:

[תהילים פרק-כה-ג]  גַּם כָּל קוֶֹיךָ לֹא יֵבֹשׁוּ יֵבֹשׁוּ הַבּוֹגְדִים רֵיקָם:

[תהילים פרק-קיא-ב]  גְּדֹלִים מַעֲשֵׂי יְהוָה דְּרוּשִׁים לְכָל חֶפְצֵיהֶם:

[תהילים פרק-קלט-טז]  גָּלְמִי רָאוּ עֵינֶיךָ וְעַל סִפְרְךָ כֻּלָּם יִכָּתֵבוּ יָמִים יֻצָּרוּ (ולא) וְלוֹ אֶחָד בָּהֶם:

[משלי פרק-א-כו]  גַּם אֲנִי בְּאֵידְכֶם אֶשְׂחָק אֶלְעַג בְּבֹא פַחְדְּכֶם:

[משלי פרק-יד-כ]  גַּם לְרֵעֵהוּ יִשָּׂנֵא רָשׁ וְאֹהֲבֵי עָשִׁיר רַבִּים:

[משלי פרק-כז-כה]  גָּלָה חָצִיר וְנִרְאָה דֶשֶׁא וְנֶאֶסְפוּ עִשְּׂבוֹת הָרִים:

[משלי פרק-כח-ג]  גֶּבֶר רָשׁ וְעֹשֵׁק דַּלִּים מָטָר סֹחֵף וְאֵין לָחֶם:

[איוב פרק-טו-י]  גַּם שָׂב גַּם יָשִׁישׁ בָּנוּ כַּבִּיר מֵאָבִיךָ יָמִים:

[איוב פרק-טז-יט]  גַּם עַתָּה הִנֵּה בַשָּׁמַיִם עֵדִי וְשָׂהֲדִי בַּמְּרוֹמִים:

[איוב פרק-יט-טו]  גָּרֵי בֵיתִי וְאַמְהֹתַי לְזָר תַּחְשְׁבֻנִי נָכְרִי הָיִיתִי בְעֵינֵיהֶם:

[שיר השירים פרק-ד-יב]  גַּן נָעוּל אֲחֹתִי כַלָּה גַּל נָעוּל מַעְיָן חָתוּם:

[איכה פרק-א-ג]  גָּלְתָה יְהוּדָה מֵעֹנִי וּמֵרֹב עֲבֹדָה הִיא יָשְׁבָה בַגּוֹיִם לֹא מָצְאָה מָנוֹחַ כָּל רֹדְפֶיהָ הִשִּׂיגוּהָ בֵּין הַמְּצָרִים:

[קהלת פרק-ד-יא]  גַּם אִם יִשְׁכְּבוּ שְׁנַיִם וְחַם לָהֶם וּלְאֶחָד אֵיךְ יֵחָם:

[קהלת פרק-יב-ה]  גַּם מִגָּבֹהַּ יִרָאוּ וְחַתְחַתִּים בַּדֶּרֶךְ וְיָנֵאץ הַשָּׁקֵד וְיִסְתַּבֵּל הֶחָגָב וְתָפֵר הָאֲבִיּוֹנָה כִּי הֹלֵךְ הָאָדָם אֶל בֵּית עוֹלָמוֹ וְסָבְבוּ בַשּׁוּק הַסֹּפְדִים:

[עזרא פרק-ב-סז]  גְּמַלֵּיהֶם אַרְבַּע מֵאוֹת שְׁלֹשִׁים וַחֲמִשָּׁה חֲמֹרִים שֵׁשֶׁת אֲלָפִים שְׁבַע מֵאוֹת וְעֶשְׂרִים:

[נחמיה פרק-ו-יז]  גַּם בַּיָּמִים הָהֵם מַרְבִּים חֹרֵי יְהוּדָה אִגְּרֹתֵיהֶם הוֹלְכוֹת עַל טוֹבִיָּה וַאֲשֶׁר לְטוֹבִיָּה בָּאוֹת אֲלֵיהֶם:

[נחמיה פרק-ז-סח]  גְּמַלִּים אַרְבַּע מֵאוֹת שְׁלֹשִׁים וַחֲמִשָּׁה חֲמֹרִים שֵׁשֶׁת אֲלָפִים שְׁבַע מֵאוֹת וְעֶשְׂרִים:

[דברי הימים ב פרק-כב-ה]  גַּם בַּעֲצָתָם הָלַךְ וַיֵּלֶךְ אֶת יְהוֹרָם בֶּן אַחְאָב מֶלֶךְ יִשְׂרָאֵל לַמִּלְחָמָה עַל חֲזָאֵל מֶלֶךְ אֲרָם בְּרָמוֹת גִּלְעָד וַיַּכּוּ הָרַמִּים אֶת יוֹרָם:

[דברי הימים ב פרק-לו-יד]  גַּם כָּל שָׂרֵי הַכֹּהֲנִים וְהָעָם הִרְבּוּ (למעול) לִמְעָל מַעַל כְּכֹל תֹּעֲבוֹת הַגּוֹיִם וַיְטַמְּאוּ אֶת בֵּית יְהוָה אֲשֶׁר הִקְדִּישׁ בִּירוּשָׁלִָם:


ג-נ


[דברים פרק-כח-נ]  גּוֹי עַז פָּנִים אֲשֶׁר לֹא יִשָּׂא פָנִים לְזָקֵן וְנַעַר לֹא יָחֹן:

[שופטים פרק-ה-יז]  גִּלְעָד בְּעֵבֶר הַיַּרְדֵּן שָׁכֵן וְדָן לָמָּה יָגוּר אֳנִיּוֹת אָשֵׁר יָשַׁב לְחוֹף יַמִּים וְעַל מִפְרָצָיו יִשְׁכּוֹן:

[ישעיה פרק-ג-ב]  גִּבּוֹר וְאִישׁ מִלְחָמָה שׁוֹפֵט וְנָבִיא וְקֹסֵם וְזָקֵן:

[תהילים פרק-מט-ג]  גַּם בְּנֵי אָדָם גַּם בְּנֵי אִישׁ יַחַד עָשִׁיר וְאֶבְיוֹן:

[משלי פרק-יז-כח]  גַּם אֱוִיל מַחֲרִישׁ חָכָם יֵחָשֵׁב אֹטֵם שְׂפָתָיו נָבוֹן:

[משלי פרק-כב-י]  גָּרֵשׁ לֵץ וְיֵצֵא מָדוֹן וְיִשְׁבֹּת דִּין וְקָלוֹן:

[איוב פרק-יט-כט]  גּוּרוּ לָכֶם מִפְּנֵי חֶרֶב כִּי חֵמָה עֲוֹנוֹת חָרֶב לְמַעַן תֵּדְעוּן (שדין) שַׁדּוּן:


ג-ע


[איוב פרק-יא-ח]  גָּבְהֵי שָׁמַיִם מַה תִּפְעָל עֲמֻקָּה מִשְּׁאוֹל מַה תֵּדָע:

[דברי הימים ב פרק-כב-ג]  גַּם הוּא הָלַךְ בְּדַרְכֵי בֵּית אַחְאָב כִּי אִמּוֹ הָיְתָה יוֹעַצְתּוֹ לְהַרְשִׁיעַ:


ג-פ


[משלי פרק-יט-יט]  (גרל) גְּדָל חֵמָה נֹשֵׂא עֹנֶשׁ כִּי אִם תַּצִּיל וְעוֹד תּוֹסִף:

[קהלת פרק-ה-טז]  גַּם כָּל יָמָיו בַּחֹשֶׁךְ יֹאכֵל וְכָעַס הַרְבֵּה וְחָלְיוֹ וָקָצֶף:


ג-צ


[בראשית פרק-ז-ג]  גַּם מֵעוֹף הַשָּׁמַיִם שִׁבְעָה שִׁבְעָה זָכָר וּנְקֵבָה לְחַיּוֹת זֶרַע עַל פְּנֵי כָל הָאָרֶץ:

[מלכים א פרק-ד-יט]  גֶּבֶר בֶּן אֻרִי בְּאֶרֶץ גִּלְעָד אֶרֶץ סִיחוֹן מֶלֶךְ הָאֱמֹרִי וְעֹג מֶלֶךְ הַבָּשָׁן וּנְצִיב אֶחָד אֲשֶׁר בָּאָרֶץ:

[ירמיה פרק-נא-מט]  גַּם בָּבֶל לִנְפֹּל חַלְלֵי יִשְׂרָאֵל גַּם לְבָבֶל נָפְלוּ חַלְלֵי כָל הָאָרֶץ:

[צפניה פרק-א-יח]  גַּם כַּסְפָּם גַּם זְהָבָם לֹא יוּכַל לְהַצִּילָם בְּיוֹם עֶבְרַת יְהוָה וּבְאֵשׁ קִנְאָתוֹ תֵּאָכֵל כָּל הָאָרֶץ כִּי כָלָה אַךְ נִבְהָלָה יַעֲשֶׂה אֵת כָּל יֹשְׁבֵי הָאָרֶץ:

[תהילים פרק-קד-ט]  גְּבוּל שַׂמְתָּ בַּל יַעֲבֹרוּן בַּל יְשׁוּבוּן לְכַסּוֹת הָאָרֶץ:


ג-ק


[ישעיה פרק-נ-ו]  גֵּוִי נָתַתִּי לְמַכִּים וּלְחָיַי לְמֹרְטִים פָּנַי לֹא הִסְתַּרְתִּי מִכְּלִמּוֹת וָרֹק:

[דברי הימים א פרק-יח-יא]  גַּם אֹתָם הִקְדִּישׁ הַמֶּלֶךְ דָּוִיד לַיהוָה עִם הַכֶּסֶף וְהַזָּהָב אֲשֶׁר נָשָׂא מִכָּל הַגּוֹיִם מֵאֱדוֹם וּמִמּוֹאָב וּמִבְּנֵי עַמּוֹן וּמִפְּלִשְׁתִּים וּמֵעֲמָלֵק:


ג-ר


[יואל פרק-א-כ]  גַּם בַּהֲמוֹת שָׂדֶה תַּעֲרוֹג אֵלֶיךָ כִּי יָבְשׁוּ אֲפִיקֵי מָיִם וְאֵשׁ אָכְלָה נְאוֹת הַמִּדְבָּר:

[תהילים פרק-פג-ח]  גְּבָל וְעַמּוֹן וַעֲמָלֵק פְּלֶשֶׁת עִם יֹשְׁבֵי צוֹר:

[תהילים פרק-קמה-ג]  גָּדוֹל יְהוָה וּמְהֻלָּל מְאֹד וְלִגְדֻלָּתוֹ אֵין חֵקֶר:

[תהילים פרק-קמז-ה]  גָּדוֹל אֲדוֹנֵינוּ וְרַב כֹּחַ לִתְבוּנָתוֹ אֵין מִסְפָּר:

[משלי פרק-יז-כו]  גַּם עֲנוֹשׁ לַצַּדִּיק לֹא טוֹב לְהַכּוֹת נְדִיבִים עַל יֹשֶׁר:

[איוב פרק-מא-ז]  גַּאֲוָה אֲפִיקֵי מָגִנִּים סָגוּר חוֹתָם צָר:

[איכה פרק-ד-ג]  גַּם (תנין) תַּנִּים חָלְצוּ שַׁד הֵינִיקוּ גּוּרֵיהֶן בַּת עַמִּי לְאַכְזָר (כי) (ענים) כַּיְעֵנִים בַּמִּדְבָּר:

[קהלת פרק-י-כ]  גַּם בְּמַדָּעֲךָ מֶלֶךְ אַל תְּקַלֵּל וּבְחַדְרֵי מִשְׁכָּבְךָ אַל תְּקַלֵּל עָשִׁיר כִּי עוֹף הַשָּׁמַיִם יוֹלִיךְ אֶת הַקּוֹל וּבַעַל (הכנפים) כְּנָפַיִם יַגֵּיד דָּבָר:


ג-ש


[שמואל ב פרק-ח-יא]  גַּם אֹתָם הִקְדִּישׁ הַמֶּלֶךְ דָּוִד לַיהוָה עִם הַכֶּסֶף וְהַזָּהָב אֲשֶׁר הִקְדִּישׁ מִכָּל הַגּוֹיִם אֲשֶׁר כִּבֵּשׁ:

[קהלת פרק-ט-ו]  גַּם אַהֲבָתָם גַּם שִׂנְאָתָם גַּם קִנְאָתָם כְּבָר אָבָדָה וְחֵלֶק אֵין לָהֶם עוֹד לְעוֹלָם בְּכֹל אֲשֶׁר נַעֲשָׂה תַּחַת הַשָּׁמֶשׁ:

[אסתר פרק-א-ט]  גַּם וַשְׁתִּי הַמַּלְכָּה עָשְׂתָה מִשְׁתֵּה נָשִׁים בֵּית הַמַּלְכוּת אֲשֶׁר לַמֶּלֶךְ אֲחַשְׁוֵרוֹשׁ:


ג-ת


[יהושע פרק-יח-כה]  גִּבְעוֹן וְהָרָמָה וּבְאֵרוֹת:

[שופטים פרק-ב-כא]  גַּם אֲנִי לֹא אוֹסִיף לְהוֹרִישׁ אִישׁ מִפְּנֵיהֶם מִן הַגּוֹיִם אֲשֶׁר עָזַב יְהוֹשֻׁעַ וַיָּמֹת:

[יחזקאל פרק-כ-כג]  גַּם אֲנִי נָשָׂאתִי אֶת יָדִי לָהֶם בַּמִּדְבָּר לְהָפִיץ אֹתָם בַּגּוֹיִם וּלְזָרוֹת אוֹתָם בָּאֲרָצוֹת:

[הושע פרק-י-א]  גֶּפֶן בּוֹקֵק יִשְׂרָאֵל פְּרִי יְשַׁוֶּה לּוֹ כְּרֹב לְפִרְיוֹ הִרְבָּה לַמִּזְבְּחוֹת כְּטוֹב לְאַרְצוֹ הֵיטִיבוּ מַצֵּבוֹת:

[חגי פרק-ב-ט]  גָּדוֹל יִהְיֶה כְּבוֹד הַבַּיִת הַזֶּה הָאַחֲרוֹן מִן הָרִאשׁוֹן אָמַר יְהוָה צְבָאוֹת וּבַמָּקוֹם הַזֶּה אֶתֵּן שָׁלוֹם נְאֻם יְהוָה צְבָאוֹת:

[זכריה פרק-ט-ט]  גִּילִי מְאֹד בַּת צִיּוֹן הָרִיעִי בַּת יְרוּשָׁלִַם הִנֵּה מַלְכֵּךְ יָבוֹא לָךְ צַדִּיק וְנוֹשָׁע הוּא עָנִי וְרֹכֵב עַל חֲמוֹר וְעַל עַיִר בֶּן אֲתֹנוֹת:

[תהילים פרק-קיט-כ]  גָּרְסָה נַפְשִׁי לְתַאֲבָה אֶל מִשְׁפָּטֶיךָ בְכָל עֵת:

[משלי פרק-יח-ט]  גַּם מִתְרַפֶּה בִמְלַאכְתּוֹ אָח הוּא לְבַעַל מַשְׁחִית:

[משלי פרק-כח-כד]  גּוֹזֵל אָבִיו וְאִמּוֹ וְאֹמֵר אֵין פָּשַׁע חָבֵר הוּא לְאִישׁ מַשְׁחִית:

[נחמיה פרק-יג-כג]  גַּם בַּיָּמִים הָהֵם רָאִיתִי אֶת הַיְּהוּדִים הֹשִׁיבוּ נָשִׁים (אשדודיות) אַשְׁדֳּדִיּוֹת (עמוניות) עַמֳּנִיּוֹת מוֹאֲבִיּוֹת:
