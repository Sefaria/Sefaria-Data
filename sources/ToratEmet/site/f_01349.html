פרק א - משנה א

{א} שֶׁאֵין בַּמִּזְבֵּחַ כִּי אִם שְׁנֵי מְקוֹמוֹת. תּוֹסָפוֹת: {ב} וּבְמִצּוּי פְּלוּגְתָּא בִּמְעִילָה דַּף ח'. דְּאִיכָּא לְמַאן דְּאָמַר דְּלֹא מְעַכֵּב. מְפָרֵשׁ: {ג} קִנִּים. קֵן קָרוּי שְׁתֵּי תוֹרִים אוֹ שְׁנֵי בְנֵי יוֹנָה. פְּרִידָה קְרוּיָה יוֹנָה אַחַת אוֹ תּוֹר אֶחָד. מְפָרֵשׁ: {ד} קִנִּים. הַקּוֹ"ף בְּחִירִי"ק, כְּמוֹ חֵץ חִצִּים. וְקַן צִפּוֹר שֶׁבְּפַתַּ"ח, מִפְּנֵי הַסְּמִיכוּת: {ה} הָרַמְבַּ"ם. וְגַם כֵּן שֶׁאֵינוֹ מְפֹרָשׁ בַּתּוֹרָה. וְכָתַב עוֹד הָרַמְבַּ"ם, וּמַה שֶּׁאָמַרְנוּ שֶׁהַחוֹבָה אֶחָד עוֹלָה וְאֶחָד חַטָּאת הֲרֵי הוּא כְּפִי עִקַּר הַחִיּוּב הָרִאשׁוֹן. וְאִם כְּבָר הִקְרִיב מִקְצָת קָרְבְּנוֹתָיו עַד שֶׁיִּהְיֶה הָאִישׁ הַזֶּה דֶּרֶךְ מָשָׁל יֵשׁ עָלָיו עֶשְׂרִים עוֹלוֹת וְעֶשֶׂר חַטָּאוֹת כֻּלָּן חוֹבָה, אוֹ בְהֶפֶךְ, וּלְפִיכָךְ אָמַר בְּמַסֶּכֶת זוֹ מִנְיַן חַטָּאוֹת שֶׁבַּחוֹבָה וּמִנְיַן עוֹלוֹת שֶׁבַּחוֹבָה, וְלֹא אָמַר חֲצִי הַחוֹבָה:

{ו} יָמוּתוּ כוּ'. דִּבְכָל אֶחָד אִיכָּא לְסַפּוֹקֵי שֶׁמָּא עוֹלָה וּפְסוּלָה לְמַטָּה, אוֹ שֶׁמָּא חַטָּאת וּפְסוּלָה לְמַעְלָה. רַשִׁ"י: {ז} וּנְדָבָה דְנָקַט, וְלֹא נָקַט חַטָּאת מֻעֶטֶת בַּחוֹבָה עוֹלָה מֻעֶטֶת בַּחוֹבָה, מִשּׁוּם דְּקָתָנֵי בַסֵּיפָא אֲבָל חוֹבָה שֶׁנִּתְעָרְבָה זוֹ בָזוֹ, לְהָכִי נָקַט נְדָבָה. מְפָרֵשׁ:

{ח} לָזוֹ כוּ'. נָתַן דֻּגְמָאוֹת לְכָל הַמַּסֶּכְתָּא בְּקָרְבְּנוֹת הַנָּשִׁים, לְפִי שֶׁהֵן יוֹתֵר חַיָּבוֹת בְּקִנִּים, מִפְּנֵי שֶׁחַיָּבוֹת בָּהֶן כְּמוֹ שֶׁחַיָּבִים הָאֲנָשִׁים, וּמוֹסִיף עַל הָאֲנָשִׁים יוֹלְדוֹת, וְגַם כֵּן שֶׁהַזִּיבוּת מָצוּי הַרְבֵּה בְנָשִׁים יוֹתֵר מִמַּה שֶׁהוּא מָצוּי בָּאֲנָשִׁים כְּמוֹ שֶׁהוּא מְפֻרְסָם וְנִגְלֶה לָעַיִן. הָרַמְבַּ"ם: {ט} מֶחֱצָה כָּשֵׁר. אַף עַל גַּב דְּאֶפְשָׁר שֶׁמַּקְרִיב מִשֶּׁל שְׁתֵּי נָשִׁים, הֲרֵי יָכוֹל לְהַקְרִיב לְשֵׁם מִי שֶׁהוּא, וַאֲנַן לֹא קָיְמִינַן אֶלָּא בַפְּרִידוֹת גּוּפַיְהוּ כַּמָּה כְשֵׁרִים וְכַמָּה פְסוּלִין מִצַּד עַצְמָן. וְעַיֵּן בְּסוֹף פִּרְקִין. וְכָל זֶה בְּכֹהֵן נִמְלָךְ כְּדִתְנַן בְּרֵישׁ פֶּרֶק ג':

{י} דְּאִי אֵין בְּרֵרָה, אִיכָּא לְמֵימַר כָּל קַן רָחֵל הִקְרִיב לְמַעְלָה, וְכָל קַן לֵאָה הִקְרִיב לְמַטָּה, וְהִקְרִיב עוֹלָה לְמַטָּה וְחַטָּאת לְמַעְלָה כוּ', וְקָא מַשְׁמַע לָן כִּדְרַב חִסְדָּא דְּאָמַר אֵין הַקִּנִּים מִתְפָּרְשׁוֹת אֶלָּא אוֹ בִּלְקִיחַת בְּעָלִים אוֹ בַּעֲשִׂיַּת כֹּהֵן, דִּכְתִיב [וַיִּקְרָא יב] וְלָקַח וְגוֹ' אֶחָד לְעוֹלָה וְאֶחָד לְחַטָּאת, וּכְתִיב [שָׁם טו] וְעָשָׂה הַכֹּהֵן אֶת הָאֶחָד חַטָּאת וְאֶת הָאֶחָד עוֹלָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{א} מְפָרֵשׁ. וּקְצָת קָשֶׁה, דִּלְפִי זֶה רֵישָׁא נָקַט סְתוּמָה בְּלָאו דַּוְקָא, וּבַסֵּיפָא פָּרַח לְבֵין הַקְּרֵבוֹת אִי אֶפְשָׁר לְפָרֵשׁ אֶלָּא דַּוְקָא סְתוּמָה. וְהַתּוֹסָפוֹת כָּתְבוּ קֵן סְתוּמָה הוּא הַדִּין בְּקֵן מְפֹרָשׁ הָוֵי דִּין זֶה, אֶלָּא מִשּׁוּם דִּבְסֵיפָא קָאָמַר פָּרַח כוּ', דְּדַוְקָא פְרִידָה שֶׁל קֵן סְתוּמָה שֶׁפָּרְחָה וְנִתְעָרְבָה לְבֵין הַקְּרֵבוֹת, אֲבָל פְּרִידָה הַמְּפֹרֶשֶׁת לְעוֹלָה שֶׁפָּרְחָה בֵּין שְׁאָר קִנִּין אֵינוֹ יָכוֹל לְהַקְרִיב אֶלָּא כְמִנְיַן עוֹלוֹת שֶׁבַּקִּנִּים שֶׁפָּרַח לְשָׁם, כִּדְאָמַר סוֹף פֶּרֶק קַמָּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} הַמֵּתוֹת. לְהָכִי קָאָמַר בֵּין חַטָּאוֹת הַמֵּתוֹת, דְּאִלּוּ נִתְעָרְבָה בֵּין חַטָּאוֹת חַיּוֹת וְהָנָךְ חַטָּאוֹת דִּידֵיהּ נִינְהוּ, מָצֵי לְמִסַּב חַד מִנַּיְהוּ וַעֲבִיד לֵיהּ זוּג לְהָא, וְהָא דְּלֹא נִתְעָרֵב הֲוָה קָרֵי לֵיהּ עוֹלָה. גִּלְיוֹן רַשִׁ"י:


פרק ב - משנה ב

{ג} תֵּימַהּ, כֵּיוָן דְּבִסְתוּמוֹת אָיְרִינַן וּלְכָל אִשָּׁה שְׁנֵי קִנִּים, נִמְצְאוּ שְׁתֵּי חֲטָאוֹת הֵן בְּאֵלוּ הַשְּׁנֵי קִנִּים וְכַדִּין הוּא שֶׁהִקְרִיב שְׁתֵּי חַטָּאוֹת, וְלֹא שֶׁבָּאוּ מִקֵּן אֶחָד אֶלָּא מִשְּׁנֵי קִנִּים. וּבְהַמְּפָרֵשׁ לֵיתָא לְהַאי אִי נַמִּי. וְיֵשׁ לְיַשֵּׁב וּלְהַגִּיהַּ שֶׁצָּרִיךְ לוֹמַר וְנִמְצְאוּ שָׁלֹשׁ חַטָּאוֹת מִשְּׁנֵי קִנִּין, וְהָכִי קָאָמַר, אִי נַמִּי אִם יַעֲשֶׂה אֶחָד חַטָּאת וְאֶחָד עוֹלָה שׁוּב לֹא יַקְרִיב הַשְּׁלִישִׁי דְּחָיְשִׁינַן כוּ', וְנִמְצְאוּ שָׁלֹשׁ חַטָּאוֹת מִשְּׁנֵי קִנִּין, וְהָכִי נַמִּי מָצֵי לְאַסְבּוֹרָהּ בְּשָׁלֹשׁ עוֹלוֹת: {ד} שֶׁאֲפִלּוּ כוּ'. שֶׁאֲפִלּוּ מְעֹרָבוֹת כָּל הַשְּׁמֹנֶה פְרִידִין שֶׁבְּאַרְבָּעָה הַקִּנִּין יַחַד, אֵין פָּחוֹת מִשְּׁנַיִם. פֵּרוּשׁ, שְׁנֵי קִנִּין כְּשֵׁרִים בָּהֶן שְׁתֵּי עוֹלוֹת וּשְׁתֵּי חַטָּאוֹת. אֲבָל שָׁלֹשׁ חַטָּאוֹת וְשָׁלֹשׁ עוֹלוֹת, לֹא מָצֵי עֲבִיד, דְּדִלְמָא עֲבִיד לְהוּ מִשְּׁנֵי קִנִּין שֶׁל אִשָּׁה אַחַת שֶׁהֵן שְׁתֵּי חַטָּאוֹת וּשְׁתֵּי עוֹלוֹת. מְפָרֵשׁ:

{ה} אֲבָל אֵין לְפָרֵשׁ לֹא הִפְסִידָה כְּלוּם וְיַקְרִיבוּ בָּהּ שֶׁבַע חַטָּאוֹת וְשֶׁבַע עוֹלוֹת דְּמָצֵי עֲבִיד לְהוּ כֵּיוָן דִּבְקִנִּין אֲחֵרִים אֵין בָּהֶם כְּלוּם וְאוֹתָן שָׁלֹשׁ שֶׁנִּתְעָרְבוּ בָהּ מֵאֲחֵרִים רְאוּיִין הֵם לְהַקְרִיב כֻּלָּם בֵּין עוֹלוֹת בֵּין חַטָּאוֹת, שֶׁבְּכָל פַּעַם שֶׁבָּא לָהּ אֶחָד הָיָה בָא מִמָּקוֹם שֶׁרָאוּי לְהַקְרִיב בֵּין עוֹלָה בֵּין חַטָּאת, דְּאֵין זֶה הַלָּשׁוֹן, דַּהֲוָה לֵיהּ לְמֵימַר נִכְשְׁרָה [נִשְׂכְּרָה] שְׁבִיעִית בִּפְרִיחָה שְׁלִישִׁית, שֶׁעַל יְדֵי פְרִיחָה שְׁלִישִׁית הֻכְשְׁרוּ כָל הַקִּנִּין שֶׁבַּשְּׁבִיעִית, אֶלָּא הַלָּשׁוֹן מַשְׁמַע לֹא הִפְסִידָה כְלוּם בְּזֹאת הַפְּרִיחָה שְׁלִישִׁית. מְפָרֵשׁ:


פרק ב - משנה ה

{ו} כֵּן כָּתְבוּ הָרַמְבַּ"ם וְהַמְּפָרֵשׁ. וַהֲוָה מָצֵי לְמֵימַר דְּיָמוּתוּ מִפְּנֵי שֶׁפָּרַח בָּהֶן מִבֵּין הַמֵּתוֹת: {ז} שֶׁפָּרַח כוּ'. שֶׁפָּרַח לְצַד זֶה שֶׁשָּׁם הַחַטָּאוֹת אוֹ הָעוֹלוֹת וּפָרַח מִשָּׁם לַקָּצֶה הָאַחֵר, הֲרֵי כְּבָר הָיָה תַעֲרֹבֶת הַחַטָּאת בָּעוֹלָה, וּלְפִיכָךְ כֻּלָּם יָמוּתוּ כְּמוֹ שֶׁנִּתְבָּאֵר בִּתְחִלַּת הַמַּסֶּכֶת. הָרַמְבַּ"ם. וְהָרַ"ב וְהַמְּפָרֵשׁ נִרְאֶה דְלֹא גָרְסֵי אוֹ שֶׁפָּרַח: {ח} עַיֵּן פֶּרֶק י' דִּזְבָחִים מִשְׁנָה ד'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ט} יָבִיאוּ כוּ'. דְּשִׁעְבּוּדָא דְּאוֹרַיְתָא. וּמִשּׁוּם הָכִי אֲפִלּוּ לֹא הִפְרִישַׁתּוּ מֵחַיִּים. גְּמָרָא:

{א} וְכֵן כָּתַב הַמְּפָרֵשׁ. וּמִשּׁוּם מִשְׁנָה ג' הוּא דְּנָקְטֵי הָכִי. דְּהָא הָכָא בִּסְתוּמָה אֲנַן קָיְמִינַן: {ב} וְיַעֲשׂוּ כְּמוֹ שֶׁכָּתַב הָרַ"ב בְּסוֹף מַתְנִיתִין. וְכֻלֵּיהּ פִּרְקִין כְּמַאן דְּאָמַר בַּעֲלֵי חַיִּים אֵינָן נִדְחִין, וּמִשּׁוּם הָכִי כִּי אַקְרִיב אִרַצֵּי:


פרק ג - משנה ג

{ג} רוֹצֶה לוֹמַר שְׁתֵּי פְרִידוֹת שֶׁהָאַחַת חַטָּאת וְאַחַת לְעוֹלָה:


פרק ג - משנה ד

{ד} וּמִיהוּ כִּי אַקְרִיב בִּסְתָמָא, כָּל אַחַת עָלְתָה לְמִי שֶׁהִיא כוּ'. וּמִשּׁוּם הָכִי מַשְׁכַּחַת לְהוּ מֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל כְּדִתְנִינַן. וְכָתְבוּ הַתּוֹסָפוֹת דְּלֹא הֲוָה צָרִיךְ לְמִתְנֵי אֶלָּא סְתוּמָה וּמְפֹרֶשֶׁת, דִּבְחַטָּאת וְעוֹלָה אֵין מַרְוִיחַ כְּלוּם:


פרק ג - משנה ה

{ה} מְפָרֵשׁ. וְרוֹצֶה לוֹמַר שֶׁמָּא זוֹהִי אוֹתָהּ הַמְּעֹרֶבֶת שֶׁכְּבָר נִקְבְּעָה לְחַטָּאת: {ו} כְּלוֹמַר, אַחַר שֶׁהִקְרִיב שְׁתֵּי חַטָּאוֹת מֵהַתַּעֲרֹבוֹת שׁוּב לֹא יוּכַל לְהַקְרִיב עוֹד חַטָּאת עַל דַּעַת שֶׁהִיא זוֹ שֶׁנִּתְעָרְבָה, דְּשֶׁמָּא זוֹ הִיא עוֹלָה הַנִּשְׁאֶרֶת, דְּהוֹאִיל וְנִקְרְבוּ שְׁתֵּי חַטָּאוֹת מִן הַתַּעֲרֹבוֹת הֲרֵי הֻקְבְּעָה הַשְּׁלִישִׁית לְעוֹלָה וְשֶׁמָּא זוֹ הִיא, וְלֹא הִיא אוֹתָהּ הַחַטָּאת שֶׁנִּתְעָרְבָה: {ז} וְיִהְיֶה פִּי שְׁנַיִם בַּחוֹבָה, שֶׁיֵּשׁ בַּחוֹבָה כִּפְלַיִם עוֹלוֹת כְּנֶגֶד הַחַטָּאוֹת: {ח} רְצוֹנוֹ לוֹמַר, גַּם שְׁתֵּיהֶן, וַאֲפִלּוּ אַחַת מֵהֶן לֹא: {ט} רְצוֹנוֹ לוֹמַר, שְׁתַּיִם בְּיַחַד אֵינוֹ יָכוֹל לְהַקְרִיב, אֲבָל אַחַת מַקְרִיב. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:


פרק ג - משנה ו

{י} כְּסָבוּר שְׁנֵי הַקִּנִּין חוֹבָה עָלֶיהָ שֶׁמָּא יוֹלֶדֶת בְּזוֹב הָיְתָה וְהֻזְקְקָה לְקַן לֵידָה וְקַן זִיבָה. רַשִׁ"י: {יא} וְכֵן לְכָל הַחַטָּאוֹת דְּמַתְנִיתִין הַבָּאוֹת עַל הַסָּפֵק: {יב} הָרַמְבַּ"ם לֹא כָתַב שֶׁקָּבְעָה לְחוֹבָתָהּ. וְדוֹ"ק: {יג} לִרְבוּתָא כָּתַב כֵּן, דְּלֹא תֵימָא דְּמִכָּל מִין הִקְרִיב עוֹלָה אַחַת כוּ', קָא מַשְׁמַע לָן דְּלֹא, דְּהָא בְּפֵרְשָׁה וְקָבְעָה מַיְרֵי: {יד} לָאו דַּוְקָא. דִּיכוֹלָה לְהָבִיא שְׁתֵּי פְרִידוֹת מֵאֵיזֶה מִין שֶׁתִּרְצֶה אַחַת חַטָּאת וְאַחַת עוֹלָה, וּכְמוֹ שֶׁכָּתַבְתִּי בְּסָמוּךְ. וְכֵן כָּתַב הָרַמְבַּ"ם: {טו} מֵאֵיזֶה מִין שֶׁיִּהְיֶה לְהוֹסִיף עַל כֶּבֶשׂ לְעוֹלָה דַּעֲשִׁירוֹת, הוֹאִיל וְחוֹבָתָהּ גַּם כֵּן קְבוּעָה וְהִיא לֹא יָדְעָה מֵאֵיזֶה מִין. הָרַמְבַּ"ם. וּבֶן עַזַּאי מִסְתַּפֵּק עוֹד בְּאֵיזֶה מִין קָבְעָה עִם הַכֶּבֶשׂ, וּלְכָךְ אָמַר שְׁתֵּי, וְרוֹצֶה לוֹמַר תּוֹר וּבֶן יוֹנָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב שֶׁהֶאֱרִיךְ בְּפֵרוּשׁ הָרַמְבַּ"ם הַרְבֵּה, וְתָפַס עִקָּר לְפֵרוּשׁ רַשִׁ"י בִּזְבָחִים דַּף ס"ז. דּוֹק בָּהֶם אִם בַּעַל נֶפֶשׁ אָתָּה: {טז} רַשִׁ"י. וְלִרְבוּתָא נָקְטֵי מִנְיָן הַמְרֻבֶּה אַף עַל גַּב דְּהִנּוֹ לְבֶן עַזַּאי. אֲבָל הָיוּ יְכוֹלִים לְפָרֵשׁ אַלִּיבָּא דְתַנָּא קַמָּא, וְשָׁלֹשׁ לְחוֹבָתָהּ, וַהֲווּ מַמָּשׁ שֶׁבַע: {יז} דְּאִי חֲצוֹצְרוֹת מַמָּשׁ, הָא תְנַן בְּסוֹף תָּמִיד וּשְׁתֵּי חֲצוֹצְרוֹת שֶׁל כֶּסֶף בְּיָדָם. עוֹד יֵשׁ לוֹמַר, דְּהַהִיא דְּתָמִיד שֶׁל כֹּהֲנִים וְהָנֵי דְהָכָא שֶׁל לְוִיִּים לִכְלֵי שִׁיר, וּשְׁנֵי מִינֵי חֲצוֹצְרוֹת הָיוּ. תּוֹסָפוֹת: {יח} בְּנֵי כוּ'. מֵעַיִם הַדַּקִּים. כְּמוֹ הָדְרָא דְכַנְתָּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יט} שֶׁבְּשָׁעָה שֶׁהֵן מַזְקִינִין וְנֶחְלָשִׁים וְיַפְסִיד גּוּפָם, תִּרְבֶּה חָכְמָתָם וְיִתְחַזֵּק שִׂכְלָם וְיוֹסִיפוּ שְׁלֵמוּת. הָרַמְבַּ"ם:
