פרק א - משנה א

כָּל מַשְׁקֶה שֶׁתְּחִלָּתוֹ לְרָצוֹן אַף עַל פִּי שֶׁאֵין סוֹפוֹ לְרָצוֹן, אוֹ שֶׁסּוֹפוֹ לְרָצוֹן אַף עַל פִּי שֶׁאֵין תְּחִלָּתוֹ לְרָצוֹן, הֲרֵי זֶה בְכִי יֻתַּן. מַשְׁקִין טְמֵאִים מְטַמְּאִין לְרָצוֹן וְשֶׁלֹּא לְרָצוֹן:

 ר"ע מברטנורה  כָּל מַשְׁקֶה שֶׁתְּחִלָּתוֹ לְרָצוֹן. דְּנִיחָא לֵיהּ בְּהָנָךְ מַשְׁקִין, בִּשְׁבִיל שֶׁתּוּדַח הַקְּעָרָה אוֹ לִשְׁאָר תַּשְׁמִישִׁין {א}: אַף עַל פִּי שֶׁאֵין סוֹפוֹ לְרָצוֹן. כְּשֶׁנָּפְלוּ עַל הַפֵּרוֹת לֹא נִיחָא לֵיהּ, אֲפִלּוּ הָכִי הָוֵי הֶכְשֵׁר. אִי נַמִּי, סוֹפוֹ לְרָצוֹן, כְּשֶׁנָּפְלוּ עַל הַפֵּרוֹת. אֲבָל בִּתְחִלַּת בִּיאָתָן הָיָה הַדָּבָר קָשֶׁה בְּעֵינָיו, הַכֹּל הָוֵי הֶכְשֵׁר: הֲרֵי זֶה בְכִי יֻתַּן. כְּלוֹמַר, הָאֳכָלִין שֶׁנָּפְלוּ עֲלֵיהֶן מַשְׁקִין הַלָּלוּ, מֻכְשָׁרִין לְקַבֵּל טֻמְאָה מִכָּאן וָאֵילָךְ. שֶׁאֵין שׁוּם אֹכֶל מְקַבֵּל טֻמְאָה עַד שֶׁיָּבֹא עָלָיו מַיִם {ב} אוֹ אֶחָד מִשִּׁבְעָה מַשְׁקִין, שֶׁהֵם, מַיִם, יַיִן, שֶׁמֶן, חָלָב, דְּבַשׁ, דָּם, טַל. דִּכְתִיב (וַיִּקְרָא יא) וְכִי יֻתַּן מַיִם עַל זֶרַע וְנָפַל מִנִּבְלָתָם עָלָיו טָמֵא הוּא, כְּתִיב יִתֵּן וְקָרִינַן יֻתַּן, מַה יִתֵּן דְּנִיחָא לֵיהּ, אַף יֻתַּן דְּנִיחָא לֵיהּ: מְטַמְּאִין לְרָצוֹן וְשֶׁלֹּא לְרָצוֹן. כֵּיוָן דִּטְמֵאִין הֵן, טֻמְאָתָן וְהֶכְשֵׁרָן בָּאִים כְּאֶחָד, וְאִם נָפְלוּ עַל הַפֵּרוֹת אֲפִלּוּ שֶׁלֹּא לְרָצוֹן, מְטַמְּאִין אוֹתָם, דְּלֹא בָעִינַן דְּנִיחָא לֵיהּ אֶלָּא כְשֶׁבָּא הַמַּשְׁקֶה קֹדֶם לַטֻּמְאָה, דִּכְתִיב וְכִי יֻתַּן מַיִם עַל זֶרַע, וַהֲדַר וְנָפַל מִנִּבְלָתָם עָלָיו. אֲבָל כְּשֶׁהַמַּשְׁקֶה וְהַטֻּמְאָה בָּאִין כְּאֶחָד כְּגוֹן שֶׁהַמַּשְׁקִין טְמֵאִין, לֹא בָעִינַן דְּנִיחָא לֵיהּ: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} שֶׁל תָּלוּשׁ: {ב} מִשֶּׁנִּתְלְשׁוּ. דְּאִם לֹא כֵן, אֵין לְךָ דָּבָר שֶׁלֹּא בָא עָלָיו מַיִם בִּמְחֻבָּר. רַשִׁ"י בַחֻמָּשׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק א - משנה ב

הַמַּרְעִיד אֶת הָאִילָן לְהַשִּׁיר מִמֶּנּוּ אֳכָלִין אוֹ אֶת הַטֻּמְאָה, אֵינָן בְּכִי יֻתַּן. לְהַשִּׁיר מִמֶּנּוּ מַשְׁקִין, בֵּית שַׁמַּאי אוֹמְרִים, הַיּוֹצְאִין וְאֶת שֶׁבּוֹ, בְּכִי יֻתַּן. בֵּית הִלֵּל אוֹמְרִים, הַיּוֹצְאִין, בְּכִי יֻתַּן, וְאֶת שֶׁבּוֹ, אֵינָן בְּכִי יֻתַּן {ד}, מִפְּנֵי שֶׁהוּא מִתְכַּוֵּן שֶׁיֵּצְאוּ מִכֻּלּוֹ:

 ר"ע מברטנורה  הַמַּרְעִיד אֶת הָאִילָן. מְנַעְנְעוֹ: אוֹ אֶת הַטֻּמְאָה. כְּגוֹן שֶׁהֵבִיא שָׁם עוֹרֵב שֶׁרֶץ אוֹ כַזַּיִת מִן הַמֵּת אוֹ נְבֵלָה וְנִשְׁאַר בֵּין עַנְפֵי הָאִילָן: אֵינָן בְּכִי יֻתַּן. אִם מֵחֲמַת רְעָדָה זוֹ נָפְלוּ מֵי גְשָׁמִים שֶׁבָּאִילָן עַל פֵּרוֹת שֶׁתַּחְתָּיו, לֹא הֻכְשְׁרוּ: הַיּוֹצְאִים וְאֶת שֶׁבּוֹ. מַשְׁקִים הַיּוֹצְאִים מֵחֲמַת הָרְעָדָה, וּמַשְׁקִים הַנִּשְׁאָרִים בָּאִילָן וְנָפְלוּ אַחַר כָּךְ עַל הַפֵּרוֹת, הֻכְשְׁרוּ, דְּאַחְשׁוֹבֵי אַחְשְׁבִינְהוּ וְכִתְלוּשִׁים דָּמוּ {ג}: מִפְּנֵי שֶׁהוּא מִתְכַּוֵּן שֶׁיֵּצְאוּ מִכֻּלּוֹ. נִתְכַּוֵּן לְהוֹצִיא כָל הַמַּיִם מִן הָאִילָן, וְהַנִּשְׁאָרִים לָאו דַּעְתֵּיהּ עֲלַיְהוּ וְלֹא מַכְשְׁרֵי: (ר"ע מברטנורה).

 עיקר תוי"ט  {ג} עַיֵּן מַה שֶּׁכָּתַבְנוּ בְּפֶרֶק ד' מִשְׁנָה ג' וּבְהָרַ"ב שָׁם מִשְׁנָה ז': {ד} אֵינָן כוּ'. וְאַף עַל פִּי שֶׁנָּשְׁרוּ מִמָּקוֹם לְמָקוֹם. הָרַמְבַּ"ם: (עיקר תוי"ט).

פרק א - משנה ג

הַמַּרְעִיד אֶת הָאִילָן וְנָפַל עַל חֲבֵרוֹ, אוֹ סוֹכָה {ה} וְנָפְלָה עַל חֲבֶרְתָּהּ, וְתַחְתֵּיהֶן זְרָעִים אוֹ יְרָקוֹת הַמְחֻבָּרִין לַקַּרְקַע, בֵּית שַׁמַּאי אוֹמְרִים, בְּכִי יֻתַּן. בֵּית הִלֵּל אוֹמְרִים, אֵינָן בְּכִי יֻתַּן. אָמַר רַבִּי יְהוֹשֻׁעַ מִשּׁוּם אַבָּא יוֹסֵי חֲלֵי קוּפְרִי אִישׁ טִבְעוֹן, תְּמַהּ עַצְמְךָ אִם יֵשׁ מַשְׁקֶה טָמֵא {ו} בַתּוֹרָה, עַד שֶׁיִּתְכַּוֵּן וְיִתֵּן, שֶׁנֶּאֱמַר (ויקרא יא), וְכִי יֻתַּן מַיִם עַל זֶרַע:

 ר"ע מברטנורה  הַמַּרְעִיד אֶת הָאִילָן. לְהַפִּיל פֵּרוֹתָיו: וְנָפְלוּ עַל חֲבֵרוֹ. עַל אִילָן אַחֵר, וּמֵאוֹתוֹ אִילָן נָפְלוּ עַל הַזְּרָעִים וְעַל הַיְרָקוֹת הַמְחֻבָּרִים לַקַּרְקַע, וּמַשְׁקֶה טוֹפֵחַ עַל אוֹתָן הַזְּרָעִים מִן הַמַּיִם שֶׁרְגִילִין לְהַשְׁקוֹת אוֹתָן אוֹ מִגְּשָׁמִים שֶׁנָּפְלוּ עֲלֵיהֶן: בֵּית שַׁמַּאי אוֹמְרִים הֲרֵי הֵן בְּכִי יֻתַּן. דְּכֵיוָן דְּיוֹדֵעַ הָיָה שֶׁיִּפְּלוּ פֵרוֹת הַלָּלוּ עַל אוֹתָן זְרָעִים שֶׁיֵּשׁ עֲלֵיהֶן מַשְׁקֶה טוֹפֵחַ, גַּלֵּי דַעְתֵּיהּ דְּנִיחָא לֵיהּ, וְהֻכְשְׁרוּ הַפֵּרוֹת שֶׁנָּפְלוּ עַל גַּבֵּי אוֹתָן זְרָעִים, לְקַבֵּל טֻמְאָה: וּבֵית הִלֵּל אוֹמְרִים אֵינָן בְּכִי יֻתַּן. כֵּיוָן דְּלֹא נָפְלוּ לְהֶדְיָא מִן הָאִילָן עַל הַזְּרָעִים אֶלָּא מֵאִילָן לְאִילָן אַחֵר: חֲלֵי קוּפְרִי. כָּךְ הָיָה כִנּוּיוֹ: טִבְעוֹן. שֵׁם מָקוֹם: עַד שֶׁיִּתְכַּוֵּן. וְהָנֵי הוֹאִיל וּמֵאִילָן לְאִילָן אוֹ מִסּוֹכָה לְסוֹכָה נָפְלוּ, אֵין זוֹ כַוָּנָה. וְטַעֲמַיְהוּ דְבֵית הִלֵּל קָאָמַר: (ר"ע מברטנורה).

 עיקר תוי"ט  {ה} סוֹכָה. עָנָף שֶׁל אִילָן. הָרַ"שׁ. וְהָרַמְבַּ"ם פֵּרֵשׁ סֻכָּה מַמָּשׁ: {ו} טָמֵא. רוֹצֶה לוֹמַר מֻכְשָׁר. וְהַרְבֵּה הֵם בְּזֹאת הַמַּסֶּכֶת. וְכֵן טָהוֹר עַל הַדָּבָר שֶׁאֵינוֹ מֻכְשָׁר. הָרַמְבַּ"ם: (עיקר תוי"ט).

פרק א - משנה ד

הַנּוֹעֵר אֲגֻדָּה שֶׁל יָרָק וְיָרְדוּ מִצַּד הָעֶלְיוֹן לַתַּחְתּוֹן, בֵּית שַׁמַּאי אוֹמְרִים, בְּכִי יֻתַּן. בֵּית הִלֵּל אוֹמְרִים, אֵינָם בְּכִי יֻתַּן. אָמְרוּ בֵית הִלֵּל לְבֵית שַׁמַּאי, וַהֲלֹא הַנּוֹעֵר אֶת הַקֶּלַח, חוֹשְׁשִׁין אָנוּ שֶׁמָּא יָצְאוּ מִן הֶעָלֶה לֶעָלֶה. אָמְרוּ לָהֶן בֵּית שַׁמַּאי, שֶׁהַקֶּלַח אֶחָד, וַאֲגֻדָּה קְלָחִים הַרְבֵּה. אָמְרוּ לָהֶם בֵּית הִלֵּל, הֲרֵי הַמַּעֲלֶה שַׂק מָלֵא פֵרוֹת וּנְתָנוֹ עַל גַּב הַנָּהָר, חוֹשְׁשִׁין אָנוּ שֶׁמָּא יָרְדוּ מִצַּד הָעֶלְיוֹן לַתַּחְתּוֹן. אֲבָל {ח} אִם הֶעֱלָה שְׁנַיִם וּנְתָנָן זֶה עַל גַּב זֶה, הַתַּחְתּוֹן בְּכִי יֻתַּן. וְרַבִּי יוֹסֵי אוֹמֵר, הַתַּחְתּוֹן טָהוֹר {ט}:

 ר"ע מברטנורה  הַנּוֹעֵר אֲגֻדָּה שֶׁל יָרָק. כְּגוֹן יָרָק שֶׁיָּרְדוּ עָלָיו גְּשָׁמִים אוֹ שֶׁנָּפַל בַּמַּיִם שֶׁלֹּא לְרָצוֹן, וּמְנַעֲרוֹ כְּדֵי שֶׁיֵּצְאוּ הַמַּיִם {ז}: חוֹשְׁשִׁים אָנוּ. בִּתְמִיהָה. כְּלוֹמַר לְדִבְרֵיכֶם הָיָה לָנוּ לָחוּשׁ גַּם לָזֶה: הַמַּעֲלֶה שַׂק מָלֵא פֵרוֹת. שֶׁנָּפַל בְּמַיִם שֶׁלֹּא לְרָצוֹן, וְהֶעֱלָהוּ וְהִנִּיחוֹ עַל שְׂפַת הַנָּהָר: הַתַּחְתּוֹן טָהוֹר. כְּלוֹמַר, אַף הַתַּחְתּוֹן טָהוֹר, וְכָל שֶׁכֵּן הָעֶלְיוֹן: (ר"ע מברטנורה).

 עיקר תוי"ט  {ז} וְיֵרְדוּ מִצַּד הָעֶלְיוֹן לַצַּד הַתַּחְתּוֹן מִקֶּלַח לְקֶלַח. בֵּית שַׁמַּאי אוֹמְרִים כֵּיוָן דְּבִשְׁעַת נִעוּר נִיחָא לֵיהּ שֶׁיֵּרְדוּ מַיִם מִצַּד הָעֶלְיוֹן לַתַּחְתּוֹן וּמִתּוֹךְ כָּךְ יִפְּלוּ הַמַּיִם מִן הָאֲגֻדָּה, חָשׁוּב הֶכְשֵׁר. וּבֵית הִלֵּל אוֹמְרִים כֵּיוָן דְּהָיָה חָפֵץ שֶׁלֹּא יִהְיוּ מַיִם עַל זֹאת הָאֲגֻדָּה בַּצַּד הָעֶלְיוֹן, נְהִי דְבַצַּד הַתַּחְתּוֹן נִיחָא לֵיהּ שֶׁיֵּרְדוּ דֶרֶךְ שָׁם, מִכָּל מָקוֹם כָּל הָאֲגֻדָּה אַחַת הִיא וְתַחְתּוֹנָה שָׁוָה לָעֶלְיוֹנָה. אָמְרוּ לָהֶם בֵּית הִלֵּל לְבֵית שַׁמַּאי וַהֲרֵי כוּ', וְכִי חוֹשְׁשִׁין שֶׁמָּא יֵצְאוּ מֵעָלֶה לְעָלֶה, בִּתְמִיהָה, הָכָא נַמִּי לֹא נֵחוּשׁ. אָמְרוּ לָהֶם, שֶׁהַקֶּלַח אֶחָד, וְהַסְּבָרָא נוֹתֶנֶת לַעֲשׂוֹת כָּל עָלִין שֶׁבּוֹ דִּין אֶחָד לָהֶם, אֲבָל אֵין סְבָרָא לַעֲשׂוֹת כָּל קְלָחִין שֶׁבָּאֲגֻדָּה אֶחָד. מַהֲרַ"ם: {ח} אֲבָל. כְּלוֹמַר בְּהָא מוֹדִינָא. כְּמוֹ אֲבָל שָׂרָה אִשְׁתֶּךָ. מַהֲרַ"ם: {ט} טָהוֹר. שֶׁלֹּא נִתְכַּוֵּן שֶׁיִּבָּלֵל הַשַּׂק הַתַּחְתּוֹן, אֲבָל נִתְכַּוֵּן לְהָגֵן בּוֹ הַשַּׂק הָעֶלְיוֹן שֶׁלֹּא יִבָּלֵל, וְנִמְצָא בְלִילַת הַתַּחְתּוֹן שֶׁלֹּא לְרָצוֹן. וַהֲלָכָה כְּרַבִּי יוֹסֵי. הָרַמְבַּ"ם: (עיקר תוי"ט).

פרק א - משנה ה

הַמְמַחֵק אֶת הַכְּרֵישָׁה {י}, וְהַסּוֹחֵט שְׂעָרוֹ בִּכְסוּתוֹ, רַבִּי יוֹסֵי אוֹמֵר, הַיּוֹצְאִין בְּכִי יֻתַּן {יב}, וְאֶת שֶׁבּוֹ אֵינָן בְּכִי יֻתַּן, מִפְּנֵי שֶׁהוּא מִתְכַּוֵּן שֶׁיֵּצְאוּ מִכֻּלּוֹ:

 ר"ע מברטנורה  הַמְמַחֵק אֶת הַכְּרֵישָׁה. דּוֹחֵק אֶת הַכְּרֵישָׁה כְּדֵי שֶׁיֵּצֵא מִמֶּנָּה הַטַּל: כְּרֵישָׁה. חָצִיר בִּלְשׁוֹן מִקְרָא. פורו"ש בְּלַעַ"ז וּבַעֲרָבִי כורא"ת: וְהַסּוֹחֵט בִּשְׂעָרוֹ וּבִכְסוּתוֹ {יא}. שֶׁהָיָה בָא בַדֶּרֶךְ וְיָרְדוּ גְשָׁמִים בִּשְׂעָרוֹ וְעַל כְּסוּתוֹ, וְסוֹחֵט בָּהֶן כְּדֵי שֶׁיֵּצְאוּ הַמַּיִם: וְאֶת שֶׁבּוֹ. מַשְׁקִין הַנִּשְׁאָרִים וְנָפְלוּ אַחַר כָּךְ עַל הַפֵּרוֹת: מִתְכַּוֵּן שֶׁיֵּצְאוּ מִכֻּלּוֹ. דְּנִתְכַּוֵּן לְהוֹצִיא מֵהֶן כָּל הַמַּיִם וְלָאו דַּעְתֵּיהּ עַל הַנִּשְׁאָרִים, הִלְכָּךְ לֹא מַכְשִׁירִים. כְּבֵית הִלֵּל דְּרֵישׁ פִּרְקִין: (ר"ע מברטנורה).

 עיקר תוי"ט  {י} הַכְּרֵישָׁה. וְהוּא מְחֻבָּר לַקַּרְקַע. אוֹ נָפַל עָלָיו שֶׁלֹּא לְרָצוֹן. הָרַמְבַּ"ם: {יא} וְהָרַמְבַּ"ם גָּרַס כְּגִרְסַת הַסֵּפֶר. וּמְפָרֵשׁ שֶׁסּוֹחֵט שְׂעָרוֹ בִּבְגָדָיו לְנַגְּבוֹ: {יב} בְּכִי יֻתַּן. אַף עַל גַּב דְּשֶׁלֹּא יִלְקֶה הַכֹּתֶל אֵינָן בְּכִי יֻתַּן [לְהַלָּן פּ"ד מ"ג], הָתָם מִשּׁוּם דְּלֹא נִיחָא לֵיהּ דַּאֲתוּ, וְהָכָא נַמִּי אַף עַל גַּב דְּלֹא נִיחָא לֵיהּ מֵעִקָּרָא, מִכָּל מָקוֹם כֵּיוָן שֶׁהוּא עַצְמוֹ דּוֹחֲקוֹ לְהוֹצִיא מֵהֶן הַמַּיִם, אַחְשְׁבִינְהוּ. תּוֹסָפוֹת: (עיקר תוי"ט).

פרק א - משנה ו

הַנּוֹפֵחַ בַּעֲדָשִׁים לְבָדְקָן אִם יָפוֹת הֵן {יג}, רַבִּי שִׁמְעוֹן אוֹמֵר, אֵינָן בְּכִי יֻתַּן. וַחֲכָמִים אוֹמְרִים, בְּכִי יֻתַּן. וְהָאוֹכֵל שֻׁמְשְׁמִין בְּאֶצְבָּעוֹ, מַשְׁקִין שֶׁעַל יָדוֹ, רַבִּי שִׁמְעוֹן אוֹמֵר, אֵינָן בְּכִי יֻתַּן. וַחֲכָמִים אוֹמְרִים, בְּכִי יֻתַּן. הַטּוֹמֵן פֵּרוֹתָיו בַּמַּיִם מִפְּנֵי הַגַּנָּבִים, אֵינָן בְּכִי יֻתַּן. מַעֲשֶׂה בְאַנְשֵׁי יְרוּשָׁלַיִם שֶׁטָּמְנוּ דְבֵלָתָן בַּמַּיִם מִפְּנֵי הַסִּיקָרִין, וְטִהֲרוּ לָהֶן חֲכָמִים. הַנּוֹתֵן פֵּרוֹתָיו בְּשִׁבֹּלֶת הַנָּהָר לַהֲבִיאָן עִמּוֹ, אֵינָן בְּכִי יֻתַּן:

 ר"ע מברטנורה  הַנּוֹפֵחַ. בְּפִיו: בַּעֲדָשִׁים. וְיָצָא עֲלֵיהֶן רֹק מִפִּיו בִּנְפִיחָתוֹ, וְהָרֹק הַיּוֹצֵא מִן הַפֶּה מִתּוֹלְדוֹת הַמַּיִם הוּא חָשׁוּב: אֵינָן בְּכִי יֻתַּן. כֵּיוָן דְּלֹא נִתְכַּוֵּן לְכָךְ: וַחֲכָמִים אוֹמְרִים בְּכִי יֻתַּן. כֵּיוָן דְּנִתְכַּוֵּן לְנַפֵּחַ, וְדֶרֶךְ הָרֹק לָצֵאת עַל יְדֵי נְפִיחָה. וַהֲלָכָה כַּחֲכָמִים: הָאוֹכֵל שֻׁמְשְׁמִין בְּאֶצְבָּעוֹ. כְּגוֹן שֶׁהִנִּיחַ שֻׁמְשְׁמִין בְּיָדוֹ אַחַת, וְאֶצְבָּעוֹ שֶׁבְּיָדוֹ הַשֵּׁנִית מַכְנִיסָהּ לְתוֹךְ פִּיו לְלַחְלְחָהּ בָּרֹק וְנוֹתְנָהּ בַּשֻּׁמְשְׁמִין שֶׁבְּיָדוֹ הַשֵּׁנִית וְהֵן נִדְבָּקִין בָּהּ וּמַכְנִיס בְּפִיו וְאוֹכֵל, וּמֵחֲמַת כֵּן נִתְלַחְלְחָה הַיָּד שֶׁהַשֻּׁמְשְׁמִין בָּהּ, וְיֵשׁ בָּהּ מַשְׁקֶה טוֹפֵחַ: רַבִּי שִׁמְעוֹן אוֹמֵר אֵינָן בְּכִי יֻתַּן. דִּלְלַחְלֵחַ אֶצְבָּעוֹ נִתְכַּוֵּן, וְלֹא לְלַחְלֵחַ הַיָּד שֶׁהַשֻּׁמְשְׁמִין בָּהּ. וְרַבָּנָן סָבְרֵי, כֵּיוָן דְּאַחְשְׁבֵיהּ בְּאֶצְבָּעוֹ, מַכְשִׁיר אֲפִלּוּ שֶׁלֹּא לְרָצוֹן, כְּדִתְנַן בְּרֵישׁ פִּרְקִין כָּל מַשְׁקֶה שֶׁתְּחִלָּתוֹ לְרָצוֹן, אַף עַל פִּי שֶׁאֵין סוֹפוֹ לְרָצוֹן, הֲרֵי זֶה בְּכִי יֻתַּן. וְרַבִּי שִׁמְעוֹן סָבַר, דְּלֹא הָיָה מִתְּחִלָּתוֹ לְרָצוֹן אֶלָּא הַמַּשְׁקֶה הַמְלַחְלֵחַ הָאֶצְבַּע בִּלְבַד. וְהַנִּשְׁאָר שֶׁנִּתְלַחְלְחָה בוֹ הַיָּד, לֹא הָיָה לְעוֹלָם לְרָצוֹן. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן בְּכֻלַּהּ מַתְנִיתִין: הַסְּרִיקִין. וַיִּתְלַקְּטוּ אֶל יִפְתָּח אֲנָשִׁים רֵיקִים (שׁוֹפְטִים יא) מְתַרְגְּמִינַן גֻּבְרִין סְרִיקִין. לְשׁוֹן אִילָן סְרָק שֶׁאֵינוֹ עוֹשֶׂה פֵרוֹת. פֵּרוּשׁ אַחֵר, סְרִיקִין, גַּנָּבִים. שֶׁכֵּן בִּלְשׁוֹן יִשְׁמָעֵאל קוֹרִין לְגַנָּבֵי סְרָאקִין. וְיֵשׁ גּוֹרְסִין סִיקָרִין, לְשׁוֹן לֹא הָיָה סִקְרִיקוֹן בִּיהוּדָה בַּהֲרוּגֵי הַמִּלְחָמָה, בְּפֶרֶק הַנִּזָּקִין, פֵּרוּשׁ גַּזְלָנִין וְאַנָּסִין: בְּשִׁבֹּלֶת הַנָּהָר. שְׁבִיל הַנָּהָר וּמְרוּצַת מֵימָיו. לְשׁוֹן וְשִׁבֹּלֶת שְׁטָפָתְנִי (תְּהִלִּים סט): לַהֲבִיאָן עִמּוֹ. שֶׁלֹּא הָיָה יָכוֹל לְנָשְׂאָן, וּמְשִׁיטָן בַּנָּהָר: (ר"ע מברטנורה).

 עיקר תוי"ט  {יג} לְבָדְקָן כוּ'. שֶׁרוֹאִין אִם נִטְפַּח וְנִתְלַחְלַח כְּשֶׁנּוֹפְחִים בּוֹ, נוֹדָע שֶׁיִּתְבַּשֵּׁל מַהֵר, כִּי הֱיוֹתוֹ מְקַבֵּל הַלֵּחוֹת בִּמְהֵרָה אוֹת אֶל פְּתִיחַת נַקְבִּיבוּתוֹ. הָרַמְבַּ"ם. וְלִדְבָרָיו לָאו בָּרֹק הַיּוֹצֵא מִפִּיו פְּלִיגֵי, אֶלָּא בַּלֵּחוֹת וְזִיעַ שֶׁמַּזִּיעִים הָעֲדָשִׁים בְּסִבַּת הַנְּפִיחָה פְּלִיגֵי. וְדִבְרֵי הָרַ"ב מִפֵּרוּשׁ הָרַ"שׁ הוּא: (עיקר תוי"ט).

פרק ב - משנה א

זֵעַת בָּתִּים, בּוֹרוֹת, שִׁיחִין וּמְעָרוֹת, טְהוֹרָה {א}. זֵעַת הָאָדָם, טְהוֹרָה. שָׁתָה מַיִם טְמֵאִין וְהִזִּיעַ, זֵעָתוֹ טְהוֹרָה. בָּא בְמַיִם שְׁאוּבִים וְהִזִּיעַ, זֵעָתוֹ טְמֵאָה. נִסְתַּפֵּג וְאַחַר כָּךְ הִזִּיעַ, זֵעָתוֹ טְהוֹרָה:

 ר"ע מברטנורה  זֵעַת בָּתִּים טְהוֹרָה. אֵינָהּ מַכְשֶׁרֶת הַזְּרָעִים לְקַבֵּל טֻמְאָה: זֵעָתוֹ טְמֵאָה. דְּמֵחֲמַת הַמַּיִם הִיא. וְדַוְקָא בָא בְּמַיִם שְׁאוּבִין. אֲבָל בָּא בְּמַיִם מְחֻבָּרִים דְּלֹא מַכְשְׁרֵי, לֹא, דְּאַף עַל גַּב דְּהַשְׁתָּא מִיהָא תְלוּשִׁים נִינְהוּ, לֹא לְרָצוֹן נִתְלְשׁוּ {ב}: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} בָּתִּים כוּ' טְהוֹרָה. וַאֲפִלּוּ הֵן טְמֵאִים. הָרַמְבַּ"ם: {ב} וְנִרְאֶה לִי דְּמַיְרֵי שֶׁלֹּא בָא בְתוֹכָם לִטְבֹּל. דְּלֹא תִקְשֶׁה מִמַּתְנִיתִין דְּרֵישׁ פֶּרֶק ה': (עיקר תוי"ט).

פרק ב - משנה ב

מֶרְחָץ טְמֵאָה, זֵעָתָהּ טְמֵאָה. וּטְהוֹרָה, בְּכִי יֻתַּן. הַבְּרֵכָה שֶׁבַּבַּיִת, הַבַּיִת מַזִּיעַ מֵחֲמָתָהּ, אִם טְמֵאָה, זֵעַת כָּל הַבַּיִת שֶׁמֵּחֲמַת הַבְּרֵכָה, טְמֵאָה:

 ר"ע מברטנורה  מֶרְחָץ טְמֵאָה. שֶׁל מַיִם שְׁאוּבִין: זֵעָתָהּ טְמֵאָה. מַכְשֶׁרֶת אֶת הַפֵּרוֹת וּמְטַמֵּאתָן {ג}: וּטְהוֹרָה. כְּלוֹמַר, וְזֵעַת מֶרְחָץ טְהוֹרָה, כְּגוֹן מֶרְחָץ שֶׁל מֵי מַעְיָן דְּלָאו מַיִם שְׁאוּבִים נִינְהוּ, הֲרֵי הוּא בְּכִי יֻתַּן, שֶׁהַזֵּעָה מַכְשֶׁרֶת אֶת הַפֵּרוֹת: אִם טְמֵאָה. הַבְּרֵכָה. כָּל זֵעָה שֶׁיַּזִיעַ הַבַּיִת מֵחֲמָתָהּ, מַכְשֶׁרֶת וּמְטַמְּאָה. אֲבָל שֶׁלֹּא מֵחֲמָתָהּ, הַוְיָא לַהּ זֵעַת הַבַּיִת וּטְהוֹרָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ג} הָרַ"שׁ. וְקָשֶׁה לִי, דְּאַף עַל גַּב דִּשְׁאוּבִים הֵם, לֹא מִפְּנֵי כֵן הֵן מְטַמְּאִים אַף עַל פִּי שֶׁמַּכְשִׁירִים. וְצָרִיךְ לוֹמַר דִּשְׁאוּבִים וּטְמֵאִים הֵם. וְהָרַמְבַּ"ם מְפָרֵשׁ, דְּמֶרְחָץ שֶׁזֵּעָתוֹ הוּא מִן הַמַּיִם שֶׁבְּתוֹכָהּ וְאִם הָיְתָה הַמֶּרְחָץ טְמֵאָה זֵעָתָהּ טְמֵאָה כוּ': (עיקר תוי"ט).

שְׁתֵּי בְרֵכוֹת, אַחַת טְהוֹרָה וְאַחַת טְמֵאָה, הַמַּזִּיעַ קָרוֹב לַטְּמֵאָה, טָמֵא. קָרוֹב לַטְּהוֹרָה, טָהוֹר. מֶחֱצָה לְמֶחֱצָה, טָמֵא. בַּרְזֶל טָמֵא שֶׁבְּלָלוֹ עִם בַּרְזֶל טָהוֹר, אִם רֹב מִן הַטָּמֵא, טָמֵא. וְאִם רֹב מִן הַטָּהוֹר, טָהוֹר. מֶחֱצָה לְמֶחֱצָה, טָמֵא. גִּסְטְרָיוֹת שֶׁיִּשְׂרָאֵל וְנָכְרִים מַטִּילִין לְתוֹכָהּ, אִם רֹב מִן הַטָּמֵא, טָמֵא. וְאִם רֹב מִן הַטָּהוֹר, טָהוֹר. מֶחֱצָה לְמֶחֱצָה, טָמֵא. מֵי שְׁפִיכוּת שֶׁיָּרְדוּ עֲלֵיהֶן מֵי גְשָׁמִים, אִם רֹב מִן הַטָּמֵא, טָמֵא. וְאִם רֹב מִן הַטָּהוֹר, טָהוֹר. מֶחֱצָה לְמֶחֱצָה, טָמֵא. אֵימָתַי, בִּזְמַן שֶׁקָּדְמוּ מֵי שְׁפִיכוּת. אֲבָל אִם קָדְמוּ מֵי גְשָׁמִים, אֲפִלּוּ כָל שֶׁהֵן, לְמֵי שְׁפִיכוּת, טָמֵא {ד}:

 ר"ע מברטנורה  בַּרְזֶל טָמֵא. שֶׁבָּא מִשִּׁבְרֵי כֵלִים טְמֵאִין: שֶׁבְּלָלוֹ. שֶׁטְּרָפוֹ וְעֵרְבוֹ עִם בַּרְזֶל טָהוֹר שֶׁבָּא מִן הָעֶשֶׁת: אִם רֹב מִן הַטָּמֵא טָמֵא. חֲכָמִים גָּזְרוּ עַל כְּלֵי מַתָּכוֹת שֶׁנִּטְמְאוּ וּשְׁבָרָן וְהִתִּיכָן וְעָשָׂה מֵהֶן כֵּלִים, שֶׁיַּחְזְרוּ לְטֻמְאָתָן הַיְשָׁנָה עַד שֶׁיַּזֶּה עֲלֵיהֶן שְׁלִישִׁי וּשְׁבִיעִי וְיַטְבִּילֵם וְיַעֲרִיב שִׁמְשָׁן: גִּסְטְרָיוֹת. כְּמִין קְדֵרוֹת עֲשׂוּיוֹת לְהַשְׁתִּין בָּהֶן. וּבְלָשׁוֹן עֲרָבִי קוֹרִין לָהֶן בסריא"ש: שֶׁיִּשְׂרָאֵל וְנָכְרִים מַטִּילִין לְתוֹכָן. מַשְׁתִּינִים לְתוֹכָן. וְשֶׁתֶן שֶׁל נָכְרִי טָמֵא, דְּגָזְרוּ עֲלֵיהֶן שֶׁיִּהְיוּ כְזָבִין לְכָל דִּבְרֵיהֶן: מֵי שְׁפִיכוּת. מַיִם שֶׁרָחֲצוּ בָהֶן וְשׁוֹפְכִין אוֹתָן לַחוּץ, וּבְחֶזְקַת טְמֵאִין הֵן: אֵימָתַי, מִתְבַּטְּלִין מֵי הַשְּׁפִיכוּת בְּרֹב: בִּזְמַן שֶׁקָּדְמוּ. הֵן, וְאַחַר כָּךְ רַבּוּ עֲלֵיהֶן מֵי גְשָׁמִים וְטִהֲרוּם: אֲבָל אִם קָדְמוּ מֵי גְשָׁמִים. אֲפִלּוּ נָפְלוּ עֲלֵיהֶן מַשֶּׁהוּ מֵי שְׁפִיכוּת, טְמֵאִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {ד} טָמֵא. שֶׁמַּשְׁקִין טְמֵאִין שֶׁיָּרְדוּ לִטְהוֹרִים, מְטַמְּאִים בְּכָל שֶׁהֵן. הָרַמְבַּ"ם. וְקָשֶׁה, דְּאַדְּרַבָּה הֲוָה לֵיהּ לְמֵימַר קַמָּא קַמָּא בָטֵיל. [וְעַיֵּן פֶּרֶק ה' דַּעֲבוֹדָה זָרָה מִשְׁנָה ח' בְּהָרַ"ב] כוּ'. כֶּסֶף מִשְׁנֶה. וּלְשׁוֹן הָרַאֲבַ"ד, טָמֵא, וְאוּלַי מִפְּנֵי שֶׁמִּשָּׁעָה שֶׁנָּפְלוּ בַתְּחִלָּה, הֻחְזְקוּ גַם הֵם בִּשְׁפִיכוּת שֶׁכְּבָר נִשְׁכְּחָה נְפִילָתָם. וְכָתַב הַכֶּסֶף מִשְׁנֶה, וְאִם תֹּאמַר, לְפִי זֶה אֲפִלּוּ לֹא נָפְלוּ מֵי שְׁפִיכוּת נַמִּי. וְיֵשׁ לוֹמַר, דְּלֹא גַזּוּר אֶלָּא כְשֶׁיֵּשׁ שָׁם מֵי שְׁפִיכוּת: (עיקר תוי"ט).

פרק ב - משנה ד

הַטּוֹרֵף אֶת גַּגּוֹ וְהַמְכַבֵּס אֶת כְּסוּתוֹ וְיָרְדוּ עֲלֵיהֶן גְּשָׁמִים, אִם רֹב מִן הַטָּמֵא, טָמֵא. וְאִם רֹב מִן הַטָּהוֹר, טָהוֹר. מֶחֱצָה לְמֶחֱצָה, טָמֵא. רַבִּי יְהוּדָה אוֹמֵר, אִם הוֹסִיפוּ לְנַטֵּף:

 ר"ע מברטנורה  הַמְטַהֵר אֶת גַּגּוֹ. גָּרְסִינַן. וְאִית דְּגָרַס הַטּוֹרֵף אֶת גַּגּוֹ, פֵּרוּשׁ שֶׁטָּח גַּגּוֹ טִיט אוֹ סִיד לְהַשְׁווֹתוֹ. וְהַמַּיִם שֶׁנּוֹתְנִין עָלָיו כְּשֶׁמְּטַהֲרוֹ אוֹ כְשֶׁטָּחוֹ בְטִיט, סְתָמָן שׁוֹפְכִין הֵן וּמְטַמְּאִין. וְכֵן הַמַּיִם שֶׁמְּכַבֵּס בָּהֶן אֶת כְּסוּתוֹ: וְיָרְדוּ עֲלֵיהֶן גְּשָׁמִין. שֶׁהֵן מַיִם טְהוֹרִים. וְהַגַּג אוֹ הַכְּסוּת מְנַטְּפִים מַיִם: אִם רֹב מִן הַטָּהוֹר טָהוֹר. כְּגוֹן שֶׁבַּתְּחִלָּה הָיָה מְנַטֵּף טִפִּים דַּקּוֹת וּלְאַחַר כֵּן מְנַטֵּף טִפִּים עָבוֹת, בְּיָדוּעַ שֶׁרֹב מִן הַטָּהוֹר: אִם הוֹסִיפוּ לְנַטֵּף. שֶׁמְּמַהֲרִים הַטִּפִּים לֵירֵד זוֹ אַחַר זוֹ יוֹתֵר מִבַּתְּחִלָּה, טָהוֹר, דְּוַדַּאי רַבּוּ. וְאַף עַל פִּי שֶׁאֵין הַטִּפּוֹת עַצְמָן יוֹתֵר גְּדוֹלוֹת מִמַּה שֶּׁהָיוּ בַתְּחִלָּה. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: (ר"ע מברטנורה).

פרק ב - משנה ה

עִיר שֶׁיִּשְׂרָאֵל וְנָכְרִים דָּרִים בָּהּ וְהָיָה בָהּ מֶרְחָץ מַרְחֶצֶת בְּשַׁבָּת, אִם רֹב נָכְרִים, רוֹחֵץ מִיָּד. וְאִם רֹב יִשְׂרָאֵל, יַמְתִּין כְּדֵי שֶׁיֵּחַמּוּ הַחַמִּין. מֶחֱצָה לְמֶחֱצָה, יַמְתִּין כְּדֵי שֶׁיֵּחַמּוּ הַחַמִּין. רַבִּי יְהוּדָה אוֹמֵר, בְּאַמְבָּטִי קְטַנָּה, אִם יֶשׁ בָּהּ רָשׁוּת, רוֹחֵץ בָּהּ מִיָּד:

 ר"ע מברטנורה  רוֹחֵץ בָּהּ מִיָּד. לְמוֹצָאֵי שַׁבָּת, וְאֵין צָרִיךְ לְהַמְתִּין כְּדֵי שֶׁיֵּחַמּוּ. דְּכֵיוָן דְּרֹב נָכְרִים, אַדַּעְתָּא דְּנָכְרִים הוּחֲמוּ: אִם יֵשׁ בָּהּ רָשׁוּת. הַמַּלְכוּת קְרוּיָה רָשׁוּת, מִפְּנֵי שֶׁהָרְשׁוּת בְּיָדָהּ לַעֲשׂוֹת כִּרְצוֹנָהּ. וְלָאו דַּוְקָא מֶלֶךְ, אֶלָּא אִם יֵשׁ שָׁם אָדָם חָשׁוּב שֶׁיֵּשׁ לוֹ עֲשָׂרָה עֲבָדִים שֶׁמְּחַמְּמִין לוֹ עֲשָׂרָה קֻמְקוּמִיּוֹת מַיִם בְּבַת אַחַת, מֻתָּר לִרְחֹץ בָּהּ מִיָּד, וְאַף עַל פִּי שֶׁרֹב הָעִיר יִשְׂרָאֵל. דְּאָמְרִינַן לִשְׁמוֹ שֶׁל אוֹתוֹ אָדָם חָשׁוּב הוּחֲמוּ: (ר"ע מברטנורה).

פרק ב - משנה ו

מָצָא בָהּ יָרָק נִמְכָּר, אִם רֹב נָכְרִים, לוֹקֵחַ מִיָּד. וְאִם רֹב יִשְׂרָאֵל, יַמְתִּין כְּדֵי שֶׁיָּבֹאוּ מִמָּקוֹם קָרוֹב {ה}. מֶחֱצָה לְמֶחֱצָה, יַמְתִּין כְּדֵי שֶׁיָּבֹאוּ מִמָּקוֹם קָרוֹב. וְאִם יֶשׁ בּוֹ רָשׁוּת, לוֹקֵחַ מִיָּד:

 ר"ע מברטנורה  הָיָה בָהּ יָרָק נִמְכָּר. יָרָק שֶׁנִּתְלַשׁ בְּשַׁבָּת, נִמְכָּר לְמוֹצָאֵי שַׁבָּת: אִם רֹב נָכְרִים לוֹקֵחַ מִיָּד. דְּכֵיוָן דְּרֹב נָכְרִים, אַדַּעְתָּא דְנָכְרִים נִתְלָשׁ: כְּדֵי שֶׁיָּבֹאוּ מִמָּקוֹם קָרוֹב. שֶׁיֵּשׁ שָׁם יָרָק. וְאַף עַל גַּב דְּשֶׁמָּא הוּבָא מִמָּקוֹם רָחוֹק, אֵין צָרִיךְ לְהַמְתִּין אֶלָּא כְדֵי שֶׁיִּלְקְטוּ יָרָק מִמָּקוֹם קָרוֹב שֶׁיֵּשׁ בּוֹ יָרָק {ו} וִיבִיאוּהוּ מִשָּׁם. שֶׁלֹּא אָמְרוּ חֲכָמִים וְלָעֶרֶב יַמְתִּין בִּכְדֵי שֶׁיַּעֲשׂוּ, אֶלָּא גְזֵרָה שֶׁמָּא יֹאמַר לְנָכְרִי בְּשַׁבָּת לֵךְ וְהָבֵא {ז}, וּכְשֶׁרוֹאֶה שֶׁצָּרִיךְ לְהַמְתִּין לָעֶרֶב כְּדֵי שֶׁיָּבֹא הַדָּבָר הַהוּא וַאֲפִלּוּ מִמָּקוֹם קָרוֹב, לֹא יֹאמַר לַנָּכְרִי לַהֲבִיאוֹ בְּשַׁבָּת: וְאִם יֵשׁ שָׁם רָשׁוּת לוֹקֵחַ מִיָּד. דְּאַדַּעְתָּא דְאָדָם חָשׁוּב רְגִילִין תָּמִיד לְהָבִיא, וְלֹא אַדַּעְתָּא דְיִשְׂרָאֵל הֱבִיאוֹ: (ר"ע מברטנורה).

 עיקר תוי"ט  {ה} קָרוֹב. כְּדֵי שֶׁיֵּלֵךְ הַנָּכְרִי וְיִתְלֹשׁ וְיִגְמֹר הַמְּלָאכָה וְיַחֲזֹר לְכָאן. וְשִׁעוּר הַתְּלִישָׁה אֵין צָרִיךְ לְהַזְכִּיר, שֶׁהִיא נַעֲשֵׂית בְּשָׁעָה מֻעֶטֶת. הָרַ"ן: {ו} וּבְפֶרֶק כ"ג דְּשַׁבָּת מִשְׁנָה ד', אֶלָּא אִם כֵּן בָּאוּ מִמָּקוֹם קָרוֹב, פֵּרֵשׁ הָרַ"ב אֶלָּא אִם כֵּן נוֹדַע בְּבֵרוּר שֶׁבָּאוּ מִתּוֹךְ הַתְּחוּם, כָּתְבוּ הַתּוֹסָפוֹת, דְּמֵת שָׁאנֵי דְּאָוְשִׁי מִלְּתָא. וּלְמַה שֶּׁכָּתַב הָרַ"ן כוּ', יֵשׁ לוֹמַר דְּמָקוֹם קָרוֹב דְּהָכָא אֵינוֹ כְמָקוֹם קָרוֹב דְּהָתָם, אֶלָּא לַאֲפוֹקֵי מָקוֹם רָחוֹק, וּלְעוֹלָם בְּחוּץ לַתְּחוּם, וְצָרִיךְ לְהַמְתִּין עַד שֶׁיָּבֹאוּ מִמֶּנּוּ, אֲבָל אֵין חוֹשְׁשִׁין לַמָּקוֹם הַיּוֹתֵר רָחוֹק מִזֶּה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} פֵּרוּשׁ, לֵךְ וּלְקֹט: (עיקר תוי"ט).

פרק ב - משנה ז

מָצָא בָהּ תִּינוֹק מֻשְׁלָךְ, אִם רֹב נָכְרִים, נָכְרִי. וְאִם רֹב יִשְׂרָאֵל, יִשְׂרָאֵל. מֶחֱצָה לְמֶחֱצָה, יִשְׂרָאֵל. רַבִּי יְהוּדָה אוֹמֵר, הוֹלְכִין אַחַר רֹב הַמַּשְׁלִיכִין:

 ר"ע מברטנורה  אִם רֹב נָכְרִים נָכְרִי. וּמֻתָּר לְהַאֲכִילוֹ נְבֵלוֹת וּטְרֵפוֹת שְׁקָצִים וּרְמָשִׂים בַּיָּדַיִם {ח}: וְאִם רֹב יִשְׂרָאֵל יִשְׂרָאֵל. וְחַיָּבִים לְהַחֲזִיר לוֹ אֲבֵדָתוֹ {ט} כְּיִשְׂרָאֵל. וּרְבוּתָא קָא מַשְׁמַע לָן, דְּמוֹצִיאִין מָמוֹן מִיַּד יִשְׂרָאֵל שֶׁזָּכָה בוֹ, וְלֹא אָמְרִינַן אוֹקֵי מָמוֹנָא אַחַזְקָתֵיהּ שֶׁזָּכָה בוֹ עַד דְּמַיְתֵי הִיאַךְ רְאָיָה שֶׁהוּא יִשְׂרָאֵל: מֶחֱצָה עַל מֶחֱצָה יִשְׂרָאֵל. לְעִנְיַן נְזָקִין {י}, שֶׁאִם שׁוֹרוֹ נָגַח שׁוֹר שֶׁל יִשְׂרָאֵל, מְשַׁלֵּם חֲצִי נֶזֶק בִּלְבָד. וְאֵינוֹ מְשַׁלֵּם נֶזֶק שָׁלֵם כְּדִין שׁוֹר שֶׁל נָכְרִי שֶׁנָּגַח שׁוֹר שֶׁל יִשְׂרָאֵל {יא}, דְּבֵין תָּם בֵּין מוּעָד מְשַׁלֵּם נֶזֶק שָׁלֵם. דְּמָצֵי אָמַר לֵיהּ, אַיְתֵּי רְאָיָה דְּלָאו יִשְׂרָאֵל אֲנָא וּשְׁקוֹל. וְלִשְׁאָר כָּל הַדְּבָרִים הָוֵי סָפֵק, וְדָנִין בּוֹ לְהַחְמִיר. מִי שֶׁהֲרָגוֹ, אֵינוֹ נֶהֱרָג עָלָיו. קִדֵּשׁ אִשָּׁה, צְרִיכָה גֵט מִסָּפֵק. וְאֵינוֹ כְיִשְׂרָאֵל גָּמוּר עַד שֶׁיִּטְבֹּל לְשֵׁם גֵּרוּת: אַחַר רֹב הַמַּשְׁלִיכִין. הַיְנוּ אַחַר הַנָּכְרִים וְאַף עַל פִּי שֶׁהֵן מֻעָט. וַאֲפִלּוּ לֹא הָיְתָה שָׁם אֶלָּא נָכְרִית אַחַת, הִיא חֲשׁוּדָה לְהַשְׁלִיךְ. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ח} וְכָל שֶׁכֵּן לְעִנְיַן אֲבֵדָה וְלִנְזָקִין. רַשִׁ"י. וּבַגְּמָרָא, דִּלְעִנְיַן פִּקּוּחַ נֶפֶשׁ אֵינוֹ כֵן: {ט} וְאַף עַל גַּב דְּקַיְמָא לָן אֵין הוֹלְכִין בְּמָמוֹן אַחַר הָרֹב, הָנֵי מִלֵּי כְּשֶׁבָּא לְיָדוֹ הַמָּמוֹן בְּהֶתֵּר וּמִדַּעַת נְתָנָם לוֹ. הָרַ"שׁ וְתוֹסָפוֹת. וּלְיוֹחֲסִין לֹא מָצֵי לְאוֹקְמֵי, דְּבָעִינַן תְּרֵי רֻבָּא: {י} דִּלְהַחֲזִיר אֲבֵדָה אוֹמֵר לוֹ אַיְתֵּי רְאָיָה דְיִשְׂרָאֵל אַתְּ. תּוֹסָפוֹת: {יא} תֵּימַהּ, דַּהֲוָה מָצֵי לְמִתְנֵי מֶחֱצָה עַל מֶחֱצָה נָכְרִים, וּמְנַגַּח תּוֹרָא דִידָן לְתוֹרָא דִידֵיהּ. תּוֹסָפוֹת: (עיקר תוי"ט).

פרק ב - משנה ח

מָצָא בָהּ מְצִיאָה, אִם רֹב נָכְרִים, אֵינוֹ צָרִיךְ לְהַכְרִיז. וְאִם רֹב יִשְׂרָאֵל, צָרִיךְ לְהַכְרִיז. מֶחֱצָה לְמֶחֱצָה, צָרִיךְ לְהַכְרִיז. מָצָא בָהּ פַּת, הוֹלְכִין אַחַר רֹב הַנַּחְתּוֹמִין. וְאִם הָיְתָה פַת עִסָּה, הוֹלְכִים אַחַר רֹב אוֹכְלֵי פַת עִסָּה {יב}. רַבִּי יְהוּדָה אוֹמֵר, אִם הָיְתָה פַת קִבָּר, הוֹלְכִין אַחַר רֹב אוֹכְלֵי פַת קִבָּר:

 ר"ע מברטנורה  רֹב נַחְתּוֹמִים. אִם רֹב נַחְתּוֹמִים נָכְרִים, הַפַּת אֲסוּרָה, דְּפַת שֶׁל נָכְרִים הוּא מִשְּׁמֹנָה עָשָׂר דָּבָר שֶׁגָּזְרוּ בּוֹ בַיּוֹם: פַּת עִסָּה. פַּת נְקִיָּה: פַּת קִבָּר. פַּת שֶׁאֵינָהּ נְקִיָּה: אַחַר רֹב אוֹכְלֵי פַת קִבָּר. אִם רֻבָּן נָכְרִים, הַפַּת אֲסוּרָה. אִי נַמִּי יֵשׁ לְפָרֵשׁ, אַחַר רֹב נַחְתּוֹמִין, אַחַר רֹב אוֹכְלֵי פַת עִסָּה וְכוּ', אִם רֻבָּן חֲבֵרִים, הַפַּת טְהוֹרָה וְאֵין צָרִיךְ לְעַשֵּׂר. וְאִם רֻבָּן עַמֵּי הָאָרֶץ, הַפַּת טְמֵאָה וְצָרִיךְ לְעַשֵּׂר: (ר"ע מברטנורה).

 עיקר תוי"ט  {יב} עִסָּה. נִרְאֶה לִי, דְּפַת עִסָּה הָיָה מִנְהָגָם דְּבַעֲלֵי בָתִּים אוֹפִין בְּבָתֵּיהֶם יוֹתֵר מִן הַנַּחְתּוֹמִין, וּלְפִיכָךְ אֵין הוֹלְכִין בּוֹ אַחַר רֹב נַחְתּוֹמִין. וְכֵן בְּפַת קִבָּר לְרַבִּי יְהוּדָה. וְנִרְאֶה לִי, דְּתַנָּא קַמָּא וְרַבִּי יְהוּדָה לֹא פְלִיגֵי, דְּמַר כִּי אַתְרֵיהּ כוּ', דִּבְאַתְרֵיהּ דְּתַנָּא קַמָּא הַנַּחְתּוֹמִים אוֹפִים פַּת קִבָּר, וּבְאַתְרֵיהּ דְּרַבִּי יְהוּדָה בַּעֲלֵי בָתִּים אוֹפִין פַּת קִבָּר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ב - משנה ט

מָצָא בָהּ בָּשָׂר, הוֹלְכִין אַחַר רֹב הַטַּבָּחִים. אִם הָיָה מְבֻשָּׁל, הוֹלְכִים אַחַר רֹב אוֹכְלֵי בָשָׂר מְבֻשָּׁל:

 ר"ע מברטנורה  הוֹלְכִין אַחַר רֹב טַבָּחִים. אִם רֹב טַבָּחֵי יִשְׂרָאֵל, הַבָּשָׂר מֻתָּר. דְּאֵין הֲלָכָה כְּרַב דְּאָמַר בָּשָׂר שֶׁנִּתְעַלֵּם מִן הָעַיִן אָסוּר {יג}. וְרַב מוֹקֵי מַתְנִיתִין בְּעוֹמֵד וְרוֹאֶה מִשָּׁעָה שֶׁנִּשְׁחַט עַד שֶׁנָּפְלָה, אֲבָל לֹא יָדַע מִי הַשּׁוֹחֵט: (ר"ע מברטנורה).

 עיקר תוי"ט  {יג} וְזֶה דְּלֹא כִדְמַשְׁמַע מִלְּשׁוֹנוֹ בְּפֶרֶק ז' דִּשְׁקָלִים מִשְׁנָה ג': (עיקר תוי"ט).

פרק ב - משנה י

הַמּוֹצֵא פֵרוֹת בַּדֶּרֶךְ, אִם רֹב מַכְנִיסִין לְבָתֵּיהֶן, פָּטוּר. וְלִמְכֹּר בַּשּׁוּק, חַיָּב. מֶחֱצָה לְמֶחֱצָה, דְּמַאי. אוֹצָר שֶׁיִּשְׂרָאֵל וְנָכְרִים מַטִּילִין לְתוֹכוֹ, אִם רֹב נָכְרִים, וַדַּאי. וְאִם רֹב יִשְׂרָאֵל, דְּמַאי. מֶחֱצָה לְמֶחֱצָה, וַדַּאי, דִּבְרֵי רַבִּי מֵאִיר. וַחֲכָמִים אוֹמְרִים, אֲפִלּוּ כֻלָּם נָכְרִים וְיִשְׂרָאֵל אֶחָד מַטִּיל לְתוֹכוֹ, דְּמַאי {יח}:

 ר"ע מברטנורה  אִם רֹב מַכְנִיסִין לְבָתֵּיהֶן פָּטוּר. מִלְּעַשֵּׂר {יד}: לִמְכֹּר בַּשּׁוּק חַיָּב. שֶׁרֹב מְבִיאִין לִמְכֹּר בַּשּׁוּק, מִן הַבַּיִת מְבִיאִין אוֹתוֹ, וּכְבָר נִתְחַיֵּב. פֵּרוּשׁ אַחֵר, אִם רֹב מַכְנִיסִין לְבָתֵּיהֶן, פָּטוּר, לְפִי שֶׁרְגִילִין לְעַשֵּׂר קֹדֶם שֶׁיַּכְנִיס לַבַּיִת כְּדֵי שֶׁיּוּכְלוּ לֶאֱכֹל מֵהֶן. דִּלְאַחַר שֶׁהִכְנִיסוּם לַבַּיִת, מִיָּד הֻקְבְּעוּ לְמַעֲשֵׂר. הִלְכָּךְ מַה שֶּׁנּוֹפֵל מֵהֶם בַּדֶּרֶךְ פָּטוּר מִן הַמַּעֲשֵׂר, לְפִי שֶׁהוּא בְחֶזְקַת מְעֻשָּׂר {טו}. אֲבָל כְּשֶׁמַּכְנִיסִין לִמְכֹּר, דַּרְכּוֹ לְהַכְנִיסָן טֶבֶל, לְפִי שֶׁיָּכוֹל לֶאֱכֹל מֵהֶן עֲרַאי, הִלְכָּךְ הַנּוֹפֵל מֵהֶן אֵינוֹ בְחֶזְקַת מְעֻשָּׂר: אִם רֹב נָכְרִים וַדַּאי. רַבִּי מֵאִיר לְטַעְמֵיהּ, דִּסְבִירָא לֵיהּ אֵין קִנְיָן לְנָכְרִי בְּאֶרֶץ יִשְׂרָאֵל לְהַפְקִיעַ מִן הַמַּעֲשֵׂר {טז}, וּפֵרוֹת שֶׁגָּדְלוּ בְּקַרְקָעוֹת שֶׁל נָכְרִי, חַיָּבִים בְּמַעֲשֵׂר, וְהַנָּכְרִי פְּשִׁיטָא דְלָא עִשֵּׂר, הִלְכָּךְ וַדַּאי טֶבֶל הֵן: וְאִם רֹב יִשְׂרָאֵל דְּמַאי. כְּדִין פֵּרוֹת הַנִּמְצָאִים בְּיַד עַם הָאָרֶץ, שֶׁצָּרִיךְ לְהַפְרִישׁ מֵהֶן תְּרוּמַת מַעֲשֵׂר וּמַעֲשֵׂר שֵׁנִי {יז}: וַחֲכָמִים אוֹמְרִים וְכוּ'. חֲכָמִים סְבִירָא לְהוּ יֵשׁ קִנְיָן לְנָכְרִי בְּאֶרֶץ יִשְׂרָאֵל לְהַפְקִיעַ מִן הַמַּעֲשֵׂר. וְאִם לֹא הָיָה יִשְׂרָאֵל מַטִּיל לְתוֹכוֹ, הָיָה הַכֹּל פָּטוּר מִן הַמַּעֲשֵׂר. אֲבָל כְּשֶׁיֵּשׁ אֲפִלּוּ יִשְׂרָאֵל אֶחָד שֶׁמַּטִּיל לְתוֹכוֹ, הַכֹּל דְּמַאי, בִּשְׁבִיל פֵּרוֹתָיו שֶׁל אוֹתוֹ יִשְׂרָאֵל, שֶׁהֵן חַיָּבִין. וַהֲלָכָה כַּחֲכָמִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {יד} דַּעֲדַיִן לֹא רָאָה פְנֵי הַבַּיִת וְלֹא נִתְחַיֵּב. הָרַ"שׁ: {טו} הָרַמְבַּ"ם. וְצָרִיךְ עִיּוּן, דְּמִשּׁוּם דִּרְגִילִין לְעַשֵּׂר יְהֵא בְחֶזְקַת מְעֻשָּׂר וְיֵצֵא מִכְּלַל דְּמַאי. וְדֹחַק לוֹמַר דְּמַיְרֵי לְעִנְיַן אַכְסַנְיָא וְעָנִי. וּבְחִבּוּרוֹ כָתַב, פָּטוּר מִלְּעַשֵּׂר, שֶׁעֲדַיִן לֹא נִקְבַּע לְמַעֲשֵׂר: {טז} זֶהוּ שִׁיטַת הַיְרוּשַׁלְמִי בִּלְבַד. וְהַיְנוּ דְסִיֵּם הָרַ"שׁ, וַאֲפִלּוּ יֵשׁ קִנְיָן מִכָּל מָקוֹם וַדַּאי אֵינוֹ מְעֻשָּׂר הוּא: {יז} אֲבָל מַעֲשֵׂר רִאשׁוֹן, הַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה. אֲבָל נִרְאֶה דְּהָכָא אֲנַן אַלִּיבָּא דְרַבִּי מֵאִיר קַיָּמִים דִּסְבִירָא לֵיהּ דְּאַף מַעֲשֵׂר רִאשׁוֹן אָסוּר לְזָרִים, וְצָרִיךְ לְהַפְרִישׁ אַף מַעֲשֵׂר רִאשׁוֹן. אִי נַמִּי אַף לְרַבִּי מֵאִיר לֹא הֶחְמִירוּ בִדְמַאי, מִשּׁוּם דְּרֹב עַמֵּי הָאָרֶץ מְעַשְּׂרִין. וְהַשְׁתָּא נִיחָא דְּלֹא פֵרֵשׁ הָרַ"ב בְּרֵישָׁא מֶחֱצָה לְמֶחֱצָה דְּמַאי שֶׁצָּרִיךְ לְהַפְרִישׁ, אֶלָּא הָכָא אִצְטְרִיךְ לְאַשְׁמְעִינַן דְּאַף לְרַבִּי מֵאִיר אֵין צָרִיךְ לְהַפְרִישׁ מַעֲשֵׂר רִאשׁוֹן. נִרְאֶה לִי: {יח} דְּמַאי. וְלֹא יָדַעְנָא טַעֲמָא מַאי, דַּאֲפִלּוּ וַדַּאי תְרוּמָה עוֹלָה בְּמֵאָה וְאֶחָד, וּתְרוּמַת מַעֲשֵׂר אֵינָהּ אֶלָּא אֶחָד מִמֵּאָה, נִמְצָא כְּשֶׁיֵּשׁ עוֹד אֲפִלּוּ מִנָּכְרִי אֶחָד, נִתְבַּטְּלָה. וַאֲפִלּוּ לְהָרִים אֵינוֹ צָרִיךְ. דְּאֵין כָּאן גֶּזֶל הַשֵּׁבֶט, דְּסָפֵק הוּא וְהַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה: (עיקר תוי"ט).

פרק ב - משנה יא

פֵּרוֹת שְׁנִיָּה שֶׁרַבּוּ עַל שֶׁל שְׁלִישִׁית, וְשֶׁל שְׁלִישִׁית עַל שֶׁל רְבִיעִית, וְשֶׁל רְבִיעִית עַל שֶׁל חֲמִישִׁית {יט}, וְשֶׁל חֲמִישִׁית עַל שֶׁל שִׁשִּׁית, וְשֶׁל שִׁשִּׁית עַל שֶׁל שְׁבִיעִית, וְשֶׁל שְׁבִיעִית עַל שֶׁל מוֹצָאֵי שְׁבִיעִית, הוֹלְכִין אַחַר הָרֹב. מֶחֱצָה לְמֶחֱצָה, לְהַחֲמִיר:

 ר"ע מברטנורה  פֵּרוֹת שְׁנִיָּה שֶׁרַבּוּ עַל שֶׁל שְׁלִישִׁית. שָׁנָה רִאשׁוֹנָה וּשְׁנִיָּה שֶׁל שְׁמִטָּה נוֹהֵג בָּהֶן מַעֲשֵׂר רִאשׁוֹן וּמַעֲשֵׂר שֵׁנִי, וְכֵן רְבִיעִית וַחֲמִישִׁית. אֲבָל שָׁנָה שְׁלִישִׁית וְשִׁשִּׁית, מַעֲשֵׂר רִאשׁוֹן וּמַעֲשַׂר עָנִי: וְשֶׁל שְׁבִיעִית. יֵשׁ בּוֹ קְדֻשַּׁת שְׁבִיעִית וּפָטוּר מִן הַמַּעֲשֵׂר. וּלְעִנְיַן כָּל הָנָךְ דִּינֵי קָתָנֵי דְהוֹלְכִים אַחַר הָרֹב: מֶחֱצָה לְמֶחֱצָה לְהַחֲמִיר. וּלְעִנְיַן סְפֵקָא דְמַעֲשֵׂר שֵׁנִי וּמַעֲשַׂר עָנִי, יַפְרִישׁ מַעֲשֵׂר אֶחָד, וִיחַלֵּל הַפֵּרוֹת עַל הַמָּעוֹת, וִיחַלֵּק הַפֵּרוֹת לַעֲנִיִּים, וְהַמָּעוֹת יֹאכַל בִּירוּשָׁלַיִם. וּלְעִנְיַן סְפֵקָא דִשְׁבִיעִית, חַיָּבִין בְּמַעֲשֵׂר וְחַיָּבִין בִּקְדֻשַּׁת שְׁבִיעִית שֶׁלֹּא לַעֲשׂוֹת בָּהֶן סְחוֹרָה וּמְלוּגְמָא, וְחַיָּבִין בְּבִעוּר: (ר"ע מברטנורה).

 עיקר תוי"ט  {יט} רְבִיעִית עַל שֶׁל חֲמִישִׁית. בִּכְדִי נָקְטוּ, דְּבִשְׁתֵּיהֶן מַעֲשֵׂר שֵׁנִי. וּמַהֲרַ"ם כָּתַב, לְעִנְיַן מְעַשֵּׂר מֵחָדָשׁ עַל הַיָּשָׁן אוֹ לְהֵפֶךְ. וַהֲוָה מָצֵי לְמֵימַר נַמִּי מִן הָרִאשׁוֹנָה עַל הַשְּׁנִיָּה, אֶלָּא דְּהַךְ רְבִיעִית עַל חֲמִישִׁית אַגַּב גְּרָרָא לֹא רָצָה לִדְלֹג: (עיקר תוי"ט).

פרק ג - משנה א

שַׂק שֶׁהוּא מָלֵא פֵרוֹת {א} וּנְתָנוֹ עַל גַּב הַנָּהָר, אוֹ עַל פִּי הַבּוֹר, אוֹ עַל מַעֲלוֹת הַמְּעָרָה, וְשָׁאֲבוּ, כֹּל שֶׁשָּׁאֲבוּ, בְּכִי יֻתַּן. רַבִּי יְהוּדָה אוֹמֵר, כֹּל שֶׁהוּא כְנֶגֶד הַמַּיִם, בְּכִי יֻתַּן. וְכֹל שֶׁאֵינוֹ כְנֶגֶד הַמַּיִם, אֵינוֹ בְּכִי יֻתַּן:

 ר"ע מברטנורה  שַׂק שֶׁהוּא מָלֵא. מַעֲלוֹת הַמְּעָרָה. הַמְלֵאָה מַיִם: כֹּל שֶׁשָּׁאֲבוּ. הַפֵּרוֹת מִלַּחְלוּחִית הַמַּיִם, הֲרֵי זֶה בְּכִי יֻתַּן. דְּנִיחָא לֵיהּ שֶׁיִּשְׁאֲבוּ הַפֵּרוֹת מִן הַמַּיִם, שֶׁמִּתּוֹךְ כָּךְ הֵן נִרְאִין גַּסִּים וּנְפוּחִים: רַבִּי יְהוּדָה אוֹמֵר כֹּל שֶׁהוּא כְנֶגֶד הַמַּיִם כוּ'. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} פֵּרוֹת. רוֹצֶה לוֹמַר כִּי רֹב הַזְּרָעִים מוֹשְׁכִים הַמַּיִם, וְכָל שֶׁכֵּן הַחִטָּה. הָרַמְבַּ"ם. וּבְחִבּוּרוֹ כָתַב זֵרְעוֹנִים: (עיקר תוי"ט).

פרק ג - משנה ב

חָבִית שֶׁהִיא מְלֵאָה פֵרוֹת וּנְתוּנָה לְתוֹךְ הַמַּשְׁקִין, אוֹ מְלֵאָה מַשְׁקִין וּנְתוּנָה לְתוֹךְ הַפֵּרוֹת, וְשָׁאֲבוּ, כֹּל שֶׁשָּׁאֲבוּ, בְּכִי יֻתַּן. בְּאֵלּוּ מַשְׁקִים אָמְרוּ, בְּמַיִם וּבְיַיִן וּבְחֹמֶץ. וּשְׁאָר כָּל הַמַּשְׁקִין, טְהוֹרִין. רַבִּי נְחֶמְיָה מְטַהֵר בְּקִטְנִית, שֶׁאֵין הַקִּטְנִית שׁוֹאָבֶת:

 ר"ע מברטנורה  חָבִית. שֶׁל חֶרֶס, אוֹ שֶׁל כְּלֵי אֶבֶן הָרַכִּים שֶׁהֵן כְּחֶרֶס. אֲבָל שְׁאָר כֵּלִים לֹא בָלְעֵי כָּל כָּךְ עַד שֶׁיִּהְיוּ הַפֵּרוֹת שֶׁבְּתוֹכָן שׁוֹאֲבִים: שָׁאֲבוּ. נִתְלַחְלְחוּ הַפֵּרוֹת וּמַשְׁקֶה טוֹפֵחַ עֲלֵיהֶן: הָכִי גָרְסִינַן, וּשְׁאָר כָּל הַמַּשְׁקִים טְהוֹרִים. שְׁאָר שִׁבְעָה מַשְׁקִים הַמַּכְשִׁירִים לְקַבֵּל טֻמְאָה, טְהוֹרִים הַפֵּרוֹת שֶׁבְּתוֹךְ הֶחָבִית אִם נִתְלַחְלְחוּ מֵהֶן {ב}. דְּדַוְקָא מַיִם וְיַיִן וְחֹמֶץ שֶׁהֵן דַּקִּים וּצְלוּלִים, הֵן נִבְלָעִין בֶּחָבִית מִפְּנֵי דַקּוּתָן, וְהַפֵּרוֹת שׁוֹאֲבִים מֵהֶן. אֲבָל שֶׁמֶן דְּבַשׁ חָלָב דָּם טַל, הֵן עָבִין, וּמִפְּנֵי עָבְיָן אֵינָן יְכוֹלִין לְהִבָּלַע בְּתוֹךְ הַכְּלִי כָּל כָּךְ עַד שֶׁיִּהְיוּ הַפֵּרוֹת שֶׁבְּתוֹךְ הֶחָבִית שׁוֹאֲבִים מֵהֶן, הִלְכָּךְ אֵינָן בְּכִי יֻתַּן: שֶׁאֵין הַקִּטְנִית שׁוֹאָבֶת. כְּגוֹן פּוֹלִים וַאֲפוּנִים וְכַיּוֹצֵא בָהֶן. וְאֵין הֲלָכָה כְּרַבִּי נְחֶמְיָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ב} דְּהוֹאִיל וּסְתָמָן אֵינָן נִבְלָעִים בֶּחָבִית לֹא הָיוּ לְרָצוֹן. וְצָרִיךְ לוֹמַר שֶׁגַּם סוֹפָן לֹא לְרָצוֹן, דְּכָל שֶׁסּוֹפוֹ לְרָצוֹן אַף עַל פִּי שֶׁאֵין תְּחִלָּתוֹ לְרָצוֹן. וְזֶה דֹחַק, דְּכֵיוָן שֶׁהַפֵּרוֹת נִרְאִין גַּסִּין, מַאי שְׁנָא מִשְּׁלֹשָׁה הַמַּשְׁקִין דְּמִסְּתָמָא אָמְרִינַן דְּנִיחָא לֵיהּ. מִיהוּ מִדִּתְנַן וְשָׁאֲבוּ וְדִכְוָתֵיהּ בְּסֵיפָא, מוּכָח כְּדִבְרֵי הָרַ"ב. תּוּ נִרְאֶה לִי דְּהָרַ"ב מְפָרֵשׁ לְכֻלַּהּ מִלְּתָא דַּוְקָא כְשֶׁהַמַּשְׁקִים חוּץ לֶחָבִית. אֲבָל בְּתוֹךְ הֶחָבִית, אִין הָכִי נַמִּי דִּיכוֹלִין לְהִפָּלֵט. וְהָרַמְבַּ"ם מְפָרֵשׁ, דְּסֵיפָא בְּמִסְּתָמָא אָמְרִינַן שֶׁלֹּא שָׁאֲבוּ וְאֵין צָרִיךְ לָחוּשׁ וְלִבְדֹּק, אֲבָל אִם רָאִינוּ שֶׁשָּׁאֲבוּ אִין הָכִי נַמִּי דַּהֲווּ בְּכִי יֻתַּן, וְלֹא מְחַלֵּק בֵּין חוּץ לֶחָבִית לְתוֹכוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ג - משנה ג

הָרוֹדֶה פַת חַמָּה וּנְתָנָהּ עַל פִּי חָבִית שֶׁל יַיִן {ג}, רַבִּי מֵאִיר מְטַמֵּא. רַבִּי יְהוּדָה מְטַהֵר. רַבִּי יוֹסֵי מְטַהֵר בְּשֶׁל חִטִּים {ד} וּמְטַמֵּא בְשֶׁל שְׂעֹרִים, מִפְּנֵי שֶׁהַשְּׂעֹרִים שׁוֹאֲבוֹת:

 ר"ע מברטנורה  הָרוֹדֶה פַת חַמָּה. שֶׁנִּלּוֹשָׁה בְּמֵי פֵרוֹת וְלֹא הֻכְשְׁרָה לְקַבֵּל טֻמְאָה: רַבִּי מֵאִיר מְטַמֵּא. כְּלוֹמַר הֻכְשַׁר הַלֶּחֶם לְקַבֵּל טֻמְאָה מֵחֲמַת הַיַּיִן שֶׁנִּבְלַע בּוֹ. אִי נַמִּי, אִם הָיָה הַיַּיִן טָמֵא, מְטַמֵּא רַבִּי מֵאִיר מֵחֲמַת הַיַּיִן טָמֵא הַנִּבְלָע בָּהּ: רַבִּי יוֹסֵי מְטַהֵר בְּשֶׁל חִטִּים. וַהֲלָכָה כְּרַבִּי יוֹסֵי: (ר"ע מברטנורה).

 עיקר תוי"ט  {ג} פַּת כוּ'. נִרְאֶה לִי שֶׁיֵּשׁ לְחַלֵּק בֵּין פַּת לְפֵרוֹת, וּמִשּׁוּם הָכִי תְּנַן הָכָא עַל פִּי חָבִית, וּלְעֵיל אָמְרִינַן דְּמִתּוֹךְ דָּפְנֵי הֶחָבִית נִפְלַט וּמִתְלַחְלֵחַ, וְכֵן צָרִיךְ לוֹמַר לְפִי מַה שֶּׁכָּתַבְתִּי לְעֵיל [אוֹת א'] בְּשֵׁם הָרַמְבַּ"ם דְּזֶרַע חִטָּה שׁוֹאֶבֶת וְהָכָא פָסְקִינַן כְּרַבִּי יוֹסֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} מְטַהֵר כוּ'. וּמַיְרֵי כְּשֶׁלֹּא רָאוּ הַפַּת עַד אַחַר זְמַן וְאֵינוֹ יוֹדֵעַ אִם הָיָה עָלָיו מַשְׁקֶה טוֹפֵחַ. כֶּסֶף מִשְׁנֶה: (עיקר תוי"ט).

פרק ג - משנה ד

הַמְרַבֵּץ אֶת בֵּיתוֹ וְנָתַן בּוֹ חִטִּים וְטָנְנוּ, אִם מֵחֲמַת הַמַּיִם, בְּכִי יֻתַּן. וְאִם מֵחֲמַת הַסֶּלַע, אֵינָן בְּכִי יֻתַּן. הַמְכַבֵּס אֶת כְּסוּתוֹ בַּעֲרֵבָה, נָתַן בָּהּ חִטִּים וְטָנְנוּ, אִם מֵחֲמַת הַמַּיִם, בְּכִי יֻתַּן. אִם מֵחֲמַת עַצְמָן {ה}, אֵינָן בְּכִי יֻתַּן. הַמְטַנֵּן בְּחֹל, הֲרֵי זֶה בְכִי יֻתַּן. מַעֲשֶׂה בְאַנְשֵׁי הַמָּחוֹז שֶׁהָיוּ מְטִנִּין בְּחֹל, אָמְרוּ לָהֶם חֲכָמִים, אִם כָּךְ הֱיִיתֶם עוֹשִׂים, לֹא עֲשִׂיתֶם טָהֳרָה מִימֵיכֶם:

 ר"ע מברטנורה  הַמְרַבֵּץ אֶת בֵּיתוֹ. מְזַלֵּף בּוֹ מַיִם כְּדֵי שֶׁלֹּא יַעֲלֶה הָאָבָק: וְטָנְנוּ. נִתְלַחְלְחוּ. לְשׁוֹן אִם הָיְתָה שָׂדֶה מְטֻנֶּנֶת, שֶׁפֵּרוּשׁוֹ שָׂדֶה לַחָה, בְּפֶרֶק קַמָּא דְּמוֹעֵד קָטָן [דַּף ו]: אִם מֵחֲמַת הַסֶּלַע. צוּר שֶׁמִּשֵּׁשֶּׁת יְמֵי בְרֵאשִׁית, וְהוּא רִצְפַּת הַבַּיִת, וּמִתּוֹךְ כָּךְ נִתְלַחְלְחוּ הַחִטִּים: הַמְטַנֵּן בְּחֹל. מַטְמִין הַפֵּרוֹת בְּחֹל כְּדֵי שֶׁיִּטַּנְּנוּ, כְּלוֹמַר שֶׁיִּתְלַחְלְחוּ, שֶׁיְּקַבְּלוּ מִלַּחְלוּחִית הַחוֹל: הֲרֵי זֶה בְכִי יֻתַּן. דְּאֵין חֹל בְּלֹא מַיִם: (ר"ע מברטנורה).

 עיקר תוי"ט  {ה} מֵחֲמַת עַצְמָן. פֵּרוּשׁ שֶׁהָיוּ הַרְבֵּה וְנִתְחַמְּמוּ וְנִתְלַחְלְחוּ. מַהֲרַ"ם. וּמִדְּלֹא קָתָנֵי וְאִם מֵחֲמַת עֲרֵבָה, יֵשׁ לוֹמַר דְּמֵחֲמַת עֲרֵבָה הֻכְשְׁרוּ. וְכֵן כָּתַב הָרַאֲבַ"ד, אֶלָּא דְּהִתְנָה שֶׁיִּתְנֵן שָׁם מֵחֲמַת הַלֵּחוֹת שֶׁבָּהּ וְלֹא מֵחֲמַת שֶׁאֵין לוֹ מָקוֹם לְכָנְסָן. אֲבָל הָרַמְבַּ"ם הֶעְתִּיק וְאִם מֵחֲמַת הָעֲרֵבָה: (עיקר תוי"ט).

פרק ג - משנה ה

הַמְטַנֵּן בְּטִיט הַנָּגוּב, רַבִּי שִׁמְעוֹן אוֹמֵר, אִם יֶשׁ בּוֹ מַשְׁקֶה טוֹפֵחַ, בְּכִי יֻתַּן. וְאִם לָאו, אֵינוֹ בְּכִי יֻתַּן. הַמְרַבֵּץ אֶת גָּרְנוֹ, אֵינוֹ חוֹשֵׁשׁ שֶׁמָּא נָתַן בָּהּ חִטִּים וְטָנְנוּ. הַמְלַקֵּט עֲשָׂבִים כְּשֶׁהַטַּל עֲלֵיהֶם, לְהָטֵן בָּהֶם חִטִּים, אֵינָן בְּכִי יֻתַּן. אִם נִתְכַּוֵּן לְכָךְ, הֲרֵי זֶה בְכִי יֻתַּן. הַמּוֹלִיךְ חִטִּין לִטְחֹן וְיָרְדוּ עֲלֵיהֶן גְּשָׁמִים, אִם שָׂמַח, בְּכִי יֻתַּן. רַבִּי יְהוּדָה אוֹמֵר, אִי אֶפְשָׁר שֶׁלֹּא לִשְׂמֹחַ, אֶלָּא אִם עָמָד:

 ר"ע מברטנורה  נָגוּב. יָבֵשׁ: אֵינוֹ חוֹשֵׁשׁ שֶׁמָּא נָתַן בָּהּ חִטִּים וְטָנְנוּ. הָכִי קָאָמַר, אִם נָתַן בָּהּ חִטִּים, אֵינוֹ חוֹשֵׁשׁ שֶׁמָּא טָנְנוּ {ו}: כְּשֶׁהַטַּל עֲלֵיהֶם. כְּגוֹן שֶׁהָיָה מַשְׂכִּיר לִמְלַאכְתּוֹ וְלִקֵּט עֲשָׂבִים עִם הַטַּל שֶׁעֲלֵיהֶם, אֲבָל הוּא לֹא נִתְכַּוֵּן בִּשְׁבִיל הַטַּל: לְהָטֵן בָּהֶם חִטִּים. לְהַטְמִין בָּהֶם חִטִּים כְּדֵי שֶׁיִּטַּנְּנוּ: אִם נִתְכַּוֵּן לְכָךְ. לַטַּל שֶׁעֲלֵיהֶן: אִי אֶפְשָׁר שֶׁלֹּא לִשְׂמֹחַ. שֶׁהֲרֵי הָיָה צָרִיךְ לְלָתְתָן: אֶלָּא אִם עָמָד. נִתְעַכַּב שָׁם כְּדֵי שֶׁיֵּרְדוּ עֲלֵיהֶם גְּשָׁמִים {ז}: (ר"ע מברטנורה).

 עיקר תוי"ט  {ו} וְצָרִיךְ לוֹמַר דִּבְבַיִת שְׁכִיחַ טְפֵי שֶׁיִּטַּנְּנוּ, וּלְהָכִי תָּנֵי לְעֵיל וְטָנְנוּ, לְפִי שֶׁכֵּן רָגִיל לִהְיוֹת שֶׁמֵּרֹב יֹבֶשׁ הַגֹּרֶן הַמַּיִם נִבְלָעִים בַּקַּרְקַע. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} וְהָכִי קָאָמַר, לְדִבְרֵיכֶם דִּבְשִׂמְחָה לְחוּד סַגִּי, לֹא הָיָה לָכֶם לוֹמַר אִם שָׂמַח, שֶׁהֲרֵי אִי אֶפְשָׁר שֶׁלֹּא לִשְׂמֹחַ, אֶלָּא דְּשִׂמְחָה לְחוּד לֹא סַגִּי, אֲבָל צָרִיךְ שֶׁיַּעֲמֹד: (עיקר תוי"ט).

פרק ג - משנה ו

הָיוּ זֵיתָיו נְתוּנִים בַּגַּג וְיָרְדוּ עֲלֵיהֶן גְּשָׁמִים, אִם שָׂמַח, בְּכִי יֻתַּן. רַבִּי יְהוּדָה אוֹמֵר, אִי אֶפְשָׁר שֶׁלֹּא לִשְׂמֹחַ, אֶלָּא, אִם פָּקַק אֶת הַצִּנּוֹר אוֹ אִם חִלְחֵל לְתוֹכָן:

 ר"ע מברטנורה  פָּקַק הַצִּנּוֹר. סָתַם הַצִּנּוֹר כְּדֵי שֶׁלֹּא יֵצְאוּ הַמַּיִם: אוֹ אִם חִלְחֵל לְתוֹכוֹ. שֶׁנִּעֲנַע וְנִעֵר הַזֵּיתִים בַּמַּיִם שֶׁבְּתוֹךְ הַצִּנּוֹר. דִּסְבִירָא לֵיהּ לְרַבִּי יְהוּדָה, שֶׁאֵין הַמַּחֲשָׁבָה מוֹעֶלֶת לְהַכְשִׁיר עַד שֶׁיַּעֲשֶׂה מַעֲשֶׂה לְגַלּוֹת הַמַּחֲשָׁבָה שֶׁבְּלִבּוֹ. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: (ר"ע מברטנורה).

פרק ג - משנה ז

הַחַמָּרִין שֶׁהָיוּ עוֹבְרִים בַּנָּהָר וְנָפְלוּ שַׂקֵּיהֶם לַמַּיִם, אִם שָׂמְחוּ, בְּכִי יֻתַּן {ח}. רַבִּי יְהוּדָה אוֹמֵר, אִי אֶפְשָׁר שֶׁלֹּא לִשְׂמֹחַ, אֶלָּא, אִם הָפְכוּ. הָיוּ רַגְלָיו מְלֵאוֹת טִיט, וְכֵן רַגְלֵי בְהֶמְתּוֹ, עָבַר בַּנָּהָר, אִם שָׂמַח, בְּכִי יֻתַּן {ט}. רַבִּי יְהוּדָה אוֹמֵר, אִי אֶפְשָׁר שֶׁלֹּא לִשְׂמֹחַ, אֶלָּא, אִם עָמַד וְהֵדִיחַ {י}. בְּאָדָם וּבִבְהֵמָה טְמֵאָה, לְעוֹלָם טָמֵא:

 ר"ע מברטנורה  אִם הָפַךְ הַשַּׂקִּים. לְצַד אַחֵר, כְּדֵי שֶׁיִּשְׁרוּ בַמַּיִם מִשְּׁנֵי צְדָדִין, אָז הֻכְשְׁרוּ: אִם עָמַד וְהֵדִיחַ. אִם עָמַד בַּמַּיִם לְהָדִיחַ רַגְלֵי בְהֶמְתּוֹ, אָז מַכְשִׁירִים מַיִם הָעוֹלִין בְּרַגְלֵיהֶם אִם נָפְלוּ עַל הַפֵּרוֹת: בְּאָדָם וּבִבְהֵמָה טְמֵאָה. מַכְשִׁירִין הַמַּיִם הָעוֹלִים בְּרַגְלֵיהֶן בֵּין הֵדִיחַ בֵּין לֹא הֵדִיחַ {יא}. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ח} בְּכִי יֻתַּן. הֻכְשְׁרוּ הַפֵּרוֹת, וְהַמַּיִם שֶׁעַל הַשַּׂקִּין הִנָּם תְּלוּשִׁין בְּרָצוֹן. הָרַמְבַּ"ם: {ט} רַגְלָיו כוּ'. וְלֹא חֲשִׁיב אָדָם כִּמְחֻבָּר. וְאֵין לְדַמּוֹתוֹ לִשְׁחִיטָה דְּצִפֹּרֶן מְחֻבֶּרֶת פָּסוּל כוּ'. הָרַ"שׁ: {י} וְהֵדִיחַ. אֲבָל לֹא הֵדִיחַ, אַף עַל גַּב דְּמִסְּתָמָא חָפֵץ שֶׁיּוּדְחוּ כוּ' מִכָּל מָקוֹם אֵינָן בְּכִי יֻתַּן, דְּבָעִינַן שֶׁיְּהֵא שָׂמֵחַ לְגַמְרֵי שֶׁלֹּא הָיָה רוֹצֶה שֶׁלֹּא יוּדְחוּ, וּלְהָכִי צָרִיךְ שֶׁיָּדִיחַ דְּהַשְׁתָּא חָזִינַן דְּנִיחָא לֵיהּ טוּבָא מִדְּקָטָרַח לְהָדִיחַ. מַהֲרַ"ם: {יא} וְרַבִּי יְהוּדָה הִיא. הָרַמְבַּ"ם מֵהַתּוֹסֶפְתָּא. וְטַעֲמָא כָּתַב מַהֲרַ"ם, דְּבִטְהוֹרָה הָעוֹמֶדֶת לַאֲכִילָה אֵינוֹ חוֹשֵׁשׁ כָּל כָּךְ אִם תִּהְיֶה מְלֻכְלֶכֶת. אֲבָל בְּאָדָם וּבִבְהֵמָה טְמֵאָה הָעוֹמֶדֶת לִרְכֹּב וּלְטִיּוּל אוֹ לְהַטְעִין מַשָּׂאוֹת וַדַּאי מַקְפִּיד בְּלִכְלוּכָן, וּסְתָמָא נִיחָא לֵיהּ לְגַמְרֵי שֶׁיּוּדְחוּ: (עיקר תוי"ט).

פרק ג - משנה ח

הַמּוֹרִיד אֶת הַגַּלְגַּלִּים וְאֶת כְּלֵי הַבָּקָר בִּשְׁעַת הַקָּדִים לַמַּיִם בִּשְׁבִיל שֶׁיָּחוּצוּ, הֲרֵי זֶה בְכִי יֻתַּן. הַמּוֹרִיד בְּהֵמָה לִשְׁתּוֹת, הַמַּיִם הָעוֹלִים בְּפִיהָ, בְּכִי יֻתַּן. וּבְרַגְלֶיהָ, אֵינָן בְּכִי יֻתַּן. אִם חָשַׁב שֶׁיּוּדְחוּ רַגְלֶיהָ, אַף הָעוֹלִין בְּרַגְלֶיהָ, בְּכִי יֻתַּן. בִּשְׁעַת הַיַּחַף וְהַדַּיִשׁ, לְעוֹלָם טָמֵא. הוֹרִיד חֵרֵשׁ שׁוֹטֶה וְקָטָן, אַף עַל פִּי שֶׁחוֹשֵׁב שֶׁיּוּדְחוּ רַגְלֶיהָ, אֵינָן בְּכִי יֻתַּן, שֶׁיֶּשׁ לָהֶן מַעֲשֶׂה וְאֵין לָהֶן מַחֲשָׁבָה:

 ר"ע מברטנורה  הַמּוֹרִיד אֶת הַגַּלְגַּלִּים. אֹפַן מַרְכְּבֹתָיו, מְתַרְגְּמִינַן גַּלְגְּלֵי רְתִיכֵיהוֹן: בִּשְׁעַת הַקָּדִים. בִּשְׁעַת הַחֹם רוּחַ הַקָּדִים נוֹשֵׁב. וְדֶרֶךְ כְּלֵי עֵץ שֶׁהֵם מִתְבַּקְּעִים מִפְּנֵי הַחֹם, וּמוֹרִידִין אוֹתָן לַמַּיִם וְהַבְּקָעִים מִתְחַבְּרִים. וְהַיְנוּ שֶׁיָּחוּצוּ, כְּמוֹ שֶׁיָּאוּצוּ, כְּלוֹמַר כְּדֵי שֶׁיֵּהָדְקוּ וְיִדָּחֲקוּ הַבְּקָעִים זֶה בָזֶה: הֲרֵי זֶה בְכִי יֻתַּן. הַמַּיִם הַנִּתְלִים בָּהֶם מַכְשִׁירִין הַזְּרָעִים לְקַבֵּל טֻמְאָה, מִשּׁוּם דְּנִיחָא לֵיהּ שֶׁיִּהְיוּ אוֹתָם הַמַּיִם נִתְלִים בַּכְּלִי לְהַדֵּק הַבְּקָעִים: בִּשְׁעַת הַיַּחַף. שֶׁהַבְּהֵמָה יְגֵעָה וַעֲיֵפָה מִטֹּרַח הַדֶּרֶךְ, וְכֵן בְּשָׁעָה שֶׁהִיא דָשָׁה בַתְּבוּאָה, הַמַּיִם שֶׁהִיא מַעֲלָה בְרַגְלֶיהָ לְעוֹלָם מַכְשִׁירִים הַזְּרָעִים לְקַבֵּל טֻמְאָה, לְפִי שֶׁהֵם לְרָצוֹן, שֶׁהַמַּיִם יָפִים לָהּ לְצַנְּנָהּ וּלְקָרְרָהּ. פֵּרוּשׁ אַחֵר, בִּשְׁעַת הַיַּחַף וְהַדַּיִשׁ, בִּשְׁעַת הַקָּצִיר רוֹצֶה הוּא שֶׁהַבְּהֵמָה תִרְחַץ רַגְלֶיהָ בִּשְׁבִיל שֶׁיִּהְיוּ נְקִיּוֹת לָדוּשׁ הַתְּבוּאָה {יב}: (ר"ע מברטנורה).

 עיקר תוי"ט  {יב} לֹא בָא לְפָרֵשׁ פֵּרוּשׁ אַחֵר אֶלָּא עַל הַדַּיִשׁ, וְאַגַּב נָקַט נַמִּי הַיַּחַף: (עיקר תוי"ט).

פרק ד - משנה א

הַשּׁוֹחֶה לִשְׁתּוֹת, הַמַּיִם הָעוֹלִים בְּפִיו וּבִשְׂפָמוֹ, בְּכִי יֻתַּן. בְּחָטְמוֹ וּבְרֹאשׁוֹ וּבִזְקָנוֹ, אֵינָן בְּכִי יֻתַּן. הַמְמַלֵּא בְּחָבִית, הַמַּיִם הָעוֹלִים אַחֲרֶיהָ, וּבַחֶבֶל שֶׁהוּא מְכֻנָּן עַל צַוָּארָהּ, וּבַחֶבֶל שֶׁהוּא לְצָרְכָּהּ, הֲרֵי זֶה בְכִי יֻתַּן. כַּמָּה הוּא צָרְכָּהּ. רַבִּי שִׁמְעוֹן בֶּן אֶלְעָזָר אוֹמֵר, טֶפַח. נְתָנָהּ תַּחַת הַצִּנּוֹר, אֵינָן בְּכִי יֻתַּן:

 ר"ע מברטנורה  הַשּׁוֹחֶה לִשְׁתּוֹת. בְּפִיו וּבִשְׂפָמוֹ הֲרֵי הֵן בְּכִי יֻתַּן. דְּהַיְנוּ טַעֲמָא, לְפִי שֶׁאֵין הַמַּיִם נִכְנָסִים לְפִיו אֶלָּא אִם כֵּן נוֹגְעִים בִּשְׂפָמוֹ תְחִלָּה, הִלְכָּךְ אַחְשְׁבִינְהוּ: הַמְמַלֵּא בְּחָבִית הַמַּיִם הָעוֹלִים אַחֲרֶיהָ. הֲרֵי הֵן בְּכִי יֻתַּן, דְּאִי אֶפְשָׁר שֶׁיִּכָּנֵס עַד שֶׁיִּגְּעוּ לַאֲחוֹרֶיהָ. אֲבָל הִנִּיחָהּ תַּחַת הַצִּנּוֹר, אֵין שֶׁלְּאַחֲרֶיהָ בְּכִי יֻתַּן, דְּאֶפְשָׁר לַקִּלּוּחַ שֶׁיֵּרֵד לְתוֹכָהּ וְלֹא יִגַּע לַאֲחוֹרֶיהָ: מְכֻנָּן עַל צַוָּארָהּ. מְסֻבָּב כְּמִין טַבַּעַת מַקִּיף לְצַוַּאר הֶחָבִית. מְכֻנָּן לְשׁוֹן הוּא עָשְׂךָ וַיְכֹנֲנֶךָ (דְּבָרִים לב): נְתָנָהּ. לֶחָבִית: תַּחַת הַצִּנּוֹר. לְקַבֵּל הַמַּיִם: אֵינָן בְּכִי יֻתַּן. הַמַּיִם שֶׁלְּאַחֲרֶיהָ וְשֶׁבַּחֶבֶל. דְּלֹא נִיחָא לֵיהּ אֶלָּא שֶׁיִּפְּלוּ הַמַּיִם לְתוֹכָהּ בִּלְבַד: (ר"ע מברטנורה).

פרק ד - משנה ב

מִי שֶׁיָּרְדוּ עָלָיו גְּשָׁמִים, אֲפִלּוּ אַב הַטֻּמְאָה, אֵינוֹ בְכִי יֻתַּן. וְאִם נִעֵר, בְּכִי יֻתַּן. עָמַד תַּחַת הַצִּנּוֹר לְהָקֵר אוֹ לִדּוֹחַ, בְּטָמֵא, טְמֵאִין. וּבְטָהוֹר, בְּכִי יֻתַּן:

 ר"ע מברטנורה  אֲפִלּוּ אַב הַטֻּמְאָה. אֲפִלּוּ הָיָה הָאָדָם הַזֶּה אַב הַטֻּמְאָה וְנִטְמְאוּ הַמַּשְׁקִין שֶׁיָּרְדוּ עָלָיו, וַאֲנַן אָמְרִינַן לְעֵיל דְּמַשְׁקִין טְמֵאִים מְטַמְּאִים לְרָצוֹן וְשֶׁלֹּא לְרָצוֹן, אֲפִלּוּ הָכִי אֵינָן מַכְשִׁירִין, כֵּיוָן שֶׁיְּרִידַת הַגְּשָׁמִים עָלָיו הָיְתָה שֶׁלֹּא לְרָצוֹן: וְאִם נִעֵר. כְּמוֹ הַמְנַעֵר טַלִּיתוֹ [שַׁבָּת קמז]. שֶׁנִּעֵר גּוּפוֹ לְהָסִיר הַגְּשָׁמִים מֵעָלָיו: בְּכִי יֻתַּן. הַמַּיִם הַנּוֹפְלִים מִמֶּנּוּ מַכְשִׁירִין הֵן: לְהָקֵר. לְקָרֵר עַצְמוֹ: אוֹ לִדּוֹחַ. לִרְחֹץ גּוּפוֹ, כְּגוֹן שֶׁהָיָה מְלֻכְלָךְ בְּטִיט אוֹ בְצוֹאָה: בְּטָמֵא. אִם הָיָה הָאָדָם טָמֵא: טְמֵאִים. הַמַּיִם. וְאִם נָפְלוּ עַל הַפֵּרוֹת, הִכְשִׁירוּם וְטִמְּאוּם כְּאַחַת: וּבְטָהוֹר בְּכִי יֻתַּן. וּמַכְשִׁירִין: (ר"ע מברטנורה).

פרק ד - משנה ג

הַכּוֹפֶה קְעָרָה {א} עַל הַכֹּתֶל בִּשְׁבִיל שֶׁתּוּדַח {ב}, הֲרֵי זֶה בְכִי יֻתַּן. אִם בִּשְׁבִיל שֶׁלֹּא יִלְקֶה הַכֹּתֶל, אֵינָן בְּכִי יֻתַּן:

 עיקר תוי"ט  {א} קְעָרָה. זֶה לְשׁוֹן רַשִׁ"י, אִי אַחְשִׁיב לַגְּשָׁמִים בִּירִידָתָן לְצֹרֶךְ דָּבָר תָּלוּשׁ, וְאַחַר כָּךְ נָפְלוּ פֵרוֹת לְתוֹכָהּ אֲפִלּוּ שֶׁלֹּא מִדַּעַת וְלֹא נִיחָא לֵיהּ, הָווּ מֻכְשָׁרִין, דִּתְחִלָּתָן לְרָצוֹן. אֲבָל לְצֹרֶךְ מְחֻבָּר, לֹא מַכְשְׁרֵי, אֶלָּא אִם כֵּן נִיחָא לֵיהּ בִּנְפִילַת הַפֵּרוֹת לְתוֹכָן דְּהָדַר אַחְשְׁבִינְהוּ לְתָלוּשׁ: {ב} שֶׁתּוּדַח. הָא בִּשְׁבִיל שֶׁיּוּדַח הַכֹּתֶל אֵינוֹ בְּכִי יֻתַּן. בַּמֶּה דְבָרִים אֲמוּרִים, בְּכֹתֶל מְעָרָה [פֵּרֵשׁ רַשִׁ"י, שֶׁחוֹקֵק הַקַּרְקַע וְנַעֲשֶׂה כֹתֶל מֵאֵלָיו], אֲבָל בְּכֹתֶל בִּנְיָן, בִּשְׁבִיל שֶׁלֹּא יִלְקֶה הַכֹּתֶל הוּא דְּאֵינוֹ בְכִי יֻתַּן, הָא בִּשְׁבִיל שֶׁיּוּדַח הַכֹּתֶל בְּכִי יֻתַּן. גְּמָרָא. וְנִיחָא הַדִּיּוּקִים דְּלֹא סָתְרִי אַהֲדָדֵי: (עיקר תוי"ט).

פרק ד - משנה ד

חָבִית שֶׁיָּרַד הַדֶּלֶף לְתוֹכָהּ, בֵּית שַׁמַּאי אוֹמְרִים, יְשַׁבֵּר. בֵּית הִלֵּל אוֹמְרִים, יְעָרֶה. וּמוֹדִים שֶׁהוּא מוֹשִׁיט אֶת יָדוֹ וְנוֹטֵל פֵּרוֹת מִתּוֹכָהּ, וְהֵם טְהוֹרִים:

 ר"ע מברטנורה  חָבִית שֶׁיָּרַד הַדֶּלֶף לְתוֹכָהּ. שֶׁלֹּא לְרָצוֹן, וּבְתוֹכָהּ פֵּרוֹת וְאֵינוֹ רוֹצֶה שֶׁיֻּכְשְׁרוּ לְקַבֵּל טֻמְאָה: יְשַׁבֵּר. שֶׁאִם מְעָרֶה, הֲרֵי מַעֲבִיר הַמַּיִם מִזֶּה לָזֶה לְרָצוֹן וְנִמְצְאוּ הַפֵּרוֹת מֻכְשָׁרִין: וּמוֹדִים. בֵּית שַׁמַּאי לְבֵית הִלֵּל: שֶׁהוּא מוֹשִׁיט אֶת יָדוֹ וְכוּ' וְהֵן טְהוֹרִין. כְּלוֹמַר לֹא הֻכְשְׁרוּ בְכָךְ: (ר"ע מברטנורה).

פרק ד - משנה ה

עֲרֵבָה שֶׁיָּרַד הַדֶּלֶף לְתוֹכָהּ, הַנִּתָּזִין וְהַצָּפִין אֵינָן בְּכִי יֻתַּן. נְטָלָהּ לְשָׁפְכָהּ, בֵּית שַׁמַּאי אוֹמְרִים, בְּכִי יֻתַּן. בֵּית הִלֵּל אוֹמְרִים, אֵינָן בְּכִי יֻתַּן. הִנִּיחָהּ שֶׁיֵּרֵד הַדֶּלֶף לְתוֹכָהּ, הַנִּתָּזִין וְהַצָּפִין, בֵּית שַׁמַּאי אוֹמְרִים, בְּכִי יֻתַּן. בֵּית הִלֵּל אוֹמְרִים, אֵינָן בְּכִי יֻתַּן. נְטָלָהּ לְשָׁפְכָהּ, אֵלּוּ וָאֵלּוּ מוֹדִים שֶׁהֵן בְּכִי יֻתַּן. הַמַּטְבִּיל אֶת הַכֵּלִים וְהַמְכַבֵּס אֶת כְּסוּתוֹ בִּמְעָרָה, הַמַּיִם הָעוֹלִים בְּיָדָיו, בְּכִי יֻתַּן. בְּרַגְלָיו, אֵינָן בְּכִי יֻתַּן. רַבִּי אֶלְעָזָר אוֹמֵר, אִם אִי אֶפְשָׁר לוֹ שֶׁיֵּרֵד אֶלָּא אִם כֵּן נִטַּנְּפוּ רַגְלָיו, אַף הָעוֹלִין בְּרַגְלָיו בְּכִי יֻתַּן:

 ר"ע מברטנורה  הַנִּתָּזִין. מִמֶּנָּה לַחוּץ: וְהַצָּפִין. לְמַעְלָה וְיָצְאוּ לַחוּץ. כֻּלְּהוּ אֵינָן בְּכִי יֻתַּן, לְפִי שֶׁהֵן שֶׁלֹּא לְרָצוֹן, וְכֵן הַמַּיִם שֶׁבְּתוֹכָהּ: נְטָלָהּ לְשָׁפְכָן. שֶׁלֹּא שְׁפָכָן בִּמְקוֹמָן, אֶלָּא נָטַל הָעֲרֵבָה לִשְׁפֹּךְ הַמַּיִם בְּמָקוֹם אַחֵר, הֲרֵי הֵן בְּכִי יֻתַּן, דִּלְרָצוֹן הֵן הוֹאִיל וְלֹא שְׁפָכָן מִיָּד שָׁם בִּמְקוֹמָן: וּבֵית הִלֵּל אוֹמְרִים. כֵּיוָן שֶׁלֹּא הִנִּיחָן שָׁם אֶלָּא נְטָלָהּ לְשָׁפְכָן, אֵינָן בְּכִי יֻתַּן: הִנִּיחָהּ. אִם הִנִּיחַ אֶת הָעֲרֵבָה מִתְּחִלָּה כְּדֵי שֶׁיֵּרֵד הַדֶּלֶף לְתוֹכָהּ: אֵלּוּ וָאֵלּוּ. בֵּית שַׁמַּאי וּבֵית הִלֵּל מוֹדִים שֶׁאַף עַל פִּי שֶׁנְּטָלָהּ אַחַר כָּךְ וּשְׁפָכָהּ, הֲרֵי הֵן בְּכִי יֻתַּן, הוֹאִיל וּכְבָר חִשַּׁב עֲלֵיהֶן: מַיִם הָעוֹלִים בְּיָדָיו. שֶׁמְּכַבֵּס בָּהֶן, נִיחָא לֵיהּ: בְּרַגְלָיו. לֹא נִיחָא לֵיהּ: אֶלָּא אִם כֵּן נִטַּנְּפוּ רַגְלָיו. הָווּ בְּכִי יֻתַּן. מִידִי דַהֲוָה אַמְמַלֵּא בֶחָבִית דִּלְעֵיל {ג}, דְּמַיִם הָעוֹלִים אַחֲרֶיהָ בְּכִי יֻתַּן, כִּדְפָרִישְׁנָא לְעֵיל, לְפִי שֶׁאִי אֶפְשָׁר לַמַּיִם לִכָּנֵס לְתוֹכָהּ עַד שֶׁיִּגְּעוּ לַאֲחוֹרֶיהָ: (ר"ע מברטנורה).

 עיקר תוי"ט  {ג} וְאֶפְשָׁר לוֹמַר דְּלֵיכָּא דְפָלֵיג עֲלֵיהּ אַף שֶׁנִּשְׁנָה בְלָשׁוֹן פְּלֻגְתָּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ד - משנה ו

קֻפָּה שֶׁהִיא מְלֵאָה תֻרְמוֹסִין וּנְתָנוּהָ לְתוֹךְ מִקְוֶה, מוֹשִׁיט יָדוֹ וְנוֹטֵל תֻּרְמוֹסִין מִתּוֹכָהּ, וְהֵם טְהוֹרִים. הֶעֱלָם מִן הַמַּיִם {ה}, הַנּוֹגְעִים בַּקֻּפָּה, טְמֵאִים {ו}, וּשְׁאָר כָּל הַתֻּרְמוֹסִים, טְהוֹרִים. צְנוֹן שֶׁבַּמְּעָרָה, נִדָּה מְדִיחָתוֹ וְהוּא טָהוֹר. הֶעֱלַתּוּ כָל שֶׁהוּא מִן הַמַּיִם, טָמֵא:

 ר"ע מברטנורה  וְהֵם טְהוֹרִין. אַף עַל פִּי שֶׁפָּשַׁט אָדָם טָמֵא אֶת יָדוֹ וְנָטַל {ד}, לְפִי שֶׁלֹּא הֻכְשְׁרוּ בַמַּיִם שֶׁבַּמִּקְוֶה, דִּכְתִיב (וַיִּקְרָא יא) בְּכָל כְּלִי יִטְמָא, מַה כְּלִי מְיֻחָד שֶׁהוּא תָלוּשׁ וּמַכְשִׁיר, אַף כָּל שֶׁהוּא תָלוּשׁ מַכְשִׁיר, לַאֲפוֹקֵי מַיִם שֶׁבְּבוֹרוֹת שִׁיחִין וּמְעָרוֹת שֶׁאֵינָן תְּלוּשִׁים מִן הַקַּרְקַע שֶׁאֵינָן מַכְשִׁירִים כָּל זְמַן שֶׁלֹּא נִתְלְשׁוּ: הֶעֱלָם מִן הַמַּיִם. הֻכְשְׁרוּ הַתֻּרְמוֹסִין לְקַבֵּל טֻמְאָה בַּמַּיִם הַתְּלוּשִׁים: הַנּוֹגְעִים בַּקֻּפָּה טְמֵאִים. שֶׁהַקֻּפָּה נַעֲשֵׂית רִאשׁוֹן כְּשֶׁנָּגַע בָּהּ אָדָם טָמֵא, וְהַתֻּרְמוֹסִים הַנּוֹגְעִים בַּקֻּפָּה הָווּ שֵׁנִי: וּשְׁאָר כָּל הַתֻּרְמוֹסִים. שֶׁאֵין נוֹגְעִים בַּקֻּפָּה: טְהוֹרִים. דְּאֵין שֵׁנִי עוֹשֶׂה שְׁלִישִׁי בְּחֻלִּין: נִדָּה מְדִיחָתוֹ וְהוּא טָהוֹר. דְּלֹא הֻכְשַׁר לְקַבֵּל טֻמְאָה בַּמַּיִם שֶׁמְּחֻבָּרִים בַּמְּעָרָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ד} דְּדַוְקָא בִצְנוֹן שֶׁחָשַׁב לַהֲדִיחוֹ הֻכְשַׁר כְּשֶׁהֶעֱלַתּוּ, אֲבָל הָנֵי לֹא. הָרַ"שׁ: {ה} הֶעֱלָם כוּ'. כְּלוֹמַר שֶׁלֹּא נְטָלָם מִתּוֹךְ הַקֻּפָּה שֶׁדֶּרֶךְ לִטְּלָם קְצָתָם קְצָתָם אֲבָל הֶעֱלָה כֻלָּם כְּאֶחָד דְּהַיְנוּ שֶׁהֶעֱלָם עִם הַקֻּפָּה: {ו} טְמֵאִים. וְתֵימַהּ, מַאי שְׁנָא מִשַּׂק מָלֵא פֵרוֹת דְּפֶרֶק קַמָּא מִשְׁנָה ד' דְּלֹא הֻכְשְׁרוּ בַּמַּיִם הַנּוֹטְפִים מִן הַשַּׂק. וְשֶׁמָּא הָכָא נִיחָא לֵיהּ טְפֵי בַּהֲדָחָה דְקֻפָּה. הָרַ"שׁ: (עיקר תוי"ט).

פרק ד - משנה ז

פֵּרוֹת שֶׁנָּפְלוּ לְתוֹךְ אַמַּת הַמַּיִם, פָּשַׁט מִי שֶׁהָיוּ יָדָיו טְמֵאוֹת וּנְטָלָן, יָדָיו טְהוֹרוֹת וְהַפֵּרוֹת טְהוֹרִים. וְאִם חָשַׁב שֶׁיּוּדְחוּ יָדָיו, יָדָיו טְהוֹרוֹת וְהַפֵּרוֹת בְּכִי יֻתַּן:

 ר"ע מברטנורה  יָדָיו טְהוֹרוֹת. דְּלֹא בָעִינַן כַּוָּנָה לְחֻלִּין. אֲבָל אִם הָיוּ פֵרוֹת מַעֲשֵׂר {ז} וְכָל שֶׁכֵּן שֶׁל תְּרוּמָה, יָדָיו טְמֵאוֹת כְּשֶׁהָיוּ, עַד שֶׁיְּכַוֵּן לְהַטְבִּיל יָדָיו: וְהַפֵּרוֹת טְהוֹרִים. וְלֹא הֻכְשְׁרוּ, דְּלֹא אַחְשְׁבַהּ לְהַךְ נְפִילָה: וְאִם בִּשְׁבִיל שֶׁיּוּדְחוּ יָדָיו. נִתְכַּוֵּן לִטֹּל אֶת הַפֵּרוֹת מִן הַמַּיִם, הַפֵּרוֹת הֲרֵי הֵן בְּכִי יֻתַּן וּמֻכְשָׁרִים לְקַבֵּל טֻמְאָה מִכָּאן וּלְהַבָּא {ח}. דְּאַחְשְׁבַהּ לְהַךְ נְפִילַת פֵּרוֹת לִטֹּל יָדָיו בְּהַגְבָּהָתָן, וְהָוֵי לֵיהּ מַשְׁקֶה שֶׁסּוֹפוֹ לְרָצוֹן, דְּאַף עַל פִּי שֶׁאֵין תְּחִלָּתוֹ לְרָצוֹן הֲרֵי זֶה בְּכִי יֻתַּן: (ר"ע מברטנורה).

 עיקר תוי"ט  {ז} הָרַמְבַּ"ם. וְצָרִיךְ עִיּוּן, דִּלְפֵרוֹת מַעֲשֵׂר אֵין צָרִיךְ אֲפִלּוּ נְטִילַת יָדַיִם. וְנִרְאֶה לִי לְהַגִּיהַּ פַּת בִּמְקוֹם פֵּרוֹת. מהר"ו וורמיז"א. וְהִיא קֻשְׁיָא עֲצוּמָה. וּמַה שֶּׁהִגִּיהַּ פַּת, אַכַּתִּי קָשֶׁה, דִּבְפֵרוֹת קָיְמִינַן. וְעוֹד, דְּלִנְגִיעָה אֲפִלּוּ פַת אֵין צָרִיךְ כַּוָּנָה אַף לַתְּרוּמָה: {ח} הַיְנוּ לְאַחַר שֶׁהֶעֱלָם. דְּכָל זְמַן שֶׁהֵם מְחֻבָּרִים בַּמַּיִם לֹא מִתְכַּשְּׁרֵי. תּוֹסָפוֹת: (עיקר תוי"ט).

פרק ד - משנה ח

קְדֵרָה שֶׁהִיא מְלֵאָה מַיִם וּנְתוּנָה לְתוֹךְ הַמִּקְוֶה וּפָשַׁט אַב הַטֻּמְאָה אֶת יָדוֹ לְתוֹכָהּ, טְמֵאָה. מַגַּע טֻמְאוֹת, טְהוֹרָה. וּשְׁאָר כָּל הַמַּשְׁקִין, טְמֵאִין, שֶׁאֵין הַמַּיִם מְטַהֲרִים אֶת שְׁאָר הַמַּשְׁקִין:

 ר"ע מברטנורה  וּפָשַׁט אַב הַטֻּמְאָה אֶת יָדָיו לְתוֹכָהּ טְמֵאָה. דִּכְלִי חֶרֶס מְקַבֵּל טֻמְאָה מֵאַב הַטֻּמְאָה הַנּוֹגֵעַ בְּתוֹכוֹ, וְלֹא עָלְתָה לוֹ טְבִילָה, וְאַף עַל פִּי שֶׁהוּא בְתוֹךְ הַמִּקְוֶה, לְפִי שֶׁהַמַּיִם שֶׁבְּתוֹכוֹ חוֹצְצִים בֵּינוֹ וּבֵין מֵי הַמִּקְוֶה {ט}: מַגַּע טֻמְאוֹת טְהוֹרָה. אִם לֹא הָיָה אַב הַטֻּמְאָה שֶׁנָּגַע בָּהּ, אֶלָּא רִאשׁוֹן לַטֻּמְאָה וְהוּא הַנִּקְרָא מַגַּע טֻמְאוֹת, כְּלוֹמַר הַנּוֹגֵעַ בַּאֲבוֹת הַטֻּמְאָה, אָז טְהוֹרָה הַקְּדֵרָה, לְפִי שֶׁאֵין כְּלִי חֶרֶס מְקַבֵּל טֻמְאָה אֶלָּא מֵאַב הַטֻּמְאָה, וְנִמְצָא שֶׁלֹּא נִטְמֵאת הַקְּדֵרָה אֶלָּא מֵחֲמַת מַשְׁקִים טְמֵאִים שֶׁבְּתוֹכָהּ שֶׁנִּטְמְאוּ וְחוֹזְרִים וּמְטַמְּאִים אוֹתָהּ, וּמִשּׁוּם הָכִי אֵינָהּ מְטַמְּאָה, שֶׁהֲרֵי הֻשְּׁקוּ הַמַּיִם שֶׁבְּתוֹכָהּ בְּמֵי מִקְוֶה וְנִטְהֲרוּ. אֲבָל שְׁאָר מַשְׁקִים כְּגוֹן יַיִן שֶׁמֶן וְחָלָב, אֵין הַמִּקְוֶה מְטַהֵר אוֹתָם, דְּלֹא סָלְקָא לְהוּ הַשָּׁקָה, הִלְכָּךְ חוֹזְרִים וּמְטַמְּאִים אֶת הַקְּדֵרָה, דְּרַבָּנָן גָּזְרוּ עַל הַמַּשְׁקִים שֶׁיִּהְיוּ מְטַמְּאִין כֵּלִים {י}, גְּזֵרָה מִשּׁוּם מַשְׁקֶה זָב וְזָבָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ט} כְּמוֹ שֶׁנִּתְבָּאֵר בְּפֶרֶק י' דְּמִקְוָאוֹת. הָרַמְבַּ"ם. וְתֵימַהּ, דְּאַדְּרַבָּה שָׁם נִתְבָּאֵר דִּמְלֵאָה מַיִם אֵין חוֹצְצִים. וְהַכֶּסֶף מִשְׁנֶה הִקְשָׁה עוֹד, דְּמַשְׁמַע שֶׁאִם לֹא הָיוּ חוֹצְצִים הָיְתָה טְהוֹרָה. וְזֶה אֵינוֹ. וְעִקָּר כְּמוֹ שֶׁכָּתַב בְּחִבּוּרוֹ, נִטְמֵאת הַקְּדֵרָה, שֶׁאֵין הַמִּקְוֶה מְטַהֵר כְּלֵי חֶרֶס. וְכֵן הוּא בְּנֻסְחַת אֶרֶץ יִשְׂרָאֵל: {י} וַאֲפִלּוּ נִטְמְאוּ בַשֵּׁנִי. וַאֲפִלּוּ מֵחֲמַת יָדַיִם: (עיקר תוי"ט).

פרק ד - משנה ט

הַמְמַלֵּא בְקִילוֹן, עַד שְׁלשָׁה יָמִים, טְמֵאִין. רַבִּי עֲקִיבָא אוֹמֵר, אִם נִגְּבוּ, מִיָּד טְהוֹרִים. וְאִם לֹא נִגְּבוּ, אֲפִלּוּ עַד שְׁלשִׁים יוֹם, טְמֵאִים:

 ר"ע מברטנורה  הַמְמַלֵּא בְקִילוֹן. כְּשֶׁהַמַּיִם מְכֻנָּסִים בְּמָקוֹם אֶחָד וּמְבַקְּשִׁים לַהֲבִיאָן אֶל שָׂדֶה אַחֵר, עוֹשִׂים לָהֶן דֶּרֶךְ שֶׁיָּבֹאוּ בוֹ אוֹתָן הַמַּיִם, וְשֵׁם אוֹתוֹ דֶרֶךְ שֶׁאוֹתָן הַמַּיִם נִמְשָׁכִים בּוֹ נִקְרָא קִילוֹן, וְעַד שְׁלֹשָׁה יָמִים אֵין אוֹתוֹ קִילוֹן מִתְנַגֵּב מִן הַמַּיִם שֶׁהִמְשִׁיכוּ לְתוֹכוֹ, וּמַכְשִׁירִין אֶת הַפֵּרוֹת שֶׁנָּפְלוּ לְתוֹכָן, שֶׁהֲרֵי הֵן לְרָצוֹן: רַבִּי עֲקִיבָא אוֹמֵר אִם נִגְּבוּ מִיָּד. רַבִּי עֲקִיבָא לֹא פָלֵיג אַתַּנָּא קַמָּא. דִּסְתָם קִילוֹן אֵינוֹ מִתְנַגֵּב עַד שְׁלֹשָׁה יָמִים, וְתַנָּא קַמָּא בִּסְתָם אַיְרֵי {יא}. וְהָכִי מוּכָח בְּהֶדְיָא בַּתּוֹסֶפְתָּא, דְּמוֹדֶה רַבִּי עֲקִיבָא בִּסְתָם שֶׁהוּא שְׁלֹשָׁה יָמִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {יא} כְּגוֹן שֶׁהִשְׁלִיךְ פֵּרוֹת בַּקִּילוֹן וְלֹא יָדוּעַ אִם הָיָה בוֹ לְהַטְפִּיחַ. כֶּסֶף מִשְׁנֶה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ד - משנה י

עֵצִים שֶׁנָּפְלוּ עֲלֵיהֶם מַשְׁקִין וְיָרְדוּ עֲלֵיהֶם גְּשָׁמִים, אִם רַבּוּ, טְהוֹרִים. הוֹצִיאָם שֶׁיֵּרְדוּ עֲלֵיהֶם גְּשָׁמִים, אַף עַל פִּי שֶׁרַבּוּ, טְמֵאִים. בָּלְעוּ מַשְׁקִים טְמֵאִים, אַף עַל פִּי שֶׁהוֹצִיאָם שֶׁיֵּרְדוּ עֲלֵיהֶם גְּשָׁמִים, טְהוֹרִין. וְלֹא יַסִּיקֵם אֶלָּא בְיָדַיִם טְהוֹרוֹת בִּלְבָד. רַבִּי שִׁמְעוֹן אוֹמֵר, אִם הָיוּ לַחִין וְהִסִּיקָן, וְרַבּוּ הַמַּשְׁקִין הַיּוֹצְאִין מֵהֶן עַל הַמַּשְׁקִין שֶׁבָּלְעוּ, טְהוֹרִים:

 ר"ע מברטנורה  שֶׁנָּפְלוּ עֲלֵיהֶם מַשְׁקִים. טְמֵאִים: אִם רַבּוּ. מֵי גְשָׁמִים עַל הַמַּשְׁקִים הָהֵם, בִּטְּלוּם וְטִהֲרוּם: הוֹצִיאָן כְּדֵי שֶׁיֵּרְדוּ עֲלֵיהֶם גְּשָׁמִים. דְּאַחְשְׁבִינְהוּ לַגְּשָׁמִים: אַף עַל פִּי שֶׁרַבּוּ טְמֵאִין. דְּנִטְמְאוּ הַגְּשָׁמִים עַצְמָן: בָּלְעוּ מַשְׁקִין טְמֵאִים. כְּגוֹן שֶׁהָעֵצִים נְגוּבִים מִבַּחוּץ וּבְתוֹכָן מַשְׁקִין בְּלוּעִין, אַף עַל פִּי שֶׁהוֹצִיאָן כְּדֵי שֶׁיֵּרְדוּ עֲלֵיהֶן גְּשָׁמִים דְּאַחְשְׁבִינְהוּ לַגְּשָׁמִים, אֲפִלּוּ הָכִי טְהוֹרִין, דְּלֹא נָגְעוּ בְמַשְׁקִין טְמֵאִין {יב}, שֶׁהֲרֵי בָלְעוּ: לֹא יַסִּיקֵם אֶלָּא בְיָדַיִם טְהוֹרוֹת. שֶׁלֹּא יְטַמְּאוּ הַיָּדַיִם אֶת מֵי הַגְּשָׁמִים שֶׁהֻחְשְׁבוּ וְיַחְזְרוּ וִיטַמְּאוּ אֶת הַתַּנּוּר, דְּמַשְׁקֶה מְטַמֵּא כְלִי. אֲבָל בְּיָדַיִם טְהוֹרוֹת יָכוֹל לְהַסִּיקָם בַּתַּנּוּר, וְהַתַּנּוּר טָהוֹר, וְלֹא אָמְרִינַן דְּסוֹף מַשְׁקֶה טָמֵא הַבָּלוּעַ לָצֵאת מֵחֲמַת הַהֶסֵּק, דִּבְעֵצִים שֶׁהָאֵשׁ שׁוֹלֶטֶת בָּהֶן וְנֶאֱחָז בָּהֶן הָאוּר {יג} כָּלִין הַמַּשְׁקִים לְתוֹכָן, וְאֵין יוֹצְאִים לַחוּץ כְּשֶׁמַּסִּיקָן בַּתַּנּוּר: אִם הָיוּ לַחִין. עֵצִים הַלָּלוּ שֶׁבָּלְעוּ מַשְׁקִין טְמֵאִין, וּמֵחֲמַת לַחוּתָן יָצְאוּ מַשְׁקִין עַצְמָן מֵהֶן, וְאִם אוֹתָן מַשְׁקִים פָּלְטוּ מַשְׁקֶה הַטָּמֵא הַבָּלוּעַ, וְרַבּוּ מַשְׁקִין שֶׁמֵּעַצְמָן עַל מַשְׁקֶה הַבָּלוּעַ: טְהוֹרִין. שֶׁבִּטְּלוּ אוֹתָן. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: (ר"ע מברטנורה).

 עיקר תוי"ט  {יב} וְקָשֶׁה, דְּאַכַּתִּי מַאי קָא מַשְׁמַע לָן כְּשֶׁלֹּא נָגְעוּ שֶׁלֹּא נִטְמְאוּ. וְהָרַמְבַּ"ם מְסַיֵּם, וֶהֱסִיקָן הַתַּנּוּר טָהוֹר, וְאֵין הַמַּיִם שֶׁעֲלֵיהֶם מִתְטַמְּאִין מִן הַמַּשְׁקִים הַבְּלוּעִין בָּהֶן: {יג} אֲבָל בְּלַחִים שֶׁאֵין הָאוּר נֶאֱחָז בָּהֶן כָּל כָּךְ, טְמֵאִים, וְלֵיכָּא לְמֵימַר שֶׁרִבּוּי לַחְלְחוּתָן מְבַטְּלִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ה

פרק ה - משנה א

מִי שֶׁטָּבַל בַּנָּהָר, וְהָיָה לְפָנָיו נָהָר אַחֵר וְעָבַר בּוֹ, טִהֲרוּ שְׁנִיִּים אֶת הָרִאשׁוֹנִים. דָּחָהוּ חֲבֵרוֹ לְשָׁכְרוֹ, וְכֵן לִבְהֶמְתּוֹ, טִהֲרוּ שְׁנִיִּים אֶת הָרִאשׁוֹנִים. וְאִם כִּמְשַׂחֵק עִמּוֹ, הֲרֵי זֶה בְכִי יֻתַּן:

 ר"ע מברטנורה  מִי שֶׁטָּבַל בַּנָּהָר. וּמַיִם שֶׁעָלָיו מַכְשִׁירִין, שֶׁהֲרֵי לְרָצוֹן הֵן: וְהָיָה לְפָנָיו נָהָר אַחֵר וְעָבַר בּוֹ. וּמַיִם שֶׁעָלָיו מִן הַנָּהָר הַשֵּׁנִי אֵינָן לְרָצוֹן וְאֵינָן מַכְשִׁירִין: טִהֲרוּ. מֵי הַנָּהָר הַשֵּׁנִי, מַיִם שֶׁהֶעֱלָן עִמּוֹ מִן הַטְּבִילָה, וְאֵין מַכְשִׁירִין לֹא אֵלּוּ וְלֹא אֵלּוּ: דָּחָיוֹ. דָּחָהוּ: לְשָׁכְרוֹ. שֶׁהָיָה חֲבֵרוֹ שִׁכּוֹר וּמִתּוֹךְ שִׁכְרוּתוֹ דָּחָהוּ וְהִפִּילוֹ בַנָּהָר: וְכֵן לִבְהֶמְתּוֹ. שֶׁהִפִּיל אֶת הַבְּהֵמָה בְּשִׁכְרוּתוֹ בַּנָּהָר, וְהָיוּ עָלָיו מֵי טְבִילָה, וְעַל הַבְּהֵמָה מַיִם לְרָצוֹן: טִהֲרוּ שְׁנִיִּים. שֶׁהָיוּ עַל יְדֵי שִׁכְרוּת וְלֹא לְרָצוֹן: אֶת הָרִאשׁוֹנִים. שֶׁהָיוּ לְרָצוֹן: וְאִם כִּמְשַׂחֵק. דָּחָה אֶת חֲבֵרוֹ בַּנָּהָר אוֹ אֶת הַבְּהֵמָה, גַּם הַשְּׁנִיִּים בְּכִי יֻתַּן, דְּכֻלְּהוּ לְרָצוֹן נִינְהוּ: (ר"ע מברטנורה).

פרק ה - משנה ב

הַשָּׁט עַל פְּנֵי הַמַּיִם, הַנִּתָּזִין, אֵינָן בְּכִי יֻתַּן {א}. וְאִם נִתְכַּוֵּן לְהַתִּיז עַל חֲבֵרוֹ, הֲרֵי זֶה בְכִי יֻתַּן. הָעוֹשֶׂה צִפּוֹר בַּמַּיִם, הַנִּתָּזִין וְאֶת שֶׁבָּהּ, אֵינָן בְּכִי יֻתַּן:

 ר"ע מברטנורה  הַנִּתָּזִין. שֶׁלֹּא בְכַוָּנָה: הָעוֹשֶׂה צִפּוֹר בַּמַּיִם. שֶׁמַּעֲלֶה עַל פְּנֵי הַמַּיִם רְתִיחוֹת כְּמִין אֲבַעְבּוּעוֹת: הַנִּתָּזִין. בַּעֲשִׂיַּת אֲבַעְבּוּעוֹת: וְאֶת שֶׁבָּהּ. דְּהַיְנוּ מֵי אֲבַעְבּוּעוֹת עַצְמָן. וְאִית דְּגָרְסֵי, הָעוֹשֶׂה צִנּוֹר בַּמַּיִם. כְּמוֹ שֶׁעוֹשִׂים הַנְּעָרִים כְּמִין צִנּוֹר שֶׁל קָנֶה אוֹ שֶׁל עֵץ חָלוּל, לְנַתֵּז מִמֶּנּוּ הַמַּיִם בִּצְחוֹק {ב}: וְאֶת שֶׁבָּהּ. הַמַּיִם הַנִּשְׁאָרִים בַּצִּנּוֹר: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} אֵינָן כוּ'. וְהַמַּיִם הָעוֹלִים עָלָיו, בְּכִי יֻתַּן. הָרַמְבַּ"ם: {ב} שֶׁמְּנַתְּזִין עַל לֹא דָבָר בָּאֲוִיר. לֹא כְמִתְכַּוֵּן לְנַתֵּז עַל חֲבֵרוֹ דְּרֵישָׁא, וְכִמְשַׂחֵק דְּמִשְׁנָה א': (עיקר תוי"ט).

פרק ה - משנה ג

פֵּרוֹת שֶׁיָּרַד הַדֶּלֶף לְתוֹכָן וּבְלָלָן שֶׁיְּנֻגְּבוּ, רַבִּי שִׁמְעוֹן אוֹמֵר, בְּכִי יֻתַּן {ג}. וַחֲכָמִים אוֹמְרִים, אֵינָן בְּכִי יֻתַּן:

 ר"ע מברטנורה  וּבְלָלָן שֶׁיְּנֻגְּבוּ. בָּלַל כָּל הַפֵּרוֹת בְּמַיִם, שֶׁכְּשֶׁיִּהְיוּ הַמַּיִם מְפֻזָּרִים בְּכָל הַפֵּרוֹת, יִתְנַגְּבוּ יוֹתֵר מְהֵרָה: וַחֲכָמִים אוֹמְרִים אֵינָן בְּכִי יֻתַּן. וַהֲלָכָה כַּחֲכָמִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {ג} בְּכִי יֻתַּן. הַטַּעַם, דְּאַף עַל גַּב דְּלֹא נִיחָא לֵיהּ הַדֶּלֶף, מִכָּל מָקוֹם כֵּיוָן דִּבְלָלָן כְּדֵי לְהִתְפַּשֵּׁט הַמַּיִם בְּכֻלָּן, חָשׁוּב נִיחָא לֵיהּ. מַהֲרַ"ם: (עיקר תוי"ט).

פרק ה - משנה ד

הַמּוֹדֵד אֶת הַבּוֹר בֵּין לְעָמְקוֹ בֵּין לְרָחְבּוֹ, הֲרֵי זֶה בְכִי יֻתַּן, דִּבְרֵי רַבִּי טַרְפוֹן. רַבִּי עֲקִיבָא אוֹמֵר, לְעָמְקוֹ, בְּכִי יֻתַּן. וּלְרָחְבּוֹ, אֵינוֹ בְכִי יֻתַּן:

 ר"ע מברטנורה  לְעָמְקוֹ בְּכִי יֻתַּן וּלְרָחְבּוֹ אֵינוֹ בְכִי יֻתַּן. כְּשֶׁאָדָם מוֹדֵד עָמְקוֹ שֶׁל בּוֹר בְּקָנֶה, נִכָּר רֹשֶׁם הַמַּיִם בַּקָּנֶה עַד כַּמָּה הֵן מַגִּיעִים, וּמִתּוֹךְ כָּךְ יוֹדֵעַ עָמְקוֹ, לְפִיכָךְ רוֹצֶה הוּא בַּמַּשְׁקֶה שֶׁבָּרֹשֶׁם {ד}. אֲבָל לְרָחְבּוֹ אֵינוֹ רוֹצֶה בְּמַשְׁקֶה שֶׁבָּרֹשֶׁם. וַהֲלָכָה כְּרַבִּי עֲקִיבָא {ה}, שֶׁחָזַר רַבִּי טַרְפוֹן לִהְיוֹת שׁוֹנֶה כִדְבָרָיו: (ר"ע מברטנורה).

 עיקר תוי"ט  {ד} וַאֲפִלּוּ מַה שֶּׁלְּמַטָּה מִן הָרֹשֶׁם. מִשּׁוּם דְּאִי אֶפְשָׁר לָרֹשֶׁם בִּלְעָדוֹ. תּוֹסֶפְתָּא: {ה} וַאֲפִלּוּ לְאִיכָּא דְאָמְרֵי דְּרַבִּי טַרְפוֹן לָאו חֲבֵרוֹ הֲוָה: (עיקר תוי"ט).

פרק ה - משנה ה

פָּשַׁט יָדוֹ אוֹ רַגְלוֹ אוֹ קָנֶה לְבוֹר לֵידַע אִם יֶשׁ בּוֹ מַיִם, אֵינָן בְּכִי יֻתַּן. לֵידַע כַּמָּה מַיִם יֶשׁ בּוֹ, הֲרֵי זֶה בְכִי יֻתַּן. זָרַק אֶבֶן לְבוֹר לֵידַע אִם יֶשׁ בּוֹ מַיִם, הַנִּתָּזִין, אֵינָן בְּכִי יֻתַּן. וְאֶת שֶׁבָּאֶבֶן, טְהוֹרִים:

 ר"ע מברטנורה  לֵידַע כַּמָּה מַיִם יֶשׁ בּוֹ. רוֹצֶה בַּמַּשְׁקֶה שֶׁבָּרֹשֶׁם, כְּדַאֲמַרַן, לְפִיכָךְ הָוֵי בְּכִי יֻתַּן. אֲבָל לֵידַע אִם יֶשׁ בּוֹ מַיִם, אֵין צָרִיךְ רֹשֶׁם שֶׁל מַשְׁקֶה {ו}: וְשֶׁבָּאֶבֶן. כְּלוֹמַר, שֶׁבָּאֶבֶן נַמִּי טְהוֹרִים, שֶׁאֵין מַכְשִׁירִים {ז}: (ר"ע מברטנורה).

 עיקר תוי"ט  {ו} אֶלָּא עַל יְדֵי הֶרְגֵשׁ בִּלְבַד יָדַע אִם יֶשׁ בּוֹ מַיִם. וְאַף עַל פִּי דִּכְשֶׁמַּרְגִּישׁ אִי אֶפְשָׁר שֶׁלֹּא יַעֲלוּ בוֹ הַמַּיִם, מִכָּל מָקוֹם אֵין כַּוָּנָתוֹ לַעֲלִיַּת שׁוּם מַיִם כְּלָל וְאֵין צָרִיךְ לָהֶם: {ז} וְהָרַמְבַּ"ם מְפָרֵשׁ נַמִּי לְעִנְיַן שֶׁאֵין מְקַבְּלִין טֻמְאָה, וְאַף שֶׁהָאֶבֶן נִרְאֶה עַל שֶׁטַח הַמַּיִם, אֵינָן כִּתְלוּשִׁין. וְכֵן כָּתַב מַהֲרַ"ם, אִם פָּשַׁט טָמֵא יָדָיו וּנְטָלוֹ מִן הַמַּיִם, מַיִם שֶׁעָלָיו טְהוֹרִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ה - משנה ו

הַחוֹבֵט עַל הַשֶּׁלַח, חוּץ לַמַּיִם, בְּכִי יֻתַּן {ח}. לְתוֹךְ הַמַּיִם, אֵינָן בְּכִי יֻתַּן. רַבִּי יוֹסֵי אוֹמֵר, אַף לְתוֹךְ הַמַּיִם בְּכִי יֻתַּן, מִפְּנֵי שֶׁהוּא מִתְכַּוֵּן שֶׁיֵּצְאוּ עִם הַצּוֹאָה:

 ר"ע מברטנורה  הַחוֹבֵט עַל הַשֶּׁלַח. מַכֶּה בָעוֹר אַחַר רְחִיצָתוֹ כְּדֵי לְנַעֲרוֹ מִן הַמַּיִם: הַשֶּׁלַח. הָעוֹר כְּמוֹת שֶׁהוּא כְּשֶׁמַּפְשִׁיטִים אוֹתוֹ מִן הַבְּהֵמָה [בְּרֵאשִׁית לז] נִקְרָא שֶׁלַח. תַּרְגּוּם וַיַּפְשִׁיטוּ, וְאַשְׁלָחוּ: שֶׁהוּא מִתְכַּוֵּן שֶׁיֵּצְאוּ עִם הַצּוֹאָה. מִתְכַּוֵּן שֶׁיֵּצְאוּ הַמַּיִם עִם הַלִּכְלוּךְ שֶׁנִּתְחַבֵּר בָּהֶן, לְפִיכָךְ מַכְשִׁיר מַה שֶּׁנִּתָּז מִמֶּנּוּ וַאֲפִלּוּ שֶׁהוּא בְתוֹךְ הַמַּיִם. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: (ר"ע מברטנורה).

 עיקר תוי"ט  {ח} בְּכִי יֻתַּן. שֶׁכַּוָּנָתוֹ בְּלִי סָפֵק הוֹצָאַת הַמַּיִם, וְהָוֵי כְמוֹ מַרְעִיד אֶת הָאִילָן. אֲבָל כְּשֶׁנֶּחְבָּט וְהוּא בְתוֹךְ הַמַּיִם, אֵין זֶה כַוָּנָתוֹ כֵּיוָן שֶׁהוּא שָׁקוּעַ בַּמַּיִם. הָרַמְבַּ"ם: (עיקר תוי"ט).

פרק ה - משנה ז

הַמַּיִם הָעוֹלִין בַּסְּפִינָה וּבָעֵקֶל וּבַמְּשׁוֹטוֹת, אֵינָן בְּכִי יֻתַּן. בַּמְּצוֹדוֹת וּבָרְשָׁתוֹת וּבַמִּכְמָרוֹת, אֵינָן בְּכִי יֻתַּן. וְאִם נִעֵר, בְּכִי יֻתַּן {ט}. הַמּוֹלִיךְ אֶת הַסְּפִינָה לַיָּם הַגָּדוֹל {י} לְצָרְפָהּ, הַמּוֹצִיא מַסְמֵר לַגְּשָׁמִים לְצָרְפוֹ, הַמַּנִּיחַ אֶת הָאוּד בַּגְּשָׁמִים לְכַבּוֹתוֹ, הֲרֵי זֶה בְכִי יֻתַּן:

 ר"ע מברטנורה  הֶעֵקֶל. מָקוֹם בְּשׁוּלֵי הַסְּפִינָה שֶׁמִּתְקַבְּצִים שָׁם הַמַּיִם הַנִּכְנָסִים בַּסְּדָקִים שֶׁל סְפִינָה: אִם נִעֵר. לְהָסִיר הַמַּיִם שֶׁעֲלֵיהֶן. כְּמוֹ הַמְנַעֵר טַלִּיתוֹ: לְצָרְפָהּ. לְבָדְקָהּ אִם הַמַּיִם נִכְנָסִים לְתוֹכָהּ אִם לָאו. פֵּרוּשׁ אַחֵר, לְחַזְּקָהּ וּלְחָסְמָהּ. לְפִי שֶׁכְּשֶׁהִיא בַיַּבָּשָׁה, נִבְקַעַת מֵאֵלֶיהָ, וּמַכְנִיסִים אוֹתָהּ בַּיָּם כְּדֵי שֶׁתִּשְׁרֶה בַמַּיִם וְיִסָּתְמוּ בִקּוּעֶיהָ וּסְדָקֶיהָ: מַסְמֵר לַגְּשָׁמִים לְצָרְפוֹ. לְחַזְּקוֹ וּלְחָסְמוֹ, כְּשֶׁיָּבֹאוּ עָלָיו מַיִם לְאַחַר שֶׁנִּתְלַבֵּן בָּאֵשׁ: אוּד. עֵץ שֶׁמִּקְצָתוֹ אָכְלָה הָאֵשׁ: (ר"ע מברטנורה).

 עיקר תוי"ט  {ט} נִעֵר כוּ'. וְהוּא הַדִּין וּבַסְּפִינָה וְדִכְוָתַהּ. וְקָשֶׁה, אַמַּאי לֹא תְנַנְהוּ בְחַד בָּבָא: {י} הַגָּדוֹל. תַּנָּא בְאֶרֶץ יִשְׂרָאֵל קָאֵי וּבְאוֹתוֹ יָם רְגִילִין בְּכָךְ: (עיקר תוי"ט).

פרק ה - משנה ח

קַסְיָא שֶׁל שֻׁלְחָנוֹת וְהַשִּׁיפָא שֶׁל לְבֵנִים, אֵינָן בְּכִי יֻתַּן. וְאִם נִעֵר, בְּכִי יֻתַּן:

 ר"ע מברטנורה  קַסְיָא שֶׁל שֻׁלְחָנוֹת. כִּסּוּי שֶׁמְּכַסִּין עַל הַשֻּׁלְחָן שֶׁלֹּא יִפֹּל עַל הַמָּזוֹן שֶׁעָלָיו מַיִם אוֹ דָבָר שֶׁל טִנּוּף וְלִכְלוּךְ: וְהַשִּׁיפָא שֶׁל לְבֵנִים. מַחְצֶלֶת שֶׁל שִׁיפָא וְשֶׁל גֶּמִי שֶׁמְּכַסִּים עַל הַלְּבֵנִים לְהָגֵן מִן הַגְּשָׁמִים. וַהֲרֵי אֵלּוּ כְּמוֹ בִשְׁבִיל שֶׁלֹּא יִלְקֶה הַכֹּתֶל שֶׁאֵינָן בְּכִי יֻתַּן: אֲבָל אִם נִעֵר. שֶׁמִּתְכַּוֵּן לַהֲדִיחָם, הֲרֵי זֶה בְּכִי יֻתַּן: (ר"ע מברטנורה).

פרק ה - משנה ט

כָּל הַנִּצּוֹק, טָהוֹר, חוּץ מִדְּבַשׁ הַזִּיפִין {יא}, וְהַצַּפָּחַת. בֵּית שַׁמַּאי אוֹמְרִים, אַף הַמִּקְפָּה שֶׁל גְּרִיסִין וְשֶׁל פּוֹל, מִפְּנֵי שֶׁהִיא סוֹלֶדֶת לְאַחֲרֶיהָ {יג}:

 ר"ע מברטנורה  כָּל הַנִּצּוֹק טָהוֹר. אִם מְעָרֶה מִכְּלִי טָהוֹר לִכְלִי טָמֵא, מַה שֶּׁבָּעֶלְיוֹן טָהוֹר, דְּנִצּוֹק אֵינוֹ חִבּוּר. חוּץ מֵהָנֵי דַּחֲשִׁיב, מִפְּנֵי שֶׁסּוֹלְדִים לַאֲחוֹרֵיהֶם, כְּשֶׁהַקִּלּוּחַ נִפְסָק, חוֹזֵר הַנִּשְׁאָר לְמַעְלָה: מִדְּבַשׁ הַזִּיפִים. אִית דִּמְפָרְשֵׁי דְּבַשׁ טוֹב וְחָשׁוּב, שֶׁמִּתּוֹךְ חֲשִׁיבוּתוֹ מְזַיְּפִין אוֹתוֹ {יב} וּמְעָרְבִין בּוֹ פְסֹלֶת. וְאִית דְּאָמְרֵי, עַל שֵׁם מְקוֹמוֹ שֶׁבָּא מִזִּיף. כְּמוֹ בְּבֹא הַזִּיפִים וַיֹּאמְרוּ לְשָׁאוּל (תְּהִלִּים נד): וְהַצַּפָּחַת. מַאֲכָל עָשׂוּי מִקֶּמַח וּדְבַשׁ וְהוּא רַךְ כָּל כָּךְ שֶׁיְּכוֹלִים לְהַצִּיקוֹ. וְדוֹמֶה לוֹ כְּצַפִּיחִית בִּדְבַשׁ [שְׁמוֹת טז]: הַמִּקְפָּה. תַּבְשִׁיל שֶׁאֵינוֹ לֹא רַךְ וְלֹא קָשֶׁה אֶלָּא קָפוּי, קָרוּי מִקְפָּה: שֶׁל גְּרִיסִין. פּוֹלִים טְחוּנוֹת וּגְרוּסוֹת בְּרֵחַיִם: וְשֶׁל פּוֹלִים. שְׁלֵמוֹת שֶׁאֵינָן גְּרוּסוֹת בְּרֵחַיִם: מִפְּנֵי שֶׁהִיא סוֹלֶדֶת לַאֲחוֹרֶיהָ. כְּשֶׁהַקִּלּוּחַ נִפְסָק, חוֹזֵר אוֹתוֹ עַמּוּד הַשּׁוֹתֵת שֶׁהָיָה קְצָתוֹ בַמַּשְׁקִין טְמֵאִין שֶׁבַּכְּלִי הַתַּחְתּוֹן וְנִמְשָׁךְ לְמַעְלָה לַמַּשְׁקִין שֶׁבַּכְּלִי הָעֶלְיוֹן הַטָּהוֹר, וּמְטַמֵּא כָל מַה שֶּׁבַּכְּלִי הָעֶלְיוֹן: (ר"ע מברטנורה).

 עיקר תוי"ט  {יא} חוּץ כוּ'. וַאֲפִלּוּ מְעָרֶה מִצּוֹנֵן לְצוֹנֵן. הָרַמְבַּ"ם: {יב} וְהָכִי נַמִּי מְפָרְשֵׁי [תְּהִלִּים נד] בְּבֹא הַזִּיפִים, שַׁקְרָנִים וְהוֹלְכֵי רָכִיל שֶׁמְּזַיְּפִים דִּבְרֵיהֶם. גְּמָרָא: {יג} שֶׁהִיא כוּ'. וּלְתַנָּא קַמָּא כֵּיוָן שֶׁאֵין בָּהֶם רִיר כָּל כָּךְ כְּמוֹ דְּבַשׁ הַזִּיפִין וְהַצַּפַּחַת, לָא חֲשִׁיב לְהוּ. תּוֹסָפוֹת: (עיקר תוי"ט).

פרק ה - משנה י

הַמְעָרֶה מֵחַם לְחַם, וּמִצּוֹנֵן לְצוֹנֵן, וּמֵחַם לְצוֹנֵן, טָהוֹר. מִצּוֹנֵן לְחַם, טָמֵא. רַבִּי שִׁמְעוֹן אוֹמֵר, אַף הַמְעָרֶה מֵחַם לְחַם וְכֹחוֹ שֶׁל תַּחְתּוֹן יָפֶה מִשֶּׁל עֶלְיוֹן, טָמֵא:

 ר"ע מברטנורה  הַמְעָרֶה מֵחַם לְחַם. הָכִי נַמִּי אַיְרֵי שֶׁמַּשְׁקִין שֶׁבַּכְּלִי הָעֶלְיוֹן שֶׁמְּעָרֶה מִמֶּנּוּ, טְהוֹרִים. וְשֶׁבִּכְלִי תַחְתּוֹן שֶׁמְּעָרֶה לְתוֹכוֹ, הֵן טְמֵאִין: מִצּוֹנֵן לְחַם טָמֵא. שֶׁתַּחְתּוֹן הַחַם מַעֲלֶה הֶבֶל לָעֶלְיוֹן הַצּוֹנֵן וּמְטַמְּאוֹ {יד}, שֶׁהֶעָנָן הָעוֹלֶה מִן הַחַם כְּמַשְׁקֶה הוּא חָשׁוּב: וְכֹחוֹ שֶׁל תַּחְתּוֹן יָפֶה. שֶׁחֻמּוֹ גָדוֹל מִשֶּׁל עֶלְיוֹן. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן, שֶׁכֵּיוָן שֶׁהָעֶלְיוֹן חַם, אַף עַל פִּי שֶׁהַתַּחְתּוֹן יוֹתֵר חַם מִמֶּנּוּ, אֵין פְּעֻלָּתוֹ שֶׁל תַּחְתּוֹן נִכֶּרֶת בָּעֶלְיוֹן וְאֵינוֹ מְטַמְּאוֹ: (ר"ע מברטנורה).

 עיקר תוי"ט  {יד} כְּמוֹ זֵעַת הַמֶּרְחָץ. הָרַ"שׁ. וּקְצָת קָשֶׁה, דְּאִם כֵּן אֲפִלּוּ בְלֹא נִצּוֹק נַמִּי יְהֵא טָמֵא. וְאוּלַי דְאִי אֶפְשָׁר בְּלֹא נִצּוֹק. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ה - משנה יא

הָאִשָּׁה שֶׁהָיוּ יָדֶיהָ טְהוֹרוֹת, וּמְגִיסָה בִקְדֵרָה טְמֵאָה, אִם הִזִּיעוּ יָדֶיהָ, טְמֵאוֹת. הָיוּ יָדֶיהָ טְמֵאוֹת, וּמְגִיסָה בִקְדֵרָה טְהוֹרָה, אִם הִזִּיעוּ יָדֶיהָ, הַקְּדֵרָה טְמֵאָה. רַבִּי יוֹסֵי אוֹמֵר, אִם נָטְפוּ. הַשּׁוֹקֵל עֲנָבִים בְּכַף מֹאזְנַיִם, הַיַּיִן שֶׁבַּכַּף, טָהוֹר, עַד שֶׁיְּעָרֶה לְתוֹךְ הַכְּלִי. הֲרֵי זֶה דוֹמֶה לְסַלֵּי זֵיתִים וַעֲנָבִים כְּשֶׁהֵן מְנַטְּפִין:

 ר"ע מברטנורה  וּמְגִיסָה בִקְדֵרָה. מְהַפֶּכֶת בְּכַף שֶׁבְּיָדָהּ לַתַּבְשִׁיל שֶׁבַּקְּדֵרָה: אִם הִזִּיעוּ יָדֶיהָ. מֵהֶבֶל הַקְּדֵרָה, נִטְמְאוּ יָדֶיהָ כְּאִלּוּ נָגְעָה בַמַּשְׁקֶה שֶׁבַּקְּדֵרָה: הָיוּ יָדֶיהָ טְמֵאוֹת. אִם הִזִּיעוּ יָדֶיהָ, נִטְמֵאת הַזֵּעָה מֵחֲמַת יָדֶיהָ, וְנִטְמֵאת הַקְּדֵרָה, לְפִי שֶׁהַהֶבֶל מְחַבֵּר בֵּין הַזֵּעָה שֶׁבְּיָדֶיהָ לַמַּשְׁקִין שֶׁבַּקְּדֵרָה {טו}: רַבִּי יוֹסֵי אוֹמֵר אִם נָטְפוּ. אֵין הַקְּדֵרָה טְמֵאָה אֶלָּא אִם כֵּן נָטְפוּ טִפִּין מִן הַזֵּעָה שֶׁבְּיָדֶיהָ לְתוֹךְ הַקְּדֵרָה. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: הַיַּיִן שֶׁבַּכַּף טָהוֹר. וְאֵינוֹ מַכְשִׁיר, דְּלֹא הֻחְשַׁב לִהְיוֹת מַשְׁקֶה עַד שֶׁיְּעָרֶה אוֹתוֹ לְתוֹךְ הַכְּלִי. וּכְמוֹ שֶׁאֵין לוֹ תּוֹרַת מַשְׁקֶה לְעִנְיַן הֶכְשֵׁר, כָּךְ אֵין לוֹ תּוֹרַת מַשְׁקֶה לְעִנְיַן טֻמְאָה וּלְעִנְיַן יֵין נֶסֶךְ: (ר"ע מברטנורה).

 עיקר תוי"ט  {טו} דְּלֹא חִלְּקוּ לְעֵיל בֵּין עֶלְיוֹן חַם לְתַחְתּוֹן חַם אֶלָּא כְשֶׁהָעֶלְיוֹן טָהוֹר, אֲבָל כְּשֶׁהָעֶלְיוֹן טָמֵא, לְעוֹלָם מְטַמֵּא הַתַּחְתּוֹן. נִרְאֶה לִי: (עיקר תוי"ט).

פרק ו

פרק ו - משנה א

הַמַּעֲלֶה פֵרוֹתָיו לַגַּג מִפְּנֵי הַכְּנִימָה, וְיָרַד עֲלֵיהֶם טַל, אֵינָם בְּכִי יֻתַּן. אִם נִתְכַּוֵּן לְכָךְ, הֲרֵי זֶה בְכִי יֻתַּן. הֶעֱלָן חֵרֵשׁ, שׁוֹטֶה וְקָטָן, אַף עַל פִּי שֶׁחִשַּׁב שֶׁיֵּרֵד עֲלֵיהֶן הַטַּל, אֵינָן בְּכִי יֻתַּן, שֶׁיֶּשׁ לָהֶן מַעֲשֶׂה וְאֵין לָהֶן מַחֲשָׁבָה:

 ר"ע מברטנורה  הַמַּעֲלֶה פֵרוֹתָיו לַגַּג מִפְּנֵי הַכְּנִימָה. תּוֹלָעִים הַגְּדֵלִים בַּפֵּרוֹת. לְשׁוֹן וַתְּהִי הַכִּנָּם (שְׁמוֹת ח): אִם נִתְכַּוֵּן לְכָךְ. שֶׁחָשַׁב שֶׁיֵּרֵד עֲלֵיהֶן טַל {א}: שֶׁיֶּשׁ לָהֶן מַעֲשֶׂה. כְּגוֹן אִם הָיוּ שׁוֹפְכִין עֲלֵיהֶן טַל בְּכַוָּנָה, הָיוּ מֻכְשָׁרִין: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} וְהוּא הַדִּין לֹא נִתְכַּוֵּן וְשָׂמֵחַ, כִּדְאִיתָא בַתּוֹסֶפְתָּא, עוֹדֵהוּ הַטַּל עֲלֵיהֶן הֲרֵי הוּא בְּכִי יֻתַּן. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ו - משנה ב

הַמַּעֲלֶה אֶת הָאֲגֻדּוֹת וְאֶת הַקְּצִיעוֹת וְאֶת הַשּׁוּם לַגַּג בִּשְׁבִיל שֶׁיַּמְתִּינוּ, אֵינָן בְּכִי יֻתַּן. כָּל הָאֲגֻדּוֹת שֶׁל בֵּית הַשְּׁוָקִים, טְמֵאִין. רַבִּי יְהוּדָה מְטַהֵר בַּלַּחִים {ג}. אָמַר רַבִּי מֵאִיר, וְכִי מִפְּנֵי מָה טִמְּאוּ, אֶלָּא מִפְּנֵי מַשְׁקֵה הַפֶּה. כָּל הַקְּמָחִין וְהַסְּלָתוֹת שֶׁל בֵּית הַשְּׁוָקִים, טְמֵאִים. הַחִילְקָה, הַטְּרָגִיס וְהַטִּסְגִי, טְמֵאִים בְּכָל מָקוֹם:

 ר"ע מברטנורה  הַמַּעֲלֶה אֶת הָאֲגֻדּוֹת. שֶׁל יָרָק: וְאֶת הַקְּצִיעוֹת. כּוֹתְשִׁים הַתְּאֵנִים בְּמַכְתֶּשֶׁת וְעוֹשִׂים אוֹתָן כְּעֵין כִּכָּרוֹת גְּדוֹלוֹת, וְהֵן נִקְרָאוֹת קְצִיעוֹת: יַמְתִּינוּ. לְשׁוֹן לִחְלוּחַ {ב}. וְדוֹמֶה לוֹ בַגְּמָרָא פֶּרֶק לֹא יַחְפֹּר [בָּבָא בַתְרָא דַּף יח] מִיתְנָא קָשֶׁה לַכְּתָלִים: אֵינָן בְּכִי יֻתַּן. אִם יָרַד עֲלֵיהֶן טַל: כָּל הָאֲגֻדּוֹת. שֶׁל יָרָק: שֶׁל בֵּית הַשְּׁוָקִים טְמֵאִין. לְפִי שֶׁהַמּוֹכְרִים רְגִילִים לְזַלֵּף עֲלֵיהֶן מַיִם כְּדֵי לְלַחְלְחָן, וְהֵן מֻכְשָׁרִין לְקַבֵּל טֻמְאָה, וְהַיָּדַיִם מְמַשְׁמְשִׁים בָּהֶן תָּמִיד, הִלְכָּךְ הֵן בְּחֶזְקַת טֻמְאָה: מִפְּנֵי מַשְׁקֵה הַפֶּה. אַף עַל פִּי שֶׁאֵין מְזַלְּפִין עֲלֵיהֶן מַיִם כְּשֶׁהֵן לַחִין, מִכָּל מָקוֹם רְגִילִים לָתֵת עֲלֵיהֶן מַשְׁקֶה הַיּוֹצֵא מִן הַפֶּה כְּדֵי לְהָסִיר הָאָבָק שֶׁעֲלֵיהֶן, וּבוֹ הֵן מֻכְשָׁרִין. וְאִיכָּא בֵּין תַּנָּא קַמָּא לְרַבִּי מֵאִיר מָקוֹם שֶׁאֵין רְגִילִין לְלַחְלְחָם בְּמַשְׁקֵה הַפֶּה, לְרַבִּי מֵאִיר טְהוֹרִים. וּלְתַנָּא קַמָּא בְּכָל עִנְיָן טְמֵאִים, דְּאַף בְּלַחִין נוֹתְנִין בָּהֶן מַיִם. וַהֲלָכָה כְּתַנָּא קַמָּא: הַקְּמָחִין וְהַסְּלָתוֹת שֶׁל בֵּית הַשְּׁוָקִים טְמֵאִין. לְפִי שֶׁלּוֹתְתִים הַחִטִּים שֶׁלָּהֶם קֹדֶם שֶׁיִּטְחֲנוּ אוֹתָן וְהֵן מֻכְשָׁרוֹת, וְהַיָּדַיִם מְמַשְׁמְשִׁים בָּהֶן תָּמִיד, לְפִיכָךְ הֵן בְּחֶזְקַת טֻמְאָה: הַחִילְקָה. חִטָּה כְתוּשָׁה וְנֶחְלַק כָּל גַּרְעִין לִשְׁנַיִם: טְרָגִיס. שֶׁנֶּחְלַק כָּל גַּרְעִין לִשְׁלֹשָׁה: טִסְגִי. שֶׁנֶּחֱלַק כָּל גַּרְעִין לְאַרְבָּעָה: בְּכָל מָקוֹם. אֲפִלּוּ שֶׁלֹּא בְבֵית הַשְּׁוָקִים, שֶׁהַכֹּל מְזַלְּפִין עֲלֵיהֶם מַיִם: (ר"ע מברטנורה).

 עיקר תוי"ט  {ב} שֶׁיִּתְלַחְלְחוּ מֵאֲלֵיהֶן מִן הָאֲוִיר. מַהֲרַ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} בַּלַּחִים. שֶׁאֵין דֶּרֶךְ לְהַתִּיז עֲלֵיהֶן מַיִם אֶלָּא אִם עָמְדוּ יָמִים כוּ'. הָרַמְבַּ"ם: (עיקר תוי"ט).

פרק ו - משנה ג

כָּל הַבֵּיצִים בְּחֶזְקַת טָהֳרָה, חוּץ מִשֶּׁל מוֹכְרֵי מַשְׁקֶה. וְאִם הָיוּ מוֹכְרִין עִמָּהֶן {ה} פֵּרוֹת יְבֵשִׁים, טְהוֹרוֹת. כָּל הַדָּגִים בְּחֶזְקַת טֻמְאָה. רַבִּי יְהוּדָה אוֹמֵר, חֲתִיכַת אִלְתִּית וְדָג הַמִּצְרִי הַבָּא בְקֻפָּה וְקוּלְיָס הָאִסְפָּנִין, הֲרֵי אֵלּוּ בְחֶזְקַת טָהֳרָה. כָּל הַצִּיר בְּחֶזְקַת טֻמְאָה. וְעַל כֻּלָּם, עַם הָאָרֶץ נֶאֱמָן לוֹמַר טְהוֹרִים הֵן, חוּץ מִשֶּׁל דָּגָה, מִפְּנֵי שֶׁהֵן מַפְקִידִין אוֹתָהּ אֵצֶל עַם הָאָרֶץ. רַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר, צִיר טָהוֹר שֶׁנָּפַל לְתוֹכוֹ מַיִם כָּל שֶׁהֵן, טָמֵא:

 ר"ע מברטנורה  חוּץ מִשֶּׁל מוֹכְרֵי מַשְׁקֶה. מִפְּנֵי שֶׁנּוֹגְעִים בָּהֶם בְּעוֹד שֶׁמַּשְׁקֶה טוֹפֵחַ עַל יְדֵיהֶם, וְאֵין מַקְפִּידִין עַל הַלִּכְלוּךְ, שֶׁקְּלִפָּתָם מְגִנָּה עֲלֵיהֶן {ד}: וְאִם הָיוּ מוֹכְרִים עִמָּהֶם פֵּרוֹת יְבֵשִׁים. מִתּוֹךְ שֶׁמְּנַגְּבִים יְדֵיהֶם כְּדֵי שֶׁלֹּא יְטַנְּפוּ אֶת הַפֵּרוֹת, מְנַגְּבִים יְדֵיהֶם נַמִּי קֹדֶם שֶׁיִּגְּעוּ בַבֵּיצִים: כָּל הַדָּגִים בְּחֶזְקַת טֻמְאָה. לְפִי שֶׁהֵם מֻכְשָׁרִים בַּמַּיִם, וְהַטָּמֵא וְהַטָּהוֹר מְמַשְׁמְשִׁים בָּהּ: חֲתִיכַת הָאִלְתּוֹת. מִין דָּג שֶׁרְגִילִין לְמָכְרוֹ חֲתִיכוֹת חֲתִיכוֹת. וּמָקוֹם יֵשׁ שֶׁשְּׁמוֹ אִלְתּוֹת וְשָׁם צָדִין אוֹתוֹ וּמוֹכְרִין אוֹתוֹ: וְדָג הַמִּצְרִי. דָּג קָטָן מְאֹד שֶׁמְּבִיאִין מִמֶּנּוּ קֻפּוֹת מְלֵאוֹת בְּמִצְרַיִם: וְקוּלְיָס הָאִסְפָּנִין. מִין דָּג שֶׁעוֹרוֹ רַךְ מְאֹד וְקוֹרִין לוֹ אלשבי"ט. וּשְׁלֹשָׁה מִינִים הַלָּלוּ אֵין הַמַּיִם מַכְשִׁירִין אוֹתָם כִּשְׁאָר דָּגִים, לְפִי שֶׁהֵן חַיִּין זְמַן מְרֻבֶּה אַחַר צֵידָתָן, וּלְאַחַר שֶׁהֵן נַעֲשִׂין אֹכֶל דְּהַיְנוּ לְאַחַר מִיתָתָן אֵין נוֹתְנִין עֲלֵיהֶם מַיִם, לְפִי שֶׁהַמַּיִם מַפְסִידִין אוֹתָן. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: כָּל הַצִּיר בְּחֶזְקַת טֻמְאָה. שֶׁסְּתָם צִיר נָתְנוּ בוֹ מַיִם {ו}, וְהַיָּדַיִם מְמַשְׁמְשִׁים בּוֹ: וְעַל כֻּלָּם. בֵּין בֵּיצִים בֵּין פֵּרוֹת בֵּין צִיר {ז}: נֶאֱמָן. עַם הָאָרֶץ לוֹמַר לֹא הֻכְשְׁרוּ: מִפְּנֵי שֶׁמַּפְקִידִים אוֹתָן. דְּכָל הַפֵּרוֹת מַפְקִידִים אוֹתָן אֵצֶל עַם הָאָרֶץ, וְאִי לָאו דְּנֶאֱמָן הֵיאַךְ מַפְקִידִים. אֲבָל דָּגָה אֵין מַפְקִידִים בְּיָדוֹ, לְפִיכָךְ אֵינוֹ נֶאֱמָן: צִיר טָהוֹר. שֶׁיָּדוּעַ בּוֹ שֶׁלֹּא הֻכְשַׁר עַל יְדֵי מַשְׁקֶה {ח}: שֶׁנָּפַל לְתוֹכוֹ מַיִם כָּל שֶׁהוּ טָמֵא. דְּמַיִם כָּל שֶׁהוּ מַכְשִׁירִים אוֹתוֹ, וּמִשֶּׁהֻכְשַׁר חֶזְקָתוֹ טָמֵא, שֶׁהַיָּדַיִם מְמַשְׁמְשִׁים בּוֹ. וַהֲלָכָה כְּרַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב: (ר"ע מברטנורה).

 עיקר תוי"ט  {ד} וְהוּא מְטַמֵּא אוֹתָהּ בִּקְלִפָּתָהּ. הָרַמְבַּ"ם: {ה} עִמָּהֶן. עִם הַמַּשְׁקֶה וְהַבֵּיצִים. הָרַמְבַּ"ם: {ו} דְּצִיר גּוּפוֹ, לָאו בַּר קַבּוֹלֵי טֻמְאָה הוּא, דְּזֵעָה בְעָלְמָא הוּא. גְּמָרָא: {ז} הָרַ"שׁ. וְלֹא נָקְטוּ נַמִּי דָגִים, וְנִרְאֶה שֶׁדַּעְתָּן דְּחוּץ מִשֶּׁל דָּגָה לָאו בְּצִיר הוּא, אֶלָּא אַרֵישָׁא קָאֵי חוּץ מִדָּגִים. וְכֵן הוּא בָרַמְבַּ"ם בְּנֻסְחַת אֶרֶץ יִשְׂרָאֵל וּבְחִבּוּרוֹ: {ח} הָרַמְבַּ"ם. וְקָשֶׁה, דְּלֹא הֲוָה צָרִיךְ לְמִתְנֵי אֶלָּא מַיִם כָּל שֶׁהֵן מַכְשִׁירִין אֶת הַצִּיר. אֲבָל בַּגְּמָרָא וּפֵרוּשׁ רַשִׁ"י, צִיר טָהוֹר כְּגוֹן שֶׁהִשִּׁיקוֹ בְמַיִם וְטָהוֹר מִמַּה נַּפְשָׁךְ, אִי רֻבָּא מַיָּא מִטַּהֵר עַל יְדֵי הַשָּׁקָה. אִי רֻבָּא צִיר וְהָוֵי לֵיהּ אֹכֶל וְלֹא מַהֲנֵי לֵיהּ הַשָּׁקָה, צִיר לָאו בַּר קַבּוֹלֵי טֻמְאָה הוּא. וְהָנָךְ מִעוּטָא דְמַיָּא בָּטְלֵי לְהוּ בְרֻבָּא. שֶׁנָּפַל כוּ' טָמֵא, דְּעַמֵּי הָאָרֶץ נֶחְשְׁדוּ לְעָרֵב מַיִם בְּצִירָן עַד מֶחֱצָה, וְכִי נָפַל הַאי כָּל שֶׁהוּא, מָצָא מִין אֶת מִינוֹ וְרַבּוּ מַיִם עַל הַצִּיר וּטְמֵאִים. וְכֵן הִשִּׂיגוֹ הָרַאֲבַ"ד: (עיקר תוי"ט).

פרק ו - משנה ד

שִׁבְעָה מַשְׁקִין הֵן. הַטַּל וְהַמַּיִם, הַיַּיִן וְהַשֶּׁמֶן, וְהַדָּם, וְהֶחָלָב, וּדְבַשׁ דְּבוֹרִים. דְּבַשׁ צְרָעִים, טָהוֹר, וּמֻתָּר בַּאֲכִילָה:

 ר"ע מברטנורה  שִׁבְעָה מַשְׁקִין הֵן. לְפִי שֶׁנֶּאֱמַר בְּטֻמְאַת אֳכָלִים (וַיִּקְרָא יא) מִכָּל הָאֹכֶל אֲשֶׁר יֵאָכֵל אֲשֶׁר יָבֹא עָלָיו מַיִם יִטְמָא, אֵין לִי אֶלָּא מַיִם, מִנַּיִן לְרַבּוֹת הַיַּיִן וְהַשֶּׁמֶן וְהֶחָלָב וְהַדְּבַשׁ וְהַדָּם וְהַטַּל, תַּלְמוּד לוֹמַר וְכָל מַשְׁקֶה אֲשֶׁר יִשָּׁתֶה. וְאִי וְכָל מַשְׁקֶה, יָכוֹל מֵי תוּתִים מֵי רִמּוֹנִים וּשְׁאָר כָּל מֵי פֵרוֹת, תַּלְמוּד לוֹמַר מַיִם, מַה מַּיִם מְיֻחָדִים שֶׁאֵין לָהֶם שֵׁם לְוַי, אַף הַיַּיִן וְהַשֶּׁמֶן וְהֶחָלָב וְהַדְּבַשׁ וְהַדָּם וְהַטַּל שֶׁאֵין לָהֶם שֵׁם לְוַי, יָצְאוּ מֵי תוּתִים מֵי רִמּוֹנִים וּשְׁאָר כָּל מֵי פֵרוֹת שֶׁיֵּשׁ לָהֶם שֵׁם לְוַי, וּדְבַשׁ דְּבוֹרִים אֵין לוֹ שֵׁם לְוַי, דִּדְבַשׁ סְתָם אִקְּרֵי {ט}. וְעוֹד שָׁנִינוּ בַתּוֹסֶפְתָּא דְשַׁבָּת פֶּרֶק ט', מִנַּיִן לְדָם שֶׁהוּא מַשְׁקֶה, שֶׁנֶּאֱמַר (בְּמִדְבַּר כג) וְדַם חֲלָלִים יִשְׁתֶּה. מִנַּיִן לְיַיִן שֶׁהוּא מַשְׁקֶה, שֶׁנֶּאֱמַר (דְּבָרִים לב) וְדַם עֵנָב תִּשְׁתֶּה חָמֶר. מִנַּיִן לִדְבַשׁ שֶׁהוּא מַשְׁקֶה, שֶׁנֶּאֱמַר (שָׁם) וַיֵּנִקֵהוּ דְבַשׁ מִסֶּלַע. מִנַּיִן לְשֶׁמֶן שֶׁהוּא מַשְׁקֶה, שֶׁנֶּאֱמַר (יְשַׁעְיָה כה) מִשְׁתֵּה שְׁמָנִים. מִנַּיִן לְחָלָב שֶׁהוּא מַשְׁקֶה, שֶׁנֶּאֱמַר (שׁוֹפְטִים ד) וַתִּפְתַּח אֶת נֹאוד הֶחָלָב וַתַּשְׁקֵהוּ. מִנַּיִן לְטַל שֶׁהוּא מַשְׁקֶה, שֶׁנֶּאֱמַר (שָׁם ו) וַיִּמֶץ טַל מִן הַגִּזָּה מְלוֹא הַסֵּפֶל מָיִם {י}. מִנַּיִן לְדִמְעוֹת הָעַיִן שֶׁהוּא מַשְׁקֶה, שֶׁנֶּאֱמַר (תְּהִלִּים פ) וַתַּשְׁקֵמוֹ בִּדְמָעוֹת שָׁלִישׁ. מִנַּיִן לְמֵי הָאַף שֶׁהוּא מַשְׁקֶה, שֶׁנֶּאֱמַר (יִרְמְיָה ט) וְעַפְעַפֵּינוּ יִזְּלוּ מָיִם {יא}: דְּבַשׁ צִרְעָה טָהוֹר. דְּלָא חֲשִׁיב מַשְׁקֶה, וְאֵינוֹ מְטַמֵּא וְאֵינוֹ מַכְשִׁיר: וּמֻתָּר בַּאֲכִילָה. דְּלֹא תֵימָא הוֹאִיל וְהַצִּרְעָה שֶׁרֶץ הָעוֹף הוּא וְטָמֵא, הַדְּבַשׁ הַיּוֹצֵא מִמֶּנּוּ נַמִּי יְהֵא טָמֵא, דְּסָלְקָא דַעְתָּךְ אֲמֵינָא דְּדַוְקָא דְבַשׁ דְּבוֹרִים רַבְּיֵהּ קְרָא לְהֶתֵּרָא שֶׁאֵין לוֹ שֵׁם לְוַי, אֲבָל דְּבַשׁ צְרָעִים שֶׁיֵּשׁ לוֹ שֵׁם לְוַי לֹא יְהֵא מֻתָּר, קָא מַשְׁמַע לָן דְּבַשׁ צִרְעָה טָהוֹר וּמֻתָּר. דְּמִפְּנֵי מָה אָמְרוּ דְּבַשׁ דְּבוֹרִים מֻתָּר, מִפְּנֵי שֶׁמַּכְנִיסוֹת אוֹתָן לְגוּפָן וְאֵינָן מְמַצּוֹת אוֹתוֹ מִגּוּפָן, דְּבַשׁ צְרָעִים נַמִּי מַכְנִיסוֹת אוֹתוֹ לְגוּפָן וְאֵין מְמַצּוֹת אוֹתוֹ מִגּוּפָן: (ר"ע מברטנורה).

 עיקר תוי"ט  {ט} אָמְנָם לְוַי בְּכָאן שֶׁל דְּבוֹרִים, לְחַלֵּק בֵּינוֹ וּבֵין דְּבַשׁ צְרָעִים. הָרַמְבַּ"ם. וּדְבַשׁ תְּמָרִים אַף עַל גַּב דְּבַקְּרָא נִקְרָא דְבַשׁ סְתָם, מִכָּל מָקוֹם בִּלְשׁוֹן בְּנֵי אָדָם דְּבַשׁ סְתָם הוּא שֶׁל דְּבוֹרִים. וְכָתַב הָרַ"שׁ דְּבַתּוֹלָדוֹת דִּלְקַמָּן לֹא חָיְשִׁינַן אֲפִלּוּ יֵשׁ לָהֶן לְוַי: {י} מִנַּיִן לְדַם הַנִּדָּה, שֶׁנֶּאֱמַר [וַיִּקְרָא כ] מְקוֹר דָּמֶיהָ, וְאוֹמֵר כוּ'. כֵּן הוּבָא בָּרַ"שׁ: {יא} וְאִם אֵינוֹ עִנְיָן לִדְמָעוֹת, תְּנֵהוּ עִנְיָן לְאַף. וְעוֹד, דְּאוֹתִיּוֹת אָלֶף עַיִן מִתְחַלְּפִין. הָרַ"שׁ. וְכָל זֶה אַסְמַכְתּוֹת. וְעִקַּר דַּהֲלָכָה לְמֹשֶׁה מִסִּינַי הָווּ. הָרַמְבַּ"ם: (עיקר תוי"ט).

פרק ו - משנה ה

תּוֹלָדוֹת לַמַּיִם, הַיּוֹצְאִין מִן הָעַיִן, מִן הָאֹזֶן, מִן הַחֹטֶם, מִן הַפֶּה, מֵי רַגְלַיִם, בֵּין גְּדוֹלִים בֵּין קְטַנִּים, לְדַעְתּוֹ {יג} וְשֶׁלֹּא לְדַעְתּוֹ. תּוֹלָדוֹת לַדָּם, דַּם שְׁחִיטָה {יד}. בַּבְּהֵמָה וּבַחַיָּה וּבָעוֹפוֹת הַטְּהוֹרִים, וְדַם הַקָּזָה {טו} לִשְׁתִיָּה. מֵי חָלָב, כֶּחָלָב. וְהַמֹּחַל, כַּשֶּׁמֶן, שֶׁאֵין הַמֹּחַל יוֹצֵא מִידֵי שֶׁמֶן, דִּבְרֵי רַבִּי שִׁמְעוֹן. רַבִּי מֵאִיר אוֹמֵר, אַף עַל פִּי שֶׁאֵין עִמּוֹ שָׁמֶן. דַּם הַשֶּׁרֶץ, כִּבְשָׂרוֹ, מְטַמֵּא וְאֵינוֹ מַכְשִׁיר, וְאֵין לָנוּ כַּיּוֹצֵא בוֹ:

 ר"ע מברטנורה  בֵּין גְּדוֹלִים. מַשְׁקִין הַיּוֹצְאִין מִפִּי טַבַּעַת {יב}: בֵּין קְטַנִּים. מֵי רַגְלַיִם: וְדַם הַקָּזָה לִשְׁתִיָּה. דַּם הַהַקָּזָה שֶׁנִּתְכַּוֵּן עָלָיו לְהַשְׁקוֹתוֹ לִבְהֵמָה וּלְחַיָּה אוֹ לְנָכְרִי: מֵי הֶחָלָב. הוּא הַקְּרוּם הַנִּפְרָשׁ מִן הַגְּבִינָה כְּשֶׁמַּעֲמִידִים הֶחָלָב וּמַקְפִּיאִין אוֹתוֹ בַּקֵּבָה, מַכְשִׁיר אֶת הַזְּרָעִים: וְהַמֹּחַל כַּשֶּׁמֶן. מַיִם שְׁחוֹרִים הַיּוֹצְאִים מִן הַזֵּיתִים: שֶׁאֵין מֹחַל יוֹצֵא מִידֵי שֶׁמֶן. וּמֹחַל עַצְמוֹ, לְרַבִּי שִׁמְעוֹן מֵי פֵרוֹת הוּא וְאֵינוֹ חָשׁוּב מַשְׁקֶה, אֲבָל מַכְשִׁיר מִפְּנֵי שְׁיוּרֵי שֶׁמֶן שֶׁבּוֹ: רַבִּי מֵאִיר אוֹמֵר אַף עַל פִּי שֶׁאֵין בּוֹ שָׁמֶן. מֹחַל עַצְמוֹ חָשׁוּב מַשְׁקֶה וּמַכְשִׁיר וּמְטַמֵּא. וַהֲלָכָה כְּרַבִּי מֵאִיר {טז}: דַּם הַשֶּׁרֶץ כִּבְשָׂרוֹ. וּמִצְטָרְפִין לְכָעֲדָשָׁה. דִּכְתִיב בִּשְׁמֹנָה שְׁרָצִים (וַיִּקְרָא יא) וְזֶה לָכֶם הַטָּמֵא, לְהָבִיא דָמוֹ שֶׁמְּטַמֵּא כִבְשָׂרוֹ. יָכוֹל יַכְשִׁיר אֶת הַזְּרָעִים, תַּלְמוּד לוֹמַר הַטָּמֵא, טָמֵא הוּא וְאֵינוֹ מַכְשִׁיר אֶת הַזְּרָעִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {יב} וְצָרִיךְ עִיּוּן מִמֵּי טְבֶרְיָא דְּמִשְׁנָה ז', עַיֵּן שָׁם. וְיֵשׁ לוֹמַר, דְּשָׁאנֵי הָתָם שֶׁשּׁוֹתִין אוֹתָן כְּדֵי לְשַׁלְשֵׁל אֵינוֹ חָשׁוּב מַשְׁקֶה אֶלָּא כְמוֹ מִידִי דְמֵיכַל, מַה שֶּׁאֵין כֵּן שְׁאָר מַשְׁקִים הַיּוֹצְאִים מִפִּי טַבַּעַת מֵאֲלֵיהֶן. וּכְהַאי גַּוְנָא חֲלֵב אִשָּׁה דְּחָשׁוּב אֹכֶל בְּפִי הַתִּינוֹק: {יג} לְדַעְתּוֹ. פֵּרוּשׁ, בֵּין שֶׁיָּצְאוּ לְדַעְתּוֹ וְכוּ'. הָרַמְבַּ"ם: {יד} שְׁחִיטָה. אַף עַל גַּב דְּדַם מַגֵּפָתָהּ לֹא, דְּדַם חֲלָלִים גַּבֵּי אָדָם כְּתִיב, מִכָּל מָקוֹם דַּם שְׁחִיטָה מַכְשִׁיר, דְּאִתַּקַּשׁ לְמַיִם, דִּכְתִיב [דְּבָרִים יב] עַל הָאָרֶץ תִּשְׁפְּכֶנּוּ כַמַּיִם. רַשִׁ"י. וּמֵהַאי טַעֲמָא נִיחָא דְּדַוְקָא בִטְהוֹרִים, דְּבֵיהּ אַיְרֵי קְרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טו} הַקָּזָה. שֶׁל אָדָם. הָרַמְבַּ"ם. דְּבַבְּהֵמָה לֹא עֲדִיף מִדַּם מַגֵּפָתָהּ. וְהָרַ"שׁ כָּתַב, דִּבְמַקִּיז לִשְׁתִיָּה אֲפִלּוּ דִבְהֵמָה נַמִּי מַכְשִׁיר, דְּאַחְשְׁבֵיהּ: {טז} וְהוּא דַעַת חֲכָמִים [דְּטָהֳרוֹת פ"ט מ"ג]. הָרַמְבַּ"ם: (עיקר תוי"ט).

פרק ו - משנה ו

אֵלּוּ מְטַמְּאִין וּמַכְשִׁירִין. זוֹבוֹ שֶׁל זָב, וְרֻקּוֹ, וְשִׁכְבַת זַרְעוֹ, וּמֵימֵי רַגְלָיו, וּרְבִיעִית מִן הַמֵּת, וְדַם הַנִּדָּה. רַבִּי אֱלִיעֶזֶר אוֹמֵר, שִׁכְבַת זֶרַע אֵינָהּ מַכְשֶׁרֶת. רַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה אוֹמֵר, דַּם הַנִּדָּה אֵינוֹ מַכְשִׁיר. רַבִּי שִׁמְעוֹן אוֹמֵר, דַּם הַמֵּת אֵינוֹ מַכְשִׁיר. וְאִם נָפַל עַל הַדְּלַעַת {יח}, גּוֹרְדָהּ {יט} וְהִיא טְהוֹרָה {כ}:

 ר"ע מברטנורה  וְאֵלּוּ מְטַמְּאִים וּמַכְשִׁירִים. שֶׁהֵן עַצְמָן מַשְׁקִין טְמֵאִים, וּמְטַמְּאִים וּמַכְשִׁירִים כְּאֶחָד {יז}: זוֹבוֹ שֶׁל זָב. דִּכְתִיב (שָׁם טו) זוֹבוֹ טָמֵא הוּא: וְרֻקּוֹ. וְכִי יָרֹק וְגוֹ': וּמֵימֵי רַגְלָיו. וְזֹאת תִּהְיֶה טֻמְאָתוֹ, וְזֹאת, לְרַבּוֹת מֵימֵי רַגְלָיו: וּרְבִיעִית דָּם מִן הַמֵּת. דִּכְתִיב (בְּמִדְבַּר יט) כָּל הַנּוֹגֵעַ בְּמֵת בְּנֶפֶשׁ הָאָדָם, וְנֶפֶשׁ הָאָדָם הַיְנוּ רְבִיעִית דָּם, דִּכְתִיב (וַיִּקְרָא יז) כִּי הַדָּם הוּא הַנֶּפֶשׁ: וְדַם הַנִּדָּה. דִּכְתִיב (שָׁם טו) וְהַדָּוָה בְּנִדָּתָהּ, מַה שֶּׁזָּב מִמֶּנָּה טָמֵא כְמוֹתָהּ: שִׁכְבַת זֶרַע אֵינָהּ מַכְשֶׁרֶת. שֶׁאֵינָהּ חֲשׁוּבָה מַשְׁקֶה, דְּיֵשׁ לָהּ שֵׁם לְוַי: דַּם הַנִּדָּה אֵינוֹ מַכְשִׁיר. מִידִי דַּהֲוֵי אַדָּם שֶׁל הַתַּחְתּוֹנִיּוֹת שֶׁגַּם הוּא אֵינוֹ מַכְשִׁיר: דַּם הַמֵּת אֵינוֹ מַכְשִׁיר. דְּכֵיוָן שֶׁמֵּת יָצָא דָמוֹ מִתּוֹרַת מַשְׁקֶה. וְלֵית הֲלָכָה כְּחַד מֵהָנָךְ תַּנָּאֵי, אֶלָּא כֻלְּהוּ מְטַמְּאִים וּמַכְשִׁירִים כְּתַנָּא קַמָּא: (ר"ע מברטנורה).

 עיקר תוי"ט  {יז} וְלָאו דַּוְקָא אֵלּוּ שֶׁהֵן אַב הַטֻּמְאָה, אֶלָּא אֲפִלּוּ דַם מַגֵּפָתָהּ וַחֲלֵב אִשָּׁה וְדִמְעַת עַיִן שֶׁאֵינָן אֶלָּא וְלַד הַטֻּמְאָה. וְכֵן כָּתַב הָרַמְבַּ"ם בְּהֶדְיָא. וְצָרִיךְ עִיּוּן דְּלֹא תָנֵי לְהוּ דַּהֲווּ רְבוּתָא טְפֵי: {יח} הַדְּלַעַת. לֹא יָדַעְתִּי לָמָּה נָקַט דְּלַעַת: {יט} גּוֹרְדָהּ. דְּדַם הָאָדָם אָסוּר מִדִּבְרֵי סוֹפְרִים. וְעוֹד, דְּמֵת אָסוּר בַּהֲנָאָה: {כ} טְהוֹרָה. לְפִי שֶׁלֹּא הֻכְשְׁרָה. הָרַ"שׁ: (עיקר תוי"ט).

פרק ו - משנה ז

אֵלּוּ לֹא מְטַמְּאִין וְלֹא מַכְשִׁירִין. הַזֵּעָה, וְהַלֵּחָה סְרוּחָה, וְהָרְאִי, וְהַדָּם הַיּוֹצֵא עִמָּהֶם, וּמַשְׁקֵה בֶן שְׁמֹנָה. רַבִּי יוֹסֵי אוֹמֵר, חוּץ מִדָּמוֹ. וְהַשּׁוֹתֶה מֵי טְבֶרְיָה {כא}, אַף עַל פִּי שֶׁיּוֹצְאִין נְקִיִּים. דַּם שְׁחִיטָה בַּבְּהֵמָה, בַּחַיָּה וּבָעוֹפוֹת הַטְּמֵאִים, וְדַם הַקָּזָה לִרְפוּאָה. רַבִּי אֱלִיעֶזֶר מְטַמֵּא בָאֵלּוּ. רַבִּי שִׁמְעוֹן בֶּן אֶלְעָזָר אוֹמֵר, חֲלֵב הַזָּכָר, טָהוֹר:

 ר"ע מברטנורה  וְאֵלּוּ לֹא מְטַמְּאִים. אֲפִלּוּ יָצְאוּ מִן הַטָּמֵא, כְּגוֹן מִן הַזָּב וְכַיּוֹצֵא בוֹ. דִּכְתִיב בְּזוֹבוֹ טָמֵא הוּא, הוּא טָמֵא וְאֵין זֵעָה וְלֵחָה סְרוּחָה וּרְעִי שֶׁלּוֹ טְמֵאִים: וְהַלֵּחָה סְרוּחָה. הַיּוֹצֵאת מִן הַמֻּרְסָה. וְהוּא הַדִּין בְּרוֹקֵק לֵחָה סְרוּחָה: וְהָרְאִי. צוֹאָה. כְּמוֹ וְהָרְעִי: וְדָם הַיּוֹצֵא עִמָּהֶם. עִם הַלֵּחָה וְהָרְעִי: וּמַשְׁקֶה בֶן שְׁמֹנָה. כָּל מַשְׁקֶה שֶׁיָּצָא מִנֵּפֶל שֶׁנּוֹלַד בֶּן שְׁמֹנָה חֳדָשִׁים, כְּגוֹן דָּמוֹ וְרֻקּוֹ וּמֵימֵי רַגְלָיו. לְפִי שֶׁבֶּן שְׁמֹנָה חַי הֲרֵי הוּא כְאֶבֶן: חוּץ מִדָּמוֹ. שֶׁמְּטַמֵּא כְדָם שֶׁל בֶּן קְיָמָא. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: אַף עַל פִּי שֶׁיּוֹצְאִין נְקִיִּים. שֶׁמּוֹצִיא אוֹתָן דֶּרֶךְ בֵּית הָרְעִי צְלוּלִים כְּמוֹ שֶׁהָיוּ כְּשֶׁשְּׁתָאָן, אֲפִלּוּ הָכִי אֵין מַכְשִׁירִין, הוֹאִיל וְיָצְאוּ דֶרֶךְ בֵּית הָרְעִי: רַבִּי אֱלִיעֶזֶר מְטַמֵּא בָאֵלּוּ. דְּמַשְׁקִין הֵן חֲשׁוּבִין, וּמְטַמְּאִין וּמַכְשִׁירִין. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: חֲלֵב הַזָּכָר טָהוֹר. דְּלָא חֲשִׁיב מַשְׁקֶה {כב}. וְכֵן הֲלָכָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {כא} מֵי טְבֶרְיָה. יֵשׁ בָּהֶן אֵיכוּת שֶׁמְּשַׁלְשְׁלִים. הָרַמְבַּ"ם: {כב} אֶלָּא כְזֵעָה בְעָלְמָא. הָרַ"שׁ: (עיקר תוי"ט).

פרק ו - משנה ח

חֲלֵב הָאִשָּׁה מְטַמֵּא לְרָצוֹן וְשֶׁלֹּא לְרָצוֹן, וַחֲלֵב הַבְּהֵמָה אֵינוֹ מְטַמֵּא אֶלָּא לְרָצוֹן. אָמַר רַבִּי עֲקִיבָא, קַל וָחֹמֶר הַדְּבָרִים. מָה אִם חֲלֵב הָאִשָּׁה שֶׁאֵינוֹ מְיֻחָד אֶלָּא לִקְטַנִּים, מְטַמֵּא לְרָצוֹן וְשֶׁלֹּא לְרָצוֹן, חֲלֵב הַבְּהֵמָה שֶׁהוּא מְיֻחָד לִקְטַנִּים וְלִגְדוֹלִים, אֵינוֹ דִין שֶׁיְּטַמֵּא לְרָצוֹן וְשֶׁלֹּא לְרָצוֹן. אָמְרוּ לוֹ, לֹא, אִם טִמֵּא חֲלֵב הָאִשָּׁה שֶׁלֹּא לְרָצוֹן, שֶׁדַּם מַגֵּפָתָהּ טָמֵא, יְטַמֵּא חֲלֵב הַבְּהֵמָה שֶׁלֹּא לְרָצוֹן, שֶׁדַּם מַגֵּפָתָהּ טָהוֹר. אָמַר לָהֶם, מַחְמִיר אֲנִי בְּחָלָב מִבְּדָם, שֶׁהַחוֹלֵב לִרְפוּאָה {כד}, טָמֵא, וְהַמַּקִּיז לִרְפוּאָה, טָהוֹר. אָמְרוּ לוֹ, סַלֵּי זֵיתִים וַעֲנָבִים יוֹכִיחוּ, שֶׁהַמַּשְׁקִים הַיּוֹצְאִין מֵהֶן לְרָצוֹן, טְמֵאִים, וְשֶׁלֹּא לְרָצוֹן, טְהוֹרִים. אָמַר לָהֶן, לֹא, אִם אֲמַרְתֶּם בְּסַלֵּי זֵיתִים וַעֲנָבִים, שֶׁתְּחִלָּתָן אֹכֶל וְסוֹפָן מַשְׁקֶה, תֹּאמְרוּ בְחָלָב שֶׁתְּחִלָּתוֹ וְסוֹפוֹ מַשְׁקֶה. עַד כָּאן הָיְתָה תְשׁוּבָה. אָמַר רַבִּי שִׁמְעוֹן מִכָּאן וָאֵילָךְ הָיִינוּ מְשִׁיבִין לְפָנָיו, מֵי גְשָׁמִים יוֹכִיחוּ, שֶׁתְּחִלָּתָן וְסוֹפָן מַשְׁקֶה וְאֵינָן מְטַמְּאִין אֶלָּא לְרָצוֹן. אָמַר לָנוּ, לֹא, אִם אֲמַרְתֶּם בְּמֵי גְשָׁמִים, שֶׁאֵין רֻבָּן לָאָדָם, אֶלָּא לָאֲרָצוֹת וְלָאִילָנוֹת, וְרֹב הֶחָלָב, לָאָדָם:

 ר"ע מברטנורה  חֲלֵב הָאִשָּׁה מְטַמֵּא. מַכְשִׁיר לְקַבֵּל טֻמְאָה. וְאִם נִטְמָא אוֹ שֶׁהָאִשָּׁה טְמֵאָה, מְטַמֵּא כְדִין מַשְׁקִין טְמֵאִים, דְּתוֹרַת מַשְׁקֶה יֶשׁ בּוֹ: לְרָצוֹן. הַנֶּחְלָב בְּכַוָּנָה {כג}: וְשֶׁלֹּא לְרָצוֹן. כְּגוֹן חָלָב הַנּוֹטֵף מִדַּדֶּיהָ: שֶׁאֵינוֹ מְיֻחָד אֶלָּא לִקְטַנִּים. וּסְתָמֵיהּ לָא חֲשִׁיב כֻּלֵּי הַאי: שֶׁדַּם מַגֵּפָתָהּ טָמֵא. דְּהַיּוֹצֵא מִמַּכָּתָהּ טָמֵא, דְּדָמֵי לְדַם חֲלָלִים שֶׁקָּרוּי מַשְׁקֶה, דְּמַה לִּי קַטְלַהּ כֻּלַּהּ, מַה לִּי קַטְלַהּ פַּלְגָּא: שֶׁדַּם מַגֵּפָתָהּ טָהוֹר. דְּדַם בְּהֵמָה לֹא אִקְּרֵי דַם חֲלָלִים, דְּגַבֵּי אָדָם כְּתִיב: סַלֵּי זֵיתִים וַעֲנָבִים יוֹכִיחוּ. שֶׁהַסּוֹחֲטָן לִרְפוּאָה טָמֵא, וְאַף עַל פִּי כֵן לְרָצוֹן טְמֵאִים, שֶׁלֹּא לְרָצוֹן טְהוֹרִים. אַף הֶחָלָב, אַף עַל פִּי שֶׁהַחוֹלֵב לִרְפוּאָה טָמֵא, כְּשֶׁנֶּחְלַב שֶׁלֹּא לְרָצוֹן דִּין הוּא שֶׁיִּהְיֶה טָהוֹר: וְשֶׁלֹּא לְרָצוֹן טְהוֹרִים. שֶׁאֵין הַיּוֹצֵא מִן הַזֵּיתִים וַעֲנָבִים חָשׁוּב מַשְׁקֶה עַד שֶׁיֵּצֵא לְרָצוֹן: (ר"ע מברטנורה).

 עיקר תוי"ט  {כג} כְּלוֹמַר, וְלֹא הָוֵי כְמוֹ לְרָצוֹן דְּרֵישׁ פֶּרֶק קַמָּא, דְּהָתָם מַיְרֵי בִשְׁעַת נְפִילָתָן עַל הַזְּרָעִים וּמִשּׁוּם הָכִי אֲפִלּוּ סוֹפוֹ לְרָצוֹן. אֲבָל הָכָא מַיְרֵי בִשְׁעַת פְּרִישָׁה מִן הָאִשָּׁה: {כד} לִרְפוּאָה. לִבְהֶמְתּוֹ שֶׁלֹּא יַזִּיקֶנָּהּ חֲלָבָהּ, טָמֵא, דְּהָא לְרָצוֹן הוּא. וְהַמַּקִּיז לִבְהֶמְתּוֹ לִרְפוּאָה, טָהוֹר, דְּדַם מַגֵּפָתָהּ הִיא, וְהוֹאִיל וְכֵן אַף הַנּוֹטֵף שֶׁלֹּא לְרָצוֹן טָמֵא. רַשִׁ"י: (עיקר תוי"ט).
