{א} וּלְעֵיל בְּפֶרֶק קַמָּא דְפֵאָה מִשְׁנָה ו' דָּרֵישׁ מִדִּכְתִיב (דְּבָרִים יד) וּבָא הַלֵּוִי כִּי אֵין לוֹ חֵלֶק וְנַחֲלָה עִמָּךְ:

{ב} חַיָּבוֹת. פֵּרוּשׁ כְּשֶׁאוֹכֵל קֶבַע. אֲבָל עֲרַאי רַשַּׁאי עֲדַיִן לֶאֱכֹל עַד שֶׁיַּגִּיעוּ זְמַנִּים אֲחֵרִים:

{ג} לְעִנְיַן זֵיתִים כָּתַב כֵּן, וּלְעִנְיַן תְּבוּאָה כְּשֶׁבָּא לְטָחֲנָן. תּוֹסְפוֹת יוֹם טוֹב:

{ד} שֶׁיְּחַפֶּה. פֵּרוּשׁ פְּעָמִים שֶׁאֵין רְצוֹנוֹ לְמַלְּאוֹתָהּ וּמְכַסֶּה הַפֵּרוֹת שֶׁבַּכַּלְכָּלָה בְּעָלִין אוֹ בְהוּצִין וּבְכַיּוֹצֵא בָהֶם, וְהַיְנוּ דְקָאָמַר מִשֶּׁיְּחַפֶּה. תּוֹסְפוֹת יוֹם טוֹב בְּשֵׁם הַפֵּרוּשׁ:

{ח} הַטַּעַם שֶׁאֵין הַמִּנְהָג שֶׁיְּבַשֵּׁל אָדָם וְיֹאכַל וְתִקָּרֵא זֹאת הָאֲכִילָה עֲרַאי, לְפִי שֶׁכֵּיוָן שֶׁבִּשֵּׁל אֵינוֹ עֲרַאי. הָרַמְבַּ"ם:

פרק ב - משנה א

{א} דְּאִלּוּ רָאוּ פְּנֵי הַבַּיִת עַל יְדֵי בַעַל הַבַּיִת וְהֻקְבְּעוּ לְמַעֲשֵׂר טָבֵיל לְהוּ וַאֲסוּרִים לְכָל אָדָם: {ב} לְבָתֵּיכֶם. דַּוְקָא לְבָתֵּיכֶם, דְּאִלּוּ לְבֵיתִי אֵין בַּיִת אַחֵר טוֹבֵל כְּדִלְקַמָּן. תּוֹסְפוֹת יוֹם טוֹב:

פרק ב - משנה ב

{ג} וְצָרִיךְ לוֹמַר שֶׁהַפֵּרוֹת מֻנָּחִים בַּשּׁוּק, דְּאִלּוּ הַפֵּרוֹת הֵמָּה גַם כֵּן בַּשַּׁעַר אוֹ בַחֲנוּת, אִם כֵּן טְבוּלִים וְהֻקְבְּעוּ לַמַּעֲשֵׂר וַאֲסוּרִין גַּם לַאֲחֵרִים. וְהוּא הַדִּין דַּהֲוָה מָצֵי לְמִתְנֵי הָיוּ יוֹשְׁבִים בַּבַּיִת, אֶלָּא שֶׁדֶּרֶךְ הַמּוֹכֵר פֵּרוֹת לֵישֵׁב בַּשַּׁעַר אוֹ בַחֲנוּת: {ד} דִּבְרֵי רַבִּי נְחֶמְיָה בְּמִשְׁנָה ה' פֶּרֶק ג'. וּבַיְרוּשַׁלְמִי, רַבִּי יְהוּדָה וְרַבִּי נְחֶמְיָה אָמְרוּ דָּבָר אֶחָד:

פרק ב - משנה ג

{ה} דְּאִלּוּ הִגִּיעַ הַשַּׁבָּת טָבְלוּ אַף עַל פִּי שֶׁלֹּא יִחֲדָם לְצֹרֶךְ הַשַּׁבָּת. יְרוּשַׁלְמִי: {ו} הָרִאשׁוֹן. כְּלוֹמַר כֵּיוָן שֶׁהִגִּיעַ לַבַּיִת הָרִאשׁוֹן שֶׁבָּעִיר נִטְבְּלוּ הַפֵּרוֹת אַף עַל פִּי שֶׁלֹּא לָן שָׁם. הָרַ"שׁ:

פרק ב - משנה ד

{ז} וַחֲכָמִים אוֹסְרִים. טַעֲמַיְהוּ, דְּסָבְרֵי דִּכְשֶׁכּוֹנְסִין תְּאֵנִים בַּכַּלְכָּלָה יֵשׁ שֶׁמַּנִּיחִין אוֹתָן כָּךְ וַהֲוָה גְמַר מְלָאכָה, אַף עַל גַּב דְּלָאו כֻּלֵּי עַלְמָא עָבְדֵי הָכִי אֶלָּא מַחֲלִיקִים אוֹתָם כִּדְאִיתָא סוֹף פֶּרֶק דִּלְעֵיל. תּוֹסְפוֹת יוֹם טוֹב:

פרק ב - משנה ה

{ח} בַּגְּמָרָא פֶּרֶק ז' דְּבָבָא מְצִיעָא אַסְמְכִינַן אַקְּרָא כִּי קִבְּצָם כֶּעָמִיר גֹּרְנָה (מִיכָה ד), וּפֵרֵשׁ רַשִׁ"י אֵין גֹּרֶן בְּלֹא קִבּוּץ, וְאֵין קִבּוּץ פָּחוֹת מִשְּׁתַּיִם:

פרק ב - משנה ו

{ט} וְאוֹכֵל. פֵּרוּשׁ בְּעוֹדוֹ בִמְחֻבָּר כְּמוֹ שֶׁכָּתַב הָרַמְבַּ"ם, שֶׁלֹּא יִכְרֹת הָאֶשְׁכּוֹל כֻּלּוֹ כְּדֵי שֶׁלֹּא יִתְחַיֵּב בַּמַּעַשְׂרוֹת, שֶׁהֲרֵי לֹא קָנָה אֶלָּא הַנִּתְלַשׁ. מַה שֶּׁאֵין כֵּן בְּסֵיפָא דְּאָמְרִינַן אֵלּוּ בִמְחֻבָּר קָנָה, וּתְנַן בְּרֵישׁ פֶּרֶק ה' דְּהַקּוֹנֶה בִמְחֻבָּר לֹא נִקְבַּע לַמַּעֲשֵׂר: {י} בִּמְחֻבָּר. וְנַעֲשָׂה כְּבַעַל הַגִּנָּה דִּתְנַן לְעֵיל דְּאוֹכֵל מֵהֶן עֲרַאי, וְעוֹד תְּנַן בִּמְחֻבָּר לַקַּרְקַע פָּטוּר. הָרַ"שׁ:

פרק ב - משנה ז

{יא} בְּנִי בִּשְׂכָרִי. פֵּרֵשׁ רַשִׁ"י בְּמַסֶּכֶת בָּבָא מְצִיעָא דַּף צ"ב בַּשָּׂכָר שֶׁקָּצַצְתָּ לִי. וְהַתּוֹסָפוֹת מְפָרְשִׁים בִּשְׂכָרִי הַיְנוּ תַּחְתַּי. וְחַיָּב אֲפִלּוּ עַל מַה שֶּׁהָיָה אוֹכֵל הָאָב, לֹא בִלְבָד מַה שֶּׁיֹּאכַל הוּא יוֹתֵר מִן הָאָב, מִשּׁוּם דְּמֵחֲזֵי כְּמִקָּח: {יב} אוֹכֵל וְחַיָּב. סְתָמָא דְּלֹא כְרַבִּי יְהוּדָה דְּסוֹף פִּרְקִין שֶׁסּוֹבֵר דְּכָל מִקָּח דְּלִקְצוֹת פָּטוּר (וּבְסֵפֶר תּוֹסָפוֹת חֲדָשִׁים מוֹכִיחַ מִלְּשׁוֹן הָרַמְבַּ"ם דְּלִקְצוֹת דְּרַבִּי יְהוּדָה אֵינוֹ דָמִי לִדְהָכָא דְּרוֹצֶה לֶאֱכֹל מִיָּד וַהֲוָה אֶצְלוֹ דָּבָר הַגָּמוּר) וְעַיֵּן עוֹד אַחֲרֵי זֶה:

פרק ב - משנה ח

{יג} בִּלְבָסִים. מַתְנִיתִין כְּשֶׁלְּבָסִים וּתְאֵנִים מֻרְכָּבִים בְּיִחוּר אֶחָד, וַאֲפִלּוּ הָכִי לֹא יֹאכַל, דְּאִי לֹא תֵימָא הָכִי הָא תָּנֵינָא בָּבָא מְצִיעָא (דַּף צא) הָיָה עוֹשֶׂה בְּיִחוּר זֶה לֹא יֹאכַל בְּיִחוּר אַחֵר. הָרַ"שׁ בְּשֵׁם הַיְרוּשַׁלְמִי: {יד} וְטַעֲמָא דְמַתְנִיתִין דִּלְעֵיל סְתָמָא וְאַחַר כָּךְ מַחֲלוֹקֶת דְּהָכָא, וּבְכִי הָא לֵית הִלְכְתָא כִּסְתָם כוּ'. וּמִיהוּ נִרְאֶה לִי דְּאַף לְרַבִּי יְהוּדָה עַד כָּאן לֹא פָטַר בְּלִקְצוֹת אֶלָּא כְּשֶׁאוֹכֵל אַחַת אַחַת וְלֹא צֵרַף שְׁתַּיִם, דְּצֵרוּף קוֹבֵעַ לְמִקָּח כְּדִלְעֵיל מִשְׁנָה ה'. תּוֹסְפוֹת יוֹם טוֹב:

פרק ג

פרק ג - משנה א

{א} בָּנָיו וּבְנֵי בֵיתוֹ. בַּיְרוּשַׁלְמִי מְבֹאָר דְּאַף דִּמְזוֹנוֹת אִשְׁתּוֹ עָלָיו, הוֹאִיל וַאֲכִילָתָהּ אֵינָהּ אֶלָּא מִתְּנַאי בֵית דִּין, לֹא דָמִי לִתְנַאי דְפוֹעֲלִים. לְפִי שֶׁבֵּית דִּין שֶׁהִתְנוּ וְתִקְּנוּ מְזוֹנוֹת לָאִשָּׁה, לֹא תִקְּנוּ אֶלָּא שֶׁלְּעִנְיַן מַעַשְׂרוֹת לֹא תְהֵא אֶלָּא כְמַתָּנָה: {ב} אוֹכְלִין. אִם נָתַן לָהֶם: {ג} דְּהָא דְּפֵרֵשׁ סוֹף פִּרְקִין דִּלְעֵיל דְּפוֹעֵל אוֹכֵל מִן הַתּוֹרָה בְּתָלוּשׁ שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ, הַיְנוּ שֶׁלֹּא נִגְמְרָה לַמַּעֲשֵׂר. וּמִיהוּ בִּשְׁעַת גְּמַר מְלָאכָה בָּעִינַן וְלֹא לְהַעֲבִירָן כוּ'. וְעַיֵּן מִשְׁנָה ב' פֶּרֶק ז' דְּבָבָא מְצִיעָא: {ד} כְּלוֹמַר וְלֹא תוֹלֶה עֵינָיו בְּדָבָר אַחֵר לִגְמָרוֹ שֶׁל הַמְּלָאכָה, אֶלָּא אוֹכְלוֹ כָּךְ כְּשֶׁהוּא לוֹקְחוֹ:

פרק ג - משנה ב

{ה} בְּסֵפֶר בֵּית דָּוִד הִצְרִיךְ עִיּוּן לָמָּה נִדְחַק לְפָרֵשׁ כֵּן, דְּאַף בְּאוֹכְלִין מִן הַתּוֹרָה הַדִּין כֵּן. וְכֵן הוּא בָּרַמְבַּ"ם בְּחִבּוּרוֹ: {ו} וְעַיֵּן בְּמַעֲשֵׂר שֵׁנִי פֶּרֶק א' מִשְׁנָה א': {ז} אַחַת כוּ'. לְפִי שֶׁאֵין כָּאן דָּבָר שֶׁיִּקְבָּעֵהוּ בְּלֹא הַמִּקָּח, הִלְכָּךְ לֹא סַגִּי הַמִּקָּח לְקָבְעוֹ קֹדֶם שֶׁנִּגְמְרָה מְלַאכְתּוֹ, כְּרַבִּי יְהוּדָה סוֹף פֶּרֶק דִּלְעֵיל. וּשְׁתַּיִם לֹא יֹאכְלוּ, כְּמוֹ שֶׁאָמְרוּ בְּמִשְׁנָה ה' פֶּרֶק דִּלְעֵיל: {ח} הַקֻּפָּה. דִּלְגַבֵּי לוֹקֵחַ הֲוֵי כְּמוֹלִיךְ לַשּׁוּק, דְּסַגִּי לְקָבְעוֹ כְּדִתְנַן בְּמִשְׁנָה ה' פֶּרֶק קַמָּא, וְהַלּוֹקֵחַ עֵינָיו בְּמִקְחוֹ כְּדִלְעֵיל: {ט} הַמֻּקְצֶה. דְּמֻקְצֶה עָשׂוּ אוֹתוֹ כִּמְצָרֵף. יְרוּשַׁלְמִי:

{ז} רַבִּי יְהוּדָה מְחַיֵּב. וְהַיְנוּ בַּמַּעֲשֵׂר, אֲבָל בַּתְּרוּמָה לֹא, כִּדְאָמַר לְעֵיל מִשְׁנָה ה' הַתּוֹרֵם בְּלִבּוֹ עַל הַכֹּל. הָרַ"שׁ בְּשֵׁם הַיְרוּשַׁלְמִי. עוֹד כָּתַב דְּעַד כָּאן לֹא מְחַיֵּב רַבִּי יְהוּדָה אֶלָּא מִנֵּיהּ וּבֵיהּ, אֲבָל מֵעָלְמָא לֹא, דִּלְמָא אָתֵי לְאַפְרוּשֵׁי מִן הַחִיּוּב עַל הַפָּטוּר כוּ':

{ח} שֶׁאֲבִיהֶן תְּרוּמָה. כְּהַהִיא דִתְנַן בְּרֵישׁ פִּרְקִין הָעוֹקֵר לֶפֶת וּצְנוֹנוֹת מִתּוֹךְ שֶׁלּוֹ וְנוֹטֵעַ לְתוֹךְ שֶׁלּוֹ לְזֶרַע, חַיָּב מִפְּנֵי שֶׁהוּא גָּרְנָן. וְאִם עֲשָׂאָן תְּרוּמָה וּנְטָעָן לְזֶרַע, הַזֶּרַע נֶאֱכָל לְזָרִים. תּוֹסְפוֹת יוֹם טוֹב:
