ר-א


[חבקוק פרק-ג-י]  רָאוּךָ יָחִילוּ הָרִים זֶרֶם מַיִם עָבָר נָתַן תְּהוֹם קוֹלוֹ רוֹם יָדֵיהוּ נָשָׂא:

[תהילים פרק-צט-ה]  רוֹמְמוּ יְהוָה אֱלֹהֵינוּ וְהִשְׁתַּחֲווּ לַהֲדֹם רַגְלָיו קָדוֹשׁ הוּא:

[משלי פרק-כ-ו]  רָב אָדָם יִקְרָא אִישׁ חַסְדּוֹ וְאִישׁ אֱמוּנִים מִי יִמְצָא:

[איוב פרק-ח-טז]  רָטֹב הוּא לִפְנֵי שָׁמֶשׁ וְעַל גַּנָּתוֹ יֹנַקְתּוֹ תֵצֵא:

[איכה פרק-ב-כ]  רְאֵה יְהוָה וְהַבִּיטָה לְמִי עוֹלַלְתָּ כֹּה אִם תֹּאכַלְנָה נָשִׁים פִּרְיָם עֹלֲלֵי טִפֻּחִים אִם יֵהָרֵג בְּמִקְדַּשׁ אֲדֹנָי כֹּהֵן וְנָבִיא:

[דניאל פרק-ד-ח]  רְבָה אִילָנָא וּתְקִף וְרוּמֵהּ יִמְטֵא לִשְׁמַיָּא וַחֲזוֹתֵהּ לְסוֹף כָּל אַרְעָא:

[עזרא פרק-ד-ח]  רְחוּם בְּעֵל טְעֵם וְשִׁמְשַׁי סָפְרָא כְּתַבוּ אִגְּרָה חֲדָה עַל יְרוּשְׁלֶם לְאַרְתַּחְשַׁשְׂתְּא מַלְכָּא כְּנֵמָא:


ר-ב


[תהילים פרק-סט-ה]  רַבּוּ מִשַּׂעֲרוֹת רֹאשִׁי שֹׂנְאַי חִנָּם עָצְמוּ מַצְמִיתַי אֹיְבַי שֶׁקֶר אֲשֶׁר לֹא גָזַלְתִּי אָז אָשִׁיב:

[תהילים פרק-פה-ב]  רָצִיתָ יְהוָה אַרְצֶךָ שַׁבְתָּ (שבות) שְׁבִית יַעֲקֹב:

[משלי פרק-טז-יג]  רְצוֹן מְלָכִים שִׂפְתֵי צֶדֶק וְדֹבֵר יְשָׁרִים יֶאֱהָב:

[איוב פרק-כד-כא]  רֹעֶה עֲקָרָה לֹא תֵלֵד וְאַלְמָנָה לֹא יְיֵטִיב:

[שיר השירים פרק-ה-יא]  רֹאשׁוֹ כֶּתֶם פָּז קְוּצּוֹתָיו תַּלְתַּלִּים שְׁחֹרוֹת כָּעוֹרֵב:


ר-ד


[שמות פרק-ט-כו]  רַק בְּאֶרֶץ גֹּשֶׁן אֲשֶׁר שָׁם בְּנֵי יִשְׂרָאֵל לֹא הָיָה בָּרָד:

[דברים פרק-יז-טז]  רַק לֹא יַרְבֶּה לּוֹ סוּסִים וְלֹא יָשִׁיב אֶת הָעָם מִצְרַיְמָה לְמַעַן הַרְבּוֹת סוּס וַיהוָה אָמַר לָכֶם לֹא תֹסִפוּן לָשׁוּב בַּדֶּרֶךְ הַזֶּה עוֹד:

[ירמיה פרק-יח-ז]  רֶגַע אֲדַבֵּר עַל גּוֹי וְעַל מַמְלָכָה לִנְתוֹשׁ וְלִנְתוֹץ וּלְהַאֲבִיד:

[תהילים פרק-קג-ח]  רַחוּם וְחַנּוּן יְהוָה אֶרֶךְ אַפַּיִם וְרַב חָסֶד:

[תהילים פרק-קיא-י]  רֵאשִׁית חָכְמָה יִרְאַת יְהוָה שֵׂכֶל טוֹב לְכָל עֹשֵׂיהֶם תְּהִלָּתוֹ עֹמֶדֶת לָעַד:

[תהילים פרק-קיב-י]  רָשָׁע יִרְאֶה וְכָעָס שִׁנָּיו יַחֲרֹק וְנָמָס תַּאֲוַת רְשָׁעִים תֹּאבֵד:

[משלי פרק-יג-יח]  רֵישׁ וְקָלוֹן פּוֹרֵעַ מוּסָר וְשׁוֹמֵר תּוֹכַחַת יְכֻבָּד:

[משלי פרק-כא-כא]  רֹדֵף צְדָקָה וָחָסֶד יִמְצָא חַיִּים צְדָקָה וְכָבוֹד:

[איוב פרק-לד-כ]  רֶגַע יָמֻתוּ וַחֲצוֹת לָיְלָה יְגֹעֲשׁוּ עָם וְיַעֲבֹרוּ וְיָסִירוּ אַבִּיר לֹא בְיָד:


ר-ה


[שמות פרק-א-ב]  רְאוּבֵן שִׁמְעוֹן לֵוִי וִיהוּדָה:

[שמות פרק-לא-ב]  רְאֵה קָרָאתִי בְשֵׁם בְּצַלְאֵל בֶּן אוּרִי בֶן חוּר לְמַטֵּה יְהוּדָה:

[במדבר פרק-טו-כ]  רֵאשִׁית עֲרִסֹתֵכֶם חַלָּה תָּרִימוּ תְרוּמָה כִּתְרוּמַת גֹּרֶן כֵּן תָּרִימוּ אֹתָהּ:

[דברים פרק-ב-ג]  רַב לָכֶם סֹב אֶת הָהָר הַזֶּה פְּנוּ לָכֶם צָפֹנָה:

[דברים פרק-ד-ה]  רְאֵה לִמַּדְתִּי אֶתְכֶם חֻקִּים וּמִשְׁפָּטִים כַּאֲשֶׁר צִוַּנִי יְהוָה אֱלֹהָי לַעֲשׂוֹת כֵּן בְּקֶרֶב הָאָרֶץ אֲשֶׁר אַתֶּם בָּאִים שָׁמָּה לְרִשְׁתָּהּ:

[דברים פרק-י-טו]  רַק בַּאֲבֹתֶיךָ חָשַׁק יְהוָה לְאַהֲבָה אוֹתָם וַיִּבְחַר בְּזַרְעָם אַחֲרֵיהֶם בָּכֶם מִכָּל הָעַמִּים כַּיּוֹם הַזֶּה:

[דברים פרק-יא-כו]  רְאֵה אָנֹכִי נֹתֵן לִפְנֵיכֶם הַיּוֹם בְּרָכָה וּקְלָלָה:

[דברים פרק-יב-כו]  רַק קָדָשֶׁיךָ אֲשֶׁר יִהְיוּ לְךָ וּנְדָרֶיךָ תִּשָּׂא וּבָאתָ אֶל הַמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה:

[דברים פרק-כ-טז]  רַק מֵעָרֵי הָעַמִּים הָאֵלֶּה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה לֹא תְחַיֶּה כָּל נְשָׁמָה:

[דברים פרק-כ-כ]  רַק עֵץ אֲשֶׁר תֵּדַע כִּי לֹא עֵץ מַאֲכָל הוּא אֹתוֹ תַשְׁחִית וְכָרָתָּ וּבָנִיתָ מָצוֹר עַל הָעִיר אֲשֶׁר הִוא עֹשָׂה עִמְּךָ מִלְחָמָה עַד רִדְתָּהּ:

[מלכים א פרק-ח-לז]  רָעָב כִּי יִהְיֶה בָאָרֶץ דֶּבֶר כִּי יִהְיֶה שִׁדָּפוֹן יֵרָקוֹן אַרְבֶּה חָסִיל כִּי יִהְיֶה כִּי יָצַר לוֹ אֹיְבוֹ בְּאֶרֶץ שְׁעָרָיו כָּל נֶגַע כָּל מַחֲלָה:

[מלכים ב פרק-ג-ג]  רַק בְּחַטֹּאות יָרָבְעָם בֶּן נְבָט אֲשֶׁר הֶחֱטִיא אֶת יִשְׂרָאֵל דָּבֵק לֹא סָר מִמֶּנָּה:

[ישעיה פרק-מז-א]  רְדִי וּשְׁבִי עַל עָפָר בְּתוּלַת בַּת בָּבֶל שְׁבִי לָאָרֶץ אֵין כִּסֵּא בַּת כַּשְׂדִּים כִּי לֹא תוֹסִיפִי יִקְרְאוּ לָךְ רַכָּה וַעֲנֻגָּה:

[ישעיה פרק-נד-א]  רָנִּי עֲקָרָה לֹא יָלָדָה פִּצְחִי רִנָּה וְצַהֲלִי לֹא חָלָה כִּי רַבִּים בְּנֵי שׁוֹמֵמָה מִבְּנֵי בְעוּלָה אָמַר יְהוָה:

[ירמיה פרק-יב-י]  רֹעִים רַבִּים שִׁחֲתוּ כַרְמִי בֹּסְסוּ אֶת חֶלְקָתִי נָתְנוּ אֶת חֶלְקַת חֶמְדָּתִי לְמִדְבַּר שְׁמָמָה:

[ירמיה פרק-יז-יד]  רְפָאֵנִי יְהוָה וְאֵרָפֵא הוֹשִׁיעֵנִי וְאִוָּשֵׁעָה כִּי תְהִלָּתִי אָתָּה:

[ירמיה פרק-מט-כד]  רָפְתָה דַמֶּשֶׂק הִפְנְתָה לָנוּס וְרֶטֶט הֶחֱזִיקָה צָרָה וַחֲבָלִים אֲחָזַתָּה כַּיּוֹלֵדָה:

[ירמיה פרק-נא-לא]  רָץ לִקְרַאת רָץ יָרוּץ וּמַגִּיד לִקְרַאת מַגִּיד לְהַגִּיד לְמֶלֶךְ בָּבֶל כִּי נִלְכְּדָה עִירוֹ מִקָּצֶה:

[יחזקאל פרק-טז-ז]  רְבָבָה כְּצֶמַח הַשָּׂדֶה נְתַתִּיךְ וַתִּרְבִּי וַתִּגְדְּלִי וַתָּבֹאִי בַּעֲדִי עֲדָיִים שָׁדַיִם נָכֹנוּ וּשְׂעָרֵךְ צִמֵּחַ וְאַתְּ עֵרֹם וְעֶרְיָה:

[הושע פרק-ב-ד]  רִיבוּ בְאִמְּכֶם רִיבוּ כִּי הִיא לֹא אִשְׁתִּי וְאָנֹכִי לֹא אִישָׁהּ וְתָסֵר זְנוּנֶיהָ מִפָּנֶיה וְנַאֲפוּפֶיהָ מִבֵּין שָׁדֶיהָ:

[מיכה פרק-ג-יא]  רָאשֶׁיהָ בְּשֹׁחַד יִשְׁפֹּטוּ וְכֹהֲנֶיהָ בִּמְחִיר יוֹרוּ וּנְבִיאֶיהָ בְּכֶסֶף יִקְסֹמוּ וְעַל יְהוָה יִשָּׁעֵנוּ לֵאמֹר הֲלוֹא יְהוָה בְּקִרְבֵּנוּ לֹא תָבוֹא עָלֵינוּ רָעָה:

[זכריה פרק-ב-יד]  רָנִּי וְשִׂמְחִי בַּת צִיּוֹן כִּי הִנְנִי בָא וְשָׁכַנְתִּי בְתוֹכֵךְ נְאֻם יְהוָה:

[תהילים פרק-ג-ג]  רַבִּים אֹמְרִים לְנַפְשִׁי אֵין יְשׁוּעָתָה לּוֹ בֵאלֹהִים סֶלָה:

[תהילים פרק-ד-ה]  רִגְזוּ וְאַל תֶּחֱטָאוּ אִמְרוּ בִלְבַבְכֶם עַל מִשְׁכַּבְכֶם וְדֹמּוּ סֶלָה:

[תהילים פרק-ד-ז]  רַבִּים אֹמְרִים מִי יַרְאֵנוּ טוֹב נְסָה עָלֵינוּ אוֹר פָּנֶיךָ יְהוָה:

[תהילים פרק-כו-יב]  רַגְלִי עָמְדָה בְמִישׁוֹר בְּמַקְהֵלִים אֲבָרֵךְ יְהוָה:

[תהילים פרק-לג-א]  רַנְּנוּ צַדִּיקִים בַּיהוָה לַיְשָׁרִים נָאוָה תְהִלָּה:

[תהילים פרק-לד-כ]  רַבּוֹת רָעוֹת צַדִּיק וּמִכֻּלָּם יַצִּילֶנּוּ יְהוָה:

[תהילים פרק-מ-יד]  רְצֵה יְהוָה לְהַצִּילֵנִי יְהוָה לְעֶזְרָתִי חוּשָׁה:

[תהילים פרק-מח-ז]  רְעָדָה אֲחָזָתַם שָׁם חִיל כַּיּוֹלֵדָה:

[תהילים פרק-נז-ז]  רֶשֶׁת הֵכִינוּ לִפְעָמַי כָּפַף נַפְשִׁי כָּרוּ לְפָנַי שִׁיחָה נָפְלוּ בְתוֹכָהּ סֶלָה:

[תהילים פרק-פ-ב]  רֹעֵה יִשְׂרָאֵל הַאֲזִינָה נֹהֵג כַּצֹּאן יוֹסֵף יֹשֵׁב הַכְּרוּבִים הוֹפִיעָה:

[תהילים פרק-צא-ח]  רַק בְּעֵינֶיךָ תַבִּיט וְשִׁלֻּמַת רְשָׁעִים תִּרְאֶה:

[משלי פרק-ד-ז]  רֵאשִׁית חָכְמָה קְנֵה חָכְמָה וּבְכָל קִנְיָנְךָ קְנֵה בִינָה:

[משלי פרק-יג-י]  רַק בְּזָדוֹן יִתֵּן מַצָּה וְאֶת נוֹעָצִים חָכְמָה:

[משלי פרק-יח-יד]  רוּחַ אִישׁ יְכַלְכֵּל מַחֲלֵהוּ וְרוּחַ נְכֵאָה מִי יִשָּׂאֶנָּה:

[משלי פרק-כט-יג]  רָשׁ וְאִישׁ תְּכָכִים נִפְגָּשׁוּ מֵאִיר עֵינֵי שְׁנֵיהֶם יְהוָה:

[משלי פרק-לא-כט]  רַבּוֹת בָּנוֹת עָשׂוּ חָיִל וְאַתְּ עָלִית עַל כֻּלָּנָה:

[איוב פרק-כ-טז]  רֹאשׁ פְּתָנִים יִינָק תַּהַרְגֵהוּ לְשׁוֹן אֶפְעֶה:

[נחמיה פרק-י-כו]  רְחוּם חֲשַׁבְנָה מַעֲשֵׂיָה:

[דברי הימים א פרק-כח-י]  רְאֵה עַתָּה כִּי יְהוָה בָּחַר בְּךָ לִבְנוֹת בַּיִת לַמִּקְדָּשׁ חֲזַק וַעֲשֵׂה:

[דברי הימים ב פרק-ו-כח]  רָעָב כִּי יִהְיֶה בָאָרֶץ דֶּבֶר כִּי יִהְיֶה שִׁדָּפוֹן וְיֵרָקוֹן אַרְבֶּה וְחָסִיל כִּי יִהְיֶה כִּי יָצַר לוֹ אוֹיְבָיו בְּאֶרֶץ שְׁעָרָיו כָּל נֶגַע וְכָל מַחֲלָה:


ר-ו


[שמות פרק-כג-יט]  רֵאשִׁית בִּכּוּרֵי אַדְמָתְךָ תָּבִיא בֵּית יְהוָה אֱלֹהֶיךָ לֹא תְבַשֵּׁל גְּדִי בַּחֲלֵב אִמּוֹ:

[שמות פרק-כח-טז]  רָבוּעַ יִהְיֶה כָּפוּל זֶרֶת אָרְכּוֹ וְזֶרֶת רָחְבּוֹ:

[שמות פרק-לד-כו]  רֵאשִׁית בִּכּוּרֵי אַדְמָתְךָ תָּבִיא בֵּית יְהוָה אֱלֹהֶיךָ לֹא תְבַשֵּׁל גְּדִי בַּחֲלֵב אִמּוֹ:

[דברים פרק-ב-לה]  רַק הַבְּהֵמָה בָּזַזְנוּ לָנוּ וּשְׁלַל הֶעָרִים אֲשֶׁר לָכָדְנוּ:

[דברים פרק-ב-לז]  רַק אֶל אֶרֶץ בְּנֵי עַמּוֹן לֹא קָרָבְתָּ כָּל יַד נַחַל יַבֹּק וְעָרֵי הָהָר וְכֹל אֲשֶׁר צִוָּה יְהוָה אֱלֹהֵינוּ:

[דברים פרק-יח-ד]  רֵאשִׁית דְּגָנְךָ תִּירֹשְׁךָ וְיִצְהָרֶךָ וְרֵאשִׁית גֵּז צֹאנְךָ תִּתֶּן לּוֹ:

[יהושע פרק-יג-יד]  רַק לְשֵׁבֶט הַלֵּוִי לֹא נָתַן נַחֲלָה אִשֵּׁי יְהוָה אֱלֹהֵי יִשְׂרָאֵל הוּא נַחֲלָתוֹ כַּאֲשֶׁר דִּבֶּר לוֹ:

[שופטים פרק-ה-י]  רֹכְבֵי אֲתֹנוֹת צְחֹרוֹת יֹשְׁבֵי עַל מִדִּין וְהֹלְכֵי עַל דֶּרֶךְ שִׂיחוּ:

[מלכים א פרק-כא-כה]  רַק לֹא הָיָה כְאַחְאָב אֲשֶׁר הִתְמַכֵּר לַעֲשׂוֹת הָרַע בְּעֵינֵי יְהוָה אֲשֶׁר הֵסַתָּה אֹתוֹ אִיזֶבֶל אִשְׁתּוֹ:

[ישעיה פרק-ח-ט]  רֹעוּ עַמִּים וָחֹתּוּ וְהַאֲזִינוּ כֹּל מֶרְחַקֵּי אָרֶץ הִתְאַזְּרוּ וָחֹתּוּ הִתְאַזְּרוּ וָחֹתּוּ:

[ירמיה פרק-ד-כד]  רָאִיתִי הֶהָרִים וְהִנֵּה רֹעֲשִׁים וְכָל הַגְּבָעוֹת הִתְקַלְקָלוּ:

[ירמיה פרק-ד-כה]  רָאִיתִי וְהִנֵּה אֵין הָאָדָם וְכָל עוֹף הַשָּׁמַיִם נָדָדוּ:

[ירמיה פרק-ד-כו]  רָאִיתִי וְהִנֵּה הַכַּרְמֶל הַמִּדְבָּר וְכָל עָרָיו נִתְּצוּ מִפְּנֵי יְהוָה מִפְּנֵי חֲרוֹן אַפּוֹ:

[תהילים פרק-י-ד]  רָשָׁע כְּגֹבַהּ אַפּוֹ בַּל יִדְרֹשׁ אֵין אֱלֹהִים כָּל מְזִמּוֹתָיו:

[תהילים פרק-לב-י]  רַבִּים מַכְאוֹבִים לָרָשָׁע וְהַבּוֹטֵחַ בַּיהוָה חֶסֶד יְסוֹבְבֶנּוּ:

[תהילים פרק-צט-ט]  רוֹמְמוּ יְהוָה אֱלֹהֵינוּ וְהִשְׁתַּחֲווּ לְהַר קָדְשׁוֹ כִּי קָדוֹשׁ יְהוָה אֱלֹהֵינוּ:

[תהילים פרק-קיג-ד]  רָם עַל כָּל גּוֹיִם יְהוָה עַל הַשָּׁמַיִם כְּבוֹדוֹ:

[תהילים פרק-קיט-קנה]  רָחוֹק מֵרְשָׁעִים יְשׁוּעָה כִּי חֻקֶּיךָ לֹא דָרָשׁוּ:

[תהילים פרק-קיט-קנח]  רָאִיתִי בֹגְדִים וָאֶתְקוֹטָטָה אֲשֶׁר אִמְרָתְךָ לֹא שָׁמָרוּ:

[תהילים פרק-קמ-י]  רֹאשׁ מְסִבָּי עֲמַל שְׂפָתֵימוֹ (יכסומו) יְכַסֵּמוֹ:

[תהילים פרק-קמז-יא]  רוֹצֶה יְהוָה אֶת יְרֵאָיו אֶת הַמְיַחֲלִים לְחַסְדּוֹ:

[משלי פרק-ה-ה]  רַגְלֶיהָ יֹרְדוֹת מָוֶת שְׁאוֹל צְעָדֶיהָ יִתְמֹכוּ:

[משלי פרק-כד-ז]  רָאמוֹת לֶאֱוִיל חָכְמוֹת בַּשַּׁעַר לֹא יִפְתַּח פִּיהוּ:

[משלי פרק-כו-יב]  רָאִיתָ אִישׁ חָכָם בְּעֵינָיו תִּקְוָה לִכְסִיל מִמֶּנּוּ:

[איוב פרק-כד-כד]  רוֹמּוּ מְּעַט וְאֵינֶנּוּ וְהֻמְּכוּ כַּכֹּל יִקָּפְצוּן וּכְרֹאשׁ שִׁבֹּלֶת יִמָּלוּ:

[איוב פרק-כט-ח]  רָאוּנִי נְעָרִים וְנֶחְבָּאוּ וִישִׁישִׁים קָמוּ עָמָדוּ:

[איוב פרק-לג-כה]  רֻטֲפַשׁ בְּשָׂרוֹ מִנֹּעַר יָשׁוּב לִימֵי עֲלוּמָיו:

[קהלת פרק-ג-י]  רָאִיתִי אֶת הָעִנְיָן אֲשֶׁר נָתַן אֱלֹהִים לִבְנֵי הָאָדָם לַעֲנוֹת בּוֹ:

[קהלת פרק-ד-טו]  רָאִיתִי אֶת כָּל הַחַיִּים הַמְהַלְּכִים תַּחַת הַשָּׁמֶשׁ עִם הַיֶּלֶד הַשֵּׁנִי אֲשֶׁר יַעֲמֹד תַּחְתָּיו:

[קהלת פרק-ז-יג]  רְאֵה אֶת מַעֲשֵׂה הָאֱלֹהִים כִּי מִי יוּכַל לְתַקֵּן אֵת אֲשֶׁר עִוְּתוֹ:

[קהלת פרק-ז-כד]  רָחוֹק מַה שֶּׁהָיָה וְעָמֹק עָמֹק מִי יִמְצָאֶנּוּ:


ר-ז


[בראשית פרק-מט-ג]  רְאוּבֵן בְּכֹרִי אַתָּה כֹּחִי וְרֵאשִׁית אוֹנִי יֶתֶר שְׂאֵת וְיֶתֶר עָז:


ר-ח


[ישעיה פרק-סא-א]  רוּחַ אֲדֹנָי יְהוִה עָלָי יַעַן מָשַׁח יְהוָה אֹתִי לְבַשֵּׂר עֲנָוִים שְׁלָחַנִי לַחֲבֹשׁ לְנִשְׁבְּרֵי לֵב לִקְרֹא לִשְׁבוּיִם דְּרוֹר וְלַאֲסוּרִים פְּקַח קוֹחַ:

[משלי פרק-יא-טו]  רַע יֵרוֹעַ כִּי עָרַב זָר וְשֹׂנֵא תֹקְעִים בּוֹטֵחַ:

[קהלת פרק-א-יד]  רָאִיתִי אֶת כָּל הַמַּעֲשִׂים שֶׁנַּעֲשׂוּ תַּחַת הַשָּׁמֶשׁ וְהִנֵּה הַכֹּל הֶבֶל וּרְעוּת רוּחַ:


ר-ט


[עמוס פרק-ט-א]  רָאִיתִי אֶת אֲדֹנָי נִצָּב עַל הַמִּזְבֵּחַ וַיֹּאמֶר הַךְ הַכַּפְתּוֹר וְיִרְעֲשׁוּ הַסִּפִּים וּבְצַעַם בְּרֹאשׁ כֻּלָּם וְאַחֲרִיתָם בַּחֶרֶב אֶהֱרֹג לֹא יָנוּס לָהֶם נָס וְלֹא יִמָּלֵט לָהֶם פָּלִיט:

[משלי פרק-יג-כג]  רָב אֹכֶל נִיר רָאשִׁים וְיֵשׁ נִסְפֶּה בְּלֹא מִשְׁפָּט:


ר-י


[שמות פרק-טז-כט]  רְאוּ כִּי יְהוָה נָתַן לָכֶם הַשַּׁבָּת עַל כֵּן הוּא נֹתֵן לָכֶם בַּיּוֹם הַשִּׁשִּׁי לֶחֶם יוֹמָיִם שְׁבוּ אִישׁ תַּחְתָּיו אַל יֵצֵא אִישׁ מִמְּקֹמוֹ בַּיּוֹם הַשְּׁבִיעִי:

[במדבר פרק-כו-ה]  רְאוּבֵן בְּכוֹר יִשְׂרָאֵל בְּנֵי רְאוּבֵן חֲנוֹךְ מִשְׁפַּחַת הַחֲנֹכִי לְפַלּוּא מִשְׁפַּחַת הַפַּלֻּאִי:

[שמואל ב פרק-טו-כח]  רְאוּ אָנֹכִי מִתְמַהְמֵהַּ (בעברות) בְּעַרְבוֹת הַמִּדְבָּר עַד בּוֹא דָבָר מֵעִמָּכֶם לְהַגִּיד לִי:

[שמואל ב פרק-כג-ב]  רוּחַ יְהוָה דִּבֶּר בִּי וּמִלָּתוֹ עַל לְשׁוֹנִי:

[מלכים א פרק-ח-יט]  רַק אַתָּה לֹא תִבְנֶה הַבָּיִת כִּי אִם בִּנְךָ הַיֹּצֵא מֵחֲלָצֶיךָ הוּא יִבְנֶה הַבַּיִת לִשְׁמִי:

[מלכים א פרק-יא-יג]  רַק אֶת כָּל הַמַּמְלָכָה לֹא אֶקְרָע שֵׁבֶט אֶחָד אֶתֵּן לִבְנֶךָ לְמַעַן דָּוִד עַבְדִּי וּלְמַעַן יְרוּשָׁלִַם אֲשֶׁר בָּחָרְתִּי:

[ישעיה פרק-מד-כ]  רֹעֶה אֵפֶר לֵב הוּתַל הִטָּהוּ וְלֹא יַצִּיל אֶת נַפְשׁוֹ וְלֹא יֹאמַר הֲלוֹא שֶׁקֶר בִּימִינִי:

[תהילים פרק-כה-יח]  רְאֵה עָנְיִי וַעֲמָלִי וְשָׂא לְכָל חַטֹּאותָי:

[תהילים פרק-כה-יט]  רְאֵה אוֹיְבַי כִּי רָבּוּ וְשִׂנְאַת חָמָס שְׂנֵאוּנִי:

[תהילים פרק-לה-כב]  רָאִיתָה יְהוָה אַל תֶּחֱרַשׁ אֲדֹנָי אֲל תִּרְחַק מִמֶּנִּי:

[תהילים פרק-קיט-קנג]  רְאֵה עָנְיִי וְחַלְּצֵנִי כִּי תוֹרָתְךָ לֹא שָׁכָחְתִּי:

[תהילים פרק-קיט-קנד]  רִיבָה רִיבִי וּגְאָלֵנִי לְאִמְרָתְךָ חַיֵּנִי:

[תהילים פרק-קיט-קנו]  רַחֲמֶיךָ רַבִּים יְהוָה כְּמִשְׁפָּטֶיךָ חַיֵּנִי:

[תהילים פרק-קיט-קנז]  רַבִּים רֹדְפַי וְצָרָי מֵעֵדְוֹתֶיךָ לֹא נָטִיתִי:

[תהילים פרק-קיט-קנט]  רְאֵה כִּי פִקּוּדֶיךָ אָהָבְתִּי יְהוָה כְּחַסְדְּךָ חַיֵּנִי:

[תהילים פרק-קכט-ב]  רַבַּת צְרָרוּנִי מִנְּעוּרָי גַּם לֹא יָכְלוּ לִי:

[איוב פרק-יז-א]  רוּחִי חֻבָּלָה יָמַי נִזְעָכוּ קְבָרִים לִי:

[איוב פרק-יט-יז]  רוּחִי זָרָה לְאִשְׁתִּי וְחַנֹּתִי לִבְנֵי בִטְנִי:

[איוב פרק-לג-ד]  רוּחַ אֵל עָשָׂתְנִי וְנִשְׁמַת שַׁדַּי תְּחַיֵּנִי:

[איכה פרק-ג-נח]  רַבְתָּ אֲדֹנָי רִיבֵי נַפְשִׁי גָּאַלְתָּ חַיָּי:

[איכה פרק-ג-נט]  רָאִיתָה יְהוָה עַוָּתָתִי שָׁפְטָה מִשְׁפָּטִי:

[איכה פרק-ג-ס]  רָאִיתָה כָּל נִקְמָתָם כָּל מַחְשְׁבֹתָם לִי:

[נחמיה פרק-י-טו]  רָאשֵׁי הָעָם פַּרְעֹשׁ פַּחַת מוֹאָב עֵילָם זַתּוּא בָּנִי:

[דברי הימים ב פרק-ו-ט]  רַק אַתָּה לֹא תִבְנֶה הַבָּיִת כִּי בִנְךָ הַיּוֹצֵא מֵחֲלָצֶיךָ הוּא יִבְנֶה הַבַּיִת לִשְׁמִי:


ר-כ


[דברים פרק-ד-ט]  רַק הִשָּׁמֶר לְךָ וּשְׁמֹר נַפְשְׁךָ מְאֹד פֶּן תִּשְׁכַּח אֶת הַדְּבָרִים אֲשֶׁר רָאוּ עֵינֶיךָ וּפֶן יָסוּרוּ מִלְּבָבְךָ כֹּל יְמֵי חַיֶּיךָ וְהוֹדַעְתָּם לְבָנֶיךָ וְלִבְנֵי בָנֶיךָ:

[דברים פרק-כ-יד]  רַק הַנָּשִׁים וְהַטַּף וְהַבְּהֵמָה וְכֹל אֲשֶׁר יִהְיֶה בָעִיר כָּל שְׁלָלָהּ תָּבֹז לָךְ וְאָכַלְתָּ אֶת שְׁלַל אֹיְבֶיךָ אֲשֶׁר נָתַן יְהוָה אֱלֹהֶיךָ לָךְ:

[יהושע פרק-א-ז]  רַק חֲזַק וֶאֱמַץ מְאֹד לִשְׁמֹר לַעֲשׂוֹת כְּכָל הַתּוֹרָה אֲשֶׁר צִוְּךָ מֹשֶׁה עַבְדִּי אַל תָּסוּר מִמֶּנּוּ יָמִין וּשְׂמֹאול לְמַעַן תַּשְׂכִּיל בְּכֹל אֲשֶׁר תֵּלֵךְ:

[ירמיה פרק-מח-יח]  רְדִי מִכָּבוֹד (ישבי) וּשְׁבִי בַצָּמָא יֹשֶׁבֶת בַּת דִּיבוֹן כִּי שֹׁדֵד מוֹאָב עָלָה בָךְ שִׁחֵת מִבְצָרָיִךְ:

[יחזקאל פרק-כז-כב]  רֹכְלֵי שְׁבָא וְרַעְמָה הֵמָּה רֹכְלָיִךְ בְּרֹאשׁ כָּל בֹּשֶׂם וּבְכָל אֶבֶן יְקָרָה וְזָהָב נָתְנוּ עִזְבוֹנָיִךְ:

[תהילים פרק-כא-יד]  רוּמָה יְהוָה בְּעֻזֶּךָ נָשִׁירָה וּנְזַמְּרָה גְּבוּרָתֶךָ:

[תהילים פרק-נז-ו]  רוּמָה עַל הַשָּׁמַיִם אֱלֹהִים עַל כָּל הָאָרֶץ כְּבוֹדֶךָ:

[תהילים פרק-נז-יב]  רוּמָה עַל שָׁמַיִם אֱלֹהִים עַל כָּל הָאָרֶץ כְּבוֹדֶךָ:

[תהילים פרק-קח-ו]  רוּמָה עַל שָׁמַיִם אֱלֹהִים וְעַל כָּל הָאָרֶץ כְּבוֹדֶךָ:

[תהילים פרק-קיט-קס]  רֹאשׁ דְּבָרְךָ אֱמֶת וּלְעוֹלָם כָּל מִשְׁפַּט צִדְקֶךָ:

[משלי פרק-ג-ח]  רִפְאוּת תְּהִי לְשָׁרֶּךָ וְשִׁקּוּי לְעַצְמוֹתֶיךָ:


ר-ל


[שמות פרק-לט-ט]  רָבוּעַ הָיָה כָּפוּל עָשׂוּ אֶת הַחֹשֶׁן זֶרֶת אָרְכּוֹ וְזֶרֶת רָחְבּוֹ כָּפוּל:

[דברים פרק-יב-טו]  רַק בְּכָל אַוַּת נַפְשְׁךָ תִּזְבַּח וְאָכַלְתָּ בָשָׂר כְּבִרְכַּת יְהוָה אֱלֹהֶיךָ אֲשֶׁר נָתַן לְךָ בְּכָל שְׁעָרֶיךָ הַטָּמֵא וְהַטָּהוֹר יֹאכְלֶנּוּ כַּצְּבִי וְכָאַיָּל:

[דברים פרק-לב-לט]  רְאוּ עַתָּה כִּי אֲנִי אֲנִי הוּא וְאֵין אֱלֹהִים עִמָּדִי אֲנִי אָמִית וַאֲחַיֶּה מָחַצְתִּי וַאֲנִי אֶרְפָּא וְאֵין מִיָּדִי מַצִּיל:

[מיכה פרק-א-יג]  רְתֹם הַמֶּרְכָּבָה לָרֶכֶשׁ יוֹשֶׁבֶת לָכִישׁ רֵאשִׁית חַטָּאת הִיא לְבַת צִיּוֹן כִּי בָךְ נִמְצְאוּ פִּשְׁעֵי יִשְׂרָאֵל:

[משלי פרק-כ-יד]  רַע רַע יֹאמַר הַקּוֹנֶה וְאֹזֵל לוֹ אָז יִתְהַלָּל:

[משלי פרק-כה-ט]  רִיבְךָ רִיב אֶת רֵעֶךָ וְסוֹד אַחֵר אַל תְּגָל:

[דניאל פרק-ח-ד]  רָאִיתִי אֶת הָאַיִל מְנַגֵּחַ יָמָּה וְצָפוֹנָה וָנֶגְבָּה וְכָל חַיּוֹת לֹא יַעַמְדוּ לְפָנָיו וְאֵין מַצִּיל מִיָּדוֹ וְעָשָׂה כִרְצֹנוֹ וְהִגְדִּיל:


ר-מ


[בראשית פרק-מז-כב]  רַק אַדְמַת הַכֹּהֲנִים לֹא קָנָה כִּי חֹק לַכֹּהֲנִים מֵאֵת פַּרְעֹה וְאָכְלוּ אֶת חֻקָּם אֲשֶׁר נָתַן לָהֶם פַּרְעֹה עַל כֵּן לֹא מָכְרוּ אֶת אַדְמָתָם:

[דברים פרק-א-ח]  רְאֵה נָתַתִּי לִפְנֵיכֶם אֶת הָאָרֶץ בֹּאוּ וּרְשׁוּ אֶת הָאָרֶץ אֲשֶׁר נִשְׁבַּע יְהוָה לַאֲבֹתֵיכֶם לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לָתֵת לָהֶם וּלְזַרְעָם אַחֲרֵיהֶם:

[דברים פרק-ב-יא]  רְפָאִים יֵחָשְׁבוּ אַף הֵם כָּעֲנָקִים וְהַמֹּאָבִים יִקְרְאוּ לָהֶם אֵמִים:

[דברים פרק-ג-יט]  רַק נְשֵׁיכֶם וְטַפְּכֶם וּמִקְנֵכֶם יָדַעְתִּי כִּי מִקְנֶה רַב לָכֶם יֵשְׁבוּ בְּעָרֵיכֶם אֲשֶׁר נָתַתִּי לָכֶם:

[דברים פרק-יב-טז]  רַק הַדָּם לֹא תֹאכֵלוּ עַל הָאָרֶץ תִּשְׁפְּכֶנּוּ כַּמָּיִם:

[דברים פרק-טו-ה]  רַק אִם שָׁמוֹעַ תִּשְׁמַע בְּקוֹל יְהוָה אֱלֹהֶיךָ לִשְׁמֹר לַעֲשׂוֹת אֶת כָּל הַמִּצְוָה הַזֹּאת אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם:

[דברים פרק-טו-כג]  רַק אֶת דָּמוֹ לֹא תֹאכֵל עַל הָאָרֶץ תִּשְׁפְּכֶנּוּ כַּמָּיִם:

[יהושע פרק-כב-ה]  רַק שִׁמְרוּ מְאֹד לַעֲשׂוֹת אֶת הַמִּצְוָה וְאֶת הַתּוֹרָה אֲשֶׁר צִוָּה אֶתְכֶם מֹשֶׁה עֶבֶד יְהוָה לְאַהֲבָה אֶת יְהוָה אֱלֹהֵיכֶם וְלָלֶכֶת בְּכָל דְּרָכָיו וְלִשְׁמֹר מִצְוֹתָיו וּלְדָבְקָה בוֹ וּלְעָבְדוֹ בְּכָל לְבַבְכֶם וּבְכָל נַפְשְׁכֶם:

[שופטים פרק-ג-ב]  רַק לְמַעַן דַּעַת דֹּרוֹת בְּנֵי יִשְׂרָאֵל לְלַמְּדָם מִלְחָמָה רַק אֲשֶׁר לְפָנִים לֹא יְדָעוּם:

[מלכים א פרק-ג-ב]  רַק הָעָם מְזַבְּחִים בַּבָּמוֹת כִּי לֹא נִבְנָה בַיִת לְשֵׁם יְהוָה עַד הַיָּמִים הָהֵם:

[ישעיה פרק-מט-יג]  רָנּוּ שָׁמַיִם וְגִילִי אָרֶץ (יפצחו) וּפִצְחוּ הָרִים רִנָּה כִּי נִחַם יְהוָה עַמּוֹ וַעֲנִיָּו יְרַחֵם:

[ישעיה פרק-נט-ז]  רַגְלֵיהֶם לָרַע יָרֻצוּ וִימַהֲרוּ לִשְׁפֹּךְ דָּם נָקִי מַחְשְׁבוֹתֵיהֶם מַחְשְׁבוֹת אָוֶן שֹׁד וָשֶׁבֶר בִּמְסִלּוֹתָם:

[ירמיה פרק-ד-יב]  רוּחַ מָלֵא מֵאֵלֶּה יָבוֹא לִי עַתָּה גַּם אֲנִי אֲדַבֵּר מִשְׁפָּטִים אוֹתָם:

[ירמיה פרק-ד-כג]  רָאִיתִי אֶת הָאָרֶץ וְהִנֵּה תֹהוּ וָבֹהוּ וְאֶל הַשָּׁמַיִם וְאֵין אוֹרָם:

[ירמיה פרק-נא-ט]  (רפאנו) רִפִּינוּ אֶת בָּבֶל וְלֹא נִרְפָּתָה עִזְבוּהָ וְנֵלֵךְ אִישׁ לְאַרְצוֹ כִּי נָגַע אֶל הַשָּׁמַיִם מִשְׁפָּטָהּ וְנִשָּׂא עַד שְׁחָקִים:

[עמוס פרק-ג-ב]  רַק אֶתְכֶם יָדַעְתִּי מִכֹּל מִשְׁפְּחוֹת הָאֲדָמָה עַל כֵּן אֶפְקֹד עֲלֵיכֶם אֵת כָּל עֲוֹנֹתֵיכֶם:

[מיכה פרק-ז-יד]  רְעֵה עַמְּךָ בְשִׁבְטֶךָ צֹאן נַחֲלָתֶךָ שֹׁכְנִי לְבָדָד יַעַר בְּתוֹךְ כַּרְמֶל יִרְעוּ בָשָׁן וְגִלְעָד כִּימֵי עוֹלָם:

[צפניה פרק-ג-יד]  רָנִּי בַּת צִיּוֹן הָרִיעוּ יִשְׂרָאֵל שִׂמְחִי וְעָלְזִי בְּכָל לֵב בַּת יְרוּשָׁלִָם:

[זכריה פרק-א-ח]  רָאִיתִי הַלַּיְלָה וְהִנֵּה אִישׁ רֹכֵב עַל סוּס אָדֹם וְהוּא עֹמֵד בֵּין הַהֲדַסִּים אֲשֶׁר בַּמְּצֻלָה וְאַחֲרָיו סוּסִים אֲדֻמִּים שְׂרֻקִּים וּלְבָנִים:

[תהילים פרק-סט-לג]  רָאוּ עֲנָוִים יִשְׂמָחוּ דֹּרְשֵׁי אֱלֹהִים וִיחִי לְבַבְכֶם:

[תהילים פרק-קכ-ו]  רַבַּת שָׁכְנָה לָּהּ נַפְשִׁי עִם שׂוֹנֵא שָׁלוֹם:

[תהילים פרק-קכג-ד]  רַבַּת שָׂבְעָה לָּהּ נַפְשֵׁנוּ הַלַּעַג הַשַּׁאֲנַנִּים הַבּוּז לִגְאֵיוֹנִים:

[תהילים פרק-קמה-יט]  רְצוֹן יְרֵאָיו יַעֲשֶׂה וְאֶת שַׁוְעָתָם יִשְׁמַע וְיוֹשִׁיעֵם:

[תהילים פרק-קמט-ו]  רוֹמְמוֹת אֵל בִּגְרוֹנָם וְחֶרֶב פִּיפִיּוֹת בְּיָדָם:

[משלי פרק-יט-כא]  רַבּוֹת מַחֲשָׁבוֹת בְּלֶב אִישׁ וַעֲצַת יְהוָה הִיא תָקוּם:

[משלי פרק-כו-י]  רַב מְחוֹלֵל כֹּל וְשֹׂכֵר כְּסִיל וְשֹׂכֵר עֹבְרִים:

[איוב פרק-כח-יח]  רָאמוֹת וְגָבִישׁ לֹא יִזָּכֵר וּמֶשֶׁךְ חָכְמָה מִפְּנִינִים:

[איוב פרק-מ-יב]  רְאֵה כָל גֵּאֶה הַכְנִיעֵהוּ וַהֲדֹךְ רְשָׁעִים תַּחְתָּם:

[שיר השירים פרק-ז-ו]  רֹאשֵׁךְ עָלַיִךְ כַּכַּרְמֶל וְדַלַּת רֹאשֵׁךְ כָּאַרְגָּמָן מֶלֶךְ אָסוּר בָּרְהָטִים:

[איכה פרק-ד-כ]  רוּחַ אַפֵּינוּ מְשִׁיחַ יְהוָה נִלְכַּד בִּשְׁחִיתוֹתָם אֲשֶׁר אָמַרְנוּ בְּצִלּוֹ נִחְיֶה בַגּוֹיִם:

[דברי הימים ב פרק-כט-לד]  רַק הַכֹּהֲנִים הָיוּ לִמְעָט וְלֹא יָכְלוּ לְהַפְשִׁיט אֶת כָּל הָעֹלוֹת וַיְּחַזְּקוּם אֲחֵיהֶם הַלְוִיִּם עַד כְּלוֹת הַמְּלָאכָה וְעַד יִתְקַדְּשׁוּ הַכֹּהֲנִים כִּי הַלְוִיִּם יִשְׁרֵי לֵבָב לְהִתְקַדֵּשׁ מֵהַכֹּהֲנִים:


ר-נ


[מלכים ב פרק-י-כט]  רַק חֲטָאֵי יָרָבְעָם בֶּן נְבָט אֲשֶׁר הֶחֱטִיא אֶת יִשְׂרָאֵל לֹא סָר יֵהוּא מֵאַחֲרֵיהֶם עֶגְלֵי הַזָּהָב אֲשֶׁר בֵּית אֵל וַאֲשֶׁר בְּדָן:

[מלכים ב פרק-טו-לה]  רַק הַבָּמוֹת לֹא סָרוּ עוֹד הָעָם מְזַבְּחִים וּמְקַטְּרִים בַּבָּמוֹת הוּא בָּנָה אֶת שַׁעַר בֵּית יְהוָה הָעֶלְיוֹן:

[ישעיה פרק-מא-ה]  רָאוּ אִיִּים וְיִירָאוּ קְצוֹת הָאָרֶץ יֶחֱרָדוּ קָרְבוּ וַיֶּאֱתָיוּן:

[ישעיה פרק-מא-כז]  רִאשׁוֹן לְצִיּוֹן הִנֵּה הִנָּם וְלִירוּשָׁלִַם מְבַשֵּׂר אֶתֵּן:

[תהילים פרק-לז-לה]  רָאִיתִי רָשָׁע עָרִיץ וּמִתְעָרֶה כְּאֶזְרָח רַעֲנָן:

[משלי פרק-יט-ו]  רַבִּים יְחַלּוּ פְנֵי נָדִיב וְכָל הָרֵעַ לְאִישׁ מַתָּן:

[משלי פרק-כח-כה]  רְחַב נֶפֶשׁ יְגָרֶה מָדוֹן וּבוֹטֵחַ עַל יְהוָה יְדֻשָּׁן:

[קהלת פרק-ז-כז]  רְאֵה זֶה מָצָאתִי אָמְרָה קֹהֶלֶת אַחַת לְאַחַת לִמְצֹא חֶשְׁבּוֹן:


ר-ע


[דברים פרק-ל-טו]  רְאֵה נָתַתִּי לְפָנֶיךָ הַיּוֹם אֶת הַחַיִּים וְאֶת הַטּוֹב וְאֶת הַמָּוֶת וְאֶת הָרָע:

[יהושע פרק-ח-כז]  רַק הַבְּהֵמָה וּשְׁלַל הָעִיר הַהִיא בָּזְזוּ לָהֶם יִשְׂרָאֵל כִּדְבַר יְהוָה אֲשֶׁר צִוָּה אֶת יְהוֹשֻׁעַ:

[יהושע פרק-יא-יג]  רַק כָּל הֶעָרִים הָעֹמְדוֹת עַל תִּלָּם לֹא שְׂרָפָם יִשְׂרָאֵל זוּלָתִי אֶת חָצוֹר לְבַדָּהּ שָׂרַף יְהוֹשֻׁעַ:

[ישעיה פרק-א-טז]  רַחֲצוּ הִזַּכּוּ הָסִירוּ רֹעַ מַעַלְלֵיכֶם מִנֶּגֶד עֵינָי חִדְלוּ הָרֵעַ:

[ישעיה פרק-מב-כ]  (ראית) רָאוֹת רַבּוֹת וְלֹא תִשְׁמֹר פָּקוֹחַ אָזְנַיִם וְלֹא יִשְׁמָע:

[ירמיה פרק-א-י]  רְאֵה הִפְקַדְתִּיךָ הַיּוֹם הַזֶּה עַל הַגּוֹיִם וְעַל הַמַּמְלָכוֹת לִנְתוֹשׁ וְלִנְתוֹץ וּלְהַאֲבִיד וְלַהֲרוֹס לִבְנוֹת וְלִנְטוֹעַ:

[משלי פרק-טו-כט]  רָחוֹק יְהוָה מֵרְשָׁעִים וּתְפִלַּת צַדִּיקִים יִשְׁמָע:


ר-פ


[תהילים פרק-קז-ה]  רְעֵבִים גַּם צְמֵאִים נַפְשָׁם בָּהֶם תִּתְעַטָּף:


ר-צ


[ישעיה פרק-כד-יט]  רֹעָה הִתְרֹעֲעָה הָאָרֶץ פּוֹר הִתְפּוֹרְרָה אֶרֶץ מוֹט הִתְמוֹטְטָה אָרֶץ:

[קהלת פרק-י-ז]  רָאִיתִי עֲבָדִים עַל סוּסִים וְשָׂרִים הֹלְכִים כַּעֲבָדִים עַל הָאָרֶץ:


ר-ק


[משלי פרק-כז-י]  רֵעֲךָ (ורעה) וְרֵעַ אָבִיךָ אַל תַּעֲזֹב וּבֵית אָחִיךָ אַל תָּבוֹא בְּיוֹם אֵידֶךָ טוֹב שָׁכֵן קָרוֹב מֵאָח רָחוֹק:


ר-ר


[דברים פרק-יב-כג]  רַק חֲזַק לְבִלְתִּי אֲכֹל הַדָּם כִּי הַדָּם הוּא הַנָּפֶשׁ וְלֹא תֹאכַל הַנֶּפֶשׁ עִם הַבָּשָׂר:

[ישעיה פרק-מד-כג]  רָנּוּ שָׁמַיִם כִּי עָשָׂה יְהוָה הָרִיעוּ תַּחְתִּיּוֹת אָרֶץ פִּצְחוּ הָרִים רִנָּה יַעַר וְכָל עֵץ בּוֹ כִּי גָאַל יְהוָה יַעֲקֹב וּבְיִשְׂרָאֵל יִתְפָּאָר:

[חבקוק פרק-א-ה]  רְאוּ בַגּוֹיִם וְהַבִּיטוּ וְהִתַּמְּהוּ תְּמָהוּ כִּי פֹעַל פֹּעֵל בִּימֵיכֶם לֹא תַאֲמִינוּ כִּי יְסֻפָּר:

[תהילים פרק-י-יד]  רָאִתָה כִּי אַתָּה עָמָל וָכַעַס תַּבִּיט לָתֵת בְּיָדֶךָ עָלֶיךָ יַעֲזֹב חֵלֶכָה יָתוֹם אַתָּה הָיִיתָ עוֹזֵר:

[תהילים פרק-מ-ו]  רַבּוֹת עָשִׂיתָ אַתָּה יְהוָה אֱלֹהַי נִפְלְאֹתֶיךָ וּמַחְשְׁבֹתֶיךָ אֵלֵינוּ אֵין עֲרֹךְ אֵלֶיךָ אַגִּידָה וַאֲדַבֵּרָה עָצְמוּ מִסַּפֵּר:

[תהילים פרק-מה-ב]  רָחַשׁ לִבִּי דָּבָר טוֹב אֹמֵר אָנִי מַעֲשַׂי לְמֶלֶךְ לְשׁוֹנִי עֵט סוֹפֵר מָהִיר:

[משלי פרק-י-ד]  רָאשׁ עֹשֶׂה כַף רְמִיָּה וְיַד חָרוּצִים תַּעֲשִׁיר:

[משלי פרק-כה-כג]  רוּחַ צָפוֹן תְּחוֹלֵל גָּשֶׁם וּפָנִים נִזְעָמִים לְשׁוֹן סָתֶר:


ר-ש


[יהושע פרק-כג-ד]  רְאוּ הִפַּלְתִּי לָכֶם אֶת הַגּוֹיִם הַנִּשְׁאָרִים הָאֵלֶּה בְּנַחֲלָה לְשִׁבְטֵיכֶם מִן הַיַּרְדֵּן וְכָל הַגּוֹיִם אֲשֶׁר הִכְרַתִּי וְהַיָּם הַגָּדוֹל מְבוֹא הַשָּׁמֶשׁ:

[שמואל א פרק-ב-ט]  רַגְלֵי (חסידו) חֲסִידָיו יִשְׁמֹר וּרְשָׁעִים בַּחֹשֶׁךְ יִדָּמּוּ כִּי לֹא בְכֹחַ יִגְבַּר אִישׁ:

[תהילים פרק-סח-יח]  רֶכֶב אֱלֹהִים רִבֹּתַיִם אַלְפֵי שִׁנְאָן אֲדֹנָי בָם סִינַי בַּקֹּדֶשׁ:

[תהילים פרק-סח-כה]  רָאוּ הֲלִיכוֹתֶיךָ אֱלֹהִים הֲלִיכוֹת אֵלִי מַלְכִּי בַקֹּדֶשׁ:

[משלי פרק-יד-לה]  רְצוֹן מֶלֶךְ לְעֶבֶד מַשְׂכִּיל וְעֶבְרָתוֹ תִּהְיֶה מֵבִישׁ:

[משלי פרק-כט-כו]  רַבִּים מְבַקְשִׁים פְּנֵי מוֹשֵׁל וּמֵיְהוָה מִשְׁפַּט אִישׁ:

[קהלת פרק-ט-ט]  רְאֵה חַיִּים עִם אִשָּׁה אֲשֶׁר אָהַבְתָּ כָּל יְמֵי חַיֵּי הֶבְלֶךָ אֲשֶׁר נָתַן לְךָ תַּחַת הַשֶּׁמֶשׁ כֹּל יְמֵי הֶבְלֶךָ כִּי הוּא חֶלְקְךָ בַּחַיִּים וּבַעֲמָלְךָ אֲשֶׁר אַתָּה עָמֵל תַּחַת הַשָּׁמֶשׁ:


ר-ת


[דברים פרק-א-כא]  רְאֵה נָתַן יְהוָה אֱלֹהֶיךָ לְפָנֶיךָ אֶת הָאָרֶץ עֲלֵה רֵשׁ כַּאֲשֶׁר דִּבֶּר יְהוָה אֱלֹהֵי אֲבֹתֶיךָ לָךְ אַל תִּירָא וְאַל תֵּחָת:

[מלכים ב פרק-יב-ד]  רַק הַבָּמוֹת לֹא סָרוּ עוֹד הָעָם מְזַבְּחִים וּמְקַטְּרִים בַּבָּמוֹת:

[מלכים ב פרק-יד-ד]  רַק הַבָּמוֹת לֹא סָרוּ עוֹד הָעָם מְזַבְּחִים וּמְקַטְּרִים בַּבָּמוֹת:

[מלכים ב פרק-טו-ד]  רַק הַבָּמוֹת לֹא סָרוּ עוֹד הָעָם מְזַבְּחִים וּמְקַטְּרִים בַּבָּמוֹת:

[ישעיה פרק-יד-טז]  רֹאֶיךָ אֵלֶיךָ יַשְׁגִּיחוּ אֵלֶיךָ יִתְבּוֹנָנוּ הֲזֶה הָאִישׁ מַרְגִּיז הָאָרֶץ מַרְעִישׁ מַמְלָכוֹת:

[יחזקאל פרק-מא-ט]  רֹחַב הַקִּיר אֲשֶׁר לַצֵּלָע אֶל הַחוּץ חָמֵשׁ אַמּוֹת וַאֲשֶׁר מֻנָּח בֵּית צְלָעוֹת אֲשֶׁר לַבָּיִת:

[תהילים פרק-עז-יז]  רָאוּךָ מַּיִם אֱלֹהִים רָאוּךָ מַּיִם יָחִילוּ אַף יִרְגְּזוּ תְהֹמוֹת:

[משלי פרק-יא-יח]  רָשָׁע עֹשֶׂה פְעֻלַּת שָׁקֶר וְזֹרֵעַ צְדָקָה שֶׂכֶר אֱמֶת:

[משלי פרק-כא-ד]  רוּם עֵינַיִם וּרְחַב לֵב נִר רְשָׁעִים חַטָּאת:

[איכה פרק-א-כ]  רְאֵה יְהוָה כִּי צַר לִי מֵעַי חֳמַרְמָרוּ נֶהְפַּךְ לִבִּי בְּקִרְבִּי כִּי מָרוֹ מָרִיתִי מִחוּץ שִׁכְּלָה חֶרֶב בַּבַּיִת כַּמָּוֶת:
