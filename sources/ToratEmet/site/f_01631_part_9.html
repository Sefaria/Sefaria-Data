פרשת מטות


{ב} וַיְדַבֵּר מֹשֶׁה אֶל רָאשֵׁי הַמַּטּוֹת לִבְנֵי יִשְׂרָאֵל לֵאמֹר זֶה הַדָּבָר אֲשֶׁר צִוָּה יְהֹוָֽה:  מ  וַיְדַבֵּר מֹשֶׁה אֶל רָאשֵׁי הַמַּטּוֹת לִבְנֵי יִשְׂרָאֵל לֵאמֹר זֶה הַדָּבָר אֲשֶׁר צִוָּה יְהֹוָֽה:  ת  וּמַלִיל משֶׁה עִם רֵישֵׁי שִׁבְטַּיָא לִבְנֵי יִשְׂרָאֵל לְמֵימָר דֵין פִּתְגָמָא דִי פַקֵיד יְיָ:

{ג} אִישׁ כִּֽי יִדֹּר נֶדֶר לַֽיהֹוָה אֽוֹ הִשָּׁבַע שְׁבֻעָה לֶאְסֹר אִסָּר עַל נַפְשׁוֹ לֹא יַחֵל דְּבָרוֹ כְּכָל הַיֹּצֵא מִפִּיו יַֽעֲשֶֽׂה:  מ  אִישׁ כִּֽי יִדֹּר נֶדֶר לַֽיהֹוָה אֽוֹ הִשָּׁבַע שְׁבֻעָה לֶאְסֹר אִסָּר עַל נַפְשׁוֹ לֹא יַחֵל דְּבָרוֹ כְּכָל הַיֹּצֵא מִפִּיו יַֽעֲשֶֽׂה:  ת  גְבַר אֲרֵי יִדַר נְדַר קֳדָם יְיָ אוֹ יְקַיֵים קְיַם לְמֵיסַר אִסַר עַל נַפְשֵׁיהּ לָא יְבַטֵל פִּתְגָמֵיהּ כְּכָל דִי יִפּוֹק מִפּוּמֵיה יַעֲבֵּד:

{ד} וְאִשָּׁה כִּֽי תִדֹּר נֶדֶר לַֽיהֹוָה וְאָֽסְרָה אִסָּר בְּבֵית אָבִיהָ בִּנְעֻרֶֽיהָ:  מ  וְאִשָּׁה כִּֽי תִדֹּר נֶדֶר לַֽיהֹוָה וְאָֽסְרָה אִסָּר בְּבֵית אָבִיהָ בִּנְעֻרֶֽיהָ:  ת  וְאִתְּתָא אֲרֵי תִדַר נְדַר קֳדָם יְיָ וְתֵיסַר אִסָר בְּבֵית אֲבוּהָא בְּרַבְיוּתָהָא:

{ה} וְשָׁמַע אָבִיהָ אֶת נִדְרָהּ וֶֽאֱסָרָהּ אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ וְהֶֽחֱרִישׁ לָהּ אָבִיהָ וְקָמוּ כָּל נְדָרֶיהָ וְכָל אִסָּר אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ יָקֽוּם:  מ  וְשָׁמַע אָבִיהָ אֶת נִדְרָהּ וֶֽאֱסָרָהּ אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ וְהֶֽחֱרִישׁ לָהּ אָבִיהָ וְקָמוּ כָּל נְדָרֶיהָ וְכָל אִסָּר אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ יָקֽוּם:  ת  וְיִשְׁמַע אֲבוּהָא יָת נִדְרָהּ וְאֵסָרָהּ דִי אֲסָרַת עַל נַפְשָׁהּ וְיִשְׁתּוֹק לָהּ אֲבוּהָא וִיקוּמוּן כָּל נִדְרָהָא וְכָל אֱסָרֵי דִי אֲסָרַת עַל נַפְשָׁהּ יְקוּמוּן:

{ו} וְאִם הֵנִיא אָבִיהָ אֹתָהּ בְּיוֹם שָׁמְעוֹ כָּל נְדָרֶיהָ וֶֽאֱסָרֶיהָ אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ לֹא יָקוּם וַֽיהֹוָה יִֽסְלַח לָהּ כִּֽי הֵנִיא אָבִיהָ אֹתָֽהּ:  מ  וְאִם הֵנִיא אָבִיהָ אֹתָהּ בְּיוֹם שָׁמְעוֹ כָּל נְדָרֶיהָ וֶֽאֱסָרֶיהָ אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ לֹא יָקוּם וַֽיהֹוָה יִֽסְלַח לָהּ כִּֽי הֵנִיא אָבִיהָ אֹתָֽהּ:  ת  וְאִם אַעְדֵי אָבוּהָא יָתַהּ בְּיוֹמָא דִשְׁמַע כָּל נִדְרָהָא וֶאֱסָרָהָא דִי אֲסָרַת עַל נַפְשָׁהּ לָא יְקוּמוּן וּמִן קֳדָם יְיָ יִשְׁתְּבֵק לָהּ אֲרֵי אַעְדִי אֲבוּהָא יָתַהּ:

{ז} וְאִם הָיוֹ תִֽהְיֶה לְאִישׁ וּנְדָרֶיהָ עָלֶיהָ אוֹ מִבְטָא שְׂפָתֶיהָ אֲשֶׁר אָֽסְרָה עַל נַפְשָֽׁהּ:  מ  וְאִם הָיוֹ תִֽהְיֶה לְאִישׁ וּנְדָרֶיהָ עָלֶיהָ אוֹ מִבְטָא שְׂפָתֶיהָ אֲשֶׁר אָֽסְרָה עַל נַפְשָֽׁהּ:  ת  וְאִם מֶהֱוָה תֶהֱוֵי לִגְבַר וְנִדְרָהָא עֲלָהּ אוֹ פֵּירוּשׁ סִפְוָתָהָא דִי אֲסָרַת עַל נַפְשָׁהּ:

{ח} וְשָׁמַע אִישָׁהּ בְּיוֹם שָׁמְעוֹ וְהֶֽחֱרִישׁ לָהּ וְקָמוּ נְדָרֶיהָ וֶֽאֱסָרֶהָ אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ יָקֻֽמוּ:  מ  וְשָׁמַע אִישָׁהּ בְּיוֹם שָׁמְעוֹ וְהֶֽחֱרִישׁ לָהּ וְקָמוּ נְדָרֶיהָ וֶֽאֱסָרֶהָ אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ יָקֻֽמוּ:  ת  וְיִשְׁמַע בַּעֲלַהּ בְּיוֹמָא דִשְׁמַע וְיִשְׁתּוֹק לַהּ וִיקוּמוּן נִדְרָהָא וֶאֱסָרָהָא דִי אֲסָרַת עַל נַפְשַׁהּ יְקוּמוּן:

{ט} וְאִם בְּיוֹם שְׁמֹעַ אִישָׁהּ יָנִיא אוֹתָהּ וְהֵפֵר אֶת נִדְרָהּ אֲשֶׁר עָלֶיהָ וְאֵת מִבְטָא שְׂפָתֶיהָ אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ וַֽיהֹוָה יִֽסְלַֽח לָֽהּ:  מ  וְאִם בְּיוֹם שְׁמֹעַ אִישָׁהּ יָנִיא אוֹתָהּ וְהֵפֵר אֶת נִדְרָהּ אֲשֶׁר עָלֶיהָ וְאֵת מִבְטָא שְׂפָתֶיהָ אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ וַֽיהֹוָה יִֽסְלַֽח לָֽהּ:  ת  וְאִם בְּיוֹמָא דִשְׁמַע בַּעֲלַהּ אַעְדִי יָתַהּ וִיבַטֵּל יָת נִדְרָהָא דִי עֲלַהּ וְיָת פֵּירוּשׁ סִפְוָתָהָא דִי אֲסָרַת עַל נַפְשַׁהּ וּמִן קֳדָם יְיָ יִשְׁתְּבֵק לַהּ:

{י} וְנֵדֶר אַלְמָנָה וּגְרוּשָׁה כֹּל אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ יָקוּם עָלֶֽיהָ:  מ  וְנֵדֶר אַלְמָנָה וּגְרוּשָׁה כֹּל אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ יָקוּם עָלֶֽיהָ:  ת  וּנְדַר אַרְמְלָא וּמְתַרְכָא כֹּל דִי אֲסָרַת עַל נַפְשַׁהּ יְקוּם עֲלַהּ:

{יא} וְאִם בֵּית אִישָׁהּ נָדָרָה אֽוֹ אָסְרָה אִסָּר עַל נַפְשָׁהּ בִּשְׁבֻעָֽה:  מ  וְאִם בֵּית אִישָׁהּ נָדָרָה אֽוֹ אָסְרָה אִסָּר עַל נַפְשָׁהּ בִּשְׁבֻעָֽה:  ת  וְאִם בֵּית בַּעֲלַהּ נְדָרַת אוֹ אֲסָרַת אִסַר עַל נַפְשַׁהּ בְּקִיוּם:

{יב} וְשָׁמַע אִישָׁהּ וְהֶֽחֱרִשׁ לָהּ לֹא הֵנִיא אֹתָהּ וְקָמוּ כָּל נְדָרֶיהָ וְכָל אִסָּר אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ יָקֽוּם:  מ  וְשָׁמַע אִישָׁהּ וְהֶֽחֱרִשׁ לָהּ לֹא הֵנִיא אֹתָהּ וְקָמוּ כָּל נְדָרֶיהָ וְכָל אִסָּר אֲשֶׁר אָֽסְרָה עַל נַפְשָׁהּ יָקֽוּם:  ת  וְיִשְׁמַע בַּעֲלַהּ וְיִשְׁתּוֹק לַהּ לָא אַעֲדֵי יָתַהּ וִיקוּמוּן כָּל נִדְרָהָא וְכָל אֱסָרֵי דִי אֲסָרַת עַל נַפְשַׁהּ יְקוּמוּן:

{יג} וְאִם הָפֵר יָפֵר אֹתָם אִישָׁהּ בְּיוֹם שָׁמְעוֹ כָּל מוֹצָא שְׂפָתֶיהָ לִנְדָרֶיהָ וּלְאִסַּר נַפְשָׁהּ לֹא יָקוּם אִישָׁהּ הֲפֵרָם וַֽיהֹוָה יִֽסְלַח לָֽהּ:  מ  וְאִם הָפֵר יָפֵר אֹתָם אִישָׁהּ בְּיוֹם שָׁמְעוֹ כָּל מוֹצָא שְׂפָתֶיהָ לִנְדָרֶיהָ וּלְאִסַּר נַפְשָׁהּ לֹא יָקוּם אִישָׁהּ הֲפֵרָם וַֽיהֹוָה יִֽסְלַח לָֽהּ:  ת  וְאִם בַּטָלָא יְבַטֵל יָתְהוֹן בַּעֲלַהּ בְּיוֹמָא דִשְׁמַע כָּל אַפָּקוּת סִפְוָתָהָא לְנִדְרָהָא וּלְאִסַר נַפְשַׁהּ לָא יְקוּמוּן בַּעֲלַהּ בְּטֵלִינוּן וּמִן קֳדָם יְיָ יִשְׁתְּבֵק לַהּ:

{יד} כָּל נֵדֶר וְכָל שְׁבֻעַת אִסָּר לְעַנֹּת נָפֶשׁ אִישָׁהּ יְקִימֶנּוּ וְאִישָׁהּ יְפֵרֶֽנּוּ:  מ  כָּל נֵדֶר וְכָל שְׁבֻעַת אִסָּר לְעַנֹּת נָפֶשׁ אִישָׁהּ יְקִימֶנּוּ וְאִישָׁהּ יְפֵרֶֽנּוּ:  ת  כָּל נְדַר וְכָל קִיוּמַת אִסַר לְסַגָפָא נְפָשׁ בַּעֲלַהּ יְקַיְמִינוּן וּבַעֲלַהּ יְבַטְלִינוּן:

{טו} וְאִם הַֽחֲרֵשׁ יַֽחֲרִישׁ לָהּ אִישָׁהּ מִיּוֹם אֶל יוֹם וְהֵקִים אֶת כָּל נְדָרֶיהָ אוֹ אֶת כָּל אֱסָרֶיהָ אֲשֶׁר עָלֶיהָ הֵקִים אֹתָם כִּֽי הֶחֱרִשׁ לָהּ בְּיוֹם שָׁמְעֽוֹ:  מ  וְאִם הַֽחֲרֵשׁ יַֽחֲרִישׁ לָהּ אִישָׁהּ מִיּוֹם אֶל יוֹם וְהֵקִים אֶת כָּל נְדָרֶיהָ אוֹ אֶת כָּל אֱסָרֶיהָ אֲשֶׁר עָלֶיהָ הֵקִים אֹתָם כִּֽי הֶחֱרִשׁ לָהּ בְּיוֹם שָׁמְעֽוֹ:  ת  וְאִם מִשְׁתַּק יִשְׁתּוֹק לַהּ בַּעֲלַהּ מִיוֹם לְיוֹם וִיקַיֵים יָת כָּל נִדְרָהָא אוֹ יָת כָּל אֱסָרָהָא דִי עֲלָהּ קַיֵים יָתְהוֹן אֲרֵי שָׁתֵק לַהּ בְּיוֹמָא דִשְׁמָע:

{טז} וְאִם הָפֵר יָפֵר אֹתָם אַֽחֲרֵי שָׁמְעוֹ וְנָשָׂא אֶת עֲוֹנָֽהּ:  מ  וְאִם הָפֵר יָפֵר אֹתָם אַֽחֲרֵי שָׁמְעוֹ וְנָשָׂא אֶת עֲוֹנָֽהּ:  ת  וְעִם בַּטָלָא יְבַטֵּל יָתְהוֹן בָּתַר דִשְׁמָע וִיקַבַּל יָת חוֹבַהּ:

{יז} אֵלֶּה הַֽחֻקִּים אֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶׁה בֵּין אִישׁ לְאִשְׁתּוֹ בֵּֽין אָב לְבִתּוֹ בִּנְעֻרֶיהָ בֵּית אָבִֽיהָ:  מ  אֵלֶּה הַֽחֻקִּים אֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶׁה בֵּין אִישׁ לְאִשְׁתּוֹ בֵּֽין אָב לְבִתּוֹ בִּנְעֻרֶיהָ בֵּית אָבִֽיהָ:  ת  אִלֵין קְיָמַיָא דִי פַקֵיד יְיָ יָת משֶׁה בֵּין גְבַר לְאִתְּתֵיהּ בֵּין אַבָּא לִבְרַתֵּיהּ בְּרַבְיוּתָהָא בֵּית אָבוּהָא: (פ)




במדבר פרק-לא

{א}  שני  וַיְדַבֵּר יְהֹוָה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהֹוָה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר:

{ב} נְקֹם נִקְמַת בְּנֵי יִשְׂרָאֵל מֵאֵת הַמִּדְיָנִים אַחַר תֵּֽאָסֵף אֶל עַמֶּֽיךָ:  מ  נְקֹם נִקְמַת בְּנֵי יִשְׂרָאֵל מֵאֵת הַמִּדְיָנִים אַחַר תֵּֽאָסֵף אֶל עַמֶּֽיךָ:  ת  אִתְפְּרַע פֻּרְעֲנוּת בְּנֵי יִשְׂרָאֵל מִן מִדְיָנָאֵי בָּתַר כֵּן תִּתְכְּנֵישׁ לְעַמָּךְ:

{ג} וַיְדַבֵּר מֹשֶׁה אֶל הָעָם לֵאמֹר הֵחָֽלְצוּ מֵֽאִתְּכֶם אֲנָשִׁים לַצָּבָא וְיִֽהְיוּ עַל מִדְיָן לָתֵת נִקְמַת יְהֹוָה בְּמִדְיָֽן:  מ  וַיְדַבֵּר מֹשֶׁה אֶל הָעָם לֵאמֹר הֵחָֽלְצוּ מֵֽאִתְּכֶם אֲנָשִׁים לַצָּבָא וְיִֽהְיוּ עַל מִדְיָן לָתֵת נִקְמַת יְהֹוָה בְּמִדְיָֽן:  ת  וּמַלֵיל משֶׁה עִם עַמָא לְמֵימָר זְרִיזוּ מִנְכוֹן גֻבְרִין לְחֵילָא וִיהוֹן עַל מִדְיָן לְמִתַּן פּוּרְעֲנוּת דִין עַמָא דַיְיָ בְּמִדְיָן:

{ד} אֶלֶף לַמַּטֶּה אֶלֶף לַמַּטֶּה לְכֹל מַטּוֹת יִשְׂרָאֵל תִּשְׁלְחוּ לַצָּבָֽא:  מ  אֶלֶף לַמַּטֶּה אֶלֶף לַמַּטֶּה לְכֹל מַטּוֹת יִשְׂרָאֵל תִּשְׁלְחוּ לַצָּבָֽא:  ת  אַלְפָא לְשִׁבְטָא אַלְפָא לְשִׁבְטָּא לְכֹל שִׁבְטַיָא דְיִשְֹׁרָאֵל תְּשַׁלְחוּן לְחֵילָא:

{ה} וַיִּמָּֽסְרוּ מֵֽאַלְפֵי יִשְׂרָאֵל אֶלֶף לַמַּטֶּה שְׁנֵים עָשָׂר אֶלֶף חֲלוּצֵי צָבָֽא:  מ  וַיִּמָּֽסְרוּ מֵֽאַלְפֵי יִשְׂרָאֵל אֶלֶף לַמַּטֶּה שְׁנֵים עָשָׂר אֶלֶף חֲלוּצֵי צָבָֽא:  ת  וְאִתְבְּחָרוּ מֵאַלְפַיָא דְיִֹשְׂרָאֵל אַלְפָא לְשִׁבְטָא תְּרֵי עֲֹשַׂר אַלְפִין מְזָרְזֵי חֵילָא:

{ו} וַיִּשְׁלַח אֹתָם מֹשֶׁה אֶלֶף לַמַּטֶּה לַצָּבָא אֹתָם וְאֶת פִּינְחָס בֶּן אֶלְעָזָר הַכֹּהֵן לַצָּבָא וּכְלֵי הַקֹּדֶשׁ וַחֲצֹֽצְרוֹת הַתְּרוּעָה בְּיָדֽוֹ:  מ  וַיִּשְׁלַח אֹתָם מֹשֶׁה אֶלֶף לַמַּטֶּה לַצָּבָא אֹתָם וְאֶת פִּינְחָס בֶּן אֶלְעָזָר הַכֹּהֵן לַצָּבָא וּכְלֵי הַקֹּדֶשׁ וַחֲצֹֽצְרוֹת הַתְּרוּעָה בְּיָדֽוֹ:  ת  וּשְׁלַח יָתְהוֹן משֶׁה אַלְפָא לְשִׁבְטָא לְחֵילָא יָתְהוֹן וְיָת פִּינְחָס בַּר אֶלְעָזָר כַּהֲנָא לְחֵילָא וּמָנֵי דְקוּדְשָׁא וַחֲצוֹצְרוֹת יַבֶּבְתָּא בִּידֵיהּ:

{ז} וַֽיִּצְבְּאוּ עַל מִדְיָן כַּֽאֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶׁה וַיַּֽהַרְגוּ כָּל זָכָֽר:  מ  וַֽיִּצְבְּאוּ עַל מִדְיָן כַּֽאֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶׁה וַיַּֽהַרְגוּ כָּל זָכָֽר:  ת  וְאִתְחַיָילוּ עַל מִדְיָן כְּמָא דְפַקֵיד יְיָ יָת משֶׁה וּקְטָלוּ כָּל דְכוּרָא:

{ח} וְאֶת מַלְכֵי מִדְיָן הָֽרְגוּ עַל חַלְלֵיהֶם אֶת אֱוִי וְאֶת רֶקֶם וְאֶת צוּר וְאֶת חוּר וְאֶת רֶבַע חֲמֵשֶׁת מַלְכֵי מִדְיָן וְאֵת בִּלְעָם בֶּן בְּעוֹר הָֽרְגוּ בֶּחָֽרֶב:  מ  וְאֶת מַלְכֵי מִדְיָן הָֽרְגוּ עַל חַלְלֵיהֶם אֶת אֱוִי וְאֶת רֶקֶם וְאֶת צוּר וְאֶת חוּר וְאֶת רֶבַע חֲמֵשֶׁת מַלְכֵי מִדְיָן וְאֵת בִּלְעָם בֶּן בְּעוֹר הָֽרְגוּ בֶּחָֽרֶב:  ת  וְיָת מַלְכֵי מִדְיָן קְטָלוּ עַל קְטִילֵיהוֹן יָת אֱוִי וְיָת רֶקֶם וְיָת צוּר וְיָת חוּר וְיָת רֶבַע חַמְשָׁא מַלְכֵי מִדְיָן וְיָת בִּלְעָם בַּר בְּעוֹר קְטָלוּ בְּחַרְבָּא:

{ט} וַיִּשְׁבּוּ בְנֵֽי יִשְׂרָאֵל אֶת נְשֵׁי מִדְיָן וְאֶת טַפָּם וְאֵת כָּל בְּהֶמְתָּם וְאֶת כָּל מִקְנֵהֶם וְאֶת כָּל חֵילָם בָּזָֽזוּ:  מ  וַיִּשְׁבּוּ בְנֵֽי יִשְׂרָאֵל אֶת נְשֵׁי מִדְיָן וְאֶת טַפָּם וְאֵת כָּל בְּהֶמְתָּם וְאֶת כָּל מִקְנֵהֶם וְאֶת כָּל חֵילָם בָּזָֽזוּ:  ת  וּשְׁבוֹ בְנֵי יִשְׂרָאֵל יָת נְשֵׁי מִדְיָן וְיָת טַּפְלְהוֹן וְיָת כָּל בְּעִירְהוֹן וְיָת כָּל גֵיתֵיהוֹן וְיָת כָּל נִכְסֵיהוֹן בְּזוֹ:

{י} וְאֵת כָּל עָֽרֵיהֶם בְּמוֹשְׁבֹתָם וְאֵת כָּל טִֽירֹתָם שָֽׂרְפוּ בָּאֵֽשׁ:  מ  וְאֵת כָּל עָֽרֵיהֶם בְּמוֹשְׁבֹתָם וְאֵת כָּל טִֽירֹתָם שָֽׂרְפוּ בָּאֵֽשׁ:  ת  וְיָת כָּל קִרְוֵיהוֹן בְּמוֹתְבָנֵיהוֹן וְיָת כָּל בֵּית סִגְדַתְהוֹן אוֹקִידוּ בְּנוּרָא:

{יא} וַיִּקְחוּ אֶת כָּל הַשָּׁלָל וְאֵת כָּל הַמַּלְקוֹחַ בָּֽאָדָם וּבַבְּהֵמָֽה:  מ  וַיִּקְחוּ אֶת כָּל הַשָּׁלָל וְאֵת כָּל הַמַּלְקוֹחַ בָּֽאָדָם וּבַבְּהֵמָֽה:  ת  וּנְסִיבוּ יָת כָּל עֲדָאָה וְיָת כָּל דְבַרְתָּא בֶּאֱנָשָׁא וּבִבְעִירָא:

{יב} וַיָּבִאוּ אֶל מֹשֶׁה וְאֶל אֶלְעָזָר הַכֹּהֵן וְאֶל עֲדַת בְּנֵֽי יִשְׂרָאֵל אֶת הַשְּׁבִי וְאֶת הַמַּלְקוֹחַ וְאֶת הַשָּׁלָל אֶל הַֽמַּחֲנֶה אֶל עַֽרְבֹת מוֹאָב אֲשֶׁר עַל יַרְדֵּן יְרֵחֽוֹ:  מ  וַיָּבִאוּ אֶל מֹשֶׁה וְאֶל אֶלְעָזָר הַכֹּהֵן וְאֶל עֲדַת בְּנֵֽי יִשְׂרָאֵל אֶת הַשְּׁבִי וְאֶת הַמַּלְקוֹחַ וְאֶת הַשָּׁלָל אֶל הַֽמַּחֲנֶה אֶל עַֽרְבֹת מוֹאָב אֲשֶׁר עַל יַרְדֵּן יְרֵחֽוֹ:  ת  וְאַיתִיוּ לְוָת משֶׁה וּלְוָת אֶלְעָזָר כַּהֲנָא וּלְוָת כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל יָת שִׁבְיָא וְיָת דְבַרְתָּא וְיָת עֲדָאָה לְמַשְׁרִיתָא לְמֵישְׁרַיָא דְמוֹאָב דִי עַל יַרְדְנָא דִירֵחוֹ: (ס)

{יג}  שלישי - במחוברות שני  וַיֵּצְאוּ מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן וְכָל נְשִׂיאֵי הָֽעֵדָה לִקְרָאתָם אֶל מִחוּץ לַֽמַּחֲנֶֽה:  מ  וַיֵּצְאוּ מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן וְכָל נְשִׂיאֵי הָֽעֵדָה לִקְרָאתָם אֶל מִחוּץ לַֽמַּחֲנֶֽה:  ת  וּנְפָקוּ משֶׁה וְאֶלְעָזָר כַּהֲנָא וְכָל רַבְרְבֵי כְנִשְׁתָּא לְקֳדָמוּתְהוֹן לְמִבָּרָא לְמַשְׁרִיתָא:

{יד} וַיִּקְצֹף מֹשֶׁה עַל פְּקוּדֵי הֶחָיִל שָׂרֵי הָֽאֲלָפִים וְשָׂרֵי הַמֵּאוֹת הַבָּאִים מִצְּבָא הַמִּלְחָמָֽה:  מ  וַיִּקְצֹף מֹשֶׁה עַל פְּקוּדֵי הֶחָיִל שָׂרֵי הָֽאֲלָפִים וְשָׂרֵי הַמֵּאוֹת הַבָּאִים מִצְּבָא הַמִּלְחָמָֽה:  ת  וּרְגֵיז משֶׁה עַל דִמְמַנָן עַל חֵילָא רַבָּנֵי אַלְפִין וְרַבָּנֵי מַאֲוָותָא דְאָתוּ מֵחֵיל קְרָבָא:

{טו} וַיֹּאמֶר אֲלֵיהֶם מֹשֶׁה הַֽחִיִּיתֶם כָּל נְקֵבָֽה:  מ  וַיֹּאמֶר אֲלֵיהֶם מֹשֶׁה הַֽחִיִּיתֶם כָּל נְקֵבָֽה:  ת  וַאֲמַר לְהוֹן משֶׁה הֲקַיֵמְתּוּן כָּל נוּקְבָא:

{טז} הֵן הֵנָּה הָיוּ לִבְנֵי יִשְׂרָאֵל בִּדְבַר בִּלְעָם לִמְסָר מַעַל בַּֽיהֹוָה עַל דְּבַר פְּעוֹר וַתְּהִי הַמַּגֵּפָה בַּֽעֲדַת יְהֹוָֽה:  מ  הֵן הֵנָּה הָיוּ לִבְנֵי יִשְׂרָאֵל בִּדְבַר בִּלְעָם לִמְסָר מַעַל בַּֽיהֹוָה עַל דְּבַר פְּעוֹר וַתְּהִי הַמַּגֵּפָה בַּֽעֲדַת יְהֹוָֽה:  ת  הָא אִינוּן הֲוָאָה לִבְנֵי יִשְׂרָאֵל בַּעֲצַת בִּלְעָם לְשַׁקָרָא שְׁקָר קֳדָם יְיָ עַל עֵסַק פְּעוֹר וַהֲוַת מוֹתָנָא בִּכְנִשְׁתָּא דַיְיָ:

{יז} וְעַתָּה הִרְגוּ כָל זָכָר בַּטָּף וְכָל אִשָּׁה יֹדַעַת אִישׁ לְמִשְׁכַּב זָכָר הֲרֹֽגוּ:  מ  וְעַתָּה הִרְגוּ כָל זָכָר בַּטָּף וְכָל אִשָּׁה יֹדַעַת אִישׁ לְמִשְׁכַּב זָכָר הֲרֹֽגוּ:  ת  וּכְעַן קְטוּלוּ כָּל דְכוּרָא בְּטַפְלָא וְכָל אִתְּתָא דִידַעַת גְבַר לְמִשְׁכַּב דְכוּרָא קְטּוּלוּ:

{יח} וְכֹל הַטַּף בַּנָּשִׁים אֲשֶׁר לֹֽא יָדְעוּ מִשְׁכַּב זָכָר הַֽחֲיוּ לָכֶֽם:  מ  וְכֹל הַטַּף בַּנָּשִׁים אֲשֶׁר לֹֽא יָדְעוּ מִשְׁכַּב זָכָר הַֽחֲיוּ לָכֶֽם:  ת  וְכָל טַפְלָא בִנְשַׁיָא דִי לָא יְדָעוּ מִשְׁכַּב דְכוּרָא קַיְימוּ לְכוֹן:

{יט} וְאַתֶּם חֲנוּ מִחוּץ לַֽמַּחֲנֶה שִׁבְעַת יָמִים כֹּל הֹרֵג נֶפֶשׁ וְכֹל נֹגֵעַ בֶּֽחָלָל תִּֽתְחַטְּאוּ בַּיּוֹם הַשְּׁלִישִׁי וּבַיּוֹם הַשְּׁבִיעִי אַתֶּם וּשְׁבִיכֶֽם:  מ  וְאַתֶּם חֲנוּ מִחוּץ לַֽמַּחֲנֶה שִׁבְעַת יָמִים כֹּל הֹרֵג נֶפֶשׁ וְכֹל נֹגֵעַ בֶּֽחָלָל תִּֽתְחַטְּאוּ בַּיּוֹם הַשְּׁלִישִׁי וּבַיּוֹם הַשְּׁבִיעִי אַתֶּם וּשְׁבִיכֶֽם:  ת  וְאַתּוּן שְׁרוֹ מִבָּרָא לְמַשְׁרִיתָא שַׁבְעָא יוֹמִין כָּל דִי קְטַל נַפְשָׁא וְכֹל דִי קְרַב בְּקָטִילָא תַּדוּן עֲלוֹהִי בְּיוֹמָא תְלִיתָאָה וּבְיוֹמָא שְׁבִיעָאָה אַתּוּן וּשְׁבִיכוֹן:

{כ} וְכָל בֶּגֶד וְכָל כְּלִי עוֹר וְכָל מַֽעֲשֵׂה עִזִּים וְכָל כְּלִי עֵץ תִּתְחַטָּֽאוּ:  מ  וְכָל בֶּגֶד וְכָל כְּלִי עוֹר וְכָל מַֽעֲשֵׂה עִזִּים וְכָל כְּלִי עֵץ תִּתְחַטָּֽאוּ:  ת  וְכָל לְבוּשׁ וְכָל מַאן דִמְשַׁךְ וְכָל עוֹבַד מֵעַזֵי וְכָל מַאן דְעָא תַּדוּן עֲלוֹהִי: (ס)

{כא} וַיֹּאמֶר אֶלְעָזָר הַכֹּהֵן אֶל אַנְשֵׁי הַצָּבָא הַבָּאִים לַמִּלְחָמָה זֹאת חֻקַּת הַתּוֹרָה אֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶֽׁה:  מ  וַיֹּאמֶר אֶלְעָזָר הַכֹּהֵן אֶל אַנְשֵׁי הַצָּבָא הַבָּאִים לַמִּלְחָמָה זֹאת חֻקַּת הַתּוֹרָה אֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶֽׁה:  ת  וַאֲמַר אֶלְעָזָר כַּהֲנָא לְגַבְרֵי חֵילָא דַאֲתוֹ לִקְרָבָא דָא גְזֵירַת אוֹרַיְיתָא דִי פַקֵיד יְיָ יָת משֶׁה:

{כב} אַךְ אֶת הַזָּהָב וְאֶת הַכָּסֶף אֶֽת הַנְּחֹשֶׁת אֶת הַבַּרְזֶל אֶֽת הַבְּדִיל וְאֶת הָֽעֹפָֽרֶת:  מ  אַךְ אֶת הַזָּהָב וְאֶת הַכָּסֶף אֶֽת הַנְּחֹשֶׁת אֶת הַבַּרְזֶל אֶֽת הַבְּדִיל וְאֶת הָֽעֹפָֽרֶת:  ת  בְּרַם יָת דַהֲבָא וְיָת כַסְפָּא יָת נְחָשָׁא יָת פַּרְזְלָא יָת אֲבָצָא וְיָת אֲבָרָא:

{כג} כָּל דָּבָר אֲשֶׁר יָבֹא בָאֵשׁ תַּֽעֲבִירוּ בָאֵשׁ וְטָהֵר אַךְ בְּמֵי נִדָּה יִתְחַטָּא וְכֹל אֲשֶׁר לֹֽא יָבֹא בָּאֵשׁ תַּֽעֲבִירוּ בַמָּֽיִם:  מ  כָּל דָּבָר אֲשֶׁר יָבֹא בָאֵשׁ תַּֽעֲבִירוּ בָאֵשׁ וְטָהֵר אַךְ בְּמֵי נִדָּה יִתְחַטָּא וְכֹל אֲשֶׁר לֹֽא יָבֹא בָּאֵשׁ תַּֽעֲבִירוּ בַמָּֽיִם:  ת  כָּל מִדַעַם דְמִתָּעַל בְּנוּרָא תַּעֲבְרוּנֵיהּ בְּנוּרָא וְיִדְכֵּי בְּרַם בְּמֵי אַדָיוּתָא יִתָּדֵי וְכָל דִי לָא מִתָּעַל בְּנוּרָא תַּעַבְרוּנֵיהּ בְּמַיָא:

{כד} וְכִבַּסְתֶּם בִּגְדֵיכֶם בַּיּוֹם הַשְּׁבִיעִי וּטְהַרְתֶּם וְאַחַר תָּבֹאוּ אֶל הַֽמַּחֲנֶֽה:  מ  וְכִבַּסְתֶּם בִּגְדֵיכֶם בַּיּוֹם הַשְּׁבִיעִי וּטְהַרְתֶּם וְאַחַר תָּבֹאוּ אֶל הַֽמַּחֲנֶֽה:  ת  וּתְחַוְרוּן לְבוּשֵׁיכוֹן בְּיוֹמָא שְׁבִיעָאָה וְתִדְכּוּן וּבָתַר כֵּן תַּעֲלוּן לְמַשְׁרִיתָא: (ס)

{כה}  רביעי  וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וַאֲמַר יְיָ לְמשֶׁה לְמֵימָר:

{כו} שָׂא אֵת רֹאשׁ מַלְקוֹחַ הַשְּׁבִי בָּֽאָדָם וּבַבְּהֵמָה אַתָּה וְאֶלְעָזָר הַכֹּהֵן וְרָאשֵׁי אֲבוֹת הָֽעֵדָֽה:  מ  שָׂא אֵת רֹאשׁ מַלְקוֹחַ הַשְּׁבִי בָּֽאָדָם וּבַבְּהֵמָה אַתָּה וְאֶלְעָזָר הַכֹּהֵן וְרָאשֵׁי אֲבוֹת הָֽעֵדָֽה:  ת  קַבֵּל יָת חוּשְׁבַּן דַבְרַת שִׁבְיָא בֶּאֱנָשָׁא וּבִבְעִירָא אַתְּ וְאֶלְעָזָר כַּהֲנָא וְרֵישֵׁי אֲבָהַת כְּנִשְׁתָּא:

{כז} וְחָצִיתָ אֶת הַמַּלְקוֹחַ בֵּין תֹּֽפְשֵׂי הַמִּלְחָמָה הַיֹּֽצְאִים לַצָּבָא וּבֵין כָּל הָֽעֵדָֽה:  מ  וְחָצִיתָ אֶת הַמַּלְקוֹחַ בֵּין תֹּֽפְשֵׂי הַמִּלְחָמָה הַיֹּֽצְאִים לַצָּבָא וּבֵין כָּל הָֽעֵדָֽה:  ת  וּתְפַלֵג יָת דְבַרְתָּא בֵּין גַבְרֵי מַגִיחֵי קְרָבָא דְנְפָקוּ לְחֵילָא וּבֵין כָּל כְּנִשְׁתָּא:

{כח} וַהֲרֵֽמֹתָ מֶכֶס לַֽיהֹוָה מֵאֵת אַנְשֵׁי הַמִּלְחָמָה הַיֹּֽצְאִים לַצָּבָא אֶחָד נֶפֶשׁ מֵֽחֲמֵשׁ הַמֵּאוֹת מִן הָֽאָדָם וּמִן הַבָּקָר וּמִן הַֽחֲמֹרִים וּמִן הַצֹּֽאן:  מ  וַהֲרֵֽמֹתָ מֶכֶס לַֽיהֹוָה מֵאֵת אַנְשֵׁי הַמִּלְחָמָה הַיֹּֽצְאִים לַצָּבָא אֶחָד נֶפֶשׁ מֵֽחֲמֵשׁ הַמֵּאוֹת מִן הָֽאָדָם וּמִן הַבָּקָר וּמִן הַֽחֲמֹרִים וּמִן הַצֹּֽאן:  ת  וְתַפְרֵשׁ נְסִיבָא קֳדָם יְיָ מִן גַבְרֵי מְגִיחֵי קְרָבָא דִנְפָקוּ לְחֵילָא חָד נַפְשָׁא מֵחֲמֵשׁ מְאָה מִן אֱנָשָׁא וּמִן תּוֹרֵי וּמִן חֲמָרֵי וּמִן עָנָא:

{כט} מִמַּֽחֲצִיתָם תִּקָּחוּ וְנָֽתַתָּה לְאֶלְעָזָר הַכֹּהֵן תְּרוּמַת יְהֹוָֽה:  מ  מִמַּֽחֲצִיתָם תִּקָּחוּ וְנָֽתַתָּה לְאֶלְעָזָר הַכֹּהֵן תְּרוּמַת יְהֹוָֽה:  ת  מִפַּלְגוּתְהוֹן תִּסְבוּן וְתִתֵּן לְאֶלְעָזָר כַּהֲנָא אַפְרָשׁוּתָא קֳדָם יְיָ:

{ל} וּמִמַּֽחֲצִת בְּנֵֽי יִשְׂרָאֵל תִּקַּח אֶחָד אָחֻז מִן הַֽחֲמִשִּׁים מִן הָֽאָדָם מִן הַבָּקָר מִן הַֽחֲמֹרִים וּמִן הַצֹּאן מִכָּל הַבְּהֵמָה וְנָֽתַתָּה אֹתָם לַלְוִיִּם שֹֽׁמְרֵי מִשְׁמֶרֶת מִשְׁכַּן יְהֹוָֽה:  מ  וּמִמַּֽחֲצִת בְּנֵֽי יִשְׂרָאֵל תִּקַּח אֶחָד אָחֻז מִן הַֽחֲמִשִּׁים מִן הָֽאָדָם מִן הַבָּקָר מִן הַֽחֲמֹרִים וּמִן הַצֹּאן מִכָּל הַבְּהֵמָה וְנָֽתַתָּה אֹתָם לַלְוִיִּם שֹֽׁמְרֵי מִשְׁמֶרֶת מִשְׁכַּן יְהֹוָֽה:  ת  וּמִפַּלְגוּת בְּנֵי יִשְׂרָאֵל תִּסַב חָד דְאִתַּחָד מִן חַמְשִׁין מִן אֱנָשָׁא מִן תּוֹרֵי מִן חֲמָרֵי וּמִן עָנָא מִכָּל בְּעִירָא וְתִתֵּן יָתְהוֹן לְלֵוָאֵי נַטְרֵי מַטְרַת מַשְׁכְּנָא דַיְיָ:

{לא} וַיַּעַשׂ מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן כַּֽאֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶֽׁה:  מ  וַיַּעַשׂ מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן כַּֽאֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶֽׁה:  ת  וַעֲבַד משֶׁה וְאֶלְעָזָר כַּהֲנָא כְּמָא דִי פַקֵיד יְיָ יָת משֶׁה:

{לב} וַיְהִי הַמַּלְקוֹחַ יֶתֶר הַבָּז אֲשֶׁר בָּֽזְזוּ עַם הַצָּבָא צֹאן שֵׁשׁ מֵאוֹת אֶלֶף וְשִׁבְעִים אֶלֶף וַֽחֲמֵשֶׁת אֲלָפִֽים:  מ  וַיְהִי הַמַּלְקוֹחַ יֶתֶר הַבָּז אֲשֶׁר בָּֽזְזוּ עַם הַצָּבָא צֹאן שֵׁשׁ מֵאוֹת אֶלֶף וְשִׁבְעִים אֶלֶף וַֽחֲמֵשֶׁת אֲלָפִֽים:  ת  וַהֲוָה דְבַרְתָּא שְׁאָר בִּיזָא דִי בְזוֹ עַמָא דִי נְפָקוּ לְחֵילָא עָנָא שִׁית מְאָה וְשַׁבְעִין וְחַמְשָׁא אַלְפִין:

{לג} וּבָקָר שְׁנַיִם וְשִׁבְעִים אָֽלֶף:  מ  וּבָקָר שְׁנַיִם וְשִׁבְעִים אָֽלֶף:  ת  וְתוֹרֵי שַׁבְעִין וּתְרֵין אַלְפִין:

{לד} וַֽחֲמֹרִים אֶחָד וְשִׁשִּׁים אָֽלֶף:  מ  וַֽחֲמֹרִים אֶחָד וְשִׁשִּׁים אָֽלֶף:  ת  וַחֲמָרֵי שִׁיתִּין וְחָד אַלְפִין:

{לה} וְנֶפֶשׁ אָדָם מִן הַנָּשִׁים אֲשֶׁר לֹֽא יָדְעוּ מִשְׁכַּב זָכָר כָּל נֶפֶשׁ שְׁנַיִם וּשְׁלֹשִׁים אָֽלֶף:  מ  וְנֶפֶשׁ אָדָם מִן הַנָּשִׁים אֲשֶׁר לֹֽא יָדְעוּ מִשְׁכַּב זָכָר כָּל נֶפֶשׁ שְׁנַיִם וּשְׁלֹשִׁים אָֽלֶף:  ת  וְנַפְשָׁא דֶאֱנָשָׁא מִן נְשַׁיָא דִי לָא יְדָעָא מִשְׁכְּבֵי דְכוּרָא כָּל נַפְשָׁתָא תְּלָתִין וּתְרֵין אַלְפִין:

{לו} וַתְּהִי הַֽמֶּחֱצָה חֵלֶק הַיֹּֽצְאִים בַּצָּבָא מִסְפַּר הַצֹּאן שְׁלֹשׁ מֵאוֹת אֶלֶף וּשְׁלֹשִׁים אֶלֶף וְשִׁבְעַת אֲלָפִים וַֽחֲמֵשׁ מֵאֽוֹת:  מ  וַתְּהִי הַֽמֶּחֱצָה חֵלֶק הַיֹּֽצְאִים בַּצָּבָא מִסְפַּר הַצֹּאן שְׁלֹשׁ מֵאוֹת אֶלֶף וּשְׁלֹשִׁים אֶלֶף וְשִׁבְעַת אֲלָפִים וַֽחֲמֵשׁ מֵאֽוֹת:  ת  וַהֲוַת פַלְגוּתָא חוּלַק גוּבְרַיָא דִנְפָקוּ לְחֵילָא מִנְיַן עָנָא תְּלַת מְאָה וּתְלָתִין וּשְׁבַע אַלְפִין וַחֲמֵשׁ מְאָה:

{לז} וַיְהִי הַמֶּכֶס לַֽיהֹוָה מִן הַצֹּאן שֵׁשׁ מֵאוֹת חָמֵשׁ וְשִׁבְעִֽים:  מ  וַיְהִי הַמֶּכֶס לַֽיהֹוָה מִן הַצֹּאן שֵׁשׁ מֵאוֹת חָמֵשׁ וְשִׁבְעִֽים:  ת  וַהֲוָה נְסִיבָא קֳדָם יְיָ מִן עָנָא שִׁית מְאָה שַׁבְעִין וַחֲמֵשׁ:

{לח} וְהַבָּקָר שִׁשָּׁה וּשְׁלֹשִׁים אָלֶף וּמִכְסָם לַיהֹוָה שְׁנַיִם וְשִׁבְעִֽים:  מ  וְהַבָּקָר שִׁשָּׁה וּשְׁלֹשִׁים אָלֶף וּמִכְסָם לַיהֹוָה שְׁנַיִם וְשִׁבְעִֽים:  ת  וְתוֹרֵי תְּלָתִין וְשִׁתָּא אַלְפִין וּנְסֵיבְהוֹן קֳדָם יְיָ שַׁבְעִין וּתְרֵין:

{לט} וַֽחֲמֹרִים שְׁלֹשִׁים אֶלֶף וַֽחֲמֵשׁ מֵאוֹת וּמִכְסָם לַֽיהֹוָה אֶחָד וְשִׁשִּֽׁים:  מ  וַֽחֲמֹרִים שְׁלֹשִׁים אֶלֶף וַֽחֲמֵשׁ מֵאוֹת וּמִכְסָם לַֽיהֹוָה אֶחָד וְשִׁשִּֽׁים:  ת  וַחֲמָרֵי תְּלָתִין אַלְפִין וַחֲמֵשׁ מְאָה וּנְסֵיבְהוֹן קֳדָם יְיָ שִׁתִּין וְחָד:

{מ} וְנֶפֶשׁ אָדָם שִׁשָּׁה עָשָׂר אָלֶף וּמִכְסָם לַֽיהֹוָה שְׁנַיִם וּשְׁלֹשִׁים נָֽפֶשׁ:  מ  וְנֶפֶשׁ אָדָם שִׁשָּׁה עָשָׂר אָלֶף וּמִכְסָם לַֽיהֹוָה שְׁנַיִם וּשְׁלֹשִׁים נָֽפֶשׁ:  ת  וְנַפְשָׁא אֱנָשָׁא שִׁתָּא עֲשַׂר אַלְפִין וּנְסֵבְהוֹן קֳדָם יְיָ תְּלָתִין וּתְרֵין נַפְשִׁין:

{מא} וַיִּתֵּן מֹשֶׁה אֶת מֶכֶס תְּרוּמַת יְהֹוָה לְאֶלְעָזָר הַכֹּהֵן כַּֽאֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶֽׁה:  מ  וַיִּתֵּן מֹשֶׁה אֶת מֶכֶס תְּרוּמַת יְהֹוָה לְאֶלְעָזָר הַכֹּהֵן כַּֽאֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶֽׁה:  ת  וִיהַב משֶׁה יָת נְסִיב אַפְרָשׁוּתָא קֳדָם יְיָ לְאֶלְעָזָר כַּהֲנָא כְּמָא דִי פַקֵיד יְיָ יָת משֶׁה:

{מב}  חמישי  וּמִֽמַּחֲצִית בְּנֵי יִשְׂרָאֵל אֲשֶׁר חָצָה מֹשֶׁה מִן הָֽאֲנָשִׁים הַצֹּֽבְאִֽים:  מ  וּמִֽמַּחֲצִית בְּנֵי יִשְׂרָאֵל אֲשֶׁר חָצָה מֹשֶׁה מִן הָֽאֲנָשִׁים הַצֹּֽבְאִֽים:  ת  וּמִפַּלְגוּת בְּנֵי יִשְׂרָאֵל דִי פְּלַג משֶׁה מִן גוּבְרַיָא דִנְפָקוּ לְחֵילָא:

{מג} וַתְּהִי מֶֽחֱצַת הָֽעֵדָה מִן הַצֹּאן שְׁלֹשׁ מֵאוֹת אֶלֶף וּשְׁלֹשִׁים אֶלֶף שִׁבְעַת אֲלָפִים וַֽחֲמֵשׁ מֵאֽוֹת:  מ  וַתְּהִי מֶֽחֱצַת הָֽעֵדָה מִן הַצֹּאן שְׁלֹשׁ מֵאוֹת אֶלֶף וּשְׁלֹשִׁים אֶלֶף שִׁבְעַת אֲלָפִים וַֽחֲמֵשׁ מֵאֽוֹת:  ת  וַהֲוַת פַּלְגוּת כְּנִשְׁתָּא מִן עָנָא תְּלַת מְאָה וּתְלָתִין וְשַׁבְעָא אַלְפִין וַחֲמֵשׁ מְאָה:

{מד} וּבָקָר שִׁשָּׁה וּשְׁלֹשִׁים אָֽלֶף:  מ  וּבָקָר שִׁשָּׁה וּשְׁלֹשִׁים אָֽלֶף:  ת  וְתוֹרֵי תְּלָתִין וְשִׁתָּא אַלְפִין:

{מה} וַֽחֲמֹרִים שְׁלֹשִׁים אֶלֶף וַֽחֲמֵשׁ מֵאֽוֹת:  מ  וַֽחֲמֹרִים שְׁלֹשִׁים אֶלֶף וַֽחֲמֵשׁ מֵאֽוֹת:  ת  וַחֲמָרֵי תְּלָתִין אַלְפִין וַחֲמֵשׁ מְאָה:

{מו} וְנֶפֶשׁ אָדָם שִׁשָּׁה עָשָׂר אָֽלֶף:  מ  וְנֶפֶשׁ אָדָם שִׁשָּׁה עָשָׂר אָֽלֶף:  ת  וְנַפְשָׁא דְאֱנָשָׁא שִׁתָּא עֲשַׂר אַלְפִין:

{מז} וַיִּקַּח מֹשֶׁה מִמַּֽחֲצִת בְּנֵֽי יִשְׂרָאֵל אֶת הָֽאָחֻז אֶחָד מִן הַֽחֲמִשִּׁים מִן הָֽאָדָם וּמִן הַבְּהֵמָה וַיִּתֵּן אֹתָם לַלְוִיִּם שֹֽׁמְרֵי מִשְׁמֶרֶת מִשְׁכַּן יְהֹוָה כַּֽאֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶֽׁה:  מ  וַיִּקַּח מֹשֶׁה מִמַּֽחֲצִת בְּנֵֽי יִשְׂרָאֵל אֶת הָֽאָחֻז אֶחָד מִן הַֽחֲמִשִּׁים מִן הָֽאָדָם וּמִן הַבְּהֵמָה וַיִּתֵּן אֹתָם לַלְוִיִּם שֹֽׁמְרֵי מִשְׁמֶרֶת מִשְׁכַּן יְהֹוָה כַּֽאֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶֽׁה:  ת  וּנְסֵב משֶׁה מִפַּלְגוּת בְּנֵי יִשְׂרָאֵל יָת דְאִתַּחָד חָד מִן חַמְשִׁין מִן אֱנָשָׁא וּמִן בְּעִירָא וִיהַב יָתְהוֹן לְלֵוָאֵי נַטְּרֵי מַטְרַת מַשְׁכְּנָא דַיְיָ כְּמָא דִי פַקֵיד יְיָ יָת משֶׁה:

{מח} וַֽיִּקְרְבוּ אֶל מֹשֶׁה הַפְּקֻדִים אֲשֶׁר לְאַלְפֵי הַצָּבָא שָׂרֵי הָֽאֲלָפִים וְשָׂרֵי הַמֵּאֽוֹת:  מ  וַֽיִּקְרְבוּ אֶל מֹשֶׁה הַפְּקֻדִים אֲשֶׁר לְאַלְפֵי הַצָּבָא שָׂרֵי הָֽאֲלָפִים וְשָׂרֵי הַמֵּאֽוֹת:  ת  וּקְרִיבוּ לְוָת משֶׁה דִמְמַנָן עַל אַלְפֵי חֵילָא רַבָּנֵי אַלְפִין וְרַבָּנֵי מַאֲוָתָא:

{מט} וַֽיֹּאמְרוּ אֶל מֹשֶׁה עֲבָדֶיךָ נָֽשְׂאוּ אֶת רֹאשׁ אַנְשֵׁי הַמִּלְחָמָה אֲשֶׁר בְּיָדֵנוּ וְלֹֽא נִפְקַד מִמֶּנּוּ אִֽישׁ:  מ  וַֽיֹּאמְרוּ אֶל מֹשֶׁה עֲבָדֶיךָ נָֽשְׂאוּ אֶת רֹאשׁ אַנְשֵׁי הַמִּלְחָמָה אֲשֶׁר בְּיָדֵנוּ וְלֹֽא נִפְקַד מִמֶּנּוּ אִֽישׁ:  ת  וַאֲמָרוּ לְמשֶׁה עַבְדָיךְ קַבִּילוּ יָת חוּשְׁבַּן גַבְרֵי מְגִיחֵי קְרָבָא דִי עִמָנָא וְלָא שְׁגָא מִנָנָא אֱנָשׁ:

{נ} וַנַּקְרֵב אֶת קָרְבַּן יְהֹוָה אִישׁ אֲשֶׁר מָצָא כְלִֽי זָהָב אֶצְעָדָה וְצָמִיד טַבַּעַת עָגִיל וְכוּמָז לְכַפֵּר עַל נַפְשֹׁתֵינוּ לִפְנֵי יְהֹוָֽה:  מ  וַנַּקְרֵב אֶת קָרְבַּן יְהֹוָה אִישׁ אֲשֶׁר מָצָא כְלִֽי זָהָב אֶצְעָדָה וְצָמִיד טַבַּעַת עָגִיל וְכוּמָז לְכַפֵּר עַל נַפְשֹׁתֵינוּ לִפְנֵי יְהֹוָֽה:  ת  וְקָרֶבְנָא יָת קוּרְבָּנָא דַיְיָ גְבַר דְאַשְׁכַּח מַאן דִדְהַב שִׁירִין וְשִׁבְּכִין עִזְקָן קַדָשִׁין וּמָחוֹךְ לְכַפָּרָא עַל נַפְשָׁתָנָא קֳדָם יְיָ:

{נא} וַיִּקַּח מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן אֶת הַזָּהָב מֵֽאִתָּם כֹּל כְּלִי מַֽעֲשֶֽׂה:  מ  וַיִּקַּח מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן אֶת הַזָּהָב מֵֽאִתָּם כֹּל כְּלִי מַֽעֲשֶֽׂה:  ת  וּנְסֵיב משֶׁה וְאֶלְעָזר כַּהֲנָא יָת דַהֲבָא מִנְהוֹן כָּל מַאן דְעוֹבָדָא:

{נב} וַיְהִי כָּל זְהַב הַתְּרוּמָה אֲשֶׁר הֵרִימוּ לַֽיהֹוָה שִׁשָּׁה עָשָׂר אֶלֶף שְׁבַע מֵאוֹת וַֽחֲמִשִּׁים שָׁקֶל מֵאֵת שָׂרֵי הָֽאֲלָפִים וּמֵאֵת שָׂרֵי הַמֵּאֽוֹת:  מ  וַיְהִי כָּל זְהַב הַתְּרוּמָה אֲשֶׁר הֵרִימוּ לַֽיהֹוָה שִׁשָּׁה עָשָׂר אֶלֶף שְׁבַע מֵאוֹת וַֽחֲמִשִּׁים שָׁקֶל מֵאֵת שָׂרֵי הָֽאֲלָפִים וּמֵאֵת שָׂרֵי הַמֵּאֽוֹת:  ת  וַהֲוָה כָּל דְהַב אַפְרָשׁוּתָא דְאַפְרָשׁוּ קֳדָם יְיָ שִׁתָּא עַשֲׂר אַלְפִין שְׁבַע מְאָה וְחַמְשִׁין סִלְעִין מִן רַבָּנֵי אַלְפִין וּמִן רַבָּנֵי מַאֲוָתָא:

{נג} אַנְשֵׁי הַצָּבָא בָּֽזְזוּ אִישׁ לֽוֹ:  מ  אַנְשֵׁי הַצָּבָא בָּֽזְזוּ אִישׁ לֽוֹ:  ת  גַבְרֵי דְחֵילָא בָּזוּ גְבַר לְנַפְשֵׁיהּ:

{נד} וַיִּקַּח מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן אֶת הַזָּהָב מֵאֵת שָׂרֵי הָֽאֲלָפִים וְהַמֵּאוֹת וַיָּבִאוּ אֹתוֹ אֶל אֹהֶל מוֹעֵד זִכָּרוֹן לִבְנֵֽי יִשְׂרָאֵל לִפְנֵי יְהֹוָֽה:  מ  וַיִּקַּח מֹשֶׁה וְאֶלְעָזָר הַכֹּהֵן אֶת הַזָּהָב מֵאֵת שָׂרֵי הָֽאֲלָפִים וְהַמֵּאוֹת וַיָּבִאוּ אֹתוֹ אֶל אֹהֶל מוֹעֵד זִכָּרוֹן לִבְנֵֽי יִשְׂרָאֵל לִפְנֵי יְהֹוָֽה:  ת  וּנְסֵיב משֶׁה וְאֶלְעָזָר כַּהֲנָא יָת דַהֲבָא מִן רַבָּנֵי אַלְפִין וּמַאֲוָתָא וְאַיְתִיאוּ יָתֵיהּ לְמַשְׁכַּן זִמְנָא דָכְרָנָא לִבְנֵי יִשְׂרָאֵל קֳדָם יְיָ: (פ)




במדבר פרק-לב

{א}  שישי - במחוברות שלישי  וּמִקְנֶה רַב הָיָה לִבְנֵי רְאוּבֵן וְלִבְנֵי גָד עָצוּם מְאֹד וַיִּרְאוּ אֶת אֶרֶץ יַעְזֵר וְאֶת אֶרֶץ גִּלְעָד וְהִנֵּה הַמָּקוֹם מְקוֹם מִקְנֶֽה:  מ  וּמִקְנֶה רַב הָיָה לִבְנֵי רְאוּבֵן וְלִבְנֵי גָד עָצוּם מְאֹד וַיִּרְאוּ אֶת אֶרֶץ יַעְזֵר וְאֶת אֶרֶץ גִּלְעָד וְהִנֵּה הַמָּקוֹם מְקוֹם מִקְנֶֽה:  ת  וּבְעִיר סַגִי הֲוָה לִבְנֵי רְאוּבֵן וְלִבְנֵי גָד תַּקֵּיף לַחֲדָא וַחֲזוֹ יָת אֲרַע יַעְזֵר וְיָת אֲרַע גִּלְעָד וְהָא אַתְרָא אֲתַר כְּשַׁר לְבֵית בְּעִיר:

{ב} וַיָּבֹאוּ בְנֵֽי גָד וּבְנֵי רְאוּבֵן וַיֹּֽאמְרוּ אֶל מֹשֶׁה וְאֶל אֶלְעָזָר הַכֹּהֵן וְאֶל נְשִׂיאֵי הָֽעֵדָה לֵאמֹֽר:  מ  וַיָּבֹאוּ בְנֵֽי גָד וּבְנֵי רְאוּבֵן וַיֹּֽאמְרוּ אֶל מֹשֶׁה וְאֶל אֶלְעָזָר הַכֹּהֵן וְאֶל נְשִׂיאֵי הָֽעֵדָה לֵאמֹֽר:  ת  וַאֲתוֹ בְנֵי גָד וּבְנֵי רְאוּבֵן וַאֲמָרוּ לְמשֶׁה וּלְאֶלְעָזָר כַּהֲנָא וּלְרַבְרְבֵי כְּנִשְׁתָּא לְמֵימָר:

{ג} עֲטָרוֹת וְדִיבֹן וְיַעְזֵר וְנִמְרָה וְחֶשְׁבּוֹן וְאֶלְעָלֵה וּשְׂבָם וּנְבוֹ וּבְעֹֽן:  מ  עֲטָרוֹת וְדִיבֹן וְיַעְזֵר וְנִמְרָה וְחֶשְׁבּוֹן וְאֶלְעָלֵה וּשְׂבָם וּנְבוֹ וּבְעֹֽן:  ת  מַכְלֶלְתָּא וּמַלְבֶּשְׁתָּא וְכוּמְרִין וּבֵית נִמְרִין וּבֵית חוּשְׁבָּנָא וּבַעֲלֵי דְבָבָא וְסִימָא וּבֵית קְבוּרְתָּא דְמשֶׁה וּבְעוֹן:

{ד} הָאָרֶץ אֲשֶׁר הִכָּה יְהֹוָה לִפְנֵי עֲדַת יִשְׂרָאֵל אֶרֶץ מִקְנֶה הִוא וְלַֽעֲבָדֶיךָ מִקְנֶֽה:  מ  הָאָרֶץ אֲשֶׁר הִכָּה יְהֹוָה לִפְנֵי עֲדַת יִשְׂרָאֵל אֶרֶץ מִקְנֶה הִוא וְלַֽעֲבָדֶיךָ מִקְנֶֽה:  ת  אַרְעָא דִי מְחָא יְיָ יַת יָתְבָהָא קֳדָם כְּנִשְׁתָּא דְיִשְׂרָאֵל אֲרַע כָּשְׁרָא לְבֵית בְּעִיר הִיא וּלְעַבְדָיךְ אִית בְּעִיר: (ס)

{ה} וַיֹּֽאמְרוּ אִם מָצָאנוּ חֵן בְּעֵינֶיךָ יֻתַּן אֶת הָאָרֶץ הַזֹּאת לַֽעֲבָדֶיךָ לַֽאֲחֻזָּה אַל תַּֽעֲבִרֵנוּ אֶת הַיַּרְדֵּֽן:  מ  וַיֹּֽאמְרוּ אִם מָצָאנוּ חֵן בְּעֵינֶיךָ יֻתַּן אֶת הָאָרֶץ הַזֹּאת לַֽעֲבָדֶיךָ לַֽאֲחֻזָּה אַל תַּֽעֲבִרֵנוּ אֶת הַיַּרְדֵּֽן:  ת  וַאֲמָרוּ אִם אַשְׁכַּחְנָא רַחֲמִין בְּעֵינָיךְ תִּתְיְהִיב יָת אַרְעָא הָדָא לְעַבְדָךְ לְאַחֲסָנָא לָא תְעַבְרִינָנָא יָת יַרְדְנָא:

{ו} וַיֹּאמֶר מֹשֶׁה לִבְנֵי גָד וְלִבְנֵי רְאוּבֵן הַאַֽחֵיכֶם יָבֹאוּ לַמִּלְחָמָה וְאַתֶּם תֵּשְׁבוּ פֹֽה:  מ  וַיֹּאמֶר מֹשֶׁה לִבְנֵי גָד וְלִבְנֵי רְאוּבֵן הַאַֽחֵיכֶם יָבֹאוּ לַמִּלְחָמָה וְאַתֶּם תֵּשְׁבוּ פֹֽה:  ת  וַאֲמַר משֶׁה לִבְנֵי גָד וְלִבְנֵי רְאוּבֵן הַאֲחֵיכוֹן יֵיתוּן לִקְרָבָא וְאַתּוּן תֵּתְבוּן הָכָא:

{ז} וְלָמָּה (תנואון) תְנִיאוּן אֶת לֵב בְּנֵי יִשְׂרָאֵל מֵֽעֲבֹר אֶל הָאָרֶץ אֲשֶׁר נָתַן לָהֶם יְהֹוָֽה:  מ  וְלָמָּה (תנואון) תְנִיאוּן אֶת לֵב בְּנֵי יִשְׂרָאֵל מֵֽעֲבֹר אֶל הָאָרֶץ אֲשֶׁר נָתַן לָהֶם יְהֹוָֽה:  ת  וּלְמָא תוֹנוּן יָת לִבָּא דִבְנֵי יִשְׂרָאֵל מִלְמֶעֱבַר לְאַרְעָא דִיהַב לְהוֹן יְיָ:

{ח} כֹּה עָשׂוּ אֲבֹֽתֵיכֶם בְּשָׁלְחִי אֹתָם מִקָּדֵשׁ בַּרְנֵעַ לִרְאוֹת אֶת הָאָֽרֶץ:  מ  כֹּה עָשׂוּ אֲבֹֽתֵיכֶם בְּשָׁלְחִי אֹתָם מִקָּדֵשׁ בַּרְנֵעַ לִרְאוֹת אֶת הָאָֽרֶץ:  ת  כְּדֵין עֲבָדוּ אֲבָהַתְכוֹן כַּד שְׁלָחִית יָתְהוֹן מֵרְקַם גֵיאָה לְמֶחֱזֵי יָת אַרְעָא:

{ט} וַֽיַּעֲלוּ עַד נַחַל אֶשְׁכּוֹל וַיִּרְאוּ אֶת הָאָרֶץ וַיָּנִיאוּ אֶת לֵב בְּנֵי יִשְׂרָאֵל לְבִלְתִּי בֹא אֶל הָאָרֶץ אֲשֶׁר נָתַן לָהֶם יְהֹוָֽה:  מ  וַֽיַּעֲלוּ עַד נַחַל אֶשְׁכּוֹל וַיִּרְאוּ אֶת הָאָרֶץ וַיָּנִיאוּ אֶת לֵב בְּנֵי יִשְׂרָאֵל לְבִלְתִּי בֹא אֶל הָאָרֶץ אֲשֶׁר נָתַן לָהֶם יְהֹוָֽה:  ת  וּסְלִיקוּ עַד נַחֲלָא דְאַתְכָּלָא וַחֲזוֹ יָת אַרְעָא וְאוֹנִיאוּ יָת לִבָּא דִבְנֵי יִשְׂרָאֵל בְּדִיל דְלָא לְמֵיעַל לְאַרְעָא דִיהַּב לְהוֹן יְיָ:

{י} וַיִּֽחַר אַף יְהֹוָה בַּיּוֹם הַהוּא וַיִּשָּׁבַע לֵאמֹֽר:  מ  וַיִּֽחַר אַף יְהֹוָה בַּיּוֹם הַהוּא וַיִּשָּׁבַע לֵאמֹֽר:  ת  וּתְקֵיף רוּגְזָא דַיְיָ בְּיוֹמָא הַהוּא וְקַיֵים לְמֵימָר:

{יא} אִם יִרְאוּ הָֽאֲנָשִׁים הָֽעֹלִים מִמִּצְרַיִם מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה אֵת הָֽאֲדָמָה אֲשֶׁר נִשְׁבַּעְתִּי לְאַבְרָהָם לְיִצְחָק וּֽלְיַעֲקֹב כִּי לֹֽא מִלְאוּ אַֽחֲרָֽי:  מ  אִם יִרְאוּ הָֽאֲנָשִׁים הָֽעֹלִים מִמִּצְרַיִם מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה אֵת הָֽאֲדָמָה אֲשֶׁר נִשְׁבַּעְתִּי לְאַבְרָהָם לְיִצְחָק וּֽלְיַעֲקֹב כִּי לֹֽא מִלְאוּ אַֽחֲרָֽי:  ת  אִם יֶחֱזוּן גֻבְרַיָא דִסְלִיקוּ מִמִצְרַיִם מִבַּר עַשְׂרִין שְׁנִין וּלְעֵילָא יָת אַרְעָא דִי קַיֵימִית לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב אֲרֵי לָא אַשְׁלִימוּ בָּתַר דַחַלְתִּי:

{יב} בִּלְתִּי כָּלֵב בֶּן יְפֻנֶּה הַקְּנִזִּי וִֽיהוֹשֻׁעַ בִּן נוּן כִּי מִלְאוּ אַֽחֲרֵי יְהֹוָֽה:  מ  בִּלְתִּי כָּלֵב בֶּן יְפֻנֶּה הַקְּנִזִּי וִֽיהוֹשֻׁעַ בִּן נוּן כִּי מִלְאוּ אַֽחֲרֵי יְהֹוָֽה:  ת  אֱלָהֵן כָּלֵב בַּר יְפֻנֶה קְנִזָאָה וִיהוֹשֻׁעַ בַּר נוּן אֲרֵי אַשְׁלִימוּ בָּתַר דַחַלְתָּא דַיְיָ:

{יג} וַיִּֽחַר אַף יְהֹוָה בְּיִשְׂרָאֵל וַיְנִעֵם בַּמִּדְבָּר אַרְבָּעִים שָׁנָה עַד תֹּם כָּל הַדּוֹר הָֽעֹשֶׂה הָרָע בְּעֵינֵי יְהֹוָֽה:  מ  וַיִּֽחַר אַף יְהֹוָה בְּיִשְׂרָאֵל וַיְנִעֵם בַּמִּדְבָּר אַרְבָּעִים שָׁנָה עַד תֹּם כָּל הַדּוֹר הָֽעֹשֶׂה הָרָע בְּעֵינֵי יְהֹוָֽה:  ת  וּתְקֵיף רוּגְזָא דַיְיָ בְּיִשְׂרָאֵל וּטְרָדִנוּן בְּמַדְבְּרָא אַרְבְּעִין שְׁנִין עַד דְסַף כָּל דָרָא דְעָבִיד דְבִישׁ קֳדָם יְיָ:

{יד} וְהִנֵּה קַמְתֶּם תַּחַת אֲבֹתֵיכֶם תַּרְבּוּת אֲנָשִׁים חַטָּאִים לִסְפּוֹת עוֹד עַל חֲרוֹן אַף יְהֹוָה אֶל יִשְׂרָאֵֽל:  מ  וְהִנֵּה קַמְתֶּם תַּחַת אֲבֹתֵיכֶם תַּרְבּוּת אֲנָשִׁים חַטָּאִים לִסְפּוֹת עוֹד עַל חֲרוֹן אַף יְהֹוָה אֶל יִשְׂרָאֵֽל:  ת  וְהָא קַמְתּוּן חֲלַף אֲבָהַתְכוֹן תַּלְמִידֵי גֻבְרַיָא חַיָבַיָא לְאוֹסָפָא עוֹד עַל תְּקוֹף רוּגְזָא דַיְיָ לְיִשְׂרָאֵל:

{טו} כִּי תְשׁוּבֻן מֵֽאַחֲרָיו וְיָסַף עוֹד לְהַנִּיחוֹ בַּמִּדְבָּר וְשִֽׁחַתֶּם לְכָל הָעָם הַזֶּֽה:  מ  כִּי תְשׁוּבֻן מֵֽאַחֲרָיו וְיָסַף עוֹד לְהַנִּיחוֹ בַּמִּדְבָּר וְשִֽׁחַתֶּם לְכָל הָעָם הַזֶּֽה:  ת  אֲרֵי תְתוּבוּן מִבָּתַר דַחַלְתֵּיהּ וְיוֹסִיף עוֹד לְאַחֲרוּתְהוֹן בְּמַדְבְּרָא וּתְחַבְּלוּן לְכָל עַמָא הָדֵין: (ס)

{טז} וַיִּגְּשׁוּ אֵלָיו וַיֹּאמְרוּ גִּדְרֹת צֹאן נִבְנֶה לְמִקְנֵנוּ פֹּה וְעָרִים לְטַפֵּֽנוּ:  מ  וַיִּגְּשׁוּ אֵלָיו וַיֹּאמְרוּ גִּדְרֹת צֹאן נִבְנֶה לְמִקְנֵנוּ פֹּה וְעָרִים לְטַפֵּֽנוּ:  ת  וּקְרִיבוּ לְוָתֵיהּ וַאֲמָרוּ חַטְרִין דְעָאן נִבְנֵי לִבְעִירָנָא הָכָא וְקִרְוִין לְטַפְלָנָא:

{יז} וַֽאֲנַחְנוּ נֵֽחָלֵץ חֻשִׁים לִפְנֵי בְּנֵי יִשְׂרָאֵל עַד אֲשֶׁר אִם הֲבִֽיאֹנֻם אֶל מְקוֹמָם וְיָשַׁב טַפֵּנוּ בְּעָרֵי הַמִּבְצָר מִפְּנֵי יֹֽשְׁבֵי הָאָֽרֶץ:  מ  וַֽאֲנַחְנוּ נֵֽחָלֵץ חֻשִׁים לִפְנֵי בְּנֵי יִשְׂרָאֵל עַד אֲשֶׁר אִם הֲבִֽיאֹנֻם אֶל מְקוֹמָם וְיָשַׁב טַפֵּנוּ בְּעָרֵי הַמִּבְצָר מִפְּנֵי יֹֽשְׁבֵי הָאָֽרֶץ:  ת  וַאֲנַחְנָא נִזְדָרֵיז מַבְעִין קֳדָם בְּנֵי יִשְׂרָאֵל עַד דִי נְעֵלִינוּן לְאַתְרֵהוֹן וְיֵתְבוּן טַפְלָנָא בְּקִרְוִין כְּרִיכִין מִן קֳדָם יַתְבֵי אַרְעָא:

{יח} לֹא נָשׁוּב אֶל בָּתֵּינוּ עַד הִתְנַחֵל בְּנֵי יִשְׂרָאֵל אִישׁ נַֽחֲלָתֽוֹ:  מ  לֹא נָשׁוּב אֶל בָּתֵּינוּ עַד הִתְנַחֵל בְּנֵי יִשְׂרָאֵל אִישׁ נַֽחֲלָתֽוֹ:  ת  לָא נְתוּב לְבֵיתָנָא עַד דְיַחְסְנוּן בְּנֵי יִשְׂרָאֵל גְבַר אַחֲסַנְתֵּיהּ:

{יט} כִּי לֹא נִנְחַל אִתָּם מֵעֵבֶר לַיַּרְדֵּן וָהָלְאָה כִּי בָאָה נַֽחֲלָתֵנוּ אֵלֵינוּ מֵעֵבֶר הַיַּרְדֵּן מִזְרָֽחָה:  מ  כִּי לֹא נִנְחַל אִתָּם מֵעֵבֶר לַיַּרְדֵּן וָהָלְאָה כִּי בָאָה נַֽחֲלָתֵנוּ אֵלֵינוּ מֵעֵבֶר הַיַּרְדֵּן מִזְרָֽחָה:  ת  אֲרֵי לָא נַחֲסִין עִמְהוֹן מֵעִבְרָא לְיַרְדְנָא וּלְהָלָא אֲרֵי קַבֵּילְנָא אַחֲסַנְתָּנָא לָנָא מֵעִבְרָא לְיַרְדְנָא מַדִינְחָא: (פ)

{כ}  שביעי - במחוברות רביעי  וַיֹּאמֶר אֲלֵיהֶם מֹשֶׁה אִֽם תַּעֲשׂוּן אֶת הַדָּבָר הַזֶּה אִם תֵּחָלְצוּ לִפְנֵי יְהֹוָה לַמִּלְחָמָֽה:  מ  וַיֹּאמֶר אֲלֵיהֶם מֹשֶׁה אִֽם תַּעֲשׂוּן אֶת הַדָּבָר הַזֶּה אִם תֵּחָלְצוּ לִפְנֵי יְהֹוָה לַמִּלְחָמָֽה:  ת  וַאֲמַר לְהוֹן משֶׁה אִם תַּעְבְדוּן יָת פִּתְגָמָא הָדֵין אִם תִּזְדָרְזוּן קֳדָם עַמָא דַיְיָ לִקְרָבָא:

{כא} וְעָבַר לָכֶם כָּל חָלוּץ אֶת הַיַּרְדֵּן לִפְנֵי יְהֹוָה עַד הֽוֹרִישׁוֹ אֶת אֹֽיְבָיו מִפָּנָֽיו:  מ  וְעָבַר לָכֶם כָּל חָלוּץ אֶת הַיַּרְדֵּן לִפְנֵי יְהֹוָה עַד הֽוֹרִישׁוֹ אֶת אֹֽיְבָיו מִפָּנָֽיו:  ת  וְיַעְבֵר לְכוֹן כָּל דִמְזָרֵז יָת יַרְדְנָא קֳדָם עַמָא דַיְיָ עַד דִיתָרַךְ יָת בַּעֲלֵי דְבָבוֹהִי מִן קֳדָמוֹהִי:

{כב} וְנִכְבְּשָׁה הָאָרֶץ לִפְנֵי יְהֹוָה וְאַחַר תָּשֻׁבוּ וִֽהְיִיתֶם נְקִיִּם מֵֽיְהֹוָה וּמִיִּשְׂרָאֵל וְהָֽיְתָה הָאָרֶץ הַזֹּאת לָכֶם לַֽאֲחֻזָּה לִפְנֵי יְהֹוָֽה:  מ  וְנִכְבְּשָׁה הָאָרֶץ לִפְנֵי יְהֹוָה וְאַחַר תָּשֻׁבוּ וִֽהְיִיתֶם נְקִיִּם מֵֽיְהֹוָה וּמִיִּשְׂרָאֵל וְהָֽיְתָה הָאָרֶץ הַזֹּאת לָכֶם לַֽאֲחֻזָּה לִפְנֵי יְהֹוָֽה:  ת  וְתִתְכְּבֵשׁ אַרְעָא קֳדָם עַמָא דַיְיָ וּבָתַר כֵּן תְּתוּבוּן וּתְהוֹן זַכָּאִין מִן קֳדָם יְיָ וּמִיִשְׂרָאֵל וּתְהֵי אַרְעָא הָדָא לְכוֹן לְאַחֲסָנָא קֳדָם יְיָ:

{כג} וְאִם לֹא תַֽעֲשׂוּן כֵּן הִנֵּה חֲטָאתֶם לַֽיהֹוָה וּדְעוּ חַטַּאתְכֶם אֲשֶׁר תִּמְצָא אֶתְכֶֽם:  מ  וְאִם לֹא תַֽעֲשׂוּן כֵּן הִנֵּה חֲטָאתֶם לַֽיהֹוָה וּדְעוּ חַטַּאתְכֶם אֲשֶׁר תִּמְצָא אֶתְכֶֽם:  ת  וְאִם לָא תַעְבְדוּן כֵּן הָא חַבְתּוּן קֳדָם יְיָ וְתֵדְעוּן חוֹבַתְכוֹן דְתַשְׁכַּח יָתְכוֹן:

{כד} בְּנֽוּ לָכֶם עָרִים לְטַפְּכֶם וּגְדֵרֹת לְצֹנַֽאֲכֶם וְהַיֹּצֵא מִפִּיכֶם תַּֽעֲשֽׂוּ:  מ  בְּנֽוּ לָכֶם עָרִים לְטַפְּכֶם וּגְדֵרֹת לְצֹנַֽאֲכֶם וְהַיֹּצֵא מִפִּיכֶם תַּֽעֲשֽׂוּ:  ת  בְּנוּ לְכוֹן קִרְוִין לְטַפְלְכוֹן וְחַטְרִין לְעַנְכוֹן וּדְיִפּוֹק מִפּוּמְכוֹן תַּעְבְּדוּן:

{כה} וַיֹּאמֶר בְּנֵי גָד וּבְנֵי רְאוּבֵן אֶל מֹשֶׁה לֵאמֹר עֲבָדֶיךָ יַֽעֲשׂוּ כַּֽאֲשֶׁר אֲדֹנִי מְצַוֶּֽה:  מ  וַיֹּאמֶר בְּנֵי גָד וּבְנֵי רְאוּבֵן אֶל מֹשֶׁה לֵאמֹר עֲבָדֶיךָ יַֽעֲשׂוּ כַּֽאֲשֶׁר אֲדֹנִי מְצַוֶּֽה:  ת  וַאֲמָרוּ בְנֵי גָד וּבְנֵי רְאוּבֵן לְמשֶׁה לְמֵימָר עַבְדָיךְ יַעְבְדוּן כְּמָא דִי רִבּוֹנִי מְפַקֵד:

{כו} טַפֵּנוּ נָשֵׁינוּ מִקְנֵנוּ וְכָל בְּהֶמְתֵּנוּ יִֽהְיוּ שָׁם בְּעָרֵי הַגִּלְעָֽד:  מ  טַפֵּנוּ נָשֵׁינוּ מִקְנֵנוּ וְכָל בְּהֶמְתֵּנוּ יִֽהְיוּ שָׁם בְּעָרֵי הַגִּלְעָֽד:  ת  טַפְלָנָא נְשָׁנָא גֵיתָנָא וְכָל בְּעִירָנָא יְהוֹן תַּמָן בְּקִרְוֵי גִלְעָד:

{כז} וַֽעֲבָדֶיךָ יַֽעַבְרוּ כָּל חֲלוּץ צָבָא לִפְנֵי יְהֹוָה לַמִּלְחָמָה כַּֽאֲשֶׁר אֲדֹנִי דֹּבֵֽר:  מ  וַֽעֲבָדֶיךָ יַֽעַבְרוּ כָּל חֲלוּץ צָבָא לִפְנֵי יְהֹוָה לַמִּלְחָמָה כַּֽאֲשֶׁר אֲדֹנִי דֹּבֵֽר:  ת  וְעַבְדָיךְ יַעְבְּרוּן כָּל מְזָרַז חֵילָא קֳדָם עַמָא דַיְיָ לִקְרָבָא כְּמָא דִי רִבּוֹנִי מְמַלֵל:

{כח} וַיְצַו לָהֶם מֹשֶׁה אֵת אֶלְעָזָר הַכֹּהֵן וְאֵת יְהוֹשֻׁעַ בִּן נוּן וְאֶת רָאשֵׁי אֲבוֹת הַמַּטּוֹת לִבְנֵי יִשְׂרָאֵֽל:  מ  וַיְצַו לָהֶם מֹשֶׁה אֵת אֶלְעָזָר הַכֹּהֵן וְאֵת יְהוֹשֻׁעַ בִּן נוּן וְאֶת רָאשֵׁי אֲבוֹת הַמַּטּוֹת לִבְנֵי יִשְׂרָאֵֽל:  ת  וּפַקֵד לְהוֹן משֶׁה יָת אֶלְעָזָר כַּהֲנָא וְיָת יְהוֹשֻׁעַ בַּר נוּן וְיָת רֵישֵׁי אֲבָהַת שִׁבְטַיָא לִבְנֵי יִשְׂרָאֵל:

{כט} וַיֹּאמֶר מֹשֶׁה אֲלֵהֶם אִם יַֽעַבְרוּ בְנֵי גָד וּבְנֵֽי רְאוּבֵן אִתְּכֶם אֶֽת הַיַּרְדֵּן כָּל חָלוּץ לַמִּלְחָמָה לִפְנֵי יְהֹוָה וְנִכְבְּשָׁה הָאָרֶץ לִפְנֵיכֶם וּנְתַתֶּם לָהֶם אֶת אֶרֶץ הַגִּלְעָד לַֽאֲחֻזָּֽה:  מ  וַיֹּאמֶר מֹשֶׁה אֲלֵהֶם אִם יַֽעַבְרוּ בְנֵי גָד וּבְנֵֽי רְאוּבֵן אִתְּכֶם אֶֽת הַיַּרְדֵּן כָּל חָלוּץ לַמִּלְחָמָה לִפְנֵי יְהֹוָה וְנִכְבְּשָׁה הָאָרֶץ לִפְנֵיכֶם וּנְתַתֶּם לָהֶם אֶת אֶרֶץ הַגִּלְעָד לַֽאֲחֻזָּֽה:  ת  וַאֲמַר משֶׁה לְהוֹן אִם יַעְבְּרוּן בְּנֵי גָד וּבְנֵי רְאוּבֵן עִמְכוֹן יָת יַרְדְנָא כָּל דִמְזָרַז לִקְרָבָא קֳדָם עַמָא דַיְיָ וְתִתְכְּבֵשׁ אַרְעָא קֳדָמֵיכוֹן וְתִתְּנוּן לְהוֹן יָת אַרְעָא דְגִלְעָד לְאַחֲסָנָא:

{ל} וְאִם לֹא יַֽעַבְרוּ חֲלוּצִים אִתְּכֶם וְנֹֽאחֲזוּ בְתֹֽכְכֶם בְּאֶרֶץ כְּנָֽעַן:  מ  וְאִם לֹא יַֽעַבְרוּ חֲלוּצִים אִתְּכֶם וְנֹֽאחֲזוּ בְתֹֽכְכֶם בְּאֶרֶץ כְּנָֽעַן:  ת  וְאִם לָא יַעְבְרוּן מְזָרְזִין עִמְכוֹן וְיַחְסְנוּן בֵּינֵיכוֹן בְּאַרְעָא דִכְנָעַן:

{לא} וַיַּֽעֲנוּ בְנֵי גָד וּבְנֵי רְאוּבֵן לֵאמֹר אֵת אֲשֶׁר דִּבֶּר יְהֹוָה אֶל עֲבָדֶיךָ כֵּן נַֽעֲשֶֽׂה:  מ  וַיַּֽעֲנוּ בְנֵי גָד וּבְנֵי רְאוּבֵן לֵאמֹר אֵת אֲשֶׁר דִּבֶּר יְהֹוָה אֶל עֲבָדֶיךָ כֵּן נַֽעֲשֶֽׂה:  ת  וְאָתִיבוּ בְנֵי גָד וּבְנֵי רְאוּבֵן לְמֵימָר יָת דִי מַלִיל יְיָ לְעַבְדָךְ כֵּן נַעְבֵּד:

{לב} נַחְנוּ נַֽעֲבֹר חֲלוּצִים לִפְנֵי יְהֹוָה אֶרֶץ כְּנָעַן וְאִתָּנוּ אֲחֻזַּת נַֽחֲלָתֵנוּ מֵעֵבֶר לַיַּרְדֵּֽן:  מ  נַחְנוּ נַֽעֲבֹר חֲלוּצִים לִפְנֵי יְהֹוָה אֶרֶץ כְּנָעַן וְאִתָּנוּ אֲחֻזַּת נַֽחֲלָתֵנוּ מֵעֵבֶר לַיַּרְדֵּֽן:  ת  נַחְנָא נַעֲבֵר מְזָרְזִין קֳדָם עַמָא דַיְיָ לְאַרְעָא דִכְנָעַן וְעִמָנָא אֲחוּדַת אַחֲסַנְתָּנָא מֵעִבְרָא לְיַרְדְנָא:

{לג} וַיִּתֵּן לָהֶם מֹשֶׁה לִבְנֵי גָד וְלִבְנֵי רְאוּבֵן וְלַֽחֲצִי שֵׁבֶט מְנַשֶּׁה בֶן יוֹסֵף אֶת מַמְלֶכֶת סִיחֹן מֶלֶךְ הָֽאֱמֹרִי וְאֶת מַמְלֶכֶת עוֹג מֶלֶךְ הַבָּשָׁן הָאָרֶץ לְעָרֶיהָ בִּגְבֻלֹת עָרֵי הָאָרֶץ סָבִֽיב:  מ  וַיִּתֵּן לָהֶם מֹשֶׁה לִבְנֵי גָד וְלִבְנֵי רְאוּבֵן וְלַֽחֲצִי שֵׁבֶט מְנַשֶּׁה בֶן יוֹסֵף אֶת מַמְלֶכֶת סִיחֹן מֶלֶךְ הָֽאֱמֹרִי וְאֶת מַמְלֶכֶת עוֹג מֶלֶךְ הַבָּשָׁן הָאָרֶץ לְעָרֶיהָ בִּגְבֻלֹת עָרֵי הָאָרֶץ סָבִֽיב:  ת  וִיהַב לְהוֹן משֶׁה לִבְנֵי גָד וְלִבְנֵי רְאוּבֵן וּלְפַלְגוּת שִׁבְטָא דִמְנַשֶׁה בַּר יוֹסֵף יָת מַלְכְּוַת סִיחוֹן מַלְכָּא דֶאֱמוֹרָאָה וְיָת מַלְכְּוַת עוֹג מַלְכָּא דְמַתְנָן אַרְעָא לְקִרְוָהָא בִּתְחוּמִין קִרְוֵי אַרְעָא סְחוֹר סְחוֹר:

{לד} וַיִּבְנוּ בְנֵי גָד אֶת דִּיבֹן וְאֶת עֲטָרֹת וְאֵת עֲרֹעֵֽר:  מ  וַיִּבְנוּ בְנֵי גָד אֶת דִּיבֹן וְאֶת עֲטָרֹת וְאֵת עֲרֹעֵֽר:  ת  וּבְנוֹ בְנֵי גָד יָת דִיבוֹן וְיָת עֲטָּרוֹת וְיָת עֲרֹעֵר:

{לה} וְאֶת עַטְרֹת שׁוֹפָן וְאֶת יַעְזֵר וְיָגְבְּהָֽה:  מ  וְאֶת עַטְרֹת שׁוֹפָן וְאֶת יַעְזֵר וְיָגְבְּהָֽה:  ת  וְיָת עַטְרוֹת שׁוֹפָן וְיָת יַעְזֵר וְרָמָתָא:

{לו} וְאֶת בֵּית נִמְרָה וְאֶת בֵּית הָרָן עָרֵי מִבְצָר וְגִדְרֹת צֹֽאן:  מ  וְאֶת בֵּית נִמְרָה וְאֶת בֵּית הָרָן עָרֵי מִבְצָר וְגִדְרֹת צֹֽאן:  ת  וְיָת בֵּית נִמְרָה וְיָת בֵּית הָרָן קִרְוִין כְּרִיכָן וְחַטְרִין דְעָאן:

{לז} וּבְנֵי רְאוּבֵן בָּנוּ אֶת חֶשְׁבּוֹן וְאֶת אֶלְעָלֵא וְאֵת קִרְיָתָֽיִם:  מ  וּבְנֵי רְאוּבֵן בָּנוּ אֶת חֶשְׁבּוֹן וְאֶת אֶלְעָלֵא וְאֵת קִרְיָתָֽיִם:  ת  וּבְנֵי רְאוּבֵן בְּנוֹ יָת חֶשְׁבּוֹן וְיָת אֶלְעָלֵא וְיָת קִרְיָתָיִם:

{לח} וְאֶת נְבוֹ וְאֶת בַּעַל מְעוֹן מֽוּסַבֹּת שֵׁם וְאֶת שִׂבְמָה וַיִּקְרְאוּ בְשֵׁמֹת אֶת שְׁמוֹת הֶֽעָרִים אֲשֶׁר בָּנֽוּ:  מ  וְאֶת נְבוֹ וְאֶת בַּעַל מְעוֹן מֽוּסַבֹּת שֵׁם וְאֶת שִׂבְמָה וַיִּקְרְאוּ בְשֵׁמֹת אֶת שְׁמוֹת הֶֽעָרִים אֲשֶׁר בָּנֽוּ:  ת  וְיָת נְבוֹ וְיָת בַּעַל מְעוֹן מַקְפָן שְׁמָהָן וְיָת שִׂבְמָה וּקְרוֹ בִשְׁמָהָן יָת שְׁמָהַת קִרְוַיָא דִי בְנוֹ:

{לט}  מפטיר  וַיֵּלְכוּ בְּנֵי מָכִיר בֶּן מְנַשֶּׁה גִּלְעָדָה וַֽיִּלְכְּדֻהָ וַיּוֹרֶשׁ אֶת הָֽאֱמֹרִי אֲשֶׁר בָּֽהּ:  מ  וַיֵּלְכוּ בְּנֵי מָכִיר בֶּן מְנַשֶּׁה גִּלְעָדָה וַֽיִּלְכְּדֻהָ וַיּוֹרֶשׁ אֶת הָֽאֱמֹרִי אֲשֶׁר בָּֽהּ:  ת  וַאֲזָלוּ בְּנֵי מָכִיר בַּר מְנַשֶׁה לְגִלְעָד וְכִבְשׁוּהָ וְתָרִיךְ יָת אֱמוֹרָאָה דִי בָּהּ:

{מ} וַיִּתֵּן מֹשֶׁה אֶת הַגִּלְעָד לְמָכִיר בֶּן מְנַשֶּׁה וַיֵּשֶׁב בָּֽהּ:  מ  וַיִּתֵּן מֹשֶׁה אֶת הַגִּלְעָד לְמָכִיר בֶּן מְנַשֶּׁה וַיֵּשֶׁב בָּֽהּ:  ת  וִיהַב משֶׁה יָת גִלְעָד לְמָכִיר בַּר מְנַשֶׁה וִיתֵב בַּהּ:

{מא} וְיָאִיר בֶּן מְנַשֶּׁה הָלַךְ וַיִּלְכֹּד אֶת חַוֹּֽתֵיהֶם וַיִּקְרָא אֶתְהֶן חַוֹּת יָאִֽיר:  מ  וְיָאִיר בֶּן מְנַשֶּׁה הָלַךְ וַיִּלְכֹּד אֶת חַוֹּֽתֵיהֶם וַיִּקְרָא אֶתְהֶן חַוֹּת יָאִֽיר:  ת  וְיָאִיר בַּר מְנַשֶׁה אֲזַל וּכְבַשׁ יָת כַּפְרָנֵיהוֹן וּקְרָא יַתְהֶן כַּפְרָנֵי יָאִיר:

{מב} וְנֹבַח הָלַךְ וַיִּלְכֹּד אֶת קְנָת וְאֶת בְּנֹתֶיהָ וַיִּקְרָא לָה נֹבַח בִּשְׁמֽוֹ:  מ  וְנֹבַח הָלַךְ וַיִּלְכֹּד אֶת קְנָת וְאֶת בְּנֹתֶיהָ וַיִּקְרָא לָה נֹבַח בִּשְׁמֽוֹ:  ת  וְנֹבַח אֲזַל וּכְבַשׁ יָת קְנַת וְיָת כַּפְרָנָהָא וּקְרָא לָה נֹבַח בִּשְׁמֵהּ: (פפפ)
