פרשת מצורע



ויקרא פרק-יד

{א} וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} זֹאת תִּֽהְיֶה תּוֹרַת הַמְּצֹרָע בְּיוֹם טָֽהֳרָתוֹ וְהוּבָא אֶל הַכֹּהֵֽן:  מ  זֹאת תִּֽהְיֶה תּוֹרַת הַמְּצֹרָע בְּיוֹם טָֽהֳרָתוֹ וְהוּבָא אֶל הַכֹּהֵֽן:  ת  דָא תְהֵי אוֹרַיְתָא דִסְגִירָא בְּיוֹמָא דְדָכוּתֵהּ וְיִתֵּתֵי לְוָת כַּהֲנָא:

{ג} וְיָצָא הַכֹּהֵן אֶל מִחוּץ לַֽמַּֽחֲנֶה וְרָאָה הַכֹּהֵן וְהִנֵּה נִרְפָּא נֶֽגַע הַצָּרַעַת מִן הַצָּרֽוּעַ:  מ  וְיָצָא הַכֹּהֵן אֶל מִחוּץ לַֽמַּֽחֲנֶה וְרָאָה הַכֹּהֵן וְהִנֵּה נִרְפָּא נֶֽגַע הַצָּרַעַת מִן הַצָּרֽוּעַ:  ת  וְיִפּוֹק כַּהֲנָא לְמִבָּרָא לְמַשְׁרִיתָא וְיֶחֱזֵי כַהֲנָא וְהָא אִתַּסִי מַכְתַּשׁ סְגִירוּתָא מִן סְגִירָא:

{ד} וְצִוָּה הַכֹּהֵן וְלָקַח לַמִּטַּהֵר שְׁתֵּֽי צִפֳּרִים חַיּוֹת טְהֹרוֹת וְעֵץ אֶרֶז וּשְׁנִי תוֹלַעַת וְאֵזֹֽב:  מ  וְצִוָּה הַכֹּהֵן וְלָקַח לַמִּטַּהֵר שְׁתֵּֽי צִפֳּרִים חַיּוֹת טְהֹרוֹת וְעֵץ אֶרֶז וּשְׁנִי תוֹלַעַת וְאֵזֹֽב:  ת  וִיפַקֵד כַּהֲנָא וְיִסַב לְמִדַכֵּי תַּרְתֵּין צִפֳּרִין חַיִין דָכְיָן וְאָעָא דְאַרְזָא וּצְבַע זְהוֹרִי וְאֵזוֹבָא:

{ה} וְצִוָּה הַכֹּהֵן וְשָׁחַט אֶת הַצִּפּוֹר הָֽאֶחָת אֶל כְּלִי חֶרֶשׂ עַל מַיִם חַיִּֽים:  מ  וְצִוָּה הַכֹּהֵן וְשָׁחַט אֶת הַצִּפּוֹר הָֽאֶחָת אֶל כְּלִי חֶרֶשׂ עַל מַיִם חַיִּֽים:  ת  וִיפַקֵד כַּהֲנָא וְיִכּוֹס יָת צִפְּרָא חֲדָא לְמָאן דַחֲסַף עַל מֵי מַבּוּעַ:

{ו} אֶת הַצִּפֹּר הַֽחַיָּה יִקַּח אֹתָהּ וְאֶת עֵץ הָאֶרֶז וְאֶת שְׁנִי הַתּוֹלַעַת וְאֶת הָֽאֵזֹב וְטָבַל אוֹתָם וְאֵת הַצִּפֹּר הַֽחַיָּה בְּדַם הַצִּפֹּר הַשְּׁחֻטָה עַל הַמַּיִם הַֽחַיִּֽים:  מ  אֶת הַצִּפֹּר הַֽחַיָּה יִקַּח אֹתָהּ וְאֶת עֵץ הָאֶרֶז וְאֶת שְׁנִי הַתּוֹלַעַת וְאֶת הָֽאֵזֹב וְטָבַל אוֹתָם וְאֵת הַצִּפֹּר הַֽחַיָּה בְּדַם הַצִּפֹּר הַשְּׁחֻטָה עַל הַמַּיִם הַֽחַיִּֽים:  ת  יָת צִפְּרָא חַיְתָא יִסַב יָתַהּ וְיָת אָעָא דְאַרְזָא וְיָת צְבַע זְהוֹרִי וְיָת אֵזוֹבָא וְיִטְבּוֹל יָתְהוֹן וְיָת צִפְּרָא חַיְתָא בִּדְמָא דְצִפְּרָא דִנְכִיסָא עַל מֵי מַבּוּעַ:

{ז} וְהִזָּה עַל הַמִּטַּהֵר מִן הַצָּרַעַת שֶׁבַע פְּעָמִים וְטִהֲרוֹ וְשִׁלַּח אֶת הַצִּפֹּר הַֽחַיָּה עַל פְּנֵי הַשָּׂדֶֽה:  מ  וְהִזָּה עַל הַמִּטַּהֵר מִן הַצָּרַעַת שֶׁבַע פְּעָמִים וְטִהֲרוֹ וְשִׁלַּח אֶת הַצִּפֹּר הַֽחַיָּה עַל פְּנֵי הַשָּׂדֶֽה:  ת  וְיַדִי עַל דְמִדַכֵּי מִן סְגִירוּתָא שְׁבַע זִמְנִין וִידַכִּנֵהּ וִישַׁלַח יָת צִפְּרָא חַיְתָא עַל אַפֵּי חַקְלָא:

{ח} וְכִבֶּס הַמִּטַּהֵר אֶת בְּגָדָיו וְגִלַּח אֶת כָּל שְׂעָרוֹ וְרָחַץ בַּמַּיִם וְטָהֵר וְאַחַר יָבוֹא אֶל הַֽמַּֽחֲנֶה וְיָשַׁב מִחוּץ לְאָֽהֳלוֹ שִׁבְעַת יָמִֽים:  מ  וְכִבֶּס הַמִּטַּהֵר אֶת בְּגָדָיו וְגִלַּח אֶת כָּל שְׂעָרוֹ וְרָחַץ בַּמַּיִם וְטָהֵר וְאַחַר יָבוֹא אֶל הַֽמַּֽחֲנֶה וְיָשַׁב מִחוּץ לְאָֽהֳלוֹ שִׁבְעַת יָמִֽים:  ת  וִיצַבַּע דְמִדַכֵּי יָת לְבוּשׁוֹהִי וִיגַלַח יָת כָּל שַׂעֲרֵהּ וְיַסְחֵי בְמַיָא וְיִדְכֵּי וּבָתַר כֵּן יֵעוֹל לְמַשְׁרִיתָא וִיתֵב מִבָּרָא לְמַשְׁכְּנֵהּ שִׁבְעַת יוֹמִין:

{ט} וְהָיָה בַיּוֹם הַשְּׁבִיעִי יְגַלַּח אֶת כָּל שְׂעָרוֹ אֶת רֹאשׁוֹ וְאֶת זְקָנוֹ וְאֵת גַּבֹּת עֵינָיו וְאֶת כָּל שְׂעָרוֹ יְגַלֵּחַ וְכִבֶּס אֶת בְּגָדָיו וְרָחַץ אֶת בְּשָׂרוֹ בַּמַּיִם וְטָהֵֽר:  מ  וְהָיָה בַיּוֹם הַשְּׁבִיעִי יְגַלַּח אֶת כָּל שְׂעָרוֹ אֶת רֹאשׁוֹ וְאֶת זְקָנוֹ וְאֵת גַּבֹּת עֵינָיו וְאֶת כָּל שְׂעָרוֹ יְגַלֵּחַ וְכִבֶּס אֶת בְּגָדָיו וְרָחַץ אֶת בְּשָׂרוֹ בַּמַּיִם וְטָהֵֽר:  ת  וִיהֵי בְיוֹמָא שְׁבִיעָאָה יְגַלַח יָת כָּל שַׂעֲרֵהּ יָת רֵישֵׁהּ וְיָת דִקְנֵהּ וְיָת גְבִינֵי עֵינוֹהִי וְיָת כָּל שַׂעֲרֵהּ יְגַלָח וִיצַבַּע יָת לְבוּשׁוֹהִי וְיַסְחֵי יָת בִּסְרֵהּ בְּמַיָא וְיִדְכֵּי:

{י} וּבַיּוֹם הַשְּׁמִינִי יִקַּח שְׁנֵֽי כְבָשִׂים תְּמִימִם וְכַבְשָׂה אַחַת בַּת שְׁנָתָהּ תְּמִימָה וּשְׁלֹשָׁה עֶשְׂרֹנִים סֹלֶת מִנְחָה בְּלוּלָה בַשֶּׁמֶן וְלֹג אֶחָד שָֽׁמֶן:  מ  וּבַיּוֹם הַשְּׁמִינִי יִקַּח שְׁנֵֽי כְבָשִׂים תְּמִימִם וְכַבְשָׂה אַחַת בַּת שְׁנָתָהּ תְּמִימָה וּשְׁלֹשָׁה עֶשְׂרֹנִים סֹלֶת מִנְחָה בְּלוּלָה בַשֶּׁמֶן וְלֹג אֶחָד שָֽׁמֶן:  ת  וּבְיוֹמָא תְמִינָאָה יִסַב תְּרֵין אִמְרִין שַׁלְמִין וְאִמַרְתָּא חֲדָא בַּת שַׁתָּא שְׁלֶמְתָּא וּתְלָתָא עֶסְרוֹנִין סֻלְתָּא מִנְחָתָא דְפִילָא בִמְשַׁח וְלֻגָא חַד דְמִשְׁחָא:

{יא} וְהֶֽעֱמִיד הַכֹּהֵן הַֽמְטַהֵר אֵת הָאִישׁ הַמִּטַּהֵר וְאֹתָם לִפְנֵי יְהֹוָה פֶּתַח אֹהֶל מוֹעֵֽד:  מ  וְהֶֽעֱמִיד הַכֹּהֵן הַֽמְטַהֵר אֵת הָאִישׁ הַמִּטַּהֵר וְאֹתָם לִפְנֵי יְהֹוָה פֶּתַח אֹהֶל מוֹעֵֽד:  ת  וִיקִים כַּהֲנָא דִמְדַכֵּי יָת גַבְרָא דְמִדַכֵּי וְיָתְהוֹן קֳדָם יְיָ בִּתְרַע מַשְׁכַּן זִמְנָא:

{יב} וְלָקַח הַכֹּהֵן אֶת הַכֶּבֶשׂ הָֽאֶחָד וְהִקְרִיב אֹתוֹ לְאָשָׁם וְאֶת לֹג הַשָּׁמֶן וְהֵנִיף אֹתָם תְּנוּפָה לִפְנֵי יְהוָֹֽה:  מ  וְלָקַח הַכֹּהֵן אֶת הַכֶּבֶשׂ הָֽאֶחָד וְהִקְרִיב אֹתוֹ לְאָשָׁם וְאֶת לֹג הַשָּׁמֶן וְהֵנִיף אֹתָם תְּנוּפָה לִפְנֵי יְהוָֹֽה:  ת  וְיִסַב כַּהֲנָא יָת אִמְרָא חֲדָא וִיקָרֵב יָתֵיהּ לַאֲשָּׁמָא וְיָת לֻגָא דְמִשְׁחָא וִירִים יָתְהוֹן אֲרָמָא קֳדָם יְיָ:

{יג}  שני  וְשָׁחַט אֶת הַכֶּבֶשׂ בִּמְקוֹם אֲשֶׁר יִשְׁחַט אֶת הַֽחַטָּאת וְאֶת הָֽעֹלָה בִּמְקוֹם הַקֹּדֶשׁ כִּי כַּֽחַטָּאת הָֽאָשָׁם הוּא לַכֹּהֵן קֹדֶשׁ קָֽדָשִׁים הֽוּא:  מ  וְשָׁחַט אֶת הַכֶּבֶשׂ בִּמְקוֹם אֲשֶׁר יִשְׁחַט אֶת הַֽחַטָּאת וְאֶת הָֽעֹלָה בִּמְקוֹם הַקֹּדֶשׁ כִּי כַּֽחַטָּאת הָֽאָשָׁם הוּא לַכֹּהֵן קֹדֶשׁ קָֽדָשִׁים הֽוּא:  ת  וְיִכּוֹס יָת אִמְרָא בַּאֲתַר דִי יִכּוֹס יָת חַטָאתָא וְיָת עֲלָתָא בַּאֲתַר קַדִישׁ אֲרֵי כְּחַטָאתָא אֲשָׁמָא הוּא לְכַהֲנָא קֹדֶשׁ קוּדְשִׁין הוּא:

{יד} וְלָקַח הַכֹּהֵן מִדַּם הָֽאָשָׁם וְנָתַן הַכֹּהֵן עַל תְּנוּךְ אֹזֶן הַמִּטַּהֵר הַיְמָנִית וְעַל בֹּהֶן יָדוֹ הַיְמָנִית וְעַל בֹּהֶן רַגְלוֹ הַיְמָנִֽית:  מ  וְלָקַח הַכֹּהֵן מִדַּם הָֽאָשָׁם וְנָתַן הַכֹּהֵן עַל תְּנוּךְ אֹזֶן הַמִּטַּהֵר הַיְמָנִית וְעַל בֹּהֶן יָדוֹ הַיְמָנִית וְעַל בֹּהֶן רַגְלוֹ הַיְמָנִֽית:  ת  וְיִסַב כַּהֲנָא מִדְמָא דַאֲשָׁמָא וְיִתֵּן כַּהֲנָא עַל רוּם אוּדְנָא דְמִדַכֵּי דְיַמִינָא וְעַל אִלְיוֹן יְדֵהּ דְיַמִינָא וְעַל אִלְיוֹן רַגְלֵהּ דְיַמִינָא:

{טו} וְלָקַח הַכֹּהֵן מִלֹּג הַשָּׁמֶן וְיָצַק עַל כַּף הַכֹּהֵן הַשְּׂמָאלִֽית:  מ  וְלָקַח הַכֹּהֵן מִלֹּג הַשָּׁמֶן וְיָצַק עַל כַּף הַכֹּהֵן הַשְּׂמָאלִֽית:  ת  וְיִסַב כַּהֲנָא מִלֻגָא דְמִשְׁחָא וִירִיק עַל יְדָא דְכַהֲנָא דִשְׁמָאלָא:

{טז} וְטָבַל הַכֹּהֵן אֶת אֶצְבָּעוֹ הַיְמָנִית מִן הַשֶּׁמֶן אֲשֶׁר עַל כַּפּוֹ הַשְּׂמָאלִית וְהִזָּה מִן הַשֶּׁמֶן בְּאֶצְבָּעוֹ שֶׁבַע פְּעָמִים לִפְנֵי יְהוָֹֽה:  מ  וְטָבַל הַכֹּהֵן אֶת אֶצְבָּעוֹ הַיְמָנִית מִן הַשֶּׁמֶן אֲשֶׁר עַל כַּפּוֹ הַשְּׂמָאלִית וְהִזָּה מִן הַשֶּׁמֶן בְּאֶצְבָּעוֹ שֶׁבַע פְּעָמִים לִפְנֵי יְהוָֹֽה:  ת  וְיִטְבּוֹל כַּהֲנָא יָת אֶצְבְּעֵהּ דְיַמִינָא מִן מִשְׁחָא דִי עַל יְדֵהּ דִּשְׂמָאלָא וְיַדִי מִן מִשְׁחָא בְּאֶצְבְּעֵהּ שְׁבַע זִמְנִין קֳדָם יְיָ:

{יז} וּמִיֶּתֶר הַשֶּׁמֶן אֲשֶׁר עַל כַּפּוֹ יִתֵּן הַכֹּהֵן עַל תְּנוּךְ אֹזֶן הַמִּטַּהֵר הַיְמָנִית וְעַל בֹּהֶן יָדוֹ הַיְמָנִית וְעַל בֹּהֶן רַגְלוֹ הַיְמָנִית עַל דַּם הָֽאָשָֽׁם:  מ  וּמִיֶּתֶר הַשֶּׁמֶן אֲשֶׁר עַל כַּפּוֹ יִתֵּן הַכֹּהֵן עַל תְּנוּךְ אֹזֶן הַמִּטַּהֵר הַיְמָנִית וְעַל בֹּהֶן יָדוֹ הַיְמָנִית וְעַל בֹּהֶן רַגְלוֹ הַיְמָנִית עַל דַּם הָֽאָשָֽׁם:  ת  וּמִשְׁאָר מִשְׁחָא דִי עַל יְדֵהּ יִתֵּן כַּהֲנָא עַל רוּם אוּדְנָא דְמִדַכֵּי דְיַּמִינָא וְעַל אִלְיוֹן יְדֵהּ דְיַמִינָא וְעַל אִלְיוֹן רַגְלֵהּ דְיַמִינָא עַל דְמָא דַאֲשָׁמָא:

{יח} וְהַנּוֹתָר בַּשֶּׁמֶן אֲשֶׁר עַל כַּף הַכֹּהֵן יִתֵּן עַל רֹאשׁ הַמִּטַּהֵר וְכִפֶּר עָלָיו הַכֹּהֵן לִפְנֵי יְהוָֹֽה:  מ  וְהַנּוֹתָר בַּשֶּׁמֶן אֲשֶׁר עַל כַּף הַכֹּהֵן יִתֵּן עַל רֹאשׁ הַמִּטַּהֵר וְכִפֶּר עָלָיו הַכֹּהֵן לִפְנֵי יְהוָֹֽה:  ת  וּדְאִשְׁתָּאַר בְּמִשְׁחָא דִי עַל יְדָא דְכַהֲנָא יִתֵּן עַל רֵישָׁא דְמִדַכֵּי וִיכַפֵּר עֲלוֹהִי כַּהֲנָא קֳדָם יְיָ:

{יט} וְעָשָׂה הַכֹּהֵן אֶת הַחַטָּאת וְכִפֶּר עַל הַמִּטַּהֵר מִטֻּמְאָתוֹ וְאַחַר יִשְׁחַט אֶת הָֽעֹלָֽה:  מ  וְעָשָׂה הַכֹּהֵן אֶת הַחַטָּאת וְכִפֶּר עַל הַמִּטַּהֵר מִטֻּמְאָתוֹ וְאַחַר יִשְׁחַט אֶת הָֽעֹלָֽה:  ת  וְיַעְבֵּד כַּהֲנָא יָת חַטָאתָא וִיכַפֵּר עַל דְמִדַּכֵּי מִסּוֹבְתֵהּ וּבָתַר כֵּן יִכּוֹס יָת עֲלָתָא:

{כ} וְהֶֽעֱלָה הַכֹּהֵן אֶת הָֽעֹלָה וְאֶת הַמִּנְחָה הַמִּזְבֵּחָה וְכִפֶּר עָלָיו הַכֹּהֵן וְטָהֵֽר:  מ  וְהֶֽעֱלָה הַכֹּהֵן אֶת הָֽעֹלָה וְאֶת הַמִּנְחָה הַמִּזְבֵּחָה וְכִפֶּר עָלָיו הַכֹּהֵן וְטָהֵֽר:  ת  וְיַסֵק כַּהֲנָא יָת עֲלָתָא וְיָת מִנְחָתָא לְמַדְבְּחָא וִיכַפֵּר עֲלוֹהִי כַּהֲנָא וְיִדְכֵּי:

{כא}  שלישי - חמישי במחוברין  וְאִם דַּל הוּא וְאֵין יָדוֹ מַשֶּׂגֶת וְלָקַח כֶּבֶשׂ אֶחָד אָשָׁם לִתְנוּפָה לְכַפֵּר עָלָיו וְעִשָּׂרוֹן סֹלֶת אֶחָד בָּלוּל בַּשֶּׁמֶן לְמִנְחָה וְלֹג שָֽׁמֶן:  מ  וְאִם דַּל הוּא וְאֵין יָדוֹ מַשֶּׂגֶת וְלָקַח כֶּבֶשׂ אֶחָד אָשָׁם לִתְנוּפָה לְכַפֵּר עָלָיו וְעִשָּׂרוֹן סֹלֶת אֶחָד בָּלוּל בַּשֶּׁמֶן לְמִנְחָה וְלֹג שָֽׁמֶן:  ת  וְאִם מִסְכֵּן הוּא וְלֵית יְדֵהּ מַדְבְּקָא וְיִסַּב אִמַר חַד אֲשָׁמָא לַאֲרָמָא לְכַפָּרָא עֲלוֹהִי וְעִסְרוֹנָא סֻלְתָּא חַד דְפִיל בִּמְשַּׁח לְמִנְחָא וְלֻגָא דְמִשְׁחָא: (ס)

{כב} וּשְׁתֵּי תֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה אֲשֶׁר תַּשִּׂיג יָדוֹ וְהָיָה אֶחָד חַטָּאת וְהָֽאֶחָד עֹלָֽה:  מ  וּשְׁתֵּי תֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה אֲשֶׁר תַּשִּׂיג יָדוֹ וְהָיָה אֶחָד חַטָּאת וְהָֽאֶחָד עֹלָֽה:  ת  וְתַרְתֵּין שַּׁפְנִינִין אוֹ תְרֵין בְּנֵי יוֹנָה דִי תַדְבֵּק יְדֵהּ וִיהֵי חַד חַטָאתָא וְחַד עֲלָתָא:

{כג} וְהֵבִיא אֹתָם בַּיּוֹם הַשְּׁמִינִי לְטָֽהֳרָתוֹ אֶל הַכֹּהֵן אֶל פֶּתַח אֹֽהֶל מוֹעֵד לִפְנֵי יְהוָֹֽה:  מ  וְהֵבִיא אֹתָם בַּיּוֹם הַשְּׁמִינִי לְטָֽהֳרָתוֹ אֶל הַכֹּהֵן אֶל פֶּתַח אֹֽהֶל מוֹעֵד לִפְנֵי יְהוָֹֽה:  ת  וְיַיְתִי יָתְהוֹן בְּיוֹמָא תְמִינָאָה לְדַכָּיוּתֵהּ לְוָת כַּהֲנָא לִתְרַע מַשְׁכַּן זִמְנָא קֳדָם יְיָ:

{כד} וְלָקַח הַכֹּהֵן אֶת כֶּבֶשׂ הָֽאָשָׁם וְאֶת לֹג הַשָּׁמֶן וְהֵנִיף אֹתָם הַכֹּהֵן תְּנוּפָה לִפְנֵי יְהוָֹֽה:  מ  וְלָקַח הַכֹּהֵן אֶת כֶּבֶשׂ הָֽאָשָׁם וְאֶת לֹג הַשָּׁמֶן וְהֵנִיף אֹתָם הַכֹּהֵן תְּנוּפָה לִפְנֵי יְהוָֹֽה:  ת  וְיִסַב כַּהֲנָא יָת אִמְרָא דַאֲשָׁמָא וְיָת לֻגָא דְמִשְׁחָא וִירִים יָתְהוֹן כַּהֲנָא אֲרָמָא קֳדָם יְיָ:

{כה} וְשָׁחַט אֶת כֶּבֶשׂ הָֽאָשָׁם וְלָקַח הַכֹּהֵן מִדַּם הָֽאָשָׁם וְנָתַן עַל תְּנוּךְ אֹֽזֶן הַמִּטַּהֵר הַיְמָנִית וְעַל בֹּהֶן יָדוֹ הַיְמָנִית וְעַל בֹּהֶן רַגְלוֹ הַיְמָנִֽית:  מ  וְשָׁחַט אֶת כֶּבֶשׂ הָֽאָשָׁם וְלָקַח הַכֹּהֵן מִדַּם הָֽאָשָׁם וְנָתַן עַל תְּנוּךְ אֹֽזֶן הַמִּטַּהֵר הַיְמָנִית וְעַל בֹּהֶן יָדוֹ הַיְמָנִית וְעַל בֹּהֶן רַגְלוֹ הַיְמָנִֽית:  ת  וְיִכּוֹס יָת אִמְרָא דַאֲשָׁמָא וְיִסַב כַּהֲנָא מִדְמָא דַאֲשָׁמָא וְיִתֵּן עַל רוּם אוּדְנָא דְמִדַכֵּי דְיַמִינָא וְעַל אִלְיוֹן יְדֵהּ דְיַמִינָא וְעַל אִלְיוֹן רַגְלֵהּ דְיַמִינָא:

{כו} וּמִן הַשֶּׁמֶן יִצֹק הַכֹּהֵן עַל כַּף הַכֹּהֵן הַשְּׂמָאלִֽית:  מ  וּמִן הַשֶּׁמֶן יִצֹק הַכֹּהֵן עַל כַּף הַכֹּהֵן הַשְּׂמָאלִֽית:  ת  וּמִן מִשְׁחָא יְרִיק כַּהֲנָא עַל יְדָא דְכַהֲנָא דִשְׂמָאלָא:

{כז} וְהִזָּה הַכֹּהֵן בְּאֶצְבָּעוֹ הַיְמָנִית מִן הַשֶּׁמֶן אֲשֶׁר עַל כַּפּוֹ הַשְּׂמָאלִית שֶׁבַע פְּעָמִים לִפְנֵי יְהוָֹֽה:  מ  וְהִזָּה הַכֹּהֵן בְּאֶצְבָּעוֹ הַיְמָנִית מִן הַשֶּׁמֶן אֲשֶׁר עַל כַּפּוֹ הַשְּׂמָאלִית שֶׁבַע פְּעָמִים לִפְנֵי יְהוָֹֽה:  ת  וְיַדִי כַהֲנָא בְּאֶצְבְּעֵהּ דְיַמִינָא מִן מִשְׁחָא דִי עַל יְדֵהּ דִשְׂמָאלָא שְׁבַע זִמְנִין קֳדָם יְיָ:

{כח} וְנָתַן הַכֹּהֵן מִן הַשֶּׁמֶן אֲשֶׁר עַל כַּפּוֹ עַל תְּנוּךְ אֹזֶן הַמִּטַּהֵר הַיְמָנִית וְעַל בֹּהֶן יָדוֹ הַיְמָנִית וְעַל בֹּהֶן רַגְלוֹ הַיְמָנִית עַל מְקוֹם דַּם הָֽאָשָֽׁם:  מ  וְנָתַן הַכֹּהֵן מִן הַשֶּׁמֶן אֲשֶׁר עַל כַּפּוֹ עַל תְּנוּךְ אֹזֶן הַמִּטַּהֵר הַיְמָנִית וְעַל בֹּהֶן יָדוֹ הַיְמָנִית וְעַל בֹּהֶן רַגְלוֹ הַיְמָנִית עַל מְקוֹם דַּם הָֽאָשָֽׁם:  ת  וְיִתֵּן כַּהֲנָא מִן מִשְׁחָא דִי עַל יְדֵהּ עַל רוּם אוּדְנָא דְמִדַכֵּי דְיַמִינָא וְעַל אִלְיוֹן יְדֵהּ דְיַמִינָא וְעַל אִלְיוֹן רַגְלֵהּ דְיַמִינָא עַל אֲתַר דְמָא דַאֲשָׁמָא:

{כט} וְהַנּוֹתָר מִן הַשֶּׁמֶן אֲשֶׁר עַל כַּף הַכֹּהֵן יִתֵּן עַל רֹאשׁ הַמִּטַּהֵר לְכַפֵּר עָלָיו לִפְנֵי יְהוָֹֽה:  מ  וְהַנּוֹתָר מִן הַשֶּׁמֶן אֲשֶׁר עַל כַּף הַכֹּהֵן יִתֵּן עַל רֹאשׁ הַמִּטַּהֵר לְכַפֵּר עָלָיו לִפְנֵי יְהוָֹֽה:  ת  וּדְיִשְׁתָּאַר מִן מִשְׁחָא דִי עַל יְדָא דְכַהֲנָא יִתֵּן עַל רֵישָׁא דְמִדַכֵּי לְכַפָּרָא עֲלוֹהִי קֳדָם יְיָ:

{ל} וְעָשָׂה אֶת הָֽאֶחָד מִן הַתֹּרִים אוֹ מִן בְּנֵי הַיּוֹנָה מֵֽאֲשֶׁר תַּשִּׂיג יָדֽוֹ:  מ  וְעָשָׂה אֶת הָֽאֶחָד מִן הַתֹּרִים אוֹ מִן בְּנֵי הַיּוֹנָה מֵֽאֲשֶׁר תַּשִּׂיג יָדֽוֹ:  ת  וְיַעְבֵּד יָת חַד מִן שַׁפְנִינַיָא אוֹ מִן בְּנֵי יוֹנָה מִדִי תַדְבֵּק יְדֵהּ:

{לא} אֵת אֲשֶׁר תַּשִּׂיג יָדוֹ אֶת הָֽאֶחָד חַטָּאת וְאֶת הָֽאֶחָד עֹלָה עַל הַמִּנְחָה וְכִפֶּר הַכֹּהֵן עַל הַמִּטַּהֵר לִפְנֵי יְהוָֹֽה:  מ  אֵת אֲשֶׁר תַּשִּׂיג יָדוֹ אֶת הָֽאֶחָד חַטָּאת וְאֶת הָֽאֶחָד עֹלָה עַל הַמִּנְחָה וְכִפֶּר הַכֹּהֵן עַל הַמִּטַּהֵר לִפְנֵי יְהוָֹֽה:  ת  יָת דִי תַדְבֵּק יְדֵהּ יָת חַד חַטָאתָא וְיָת חַד עֲלָתָא עַל מִנְחָתָא וִיכַפֵר כַּהֲנָא עַל דְמִדַכֵּי קֳדָם יְיָ:

{לב} זֹאת תּוֹרַת אֲשֶׁר בּוֹ נֶגַע צָרָעַת אֲשֶׁר לֹֽא תַשִּׂיג יָדוֹ בְּטָֽהֳרָֽתוֹ:  מ  זֹאת תּוֹרַת אֲשֶׁר בּוֹ נֶגַע צָרָעַת אֲשֶׁר לֹֽא תַשִּׂיג יָדוֹ בְּטָֽהֳרָֽתוֹ:  ת  דָא אוֹרַיְתָא דִי בֵהּ מַכְתַּשׁ סְגִירוּ דִי לָא תַדְבֵּק יְדֵהּ בְּדָכוּתֵהּ: (פ)

{לג}  רביעי - שישי במחוברין  וַיְדַבֵּר יְהֹוָה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן לֵאמֹֽר:  מ  וַיְדַבֵּר יְהֹוָה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה וְעִם אַהֲרֹן לְמֵימָר:

{לד} כִּי תָבֹאוּ אֶל אֶרֶץ כְּנַעַן אֲשֶׁר אֲנִי נֹתֵן לָכֶם לַֽאֲחֻזָּה וְנָֽתַתִּי נֶגַע צָרַעַת בְּבֵית אֶרֶץ אֲחֻזַּתְכֶֽם:  מ  כִּי תָבֹאוּ אֶל אֶרֶץ כְּנַעַן אֲשֶׁר אֲנִי נֹתֵן לָכֶם לַֽאֲחֻזָּה וְנָֽתַתִּי נֶגַע צָרַעַת בְּבֵית אֶרֶץ אֲחֻזַּתְכֶֽם:  ת  אֲרֵי תֵעֲלוּן לְאַרְעָא דִכְנַעַן דִי אֲנָא יָהֵב לְכוֹן לְאַחֲסָנָא וְאֶתֵּן מַכְתַּשׁ סְגִירוּ בְּבֵית אַרְעָא אַחֲסַנְתְּכוֹן:

{לה} וּבָא אֲשֶׁר לוֹ הַבַּיִת וְהִגִּיד לַכֹּהֵן לֵאמֹר כְּנֶגַע נִרְאָה לִי בַּבָּֽיִת:  מ  וּבָא אֲשֶׁר לוֹ הַבַּיִת וְהִגִּיד לַכֹּהֵן לֵאמֹר כְּנֶגַע נִרְאָה לִי בַּבָּֽיִת:  ת  וְיֵיתֵי דִי דִילֵהּ בֵּיתָא וִיחַוִי לְכַהֲנָא לְמֵימָר כְּמַכְתָּשָׁא אִתְחֲזִי לִי בְּבֵיתָא:

{לו} וְצִוָּה הַכֹּהֵן וּפִנּוּ אֶת הַבַּיִת בְּטֶרֶם יָבֹא הַכֹּהֵן לִרְאוֹת אֶת הַנֶּגַע וְלֹא יִטְמָא כָּל אֲשֶׁר בַּבָּיִת וְאַחַר כֵּן יָבֹא הַכֹּהֵן לִרְאוֹת אֶת הַבָּֽיִת:  מ  וְצִוָּה הַכֹּהֵן וּפִנּוּ אֶת הַבַּיִת בְּטֶרֶם יָבֹא הַכֹּהֵן לִרְאוֹת אֶת הַנֶּגַע וְלֹא יִטְמָא כָּל אֲשֶׁר בַּבָּיִת וְאַחַר כֵּן יָבֹא הַכֹּהֵן לִרְאוֹת אֶת הַבָּֽיִת:  ת  וִיפַקֵד כַּהֲנָא וִיפַנוּן יָת בֵּיתָא עַד לָא יֵעוֹל כַּהֲנָא לְמֶחֱזֵי יָת מַכְתָּשָׁא וְלָא יִסְתָּאַב כָּל דִי בְּבֵיתָא וּבָתַר כֵּן יֵיעוֹל כַּהֲנָא לְמֶחֱזֵי יָת בֵּיתָא:

{לז} וְרָאָה אֶת הַנֶּגַע וְהִנֵּה הַנֶּגַע בְּקִירֹת הַבַּיִת שְׁקַֽעֲרוּרֹת יְרַקְרַקֹּת אוֹ אֲדַמְדַּמֹּת וּמַרְאֵיהֶן שָׁפָל מִן הַקִּֽיר:  מ  וְרָאָה אֶת הַנֶּגַע וְהִנֵּה הַנֶּגַע בְּקִירֹת הַבַּיִת שְׁקַֽעֲרוּרֹת יְרַקְרַקֹּת אוֹ אֲדַמְדַּמֹּת וּמַרְאֵיהֶן שָׁפָל מִן הַקִּֽיר:  ת  וְיֶחֱזֵי יָת מַכְתָּשָׁא וְהָא מַכְתָּשָׁא בְּכָתְלֵי בֵיתָא פַּחֲתִין יַרְקָן אוֹ סַמְקָן וּמֶחֱזֵיהוֹן מַכִּיךְ מִן כָּתְלָא:

{לח} וְיָצָא הַכֹּהֵן מִן הַבַּיִת אֶל פֶּתַֽח הַבָּיִת וְהִסְגִּיר אֶת הַבַּיִת שִׁבְעַת יָמִֽים:  מ  וְיָצָא הַכֹּהֵן מִן הַבַּיִת אֶל פֶּתַֽח הַבָּיִת וְהִסְגִּיר אֶת הַבַּיִת שִׁבְעַת יָמִֽים:  ת  וְיִפּוֹק כַּהֲנָא מִן בֵּיתָא לִתְרַע בֵּיתָא וְיַסְגַר יָת בֵּיתָא שִׁבְעָא יוֹמִין:

{לט} וְשָׁב הַכֹּהֵן בַּיּוֹם הַשְּׁבִיעִי וְרָאָה וְהִנֵּה פָּשָׂה הַנֶּגַע בְּקִירֹת הַבָּֽיִת:  מ  וְשָׁב הַכֹּהֵן בַּיּוֹם הַשְּׁבִיעִי וְרָאָה וְהִנֵּה פָּשָׂה הַנֶּגַע בְּקִירֹת הַבָּֽיִת:  ת  וִיתוּב כַּהֲנָא בְּיוֹמָא שְׁבִיעָאָה וְיֶחֱזֵי וְהָא אוֹסֵף מַכְתָּשָׁא בּכָתלֵי בֵיתָא:

{מ} וְצִוָּה הַכֹּהֵן וְחִלְּצוּ אֶת הָאֲבָנִים אֲשֶׁר בָּהֵן הַנָּגַע וְהִשְׁלִיכוּ אֶתְהֶן אֶל מִחוּץ לָעִיר אֶל מָקוֹם טָמֵֽא:  מ  וְצִוָּה הַכֹּהֵן וְחִלְּצוּ אֶת הָאֲבָנִים אֲשֶׁר בָּהֵן הַנָּגַע וְהִשְׁלִיכוּ אֶתְהֶן אֶל מִחוּץ לָעִיר אֶל מָקוֹם טָמֵֽא:  ת  וִיפַקֵד כֲּהֲנָא וִישַׁלְפוּן יָת אַבְנַיָא דִי בְהוֹן מַכְתָּשָׁא וְיִרְמוּן יָתְהוֹן לְמִבָּרָא לְקַרְתָּא לַאַתַר מְסָאָב:

{מא} וְאֶת הַבַּיִת יַקְצִעַ מִבַּיִת סָבִיב וְשָֽׁפְכוּ אֶת הֶֽעָפָר אֲשֶׁר הִקְצוּ אֶל מִחוּץ לָעִיר אֶל מָקוֹם טָמֵֽא:  מ  וְאֶת הַבַּיִת יַקְצִעַ מִבַּיִת סָבִיב וְשָֽׁפְכוּ אֶת הֶֽעָפָר אֲשֶׁר הִקְצוּ אֶל מִחוּץ לָעִיר אֶל מָקוֹם טָמֵֽא:  ת  וְיָת בֵּיתָא יְקַלְפוּן מִגָיו סְחוֹר סְחוֹר וְיִרְמוּן יָת עַפְרָא דִי קַלִיפוּ לְמִבָּרָא לְקַרְתָּא לַאֲתַר מְסָאָב:

{מב} וְלָֽקְחוּ אֲבָנִים אֲחֵרוֹת וְהֵבִיאוּ אֶל תַּחַת הָֽאֲבָנִים וְעָפָר אַחֵר יִקַּח וְטָח אֶת הַבָּֽיִת:  מ  וְלָֽקְחוּ אֲבָנִים אֲחֵרוֹת וְהֵבִיאוּ אֶל תַּחַת הָֽאֲבָנִים וְעָפָר אַחֵר יִקַּח וְטָח אֶת הַבָּֽיִת:  ת  וְיִסְבוּן אַבְנַיָא אָחֳרָנִין וְיַיְתוּן בַּאֲתַר אַבְנַיָא וַעֲפַר אָחֳרָן יִסַב וִישׁוּעַ יָת בֵּיתָא:

{מג} וְאִם יָשׁוּב הַנֶּגַע וּפָרַח בַּבַּיִת אַחַר חִלֵּץ אֶת הָֽאֲבָנִים וְאַֽחֲרֵי הִקְצוֹת אֶת הַבַּיִת וְאַֽחֲרֵי הִטּֽוֹחַ:  מ  וְאִם יָשׁוּב הַנֶּגַע וּפָרַח בַּבַּיִת אַחַר חִלֵּץ אֶת הָֽאֲבָנִים וְאַֽחֲרֵי הִקְצוֹת אֶת הַבַּיִת וְאַֽחֲרֵי הִטּֽוֹחַ:  ת  וְאִם יְתוּב מַכְתָּשָׁא וְיִסְגֵי בְּבֵיתָא בָּתַר דְשַׁלִיפוּ יָת אַבְנַיָא וּבָתַר דְקַלִיפוּ יָת בֵּיתָא וּבָתַּר דְאִתְּשָׁע:

{מד} וּבָא הַכֹּהֵן וְרָאָה וְהִנֵּה פָּשָׂה הַנֶּגַע בַּבָּיִת צָרַעַת מַמְאֶרֶת הִוא בַּבַּיִת טָמֵא הֽוּא:  מ  וּבָא הַכֹּהֵן וְרָאָה וְהִנֵּה פָּשָׂה הַנֶּגַע בַּבָּיִת צָרַעַת מַמְאֶרֶת הִוא בַּבַּיִת טָמֵא הֽוּא:  ת  וְיֵיעוֹל כַּהֲנָא וְיֶחֱזֵי וְהָא אוֹסֵף מַכְתָּשָׁא בְּבֵיתָא סְגִירוּת מְחַסְרָא הִיא בְּבֵיתָא מְסָאָב הוּא:

{מה} וְנָתַץ אֶת הַבַּיִת אֶת אֲבָנָיו וְאֶת עֵצָיו וְאֵת כָּל עֲפַר הַבָּיִת וְהוֹצִיא אֶל מִחוּץ לָעִיר אֶל מָקוֹם טָמֵֽא:  מ  וְנָתַץ אֶת הַבַּיִת אֶת אֲבָנָיו וְאֶת עֵצָיו וְאֵת כָּל עֲפַר הַבָּיִת וְהוֹצִיא אֶל מִחוּץ לָעִיר אֶל מָקוֹם טָמֵֽא:  ת  וִיתָרַע יָת בֵּיתָא יָת אַבְנוֹהִי וְיָת אָעוֹהִי וְיָת כָּל עֲפַר בֵּיתָא וְיַפֵּק לְמִבָּרָא לְקַרְתָּא לַאֲתַר מְסָאָב:

{מו} וְהַבָּא אֶל הַבַּיִת כָּל יְמֵי הִסְגִּיר אֹתוֹ יִטְמָא עַד הָעָֽרֶב:  מ  וְהַבָּא אֶל הַבַּיִת כָּל יְמֵי הִסְגִּיר אֹתוֹ יִטְמָא עַד הָעָֽרֶב:  ת  וּדְיֵיעוֹל לְבֵיתָא כָּל יוֹמִין דְיַסְגַר יָתֵהּ יְהֵי מְסָאָב עַד רַמְשָׁא:

{מז} וְהַשֹּׁכֵב בַּבַּיִת יְכַבֵּס אֶת בְּגָדָיו וְהָֽאֹכֵל בַּבַּיִת יְכַבֵּס אֶת בְּגָדָֽיו:  מ  וְהַשֹּׁכֵב בַּבַּיִת יְכַבֵּס אֶת בְּגָדָיו וְהָֽאֹכֵל בַּבַּיִת יְכַבֵּס אֶת בְּגָדָֽיו:  ת  וּדְיִשְׁכּוּב בְּבֵיתָא יְצַבַּע יָת לְבוּשׁוֹהִי וּדְיֵיכוּל בְּבֵיתָא יְצַבַּע יָת לְבוּשׁוֹהִי:

{מח} וְאִם בֹּא יָבֹא הַכֹּהֵן וְרָאָה וְהִנֵּה לֹֽא פָשָׂה הַנֶּגַע בַּבַּיִת אַֽחֲרֵי הִטֹּחַ אֶת הַבָּיִת וְטִהַר הַכֹּהֵן אֶת הַבַּיִת כִּי נִרְפָּא הַנָּֽגַע:  מ  וְאִם בֹּא יָבֹא הַכֹּהֵן וְרָאָה וְהִנֵּה לֹֽא פָשָׂה הַנֶּגַע בַּבַּיִת אַֽחֲרֵי הִטֹּחַ אֶת הַבָּיִת וְטִהַר הַכֹּהֵן אֶת הַבַּיִת כִּי נִרְפָּא הַנָּֽגַע:  ת  וְאִם מֵעַל יֵיעוֹל כַּהֲנָא וְיֶחֱזֵי וְהָא לָא אוֹסֵף מַכְתָּשָׁא בְּבֵיתָא בָּתַר דְאִתְּשַׁע יָת בֵּיתָא וידַכֵּי כַהֲנָא יָת בֵּיתָא אֲרֵי אִתַּסִי מַכְתָּשָׁא:

{מט} וְלָקַח לְחַטֵּא אֶת הַבַּיִת שְׁתֵּי צִפֳּרִים וְעֵץ אֶרֶז וּשְׁנִי תוֹלַעַת וְאֵזֹֽב:  מ  וְלָקַח לְחַטֵּא אֶת הַבַּיִת שְׁתֵּי צִפֳּרִים וְעֵץ אֶרֶז וּשְׁנִי תוֹלַעַת וְאֵזֹֽב:  ת  וְיִסַב לְדַכָּאָה יָת בֵּיתָא תַּרְתֵּין צִפֲּרִין וְאָעָא דְאַרְזָא וּצְבַע זְהוֹרִי וְאֵזוֹבָא:

{נ} וְשָׁחַט אֶת הַצִּפֹּר הָֽאֶחָת אֶל כְּלִי חֶרֶשׂ עַל מַיִם חַיִּֽים:  מ  וְשָׁחַט אֶת הַצִּפֹּר הָֽאֶחָת אֶל כְּלִי חֶרֶשׂ עַל מַיִם חַיִּֽים:  ת  וְיִכּוֹס יָת צִפֲּרָא חֲדָא לְמַאן דַחֲסַף עַל מֵי מַבּוּעַ:

{נא} וְלָקַח אֶת עֵץ הָאֶרֶז וְאֶת הָאֵזֹב וְאֵת שְׁנִי הַתּוֹלַעַת וְאֵת הַצִּפֹּר הַֽחַיָּה וְטָבַל אֹתָם בְּדַם הַצִּפֹּר הַשְּׁחוּטָה וּבַמַּיִם הַֽחַיִּים וְהִזָּה אֶל הַבַּיִת שֶׁבַע פְּעָמִֽים:  מ  וְלָקַח אֶת עֵץ הָאֶרֶז וְאֶת הָאֵזֹב וְאֵת שְׁנִי הַתּוֹלַעַת וְאֵת הַצִּפֹּר הַֽחַיָּה וְטָבַל אֹתָם בְּדַם הַצִּפֹּר הַשְּׁחוּטָה וּבַמַּיִם הַֽחַיִּים וְהִזָּה אֶל הַבַּיִת שֶׁבַע פְּעָמִֽים:  ת  וְיִסַב יָת אָעָא דְאַרְזָא וְיָת אֵזוֹבָא וְיָת צְבַע זְהוֹרִי וְיָת צִפֲּרָא חַיְתָא וְיִטְבּוֹל יָתְהוֹן בִּדְמָא דְצִפֲּרָא דִנְכִיסָא וּבְמֵי מַבּוּעַ וְיַדִי לְבֵיתָא שְׁבַע זִמְנִין:

{נב} וְחִטֵּא אֶת הַבַּיִת בְּדַם הַצִּפּוֹר וּבַמַּיִם הַֽחַיִּים וּבַצִּפֹּר הַֽחַיָּה וּבְעֵץ הָאֶרֶז וּבָֽאֵזֹב וּבִשְׁנִי הַתּוֹלָֽעַת:  מ  וְחִטֵּא אֶת הַבַּיִת בְּדַם הַצִּפּוֹר וּבַמַּיִם הַֽחַיִּים וּבַצִּפֹּר הַֽחַיָּה וּבְעֵץ הָאֶרֶז וּבָֽאֵזֹב וּבִשְׁנִי הַתּוֹלָֽעַת:  ת  וִידַכֵּי יָת בֵּיתָא בִּדְמָא דְצִפֲּרָא וּבְמֵי מַבּוּעַ וּבְצִפֲּרָא חַיְתָא וּבְאָעָא דְאַרְזָא וּבְאֵזוֹבָא וּבִצְבַע זְהוֹרִי:

{נג} וְשִׁלַּח אֶת הַצִּפֹּר הַֽחַיָּה אֶל מִחוּץ לָעִיר אֶל פְּנֵי הַשָּׂדֶה וְכִפֶּר עַל הַבַּיִת וְטָהֵֽר:  מ  וְשִׁלַּח אֶת הַצִּפֹּר הַֽחַיָּה אֶל מִחוּץ לָעִיר אֶל פְּנֵי הַשָּׂדֶה וְכִפֶּר עַל הַבַּיִת וְטָהֵֽר:  ת  וִישַׁלַח יָת צִפֲּרָא חַיְתָא לְמִבָּרָא לְקַרְתָּא לְאַפֵּי חַקְלָא וִיכַפֵּר עַל בֵּיתָא וְיִדְכֵּי:

{נד}  חמישי  זֹאת הַתּוֹרָה לְכָל נֶגַע הַצָּרַעַת וְלַנָּֽתֶק:  מ  זֹאת הַתּוֹרָה לְכָל נֶגַע הַצָּרַעַת וְלַנָּֽתֶק:  ת  דָא אוֹרַיְתָא לְכָל מַכְתַּשׁ סְגִירוּתָא וּלְנִתְקָא:

{נה} וּלְצָרַעַת הַבֶּגֶד וְלַבָּֽיִת:  מ  וּלְצָרַעַת הַבֶּגֶד וְלַבָּֽיִת:  ת  וְלִסְגִירוּת לְבוּשָׁא וּלְבֵיתָא:

{נו} וְלַשְׂאֵת וְלַסַּפַּחַת וְלַבֶּהָֽרֶת:  מ  וְלַשְׂאֵת וְלַסַּפַּחַת וְלַבֶּהָֽרֶת:  ת  וּלְעַמְקָא וּלְעַדְיָא וּלְבַהֲרָא:

{נז} לְהוֹרֹת בְּיוֹם הַטָּמֵא וּבְיוֹם הַטָּהֹר זֹאת תּוֹרַת הַצָּרָֽעַת:  מ  לְהוֹרֹת בְּיוֹם הַטָּמֵא וּבְיוֹם הַטָּהֹר זֹאת תּוֹרַת הַצָּרָֽעַת:  ת  לְאַלָפָא בְּיוֹמָא מְסָאָבָא וּבְיוֹמָא דַכְיָא דָא אוֹרַיְתָא דִסְגִירוּתָא: (פ)




ויקרא פרק-טו

{א} וַיְדַבֵּר יְהֹוָה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן לֵאמֹֽר:  מ  וַיְדַבֵּר יְהֹוָה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה וְעִם אַהֲרֹן לְמֵימָר:

{ב} דַּבְּרוּ אֶל בְּנֵי יִשְׂרָאֵל וַֽאֲמַרְתֶּם אֲלֵהֶם אִישׁ אִישׁ כִּי יִֽהְיֶה זָב מִבְּשָׂרוֹ זוֹבוֹ טָמֵא הֽוּא:  מ  דַּבְּרוּ אֶל בְּנֵי יִשְׂרָאֵל וַֽאֲמַרְתֶּם אֲלֵהֶם אִישׁ אִישׁ כִּי יִֽהְיֶה זָב מִבְּשָׂרוֹ זוֹבוֹ טָמֵא הֽוּא:  ת  מַלִילוּ עִם בְּנֵי יִשְׂרָאֵל וְתֵימְרוּן לְהוֹן גְבַר גְבַר אֲרֵי יְהֵי דָאִיב מִבִּשְׂרֵהּ דוֹבֵהּ מְסָאָב הוּא:

{ג} וְזֹאת תִּהְיֶה טֻמְאָתוֹ בְּזוֹבוֹ רָר בְּשָׂרוֹ אֶת זוֹבוֹ אֽוֹ הֶחְתִּים בְּשָׂרוֹ מִזּוֹבוֹ טֻמְאָתוֹ הִֽוא:  מ  וְזֹאת תִּהְיֶה טֻמְאָתוֹ בְּזוֹבוֹ רָר בְּשָׂרוֹ אֶת זוֹבוֹ אֽוֹ הֶחְתִּים בְּשָׂרוֹ מִזּוֹבוֹ טֻמְאָתוֹ הִֽוא:  ת  וְדָא תְּהֵי סוֹבְתֵהּ בְּדוֹבֵהּ רָר בִּשְׂרֵהּ יָת דוֹבֵהּ אוֹ חָתִים בִּשְׂרֵהּ מִדוֹבֵהּ סוֹבְתֵהּ הִיא:

{ד} כָּל הַמִּשְׁכָּב אֲשֶׁר יִשְׁכַּב עָלָיו הַזָּב יִטְמָא וְכָֽל הַכְּלִי אֲשֶׁר יֵשֵׁב עָלָיו יִטְמָֽא:  מ  כָּל הַמִּשְׁכָּב אֲשֶׁר יִשְׁכַּב עָלָיו הַזָּב יִטְמָא וְכָֽל הַכְּלִי אֲשֶׁר יֵשֵׁב עָלָיו יִטְמָֽא:  ת  כָּל מִשְׁכְּבָא דִי יִשְׁכּוּב עֲלוֹהִי דוֹבָנָא יְהֵי מְסָאָב וְכָל מָנָא דְיֵיתֵב עֲלוֹהִי יהֵי מְסָאָב:

{ה} וְאִישׁ אֲשֶׁר יִגַּע בְּמִשְׁכָּבוֹ יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  מ  וְאִישׁ אֲשֶׁר יִגַּע בְּמִשְׁכָּבוֹ יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  ת  וּגְבַר דִי יִקְרַב בְּמִשְׁכְּבֵהּ יְצַבַּע לְבוּשׁוֹהִי וְיַסְחֵי בְמַיָא וִיהֵי מְסָאָב עַד רַמְשָׁא:

{ו} וְהַיּשֵׁב עַֽל הַכְּלִי אֲשֶׁר יֵשֵׁב עָלָיו הַזָּב יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  מ  וְהַיּשֵׁב עַֽל הַכְּלִי אֲשֶׁר יֵשֵׁב עָלָיו הַזָּב יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  ת  וּדְיֵיתֵב עַל מָנָא דְיֵיתֵב עֲלוֹהִי דוֹבָנָא יְצַבַּע לְבוּשׁוֹהִי וְיַסְחֵי בְמַיָא וִיהֵי מְסָאָב עַד רַמְשָׁא:

{ז} וְהַנֹּגֵעַ בִּבְשַׂר הַזָּב יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  מ  וְהַנֹּגֵעַ בִּבְשַׂר הַזָּב יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  ת  וּדְיִקְרַב בִּבְסַר דוֹבָנָא יְצַבַּע לְבוּשׁוֹהִי וְיַסְחֵי בְמַיָא וִיהֵי מְסָאָב עַד רַמְשָׁא:

{ח} וְכִֽי יָרֹק הַזָּב בַּטָּהוֹר וְכִבֶּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  מ  וְכִֽי יָרֹק הַזָּב בַּטָּהוֹר וְכִבֶּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  ת  וַאֲרֵי יֵרוֹק דוֹבָנָא בְּדַכְיָא וִיצַבַּע לְבוּשׁוֹהִי וְיַסְחֵי בְמַיָא וִיהֵי מְסָאָב עַד רַמְשָּׁא:

{ט} וְכָל הַמֶּרְכָּב אֲשֶׁר יִרְכַּב עָלָיו הַזָּב יִטְמָֽא:  מ  וְכָל הַמֶּרְכָּב אֲשֶׁר יִרְכַּב עָלָיו הַזָּב יִטְמָֽא:  ת  וְכָל מֶרְכְּבָא דִי יִרְכּוּב עֲלוֹהִי דוֹבָנָא יְהֵי מְסָאָב:

{י} וְכָל הַנֹּגֵעַ בְּכֹל אֲשֶׁר יִהְיֶה תַחְתָּיו יִטְמָא עַד הָעָרֶב וְהַנּוֹשֵׂא אוֹתָם יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  מ  וְכָל הַנֹּגֵעַ בְּכֹל אֲשֶׁר יִהְיֶה תַחְתָּיו יִטְמָא עַד הָעָרֶב וְהַנּוֹשֵׂא אוֹתָם יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  ת  וְכָל דְיִקְרַב בְּכֹל דִי יְהֵי תְחוֹתוֹהִי יְהֵי מְסָאָב עַד רַמְשָׁא וּדְיִטוֹל יָתְהוֹן יְצַבַּע לְבוּשׁוֹהִי וְיַסְחֵי בְמַיָא וִיהֵי מְסָאָב עַד רַמְשָׁא:

{יא} וְכֹל אֲשֶׁר יִֽגַּֽע בּוֹ הַזָּב וְיָדָיו לֹֽא שָׁטַף בַּמָּיִם וְכִבֶּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  מ  וְכֹל אֲשֶׁר יִֽגַּֽע בּוֹ הַזָּב וְיָדָיו לֹֽא שָׁטַף בַּמָּיִם וְכִבֶּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  ת  וְכֹל דִי יִקְרַב בֵּהּ דוֹבָנָא וִידוֹהִי לָא שְׁטַף בְּמַיָא וִיצַבַּע לְבוּשׁוֹהִי וְיַסְחֵי בְמַיָא וִיהֵי מְסָאָב עַד רַמְשָׁא:

{יב} וּכְלִי חֶרֶשׂ אֲשֶׁר יִגַּע בּוֹ הַזָּב יִשָּׁבֵר וְכָל כְּלִי עֵץ יִשָּׁטֵף בַּמָּֽיִם:  מ  וּכְלִי חֶרֶשׂ אֲשֶׁר יִגַּע בּוֹ הַזָּב יִשָּׁבֵר וְכָל כְּלִי עֵץ יִשָּׁטֵף בַּמָּֽיִם:  ת  וּמָאן דַחֲסַף דִי יִקְרַב בֵּהּ דוֹבָנָא יִתָּבָר וְכָל מָאן דְאָע יִשְׁתַּטַּף בְּמַיָא:

{יג} וְכִֽי יִטְהַר הַזָּב מִזּוֹבוֹ וְסָפַר לוֹ שִׁבְעַת יָמִים לְטָֽהֳרָתוֹ וְכִבֶּס בְּגָדָיו וְרָחַץ בְּשָׂרוֹ בְּמַיִם חַיִּים וְטָהֵֽר:  מ  וְכִֽי יִטְהַר הַזָּב מִזּוֹבוֹ וְסָפַר לוֹ שִׁבְעַת יָמִים לְטָֽהֳרָתוֹ וְכִבֶּס בְּגָדָיו וְרָחַץ בְּשָׂרוֹ בְּמַיִם חַיִּים וְטָהֵֽר:  ת  וַאֲרֵי יִדְכֵּי דוֹבָנָא מִדוֹבֵהּ וְיִמְנֵי לֵהּ שִׁבְעַת יוֹמִין לְדָכוּתֵהּ וִיצַבַּע לְבוּשׁוֹהִי וְיַסְחֵי בִשְׂרֵהּ בְּמֵי מַבּוּעַ וְיִדְכֵּי:

{יד} וּבַיּוֹם הַשְּׁמִינִי יִֽקַּֽח לוֹ שְׁתֵּי תֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה וּבָא לִפְנֵי יְהֹוָה אֶל פֶּתַח אֹהֶל מוֹעֵד וּנְתָנָם אֶל הַכֹּהֵֽן:  מ  וּבַיּוֹם הַשְּׁמִינִי יִֽקַּֽח לוֹ שְׁתֵּי תֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה וּבָא לִפְנֵי יְהֹוָה אֶל פֶּתַח אֹהֶל מוֹעֵד וּנְתָנָם אֶל הַכֹּהֵֽן:  ת  וּבְיוֹמָא תְמִינָאָה יִסַב לֵהּ תַּרְתֵּין שַׁפְנִינִין אוֹ תְרֵין בְּנֵי יוֹנָה וְיֵיתֵי לָקֳדָם יְיָ לִתְרַע מַשְׁכַּן זִמְנָא וְיִתְּנִנוּן לְכַהֲנָא:

{טו} וְעָשָׂה אֹתָם הַכֹּהֵן אֶחָד חַטָּאת וְהָֽאֶחָד עֹלָה וְכִפֶּר עָלָיו הַכֹּהֵן לִפְנֵי יְהוָֹה מִזּוֹבֽוֹ:  מ  וְעָשָׂה אֹתָם הַכֹּהֵן אֶחָד חַטָּאת וְהָֽאֶחָד עֹלָה וְכִפֶּר עָלָיו הַכֹּהֵן לִפְנֵי יְהוָֹה מִזּוֹבֽוֹ:  ת  וְיַעְבֵּד יָתְהוֹן כַּהֲנָא חָד חַטָאתָא וְחָד עֲלָתָא וִיכַפֵּר עֲלוֹהִי כַהֲנָא קֳדָם יְיָ מִדוֹבֵהּ: (ס)

{טז}  שישי - שביעי במחוברין  וְאִישׁ כִּֽי תֵצֵא מִמֶּנּוּ שִׁכְבַת זָרַע וְרָחַץ בַּמַּיִם אֶת כָּל בְּשָׂרוֹ וְטָמֵא עַד הָעָֽרֶב:  מ  וְאִישׁ כִּֽי תֵצֵא מִמֶּנּוּ שִׁכְבַת זָרַע וְרָחַץ בַּמַּיִם אֶת כָּל בְּשָׂרוֹ וְטָמֵא עַד הָעָֽרֶב:  ת  וּגְבַר אֲרֵי תִפוֹק מִנֵהּ שִׁכְבַת זַרְעָא וְיַסְחֵי בְמַיָא יָת כָּל בִּשְׂרֵהּ וִיהֵי מְסָאָב עַד רַמְשָׁא:

{יז} וְכָל בֶּגֶד וְכָל עוֹר אֲשֶׁר יִֽהְיֶה עָלָיו שִׁכְבַת זָרַע וְכֻבַּס בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  מ  וְכָל בֶּגֶד וְכָל עוֹר אֲשֶׁר יִֽהְיֶה עָלָיו שִׁכְבַת זָרַע וְכֻבַּס בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  ת  וְכָל לְבוּשׁ וְכָל מְשַׁךְ דִי יְהֵי עֲלוֹהִי שִׁכְבַת זַרְעָא וְיִצְטַבַּע בְּמַיָא וִיהֵי מְסָאָב עַד רַמְשָׁא:

{יח} וְאִשָּׁה אֲשֶׁר יִשְׁכַּב אִישׁ אֹתָהּ שִׁכְבַת זָרַע וְרָֽחֲצוּ בַמַּיִם וְטָֽמְאוּ עַד הָעָֽרֶב:  מ  וְאִשָּׁה אֲשֶׁר יִשְׁכַּב אִישׁ אֹתָהּ שִׁכְבַת זָרַע וְרָֽחֲצוּ בַמַּיִם וְטָֽמְאוּ עַד הָעָֽרֶב:  ת  וְאִתְּתָא דִי יִשְׁכּוּב גְבַר יָתָהּ שִׁכְבַת זַרְעָא וְיַסְחוּן בְּמַיָא וְיִסְאֲבוּן עַד רַמְשָׁא:

{יט} וְאִשָּׁה כִּי תִֽהְיֶה זָבָה דָּם יִהְיֶה זֹבָהּ בִּבְשָׂרָהּ שִׁבְעַת יָמִים תִּהְיֶה בְנִדָּתָהּ וְכָל הַנֹּגֵעַ בָּהּ יִטְמָא עַד הָעָֽרֶב:  מ  וְאִשָּׁה כִּי תִֽהְיֶה זָבָה דָּם יִהְיֶה זֹבָהּ בִּבְשָׂרָהּ שִׁבְעַת יָמִים תִּהְיֶה בְנִדָּתָהּ וְכָל הַנֹּגֵעַ בָּהּ יִטְמָא עַד הָעָֽרֶב:  ת  וְאִתְּתָא אֲרֵי תְהֵי דָיְבָא דָם יְהֵי דוֹבָהּ בְּבִשְׂרָהּ שִׁבְעָא יוֹמִין תְּהֵי בְרִחוּקָהּ וְכָל דְיִקְרַב בָּהּ יְהֵי מְסָאָב עַד רַמְשָׁא:

{כ} וְכֹל אֲשֶׁר תִּשְׁכַּב עָלָיו בְּנִדָּתָהּ יִטְמָא וְכֹל אֲשֶׁר תֵּשֵׁב עָלָיו יִטְמָֽא:  מ  וְכֹל אֲשֶׁר תִּשְׁכַּב עָלָיו בְּנִדָּתָהּ יִטְמָא וְכֹל אֲשֶׁר תֵּשֵׁב עָלָיו יִטְמָֽא:  ת  וְכֹל דִי תִשְׁכּוּב עֲלוֹהִי בְּרִחוּקָהּ יְהֵי מְסָאָב וְכֹל דִי תֵיתֵב עֲלוֹהִי יְהֵי מְסָאָב:

{כא} וְכָל הַנֹּגֵעַ בְּמִשְׁכָּבָהּ יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  מ  וְכָל הַנֹּגֵעַ בְּמִשְׁכָּבָהּ יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  ת  וְכָל דְיִקְרַב בְּמִשְׁכְּבָהּ יְצַבַּע לְבוּשׁוֹהִי וְיַסְחֵי בְמַיָא וִיהֵי מְסָאָב עַד רַמְשָׁא:

{כב} וְכָל הַנֹּגֵעַ בְּכָל כְּלִי אֲשֶׁר תֵּשֵׁב עָלָיו יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  מ  וְכָל הַנֹּגֵעַ בְּכָל כְּלִי אֲשֶׁר תֵּשֵׁב עָלָיו יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  ת  וְכָל דְיִקְרַב בְּכָל מָנָא דִי תֵיתֵב עֲלוֹהִי יְצַבַּע לְבוּשׁוֹהִי וְיַסְחֵי בְמַיָא וִיהֵי מְסָאָב עַד רַמְשָׁא:

{כג} וְאִם עַֽל הַמִּשְׁכָּב הוּא אוֹ עַֽל הַכְּלִי אֲשֶׁר הִוא ישֶֽׁבֶת עָלָיו בְּנָגְעוֹ בוֹ יִטְמָא עַד הָעָֽרֶב:  מ  וְאִם עַֽל הַמִּשְׁכָּב הוּא אוֹ עַֽל הַכְּלִי אֲשֶׁר הִוא ישֶֽׁבֶת עָלָיו בְּנָגְעוֹ בוֹ יִטְמָא עַד הָעָֽרֶב:  ת  וְאִם עַל מִשְׁכְּבָא הוּא אוֹ עַל מָנָא דִי הִיא יָתְבָא עֲלוֹהִי בְּמִקְרְבֵהּ בֵּהּ יְהֵי מְסָאָב עַד רַמְשָׁא:

{כד} וְאִם שָׁכֹב יִשְׁכַּב אִישׁ אֹתָהּ וּתְהִי נִדָּתָהּ עָלָיו וְטָמֵא שִׁבְעַת יָמִים וְכָל הַמִּשְׁכָּב אֲשֶׁר יִשְׁכַּב עָלָיו יִטְמָֽא:  מ  וְאִם שָׁכֹב יִשְׁכַּב אִישׁ אֹתָהּ וּתְהִי נִדָּתָהּ עָלָיו וְטָמֵא שִׁבְעַת יָמִים וְכָל הַמִּשְׁכָּב אֲשֶׁר יִשְׁכַּב עָלָיו יִטְמָֽא:  ת  וְאִם מִשְׁכַּב יִשְׁכּוּב גְבַר יָתָהּ וּתְהֵי רִחוּקָהּ עֲלוֹהִי וִיהֵי מְסָאָב שִׁבְעַת יוֹמִין וְכָל מִשְׁכָּב דִי יִשְׁכּוּב עֲלוֹהִי יְהֵי מְסָאָב: (ס)

{כה} וְאִשָּׁה כִּֽי יָזוּב זוֹב דָּמָהּ יָמִים רַבִּים בְּלֹא עֶת נִדָּתָהּ אוֹ כִֽי תָזוּב עַל נִדָּתָהּ כָּל יְמֵי זוֹב טֻמְאָתָהּ כִּימֵי נִדָּתָהּ תִּֽהְיֶה טְמֵאָה הִֽוא:  מ  וְאִשָּׁה כִּֽי יָזוּב זוֹב דָּמָהּ יָמִים רַבִּים בְּלֹא עֶת נִדָּתָהּ אוֹ כִֽי תָזוּב עַל נִדָּתָהּ כָּל יְמֵי זוֹב טֻמְאָתָהּ כִּימֵי נִדָּתָהּ תִּֽהְיֶה טְמֵאָה הִֽוא:  ת  וְאִתְּתָא אֲרֵי יְדוּב דוֹב דְמָהּ יוֹמִין סַגִיאִין בְּלָא עִדַן רִחוּקָהּ אוֹ אֲרֵי תְדוּב עַל רִחוּקָהּ כָּל יוֹמֵי דוֹב סוֹבְתָהּ כְּיוֹמֵי רִחוּקָהּ תְּהֵי מְסָאֳבָא הִיא:

{כו} כָּל הַמִּשְׁכָּב אֲשֶׁר תִּשְׁכַּב עָלָיו כָּל יְמֵי זוֹבָהּ כְּמִשְׁכַּב נִדָּתָהּ יִֽהְיֶה לָּהּ וְכָֽל הַכְּלִי אֲשֶׁר תֵּשֵׁב עָלָיו טָמֵא יִֽהְיֶה כְּטֻמְאַת נִדָּתָֽהּ:  מ  כָּל הַמִּשְׁכָּב אֲשֶׁר תִּשְׁכַּב עָלָיו כָּל יְמֵי זוֹבָהּ כְּמִשְׁכַּב נִדָּתָהּ יִֽהְיֶה לָּהּ וְכָֽל הַכְּלִי אֲשֶׁר תֵּשֵׁב עָלָיו טָמֵא יִֽהְיֶה כְּטֻמְאַת נִדָּתָֽהּ:  ת  כָּל מִשְׁכְּבָא דִי תִשְׁכּוּב עֲלוֹהִי כָּל יוֹמֵי דוֹבָהּ כְּמִשְׁכַּב רִחוּקָהּ יְהֵי לָהּ וְכָל מָנָא דִי תֵתֵב עֲלוֹהִי מְסָאָב יְהֵי כְּסוֹאֲבַת רִחוּקָהּ:

{כז} וְכָל הַנּוֹגֵעַ בָּם יִטְמָא וְכִבֶּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  מ  וְכָל הַנּוֹגֵעַ בָּם יִטְמָא וְכִבֶּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעָֽרֶב:  ת  וְכָל דְיִקְרַב בְּהוֹן יְהֵי מְסָאָב וִיצַבַּע לְבוּשׁוֹהִי וְיַסְחֵי בְמַיָא וִיהֵי מְסָאָב עַד רַמְשָׁא:

{כח} וְאִם טָֽהֲרָה מִזּוֹבָהּ וְסָֽפְרָה לָּהּ שִׁבְעַת יָמִים וְאַחַר תִּטְהָֽר:  מ  וְאִם טָֽהֲרָה מִזּוֹבָהּ וְסָֽפְרָה לָּהּ שִׁבְעַת יָמִים וְאַחַר תִּטְהָֽר:  ת  וְאִם דְכִיאַת מִדוֹבָהּ וְתִמְנֵי לָהּ שִׁבְעַת יוֹמִין וּבָתַר כֵּן תִדְכֵּי:

{כט}  שביעי  וּבַיּוֹם הַשְּׁמִינִי תִּֽקַּֽח לָהּ שְׁתֵּי תֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה וְהֵֽבִיאָה אוֹתָם אֶל הַכֹּהֵן אֶל פֶּתַח אֹהֶל מוֹעֵֽד:  מ  וּבַיּוֹם הַשְּׁמִינִי תִּֽקַּֽח לָהּ שְׁתֵּי תֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה וְהֵֽבִיאָה אוֹתָם אֶל הַכֹּהֵן אֶל פֶּתַח אֹהֶל מוֹעֵֽד:  ת  וּבְיוֹמָא תְמִינָאָה תִּסַב לָהּ תַּרְתֵּין שַׁפְנִינִין אוֹ תְרֵין בְּנֵי יוֹנָה וְתַיְתִי יָתְהוֹן לְוָת כַּהֲנָא לִתְרַע מַשְׁכַּן זִמְנָא:

{ל} וְעָשָׂה הַכֹּהֵן אֶת הָֽאֶחָד חַטָּאת וְאֶת הָֽאֶחָד עֹלָה וְכִפֶּר עָלֶיהָ הַכֹּהֵן לִפְנֵי יְהֹוָה מִזּוֹב טֻמְאָתָֽהּ:  מ  וְעָשָׂה הַכֹּהֵן אֶת הָֽאֶחָד חַטָּאת וְאֶת הָֽאֶחָד עֹלָה וְכִפֶּר עָלֶיהָ הַכֹּהֵן לִפְנֵי יְהֹוָה מִזּוֹב טֻמְאָתָֽהּ:  ת  וְיַעְבֵּד כַּהֲנָא יָת חָד חַטָאתָא וְיָת חָד עֲלָתָא וִיכַפֵּר עֲלָהּ כַּהֲנָא קֳדָם יְיָ מִדוֹב סוֹבְתָהּ:

{לא}  מפטיר  וְהִזַּרְתֶּם אֶת בְּנֵֽי יִשְׂרָאֵל מִטֻּמְאָתָם וְלֹא יָמֻתוּ בְּטֻמְאָתָם בְּטַמְּאָם אֶת מִשְׁכָּנִי אֲשֶׁר בְּתוֹכָֽם:  מ  וְהִזַּרְתֶּם אֶת בְּנֵֽי יִשְׂרָאֵל מִטֻּמְאָתָם וְלֹא יָמֻתוּ בְּטֻמְאָתָם בְּטַמְּאָם אֶת מִשְׁכָּנִי אֲשֶׁר בְּתוֹכָֽם:  ת  וְתַפְרְשׁוּן יָת בְּנֵי יִשְׂרָאֵל מִסוֹאֲבַתְהוֹן וְלָא יְמוּתוּן בְּסוֹאֲבַתְהוֹן בְּסָאָבְהוֹן יָת מַשְׁכְּנִי דִי בֵינֵיהוֹן:

{לב} זֹאת תּוֹרַת הַזָּב וַֽאֲשֶׁר תֵּצֵא מִמֶּנּוּ שִׁכְבַת זֶרַע לְטָמְאָה בָֽהּ:  מ  זֹאת תּוֹרַת הַזָּב וַֽאֲשֶׁר תֵּצֵא מִמֶּנּוּ שִׁכְבַת זֶרַע לְטָמְאָה בָֽהּ:  ת  דָא אוֹרַיְתָא דְדוֹבָנָא וְדִי תִפּוֹק מִנֵהּ שִּׁכְבַת זַרְעָא לְאִסְתָּאָבָא בָהּ:

{לג} וְהַדָּוָה בְּנִדָּתָהּ וְהַזָּב אֶת זוֹבוֹ לַזָּכָר וְלַנְּקֵבָה וּלְאִישׁ אֲשֶׁר יִשְׁכַּב עִם טְמֵאָֽה:  מ  וְהַדָּוָה בְּנִדָּתָהּ וְהַזָּב אֶת זוֹבוֹ לַזָּכָר וְלַנְּקֵבָה וּלְאִישׁ אֲשֶׁר יִשְׁכַּב עִם טְמֵאָֽה:  ת  וְלִדְסוֹבְתָה בְּרִחוּקָהּ וְלִדְדָאִיב יָת דוֹבֵהּ לִדְכַר וּלְנֻקְבָא וְלִגְבַר דִי יִשְׁכּוּב עִם מְסָאָבָא: (פפפ)
