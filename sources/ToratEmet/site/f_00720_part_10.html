י-א


[דברים פרק-כח-כז]  יַכְּכָה יְהוָה בִּשְׁחִין מִצְרַיִם (ובעפלים) וּבַטְּחֹרִים וּבַגָּרָב וּבֶחָרֶס אֲשֶׁר לֹא תוּכַל לְהֵרָפֵא:

[ישעיה פרק-לב-י]  יָמִים עַל שָׁנָה תִּרְגַּזְנָה בֹּטְחוֹת כִּי כָּלָה בָצִיר אֹסֶף בְּלִי יָבוֹא:

[ישעיה פרק-מא-ג]  יִרְדְּפֵם יַעֲבוֹר שָׁלוֹם אֹרַח בְּרַגְלָיו לֹא יָבוֹא:

[תהילים פרק-כ-ה]  יִתֶּן לְךָ כִלְבָבֶךָ וְכָל עֲצָתְךָ יְמַלֵּא:

[תהילים פרק-כח-ח]  יְהוָה עֹז לָמוֹ וּמָעוֹז יְשׁוּעוֹת מְשִׁיחוֹ הוּא:

[תהילים פרק-צט-ג]  יוֹדוּ שִׁמְךָ גָּדוֹל וְנוֹרָא קָדוֹשׁ הוּא:

[תהילים פרק-קב-ב]  יְהוָה שִׁמְעָה תְפִלָּתִי וְשַׁוְעָתִי אֵלֶיךָ תָבוֹא:

[תהילים פרק-קטו-ט]  יִשְׂרָאֵל בְּטַח בַּיהוָה עֶזְרָם וּמָגִנָּם הוּא:

[תהילים פרק-קטו-יא]  יִרְאֵי יְהוָה בִּטְחוּ בַיהוָה עֶזְרָם וּמָגִנָּם הוּא:

[משלי פרק-יב-יח]  יֵשׁ בּוֹטֶה כְּמַדְקְרוֹת חָרֶב וּלְשׁוֹן חֲכָמִים מַרְפֵּא:

[קהלת פרק-ד-ח]  יֵשׁ אֶחָד וְאֵין שֵׁנִי גַּם בֵּן וָאָח אֵין לוֹ וְאֵין קֵץ לְכָל עֲמָלוֹ גַּם (עיניו) עֵינוֹ לֹא תִשְׂבַּע עֹשֶׁר וּלְמִי אֲנִי עָמֵל וּמְחַסֵּר אֶת נַפְשִׁי מִטּוֹבָה גַּם זֶה הֶבֶל וְעִנְיַן רָע הוּא:


י-ב


[שמות פרק-טו-ו]  יְמִינְךָ יְהוָה נֶאְדָּרִי בַּכֹּחַ יְמִינְךָ יְהוָה תִּרְעַץ אוֹיֵב:

[דברים פרק-א-י]  יְהוָה אֱלֹהֵיכֶם הִרְבָּה אֶתְכֶם וְהִנְּכֶם הַיּוֹם כְּכוֹכְבֵי הַשָּׁמַיִם לָרֹב:

[דברים פרק-ה-ב]  יְהוָה אֱלֹהֵינוּ כָּרַת עִמָּנוּ בְּרִית בְּחֹרֵב:

[דברים פרק-כח-כח]  יַכְּכָה יְהוָה בְּשִׁגָּעוֹן וּבְעִוָּרוֹן וּבְתִמְהוֹן לֵבָב:

[דברים פרק-לב-ב]  יַעֲרֹף כַּמָּטָר לִקְחִי תִּזַּל כַּטַּל אִמְרָתִי כִּשְׂעִירִם עֲלֵי דֶשֶׁא וְכִרְבִיבִים עֲלֵי עֵשֶׂב:

[ירמיה פרק-ה-ג]  יְהוָֹה עֵינֶיךָ הֲלוֹא לֶאֱמוּנָה הִכִּיתָה אֹתָם וְלֹא חָלוּ כִּלִּיתָם מֵאֲנוּ קַחַת מוּסָר חִזְּקוּ פְנֵיהֶם מִסֶּלַע מֵאֲנוּ לָשׁוּב:

[יחזקאל פרק-לב-כא]  יְדַבְּרוּ לוֹ אֵלֵי גִבּוֹרִים מִתּוֹךְ שְׁאוֹל אֶת עֹזְרָיו יָרְדוּ שָׁכְבוּ הָעֲרֵלִים חַלְלֵי חָרֶב:

[יחזקאל פרק-מה-ב]  יִהְיֶה מִזֶּה אֶל הַקֹּדֶשׁ חֲמֵשׁ מֵאוֹת בַּחֲמֵשׁ מֵאוֹת מְרֻבָּע סָבִיב וַחֲמִשִּׁים אַמָּה מִגְרָשׁ לוֹ סָבִיב:

[יואל פרק-ד-יב]  יֵעוֹרוּ וְיַעֲלוּ הַגּוֹיִם אֶל עֵמֶק יְהוֹשָׁפָט כִּי שָׁם אֵשֵׁב לִשְׁפֹּט אֶת כָּל הַגּוֹיִם מִסָּבִיב:

[תהילים פרק-כ-ב]  יַעַנְךָ יְהוָה בְּיוֹם צָרָה יְשַׂגֶּבְךָ שֵׁם אֱלֹהֵי יַעֲקֹב:

[תהילים פרק-מח-ג]  יְפֵה נוֹף מְשׂוֹשׂ כָּל הָאָרֶץ הַר צִיּוֹן יַרְכְּתֵי צָפוֹן קִרְיַת מֶלֶךְ רָב:

[תהילים פרק-סד-יא]  יִשְׂמַח צַדִּיק בַּיהוָה וְחָסָה בוֹ וְיִתְהַלְלוּ כָּל יִשְׁרֵי לֵב:

[תהילים פרק-עג-ז]  יָצָא מֵחֵלֶב עֵינֵמוֹ עָבְרוּ מַשְׂכִּיּוֹת לֵבָב:

[תהילים פרק-קד-כג]  יֵצֵא אָדָם לְפָעֳלוֹ וְלַעֲבֹדָתוֹ עֲדֵי עָרֶב:

[משלי פרק-א-כד]  יַעַן קָרָאתִי וַתְּמָאֵנוּ נָטִיתִי יָדִי וְאֵין מַקְשִׁיב:

[משלי פרק-יג-ז]  יֵשׁ מִתְעַשֵּׁר וְאֵין כֹּל מִתְרוֹשֵׁשׁ וְהוֹן רָב:

[משלי פרק-כד-כא]  יְרָא אֶת יְהוָה בְּנִי וָמֶלֶךְ עִם שׁוֹנִים אַל תִּתְעָרָב:

[איוב פרק-כא-יב]  יִשְׂאוּ כְּתֹף וְכִנּוֹר וְיִשְׂמְחוּ לְקוֹל עוּגָב:

[איוב פרק-לד-טו]  יִגְוַע כָּל בָּשָׂר יָחַד וְאָדָם עַל עָפָר יָשׁוּב:

[איוב פרק-לט-כב]  יִשְׂחַק לְפַחַד וְלֹא יֵחָת וְלֹא יָשׁוּב מִפְּנֵי חָרֶב:


י-ג


[תהילים פרק-קז-כו]  יַעֲלוּ שָׁמַיִם יֵרְדוּ תְהוֹמוֹת נַפְשָׁם בְּרָעָה תִתְמוֹגָג:


י-ד


[שמות פרק-טו-יח]  יְהוָה יִמְלֹךְ לְעֹלָם וָעֶד:

[ישעיה פרק-מה-יז]  יִשְׂרָאֵל נוֹשַׁע בַּיהוָה תְּשׁוּעַת עוֹלָמִים לֹא תֵבֹשׁוּ וְלֹא תִכָּלְמוּ עַד עוֹלְמֵי עַד:

[תהילים פרק-ז-יז]  יָשׁוּב עֲמָלוֹ בְרֹאשׁוֹ וְעַל קָדְקֳדוֹ חֲמָסוֹ יֵרֵד:

[תהילים פרק-כא-ב]  יְהוָה בְּעָזְּךָ יִשְׂמַח מֶלֶךְ וּבִישׁוּעָתְךָ מַה יָּגֶיל מְאֹד:

[תהילים פרק-כב-כז]  יֹאכְלוּ עֲנָוִים וְיִשְׂבָּעוּ יְהַלְלוּ יְהוָה דֹּרְשָׁיו יְחִי לְבַבְכֶם לָעַד:

[תהילים פרק-נ-ג]  יָבֹא אֱלֹהֵינוּ וְאַל יֶחֱרַשׁ אֵשׁ לְפָנָיו תֹּאכֵל וּסְבִיבָיו נִשְׂעֲרָה מְאֹד:

[תהילים פרק-קד-ה]  יָסַד אֶרֶץ עַל מְכוֹנֶיהָ בַּל תִּמּוֹט עוֹלָם וָעֶד:

[משלי פרק-לא-ז]  יִשְׁתֶּה וְיִשְׁכַּח רִישׁוֹ וַעֲמָלוֹ לֹא יִזְכָּר עוֹד:

[דברי הימים א פרק-כז-כד]  יוֹאָב בֶּן צְרוּיָה הֵחֵל לִמְנוֹת וְלֹא כִלָּה וַיְהִי בָזֹאת קֶצֶף עַל יִשְׂרָאֵל וְלֹא עָלָה הַמִּסְפָּר בְּמִסְפַּר דִּבְרֵי הַיָּמִים לַמֶּלֶךְ דָּוִיד:


י-ה


[בראשית פרק-לג-יד]  יַעֲבָר נָא אֲדֹנִי לִפְנֵי עַבְדּוֹ וַאֲנִי אֶתְנָהֲלָה לְאִטִּי לְרֶגֶל הַמְּלָאכָה אֲשֶׁר לְפָנַי וּלְרֶגֶל הַיְלָדִים עַד אֲשֶׁר אָבֹא אֶל אֲדֹנִי שֵׂעִירָה:

[ויקרא פרק-ז-ל]  יָדָיו תְּבִיאֶינָה אֵת אִשֵּׁי יְהוָה אֶת הַחֵלֶב עַל הֶחָזֶה יְבִיאֶנּוּ אֵת הֶחָזֶה לְהָנִיף אֹתוֹ תְּנוּפָה לִפְנֵי יְהוָה:

[ויקרא פרק-כה-יא]  יוֹבֵל הִוא שְׁנַת הַחֲמִשִּׁים שָׁנָה תִּהְיֶה לָכֶם לֹא תִזְרָעוּ וְלֹא תִקְצְרוּ אֶת סְפִיחֶיהָ וְלֹא תִבְצְרוּ אֶת נְזִרֶיהָ:

[במדבר פרק-כ-כד]  יֵאָסֵף אַהֲרֹן אֶל עַמָּיו כִּי לֹא יָבֹא אֶל הָאָרֶץ אֲשֶׁר נָתַתִּי לִבְנֵי יִשְׂרָאֵל עַל אֲשֶׁר מְרִיתֶם אֶת פִּי לְמֵי מְרִיבָה:

[במדבר פרק-כז-טז]  יִפְקֹד יְהוָה אֱלֹהֵי הָרוּחֹת לְכָל בָּשָׂר אִישׁ עַל הָעֵדָה:

[דברים פרק-א-ו]  יְהוָה אֱלֹהֵינוּ דִּבֶּר אֵלֵינוּ בְּחֹרֵב לֵאמֹר רַב לָכֶם שֶׁבֶת בָּהָר הַזֶּה:

[דברים פרק-ג-יד]  יָאִיר בֶּן מְנַשֶּׁה לָקַח אֶת כָּל חֶבֶל אַרְגֹּב עַד גְּבוּל הַגְּשׁוּרִי וְהַמַּעֲכָתִי וַיִּקְרָא אֹתָם עַל שְׁמוֹ אֶת הַבָּשָׁן חַוֹּת יָאִיר עַד הַיּוֹם הַזֶּה:

[דברים פרק-כח-יב]  יִפְתַּח יְהוָה לְךָ אֶת אוֹצָרוֹ הַטּוֹב אֶת הַשָּׁמַיִם לָתֵת מְטַר אַרְצְךָ בְּעִתּוֹ וּלְבָרֵךְ אֵת כָּל מַעֲשֵׂה יָדֶךָ וְהִלְוִיתָ גּוֹיִם רַבִּים וְאַתָּה לֹא תִלְוֶה:

[דברים פרק-כח-כא]  יַדְבֵּק יְהוָה בְּךָ אֶת הַדָּבֶר עַד כַּלֹּתוֹ אֹתְךָ מֵעַל הָאֲדָמָה אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ:

[דברים פרק-לא-ג]  יְהוָה אֱלֹהֶיךָ הוּא עֹבֵר לְפָנֶיךָ הוּא יַשְׁמִיד אֶת הַגּוֹיִם הָאֵלֶּה מִלְּפָנֶיךָ וִירִשְׁתָּם יְהוֹשֻׁעַ הוּא עֹבֵר לְפָנֶיךָ כַּאֲשֶׁר דִּבֶּר יְהוָה:

[יהושע פרק-יא-יח]  יָמִים רַבִּים עָשָׂה יְהוֹשֻׁעַ אֶת כָּל הַמְּלָכִים הָאֵלֶּה מִלְחָמָה:

[יהושע פרק-טו-לה]  יַרְמוּת וַעֲדֻלָּם שׂוֹכֹה וַעֲזֵקָה:

[שמואל ב פרק-ג-כה]  יָדַעְתָּ אֶת אַבְנֵר בֶּן נֵר כִּי לְפַתֹּתְךָ בָּא וְלָדַעַת אֶת מוֹצָאֲךָ וְאֶת (מבואך) מוֹבָאֶךָ וְלָדַעַת אֵת כָּל אֲשֶׁר אַתָּה עֹשֶׂה:

[מלכים א פרק-י-ט]  יְהִי יְהוָה אֱלֹהֶיךָ בָּרוּךְ אֲשֶׁר חָפֵץ בְּךָ לְתִתְּךָ עַל כִּסֵּא יִשְׂרָאֵל בְּאַהֲבַת יְהוָה אֶת יִשְׂרָאֵל לְעֹלָם וַיְשִׂימְךָ לְמֶלֶךְ לַעֲשׂוֹת מִשְׁפָּט וּצְדָקָה:

[מלכים ב פרק-יט-כח]  יַעַן הִתְרַגֶּזְךָ אֵלַי וְשַׁאֲנַנְךָ עָלָה בְאָזְנָי וְשַׂמְתִּי חַחִי בְּאַפֶּךָ וּמִתְגִּי בִּשְׂפָתֶיךָ וַהֲשִׁבֹתִיךָ בַּדֶּרֶךְ אֲשֶׁר בָּאתָ בָּהּ:

[מלכים ב פרק-כא-טו]  יַעַן אֲשֶׁר עָשׂוּ אֶת הָרַע בְּעֵינַי וַיִּהְיוּ מַכְעִסִים אֹתִי מִן הַיּוֹם אֲשֶׁר יָצְאוּ אֲבוֹתָם מִמִּצְרַיִם וְעַד הַיּוֹם הַזֶּה:

[מלכים ב פרק-כב-יט]  יַעַן רַךְ לְבָבְךָ וַתִּכָּנַע מִפְּנֵי יְהוָה בְּשָׁמְעֲךָ אֲשֶׁר דִּבַּרְתִּי עַל הַמָּקוֹם הַזֶּה וְעַל יֹשְׁבָיו לִהְיוֹת לְשַׁמָּה וְלִקְלָלָה וַתִּקְרַע אֶת בְּגָדֶיךָ וַתִּבְכֶּה לְפָנָי וְגַם אָנֹכִי שָׁמַעְתִּי נְאֻם יְהוָה:

[ישעיה פרק-כג-יא]  יָדוֹ נָטָה עַל הַיָּם הִרְגִּיז מַמְלָכוֹת יְהוָה צִוָּה אֶל כְּנַעַן לַשְׁמִד מָעֻזְנֶיהָ:

[ישעיה פרק-כו-י]  יֻחַן רָשָׁע בַּל לָמַד צֶדֶק בְּאֶרֶץ נְכֹחוֹת יְעַוֵּל וּבַל יִרְאֶה גֵּאוּת יְהוָה:

[ישעיה פרק-לג-ב]  יְהוָה חָנֵּנוּ לְךָ קִוִּינוּ הֱיֵה זְרֹעָם לַבְּקָרִים אַף יְשׁוּעָתֵנוּ בְּעֵת צָרָה:

[ישעיה פרק-לז-כט]  יַעַן הִתְרַגֶּזְךָ אֵלַי וְשַׁאֲנַנְךָ עָלָה בְאָזְנָי וְשַׂמְתִּי חַחִי בְּאַפֶּךָ וּמִתְגִּי בִּשְׂפָתֶיךָ וַהֲשִׁיבֹתִיךָ בַּדֶּרֶךְ אֲשֶׁר בָּאתָ בָּהּ:

[ישעיה פרק-לח-כ]  יְהוָה לְהוֹשִׁיעֵנִי וּנְגִנוֹתַי נְנַגֵּן כָּל יְמֵי חַיֵּינוּ עַל בֵּית יְהוָה:

[ישעיה פרק-מה-ז]  יוֹצֵר אוֹר וּבוֹרֵא חֹשֶׁךְ עֹשֶׂה שָׁלוֹם וּבוֹרֵא רָע אֲנִי יְהוָה עֹשֶׂה כָל אֵלֶּה:

[ירמיה פרק-כב-כג]  (ישבתי) יֹשַׁבְתְּ בַּלְּבָנוֹן (מקננתי) מְקֻנַּנְתְּ בָּאֲרָזִים מַה נֵּחַנְתְּ בְּבֹא לָךְ חֲבָלִים חִיל כַּיֹּלֵדָה:

[ירמיה פרק-כט-כג]  יַעַן אֲשֶׁר עָשׂוּ נְבָלָה בְּיִשְׂרָאֵל וַיְנַאֲפוּ אֶת נְשֵׁי רֵעֵיהֶם וַיְדַבְּרוּ דָבָר בִּשְׁמִי שֶׁקֶר אֲשֶׁר לוֹא צִוִּיתִם וְאָנֹכִי (הוידע) הַיּוֹדֵעַ וָעֵד נְאֻם יְהוָה:

[יחזקאל פרק-לד-כא]  יַעַן בְּצַד וּבְכָתֵף תֶּהְדֹּפוּ וּבְקַרְנֵיכֶם תְּנַגְּחוּ כָּל הַנַּחְלוֹת עַד אֲשֶׁר הֲפִיצוֹתֶם אוֹתָנָה אֶל הַחוּצָה:

[יחזקאל פרק-לה-י]  יַעַן אֲמָרְךָ אֶת שְׁנֵי הַגּוֹיִם וְאֶת שְׁתֵּי הָאֲרָצוֹת לִי תִהְיֶינָה וִירַשְׁנוּהָ וַיהוָה שָׁם הָיָה:

[הושע פרק-יא-יא]  יֶחֶרְדוּ כְצִפּוֹר מִמִּצְרַיִם וּכְיוֹנָה מֵאֶרֶץ אַשּׁוּר וְהוֹשַׁבְתִּים עַל בָּתֵּיהֶם נְאֻם יְהוָה:

[מיכה פרק-ז-טז]  יִרְאוּ גוֹיִם וְיֵבֹשׁוּ מִכֹּל גְּבוּרָתָם יָשִׂימוּ יָד עַל פֶּה אָזְנֵיהֶם תֶּחֱרַשְׁנָה:

[חבקוק פרק-ג-יג]  יָצָאתָ לְיֵשַׁע עַמֶּךָ לְיֵשַׁע אֶת מְשִׁיחֶךָ מָחַצְתָּ רֹּאשׁ מִבֵּית רָשָׁע עָרוֹת יְסוֹד עַד צַוָּאר סֶלָה:

[צפניה פרק-ג-יז]  יְהוָה אֱלֹהַיִךְ בְּקִרְבֵּך גִּבּוֹר יוֹשִׁיעַ יָשִׂישׂ עָלַיִךְ בְּשִׂמְחָה יַחֲרִישׁ בְּאַהֲבָתוֹ יָגִיל עָלַיִךְ בְּרִנָּה:

[תהילים פרק-ה-ד]  יְהוָה בֹּקֶר תִּשְׁמַע קוֹלִי בֹּקֶר אֶעֱרָךְ לְךָ וַאֲצַפֶּה:

[תהילים פרק-ו-ז]  יָגַעְתִּי בְּאַנְחָתִי אַשְׂחֶה בְכָל לַיְלָה מִטָּתִי בְּדִמְעָתִי עַרְשִׂי אַמְסֶה:

[תהילים פרק-ז-ו]  יִרַדֹּף אוֹיֵב נַפְשִׁי וְיַשֵּׂג וְיִרְמֹס לָאָרֶץ חַיָּי וּכְבוֹדִי לֶעָפָר יַשְׁכֵּן סֶלָה:

[תהילים פרק-כ-ד]  יִזְכֹּר כָּל מִנְחֹתֶךָ וְעוֹלָתְךָ יְדַשְּׁנֶה סֶלָה:

[תהילים פרק-כב-לב]  יָבֹאוּ וְיַגִּידוּ צִדְקָתוֹ לְעַם נוֹלָד כִּי עָשָׂה:

[תהילים פרק-לה-ה]  יִהְיוּ כְּמֹץ לִפְנֵי רוּחַ וּמַלְאַךְ יְהוָה דּוֹחֶה:

[תהילים פרק-לז-יח]  יוֹדֵעַ יְהוָה יְמֵי תְמִימִם וְנַחֲלָתָם לְעוֹלָם תִּהְיֶה:

[תהילים פרק-מו-ד]  יֶהֱמוּ יֶחְמְרוּ מֵימָיו יִרְעֲשׁוּ הָרִים בְּגַאֲוָתוֹ סֶלָה:

[תהילים פרק-מו-ח]  יְהוָה צְבָאוֹת עִמָּנוּ מִשְׂגָּב לָנוּ אֱלֹהֵי יַעֲקֹב סֶלָה:

[תהילים פרק-מו-יב]  יְהוָה צְבָאוֹת עִמָּנוּ מִשְׂגָּב לָנוּ אֱלֹהֵי יַעֲקֹב סֶלָה:

[תהילים פרק-מז-ה]  יִבְחַר לָנוּ אֶת נַחֲלָתֵנוּ אֶת גְּאוֹן יַעֲקֹב אֲשֶׁר אָהֵב סֶלָה:

[תהילים פרק-נה-יא]  יוֹמָם וָלַיְלָה יְסוֹבְבֻהָ עַל חוֹמֹתֶיהָ וְאָוֶן וְעָמָל בְּקִרְבָּהּ:

[תהילים פרק-סה-יג]  יִרְעֲפוּ נְאוֹת מִדְבָּר וְגִיל גְּבָעוֹת תַּחְגֹּרְנָה:

[תהילים פרק-סז-ה]  יִשְׂמְחוּ וִירַנְּנוּ לְאֻמִּים כִּי תִשְׁפֹּט עַמִּים מִישׁוֹר וּלְאֻמִּים בָּאָרֶץ תַּנְחֵם סֶלָה:

[תהילים פרק-עב-ג]  יִשְׂאוּ הָרִים שָׁלוֹם לָעָם וּגְבָעוֹת בִּצְדָקָה:

[תהילים פרק-עח-טו]  יְבַקַּע צֻרִים בַּמִּדְבָּר וַיַּשְׁקְ כִּתְהֹמוֹת רַבָּה:

[תהילים פרק-פ-יד]  יְכַרְסְמֶנָּה חֲזִיר מִיָּ וְזִיז שָׂדַי יִרְעֶנָּה:

[תהילים פרק-פ-כ]  יְהוָה אֱלֹהִים צְבָאוֹת הֲשִׁיבֵנוּ הָאֵר פָּנֶיךָ וְנִוָּשֵׁעָה:

[תהילים פרק-פד-ט]  יְהוָה אֱלֹהִים צְבָאוֹת שִׁמְעָה תְפִלָּתִי הַאֲזִינָה אֱלֹהֵי יַעֲקֹב סֶלָה:

[תהילים פרק-פז-ו]  יְהוָה יִסְפֹּר בִּכְתוֹב עַמִּים זֶה יֻלַּד שָׁם סֶלָה:

[תהילים פרק-צ-י]  יְמֵי שְׁנוֹתֵינוּ בָהֶם שִׁבְעִים שָׁנָה וְאִם בִּגְבוּרֹת שְׁמוֹנִים שָׁנָה וְרָהְבָּם עָמָל וָאָוֶן כִּי גָז חִישׁ וַנָּעֻפָה:

[תהילים פרק-צח-ז]  יִרְעַם הַיָּם וּמְלֹאוֹ תֵּבֵל וְיֹשְׁבֵי בָהּ:

[תהילים פרק-קג-יט]  יְהוָה בַּשָּׁמַיִם הֵכִין כִּסְאוֹ וּמַלְכוּתוֹ בַּכֹּל מָשָׁלָה:

[תהילים פרק-קד-לד]  יֶעֱרַב עָלָיו שִׂיחִי אָנֹכִי אֶשְׂמַח בַּיהוָה:

[תהילים פרק-קד-לה]  יִתַּמּוּ חַטָּאִים מִן הָאָרֶץ וּרְשָׁעִים עוֹד אֵינָם בָּרֲכִי נַפְשִׁי אֶת יְהוָה הַלְלוּ יָהּ:

[תהילים פרק-קו-יט]  יַעֲשׂוּ עֵגֶל בְּחֹרֵב וַיִּשְׁתַּחֲווּ לְמַסֵּכָה:

[תהילים פרק-קז-מב]  יִרְאוּ יְשָׁרִים וְיִשְׂמָחוּ וְכָל עַוְלָה קָפְצָה פִּיהָ:

[תהילים פרק-קט-ט]  יִהְיוּ בָנָיו יְתוֹמִים וְאִשְׁתּוֹ אַלְמָנָה:

[תהילים פרק-קי-ו]  יָדִין בַּגּוֹיִם מָלֵא גְוִיּוֹת מָחַץ רֹאשׁ עַל אֶרֶץ רַבָּה:

[תהילים פרק-קכ-ב]  יְהוָה הַצִּילָה נַפְשִׁי מִשְּׂפַת שֶׁקֶר מִלָּשׁוֹן רְמִיָּה:

[תהילים פרק-קכא-ו]  יוֹמָם הַשֶּׁמֶשׁ לֹא יַכֶּכָּה וְיָרֵחַ בַּלָּיְלָה:

[תהילים פרק-קמה-י]  יוֹדוּךָ יְהוָה כָּל מַעֲשֶׂיךָ וַחֲסִידֶיךָ יְבָרֲכוּכָה:

[תהילים פרק-קמו-י]  יִמְלֹךְ יְהוָה לְעוֹלָם אֱלֹהַיִךְ צִיּוֹן לְדֹר וָדֹר הַלְלוּ יָהּ:

[משלי פרק-א-ה]  יִשְׁמַע חָכָם וְיוֹסֶף לֶקַח וְנָבוֹן תַּחְבֻּלוֹת יִקְנֶה:

[משלי פרק-ג-טו]  יְקָרָה הִיא (מפניים) מִפְּנִינִים וְכָל חֲפָצֶיךָ לֹא יִשְׁווּ בָהּ:

[משלי פרק-ג-יט]  יְהוָה בְּחָכְמָה יָסַד אָרֶץ כּוֹנֵן שָׁמַיִם בִּתְבוּנָה:

[משלי פרק-י-כז]  יִרְאַת יְהוָה תּוֹסִיף יָמִים וּשְׁנוֹת רְשָׁעִים תִּקְצֹרְנָה:

[משלי פרק-יב-יז]  יָפִיחַ אֱמוּנָה יַגִּיד צֶדֶק וְעֵד שְׁקָרִים מִרְמָה:

[משלי פרק-טו-לג]  יִרְאַת יְהוָה מוּסַר חָכְמָה וְלִפְנֵי כָבוֹד עֲנָוָה:

[איוב פרק-ז-ו]  יָמַי קַלּוּ מִנִּי אָרֶג וַיִּכְלוּ בְּאֶפֶס תִּקְוָה:

[איוב פרק-כ-כד]  יִבְרַח מִנֵּשֶׁק בַּרְזֶל תַּחְלְפֵהוּ קֶשֶׁת נְחוּשָׁה:

[איוב פרק-כא-יח]  יִהְיוּ כְּתֶבֶן לִפְנֵי רוּחַ וּכְמֹץ גְּנָבַתּוּ סוּפָה:

[איוב פרק-כא-כ]  יִרְאוּ (עינו) עֵינָיו כִּידוֹ וּמֵחֲמַת שַׁדַּי יִשְׁתֶּה:

[איוב פרק-כד-כ]  יִשְׁכָּחֵהוּ רֶחֶם מְתָקוֹ רִמָּה עוֹד לֹא יִזָּכֵר וַתִּשָּׁבֵר כָּעֵץ עַוְלָה:

[איוב פרק-לו-לג]  יַגִּיד עָלָיו רֵעוֹ מִקְנֶה אַף עַל עוֹלֶה:

[איוב פרק-מא-יט]  יַחְשֹׁב לְתֶבֶן בַּרְזֶל לְעֵץ רִקָּבוֹן נְחוּשָׁה:

[איוב פרק-מא-כג]  יַרְתִּיחַ כַּסִּיר מְצוּלָה יָם יָשִׂים כַּמֶּרְקָחָה:

[איוב פרק-מב-ב]  (ידעת) יָדַעְתִּי כִּי כֹל תּוּכָל וְלֹא יִבָּצֵר מִמְּךָ מְזִמָּה:

[שיר השירים פרק-ב-יד]  יוֹנָתִי בְּחַגְוֵי הַסֶּלַע בְּסֵתֶר הַמַּדְרֵגָה הַרְאִינִי אֶת מַרְאַיִךְ הַשְׁמִיעִינִי אֶת קוֹלֵךְ כִּי קוֹלֵךְ עָרֵב וּמַרְאֵיךְ נָאוֶה:

[רות פרק-א-ט]  יִתֵּן יְהוָה לָכֶם וּמְצֶאןָ מְנוּחָה אִשָּׁה בֵּית אִישָׁהּ וַתִּשַּׁק לָהֶן וַתִּשֶּׂאנָה קוֹלָן וַתִּבְכֶּינָה:

[איכה פרק-ג-כט]  יִתֵּן בֶּעָפָר פִּיהוּ אוּלַי יֵשׁ תִּקְוָה:

[איכה פרק-ג-ל]  יִתֵּן לְמַכֵּהוּ לֶחִי יִשְׂבַּע בְּחֶרְפָּה:

[עזרא פרק-י-יד]  יַעֲמְדוּ נָא שָׂרֵינוּ לְכָל הַקָּהָל וְכֹל אֲשֶׁר בֶּעָרֵינוּ הַהֹשִׁיב נָשִׁים נָכְרִיּוֹת יָבֹא לְעִתִּים מְזֻמָּנִים וְעִמָּהֶם זִקְנֵי עִיר וָעִיר וְשֹׁפְטֶיהָ עַד לְהָשִׁיב חֲרוֹן אַף אֱלֹהֵינוּ מִמֶּנּוּ עַד לַדָּבָר הַזֶּה:

[נחמיה פרק-יב-לד]  יְהוּדָה וּבִנְיָמִן וּשְׁמַעְיָה וְיִרְמְיָה:

[דברי הימים ב פרק-ט-ח]  יְהִי יְהוָה אֱלֹהֶיךָ בָּרוּךְ אֲשֶׁר חָפֵץ בְּךָ לְתִתְּךָ עַל כִּסְאוֹ לְמֶלֶךְ לַיהוָה אֱלֹהֶיךָ בְּאַהֲבַת אֱלֹהֶיךָ אֶת יִשְׂרָאֵל לְהַעֲמִידוֹ לְעוֹלָם וַיִּתֶּנְךָ עֲלֵיהֶם לְמֶלֶךְ לַעֲשׂוֹת מִשְׁפָּט וּצְדָקָה:

[דברי הימים ב פרק-לד-כז]  יַעַן רַךְ לְבָבְךָ וַתִּכָּנַע מִלִּפְנֵי אֱלֹהִים בְּשָׁמְעֲךָ אֶת דְּבָרָיו עַל הַמָּקוֹם הַזֶּה וְעַל יֹשְׁבָיו וַתִּכָּנַע לְפָנַי וַתִּקְרַע אֶת בְּגָדֶיךָ וַתֵּבְךְּ לְפָנָי וְגַם אֲנִי שָׁמַעְתִּי נְאֻם יְהוָה:

[בראשית פרק-ט-כז]  יַפְתְּ אֱלֹהִים לְיֶפֶת וְיִשְׁכֹּן בְּאָהֳלֵי שֵׁם וִיהִי כְנַעַן עֶבֶד לָמוֹ:

[שמות פרק-טו-ג]  יְהוָה אִישׁ מִלְחָמָה יְהוָה שְׁמוֹ:

[במדבר פרק-כד-ז]  יִזַּל מַיִם מִדָּלְיָו וְזַרְעוֹ בְּמַיִם רַבִּים וְיָרֹם מֵאֲגַג מַלְכּוֹ וְתִנַּשֵּׂא מַלְכֻתוֹ:

[דברים פרק-כח-ט]  יְקִימְךָ יְהוָה לוֹ לְעַם קָדוֹשׁ כַּאֲשֶׁר נִשְׁבַּע לָךְ כִּי תִשְׁמֹר אֶת מִצְוֹת יְהוָה אֱלֹהֶיךָ וְהָלַכְתָּ בִּדְרָכָיו:

[דברים פרק-כח-מט]  יִשָּׂא יְהוָה עָלֶיךָ גּוֹי מֵרָחוֹק מִקְצֵה הָאָרֶץ כַּאֲשֶׁר יִדְאֶה הַנָּשֶׁר גּוֹי אֲשֶׁר לֹא תִשְׁמַע לְשֹׁנוֹ:

[דברים פרק-לב-י]  יִמְצָאֵהוּ בְּאֶרֶץ מִדְבָּר וּבְתֹהוּ יְלֵל יְשִׁמֹן יְסֹבְבֶנְהוּ יְבוֹנְנֵהוּ יִצְּרֶנְהוּ כְּאִישׁוֹן עֵינוֹ:

[דברים פרק-לב-טז]  יַקְנִאֻהוּ בְּזָרִים בְּתוֹעֵבֹת יַכְעִיסֻהוּ:

[שופטים פרק-ה-כו]  יָדָהּ לַיָּתֵד תִּשְׁלַחְנָה וִימִינָהּ לְהַלְמוּת עֲמֵלִים וְהָלְמָה סִיסְרָא מָחֲקָה רֹאשׁוֹ וּמָחֲצָה וְחָלְפָה רַקָּתוֹ:

[שמואל א פרק-ב-י]  יְהוָה יֵחַתּוּ (מריבו) מְרִיבָיו (עלו) עָלָיו בַּשָּׁמַיִם יַרְעֵם יְהוָה יָדִין אַפְסֵי אָרֶץ וְיִתֶּן עֹז לְמַלְכּוֹ וְיָרֵם קֶרֶן מְשִׁיחוֹ:

[שמואל ב פרק-ג-לד]  יָדֶךָ לֹא אֲסֻרוֹת וְרַגְלֶיךָ לֹא לִנְחֻשְׁתַּיִם הֻגָּשׁוּ כִּנְפוֹל לִפְנֵי בְנֵי עַוְלָה נָפָלְתָּ וַיֹּסִפוּ כָל הָעָם לִבְכּוֹת עָלָיו:

[שמואל ב פרק-כב-יד]  יַרְעֵם מִן שָׁמַיִם יְהוָה וְעֶלְיוֹן יִתֵּן קוֹלוֹ:

[מלכים א פרק-ח-נז]  יְהִי יְהוָה אֱלֹהֵינוּ עִמָּנוּ כַּאֲשֶׁר הָיָה עִם אֲבֹתֵינוּ אַל יַעַזְבֵנוּ וְאַל יִטְּשֵׁנוּ:

[מלכים א פרק-יא-לג]  יַעַן אֲשֶׁר עֲזָבוּנִי וַיִּשְׁתַּחֲווּ לְעַשְׁתֹּרֶת אֱלֹהֵי צִדֹנִין לִכְמוֹשׁ אֱלֹהֵי מוֹאָב וּלְמִלְכֹּם אֱלֹהֵי בְנֵי עַמּוֹן וְלֹא הָלְכוּ בִדְרָכַי לַעֲשׂוֹת הַיָּשָׁר בְּעֵינַי וְחֻקֹּתַי וּמִשְׁפָּטַי כְּדָוִד אָבִיו:

[מלכים ב פרק-כא-יא]  יַעַן אֲשֶׁר עָשָׂה מְנַשֶּׁה מֶלֶךְ יְהוּדָה הַתֹּעֵבוֹת הָאֵלֶּה הֵרַע מִכֹּל אֲשֶׁר עָשׂוּ הָאֱמֹרִי אֲשֶׁר לְפָנָיו וַיַּחֲטִא גַם אֶת יְהוּדָה בְּגִלּוּלָיו:

[ישעיה פרק-ח-ו]  יַעַן כִּי מָאַס הָעָם הַזֶּה אֵת מֵי הַשִּׁלֹחַ הַהֹלְכִים לְאַט וּמְשׂוֹשׂ אֶת רְצִין וּבֶן רְמַלְיָהוּ:

[ישעיה פרק-יט-יד]  יְהוָה מָסַךְ בְּקִרְבָּהּ רוּחַ עִוְעִים וְהִתְעוּ אֶת מִצְרַיִם בְּכָל מַעֲשֵׂהוּ כְּהִתָּעוֹת שִׁכּוֹר בְּקִיאוֹ:

[ישעיה פרק-כו-יב]  יְהוָה תִּשְׁפֹּת שָׁלוֹם לָנוּ כִּי גַּם כָּל מַעֲשֵׂינוּ פָּעַלְתָּ לָּנוּ:

[ישעיה פרק-כו-טז]  יְהוָה בַּצַּר פְּקָדוּךָ צָקוּן לַחַשׁ מוּסָרְךָ לָמוֹ:

[ישעיה פרק-מא-כב]  יַגִּישׁוּ וְיַגִּידוּ לָנוּ אֵת אֲשֶׁר תִּקְרֶינָה הָרִאשֹׁנוֹת מָה הֵנָּה הַגִּידוּ וְנָשִׂימָה לִבֵּנוּ וְנֵדְעָה אַחֲרִיתָן אוֹ הַבָּאוֹת הַשְׁמִיעֻנוּ:

[ישעיה פרק-מב-יא]  יִשְׂאוּ מִדְבָּר וְעָרָיו חֲצֵרִים תֵּשֵׁב קֵדָר יָרֹנּוּ יֹשְׁבֵי סֶלַע מֵרֹאשׁ הָרִים יִצְוָחוּ:

[ישעיה פרק-מב-יב]  יָשִׂימוּ לַיהוָה כָּבוֹד וּתְהִלָּתוֹ בָּאִיִּים יַגִּידוּ:

[ישעיה פרק-מד-ט]  יֹצְרֵי פֶסֶל כֻּלָּם תֹּהוּ וַחֲמוּדֵיהֶם בַּל יוֹעִילוּ וְעֵדֵיהֶם הֵמָּה בַּל יִרְאוּ וּבַל יֵדְעוּ לְמַעַן יֵבֹשׁוּ:

[ישעיה פרק-מו-ז]  יִשָּׂאֻהוּ עַל כָּתֵף יִסְבְּלֻהוּ וְיַנִּיחֻהוּ תַחְתָּיו וְיַעֲמֹד מִמְּקוֹמוֹ לֹא יָמִישׁ אַף יִצְעַק אֵלָיו וְלֹא יַעֲנֶה מִצָּרָתוֹ לֹא יוֹשִׁיעֶנּוּ:

[ישעיה פרק-נז-ב]  יָבוֹא שָׁלוֹם יָנוּחוּ עַל מִשְׁכְּבוֹתָם הֹלֵךְ נְכֹחוֹ:

[ירמיה פרק-י-כג]  יָדַעְתִּי יְהוָה כִּי לֹא לָאָדָם דַּרְכּוֹ לֹא לְאִישׁ הֹלֵךְ וְהָכִין אֶת צַעֲדוֹ:

[יחזקאל פרק-יג-כב]  יַעַן הַכְאוֹת לֵב צַדִּיק שֶׁקֶר וַאֲנִי לֹא הִכְאַבְתִּיו וּלְחַזֵּק יְדֵי רָשָׁע לְבִלְתִּי שׁוּב מִדַּרְכּוֹ הָרָע לְהַחֲיֹתוֹ:

[הושע פרק-ו-ב]  יְחַיֵּנוּ מִיֹּמָיִם בַּיּוֹם הַשְּׁלִישִׁי יְקִמֵנוּ וְנִחְיֶה לְפָנָיו:

[נחום פרק-א-ג]  יְהוָֹה אֶרֶךְ אַפַּיִם (וגדול) וּגְדָל כֹּחַ וְנַקֵּה לֹא יְנַקֶּה יְהוָה בְּסוּפָה וּבִשְׂעָרָה דַּרְכּוֹ וְעָנָן אֲבַק רַגְלָיו:

[תהילים פרק-ב-ב]  יִתְיַצְּבוּ מַלְכֵי אֶרֶץ וְרוֹזְנִים נוֹסְדוּ יָחַד עַל יְהוָה וְעַל מְשִׁיחוֹ:

[תהילים פרק-ב-ד]  יוֹשֵׁב בַּשָּׁמַיִם יִשְׂחָק אֲדֹנָי יִלְעַג לָמוֹ:

[תהילים פרק-י-ח]  יֵשֵׁב בְּמַאְרַב חֲצֵרִים בַּמִּסְתָּרִים יַהֲרֹג נָקִי עֵינָיו לְחֵלְכָה יִצְפֹּנוּ:

[תהילים פרק-י-ט]  יֶאֱרֹב בַּמִּסְתָּר כְּאַרְיֵה בְסֻכֹּה יֶאֱרֹב לַחֲטוֹף עָנִי יַחְטֹף עָנִי בְּמָשְׁכוֹ בְרִשְׁתּוֹ:

[תהילים פרק-י-טז]  יְהוָה מֶלֶךְ עוֹלָם וָעֶד אָבְדוּ גוֹיִם מֵאַרְצוֹ:

[תהילים פרק-יא-ה]  יְהוָה צַדִּיק יִבְחָן וְרָשָׁע וְאֹהֵב חָמָס שָׂנְאָה נַפְשׁוֹ:

[תהילים פרק-יט-י]  יִרְאַת יְהוָה טְהוֹרָה עוֹמֶדֶת לָעַד מִשְׁפְּטֵי יְהוָה אֱמֶת צָדְקוּ יַחְדָּו:

[תהילים פרק-כ-י]  יְהוָה הוֹשִׁיעָה הַמֶּלֶךְ יַעֲנֵנוּ בְיוֹם קָרְאֵנוּ:

[תהילים פרק-כד-ה]  יִשָּׂא בְרָכָה מֵאֵת יְהוָה וּצְדָקָה מֵאֱלֹהֵי יִשְׁעוֹ:

[תהילים פרק-כה-ט]  יַדְרֵךְ עֲנָוִים בַּמִּשְׁפָּט וִילַמֵּד עֲנָוִים דַּרְכּוֹ:

[תהילים פרק-כח-ז]  יְהוָה עֻזִּי וּמָגִנִּי בּוֹ בָטַח לִבִּי וְנֶעֱזָרְתִּי וַיַּעֲלֹז לִבִּי וּמִשִּׁירִי אֲהוֹדֶנּוּ:

[תהילים פרק-לד-י]  יְראוּ אֶת יְהוָה קְדֹשָׁיו כִּי אֵין מַחְסוֹר לִירֵאָיו:

[תהילים פרק-לה-כז]  יָרֹנּוּ וְיִשְׂמְחוּ חֲפֵצֵי צִדְקִי וְיֹאמְרוּ תָמִיד יִגְדַּל יְהוָה הֶחָפֵץ שְׁלוֹם עַבְדּוֹ:

[תהילים פרק-לז-לג]  יְהוָה לֹא יַעַזְבֶנּוּ בְיָדוֹ וְלֹא יַרְשִׁיעֶנּוּ בְּהִשָּׁפְטוֹ:

[תהילים פרק-מא-ג]  יְהוָה יִשְׁמְרֵהוּ וִיחַיֵּהוּ (יאשר) וְאֻשַּׁר בָּאָרֶץ וְאַל תִּתְּנֵהוּ בְּנֶפֶשׁ אֹיְבָיו:

[תהילים פרק-מא-ד]  יְהוָה יִסְעָדֶנּוּ עַל עֶרֶשׂ דְּוָי כָּל מִשְׁכָּבוֹ הָפַכְתָּ בְחָלְיוֹ:

[תהילים פרק-מז-ד]  יַדְבֵּר עַמִּים תַּחְתֵּינוּ וּלְאֻמִּים תַּחַת רַגְלֵינוּ:

[תהילים פרק-נ-ד]  יִקְרָא אֶל הַשָּׁמַיִם מֵעָל וְאֶל הָאָרֶץ לָדִין עַמּוֹ:

[תהילים פרק-נז-ד]  יִשְׁלַח מִשָּׁמַיִם וְיוֹשִׁיעֵנִי חֵרֵף שֹׁאֲפִי סֶלָה יִשְׁלַח אֱלֹהִים חַסְדּוֹ וַאֲמִתּוֹ:

[תהילים פרק-נח-ח]  יִמָּאֲסוּ כְמוֹ מַיִם יִתְהַלְּכוּ לָמוֹ יִדְרֹךְ (חצו) חִצָּיו כְּמוֹ יִתְמֹלָלוּ:

[תהילים פרק-סא-ח]  יֵשֵׁב עוֹלָם לִפְנֵי אֱלֹהִים חֶסֶד וֶאֱמֶת מַן יִנְצְרֻהוּ:

[תהילים פרק-סג-יא]  יַגִּירֻהוּ עַל יְדֵי חָרֶב מְנָת שֻׁעָלִים יִהְיוּ:

[תהילים פרק-סד-ו]  יְחַזְּקוּ לָמוֹ דָּבָר רָע יְסַפְּרוּ לִטְמוֹן מוֹקְשִׁים אָמְרוּ מִי יִרְאֶה לָּמוֹ:

[תהילים פרק-סח-ב]  יָקוּם אֱלֹהִים יָפוּצוּ אוֹיְבָיו וְיָנוּסוּ מְשַׂנְאָיו מִפָּנָיו:

[תהילים פרק-סט-כט]  יִמָּחוּ מִסֵּפֶר חַיִּים וְעִם צַדִּיקִים אַל יִכָּתֵבוּ:

[תהילים פרק-עב-יז]  יְהִי שְׁמוֹ לְעוֹלָם לִפְנֵי שֶׁמֶשׁ (ינין) יִנּוֹן שְׁמוֹ וְיִתְבָּרְכוּ בוֹ כָּל גּוֹיִם יְאַשְּׁרוּהוּ:

[תהילים פרק-עג-ח]  יָמִיקוּ וִידַבְּרוּ בְרָע עֹשֶׁק מִמָּרוֹם יְדַבֵּרוּ:

[תהילים פרק-פג-יח]  יֵבֹשׁוּ וְיִבָּהֲלוּ עֲדֵי עַד וְיַחְפְּרוּ וְיֹאבֵדוּ:

[תהילים פרק-צא-טו]  יִקְרָאֵנִי וְאֶעֱנֵהוּ עִמּוֹ אָנֹכִי בְצָרָה אֲחַלְּצֵהוּ וַאֲכַבְּדֵהוּ:

[תהילים פרק-צד-כא]  יָגוֹדּוּ עַל נֶפֶשׁ צַדִּיק וְדָם נָקִי יַרְשִׁיעוּ:

[תהילים פרק-צו-יא]  יִשְׂמְחוּ הַשָּׁמַיִם וְתָגֵל הָאָרֶץ יִרְעַם הַיָּם וּמְלֹאוֹ:

[תהילים פרק-קג-ז]  יוֹדִיעַ דְּרָכָיו לְמֹשֶׁה לִבְנֵי יִשְׂרָאֵל עֲלִילוֹתָיו:

[תהילים פרק-קד-לא]  יְהִי כְבוֹד יְהוָה לְעוֹלָם יִשְׂמַח יְהוָה בְּמַעֲשָׂיו:

[תהילים פרק-קט-יא]  יְנַקֵּשׁ נוֹשֶׁה לְכָל אֲשֶׁר לוֹ וְיָבֹזּוּ זָרִים יְגִיעוֹ:

[תהילים פרק-קטז-טו]  יָקָר בְּעֵינֵי יְהוָה הַמָּוְתָה לַחֲסִידָיו:

[תהילים פרק-קיח-ב]  יֹאמַר נָא יִשְׂרָאֵל כִּי לְעוֹלָם חַסְדּוֹ:

[תהילים פרק-קיח-ג]  יֹאמְרוּ נָא בֵית אַהֲרֹן כִּי לְעוֹלָם חַסְדּוֹ:

[תהילים פרק-קיח-ד]  יֹאמְרוּ נָא יִרְאֵי יְהוָה כִּי לְעוֹלָם חַסְדּוֹ:

[תהילים פרק-קכב-ג]  יְרוּשָׁלִַם הַבְּנוּיָה כְּעִיר שֶׁחֻבְּרָה לָּהּ יַחְדָּו:

[תהילים פרק-קמ-יא]  (ימיטו) יִמּוֹטוּ עֲלֵיהֶם גֶּחָלִים בָּאֵשׁ יַפִּלֵם בְּמַהֲמֹרוֹת בַּל יָקוּמוּ:

[תהילים פרק-קמד-ג]  יְהוָה מָה אָדָם וַתֵּדָעֵהוּ בֶּן אֱנוֹשׁ וַתְּחַשְּׁבֵהוּ:

[תהילים פרק-קמד-ה]  יְהוָה הַט שָׁמֶיךָ וְתֵרֵד גַּע בֶּהָרִים וְיֶעֱשָׁנוּ:

[תהילים פרק-קמח-ה]  יְהַלְלוּ אֶת שֵׁם יְהוָה כִּי הוּא צִוָּה וְנִבְרָאוּ:

[תהילים פרק-קמט-ג]  יְהַלְלוּ שְׁמוֹ בְמָחוֹל בְּתֹף וְכִנּוֹר יְזַמְּרוּ לוֹ:

[משלי פרק-א-ז]  יִרְאַת יְהוָה רֵאשִׁית דָּעַת חָכְמָה וּמוּסָר אֱוִילִים בָּזוּ:

[משלי פרק-ט-ז]  יֹסֵר לֵץ לֹקֵחַ לוֹ קָלוֹן וּמוֹכִיחַ לְרָשָׁע מוּמוֹ:

[איוב פרק-ו-יח]  יִלָּפְתוּ אָרְחוֹת דַּרְכָּם יַעֲלוּ בַתֹּהוּ וְיֹאבֵדוּ:

[איוב פרק-יב-ו]  יִשְׁלָיוּ אֹהָלִים לְשֹׁדְדִים וּבַטֻּחוֹת לְמַרְגִּיזֵי אֵל לַאֲשֶׁר הֵבִיא אֱלוֹהַּ בְּיָדוֹ:

[איוב פרק-יד-כא]  יִכְבְּדוּ בָנָיו וְלֹא יֵדָע וְיִצְעֲרוּ וְלֹא יָבִין לָמוֹ:

[איוב פרק-טו-כו]  יָרוּץ אֵלָיו בְּצַוָּאר בַּעֲבִי גַּבֵּי מָגִנָּיו:

[איוב פרק-טו-לג]  יַחְמֹס כַּגֶּפֶן בִּסְרוֹ וְיַשְׁלֵךְ כַּזַּיִת נִצָּתוֹ:

[איוב פרק-יח-ז]  יֵצְרוּ צַעֲדֵי אוֹנוֹ וְתַשְׁלִיכֵהוּ עֲצָתוֹ:

[איוב פרק-יח-יב]  יְהִי רָעֵב אֹנוֹ וְאֵיד נָכוֹן לְצַלְעוֹ:

[איוב פרק-יח-יח]  יֶהְדְּפֻהוּ מֵאוֹר אֶל חֹשֶׁךְ וּמִתֵּבֵל יְנִדֻּהוּ:

[איוב פרק-כ-יג]  יַחְמֹל עָלֶיהָ וְלֹא יַעַזְבֶנָּה וְיִמְנָעֶנָּה בְּתוֹךְ חִכּוֹ:

[איוב פרק-כ-כג]  יְהִי לְמַלֵּא בִטְנוֹ יְשַׁלַּח בּוֹ חֲרוֹן אַפּוֹ וְיַמְטֵר עָלֵימוֹ בִּלְחוּמוֹ:

[איוב פרק-כ-כז]  יְגַלּוּ שָׁמַיִם עֲוֹנוֹ וְאֶרֶץ מִתְקוֹמָמָה לוֹ:

[איוב פרק-כ-כח]  יִגֶל יְבוּל בֵּיתוֹ נִגָּרוֹת בְּיוֹם אַפּוֹ:

[איוב פרק-כא-יג]  (יבלו) יְכַלּוּ בַטּוֹב יְמֵיהֶם וּבְרֶגַע שְׁאוֹל יֵחָתּוּ:

[איוב פרק-כב-יט]  יִרְאוּ צַדִּיקִים וְיִשְׂמָחוּ וְנָקִי יִלְעַג לָמוֹ:

[איוב פרק-כד-ט]  יִגְזְלוּ מִשֹּׁד יָתוֹם וְעַל עָנִי יַחְבֹּלוּ:

[איוב פרק-כז-כא]  יִשָּׂאֵהוּ קָדִים וְיֵלַךְ וִישָׂעֲרֵהוּ מִמְּקֹמוֹ:

[איוב פרק-כז-כג]  יִשְׂפֹּק עָלֵימוֹ כַפֵּימוֹ וְיִשְׁרֹק עָלָיו מִמְּקֹמוֹ:

[איוב פרק-לג-ג]  יֹשֶׁר לִבִּי אֲמָרָי וְדַעַת שְׂפָתַי בָּרוּר מִלֵּלוּ:

[איוב פרק-לג-כא]  יִכֶל בְּשָׂרוֹ מֵרֹאִי (ושפי) וְשֻׁפּוּ עַצְמוֹתָיו לֹא רֻאּוּ:

[איוב פרק-לג-כו]  יֶעְתַּר אֶל אֱלוֹהַּ וַיִּרְצֵהוּ וַיַּרְא פָּנָיו בִּתְרוּעָה וַיָּשֶׁב לֶאֱנוֹשׁ צִדְקָתוֹ:

[איוב פרק-לט-ד]  יַחְלְמוּ בְנֵיהֶם יִרְבּוּ בַבָּר יָצְאוּ וְלֹא שָׁבוּ לָמוֹ:

[איוב פרק-מ-יז]  יַחְפֹּץ זְנָבוֹ כְמוֹ אָרֶז גִּידֵי (פחדו) פַחֲדָיו יְשֹׂרָגוּ:

[רות פרק-ב-יב]  יְשַׁלֵּם יְהוָה פָּעֳלֵךְ וּתְהִי מַשְׂכֻּרְתֵּךְ שְׁלֵמָה מֵעִם יְהוָה אֱלֹהֵי יִשְׂרָאֵל אֲשֶׁר בָּאת לַחֲסוֹת תַּחַת כְּנָפָיו:

[איכה פרק-ג-כח]  יֵשֵׁב בָּדָד וְיִדֹּם כִּי נָטַל עָלָיו:

[קהלת פרק-א-י]  יֵשׁ דָּבָר שֶׁיֹּאמַר רְאֵה זֶה חָדָשׁ הוּא כְּבָר הָיָה לְעֹלָמִים אֲשֶׁר הָיָה מִלְּפָנֵנוּ:

[קהלת פרק-ג-יב]  יָדַעְתִּי כִּי אֵין טוֹב בָּם כִּי אִם לִשְׂמוֹחַ וְלַעֲשׂוֹת טוֹב בְּחַיָּיו:

[קהלת פרק-ג-יד]  יָדַעְתִּי כִּי כָּל אֲשֶׁר יַעֲשֶׂה הָאֱלֹהִים הוּא יִהְיֶה לְעוֹלָם עָלָיו אֵין לְהוֹסִיף וּמִמֶּנּוּ אֵין לִגְרֹעַ וְהָאֱלֹהִים עָשָׂה שֶׁיִּרְאוּ מִלְּפָנָיו:

[קהלת פרק-ה-יב]  יֵשׁ רָעָה חוֹלָה רָאִיתִי תַּחַת הַשָּׁמֶשׁ עֹשֶׁר שָׁמוּר לִבְעָלָיו לְרָעָתוֹ:

[אסתר פרק-ו-ח]  יָבִיאוּ לְבוּשׁ מַלְכוּת אֲשֶׁר לָבַשׁ בּוֹ הַמֶּלֶךְ וְסוּס אֲשֶׁר רָכַב עָלָיו הַמֶּלֶךְ וַאֲשֶׁר נִתַּן כֶּתֶר מַלְכוּת בְּרֹאשׁוֹ:

[דניאל פרק-יב-י]  יִתְבָּרֲרוּ וְיִתְלַבְּנוּ וְיִצָּרְפוּ רַבִּים וְהִרְשִׁיעוּ רְשָׁעִים וְלֹא יָבִינוּ כָּל רְשָׁעִים וְהַמַּשְׂכִּלִים יָבִינוּ:

[עזרא פרק-ד-יב]  יְדִיעַ לֶהֱוֵא לְמַלְכָּא דִּי יְהוּדָיֵא דִּי סְלִקוּ מִן לְוָתָךְ עֲלֶינָא אֲתוֹ לִירוּשְׁלֶם קִרְיְתָא מָרָדְתָּא וּבִאישְׁתָּא בָּנַיִן (ושורי) וְשׁוּרַיָּא (אשכללו) שַׁכְלִלוּ וְאֻשַּׁיָּא יַחִיטוּ:

[דברי הימים א פרק-ג-יא]  יוֹרָם בְּנוֹ אֲחַזְיָהוּ בְנוֹ יוֹאָשׁ בְּנוֹ:

[דברי הימים א פרק-ו-ו]  יוֹאָח בְּנוֹ עִדּוֹ בְנוֹ זֶרַח בְּנוֹ יְאָתְרַי בְּנוֹ:

[דברי הימים א פרק-טז-לב]  יִרְעַם הַיָּם וּמְלוֹאוֹ יַעֲלֹץ הַשָּׂדֶה וְכָל אֲשֶׁר בּוֹ:

[דברי הימים א פרק-יז-כ]  יְהוָה אֵין כָּמוֹךָ וְאֵין אֱלֹהִים זוּלָתֶךָ בְּכֹל אֲשֶׁר שָׁמַעְנוּ בְּאָזְנֵינוּ:

[דברי הימים ב פרק-כט-א]  יְחִזְקִיָּהוּ מָלַךְ בֶּן עֶשְׂרִים וְחָמֵשׁ שָׁנָה וְעֶשְׂרִים וָתֵשַׁע שָׁנָה מָלַךְ בִּירוּשָׁלִָם וְשֵׁם אִמּוֹ אֲבִיָּה בַּת זְכַרְיָהוּ:


י-ז


[משלי פרק-ח-כב]  יְהוָה קָנָנִי רֵאשִׁית דַּרְכּוֹ קֶדֶם מִפְעָלָיו מֵאָז:


י-ח


[ישעיה פרק-כו-ג]  יֵצֶר סָמוּךְ תִּצֹּר שָׁלוֹם שָׁלוֹם כִּי בְךָ בָּטוּחַ:

[ישעיה פרק-נה-ז]  יַעֲזֹב רָשָׁע דַּרְכּוֹ וְאִישׁ אָוֶן מַחְשְׁבֹתָיו וְיָשֹׁב אֶל יְהוָה וִירַחֲמֵהוּ וְאֶל אֱלֹהֵינוּ כִּי יַרְבֶּה לִסְלוֹחַ:

[זכריה פרק-ט-טו]  יְהוָה צְבָאוֹת יָגֵן עֲלֵיהֶם וְאָכְלוּ וְכָבְשׁוּ אַבְנֵי קֶלַע וְשָׁתוּ הָמוּ כְּמוֹ יָיִן וּמָלְאוּ כַּמִּזְרָק כְּזָוִיּוֹת מִזְבֵּחַ:

[תהילים פרק-מ-טז]  יָשֹׁמּוּ עַל עֵקֶב בָּשְׁתָּם הָאֹמְרִים לִי הֶאָח הֶאָח:

[תהילים פרק-נו-ד]  יוֹם אִירָא אֲנִי אֵלֶיךָ אֶבְטָח:

[תהילים פרק-ע-ד]  יָשׁוּבוּ עַל עֵקֶב בָּשְׁתָּם הָאֹמְרִים הֶאָח הֶאָח:

[תהילים פרק-עב-ז]  יִפְרַח בְּיָמָיו צַדִּיק וְרֹב שָׁלוֹם עַד בְּלִי יָרֵחַ:

[תהילים פרק-קט-יד]  יִזָּכֵר עֲוֹן אֲבֹתָיו אֶל יְהוָה וְחַטַּאת אִמּוֹ אַל תִּמָּח:

[תהילים פרק-קט-כח]  יְקַלְלוּ הֵמָּה וְאַתָּה תְבָרֵךְ קָמוּ וַיֵּבֹשׁוּ וְעַבְדְּךָ יִשְׂמָח:

[איוב פרק-לג-יח]  יַחְשֹׂךְ נַפְשׁוֹ מִנִּי שָׁחַת וְחַיָּתוֹ מֵעֲבֹר בַּשָּׁלַח:


י-ט


[תהילים פרק-עב-ב]  יָדִין עַמְּךָ בְצֶדֶק וַעֲנִיֶּיךָ בְמִשְׁפָּט:

[תהילים פרק-צג-א]  יְהוָה מָלָךְ גֵּאוּת לָבֵשׁ לָבֵשׁ יְהוָה עֹז הִתְאַזָּר אַף תִּכּוֹן תֵּבֵל בַּל תִּמּוֹט:

[משלי פרק-יא-כא]  יָד לְיָד לֹא יִנָּקֶה רָּע וְזֶרַע צַדִּיקִים נִמְלָט:

[קהלת פרק-י-ה]  יֵשׁ רָעָה רָאִיתִי תַּחַת הַשָּׁמֶשׁ כִּשְׁגָגָה שֶׁיֹּצָא מִלִּפְנֵי הַשַּׁלִּיט:


י-י


[דברים פרק-כח-כ]  יְשַׁלַּח יְהוָה בְּךָ אֶת הַמְּאֵרָה אֶת הַמְּהוּמָה וְאֶת הַמִּגְעֶרֶת בְּכָל מִשְׁלַח יָדְךָ אֲשֶׁר תַּעֲשֶׂה עַד הִשָּׁמֶדְךָ וְעַד אֲבָדְךָ מַהֵר מִפְּנֵי רֹעַ מַעֲלָלֶיךָ אֲשֶׁר עֲזַבְתָּנִי:

[שמואל ב פרק-כב-יח]  יַצִּילֵנִי מֵאֹיְבִי עָז מִשֹּׂנְאַי כִּי אָמְצוּ מִמֶּנִּי:

[שמואל ב פרק-כב-יט]  יְקַדְּמֻנִי בְּיוֹם אֵידִי וַיְהִי יְהוָה מִשְׁעָן לִי:

[שמואל ב פרק-כב-כא]  יִגְמְלֵנִי יְהוָה כְּצִדְקָתִי כְּבֹר יָדַי יָשִׁיב לִי:

[שמואל ב פרק-כג-לו]  יִגְאָל בֶּן נָתָן מִצֹּבָה בָּנִי הַגָּדִי:

[מלכים א פרק-כב-מב]  יְהוֹשָׁפָט בֶּן שְׁלֹשִׁים וְחָמֵשׁ שָׁנָה בְּמָלְכוֹ וְעֶשְׂרִים וְחָמֵשׁ שָׁנָה מָלַךְ בִּירוּשָׁלִָם וְשֵׁם אִמּוֹ עֲזוּבָה בַּת שִׁלְחִי:

[ירמיה פרק-י-כד]  יַסְּרֵנִי יְהוָה אַךְ בְּמִשְׁפָּט אַל בְּאַפְּךָ פֶּן תַּמְעִטֵנִי:

[חבקוק פרק-ג-יט]  יְהוִה אֲדֹנָי חֵילִי וַיָּשֶׂם רַגְלַי כָּאַיָּלוֹת וְעַל בָּמוֹתַי יַדְרִכֵנִי לַמְנַצֵּחַ בִּנְגִינוֹתָי:

[תהילים פרק-ג-ב]  יְהוָה מָה רַבּוּ צָרָי רַבִּים קָמִים עָלָי:

[תהילים פרק-ו-ב]  יְהוָה אַל בְּאַפְּךָ תוֹכִיחֵנִי וְאַל בַּחֲמָתְךָ תְיַסְּרֵנִי:

[תהילים פרק-ז-ב]  יְהוָה אֱלֹהַי בְּךָ חָסִיתִי הוֹשִׁיעֵנִי מִכָּל רֹדְפַי וְהַצִּילֵנִי:

[תהילים פרק-ז-ד]  יְהוָה אֱלֹהַי אִם עָשִׂיתִי זֹאת אִם יֶשׁ עָוֶל בְּכַפָּי:

[תהילים פרק-ז-ט]  יְהוָה יָדִין עַמִּים שָׁפְטֵנִי יְהוָה כְּצִדְקִי וּכְתֻמִּי עָלָי:

[תהילים פרק-טז-ד]  יִרְבּוּ עַצְּבוֹתָם אַחֵר מָהָרוּ בַּל אַסִּיךְ נִסְכֵּיהֶם מִדָּם וּבַל אֶשָּׂא אֶת שְׁמוֹתָם עַל שְׂפָתָי:

[תהילים פרק-טז-ה]  יְהוָה מְנָת חֶלְקִי וְכוֹסִי אַתָּה תּוֹמִיךְ גּוֹרָלִי:

[תהילים פרק-יח-ג]  יְהוָה סַלְעִי וּמְצוּדָתִי וּמְפַלְטִי אֵלִי צוּרִי אֶחֱסֶה בּוֹ מָגִנִּי וְקֶרֶן יִשְׁעִי מִשְׂגַּבִּי:

[תהילים פרק-יח-יח]  יַצִּילֵנִי מֵאֹיְבִי עָז וּמִשֹּׂנְאַי כִּי אָמְצוּ מִמֶּנִּי:

[תהילים פרק-יח-יט]  יְקַדְּמוּנִי בְיוֹם אֵידִי וַיְהִי יְהוָה לְמִשְׁעָן לִי:

[תהילים פרק-יח-כא]  יִגְמְלֵנִי יְהוָה כְּצִדְקִי כְּבֹר יָדַי יָשִׁיב לִי:

[תהילים פרק-יט-טו]  יִהְיוּ לְרָצוֹן אִמְרֵי פִי וְהֶגְיוֹן לִבִּי לְפָנֶיךָ יְהוָה צוּרִי וְגֹאֲלִי:

[תהילים פרק-כב-טז]  יָבֵשׁ כַּחֶרֶשׂ כֹּחִי וּלְשׁוֹנִי מֻדְבָּק מַלְקוֹחָי וְלַעֲפַר מָוֶת תִּשְׁפְּתֵנִי:

[תהילים פרק-ל-ג]  יְהוָה אֱלֹהָי שִׁוַּעְתִּי אֵלֶיךָ וַתִּרְפָּאֵנִי:

[תהילים פרק-לה-ד]  יֵבֹשׁוּ וְיִכָּלְמוּ מְבַקְשֵׁי נַפְשִׁי יִסֹּגוּ אָחוֹר וְיַחְפְּרוּ חֹשְׁבֵי רָעָתִי:

[תהילים פרק-לה-יא]  יְקוּמוּן עֵדֵי חָמָס אֲשֶׁר לֹא יָדַעְתִּי יִשְׁאָלוּנִי:

[תהילים פרק-לה-יב]  יְשַׁלְּמוּנִי רָעָה תַּחַת טוֹבָה שְׁכוֹל לְנַפְשִׁי:

[תהילים פרק-לה-כו]  יֵבֹשׁוּ וְיַחְפְּרוּ יַחְדָּו שְׂמֵחֵי רָעָתִי יִלְבְּשׁוּ בֹשֶׁת וּכְלִמָּה הַמַּגְדִּילִים עָלָי:

[תהילים פרק-לח-ב]  יְהוָה אַל בְּקֶצְפְּךָ תוֹכִיחֵנִי וּבַחֲמָתְךָ תְיַסְּרֵנִי:

[תהילים פרק-מ-טו]  יֵבֹשׁוּ וְיַחְפְּרוּ יַחַד מְבַקְשֵׁי נַפְשִׁי לִסְפּוֹתָהּ יִסֹּגוּ אָחוֹר וְיִכָּלְמוּ חֲפֵצֵי רָעָתִי:

[תהילים פרק-מא-ח]  יַחַד עָלַי יִתְלַחֲשׁוּ כָּל שֹׂנְאָי עָלַי יַחְשְׁבוּ רָעָה לִי:

[תהילים פרק-מב-ט]  יוֹמָם יְצַוֶּה יְהוָה חַסְדּוֹ וּבַלַּיְלָה (שירה) שִׁירוֹ עִמִּי תְּפִלָּה לְאֵל חַיָּי:

[תהילים פרק-נ-יא]  יָדַעְתִּי כָּל עוֹף הָרִים וְזִיז שָׂדַי עִמָּדִי:

[תהילים פרק-נו-ז]  יָגוּרוּ (יצפינו) יִצְפּוֹנוּ הֵמָּה עֲקֵבַי יִשְׁמֹרוּ כַּאֲשֶׁר קִוּוּ נַפְשִׁי:

[תהילים פרק-סט-ד]  יָגַעְתִּי בְקָרְאִי נִחַר גְּרוֹנִי כָּלוּ עֵינַי מְיַחֵל לֵאלֹהָי:

[תהילים פרק-ע-ג]  יֵבֹשׁוּ וְיַחְפְּרוּ מְבַקְשֵׁי נַפְשִׁי יִסֹּגוּ אָחוֹר וְיִכָּלְמוּ חֲפֵצֵי רָעָתִי:

[תהילים פרק-עא-יג]  יֵבֹשׁוּ יִכְלוּ שֹׂטְנֵי נַפְשִׁי יַעֲטוּ חֶרְפָּה וּכְלִמָּה מְבַקְשֵׁי רָעָתִי:

[תהילים פרק-קיח-ז]  יְהוָה לִי בְּעֹזְרָי וַאֲנִי אֶרְאֶה בְשֹׂנְאָי:

[תהילים פרק-קיח-יח]  יַסֹּר יִסְּרַנִּי יָּהּ וְלַמָּוֶת לֹא נְתָנָנִי:

[תהילים פרק-קיט-עד]  יְרֵאֶיךָ יִרְאוּנִי וְיִשְׂמָחוּ כִּי לִדְבָרְךָ יִחָלְתִּי:

[תהילים פרק-קיט-עה]  יָדַעְתִּי יְהוָה כִּי צֶדֶק מִשְׁפָּטֶיךָ וֶאֱמוּנָה עִנִּיתָנִי:

[תהילים פרק-קיט-עז]  יְבֹאוּנִי רַחֲמֶיךָ וְאֶחְיֶה כִּי תוֹרָתְךָ שַׁעֲשֻׁעָי:

[משלי פרק-ח-יג]  יִרְאַת יְהוָה שְׂנֹאת רָע גֵּאָה וְגָאוֹן וְדֶרֶךְ רָע וּפִי תַהְפֻּכוֹת שָׂנֵאתִי:

[משלי פרק-יב-י]  יוֹדֵעַ צַדִּיק נֶפֶשׁ בְּהֶמְתּוֹ וְרַחֲמֵי רְשָׁעִים אַכְזָרִי:

[איוב פרק-ט-כח]  יָגֹרְתִּי כָל עַצְּבֹתָי יָדַעְתִּי כִּי לֹא תְנַקֵּנִי:

[איוב פרק-ט-לד]  יָסֵר מֵעָלַי שִׁבְטוֹ וְאֵמָתוֹ אַל תְּבַעֲתַנִּי:

[איוב פרק-י-ח]  יָדֶיךָ עִצְּבוּנִי וַיַּעֲשׂוּנִי יַחַד סָבִיב וַתְּבַלְּעֵנִי:

[איוב פרק-טז-יא]  יַסְגִּירֵנִי אֵל אֶל עֲוִיל וְעַל יְדֵי רְשָׁעִים יִרְטֵנִי:

[איוב פרק-טז-יג]  יָסֹבּוּ עָלַי רַבָּיו יְפַלַּח כִּלְיוֹתַי וְלֹא יַחְמוֹל יִשְׁפֹּךְ לָאָרֶץ מְרֵרָתִי:

[בראשית פרק-כז-כט]  יַעַבְדוּךָ עַמִּים (וישתחו) וְיִשְׁתַּחֲווּ לְךָ לְאֻמִּים הֱוֵה גְבִיר לְאַחֶיךָ וְיִשְׁתַּחֲוּוּ לְךָ בְּנֵי אִמֶּךָ אֹרְרֶיךָ אָרוּר וּמְבָרֲכֶיךָ בָּרוּךְ:

[דברים פרק-יז-ז]  יַד הָעֵדִים תִּהְיֶה בּוֹ בָרִאשֹׁנָה לַהֲמִיתוֹ וְיַד כָּל הָעָם בָּאַחֲרֹנָה וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ:

[דברים פרק-כח-ז]  יִתֵּן יְהוָה אֶת אֹיְבֶיךָ הַקָּמִים עָלֶיךָ נִגָּפִים לְפָנֶיךָ בְּדֶרֶךְ אֶחָד יֵצְאוּ אֵלֶיךָ וּבְשִׁבְעָה דְרָכִים יָנוּסוּ לְפָנֶיךָ:

[דברים פרק-כח-ח]  יְצַו יְהוָה אִתְּךָ אֶת הַבְּרָכָה בַּאֲסָמֶיךָ וּבְכֹל מִשְׁלַח יָדֶךָ וּבֵרַכְךָ בָּאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:

[דברים פרק-כח-כב]  יַכְּכָה יְהוָה בַּשַּׁחֶפֶת וּבַקַּדַּחַת וּבַדַּלֶּקֶת וּבַחַרְחֻר וּבַחֶרֶב וּבַשִּׁדָּפוֹן וּבַיֵּרָקוֹן וּרְדָפוּךָ עַד אָבְדֶךָ:

[דברים פרק-כח-כד]  יִתֵּן יְהוָה אֶת מְטַר אַרְצְךָ אָבָק וְעָפָר מִן הַשָּׁמַיִם יֵרֵד עָלֶיךָ עַד הִשָּׁמְדָךְ:

[דברים פרק-כח-לה]  יַכְּכָה יְהוָה בִּשְׁחִין רָע עַל הַבִּרְכַּיִם וְעַל הַשֹּׁקַיִם אֲשֶׁר לֹא תוּכַל לְהֵרָפֵא מִכַּף רַגְלְךָ וְעַד קָדְקֳדֶךָ:

[דברים פרק-לג-י]  יוֹרוּ מִשְׁפָּטֶיךָ לְיַעֲקֹב וְתוֹרָתְךָ לְיִשְׂרָאֵל יָשִׂימוּ קְטוֹרָה בְּאַפֶּךָ וְכָלִיל עַל מִזְבְּחֶךָ:

[שמואל א פרק-כד-יג]  יִשְׁפֹּט יְהוָה בֵּינִי וּבֵינֶךָ וּנְקָמַנִי יְהוָה מִמֶּךָּ וְיָדִי לֹא תִהְיֶה בָּךְ:

[שמואל ב פרק-יט-לח]  יָשָׁב נָא עַבְדְּךָ וְאָמֻת בְּעִירִי עִם קֶבֶר אָבִי וְאִמִּי וְהִנֵּה עַבְדְּךָ כִמְהָם יַעֲבֹר עִם אֲדֹנִי הַמֶּלֶךְ וַעֲשֵׂה לוֹ אֵת אֲשֶׁר טוֹב בְּעֵינֶיךָ:

[ישעיה פרק-כו-יג]  יְהוָה אֱלֹהֵינוּ בְּעָלוּנוּ אֲדֹנִים זוּלָתֶךָ לְבַד בְּךָ נַזְכִּיר שְׁמֶךָ:

[ירמיה פרק-יד-כ]  יָדַעְנוּ יְהוָה רִשְׁעֵנוּ עֲוֹן אֲבוֹתֵינוּ כִּי חָטָאנוּ לָךְ:

[יחזקאל פרק-כז-יג]  יָוָן תֻּבַל וָמֶשֶׁךְ הֵמָּה רֹכְלָיִךְ בְּנֶפֶשׁ אָדָם וּכְלֵי נְחֹשֶׁת נָתְנוּ מַעֲרָבֵךְ:

[תהילים פרק-מח-יב]  יִשְׂמַח הַר צִיּוֹן תָּגֵלְנָה בְּנוֹת יְהוּדָה לְמַעַן מִשְׁפָּטֶיךָ:

[תהילים פרק-ע-ה]  יָשִׂישׂוּ וְיִשְׂמְחוּ בְּךָ כָּל מְבַקְשֶׁיךָ וְיֹאמְרוּ תָמִיד יִגְדַּל אֱלֹהִים אֹהֲבֵי יְשׁוּעָתֶךָ:

[תהילים פרק-פד-יג]  יְהוָה צְבָאוֹת אַשְׁרֵי אָדָם בֹּטֵחַ בָּךְ:

[תהילים פרק-פח-ב]  יְהוָה אֱלֹהֵי יְשׁוּעָתִי יוֹם צָעַקְתִּי בַלַּיְלָה נֶגְדֶּךָ:

[תהילים פרק-קיט-עו]  יְהִי נָא חַסְדְּךָ לְנַחֲמֵנִי כְּאִמְרָתְךָ לְעַבְדֶּךָ:

[תהילים פרק-קיט-עח]  יֵבֹשׁוּ זֵדִים כִּי שֶׁקֶר עִוְּתוּנִי אֲנִי אָשִׂיחַ בְּפִקּוּדֶיךָ:

[תהילים פרק-קכא-ה]  יְהוָה שֹׁמְרֶךָ יְהוָה צִלְּךָ עַל יַד יְמִינֶךָ:

[תהילים פרק-קכא-ז]  יְהוָה יִשְׁמָרְךָ מִכָּל רָע יִשְׁמֹר אֶת נַפְשֶׁךָ:

[תהילים פרק-קכב-ז]  יְהִי שָׁלוֹם בְּחֵילֵךְ שַׁלְוָה בְּאַרְמְנוֹתָיִךְ:

[תהילים פרק-קכח-ב]  יְגִיעַ כַּפֶּיךָ כִּי תֹאכֵל אַשְׁרֶיךָ וְטוֹב לָךְ:

[תהילים פרק-קכח-ה]  יְבָרֶכְךָ יְהוָה מִצִּיּוֹן וּרְאֵה בְּטוּב יְרוּשָׁלִָם כֹּל יְמֵי חַיֶּיךָ:

[תהילים פרק-קלח-ד]  יוֹדוּךָ יְהוָה כָּל מַלְכֵי אָרֶץ כִּי שָׁמְעוּ אִמְרֵי פִיךָ:

[משלי פרק-ה-יז]  יִהְיוּ לְךָ לְבַדֶּךָ וְאֵין לְזָרִים אִתָּךְ:

[משלי פרק-ה-יח]  יְהִי מְקוֹרְךָ בָרוּךְ וּשְׂמַח מֵאֵשֶׁת נְעוּרֶךָ:

[משלי פרק-יט-יח]  יַסֵּר בִּנְךָ כִּי יֵשׁ תִּקְוָה וְאֶל הֲמִיתוֹ אַל תִּשָּׂא נַפְשֶׁךָ:

[משלי פרק-כג-כה]  יִשְׂמַח אָבִיךָ וְאִמֶּךָ וְתָגֵל יוֹלַדְתֶּךָ:

[משלי פרק-כז-ב]  יְהַלֶּלְךָ זָר וְלֹא פִיךָ נָכְרִי וְאַל שְׂפָתֶיךָ:

[משלי פרק-כט-יז]  יַסֵּר בִּנְךָ וִינִיחֶךָ וְיִתֵּן מַעֲדַנִּים לְנַפְשֶׁךָ:

[משלי פרק-לא-יט]  יָדֶיהָ שִׁלְּחָה בַכִּישׁוֹר וְכַפֶּיהָ תָּמְכוּ פָלֶךְ:

[איוב פרק-טו-ו]  יַרְשִׁיעֲךָ פִיךָ וְלֹא אָנִי וּשְׂפָתֶיךָ יַעֲנוּ בָךְ:

[איוב פרק-כב-ל]  יְמַלֵּט אִי נָקִי וְנִמְלַט בְּבֹר כַּפֶּיךָ:

[איכה פרק-א-י]  יָדוֹ פָּרַשׂ צָר עַל כָּל מַחֲמַדֶּיהָ כִּי רָאֲתָה גוֹיִם בָּאוּ מִקְדָּשָׁהּ אֲשֶׁר צִוִּיתָה לֹא יָבֹאוּ בַקָּהָל לָךְ:

[דניאל פרק-ט-ח]  יְהוָה לָנוּ בֹּשֶׁת הַפָּנִים לִמְלָכֵינוּ לְשָׂרֵינוּ וְלַאֲבֹתֵינוּ אֲשֶׁר חָטָאנוּ לָךְ:

[דברי הימים א פרק-טז-לא]  יִשְׂמְחוּ הַשָּׁמַיִם וְתָגֵל הָאָרֶץ וְיֹאמְרוּ בַגּוֹיִם יְהוָה מָלָךְ:

[דברי הימים א פרק-כט-יח]  יְהוָה אֱלֹהֵי אַבְרָהָם יִצְחָק וְיִשְׂרָאֵל אֲבֹתֵינוּ שֳׁמְרָה זֹּאת לְעוֹלָם לְיֵצֶר מַחְשְׁבוֹת לְבַב עַמֶּךָ וְהָכֵן לְבָבָם אֵלֶיךָ:

[דברי הימים ב פרק-ו-מב]  יְהוָה אֱלֹהִים אַל תָּשֵׁב פְּנֵי מְשִׁיחֶיךָ זָכְרָה לְחַסְדֵי דָּוִיד עַבְדֶּךָ:


י-ל


[דברים פרק-א-לח]  יְהוֹשֻׁעַ בִּן נוּן הָעֹמֵד לְפָנֶיךָ הוּא יָבֹא שָׁמָּה אֹתוֹ חַזֵּק כִּי הוּא יַנְחִלֶנָּה אֶת יִשְׂרָאֵל:

[שופטים פרק-ה-ח]  יִבְחַר אֱלֹהִים חֲדָשִׁים אָז לָחֶם שְׁעָרִים מָגֵן אִם יֵרָאֶה וָרֹמַח בְּאַרְבָּעִים אֶלֶף בְּיִשְׂרָאֵל:

[שמואל א פרק-ב-ו]  יְהוָה מֵמִית וּמְחַיֶּה מוֹרִיד שְׁאוֹל וַיָּעַל:

[ישעיה פרק-כו-יט]  יִחְיוּ מֵתֶיךָ נְבֵלָתִי יְקוּמוּן הָקִיצוּ וְרַנְּנוּ שֹׁכְנֵי עָפָר כִּי טַל אוֹרֹת טַלֶּךָ וָאָרֶץ רְפָאִים תַּפִּיל:

[ירמיה פרק-טז-יט]  יְהוָה עֻזִּי וּמָעֻזִּי וּמְנוּסִי בְּיוֹם צָרָה אֵלֶיךָ גּוֹיִם יָבֹאוּ מֵאַפְסֵי אָרֶץ וְיֹאמְרוּ אַךְ שֶׁקֶר נָחֲלוּ אֲבוֹתֵינוּ הֶבֶל וְאֵין בָּם מוֹעִיל:

[יחזקאל פרק-יג-י]  יַעַן וּבְיַעַן הִטְעוּ אֶת עַמִּי לֵאמֹר שָׁלוֹם וְאֵין שָׁלוֹם וְהוּא בֹּנֶה חַיִץ וְהִנָּם טָחִים אֹתוֹ תָּפֵל:

[יואל פרק-א-ד]  יֶתֶר הַגָּזָם אָכַל הָאַרְבֶּה וְיֶתֶר הָאַרְבֶּה אָכַל הַיָּלֶק וְיֶתֶר הַיֶּלֶק אָכַל הֶחָסִיל:

[צפניה פרק-א-טו]  יוֹם עֶבְרָה הַיּוֹם הַהוּא יוֹם צָרָה וּמְצוּקָה יוֹם שֹׁאָה וּמְשׁוֹאָה יוֹם חֹשֶׁךְ וַאֲפֵלָה יוֹם עָנָן וַעֲרָפֶל:

[תהילים פרק-כב-יט]  יְחַלְּקוּ בְגָדַי לָהֶם וְעַל לְבוּשִׁי יַפִּילוּ גוֹרָל:

[תהילים פרק-כב-כד]  יִרְאֵי יְהוָה הַלְלוּהוּ כָּל זֶרַע יַעֲקֹב כַּבְּדוּהוּ וְגוּרוּ מִמֶּנּוּ כָּל זֶרַע יִשְׂרָאֵל:

[תהילים פרק-ל-ח]  יְהוָה בִּרְצוֹנְךָ הֶעֱמַדְתָּה לְהַרְרִי עֹז הִסְתַּרְתָּ פָנֶיךָ הָיִיתִי נִבְהָל:

[תהילים פרק-לא-יח]  יְהוָה אַל אֵבוֹשָׁה כִּי קְרָאתִיךָ יֵבֹשׁוּ רְשָׁעִים יִדְּמוּ לִשְׁאוֹל:

[תהילים פרק-לג-ח]  יִירְאוּ מֵיְהוָה כָּל הָאָרֶץ מִמֶּנּוּ יָגוּרוּ כָּל יֹשְׁבֵי תֵבֵל:

[תהילים פרק-עח-מז]  יַהֲרֹג בַּבָּרָד גַּפְנָם וְשִׁקְמוֹתָם בַּחֲנָמַל:

[תהילים פרק-צד-יא]  יְהוָה יֹדֵעַ מַחְשְׁבוֹת אָדָם כִּי הֵמָּה הָבֶל:

[תהילים פרק-קז-י]  יֹשְׁבֵי חֹשֶׁךְ וְצַלְמָוֶת אֲסִירֵי עֳנִי וּבַרְזֶל:

[תהילים פרק-קיח-טז]  יְמִין יְהוָה רוֹמֵמָה יְמִין יְהוָה עֹשָׂה חָיִל:

[משלי פרק-יז-כא]  יֹלֵד כְּסִיל לְתוּגָה לוֹ וְלֹא יִשְׂמַח אֲבִי נָבָל:

[איוב פרק-ה-ד]  יִרְחֲקוּ בָנָיו מִיֶּשַׁע וְיִדַּכְּאוּ בַשַּׁעַר וְאֵין מַצִּיל:

[איוב פרק-כז-ז]  יְהִי כְרָשָׁע אֹיְבִי וּמִתְקוֹמְמִי כְעַוָּל:

[איוב פרק-מ-כב]  יְסֻכֻּהוּ צֶאֱלִים צִלֲלוֹ יְסֻבּוּהוּ עַרְבֵי נָחַל:

[קהלת פרק-ח-יד]  יֶשׁ הֶבֶל אֲשֶׁר נַעֲשָׂה עַל הָאָרֶץ אֲשֶׁר יֵשׁ צַדִּיקִים אֲשֶׁר מַגִּיעַ אֲלֵהֶם כְּמַעֲשֵׂה הָרְשָׁעִים וְיֵשׁ רְשָׁעִים שֶׁמַּגִּיעַ אֲלֵהֶם כְּמַעֲשֵׂה הַצַּדִּיקִים אָמַרְתִּי שֶׁגַּם זֶה הָבֶל:

[דברי הימים א פרק-א-לא]  יְטוּר נָפִישׁ וָקֵדְמָה אֵלֶּה הֵם בְּנֵי יִשְׁמָעֵאל:

[דברי הימים א פרק-כט-טז]  יְהוָה אֱלֹהֵינוּ כֹל הֶהָמוֹן הַזֶּה אֲשֶׁר הֲכִינֹנוּ לִבְנוֹת לְךָ בַיִת לְשֵׁם קָדְשֶׁךָ מִיָּדְךָ (היא) הוּא וּלְךָ הַכֹּל:


י-מ


[בראשית פרק-כד-ז]  יְהוָה אֱלֹהֵי הַשָּׁמַיִם אֲשֶׁר לְקָחַנִי מִבֵּית אָבִי וּמֵאֶרֶץ מוֹלַדְתִּי וַאֲשֶׁר דִּבֶּר לִי וַאֲשֶׁר נִשְׁבַּע לִי לֵאמֹר לְזַרְעֲךָ אֶתֵּן אֶת הָאָרֶץ הַזֹּאת הוּא יִשְׁלַח מַלְאָכוֹ לְפָנֶיךָ וְלָקַחְתָּ אִשָּׁה לִבְנִי מִשָּׁם:

[בראשית פרק-מט-יד]  יִשָּׂשכָר חֲמֹר גָּרֶם רֹבֵץ בֵּין הַמִּשְׁפְּתָיִם:

[ויקרא פרק-י-ט]  יַיִן וְשֵׁכָר אַל תֵּשְׁתְּ אַתָּה וּבָנֶיךָ אִתָּךְ בְּבֹאֲכֶם אֶל אֹהֶל מוֹעֵד וְלֹא תָמֻתוּ חֻקַּת עוֹלָם לְדֹרֹתֵיכֶם:

[במדבר פרק-ו-כו]  יִשָּׂא יְהוָה פָּנָיו אֵלֶיךָ וְיָשֵׂם לְךָ שָׁלוֹם:

[במדבר פרק-יד-יח]  יְהוָה אֶרֶךְ אַפַּיִם וְרַב חֶסֶד נֹשֵׂא עָוֹן וָפָשַׁע וְנַקֵּה לֹא יְנַקֶּה פֹּקֵד עֲוֹן אָבוֹת עַל בָּנִים עַל שִׁלֵּשִׁים וְעַל רִבֵּעִים:

[דברים פרק-א-יא]  יְהוָה אֱלֹהֵי אֲבוֹתֵכֶם יֹסֵף עֲלֵיכֶם כָּכֶם אֶלֶף פְּעָמִים וִיבָרֵךְ אֶתְכֶם כַּאֲשֶׁר דִּבֶּר לָכֶם:

[דברים פרק-א-ל]  יְהוָה אֱלֹהֵיכֶם הַהֹלֵךְ לִפְנֵיכֶם הוּא יִלָּחֵם לָכֶם כְּכֹל אֲשֶׁר עָשָׂה אִתְּכֶם בְּמִצְרַיִם לְעֵינֵיכֶם:

[דברים פרק-יג-יד]  יָצְאוּ אֲנָשִׁים בְּנֵי בְלִיַּעַל מִקִּרְבֶּךָ וַיַּדִּיחוּ אֶת יֹשְׁבֵי עִירָם לֵאמֹר נֵלְכָה וְנַעַבְדָה אֱלֹהִים אֲחֵרִים אֲשֶׁר לֹא יְדַעְתֶּם:

[דברים פרק-לב-יז]  יִזְבְּחוּ לַשֵּׁדִים לֹא אֱלֹהַ אֱלֹהִים לֹא יְדָעוּם חֲדָשִׁים מִקָּרֹב בָּאוּ לֹא שְׂעָרוּם אֲבֹתֵיכֶם:

[יהושע פרק-כג-יג]  יָדוֹעַ תֵּדְעוּ כִּי לֹא יוֹסִיף יְהוָה אֱלֹהֵיכֶם לְהוֹרִישׁ אֶת הַגּוֹיִם הָאֵלֶּה מִלִּפְנֵיכֶם וְהָיוּ לָכֶם לְפַח וּלְמוֹקֵשׁ וּלְשֹׁטֵט בְּצִדֵּיכֶם וְלִצְנִנִים בְּעֵינֵיכֶם עַד אֲבָדְכֶם מֵעַל הָאֲדָמָה הַטּוֹבָה הַזֹּאת אֲשֶׁר נָתַן לָכֶם יְהוָה אֱלֹהֵיכֶם:

[שופטים פרק-ה-ד]  יְהוָה בְּצֵאתְךָ מִשֵּׂעִיר בְּצַעְדְּךָ מִשְּׂדֵה אֱדוֹם אֶרֶץ רָעָשָׁה גַּם שָׁמַיִם נָטָפוּ גַּם עָבִים נָטְפוּ מָיִם:

[שמואל א פרק-ב-ז]  יְהוָה מוֹרִישׁ וּמַעֲשִׁיר מַשְׁפִּיל אַף מְרוֹמֵם:

[שמואל ב פרק-ג-כט]  יָחֻלוּ עַל רֹאשׁ יוֹאָב וְאֶל כָּל בֵּית אָבִיו וְאַל יִכָּרֵת מִבֵּית יוֹאָב זָב וּמְצֹרָע וּמַחֲזִיק בַּפֶּלֶךְ וְנֹפֵל בַּחֶרֶב וַחֲסַר לָחֶם:

[שמואל ב פרק-כב-יז]  יִשְׁלַח מִמָּרוֹם יִקָּחֵנִי יַמְשֵׁנִי מִמַּיִם רַבִּים:

[שמואל ב פרק-כב-מב]  יִשְׁעוּ וְאֵין מֹשִׁיעַ אֶל יְהוָה וְלֹא עָנָם:

[מלכים א פרק-ד-כ]  יְהוּדָה וְיִשְׂרָאֵל רַבִּים כַּחוֹל אֲשֶׁר עַל הַיָּם לָרֹב אֹכְלִים וְשֹׁתִים וּשְׂמֵחִים:

[מלכים א פרק-טז-ב]  יַעַן אֲשֶׁר הֲרִימֹתִיךָ מִן הֶעָפָר וָאֶתֶּנְךָ נָגִיד עַל עַמִּי יִשְׂרָאֵל וַתֵּלֶךְ בְּדֶרֶךְ יָרָבְעָם וַתַּחֲטִא אֶת עַמִּי יִשְׂרָאֵל לְהַכְעִיסֵנִי בְּחַטֹּאתָם:

[ישעיה פרק-ג-ז]  יִשָּׂא בַיּוֹם הַהוּא לֵאמֹר לֹא אֶהְיֶה חֹבֵשׁ וּבְבֵיתִי אֵין לֶחֶם וְאֵין שִׂמְלָה לֹא תְשִׂימֻנִי קְצִין עָם:

[ישעיה פרק-ג-יד]  יְהוָה בְּמִשְׁפָּט יָבוֹא עִם זִקְנֵי עַמּוֹ וְשָׂרָיו וְאַתֶּם בִּעַרְתֶּם הַכֶּרֶם גְּזֵלַת הֶעָנִי בְּבָתֵּיכֶם:

[ישעיה פרק-כו-יא]  יְהוָה רָמָה יָדְךָ בַּל יֶחֱזָיוּן יֶחֱזוּ וְיֵבֹשׁוּ קִנְאַת עָם אַף אֵשׁ צָרֶיךָ תֹאכְלֵם:

[ישעיה פרק-מ-ז]  יָבֵשׁ חָצִיר נָבֵל צִיץ כִּי רוּחַ יְהוָה נָשְׁבָה בּוֹ אָכֵן חָצִיר הָעָם:

[ישעיה פרק-מ-ח]  יָבֵשׁ חָצִיר נָבֵל צִיץ וּדְבַר אֱלֹהֵינוּ יָקוּם לְעוֹלָם:

[ירמיה פרק-יז-יח]  יֵבֹשׁוּ רֹדְפַי וְאַל אֵבֹשָׁה אָנִי יֵחַתּוּ הֵמָּה וְאַל אֵחַתָּה אָנִי הָבִיא עֲלֵיהֶם יוֹם רָעָה וּמִשְׁנֶה שִׁבָּרוֹן שָׁבְרֵם:

[ירמיה פרק-יט-ד]  יַעַן אֲשֶׁר עֲזָבֻנִי וַיְנַכְּרוּ אֶת הַמָּקוֹם הַזֶּה וַיְקַטְּרוּ בוֹ לֵאלֹהִים אֲחֵרִים אֲשֶׁר לֹא יְדָעוּם הֵמָּה וַאֲבוֹתֵיהֶם וּמַלְכֵי יְהוּדָה וּמָלְאוּ אֶת הַמָּקוֹם הַזֶּה דַּם נְקִיִּם:

[יחזקאל פרק-כ-כד]  יַעַן מִשְׁפָּטַי לֹא עָשׂוּ וְחֻקּוֹתַי מָאָסוּ וְאֶת שַׁבְּתוֹתַי חִלֵּלוּ וְאַחֲרֵי גִּלּוּלֵי אֲבוֹתָם הָיוּ עֵינֵיהֶם:

[יחזקאל פרק-לא-ט]  יָפֶה עֲשִׂיתִיו בְּרֹב דָּלִיּוֹתָיו וַיְקַנְאֻהוּ כָּל עֲצֵי עֵדֶן אֲשֶׁר בְּגַן הָאֱלֹהִים:

[יחזקאל פרק-מד-יב]  יַעַן אֲשֶׁר יְשָׁרְתוּ אוֹתָם לִפְנֵי גִלּוּלֵיהֶם וְהָיוּ לְבֵית יִשְׂרָאֵל לְמִכְשׁוֹל עָוֹן עַל כֵּן נָשָׂאתִי יָדִי עֲלֵיהֶם נְאֻם אֲדֹנָי יְהוִה וְנָשְׂאוּ עֲוֹנָם:

[הושע פרק-ז-ה]  יוֹם מַלְכֵּנוּ הֶחֱלוּ שָׂרִים חֲמַת מִיָּיִן מָשַׁךְ יָדוֹ אֶת לֹצְצִים:

[הושע פרק-ז-טז]  יָשׁוּבוּ לֹא עָל הָיוּ כְּקֶשֶׁת רְמִיָּה יִפְּלוּ בַחֶרֶב שָׂרֵיהֶם מִזַּעַם לְשׁוֹנָם זוֹ לַעְגָּם בְּאֶרֶץ מִצְרָיִם:

[הושע פרק-ט-יז]  יִמְאָסֵם אֱלֹהַי כִּי לֹא שָׁמְעוּ לוֹ וְיִהְיוּ נֹדְדִים בַּגּוֹיִם:

[מיכה פרק-ז-יט]  יָשׁוּב יְרַחֲמֵנוּ יִכְבֹּשׁ עֲוֹנֹתֵינוּ וְתַשְׁלִיךְ בִּמְצֻלוֹת יָם כָּל חַטֹּאותָם:

[זכריה פרק-ד-ט]  יְדֵי זְרֻבָּבֶל יִסְּדוּ הַבַּיִת הַזֶּה וְיָדָיו תְּבַצַּעְנָה וְיָדַעְתָּ כִּי יְהוָה צְבָאוֹת שְׁלָחַנִי אֲלֵיכֶם:

[תהילים פרק-ח-ב]  יְהוָה אֲדֹנֵינוּ מָה אַדִּיר שִׁמְךָ בְּכָל הָאָרֶץ אֲשֶׁר תְּנָה הוֹדְךָ עַל הַשָּׁמָיִם:

[תהילים פרק-ט-יח]  יָשׁוּבוּ רְשָׁעִים לִשְׁאוֹלָה כָּל גּוֹיִם שְׁכֵחֵי אֱלֹהִים:

[תהילים פרק-י-ה]  יָחִילוּ (דרכו) דְרָכָיו בְּכָל עֵת מָרוֹם מִשְׁפָּטֶיךָ מִנֶּגְדּוֹ כָּל צוֹרְרָיו יָפִיחַ בָּהֶם:

[תהילים פרק-יא-ד]  יְהוָה בְּהֵיכַל קָדְשׁוֹ יְהוָה בַּשָּׁמַיִם כִּסְאוֹ עֵינָיו יֶחֱזוּ עַפְעַפָּיו יִבְחֲנוּ בְּנֵי אָדָם:

[תהילים פרק-יא-ו]  יַמְטֵר עַל רְשָׁעִים פַּחִים אֵשׁ וְגָפְרִית וְרוּחַ זִלְעָפוֹת מְנָת כּוֹסָם:

[תהילים פרק-יד-ב]  יְהוָה מִשָּׁמַיִם הִשְׁקִיף עַל בְּנֵי אָדָם לִרְאוֹת הֲיֵשׁ מַשְׂכִּיל דֹּרֵשׁ אֶת אֱלֹהִים:

[תהילים פרק-יח-יב]  יָשֶׁת חֹשֶׁךְ סִתְרוֹ סְבִיבוֹתָיו סֻכָּתוֹ חֶשְׁכַת מַיִם עָבֵי שְׁחָקִים:

[תהילים פרק-יח-יז]  יִשְׁלַח מִמָּרוֹם יִקָּחֵנִי יַמְשֵׁנִי מִמַּיִם רַבִּים:

[תהילים פרק-יח-מב]  יְשַׁוְּעוּ וְאֵין מוֹשִׁיעַ עַל יְהוָה וְלֹא עָנָם:

[תהילים פרק-כב-כח]  יִזְכְּרוּ וְיָשֻׁבוּ אֶל יְהוָה כָּל אַפְסֵי אָרֶץ וְיִשְׁתַּחֲווּ לְפָנֶיךָ כָּל מִשְׁפְּחוֹת גּוֹיִם:

[תהילים פרק-כט-י]  יְהוָה לַמַּבּוּל יָשָׁב וַיֵּשֶׁב יְהוָה מֶלֶךְ לְעוֹלָם:

[תהילים פרק-כט-יא]  יְהוָה עֹז לְעַמּוֹ יִתֵּן יְהוָה יְבָרֵךְ אֶת עַמּוֹ בַשָּׁלוֹם:

[תהילים פרק-לג-י]  יְהוָה הֵפִיר עֲצַת גּוֹיִם הֵנִיא מַחְשְׁבוֹת עַמִּים:

[תהילים פרק-לה-ו]  יְהִי דַרְכָּם חֹשֶׁךְ וַחֲלַקְלַקּוֹת וּמַלְאַךְ יְהוָה רֹדְפָם:

[תהילים פרק-לו-ו]  יְהוָה בְּהַשָּׁמַיִם חַסְדֶּךָ אֱמוּנָתְךָ עַד שְׁחָקִים:

[תהילים פרק-לו-ט]  יִרְוְיֻן מִדֶּשֶׁן בֵּיתֶךָ וְנַחַל עֲדָנֶיךָ תַשְׁקֵם:

[תהילים פרק-מה-ג]  יָפְיָפִיתָ מִבְּנֵי אָדָם הוּצַק חֵן בְּשְׂפְתוֹתֶיךָ עַל כֵּן בֵּרַכְךָ אֱלֹהִים לְעוֹלָם:

[תהילים פרק-נד-ז]  (ישוב) יָשִׁיב הָרַע לְשֹׁרְרָי בַּאֲמִתְּךָ הַצְמִיתֵם:

[תהילים פרק-נה-טז]  (ישימות) יַשִּׁי מָוֶת עָלֵימוֹ יֵרְדוּ שְׁאוֹל חַיִּים כִּי רָעוֹת בִּמְגוּרָם בְּקִרְבָּם:

[תהילים פרק-נה-כ]  יִשְׁמַע אֵל וְיַעֲנֵם וְיֹשֵׁב קֶדֶם סֶלָה אֲשֶׁר אֵין חֲלִיפוֹת לָמוֹ וְלֹא יָרְאוּ אֱלֹהִים:

[תהילים פרק-סז-ד]  יוֹדוּךָ עַמִּים אֱלֹהִים יוֹדוּךָ עַמִּים כֻּלָּם:

[תהילים פרק-סז-ו]  יוֹדוּךָ עַמִּים אֱלֹהִים יוֹדוּךָ עַמִּים כֻּלָּם:

[תהילים פרק-סח-לב]  יֶאֱתָיוּ חַשְׁמַנִּים מִנִּי מִצְרָיִם כּוּשׁ תָּרִיץ יָדָיו לֵאלֹהִים:

[תהילים פרק-סט-לה]  יְהַלְלוּהוּ שָׁמַיִם וָאָרֶץ יַמִּים וְכָל רֹמֵשׂ בָּם:

[תהילים פרק-עב-ה]  יִירָאוּךָ עִם שָׁמֶשׁ וְלִפְנֵי יָרֵחַ דּוֹר דּוֹרִים:

[תהילים פרק-עח-מה]  יְשַׁלַּח בָּהֶם עָרֹב וַיֹּאכְלֵם וּצְפַרְדֵּעַ וַתַּשְׁחִיתֵם:

[תהילים פרק-עח-מט]  יְשַׁלַּח בָּם חֲרוֹן אַפּוֹ עֶבְרָה וָזַעַם וְצָרָה מִשְׁלַחַת מַלְאֲכֵי רָעִים:

[תהילים פרק-צ-טז]  יֵרָאֶה אֶל עֲבָדֶיךָ פָעֳלֶךָ וַהֲדָרְךָ עַל בְּנֵיהֶם:

[תהילים פרק-צז-א]  יְהוָה מָלָךְ תָּגֵל הָאָרֶץ יִשְׂמְחוּ אִיִּים רַבִּים:

[תהילים פרק-צז-ז]  יֵבֹשׁוּ כָּל עֹבְדֵי פֶסֶל הַמִּתְהַלְלִים בָּאֱלִילִים הִשְׁתַּחֲווּ לוֹ כָּל אֱלֹהִים:

[תהילים פרק-צט-ב]  יְהוָה בְּצִיּוֹן גָּדוֹל וְרָם הוּא עַל כָּל הָעַמִּים:

[תהילים פרק-צט-ח]  יְהוָה אֱלֹהֵינוּ אַתָּה עֲנִיתָם אֵל נֹשֵׂא הָיִיתָ לָהֶם וְנֹקֵם עַל עֲלִילוֹתָם:

[תהילים פרק-קד-ח]  יַעֲלוּ הָרִים יֵרְדוּ בְקָעוֹת אֶל מְקוֹם זֶה יָסַדְתָּ לָהֶם:

[תהילים פרק-קד-יא]  יַשְׁקוּ כָּל חַיְתוֹ שָׂדָי יִשְׁבְּרוּ פְרָאִים צְמָאָם:

[תהילים פרק-קז-ח]  יוֹדוּ לַיהוָה חַסְדּוֹ וְנִפְלְאוֹתָיו לִבְנֵי אָדָם:

[תהילים פרק-קז-טו]  יוֹדוּ לַיהוָה חַסְדּוֹ וְנִפְלְאוֹתָיו לִבְנֵי אָדָם:

[תהילים פרק-קז-כ]  יִשְׁלַח דְּבָרוֹ וְיִרְפָּאֵם וִימַלֵּט מִשְּׁחִיתוֹתָם:

[תהילים פרק-קז-כא]  יוֹדוּ לַיהוָה חַסְדּוֹ וְנִפְלְאוֹתָיו לִבְנֵי אָדָם:

[תהילים פרק-קז-כג]  יוֹרְדֵי הַיָּם בָּאֳנִיּוֹת עֹשֵׂי מְלָאכָה בְּמַיִם רַבִּים:

[תהילים פרק-קז-כט]  יָקֵם סְעָרָה לִדְמָמָה וַיֶּחֱשׁוּ גַּלֵּיהֶם:

[תהילים פרק-קז-לא]  יוֹדוּ לַיהוָה חַסְדּוֹ וְנִפְלְאוֹתָיו לִבְנֵי אָדָם:

[תהילים פרק-קז-לה]  יָשֵׂם מִדְבָּר לַאֲגַם מַיִם וְאֶרֶץ צִיָּה לְמֹצָאֵי מָיִם:

[תהילים פרק-קט-יג]  יְהִי אַחֲרִיתוֹ לְהַכְרִית בְּדוֹר אַחֵר יִמַּח שְׁמָם:

[תהילים פרק-קט-טו]  יִהְיוּ נֶגֶד יְהוָה תָּמִיד וְיַכְרֵת מֵאֶרֶץ זִכְרָם:

[תהילים פרק-קט-כט]  יִלְבְּשׁוּ שׂוֹטְנַי כְּלִמָּה וְיַעֲטוּ כַמְעִיל בָּשְׁתָּם:

[תהילים פרק-קיג-ב]  יְהִי שֵׁם יְהוָה מְבֹרָךְ מֵעַתָּה וְעַד עוֹלָם:

[תהילים פרק-קטו-ז]  יְדֵיהֶם וְלֹא יְמִישׁוּן רַגְלֵיהֶם וְלֹא יְהַלֵּכוּ לֹא יֶהְגּוּ בִּגְרוֹנָם:

[תהילים פרק-קטו-יג]  יְבָרֵךְ יִרְאֵי יְהוָה הַקְּטַנִּים עִם הַגְּדֹלִים:

[תהילים פרק-קטו-יד]  יֹסֵף יְהוָה עֲלֵיכֶם עֲלֵיכֶם וְעַל בְּנֵיכֶם:

[תהילים פרק-קיח-ו]  יְהוָה לִי לֹא אִירָא מַה יַּעֲשֶׂה לִי אָדָם:

[תהילים פרק-קכא-ח]  יְהוָה יִשְׁמָר צֵאתְךָ וּבוֹאֶךָ מֵעַתָּה וְעַד עוֹלָם:

[תהילים פרק-קכה-ב]  יְרוּשָׁלִַם הָרִים סָבִיב לָהּ וַיהוָה סָבִיב לְעַמּוֹ מֵעַתָּה וְעַד עוֹלָם:

[תהילים פרק-קכט-ד]  יְהוָה צַדִּיק קִצֵּץ עֲבוֹת רְשָׁעִים:

[תהילים פרק-קלא-ג]  יַחֵל יִשְׂרָאֵל אֶל יְהוָה מֵעַתָּה וְעַד עוֹלָם:

[תהילים פרק-קמ-יג]  (ידעת) יָדַעְתִּי כִּי יַעֲשֶׂה יְהוָה דִּין עָנִי מִשְׁפַּט אֶבְיֹנִים:

[תהילים פרק-קמא-ה]  יֶהֶלְמֵנִי צַדִּיק חֶסֶד וְיוֹכִיחֵנִי שֶׁמֶן רֹאשׁ אַל יָנִי רֹאשִׁי כִּי עוֹד וּתְפִלָּתִי בְּרָעוֹתֵיהֶם:

[תהילים פרק-קמו-ח]  יְהוָה פֹּקֵחַ עִוְרִים יְהוָה זֹקֵף כְּפוּפִים יְהוָה אֹהֵב צַדִּיקִים:

[תהילים פרק-קמז-יח]  יִשְׁלַח דְּבָרוֹ וְיַמְסֵם יַשֵּׁב רוּחוֹ יִזְּלוּ מָיִם:

[תהילים פרק-קמח-יג]  יְהַלְלוּ אֶת שֵׁם יְהוָה כִּי נִשְׂגָּב שְׁמוֹ לְבַדּוֹ הוֹדוֹ עַל אֶרֶץ וְשָׁמָיִם:

[תהילים פרק-קמט-ב]  יִשְׂמַח יִשְׂרָאֵל בְּעֹשָׂיו בְּנֵי צִיּוֹן יָגִילוּ בְמַלְכָּם:

[תהילים פרק-קמט-ה]  יַעְלְזוּ חֲסִידִים בְּכָבוֹד יְרַנְּנוּ עַל מִשְׁכְּבוֹתָם:

[משלי פרק-ה-טז]  יָפוּצוּ מַעְיְנֹתֶיךָ חוּצָה בָּרְחֹבוֹת פַּלְגֵי מָיִם:

[משלי פרק-ו-יט]  יָפִיחַ כְּזָבִים עֵד שָׁקֶר וּמְשַׁלֵּחַ מְדָנִים בֵּין אַחִים:

[משלי פרק-יב-כו]  יָתֵר מֵרֵעֵהוּ צַדִּיק וְדֶרֶךְ רְשָׁעִים תַּתְעֵם:

[משלי פרק-כז-כג]  יָדֹעַ תֵּדַע פְּנֵי צֹאנֶךָ שִׁית לִבְּךָ לַעֲדָרִים:

[איוב פרק-ג-ה]  יִגְאָלֻהוּ חֹשֶׁךְ וְצַלְמָוֶת תִּשְׁכָּן עָלָיו עֲנָנָה יְבַעֲתֻהוּ כִּמְרִירֵי יוֹם:

[איוב פרק-ה-יד]  יוֹמָם יְפַגְּשׁוּ חֹשֶׁךְ וְכַלַּיְלָה יְמַשְׁשׁוּ בַצָּהֳרָיִם:

[איוב פרק-ח-טו]  יִשָּׁעֵן עַל בֵּיתוֹ וְלֹא יַעֲמֹד יַחֲזִיק בּוֹ וְלֹא יָקוּם:

[איוב פרק-יח-ט]  יֹאחֵז בְּעָקֵב פָּח יַחֲזֵק עָלָיו צַמִּים:

[איוב פרק-כא-כו]  יַחַד עַל עָפָר יִשְׁכָּבוּ וְרִמָּה תְּכַסֶּה עֲלֵיהֶם:

[איוב פרק-כד-כג]  יִתֶּן לוֹ לָבֶטַח וְיִשָּׁעֵן וְעֵינֵיהוּ עַל דַּרְכֵיהֶם:

[איוב פרק-לד-כד]  יָרֹעַ כַּבִּירִים לֹא חֵקֶר וַיַּעֲמֵד אֲחֵרִים תַּחְתָּם:

[איוב פרק-לו-טו]  יְחַלֵּץ עָנִי בְעָנְיוֹ וְיִגֶל בַּלַּחַץ אָזְנָם:

[איוב פרק-לח-כא]  יָדַעְתָּ כִּי אָז תִּוָּלֵד וּמִסְפַּר יָמֶיךָ רַבִּים:

[איוב פרק-מ-ל]  יִכְרוּ עָלָיו חַבָּרִים יֶחֱצוּהוּ בֵּין כְּנַעֲנִים:

[שיר השירים פרק-ה-יד]  יָדָיו גְּלִילֵי זָהָב מְמֻלָּאִים בַּתַּרְשִׁישׁ מֵעָיו עֶשֶׁת שֵׁן מְעֻלֶּפֶת סַפִּירִים:

[איכה פרק-ב-י]  יֵשְׁבוּ לָאָרֶץ יִדְּמוּ זִקְנֵי בַת צִיּוֹן הֶעֱלוּ עָפָר עַל רֹאשָׁם חָגְרוּ שַׂקִּים הוֹרִידוּ לָאָרֶץ רֹאשָׁן בְּתוּלֹת יְרוּשָׁלִָם:

[קהלת פרק-ו-א]  יֵשׁ רָעָה אֲשֶׁר רָאִיתִי תַּחַת הַשָּׁמֶשׁ וְרַבָּה הִיא עַל הָאָדָם:

[דניאל פרק-א-ד]  יְלָדִים אֲשֶׁר אֵין בָּהֶם כָּל (מאום) מוּם וְטוֹבֵי מַרְאֶה וּמַשְׂכִּילִים בְּכָל חָכְמָה וְיֹדְעֵי דַעַת וּמְבִינֵי מַדָּע וַאֲשֶׁר כֹּחַ בָּהֶם לַעֲמֹד בְּהֵיכַל הַמֶּלֶךְ וּלֲלַמְּדָם סֵפֶר וּלְשׁוֹן כַּשְׂדִּים:

[עזרא פרק-ה-ח]  יְדִיעַ לֶהֱוֵא לְמַלְכָּא דִּי אֲזַלְנָא לִיהוּד מְדִינְתָּא לְבֵית אֱלָהָא רַבָּא וְהוּא מִתְבְּנֵא אֶבֶן גְּלָל וְאָע מִתְּשָׂם בְּכֻתְלַיָּא וַעֲבִידְתָּא דָךְ אָסְפַּרְנָא מִתְעַבְדָא וּמַצְלַח בְּיֶדְהֹם:


י-נ


[שמות פרק-א-ג]  יִשָּׂשכָר זְבוּלֻן וּבְנְיָמִן:

[שמות פרק-יד-יד]  יְהוָה יִלָּחֵם לָכֶם וְאַתֶּם תַּחֲרִישׁוּן:

[דברים פרק-ד-י]  יוֹם אֲשֶׁר עָמַדְתָּ לִפְנֵי יְהוָה אֱלֹהֶיךָ בְּחֹרֵב בֶּאֱמֹר יְהוָה אֵלַי הַקְהֶל לִי אֶת הָעָם וְאַשְׁמִעֵם אֶת דְּבָרָי אֲשֶׁר יִלְמְדוּן לְיִרְאָה אֹתִי כָּל הַיָּמִים אֲשֶׁר הֵם חַיִּים עַל הָאֲדָמָה וְאֶת בְּנֵיהֶם יְלַמֵּדוּן:

[דברים פרק-כח-לו]  יוֹלֵךְ יְהוָה אֹתְךָ וְאֶת מַלְכְּךָ אֲשֶׁר תָּקִים עָלֶיךָ אֶל גּוֹי אֲשֶׁר לֹא יָדַעְתָּ אַתָּה וַאֲבֹתֶיךָ וְעָבַדְתָּ שָּׁם אֱלֹהִים אֲחֵרִים עֵץ וָאָבֶן:

[שמואל ב פרק-כא-ו]  (ינתן) יֻתַּן לָנוּ שִׁבְעָה אֲנָשִׁים מִבָּנָיו וְהוֹקַעֲנוּם לַיהוָה בְּגִבְעַת שָׁאוּל בְּחִיר יְהוָה וַיֹּאמֶר הַמֶּלֶךְ אֲנִי אֶתֵּן:

[ישעיה פרק-א-ג]  יָדַע שׁוֹר קֹנֵהוּ וַחֲמוֹר אֵבוּס בְּעָלָיו יִשְׂרָאֵל לֹא יָדַע עַמִּי לֹא הִתְבּוֹנָן:

[ישעיה פרק-כה-א]  יְהוָה אֱלֹהַי אַתָּה אֲרוֹמִמְךָ אוֹדֶה שִׁמְךָ כִּי עָשִׂיתָ פֶּלֶא עֵצוֹת מֵרָחוֹק אֱמוּנָה אֹמֶן:

[יחזקאל פרק-כא-ט]  יַעַן אֲשֶׁר הִכְרַתִּי מִמֵּךְ צַדִּיק וְרָשָׁע לָכֵן תֵּצֵא חַרְבִּי מִתַּעְרָהּ אֶל כָּל בָּשָׂר מִנֶּגֶב צָפוֹן:

[הושע פרק-יד-ז]  יֵלְכוּ יֹנְקוֹתָיו וִיהִי כַזַּיִת הוֹדוֹ וְרֵיחַ לוֹ כַּלְּבָנוֹן:

[הושע פרק-יד-ח]  יָשֻׁבוּ יֹשְׁבֵי בְצִלּוֹ יְחַיּוּ דָגָן וְיִפְרְחוּ כַגָּפֶן זִכְרוֹ כְּיֵין לְבָנוֹן:

[תהילים פרק-עח-כו]  יַסַּע קָדִים בַּשָּׁמָיִם וַיְנַהֵג בְּעֻזּוֹ תֵימָן:

[תהילים פרק-פד-ח]  יֵלְכוּ מֵחַיִל אֶל חָיִל יֵרָאֶה אֶל אֱלֹהִים בְּצִיּוֹן:

[תהילים פרק-צא-א]  יֹשֵׁב בְּסֵתֶר עֶלְיוֹן בְּצֵל שַׁדַּי יִתְלוֹנָן:

[תהילים פרק-צד-ד]  יַבִּיעוּ יְדַבְּרוּ עָתָק יִתְאַמְּרוּ כָּל פֹּעֲלֵי אָוֶן:

[תהילים פרק-קז-לג]  יָשֵׂם נְהָרוֹת לְמִדְבָּר וּמֹצָאֵי מַיִם לְצִמָּאוֹן:

[תהילים פרק-קטו-יב]  יְהוָה זְכָרָנוּ יְבָרֵךְ יְבָרֵךְ אֶת בֵּית יִשְׂרָאֵל יְבָרֵךְ אֶת בֵּית אַהֲרֹן:

[תהילים פרק-קכט-ה]  יֵבֹשׁוּ וְיִסֹּגוּ אָחוֹר כֹּל שֹׂנְאֵי צִיּוֹן:

[איוב פרק-ג-ח]  יִקְּבֻהוּ אֹרְרֵי יוֹם הָעֲתִידִים עֹרֵר לִוְיָתָן:

[איוב פרק-כא-יא]  יְשַׁלְּחוּ כַצֹּאן עֲוִילֵיהֶם וְיַלְדֵיהֶם יְרַקֵּדוּן:

[שיר השירים פרק-א-ב]  יִשָּׁקֵנִי מִנְּשִׁיקוֹת פִּיהוּ כִּי טוֹבִים דֹּדֶיךָ מִיָּיִן:

[דברי הימים א פרק-ה-יב]  יוֹאֵל הָרֹאשׁ וְשָׁפָם הַמִּשְׁנֶה וְיַעְנַי וְשָׁפָט בַּבָּשָׁן:


י-ס


[משלי פרק-יב-כד]  יַד חָרוּצִים תִּמְשׁוֹל וּרְמִיָּה תִּהְיֶה לָמַס:


י-ע


[בראשית פרק-לא-כט]  יֶשׁ לְאֵל יָדִי לַעֲשׂוֹת עִמָּכֶם רָע וֵאלֹהֵי אֲבִיכֶם אֶמֶשׁ אָמַר אֵלַי לֵאמֹר הִשָּׁמֶר לְךָ מִדַּבֵּר עִם יַעֲקֹב מִטּוֹב עַד רָע:

[בראשית פרק-מא-לד]  יַעֲשֶׂה פַרְעֹה וְיַפְקֵד פְּקִדִים עַל הָאָרֶץ וְחִמֵּשׁ אֶת אֶרֶץ מִצְרַיִם בְּשֶׁבַע שְׁנֵי הַשָּׂבָע:

[תהילים פרק-ו-יא]  יֵבֹשׁוּ וְיִבָּהֲלוּ מְאֹד כָּל אֹיְבָי יָשֻׁבוּ יֵבֹשׁוּ רָגַע:

[תהילים פרק-נח-יא]  יִשְׂמַח צַדִּיק כִּי חָזָה נָקָם פְּעָמָיו יִרְחַץ בְּדַם הָרָשָׁע:

[תהילים פרק-עב-יג]  יָחֹס עַל דַּל וְאֶבְיוֹן וְנַפְשׁוֹת אֶבְיוֹנִים יוֹשִׁיעַ:

[תהילים פרק-קד-טז]  יִשְׂבְּעוּ עֲצֵי יְהוָה אַרְזֵי לְבָנוֹן אֲשֶׁר נָטָע:

[תהילים פרק-קז-כז]  יָחוֹגּוּ וְיָנוּעוּ כַּשִּׁכּוֹר וְכָל חָכְמָתָם תִּתְבַּלָּע:

[משלי פרק-יט-כג]  יִרְאַת יְהוָה לְחַיִּים וְשָׂבֵעַ יָלִין בַּל יִפָּקֶד רָע:

[איוב פרק-ד-טז]  יַעֲמֹד וְלֹא אַכִּיר מַרְאֵהוּ תְּמוּנָה לְנֶגֶד עֵינָי דְּמָמָה וָקוֹל אֶשְׁמָע:

[איוב פרק-לז-ה]  יַרְעֵם אֵל בְּקוֹלוֹ נִפְלָאוֹת עֹשֶׂה גְדֹלוֹת וְלֹא נֵדָע:

[איוב פרק-לט-ז]  יִשְׂחַק לַהֲמוֹן קִרְיָה תְּשֻׁאוֹת נוֹגֵשׂ לֹא יִשְׁמָע:


י-פ


[ישעיה פרק-יח-ו]  יֵעָזְבוּ יַחְדָּו לְעֵיט הָרִים וּלְבֶהֱמַת הָאָרֶץ וְקָץ עָלָיו הָעַיִט וְכָל בֶּהֱמַת הָאָרֶץ עָלָיו תֶּחֱרָף:

[תהילים פרק-קלח-ח]  יְהוָה יִגְמֹר בַּעֲדִי יְהוָה חַסְדְּךָ לְעוֹלָם מַעֲשֵׂי יָדֶיךָ אַל תֶּרֶף:


י-צ


[בראשית פרק-יח-ד]  יֻקַּח נָא מְעַט מַיִם וְרַחֲצוּ רַגְלֵיכֶם וְהִשָּׁעֲנוּ תַּחַת הָעֵץ:

[דברים פרק-כח-כה]  יִתֶּנְךָ יְהוָה נִגָּף לִפְנֵי אֹיְבֶיךָ בְּדֶרֶךְ אֶחָד תֵּצֵא אֵלָיו וּבְשִׁבְעָה דְרָכִים תָּנוּס לְפָנָיו וְהָיִיתָ לְזַעֲוָה לְכֹל מַמְלְכוֹת הָאָרֶץ:

[ישעיה פרק-טז-ד]  יָגוּרוּ בָךְ נִדָּחַי מוֹאָב הֱוִי סֵתֶר לָמוֹ מִפְּנֵי שׁוֹדֵד כִּי אָפֵס הַמֵּץ כָּלָה שֹׁד תַּמּוּ רֹמֵס מִן הָאָרֶץ:

[ישעיה פרק-כג-ט]  יְהוָה צְבָאוֹת יְעָצָהּ לְחַלֵּל גְּאוֹן כָּל צְבִי לְהָקֵל כָּל נִכְבַּדֵּי אָרֶץ:

[ישעיה פרק-כו-טו]  יָסַפְתָּ לַגּוֹי יְהוָה יָסַפְתָּ לַגּוֹי נִכְבָּדְתָּ רִחַקְתָּ כָּל קַצְוֵי אָרֶץ:

[ישעיה פרק-לז-טז]  יְהוָה צְבָאוֹת אֱלֹהֵי יִשְׂרָאֵל יֹשֵׁב הַכְּרֻבִים אַתָּה הוּא הָאֱלֹהִים לְבַדְּךָ לְכֹל מַמְלְכוֹת הָאָרֶץ אַתָּה עָשִׂיתָ אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ:

[יחזקאל פרק-לה-ה]  יַעַן הֱיוֹת לְךָ אֵיבַת עוֹלָם וַתַּגֵּר אֶת בְּנֵי יִשְׂרָאֵל עַל יְדֵי חָרֶב בְּעֵת אֵידָם בְּעֵת עֲוֹן קֵץ:

[תהילים פרק-ח-י]  יְהוָה אֲדֹנֵינוּ מָה אַדִּיר שִׁמְךָ בְּכָל הָאָרֶץ:

[תהילים פרק-סז-ח]  יְבָרְכֵנוּ אֱלֹהִים וְיִירְאוּ אֹתוֹ כָּל אַפְסֵי אָרֶץ:

[תהילים פרק-עב-ו]  יֵרֵד כְּמָטָר עַל גֵּז כִּרְבִיבִים זַרְזִיף אָרֶץ:

[תהילים פרק-עב-טז]  יְהִי פִסַּת בַּר בָּאָרֶץ בְּרֹאשׁ הָרִים יִרְעַשׁ כַּלְּבָנוֹן פִּרְיוֹ וְיָצִיצוּ מֵעִיר כְּעֵשֶׂב הָאָרֶץ:

[תהילים פרק-עו-יג]  יִבְצֹר רוּחַ נְגִידִים נוֹרָא לְמַלְכֵי אָרֶץ:

[תהילים פרק-צט-א]  יְהוָה מָלָךְ יִרְגְּזוּ עַמִּים יֹשֵׁב כְּרוּבִים תָּנוּט הָאָרֶץ:

[תהילים פרק-קלד-ג]  יְבָרֶכְךָ יְהוָה מִצִּיּוֹן עֹשֵׂה שָׁמַיִם וָאָרֶץ:

[איוב פרק-כד-ד]  יַטּוּ אֶבְיוֹנִים מִדָּרֶךְ יַחַד חֻבְּאוּ עֲנִיֵּי אָרֶץ:


י-ק


[מלכים ב פרק-יב-ו]  יִקְחוּ לָהֶם הַכֹּהֲנִים אִישׁ מֵאֵת מַכָּרוֹ וְהֵם יְחַזְּקוּ אֶת בֶּדֶק הַבַּיִת לְכֹל אֲשֶׁר יִמָּצֵא שָׁם בָּדֶק:

[ירמיה פרק-כט-כו]  יְהוָה נְתָנְךָ כֹהֵן תַּחַת יְהוֹיָדָע הַכֹּהֵן לִהְיוֹת פְּקִדִים בֵּית יְהוָה לְכָל אִישׁ מְשֻׁגָּע וּמִתְנַבֵּא וְנָתַתָּה אֹתוֹ אֶל הַמַּהְפֶּכֶת וְאֶל הַצִּינֹק:

[מיכה פרק-ז-יא]  יוֹם לִבְנוֹת גְּדֵרָיִךְ יוֹם הַהוּא יִרְחַק חֹק:

[תהילים פרק-ז-י]  יִגְמָר נָא רַע רְשָׁעִים וּתְכוֹנֵן צַדִּיק וּבֹחֵן לִבּוֹת וּכְלָיוֹת אֱלֹהִים צַדִּיק:

[תהילים פרק-סד-ז]  יַחְפְּשׂוּ עוֹלֹת תַּמְנוּ חֵפֶשׂ מְחֻפָּשׂ וְקֶרֶב אִישׁ וְלֵב עָמֹק:

[תהילים פרק-עב-ד]  יִשְׁפֹּט עֲנִיֵּי עָם יוֹשִׁיעַ לִבְנֵי אֶבְיוֹן וִידַכֵּא עוֹשֵׁק:

[תהילים פרק-קז-יד]  יוֹצִיאֵם מֵחֹשֶׁךְ וְצַלְמָוֶת וּמוֹסְרוֹתֵיהֶם יְנַתֵּק:

[תהילים פרק-קמ-ח]  יְהוִֹה אֲדֹנָי עֹז יְשׁוּעָתִי סַכֹּתָה לְרֹאשִׁי בְּיוֹם נָשֶׁק:

[איוב פרק-כז-יז]  יָכִין וְצַדִּיק יִלְבָּשׁ וְכֶסֶף נָקִי יַחֲלֹק:

[איוב פרק-לט-כא]  יַחְפְּרוּ בָעֵמֶק וְיָשִׂישׂ בְּכֹחַ יֵצֵא לִקְרַאת נָשֶׁק:


י-ר


[בראשית פרק-מט-יז]  יְהִי דָן נָחָשׁ עֲלֵי דֶרֶךְ שְׁפִיפֹן עֲלֵי אֹרַח הַנֹּשֵׁךְ עִקְּבֵי סוּס וַיִּפֹּל רֹכְבוֹ אָחוֹר:

[דברים פרק-לב-יב]  יְהוָה בָּדָד יַנְחֶנּוּ וְאֵין עִמּוֹ אֵל נֵכָר:

[דברים פרק-לב-יג]  יַרְכִּבֵהוּ עַל (במותי) בָּמֳתֵי אָרֶץ וַיֹּאכַל תְּנוּבֹת שָׂדָי וַיֵּנִקֵהוּ דְבַשׁ מִסֶּלַע וְשֶׁמֶן מֵחַלְמִישׁ צוּר:

[דברים פרק-לג-ו]  יְחִי רְאוּבֵן וְאַל יָמֹת וִיהִי מְתָיו מִסְפָּר:

[מלכים א פרק-ד-יז]  יְהוֹשָׁפָט בֶּן פָּרוּחַ בְּיִשָׂשכָר:

[מלכים א פרק-כב-מט]  יְהוֹשָׁפָט (עשר) עָשָׂה אֳנִיּוֹת תַּרְשִׁישׁ לָלֶכֶת אוֹפִירָה לַזָּהָב וְלֹא הָלָךְ כִּי (נשברה) נִשְׁבְּרוּ אֳנִיּוֹת בְּעֶצְיוֹן גָּבֶר:

[ישעיה פרק-ז-ה]  יַעַן כִּי יָעַץ עָלֶיךָ אֲרָם רָעָה אֶפְרַיִם וּבֶן רְמַלְיָהוּ לֵאמֹר:

[ישעיה פרק-ז-יז]  יָבִיא יְהוָה עָלֶיךָ וְעַל עַמְּךָ וְעַל בֵּית אָבִיךָ יָמִים אֲשֶׁר לֹא בָאוּ לְמִיּוֹם סוּר אֶפְרַיִם מֵעַל יְהוּדָה אֵת מֶלֶךְ אַשּׁוּר:

[ישעיה פרק-מב-יג]  יְהוָה כַּגִּבּוֹר יֵצֵא כְּאִישׁ מִלְחָמוֹת יָעִיר קִנְאָה יָרִיעַ אַף יַצְרִיחַ עַל אֹיְבָיו יִתְגַּבָּר:

[ישעיה פרק-מב-כא]  יְהוָה חָפֵץ לְמַעַן צִדְקוֹ יַגְדִּיל תּוֹרָה וְיַאְדִּיר:

[יואל פרק-ב-ב]  יוֹם חֹשֶׁךְ וַאֲפֵלָה יוֹם עָנָן וַעֲרָפֶל כְּשַׁחַר פָּרֻשׂ עַל הֶהָרִים עַם רַב וְעָצוּם כָּמֹהוּ לֹא נִהְיָה מִן הָעוֹלָם וְאַחֲרָיו לֹא יוֹסֵף עַד שְׁנֵי דּוֹר וָדוֹר:

[מיכה פרק-ז-יב]  יוֹם הוּא וְעָדֶיךָ יָבוֹא לְמִנִּי אַשּׁוּר וְעָרֵי מָצוֹר וּלְמִנִּי מָצוֹר וְעַד נָהָר וְיָם מִיָּם וְהַר הָהָר:

[חבקוק פרק-ג-ב]  יְהוָה שָׁמַעְתִּי שִׁמְעֲךָ יָרֵאתִי יְהוָה פָּעָלְךָ בְּקֶרֶב שָׁנִים חַיֵּיהוּ בְּקֶרֶב שָׁנִים תּוֹדִיעַ בְּרֹגֶז רַחֵם תִּזְכּוֹר:

[תהילים פרק-ל-ד]  יְהוָה הֶעֱלִיתָ מִן שְׁאוֹל נַפְשִׁי חִיִּיתַנִי (מיורדי) מִיָּרְדִי בוֹר:

[תהילים פרק-נט-ז]  יָשׁוּבוּ לָעֶרֶב יֶהֱמוּ כַכָּלֶב וִיסוֹבְבוּ עִיר:

[תהילים פרק-סא-ז]  יָמִים עַל יְמֵי מֶלֶךְ תּוֹסִיף שְׁנוֹתָיו כְּמוֹ דֹר וָדֹר:

[תהילים פרק-סט-יג]  יָשִׂיחוּ בִי יֹשְׁבֵי שָׁעַר וּנְגִינוֹת שׁוֹתֵי שֵׁכָר:

[תהילים פרק-עח-נ]  יְפַלֵּס נָתִיב לְאַפּוֹ לֹא חָשַׂךְ מִמָּוֶת נַפְשָׁם וְחַיָּתָם לַדֶּבֶר הִסְגִּיר:

[תהילים פרק-צו-יב]  יַעֲלֹז שָׂדַי וְכָל אֲשֶׁר בּוֹ אָז יְרַנְּנוּ כָּל עֲצֵי יָעַר:

[תהילים פרק-קז-ב]  יֹאמְרוּ גְּאוּלֵי יְהוָה אֲשֶׁר גְּאָלָם מִיַּד צָר:

[תהילים פרק-קט-ח]  יִהְיוּ יָמָיו מְעַטִּים פְּקֻדָּתוֹ יִקַּח אַחֵר:

[תהילים פרק-קלה-יג]  יְהוָה שִׁמְךָ לְעוֹלָם יְהוָה זִכְרְךָ לְדֹר וָדֹר:

[תהילים פרק-קמא-י]  יִפְּלוּ בְמַכְמֹרָיו רְשָׁעִים יַחַד אָנֹכִי עַד אֶעֱבוֹר:

[משלי פרק-יא-כד]  יֵשׁ מְפַזֵּר וְנוֹסָף עוֹד וְחוֹשֵׂךְ מִיֹּשֶׁר אַךְ לְמַחְסוֹר:

[איוב פרק-ג-ג]  יֹאבַד יוֹם אִוָּלֶד בּוֹ וְהַלַּיְלָה אָמַר הֹרָה גָבֶר:

[איוב פרק-ג-ט]  יֶחְשְׁכוּ כּוֹכְבֵי נִשְׁפּוֹ יְקַו לְאוֹר וָאַיִן וְאַל יִרְאֶה בְּעַפְעַפֵּי שָׁחַר:

[איוב פרק-יב-כה]  יְמַשְׁשׁוּ חֹשֶׁךְ וְלֹא אוֹר וַיַּתְעֵם כַּשִּׁכּוֹר:

[איוב פרק-טו-כד]  יְבַעֲתֻהוּ צַר וּמְצוּקָה תִּתְקְפֵהוּ כְּמֶלֶךְ עָתִיד לַכִּידוֹר:

[איוב פרק-טז-יד]  יִפְרְצֵנִי פֶרֶץ עַל פְּנֵי פָרֶץ יָרֻץ עָלַי כְּגִבּוֹר:

[איוב פרק-יז-ח]  יָשֹׁמּוּ יְשָׁרִים עַל זֹאת וְנָקִי עַל חָנֵף יִתְעֹרָר:

[דברי הימים א פרק-יב-יד]  יִרְמְיָהוּ הָעֲשִׂירִי מַכְבַּנַּי עַשְׁתֵּי עָשָׂר:


י-ש


[יחזקאל פרק-לד-ו]  יִשְׁגּוּ צֹאנִי בְּכָל הֶהָרִים וְעַל כָּל גִּבְעָה רָמָה וְעַל כָּל פְּנֵי הָאָרֶץ נָפֹצוּ צֹאנִי וְאֵין דּוֹרֵשׁ וְאֵין מְבַקֵּשׁ:

[תהילים פרק-סט-כג]  יְהִי שֻׁלְחָנָם לִפְנֵיהֶם לְפָח וְלִשְׁלוֹמִים לְמוֹקֵשׁ:

[תהילים פרק-צא-ז]  יִפֹּל מִצִּדְּךָ אֶלֶף וּרְבָבָה מִימִינֶךָ אֵלֶיךָ לֹא יִגָּשׁ:

[תהילים פרק-קב-יב]  יָמַי כְּצֵל נָטוּי וַאֲנִי כָּעֵשֶׂב אִיבָשׁ:

[תהילים פרק-קיט-פ]  יְהִי לִבִּי תָמִים בְּחֻקֶּיךָ לְמַעַן לֹא אֵבוֹשׁ:

[תהילים פרק-קכט-ו]  יִהְיוּ כַּחֲצִיר גַּגּוֹת שֶׁקַּדְמַת שָׁלַף יָבֵשׁ:

[איוב פרק-ג-יח]  יַחַד אֲסִירִים שַׁאֲנָנוּ לֹא שָׁמְעוּ קוֹל נֹגֵשׂ:

[איוב פרק-לט-ח]  יְתוּר הָרִים מִרְעֵהוּ וְאַחַר כָּל יָרוֹק יִדְרוֹשׁ:


י-ת


[ישעיה פרק-לה-א]  יְשֻׂשׂוּם מִדְבָּר וְצִיָּה וְתָגֵל עֲרָבָה וְתִפְרַח כַּחֲבַצָּלֶת:

[ירמיה פרק-נ-כד]  יָקֹשְׁתִּי לָךְ וְגַם נִלְכַּדְתְּ בָּבֶל וְאַתְּ לֹא יָדָעַתְּ נִמְצֵאת וְגַם נִתְפַּשְׂתְּ כִּי בַיהוָה הִתְגָּרִית:

[ירמיה פרק-נא-לח]  יַחְדָּו כַּכְּפִרִים יִשְׁאָגוּ נָעֲרוּ כְּגוֹרֵי אֲרָיוֹת:

[צפניה פרק-א-טז]  יוֹם שׁוֹפָר וּתְרוּעָה עַל הֶעָרִים הַבְּצֻרוֹת וְעַל הַפִּנּוֹת הַגְּבֹהוֹת:

[צפניה פרק-ג-ה]  יְהוָה צַדִּיק בְּקִרְבָּהּ לֹא יַעֲשֶׂה עַוְלָה בַּבֹּקֶר בַּבֹּקֶר מִשְׁפָּטוֹ יִתֵּן לָאוֹר לֹא נֶעְדָּר וְלֹא יוֹדֵעַ עַוָּל בֹּשֶׁת:

[מלאכי פרק-ב-יב]  יַכְרֵת יְהוָה לָאִישׁ אֲשֶׁר יַעֲשֶׂנָּה עֵר וְעֹנֶה מֵאָהֳלֵי יַעֲקֹב וּמַגִּישׁ מִנְחָה לַיהוָה צְבָאוֹת:

[תהילים פרק-יב-ד]  יַכְרֵת יְהוָה כָּל שִׂפְתֵי חֲלָקוֹת לָשׁוֹן מְדַבֶּרֶת גְּדֹלוֹת:

[תהילים פרק-יט-ג]  יוֹם לְיוֹם יַבִּיעַ אֹמֶר וְלַיְלָה לְּלַיְלָה יְחַוֶּה דָּעַת:

[תהילים פרק-נה-ו]  יִרְאָה וָרַעַד יָבֹא בִי וַתְּכַסֵּנִי פַּלָּצוּת:

[תהילים פרק-עד-ה]  יִוָּדַע כְּמֵבִיא לְמָעְלָה בִּסֲבָךְ עֵץ קַרְדֻּמּוֹת:

[תהילים פרק-קט-טז]  יַעַן אֲשֶׁר לֹא זָכַר עֲשׂוֹת חָסֶד וַיִּרְדֹּף אִישׁ עָנִי וְאֶבְיוֹן וְנִכְאֵה לֵבָב לְמוֹתֵת:

[תהילים פרק-קל-ז]  יַחֵל יִשְׂרָאֵל אֶל יְהוָה כִּי עִם יְהוָה הַחֶסֶד וְהַרְבֵּה עִמּוֹ פְדוּת:

[תהילים פרק-קמו-ט]  יְהוָה שֹׁמֵר אֶת גֵּרִים יָתוֹם וְאַלְמָנָה יְעוֹדֵד וְדֶרֶךְ רְשָׁעִים יְעַוֵּת:

[משלי פרק-יד-יב]  יֵשׁ דֶּרֶךְ יָשָׁר לִפְנֵי אִישׁ וְאַחֲרִיתָהּ דַּרְכֵי מָוֶת:

[משלי פרק-יד-כז]  יִרְאַת יְהוָה מְקוֹר חַיִּים לָסוּר מִמֹּקְשֵׁי מָוֶת:

[משלי פרק-טז-כה]  יֵשׁ דֶּרֶךְ יָשָׁר לִפְנֵי אִישׁ וְאַחֲרִיתָהּ דַּרְכֵי מָוֶת:

[משלי פרק-כ-טו]  יֵשׁ זָהָב וְרָב פְּנִינִים וּכְלִי יְקָר שִׂפְתֵי דָעַת:

[משלי פרק-כט-ז]  יֹדֵעַ צַדִּיק דִּין דַּלִּים רָשָׁע לֹא יָבִין דָּעַת:

[איוב פרק-יח-יג]  יֹאכַל בַּדֵּי עוֹרוֹ יֹאכַל בַּדָּיו בְּכוֹר מָוֶת:

[איוב פרק-יח-יד]  יִנָּתֵק מֵאָהֳלוֹ מִבְטַחוֹ וְתַצְעִדֵהוּ לְמֶלֶךְ בַּלָּהוֹת:

[שיר השירים פרק-ו-ד]  יָפָה אַתְּ רַעְיָתִי כְּתִרְצָה נָאוָה כִּירוּשָׁלִָם אֲיֻמָּה כַּנִּדְגָּלוֹת:

[איכה פרק-ה-ג]  יְתוֹמִים הָיִינוּ (אין) וְאֵין אָב אִמֹּתֵינוּ כְּאַלְמָנוֹת:

[עזרא פרק-ט-טו]  יְהוָה אֱלֹהֵי יִשְׂרָאֵל צַדִּיק אַתָּה כִּי נִשְׁאַרְנוּ פְלֵיטָה כְּהַיּוֹם הַזֶּה הִנְנוּ לְפָנֶיךָ בְּאַשְׁמָתֵינוּ כִּי אֵין לַעֲמוֹד לְפָנֶיךָ עַל זֹאת:

[דברי הימים א פרק-יז-יט]  יְהוָה בַּעֲבוּר עַבְדְּךָ וּכְלִבְּךָ עָשִׂיתָ אֵת כָּל הַגְּדוּלָּה הַזֹּאת לְהֹדִיעַ אֶת כָּל הַגְּדֻלּוֹת:
