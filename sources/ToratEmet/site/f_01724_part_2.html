פרשת צו




ויקרא פרק-ו

{א} וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} צַ֤ו אֶֽת-אַֽהֲרֹן֙ וְאֶת-בָּנָ֣יו לֵאמֹ֔ר זֹ֥את תּוֹרַ֖ת הָֽעֹלָ֑ה הִ֣וא הָֽעֹלָ֡ה עַל֩ מֽוֹקְדָ֨ה עַל-הַמִּזְבֵּ֤חַ כָּל-הַלַּ֨יְלָה֙ עַד-הַבֹּ֔קֶר וְאֵ֥שׁ הַמִּזְבֵּ֖חַ תּ֥וּקַד בּֽוֹ:  מ  צַ֤ו אֶֽת-אַֽהֲרֹן֙ וְאֶת-בָּנָ֣יו לֵאמֹ֔ר זֹ֥את תּוֹרַ֖ת הָֽעֹלָ֑ה הִ֣וא הָֽעֹלָ֡ה עַל֩ מֽוֹקְדָ֨ה עַל-הַמִּזְבֵּ֤חַ כָּל-הַלַּ֨יְלָה֙ עַד-הַבֹּ֔קֶר וְאֵ֥שׁ הַמִּזְבֵּ֖חַ תּ֥וּקַד בּֽוֹ:  ת  פַּקֵד יָת אַהֲרֹן וְיָת בְּנוֹהִי לְמֵימָר דָא אוֹרַיְתָא דַעֲלָתָא הִיא עֲלָתָא דְמִתּוֹקְדָא עַל מַדְבְּחָא כָל לֵילְיָא עַד צַפְרָא וְאֶשָׁתָא דְמַדְבְּחָא תְּהֵא יָקְדָא בֵהּ:

{ג} וְלָבַ֨שׁ הַכֹּהֵ֜ן מִדּ֣וֹ בַ֗ד וּמִֽכְנְסֵי-בַד֘ יִלְבַּ֣שׁ עַל-בְּשָׂרוֹ֒ וְהֵרִ֣ים אֶת-הַדֶּ֗שֶׁן אֲשֶׁ֨ר תֹּאכַ֥ל הָאֵ֛שׁ אֶת-הָֽעֹלָ֖ה עַל-הַמִּזְבֵּ֑חַ וְשָׂמ֕וֹ אֵ֖צֶל הַמִּזְבֵּֽחַ:  מ  וְלָבַ֨שׁ הַכֹּהֵ֜ן מִדּ֣וֹ בַ֗ד וּמִֽכְנְסֵי-בַד֘ יִלְבַּ֣שׁ עַל-בְּשָׂרוֹ֒ וְהֵרִ֣ים אֶת-הַדֶּ֗שֶׁן אֲשֶׁ֨ר תֹּאכַ֥ל הָאֵ֛שׁ אֶת-הָֽעֹלָ֖ה עַל-הַמִּזְבֵּ֑חַ וְשָׂמ֕וֹ אֵ֖צֶל הַמִּזְבֵּֽחַ:  ת  וְיַלְבֵּשׁ כַּהֲנָא לְבוּשִׁין דְבוּץ וּמִכְנְסִין דְבוּץ יַלְבֵּשׁ עַל בִּסְרֵהּ וְיַפְרֵשׁ יָת קִטְמָא דִי תֵיכוּל אֶשָׁתָא יָת עֲלָתָא עַל מַדְבְּחָא וִישַׁוִנֵהּ בִּסְטַּר מַדְבְּחָא:

{ד} וּפָשַׁט֙ אֶת-בְּגָדָ֔יו וְלָבַ֖שׁ בְּגָדִ֣ים אֲחֵרִ֑ים וְהוֹצִ֤יא אֶת-הַדֶּ֨שֶׁן֙ אֶל-מִח֣וּץ לַֽמַּֽחֲנֶ֔ה אֶל-מָק֖וֹם טָהֽוֹר:  מ  וּפָשַׁט֙ אֶת-בְּגָדָ֔יו וְלָבַ֖שׁ בְּגָדִ֣ים אֲחֵרִ֑ים וְהוֹצִ֤יא אֶת-הַדֶּ֨שֶׁן֙ אֶל-מִח֣וּץ לַֽמַּֽחֲנֶ֔ה אֶל-מָק֖וֹם טָהֽוֹר:  ת  וְיַשְׁלַח יָת לְבוּשׁוֹהִי וְיִלְבַּשׁ לְבוּשִׁין אָחֳרָנִין וְיַפֵּק יָת קִטְמָא לְמִבָּרָא לְמַשְׁרִיתָא לַאֲתַר דְכֵי:

{ה} וְהָאֵ֨שׁ עַל-הַמִּזְבֵּ֤חַ תּֽוּקַד-בּוֹ֙ לֹ֣א תִכְבֶּ֔ה וּבִעֵ֨ר עָלֶ֧יהָ הַכֹּהֵ֛ן עֵצִ֖ים בַּבֹּ֣קֶר בַּבֹּ֑קֶר וְעָרַ֤ךְ עָלֶ֨יהָ֙ הָֽעֹלָ֔ה וְהִקְטִ֥יר עָלֶ֖יהָ חֶלְבֵ֥י הַשְּׁלָמִֽים:  מ  וְהָאֵ֨שׁ עַל-הַמִּזְבֵּ֤חַ תּֽוּקַד-בּוֹ֙ לֹ֣א תִכְבֶּ֔ה וּבִעֵ֨ר עָלֶ֧יהָ הַכֹּהֵ֛ן עֵצִ֖ים בַּבֹּ֣קֶר בַּבֹּ֑קֶר וְעָרַ֤ךְ עָלֶ֨יהָ֙ הָֽעֹלָ֔ה וְהִקְטִ֥יר עָלֶ֖יהָ חֶלְבֵ֥י הַשְּׁלָמִֽים:  ת  וְאֶשָׁתָא עַל מַדְבְּחָא תְהֵי יָקְדָא בֵהּ לָא תִטְּפֵי וְיַבְעַר עֲלַהּ כַּהֲנָא אָעַיָא בִּצְפַר בִּצְפָר וְיַסְדַר עֲלַהּ עֲלָתָא וְיַסֵק עֲלַהּ תַּרְבֵּי נִכְסַת קוּדְשַׁיָא:

{ו} אֵ֗שׁ תָּמִ֛יד תּוּקַ֥ד עַל-הַמִּזְבֵּ֖חַ לֹ֥א תִכְבֶּֽה:  מ  אֵ֗שׁ תָּמִ֛יד תּוּקַ֥ד עַל-הַמִּזְבֵּ֖חַ לֹ֥א תִכְבֶּֽה:  ת  אֶשָׁתָא תְּדִירָא תְּהֵי יָקְדָא עַל מַדְבְּחָא לָא תִטְפֵי: (ס)

{ז} וְזֹ֥את תּוֹרַ֖ת הַמִּנְחָ֑ה הַקְרֵ֨ב אֹתָ֤הּ בְּנֵי-אַֽהֲרֹן֙ לִפְנֵ֣י יְהֹוָ֔ה אֶל-פְּנֵ֖י הַמִּזְבֵּֽחַ:  מ  וְזֹ֥את תּוֹרַ֖ת הַמִּנְחָ֑ה הַקְרֵ֨ב אֹתָ֤הּ בְּנֵי-אַֽהֲרֹן֙ לִפְנֵ֣י יְהֹוָ֔ה אֶל-פְּנֵ֖י הַמִּזְבֵּֽחַ:  ת  וְדָא אוֹרַיְתָא דְמִנְחָתָא דִיקָרְבוּן יָתַהּ בְּנֵי אַהֲרֹן קֳדָם יְיָ לָקֳדָם מַדְבְּחָא:

{ח} וְהֵרִ֨ים מִמֶּ֜נּוּ בְּקֻמְצ֗וֹ מִסֹּ֤לֶת הַמִּנְחָה֙ וּמִשַּׁמְנָ֔הּ וְאֵת֙ כָּל-הַלְּבֹנָ֔ה אֲשֶׁ֖ר עַל-הַמִּנְחָ֑ה וְהִקְטִ֣יר הַמִּזְבֵּ֗חַ רֵ֧יחַ נִיחֹ֛חַ אַזְכָּֽרָתָ֖הּ לַֽיהוָֹֽה:  מ  וְהֵרִ֨ים מִמֶּ֜נּוּ בְּקֻמְצ֗וֹ מִסֹּ֤לֶת הַמִּנְחָה֙ וּמִשַּׁמְנָ֔הּ וְאֵת֙ כָּל-הַלְּבֹנָ֔ה אֲשֶׁ֖ר עַל-הַמִּנְחָ֑ה וְהִקְטִ֣יר הַמִּזְבֵּ֗חַ רֵ֧יחַ נִיחֹ֛חַ אַזְכָּֽרָתָ֖הּ לַֽיהוָֹֽה:  ת  וְיַפְרֵשׁ מִנֵהּ בְּקֻמְצֵהּ מִסֻלְתָּא דְמִנְחָתָא וּמִמִשְׁחַהּ וְיָת כָּל לְבוּנְתָּא דִי עַל מִנְחָתָא וְיַסֵק לְמַדְבְּחָא לְאִתְקַבָּלָא בְרַעֲוָא אַדְכַּרְתָּא קֳדָם יְיָ:

{ט} וְהַנּוֹתֶ֣רֶת מִמֶּ֔נָּה יֹֽאכְל֖וּ אַֽהֲרֹ֣ן וּבָנָ֑יו מַצּ֤וֹת תֵּֽאָכֵל֙ בְּמָק֣וֹם קָדֹ֔שׁ בַּֽחֲצַ֥ר אֹֽהֶל-מוֹעֵ֖ד יֹֽאכְלֽוּהָ:  מ  וְהַנּוֹתֶ֣רֶת מִמֶּ֔נָּה יֹֽאכְל֖וּ אַֽהֲרֹ֣ן וּבָנָ֑יו מַצּ֤וֹת תֵּֽאָכֵל֙ בְּמָק֣וֹם קָדֹ֔שׁ בַּֽחֲצַ֥ר אֹֽהֶל-מוֹעֵ֖ד יֹֽאכְלֽוּהָ:  ת  וּדְאִשְׁתָּאַר מִנַהּ יֵכְלוּן אַהֲרֹן וּבְנוֹהִי פַּטִיר תִּתְאַכֵל בַּאֲתַר קַדִּישׁ בְּדָרַת מַשְׁכַּן זִמְנָא יֵכְלֻנַהּ:

{י} לֹ֤א תֵֽאָפֶה֙ חָמֵ֔ץ חֶלְקָ֛ם נָתַ֥תִּי אֹתָ֖הּ מֵֽאִשָּׁ֑י קֹ֤דֶשׁ קָֽדָשִׁים֙ הִ֔וא כַּֽחַטָּ֖את וְכָֽאָשָֽׁם:  מ  לֹ֤א תֵֽאָפֶה֙ חָמֵ֔ץ חֶלְקָ֛ם נָתַ֥תִּי אֹתָ֖הּ מֵֽאִשָּׁ֑י קֹ֤דֶשׁ קָֽדָשִׁים֙ הִ֔וא כַּֽחַטָּ֖את וְכָֽאָשָֽׁם:  ת  לָא תִתְאֲפֵי חֲמִיעַ חָלָקְהוֹן יְהָבִית יָתַהּ מִקֻרְבָּנָי קֹדֶשׁ קוּדְשִׁין הִיא כְּחַטָאתָא וְכַאֲשָׁמָא:

{יא} כָּל-זָכָ֞ר בִּבְנֵ֤י אַֽהֲרֹן֙ יֹֽאכֲלֶ֔נָּה חָק-עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם מֵֽאִשֵּׁ֖י יְהוָֹ֑ה כֹּ֛ל אֲשֶׁר-יִגַּ֥ע בָּהֶ֖ם יִקְדָּֽשׁ:  מ  כָּל-זָכָ֞ר בִּבְנֵ֤י אַֽהֲרֹן֙ יֹֽאכֲלֶ֔נָּה חָק-עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם מֵֽאִשֵּׁ֖י יְהוָֹ֑ה כֹּ֛ל אֲשֶׁר-יִגַּ֥ע בָּהֶ֖ם יִקְדָּֽשׁ:  ת  כָּל דְכוּרָא בִּבְנֵי אַהֲרֹן יֵכְלֻנַהּ קְיָם עָלָם לְדָרָתֵיכוֹן מִקֻרָבָּנַיָא דַיְיָ כֹּל דְיִקְרַב בְּהוֹן יִתְקַדָּשׁ: (פ)

{יב}  שני  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{יג} זֶ֡ה קָרְבַּן֩ אַֽהֲרֹ֨ן וּבָנָ֜יו אֲשֶׁר-יַקְרִ֣יבוּ לַֽיהֹוָ֗ה בְּיוֹם֙ הִמָּשַׁ֣ח אֹת֔וֹ עֲשִׂירִ֨ת הָֽאֵפָ֥ה סֹ֛לֶת מִנְחָ֖ה תָּמִ֑יד מַֽחֲצִיתָ֣הּ בַּבֹּ֔קֶר וּמַֽחֲצִיתָ֖הּ בָּעָֽרֶב:  מ  זֶ֡ה קָרְבַּן֩ אַֽהֲרֹ֨ן וּבָנָ֜יו אֲשֶׁר-יַקְרִ֣יבוּ לַֽיהֹוָ֗ה בְּיוֹם֙ הִמָּשַׁ֣ח אֹת֔וֹ עֲשִׂירִ֨ת הָֽאֵפָ֥ה סֹ֛לֶת מִנְחָ֖ה תָּמִ֑יד מַֽחֲצִיתָ֣הּ בַּבֹּ֔קֶר וּמַֽחֲצִיתָ֖הּ בָּעָֽרֶב:  ת  דֵין קֻרְבַּן אַהֲרֹן וּבְנוֹהִי דִי יְקָרְבוּן קֳדָם יְיָ בְּיוֹמָא דְרַבִּיאוּ יָתֵהּ חַד מִן עַסְרָא בִתְלַת סְאִין סֻלְתָּא מִנְחָתָא תְּדִירָא פַּלְגוּתַהּ בְּצַפְרָא וּפַלְגוּתַהּ בְּרַמְשָׁא:

{יד} עַֽל-מַֽחֲבַ֗ת בַּשֶּׁ֛מֶן תֵּֽעָשֶׂ֖ה מֻרְבֶּ֣כֶת תְּבִיאֶ֑נָּה תֻּֽפִינֵי֙ מִנְחַ֣ת פִּתִּ֔ים תַּקְרִ֥יב רֵֽיחַ-נִיחֹ֖חַ לַֽיהוָֹֽה:  מ  עַֽל-מַֽחֲבַ֗ת בַּשֶּׁ֛מֶן תֵּֽעָשֶׂ֖ה מֻרְבֶּ֣כֶת תְּבִיאֶ֑נָּה תֻּֽפִינֵי֙ מִנְחַ֣ת פִּתִּ֔ים תַּקְרִ֥יב רֵֽיחַ-נִיחֹ֖חַ לַֽיהוָֹֽה:  ת  עַל מַסְרֵיתָא בִּמְשַׁח תִּתְעֲבֵד רְבִיכָא תַיְתִנַהּ תּוּפִינֵי מִנְחַת בִּצוּעִין תְּקָרֵב לְאִתְקַבָּלָא בְרַעֲוָא קֳדָם יְיָ:

{יז} וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{יח} דַּבֵּ֤ר אֶֽל-אַֽהֲרֹן֙ וְאֶל-בָּנָ֣יו לֵאמֹ֔ר זֹ֥את תּוֹרַ֖ת הַֽחַטָּ֑את בִּמְק֡וֹם אֲשֶׁר֩ תִּשָּׁחֵ֨ט הָֽעֹלָ֜ה תִּשָּׁחֵ֤ט הַֽחַטָּאת֙ לִפְנֵ֣י יְהֹוָ֔ה קֹ֥דֶשׁ קָֽדָשִׁ֖ים הִֽוא:  מ  דַּבֵּ֤ר אֶֽל-אַֽהֲרֹן֙ וְאֶל-בָּנָ֣יו לֵאמֹ֔ר זֹ֥את תּוֹרַ֖ת הַֽחַטָּ֑את בִּמְק֡וֹם אֲשֶׁר֩ תִּשָּׁחֵ֨ט הָֽעֹלָ֜ה תִּשָּׁחֵ֤ט הַֽחַטָּאת֙ לִפְנֵ֣י יְהֹוָ֔ה קֹ֥דֶשׁ קָֽדָשִׁ֖ים הִֽוא:  ת  מַלֵל עִם אַהֲרֹן וְעִם בְּנוֹהִי לְמֵימָר דָא אוֹרַיְתָא דְחַטָאתָא בַּאֲתַר דִי תִתְנְכֵס עֲלָתָא תִּתְנְכֵס חַטָאתָא קֳדָם יְיָ קֹדֶשׁ קוּדְשִׁין הִיא:

{יט} הַכֹּהֵ֛ן הַֽמְחַטֵּ֥א אֹתָ֖הּ יֹֽאכֲלֶ֑נָּה בְּמָק֤וֹם קָדֹשׁ֙ תֵּֽאָכֵ֔ל בַּֽחֲצַ֖ר אֹ֥הֶל מוֹעֵֽד:  מ  הַכֹּהֵ֛ן הַֽמְחַטֵּ֥א אֹתָ֖הּ יֹֽאכֲלֶ֑נָּה בְּמָק֤וֹם קָדֹשׁ֙ תֵּֽאָכֵ֔ל בַּֽחֲצַ֖ר אֹ֥הֶל מוֹעֵֽד:  ת  כַּהֲנָא דִמְכַפֵּר בִּדְמַהּ יֵיכְלִנַהּ בַּאֲתַר קַדִישׁ תִּתְאֲכֵל בְּדָרַת מַשְׁכַּן זִמְנָא:

{כ} כֹּ֛ל אֲשֶׁר-יִגַּ֥ע בִּבְשָׂרָ֖הּ יִקְדָּ֑שׁ וַֽאֲשֶׁ֨ר יִזֶּ֤ה מִדָּמָהּ֙ עַל-הַבֶּ֔גֶד אֲשֶׁר֙ יִזֶּ֣ה עָלֶ֔יהָ תְּכַבֵּ֖ס בְּמָק֥וֹם קָדֹֽשׁ:  מ  כֹּ֛ל אֲשֶׁר-יִגַּ֥ע בִּבְשָׂרָ֖הּ יִקְדָּ֑שׁ וַֽאֲשֶׁ֨ר יִזֶּ֤ה מִדָּמָהּ֙ עַל-הַבֶּ֔גֶד אֲשֶׁר֙ יִזֶּ֣ה עָלֶ֔יהָ תְּכַבֵּ֖ס בְּמָק֥וֹם קָדֹֽשׁ:  ת  כֹּל דִי יִקְרַב בְּבִסְרַהּ יִתְקַדַשׁ וְדִי יַדִי מִדְמַהּ עַל לְבוּשׁ דִי יַדִי עֲלַהּ תִּתְחַוָר בַּאֲתַר קַדִישׁ:

{כא} וּכְלִי-חֶ֛רֶשׂ אֲשֶׁ֥ר תְּבֻשַּׁל-בּ֖וֹ יִשָּׁבֵ֑ר וְאִם-בִּכְלִ֤י נְח֨שֶׁת֙ בֻּשָּׁ֔לָה וּמֹרַ֥ק וְשֻׁטַּ֖ף בַּמָּֽיִם:  מ  וּכְלִי-חֶ֛רֶשׂ אֲשֶׁ֥ר תְּבֻשַּׁל-בּ֖וֹ יִשָּׁבֵ֑ר וְאִם-בִּכְלִ֤י נְח֨שֶׁת֙ בֻּשָּׁ֔לָה וּמֹרַ֥ק וְשֻׁטַּ֖ף בַּמָּֽיִם:  ת  וּמָן דַחֲסַף דִי תִתְבַּשֵׁל בֵּהּ יִתָּבָר וְאִם בְּמָנָא דִנְחָשָׁא תִּתְבַּשֵׁל וְיִתְמְרֵק וְיִשְׁתַּטֵף בְּמַיָא:

{כב} כָּל-זָכָ֥ר בַּכֹּֽהֲנִ֖ים יֹאכַ֣ל אֹתָ֑הּ קֹ֥דֶשׁ קָֽדָשִׁ֖ים הִֽוא:  מ  כָּל-זָכָ֥ר בַּכֹּֽהֲנִ֖ים יֹאכַ֣ל אֹתָ֑הּ קֹ֥דֶשׁ קָֽדָשִׁ֖ים הִֽוא:  ת  כָּל דְכוּרָא בְּכַהֲנַיָא יֵכוּל יָתַהּ קֹדֶשׁ קוּדְשִׁין הִיא:

{כג} וְכָל-חַטָּ֡את אֲשֶׁר֩ יוּבָ֨א מִדָּמָ֜הּ אֶל-אֹ֧הֶל מוֹעֵ֛ד לְכַפֵּ֥ר בַּקֹּ֖דֶשׁ לֹ֣א תֵֽאָכֵ֑ל בָּאֵ֖שׁ תִּשָּׂרֵֽף:  מ  וְכָל-חַטָּ֡את אֲשֶׁר֩ יוּבָ֨א מִדָּמָ֜הּ אֶל-אֹ֧הֶל מוֹעֵ֛ד לְכַפֵּ֥ר בַּקֹּ֖דֶשׁ לֹ֣א תֵֽאָכֵ֑ל בָּאֵ֖שׁ תִּשָּׂרֵֽף:  ת  וְכָל חַטָאתָא דִי יִתָּעַל מִדְמַהּ לְמַשְׁכַּן זִמְנָא לְכַפָּרָא בְקוּדְשָׁא לָא תִתְאֲכֵל בְּנוּרָא תִּתּוֹקָד: (פ)




ויקרא פרק-ז

{א} וְזֹ֥את תּוֹרַ֖ת הָֽאָשָׁ֑ם קֹ֥דֶשׁ קָֽדָשִׁ֖ים הֽוּא:  מ  וְזֹ֥את תּוֹרַ֖ת הָֽאָשָׁ֑ם קֹ֥דֶשׁ קָֽדָשִׁ֖ים הֽוּא:  ת  וְדָא אוֹרַיְתָא דַאֲשָׁמָא קֹדֶשׁ קוּדְשִׁין הוּא:

{ב} בִּמְק֗וֹם אֲשֶׁ֤ר יִשְׁחֲטוּ֙ אֶת-הָ֣עֹלָ֔ה יִשְׁחֲט֖וּ אֶת-הָֽאָשָׁ֑ם וְאֶת-דָּמ֛וֹ יִזְרֹ֥ק עַל-הַמִּזְבֵּ֖חַ סָבִֽיב:  מ  בִּמְק֗וֹם אֲשֶׁ֤ר יִשְׁחֲטוּ֙ אֶת-הָ֣עֹלָ֔ה יִשְׁחֲט֖וּ אֶת-הָֽאָשָׁ֑ם וְאֶת-דָּמ֛וֹ יִזְרֹ֥ק עַל-הַמִּזְבֵּ֖חַ סָבִֽיב:  ת  בְּאַתְרָא דִי יִכְּסוּן יָת עֲלָתָא יִכְּסוּן יָת אֲשָׁמָא וְיָת דְמֵהּ יִזְרַק עַל מַדְבְּחָא סְחוֹר סְחוֹר:

{ג} וְאֵ֥ת כָּל-חֶלְבּ֖וֹ יַקְרִ֣יב מִמֶּ֑נּוּ אֵ֚ת הָֽאַלְיָ֔ה וְאֶת-הַחֵ֖לֶב הַֽמְכַסֶּ֥ה אֶת-הַקֶּֽרֶב:  מ  וְאֵ֥ת כָּל-חֶלְבּ֖וֹ יַקְרִ֣יב מִמֶּ֑נּוּ אֵ֚ת הָֽאַלְיָ֔ה וְאֶת-הַחֵ֖לֶב הַֽמְכַסֶּ֥ה אֶת-הַקֶּֽרֶב:  ת  וְיָת כָּל תַּרְבֵּהּ יַפְרֵשׁ מִנֵהּ יָת אֲלִיתָא וְיָת תַּרְבָּא דְחָפֵי יָת גַוָא:

{ד} וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת-הַחֵ֨לֶב֙ אֲשֶׁ֣ר עֲלֵיהֶ֔ן אֲשֶׁ֖ר עַל-הַכְּסָלִ֑ים וְאֶת-הַיֹּתֶ֨רֶת֙ עַל-הַכָּבֵ֔ד עַל-הַכְּלָיֹ֖ת יְסִירֶֽנָּה:  מ  וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת-הַחֵ֨לֶב֙ אֲשֶׁ֣ר עֲלֵיהֶ֔ן אֲשֶׁ֖ר עַל-הַכְּסָלִ֑ים וְאֶת-הַיֹּתֶ֨רֶת֙ עַל-הַכָּבֵ֔ד עַל-הַכְּלָיֹ֖ת יְסִירֶֽנָּה:  ת  וְיָת תַּרְתֵּין כָּלְיָן וְיָת תַּרְבָּא דִי עֲלֵיהֶן דִי עַל גִסְסַיָא וְיָת חִצְרָא דְעַל כַּבְדָא עַל כָּלְיֵתָא יֶעְדִנַהּ:

{ה} וְהִקְטִ֨יר אֹתָ֤ם הַכֹּהֵן֙ הַמִּזְבֵּ֔חָה אִשֶּׁ֖ה לַֽיהוָֹ֑ה אָשָׁ֖ם הֽוּא:  מ  וְהִקְטִ֨יר אֹתָ֤ם הַכֹּהֵן֙ הַמִּזְבֵּ֔חָה אִשֶּׁ֖ה לַֽיהוָֹ֑ה אָשָׁ֖ם הֽוּא:  ת  וְיַסֵיק יָתְהוֹן כַּהֲנָא לְמַדְבְּחָא קוּרְבָּנָא קֳדָם יְיָ אֲשָׁמָא הוּא:

{ו} כָּל-זָכָ֥ר בַּכֹּֽהֲנִ֖ים יֹֽאכֲלֶ֑נּוּ בְּמָק֤וֹם קָדוֹשׁ֙ יֵֽאָכֵ֔ל קֹ֥דֶשׁ קָֽדָשִׁ֖ים הֽוּא:  מ  כָּל-זָכָ֥ר בַּכֹּֽהֲנִ֖ים יֹֽאכֲלֶ֑נּוּ בְּמָק֤וֹם קָדוֹשׁ֙ יֵֽאָכֵ֔ל קֹ֥דֶשׁ קָֽדָשִׁ֖ים הֽוּא:  ת  כָּל דְכוּרָא בְּכַהֲנַיָא יֵיכְלִנֵהּ בַּאֲתַר קַדִישׁ יִתְאֲכָל קוֹדֶשׁ קוּדְשִׁין הוּא:

{ז} כַּֽחַטָּאת֙ כָּֽאָשָׁ֔ם תּוֹרָ֥ה אַחַ֖ת לָהֶ֑ם הַכֹּהֵ֛ן אֲשֶׁ֥ר יְכַפֶּר-בּ֖וֹ ל֥וֹ יִֽהְיֶֽה:  מ  כַּֽחַטָּאת֙ כָּֽאָשָׁ֔ם תּוֹרָ֥ה אַחַ֖ת לָהֶ֑ם הַכֹּהֵ֛ן אֲשֶׁ֥ר יְכַפֶּר-בּ֖וֹ ל֥וֹ יִֽהְיֶֽה:  ת  כְּחַטָאתָא כַּאֲשָׁמָא אוֹרַיְתָא חֲדָא לְהוֹן כַּהֲנָא דִי יְכַפֵּר בֵּהּ דִילֵהּ יְהֵי:

{ח} וְהַ֨כֹּהֵ֔ן הַמַּקְרִ֖יב אֶת-עֹ֣לַת אִ֑ישׁ ע֤וֹר הָֽעֹלָה֙ אֲשֶׁ֣ר הִקְרִ֔יב לַכֹּהֵ֖ן ל֥וֹ יִֽהְיֶֽה:  מ  וְהַ֨כֹּהֵ֔ן הַמַּקְרִ֖יב אֶת-עֹ֣לַת אִ֑ישׁ ע֤וֹר הָֽעֹלָה֙ אֲשֶׁ֣ר הִקְרִ֔יב לַכֹּהֵ֖ן ל֥וֹ יִֽהְיֶֽה:  ת  וְכַהֲנָא דִמְקָרֵב יַת עֲלַת גְבַר מְשַׁךְ עֲלָתָא דִי יְקָרֵב לְכַהֲנָא דִילֵהּ יְהֵי:

{ט} וְכָל-מִנְחָ֗ה אֲשֶׁ֤ר תֵּֽאָפֶה֙ בַּתַּנּ֔וּר וְכָל-נַֽעֲשָׂ֥ה בַמַּרְחֶ֖שֶׁת וְעַֽל-מַֽחֲבַ֑ת לַכֹּהֵ֛ן הַמַּקְרִ֥יב אֹתָ֖הּ ל֥וֹ תִֽהְיֶֽה:  מ  וְכָל-מִנְחָ֗ה אֲשֶׁ֤ר תֵּֽאָפֶה֙ בַּתַּנּ֔וּר וְכָל-נַֽעֲשָׂ֥ה בַמַּרְחֶ֖שֶׁת וְעַֽל-מַֽחֲבַ֑ת לַכֹּהֵ֛ן הַמַּקְרִ֥יב אֹתָ֖הּ ל֥וֹ תִֽהְיֶֽה:  ת  וְכָל מִנְחָתָא דִי תִתְאֲפֵי בְּתַנוּרָא וְכָל דְתִתְעֲבֵד בְּרִדְתָּא וְעַל מַסְרֵיתָא לְכַהֲנָא דִמְקָרֵב יָתַהּ דִילֵהּ תְּהֵי:

{י} וְכָל-מִנְחָ֥ה בְלוּלָֽה-בַשֶּׁ֖מֶן וַֽחֲרֵבָ֑ה לְכָל-בְּנֵ֧י אַֽהֲרֹ֛ן תִּֽהְיֶ֖ה אִ֥ישׁ כְּאָחִֽיו:  מ  וְכָל-מִנְחָ֥ה בְלוּלָֽה-בַשֶּׁ֖מֶן וַֽחֲרֵבָ֑ה לְכָל-בְּנֵ֧י אַֽהֲרֹ֛ן תִּֽהְיֶ֖ה אִ֥ישׁ כְּאָחִֽיו:  ת  וְכָל מִנְחָתָא דְפִילָא בִמְשַׁח וּדְלָא פִילָא לְכָל בְּנֵי אַהֲרֹן תְּהֵי גְבַר כַּאָחוֹהִי: (פ)

{יא}  שלישי  וְזֹ֥את תּוֹרַ֖ת זֶ֣בַח הַשְּׁלָמִ֑ים אֲשֶׁ֥ר יַקְרִ֖יב לַֽיהוָֹֽה:  מ  וְזֹ֥את תּוֹרַ֖ת זֶ֣בַח הַשְּׁלָמִ֑ים אֲשֶׁ֥ר יַקְרִ֖יב לַֽיהוָֹֽה:  ת  וְדָא אוֹרַיְתָא דְנִכְסַת קוּדְשַׁיָא דִי יְקָרֵב קֳדָם יְיָ:

{יב} אִ֣ם עַל-תּוֹדָה֘ יַקְרִיבֶ֒נּוּ֒ וְהִקְרִ֣יב | עַל-זֶ֣בַח הַתּוֹדָ֗ה חַלּ֤וֹת מַצּוֹת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן וְסֹ֣לֶת מֻרְבֶּ֔כֶת חַלֹּ֖ת בְּלוּלֹ֥ת בַּשָּֽׁמֶן:  מ  אִ֣ם עַל-תּוֹדָה֘ יַקְרִיבֶ֒נּוּ֒ וְהִקְרִ֣יב | עַל-זֶ֣בַח הַתּוֹדָ֗ה חַלּ֤וֹת מַצּוֹת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן וְסֹ֣לֶת מֻרְבֶּ֔כֶת חַלֹּ֖ת בְּלוּלֹ֥ת בַּשָּֽׁמֶן:  ת  אִם עַל תּוֹדְתָא יְקָרְבִנֵהּ וִיקָרֵב עַל נִכְסַת תּוֹדְתָא גְרִיצָן פַּטִירָן דְפִילָן בִּמְשַׁח וְאֶסְפּוֹגִין פַּטִירִין דִּמְשִׁיחִין בִּמְשָׁח וְסוּלְתָּא רְבִיכָא גְרִיצָן דְפִילָן בִּמְשָׁח:

{יג} עַל-חַלֹּת֙ לֶ֣חֶם חָמֵ֔ץ יַקְרִ֖יב קָרְבָּנ֑וֹ עַל-זֶ֖בַח תּוֹדַ֥ת שְׁלָמָֽיו:  מ  עַל-חַלֹּת֙ לֶ֣חֶם חָמֵ֔ץ יַקְרִ֖יב קָרְבָּנ֑וֹ עַל-זֶ֖בַח תּוֹדַ֥ת שְׁלָמָֽיו:  ת  עַל גְּרִיצָן דִלְחֵם חָמִיעַ יְקָרֵב קוּרְבָּנֵיהּ עַל נִכְסַת תּוֹדַת קוּדְשׁוֹהִי:

{יד} וְהִקְרִ֨יב מִמֶּ֤נּוּ אֶחָד֙ מִכָּל-קָרְבָּ֔ן תְּרוּמָ֖ה לַֽיהוָֹ֑ה לַכֹּהֵ֗ן הַזֹּרֵ֛ק אֶת-דַּ֥ם הַשְּׁלָמִ֖ים ל֥וֹ יִֽהְיֶֽה:  מ  וְהִקְרִ֨יב מִמֶּ֤נּוּ אֶחָד֙ מִכָּל-קָרְבָּ֔ן תְּרוּמָ֖ה לַֽיהוָֹ֑ה לַכֹּהֵ֗ן הַזֹּרֵ֛ק אֶת-דַּ֥ם הַשְּׁלָמִ֖ים ל֥וֹ יִֽהְיֶֽה:  ת  וִיקָרֵב מִנֵהּ חַד מִכָּל קוּרְבַּן אַפְרָשׁוּתָא קֳדָם יְיָ לְכַהֲנָא דְיִזְרַק יָת דַם נִכְסַת קוּדְשַׁיָא דִילֵהּ יְהֵי:

{טו} וּבְשַׂ֗ר זֶ֚בַח תּוֹדַ֣ת שְׁלָמָ֔יו בְּי֥וֹם קָרְבָּנ֖וֹ יֵֽאָכֵ֑ל לֹֽא-יַנִּ֥יחַ מִמֶּ֖נּוּ עַד-בֹּֽקֶר:  מ  וּבְשַׂ֗ר זֶ֚בַח תּוֹדַ֣ת שְׁלָמָ֔יו בְּי֥וֹם קָרְבָּנ֖וֹ יֵֽאָכֵ֑ל לֹֽא-יַנִּ֥יחַ מִמֶּ֖נּוּ עַד-בֹּֽקֶר:  ת  וּבְשַׂר נִכְסַת תּוֹדַת קוּדְשׁוֹהִי בְּיוֹם קֻרְבָּנֵהּ יִתְאֲכֵל לָא יַצְנַע מִנֵהּ עַד צַפְרָא:

{טז} וְאִם-נֶ֣דֶר | א֣וֹ נְדָבָ֗ה זֶ֚בַח קָרְבָּנ֔וֹ בְּי֛וֹם הַקְרִיב֥וֹ אֶת-זִבְח֖וֹ יֵֽאָכֵ֑ל וּמִֽמָּ֣חֳרָ֔ת וְהַנּוֹתָ֥ר מִמֶּ֖נּוּ יֵֽאָכֵֽל:  מ  וְאִם-נֶ֣דֶר | א֣וֹ נְדָבָ֗ה זֶ֚בַח קָרְבָּנ֔וֹ בְּי֛וֹם הַקְרִיב֥וֹ אֶת-זִבְח֖וֹ יֵֽאָכֵ֑ל וּמִֽמָּ֣חֳרָ֔ת וְהַנּוֹתָ֥ר מִמֶּ֖נּוּ יֵֽאָכֵֽל:  ת  וְאִם נִדְרָא אוֹ נְדַבְתָּא נִכְסַת קֻרְבָּנֵהּ בְּיוֹמָא דִי יְקָרֵב יָת נִכְסְתֵיהּ יִתְאֲכֵל וּבְיוֹמָא דְבַתְרוֹהִי וּדְאִשְׁתָּאַר מִנֵהּ יִתְאֲכֵלּ:

{יז} וְהַנּוֹתָ֖ר מִבְּשַׂ֣ר הַזָּ֑בַח בַּיּוֹם֙ הַשְּׁלִישִׁ֔י בָּאֵ֖שׁ יִשָּׂרֵֽף:  מ  וְהַנּוֹתָ֖ר מִבְּשַׂ֣ר הַזָּ֑בַח בַּיּוֹם֙ הַשְּׁלִישִׁ֔י בָּאֵ֖שׁ יִשָּׂרֵֽף:  ת  וּדְאִשְׁתָּאַר מִבְּסַר נִכְסָתָא בְּיוֹמָא תְּלִיתָאָה בְּנוּרָא יִתּוֹקָד:

{יח} וְאִ֣ם הֵֽאָכֹ֣ל יֵֽ֠אָכֵ֠ל מִבְּשַׂר-זֶ֨בַח שְׁלָמָ֜יו בַּיּ֣וֹם הַשְּׁלִישִׁי֘ לֹ֣א יֵֽרָצֶה֒ הַמַּקְרִ֣יב אֹת֗וֹ לֹ֧א יֵֽחָשֵׁ֛ב ל֖וֹ פִּגּ֣וּל יִֽהְיֶ֑ה וְהַנֶּ֛פֶשׁ הָֽאֹכֶ֥לֶת מִמֶּ֖נּוּ עֲוֹנָ֥הּ תִּשָּֽׂא:  מ  וְאִ֣ם הֵֽאָכֹ֣ל יֵֽ֠אָכֵ֠ל מִבְּשַׂר-זֶ֨בַח שְׁלָמָ֜יו בַּיּ֣וֹם הַשְּׁלִישִׁי֘ לֹ֣א יֵֽרָצֶה֒ הַמַּקְרִ֣יב אֹת֗וֹ לֹ֧א יֵֽחָשֵׁ֛ב ל֖וֹ פִּגּ֣וּל יִֽהְיֶ֑ה וְהַנֶּ֛פֶשׁ הָֽאֹכֶ֥לֶת מִמֶּ֖נּוּ עֲוֹנָ֥הּ תִּשָּֽׂא:  ת  וְאִם אִתְאֲכָלָא יִתְאֲכֵל מִבְּסַר נִכְסַת קוּדְשׁוֹהִי בְּיוֹמָא תְּלִיתָאָה לָא יְהֵי לְרַעֲוָא דִמְקָרֵב יָתֵהּ לָא יִתְחַשֵׁב לֵהּ מְרָחָק יְהֵי וֶאֱנַשׁ דְתֵכוּל מִנֵהּ חוֹבֵהּ יְקַבֵּל:

{יט} וְהַבָּשָׂ֞ר אֲשֶׁר-יִגַּ֤ע בְּכָל-טָמֵא֙ לֹ֣א יֵֽאָכֵ֔ל בָּאֵ֖שׁ יִשָּׂרֵ֑ף וְהַ֨בָּשָׂ֔ר כָּל-טָה֖וֹר יֹאכַ֥ל בָּשָֽׂר:  מ  וְהַבָּשָׂ֞ר אֲשֶׁר-יִגַּ֤ע בְּכָל-טָמֵא֙ לֹ֣א יֵֽאָכֵ֔ל בָּאֵ֖שׁ יִשָּׂרֵ֑ף וְהַ֨בָּשָׂ֔ר כָּל-טָה֖וֹר יֹאכַ֥ל בָּשָֽׂר:  ת  וּבְסַר קוּדְשָׁא דִי יִקְרַב בְּכָל מְסָאָב לָא יִתְאֲכֵל בְּנוּרָא יִתּוֹקָד וּבְסַר קוּדְשָׁא כָּל דִי דְכֵי לְקוּדְשָׁא יֵיכוּל בְּסַר קוּדְשָׁא:

{כ} וְהַנֶּ֜פֶשׁ אֲשֶׁר-תֹּאכַ֣ל בָּשָׂ֗ר מִזֶּ֤בַֽח הַשְּׁלָמִים֙ אֲשֶׁ֣ר לַֽיהוָֹ֔ה וְטֻמְאָת֖וֹ עָלָ֑יו וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ:  מ  וְהַנֶּ֜פֶשׁ אֲשֶׁר-תֹּאכַ֣ל בָּשָׂ֗ר מִזֶּ֤בַֽח הַשְּׁלָמִים֙ אֲשֶׁ֣ר לַֽיהוָֹ֔ה וְטֻמְאָת֖וֹ עָלָ֑יו וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ:  ת  וֶאֱנַשׁ דִי תֵכוּל בִּסְרָא מִנִכְסַת קוּדְשַׁיָא דִי קֳדָם יְיָ וְסוֹבְתֵהּ עֲלוֹהִי וְיִשְׁתֵּיצֵי אֱנָשָׁא הַהוּא מֵעַמַּהּ:

{כא} וְנֶ֜פֶשׁ כִּֽי-תִגַּ֣ע בְּכָל-טָמֵ֗א בְּטֻמְאַ֤ת אָדָם֙ א֣וֹ | בִּבְהֵמָ֣ה טְמֵאָ֗ה א֚וֹ בְּכָל-שֶׁ֣קֶץ טָמֵ֔א וְאָכַ֛ל מִבְּשַׂר-זֶ֥בַֽח הַשְּׁלָמִ֖ים אֲשֶׁ֣ר לַֽיהוָֹ֑ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ:  מ  וְנֶ֜פֶשׁ כִּֽי-תִגַּ֣ע בְּכָל-טָמֵ֗א בְּטֻמְאַ֤ת אָדָם֙ א֣וֹ | בִּבְהֵמָ֣ה טְמֵאָ֗ה א֚וֹ בְּכָל-שֶׁ֣קֶץ טָמֵ֔א וְאָכַ֛ל מִבְּשַׂר-זֶ֥בַֽח הַשְּׁלָמִ֖ים אֲשֶׁ֣ר לַֽיהוָֹ֑ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ:  ת  וֶאֱנַשׁ אֲרֵי תִקְרַב בְּכָל מְסָאָב בְסוֹאֲבַת אֱנָשָׁא אוֹ בִּבְעִירָא מְסָאֲבָא אוֹ בְּכָל שִׁקְצָא מְסָאָב וְיֵיכוּל מִבְּסַר נִכְסַת קוּדְשַׁיָא דִי קֳדָם יְיָ וְיִשְׁתֵּיצֵי אֱנָשָׁא הַהוּא מֵעַמַּהּ:

{כב} וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלֵיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{כג} דַּבֵּ֛ר אֶל-בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר כָּל-חֵ֜לֶב שׁ֥וֹר וְכֶ֛שֶׂב וָעֵ֖ז לֹ֥א תֹאכֵֽלוּ:  מ  דַּבֵּ֛ר אֶל-בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר כָּל-חֵ֜לֶב שׁ֥וֹר וְכֶ֛שֶׂב וָעֵ֖ז לֹ֥א תֹאכֵֽלוּ:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל לְמֵימָר כָּל תְּרַב תּוֹר וְאִמַר וְעִזָא לָא תֵיכְלוּן:

{כד} וְחֵ֤לֶב נְבֵלָה֙ וְחֵ֣לֶב טְרֵפָ֔ה יֵֽעָשֶׂ֖ה לְכָל-מְלָאכָ֑ה וְאָכֹ֖ל לֹ֥א תֹֽאכְלֻֽהוּ:  מ  וְחֵ֤לֶב נְבֵלָה֙ וְחֵ֣לֶב טְרֵפָ֔ה יֵֽעָשֶׂ֖ה לְכָל-מְלָאכָ֑ה וְאָכֹ֖ל לֹ֥א תֹֽאכְלֻֽהוּ:  ת  וּתְרַב נְבִילָא וּתְרַב תְּבִירָא יִתְעֲבֵד לְכָל עִבִדְתָּא וּמֵיכַל לָא תֵיכְלוּנֵהּ:

{כה} כִּ֚י כָּל-אֹכֵ֣ל חֵ֔לֶב מִ֨ן-הַבְּהֵמָ֔ה אֲשֶׁ֨ר יַקְרִ֥יב מִמֶּ֛נָּה אִשֶּׁ֖ה לַֽיהוָֹ֑ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הָֽאֹכֶ֖לֶת מֵֽעַמֶּֽיהָ:  מ  כִּ֚י כָּל-אֹכֵ֣ל חֵ֔לֶב מִ֨ן-הַבְּהֵמָ֔ה אֲשֶׁ֨ר יַקְרִ֥יב מִמֶּ֛נָּה אִשֶּׁ֖ה לַֽיהוָֹ֑ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הָֽאֹכֶ֖לֶת מֵֽעַמֶּֽיהָ:  ת  אֲרֵי כָל דְיֵכוּל תַּרְבָּא מִן בְּעִירָא דִי יִקָרְבוּן מִנַהּ קוּרְבָּנָא קֳדָם יְיָ וְיִשְׁתֵּיצֵי אֱנָשָׁא דִי תֵיכוּל מֵעַמַּהּ:

{כו} וְכָל-דָּם֙ לֹ֣א תֹֽאכְל֔וּ בְּכֹ֖ל מֽוֹשְׁבֹֽתֵיכֶ֑ם לָע֖וֹף וְלַבְּהֵמָֽה:  מ  וְכָל-דָּם֙ לֹ֣א תֹֽאכְל֔וּ בְּכֹ֖ל מֽוֹשְׁבֹֽתֵיכֶ֑ם לָע֖וֹף וְלַבְּהֵמָֽה:  ת  וְכָל דְמָא לָא תֵיכְלוּן בְּכֹל מוֹתְבָנֵיכוֹן דְעוֹפָא וְדִבְעִירָא:

{כז} כָּל-נֶ֖פֶשׁ אֲשֶׁר-תֹּאכַ֣ל כָּל-דָּ֑ם וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ:  מ  כָּל-נֶ֖פֶשׁ אֲשֶׁר-תֹּאכַ֣ל כָּל-דָּ֑ם וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ:  ת  כָּל אֱנָשָׁא דִי תֵיכוּל כָּל דְמָא וְיִשְׁתֵּיצֵי אֱנָשָׁא הַהוּא מֵעַמַהּ: (פ)

{כח} וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{כט} דַּבֵּ֛ר אֶל-בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר הַמַּקְרִ֞יב אֶת-זֶ֤בַח שְׁלָמָיו֙ לַֽיהֹוָ֔ה יָבִ֧יא אֶת-קָרְבָּנ֛וֹ לַֽיהוָֹ֖ה מִזֶּ֥בַח שְׁלָמָֽיו:  מ  דַּבֵּ֛ר אֶל-בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר הַמַּקְרִ֞יב אֶת-זֶ֤בַח שְׁלָמָיו֙ לַֽיהֹוָ֔ה יָבִ֧יא אֶת-קָרְבָּנ֛וֹ לַֽיהוָֹ֖ה מִזֶּ֥בַח שְׁלָמָֽיו:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל לְמֵימָר דִמְקָרֵב יָת נִכְסַת קוּדְשׁוֹהִי קֳדָם יְיָ יַיְתִי יָת קֻרְבָּנֵהּ לָקֳדָם יְיָ מִנִכְסַת קוּדְשׁוֹהִי:

{ל} יָדָ֣יו תְּבִיאֶ֔ינָה אֵ֖ת אִשֵּׁ֣י יְהוָֹ֑ה אֶת-הַחֵ֤לֶב עַל-הֶֽחָזֶה֙ יְבִיאֶ֔נּוּ אֵ֣ת הֶֽחָזֶ֗ה לְהָנִ֥יף אֹת֛וֹ תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֹֽה:  מ  יָדָ֣יו תְּבִיאֶ֔ינָה אֵ֖ת אִשֵּׁ֣י יְהוָֹ֑ה אֶת-הַחֵ֤לֶב עַל-הֶֽחָזֶה֙ יְבִיאֶ֔נּוּ אֵ֣ת הֶֽחָזֶ֗ה לְהָנִ֥יף אֹת֛וֹ תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֹֽה:  ת  יְדוֹהִי תַיְתוּן יָת קֻרְבָּנַיָא דַיְיָ יָת תַּרְבָּא עַל חַדְיָא יַיְתִנֵהּ יָת חֶדְיָא לַאֲרָמָא יָתֵהּ אֲרוּמָא קֳדָם יְיָ:

{לא} וְהִקְטִ֧יר הַכֹּהֵ֛ן אֶת-הַחֵ֖לֶב הַמִּזְבֵּ֑חָה וְהָיָה֙ הֶֽחָזֶ֔ה לְאַֽהֲרֹ֖ן וּלְבָנָֽיו:  מ  וְהִקְטִ֧יר הַכֹּהֵ֛ן אֶת-הַחֵ֖לֶב הַמִּזְבֵּ֑חָה וְהָיָה֙ הֶֽחָזֶ֔ה לְאַֽהֲרֹ֖ן וּלְבָנָֽיו:  ת  וְיַסֵק כַּהֲנָא יָת תַּרְבָּא לְמַדְבְּחָא וִיהֵי חֶדְיָא לְאַהֲרֹן וְלִבְנוֹהִי:

{לב} וְאֵת֙ שׁ֣וֹק הַיָּמִ֔ין תִּתְּנ֥וּ תְרוּמָ֖ה לַכֹּהֵ֑ן מִזִּבְחֵ֖י שַׁלְמֵיכֶֽם:  מ  וְאֵת֙ שׁ֣וֹק הַיָּמִ֔ין תִּתְּנ֥וּ תְרוּמָ֖ה לַכֹּהֵ֑ן מִזִּבְחֵ֖י שַׁלְמֵיכֶֽם:  ת  וְיָת שׁוֹקָא דְיַמִינָא תִּתְּנוּן אַפְרָשׁוּתָא לְכַהֲנָא מִנִכְסַת קוּדְשֵׁיכוֹן:

{לג} הַמַּקְרִ֞יב אֶת-דַּ֧ם הַשְּׁלָמִ֛ים וְאֶת-הַחֵ֖לֶב מִבְּנֵ֣י אַֽהֲרֹ֑ן ל֧וֹ תִֽהְיֶ֛ה שׁ֥וֹק הַיָּמִ֖ין לְמָנָֽה:  מ  הַמַּקְרִ֞יב אֶת-דַּ֧ם הַשְּׁלָמִ֛ים וְאֶת-הַחֵ֖לֶב מִבְּנֵ֣י אַֽהֲרֹ֑ן ל֧וֹ תִֽהְיֶ֛ה שׁ֥וֹק הַיָּמִ֖ין לְמָנָֽה:  ת  דִמְקָרֵב יָת דָם נִכְסַת קוּדְשַׁיָא וְיָת תַּרְבָּא מִבְּנֵי אַהֲרֹן דִילֵהּ תְּהֵי שׁוֹקָא דְיַמִינָא לָחֳלָק:

{לד} כִּי֩ אֶת-חֲזֵ֨ה הַתְּנוּפָ֜ה וְאֵ֣ת | שׁ֣וֹק הַתְּרוּמָ֗ה לָקַ֨חְתִּי֙ מֵאֵ֣ת בְּנֵֽי-יִשְׂרָאֵ֔ל מִזִּבְחֵ֖י שַׁלְמֵיהֶ֑ם וָֽאֶתֵּ֣ן אֹ֠תָ֠ם לְאַֽהֲרֹ֨ן הַכֹּהֵ֤ן וּלְבָנָיו֙ לְחָק-עוֹלָ֔ם מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  מ  כִּי֩ אֶת-חֲזֵ֨ה הַתְּנוּפָ֜ה וְאֵ֣ת | שׁ֣וֹק הַתְּרוּמָ֗ה לָקַ֨חְתִּי֙ מֵאֵ֣ת בְּנֵֽי-יִשְׂרָאֵ֔ל מִזִּבְחֵ֖י שַׁלְמֵיהֶ֑ם וָֽאֶתֵּ֣ן אֹ֠תָ֠ם לְאַֽהֲרֹ֨ן הַכֹּהֵ֤ן וּלְבָנָיו֙ לְחָק-עוֹלָ֔ם מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  ת  אֲרֵי יָת חֶדְיָא דַאֲרָמוּתָא וְיָת שׁוֹקָא דְאַפְרָשׁוּתָא נְסֵבִית מִן בְּנֵי יִשְׂרָאֵל מִנִכְסַת קוּדְשֵׁיהוֹן וִיהָבִית יָתְהוֹן לְאַהֲרֹן כַּהֲנָא וְלִבְנוֹהִי לִקְיָם עָלָם מִן בְּנֵי יִשְׂרָאֵל:

{לה} זֹ֣את מִשְׁחַ֤ת אַֽהֲרֹן֙ וּמִשְׁחַ֣ת בָּנָ֔יו מֵֽאִשֵּׁ֖י יְהוָֹ֑ה בְּיוֹם֙ הִקְרִ֣יב אֹתָ֔ם לְכַהֵ֖ן לַֽיהוָֹֽה:  מ  זֹ֣את מִשְׁחַ֤ת אַֽהֲרֹן֙ וּמִשְׁחַ֣ת בָּנָ֔יו מֵֽאִשֵּׁ֖י יְהוָֹ֑ה בְּיוֹם֙ הִקְרִ֣יב אֹתָ֔ם לְכַהֵ֖ן לַֽיהוָֹֽה:  ת  דָא רְבוּת אַהֲרֹן וּרְבוּת בְּנוֹהִי מִקוּרְבָּנַיָא דַיְיָ בְּיוֹמָא דְקָרֵב יָתְהוֹן לְשַׁמָשָׁא קֳדָם יְיָ:

{לו} אֲ֠שֶׁ֠ר צִוָּ֨ה יְהֹוָ֜ה לָתֵ֣ת לָהֶ֗ם בְּיוֹם֙ מָשְׁח֣וֹ אֹתָ֔ם מֵאֵ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל חֻקַּ֥ת עוֹלָ֖ם לְדֹֽרֹתָֽם:  מ  אֲ֠שֶׁ֠ר צִוָּ֨ה יְהֹוָ֜ה לָתֵ֣ת לָהֶ֗ם בְּיוֹם֙ מָשְׁח֣וֹ אֹתָ֔ם מֵאֵ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל חֻקַּ֥ת עוֹלָ֖ם לְדֹֽרֹתָֽם:  ת  דִי פַקִיד יְיָ לְמִתַּן לְהוֹן בְּיוֹמָא דְרַבִּי יָתְהוֹן מִן בְּנֵי יִשְׂרָאֵל קְיָם עָלָם לְדָרֵיהוֹן:

{לז} זֹ֣את הַתּוֹרָ֗ה לָֽעֹלָה֙ לַמִּנְחָ֔ה וְלַֽחַטָּ֖את וְלָֽאָשָׁ֑ם וְלַמִּלּוּאִ֔ים וּלְזֶ֖בַח הַשְּׁלָמִֽים:  מ  זֹ֣את הַתּוֹרָ֗ה לָֽעֹלָה֙ לַמִּנְחָ֔ה וְלַֽחַטָּ֖את וְלָֽאָשָׁ֑ם וְלַמִּלּוּאִ֔ים וּלְזֶ֖בַח הַשְּׁלָמִֽים:  ת  דָא אוֹרַיְתָא לַעֲלָתָא לְמִנְחָתָא וּלְחַטָּאתָא וְלַאֲשָׁמָא וּלְקוּרְבָּנַיָא וּלְנִכְסַת קוּדְשַׁיָא:

{לח} אֲשֶׁ֨ר צִוָּ֧ה יְהוָֹ֛ה אֶת-מֹשֶׁ֖ה בְּהַ֣ר סִינָ֑י בְּי֨וֹם צַוֹּת֜וֹ אֶת-בְּנֵ֣י יִשְׂרָאֵ֗ל לְהַקְרִ֧יב אֶת-קָרְבְּנֵיהֶ֛ם לַֽיהוָֹ֖ה בְּמִדְבַּ֥ר סִינָֽי:  מ  אֲשֶׁ֨ר צִוָּ֧ה יְהוָֹ֛ה אֶת-מֹשֶׁ֖ה בְּהַ֣ר סִינָ֑י בְּי֨וֹם צַוֹּת֜וֹ אֶת-בְּנֵ֣י יִשְׂרָאֵ֗ל לְהַקְרִ֧יב אֶת-קָרְבְּנֵיהֶ֛ם לַֽיהוָֹ֖ה בְּמִדְבַּ֥ר סִינָֽי:  ת  דִי פַקִיד יְיָ יָת מֹשֶׁה בְּטוּרָא דְסִינָי בְּיוֹמָא דְפַקְדֵהּ יָת בְּנֵי יִשְׂרָאֵל לְקָרָבָא יָת קֻרְבָּנְהוֹן קֳדָם יְיָ בְּמַדְבְּרָא דְסִינָי: (פ)




ויקרא פרק-ח

{א}  רביעי  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} קַ֤ח אֶֽת-אַֽהֲרֹן֙ וְאֶת-בָּנָ֣יו אִתּ֔וֹ וְאֵת֙ הַבְּגָדִ֔ים וְאֵ֖ת שֶׁ֣מֶן הַמִּשְׁחָ֑ה וְאֵ֣ת | פַּ֣ר הַֽחַטָּ֗את וְאֵת֙ שְׁנֵ֣י הָֽאֵילִ֔ים וְאֵ֖ת סַ֥ל הַמַּצּֽוֹת:  מ  קַ֤ח אֶֽת-אַֽהֲרֹן֙ וְאֶת-בָּנָ֣יו אִתּ֔וֹ וְאֵת֙ הַבְּגָדִ֔ים וְאֵ֖ת שֶׁ֣מֶן הַמִּשְׁחָ֑ה וְאֵ֣ת | פַּ֣ר הַֽחַטָּ֗את וְאֵת֙ שְׁנֵ֣י הָֽאֵילִ֔ים וְאֵ֖ת סַ֥ל הַמַּצּֽוֹת:  ת  קָרֵב יָת אַהֲרֹן וְיָת בְּנוֹהִי עִמֵהּ וְיָת לְבוּשַׁיָא וְיָת מִשְׁחָא דִרְבוּתָא וְיָת תּוֹרָא דְחַטָאתָא וְיָת תְּרֵין דִכְרִין וְיָת סַלָא דְפַטִירַיָא:

{ג} וְאֵ֥ת כָּל-הָֽעֵדָ֖ה הַקְהֵ֑ל אֶל-פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  מ  וְאֵ֥ת כָּל-הָֽעֵדָ֖ה הַקְהֵ֑ל אֶל-פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  ת  וְיָת כָּל כְּנִשְׁתָּא כְּנֵשׁ לִתְרַע מַשְׁכַּן זִמְנָא:

{ד} וַיַּ֣עַשׂ מֹשֶׁ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֹת֑וֹ וַתִּקָּהֵל֙ הָֽעֵדָ֔ה אֶל-פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  מ  וַיַּ֣עַשׂ מֹשֶׁ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֹת֑וֹ וַתִּקָּהֵל֙ הָֽעֵדָ֔ה אֶל-פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  ת  וַעֲבַד מֹשֶׁה כְּמָא דִי פַקִיד יְיָ יָתֵהּ וְאִתְכְּנִשַׁת כְּנִשְׁתָּא לִתְרַע מַשְׁכַּן זִמְנָא:

{ה} וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל-הָֽעֵדָ֑ה זֶ֣ה הַדָּבָ֔ר אֲשֶׁר-צִוָּ֥ה יְהוָֹה֖ לַֽעֲשֽׂוֹת:  מ  וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל-הָֽעֵדָ֑ה זֶ֣ה הַדָּבָ֔ר אֲשֶׁר-צִוָּ֥ה יְהוָֹה֖ לַֽעֲשֽׂוֹת:  ת  וַאֲמַר מֹשֶׁה לִכְנִשְׁתָּא דֵין פִּתְגָמָא דִי פַקִיד יְיָ לְמֶעְבָּד:

{ו} וַיַּקְרֵ֣ב מֹשֶׁ֔ה אֶֽת-אַֽהֲרֹ֖ן וְאֶת-בָּנָ֑יו וַיִּרְחַ֥ץ אֹתָ֖ם בַּמָּֽיִם:  מ  וַיַּקְרֵ֣ב מֹשֶׁ֔ה אֶֽת-אַֽהֲרֹ֖ן וְאֶת-בָּנָ֑יו וַיִּרְחַ֥ץ אֹתָ֖ם בַּמָּֽיִם:  ת  וְקָרִיב מֹשֶׁה יָת אַהֲרֹן וְיָת בְּנוֹהִי וְאַסְחֵי יָתְהוֹן בְּמַיָא:

{ז} וַיִּתֵּ֨ן עָלָ֜יו אֶת-הַכֻּתֹּ֗נֶת וַיַּחְגֹּ֤ר אֹתוֹ֙ בָּֽאַבְנֵ֔ט וַיַּלְבֵּ֤שׁ אֹתוֹ֙ אֶֽת-הַמְּעִ֔יל וַיִּתֵּ֥ן עָלָ֖יו אֶת-הָֽאֵפֹ֑ד וַיַּחְגֹּ֣ר אֹת֗וֹ בְּחֵ֨שֶׁב֙ הָֽאֵפֹ֔ד וַיֶּאְפֹּ֥ד ל֖וֹ בּֽוֹ:  מ  וַיִּתֵּ֨ן עָלָ֜יו אֶת-הַכֻּתֹּ֗נֶת וַיַּחְגֹּ֤ר אֹתוֹ֙ בָּֽאַבְנֵ֔ט וַיַּלְבֵּ֤שׁ אֹתוֹ֙ אֶֽת-הַמְּעִ֔יל וַיִּתֵּ֥ן עָלָ֖יו אֶת-הָֽאֵפֹ֑ד וַיַּחְגֹּ֣ר אֹת֗וֹ בְּחֵ֨שֶׁב֙ הָֽאֵפֹ֔ד וַיֶּאְפֹּ֥ד ל֖וֹ בּֽוֹ:  ת  וִיהַב עֲלוֹהִי יָת כִּתּוּנָא וְזָרֵז יָתֵהּ בְּהֶמְיָנָא וְאַלְבֵּשׁ יָתֵהּ יָת מְעִילָא וִיהַב עֲלוֹהִי יָת אֵפוֹדָא וְזָרֵז יָתֵהּ בְּהֶמְיַן אֵפוֹדָא וְאַתְקֵן לֵהּ בֵּהּ:

{ח} וַיָּ֥שֶׂם עָלָ֖יו אֶת-הַח֑שֶׁן וַיִּתֵּן֙ אֶל-הַחֹ֔שֶׁן אֶת-הָֽאוּרִ֖ים וְאֶת-הַתֻּמִּֽים:  מ  וַיָּ֥שֶׂם עָלָ֖יו אֶת-הַח֑שֶׁן וַיִּתֵּן֙ אֶל-הַחֹ֔שֶׁן אֶת-הָֽאוּרִ֖ים וְאֶת-הַתֻּמִּֽים:  ת  וְשַׁוִי עֲלוֹהִי יָת חוּשְׁנָא וִיהַב בְּחוּשְׁנָא יָת אוּרַיָא וְיָת תֻּמַיָא:

{ט} וַיָּ֥שֶׂם אֶת-הַמִּצְנֶ֖פֶת עַל-רֹאשׁ֑וֹ וַיָּ֨שֶׂם עַל-הַמִּצְנֶ֜פֶת אֶל-מ֣וּל פָּנָ֗יו אֵ֣ת צִ֤יץ הַזָּהָב֙ נֵ֣זֶר הַקֹּ֔דֶשׁ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיָּ֥שֶׂם אֶת-הַמִּצְנֶ֖פֶת עַל-רֹאשׁ֑וֹ וַיָּ֨שֶׂם עַל-הַמִּצְנֶ֜פֶת אֶל-מ֣וּל פָּנָ֗יו אֵ֣ת צִ֤יץ הַזָּהָב֙ נֵ֣זֶר הַקֹּ֔דֶשׁ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְשַׁוִי יָת מִצְנַפְתָּא עַל רֵישֵׁיהּ וְשַׁוִי עַל מִצְנֶפְתָּא לָקֳבֵיל אַפּוֹהִי יָת צִיצָא דְדַהֲבָא כְּלִילָא דְקוּדְשָׁא כְּמָא דִי פַקִיד יְיָ יָת מֹשֶׁה:

{י} וַיִּקַּ֤ח מֹשֶׁה֙ אֶת-שֶׁ֣מֶן הַמִּשְׁחָ֔ה וַיִּמְשַׁ֥ח אֶת-הַמִּשְׁכָּ֖ן וְאֶת-כָּל-אֲשֶׁר-בּ֑וֹ וַיְקַדֵּ֖שׁ אֹתָֽם:  מ  וַיִּקַּ֤ח מֹשֶׁה֙ אֶת-שֶׁ֣מֶן הַמִּשְׁחָ֔ה וַיִּמְשַׁ֥ח אֶת-הַמִּשְׁכָּ֖ן וְאֶת-כָּל-אֲשֶׁר-בּ֑וֹ וַיְקַדֵּ֖שׁ אֹתָֽם:  ת  וּנְסִיב מֹשֶׁה יָתּ מִשְׁחָא דִרְבוּתָא וְרַבִּי יָת מַשְׁכְּנָא וְיָת כָּל דִי בֵהּ וְקַדִישׁ יָתְהוֹן:

{יא} וַיַּ֥ז מִמֶּ֛נּוּ עַל-הַמִּזְבֵּ֖חַ שֶׁ֣בַע פְּעָמִ֑ים וַיִּמְשַׁ֨ח אֶת-הַמִּזְבֵּ֜חַ וְאֶת-כָּל-כֵּלָ֗יו וְאֶת-הַכִּיֹּ֛ר וְאֶת-כַּנּ֖וֹ לְקַדְּשָֽׁם:  מ  וַיַּ֥ז מִמֶּ֛נּוּ עַל-הַמִּזְבֵּ֖חַ שֶׁ֣בַע פְּעָמִ֑ים וַיִּמְשַׁ֨ח אֶת-הַמִּזְבֵּ֜חַ וְאֶת-כָּל-כֵּלָ֗יו וְאֶת-הַכִּיֹּ֛ר וְאֶת-כַּנּ֖וֹ לְקַדְּשָֽׁם:  ת  וְאַדִי מִנֵהּ עַל מַדְבְּחָא שְׁבַע זִמְנִין וְרַבִּי יָת מַדְבְּחָא וְיָת כָּל מָנוֹהִי וְיָת כִּיוֹרָא וְיָת בְּסִיסֵהּ לְקַדָשׁוּתְהוֹן:

{יב} וַיִּצֹק֙ מִשֶּׁ֣מֶן הַמִּשְׁחָ֔ה עַ֖ל רֹ֣אשׁ אַֽהֲרֹ֑ן וַיִּמְשַׁ֥ח אֹת֖וֹ לְקַדְּשֽׁוֹ:  מ  וַיִּצֹק֙ מִשֶּׁ֣מֶן הַמִּשְׁחָ֔ה עַ֖ל רֹ֣אשׁ אַֽהֲרֹ֑ן וַיִּמְשַׁ֥ח אֹת֖וֹ לְקַדְּשֽׁוֹ:  ת  וַאֲרֵיק מִמִשְׁחָא דִרְבוּתָא עַל רֵישָׁא דְאַהֲרֹן וְרַבִּי יָתֵהּ לְקַדָשׁוּתֵהּ:

{יג} וַיַּקְרֵ֨ב מֹשֶׁ֜ה אֶת-בְּנֵ֣י אַֽהֲרֹ֗ן וַיַּלְבִּשֵׁ֤ם כֻּתֳּנֹת֙ וַיַּחְגֹּ֤ר אֹתָם֙ אַבְנֵ֔ט וַיַּֽחֲבֹ֥שׁ לָהֶ֖ם מִגְבָּע֑וֹת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיַּקְרֵ֨ב מֹשֶׁ֜ה אֶת-בְּנֵ֣י אַֽהֲרֹ֗ן וַיַּלְבִּשֵׁ֤ם כֻּתֳּנֹת֙ וַיַּחְגֹּ֤ר אֹתָם֙ אַבְנֵ֔ט וַיַּֽחֲבֹ֥שׁ לָהֶ֖ם מִגְבָּע֑וֹת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְקָרִיב מֹשֶׁה יָת בְּנֵי אַהֲרֹן וְאַלְבֵּשִׁנוּן כִּתּוּנִין וְזָרֵז יָתְהוֹן הֶמְיָנִין וְאַתְקֵן לְהוֹן כּוֹבָעִין כְּמָא דִי פַקִיד יְיָ יָת מֹשֶׁה:

{יד}  חמישי  וַיַּגֵּ֕שׁ אֵ֖ת פַּ֣ר הַֽחַטָּ֑את וַיִּסְמֹ֨ךְ אַֽהֲרֹ֤ן וּבָנָיו֙ אֶת-יְדֵיהֶ֔ם עַל-רֹ֖אשׁ פַּ֥ר הַֽחַטָּֽאת:  מ  וַיַּגֵּ֕שׁ אֵ֖ת פַּ֣ר הַֽחַטָּ֑את וַיִּסְמֹ֨ךְ אַֽהֲרֹ֤ן וּבָנָיו֙ אֶת-יְדֵיהֶ֔ם עַל-רֹ֖אשׁ פַּ֥ר הַֽחַטָּֽאת:  ת  וְקָרֵב יָת תּוֹרָא דְחַטָאתָא וּסְמַךְ אַהֲרֹן וּבְנוֹהִי יָת יְדֵיהוֹן עַל רֵישׁ תּוֹרָא דְחַטָאתָא:

{טו} וַיִּשְׁחָ֗ט וַיִּקַּ֨ח מֹשֶׁ֤ה אֶת-הַדָּם֙ וַ֠יִּתֵּ֠ן עַל-קַרְנ֨וֹת הַמִּזְבֵּ֤חַ סָבִיב֙ בְּאֶצְבָּע֔וֹ וַיְחַטֵּ֖א אֶת-הַמִּזְבֵּ֑חַ וְאֶת-הַדָּ֗ם יָצַק֙ אֶל-יְס֣וֹד הַמִּזְבֵּ֔חַ וַֽיְקַדְּשֵׁ֖הוּ לְכַפֵּ֥ר עָלָֽיו:  מ  וַיִּשְׁחָ֗ט וַיִּקַּ֨ח מֹשֶׁ֤ה אֶת-הַדָּם֙ וַ֠יִּתֵּ֠ן עַל-קַרְנ֨וֹת הַמִּזְבֵּ֤חַ סָבִיב֙ בְּאֶצְבָּע֔וֹ וַיְחַטֵּ֖א אֶת-הַמִּזְבֵּ֑חַ וְאֶת-הַדָּ֗ם יָצַק֙ אֶל-יְס֣וֹד הַמִּזְבֵּ֔חַ וַֽיְקַדְּשֵׁ֖הוּ לְכַפֵּ֥ר עָלָֽיו:  ת  וּנְכֵס וּנְסִיב מֹשֶׁה יָת דְמָא וִיהַב עַל קַרְנַת מַדְבְּחָא סְחוֹר סְחוֹר בְּאֶצְבְּעֵהּ וְדַכִּי יָת מַדְבְּחָא וְיָת דְמָא אֲרִיק לִיסוֹדָא דְמַדְבְּחָא וְקַדְשֵׁהּ לְכַפָּרָא עֲלוֹהִי:

{טז} וַיִּקַּ֗ח אֶֽת-כָּל-הַחֵ֘לֶב֘ אֲשֶׁ֣ר עַל-הַקֶּ֒רֶב֒ וְאֵת֙ יֹתֶ֣רֶת הַכָּבֵ֔ד וְאֶת-שְׁתֵּ֥י הַכְּלָיֹ֖ת וְאֶת-חֶלְבְּהֶ֑ן וַיַּקְטֵ֥ר מֹשֶׁ֖ה הַמִּזְבֵּֽחָה:  מ  וַיִּקַּ֗ח אֶֽת-כָּל-הַחֵ֘לֶב֘ אֲשֶׁ֣ר עַל-הַקֶּ֒רֶב֒ וְאֵת֙ יֹתֶ֣רֶת הַכָּבֵ֔ד וְאֶת-שְׁתֵּ֥י הַכְּלָיֹ֖ת וְאֶת-חֶלְבְּהֶ֑ן וַיַּקְטֵ֥ר מֹשֶׁ֖ה הַמִּזְבֵּֽחָה:  ת  וּנְסִיב יָת כָּל תַּרְבָּא דִי עַל גַוָא וְיָת חֲצַר כַּבְדָא וְיָת תַּרְתֵּין כָּלְיָן וְיָת תַּרְבְּהֶן וְאַסֵק מֹשֶׁה לְמַדְבְּחָא:

{יז} וְאֶת-הַפָּ֤ר וְאֶת-עֹרוֹ֙ וְאֶת-בְּשָׂר֣וֹ וְאֶת-פִּרְשׁ֔וֹ שָׂרַ֣ף בָּאֵ֔שׁ מִח֖וּץ לַֽמַּֽחֲנֶה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וְאֶת-הַפָּ֤ר וְאֶת-עֹרוֹ֙ וְאֶת-בְּשָׂר֣וֹ וְאֶת-פִּרְשׁ֔וֹ שָׂרַ֣ף בָּאֵ֔שׁ מִח֖וּץ לַֽמַּֽחֲנֶה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְיָת תּוֹרָא וְיָת מַשְׁכֵהּ וְיָת בִּסְרֵהּ וְיָת אָכְלֵהּ אוֹקִיד בְּנוּרָא מִבָּרָא לְמַשְׁרִיתָא כְּמָא דִי פַקִיד יְיָ יָת מֹשֶׁה:

{יח} וַיַּקְרֵ֕ב אֵ֖ת אֵ֣יל הָֽעֹלָ֑ה וַֽיִּסְמְכ֞וּ אַֽהֲרֹ֧ן וּבָנָ֛יו אֶת-יְדֵיהֶ֖ם עַל-רֹ֥אשׁ הָאָֽיִל:  מ  וַיַּקְרֵ֕ב אֵ֖ת אֵ֣יל הָֽעֹלָ֑ה וַֽיִּסְמְכ֞וּ אַֽהֲרֹ֧ן וּבָנָ֛יו אֶת-יְדֵיהֶ֖ם עַל-רֹ֥אשׁ הָאָֽיִל:  ת  וְקָרִיב יָת דִכְרָא דַעֲלָתָא וּסְמָכוּ אַהֲרֹן וּבְנוֹהִי יָת יְדֵיהוֹן עַל רֵישׁ דִכְרָא:

{יט} וַיִּשְׁחָ֑ט וַיִּזְרֹ֨ק מֹשֶׁ֧ה אֶת-הַדָּ֛ם עַל-הַמִּזְבֵּ֖חַ סָבִֽיב:  מ  וַיִּשְׁחָ֑ט וַיִּזְרֹ֨ק מֹשֶׁ֧ה אֶת-הַדָּ֛ם עַל-הַמִּזְבֵּ֖חַ סָבִֽיב:  ת  וּנְכֵס וּזְרַק מֹשֶׁה יָת דְמָא עַל מַדְבְּחָא סְחוֹר סְחוֹר:

{כ} וְאֶ֨ת-הָאַ֔יִל נִתַּ֖ח לִנְתָחָ֑יו וַיַּקְטֵ֤ר מֹשֶׁה֙ אֶת-הָרֹ֔אשׁ וְאֶת-הַנְּתָחִ֖ים וְאֶת-הַפָּֽדֶר:  מ  וְאֶ֨ת-הָאַ֔יִל נִתַּ֖ח לִנְתָחָ֑יו וַיַּקְטֵ֤ר מֹשֶׁה֙ אֶת-הָרֹ֔אשׁ וְאֶת-הַנְּתָחִ֖ים וְאֶת-הַפָּֽדֶר:  ת  וְיָת דִכְרָא פַּלֵג לְאֶבְרוֹהִי וְאַסֵק מֹשֶׁה יָת רֵישָׁא וְיָת אֶבְרַיָא וְיָת תַּרְבָּא:

{כא} וְאֶת-הַקֶּ֥רֶב וְאֶת-הַכְּרָעַ֖יִם רָחַ֣ץ בַּמָּ֑יִם וַיַּקְטֵר֩ מֹשֶׁ֨ה אֶת-כָּל-הָאַ֜יִל הַמִּזְבֵּ֗חָה עֹלָ֨ה ה֤וּא לְרֵֽיחַ-נִיחֹ֨חַ֙ אִשֶּׁ֥ה הוּא֙ לַֽיהֹוָ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וְאֶת-הַקֶּ֥רֶב וְאֶת-הַכְּרָעַ֖יִם רָחַ֣ץ בַּמָּ֑יִם וַיַּקְטֵר֩ מֹשֶׁ֨ה אֶת-כָּל-הָאַ֜יִל הַמִּזְבֵּ֗חָה עֹלָ֨ה ה֤וּא לְרֵֽיחַ-נִיחֹ֨חַ֙ אִשֶּׁ֥ה הוּא֙ לַֽיהֹוָ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְיָת גַוָא וְיָת כְּרָעַיָא חַלִיל בְּמַיָא וְאַסֵק מֹשֶׁה יָת כָּל דִכְרָא לְמַדְבְּחָא עֲלָתָא הוּא לְאִתְקַבָּלָא בְרַעִוָא קוּרְבָּנָא הוּא קֳדָם יְיָ כְּמָא דִי פַקִיד יְיָ יָת מֹשֶׁה:

{כב}  שישי  וַיַּקְרֵב֙ אֶת-הָאַ֣יִל הַשֵּׁנִ֔י אֵ֖יל הַמִּלֻּאִ֑ים וַֽיִּסְמְכ֞וּ אַֽהֲרֹ֧ן וּבָנָ֛יו אֶת-יְדֵיהֶ֖ם עַל-רֹ֥אשׁ הָאָֽיִל:  מ  וַיַּקְרֵב֙ אֶת-הָאַ֣יִל הַשֵּׁנִ֔י אֵ֖יל הַמִּלֻּאִ֑ים וַֽיִּסְמְכ֞וּ אַֽהֲרֹ֧ן וּבָנָ֛יו אֶת-יְדֵיהֶ֖ם עַל-רֹ֥אשׁ הָאָֽיִל:  ת  וְקָרִיב יָת דִכְרָא תִנְיָנָא דְכַר קוּרְבָּנַיָא וּסְמָכוּ אַהֲרֹן וּבְנוֹהִי יָת יְדֵיהוֹן עַל רֵישׁ דִכְרָא:

{כג} וַיִּשְׁחָ֓ט | וַיִּקַּ֤ח מֹשֶׁה֙ מִדָּמ֔וֹ וַיִּתֵּ֛ן עַל-תְּנ֥וּךְ אֹֽזֶן-אַֽהֲרֹ֖ן הַיְמָנִ֑ית וְעַל-בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל-בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִֽית:  מ  וַיִּשְׁחָ֓ט | וַיִּקַּ֤ח מֹשֶׁה֙ מִדָּמ֔וֹ וַיִּתֵּ֛ן עַל-תְּנ֥וּךְ אֹֽזֶן-אַֽהֲרֹ֖ן הַיְמָנִ֑ית וְעַל-בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל-בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִֽית:  ת  וּנְכֵס וּנְסִיב מֹשֶׁה מִן דְמָא וִיהַב עַל רוּם אוּדְנָא דְאַהֲרֹן דְיַמִינָא וְעַל אִלְיוֹן יְדֵהּ דְיַמִינָא וְעַל אִלְיוֹן רַגְלֵהּ דְיַמִינָא:

{כד} וַיַּקְרֵ֞ב אֶת-בְּנֵ֣י אַֽהֲרֹ֗ן וַיִּתֵּ֨ן מֹשֶׁ֤ה מִן-הַדָּם֙ עַל-תְּנ֤וּךְ אָזְנָם֙ הַיְמָנִ֔ית וְעַל-בֹּ֤הֶן יָדָם֙ הַיְמָנִ֔ית וְעַל-בֹּ֥הֶן רַגְלָ֖ם הַיְמָנִ֑ית וַיִּזְרֹ֨ק מֹשֶׁ֧ה אֶת-הַדָּ֛ם עַל-הַמִּזְבֵּ֖חַ סָבִֽיב:  מ  וַיַּקְרֵ֞ב אֶת-בְּנֵ֣י אַֽהֲרֹ֗ן וַיִּתֵּ֨ן מֹשֶׁ֤ה מִן-הַדָּם֙ עַל-תְּנ֤וּךְ אָזְנָם֙ הַיְמָנִ֔ית וְעַל-בֹּ֤הֶן יָדָם֙ הַיְמָנִ֔ית וְעַל-בֹּ֥הֶן רַגְלָ֖ם הַיְמָנִ֑ית וַיִּזְרֹ֨ק מֹשֶׁ֧ה אֶת-הַדָּ֛ם עַל-הַמִּזְבֵּ֖חַ סָבִֽיב:  ת  וְקָרֵיב יָת בְּנֵי אַהֲרֹן וִיהַב מֹשֶׁה מִן דְמָא עַל רוּם אוּדְנֵהוֹן דְיַמִינָא וְעַל אִלְיוֹן אוּדְנֵהוֹן דְיַמִינָא וְעַל אִלְיוֹן רַגְלֵיהוֹן דְיַמִינָא וּזְרַק מֹשֶׁה יָת דְמָא עַל מַדְבְּחָא סְחוֹר סְחוֹר:

{כה} וַיִּקַּ֞ח אֶת-הַחֵ֣לֶב וְאֶֽת-הָֽאַלְיָ֗ה וְאֶֽת-כָּל-הַחֵ֘לֶב֘ אֲשֶׁ֣ר עַל-הַקֶּ֒רֶב֒ וְאֵת֙ יֹתֶ֣רֶת הַכָּבֵ֔ד וְאֶת-שְׁתֵּ֥י הַכְּלָיֹ֖ת וְאֶת-חֶלְבְּהֶ֑ן וְאֵ֖ת שׁ֥וֹק הַיָּמִֽין:  מ  וַיִּקַּ֞ח אֶת-הַחֵ֣לֶב וְאֶֽת-הָֽאַלְיָ֗ה וְאֶֽת-כָּל-הַחֵ֘לֶב֘ אֲשֶׁ֣ר עַל-הַקֶּ֒רֶב֒ וְאֵת֙ יֹתֶ֣רֶת הַכָּבֵ֔ד וְאֶת-שְׁתֵּ֥י הַכְּלָיֹ֖ת וְאֶת-חֶלְבְּהֶ֑ן וְאֵ֖ת שׁ֥וֹק הַיָּמִֽין:  ת  וּנְסִיב יָת תַּרְבָּא וְיָת אֲלִיתָא וְיָת כָּל תַּרְבָּא דִי עַל גַוָא וְיָת חֲצַר כַּבְדָא וְיָת תַּרְתֵּין כָּלְיָן וְיָת תַּרְבְּהֶן וְיָת שׁוֹקָא דְיַמִינָא:

{כו} וּמִסַּ֨ל הַמַּצּ֜וֹת אֲשֶׁ֣ר | לִפְנֵ֣י יְהֹוָ֗ה לָ֠קַ֠ח חַלַּ֨ת מַצָּ֤ה אַחַת֙ וְֽחַלַּ֨ת לֶ֥חֶם שֶׁ֛מֶן אַחַ֖ת וְרָקִ֣יק אֶחָ֑ד וַיָּ֨שֶׂם֙ עַל-הַ֣חֲלָבִ֔ים וְעַ֖ל שׁ֥וֹק הַיָּמִֽין:  מ  וּמִסַּ֨ל הַמַּצּ֜וֹת אֲשֶׁ֣ר | לִפְנֵ֣י יְהֹוָ֗ה לָ֠קַ֠ח חַלַּ֨ת מַצָּ֤ה אַחַת֙ וְֽחַלַּ֨ת לֶ֥חֶם שֶׁ֛מֶן אַחַ֖ת וְרָקִ֣יק אֶחָ֑ד וַיָּ֨שֶׂם֙ עַל-הַ֣חֲלָבִ֔ים וְעַ֖ל שׁ֥וֹק הַיָּמִֽין:  ת  וּמִסַלָא דְפַטִירַיָא דִי קֳדָם יְיָ נְסִיב גְרִצְתָּא פַטִירָא חֲדָא וּגְרִיצְתָּא דִלְחֵם מְשַׁח חֲדָא וְאֶסְפּוֹג חָד וְשַׁוִי עַל תַּרְבַּיָא וְעַל שׁוֹקָא דְיַמִינָא:

{כז} וַיִּתֵּ֣ן אֶת-הַכֹּ֔ל עַ֚ל כַּפֵּ֣י אַֽהֲרֹ֔ן וְעַ֖ל כַּפֵּ֣י בָנָ֑יו וַיָּ֧נֶף אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֹֽה:  מ  וַיִּתֵּ֣ן אֶת-הַכֹּ֔ל עַ֚ל כַּפֵּ֣י אַֽהֲרֹ֔ן וְעַ֖ל כַּפֵּ֣י בָנָ֑יו וַיָּ֧נֶף אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֹֽה:  ת  וִיהַב יָת כֹּלָא עַל יְדֵי אַהֲרֹן וְעַל יְדֵי בְנוֹהִי וַאֲרֵים יָתְהוֹן אֲרָמָא קֳדָם יְיָ:

{כח} וַיִּקַּ֨ח מֹשֶׁ֤ה אֹתָם֙ מֵעַ֣ל כַּפֵּיהֶ֔ם וַיַּקְטֵ֥ר הַמִּזְבֵּ֖חָה עַל-הָֽעֹלָ֑ה מִ֨לֻּאִ֥ים הֵם֙ לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֥ה ה֖וּא לַֽיהוָֹֽה:  מ  וַיִּקַּ֨ח מֹשֶׁ֤ה אֹתָם֙ מֵעַ֣ל כַּפֵּיהֶ֔ם וַיַּקְטֵ֥ר הַמִּזְבֵּ֖חָה עַל-הָֽעֹלָ֑ה מִ֨לֻּאִ֥ים הֵם֙ לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֥ה ה֖וּא לַֽיהוָֹֽה:  ת  וּנְסִיב מֹשֶׁה יָתְהוֹן מֵעַל יְדֵיהוֹן וְאַסֵק לְמַדְבְּחָא עַל עֲלָתָא קוּרְבָּנַיָא אִנוּן לְאִתְקַבָּלָא בְרַעֲוָא קוּרְבָּנָא הוּא קֳדָם יְיָ:

{כט} וַיִּקַּ֤ח מֹשֶׁה֙ אֶת-הֶ֣חָזֶ֔ה וַיְנִיפֵ֥הוּ תְנוּפָ֖ה לִפְנֵ֣י יְהוָֹ֑ה מֵאֵ֣יל הַמִּלֻּאִ֗ים לְמֹשֶׁ֤ה הָיָה֙ לְמָנָ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיִּקַּ֤ח מֹשֶׁה֙ אֶת-הֶ֣חָזֶ֔ה וַיְנִיפֵ֥הוּ תְנוּפָ֖ה לִפְנֵ֣י יְהוָֹ֑ה מֵאֵ֣יל הַמִּלֻּאִ֗ים לְמֹשֶׁ֤ה הָיָה֙ לְמָנָ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וּנְסִיב מֹשֶׁה יָת חֶדְיָא וַאֲרִימֵהּ אֲרָמָא קֳדָם יְיָ מִדְכַר קוּרְבָּנַיָא לְמֹשֶׁה הֲוָה לָחֳלָק כְּמָא דִי פַקִיד יְיָ יָת מֹשֶׁה:

{ל}  שביעי  וַיִּקַּ֨ח מֹשֶׁ֜ה מִשֶּׁ֣מֶן הַמִּשְׁחָ֗ה וּמִן-הַדָּם֘ אֲשֶׁ֣ר עַל-הַמִּזְבֵּ֒חַ֒ וַיַּ֤ז עַֽל-אַֽהֲרֹן֙ עַל-בְּגָדָ֔יו וְעַל-בָּנָ֛יו וְעַל-בִּגְדֵ֥י בָנָ֖יו אִתּ֑וֹ וַיְקַדֵּ֤שׁ אֶֽת-אַֽהֲרֹן֙ אֶת-בְּגָדָ֔יו וְאֶת-בָּנָ֛יו וְאֶת-בִּגְדֵ֥י בָנָ֖יו אִתּֽוֹ:  מ  וַיִּקַּ֨ח מֹשֶׁ֜ה מִשֶּׁ֣מֶן הַמִּשְׁחָ֗ה וּמִן-הַדָּם֘ אֲשֶׁ֣ר עַל-הַמִּזְבֵּ֒חַ֒ וַיַּ֤ז עַֽל-אַֽהֲרֹן֙ עַל-בְּגָדָ֔יו וְעַל-בָּנָ֛יו וְעַל-בִּגְדֵ֥י בָנָ֖יו אִתּ֑וֹ וַיְקַדֵּ֤שׁ אֶֽת-אַֽהֲרֹן֙ אֶת-בְּגָדָ֔יו וְאֶת-בָּנָ֛יו וְאֶת-בִּגְדֵ֥י בָנָ֖יו אִתּֽוֹ:  ת  וּנְסִיב מֹשֶׁה מִמִשְׁחָא דִרְבוּתָא וּמִן דְמָא דִי עַל מַדְבְּחָא וְאַדִי עַל אַהֲרֹן עַל לְבוּשׁוֹהִי וְעַל בְּנוֹהִי וְעַל לְבוּשֵׁי בְנוֹהִי עִמֵהּ וְקַדִישׁ יָת אַהֲרֹן יָת לְבוּשׁוֹהִי וְיָת בְּנוֹהִי וְיָת לְבוּשֵׁי בְנוֹהִי עִמֵהּ:

{לא} וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל-אַֽהֲרֹ֣ן וְאֶל-בָּנָ֗יו בַּשְּׁל֣וּ אֶת-הַבָּשָׂר֘ פֶּ֣תַח אֹ֣הֶל מוֹעֵד֒ וְשָׁם֙ תֹּֽאכְל֣וּ אֹת֔וֹ וְאֶ֨ת-הַלֶּ֔חֶם אֲשֶׁ֖ר בְּסַ֣ל הַמִּלֻּאִ֑ים כַּֽאֲשֶׁ֤ר צִוֵּ֨יתִי֙ לֵאמֹ֔ר אַֽהֲרֹ֥ן וּבָנָ֖יו יֹֽאכְלֻֽהוּ:  מ  וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל-אַֽהֲרֹ֣ן וְאֶל-בָּנָ֗יו בַּשְּׁל֣וּ אֶת-הַבָּשָׂר֘ פֶּ֣תַח אֹ֣הֶל מוֹעֵד֒ וְשָׁם֙ תֹּֽאכְל֣וּ אֹת֔וֹ וְאֶ֨ת-הַלֶּ֔חֶם אֲשֶׁ֖ר בְּסַ֣ל הַמִּלֻּאִ֑ים כַּֽאֲשֶׁ֤ר צִוֵּ֨יתִי֙ לֵאמֹ֔ר אַֽהֲרֹ֥ן וּבָנָ֖יו יֹֽאכְלֻֽהוּ:  ת  וַאֲמַר מֹשֶׁה לְאַהֲרֹן וְלִבְנוֹהִי בַּשִׁילוּ יָת בִּסְרָא בִּתְרַע מַשְׁכַּן זִמְנָא וְתַמָן תֵּיכְלוּן יָתֵהּ וְיָת לַחְמָא דִי בְּסַל קוּרְבָּנַיָא כְּמָא דִי פַקֵדִית לְמֵימָר אַהֲרֹן וּבְנוֹהִי יֵיכְלוּנֵהּ:

{לב} וְהַנּוֹתָ֥ר בַּבָּשָׂ֖ר וּבַלָּ֑חֶם בָּאֵ֖שׁ תִּשְׂרֹֽפוּ:  מ  וְהַנּוֹתָ֥ר בַּבָּשָׂ֖ר וּבַלָּ֑חֶם בָּאֵ֖שׁ תִּשְׂרֹֽפוּ:  ת  וּדְאִשְׁתָּאַר בְּבִשְׂרָא וּבְלַחְמָא בְּנוּרָא תּוֹקְדוּן:

{לג}  מפטיר  וּמִפֶּ֩תַח֩ אֹ֨הֶל מוֹעֵ֜ד לֹ֤א תֵֽצְאוּ֙ שִׁבְעַ֣ת יָמִ֔ים עַ֚ד י֣וֹם מְלֹ֔את יְמֵ֖י מִלֻּֽאֵיכֶ֑ם כִּ֚י שִׁבְעַ֣ת יָמִ֔ים יְמַלֵּ֖א אֶת-יֶדְכֶֽם:  מ  וּמִפֶּ֩תַח֩ אֹ֨הֶל מוֹעֵ֜ד לֹ֤א תֵֽצְאוּ֙ שִׁבְעַ֣ת יָמִ֔ים עַ֚ד י֣וֹם מְלֹ֔את יְמֵ֖י מִלֻּֽאֵיכֶ֑ם כִּ֚י שִׁבְעַ֣ת יָמִ֔ים יְמַלֵּ֖א אֶת-יֶדְכֶֽם:  ת  וּמִתְּרַע מַשְׁכַּן זִמְנָא לָא תִפְּקוּן שִׁבְעָא יוֹמִין עַד יוֹם מִשְׁלַם יוֹמֵי קוּרְבָּנֵיכוֹן אֲרֵי שִׁבְעָא יוֹמִין יְקָרֵב יָת קֻרְבָּנֵיכוֹן:

{לד} כַּֽאֲשֶׁ֥ר עָשָׂ֖ה בַּיּ֣וֹם הַזֶּ֑ה צִוָּ֧ה יְהוָֹ֛ה לַֽעֲשֹׂ֖ת לְכַפֵּ֥ר עֲלֵיכֶֽם:  מ  כַּֽאֲשֶׁ֥ר עָשָׂ֖ה בַּיּ֣וֹם הַזֶּ֑ה צִוָּ֧ה יְהוָֹ֛ה לַֽעֲשֹׂ֖ת לְכַפֵּ֥ר עֲלֵיכֶֽם:  ת  כְּמָא דִי עֲבַד בְּיוֹמָא הָדֵין פָּקִיד יְיָ לְמֶעְבַּד לְכַפָּרָא עֲלֵיכוֹן:

{לה} וּפֶ֩תַח֩ אֹ֨הֶל מוֹעֵ֜ד תֵּֽשְׁב֨וּ יוֹמָ֤ם וָלַ֨יְלָה֙ שִׁבְעַ֣ת יָמִ֔ים וּשְׁמַרְתֶּ֛ם אֶת-מִשְׁמֶ֥רֶת יְהוָֹ֖ה וְלֹ֣א תָמ֑וּתוּ כִּי-כֵ֖ן צֻוֵּֽיתִי:  מ  וּפֶ֩תַח֩ אֹ֨הֶל מוֹעֵ֜ד תֵּֽשְׁב֨וּ יוֹמָ֤ם וָלַ֨יְלָה֙ שִׁבְעַ֣ת יָמִ֔ים וּשְׁמַרְתֶּ֛ם אֶת-מִשְׁמֶ֥רֶת יְהוָֹ֖ה וְלֹ֣א תָמ֑וּתוּ כִּי-כֵ֖ן צֻוֵּֽיתִי:  ת  וּבִתְרַע מַשְׁכַּן זִמְנָא תֵּיתְבוּן יְמָם וְלֵילֵי שִׁבְעָא יוֹמִין וְתִטְרוּן יָת מַטְרַת מֵימְרָא דַיְיָ וְלָא תְמוּתוּן אֲרֵי כֵן אִתְפַּקָדִית:

{לו} וַיַּ֥עַשׂ אַֽהֲרֹ֖ן וּבָנָ֑יו אֵ֚ת כָּל-הַדְּבָרִ֔ים אֲשֶׁר-צִוָּ֥ה יְהוָֹ֖ה בְּיַד-מֹשֶֽׁה:  מ  וַיַּ֥עַשׂ אַֽהֲרֹ֖ן וּבָנָ֑יו אֵ֚ת כָּל-הַדְּבָרִ֔ים אֲשֶׁר-צִוָּ֥ה יְהוָֹ֖ה בְּיַד-מֹשֶֽׁה:  ת  וַעֲבַד אַהֲרֹן וּבְנוֹהִי יָת כָּל פִּתְגָמַיָא דִי פַקִיד יְיָ בִּידָא דְמֹשֶׁה: (ססס)
