פרשת פקודי


{כא} אֵ֣לֶּה פְקוּדֵ֤י הַמִּשְׁכָּן֙ מִשְׁכַּ֣ן הָֽעֵדֻ֔ת אֲשֶׁ֥ר פֻּקַּ֖ד עַל-פִּ֣י מֹשֶׁ֑ה עֲבֹדַת֙ הַֽלְוִיִּ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן-אַֽהֲרֹ֖ן הַכֹּהֵֽן:  מ  אֵ֣לֶּה פְקוּדֵ֤י הַמִּשְׁכָּן֙ מִשְׁכַּ֣ן הָֽעֵדֻ֔ת אֲשֶׁ֥ר פֻּקַּ֖ד עַל-פִּ֣י מֹשֶׁ֑ה עֲבֹדַת֙ הַֽלְוִיִּ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן-אַֽהֲרֹ֖ן הַכֹּהֵֽן:  ת  אִלֵין מִנְיָנֵי מַשְׁכְּנָא מַשְׁכְּנָא דְסַהֲדוּתָא דִי אִתְמְנֵי עַל מֵימְרָא דְמשֶׁה פָּלְחַן לֵיוָאֵי בִּידָא דְאִיתָמָר בַּר אַהֲרֹן כַּהֲנָא:

{כב} וּבְצַלְאֵ֛ל בֶּן-אוּרִ֥י בֶן-ח֖וּר לְמַטֵּ֣ה יְהוּדָ֑ה עָשָׂ֕ה אֵ֛ת כָּל-אֲשֶׁר-צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וּבְצַלְאֵ֛ל בֶּן-אוּרִ֥י בֶן-ח֖וּר לְמַטֵּ֣ה יְהוּדָ֑ה עָשָׂ֕ה אֵ֛ת כָּל-אֲשֶׁר-צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וּבְצַלְאֵל בַּר אוּרִי בַר חוּר לְשִׁבְטָא דִיהוּדָה עֲבַד יָת כָּל דִי פַקֵיד יְיָ יָת משֶׁה:

{כג} וְאִתּ֗וֹ אָֽהֳלִיאָ֞ב בֶּן-אֲחִֽיסָמָ֛ךְ לְמַטֵּה-דָ֖ן חָרָ֣שׁ וְחֹשֵׁ֑ב וְרֹקֵ֗ם בַּתְּכֵ֨לֶת֙ וּבָ֣אַרְגָּמָ֔ן וּבְתוֹלַ֥עַת הַשָּׁנִ֖י וּבַשֵּֽׁשׁ:  מ  וְאִתּ֗וֹ אָֽהֳלִיאָ֞ב בֶּן-אֲחִֽיסָמָ֛ךְ לְמַטֵּה-דָ֖ן חָרָ֣שׁ וְחֹשֵׁ֑ב וְרֹקֵ֗ם בַּתְּכֵ֨לֶת֙ וּבָ֣אַרְגָּמָ֔ן וּבְתוֹלַ֥עַת הַשָּׁנִ֖י וּבַשֵּֽׁשׁ:  ת  וְעִמֵהּ אָהֳלִיאָב בַּר אֲחִיסָמָךְ לְשִׁבְטָא דְדָן נַגָר וָאֳמָן וְצַיָר בְּתִכְלָא וּבְאַרְגְוָנָא וּבִצְבַע זְהוֹרִי וּבְבוּצָא: (ס)

{כד} כָּל-הַזָּהָ֗ב הֶֽעָשׂוּי֙ לַמְּלָאכָ֔ה בְּכֹ֖ל מְלֶ֣אכֶת הַקֹּ֑דֶשׁ וַיְהִ֣י | זְהַ֣ב הַתְּנוּפָ֗ה תֵּ֤שַׁע וְעֶשְׂרִים֙ כִּכָּ֔ר וּשְׁבַ֨ע מֵא֧וֹת וּשְׁלֹשִׁ֛ים שֶׁ֖קֶל בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ:  מ  כָּל-הַזָּהָ֗ב הֶֽעָשׂוּי֙ לַמְּלָאכָ֔ה בְּכֹ֖ל מְלֶ֣אכֶת הַקֹּ֑דֶשׁ וַיְהִ֣י | זְהַ֣ב הַתְּנוּפָ֗ה תֵּ֤שַׁע וְעֶשְׂרִים֙ כִּכָּ֔ר וּשְׁבַ֨ע מֵא֧וֹת וּשְׁלֹשִׁ֛ים שֶׁ֖קֶל בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ:  ת  כָּל דַהֲבָא דְאִתְעֲבֵד לְעִבִידָא בְּכֹל עִבִידַת קוּדְשָׁא וַהֲוָה דְהַב אֲרָמוּתָא עֶסְרִין וּתְשַׁע כִּכְּרִין וּשְׁבַע מְאָה וּתְלָתִין סִלְעִין בְּסִלְעֵי קוּדְשָׁא:

{כה} וְכֶ֛סֶף פְּקוּדֵ֥י הָֽעֵדָ֖ה מְאַ֣ת כִּכָּ֑ר וְאֶ֩לֶף֩ וּשְׁבַ֨ע מֵא֜וֹת וַֽחֲמִשָּׁ֧ה וְשִׁבְעִ֛ים שֶׁ֖קֶל בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ:  מ  וְכֶ֛סֶף פְּקוּדֵ֥י הָֽעֵדָ֖ה מְאַ֣ת כִּכָּ֑ר וְאֶ֩לֶף֩ וּשְׁבַ֨ע מֵא֜וֹת וַֽחֲמִשָּׁ֧ה וְשִׁבְעִ֛ים שֶׁ֖קֶל בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ:  ת  וּכְסַף מִנְיָנֵי כְנִשְׁתָּא מְאָה כִכְּרִין וְאֶלֶף וּשְׁבַע מְאָה וְשִׁבְעִין וְחַמְשָׁא סִלְעִין בְּסִלְעֵי קוּדְשָׁא:

{כו} בֶּ֚קַע לַגֻּלְגֹּ֔לֶת מַֽחֲצִ֥ית הַשֶּׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ לְכֹ֨ל הָֽעֹבֵ֜ר עַל-הַפְּקֻדִ֗ים מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה לְשֵׁשׁ-מֵא֥וֹת אֶ֨לֶף֙ וּשְׁלֹ֣שֶׁת אֲלָפִ֔ים וַֽחֲמֵ֥שׁ מֵא֖וֹת וַֽחֲמִשִּֽׁים:  מ  בֶּ֚קַע לַגֻּלְגֹּ֔לֶת מַֽחֲצִ֥ית הַשֶּׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ לְכֹ֨ל הָֽעֹבֵ֜ר עַל-הַפְּקֻדִ֗ים מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה לְשֵׁשׁ-מֵא֥וֹת אֶ֨לֶף֙ וּשְׁלֹ֣שֶׁת אֲלָפִ֔ים וַֽחֲמֵ֥שׁ מֵא֖וֹת וַֽחֲמִשִּֽׁים:  ת  תִּקְלָּא לְגֻלְגַלְתָּא פַלְגוּת סִלְעָא בְּסִלְעֵי קוּדְשָׁא לְכֹל דְעָבַר עַל מִנְיָנַיָא מִבַּר עֶסְרִין שְׁנִין וּלְעֵלָא לְשִׁית מְאָה וּתְלָתָא אַלְפִין וַחֲמֵשׁ מְאָה וְחַמְשִׁין:

{כז} וַיְהִ֗י מְאַת֙ כִּכַּ֣ר הַכֶּ֔סֶף לָצֶ֗קֶת אֵ֚ת אַדְנֵ֣י הַקֹּ֔דֶשׁ וְאֵ֖ת אַדְנֵ֣י הַפָּרֹ֑כֶת מְאַ֧ת אֲדָנִ֛ים לִמְאַ֥ת הַכִּכָּ֖ר כִּכָּ֥ר לָאָֽדֶן:  מ  וַיְהִ֗י מְאַת֙ כִּכַּ֣ר הַכֶּ֔סֶף לָצֶ֗קֶת אֵ֚ת אַדְנֵ֣י הַקֹּ֔דֶשׁ וְאֵ֖ת אַדְנֵ֣י הַפָּרֹ֑כֶת מְאַ֧ת אֲדָנִ֛ים לִמְאַ֥ת הַכִּכָּ֖ר כִּכָּ֥ר לָאָֽדֶן:  ת  וַהֲוָה מְאָה כִכְּרִין דִכְסַף לְאַתָּכָא יָת סַמְכֵי קוּדְשָׁא וְיָת סַמְכֵי דְפָרֻכְתָּא מְאָה סַמְכִין לִמְאָה כִכְּרִין כִּכְּרָא לְסַמְכָא:

{כח} וְאֶת-הָאֶ֜לֶף וּשְׁבַ֤ע הַמֵּאוֹת֙ וַֽחֲמִשָּׁ֣ה וְשִׁבְעִ֔ים עָשָׂ֥ה וָוִ֖ים לָֽעַמּוּדִ֑ים וְצִפָּ֥ה רָֽאשֵׁיהֶ֖ם וְחִשַּׁ֥ק אֹתָֽם:  מ  וְאֶת-הָאֶ֜לֶף וּשְׁבַ֤ע הַמֵּאוֹת֙ וַֽחֲמִשָּׁ֣ה וְשִׁבְעִ֔ים עָשָׂ֥ה וָוִ֖ים לָֽעַמּוּדִ֑ים וְצִפָּ֥ה רָֽאשֵׁיהֶ֖ם וְחִשַּׁ֥ק אֹתָֽם:  ת  וְיָת אַלְפָא וּשְׁבַע מְאָה וְשִׁבְעִין וְחַמְשָׁא עֲבַד וָוִין לְעַמוּדַיָא וְחִפָּא רֵישֵׁיהוֹן וְכַבֵּשׁ יָתְהוֹן:

{כט} וּנְחֹ֥שֶׁת הַתְּנוּפָ֖ה שִׁבְעִ֣ים כִּכָּ֑ר וְאַלְפַּ֥יִם וְאַרְבַּע-מֵא֖וֹת שָֽׁקֶל:  מ  וּנְחֹ֥שֶׁת הַתְּנוּפָ֖ה שִׁבְעִ֣ים כִּכָּ֑ר וְאַלְפַּ֥יִם וְאַרְבַּע-מֵא֖וֹת שָֽׁקֶל:  ת  וּנְחָשָׁא דַאֲרָמוּתָא שִׁבְעִין כִּכְּרִין וּתְרֵין אַלְפִין וְאַרְבַּע מְאָה סִלְעִין:

{ל} וַיַּ֣עַשׂ בָּ֗הּ אֶת-אַדְנֵי֙ פֶּ֚תַח אֹ֣הֶל מוֹעֵ֔ד וְאֵת֙ מִזְבַּ֣ח הַנְּחֹ֔שֶׁת וְאֶת-מִכְבַּ֥ר הַנְּחֹ֖שֶׁת אֲשֶׁר-ל֑וֹ וְאֵ֖ת כָּל-כְּלֵ֥י הַמִּזְבֵּֽחַ:  מ  וַיַּ֣עַשׂ בָּ֗הּ אֶת-אַדְנֵי֙ פֶּ֚תַח אֹ֣הֶל מוֹעֵ֔ד וְאֵת֙ מִזְבַּ֣ח הַנְּחֹ֔שֶׁת וְאֶת-מִכְבַּ֥ר הַנְּחֹ֖שֶׁת אֲשֶׁר-ל֑וֹ וְאֵ֖ת כָּל-כְּלֵ֥י הַמִּזְבֵּֽחַ:  ת  וַעֲבַד בַּהּ יָת סַמְכֵי תְּרַע מַשְׁכַּן זִמְנָא וְיָת מַדְבְּחָא דִנְחָשָׁא וְיָת סְרָדָא דִנְחָשָׁא דִי לֵהּ וְיָת כָּל מָנֵי מַדְבְּחָא:

{לא} וְאֶת-אַדְנֵ֤י הֶֽחָצֵר֙ סָבִ֔יב וְאֶת-אַדְנֵ֖י שַׁ֣עַר הֶֽחָצֵ֑ר וְאֵ֨ת כָּל-יִתְדֹ֧ת הַמִּשְׁכָּ֛ן וְאֶת-כָּל-יִתְדֹ֥ת הֶֽחָצֵ֖ר סָבִֽיב:  מ  וְאֶת-אַדְנֵ֤י הֶֽחָצֵר֙ סָבִ֔יב וְאֶת-אַדְנֵ֖י שַׁ֣עַר הֶֽחָצֵ֑ר וְאֵ֨ת כָּל-יִתְדֹ֧ת הַמִּשְׁכָּ֛ן וְאֶת-כָּל-יִתְדֹ֥ת הֶֽחָצֵ֖ר סָבִֽיב:  ת  וְיָת סַמְכֵי דְדַרְתָּא סְחוֹר סְחוֹר וְיָת סַמְכֵי תְּרַע דַרְתָּא וְיָת כָּל סִכֵּי מַשְׁכְּנָא וְיָת כָּל סִכֵּי דַרְתָּא סְחוֹר סְחוֹר:




שמות פרק-לט

{א} וּמִן-הַתְּכֵ֤לֶת וְהָֽאַרְגָּמָן֙ וְתוֹלַ֣עַת הַשָּׁנִ֔י עָשׂ֥וּ בִגְדֵֽי-שְׂרָ֖ד לְשָׁרֵ֣ת בַּקֹּ֑דֶשׁ וַֽיַּֽעֲשׂ֞וּ אֶת-בִּגְדֵ֤י הַקֹּ֨דֶשׁ֙ אֲשֶׁ֣ר לְאַֽהֲרֹ֔ן כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וּמִן-הַתְּכֵ֤לֶת וְהָֽאַרְגָּמָן֙ וְתוֹלַ֣עַת הַשָּׁנִ֔י עָשׂ֥וּ בִגְדֵֽי-שְׂרָ֖ד לְשָׁרֵ֣ת בַּקֹּ֑דֶשׁ וַֽיַּֽעֲשׂ֞וּ אֶת-בִּגְדֵ֤י הַקֹּ֨דֶשׁ֙ אֲשֶׁ֣ר לְאַֽהֲרֹ֔ן כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וּמִן תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי עֲבָדוּ לְבוּשֵׁי שִׁמוּשָׁא לְשַׁמָשָׁא בְּקוּדְשָׁא וַעֲבָדוּ יָת לְבוּשֵׁי קוּדְשָׁא דִי לְאַהֲרֹן כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (פ)

{ב}  שני - חמישי במחוברין  וַיַּ֖עַשׂ אֶת-הָֽאֵפֹ֑ד זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מָשְׁזָֽר:  מ  וַיַּ֖עַשׂ אֶת-הָֽאֵפֹ֑ד זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מָשְׁזָֽר:  ת  וַעֲבַד יַת אֵפוֹדָא דַהֲבָא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר:

{ג} וַֽיְרַקְּע֞וּ אֶת-פַּחֵ֣י הַזָּהָב֘ וְקִצֵּ֣ץ פְּתִילִם֒ לַֽעֲשׂ֗וֹת בְּת֤וֹךְ הַתְּכֵ֨לֶת֙ וּבְת֣וֹךְ הָֽאַרְגָּמָ֔ן וּבְת֛וֹךְ תּוֹלַ֥עַת הַשָּׁנִ֖י וּבְת֣וֹךְ הַשֵּׁ֑שׁ מַֽעֲשֵׂ֖ה חֹשֵֽׁב:  מ  וַֽיְרַקְּע֞וּ אֶת-פַּחֵ֣י הַזָּהָב֘ וְקִצֵּ֣ץ פְּתִילִם֒ לַֽעֲשׂ֗וֹת בְּת֤וֹךְ הַתְּכֵ֨לֶת֙ וּבְת֣וֹךְ הָֽאַרְגָּמָ֔ן וּבְת֛וֹךְ תּוֹלַ֥עַת הַשָּׁנִ֖י וּבְת֣וֹךְ הַשֵּׁ֑שׁ מַֽעֲשֵׂ֖ה חֹשֵֽׁב:  ת  וְרַדִידוּ יָת טַסֵי דַהֲבָא וְקַצִיצוּ חוּטִין לְמֶעְבַּד בְּגוֹ תִּכְלָא וּבְגוֹ אַרְגְוָנָא וּבְגוֹ צְבַע זְהוֹרִי וּבְגוֹ בוּצָא עוֹבַד אֳמָן:

{ד} כְּתֵפֹ֥ת עָֽשׂוּ-ל֖וֹ חֹֽבְרֹ֑ת עַל-שְׁנֵ֥י (קצוותו) קְצוֹתָ֖יו חֻבָּֽר:  מ  כְּתֵפֹ֥ת עָֽשׂוּ-ל֖וֹ חֹֽבְרֹ֑ת עַל-שְׁנֵ֥י (קצוותו) קְצוֹתָ֖יו חֻבָּֽר:  ת  כַּתְפָּן עֲבָדוּ לֵהּ מְלַפְּפָן עַל תְּרֵין סִטְרוֹהִי יִתְלָפֵף:

{ה} וְחֵ֨שֶׁב אֲפֻדָּת֜וֹ אֲשֶׁ֣ר עָלָ֗יו מִמֶּ֣נּוּ הוּא֘ כְּמַֽעֲשֵׂ֒הוּ֒ זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מָשְׁזָ֑ר כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וְחֵ֨שֶׁב אֲפֻדָּת֜וֹ אֲשֶׁ֣ר עָלָ֗יו מִמֶּ֣נּוּ הוּא֘ כְּמַֽעֲשֵׂ֒הוּ֒ זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מָשְׁזָ֑ר כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְהֶמְיַן תִּקוּנֵהּ דִי עֲלוֹהִי מִנֵהּ הוּא כְּעוֹבָדוֹהִי דַהֲבָא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{ו} וַֽיַּֽעֲשׂוּ֙ אֶת-אַבְנֵ֣י הַשֹּׁ֔הַם מֻֽסַבֹּ֖ת מִשְׁבְּצֹ֣ת זָהָ֑ב מְפֻתָּחֹת֙ פִּתּוּחֵ֣י חוֹתָ֔ם עַל-שְׁמ֖וֹת בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וַֽיַּֽעֲשׂוּ֙ אֶת-אַבְנֵ֣י הַשֹּׁ֔הַם מֻֽסַבֹּ֖ת מִשְׁבְּצֹ֣ת זָהָ֑ב מְפֻתָּחֹת֙ פִּתּוּחֵ֣י חוֹתָ֔ם עַל-שְׁמ֖וֹת בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וַעֲבָדוּ יָת אַבְנֵי בוּרְלָא מְשַׁקְעָן מְרַמְצָן דִדְהָב גְלִיפָן כְּתַב מְפָרַשׁ עַל שְׁמָהַת בְּנֵי יִשְׂרָאֵל:

{ז} וַיָּ֣שֶׂם אֹתָ֗ם עַ֚ל כִּתְפֹ֣ת הָֽאֵפֹ֔ד אַבְנֵ֥י זִכָּר֖וֹן לִבְנֵ֣י יִשְׂרָאֵ֑ל כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיָּ֣שֶׂם אֹתָ֗ם עַ֚ל כִּתְפֹ֣ת הָֽאֵפֹ֔ד אַבְנֵ֥י זִכָּר֖וֹן לִבְנֵ֣י יִשְׂרָאֵ֑ל כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְשַׁוִי יָתְהוֹן עַל כִּתְפָא דְאֵפוֹדָא אַבְנֵי דוּכְרָנָא לִבְנֵי יִשְׂרָאֵל כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (פ)

{ח} וַיַּ֧עַשׂ אֶת-הַחֹ֛שֶׁן מַֽעֲשֵׂ֥ה חֹשֵׁ֖ב כְּמַֽעֲשֵׂ֣ה אֵפֹ֑ד זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מָשְׁזָֽר:  מ  וַיַּ֧עַשׂ אֶת-הַחֹ֛שֶׁן מַֽעֲשֵׂ֥ה חֹשֵׁ֖ב כְּמַֽעֲשֵׂ֣ה אֵפֹ֑ד זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מָשְׁזָֽר:  ת  וַעֲבַד יָת חוּשְׁנָא עוֹבַד אֳמָן כְּעוֹבְדֵי אֵפוֹדָא דַהֲבָּא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר:

{ט} רָב֧וּעַ הָיָ֛ה כָּפ֖וּל עָשׂ֣וּ אֶת-הַחֹ֑שֶׁן זֶ֧רֶת אָרְכּ֛וֹ וְזֶ֥רֶת רָחְבּ֖וֹ כָּפֽוּל:  מ  רָב֧וּעַ הָיָ֛ה כָּפ֖וּל עָשׂ֣וּ אֶת-הַחֹ֑שֶׁן זֶ֧רֶת אָרְכּ֛וֹ וְזֶ֥רֶת רָחְבּ֖וֹ כָּפֽוּל:  ת  מְרַבַּע הֲוָה עֵיף עֲבָדוּ יָת חוּשְׁנָא זֶרְתָּא אֻרְכֵּהּ וְזֶרְתָּא פְתָיֵהּ עֵיף:

{י} וַיְמַ֨לְאוּ-ב֔וֹ אַרְבָּעָ֖ה ט֣וּרֵי אָ֑בֶן ט֗וּר אֹ֤דֶם פִּטְדָה֙ וּבָרֶ֔קֶת הַטּ֖וּר הָֽאֶחָֽד:  מ  וַיְמַ֨לְאוּ-ב֔וֹ אַרְבָּעָ֖ה ט֣וּרֵי אָ֑בֶן ט֗וּר אֹ֤דֶם פִּטְדָה֙ וּבָרֶ֔קֶת הַטּ֖וּר הָֽאֶחָֽד:  ת  וְאַשְׁלִימוּ בֵהּ אַרְבְּעָא סִדְרֵי אֶבֶן טָבָא סִדְרָא קַדְמָאָה סַמְקָן יַרְקָן וּבַרְקָן סִדְרָא חָד:

{יא} וְהַטּ֖וּר הַשֵּׁנִ֑י נֹ֥פֶךְ סַפִּ֖יר וְיָֽהֲלֹֽם:  מ  וְהַטּ֖וּר הַשֵּׁנִ֑י נֹ֥פֶךְ סַפִּ֖יר וְיָֽהֲלֹֽם:  ת  וְסִדְרָא תִּנְיָנָא אִזְמָרַגְדִין שַׁבְזִיז וְסַבְהֲלוֹם:

{יב} וְהַטּ֖וּר הַשְּׁלִישִׁ֑י לֶ֥שֶׁם שְׁב֖וֹ וְאַחְלָֽמָה:  מ  וְהַטּ֖וּר הַשְּׁלִישִׁ֑י לֶ֥שֶׁם שְׁב֖וֹ וְאַחְלָֽמָה:  ת  וְסִדְרָא תְּלִיתָאָה קַנְכֵּרֵי טַרְקְיָא וְעֵין עֶגְלָא:

{יג} וְהַטּוּר֙ הָֽרְבִיעִ֔י תַּרְשִׁ֥ישׁ שֹׁ֖הַם וְיָֽשְׁפֵ֑ה מֽוּסַבֹּ֛ת מִשְׁבְּצֹ֥ת זָהָ֖ב בְּמִלֻּֽאֹתָֽם:  מ  וְהַטּוּר֙ הָֽרְבִיעִ֔י תַּרְשִׁ֥ישׁ שֹׁ֖הַם וְיָֽשְׁפֵ֑ה מֽוּסַבֹּ֛ת מִשְׁבְּצֹ֥ת זָהָ֖ב בְּמִלֻּֽאֹתָֽם:  ת  וְסִדְרָא רְבִיעָאָה כְּרוּם יַמָּא וּבוּרְלָא וּפַנְטֵּרֵי מְשַׁקְעָן מְרַמְצָן דִדְהַב בְּאַשְׁלָמוּתְהוֹן:

{יד} וְ֠הָֽאֲבָנִ֠ים עַל-שְׁמֹ֨ת בְּנֵֽי-יִשְׂרָאֵ֥ל הֵ֛נָּה שְׁתֵּ֥ים עֶשְׂרֵ֖ה עַל-שְׁמֹתָ֑ם פִּתּוּחֵ֤י חֹתָם֙ אִ֣ישׁ עַל-שְׁמ֔וֹ לִשְׁנֵ֥ים עָשָׂ֖ר שָֽׁבֶט:  מ  וְ֠הָֽאֲבָנִ֠ים עַל-שְׁמֹ֨ת בְּנֵֽי-יִשְׂרָאֵ֥ל הֵ֛נָּה שְׁתֵּ֥ים עֶשְׂרֵ֖ה עַל-שְׁמֹתָ֑ם פִּתּוּחֵ֤י חֹתָם֙ אִ֣ישׁ עַל-שְׁמ֔וֹ לִשְׁנֵ֥ים עָשָׂ֖ר שָֽׁבֶט:  ת  וְאַבְנַיָא עַל שְׁמָהַת בְּנֵי יִשְׂרָאֵל אִנוּן תַּרְתֵּי עֶסְרֵי עַל שְׁמָהַתְהוֹן כְּתַב מְפָרַשׁ כִּגְלוֹף דְעִזְקָא גְבַר עַל שְׁמֵהּ לִתְרֵין עֲסַר שִׁבְטִין:

{טו} וַיַּֽעֲשׂ֧וּ עַל-הַחֹ֛שֶׁן שַׁרְשְׁרֹ֥ת גַּבְלֻ֖ת מַֽעֲשֵׂ֣ה עֲבֹ֑ת זָהָ֖ב טָהֽוֹר:  מ  וַיַּֽעֲשׂ֧וּ עַל-הַחֹ֛שֶׁן שַׁרְשְׁרֹ֥ת גַּבְלֻ֖ת מַֽעֲשֵׂ֣ה עֲבֹ֑ת זָהָ֖ב טָהֽוֹר:  ת  וַעֲבָדוּ עַל חוּשְׁנָא תִּכִּין מְתַחְמָן עוֹבַד גְדִילוּ דְהַב דְכֵי:

{טז} וַֽיַּֽעֲשׂ֗וּ שְׁתֵּי֙ מִשְׁבְּצֹ֣ת זָהָ֔ב וּשְׁתֵּ֖י טַבְּעֹ֣ת זָהָ֑ב וַֽיִּתְּנ֗וּ אֶת-שְׁתֵּי֙ הַטַּבָּעֹ֔ת עַל-שְׁנֵ֖י קְצ֥וֹת הַחֹֽשֶׁן:  מ  וַֽיַּֽעֲשׂ֗וּ שְׁתֵּי֙ מִשְׁבְּצֹ֣ת זָהָ֔ב וּשְׁתֵּ֖י טַבְּעֹ֣ת זָהָ֑ב וַֽיִּתְּנ֗וּ אֶת-שְׁתֵּי֙ הַטַּבָּעֹ֔ת עַל-שְׁנֵ֖י קְצ֥וֹת הַחֹֽשֶׁן:  ת  וַעֲבָדוּ תַּרְתֵּין מְרַמְצָן דִדְהַב וְתַרְתֵּין עִזְקָן דִדְהָב וִיהָבוּ יָת תַּרְתֵּין עִזְקָתָא עַל תְּרֵין סִטְרֵי חוּשְׁנָא:

{יז} וַֽיִּתְּנ֗וּ שְׁתֵּי֙ הָֽעֲבֹתֹ֣ת הַזָּהָ֔ב עַל-שְׁתֵּ֖י הַטַּבָּעֹ֑ת עַל-קְצ֖וֹת הַחֹֽשֶׁן:  מ  וַֽיִּתְּנ֗וּ שְׁתֵּי֙ הָֽעֲבֹתֹ֣ת הַזָּהָ֔ב עַל-שְׁתֵּ֖י הַטַּבָּעֹ֑ת עַל-קְצ֖וֹת הַחֹֽשֶׁן:  ת  וִיהָבוּ תַּרְתֵּין גְדִילָתָא דִדְהַב עַל תַּרְתֵּין עִזְקָתָא עַל סִטְרֵי חוּשְׁנָא:

{יח} וְאֵ֨ת שְׁתֵּ֤י קְצוֹת֙ שְׁתֵּ֣י הָֽעֲבֹתֹ֔ת נָתְנ֖וּ עַל-שְׁתֵּ֣י הַֽמִּשְׁבְּצֹ֑ת וַֽיִּתְּנֻ֛ם עַל-כִּתְפֹ֥ת הָֽאֵפֹ֖ד אֶל-מ֥וּל פָּנָֽיו:  מ  וְאֵ֨ת שְׁתֵּ֤י קְצוֹת֙ שְׁתֵּ֣י הָֽעֲבֹתֹ֔ת נָתְנ֖וּ עַל-שְׁתֵּ֣י הַֽמִּשְׁבְּצֹ֑ת וַֽיִּתְּנֻ֛ם עַל-כִּתְפֹ֥ת הָֽאֵפֹ֖ד אֶל-מ֥וּל פָּנָֽיו:  ת  וְיָת תַּרְתֵּין גְדִילָן דְעַל תַּרְתֵּין סִטְרוֹהִי יְהָבוּ עַל תַּרְתֵּין מְרַמְצָתָא וִיהָבֻנוּן עַל כַּתְפֵּי אֵפוֹדָא לָקֳבֵל אַפּוֹהִי:

{יט} וַֽיַּֽעֲשׂ֗וּ שְׁתֵּי֙ טַבְּעֹ֣ת זָהָ֔ב וַיָּשִׂ֕ימוּ עַל-שְׁנֵ֖י קְצ֣וֹת הַחֹ֑שֶׁן עַל-שְׂפָת֕וֹ אֲשֶׁ֛ר אֶל-עֵ֥בֶר הָֽאֵפֹ֖ד בָּֽיְתָה:  מ  וַֽיַּֽעֲשׂ֗וּ שְׁתֵּי֙ טַבְּעֹ֣ת זָהָ֔ב וַיָּשִׂ֕ימוּ עַל-שְׁנֵ֖י קְצ֣וֹת הַחֹ֑שֶׁן עַל-שְׂפָת֕וֹ אֲשֶׁ֛ר אֶל-עֵ֥בֶר הָֽאֵפֹ֖ד בָּֽיְתָה:  ת  וַעֲבָדוּ תַּרְתֵּין עִזְקַן דִדְהַב וְשַׁוִיאוּ עַל תְּרֵין סִטְרֵי חוּשְׁנָא עַל סִפְתֵהּ דִי לְעִבְרָא דְאֵפוֹדָא מִלְגָיו:

{כ} וַֽיַּֽעֲשׂוּ֘ שְׁתֵּ֣י טַבְּעֹ֣ת זָהָב֒ וַֽיִּתְּנֻ֡ם עַל-שְׁתֵּי֩ כִתְפֹ֨ת הָֽאֵפֹ֤ד מִלְּמַ֨טָּה֙ מִמּ֣וּל פָּנָ֔יו לְעֻמַּ֖ת מַחְבַּרְתּ֑וֹ מִמַּ֕עַל לְחֵ֖שֶׁב הָֽאֵפֹֽד:  מ  וַֽיַּֽעֲשׂוּ֘ שְׁתֵּ֣י טַבְּעֹ֣ת זָהָב֒ וַֽיִּתְּנֻ֡ם עַל-שְׁתֵּי֩ כִתְפֹ֨ת הָֽאֵפֹ֤ד מִלְּמַ֨טָּה֙ מִמּ֣וּל פָּנָ֔יו לְעֻמַּ֖ת מַחְבַּרְתּ֑וֹ מִמַּ֕עַל לְחֵ֖שֶׁב הָֽאֵפֹֽד:  ת  וַעֲבָדוּ תַּרְתֵּין עִזְקַן דִדְהַב וִיהָבֻנוּן עַל תַּרְתֵּין כִּתְפֵי אֵפוֹדָא מִלְרַע מִלָקֳבֵל אַפּוֹהִי לָקֳבֵל בֵּית לוֹפֵי מֵעִלָוֵי לְהֶמְיַן אֵפוֹדָא:

{כא} וַיִּרְכְּס֣וּ אֶת-הַח֡שֶׁן מִטַּבְּעֹתָיו֩ אֶל-טַבְּעֹ֨ת הָֽאֵפֹ֜ד בִּפְתִ֣יל תְּכֵ֗לֶת לִֽהְיֹת֙ עַל-חֵ֣שֶׁב הָֽאֵפֹ֔ד וְלֹֽא-יִזַּ֣ח הַח֔שֶׁן מֵעַ֖ל הָֽאֵפֹ֑ד כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיִּרְכְּס֣וּ אֶת-הַח֡שֶׁן מִטַּבְּעֹתָיו֩ אֶל-טַבְּעֹ֨ת הָֽאֵפֹ֜ד בִּפְתִ֣יל תְּכֵ֗לֶת לִֽהְיֹת֙ עַל-חֵ֣שֶׁב הָֽאֵפֹ֔ד וְלֹֽא-יִזַּ֣ח הַח֔שֶׁן מֵעַ֖ל הָֽאֵפֹ֑ד כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְיַחֲדוּן יָת חוּשְׁנָא מֵעִזְקְתֵהּ לְעִזְקָתָא דְאֵפוֹדָא בְּחוּטָּא דִתְכֶלְתָּא לְמֶהֱוֵי עַל הֶמְיַן אֵפוֹדָא וְלָא יִתְפָּרֵק חוּשְׁנָא מֵעִלָוֵי אֵפוֹדָא כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (פ)

{כב}  שלישי - ששי במחוברין  וַיַּ֛עַשׂ אֶת-מְעִ֥יל הָֽאֵפֹ֖ד מַֽעֲשֵׂ֣ה אֹרֵ֑ג כְּלִ֖יל תְּכֵֽלֶת:  מ  וַיַּ֛עַשׂ אֶת-מְעִ֥יל הָֽאֵפֹ֖ד מַֽעֲשֵׂ֣ה אֹרֵ֑ג כְּלִ֖יל תְּכֵֽלֶת:  ת  וַעֲבַד יָת מְעִילָא דְאֵפוֹדָא עוֹבַד מָחֵי גְמִיר תִּכְלָא:

{כג} וּפִֽי-הַמְּעִ֥יל בְּתוֹכ֖וֹ כְּפִ֣י תַחְרָ֑א שָׂפָ֥ה לְפִ֛יו סָבִ֖יב לֹ֥א יִקָּרֵֽעַ:  מ  וּפִֽי-הַמְּעִ֥יל בְּתוֹכ֖וֹ כְּפִ֣י תַחְרָ֑א שָׂפָ֥ה לְפִ֛יו סָבִ֖יב לֹ֥א יִקָּרֵֽעַ:  ת  וּפוּם מְעִילָא כָּפִיל לְגַוֵהּ כְּפוּם שִׁרְיָן תּוֹרָא מַקַף לְפוּמֵהּ סְחוֹר סְחוֹר לָא יִתְבְּזָע:

{כד} וַֽיַּֽעֲשׂוּ֙ עַל-שׁוּלֵ֣י הַמְּעִ֔יל רִמּוֹנֵ֕י תְּכֵ֥לֶת וְאַרְגָּמָ֖ן וְתוֹלַ֣עַת שָׁנִ֑י מָשְׁזָֽר:  מ  וַֽיַּֽעֲשׂוּ֙ עַל-שׁוּלֵ֣י הַמְּעִ֔יל רִמּוֹנֵ֕י תְּכֵ֥לֶת וְאַרְגָּמָ֖ן וְתוֹלַ֣עַת שָׁנִ֑י מָשְׁזָֽר:  ת  וַעֲבָדוּ עַל שִׁפּוֹלֵי מְעִילָא רִמוֹנֵי תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי שְׁזִיר:

{כה} וַֽיַּֽעֲשׂ֥וּ פַֽעֲמֹנֵ֖י זָהָ֣ב טָה֑וֹר וַיִּתְּנ֨וּ אֶת-הַפַּֽעֲמֹנִ֜ים בְּת֣וֹךְ הָֽרִמֹּנִ֗ים עַל-שׁוּלֵ֤י הַמְּעִיל֙ סָבִ֔יב בְּת֖וֹךְ הָֽרִמֹּנִֽים:  מ  וַֽיַּֽעֲשׂ֥וּ פַֽעֲמֹנֵ֖י זָהָ֣ב טָה֑וֹר וַיִּתְּנ֨וּ אֶת-הַפַּֽעֲמֹנִ֜ים בְּת֣וֹךְ הָֽרִמֹּנִ֗ים עַל-שׁוּלֵ֤י הַמְּעִיל֙ סָבִ֔יב בְּת֖וֹךְ הָֽרִמֹּנִֽים:  ת  וַעֲבָדוּ זַגַיָא דְהַב דְכֵי וִיהָבוּ יָת זַגַיָא בְּגוֹ רִמוֹנַיָא עַל שִׁפּוֹלֵי מְעִילָא סְחוֹר סְחוֹר בְּגוֹ רִמוֹנַיָא:

{כו} פַּֽעֲמֹ֤ן וְרִמֹּן֙ פַּֽעֲמֹ֣ן וְרִמֹּ֔ן עַל-שׁוּלֵ֥י הַמְּעִ֖יל סָבִ֑יב לְשָׁרֵ֕ת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  פַּֽעֲמֹ֤ן וְרִמֹּן֙ פַּֽעֲמֹ֣ן וְרִמֹּ֔ן עַל-שׁוּלֵ֥י הַמְּעִ֖יל סָבִ֑יב לְשָׁרֵ֕ת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  זַגָא וְרִמוֹנָא זַגָא וְרִמוֹנָא עַל שִׁפּוֹלֵי מְעִילָא סְחוֹר סְחוֹר לְשַׁמָשָׁא כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{כז} וַֽיַּֽעֲשׂ֛וּ אֶת-הַכָּתְנֹ֥ת שֵׁ֖שׁ מַֽעֲשֵׂ֣ה אֹרֵ֑ג לְאַֽהֲרֹ֖ן וּלְבָנָֽיו:  מ  וַֽיַּֽעֲשׂ֛וּ אֶת-הַכָּתְנֹ֥ת שֵׁ֖שׁ מַֽעֲשֵׂ֣ה אֹרֵ֑ג לְאַֽהֲרֹ֖ן וּלְבָנָֽיו:  ת  וַעֲבָדוּ יָת כִּתּוּנִין דְבוּצָא עוֹבַד מָחֵי לְאַהֲרֹן וִלִבְנוֹהִי:

{כח} וְאֵת֙ הַמִּצְנֶ֣פֶת שֵׁ֔שׁ וְאֶת-פַּֽאֲרֵ֥י הַמִּגְבָּעֹ֖ת שֵׁ֑שׁ וְאֶת-מִכְנְסֵ֥י הַבָּ֖ד שֵׁ֥שׁ מָשְׁזָֽר:  מ  וְאֵת֙ הַמִּצְנֶ֣פֶת שֵׁ֔שׁ וְאֶת-פַּֽאֲרֵ֥י הַמִּגְבָּעֹ֖ת שֵׁ֑שׁ וְאֶת-מִכְנְסֵ֥י הַבָּ֖ד שֵׁ֥שׁ מָשְׁזָֽר:  ת  וְיָת מִצְנֶפְתָּא דְבוּצָא וְיָת שְׁבַח כּוֹבָעַיָא דְבוּצָא וְיָת מִכְנְסֵי בוּצָא דְבוּץ שְׁזִיר:

{כט} וְֽאֶת-הָֽאַבְנֵ֞ט שֵׁ֣שׁ מָשְׁזָ֗ר וּתְכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י מַֽעֲשֵׂ֣ה רֹקֵ֑ם כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וְֽאֶת-הָֽאַבְנֵ֞ט שֵׁ֣שׁ מָשְׁזָ֗ר וּתְכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י מַֽעֲשֵׂ֣ה רֹקֵ֑ם כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְיָת הֶמְיָנָא דְבוּץ שְּׁזִיר וְתִכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי עוֹבַד צַיָר כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{ל} וַֽיַּֽעֲשׂ֛וּ אֶת-צִ֥יץ נֶֽזֶר-הַקֹּ֖דֶשׁ זָהָ֣ב טָה֑וֹר וַיִּכְתְּב֣וּ עָלָ֗יו מִכְתַּב֙ פִּתּוּחֵ֣י חוֹתָ֔ם קֹ֖דֶשׁ לַֽיהוָֹֽה:  מ  וַֽיַּֽעֲשׂ֛וּ אֶת-צִ֥יץ נֶֽזֶר-הַקֹּ֖דֶשׁ זָהָ֣ב טָה֑וֹר וַיִּכְתְּב֣וּ עָלָ֗יו מִכְתַּב֙ פִּתּוּחֵ֣י חוֹתָ֔ם קֹ֖דֶשׁ לַֽיהוָֹֽה:  ת  וַעֲבָדוּ יָת צִיצָא כְלִילָא דְקוּדְשָׁא דְהַב דְכֵי וּכְתָבוּ עֲלוֹהִי גְלִיפָן כְּתַב מְפָרַשׁ קֹדֶשׁ לַיְיָ:

{לא} וַיִּתְּנ֤וּ עָלָיו֙ פְּתִ֣יל תְּכֵ֔לֶת לָתֵ֥ת עַל-הַמִּצְנֶ֖פֶת מִלְמָ֑עְלָה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיִּתְּנ֤וּ עָלָיו֙ פְּתִ֣יל תְּכֵ֔לֶת לָתֵ֥ת עַל-הַמִּצְנֶ֖פֶת מִלְמָ֑עְלָה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וִיהָבוּ עֲלוֹהִי חוּטָא תְכֶלְתָּא לְמִתַּן עַל מִצְנֶפְתָּא מִלְעֵלָא כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{לב} וַתֵּ֕כֶל כָּל-עֲבֹדַ֕ת מִשְׁכַּ֖ן אֹ֣הֶל מוֹעֵ֑ד וַֽיַּֽעֲשׂוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כְּ֠כֹ֠ל אֲשֶׁ֨ר צִוָּ֧ה יְהוָֹ֛ה אֶת-מֹשֶׁ֖ה כֵּ֥ן עָשֽׂוּ:  מ  וַתֵּ֕כֶל כָּל-עֲבֹדַ֕ת מִשְׁכַּ֖ן אֹ֣הֶל מוֹעֵ֑ד וַֽיַּֽעֲשׂוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כְּ֠כֹ֠ל אֲשֶׁ֨ר צִוָּ֧ה יְהוָֹ֛ה אֶת-מֹשֶׁ֖ה כֵּ֥ן עָשֽׂוּ:  ת  וּשְׁלִים כָּל פָּלְחַן מַשְׁכְּנָא מַשְׁכַּן זִמְנָא וַעֲבָדוּ בְּנֵי יִשְׂרָאֵל כְּכֹל דִי פַקִיד יְיָ יָת משֶׁה כֵּן עֲבָדוּ:

{לג}  רביעי  וַיָּבִ֤יאוּ אֶת-הַמִּשְׁכָּן֙ אֶל-מֹשֶׁ֔ה אֶת-הָאֹ֖הֶל וְאֶת-כָּל-כֵּלָ֑יו קְרָסָ֣יו קְרָשָׁ֔יו (בריחו) בְּרִיחָ֖יו וְעַמֻּדָ֥יו וַֽאֲדָנָֽיו:  מ  וַיָּבִ֤יאוּ אֶת-הַמִּשְׁכָּן֙ אֶל-מֹשֶׁ֔ה אֶת-הָאֹ֖הֶל וְאֶת-כָּל-כֵּלָ֑יו קְרָסָ֣יו קְרָשָׁ֔יו (בריחו) בְּרִיחָ֖יו וְעַמֻּדָ֥יו וַֽאֲדָנָֽיו:  ת  וְאַיְתִיוּ יָת מַשְׁכְּנָא לְמשֶׁה יָת מַשְׁכְּנָא וְיָת כָּל מָנוֹהִי פוּרְפוֹהִי דַפּוֹהִי עַבְּרוֹהִי וְעַמוּדוֹהִי וְסַמְכוֹהִי:

{לד} וְאֶת-מִכְסֵ֞ה עוֹרֹ֤ת הָֽאֵילִם֙ הַֽמְאָדָּמִ֔ים וְאֶת-מִכְסֵ֖ה עֹרֹ֣ת הַתְּחָשִׁ֑ים וְאֵ֖ת פָּרֹ֥כֶת הַמָּסָֽךְ:  מ  וְאֶת-מִכְסֵ֞ה עוֹרֹ֤ת הָֽאֵילִם֙ הַֽמְאָדָּמִ֔ים וְאֶת-מִכְסֵ֖ה עֹרֹ֣ת הַתְּחָשִׁ֑ים וְאֵ֖ת פָּרֹ֥כֶת הַמָּסָֽךְ:  ת  וְיָת חוֹפָאָה מַשְׁכֵי דְדִכְרֵי מְסַמְקֵי וְיָת חוֹפָאָה מַשְׁכֵי דְסַסְגוֹנָא וְיָת פָּרֻכְתָּא דִפְרָסָא:

{לה} אֶת-אֲר֥וֹן הָֽעֵדֻ֖ת וְאֶת-בַּדָּ֑יו וְאֵ֖ת הַכַּפֹּֽרֶת:  מ  אֶת-אֲר֥וֹן הָֽעֵדֻ֖ת וְאֶת-בַּדָּ֑יו וְאֵ֖ת הַכַּפֹּֽרֶת:  ת  יָת אֲרוֹנָא דְסַהֲדוּתָא וְיָת אֲרִיחוֹהִי וְיָת כַּפֻּרְתָּא:

{לו} אֶת-הַשֻּׁלְחָן֙ אֶת-כָּל-כֵּלָ֔יו וְאֵ֖ת לֶ֥חֶם הַפָּנִֽים:  מ  אֶת-הַשֻּׁלְחָן֙ אֶת-כָּל-כֵּלָ֔יו וְאֵ֖ת לֶ֥חֶם הַפָּנִֽים:  ת  יָת פָּתוֹרָא יָת כָּל מָנוֹהִי וְיָת לְחֵם אַפַּיָא:

{לז} אֶת-הַמְּנֹרָ֨ה הַטְּהֹרָ֜ה אֶת-נֵֽרֹתֶ֗יהָ נֵרֹ֛ת הַמַּֽעֲרָכָ֖ה וְאֶת-כָּל-כֵּלֶ֑יהָ וְאֵ֖ת שֶׁ֥מֶן הַמָּאֽוֹר:  מ  אֶת-הַמְּנֹרָ֨ה הַטְּהֹרָ֜ה אֶת-נֵֽרֹתֶ֗יהָ נֵרֹ֛ת הַמַּֽעֲרָכָ֖ה וְאֶת-כָּל-כֵּלֶ֑יהָ וְאֵ֖ת שֶׁ֥מֶן הַמָּאֽוֹר:  ת  יָת מְנַרְתָּא דָכִיתָא יָת בּוֹצִינָהָא בּוֹצִנֵי סִדְרָא וְיָת כָּל מָנָהָא וְיָת מִשְׁחָא דְאַנְהָרוּתָא:

{לח} וְאֵת֙ מִזְבַּ֣ח הַזָּהָ֔ב וְאֵת֙ שֶׁ֣מֶן הַמִּשְׁחָ֔ה וְאֵ֖ת קְטֹ֣רֶת הַסַּמִּ֑ים וְאֵ֕ת מָסַ֖ךְ פֶּ֥תַֽח הָאֹֽהֶל:  מ  וְאֵת֙ מִזְבַּ֣ח הַזָּהָ֔ב וְאֵת֙ שֶׁ֣מֶן הַמִּשְׁחָ֔ה וְאֵ֖ת קְטֹ֣רֶת הַסַּמִּ֑ים וְאֵ֕ת מָסַ֖ךְ פֶּ֥תַֽח הָאֹֽהֶל:  ת  וְיָת מַדְבְּחָא דְדַהֲבָא וְיָת מִשְׁחָא דִרְבוּתָא וְיָת קְטֹרֶת בּוּסְמַיָא וְיָת פְּרָסָא דִתְרַע מַשְׁכְּנָא:

{לט} אֵ֣ת | מִזְבַּ֣ח הַנְּחֹ֗שֶׁת וְאֶת-מִכְבַּ֤ר הַנְּחֹ֨שֶׁת֙ אֲשֶׁר-ל֔וֹ אֶת-בַּדָּ֖יו וְאֶת-כָּל-כֵּלָ֑יו אֶת-הַכִּיֹּ֖ר וְאֶת-כַּנּֽוֹ:  מ  אֵ֣ת | מִזְבַּ֣ח הַנְּחֹ֗שֶׁת וְאֶת-מִכְבַּ֤ר הַנְּחֹ֨שֶׁת֙ אֲשֶׁר-ל֔וֹ אֶת-בַּדָּ֖יו וְאֶת-כָּל-כֵּלָ֑יו אֶת-הַכִּיֹּ֖ר וְאֶת-כַּנּֽוֹ:  ת  יָת מַדְבְּחָא דִנְחָשָׁא וְיָת סְרָדָא דִנְחָשָׁא דִילֵהּ יָת אֲרִיחוֹהִי וְיָת כָּל מָנוֹהִי יָת כִּיוֹרָא וְיָת בְּסִיסֵהּ:

{מ} אֵת֩ קַלְעֵ֨י הֶֽחָצֵ֜ר אֶת-עַמֻּדֶ֣יהָ וְאֶת-אֲדָנֶ֗יהָ וְאֶת-הַמָּסָךְ֙ לְשַׁ֣עַר הֶֽחָצֵ֔ר אֶת-מֵֽיתָרָ֖יו וִיתֵֽדֹתֶ֑יהָ וְאֵ֗ת כָּל-כְּלֵ֛י עֲבֹדַ֥ת הַמִּשְׁכָּ֖ן לְאֹ֥הֶל מוֹעֵֽד:  מ  אֵת֩ קַלְעֵ֨י הֶֽחָצֵ֜ר אֶת-עַמֻּדֶ֣יהָ וְאֶת-אֲדָנֶ֗יהָ וְאֶת-הַמָּסָךְ֙ לְשַׁ֣עַר הֶֽחָצֵ֔ר אֶת-מֵֽיתָרָ֖יו וִיתֵֽדֹתֶ֑יהָ וְאֵ֗ת כָּל-כְּלֵ֛י עֲבֹדַ֥ת הַמִּשְׁכָּ֖ן לְאֹ֥הֶל מוֹעֵֽד:  ת  יָת סְרָדֵי דְדַרְתָּא יָת עַמוּדָהָא וְיָת סַמְכָהָא וְיָת פְּרָסָא לִתְרַע דַרְתָּא יָת אֲטוּנוֹהִי וְסִכָּהָא וְיָת כָּל מָנֵי פָּלְחַן מַשְׁכְּנָא לְמַשְׁכַּן זִמְנָא:

{מא} אֶת-בִּגְדֵ֥י הַשְּׂרָ֖ד לְשָׁרֵ֣ת בַּקֹּ֑דֶשׁ אֶת-בִּגְדֵ֤י הַקֹּ֨דֶשׁ֙ לְאַֽהֲרֹ֣ן הַכֹּהֵ֔ן וְאֶת-בִּגְדֵ֥י בָנָ֖יו לְכַהֵֽן:  מ  אֶת-בִּגְדֵ֥י הַשְּׂרָ֖ד לְשָׁרֵ֣ת בַּקֹּ֑דֶשׁ אֶת-בִּגְדֵ֤י הַקֹּ֨דֶשׁ֙ לְאַֽהֲרֹ֣ן הַכֹּהֵ֔ן וְאֶת-בִּגְדֵ֥י בָנָ֖יו לְכַהֵֽן:  ת  יָת לְבוּשֵׁי שִׁמוּשָׁא לְשַׁמָשָׁא בְקוּדְשָׁא יָת לְבוּשֵׁי קוּדְשָׁא לְאַהֲרֹן כַּהֲנָא וְיָת לְבוּשֵׁי בְנוֹהִי לְשַׁמָשָׁא:

{מב} כְּכֹ֛ל אֲשֶׁר-צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶׁ֑ה כֵּ֤ן עָשׂוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֵ֖ת כָּל-הָֽעֲבֹדָֽה:  מ  כְּכֹ֛ל אֲשֶׁר-צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶׁ֑ה כֵּ֤ן עָשׂוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֵ֖ת כָּל-הָֽעֲבֹדָֽה:  ת  כְּכֹל דִי פַקִיד יְיָ יָת משֶׁה כֵּן עֲבָדוּ בְּנֵי יִשְׂרָאֵל יָת כָּל פָּלְחָנָא:

{מג} וַיַּ֨רְא מֹשֶׁ֜ה אֶת-כָּל-הַמְּלָאכָ֗ה וְהִנֵּה֙ עָשׂ֣וּ אֹתָ֔הּ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה כֵּ֣ן עָשׂ֑וּ וַיְבָ֥רֶךְ אֹתָ֖ם מֹשֶֽׁה:  מ  וַיַּ֨רְא מֹשֶׁ֜ה אֶת-כָּל-הַמְּלָאכָ֗ה וְהִנֵּה֙ עָשׂ֣וּ אֹתָ֔הּ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה כֵּ֣ן עָשׂ֑וּ וַיְבָ֥רֶךְ אֹתָ֖ם מֹשֶֽׁה:  ת  וַחֲזָא משֶׁה יָת כָּל עִבִדְתָּא וְהָא עֲבָדוּ יָתַהּ כְּמָא דִי פַקִיד יְיָ כֵּן עֲבָדוּ וּבָרִיךְ יָתְהוֹן משֶׁה: (פ)




שמות פרק-מ

{א}  חמישי  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{ב} בְּיוֹם-הַחֹ֥דֶשׁ הָֽרִאשׁ֖וֹן בְּאֶחָ֣ד לַחֹ֑דֶשׁ תָּקִ֕ים אֶת-מִשְׁכַּ֖ן אֹ֥הֶל מוֹעֵֽד:  מ  בְּיוֹם-הַחֹ֥דֶשׁ הָֽרִאשׁ֖וֹן בְּאֶחָ֣ד לַחֹ֑דֶשׁ תָּקִ֕ים אֶת-מִשְׁכַּ֖ן אֹ֥הֶל מוֹעֵֽד:  ת  בְּיוֹמָא יַרְחָא קַדְמָאָה בְּחַד לְיַרְחָא תְּקִים יָת מַשְׁכְּנָא מַשְּׁכַּן זִמְנָא:

{ג} וְשַׂמְתָּ֣ שָׁ֔ם אֵ֖ת אֲר֣וֹן הָֽעֵד֑וּת וְסַכֹּתָ֥ עַל-הָֽאָרֹ֖ן אֶת-הַפָּרֹֽכֶת:  מ  וְשַׂמְתָּ֣ שָׁ֔ם אֵ֖ת אֲר֣וֹן הָֽעֵד֑וּת וְסַכֹּתָ֥ עַל-הָֽאָרֹ֖ן אֶת-הַפָּרֹֽכֶת:  ת  וּתְשַׁוֵי תַמָן יַת אֲרוֹנָא דְסַהֲדוּתָא וְתַטֵּל עַל אֲרוֹנָא יָת פָּרֻכְתָּא:

{ד} וְהֵֽבֵאתָ֙ אֶת-הַשֻּׁלְחָ֔ן וְעָֽרַכְתָּ֖ אֶת-עֶרְכּ֑וֹ וְהֵֽבֵאתָ֙ אֶת-הַמְּנֹרָ֔ה וְהַֽעֲלֵיתָ֖ אֶת-נֵֽרֹתֶֽיהָ:  מ  וְהֵֽבֵאתָ֙ אֶת-הַשֻּׁלְחָ֔ן וְעָֽרַכְתָּ֖ אֶת-עֶרְכּ֑וֹ וְהֵֽבֵאתָ֙ אֶת-הַמְּנֹרָ֔ה וְהַֽעֲלֵיתָ֖ אֶת-נֵֽרֹתֶֽיהָ:  ת  וְתָעֵל יָת פָּתוֹרָא וְתַסְדַר יָת סִדְרֵהּ וְתָעֵל יָת מְנַרְתָּא וְתַדְלֵק יָת בּוֹצִינָהָא:

{ה} וְנָֽתַתָּ֞ה אֶת-מִזְבַּ֤ח הַזָּהָב֙ לִקְטֹ֔רֶת לִפְנֵ֖י אֲר֣וֹן הָֽעֵדֻ֑ת וְשַׂמְתָּ֛ אֶת-מָסַ֥ךְ הַפֶּ֖תַח לַמִּשְׁכָּֽן:  מ  וְנָֽתַתָּ֞ה אֶת-מִזְבַּ֤ח הַזָּהָב֙ לִקְטֹ֔רֶת לִפְנֵ֖י אֲר֣וֹן הָֽעֵדֻ֑ת וְשַׂמְתָּ֛ אֶת-מָסַ֥ךְ הַפֶּ֖תַח לַמִּשְׁכָּֽן:  ת  וְתִתֵּן יָת מַדְבְּחָא דְדַהֲבָא לִקְטֹּרֶת בּוּסְמַיָא קֳדָם אֲרוֹנָא דְסַהֲדוּתָא וּתְשַׁוֵי יָת פְּרָסָא דְתַרְעָא לְמַשְׁכְּנָא:

{ו} וְנָ֣תַתָּ֔ה אֵ֖ת מִזְבַּ֣ח הָֽעֹלָ֑ה לִפְנֵ֕י פֶּ֖תַח מִשְׁכַּ֥ן אֹֽהֶל-מוֹעֵֽד:  מ  וְנָ֣תַתָּ֔ה אֵ֖ת מִזְבַּ֣ח הָֽעֹלָ֑ה לִפְנֵ֕י פֶּ֖תַח מִשְׁכַּ֥ן אֹֽהֶל-מוֹעֵֽד:  ת  וְתִתֵּן יָת מַדְבְּחָא דַעֲלָתָא קֳדָם תְּרַע מַשְׁכְּנָא מַשְׁכַּן זִמְנָא:

{ז} וְנָֽתַתָּ֙ אֶת-הַכִּיֹּ֔ר בֵּֽין-אֹ֥הֶל מוֹעֵ֖ד וּבֵ֣ין הַמִּזְבֵּ֑חַ וְנָֽתַתָּ֥ שָׁ֖ם מָֽיִם:  מ  וְנָֽתַתָּ֙ אֶת-הַכִּיֹּ֔ר בֵּֽין-אֹ֥הֶל מוֹעֵ֖ד וּבֵ֣ין הַמִּזְבֵּ֑חַ וְנָֽתַתָּ֥ שָׁ֖ם מָֽיִם:  ת  וְתִתֵּן יָת כִּיוֹרָא בֵּין מַשְׁכַּן זִמְנָא וּבֵין מַדְבְּחָא וְתִתֵּן תַּמָן מַיָא:

{ח} וְשַׂמְתָּ֥ אֶת-הֶֽחָצֵ֖ר סָבִ֑יב וְנָ֣תַתָּ֔ אֶת-מָסַ֖ךְ שַׁ֥עַר הֶֽחָצֵֽר:  מ  וְשַׂמְתָּ֥ אֶת-הֶֽחָצֵ֖ר סָבִ֑יב וְנָ֣תַתָּ֔ אֶת-מָסַ֖ךְ שַׁ֥עַר הֶֽחָצֵֽר:  ת  וּתְשַׁוֵי יָת דַרְתָּא סְחוֹר סְחוֹר וְתִתֵּן יָת פְּרָסָא דִתְרַע דַרְתָּא:

{ט} וְלָֽקַחְתָּ֙ אֶת-שֶׁ֣מֶן הַמִּשְׁחָ֔ה וּמָֽשַׁחְתָּ֥ אֶת-הַמִּשְׁכָּ֖ן וְאֶת-כָּל-אֲשֶׁר-בּ֑וֹ וְקִדַּשְׁתָּ֥ אֹת֛וֹ וְאֶת-כָּל-כֵּלָ֖יו וְהָ֥יָה קֹֽדֶשׁ:  מ  וְלָֽקַחְתָּ֙ אֶת-שֶׁ֣מֶן הַמִּשְׁחָ֔ה וּמָֽשַׁחְתָּ֥ אֶת-הַמִּשְׁכָּ֖ן וְאֶת-כָּל-אֲשֶׁר-בּ֑וֹ וְקִדַּשְׁתָּ֥ אֹת֛וֹ וְאֶת-כָּל-כֵּלָ֖יו וְהָ֥יָה קֹֽדֶשׁ:  ת  וְתִסַב יָת מִשְׁחָא דִרְבוּתָא וּתְרַבֵּי יָת מַשְׁכְּנָא וְיָת כָּל דִי בֵהּ וּתְקַדֵשׁ יָתֵהּ וְיָת כָּל מָנוֹהִי וִיהֵי קוּדְשָׁא:

{י} וּמָֽשַׁחְתָּ֛ אֶת-מִזְבַּ֥ח הָֽעֹלָ֖ה וְאֶת-כָּל-כֵּלָ֑יו וְקִדַּשְׁתָּ֙ אֶת-הַמִּזְבֵּ֔חַ וְהָיָ֥ה הַמִּזְבֵּ֖חַ קֹ֥דֶשׁ קָֽדָשִֽׁים:  מ  וּמָֽשַׁחְתָּ֛ אֶת-מִזְבַּ֥ח הָֽעֹלָ֖ה וְאֶת-כָּל-כֵּלָ֑יו וְקִדַּשְׁתָּ֙ אֶת-הַמִּזְבֵּ֔חַ וְהָיָ֥ה הַמִּזְבֵּ֖חַ קֹ֥דֶשׁ קָֽדָשִֽׁים:  ת  וּתְרַבֵי יָת מַדְבְּחָא דַעֲלָתָא וְיָת כָּל מָנוֹהִי וּתְקַדֵשׁ יָת מַדְבְּחָא וִיהֵי מַדְבְּחָא קֹדֶשׁ קוּדְשִׁין:

{יא} וּמָֽשַׁחְתָּ֥ אֶת-הַכִּיֹּ֖ר וְאֶת-כַּנּ֑וֹ וְקִדַּשְׁתָּ֖ אֹתֽוֹ:  מ  וּמָֽשַׁחְתָּ֥ אֶת-הַכִּיֹּ֖ר וְאֶת-כַּנּ֑וֹ וְקִדַּשְׁתָּ֖ אֹתֽוֹ:  ת  וּתְרַבֵּי יָת כִּיוֹרָא וְיָת בְּסִיסֵהּ וּתְקַדֵשׁ יָתֵהּ:

{יב} וְהִקְרַבְתָּ֤ אֶֽת-אַֽהֲרֹן֙ וְאֶת-בָּנָ֔יו אֶל-פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְרָֽחַצְתָּ֥ אֹתָ֖ם בַּמָּֽיִם:  מ  וְהִקְרַבְתָּ֤ אֶֽת-אַֽהֲרֹן֙ וְאֶת-בָּנָ֔יו אֶל-פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְרָֽחַצְתָּ֥ אֹתָ֖ם בַּמָּֽיִם:  ת  וּתְקָרֵב יָת אַהֲרֹן וְיָת בְּנוֹהִי לִתְרַע מַשְׁכַּן זִמְנָא וְתַסְחֵי יָתְהוֹן בְּמַיָא:

{יג} וְהִלְבַּשְׁתָּ֙ אֶֽת-אַ֣הֲרֹ֔ן אֵ֖ת בִּגְדֵ֣י הַקֹּ֑דֶשׁ וּמָֽשַׁחְתָּ֥ אֹת֛וֹ וְקִדַּשְׁתָּ֥ אֹת֖וֹ וְכִהֵ֥ן לִֽי:  מ  וְהִלְבַּשְׁתָּ֙ אֶֽת-אַ֣הֲרֹ֔ן אֵ֖ת בִּגְדֵ֣י הַקֹּ֑דֶשׁ וּמָֽשַׁחְתָּ֥ אֹת֛וֹ וְקִדַּשְׁתָּ֥ אֹת֖וֹ וְכִהֵ֥ן לִֽי:  ת  וְתַלְבֵּשׁ יָת אַהֲרֹן יָת לְבוּשֵׁי קוּדְשָׁא וּתְרַבֵּי יָתֵהּ וּתְקַדֵשׁ יָתֵהּ וִישַׁמֵשׁ קֳדָמָי:

{יד} וְאֶת-בָּנָ֖יו תַּקְרִ֑יב וְהִלְבַּשְׁתָּ֥ אֹתָ֖ם כֻּתֳּנֹֽת:  מ  וְאֶת-בָּנָ֖יו תַּקְרִ֑יב וְהִלְבַּשְׁתָּ֥ אֹתָ֖ם כֻּתֳּנֹֽת:  ת  וְיָת בְּנוֹהִי תְּקָרֵב וְתַלְבֵּשּׁ יָתְהוֹן כִּתּוּנִין:

{טו} וּמָֽשַׁחְתָּ֣ אֹתָ֗ם כַּֽאֲשֶׁ֤ר מָשַׁ֨חְתָּ֙ אֶת-אֲבִיהֶ֔ם וְכִֽהֲנ֖וּ לִ֑י וְ֠הָֽיְתָ֠ה לִהְיֹ֨ת לָהֶ֧ם מָשְׁחָתָ֛ם לִכְהֻנַּ֥ת עוֹלָ֖ם לְדֹֽרֹתָֽם:  מ  וּמָֽשַׁחְתָּ֣ אֹתָ֗ם כַּֽאֲשֶׁ֤ר מָשַׁ֨חְתָּ֙ אֶת-אֲבִיהֶ֔ם וְכִֽהֲנ֖וּ לִ֑י וְ֠הָֽיְתָ֠ה לִהְיֹ֨ת לָהֶ֧ם מָשְׁחָתָ֛ם לִכְהֻנַּ֥ת עוֹלָ֖ם לְדֹֽרֹתָֽם:  ת  וּתְרַבִּי יָתְהוֹן כְּמָא דִי רַבִּיתָא יָת אֲבוּהוֹן וִישַׁמְשׁוּן קֳדָמָי וּתְהֵי לְמֶהֱוֵי לְּהוֹן רְבוּתְהוֹן לְכַהֲנוּת עֲלָם לְדָרֵיהוֹן:

{טז} וַיַּ֖עַשׂ מֹשֶׁ֑ה כְּ֠כֹ֠ל אֲשֶׁ֨ר צִוָּ֧ה יְהוָֹ֛ה אֹת֖וֹ כֵּ֥ן עָשָֽׂה:  מ  וַיַּ֖עַשׂ מֹשֶׁ֑ה כְּ֠כֹ֠ל אֲשֶׁ֨ר צִוָּ֧ה יְהוָֹ֛ה אֹת֖וֹ כֵּ֥ן עָשָֽׂה:  ת  וַעֲבַד משֶׁה כְּכֹל דִי פַקִיד יְיָ יָתֵהּ כֵּן עֲבָד: (ס)

{יז}  שישי  וַיְהִ֞י בַּחֹ֧דֶשׁ הָֽרִאשׁ֛וֹן בַּשָּׁנָ֥ה הַשֵּׁנִ֖ית בְּאֶחָ֣ד לַחֹ֑דֶשׁ הוּקַ֖ם הַמִּשְׁכָּֽן:  מ  וַיְהִ֞י בַּחֹ֧דֶשׁ הָֽרִאשׁ֛וֹן בַּשָּׁנָ֥ה הַשֵּׁנִ֖ית בְּאֶחָ֣ד לַחֹ֑דֶשׁ הוּקַ֖ם הַמִּשְׁכָּֽן:  ת  וַהֲוָה בְּיַרְחָא קַדְמָאָה בְּשַׁתָּא תִנְיֵתָא בְּחַד לְיַרְחָא אִתָּקַם מַשְׁכְּנָא:

{יח} וַיָּ֨קֶם מֹשֶׁ֜ה אֶת-הַמִּשְׁכָּ֗ן וַיִּתֵּן֙ אֶת-אֲדָנָ֔יו וַיָּ֨שֶׂם֙ אֶת-קְרָשָׁ֔יו וַיִּתֵּ֖ן אֶת-בְּרִיחָ֑יו וַיָּ֖קֶם אֶת-עַמּוּדָֽיו:  מ  וַיָּ֨קֶם מֹשֶׁ֜ה אֶת-הַמִּשְׁכָּ֗ן וַיִּתֵּן֙ אֶת-אֲדָנָ֔יו וַיָּ֨שֶׂם֙ אֶת-קְרָשָׁ֔יו וַיִּתֵּ֖ן אֶת-בְּרִיחָ֑יו וַיָּ֖קֶם אֶת-עַמּוּדָֽיו:  ת  וַאֲקֵם משֶׁה יָת מַשְׁכְּנָא וִיהַב יָת סַמְכוֹהִי וְשַׁוִי יָת דַפּוֹהִי וִיהַב יָת עַבְרוֹהִי וַאֲקֵם יָת עַמוּדוֹהִי:

{יט} וַיִּפְרֹ֤שׂ אֶת-הָאֹ֨הֶל֙ עַל-הַמִּשְׁכָּ֔ן וַיָּ֜שֶׂם אֶת-מִכְסֵ֥ה הָאֹ֛הֶל עָלָ֖יו מִלְמָ֑עְלָה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיִּפְרֹ֤שׂ אֶת-הָאֹ֨הֶל֙ עַל-הַמִּשְׁכָּ֔ן וַיָּ֜שֶׂם אֶת-מִכְסֵ֥ה הָאֹ֛הֶל עָלָ֖יו מִלְמָ֑עְלָה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וּפְרַס יָת פְרָסָא עַל מַשְׁכְּנָא וְשַׁוִי יָת חוֹפָאָה דְמַשְׁכְּנָא עֲלוֹהִי מִלְעֵלָא כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{כ} וַיִּקַּ֞ח וַיִּתֵּ֤ן אֶת-הָֽעֵדֻת֙ אֶל-הָ֣אָרֹ֔ן וַיָּ֥שֶׂם אֶת-הַבַּדִּ֖ים עַל-הָֽאָרֹ֑ן וַיִּתֵּ֧ן אֶת-הַכַּפֹּ֛רֶת עַל-הָֽאָרֹ֖ן מִלְמָֽעְלָה:  מ  וַיִּקַּ֞ח וַיִּתֵּ֤ן אֶת-הָֽעֵדֻת֙ אֶל-הָ֣אָרֹ֔ן וַיָּ֥שֶׂם אֶת-הַבַּדִּ֖ים עַל-הָֽאָרֹ֑ן וַיִּתֵּ֧ן אֶת-הַכַּפֹּ֛רֶת עַל-הָֽאָרֹ֖ן מִלְמָֽעְלָה:  ת  וּנְסִיב וִיהַב יָת סַהֲדוּתָא לַאֲרוֹנָא וְשַׁוִי יָת אֲרִיחַיָא עַל אֲרוֹנָא וִיהַב יָת כַּפֻּרְתָּא עַל אֲרוֹנָא מִלְעֵלָא:

{כא} וַיָּבֵ֣א אֶת-הָֽאָרֹן֘ אֶל-הַמִּשְׁכָּן֒ וַיָּ֗שֶׂם אֵ֚ת פָּרֹ֣כֶת הַמָּסָ֔ךְ וַיָּ֕סֶךְ עַ֖ל אֲר֣וֹן הָֽעֵד֑וּת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיָּבֵ֣א אֶת-הָֽאָרֹן֘ אֶל-הַמִּשְׁכָּן֒ וַיָּ֗שֶׂם אֵ֚ת פָּרֹ֣כֶת הַמָּסָ֔ךְ וַיָּ֕סֶךְ עַ֖ל אֲר֣וֹן הָֽעֵד֑וּת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְאָעֵלּ יָת אֲרוֹנָא לְמַשְׁכְּנָא וְשַׁוִי יָת פָּרֻכְתָּא דִפְרָסָא וְאַטֵל עַל אֲרוֹנָא דְסַהֲדוּתָא כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{כב} וַיִּתֵּ֤ן אֶת-הַשֻּׁלְחָן֙ בְּאֹ֣הֶל מוֹעֵ֔ד עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן צָפֹ֑נָה מִח֖וּץ לַפָּרֹֽכֶת:  מ  וַיִּתֵּ֤ן אֶת-הַשֻּׁלְחָן֙ בְּאֹ֣הֶל מוֹעֵ֔ד עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן צָפֹ֑נָה מִח֖וּץ לַפָּרֹֽכֶת:  ת  וִיהַב יָת פָּתוֹרָא בְּמַשְׁכַּן זִמְנָא עַל צִדָא דְמַשְׁכְּנָא צִפוּנָא מִבָּרָא לְפָרֻכְתָּא:

{כג} וַיַּֽעֲרֹ֥ךְ עָלָ֛יו עֵ֥רֶךְ לֶ֖חֶם לִפְנֵ֣י יְהוָֹ֑ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיַּֽעֲרֹ֥ךְ עָלָ֛יו עֵ֥רֶךְ לֶ֖חֶם לִפְנֵ֣י יְהוָֹ֑ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְסַדַר עֲלוֹהִי סִדוּר דִלְחֵם קֳדָם יְיָ כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{כד} וַיָּ֤שֶׂם אֶת-הַמְּנֹרָה֙ בְּאֹ֣הֶל מוֹעֵ֔ד נֹ֖כַח הַשֻּׁלְחָ֑ן עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן נֶֽגְבָּה:  מ  וַיָּ֤שֶׂם אֶת-הַמְּנֹרָה֙ בְּאֹ֣הֶל מוֹעֵ֔ד נֹ֖כַח הַשֻּׁלְחָ֑ן עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן נֶֽגְבָּה:  ת  וְשַׁוִי יָת מְנַרְתָּא בְּמַשְׁכַּן זִמְנָא לָקֳבֵל פָּתוֹרָא עַל צִדָא דְמַשְׁכְּנָא לְדָרוֹמָא:

{כה} וַיַּ֥עַל הַנֵּרֹ֖ת לִפְנֵ֣י יְהוָֹה֑ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיַּ֥עַל הַנֵּרֹ֖ת לִפְנֵ֣י יְהוָֹה֑ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְאַדְלֵק בּוֹצִינָהָא קֳדָם יְיָ כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{כו} וַיָּ֛שֶׂם אֶת-מִזְבַּ֥ח הַזָּהָ֖ב בְּאֹ֣הֶל מוֹעֵ֑ד לִפְנֵ֖י הַפָּרֹֽכֶת:  מ  וַיָּ֛שֶׂם אֶת-מִזְבַּ֥ח הַזָּהָ֖ב בְּאֹ֣הֶל מוֹעֵ֑ד לִפְנֵ֖י הַפָּרֹֽכֶת:  ת  וְשַׁוִי יָת מַדְבְּחָא דְדַהֲבָא בְּמַשְׁכַּן זִמְנָא קֳדָם פָּרֻכְתָּא:

{כז} וַיַּקְטֵ֥ר עָלָ֖יו קְטֹ֣רֶת סַמִּ֑ים כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וַיַּקְטֵ֥ר עָלָ֖יו קְטֹ֣רֶת סַמִּ֑ים כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְאַקְטַּר עֲלוֹהִי קְטֹרֶת בּוּסְמִין כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{כח}  שביעי  וַיָּ֛שֶׂם אֶת-מָסַ֥ךְ הַפֶּ֖תַח לַמִּשְׁכָּֽן:  מ  וַיָּ֛שֶׂם אֶת-מָסַ֥ךְ הַפֶּ֖תַח לַמִּשְׁכָּֽן:  ת  וְשַׁוִי יָת פְּרָסָא דְתַרְעָא לְמַשְׁכְּנָא:

{כט} וְאֵת֙ מִזְבַּ֣ח הָֽעֹלָ֔ה שָׂ֕ם פֶּ֖תַח מִשְׁכַּ֣ן אֹֽהֶל-מוֹעֵ֑ד וַיַּ֣עַל עָלָ֗יו אֶת-הָֽעֹלָה֙ וְאֶת-הַמִּנְחָ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  וְאֵת֙ מִזְבַּ֣ח הָֽעֹלָ֔ה שָׂ֕ם פֶּ֖תַח מִשְׁכַּ֣ן אֹֽהֶל-מוֹעֵ֑ד וַיַּ֣עַל עָלָ֗יו אֶת-הָֽעֹלָה֙ וְאֶת-הַמִּנְחָ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  וְיָת מַדְבְּחָא דַעֲלָתָא שַׁוִי בִּתְרַע מַשְׁכְּנָא מַשְׁכַּן זִמְנָא וְאַסֵק עֲלוֹהִי יָת עֲלָתָא וְיָת מִנְחָתָּא כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{ל} וַיָּ֨שֶׂם֙ אֶת-הַכִּיֹּ֔ר בֵּין-אֹ֥הֶל מוֹעֵ֖ד וּבֵ֣ין הַמִּזְבֵּ֑חַ וַיִּתֵּ֥ן שָׁ֛מָּה מַ֖יִם לְרָחְצָֽה:  מ  וַיָּ֨שֶׂם֙ אֶת-הַכִּיֹּ֔ר בֵּין-אֹ֥הֶל מוֹעֵ֖ד וּבֵ֣ין הַמִּזְבֵּ֑חַ וַיִּתֵּ֥ן שָׁ֛מָּה מַ֖יִם לְרָחְצָֽה:  ת  וְשַׁוִי יָת כִּיוֹרָא בֵּין מַשְׁכַּן זִמְנָא וּבֵין מַדְבְּחָא וִיהַב תַּמָן מַיָא לְקִדוּשׁ:

{לא} וְרָֽחֲצ֣וּ מִמֶּ֔נּוּ מֹשֶׁ֖ה וְאַֽהֲרֹ֣ן וּבָנָ֑יו אֶת-יְדֵיהֶ֖ם וְאֶת-רַגְלֵיהֶֽם:  מ  וְרָֽחֲצ֣וּ מִמֶּ֔נּוּ מֹשֶׁ֖ה וְאַֽהֲרֹ֣ן וּבָנָ֑יו אֶת-יְדֵיהֶ֖ם וְאֶת-רַגְלֵיהֶֽם:  ת  וּמְקַדְשִׁין מִנֵהּ משֶׁה וְאַהֲרֹן וּבְנוֹהִי יָת יְדֵיהוֹן וְיָת רַגְלֵיהוֹן:

{לב} בְּבֹאָ֞ם אֶל-אֹ֣הֶל מוֹעֵ֗ד וּבְקָרְבָתָ֛ם אֶל-הַמִּזְבֵּ֖חַ יִרְחָ֑צוּ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  מ  בְּבֹאָ֞ם אֶל-אֹ֣הֶל מוֹעֵ֗ד וּבְקָרְבָתָ֛ם אֶל-הַמִּזְבֵּ֖חַ יִרְחָ֑צוּ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֶת-מֹשֶֽׁה:  ת  בְּמֵעַלְהוֹן לְמַשְׁכַּן זִמְנָא וּבְמִקְרָבְהוֹן לְמַדְבְּחָא יְקַדְשׁוּן כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (ס)

{לג} וַיָּ֣קֶם אֶת-הֶֽחָצֵ֗ר סָבִיב֙ לַמִּשְׁכָּ֣ן וְלַמִּזְבֵּ֔חַ וַיִּתֵּ֕ן אֶת-מָסַ֖ךְ שַׁ֣עַר הֶֽחָצֵ֑ר וַיְכַ֥ל מֹשֶׁ֖ה אֶת-הַמְּלָאכָֽה:  מ  וַיָּ֣קֶם אֶת-הֶֽחָצֵ֗ר סָבִיב֙ לַמִּשְׁכָּ֣ן וְלַמִּזְבֵּ֔חַ וַיִּתֵּ֕ן אֶת-מָסַ֖ךְ שַׁ֣עַר הֶֽחָצֵ֑ר וַיְכַ֥ל מֹשֶׁ֖ה אֶת-הַמְּלָאכָֽה:  ת  וַאֲקֵם יָת דַרְתָּא סְחוֹר סְחוֹר לְמַשְׁכְּנָא וּלְמַדְבְּחָא וִיהַב יָת פְּרָסָא דִתְרַע דַרְתָּא וְשֵׁיצֵי משֶׁה יָת עִבִידָא: (פ)

{לד}  מפטיר  וַיְכַ֥ס הֶֽעָנָ֖ן אֶת-אֹ֣הֶל מוֹעֵ֑ד וּכְב֣וֹד יְהֹוָ֔ה מָלֵ֖א אֶת-הַמִּשְׁכָּֽן:  מ  וַיְכַ֥ס הֶֽעָנָ֖ן אֶת-אֹ֣הֶל מוֹעֵ֑ד וּכְב֣וֹד יְהֹוָ֔ה מָלֵ֖א אֶת-הַמִּשְׁכָּֽן:  ת  וַחֲפָא עֲנָנָא יָת מַשְׁכַּן זִמְנָא וִיקָרָא דַיְיָ אִתְמְלֵי יָת מַשְׁכְּנָא:

{לה} וְלֹֽא-יָכֹ֣ל מֹשֶׁ֗ה לָבוֹא֙ אֶל-אֹ֣הֶל מוֹעֵ֔ד כִּֽי-שָׁכַ֥ן עָלָ֖יו הֶֽעָנָ֑ן וּכְב֣וֹד יְהֹוָ֔ה מָלֵ֖א אֶת-הַמִּשְׁכָּֽן:  מ  וְלֹֽא-יָכֹ֣ל מֹשֶׁ֗ה לָבוֹא֙ אֶל-אֹ֣הֶל מוֹעֵ֔ד כִּֽי-שָׁכַ֥ן עָלָ֖יו הֶֽעָנָ֑ן וּכְב֣וֹד יְהֹוָ֔ה מָלֵ֖א אֶת-הַמִּשְׁכָּֽן:  ת  וְלָא יָכִיל משֶׁה לְמֵיעַל לְמַשְׁכַּן זִמְנָא אֲרֵי שְׁרָא עֲלוֹהִי עֲנָנָא וִיקָרָא דַיְיָ אִתְמְלִי יָת מַשְׁכְּנָא:

{לו} וּבְהֵֽעָל֤וֹת הֶֽעָנָן֙ מֵעַ֣ל הַמִּשְׁכָּ֔ן יִסְע֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל בְּכֹ֖ל מַסְעֵיהֶֽם:  מ  וּבְהֵֽעָל֤וֹת הֶֽעָנָן֙ מֵעַ֣ל הַמִּשְׁכָּ֔ן יִסְע֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל בְּכֹ֖ל מַסְעֵיהֶֽם:  ת  וּבְאִסְתַּלָקוּת עֲנָנָא מֵעִלָוֵי מַשְׁכְּנָא נָטְלִין בְּנֵי יִשְׂרָאֵל בְּכֹל מַטְלָנֵיהוֹן:

{לז} וְאִם-לֹ֥א יֵֽעָלֶ֖ה הֶֽעָנָ֑ן וְלֹ֣א יִסְע֔וּ עַד-י֖וֹם הֵעָֽלֹתֽוֹ:  מ  וְאִם-לֹ֥א יֵֽעָלֶ֖ה הֶֽעָנָ֑ן וְלֹ֣א יִסְע֔וּ עַד-י֖וֹם הֵעָֽלֹתֽוֹ:  ת  וְאִם לָא יִסְתַּלַק עֲנָנָא וְלָא נָטְלִין עַד יוֹמָא דְאִסְתַּלָקוּתֵהּ:

{לח} כִּי֩ עֲנַ֨ן יְהֹוָ֤ה עַל-הַמִּשְׁכָּן֙ יוֹמָ֔ם וְאֵ֕שׁ תִּהְיֶ֥ה לַ֖יְלָה בּ֑וֹ לְעֵינֵ֥י כָל-בֵּית-יִשְׂרָאֵ֖ל בְּכָל-מַסְעֵיהֶֽם:  מ  כִּי֩ עֲנַ֨ן יְהֹוָ֤ה עַל-הַמִּשְׁכָּן֙ יוֹמָ֔ם וְאֵ֕שׁ תִּהְיֶ֥ה לַ֖יְלָה בּ֑וֹ לְעֵינֵ֥י כָל-בֵּית-יִשְׂרָאֵ֖ל בְּכָל-מַסְעֵיהֶֽם:  ת  אֲרֵי עֲנָנָא יְקָרָא דַיְיָ עַל מַשְׁכְּנָא בִּימָמָא וְחֵיזוּ אֶשָׁתָא הֲוֵי בְלֵילְיָא בֵּהּ לְעֵינֵי כָל בֵּית יִשְׂרָאֵל בְּכָל מַטְלָנֵיהוֹן: (חזק)
