פרק א - משנה א

בֵּיצָה שֶׁנּוֹלְדָה בְיוֹם טוֹב, בֵּית שַׁמַּאי אוֹמְרִים, תֵּאָכֵל. וּבֵית הִלֵּל אוֹמְרִים, לֹא תֵאָכֵל {ט}. בֵּית שַׁמַּאי אוֹמְרִים, שְׂאֹר בְּכַזַּיִת וְחָמֵץ בְּכַכּוֹתֶבֶת {יא}. וּבֵית הִלֵּל אוֹמְרִים, זֶה וָזֶה בְּכַזָּיִת {יב}:

 ר"ע מברטנורה  בֵּיצָה שֶׁנּוֹלְדָה בְיוֹם טוֹב. [בְּיוֹם טוֹב] שֶׁלְאַחַר שַׁבָּת עָסְקִינַן {א}. וְטַעֲמַיְהוּ דְּבֵית הִלֵּל דְּאָמְרֵי לֹא תֵאָכֵל, מִשּׁוּם דְּכָל בֵּיצָה דְּמִתְיַלְדָא הָאִידְנָא, מֵאֶתְמוֹל {ב} גָּמְרָה לַהּ, וְנִמְצֵאת שַׁבָּת מְכִינָה לְיוֹם טוֹב, וְהַתּוֹרָה אָמְרָה (שְׁמוֹת טז) וְהָיָה בַּיּוֹם הַשִּׁשִּׁי וְהֵכִינוּ {ג}, וּסְתָם שִׁשִּׁי חֹל הוּא, חֹל מֵכִין לְשַׁבָּת, וְחֹל מֵכִין לְיוֹם טוֹב, דְּיוֹם טוֹב נַמִּי אִקְרֵי שַׁבָּת {ד}, וְאֵין יוֹם טוֹב מֵכִין לְשַׁבָּת וְלֹא שַׁבָּת מְכִינָה לְיוֹם טוֹב {ה}. וַהֲכָנָה כִּי הַךְ דְּבֵיצָה אַף עַל גַּב דְּבִידֵי שָׁמָיִם הִיא, מִקַּרְיָא הֲכָנָה. וּמִיהוּ שַׁבָּת וְיוֹם טוֹב דַּחֲשִׁיבֵי, בָּעֲיָא סְעוּדָתָן הַזְמָנָה בְּחֹל, וְאֵין אֶחָד מֵהֶן מֵכִין לַחֲבֵרוֹ אֲפִלּוּ בַּהֲכָנָה דְּבִידֵי שָׁמָיִם, אֲבָל סְעוּדַת חֹל לֹא חֲשִׁיבָא וְלֹא בָּעֲיָא הַזְמָנָה, הִלְכָּךְ בְּאֶחָד בְּשַׁבָּת בְּעָלְמָא שֶׁאֵינוֹ יוֹם טוֹב לֵית לָן לְמֵיסַר בֵּיצָה שֶׁנּוֹלְדָה בּוֹ לְפִי שֶׁנִּגְמְרָה בְּשַׁבָּת, דִּסְעוּדַת חֹל לֹא אַצְרְכָהּ רַחֲמָנָא זִמּוּן מִבְּעוֹד יוֹם. וְאָסְרוּ בֵּיצָה שֶׁנּוֹלְדָה בְּכָל יוֹם טוֹב, וְאַף עַל פִּי שֶׁאֵינוֹ אַחַר שַׁבָּת, גְּזֵרָה מִשּׁוּם יוֹם טוֹב שֶׁאַחַר שַׁבָּת. וְכֵן אָסְרוּ בֵּיצָה שֶׁנּוֹלְדָה בְּכָל שַׁבָּת לְגָמְעָהּ חַיָּה, גְּזֵרָה מִשּׁוּם שַׁבָּת שֶׁאַחַר יוֹם טוֹב. וְשַׁבָּת וְיוֹם טוֹב הַסְּמוּכִים זֶה לָזֶה, בֵּיצָה שֶׁנּוֹלְדָה בָּזֶה אֲסוּרָה בָּזֶה {ו}. וְכֵן שְׁנֵי יָמִים טוֹבִים שֶׁל רֹאשׁ הַשָּׁנָה {ז}. אֲבָל שְׁנֵי יָמִים טוֹבִים שֶׁל גָּלֻיּוֹת, דְּמִמַּה נַּפְשָׁךְ אֶחָד מֵהֶם חֹל, בֵּיצָה שֶׁנּוֹלְדָה בָּזֶה מֻתֶּרֶת בָּזֶה {ח}: שְׂאֹר בְּכַזַּיִת. לְעִנְיַן אֲכִילָה כֻּלֵּי עָלְמָא לֹא פְלִיגֵי דְּזֶה וָזֶה בְּכַזַּיִת, מִדְּפָתַח הַכָּתוּב בִּשְׂאֹר וְסִיֵּם בְּחָמֵץ, שְׂאֹר לֹא יִמָּצֵא בְּבָתֵּיכֶם כִּי כָּל אוֹכֵל מַחֲמֶצֶת (שְׁמוֹת יב), לוֹמַר לְךָ זֶהוּ שְׂאֹר וְזֶהוּ חָמֵץ, כְּזֶה כֵּן זֶה. כִּי פְלִיגֵי לְעִנְיַן בִּעוּר, בֵּית שַׁמַּאי סָבְרֵי מִדְּכָתַב רַחֲמָנָא לְתַרְוַיְהוּ, שְׂאֹר וְחָמֵץ {י}, וְלִכְתּוּב חָמֵץ שֶׁאֵין חִמּוּצוֹ קָשֶׁה וַאֲפִלּוּ הָכִי חַיָּב לְבַעֵר וְקַל וָחֹמֶר לַשְּׂאֹר שֶׁחִמּוּצוֹ קָשֶׁה, אֶלָּא שְׁמַע מִנָּהּ שִׁעוּרוֹ שֶׁל זֶה לֹא כְשִׁעוּרוֹ שֶׁל זֶה, וְלֹא יָלְפִינַן בִּעוּר מֵאֲכִילָה. וּבֵית הִלֵּל סָבְרֵי זֶה וָזֶה בְּכַזַּיִת, דְּיָלְפִינַן בִּעוּר מֵאֲכִילָה. וּמִשּׁוּם דְּהָנָךְ תְּלַת מִלֵּי מִקֻלֵּי בֵּית שַׁמַּאי וּמֵחֻמְרֵי בֵּית הִלֵּל גַּבֵּי יוֹם טוֹב, תָּנָא לְהוּ הָכָא גַּבֵּי הֲדָדֵי: בְּכַכּוֹתֶבֶת. תְּמָרָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} בַּגְּמָרָא. וְאַף עַל גַּב דְּמַסְקִינַן דְּיוֹם טוֹב דְּעָלְמָא נַמִּי אֲסוּרָה, אֲפִלּוּ הָכִי מוֹקְמִינַן לְמַתְנִיתִין בְּעִקַּר הַדִּין. וְהָא דְנָקַט יוֹם טוֹב וְלֹא שַׁבָּת וְיִהְיֶה פֵּרוּשׁוֹ שַׁבָּת שֶׁאַחַר יוֹם טוֹב, דַּהֲוָה אֲמִינָא דַּוְקָא שַׁבָּת סְעוּדָתוֹ בָּעֲיָא הֲכָנָה וְאֵין יוֹם טוֹב מֵכִין לְשַׁבָּת, אֲבָל יוֹם טוֹב שֶׁאַחַר שַׁבָּת אֵימָא לֹא לִתְּסַר, דִּכְחֹל שַׁוְוִינְהוּ שֶׁאֵין סְעוּדָתוֹ בָּעֵי הֲכָנָה, הִלְכָּךְ אִיצְטְרִיךְ לְמִתְנֵי יוֹם טוֹב, וְשַׁבָּת אָתְיָא בְּמִכָּל שֶׁכֵּן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} גְּמָרָא. דְּאִלּוּ הַיּוֹם גָּמְרָה קַיְמָא לָן בְּעֵרוּבִין דְּשַׁבָּת מְכִינָה לְעַצְמָהּ וְהוּא הַדִּין יוֹם טוֹב. וְקָאָמַר גָּמְרָה, דְּאִלּוּ לֹא גָּמְרָה אַף עַל פִּי שֶׁנִתְגַּדֵּל לֹא מִיתַּסְרִי, דְּאִי הָכִי בְּיוֹם טוֹב אַחַר שַׁבָּת הֵיכִי שָׁחֲטִינַן, וַהֲלֹא כְּבָר נִתְגַּדְלוּ בְּשַׁבָּת, אֶלָּא דְּלֹא חָשְׁבוּ לַהּ הֲכָנָה אֶלָּא כִּי גָּמְרָה, שֶׁאָז מִתְחַדֶּשֶׁת בְּרִיָּה חֲדָשָׁה לְגַמְרֵי. הָרַ"ן: {ג} דְּעַל כָּרְחֲךָ בָּא לְאַשְׁמְעִינַן דַּאֲפִלּוּ הֲכָנָה בִּידֵי שָׁמַיִם כְּגוֹן בֵּיצָה אֲסוּרָה, דְּאִי מִשּׁוּם הֲכָנָה בִּידֵי אָדָם כְּלוֹמַר שֶׁיֹּאפוּ וִיבַשְּׁלוּ מִבְּעוֹד יוֹם, הָא בְּהֶדְיָא כְתִיב אֶת אֲשֶׁר תֹּאפוּ אֵפוּ. רַשִׁ"י: {ד} שֶׁכֵּן כְּתִיב בְּהוּ שַׁבָּתוֹן בְּפָרָשַׁת אֱמֹר, גַּם מִמָּחֳרָת הַשַּׁבָּת דְּעֹמֶר פֵּרְשׁוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה מִמָּחֳרָת יוֹם טוֹב: {ה} גְּמָרָא. דְּכֵיוָן דְּשָׁמְעִינַן דְּאֵין יוֹם טוֹב מֵכִין לְשַׁבָּת, יָלְפִינַן מִקַּל וָחֹמֶר דְּאֵין שַׁבָּת מְכִינָה לְיוֹם טוֹב. הָרַ"ן: {ו} וְאַף עַל גַּב דְּמֵאֶתְמוֹל הִיא נִגְמֶרֶת, מִכָּל מָקוֹם כְּשֶׁנּוֹלְדָה בְּשַׁבָּת הֲרֵי זוֹ הֲכָנָה, דְּגַם בִּשְׁעַת לֵידָה שַׁיָּךְ הֲכָנָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} שֶׁמִּתְּחִלָּה לֹא מֵחֲמַת סָפֵק הִתְחִילוּ לַעֲשׂוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} מַה שֶּׁאֵין כֵּן שַׁבָּת וְיוֹם טוֹב הַסְּמוּכִים, וְאַף עַל פִּי שֶׁהַיּוֹם טוֹב הוּא יוֹם טוֹב שֵׁנִי, דְּשַׁבָּת מִיהוּ וַדַּאי קֹדֶשׁ הוּא, וְיוֹם טוֹב הַסָּמוּךְ לָהּ נַמִּי שֶׁמָּא יוֹם טוֹב גָּמוּר הוּא. הָרַ"ן: {ט} לֹא תֵאָכֵל. וַאֲפִלּוּ לְטַלְטֵל נַמִּי לֹא כוּ', אֶלָּא מִשּׁוּם דִּסְתָמָא בֵּיצָה לָאֲכִילָה קַיְמָא, נָקַט כִּי הַאי לִישָׁנָא. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} גְּמָרָא. וּרְצוֹנוֹ לוֹמַר לֹא יֵרָאֶה לְךָ חָמֵץ וְלֹא יֵרָאֶה לְךָ שְׂאֹר [שְׁמוֹת יג] דְּכָתַב לָאו בְּכָל חֲדָא, מַה שֶּׁאֵין כֵּן בָּאֲכִילָה, דִּנְתִינַת טַעַם, כִּי כָּל אֹכֶל כְּתִיב [שָׁם יב]: {יא} בְּכַכּוֹתֶבֶת. דְּהוֹאִיל וְיָדַעְנוּ דְּשִׁעוּרוֹ שֶׁל זֶה לֹא כְּשִׁעוּרוֹ שֶׁל זֶה, נוֹקִים לֵיהּ שִׁעוּרָא כִּדְאַשְׁכְּחָנָא בְּעָלְמָא: {יב} בְּכַזָּיִת. וּקְרָאֵי צְרִיכֵי דְאִי כָּתַב שְׂאֹר הֲוָה אֲמִינָא מִשּׁוּם דְּחִמּוּצוֹ קָשֶׁה כוּ', וְאִי כָּתַב חָמֵץ מִשּׁוּם דְּרָאוּי לָאֲכִילָה. גְּמָרָא: (עיקר תוי"ט).

פרק א - משנה ב

הַשּׁוֹחֵט חַיָּה וָעוֹף בְּיוֹם טוֹב, בֵּית שַׁמַּאי אוֹמְרִים, יַחְפֹּר בְּדֶקֶר וִיכַסֶּה, וּבֵית הִלֵּל אוֹמְרִים, לֹא יִשְׁחֹט, אֶלָּא אִם כֵּן הָיָה לוֹ עָפָר מוּכָן מִבְּעוֹד יוֹם. וּמוֹדִים, שֶׁאִם שָׁחַט, שֶׁיַּחְפֹּר בְּדֶקֶר וִיכַסֶּה, שֶׁאֵפֶר כִּירָה מוּכָן הוּא:

 ר"ע מברטנורה  הַשּׁוֹחֵט חַיָּה וָעוֹף. הַבָּא לִשְׁחֹט חַיָּה וָעוֹף וְנִמְלַךְ בְּבֵית דִּין כֵּיצַד יַעֲשֶׂה: בֵּית שַׁמַּאי אוֹמְרִים. בֵּית דִּין מוֹרִין לוֹ שֶׁיִּשְׁחַט לְכַתְּחִלָּה וְיַחְפֹּר בְּדֶקֶר נָעוּץ שֶׁהֱכִינוֹ מִבְּעוֹד יוֹם, כְּלוֹמַר שֶׁיַּעֲקֹר אוֹתוֹ מִמְּקוֹם נְעִיצָתוֹ וְיַעֲלֶה עָפָר וִיכַסֶּה בּוֹ. וּמַיְרֵי כְּשֶׁהוּא נָעוּץ בְּעָפָר תִּחוֹחַ הָרָאוּי לְכִסּוּי, שֶׁאֵינוֹ מְחֻסָּר כְּתִישָׁה {יג}: דֶּקֶר. יָתֵד שֶׁנּוֹעֲצִים בָּאָרֶץ. לְשׁוֹן וַיִּדְקֹר אֶת שְׁנֵיהֶם (בְּמִדְבָּר כח): שֶׁאֵפֶר כִּירָה מוּכָן הוּא. לָאו אַמִלְּתַיְהוּ דְּבֵית שַׁמַּאי וּבֵית הִלֵּל קָאֵי, אֶלָּא מִלְּתָא בְּאַנְפֵּי נַפְשָׁהּ הִיא, וְהָכִי קָאָמַר וְאֵפֶר כִּירָה מוּכָן הוּא וְאֵינוֹ צָרִיךְ הַזְמָנָה, דְּדַעְתֵּיהּ עִלּוּיֵהּ. וְלֹא שָׁנוּ אֶלָּא שֶׁהֻסַּק מֵעֶרֶב יוֹם טוֹב, אֲבָל הֻסַּק בְּיוֹם טוֹב אָסוּר {יד}, דְּלֵיכָּא לְמֵימַר דַּעְתֵּיהּ עִלּוּיֵהּ מֵאֶתְמוֹל. וְאִם רָאוּי לִצְלוֹת בּוֹ בֵּיצָה, שֶׁעֲדַיִן הוּא רֶמֶץ חַם, אַף עַל פִּי שֶׁהֻסַּק בְּיוֹם טוֹב מֻתָּר לְכַסּוֹת בּוֹ, דְּאַיְדֵי דַּחֲזִי לְהִפּוּכֵי בֵּיהּ לִצְלוֹת בּוֹ בֵּיצָה, שָׁקֵיל לֵיהּ נַמִּי וּמְכַסֶּה בּוֹ: (ר"ע מברטנורה).

 עיקר תוי"ט  {יג} דִּכְתִישָׁה תּוֹלָדָה דְּטוֹחֵן הִיא. אֲבָל מַה שֶּׁנּוֹטֵל הֶעָפָר וְעוֹשֶׂה גֻמָּא, אַף עַל גַּב דְּקָיְימָא לָן חוֹרֵשׁ בְּכָל שֶׁהוּא כוּ', אָמְרִינַן בַּגְּמָרָא הָנֵי מִלֵּי הֵיכָא דִּמְתַקֵּן, הָכָא מְקַלְקֵל (וְשָׁרֵי לְכַתְּחִלָּה מִשּׁוּם שִׂמְחַת יוֹם טוֹב), כְּלוֹמַר דְּמַיְרֵי בְּמָקוֹם שֶׁהַגֻּמָּא מְקַלְקֵל כְּמוֹ בַּבַּיִת אוֹ בְּחָצֵר, אֲבָל בַּשָּׂדֶה לֹא: {יד} דִּבְיוֹם טוֹב קַיְמָא לָן כְּמַאן דְּאָמַר מֻקְצֶה אָסוּר. וְהַתּוֹסָפוֹת כָּתְבוּ דְּהָכָא לְכֻלֵּי עָלְמָא אָסוּר, דַּהֲוֵי נוֹלָד גָּמוּר, דְּמֵעִקָּרָא עֵצִים וְהַשְׁתָּא אֵפֶר: (עיקר תוי"ט).

פרק א - משנה ג

בֵּית שַׁמַּאי אוֹמְרִים, אֵין מוֹלִיכִין אֶת הַסֻּלָּם מִשּׁוֹבָךְ לְשׁוֹבָךְ, אֲבָל מַטֵּהוּ מֵחַלּוֹן לְחַלּוֹן. וּבֵית הִלֵּל מַתִּירִין. בֵּית שַׁמַּאי אוֹמְרִים, לֹא יִטֹּל, אֶלָּא אִם כֵּן נִעֲנֵעַ מִבְּעוֹד יוֹם. וּבֵית הִלֵּל אוֹמְרִים, עוֹמֵד וְאוֹמֵר {טז} זֶה וָזֶה אֲנִי נוֹטֵל:

 ר"ע מברטנורה  אֵין מוֹלִיכִין אֶת הַסֻּלָּם מִשּׁוֹבָךְ לְשׁוֹבָךְ. בְּסֻלָּם שֶׁל עֲלִיָּה כֻּלֵּי עָלְמָא לֹא פְלִיגֵי דְּאָסוּר, וּבְסֻלָּם שֶׁל שׁוֹבָךְ נַמִּי כְּשֶׁהוּא מוֹלִיכוֹ בִּרְשׁוּת הָרַבִּים מוֹדוּ בֵּית הִלֵּל דְּאָסוּר, שֶׁהָרוֹאֶה אוֹמֵר לְתַקֵּן גַּגּוֹ הוּא מוֹלִיכוֹ וְעוֹשֶׂה מְלָאכָה בְּיוֹם טוֹב. כִּי פְלִיגֵי שֶׁמּוֹלִיכוֹ בִּרְשׁוּת הַיָּחִיד וְאֵין בְּנֵי אָדָם רוֹאִים אוֹתוֹ. בֵּית שַׁמַּאי סָבְרֵי אֵין מוֹלִיכִין, שֶׁכָּל דָּבָר שֶׁאָסְרוּ חֲכָמִים מִפְּנֵי מַרְאִית הָעַיִן אֲפִלּוּ בְּחַדְרֵי חֲדָרִים אָסוּר. וּבֵית הִלֵּל מַתִּירִין, אַף עַל גַּב דִּבְעָלְמָא דִּינָא הָכִי הוּא, הָכָא שָׁרֵי מִשּׁוּם שִׂמְחַת יוֹם טוֹב {טו}: מֵחַלּוֹן לְחַלּוֹן. בְּאוֹתוֹ שׁוֹבָךְ עַצְמוֹ, שֶׁכָּךְ דֶּרֶךְ שׁוֹבָךְ לִהְיוֹת בּוֹ [מְחִצּוֹת הַרְבֵּה] קֵן לְכָל זוּג וְזוּג, וְחַלּוֹן לְכָל קֵן וָקֵן: זֶה וָזֶה אֲנִי נוֹטֵל. פְּלוּגְתַּיְהוּ דְּבֵית שַׁמַּאי וּבֵית הִלֵּל דַּוְקָא בִּבְרֵכָה רִאשׁוֹנָה. שֶׁדֶּרֶךְ מְגַדְּלֵי יוֹנִים תָּמִיד לְהַנִּיחַ בְּרֵכָה רִאשׁוֹנָה, דְּהַיְנוּ שְׁתֵּי גּוֹזָלוֹת רִאשׁוֹנוֹת, שֶׁהָאִמָּהוֹת מִשְׁתַּעְשְׁעוֹת בָּהֶן וְלֹא יִבְרְחוּ. וּבֵית שַׁמַּאי סָבְרֵי דִּבְדִבּוּר בְּעָלְמָא לֹא סַגִּי, דִּלְמָא כִּי שָׁקֵיל לְהוּ לְמָחָר אָתֵי לְאַמְלוּכֵי וְחָיִס עֲלַיְהוּ וְהָוֵי טִלְטוּל שֶׁלֹּא לְצֹרֶךְ, אֲבָל מִשֶּׁנִּעְנֵעַ וּמִשְׁמֵשׁ בָּהֶן מִבְּעוֹד יוֹם לִשְׁחִיטָה וְלֹא חָס עֲלַיְהוּ תּוּ לֹא חָיְשִׁינַן דְּחָיִס. וּבֵית הִלֵּל סָבְרֵי לֹא גָּזְרִינַן דִּלְמָא חָיִס עֲלַיְהוּ, וַאֲפִלּוּ בִּבְרֵכָה רִאשׁוֹנָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {טו} וְאַף עַל גַּב דִּלְעֵיל לֹא הִתִּירוּ אֶלָּא בְּדֶקֶר נָעוּץ (וַעֲפַר תִּחוֹחַ לֹא מַהֲנֵי מִשּׁוּם חֲשָׁדָא דַחֲפִירָה. רַשִׁ"י), וְהָכָא נַמִּי אִיכָּא חֲשָׁדָא דִּלְהָטִיחַ גַּגּוֹ, תֵּרְצוּ בַּגְּמָרָא דְהָכָא שׁוֹבְכוֹ מוֹכִיחַ עָלָיו. וּפֵרֵשׁ הָרַ"ן כְּלוֹמַר שׁוֹבְכוֹ שֶׁל סֻלָּם מוֹכִיח,ַ שֶׁהוּא נִכָּר שֶׁהוּא לְשֵׁם שׁוֹבָךְ: {טז} וְאוֹמֵר. מִבְּעוֹד יוֹם. רַשִׁ"י: (עיקר תוי"ט).

פרק א - משנה ד

זִמֵּן שְׁחוֹרִים וּמָצָא לְבָנִים, לְבָנִים וּמָצָא שְׁחוֹרִים, שְׁנַיִם וּמָצָא שְׁלֹשָׁה, אֲסוּרִים {יח}. שְׁלֹשָׁה וּמָצָא שְׁנַיִם, מֻתָּרִים. בְּתוֹךְ הַקֵּן וּמָצָא לִפְנֵי הַקֵּן, אֲסוּרִים. וְאִם אֵין שָׁם אֶלָּא הֵם, הֲרֵי אֵלּוּ מֻתָּרִים:

 ר"ע מברטנורה  זִמֵּן שְׁחוֹרִים וּמָצָא לְבָנִים. הָא מִלְּתָא דִּפְשִׁיטָא הוּא וְלֹא אִצְטְרִיךְ לְמַתְנֵיהּ. אֶלָּא הָכִי קָאָמַר, זִמֵּן שְׁחוֹרִים וּלְבָנִים וּמָצָא שְׁחוֹרִים בִּמְקוֹם לְבָנִים וּלְבָנִים בִּמְקוֹם שְׁחוֹרִים. וְהָכִי פֵּרוּשָׁא דְּמַתְנִיתִין, זִמֵּן שְׁחוֹרִים בְּקֵן זֶה וּמָצָא בּוֹ לְבָנִים, זִמֵּן בַּשֵּׁנִי לְבָנִים וּמָצָא בּוֹ שְׁחוֹרִים, מַהוּ דְּתֵימָא הָנֵי אִינְהוּ וְאִתְהֲפִיךְ אִתְהֲפוֹכֵי, קָא מַשְׁמַע לָן {יז}: שְׁנַיִם וּמָצָא שְׁלֹשָׁה. וְאֵינוֹ מַכִּיר הַמּוּכָנִין, כֻּלָּן אֲסוּרִים: שְׁלֹשָׁה וּמָצָא שְׁנַיִם מֻתָּרִים. הָאֶחָד הָלַךְ וְהַשְּׁנַיִם נִשְׁאֲרוּ. וְלֹא אָמְרִינַן כְּשֵׁם שֶׁהָאֶחָד הָלַךְ כֵּן הָלְכוּ כֻּלָּם וְהָנֵי אַחֲרִינֵי נִינְהוּ: זִמֵּן בְּתוֹךְ הַקֵּן וּמָצָא לִפְנֵי הַקֵּן. וּבְתוֹךְ הַקֵּן לֹא מָצָא כְּלוּם, אֲסוּרִים: וְאִם אֵין בְּאוֹתוֹ שׁוֹבָךְ אֶלָּא הֵן הֲרֵי אֵלּוּ מֻתָּרִין. וְאַף עַל פִּי שֶׁיֵּשׁ שׁוֹבָךְ אַחֵר קָרוֹב לוֹ בְּתוֹךְ חֲמִשִּׁים אַמָּה {יט}, אִם אֵין אוֹתוֹ שׁוֹבָךְ בְּשׁוּרָה שֶׁל שׁוֹבָךְ זֶה בְּשָׁוֶה אֶלָּא בְּקֶרֶן זָוִית {כ}, לֹא אָמְרִינַן מֵהַהוּא שׁוֹבָךְ הַקָּרוֹב אָתוּ. שֶׁהַגּוֹזָלוֹת כָּל זְמַן שֶׁאֵינָם פּוֹרְחִים אֶלָּא מְדַדִּין בְּרַגְלֵיהֶן אֵינָן מְדַדִּין אֶלָּא כְּנֶגֶד הַקֵּן שֶׁלָּהֶן בְּשָׁוֶה, שֶׁאִם רוֹאִין קִנֵּיהֶן כְּשֶׁהוֹפְכִים לַאֲחוֹרֵיהֶם מְדַדִּין, וְאִם לָאו אֵין מְדַדִּין: (ר"ע מברטנורה).

 עיקר תוי"ט  {יז} וְלֹא פֵּרֵשׁ הָרַ"ב טַעְמָא מַאי אָמְרִינַן הָכִי. וְלִכְאוֹרָה טַעֲמֵיהּ כְּמַאן דְּאָמַר רֹב וְקָרוֹב הוֹלְכִים אַחַר הָרֹב כוּ'. אֶלָּא שֶׁהָרַמְבַּ"ם תָּלָה הַדָּבָר בְּסָפֵק מוּכָן שֶׁאָסוּר, וְכָתַב הַמַּגִּיד הַטַּעַם מִפְּנֵי שֶׁהוּא דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין, אוֹ שֶׁרָצוּ חֲכָמִים לְהַחְמִיר בְּסָפֵק זֶה: {יח} אֲסוּרִים. דְּמִמָּה נַפְשָׁךְ אִיכָּא חַד דִּמְעֹרָב בְּהוּ, וְלֵיכָּא לְמִישְׁרִינְהוּ מִשּׁוּם דְּחַד בִּתְרֵי בָּטִיל (דִּכְתִיב אַחֲרֵי רַבִּים לְהַטּוֹת. רַשִׁ"י), דְּבַעֲלֵי חַיִּים חֲשִׁיבֵי וְלֹא בָּטְלֵי, וְעוֹד כָּל דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין אֲפִלּוּ בְּאֶלֶף לֹא בָטִיל. תּוֹסָפוֹת: {יט} דְּמַתְנִיתִין בִּמְדַדִּין שֶׁאֵין יְכוֹלִין לִפְרֹחַ, וְאִי לֵיכָּא קֵן בְּתוֹךְ חֲמִשִּׁים אַמָּה פְּשִׁיטָא דְמֻתָּרִין, דְּכָל הַמְּדַדֶּה אֵין מְדַדֶּה יוֹתֵר מֵחֲמִשִּׁים אַמָּה. גְּמָרָא: {כ} שֶׁזָּוִית הַבַּיִת בּוֹלֵט לְהָלְאָה מִן הַשּׁוֹבָךְ וְקֵן זֶה שֶׁזִּמֵּן בּוֹ עוֹמֵד לִצְדָדִין, דְּאִי הֲוָה אָתֵי לְהָכִי לֹא הֲוָה חֲזִי לְקִנֵּיהּ, שֶׁהַזָּוִית שֶׁל בַּיִת עוֹמֶדֶת לְפָנָיו. רַשִׁ"י: (עיקר תוי"ט).

פרק א - משנה ה

בֵּית שַׁמַּאי אוֹמְרִים, אֵין מְסַלְּקִין אֶת הַתְּרִיסִין בְּיוֹם טוֹב. וּבֵית הִלֵּל מַתִּירִין אַף לְהַחֲזִיר. בֵּית שַׁמַּאי אוֹמְרִים, אֵין נוֹטְלִין אֶת הָעֱלִי לְקַצֵּב עָלָיו בָּשָׂר. וּבֵית הִלֵּל מַתִּירִין. בֵּית שַׁמַּאי אוֹמְרִים, אֵין נוֹתְנִין אֶת הָעוֹר לִפְנֵי הַדּוֹרְסָן וְלֹא יַגְבִּיהֶנּוּ, אֶלָּא אִם כֵּן יֵשׁ עִמּוֹ כַזַּיִת בָּשָׂר. וּבֵית הִלֵּל מַתִּירִין. בֵּית שַׁמַּאי אוֹמְרִים, אֵין מוֹצִיאִין לֹא אֶת הַקָּטָן וְלֹא אֶת הַלּוּלָב וְלֹא אֶת סֵפֶר תּוֹרָה לִרְשׁוּת הָרַבִּים. וּבֵית הִלֵּל מַתִּירִין:

 ר"ע מברטנורה  תְּרִיסִין. מוֹכְרֵי בְשָׂמִים יֵשׁ לָהֵן חֲנֻיּוֹת הָעֲשׂוּיוֹת כְּמִין תֵּבוֹת, וְעוֹמְדוֹת בַּשּׁוּק, וְאֵינָן מְחֻבָּרוֹת לַקַּרְקַע. וּתְרִיסִין הֵן דְּלָתוֹת שֶׁסּוֹגְרִין בָּהֶן פִּתְחֵי אוֹתָן תֵּבוֹת, וּפְעָמִים שֶׁמְּסַלְּקִים אוֹתָן תְּרִיסִין מִפֶּתַח הַתֵּבָה וְשׁוֹטְחִין עֲלֵיהֶן הַסַּמְּמָנִין שֶׁבַּחֲנוּת: מַתִּירִין אַף לְהַחֲזִיר. בִּתְרִיסִין שֶׁאֵין לָהֶם צִיר כְּלָל כֻּלֵּי עָלְמָא לֹא פְלִיגֵי דִשְׁרִי אֲפִלּוּ לִנְעֹל בָּהֶן פֶּתַח [בַּיִת שֶׁבְּתוֹךְ] הֶחָצֵר. וּבִתְרִיסִין שֶׁיֵּשׁ לָהֵן צִיר מִן הַצָּד כֻּלֵּי עָלְמָא לֹא פְלִיגֵי דְּאָסוּר לִנְעֹל בָּהֶן אֲפִלּוּ פִּתְחֵי חֲנֻיּוֹת, דְּדָמֵי לְבִנְיָן {כא}. כִּי פְלִיגֵי בִּתְרִיסִין שֶׁיֵּשׁ לָהֵן צִיר בָּאֶמְצַע כְּעֵין בְּלִיטָה וְתוֹחֲבִין אוֹתָהּ בְּחֹר שֶׁבְּאֶמְצַע דֹּפֶן פֶּתַח הַחֲנוּת, בֵּית שַׁמַּאי סָבְרֵי גָּזְרִינַן צִיר בָּאֶמְצַע אַטּוּ צִיר מִן הַצָּד, וּבֵית הִלֵּל סָבְרֵי לֹא גָּזְרִינַן בִּתְרִיסֵי חֲנֻיּוֹת, וְשָׁרֵי לְהַחֲזִיר, מִשּׁוּם דְּבָעֵי לַאֲפוֹקֵי תַּבְלִין, וְאִי לֹא שָׁרֵית לֵיהּ לְאַהֲדוֹרֵי לֹא פָּתַח, וְאָתֵי לְאִמְּנוּעֵי מִשִּׂמְחַת יוֹם טוֹב: הָעֱלִי. יָד שֶׁל עֵץ גָּדוֹל הַרְבֵּה וְעָגֹל שֶׁכּוֹתְשִׁים בּוֹ אֶת הָרִיפוֹת וְדוֹמֵיהֶן. וּמִפְּנֵי כָּבְדּוֹ וְגָדְלוֹ אֵין תּוֹרַת כְּלִי עָלָיו: לְקַצֵּב. לַחֲתֹךְ: וּבֵית הִלֵּל מַתִּירִין. מִשּׁוּם שִׂמְחַת יוֹם טוֹב, וְאַף עַל פִּי שֶׁאֵין עָלָיו תּוֹרַת כְּלִי {כב}: לִפְנֵי בֵית הַדְּרִיסָה. שֶׁיִּדְרְסוּ עָלָיו בְּנֵי אָדָם, כְּדֵי שֶׁלֹּא יַפְסִיד: וְלֹא יַגְבִּיהֶנּוּ. לְטַלְטְלוֹ לְאַחַר הַפְשָׁטָתוֹ: וּבֵית הִלֵּל מַתִּירִין. דְּאִי לֹא שָׁרֵית לֵיהּ לֹא שָׁחֵיט כְּדֵי שֶׁלֹּא לְהַפְסִיד הָעוֹר, וּמִמְנַע מִשִּׂמְחַת יוֹם טוֹב: לֹא אֶת הַקָּטָן. וְלֹא כָּל מִידֵי דְּלָאו צֹרֶךְ אֲכִילָה: וּבֵית הִלֵּל מַתִּירִין. דְּאָמְרֵי מִתּוֹךְ {כג} שֶׁהֻתְּרָה הוֹצָאָה לְצֹרֶךְ אֲכִילָה, הֻתְּרָה נַמִּי שֶׁלֹּא לְצֹרֶךְ אֲכִילָה. וְהוּא שֶׁיִּהְיֶה בָּהּ צֹרֶךְ מִצְוָה {כד} כְּגוֹן הָנֵי דַחֲשִׁיבֵי בְּמַתְנִיתִין, אוֹ צֹרֶךְ הֲנָאָה, כְּגוֹן מִפֶּתַח בֵּיתוֹ {כה} וְכַיּוֹצֵא בָּהֶן. לַאֲפוֹקֵי הוֹצָאַת אֲבָנִים וְכַיּוֹצֵא בָּזֶה, דִּבְהָא מוֹדוּ בֵּית הִלֵּל דְּאָסוּר: (ר"ע מברטנורה).

 עיקר תוי"ט  {כא} וּמִיהוּ בִּנְיָן מַמָּשׁ לֹא הָוֵי, דְּאֵין בִּנְיָן בְּכֵלִים, וְקָאָסְרֵי לֵיהּ בֵּית הִלֵּל מִדְּרַבָּנָן. וּמִן הַצַּד הַיְנוּ יָדוֹת לִשְׁנֵי רָאשֵׁי הַדַּף. רַשִׁ"י: {כב} וּבַגְּמָרָא רָמֵי אַמַּתְנִיתִין ב' דְּלֹא שָׁרֵי בֵּית הִלֵּל מִשּׁוּם שִׂמְחַת יוֹם טוֹב אֶלָּא בְּדֶקֶר נָעוּץ, וַאֲפִלּוּ הָכִי לְכַתְּחִלָּה לֹא. וּמְשָׁנֵי טַעֲמַיְיהוּ דְּבֵית הִלֵּל מִשּׁוּם דְּיֵשׁ תּוֹרַת כְּלִי עָלָיו, וּסְבִירָא לֵיהּ דְּאַף עַל פִּי שֶׁמְּלַאכְתּוֹ לְאִסּוּר מֻתָּר לְטַלְטְלוֹ לְצֹרֶךְ גּוּפוֹ כוּ' (אִי נַמִּי מֻחְלֶפֶת הַשִּׁיטָה דְהָכָא. רַשִׁ"י) וְזֶה הֲוָה לֵיהּ לְהָרַ"ב לִכְתֹּב דְּלֹא נִטְעֶה בַּהֲלָכָה כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} דְּאַךְ אֲשֶׁר יֵאָכֵל לְכָל נֶפֶשׁ כְּתִיב [שְׁמוֹת יב], וְהַהוֹצָאָה אַף הִיא בְּאֹכֶל וּבְמַשְׁקֶה, וְאַף עַל פִּי שֶׁהוּא בִּדְבָרִים אֲחֵרִים אָמְרִינַן מִתּוֹךְ. הַמַּגִּיד: {כד} דְּמִדְּאוֹרַיְתָא הֻתְּרָה לְגַמְרֵי, אֶלָּא רַבָּנָן הוּא דְּגָזוּר בְּמִידֵי דְּהִיא טִרְחָא שֶׁלֹּא לְצֹרֶךְ, וּבְכֻלָּהּ לֹא בָּעֵי לְמִגְּזַר, מִשּׁוּם דְּאֵין גּוֹזְרִין גְּזֵרָה עַל הַצִּבּוּר אֶלָּא אִם כֵּן רֹב הַצִּבּוּר יְכוֹלִין לַעֲמֹד בָּהּ. רַשִׁ"י: {כה} יֵשׁ לְהִסְתַּפֵּק אִי מַיְרֵי דַּוְקָא בְּיֵשׁ אֳכָלִין בַּבַּיִת, אִי נַמִּי כֵּלִים שֶׁיֵּשׁ לַיּוֹם צֹרֶךְ בָּהֶן, אוֹ דְּכֵיוָן דְּלִבּוֹ דוֹאֵג מִשּׁוּם גְּנֵבָה חָשֵׁיב צֹרֶךְ קְצָת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק א - משנה ו

בֵּית שַׁמַּאי אוֹמְרִים, אֵין מוֹלִיכִין חַלָּה וּמַתָּנוֹת לַכֹּהֵן בְּיוֹם טוֹב, בֵּין שֶׁהוּרְמוּ מֵאֶמֶשׁ, בֵּין שֶׁהוּרְמוּ מֵהַיּוֹם. וּבֵית הִלֵּל מַתִּירִין. אָמְרוּ לָהֶם בֵּית שַׁמַּאי, גְּזֵרָה שָׁוָה, חַלָּה וּמַתָּנוֹת מַתָּנָה לַכֹּהֵן, וּתְרוּמָה מַתָּנָה לַכֹּהֵן, כְּשֵׁם שֶׁאֵין מוֹלִיכִין אֶת הַתְּרוּמָה, כָּךְ אֵין מוֹלִיכִין אֶת הַמַּתָּנוֹת. אָמְרוּ לָהֶם בֵּית הִלֵּל, לֹא, אִם אֲמַרְתֶּם בַּתְּרוּמָה, שֶׁאֵינוֹ זַכַּאי בַּהֲרָמָתָהּ, תֹּאמְרוּ בַמַּתָּנוֹת, שֶׁזַּכַּאי בַּהֲרָמָתָן:

 ר"ע מברטנורה  אֵין מוֹלִיכִין חַלָּה. בְּיוֹם טוֹב. אַף עַל פִּי שֶׁמֻּתָּר בְּהַפְרָשָׁתָהּ לֹא שָׁרוּ לֵיהּ בְּהוֹלָכָתָהּ, אֶלָּא תִּקּוּן עִסָּתוֹ הִתִּירוּ לוֹ {כו} וְלֹא יוֹתֵר: מַתָּנוֹת. הַזְּרוֹעַ וְהַלְּחָיַיִם וְהַקֵּבָה: גְּזֵרָה שָׁוָה. לָאו דַּוְקָא, דְּכֻלָּהּ מִדְּרַבָּנָן גָּזְרוּ מִשּׁוּם גְּזֵרָה דְּאֵין מַגְבִּיהִין תְּרוּמוֹת וּמַעַשְׂרוֹת בְּיוֹם טוֹב. אֶלָּא דּוּמְיָא דִּגְזֵרָה שָׁוָה: מַתָּנָה לַכֹּהֵן. מֵעֶשְׂרִים וְאַרְבַּע מַתְּנוֹת כְּהוּנָה: כְּשֵׁם שֶׁאֵין מוֹלִיכִין אֶת הַתְּרוּמָה וְכוּ'. הַךְ מַתְנִיתִין אִדַּחְיָא לַהּ בַּגְּמָרָא, דְּלֹא נֶחְלְקוּ בֵּית שַׁמַּאי וּבֵית הִלֵּל שֶׁמּוֹלִיכִין חַלָּה וּמַתָּנוֹת. לֹא נֶחְלְקוּ אֶלָּא בְּהוֹלָכַת תְּרוּמָה, שֶׁבֵּית שַׁמַּאי אוֹמְרִים אֵין מוֹלִיכִין, וּבֵית הִלֵּל אוֹמְרִים מוֹלִיכִין. אָמְרוּ לָהֶן בֵּית הִלֵּל לְבֵית שַׁמַּאי, חַלָּה וּמַתָּנוֹת מַתָּנָה לַכֹּהֵן, וּתְרוּמָה מַתָּנָה לַכֹּהֵן, כְּשֵׁם שֶׁמּוֹלִיכִין אֶת הַמַּתָּנוֹת כָּךְ מוֹלִיכִין אֶת הַתְּרוּמָה. אָמְרוּ לָהֶם בֵּית שַׁמַּאי, לֹא, אִם אֲמַרְתֶּם בְּחַלָּה וּמַתָּנוֹת שֶׁזַּכַּאי בַּהֲרָמָתָן, שֶׁנָּתְנוּ חֲכָמִים כֹּחַ וּרְשׁוּת בְּהַפְרָשָׁתָן, לְפִי שֶׁהַחִיּוּב שֶׁלָּהֶן בָּא בְּיוֹם טוֹב דְּלָשִׁין וְשׁוֹחֲטִין בְּיוֹם טוֹב, וְכֵיוָן שֶׁהֻתְּרוּ אוֹתָן שֶׁל יוֹם טוֹב שָׁרִינַן לְהוּ לְהוֹלִיךְ אֲפִלּוּ אוֹתָן שֶׁנַּעֲשׂוּ מֵעֶרֶב יוֹם טוֹב, תֹּאמְרוּ בִּתְרוּמָה שֶׁאֵינוֹ זַכַּאי בַּהֲרָמָתָהּ, שֶׁהֲרֵי אִי אֶפְשָׁר שֶׁיָּבֹא חִיּוּב תְּרוּמָה לַכְּרִי בְּיוֹם טוֹב, לְפִי שֶׁאֵין חִיּוּב תְּרוּמָה אֶלָּא מִשְּׁעַת מְרִיחָה בִּכְרִי, וְאֵין מְמָרְחִין בְּיוֹם טוֹב {כז}, הִלְכָּךְ אֵין מוֹלִיכִין תְּרוּמָה בְּיוֹם טוֹב: (ר"ע מברטנורה).

 עיקר תוי"ט  {כו} וּמַתָּנוֹת אַף הֵן הֲרָמָתָן הֻתְּרוּ לוֹ, וְאַף עַל פִּי שֶׁאֵין טוֹבְלִין כִּשְׁאָר טֶבֶל לֶאֱסֹר הַבָּשָׂר, וּכְמוֹ שֶׁכָּתַב הָרַ"ב בְּמִשְׁנָה ג' פֶּרֶק י' דְּחֻלִּין. מִיהוּ אִם לֹא פֵּרַשׁ שְׁאָר הַבָּשָׂר מֵהֶן הֲרֵי הוּא אוֹכְלָן וְעוֹבֵר עֲשֵׂה. רַשִׁ"י: {כז} בַּתְּבוּאָה שַׁיָּךְ מֵרוּחַ, וּשְׁאֵרָהּ עַיֵּן פֶּרֶק קַמָּא דְּמַעַשְׂרוֹת אֵימָתַי גָּרְנָן, אַף עַל פִּי דִּבְיֶרֶק מִלּוּי כְּלִי סַגִּי כְּדִתְנַן הָתָם בְּמִשְׁנָה ה', יֶרֶק דְּחִיּוּב תְּרוּמָתוֹ מִדְּרַבָּנָן לֹא אָירְיָא. תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק א - משנה ז

בֵּית שַׁמַּאי אוֹמְרִים, תְּבָלִין נִדּוֹכִין בְּמָדוֹךְ שֶׁל עֵץ, וְהַמֶּלַח {כח} בְּפַךְ, וּבְעֵץ הַפָּרוּר. וּבֵית הִלֵּל אוֹמְרִים, תְּבָלִין נִדּוֹכִין כְּדַרְכָּן בְּמָדוֹךְ שֶׁל אֶבֶן, וְהַמֶּלַח בְּמָדוֹךְ שֶׁל עֵץ:

 ר"ע מברטנורה  בְּמָדוֹךְ שֶׁל עֵץ. אֲבָל לֹא שֶׁל אֶבֶן. וּמִיהוּ בְּמָדוֹךְ שֶׁל עֵץ נִדּוֹכִין כְּדַרְכָּן וְאֵין צָרִיךְ שִׁנּוּי, מִפְּנֵי שֶׁהֵן מְפִיגִין טַעֲמָן אִם נִדּוֹכִין מֵאֶתְמוֹל: וְהַמֶּלַח בְּפַךְ. שֶׁל חֶרֶס יְדוּכֶנּוּ: אוֹ בְּעֵץ פָּרוּר. הוּא הַכַּף שֶׁמַּגִּיסִים בּוֹ הַקְּדֵרָה. דְּמֶלַח בָּעֵי שִׁנּוּי, שֶׁהָיָה לוֹ לָדוּךְ מֵאֶתְמוֹל, שֶׁאֵינוֹ מֵפִיג טַעֲמוֹ. וַהֲלָכָה שֶׁהַבָּא לָדוּךְ מֶלַח בְּיוֹם טוֹב מַטֶּה הַמְּדוֹכָה עַל צִדָּהּ וְדַךְ עַל יְדֵי שִׁנּוּי. וְהַתַּבְלִין נִדּוֹכִין כְּדַרְכָּן וְאֵין צְרִיכִין שִׁנּוּי: (ר"ע מברטנורה).

 עיקר תוי"ט  {כח} וְהַמֶּלַח. בְּמָקוֹם שֶׁיֵּשׁ מֶלַח גַּסָּה. רַשִׁ"י: (עיקר תוי"ט).

פרק א - משנה ח

הַבּוֹרֵר קִטְנִית בְּיוֹם טוֹב, בֵּית שַׁמַּאי אוֹמְרִים, בּוֹרֵר אֹכֶל וְאוֹכֵל. וּבֵית הִלֵּל אוֹמְרִים, בּוֹרֵר כְּדַרְכּוֹ בְּחֵיקוֹ, בְּקָנוֹן וּבְתַמְחוּי, אֲבָל לֹא בְטַבְלָא {כט} וְלֹא בְנָפָה וְלֹא בִכְבָרָה. רַבָּן גַּמְלִיאֵל אוֹמֵר, אַף מֵדִיחַ וְשׁוֹלֶה:

 ר"ע מברטנורה  בּוֹרֵר אֹכֶל וְאוֹכֵל. מְלַקֵּט הָאֹכֶל אֶחָד אֶחָד מִן הַפְּסֹלֶת: בּוֹרֵר כְּדַרְכּוֹ בְּחֵיקוֹ. וּבִלְבַד שֶׁלֹּא יִהְיֶה טֹרַח בְּרֵרַת הַפְּסֹלֶת מְרֻבֶּה. אֲבָל אִם טֹרַח בְּרֵרַת הַפְּסֹלֶת מְרֻבֶּה עַל טֹרַח בְּרֵרַת הָאֹכֶל, כְּגוֹן שֶׁהַפְּסֹלֶת דַּק בְּיוֹתֵר, אַף עַל פִּי שֶׁהָאֹכֶל מְרֻבֶּה בְּשִׁעוּרוֹ עַל הַפְּסֹלֶת, מוֹדִים בֵּית הִלֵּל שֶׁבּוֹרֵר אֶת הָאֹכֶל וּמַנִּיחַ אֶת הַפְּסֹלֶת, דִּלְמִעוּטֵי בְּטִרְחָא עָדִיף: אַף מֵדִיחַ וְשׁוֹלֶה. שֶׁמֵּבִיא כְּלִי מָלֵא קִטְנִית וְנוֹתֵן עָלָיו מַיִם, וְהַפְּסֹלֶת צָף עַל פְּנֵי הַמַּיִם וּמֵסִיר אוֹתוֹ בְּיָדוֹ. שׁוֹלֶה, כְּמוֹ שַׁל נְעָלֶיךָ (שְׁמוֹת ג). וְאֵין הֲלָכָה כְּרַבָּן גַּמְלִיאֵל: (ר"ע מברטנורה).

 עיקר תוי"ט  {כט} בְּטַבְלָא. דְּמֵחֲזֵי כְּמַאן דְּעָבֵיד לְצֹרֶךְ מָחָר, שֶׁאֵין דֶּרֶךְ לַעֲשׂוֹת בַּכֵּלִים הַלָּלוּ אֶלָּא הַרְבֵּה. רַשִׁ"י: (עיקר תוי"ט).

פרק א - משנה ט

בֵּית שַׁמַּאי אוֹמְרִים, אֵין מְשַׁלְּחִין בְּיוֹם טוֹב אֶלָּא מָנוֹת. וּבֵית הִלֵּל אוֹמְרִים, מְשַׁלְּחִין בְּהֵמָה חַיָּה וָעוֹף, בֵּין חַיִּין בֵּין שְׁחוּטִין. מְשַׁלְּחִין יֵינוֹת שְׁמָנִים וּסְלָתוֹת וְקִטְנִיּוֹת, אֲבָל לֹא תְבוּאָה, וְרַבִּי שִׁמְעוֹן מַתִּיר בִּתְבוּאָה:

 ר"ע מברטנורה  אֵין מְשַׁלְּחִין בְּיוֹם טוֹב. דּוֹרוֹן אִישׁ לְרֵעֵהוּ: אֶלָּא מָנוֹת. דָּבָר מוּכָן שֶׁאֵינוֹ עָשׂוּי לְהַנִּיחוֹ לְמָחָר, כְּגוֹן חֲתִיכוֹת בָּשָׂר וְדָגִים: וּבֵית הִלֵּל אוֹמְרִים כוּ'. וְלֹא שָׁרוּ בֵּית שַׁמַּאי אֶלָּא לִשְׁלֹחַ עַל יְדֵי אָדָם אֶחָד אוֹ שְׁנַיִם, אֲבָל שְׁלֹשָׁה בְּנֵי אָדָם אוֹ יוֹתֵר נוֹשְׂאֵי מִנְחָה אָסוּר, דְּאַוְשָׁא מִלְּתָא טוּבָא וְנִרְאִין כְּמוֹלִיכִין לִמְכֹּר בַּשּׁוּק: אֲבָל לֹא תְבוּאָה. שֶׁאֵינָהּ רְאוּיָה לַאֲכִילָה, שֶׁהֲרֵי צְרִיכָה טְחִינָה, וְאֵין טוֹחֲנִין בְּיוֹם טוֹב {ל}: וְרַבִּי שִׁמְעוֹן מַתִּיר בִּתְבוּאָה. שֶׁיָּכוֹל לְכָתְשָׁהּ בְּמַכְתֶּשֶׁת קְטַנָּה {לא} וִיבַשְּׁלָהּ מַעֲשֵׂה קְדֵרָה. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: (ר"ע מברטנורה).

 עיקר תוי"ט  {ל} וּמְסַיְּמֵי רַשִׁ"י וְהָרַ"ן שֶׁהֲוָה לוֹ לִטְחֹן מֵאֶתְמוֹל וְלֹא תָפִיג טַעֲמָהּ. וְקַשְׁיָא לִי אִם כֵּן לִשְׁתְּרֵי בְּהַטָּיָה מִידֵי דַּהֲוָה אַשְּׁחִיקַת מֶלַח, דְּוַדַּאי שֶׁהִיא תּוֹלָדָה דְּטוֹחֵן כְּכָל שְׁחִיקַת תַּבְלִין וְסַמְמָנִים שֶׁהֵם תּוֹלָדָה דְּטוֹחֵן. וְהָרַ"ן בְּפֶרֶק ג' נָתַן טַעַם אַחֵר וְאָמַר אָסְרוּ חֲכָמִים טְחִינָה לְפִי שֶׁהוּא מְלָאכָה הַנַּעֲשֵׂית לְיָמִים הַרְבֵּה. וּשְׁחִיקַת פִּלְפְּלִין אַף עַל פִּי שֶׁהִיא תּוֹלְדָתָהּ, אֵין אוֹתָהּ שְׁחִיקָה נַעֲשֵׂית בְּאוֹתוֹ עִנְיָן שֶׁהִיא נַעֲשֵׂית לְיָמִים הַרְבֵּה, שֶׁזּוֹ בְּמָדוֹךְ וְזוֹ בָּרֵחַיִם, וְאַף פִּלְפְּלִין בָּרֵחַיִם שֶׁלָּהֶם אֵינָהּ כְּדֶרֶךְ טְחִינַת חִטִּים לְיָמִים הַרְבֵּה. וְכָתַב הָרֹא"שׁ דְּדָמֵי לְעֻבְדָּא דְּחֹל: {לא} וְתַנָּא קַמָּא אָזֵיל בָּתַר רֻבָּא. הַמַּגִּיד: (עיקר תוי"ט).

פרק א - משנה י

מְשַׁלְּחִין כֵּלִים, בֵּין תְּפוּרִין בֵּין שֶׁאֵינָן תְּפוּרִין, וְאַף עַל פִּי שֶׁיֵּשׁ בָּהֶן כִּלְאַיִם, וְהֵן לְצֹרֶךְ הַמּוֹעֵד, אֲבָל לֹא סַנְדָּל הַמְסֻמָּר וְלֹא מִנְעָל שֶׁאֵינוֹ תָפוּר. רַבִּי יְהוּדָה אוֹמֵר, אַף לֹא מִנְעָל לָבָן, מִפְּנֵי שֶׁצָּרִיךְ אֻמָּן. זֶה הַכְּלָל, כֹּל שֶׁנֵּאוֹתִין בּוֹ, בְּיוֹם טוֹב מְשַׁלְּחִין אוֹתוֹ:

 ר"ע מברטנורה  מְשַׁלְּחִין כֵּלִים תְּפוּרִין. דַּחֲזוּ לְמַלְבּוּשׁ. וְשֶׁאֵינָן תְּפוּרִין חֲזוּ לְאַכְסוּיֵי: כִּלְאַיִם. כְּשֶׁהֵן קַשִּׁין שֶׁאֵין מְחַמְּמִין מֻתָּר לִשְׁכַּב עֲלֵיהֶם: סַנְדָּל מְסֻמָּר. שֶׁל עֵץ מְחֻפֶּה מַסְמְרִים. שֶׁגָּזְרוּ חֲכָמִים עָלָיו שֶׁלֹּא לְנַעֲלוֹ בְּשַׁבָּת וְיוֹם טוֹב מִשּׁוּם מַעֲשֶׂה שֶׁהָיָה שֶׁנֶּהֶרְגוּ הֲרוּגֵי מַלְכוּת עַל יָדוֹ, כַּמְּבֹאָר בְּמַסֶּכֶת שַׁבָּת פֶּרֶק בַּמֶּה אִשָּׁה יוֹצְאָה [דַּף ס]: וְלֹא מִנְעָל שֶׁאֵינוֹ תָפוּר. וַאֲפִלּוּ הוּא מְחֻבָּר בִּיתֵדוֹת שֶׁל עֵץ וְכַיּוֹצֵא בָּהֶן: אַף לֹא מִנְעָל לָבָן. בְּאַתְרֵיהּ דְּרַבִּי יְהוּדָה לֹא הָיוּ נוֹעֲלִים מִנְעָל לָבָן עַד שֶׁמַּשְׁחִירִים אוֹתוֹ: שֶׁצָּרִיךְ אֻמָּן. לְהַשְׁחִירוֹ: כָּל שֶׁנֵּאוֹתִין בּוֹ. הָכִי קָאָמַר, כָּל שֶׁנְּאוֹתִין בּוֹ בַּחֹל כְּמוֹת שֶׁהוּא וְאֵינוֹ צָרִיךְ לִמְלָאכָה אַחֶרֶת, אַף עַל פִּי שֶׁאֵין נֵאוֹתִין בּוֹ בְּיוֹם טוֹב, כְּגוֹן תְּפִלִּין {לב}, שֶׁבְּחֹל לוֹבְשִׁין אוֹתָן כְּמוֹת שֶׁהֵן וּבְיוֹם טוֹב אֵין לוֹבְשִׁין אוֹתָן, מְשַׁלְּחִין אוֹתָן בְּיוֹם טוֹב: (ר"ע מברטנורה).

 עיקר תוי"ט  {לב} כֵּיוָן דְּאִיכָּא שִׂמְחַת יוֹם טוֹב בְּשִׁלּוּחָן שָׁרֵי. בֵּית יוֹסֵף. אֲבָל סַנְדָּל אָסוּר, הוֹאִיל וּבָאת תַּקָּלָה עַל יָדוֹ. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ב - משנה א

יוֹם טוֹב שֶׁחָל לִהְיוֹת עֶרֶב שַׁבָּת, לֹא יְבַשֵּׁל אָדָם בַּתְּחִלָּה מִיּוֹם טוֹב לַשַּׁבָּת, אֲבָל מְבַשֵּׁל הוּא לְיוֹם טוֹב, וְאִם הוֹתִיר, הוֹתִיר לַשַּׁבָּת, וְעוֹשֶׂה תַבְשִׁיל מֵעֶרֶב יוֹם טוֹב וְסוֹמֵךְ עָלָיו לַשַּׁבָּת. בֵּית שַׁמַּאי אוֹמְרִים, שְׁנֵי תַבְשִׁילִין. וּבֵית הִלֵּל אוֹמְרִים, תַּבְשִׁיל אֶחָד. וְשָׁוִין בְּדָג וּבֵיצָה {ג} שֶׁעָלָיו שֶׁהֵן שְׁנֵי תַבְשִׁילִין. אֲכָלוֹ אוֹ שֶׁאָבַד, לֹא {ה} יְבַשֵּׁל עָלָיו בַּתְּחִלָּה. וְאִם שִׁיֵּר מִמֶּנּוּ כָל שֶׁהוּא, סוֹמֵךְ עָלָיו לַשַּׁבָּת:

 ר"ע מברטנורה  יוֹם טוֹב שֶׁחָל לִהְיוֹת עֶרֶב שַׁבָּת לֹא יְבַשֵּׁל בַּתְּחִלָּה. כְּלוֹמַר לֹא יְהֵא תְּחִלַּת בִּשּׁוּלוֹ וְעִקָּרוֹ לְשׁוּם שַׁבָּת, אֶלָּא לְשׁוּם יוֹם טוֹב יְהֵא תְּחִלַּת בִּשּׁוּלוֹ, וּמַה שֶּׁמּוֹתִיר יְהֵא לְשַׁבָּת. וְעוֹשֶׂה תַבְשִׁיל מֵעֶרֶב יוֹם טוֹב. לְשֵׁם עֵרוּבֵי תַּבְשִׁילִין {א}. אִית דְּאָמְרֵי טַעֲמָא דְּהַאי תַּבְשִׁיל מֵעֶרֶב יוֹם טוֹב, מִשּׁוּם כְּבוֹד שַׁבָּת הוּא, שֶׁמִּתּוֹךְ שֶׁמֵּעֶרֶב יוֹם טוֹב זוֹכֵר הַשַּׁבָּת, יִבְרֹר מָנָה יָפָה לְשַׁבָּת {ב} וְלֹא יִשְׁכַּח שַׁבָּת מִשּׁוּם טִרְדַּת יוֹם טוֹב. וְאִית דְּאָמְרֵי מִשּׁוּם כְּבוֹד יוֹם טוֹב הוּא, כְּדֵי שֶׁיִּרְאוּ שֶׁאֵין אוֹפִין מִיּוֹם טוֹב לְשַׁבָּת אֶלָּא אִם כֵּן הִתְחִיל מִבְּעוֹד יוֹם, דִּבְיוֹם טוֹב אֵינוֹ אֶלָּא גּוֹמֵר, קַל וָחֹמֶר מִיּוֹם טוֹב לְחֹל דִּלְגַמְרֵי אֵין אוֹפִין וְלֹא מְבַשְּׁלִין. וּמְבָרְכִים עַל עֵרוּבֵי תַּבְשִׁילִין כְּדֶרֶךְ שֶׁמְּבָרְכִים עַל עֵרוּבֵי חֲצֵרוֹת. וְצָרִיךְ לְזַכּוֹת עַל יְדֵי אַחֵר לְכָל מִי שֶׁרוֹצֶה אַחַר כָּךְ לִסְמֹךְ עַל עֵרוּבוֹ, וַאֲפִלּוּ שֶׁלֹּא מִדַּעְתּוֹ, דְּזָכִין לָאָדָם שֶׁלֹּא בְּפָנָיו: אֲכָלוֹ אוֹ שֶׁאָבַד. אוֹתוֹ תַּבְשִׁיל שֶׁבִּשֵּׁל, מֵעֶרֶב שַׁבָּת {ד}: סוֹמֵךְ עָלָיו. לְבַשֵּׁל בִּשּׁוּלוֹ בְּיוֹם טוֹב לְצֹרֶךְ שַׁבָּת. אֲבָל מִתְּחִלָּתוֹ אֵין עֵרוּבֵי תַּבְשִׁילִין פָּחוֹת מִכַּזַּיִת {ו}, בֵּין לְאֶחָד בֵּין לְמֵאָה. וְאִם הִתְחִיל בְּעִסָּתוֹ וְנֶאֱבַד עֵרוּבוֹ, מַה שֶּׁהִתְחִיל גּוֹמֵר: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} וְנִקְרָא שְׁמוֹ עֵרוּב, שֶׁהוּא מְעָרֵב צָרְכֵי שַׁבָּת עַל צָרְכֵי יוֹם טוֹב לַעֲשׂוֹתָן יַחַד. הָרַאֲבַ"ד: {ב} וַאֲפִלּוּ בְּיוֹם טוֹב עַצְמוֹ הָיָה יָכוֹל לַעֲשׂוֹת הָעֵרוּב לִפְנֵי סְעוּדַת יוֹם טוֹב, אֶלָּא גְּזֵרָה שֶׁמָּא יִפְשַׁע וְיִשְׁכַּח וְלֹא יְעָרֵב מִשּׁוּם דְּטָרִיד. גְּמָרָא. וְלִשְׁנֵי הַפֵּרוּשִׁים שֶׁכָּתַב הָרַ"ב אֵינוֹ אֶלָּא מִדִּבְרֵי סוֹפְרִים, וְאַף עַל גַּב דַּהֲכָנָה דְּאוֹרַיְתָא, מֻתָּר לְבַשֵּׁל מִיּוֹם טוֹב לְשַׁבָּת, מִשּׁוּם הוֹאִיל וְאִי מִקַּלְעֵי אוֹרְחִים. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} בְּדָג וּבֵיצָה. שֶׁהָיוּ טָחִין בֵּיצָה עַל הַדָּג כְּשֶׁצּוֹלִין אוֹתוֹ. רַשִׁ"י: {ד} וְזֶה לְשׁוֹן רַשִׁ"י, אוֹ שֶׁאָבַד קֹדֶם שֶׁבִּשֵּׁל כָּל צָרְכֵי שַׁבָּת. אִם כֵּן יֵשׁ לְפָרֵשׁ דִּבְרֵי הָרַ"ב דְּהָכִי קָאָמַר, מֵעֶרֶב שַׁבָּת קֹדֶם שֶׁבִּשֵּׁל כוּ', וְאֵין צָרִיךְ לְהַגִּיהַּ מֵעֶרֶב יוֹם טוֹב: {ה} לֹא כוּ'. וּלְפֵרוּשׁ רִאשׁוֹן דְּטַעְמָא דְּעֵרוּב תַּבְשִׁילִין מִשּׁוּם כְּבוֹד שַׁבָּת, אֲפִלּוּ הָכִי מוּטָב שֶׁלֹּא יִתְעַדֵּן שַׁבָּת אַחַת וְיִתְעַדֵּן שַׁבָּתוֹת הַרְבֵּה. שֶׁאִם אַתָּה מַתִּיר לוֹ לְבַשֵּׁל כְּשֶׁנֶּאֱבַד עֵרוּבוֹ אַף הוּא יִתְרַשֵּׁל בּוֹ וְלֹא יִשְׁמְרֶנּוּ אוֹ יֵאָבֵד אוֹ יֵאָכֵל וְלֹא יִזְכֹּר לְשַׁבָּת כְּלָל. הָרַ"ן: {ו} אֶלָּא דְּבַגְּמָרָא גָּרְסִינַן מַאי כָּל שֶׁהוּא, דְּאִית בֵּיהּ כַּזַּיִת, וּפֵרֵשׁ רַשִׁ"י וּלְגַבֵּי כִּכָּר שָׁלֵם קָרֵי לֵיהּ כָּל שֶׁהוּא. וְכֵן הָרַמְבַּ"ם סָתַם וְלֹא חִלֵּק בֵּין תְּחִלָּתוֹ לְסוֹפוֹ, וְכֵן כָּתַב הַטּוּר. וְצָרִיךְ עִיּוּן עַל הָרַ"ב. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ב - משנה ב

חָל לִהְיוֹת אַחַר הַשַּׁבָּת, בֵּית שַׁמַּאי אוֹמְרִים, מַטְבִּילִין אֶת הַכֹּל מִלִּפְנֵי הַשַּׁבָּת, וּבֵית הִלֵּל אוֹמְרִים, כֵּלִים מִלִּפְנֵי הַשַּׁבָּת, וְאָדָם בַּשַּׁבָּת:

 ר"ע מברטנורה  מַטְבִּילִין אֶת הַכֹּל. שֶׁחַיָּב אָדָם לְטַהֵר עַצְמוֹ בָּרֶגֶל {ז}. וְכָל מַה שֶּׁצָּרִיךְ טְבִילָה, בֵּין אָדָם בֵּין כֵּלִים, מַטְבִּילִין מִלִּפְנֵי הַשַּׁבָּת: כֵּלִים מִלִּפְנֵי הַשַּׁבָּת. שֶׁכֵּיוָן שֶׁהֵם רְאוּיִן לְאַחַר טְבִילָה לְדָבָר שֶׁלֹּא הָיוּ רְאוּיִן לוֹ קֹדֶם טְבִילָה, אִי מַטְבִּיל לְהוּ בְּשַׁבָּת וּבְיוֹם טוֹב הֲוָה לֵיהּ מְתַקֵּן מָנָא וְאָסוּר. וְשַׁבָּת דְּנָקַט הָכָא לְהוֹדִיעֲךָ כֹּחָן דְּבֵית הִלֵּל, דַּאֲפִלּוּ בְּשַׁבָּת שָׁרוּ לְהַטְבִּיל אָדָם, מִשּׁוּם דְּנִרְאֶה כְּמֵיקֵר וּכְמִצְטַנֵּן {ח} וְרוֹחֵץ לְתַעֲנוּג וְלֹא לְשֵׁם טְבִילָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ז} דִּכְתִיב וּבְנִבְלָתָם לֹא תִגָּעוּ [וַיִּקְרָא יא], וְמוֹקִים לַהּ בְּמַסֶּכֶת רֹאשׁ הַשָּׁנָה דַּף ט"ז בָּרֶגֶל: {ח} וַאֲפִלּוּ אִם טוֹבֵל בְּמַיִם סְרוּחִים כוּ' וּבִימוֹת הַגְּשָׁמִים דְּלֵיכָּא לְמֵימַר דְּמֵחֲזֵי כְּמֵיקֵר, נִרְאֶה כְיוֹרֵד לִרְחֹץ מִטִּיט וְצוֹאָה שֶׁבְּגוּפוֹ. גְּמָרָא. וְכָתַב בֵּית יוֹסֵף דְּזֶה שַׁיָּךְ גַּם כֵּן בְּמַיִם סְרוּחִים: (עיקר תוי"ט).

פרק ב - משנה ג

וְשָׁוִין שֶׁמַּשִּׁיקִין אֶת הַמַּיִם בִּכְלִי אֶבֶן לְטַהֲרָן, אֲבָל לֹא מַטְבִּילִין. וּמַטְבִּילִין מִגַּב לְגַב וּמֵחֲבוּרָה לַחֲבוּרָה:

 ר"ע מברטנורה  וְשָׁוִין שֶׁמַּשִּׁיקִין אֶת הַמַּיִם בִּכְלִי אֶבֶן. מִי שֶׁיֵּשׁ לוֹ מַיִם יָפִים לִשְׁתּוֹת וְנִטְמְאוּ, מְמַלֵּא מֵהֶן כְּלִי אֶבֶן שֶׁאֵינוֹ מְקַבֵּל טֻמְאָה, וְנוֹתְנָן בְּמִקְוֵה מַיִם מְלוּחִים {ט} אוֹ עֲכוּרִים עַד שֶׁנּוֹשְׁקִים מַיִם לְמַיִם, נִמְצְאוּ אֵלּוּ זְרוּעִים וּמְחֻבָּרִים לְמֵי מִקְוֶה וּבָטְלוּ אַגַּבַּיְהוּ וְטָהֲרוּ. וְאֵין טָהֳרָה בַּמִּקְוֶה לְשׁוּם אֹכֶל וּמַשְׁקֶה אֶלָּא לְמַיִם בִּלְבַד, וְלֹא בְּתוֹרַת טְבִילָה אֶלָּא בְּתוֹרַת זְרִיעָה: אֲבָל לֹא מַטְבִּילִין. שֶׁאֵין נוֹתְנִין אוֹתָן בִּכְלִי עֵץ טָמֵא, שֶׁצָּרִיךְ טְבִילָה לַכְּלִי, לְהַשִּׁיק בּוֹ כְּדֵי לַעֲלוֹת טְבִילָה לִכְלִי {י} אַגַּב הַשָּׁקַת מַיִם: וּמַטְבִּילִין מִגַּב לְגַב. מִי שֶׁהִטְבִּיל כֵּלִים לְדַעַת שֶׁיִּדְרֹךְ בָּהֶן זֵיתִים בְּבֵית הַבַּד לְשֶׁמֶן שֶׁל חֻלִּין, וְנִמְלַךְ לִדְרֹךְ בָּהֶן עֲנָבִים בַּגַּת לְיַיִן שֶׁל תְּרוּמָה, צָרִיךְ לְהַטְבִּילָן פַּעַם שְׁנִיָּה לְשֵׁם תְּרוּמָה. וְכֵן אִם הִטְבִּיל כֵּלִים לְשֵׁם תְּרוּמָה וְנִמְלַךְ לַעֲשׂוֹת בָּהֶם קֹדֶשׁ, צְרִיכִים טְבִילָה שְׁנִיָּה לְשֵׁם קֹדֶשׁ. וְאוֹתָהּ טְבִילָה יָכֹל לְהַטְבִּיל בְּיוֹם טוֹב, שֶׁאֵין כָּאן תִּקּוּן כְּלִי, לְפִי שֶׁאֵין טְבִילָה זוֹ לְהַעֲלוֹת אוֹתָן מִיּדֵי טֻמְאָה אֶלָּא לְתוֹסֶפֶת טָהֳרָה: וּמֵחֲבוּרָה לַחֲבוּרָה. אִם הִטְבִּיל כֵּלִים עַל מְנַת לֶאֱכֹל פִּסְחוֹ עִם חֲבוּרָה זוֹ וְנִמְלַךְ לְהִמָּנוֹת עִם חֲבוּרָה אַחֶרֶת וּבָא לְהַטְבִּיל כֵּלָיו פַּעַם שְׁנִיָּה, אוֹתָהּ טְבִילָה מֻתָּר לַעֲשׂוֹתָהּ בְּיוֹם טוֹב {יא}: (ר"ע מברטנורה).

 עיקר תוי"ט  {ט} דְּאִי לֹא תֵימָא הָכִי לָמָּה לִי לְהַשִּׁיקָם, כֵּיוָן שֶׁיֵּשׁ לוֹ בְּלָאו הָכִי מַיִם יָפִים לִשְׁתּוֹת. הָרַמְבַּ"ם: {י} רַשִׁ"י. וּמַשְׁמַע דְּאִלּוּ בִּכְלִי עֵץ שֶׁאֵינוֹ טָמֵא רַשַּׁאי לְמַלְּאוֹתָן וּלְטָבְלָן בּוֹ, וְאַף עַל גַּב דְּהַשְׁתָּא מְקַבֵּל טֻמְאָה מֵהַמַּיִם, הָךְ טֻמְאָה לֵיתָא אֶלָּא מִדְּרַבָּנָן, וְגַם נִטְמָא בְּיוֹם טוֹב עַצְמוֹ, וְקַיְמָא לָן דִּבְחַד מְהָנֵי גַוָּנֵי מֻתָּר לְטָבְלוֹ בְּיוֹם טוֹב. וְקָשֶׁה דְאִם כֵּן בָּרֵישָׁא כְּלִי אֶבֶן לָמָּה לִי, לָכֵן נִרְאֶה לִי דְּטָמֵא רוֹצֶה לוֹמַר שֶׁעַכְשָׁיו נִטְמָא מֵהַמַּיִם, וְטַעְמָא דְּאָסוּר לִגְרֹם טֻמְאָה לִכְלִי בַּיָּדַיִם כְּמוֹ שֶׁכָּתְבוּ הַתּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} רַשִׁ"י. וְהַךְ טְבִילָה לֹא מִתַּנְיָא בְּשׁוּם דּוּכְתָּא, וְרַשִׁ"י לְטַעְמֵיהּ דְּגָרַס בַּגְּמָרָא כַּדּוֹ (וְלֹא בַּדּוֹ) וּמְפָרֵשׁ לָרֵישָׁא גַם כֵּן בְּעִנְיָן אַחֵר, עַיֵּן שָׁם. וְהָרַ"ב הֲוָה לֵיהּ לְפָרֵשׁ כְּמוֹ שֶׁכָּתַב הָרַמְבַּ"ם מֵחֲבוּרַת אוֹכְלֵי חֻלִּין לְאוֹכְלֵי מַעֲשֵׂר שֵׁנִי אוֹ כְּמוֹ שֶׁכָּתַב הַמַּגִּיד מִשְׁנֶה מִתְּרוּמָה לִשְׁלָמִים. וְאִי קַשְׁיָא הַיְנוּ רֵישָׁא, הָא לְפֵרוּשׁוֹ גַּם כֵּן הָוֵי זוֹ וְאֵין צָרִיךְ לוֹמַר זוֹ: (עיקר תוי"ט).

פרק ב - משנה ד

בֵּית שַׁמַּאי אוֹמְרִים, מְבִיאִין שְׁלָמִים וְאֵין סוֹמְכִין עֲלֵיהֶן, אֲבָל לֹא עוֹלוֹת. וּבֵית הִלֵּל אוֹמְרִים, מְבִיאִין שְׁלָמִים וְעוֹלוֹת וְסוֹמְכִין {יד} עֲלֵיהֶם:

 ר"ע מברטנורה  מְבִיאִין שְׁלָמִים בְּיוֹם טוֹב. שַׁלְמֵי חֲגִיגָה, לְפִי שֶׁיֵּשׁ בָּהֶם צֹרֶךְ אֲכִילָה לָאָדָם: וְאֵין סוֹמְכִין עֲלֵיהֶם. שֶׁהַסְּמִיכָה אֲסוּרָה מִשּׁוּם שְׁבוּת, שֶׁבְּכָל כֹּחוֹ הָיָה סוֹמֵךְ וַהֲוָה לֵיהּ מִשְׁתַּמֵּשׁ בְּבַעֲלֵי חַיִּים {יב}. אֶלָּא סוֹמְכִין עֲלֵיהֶם מֵעֶרֶב יוֹם טוֹב, דְּלֵית לְהוּ לְבֵית שַׁמַּאי תֵּכֶף לַסְּמִיכָה שְׁחִיטָה: אֲבָל לֹא עוֹלוֹת. חוּץ מִתְּמִידִין וּמוּסָפִין שֶׁהֵן קָרְבַּן צִבּוּר וּזְמַנָּן קָבוּעַ. אֲבָל לֹא יָבִיא עוֹלַת יָחִיד, שֶׁאֵין בָּהּ אֲכִילַת הֶדְיוֹט. וַאֲפִלּוּ עוֹלוֹת רְאִיָּה קְרֵבוֹת בִּשְׁאָר יְמוֹת הָרֶגֶל וְלֹא בְּיוֹם טוֹב, דְּרַחֲמָנָא אָמַר (בְּמִדְבָּר כט) עֲצֶרֶת תִּהְיֶה לָכֶם, לָכֶם וְלֹא לְגָבֹהַּ: וּבֵית הִלֵּל אוֹמְרִים. מְבִיאִים שַׁלְמֵי חֲגִיגָה וְעוֹלוֹת רְאִיָּה, דִּכְתִיב (דְּבָרִים טז) עֲצֶרֶת לַה', כָּל דְּלַה'. אֲבָל נְדָרִים וּנְדָבוֹת דִּבְרֵי הַכֹּל אֵין קְרֵבִין {יג} בְּיוֹם טוֹב, לֹא עוֹלוֹת וְלֹא שְׁלָמִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {יב} וּתְנַן אֵין רוֹכְבִין עַל גַּבֵּי בְּהֵמָה. רַשִׁ"י: {יג} הוֹאִיל וְאֵין זְמַנָּן הַיּוֹם כְּלָל. וְאַף עַל גַּב דַּחֲגִיגָה וּרְאִיָּה נַמִּי יֵשׁ לָהֶן תַּשְׁלוּמִין כָּל שִׁבְעָה, שָׁרוּ בָּהּ שֶׁמָּא יֵאָנֵס וְלֹא יַקְרִיבוּהוּ בִּשְׁאָר הַיָּמִים, וּתְנַן עָבַר הָרֶגֶל וְלֹא חָג אֵינוֹ חַיָּב בְּאַחֲרָיוּתוֹ. וְאִם תֹּאמַר יְהֵא רַשַּׁאי לִשְׁחֹט נְדָרִים וּנְדָבוֹת הוֹאִיל וְהַבְּעָלִים וְגַם הַכֹּהֲנִים אוֹכְלִים מִמֶּנּוּ וְנִמְצָא שֶׁכָּל הַבָּשָׂר נֶאֱכַל. וְיֵשׁ לוֹמַר דְּעִקַּר הַהַקְרָבָה בִּשְׁבִיל גָּבוֹהַּ, וְכֹהֲנִים וּבְעָלִים מִשֶּׁל גָּבוֹהַּ קָא זָכוּ. תּוֹסָפוֹת: {יד} וְסוֹמְכִין. דִּסְבִירָא לְהוּ תֵכֶף לַסְּמִיכָה שְׁחִיטָה, וּמִדְּרַשַּׁאי לַהֲבִיאָן רַשַּׁאי לִסְמֹךְ: (עיקר תוי"ט).

פרק ב - משנה ה

בֵּית שַׁמַּאי אוֹמְרִים, לֹא יָחֵם אָדָם חַמִּין לְרַגְלָיו {טו}, אֶלָּא אִם כֵּן רְאוּיִין לִשְׁתִיָּה {טז}. וּבֵית הִלֵּל מַתִּירִין. עוֹשֶׂה {יז} אָדָם מְדוּרָה וּמִתְחַמֵּם כְּנֶגְדָּהּ:

 ר"ע מברטנורה  לֹא יָחֵם אָדָם חַמִּין לְרַגְלָיו. אֹכֶל נֶפֶשׁ הִתִּירָה הַתּוֹרָה, וְלֹא לְהַבְעִיר אוּר בִּשְׁבִיל רְחִיצָה: וּבֵית הִלֵּל מַתִּירִין. דְּמִתּוֹךְ שֶׁהֻתְּרָה הַבְעָרָה לְצֹרֶךְ אֲכִילָה, הֻתְּרָה נַמִּי שֶׁלֹּא לְצֹרֶךְ אֲכִילָה כְּשֶׁיֵּשׁ בָּהּ צֹרֶךְ אֵיזוֹ הֲנָאָה. וּפְסַק הֲלָכָה, דְּחַמִּין שֶׁהוּחְמוּ בְּיוֹם טוֹב רוֹחֵץ בָּהֶן פָּנָיו יָדָיו וְרַגְלָיו, אֲבָל לֹא כָּל גּוּפוֹ, דְּגָזְרוּ יוֹם טוֹב אַטּוּ שַׁבָּת. וְחַמִּין שֶׁהוּחְמוּ מֵעֶרֶב יוֹם טוֹב רוֹחֵץ בָּהֶן כָּל גּוּפוֹ בְּיוֹם טוֹב: (ר"ע מברטנורה).

 עיקר תוי"ט  {טו} לְרַגְלָיו. הוּא הַדִּין לְפָנָיו וְיָדָיו, וַחֲדָא נָקַט. וְאֶפְשָׁר דְּלִרְבוּתָא נָקַט, דְּלִרְחֹץ הָרַגְלַיִם בְּחַמִּין הֲוָה דָבָר שָׁוֶה לְכָל נֶפֶשׁ יוֹתֵר מִפָּנָיו וְיָדָיו, שֶׁכֵּן פָּנָיו וְיָדָיו רְגִילִים יוֹתֵר לִרְחֹץ בְּצוֹנֵן: {טז} לִשְׁתִיָּה. וּמַיְרֵי כְגוֹן שֶׁשָּׁתָה מֵהֶן וּמַרְבֶּה לִרְחֹץ רַגְלָיו, דְּאִי לֹא תֵימָא הָכִי, הָא לֵית לְהוּ לְבֵית שַׁמַּאי מִתּוֹךְ. וְהָכִי אִיתָא בְּהֶדְיָא בַּיְרוּשַׁלְמִי, דִּלְבֵית שַׁמַּאי צָרִיךְ לִשְׁתּוֹת מֵהֶן. תּוֹסָפוֹת: {יז} עוֹשֶׂה כוּ'. וּבֵית הִלֵּל קָתָנֵי לַהּ כִּדְאִיתָא בַּבְּרַיְתָא: (עיקר תוי"ט).

פרק ב - משנה ו

שְׁלשָׁה דְבָרִים רַבָּן גַּמְלִיאֵל מַחֲמִיר כְּדִבְרֵי בֵית שַׁמַּאי {יח}, אֵין טוֹמְנִין אֶת הַחַמִּין מִיּוֹם טוֹב לַשַּׁבָּת, וְאֵין זוֹקְפִין אֶת הַמְּנוֹרָה בְיוֹם טוֹב, וְאֵין אוֹפִין פִּתִּין גְּרִיצִין אֶלָּא רְקִיקִין. אָמַר רַבָּן גַּמְלִיאֵל, מִימֵיהֶן שֶׁל בֵּית אַבָּא לֹא הָיוּ אוֹפִין פִּתִּין גְּרִיצִין, אֶלָּא רְקִיקִין. אָמְרוּ לוֹ, מַה נַּעֲשֶׂה לְבֵית אָבִיךָ, שֶׁהָיוּ מַחֲמִירִין עַל עַצְמָן וּמְקִלִּין לְכָל יִשְׂרָאֵל, לִהְיוֹת אוֹפִין פִּתִּין גְּרִיצִין וְחֹרִי:

 ר"ע מברטנורה  אֵין טוֹמְנִין אֶת הַחַמִּין מִיּוֹם טוֹב לַשַּׁבָּת. דְּסָבְרֵי בֵּית שַׁמַּאי אֵין אוֹפִין אֶלָּא אִם כֵּן עֵרֵב בְּפַת, וְאֵין מְבַשְּׁלִין אֶלָּא אִם כֵּן עֵרֵב עַל הַתַּבְשִׁיל, וְאֵין טוֹמְנִין אֶלָּא אִם כֵּן הָיוּ לוֹ חַמִּין טְמוּנִין מֵעֶרֶב יוֹם טוֹב. וּבֵית הִלֵּל סָבְרֵי דְּאוֹפִין וּמְבַשְּׁלִין וְטוֹמְנִין עַל עֵרוּב שֶׁל תַּבְשִׁיל בִּלְבַד. וּמִיהוּ בַּגְּמָרָא מוּכָח דְּבֵית הִלֵּל נַמִּי בָּעוּ פַּת וּבָשָׂר. וְכֵן הֲלָכָה: אֵין זוֹקְפִין מְנוֹרָה. שֶׁל חֻלְיוֹת שֶׁנִּתְפָּרְקוּ חֻלְיוֹתֶיהָ, אֵין מַחֲזִירִין אוֹתָן, דַּהֲוֵי כְּבוֹנֶה, וְיֵשׁ בִּנְיָן בְּכֵלִים. וּבֵית הִלֵּל סָבְרֵי אֵין בִּנְיָן בְּכֵלִים {יט}: גְּרִיצִין. כִּכְּרוֹת עָבוֹת {כ}: אֶלָּא רְקִיקִין. כִּכְּרוֹת דַּקּוֹת, דְּסָבְרֵי בֵּית שַׁמַּאי אֵין אוֹפִין פַּת מְרֻבָּה בְּיוֹם טוֹב מִשּׁוּם טִרְחָא. וּבֵית הִלֵּל אוֹמְרִים אוֹפִין פַּת מְרֻבָּה בְּיוֹם טוֹב, שֶׁבִּזְמַן שֶׁהַפַּת מְרֻבָּה הִיא נֶאֱפֵית יָפֶה: וְחֹרִי. עִסָּה גְּדוֹלָה הַנֶאֱפֵית עַל גַּבֵּי גֶּחָלִים, מִשּׁוּם דְּבָעֲיָא הֶסֵּק גָּדוֹל, שֶׁהַגֶּחָלִים הוֹלְכוֹת וְעוֹמְמוֹת, וְטִרְחָא אִיכָּא. וְאֵין הֲלָכָה כְּרַבָּן גַּמְלִיאֵל בְּכָל מַה שֶּׁמַּחְמִיר כְּדִבְרֵי בֵּית שַׁמַּאי: (ר"ע מברטנורה).

 עיקר תוי"ט  {יח} כְּדִבְרֵי בֵית שַׁמַּאי. אַף עַל גַּב שֶׁהוּא הָיָה מִזֶּרַע הִלֵּל: {יט} וְאֵינוֹ חַיָּב אֶלָּא אִם כֵּן עוֹשֶׂה מְלָאכָה כְּגוֹן מְמַחֵק אוֹ מְחַתֵּךְ כוּ' שֶׁהֵן אֲבוֹת מְלָאכוֹת לְעַצְמָן, אֲבָל חֲזָרַת חֻלְיוֹת שֶׁאֵין בָּהֶם אֶלָּא מִשּׁוּם בִּנְיָן, אֵין בּוֹנֶה אֶלָּא בְּבַיִת וְאֹהָלִים. רַשִׁ"י: {כ} וּרְצוֹנוֹ לוֹמַר פַּת מְרֻבָּה כְּמוֹ שֶׁכּוֹתֵב לְקַמָּן, אֶלָּא לְהָכִי קָרֵי לַהּ עָבָה בַּבְּרַיְתָא, מִשּׁוּם דִּנְפִישֵׁי בְּלִישָׁה. אִי נַמִּי בְּאַתְרָא דְהַאי תַנָּא פַּת מְרֻבָּה פַּת עָבָה קָרֵי לַהּ: (עיקר תוי"ט).

פרק ב - משנה ז

אַף הוּא אָמַר שְׁלֹשָׁה דְבָרִים לְהָקֵל, מְכַבְּדִין בֵּין הַמִּטּוֹת, וּמַנִּיחִין אֶת הַמֻּגְמָר בְּיוֹם טוֹב, וְעוֹשִׂין גְּדִי מְקֻלָּס בְּלֵילֵי פְסָחִים. וַחֲכָמִים אוֹסְרִין:

 ר"ע מברטנורה  מְכַבְּדִין בֵּין הַמִּטּוֹת. שֶׁאוֹכְלִין שָׁם. שֶׁרְגִילִים לְהָסֵב וְלֶאֱכֹל עַל גַּבֵּי מִטּוֹת. וּמִפְּנֵי שֶׁהוּא מָקוֹם מֻעָט לֹא חָיְשִׁינַן שֶׁמָּא יַשְׁוֶה גֻּמּוֹת {כא} כִּדְחָיְשִׁינַן גַּבֵּי בַּיִת שֶׁהוּא גָּדוֹל וְאִי אֶפְשָׁר שֶׁלֹּא יִהְיֶה שָׁם גֻּמָּא: מֻגְמָר. לְבוֹנָה עַל גַּבֵּי גֶּחָלִים לְהָרִיחַ בָּהּ. אֲבָל לְגַמֵּר אֶת הַכֵּלִים דִּבְרֵי הַכֹּל אָסוּר: מְקֻלָּס. כְּרָעָיו וּבְנֵי מֵעָיו תְּלוּיִין חוּצָה לוֹ [בְּצִדּוֹ] כְּשֶׁצּוֹלֵהוּ, וְהָיוּ עוֹשִׂים זֵכֶר לַפֶּסַח דִּכְתִיב בֵּיהּ (שְׁמוֹת יב) רֹאשׁוֹ עַל כְּרָעָיו וְעַל קִרְבּוֹ. מְקֻלָּס, כְּגִבּוֹר זֶה שֶׁכְּלֵי זֵינָיו עִמּוֹ. תַּרְגּוּם וְכוֹבַע נְחֹשֶׁת, וְקוּלְסָא דִנְחָשָׁא. וְרַמְבַּ"ם פֵּרֵשׁ מְקֻלָּס, מְכֻבָּד, לְשׁוֹן קִלּוּס: וַחֲכָמִים אוֹסְרִין. בִּשְׁלָשְׁתָּן. כִּבּוּד, מִשּׁוּם אַשְׁווּיֵי גֻּמּוֹת. וּמֻגְמָר, מִשּׁוּם דְּלָאו צֹרֶךְ לְכָל נֶפֶשׁ הוּא, שֶׁאֵינוֹ אֶלָּא לַמְּפֻנָּקִים וְלַמְּעֻנָּגִים וּלְמִי שֶׁרֵיחוֹ רַע. מְקֻלָּס, מִפְּנֵי שֶׁנִּרְאֶה כְּאוֹכֵל קָדָשִׁים בַּחוּץ. וַהֲלָכָה כַּחֲכָמִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {כא} וְכִי תֵימָא מִכָּל מָקוֹם הֲרֵי הוּא מֵזִיז עָפָר מִמְּקוֹמוֹ, אִיכָּא לְמֵימַר דְּטִלְטוּל מִן הַצַּד עַל יְדֵי דָבָר אַחֵר מִשּׁוּם כְּבוֹד יוֹם טוֹב הִתִּירוּהוּ. הָרַ"ן. וּמֵהַאי טַעְמָא הִתִּירוּ לְכַבֵּד עֲצָמוֹת כוּ' אַף עַל גַּב דְּאָסוּר לְטַלְטְלָן: (עיקר תוי"ט).

פרק ב - משנה ח

שְׁלשָׁה דְבָרִים רַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה מַתִּיר, וַחֲכָמִים אוֹסְרִין. פָּרָתוֹ יוֹצְאָה בִרְצוּעָה שֶׁבֵּין קַרְנֶיהָ, וּמְקָרְדִין אֶת הַבְּהֵמָה בְיוֹם טוֹב {כה}, וְשׁוֹחֲקִין אֶת הַפִּלְפְּלִין בָּרֵחַיִם שֶׁלָּהֶם. רַבִּי יְהוּדָה אוֹמֵר, אֵין מְקָרְדִין אֶת הַבְּהֵמָה בְיוֹם טוֹב, מִפְּנֵי שֶׁעוֹשֶׂה חַבּוּרָה, אֲבָל מְקַרְצְפִין {כו}. וַחֲכָמִים אוֹמְרִים, אֵין מְקָרְדִין, אַף לֹא מְקַרְצְפִין:

 ר"ע מברטנורה  פָּרָתוֹ יוֹצְאָה בִרְצוּעָה שֶׁבֵּין קַרְנֶיהָ. לְנוֹי {כב}. וְאָמְרוּ חֲכָמִים מַשְּׂאוֹי הוּא וְאֵינוֹ תַּכְשִׁיט לָהּ. וְלֹא פָּרָתוֹ שֶׁל רַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה הָיְתָה אֶלָּא שֶׁל שְׁכֵנָתוֹ, וְעַל שֶׁלֹּא מִחָה בָּהּ נִקְרֵאת עַל שְׁמוֹ: וּמְקָרְדִין בְּיוֹם טוֹב. כְּמִין מְגֵרָה קְטַנָּה שֶׁל בַּרְזֶל שֶׁשִּׁנֶּיהָ דַּקּוֹת וּמְחַכְּכִין וּמְגָרְדִין {כג} בָּהּ הַבְּהֵמָה, וְאַף עַל גַּב דְּעָבֵיד חֲבוּרָה {כד}: בָּרֵחַיִם שֶׁלָּהֶן. קְטַנּוֹת הָעֲשׂוּיוֹת לְכָךְ: מְקַרְצְפִין. בִּמְגֵרָה שֶׁל עֵץ שֶׁשִּׁנֶּיהָ גַּסּוֹת וְאֵין עוֹשִׂין חַבּוּרָה: אַף לֹא מְקַרְצְפִין. דְּגָזְרִינַן קִרְצוּף אַטּוּ קִירוּד. וְאֵין הֲלָכָה כְּרַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה בִּשְׁלֹשָׁה דְּבָרִים הַלָּלוּ, אֶלָּא בִּמְקָרְדִין אֶת הַבְּהֵמָה בִּלְבַד, מִשּׁוּם דִּבְהָא קָם לֵיהּ בְּשִׁטָּתֵיהּ דְּרַבִּי שִׁמְעוֹן דְּאָמַר דָּבָר שֶׁאֵין מִתְכַּוֵּן מֻתָּר, וְקַיְמָא לָן כְּוָתֵיהּ. וַחֲכָמִים דִּפְלִיגֵי עֲלֵיהּ סָבְרֵי כְּרַבִּי יְהוּדָה דְּאָמַר דָּבָר שֶׁאֵין מִתְכַּוֵּן אָסוּר, וְאֵינָהּ הֲלָכָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {כב} וּבְסוֹף פֶּרֶק ה' דְשַׁבָּת כָּתַב בֵּין לְנוֹי בֵּין לְשִׁמּוּר, דְּכָל נְטִירוּתָא יְתֵירְתָא מַשְּׂאוֹי הוּא, וּפְלֻגְתָּא הוּא בַּגְּמָרָא וַהֲלָכָה כְּמוֹ שֶׁכָּתַב שָׁם: {כג} מִלְּשׁוֹן וַיִּקַּח לוֹ חֶרֶשׂ לְהִתְגָּרֵד בּוֹ [אִיּוֹב ב] וְהַגִּימֶ"ל מִתְחַלֵּף בְּקוּ"ף, שֶׁהֵם מִמּוֹצָא גיכ"ק: {כד} וּלְשׁוֹן רַשִׁ"י בַּגְּמָרָא, מִיהוּ הַךְ חֲשָׁשָׁא [דְּחַבּוּרָה] בְּקִירוּד אִיתָא, וְנִמּוּקוֹ עִמּוֹ, דִּפְסִיק רֵישָׁא מוֹדֶה רַבִּי שִׁמְעוֹן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כה} בְּיוֹם טוֹב. מִדְּאִיצְטְרִיךְ לְמִתְנֵי בְּיוֹם טוֹב שְׁמַע מִנָּהּ דְּבָבָא קַמַּיְתָא דְּהַיְנוּ פָּרָתוֹ כוּ' לָאו בְּיוֹם טוֹב אֶלָּא בְּשַׁבָּת. דְּאֵין מְצֻוִּין עַל שְׁבִיתַת בְּהֶמְתּוֹ בְּיוֹם טוֹב. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כו} מְקַרְצְפִין. מִלָּה אֲרַמִּית הוּא, לְהִתְגָּרֵד בּוֹ לְאִתְקַרְצְפָה בֵּיהּ. עָרוּךְ. וּבַגְּמָרָא דִּפְסָחִים דַּף י"א מַאי שְׁנָא גַבֵּי חָמֵץ דְּגָזוּר וּמַאי שְׁנָא גַּבֵּי קִרְצוּף דְּלֹא גָּזוּר. לֶחֶם בְּלֶחֶם מִיחַלַּף, קִירוּד בְּקִרְצוּף לֹא מִיחַלַּף: (עיקר תוי"ט).

פרק ב - משנה ט

הָרֵחַיִם שֶׁל פִּלְפְּלִין טְמֵאָה, מִשּׁוּם שְׁלשָׁה כֵלִים, מִשּׁוּם כְּלִי קִבּוּל, וּמִשּׁוּם כְּלִי מַתָּכוֹת, וּמִשּׁוּם כְּלִי כְבָרָה:

 ר"ע מברטנורה  מִשּׁוּם שְׁלֹשָׁה כֵלִים. שֶׁאִם פֵּרַשׁ אֶחָד מֵהֶן אֵינָן שִׁבְרֵי כֵּלִים, אֶלָּא כָּל אֶחָד מֵהֶם חָשׁוּב כְּלִי בִּפְנֵי עַצְמוֹ וְטָמֵא, אַף עַל פִּי שֶׁבְּעוֹדָן בָּרֵחַיִם הֵן מְחֻבָּרִים וְנִרְאִים כִּכְלִי אֶחָד: מִשּׁוּם כְּלִי קִבּוּל. הַתַּחְתּוֹנָה שֶׁמְּקַבֶּלֶת אֲבַק שְׁחִיקַת הַפִּלְפְּלִים דֶּרֶךְ נִקְבֵי הַכְּבָרָה, טְמֵאָה מִשּׁוּם כְּלִי קִבּוּל, שֶׁכְּלִי עֵץ שֶׁיֵּשׁ לוֹ בֵּית קִבּוּל הוּא: וּמִשּׁוּם כְּלִי מַתָּכוֹת. הָעֶלְיוֹנָה שֶׁכּוֹתְשִׁין וְטוֹחֲנִין בָּהּ הַפִּלְפְּלִין, טְמֵאָה מִשּׁוּם כְּלִי מַתֶּכֶת, דְּמִשּׁוּם כְּלִי עֵץ לֵיכָּא לִטְמוּיַהּ, דִּפְשׁוּטֵיהֶן טְהוֹרִין, אֶלָּא מִשּׁוּם צִפּוּיָהּ הַתַּחְתּוֹן שֶׁהוּא שֶׁל מַתֶּכֶת {כז}: וּמִשּׁוּם כְּלִי כְבָרָה. אֶמְצָעִית שֶׁהִיא מַקֶּפֶת אֶת הַכְּבָרָה {כח}. מִשּׁוּם כְּלִי עֵץ לֹא מְטַמְּאָה, דְּאֵין קִבּוּלָהּ קִבּוּל, אֶלָּא חֲכָמִים גָּזְרוּ טֻמְאָה עַל הַכְּבָרָה מִשּׁוּם כְּלִי אָרוּג {כט}. וַאֲפִלּוּ אֵין הַכְּבָרָה שֶׁל מַתֶּכֶת {ל}, מְטַמְּאָה מִשּׁוּם כְּלִי כְּבָרָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {כז} מְסַיֵּם רַשִׁ"י שֶׁהוּא עִקָּר וְהָעֵץ בָּטֵל אֶצְלוֹ: {כח} וְנִרְאֶה שֶׁרְצוֹנוֹ לוֹמַר הָאֶמְצָעִית הִיא הַכְּבָרָה, אֶלָּא לְפִי שֶׁיֵּשׁ לָהּ בְּהֶכְרֵחַ סוֹבֵב הַמַּקִּיפָהּ, אַסְבְּרַה לַהּ בְּהַאי לִישָׁנָא: {כט} כְּלוֹמַר בֶּגֶד דִּכְתִיב בַּתּוֹרָה שֶׁהוּא אָרוּג: {ל} דְּאִלּוּ הִיא שֶׁל מַתֶּכֶת וַדַּאי טְמֵאָה מִשּׁוּם פְּשׁוּטֵי כְּלֵי מַתֶּכֶת, כֵּן כָּתַב רַשִׁ"י. כְּלוֹמַר אַף עַל פִּי שֶׁהַכְּבָרָה אֵינָהּ יְכוֹלָה לְקַבֵּל כְּלוּם: (עיקר תוי"ט).

פרק ב - משנה י

עֲגָלָה שֶׁל קָטָן טְמֵאָה מִדְרָס וְנִטֶּלֶת בְּשַׁבָּת, וְאֵינָהּ נִגְרֶרֶת אֶלָּא עַל גַּבֵּי כֵלִים. רַבִּי יְהוּדָה אוֹמֵר, כָּל הַכֵּלִים אֵין נִגְרָרִין חוּץ מִן הָעֲגָלָה, מִפְּנֵי שֶׁהִיא כוֹבֶשֶׁת:

 ר"ע מברטנורה  עֲגָלָה. הָעֲשׂוּיָה לְשַׂחֵק לְקָטָן, וּמְיֻחֶדֶת לוֹ לֵישֵׁב עָלֶיהָ {לא}: טְמֵאָה מִדְרָס. אִם הָיָה הַקָּטָן זָב נַעֲשֵׂית הָעֲגָלָה אַב הַטֻּמְאָה: וְנִטֶּלֶת בְּשַׁבָּת. דְּתוֹרַת כְּלִי עָלֶיהָ: וְאֵינָהּ נִגְרֶרֶת. בְּשַׁבָּת: אֶלָּא עַל גַּבֵּי כֵלִים. עַל גַּבֵּי בְּגָדִים, מִפְּנֵי שֶׁעוֹשָׂה חָרִיץ בַּקַּרְקַע, וְחוֹפֵר חַיָּב מִשּׁוּם חוֹרֵשׁ: רַבִּי יְהוּדָה אוֹמֵר כוּ'. תְּרֵי תַּנָּאֵי נִינְהוּ וְאַלִּיבָּא דְּרַבִּי יְהוּדָה, דְּתַנָּא קַמָּא נַמִּי רַבִּי יְהוּדָה הִיא דְּאָמַר דָּבָר שֶׁאֵין מִתְכַּוֵּן אָסוּר. וַאֲתָא הַאי תַּנָּא אַחֲרִיתִי וְאָמַר דְּלֹא אָסַר רַבִּי יְהוּדָה בַּעֲגָלָה שֶׁל קָטָן, מִפְּנֵי שֶׁאֵינָהּ עוֹשָׂה חָרִיץ בְּהִלּוּכָהּ {לב} עַל יְדֵי חֲפִירָה, אֶלָּא כּוֹבֶשֶׁת הַקַּרְקַע וְנָדוֹשׁ תַּחְתֶּיהָ וְנַעֲשֶׂה עֲפָרָהּ נָמוּךְ, אֲבָל אֵינָהּ מְזִיזָה עָפָר מִמְּקוֹמָהּ. וּכְבָר אִפְסִיקָא הִלְכָתָא כְּרַבִּי שִׁמְעוֹן דְּאָמַר גּוֹרֵר אָדָם מִטָּה כִּסֵּא וְסַפְסָל, וּבִלְבַד שֶׁלֹּא יִתְכַּוֵּן לַעֲשׂוֹת חָרִיץ: (ר"ע מברטנורה).

 עיקר תוי"ט  {לא} וְכָתְבוּ הַתּוֹסָפוֹת דְּלֹא נְהִירָא, דְּמַאי קָא מַשְׁמַע לָן, פְּשִׁיטָא, הָא אֵין לְךָ כְּלִי מְיֻחָד גָּדוֹל מִזֶּה, לְכָךְ נִרְאֶה לִי דְּהַיְנוּ כְּלִי שֶׁעוֹשִׂין לַקְּטַנִּים לְהִתְלַמֵּד לַהֲלוֹךְ, וְהִיא עוֹמֶדֶת עַל שְׁלֹשָׁה אוֹפַנִים, וְאוֹחֵז בָּהּ בְּיָדוֹ וְהִיא מִתְגַּלְגֶּלֶת לְפָנָיו: {לב} וְאִידָךְ סָבַר פְּעָמִים שֶׁאֵין הָאוֹפָן מִתְגַּלְגֵּל וְהוּא נִגְרָר וְחוֹפֵר, וְאִידָךְ סָבַר לֹא שְׁכִיחַ הָכִי. רַשִׁ"י: (עיקר תוי"ט).

פרק ג - משנה א

אֵין צָדִין דָּגִים מִן הַבֵּיבָרִין בְּיוֹם טוֹב, וְאֵין נוֹתְנִין לִפְנֵיהֶם מְזוֹנוֹת. אֲבָל צָדִין חַיָּה וָעוֹף מִן הַבֵּיבָרִין, וְנוֹתְנִין לִפְנֵיהֶם מְזוֹנוֹת. רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר, לֹא כָל הַבֵּיבָרִין שָׁוִין. זֶה הַכְּלָל, כָּל הַמְחֻסָּר צִידָה אָסוּר, וְשֶׁאֵינוֹ מְחֻסָּר צִידָה מֻתָּר:

 ר"ע מברטנורה  אֵין צָדִין דָּגִים. אַף עַל גַּב דִּשְׁחִיטָה וַאֲפִיָּה וּבִשּׁוּל מֵאֲבוֹת מְלָאכוֹת הֵן וְהֻתְּרוּ לְצֹרֶךְ יוֹם טוֹב, צִידָה דָּמְיָא לִקְצִירָה, וּקְצִירָה לֹא הֻתְּרָה בְּיוֹם טוֹב {א}. בֵּיבָרִין שֶׁל דָּגִים, הֵן בְּרֵכוֹת שֶׁל מַיִם שֶׁדָּגִים מִתְגַּדְּלִים שָׁם. בֵּיבָרִים שֶׁל חַיָּה, קַרְפִּיפוֹת מֻקָּפִין גָּדֵר סָבִיב וּמְבִיאִים שָׁם חַיּוֹת וְיוֹלְדוֹת וּמִתְגַּדְּלוֹת שָׁם: וְאֵין נוֹתְנִין. לִפְנֵי הַדָּגִים מְזוֹנוֹת, דְּאֶפְשָׁר לָהֶם בְּלֹא מְזוֹנוֹת וְאֵין מְזוֹנוֹתָן עָלֶיךָ: אֲבָל צָדִין חַיָּה וָעוֹף. הַמּוּכָנִים מֵאֶתְמוֹל: מִן הַבֵּיבָרִין. הַקְּטַנִּים, שֶׁאֵין מְחֻסָּרִים צִידָה. וְרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל לָאו לְאַפְלוּגֵי אַתַּנָּא קַמָּא אָתָא, אֶלָּא לִפְרוּשֵׁי {ב}: כָּל הַמְּחֻסָּר צִידָה. שֶׁצָּרִיךְ לְבַקֵּשׁ תַּחְבּוּלוֹת כְּדֵי לְתָפְסוֹ. וְכָל דְּרָהֵיט בַּתְרֵיהּ וּמָטֵי לֵיהּ בְּחַד שְׁחִיָּה לֹא הָוֵי מְחֻסַּר צִידָה, וְאִי לֹא, הָוֵי מְחֻסַּר צִידָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} וְטַעְמָא כְּמוֹ בִּטְחִינָה, כְּמוֹ שֶׁכָּתַבְנוּ בְּפֶרֶק קַמָּא בְּאוֹת ל'. וְכֵן צִידָה אָסְרוּ, שֶׁפְּעָמִים שֶׁתַּעֲלֶה בִּמְצוּדָתוֹ דָּגִים הַרְבֵּה כוּ'. הָרֹא"שׁ: {ב} וּלְפִי זֶה בְּדָגִים לְעוֹלָם אֵין צָדִין, דְּאִי לֹא תֵימָא הָכִי לָמָּה חִלְּקָן. וְהַטַּעַם, מִפְּנֵי שֶׁהֵם מְכֻסִּים מִן הָעַיִן: (עיקר תוי"ט).

פרק ג - משנה ב

מְצוּדוֹת חַיָּה וָעוֹף וְדָגִים שֶׁעֲשָׂאָן מֵעֶרֶב יוֹם טוֹב, לֹא יִטֹּל מֵהֶן בְּיוֹם טוֹב, אֶלָּא אִם כֵּן יוֹדֵעַ שֶׁנִּצּוֹדוּ מֵעֶרֶב יוֹם טוֹב. וּמַעֲשֶׂה בְנָכְרִי אֶחָד, שֶׁהֵבִיא דָגִים לְרַבָּן גַּמְלִיאֵל, וְאָמַר, מֻתָּרִין הֵן, אֶלָּא שֶׁאֵין רְצוֹנִי לְקַבֵּל הֵימֶנּוּ:

 ר"ע מברטנורה  וּמַעֲשֶׂה בְנָכְרִי אֶחָד. חִסּוּרֵי מִחַסְּרָא וְהָכִי קָתָנֵי, סָפֵק מוּכָן אָסוּר {ג} וְרַבָּן גַּמְלִיאֵל מַתִּיר, וּמַעֲשֶׂה בְּנָכְרִי אֶחָד וְכוּ'. וְאֵין הֲלָכָה כְּרַבָּן גַּמְלִיאֵל, אֶלָּא פֵּרוֹת וְדָגִים שֶׁהוּבְאוּ בְּיוֹם טוֹב, סָפֵק נִלְקְטוּ {ד} הַיּוֹם סָפֵק מֵאֶתְמוֹל, סָפֵק נִצּוֹדוּ הַיּוֹם סָפֵק מֵאֶתְמוֹל, אֲסוּרִים. וְכָל שֶׁאָסוּר לְאָכְלוֹ אָסוּר לְטַלְטְלוֹ. וְאִם צוּרָתָן מוֹכַחַת עֲלֵיהֶן, כְּגוֹן פֵּרוֹת דִּכְמוּשִׁין שֶׁאִי אֶפְשָׁר שֶׁנִּלְקְטוּ הַיּוֹם, וְכֵן דָּגִים שֶׁהֱבִיאוּם מִמָּקוֹם רָחוֹק שֶׁאִי אֶפְשָׁר שֶׁנִּצּוֹדוּ הַיּוֹם, מֻתָּרִין. וְדָבָר מוּכָן שֶׁבָּא מִחוּץ לַתְּחוּם {ה} בִּשְׁבִיל יִשְׂרָאֵל, אָסוּר לְאוֹתוֹ יִשְׂרָאֵל שֶׁהוּבָא בִּשְׁבִילוֹ וּלְכָל בְּנֵי בֵיתוֹ, וּמֻתָּר לְיִשְׂרָאֵל אַחֵר. וּפֵרוֹת אוֹ דָּגִים שֶׁנִּתְלְשׁוּ אוֹ נִצּוֹדוּ בְּיוֹם טוֹב רִאשׁוֹן מֻתָּרִין לָעֶרֶב בְּלֵיל יוֹם טוֹב שֵׁנִי בִּכְדֵי שֶׁיַּעֲשׂוּ, חוּץ מִשְּׁנֵי יָמִים טוֹבִים שֶׁל רֹאשׁ הַשָּׁנָה שֶׁהֵם אֲסוּרִים עַד מוֹצָאֵי יוֹם טוֹב שֵׁנִי בִּכְדֵי שֶׁיַּעֲשׂוּ. וּפֵרוּשׁ כְּדֵי שֶׁיַּעֲשׂוּ, כְּדֵי שֶׁיִּתְלְשׁוּ הַפֵּרוֹת מִמָּקוֹם שֶׁנִּתְלְשׁוּ וְיָבִיאוּ אוֹתָם מִן הַמָּקוֹם שֶׁהֱבִיאוּם. וְעִיר שֶׁדַּלְתוֹתֶיהָ נְעוּלוֹת בַּלַּיְלָה צָרִיךְ שֶׁיַּמְתִּין לַבֹּקֶר בִּכְדֵי שֶׁיַּעֲשׂוּ: אֶלָּא שֶׁאֵין רְצוֹנִי לְקַבֵּל מִמֶּנּוּ. שֶׁאֲנִי שׂוֹנֵא אוֹתוֹ: (ר"ע מברטנורה).

 עיקר תוי"ט  {ג} מִשּׁוּם דְּכָל שֶׁיֵּשׁ בְּאוֹתוֹ מִין בִּמְחֻבָּר, דַּרְכָּן שֶׁל בְּנֵי אָדָם לִלְקוֹט אוֹתוֹ בְּיוֹמוֹ וְלֹא מִבָּעֶרֶב. הָרַ"ן: {ד} לְפִי שֶׁלְּקִיטָה דּוֹמָה לִקְצִירָה וְאַף לְקִיטַת פֵּרוֹת הַנִּפְסָדִים כְּתוּתִים וַעֲנָבִים אָסְרוּ, לְפִי שֶׁהָלְכוּ אַחַר רֻבָּהּ שֶׁל לְקִיטָה. הָרַ"ן. וּבְכָאן לֹא הִתְנָה הָרַ"ב שֶׁיְּהֵא בִּשְׁבִיל יִשְׂרָאֵל כְּדִלְקַמָּן בִּתְחוּם, מִשּׁוּם דְּכָל שֶׁבִּמְחֻבָּר יֵשׁ בּוֹ מִשּׁוּם מֻקְצֶה, וּמִשּׁוּם הָכִי אֲפִלּוּ תְּלָשָׁן לְעַצְמוֹ אֲסוּרִים. וְכֵן כָּל שֶׁשַּׁיָּךְ בּוֹ אִסּוּר מֻקְצֶה אָסוּר לוֹ לֵהָנוֹת בְּיוֹם טוֹב מִמְּלַאכְתּוֹ שֶׁל נָכְרִי, אַף עַל פִּי שֶׁעֲשָׂאָהּ לְעַצְמוֹ (וְעַיֵּן מַה שֶּׁכָּתוּב בְּפֶרֶק ט"ז דְּשַׁבָּת. וְכֶבֶשׁ יֵשׁ לוֹמַר דַּעֲשָׂאוֹ מִדָּבָר מוּכָן) וּלְכָךְ לֹא הִצְרִיךְ הָרַ"ב כָּאן לְהַמְתִּין בִּכְדֵי שֶׁיַּעֲשׂוּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} בְּוַדַּאי בָּא מִחוּץ לַתְּחוּם מַיְרֵי, וְאִסּוּר זֶה פֵּרֵשׁ הָרֹא"שׁ בַּאֲכִילָה, אֲבָל בְּטִלְטוּל שָׁרֵי, דְּכֵיוָן דְּשָׁרֵי לְיִשְׂרָאֵל אַחֵר בַּאֲכִילָה אֵין לוֹ דִין מֻקְצֶה לֶאֱסֹר בְּטִלְטוּל: (עיקר תוי"ט).

פרק ג - משנה ג

בְּהֵמָה מְסֻכֶּנֶת לֹא יִשְׁחֹט, אֶלָּא אִם כֵּן יֵשׁ שָׁהוּת {ו} בַּיּוֹם לֶאֱכֹל מִמֶּנָּה כַּזַּיִת צָלִי. רַבִּי עֲקִיבָא אוֹמֵר, אֲפִלּוּ כַזַּיִת חַי מִבֵּית טְבִיחָתָהּ. שְׁחָטָהּ בַּשָּׂדֶה, לֹא יְבִיאֶנָּה בְמוֹט וּבְמוֹטָה. אֲבָל מֵבִיא בְיָדוֹ אֵבָרִים אֵבָרִים:

 ר"ע מברטנורה  בְּהֵמָה מְסֻכֶּנֶת. שֶׁהוּא יָרֵא שֶׁמָּא תָּמוּת. וְאֵין צָרִיךְ לָהּ, שֶׁכְּבָר סָעַד סְעוּדָתוֹ: לֹא יִשְׁחֹט. אֶלָּא אִם כֵּן יוֹדֵעַ שֶׁיֵּשׁ שָׁהוּת בַּיּוֹם לֶאֱכֹל כַּזַּיִת צָלִי הֵימֶנָּה: מִבֵּית טְבִיחָתָהּ. שֶׁהוּא מְזֻמָּן וּמֻפְשָׁט מֵעוֹרוֹ וְעוֹמֵד. וְאֵין הֲלָכָה כְּרַבִּי עֲקִיבָא: לֹא יְבִיאֶנָּה בְּמוֹט וּבְמוֹטָה. בִּשְׁנֵי בְּנֵי אָדָם. מִשּׁוּם דְּאַוְשָׁא מִלְּתָא {ז} וּמְזַלְזֵל בִּכְבוֹד יוֹם טוֹב: (ר"ע מברטנורה).

 עיקר תוי"ט  {ו} יֵשׁ שָׁהוּת. לִישְׁנָא דְמַתְנִיתִין דַּיְקָא, דְּאַף עַל פִּי שֶׁאֵינוֹ אוֹכֵל כֵּיוָן שֶׁהָיָה יָכֹל לֶאֱכֹל סַגִּי דְּהָא אָמְרִינַן הוֹאִיל כוּ'. בֵּית יוֹסֵף: {ז} וְאֵין רְצוֹנוֹ לוֹמַר שֶׁהָאֶחָד נוֹשֵׂא בְּמוֹט וְהַשֵּׁנִי בְּמוֹטָה, אֶלָּא שֶׁשְּׁנֵיהֶם נוֹשְׂאִים בְּמוֹט כְּאַחַת אוֹ בְּמוֹטָה: (עיקר תוי"ט).

פרק ג - משנה ד

בְּכוֹר שֶׁנָּפַל לְבוֹר, רַבִּי יְהוּדָה אוֹמֵר, יֵרֵד מֻמְחֶה וְיִרְאֶה, אִם יֶשׁ בּוֹ מוּם, יַעֲלֶה וְיִשְׁחֹט. וְאִם לָאו, לֹא יִשְׁחֹט. רַבִּי שִׁמְעוֹן אוֹמֵר, כֹּל שֶׁאֵין מוּמוֹ נִכָּר מִבְּעוֹד יוֹם, אֵין זֶה מִן הַמּוּכָן:

 ר"ע מברטנורה  בְּכוֹר שֶׁנָּפַל לְבוֹר. בְּכוֹר בַּעַל מוּם שֶׁלֹּא הֶרְאָהוּ לְחָכָם מֵעֶרֶב יוֹם טוֹב מִבְּעוֹד יוֹם לְהַתִּירוֹ, וְנָפַל לְבוֹר בְּיוֹם טוֹב וְיָרֵא שֶׁמָּא יָמוּת שָׁם: יֵרֵד מֻמְחֶה. שֶׁבָּקִי לְהַבְחִין בֵּין מוּם קָבוּעַ לְמוּם עוֹבֵר: וְיִרְאֶה. מוּם שֶׁהָיָה בּוֹ אֶתְמוֹל, אִם מוּם קָבוּעַ הוּא: יַעֲלֶה וְיִשְׁחֹט. דְּמִשּׁוּם מֻקְצֶה לֵיכָּא, דְּמֵאֶתְמוֹל דַּעְתֵּיהּ עִילָּוֵיהּ: וְאִם לָאו לֹא יִשְׁחֹט. הָכִי קָאָמַר, וְאִם לָאו, דְּאוֹתוֹ מוּם שֶׁהָיָה בּוֹ מֵעֶרֶב יוֹם טוֹב לֹא הָיָה קָבוּעַ אֶלָּא הַיּוֹם נַעֲשָׂה בּוֹ מוּם קָבוּעַ, אֲפִלּוּ עָבַר וְהֶעֱלָהוּ לֹא יִשְׁחֹט. וְלֹא אָמְרִינַן מֵאֶתְמוֹל הֲוָה דַּעְתֵּיהּ עִילָּוֵיהּ וְכֵיוָן שֶׁיֵּשׁ בּוֹ עַכְשָׁיו מוּם קָבוּעַ נִשְׁחָטֵיהּ, דְּהוֹאִיל וְהַמּוּם שֶׁל אֶתְמוֹל לֹא הָיָה קָבוּעַ, מֻקְצֶה מֵחֲמַת אִסּוּר הֲוָה: אֵין זֶה מִן הַמּוּכָן. לָאו מִשּׁוּם מֻקְצֶה אָסַר לֵיהּ, דְּהָא לֵית לֵיהּ לְרַבִּי שִׁמְעוֹן מֻקְצֶה, אֶלָּא מִפְּנֵי שֶׁמַּתִּירוֹ בְּיוֹם טוֹב הֲוָה לֵיהּ כִּמְתַקְּנוֹ, וְנִרְאֶה כְּדָן דִּין, דְּגָזוּר בֵּיהּ מִשּׁוּם שְׁבוּת. וְהָכִי קָאָמַר, אֵין הַתָּרָתוֹ הֶתֵּר {ח}, וְאֵינוֹ מוּכָן לִהְיוֹת כָּשֵׁר: (ר"ע מברטנורה).

 עיקר תוי"ט  {ח} מַשְׁמַע דַּאֲפִלּוּ בְּעָבַר וְדָן וְכֵן מַשְׁמַע לִישְׁנָא דְּמַתְנִיתִין. וְקַשְׁיָא לִי כוּ' דַּהֲרֵי אִם עָבַר וְדָן דִּין מִי לֹא הֲוָה דִינֵיהּ דִּין, וּכְמוֹ בְּאִינָךְ דִּתְנִינַן נַמִּי גַבֵּי דְלָא דָּנִין לְקַמָּן בְּפֶרֶק ה', דְּוַדַּאי אִם קִדֵּשׁ דִּמְקֻדֶּשֶׁת כוּ'. וְהַתּוֹסָפוֹת כָּתְבוּ דְאָסוּר מִשּׁוּם מֻקְצֶה אַף לְרַבִּי שִׁמְעוֹן, דְּלֹא הָוֵי דַעְתֵּיהּ עִלָּוֵיהּ, וְכָתַב הָרַ"ן מִשּׁוּם דְּמִי יֵימַר דְּמִזְדַּקֵּק לֵיהּ חָכָם. תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ג - משנה ה

בְּהֵמָה שֶׁמֵּתָה, לֹא יְזִיזֶנָּה מִמְּקוֹמָהּ. וּמַעֲשֶׂה וְשָׁאֲלוּ אֶת רַבִּי טַרְפוֹן עָלֶיהָ וְעַל הַחַלָּה שֶׁנִּטְמְאָה, וְנִכְנַס לְבֵית הַמִּדְרָשׁ וְשָׁאַל, וְאָמְרוּ לוֹ, לֹא יְזִיזֵם מִמְּקוֹמָם:

 ר"ע מברטנורה  בְּהֵמָה שֶׁמֵּתָה. בְּבֶהֱמַת קָדָשִׁים מַיְרֵי, כִּדְקָתָנֵי עֲלֵיהּ וְעַל הַחַלָּה שֶׁנִּטְמֵאת. וּבֶהֱמַת קָדָשִׁים שֶׁמֵּתָה אֲסוּרָה בַּהֲנָאָה וּטְעוּנָה קְבוּרָה, הִלְכָּךְ לֹא יְזִיזֶנָּה מִמְּקוֹמָהּ. אֲבָל בֶּהֱמַת חֻלִּין מְחַתְּכִין אוֹתָהּ לִפְנֵי הַכְּלָבִים. וְהָנֵי מִלֵּי שֶׁהָיְתָה מְסֻכֶּנֶת מֵאֶתְמוֹל, דְּדַעְתֵּיהּ עִילָּוָהּ, אֲבָל לֹא הָיְתָה מְסֻכֶּנֶת מֵאֶתְמוֹל אַף בֶּהֱמַת חֻלִּין לֹא יְזִיזֶנָּה מִמְּקוֹמָהּ {ט}: הַחַלָּה שֶׁנִּטְמֵאת. אֵינָהּ רְאוּיָה לְכֹהֵן בְּיוֹם טוֹב, אַף לַהַסָּקָה אוֹ לְתִתָּהּ לְכַלְבּוֹ, דְּאֵין מְבַעֲרִין קָדָשִׁים מִן הָעוֹלָם בְּיוֹם טוֹב, וַאֲפִלּוּ עַל יְדֵי אֲכִילַת בְּהֵמָה, דִּגְזֵרַת הַכָּתוּב הִיא שֶׁאֵין קָדָשִׁים טְמֵאִים מִתְבַּעֲרִים מִן הָעוֹלָם בְּיוֹם טוֹב: (ר"ע מברטנורה).

 עיקר תוי"ט  {ט} דִּבְיוֹם טוֹב קַיְמָא לָן כְּרַבִּי יְהוּדָה דְמֻקְצֶה אָסוּר. וּבִמְסֻכֶּנֶת מִיהָא מוֹדֶה רַבִּי יְהוּדָה דְּשָׁרֵי: (עיקר תוי"ט).

פרק ג - משנה ו

אֵין נִמְנִין עַל הַבְּהֵמָה לְכַתְּחִלָּה בְּיוֹם טוֹב, אֲבָל נִמְנִין {יא} עָלֶיהָ מֵעֶרֶב יוֹם טוֹב וְשׁוֹחֲטִין וּמְחַלְּקִין בֵּינֵיהֶן. רַבִּי יְהוּדָה אוֹמֵר, שׁוֹקֵל אָדָם בָּשָׂר כְּנֶגֶד הַכְּלִי אוֹ כְנֶגֶד הַקּוֹפִיץ. וַחֲכָמִים אוֹמְרִים, אֵין מַשְׁגִּיחִין בְּכַף מֹאזְנַיִם כָּל עִקָּר:

 ר"ע מברטנורה  אֵין נִמְנִין. אֵין פּוֹסְקִים דָּמִים לִהְיוֹת שְׁנַיִם שְׁלֹשָׁה בְּנֵי אָדָם אוֹמְרִים בְּהֵמָה זוֹ בִּשְׁלֹשָׁה זוּזִים, זוּז לְכָל אֶחָד {י}: וְשׁוֹחֲטִים וּמְחַלְּקִים בֵּינֵיהֶן. לְמֶחֱצָה לִשְׁלִישׁ וְלִרְבִיעַ, וְאֵין מַזְכִּירִין פִּסּוּק דָּמִים: שׁוֹקֵל אָדָם בָּשָׂר. אַף עַל פִּי שֶׁאָסוּר לִשְׁקֹל בְּלִיטְרָא, דַּהֲוֵי מַעֲשֵׂה חֹל, מֻתָּר לִשְׁקֹל כְּנֶגֶד הַכְּלִי וּכְנֶגֶד הַקּוֹפִיץ, וּלְמָחָר יִרְאוּ הַקּוֹפִיץ וְהַכְּלִי כַּמָּה מִשְׁקָלוֹ: חֲכָמִים אוֹמְרִים אֵין מַשְׁגִּיחִין. אֵין מְעַיְּנִין: בְּכַף מֹאזְנַיִם כָּל עִקָּר. וַאֲפִלּוּ לִשְׁמֹר הַבָּשָׂר מִן הָעַכְבָּרִין אָסוּר לָשׂוּם אוֹתוֹ בְּכַף מֹאזְנַיִם {יב}. וַהֲלָכָה כַּחֲכָמִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {י} וְכָתַב רַשִׁ"י דְמִקַּח וּמִמְכָּר בְּשַׁבָּת וְיוֹם טוֹב אָסוּר, בְּסֵפֶר עֶזְרָא ב' (הַיְנוּ נְחֶמְיָה) י"ג. וּבְדַף ל"ז פֵּרֵשׁ מִדִּכְתִיב מִמְּצוֹא חֶפְצְךָ וְדַבֵּר דָּבָר [יְשַׁעְיָה נח], אִי נַמִּי מִקַּח וּמִמְכָּר אָתֵי לִידֵי כְּתִיבַת שִׁטְרֵי מְכִירָה. וְכֻלָּהּ חֲדָא גְּזֵרָה הוּא: {יא} נִמְנִין. כְּלוֹמַר וּבְהָכִי יְכוֹלִין לְמָחָר לְחַלֵּק בֵּינֵיהֶם: {יב} דְּמֵחֲזֵי כְּנוֹתְנוֹ לְשֵׁם מִשְׁקָל. רַשִׁ"י: (עיקר תוי"ט).

פרק ג - משנה ז

אֵין מַשְׁחִיזִין אֶת הַסַּכִּין בְּיוֹם טוֹב, אֲבָל מַשִּׂיאָהּ עַל גַּבֵּי חֲבֶרְתָּהּ. לֹא יֹאמַר אָדָם לַטַּבָּח, שְׁקוֹל לִי בְדִינָר {יג} בָּשָׂר. אֲבָל שׁוֹחֵט וּמְחַלְּקִים בֵּינֵיהֶן {יד}:

 ר"ע מברטנורה  אֵין מַשְׁחִיזִין אֶת הַסַּכִּין. בְּרֵחַיִם אוֹ בְּמַשְׁחֶזֶת: אֲבָל מַשִּׂיאָהּ עַל גַּבֵּי חֲבֶרְתָּהּ. דִּמְשַׁנֶּה מִדֶּרֶךְ חֹל: (ר"ע מברטנורה).

 עיקר תוי"ט  {יג} בְּדִינָר. שֶׁאָסוּר לְהַזְכִּיר שֵׁם דָּמִים. רַשִׁ"י. וּתְמִיהָנִי דְמִשְׁנָה שֶׁאֵינָהּ צְרִיכָה הִיא, דִּכְבָר נִשְׁנָה לְעֵיל אֵין נִמְנִין: {יד} וּמְחַלְּקִין בֵּינֵיהֶן. שֶׁיֹּאמַר תֵּן לִי חֵלֶק אוֹ חֲצִי חֵלֶק, כְּדֶרֶךְ שֶׁרְגִילִין לְנַתְּחָן בַּמָּקוֹם הַהוּא, וּמֵבִיא שְׁתֵּי בְּהֵמוֹת וּמַעֲמִידָן זוֹ אֵצֶל זוֹ וְאוֹמֵר זוֹ כָּזוֹ, וּלְמָחָר שָׁמִין הַנִּשְׁאֶרֶת כְּפִי מַה שֶּׁשָּׁוָה כָּךְ יִפְרַע לוֹ. גְּמָרָא: (עיקר תוי"ט).

פרק ג - משנה ח

אוֹמֵר אָדָם לַחֲבֵרוֹ, מַלֵּא לִי כְלִי זֶה, אֲבָל לֹא בַמִּדָּה. רַבִּי יְהוּדָה אוֹמֵר, אִם הָיָה כְלִי שֶׁל מִדָּה, לֹא יְמַלְאֶנּוּ. מַעֲשֶׂה בְאַבָּא שָׁאוּל בֶּן בָּטְנִית, שֶׁהָיָה מְמַלֵּא מִדּוֹתָיו מֵעֶרֶב יוֹם טוֹב וְנוֹתְנָן לַלָּקוֹחוֹת בְּיוֹם טוֹב. אַבָּא שָׁאוּל אוֹמֵר, אַף בַּמּוֹעֵד עוֹשֶׂה כֵן, מִפְּנֵי בֵרוּרֵי הַמִּדּוֹת. וַחֲכָמִים אוֹמְרִים, אַף בְּחֹל עוֹשֶׂה כֵן, מִפְּנֵי מִצּוּי הַמִּדּוֹת. הוֹלֵךְ אָדָם אֵצֶל חֶנְוָנִי הָרָגִיל אֶצְלוֹ {יח}, וְאוֹמֵר לוֹ, תֵּן לִי בֵּיצִים וֶאֱגוֹזִים בְּמִנְיָן, שֶׁכֵּן דֶּרֶךְ בַּעַל הַבַּיִת לִהְיוֹת מוֹנֶה {יט} בְּתוֹךְ בֵּיתוֹ:

 ר"ע מברטנורה  אֲבָל לֹא בַמִּדָּה. בִּכְלִי הַמְּיֻחָד לְמִדָּה שֶׁמּוֹדֵד וּמוֹכֵר בּוֹ. אֲבָל אִם אֵינוֹ מְיֻחָד לְכָךְ, אַף עַל פִּי שֶׁעוֹמֵד לְמִדָּה, שֶׁכְּשֶׁיִּשָּׁבֵר זֶה שֶׁהוּא מוֹדֵד בּוֹ יַעֲמֹד זֶה תַּחְתָּיו, מֻתָּר לְמַלְאוֹתוֹ {טו}: אִם הָיָה כְלִי שֶׁל מִדָּה. אַף עַל פִּי שֶׁעֲדַיִן לֹא יִחֲדוֹ לְכָךְ, לֹא יְמַלְּאֶנּוּ {טז}. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: מְמַלֵּא מִדּוֹתָיו מֵעֶרֶב יוֹם טוֹב. לְפִי שֶׁאֵין מוֹדְדִין בְּיוֹם טוֹב: אַף בְּחֹל הַמּוֹעֵד עוֹשֶׂה כֵן. מִפְּנֵי בִּטּוּל בֵּית הַמִּדְרָשׁ, שֶׁהָיוּ רַבִּים בָּאִים לִשְׁאֹל הֵימֶנּוּ בַּמּוֹעֵד, שֶׁאֵין טְרוּדִין בִּמְלָאכָה, וְהָיָה מְמַלֵּא מִדּוֹתָיו בַּלַּיְלָה שֶׁאֵינוֹ זְמַן בֵּית הַמִּדְרָשׁ {יז} כְּדֵי שֶׁיְּהֵא פָּנוּי בַּיּוֹם: מִפְּנֵי מִצּוּי הַמִּדּוֹת. כְּשֶׁהָיָה מוֹכֵר שֶׁמֶן הָיוּ לוֹ מִדּוֹת הַרְבֵּה, וּמְבִיאִין הַלָּקוֹחוֹת כְּלֵיהֶן וּמוֹדֵד לְכָל אֶחָד וְאֶחָד בְּמִדָּה לְעַצְמוֹ, וּמִתְמַצּוֹת וְהוֹלְכוֹת לְתוֹךְ כְּלֵיהֶן כָּל הַלַּיְלָה, כְּדֵי שֶׁלֹּא יִשָּׁאֵר לוֹ הַשֶּׁמֶן הַמֻּדְבָּק בְּשׁוּלֵי הַמִּדָּה וּבְדָפְנוֹתֶיהָ וְנִמְצָא גּוֹזֵל אֶת הַלָּקוֹחוֹת: (ר"ע מברטנורה).

 עיקר תוי"ט  {טו} דְּהוֹאִיל וְאֵינוֹ מְיֻחָד לֹא קָא עָבֵיד כִּדְעַבְדִין בְּחֹל שֶׁדֶּרֶךְ לָקוֹחוֹת לוֹמַר תֵּן לִי לֹג וְאֵין דַּרְכָּן לוֹמַר כְּלִי זֶה מַלֵּא לִי בְּדֶרֶךְ מֵקַּח אֶלָּא בְּדֶרֶךְ הַלְוָאָה אוֹ מַתָּנָה. וְאַף עַל גַּב דִּמְיֻחָד לְמִדָּה מַה שֶּׁאֵין כֵּן בְּמִשְׁנָה ו' דְּאָסְרִי לִשְׁקֹל בַּכְּלִי מִשּׁוּם דְּקָא עֲבִיד עוֹבָדִין דְּחֹל דִּפְעָמִים שֶׁאֵין הַלִּיטְרָא לְפָנָיו וְכֵיוָן שֶׁיּוֹדֵעַ מִשְׁקַל כֵּלָיו שׁוֹקֵל בּוֹ. גְּמָרָא: {טז} הוֹאִיל וְעוֹמֵד לְמִדָּה, מַה שֶּׁאֵין כֵּן בְּמִשְׁנָה ו' דְּמַתִּיר לִשְׁקֹל כְּנֶגֶד כְּלִי לְפִי שֶׁאֵינוֹ עוֹמֵד לִשְׁקֹל בּוֹ. גְּמָרָא: {יז} וּלְמַאן דְּגָרַס מִפְּנֵי בֵּרוּרֵי הַמִּדּוֹת הָכִי פֵּרוּשׁוֹ, מִשּׁוּם בִּטּוּל בֵּית הַמִּדְרָשׁ שֶׁל בָּאֵי מוֹעֵד לֹא יִהְיֶה פָּנוּי לִשְׁהוֹת וּלְבָרֵר מִדּוֹתָיו שֶׁלֹּא יַרְתִּיחוּ, לְפִיכָךְ מְמַלְּאָן בַּלַּיְלָה. וְטַעְמָא שֶׁרָצָה שֶׁלֹּא יַרְתִּיחוּ, לְפִי שֶׁעַל יְדֵי הָרְתִיחוֹת שֶׁמַּעֲלֶה הַיַּיִן נִמְצָא שֶׁאֵין הַמִּדָּה מָלֵא: {יח} חֶנְוָנִי כוּ'. שֶׁמִּתּוֹךְ שֶׁרָגִיל אֶצְלוֹ מַאֲמִינוֹ וְנוֹתְנוֹ לוֹ בְּלֹא פִסּוּק דָּמִים. רַשִׁ"י: {יט} מוֹנֶה. הִלְכָּךְ לֹא מוּכָח דְּמִשּׁוּם דָּמִים הוּא דְּמִדְּכַּר לֵיהּ מִנְיָן, אֶלָּא שֶׁכָּךְ צָרִיךְ. הָרַ"ן: (עיקר תוי"ט).

פרק ד - משנה א

הַמֵּבִיא כַדֵּי יַיִן מִמָּקוֹם לְמָקוֹם, לֹא יְבִיאֵם בְּסַל וּבְקֻפָּה, אֲבָל מֵבִיא הוּא עַל כְּתֵפוֹ אוֹ לְפָנָיו. וְכֵן הַמּוֹלִיךְ אֶת הַתֶּבֶן, לֹא יַפְשִׁיל אֶת הַקֻּפָּה לַאֲחוֹרָיו, אֲבָל מְבִיאָהּ הוּא בְיָדוֹ. וּמַתְחִילִין בַּעֲרֵמַת הַתֶּבֶן, אֲבָל לֹא בָעֵצִים שֶׁבַּמֻּקְצֶה:

 ר"ע מברטנורה  הַמֵּבִיא. מִמָּקוֹם לְמָקוֹם. בְּתוֹךְ הַתְּחוּם {א}: לֹא יְבִיאֵם וְכוּ'. לָתֵת שָׁלֹשׁ וְאַרְבַּע כַּדּוֹת בְּתוֹךְ סַל אוֹ קֻפָּה וְיִשָּׂאֵם, מִשּׁוּם דְּנִרְאֶה כְּמַעֲשֵׂה חֹל לָשֵׂאת מַשָּׂאוֹת. וְאִם אִי אֶפְשָׁר לוֹ לְשַׁנּוֹת מֻתָּר: אֲבָל מֵבִיא הוּא עַל כְּתֵפוֹ. אֶחָד אוֹ שְׁנַיִם, דְּמוּכַח דִּלְצֹרֶךְ יוֹם טוֹב {ב}: אוֹ לְפָנָיו. בְּיָדוֹ: וְכֵן הַמּוֹלִיךְ אֶת הַתֶּבֶן. לְהֶסֵּק אוֹ לִבְהֵמָה: לֹא יַפְשִׁיל קֻפָּה לַאֲחוֹרָיו. שֶׁנִּרְאֶה כְּדֶרֶךְ חֹל: וּמַתְחִילִין בַּעֲרֵמַת הַתֶּבֶן. וְאַף עַל פִּי שֶׁלֹּא זִמְּנָהּ מִבְּעוֹד יוֹם וְלֹא הָיָה רָגִיל לְהַסִּיק מִמֶּנָּה, דְּלֵית לֵיהּ לְהַאי תַּנָּא מֻקְצֶה: אֲבָל לֹא בָעֵצִים [שֶׁבַּמֻּקְצֶה]. רְחָבָה שֶׁאֲחוֹרֵי הַבָּתִּים קְרוּיָה מֻקְצֶה, עַל שֵׁם שֶׁהִיא מֻקְצָה לְאָחוֹר וְאֵין נִכְנָסִים וְיוֹצְאִים לָהּ תָּדִיר {ג}. וְהָנָךְ עֵצִים דְּאָיְרֵי בְּהוּ הָכָא הֵן קוֹרוֹת גְּדוֹלוֹת שֶׁל אֲרָזִים הָעוֹמְדוֹת לְבִנְיָן, דְּמֻקְצֶה מֵחֲמַת חֶסְרוֹן כִּיס נִינְהוּ, שֶׁדְּמֵיהֶן יְקָרִים, וּבְהָא אֲפִלּוּ רַבִּי שִׁמְעוֹן דְּלֵית לֵיהּ מֻקְצֶה מוֹדֶה: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} וּמְסַיֵּם רַשִׁ"י אוֹ עַל יְדֵי עֵרוּב. וְכָתַב בֵּית יוֹסֵף דְּמַשְׁמַע מִדְּבָרָיו דִּבְמוֹלִיךְ הַכַּדִּים בַּחוּץ בִּמְקוֹם הָרוֹאִים דַּוְקָא הוּא דְּאָמְרִינַן דְּלֹא יְבִיאֵם בְּסַל וּבְקֻפָּה, אֲבָל לְטַלְטֵל בִּמְקוֹם שֶׁאֵין רוֹאִין כְּגוֹן מִזָּוִית לְזָוִית בְּאוֹתוֹ בַּיִת, מֵבִיא כִּי הֵיכִי דְבָעֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} אוֹרְחָא דְמִלְּתָא נָקַט, דְּעַל כְּתֵפוֹ מֵבִיא אֶחָד אוֹ שְׁנַיִם, וּבְסַל וּבְקֻפָּה מֵבִיא שְׁלֹשָׁה וְאַרְבָּעָה, וְהוּא הַדִּין אִיפְּכָא, דִּלְעוֹלָם אִסּוּרָא בְּסַל וְקֻפָּה וְהַהֶתֵּר בִּכְתֵפוֹ, דִּבְשִׁנּוּי תַּלְיָא מִלְּתָא, לֹא בַּמִּסְפָּר: {ג} וְשָׁם נוֹתְנִין עֵצִים וְכָל דָּבָר שֶׁאֵין דַּעְתּוֹ לִטֹּל עַד זְמַן מְרֻבֶּה. רַשִׁ"י: (עיקר תוי"ט).

פרק ד - משנה ב

אֵין נוֹטְלִין עֵצִים מִן הַסֻּכָּה, אֶלָּא מִן הַסָּמוּךְ לָהּ. מְבִיאִין עֵצִים מִן הַשָּׂדֶה מִן הַמְכֻנָּס {ו}, וּמִן הַקַּרְפֵּף אֲפִלּוּ מִן הַמְפֻזָּר. אֵיזֶהוּ קַרְפֵּף, כֹּל שֶׁסָּמוּךְ לָעִיר, דִּבְרֵי רַבִּי יְהוּדָה. רַבִּי יוֹסֵי אוֹמֵר, כֹּל שֶׁנִּכְנָסִין לוֹ בְפוֹתַחַת, וַאֲפִלּוּ בְתוֹךְ תְּחוּם שַׁבָּת:

 ר"ע מברטנורה  אֵין נוֹטְלִין עֵצִים מִן הַסֻּכָּה. אֲפִלּוּ סֻכָּה שֶׁאֵינָהּ שֶׁל מִצְוָה {ד} כְּגוֹן בְּפֶסַח אוֹ בַּעֲצֶרֶת, אֵין נוֹטְלִים מִמֶּנָּה עֵצִים בְּיוֹם טוֹב, מִשּׁוּם סְתִירַת אֹהֶל: אֶלָּא מִן הַסָּמוּךְ לָהּ. כְּגוֹן קָנִים הַזְּקוּפִים סָמוּךְ לַדְּפָנוֹת וְלֹא נֶאֶרְגוּ עִם הַדֹּפֶן, דְּלֹא בָּטְלוּ לְגַבֵּי דֹּפֶן {ה}, לְפִיכָךְ נוֹטְלִים מֵהֶן בְּיוֹם טוֹב: מְבִיאִין עֵצִים. תְּלוּשִׁים מִן הַשָּׂדֶה שֶׁבְּתוֹךְ הַתְּחוּם, מִן הַמְּכֻנָּס: וּמִן הַקַּרְפֵּף אֲפִלּוּ מִן הַמְּפֻזָּר. מַתְנִיתִין יְחִידָאָה הִיא וְאֵינָהּ הֲלָכָה, אֶלָּא הִלְכָתָא אֵין מְבִיאִין עֵצִים אֶלָּא מִן הַמְּכֻנָּסִים שֶׁבַּקַרְפֵּף. וּמִן הַשָּׂדֶה לֹא יָבִיא כְּלָל, וַאֲפִלּוּ מִן הַמְּכֻנָּסִים, דְּלָאו דַּעְתֵּיהּ עִלָּוָיְהוּ הוֹאִיל וְאֵינָן מִשְׁתַּמְּרִים שָׁם. אֲבָל מִן הַקַּרְפֵּף שֶׁהוּא מִשְׁתַּמֵּר וּמֻקָּף סָבִיב, כְּשֶׁהֵן מְכֻנָּסִין דַּעְתֵּיהּ עִלָּוָיְהוּ: כֹּל שֶׁסָּמוּךְ לָעִיר. מַמָּשׁ. וְהוּא דְּאִית לֵיהּ פּוֹתַחַת מִפֶּתַח שֶׁמִּשְׁתַּמֵּר {ז}, דְּרַבִּי יְהוּדָה תַּרְתֵּי בָעֵי, סָמוּךְ לָעִיר וּפוֹתַחַת. וּמִשּׁוּם דְּקָסָבַר סְתָם קַרְפִּיפוֹת יֵשׁ לָהֶם פּוֹתַחַת, מִשּׁוּם הָכִי לֹא חָשׁ לְהַזְכִּירָהּ: רַבִּי יוֹסֵי אוֹמֵר כֹּל שֶׁנִּכְנָסִין לוֹ בְּפוֹתַחַת, וַאֲפִלּוּ בְתוֹךְ תְּחוּם שַׁבָּת. כֵּיוָן דְּאִית לֵיהּ פּוֹתַחַת לֹא בָּעִינַן סָמוּךְ, אֶלָּא אֲפִלּוּ רָחוֹק עַד קָרוֹב לְסוֹף תְּחוּם שַׁבָּת. וְאִם הוּא סָמוּךְ לֹא בָּעִינַן פּוֹתַחַת. וּבְסָמוּךְ בְּלֹא פּוֹתַחַת אוֹ פּוֹתַחַת בְּלֹא סָמוּךְ שָׁרֵי. וַהֲלָכָה כְּרַבִּי יוֹסֵי {ח}: (ר"ע מברטנורה).

 עיקר תוי"ט  {ד} דִּבְשֶׁל מִצְוָה אָסוּר לֵהָנוֹת כָּל יְמֵי הַסֻּכּוֹת כִּדְיָלְפִינַן בַּגְּמָרָא: {ה} וְאֵינוֹ דּוֹמֶה לַסָּמוּךְ לַסְּכָךְ, שֶׁהַסְּכָךְ אֵינוֹ אָרוּג, לְפִיכָךְ הַכֹּל שָׁוֶה הָעֶלְיוֹן כַּתַּחְתּוֹן סְכָךְ הוּא. רַשִׁ"י: {ו} הַמְכֻנָּס. אִם כִּנְסָן מֵעֶרֶב יוֹם טוֹב, דְּגָלֵי דַעְתֵּיהּ דַּעֲלַיְיהוּ סָמַךְ. רַשִׁ"י. וְלָאו דַוְקָא מֵעֶרֶב יוֹם טוֹב, אֶלָּא כָּל שֶׁהֵם מְכֻנָּסִים קֹדֶם יוֹם טוֹב דַּעְתֵּיהּ עִלָּוֵיהּ: {ז} וְטַעְמָא מִשּׁוּם דְּכָל שֶׁיֵּשׁ לוֹ מִסְגֶּרֶת מֵחֲזֵי כְּחָצֵר וְלֹא הָוֵי עֵצִים שֶׁבְּתוֹכוֹ כְּמֻקְצִים. בֵּית יוֹסֵף: {ח} וּשְׁמַע מִנָּהּ דְּבָעִינַן מִיהַת שֶׁלֹּא יְהוּ מֻקְצִים דְּהָא חֲדָא מִתַּרְתֵּי מִיהוּ בָעֵי רַבִּי יוֹסֵי: (עיקר תוי"ט).

פרק ד - משנה ג

אֵין מְבַקְּעִין עֵצִים, לֹא מִן הַקּוֹרוֹת, וְלֹא מִן הַקּוֹרָה שֶׁנִּשְׁבְּרָה בְיוֹם טוֹב. וְאֵין מְבַקְּעִין לֹא בְקַרְדֹּם {יא} וְלֹא בִמְגֵרָה וְלֹא בְמַגָּל {יג}, אֶלָּא בְקוֹפִיץ. בַּיִת שֶׁהוּא מָלֵא פֵרוֹת, סָתוּם וְנִפְחַת, נוֹטֵל מִמְּקוֹם הַפְּחָת. רַבִּי מֵאִיר אוֹמֵר, אַף פּוֹחֵת לְכַתְּחִלָּה וְנוֹטֵל:

 ר"ע מברטנורה  אֵין מְבַקְּעִין עֵצִים מִן הַקּוֹרוֹת. הַסְּדוּרוֹת בָּאָרֶץ, כְּדֵי שֶׁלֹּא יִתְעַקְּמוּ, וְעוֹמְדוֹת לְבִנְיָן {ט}: וְלֹא מִן הַקּוֹרָה שֶׁנִּשְׁבְּרָה בְיוֹם טוֹב. וְאַף עַל גַּב דְּהַשְׁתָּא לַהַסָּקָה קַיְמָא, בֵּין הַשְּׁמָשׁוֹת לָאו לְהָכִי קַיְמָא {י}: וְאֵין מְבַקְּעִין לֹא בְקַרְדֹּם. מַתְנִיתִין חִסּוּרֵי מִחַסְּרָא וְהָכִי קָתָנֵי, אֲבָל מְבַקְּעִין מִן הַקּוֹרָה {יב} שֶׁנִּשְׁבְּרָה מֵעֶרֶב יוֹם טוֹב, וּכְשֶׁהֵן מְבַקְּעִין אֵין מְבַקְּעִין לֹא בַקַּרְדֹּם: וְלֹא בִמְגֵרָה. כְּעֵין סַכִּין אָרֹךְ מָלֵא פְּגִימוֹת וְקוֹצְצִין בּוֹ עֵצִים עָבִים. וּכְלִי אֻמָּן הוּא: מַגָּל. אַף הוּא כְּלִי אֻמָּן, וְנִרְאֶה כְּרוֹצֶה לַעֲשׂוֹת מְלָאכָה: אֶלָּא בְקוֹפִיץ. סְתָם קוֹפִיץ הוּא סַכִּין שֶׁל קַצָּבִים, וְאֵינוֹ כְּלִי אֻמָּן. וְיֵשׁ מֵהֶן שֶׁיֵּשׁ לָהֶן שְׁנֵי רָאשִׁין, רֹאשׁ אֶחָד רָחָב וְנִקְרָא צַד נַקְבוּת, וְרֹאשׁ אֶחָד צָר וְנִקְרָא צַד זַכְרוּת, וְאֵין מְבַקְּעִין בּוֹ אֶלָּא בְּצַד זַכְרוּת שֶׁלּוֹ {יד}: נוֹטֵל מִמְּקוֹם הַפְּחָת. וְלֹא אָמְרִינַן מֻקְצִים מֵחֲמַת אִסּוּר הֵן דְּאֵין יָכֹל לְפָחֲתוֹ בְּיוֹם טוֹב וַאֲסַח דַּעְתֵּיהּ מִנַּיְהוּ, דְּבַיִת דִּתְנַן בְּמַתְנִיתִין מַיְרֵי שֶׁאֵינוֹ בָּנוּי בְּטִיט וּבְסִיד אֶלָּא סָדוּר שֶׁל לְבֵנִים זוֹ עַל זוֹ בְּלֹא טִיט, וְהַשְׁתָּא אֵין בִּפְחִיתָתוֹ אִסּוּרָא דְּאוֹרַיְתָא {טו}, הִלְכָּךְ לָאו מֻקְצִים נִינְהוּ, כִּי הֵיכִי דְאָמְרִינַן גַּבֵּי טֶבֶל דְּלָאו מֻקְצֶה הוּא שֶׁאִם עָבַר וְתִקְּנוֹ מְתֻקָּן: אַף פּוֹחֵת לְכַתְּחִלָּה. דְּהוֹאִיל וְאֵין שָׁם טִיט אֶלָּא לְבֵנִים סְדוּרִין זוֹ עַל זוֹ, לֹא הָוֵי סוֹתֵר, וְשָׁרֵי לִפְחוֹת לְכַתְּחִלָּה. וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר: (ר"ע מברטנורה).

 עיקר תוי"ט  {ט} וַאֲפִלּוּ לְרַבִּי שִׁמְעוֹן, וְטַעְמָא מִפְּנֵי שֶׁהֵם מֻקְצִים מֵחֲמַת חֶסְרוֹן כִּיס דְּמוֹדֶה רַבִּי שִׁמְעוֹן. בֵּית יוֹסֵף: {י} וְכֵיוָן דְּאִתְקְצַאי לְבֵין הַשְּׁמָשׁוֹת אִתְקְצַאי לְכֻלֵּי יוֹמָא. רַשִׁ"י. וּשְׁמַע מִנָּהּ דְּמַתְנִיתִין כְּרַבִּי יְהוּדָה דְּמֻקְצֶה אָסוּר, וְאַף עַל גַּב דִּבְשַׁבָּת [פֶּרֶק כד מִשְׁנָה ד] סָתִים לָן כְּרַבִּי שִׁמְעוֹן (עַיֵּן שָׁם), הַיְנוּ טַעְמָא דְּשַׁבָּת חֲמִירָא וְלֹא אָתְיָא לְזִלְזוּלֵי בֵּיהּ, סָתַם כְּרַבִּי שִׁמְעוֹן דְּמֵקֵל. יוֹם טוֹב דְּקִיל וְאָתוּ לְזִלְזוּלֵי בֵּיהּ, סָתַם כְּרַבִּי יְהוּדָה דְּמַחְמִיר. גְּמָרָא: {יא} בְּקַרְדֹּם. סְתָם קַרְדֹּם עָשׂוּי כְּמִין כְּלִי אֻמָּנוּת שֶׁעוֹשִׂין בּוֹ אֻכָּפוֹת וְסָרָגוֹת כוּ', וְיֵשׁ לוֹ שְׁתֵּי פִיּוֹת. רַשִׁ"י: {יב} וְכָתַב הָרַמְבַּ"ם וְלָמָּה לֹא נֶאֱסַר הַבִּקּוּעַ כְּלָל, מִפְּנֵי שֶׁאֶפְשָׁר שֶׁיִּפְגַּע בְּעֵץ עָבֶה וְלֹא יוּכַל לְהַבְעִירוֹ וְיִמָּנַע מִלְּבַשֵּׁל לְפִיכָךְ הִתִּירוּ לְבַקֵּעַ בְּשִׁנּוּי. וְכָל הַדְּבָרִים הַדּוֹמִין לָזֶה, מִזֶּה הַטַּעַם הִתִּירוּ מַה שֶּׁהִתִּירוּ וְאָסְרוּ מַה שֶּׁאָסְרוּ. וְהָרַאֲבַ"ד כָּתַב כֵּיוָן דְּאִי אֶפְשָׁר לְבַשֵּׁל וְלֶאֱפוֹת בְּלֹא עֵצִים, עָשׂוּ הַבְּקִיעָה כְּדִיכַת מֶלַח שֶׁהִתִּירוּ אוֹתוֹ עַל יְדֵי שִׁנּוּי: {יג} מַגָּל. פֵּרֵשׁ בָּעָרוּךְ מַגָּל יָד עָשׂוּי לְשַׁבֵּר עֲצָמוֹת וְעֵצִים, וְהוּא חָלָק: {יד} שֶׁזֶּה הַצַּד עָשׂוּי לְבַקֵּעַ בּוֹ עֵצִים, וְצַד הָרָחָב קוֹצְצִין בּוֹ הָעֵץ לְרָחְבּוֹ, דְּדָמֵי לִמְלֶאכֶת הָאֻמָּן. רַשִׁ"י: {טו} וְהִקְשׁוּ בַּתּוֹסָפוֹת מִמָּעוֹת שֶׁעַל הַכַּר דְּמִשְׁנָה ב' פֶּרֶק כ"א דְשַׁבָּת, דְּאִם הוּנְחוּ שָׁם מֵעֶרֶב שַׁבָּת הָוֵי בָּסִיס וְאָסוּר, וְאַף עַל גַּב דְּלֹא הֲוָה אֶלָּא מֻקְצֶה מִדְּרַבָּנָן. וְתֵרְצוּ הָרַמְבַּ"ן וְהָרַ"ן דְּהָכָא הַפֵּרוֹת עַצְמָן רְאוּיִן וְלֹא נַעֲשׂוּ בָּסִיס לַבַּיִת, אֶלָּא שֶׁדָּבָר אַחֵר גּוֹרֵם שֶׁאֵינוֹ יָכֹל לֵהָנוֹת מֵהֶן, וְאֵינוֹ אֶלָּא כְּאִלּוּ הָיָה פֵּרוֹתָיו בְּמָקוֹם שֶׁאֵינוֹ יָכֹל לַהֲלוֹךְ שָׁם, וְהָכִי נַמִּי גַּבֵּי טֶבֶל: (עיקר תוי"ט).

פרק ד - משנה ד

אֵין פּוֹתְחִין אֶת הַנֵּר, מִפְּנֵי שֶׁהוּא עוֹשֶׂה כְלִי. וְאֵין עוֹשִׂין פֶּחָמִין בְּיוֹם טוֹב, וְאֵין חוֹתְכִין אֶת הַפְּתִילָה לִשְׁנַיִם. רַבִּי יְהוּדָה אוֹמֵר, חוֹתְכָהּ בָּאוּר לִשְׁתֵּי נֵרוֹת:

 ר"ע מברטנורה  אֵין פּוֹתְחִין אֶת הַנֵּר. לִטֹּל אֶחָד מִן הַבֵּיצִים שֶׁל יוֹצֵר חֶרֶשׂ וְלִתְחֹב אֶגְרוֹפוֹ לְתוֹכוֹ לַחְקֹק נֵר, מִפְּנֵי שֶׁעוֹשֶׂה כְּלִי {טז}: וְאֵין עוֹשִׂין פֶּחָמִין. דְּאִינְהוּ נַמִּי כְּלִי נִינְהוּ לְצוֹרְפֵי זָהָב, וּפְתִילָה נַמִּי כְּלִי הִיא לְהַדְלָקָה, שֶׁצְּרִיכָה עֲשִׂיָּה וְתִקּוּן: וְאֵין חוֹתְכִין אוֹתָהּ לִשְׁנָיִם. לְפִי שֶׁהוּא מְתַקֵּן כְּלִי. וּלְמַעֲכָהּ בַּיָּד מֻתָּר: חוֹתְכָהּ בָּאוּר. נוֹתֵן שְׁנֵי רָאשֵׁי הַפְּתִילָה בְּפִי שְׁתֵּי נֵרוֹת שֶׁצָּרִיךְ לְהַדְלִיקָן כְּאַחַת וּמַדְלִיק בָּאֶמְצַע, דְּהַשְׁתָּא לֹא מוּכָח דִּלְתִקּוּנָא מָנָא מְכַוֵּן אֶלָּא לְהַדְלָקָה בְּעָלְמָא. וַהֲלָכָה כְּרַבִּי יְהוּדָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {טז} הָא קָא מַשְׁמַע לָן דְּאַף עַל גַּב דְּאֵינוֹ כְּלִי לְעִנְיָן קַבָּלַת טֻמְאָה עַד שֶׁיִּצְרְפֶנּוּ בַּכִּבְשָׁן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ד - משנה ה

אֵין שׁוֹבְרִין {יז} אֶת הַחֶרֶס, וְאֵין חוֹתְכִין אֶת הַנְּיָר לִצְלוֹת בּוֹ מָלִיחַ, וְאֵין גּוֹרְפִין תַּנּוּר וְכִירַיִם, אֲבָל מְכַבְּשִׁין, וְאֵין מַקִּיפִין שְׁתֵּי חָבִיּוֹת לִשְׁפֹּת עֲלֵיהֶן אֶת הַקְּדֵרָה, וְאֵין סוֹמְכִין אֶת הַקְּדֵרָה בְבַקַּעַת, וְכֵן בְּדֶלֶת, וְאֵין מַנְהִיגִין אֶת הַבְּהֵמָה בְמַקֵּל בְּיוֹם טוֹב, וְרַבִּי אֶלְעָזָר בְּרַבִּי שִׁמְעוֹן מַתִּיר:

 ר"ע מברטנורה  וְאֵין חוֹתְכִין אֶת הַנְּיָר. שֶׁנּוֹתְנִין בּוֹ שֶׁמֶן, וְעוֹמֵד בִּפְנֵי הָאֵשׁ כָּל זְמַן שֶׁהַשֶּׁמֶן בְּתוֹכוֹ, וְאֵינוֹ נִשְׂרָף: מליח. דג מָלִיחַ. דָּג: אֵין גּוֹרְפִין תַּנּוּר וְכִירַיִם. אִם נָפַל לְתוֹכָן מִטְּפוּלָהּ שֶׁל תַּנּוּר וּמִן הַטִּיחַ, אֵין גּוֹרְפִין אוֹתָן, דִּמְתַקֵּן מָנָא הוּא. וְאָתְיָא כְּרַבָּנָן דְּאָמְרֵי מַכְשִׁירֵי אֹכֶל נֶפֶשׁ אֲסִירֵי: מְכַבְּשִׁין. מַשְׁכִּיבִין אֶת הָאֵפֶר וְאֶת הֶעָפָר כְּדֵי שֶׁיִּהְיֶה חָלָק {יח}. וְאִם אִי אֶפְשָׁר לוֹ לֶאֱפוֹת אֶלָּא אִם כֵּן גָּרַף, מֻתָּר: מַקִּיפִין. מְקָרְבִין זוֹ אֵצֶל זוֹ. כְּמוֹ אֵין מַקִּיפִין בְּרֵיאָה: לִשְׁפֹּת. לַעֲרֹךְ וּלְהוֹשִׁיב עֲלֵיהֶם קְדֵרָה {יט}, וְהָאוּר בֵּין שְׁתֵּי הֶחָבִיּוֹת. וְאָסַר לַהּ מִשּׁוּם דְּדָמֵי לְבִנְיָן {כ}: [וְאֵין סוֹמְכִין] אֶת הַקְּדֵרָה בְבַקַּעַת. דְּלֹא נִתְּנוּ עֵצִים אֶלָּא לַהַסָּקָה {כא}. וּבְקַעַת, עֵצִים שֶׁנִּתְבַּקְעוּ: וְכֵן הַדֶּלֶת. אֵין סוֹמְכִין אוֹתָהּ בִּבְקַעַת, דְּעֵצִים מֻקְצִים הֵם אֵצֶל כָּל תַּשְׁמִישׁ חוּץ מִן הַהַסָּקָה: וְאֵין מַנְהִיגִים אֶת הַבְּהֵמָה בְּמַקֵּל. מִשּׁוּם דְּנִרְאֶה כְּמוֹלִיכָהּ לִמְכֹּר בַּשּׁוּק {כב}: (ר"ע מברטנורה).

 עיקר תוי"ט  {יז} אֵין שׁוֹבְרִין. כְּשֶׁצּוֹלִין דָּגִים עַל הָאַסְכָּלָה חוֹתְכִים קָנִים אוֹ קַשׁ שִׁבָּלִין אוֹ שׁוֹבְרִים חֲרָסִים אוֹ חוֹתְכִין נְיָר כוּ' וְסוֹדְרִים עַל גַּבֵּי הָאַסְכָּלָה, מִפְּנֵי שֶׁהַמַּתֶּכֶת מִתְחַמֵּם וְשׂוֹרֵף אֶת הַדָּג. רַשִׁ"י: {יח} מְסַיֵּם רַשִׁ"י מַשְׁפִּילוֹ וּמַשְׁוֵהוּ שֶׁלֹּא יִגַּע בַּפַּת שֶׁמְּדֻבָּקִין בְּדָפְנוֹתָיו וְיִשְׂרְפֶנּוּ: {יט} וְלָשׁוֹן עִבְרִי הוּא, כְּמוֹ שְׁפֹת הַסִּיר (מְלָכִים ב ד): {כ} מִדְּרַבָּנָן, מִשּׁוּם דְּמִתְחַזֵי כְּאָהֳלָא. רַשִׁ"י: {כא} גְּמָרָא. וּפֵרֵשׁ הָרַ"ן שֶׁכֵּיוָן שֶׁאֵין מְיֻחָדִין אֶלָּא לַהַסָּקָה בִּלְבַד אֵין תּוֹרַת כְּלִי עֲלֵיהֶם וְיֵשׁ לָהֶם דִּין מֻקְצֶה גַבֵּי שְׁאָר מְלָאכוֹת: {כב} גְּמָרָא, וְכִי תֵימָא וְלָמָּה לָן הַאי טַעְמָא, תֵּיפּוּק לֵיהּ מִשּׁוּם מֻקְצֶה דְהָעֵצִים כַּנִּזְכָּר לְעֵיל. אִיכָּא לְמֵימַר נָפְקָא מִנָּהּ דַּאֲפִלּוּ אַזַמְּנֵיהּ מֵאֶתְמוֹל דְּלָאו מֻקְצֶה הוּא, אֲפִלּוּ הָכִי אָסוּר מִשּׁוּם דְּנִרְאֶה כוּ'. הָרַ"ן: (עיקר תוי"ט).

פרק ד - משנה ו

רַבִּי אֱלִיעֶזֶר אוֹמֵר, נוֹטֵל אָדָם קֵיסָם מִשֶּׁלְּפָנָיו לַחֲצֹץ בּוֹ שִׁנָּיו. וּמְגַבֵּב מִן הֶחָצֵר וּמַדְלִיק, שֶׁכָּל מַה שֶּׁבֶּחָצֵר מוּכָן הוּא. וַחֲכָמִים אוֹמְרִים, מְגַבֵּב מִשֶּׁלְּפָנָיו וּמַדְלִיק:

 ר"ע מברטנורה  מִשֶּׁלְּפָנָיו. מִמַּה שֶּׁלְּפָנָיו בַּבַּיִת: לַחֲצֹץ בּוֹ שִׁנָּיו. לִטֹּל בָּשָׂר הַחוֹצֵץ בֵּין שִׁנָּיו. וּמִשֶּׁלְּפָנָיו לָאו דַוְקָא, דִּלְרַבִּי אֱלִיעֶזֶר אַף מִן הֶחָצֵר נַמִּי מֻתָּר לִטֹּל, דְּקָאָמַר כָּל מַה שֶּׁבֶּחָצֵר מוּכָן הוּא {כג}. וְהָא דְּנָקַט מִשֶּׁלְּפָנָיו, מִשּׁוּם רַבָּנָן דִּפְלִיגֵי דַּאֲפִלּוּ מִשֶּׁלְּפָנָיו לְהַדְלִיק אֵין, לַחֲצֹץ לֹא, דְּסָבְרֵי לֹא נִתְּנוּ עֵצִים אֶלָּא לַהַסָּקָה: וַחֲכָמִים אוֹמְרִים מְגַבֵּב מִשֶּׁלְּפָנָיו וּמַדְלִיק. אֲבָל מִן הֶחָצֵר לֹא, דְּהוֹאִיל וְקִסְמִין דַּקִּין הֵן וְטוֹרֵחַ לְקוֹשְׁשָׁן וּלְגַבְּבָן, מֵאֶתְמוֹל לָאו לְהָכִי קַיְמֵי. וּדְקָאָמַרְתְּ נוֹטֵל קֵיסָם לַחֲצֹץ בּוֹ שִׁנָּיו, אֲנַן אִית לָן דְּאַף מִשֶּׁלְּפָנָיו לֹא יִטֹּל אֶלָּא לְהַדְלִיק, דְּלֹא נִתְּנוּ עֵצִים אֶלָּא לַהַסָּקָה. וּבְתַרְתֵּי פְלִיגֵי. וַהֲלָכָה כַּחֲכָמִים. וְאֵין מֻתָּר לִטֹּל קֵיסָם לַחֲצֹץ בּוֹ שִׁנָּיו אֶלָּא מֵאֵבוּס שֶׁל בְּהֵמָה. וְלֹא יְחַתְּכֶנּוּ וִיתַקְּנֶנּוּ לְכָךְ, אֶלָּא אִם כֵּן הָיָה לַח וְרָאוּי לְמַאֲכַל בְּהֵמָה, שֶׁכָּל דָּבָר הָרָאוּי לְמַאֲכַל בְּהֵמָה מֻתָּר לְחָתְכוֹ בֵּין בְּיוֹם טוֹב בֵּין בְּשַׁבָּת, וְאֵין בָּהֶם מִשּׁוּם תִּקּוּן כְּלִי: (ר"ע מברטנורה).

 עיקר תוי"ט  {כג} וְרֵישָׁא לֹא אָתֵי לְאַשְׁמוֹעִינַן אֶלָּא הֶתֵּר טִלְטוּל עֵצִים אַף שֶׁלֹּא לַהַסָּקָה. רַשִׁ"י. כְּלוֹמַר וּכְדִתְנַן בַּסֵּיפָא וּמְגַבֵּב כוּ' וּמַדְלִיק הוּא הַדִּין שֶׁלֹּא לְהַדְלִיק אֶלָּא לַחֲצֹץ נָמִי שָׁרֵי כְּמוֹ בָּרֵישָׁא: (עיקר תוי"ט).

פרק ד - משנה ז

אֵין מוֹצִיאִין אֶת הָאוּר לֹא מִן הָעֵצִים {כד}, וְלֹא מִן הָאֲבָנִים, וְלֹא מִן הֶעָפָר, וְלֹא מִן הַמַּיִם, וְאֵין מְלַבְּנִין אֶת הָרְעָפִים לִצְלוֹת בָּהֶן. וְעוֹד אָמַר רַבִּי אֱלִיעֶזֶר, עוֹמֵד אָדָם עַל הַמֻּקְצֶה עֶרֶב שַׁבָּת {כה} בַּשְּׁבִיעִית, וְאוֹמֵר, מִכָּאן אֲנִי אוֹכֵל לְמָחָר. וַחֲכָמִים אוֹמְרִים, עַד שֶׁיִּרְשֹׁם {כז} וְיֹאמַר, מִכָּאן וְעַד כָּאן:

 ר"ע מברטנורה  אֵין מוֹצִיאִין אֶת הָאוּר. מִשּׁוּם דְּמוֹלִיד וְדָמֵי לִמְלָאכָה, שֶׁבּוֹרֵא הָאֵשׁ הַזֶּה בְּיוֹם טוֹב: מִן הֶעָפָר. יֵשׁ קַרְקַע כְּשֶׁחוֹפְרִין אוֹתָהּ מוֹצִיאָה אוּר מִמְּקוֹם מַחְפֹּרֶת שֶׁלָּהּ: וְלֹא מִן הַמַּיִם. נוֹתְנִים מַיִם בִּכְלִי זְכוּכִית לְבָנָה וְנוֹתְנוֹ בַּחַמָּה כְּשֶׁהַשֶּׁמֶשׁ חַם מְאֹד וְהַזְּכוּכִית מוֹצִיאָה שַׁלְהֶבֶת, וּמְבִיאִים נְעֹרֶת וּמַגִּיעִים בַּזְּכוּכִית וְהִיא בּוֹעֶרֶת: אֵין מְלַבְּנִין. בָּאוּר: אֶת הָרְעָפִים. דַּוְקָא בִּרְעָפִים חֲדָשִׁים, מִשּׁוּם דְּמַשְׁוֵי לֵיהּ מָנָא בְּהֶסֵּק זֶה, שֶׁהֵן מִתְחַסְּמִין וּמִתְחַזְּקִים עַל יְדֵי הָאוּר: רְעָפִים. כְּלֵי חֶרֶס חֲלוּלִים בְּאֶמְצָעָן שֶׁמְּכַסִּים בָּהֶן הַגַּגּוֹת: וְעוֹד אָמַר רַבִּי אֱלִיעֶזֶר. מִשּׁוּם דְּתָנָא אַחַת לְקֻלָּא גַּבֵּי מֻקְצֶה וְהָדָר אָמַר אַחֲרִיתִי, קָאָמַר וְעוֹד: עוֹמֵד אָדָם עַל הַמֻּקְצֶה. הַצָּרִיךְ הַזְמָנָה, וְהַזְמָנָה מוֹעֶלֶת לוֹ: עֶרֶב שַׁבָּת בַּשְּׁבִיעִית. שֶׁאֵין מַעֲשֵׂר נוֹהֵג בָּהּ וְאֵין מְחֻסָּר אֶלָּא הַזְמָנָה. וְהוּא הַדִּין בִּמְעֻשָּׂר וּבִשְׁאָר שְׁנֵי הַשָּׁבוּעַ. אֶלָּא אוֹרְחָא דְּמִלְּתָא נָקַט, דִּסְתָם מֻקְצֶה לָאו מְעֻשָּׂר הוּא, דִּגְרוֹגָרוֹת וְצִמּוּקִים הֵן סְתָם מֻקְצֶה וְאֵין רְגִילִין לְעַשֵּׂר קֹדֶם גְּמַר מְלָאכָה {כו}: מִכָּאן אֲנִי נוֹטֵל לְמָחָר. וְסַגִּי בְּהָכִי, דְּיֵשׁ בְּרֵרָה: עַד שֶׁיִּרְשֹׁם. בְּסִימָן, דְּאֵין בְּרֵרָה. וַהֲלָכָה כַּחֲכָמִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {כד} הָעֵצִים. כְּגוֹן שֶׁחוֹכְכִין אוֹתָן זוֹ בְזוֹ אוֹ מַכִּין זוֹ בְזוֹ עַד שֶׁתֵּצֵא הָאֵשׁ. הָרַמְבַּ"ם: {כה} עֶרֶב שַׁבָּת. לִרְבוּתָא נָקַט עֶרֶב שַׁבָּת, אַף עַל גַּב דְּסָתִים לָן תַּנָּא כְּרַבִּי שִׁמְעוֹן דְּלֵית לֵיהּ מֻקְצֶה, אֲפִלּוּ הָכִי בְּהַךְ מֻקְצֶה דִּגְרוֹגָרוֹת וְצִמּוּקִים מוֹדֶה רַבִּי שִׁמְעוֹן מִשּׁוּם דְּדָחִינְהוּ בַּיָּדַיִם וְלֹא חֲזוּ, שֶׁמַּסְרִיחוֹת כְּשֶׁמִתְיַבְּשִׁין קֹדֶם שֶׁיָּבְשׁוּ: {כו} וּגְמַר מְלֶאכֶת הַגְּרוֹגָרוֹת בְּסוֹף פֶּרֶק קַמָּא דְּמַעַשְׂרוֹת, וַאֲפִלּוּ הָכִי שַׁבָּת קוֹבַעַת לַמַּעַשְׂרוֹת לְרַבִּי אֱלִיעֶזֶר. גְּמָרָא: {כז} עַד שֶׁיִּרְשֹׁם. כְּבֵית הִלֵּל דִּלְעֵיל, וְהָכָא מַחְמִירִין וּבָעֵי שֶׁיִּרְשׁוֹם מִשּׁוּם דְּאַקְצֵיהּ בַּיָּדַיִם. וְדַע דְּמַסְקִינַן בַּגְּמָרָא דְּלַחֲכָמִים אֵין שַׁבָּת קוֹבַעַת אֶלָּא בְּדָבָר שֶׁנִּגְמְרָה מְלַאכְתּוֹ, וּמַהֲנֵי לֵיהּ דְּלָא בָּעִינַן שֶׁיִּרְאֶה פְּנֵי הַבַּיִת, וְהָכָא שָׁאנֵי דְּכֵיוָן שֶׁאָמַר מִכָּאן כוּ' קָבְעָה לֵיהּ. וְהוּא הַדִּין בְּחֹל, אֶלָּא דְגַבֵּי שַׁבָּת קָא מַשְׁמַע לָן דְּטֶבֶל מוּכָן הוּא, שֶׁאִם עָבַר וְתִקְּנוֹ מְתֻקָּן, וְלֹא אָסְרוּ אֶלָּא לְכַתְּחִלָּה. תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

מַשִּׁילִין פֵּרוֹת דֶּרֶךְ אֲרֻבָּה בְּיוֹם טוֹב, אֲבָל לֹא בְשַׁבָּת, וּמְכַסִּים פֵּרוֹת בְּכֵלִים מִפְּנֵי הַדֶּלֶף, וְכֵן כַּדֵּי יַיִן וְכַדֵּי שֶׁמֶן {ג}. וְנוֹתְנִין כְּלִי תַּחַת הַדֶּלֶף {ד} בְּשַׁבָּת:

 ר"ע מברטנורה  מַשִּׁילִין פֵּרוֹת דֶּרֶךְ אֲרֻבָּה. מִי שֶׁיֵּשׁ לוֹ פֵּרוֹת אוֹ תְּבוּאָה שָׁטוּחַ עַל גַּגּוֹ לְהִתְיַבֵּשׁ וְרָאָה גְּשָׁמִים מְמַשְׁמְשִׁים וּבָאִים, הִתִּירוּ לוֹ לִטְרֹחַ וּלְהַשְׁלִיךְ {א} דֶּרֶךְ אֲרֻבָּה שֶׁבַּגַּג וְהֵם נוֹפְלִים לָאָרֶץ, דְּלֵיכָּא טִרְחָא יְתֵירָא. וְדַוְקָא אֲרֻבָּה, דְּכָל אֲרֻבָּה מִלְּמַעְלָה לְמַטָּה [בְּתִקְרַת הַגַּג], אֲבָל חַלּוֹן כְּגוֹן גַּג שֶׁמֻּקָּף מְחִצּוֹת וְחַלּוֹן בַּכֹּתֶל וְצָרִיךְ לְהַגְבִּיהּוֹ עַד הַחַלּוֹן וּלְהַשְׁלִיכוֹ, כֻּלֵּי הַאי לֹא שָׁרוּ רַבָּנָן לְמִטְרָח: וּמְכַסִּין אֶת הַפֵּרוֹת. וְלֹא אָמְרִינַן טֹרַח שֶׁלֹּא לְצֹרֶךְ יוֹם טוֹב הוּא {ב}, דְּמִשּׁוּם הֶפְסֵד מָמוֹן שָׁרוּ לֵיהּ רַבָּנָן: דֶּלֶף. גְּשָׁמִים הַנּוֹטְפִים מִן הַגַּג: וְכֵן כַּדֵּי יַיִן וְכַדֵּי שָׁמֶן. מְכַסִּין אוֹתָן מִפְּנֵי הַדֶּלֶף: תַּחַת הַדֶּלֶף. לְקַבֵּל אֶת הַמַּיִם שֶׁלֹּא יְטַנְּפוּ הַבַּיִת. וְאִם נִתְמַלֵּא הַכְּלִי שׁוֹפֵךְ וְשׁוֹנֶה וְאֵינוֹ נִמְנָע: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} לִישְׁנָא דְּכִי יִשַּׁל זֵיתֶךָ [דְּבָרִים כח]. גְּמָרָא: {ב} וְכָתַב הָרֹא"שׁ, וְיוֹתֵר נִרְאֶה כְּדִבְרֵי רִ"י דִּבְשַׁבָּת דְּקָתָנֵי בַּסֵּיפָא קָאֵי אַהָנֵי תְּרֵי בָּבֵי, דְּמִדְּקָתָנֵי רֵישָׁא בְּיוֹם טוֹב אֲבָל לֹא בְּשַׁבָּת וּבַסֵּיפָא קָתָנֵי בְּשַׁבָּת, מַשְׁמַע דְּקָאֵי אַתְּרֵי בָּבֵי: {ג} כַּדֵּי יַיִן כוּ'. מוֹקְמִינַן בַּגְּמָרָא בִּדְטִבְלָא, דְּאָסוּר לְטַלְטְלָן, וְהָא קָא מַשְׁמַע לָן דְּלֹא תֵימָא בְּפֵרוֹת הָרְאוּיִן לְטַלְטֵל דַּוְקָא קָא מַשְׁמַע לָן, דַּאֲפִלּוּ בְּאֵין רְאוּיִן, דִּכְלִי נִטַּל לְצֹרֶךְ דָּבָר שֶׁאֵינוֹ נִטַּל. וּלְמַאן דְּאָמַר שֶׁאֵינוֹ נִטַּל, מוֹקְמִינַן דַּוְקָא בְּפֵרוֹת הָרְאוּיִן, וְהָדַר תָּנֵי כַּדֵּי יַיִן, לְאַשְׁמְעִינַן דַּאֲפִלּוּ בְּהֶפְסֵד מֻעָט הִתִּירוּ: {ד} הַדֶּלֶף. דַּוְקָא בְּדֶלֶף הָרָאוּי לִרְחִיצָה, דְּאִי בְּשֶׁאֵינוֹ רָאוּי אָסוּר, דִּמְבַטֵּל כְּלִי מֵהֵיכָנוֹ. הָרַ"ן: (עיקר תוי"ט).

כֹּל שֶׁחַיָּבִין עָלָיו מִשּׁוּם שְׁבוּת, מִשּׁוּם רְשׁוּת, מִשּׁוּם מִצְוָה, בְּשַׁבָּת, חַיָּבִין עָלָיו בְּיוֹם טוֹב. וְאֵלּוּ הֵן מִשּׁוּם שְׁבוּת, לֹא עוֹלִין בָּאִילָן, וְלֹא רוֹכְבִין עַל גַּבֵּי בְהֵמָה, וְלֹא שָׁטִין עַל פְּנֵי הַמַּיִם, וְלֹא מְטַפְּחִין, וְלֹא מְסַפְּקִין, וְלֹא מְרַקְּדִין {ט}. וְאֵלּוּ הֵן מִשּׁוּם רְשׁוּת, לֹא דָנִין, וְלֹא מְקַדְּשִׁין, וְלֹא חוֹלְצִין, וְלֹא מְיַבְּמִין. וְאֵלּוּ הֵן מִשּׁוּם מִצְוָה, לֹא מַקְדִּישִׁין, וְלֹא מַעֲרִיכִין, וְלֹא מַחֲרִימִין, וְלֹא מַגְבִּיהִין תְּרוּמָה וּמַעֲשֵׂר. כָּל אֵלּוּ בְּיוֹם טוֹב אָמְרוּ, קַל וָחֹמֶר בְּשַׁבָּת {יד}. אֵין בֵּין יוֹם טוֹב לְשַׁבָּת {טו} אֶלָּא אֹכֶל נֶפֶשׁ בִּלְבָד:

 ר"ע מברטנורה  כֹּל שֶׁחַיָּבִין עָלָיו. שֶׁאָסְרוּ חֲכָמִים לַעֲשׂוֹתוֹ בְּשַׁבָּת מִשּׁוּם שְׁבוּת: אוֹ מִשּׁוּם רְשׁוּת. שֶׁיֵּשׁ בּוֹ קְצָת מִצְוָה אֲבָל לֹא מִצְוָה גְדוֹלָה, וְקָרוֹב הוּא צָרִיךְ לָהֶן הוּא מַפְרִישָׁן, אֲפִלּוּ הָכִי אָסוּר, דַּהֲוָה לֵיהּ מְתַקֵּן. וְהָנֵי מִלֵּי פֵּרוֹת דִּטְבִילֵי מֵאֶתְמוֹל, אֲבָל פֵּרוֹת דִּטְבִילֵי הָאִידְנָא כְּגוֹן עִסָּה לְאַפְרוּשֵׁי מִינָהּ חַלָּה מַפְרִישִׁינַן לַהּ בְּיוֹם טוֹב וְיָהֲבִינַן לַכֹּהֵן: אֵין בֵּין יוֹם טוֹב לְשַׁבָּת אֶלָּא אֹכֶל נֶפֶשׁ בִּלְבַד. הַךְ סְתָּמָא כְּבֵית שַׁמַּאי דְּאָמְרֵי אֵין מוֹצִיאִין לֹא אֶת הַקָּטָן וְלֹא אֶת הַלּוּלָב וְלֹא אֶת סֵפֶר הַתּוֹרָה לִרְשׁוּת הָרַבִּים, דְּלֹא הֻתְּרָה הוֹצָאָה אֶלָּא לְצֹרֶךְ אֹכֶל נֶפֶשׁ. אֲבָל אֲנַן קַיְמָא לָן כְּבֵית הִלֵּל דְּאָמְרֵי מִתּוֹךְ שֶׁהֻתְּרָה הוֹצָאָה לְצֹרֶךְ אֹכֶל נֶפֶשׁ הֻתְּרָה נַמִּי שֶׁלֹּא לְצֹרֶךְ {טז}. וְאִיכָּא נַמִּי מַשִּׁילִין פֵּרוֹת דֶּרֶךְ אֲרֻבָּה דְּרֵישׁ פִּרְקִין, דִּבְשַׁבָּת אָסוּר וּבְיוֹם טוֹב מֻתָּר: (ר"ע בּוֹ לָשׁוּט. רַשִׁ"י: {ז} וַיַּכּוּ כָף וַיֹּאמְרוּ יְחִי הַמֶּלֶךְ [מְלָכִים ב יא], מְתֻרְגָּם וּטְפָחוּ יַד: {ח} כְּמוֹ סָפַקְתִּי עַל יָרֵךְ [יִרְמְיָה יח]: {ט} מְרַקְּדִין. עֵת לִרְקֹד [קֹהֶלֶת פרק ה - משנה ג

הַבְּהֵמָה וְהַכֵּלִים כְּרַגְלֵי הַבְּעָלִים. הַמּוֹסֵר בְּהֶמְתּוֹ לִבְנוֹ {יז} אוֹ לְרוֹעֶה, הֲרֵי אֵלּוּ כְרַגְלֵי הַבְּעָלִים. כֵּלִים הַמְיֻחָדִין לְאַחַד מִן הָאַחִין שֶׁבַּבַּיִת, הֲרֵי אֵלּוּ כְרַגְלָיו. וְשֶׁאֵין מְיֻחָדִין, הֲרֵי אֵלּוּ כִמְקוֹם שֶׁהוֹלְכִין:

 ר"ע מברטנורה  כְּרַגְלֵי הַבְּעָלִים. אֵין אָדָם יָכוֹל לְהוֹלִיכָהּ בְּיוֹם טוֹב אֶלָּא בְּמָקוֹם שֶׁבְּעָלֶיהָ יְכוֹלִים לֵילֵךְ: אוֹ לְרוֹעֶה הֲרֵי אֵלּוּ כְרַגְלֵי הַבְּעָלִים. מַתְנִיתִין כְּשֶׁיֵּשׁ בָּעִיר שְׁנֵי רוֹעִים, דְּלֹא יָדְעִינַן לְהֵי מִנַּיְהוּ הַוְיָא לַהּ דַּעַת בְּעָלִים, הִלְכָּךְ הֲרֵי הֵן כְּרַגְלֵי הַבְּעָלִים אִם אֵינָהּ אֵצֶל הָרוֹעֶה מִבְּעוֹד יוֹם. אֲבָל כְּשֶׁאֵין בָּעִיר אֶלָּא רוֹעֶה אֶחָד, כָּל בְּנֵי הָעִיר מַעֲמִידִין בְּהֶמְתָּן בִּרְשׁוּתוֹ וּבִקְנִיַּת שְׁבִיתָתוֹ, וְהַוְיָא לַהּ כְּרַגְלֵי הָרוֹעֶה: וְשֶׁאֵינָן מְיֻחָדִים. לְאֶחָד מֵהֶן, אֶלָּא לְכֻלָּן: הֲרֵי אֵלּוּ כִמְקוֹם שֶׁהוֹלְכִין. לִמְקוֹם שֶׁכֻּלָּן יְכוֹלִין לֵילֵךְ מֻתָּר לְהוֹלִיךְ אֶת הַכֵּלִים. אֲבָל אִם עֵרֵב אֶחָד מֵהֶן לְסוֹף אַלְפַּיִם לַצָּפוֹן וְהַשְּׁאָר לֹא עֵרְבוּ, הוּא מְעַכֵּב עַל יָדָם מִלְּהוֹלִיכָהּ לַדָּרוֹם אֲפִלּוּ פְּסִיעָה אַחַת מִפְּנֵי חֶלְקוֹ, וְהֵן מְעַכְּבִין עַל יָדוֹ מִלְּהוֹלִיכָהּ לַצָּפוֹן אֶלָּא אַלְפַּיִם אַמָּה שֶׁהֵן מֻתָּרִין בָּהֶן: (ר"ע מברטנורה).

 עיקר תוי"ט  {יז} לִבְנוֹ. וְאַף עַל גַּב דְּהוּא יָחִיד לֹא קָנָה, לְפִי שֶׁדֶּרֶךְ בְּנֵי אָדָם לִמְסֹר כְּלֵיהֶן וְחֶפְצֵיהֶן לִבְנֵיהֶן וְאֵינָם מַחְשִׁיבִין אוֹתָם כְּאִלּוּ הוֹצִיאוּם מֵרְשׁוּתָם. הַמַּגִּיד מִשְׁנֶה: (עיקר תוי"ט).

פרק ה - משנה ד

הַשּׁוֹאֵל כְּלִי מֵחֲבֵרוֹ מֵעֶרֶב יוֹם טוֹב, כְּרַגְלֵי הַשּׁוֹאֵל. בְּיוֹם טוֹב, כְּרַגְלֵי הַמַּשְׁאִיל. הָאִשָּׁה שֶׁשָּׁאֲלָה מֵחֲבֶרְתָּהּ תְּבָלִין וּמַיִם וּמֶלַח לְעִסָּתָהּ, הֲרֵי אֵלּוּ כְרַגְלֵי שְׁתֵּיהֶן {כ}. רַבִּי יְהוּדָה פוֹטֵר בְּמַיִם, מִפְּנֵי שֶׁאֵין בָּהֶן מַמָּשׁ:

 ר"ע מברטנורה  כְּרַגְלֵי הַשּׁוֹאֵל. שֶׁהֲרֵי קָנָה שְׁבִיתָתוֹ אֶצְלוֹ בֵּין הַשְּׁמָשׁוֹת, דְּבֵין הַשְּׁמָשׁוֹת שֶׁהוּא כְּנִיסַת הַיּוֹם קוֹנֶה שְׁבִיתָה. וַאֲפִלּוּ לֹא בָּא לְיַד הַשּׁוֹאֵל אֶלָּא בְּיוֹם טוֹב {יח}, דְּלָאו בִּרְשׁוּתֵיהּ קַיָּמִין בְּבֵין הַשְּׁמָשׁוֹת, הֲרֵי אֵלּוּ כְּרַגְלָיו שֶׁל שׁוֹאֵל. וְאִם שְׁאָלוֹ בְּיוֹם טוֹב מִשֶּׁחָשְׁכָה, הֲרֵי הֵן כְּרַגְלֵי הַמַּשְׁאִיל, לְפִי שֶׁקָּנָה שְׁבִיתָה אֵצֶל בְּעָלָיו. וַאֲפִלּוּ הָיָה רָגִיל לְהַשְׁאִיל בְּכָל יוֹם טוֹב {יט}: הָאִשָּׁה שֶׁשָּׁאֲלָה מֵחֲבֶרְתָּהּ תְּבָלִין. לִקְדֵרָתָהּ: וּמַיִם וּמֶלַח. לְעִסָּתָהּ: הֲרֵי אֵלּוּ. הַקְּדֵרָה וְהָעִסָּה: כְּרַגְלֵי שְׁתֵּיהֶן. אֵין מוֹלִיכִין אוֹתָן אֶלָּא בְּמָקוֹם שֶׁשְּׁתֵּיהֶן יְכוֹלוֹת לֵילֵךְ. דְּכֵיוָן דִּבְיוֹם טוֹב שָׁאֲלָה, קָנוּ הַתַּבְלִין אוֹ הַמַּיִם וְהַמֶּלַח שְׁבִיתָה אֵצֶל בַּעֲלֵיהֶן: שֶׁאֵין בָּהֶם מַמָּשׁ. אֵין נִכָּרוֹת לֹא בַּקְּדֵרָה וְלֹא בָּעִסָּה, כְּגוֹן שֶׁהַתַּבְשִׁיל עָבֶה וְאֵין הַמַּיִם נִכָּרִים בָּהּ, הִלְכָּךְ אֵין מְעַכְּבוֹת הוֹלָכָתָן. וּבְמֶלַח לֹא פָלִיג רַבִּי יְהוּדָה, דְּמַיְרֵי בְּעִסָּה שֶׁנִּלּוֹשָׁה בְּמֶלַח שֶׁהִיא גַסָּה וְעָבָה, שֶׁהִיא נִכֶּרֶת וְיֵשׁ בָּהּ מַמָּשׁ. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {יח} וְהַשּׁוֹאֵל דְּקָתָנֵי בְּדִבּוּרָא בְּעָלְמָא וְלֹא מָשַׁךְ מִבְּעוֹד יוֹם, וְקָא מַשְׁמַע לָן כֵּיוָן דְּאַסְמְכֵיהּ בִּרְשׁוּתֵיהּ אוֹקְמֵיהּ. רַשִׁ"י: {יט} פֵּרוּשׁ לְהַשְׁאִיל כְּלִי זֶה לְהַשּׁוֹאֵל הַזֶּה, וְכֵיוָן דִּבְעֶרֶב יוֹם טוֹב זֶה לֹא אָמַר לֵיהּ לֹא אוֹקְמֵיהּ בִּרְשׁוּתֵיהּ, מִשּׁוּם דְּמֵימַר אָמַר דִּלְמָא מַשְׁכַּח אִינִישׁ אַחֲרִינָא וְאַזִּיל וְשָׁאִיל מִנֵּיהּ. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} שְׁתֵּיהֶן. וְלֹא בַּטְלֵי מַיִם וּמֶלַח אַגַּב הָעִסָּה, מִשּׁוּם דַּהֲוֵי דָּבָר שֶׁיֵּשׁ לוֹ מַתִּירִין. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ה - משנה ה

הַגַּחֶלֶת כְּרַגְלֵי הַבְּעָלִים, וְשַׁלְהֶבֶת בְּכָל מָקוֹם. גַּחֶלֶת שֶׁל הֶקְדֵּשׁ מוֹעֲלִין בָּהּ, וְשַׁלְהֶבֶת לֹא נֶהֱנִין וְלֹא מוֹעֲלִין. הַמּוֹצִיא גַחֶלֶת לִרְשׁוּת הָרַבִּים, חַיָּב. וְשַׁלְהֶבֶת, פָּטוּר. בּוֹר {כא} שֶׁל יָחִיד, כְּרַגְלֵי הַיָּחִיד. וְשֶׁל אַנְשֵׁי אוֹתָהּ הָעִיר, כְּרַגְלֵי אַנְשֵׁי אוֹתָהּ הָעִיר. וְשֶׁל עוֹלֵי בָבֶל, כְּרַגְלֵי הַמְמַלֵּא:

 ר"ע מברטנורה  וְשַׁלְהֶבֶת. כְּגוֹן הִדְלִיק נֵרוֹ בְּשַׁלְהֶבֶת חֲבֵרוֹ, אֵינוֹ מְעַכֵּב בְּאִסּוּר תְּחוּמִין: מוֹעֲלִין בָּהּ. הַנֶּהֱנֶה מִמֶּנָּה מֵבִיא אֲשַׁם מְעִילוֹת: וְשַׁלְהֶבֶת לֹא נֶהֱנִין. לְכַתְּחִלָּה מִדְּרַבָּנָן. וְאִם נֶהֱנוּ לֹא מוֹעֲלִין, אֵין חַיָּבִין קָרְבַּן מְעִילָה, דְּלֵית בָּהּ מַמָּשָׁא. וְכֵן הַמּוֹצִיא גַּחֶלֶת בְּשַׁבָּת לִרְשׁוּת הָרַבִּים חַיָּב, וְשַׁלְהֶבֶת שֶׁדְּחָפָהּ בְּיָדוֹ מֵרְשׁוּת הַיָּחִיד לִרְשׁוּת הָרַבִּים פָּטוּר: כְּרַגְלֵי הַיָּחִיד. אֵין מוֹלִיכִין הַמַּיִם אֶלָּא כְּרַגְלֵי בַּעַל הַבּוֹר: כְּרַגְלֵי אוֹתָהּ הָעִיר. אַלְפַּיִם אַמָּה לְכָל רוּחַ חוּץ לְעִבּוּרָהּ {כב}: וְשֶׁל עוֹלֵי בָבֶל. הֶעָשׂוּי לְעוֹבְרֵי דְרָכִים בְּאֶמְצַע הַדֶּרֶךְ, וַעֲשָׂאוּם בְּנֵי הַגּוֹלָה לִשְׁתּוֹתָם בַּעֲלוֹתָם: כְּרַגְלֵי הַמְּמַלֵּא. מִפְּנֵי שֶׁהוּא הֶפְקֵר וְהֶפְקֵר נִקְנֶה בְּהַגְבָּהָה. וְאִם בָּא אֶחָד וְשָׁאַל לוֹ מֵימָיו אֵינוֹ מוֹלִיכָן אֶלָּא כְּרַגְלָיו, דְּקָסָבַר הַאי תַּנָּא יֵשׁ בְּרֵרָה לְהַחְמִיר {כג} וּמֵאֶתְמוֹל הֻבְרַר הַדָּבָר דִּלְהַאי גַבְרָא חֲזוּ וּבִרְשׁוּתֵיהּ קַיְּמֵי. וּדְלֹא כְּרַבִּי יוֹחָנָן בֶּן נוּרִי דְּאָמַר חֶפְצֵי הֶפְקֵר קוֹנִין שְׁבִיתָה לְעַצְמָן בִּמְקוֹמָן: (ר"ע מברטנורה).

 עיקר תוי"ט  {כא} בּוֹר כוּ'. כָּל מָקוֹם שֶׁשּׁוֹנֶה בּוֹר הוּא מַיִם מְכֻנָּסִים, וְהָכִי אִיתָא בַּגְּמָרָא, דְּאִלּוּ הַנּוֹבְעִים הֲרֵי הֵן כְּרַגְלֵי כָּל אָדָם, פֵּרוּשׁ הַמְּמַלֵּא רִאשׁוֹן: {כב} רַשִׁ"י. וְנִרְאֶה בְּעֵינַי דְּדַוְקָא כְּשֶׁלֹּא עֵרֵב אֶחָד מֵהֶן, דְּאִם לֹא כֵן הוּא מְעַכֵּב. עוֹד יֵשׁ לוֹמַר דְּשָׁאנֵי מַיִם דְּכָלִין וְאֵינָם נִשְׁאָרִים לְאַחַר עֲשִׂיַּת צְרָכָיו בָּהֶם, שַׁיָּךְ לוֹמַר בָּהֶם יֵשׁ בְּרֵרָה, מַה שֶּׁאֵין כֵּן בְּכֵלִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} דִּתְחוּמִין דְּרַבָּנָן, וּבִדְרַבָּנָן קַיְמָא לָן יֵשׁ בְּרֵרָה: (עיקר תוי"ט).

פרק ה - משנה ו

מִי שֶׁהָיוּ פֵרוֹתָיו בְּעִיר אַחֶרֶת, וְעֵרְבוּ בְנֵי אוֹתָהּ הָעִיר לְהָבִיא אֶצְלוֹ מִפֵּרוֹתָיו, לֹא יָבִיאוּ לוֹ. וְאִם עֵרַב הוּא, פֵּרוֹתָיו כָּמוֹהוּ:

 ר"ע מברטנורה  לֹא יָבִיאוּ לוֹ. מִפֵּרוֹתָיו, הוֹאִיל וְהוּא לֹא עֵרֵב לְשָׁם, דְּפֵרוֹתָיו כָּמוֹהוּ: (ר"ע מברטנורה).

פרק ה - משנה ז

מִי שֶׁזִּמֵּן אֶצְלוֹ אוֹרְחִים, לֹא יוֹלִיכוּ בְיָדָם מָנוֹת, אֶלָּא אִם כֵּן {כד} זִכָּה לָהֶם מָנוֹתֵיהֶם מֵעֶרֶב יוֹם טוֹב. אֵין מַשְׁקִין {כה} וְשׁוֹחֲטִין אֶת הַמִּדְבָּרִיּוֹת, אֲבָל מַשְׁקִין וְשׁוֹחֲטִין אֶת הַבַּיָתוֹת. אֵלּוּ הֵן בַּיָתוֹת, הַלָּנוֹת בָּעִיר {כו}. מִדְבָּרִיּוֹת, הַלָּנוֹת בָּאֲפָר {כז}:

 ר"ע מברטנורה  מִי שֶׁזִּמֵּן אֶצְלוֹ אוֹרְחִים. מֵעִיר אַחֶרֶת, וְיָבֹאוּ אֶצְלוֹ עַל יְדֵי עֵרוּב: לֹא יוֹלִיכוּ. לְאַחַר סְעוּדָה מָנוֹת בְּיָדָם לְבֵיתָם: אֶלָּא אִם כֵּן זִכָּה לָהֶם. עַל יְדֵי אַחֵר: (ר"ע מברטנורה).

 עיקר תוי"ט  {כד} אֶלָּא אִם כֵּן. וְאַף עַל גַּב דִּבְמִשְׁנָה ד' בְּדִבּוּרָא קָנִי לֵיהּ הַשּׁוֹאֵל, שָׁאנֵי הָתָם דְּאָמַר לֵיהּ מַשְׁאִיל לַשּׁוֹאֵל גּוּפֵיהּ, הִלְכָּךְ סָמְכָה דַעְתֵּיהּ, אֲבָל הָכָא דְּלֹא אָמַר לֵיהּ בָּעִינַן זִכּוּי. הָרַ"ן: {כה} אֵין מַשְׁקִין. לְעִנְיַן הַהַשְׁקָאָה בְּעָלְמָא וַדַּאי אֵלּוּ וְאֵלּוּ מְזוֹנוֹתֵיהֶם עָלֶיךָ וּמֻתָּר לְהַשְׁקוֹתָם, מִשּׁוּם הָכִי מְפֹרָשׁ בַּגְּמָרָא דְּמִשּׁוּם שְׁחִיטָה נָקַט, וְאַגַּב אוֹרְחֵיהּ קָא מַשְׁמַע לָן דְּלִשְׁקֵי וְהָדַר לִשְׁחֹט, מִשּׁוּם סִירְכָא דְּמַשְׁכָּא כוּ'. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כו} בָּעִיר. כְּתַנָּא קַמָּא דִּבְרַיְתָא שֶׁפֵּרֵשׁ שֶׁיּוֹצְאוֹת וְרוֹעוֹת חוּץ לַתְּחוּם וּבָאוֹת וְלָנוֹת בְּתוֹךְ הַתְּחוּם, וְנִמְצָא שֶׁהֵן מוּכָנוֹת: {כז} בָּאֲפָר. פֵּרֵשׁ הָרַמְבַּ"ם שָׁם הַכְּפָרִים. וְהֶעָרוּךְ פֵּרֵשׁ כְּמוֹ וַתִּרְעֶינָה בָּאָחוּ [בְּרֵאשִׁית מא], וּרְצוֹנוֹ לוֹמַר שֶׁהֵם חוּץ לַתְּחוּם, וְהֵם מֻקְצִים, דְּאֵין דַּעַת אַנְשֵׁי הָעִיר עֲלֵיהֶן. וְעַיֵּן מַה שֶּׁכָּתַבְנוּ פֶּרֶק ד' מִשְׁנָה ג'. וּמִיהוּ הָכָא אִיתָא מַאן דְּאָמַר בַּגְּמָרָא דְּאַף רַבִּי שִׁמְעוֹן מוֹדֶה דַּהֲוֵי כִּגְרוֹגָרוֹת וְצִמּוּקִים דְּדָחִינְהוּ בַּיָּדַיִם: (עיקר תוי"ט).
