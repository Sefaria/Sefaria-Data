ד-א


[ויקרא פרק-ו-יח]  דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו לֵאמֹר זֹאת תּוֹרַת הַחַטָּאת בִּמְקוֹם אֲשֶׁר תִּשָּׁחֵט הָעֹלָה תִּשָּׁחֵט הַחַטָּאת לִפְנֵי יְהוָה קֹדֶשׁ קָדָשִׁים הִוא:

[ויקרא פרק-יב-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר אִשָּׁה כִּי תַזְרִיעַ וְיָלְדָה זָכָר וְטָמְאָה שִׁבְעַת יָמִים כִּימֵי נִדַּת דְּוֹתָהּ תִּטְמָא:

[ויקרא פרק-טו-ב]  דַּבְּרוּ אֶל בְּנֵי יִשְׂרָאֵל וַאֲמַרְתֶּם אֲלֵהֶם אִישׁ אִישׁ כִּי יִהְיֶה זָב מִבְּשָׂרוֹ זוֹבוֹ טָמֵא הוּא:

[במדבר פרק-ה-ו]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל אִישׁ אוֹ אִשָּׁה כִּי יַעֲשׂוּ מִכָּל חַטֹּאת הָאָדָם לִמְעֹל מַעַל בַּיהוָה וְאָשְׁמָה הַנֶּפֶשׁ הַהִוא:

[דניאל פרק-ב-לו]  דְּנָה חֶלְמָא וּפִשְׁרֵהּ נֵאמַר קֳדָם מַלְכָּא:

[דניאל פרק-ב-מג]  (די) וְדִי חֲזַיְתָ פַּרְזְלָא מְעָרַב בַּחֲסַף טִינָא מִתְעָרְבִין לֶהֱוֹן בִּזְרַע אֲנָשָׁא וְלָא לֶהֱוֹן דָּבְקִין דְּנָה עִם דְּנָה הֵא כְדִי פַרְזְלָא לָא מִתְעָרַב עִם חַסְפָּא:

[דניאל פרק-ד-כא]  דְּנָה פִשְׁרָא מַלְכָּא וּגְזֵרַת (עליא) עִלָּאָה הִיא דִּי מְטָת עַל (מראי) מָרִי מַלְכָּא:


ד-ב


[שמות פרק-יא-ב]  דַּבֶּר נָא בְּאָזְנֵי הָעָם וְיִשְׁאֲלוּ אִישׁ מֵאֵת רֵעֵהוּ וְאִשָּׁה מֵאֵת רְעוּתָהּ כְּלֵי כֶסֶף וּכְלֵי זָהָב:

[ישעיה פרק-נה-ו]  דִּרְשׁוּ יְהוָה בְּהִמָּצְאוֹ קְרָאֻהוּ בִּהְיוֹתוֹ קָרוֹב:

[תהילים פרק-לו-ד]  דִּבְרֵי פִיו אָוֶן וּמִרְמָה חָדַל לְהַשְׂכִּיל לְהֵיטִיב:


ד-ד


[במדבר פרק-ב-יח]  דֶּגֶל מַחֲנֵה אֶפְרַיִם לְצִבְאֹתָם יָמָּה וְנָשִׂיא לִבְנֵי אֶפְרַיִם אֱלִישָׁמָע בֶּן עַמִּיהוּד:

[תהילים פרק-קה-ד]  דִּרְשׁוּ יְהוָה וְעֻזּוֹ בַּקְּשׁוּ פָנָיו תָּמִיד:

[קהלת פרק-יב-יא]  דִּבְרֵי חֲכָמִים כַּדָּרְבֹנוֹת וּכְמַשְׂמְרוֹת נְטוּעִים בַּעֲלֵי אֲסֻפּוֹת נִתְּנוּ מֵרֹעֶה אֶחָד:

[דברי הימים א פרק-טז-יא]  דִּרְשׁוּ יְהוָה וְעֻזּוֹ בַּקְּשׁוּ פָנָיו תָּמִיד:


ד-ה


[ויקרא פרק-ד-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר נֶפֶשׁ כִּי תֶחֱטָא בִשְׁגָגָה מִכֹּל מִצְוֹת יְהוָה אֲשֶׁר לֹא תֵעָשֶׂינָה וְעָשָׂה מֵאַחַת מֵהֵנָּה:

[ויקרא פרק-כב-ב]  דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו וְיִנָּזְרוּ מִקָּדְשֵׁי בְנֵי יִשְׂרָאֵל וְלֹא יְחַלְּלוּ אֶת שֵׁם קָדְשִׁי אֲשֶׁר הֵם מַקְדִּשִׁים לִי אֲנִי יְהוָה:

[ויקרא פרק-כב-יח]  דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו וְאֶל כָּל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם אִישׁ אִישׁ מִבֵּית יִשְׂרָאֵל וּמִן הַגֵּר בְּיִשְׂרָאֵל אֲשֶׁר יַקְרִיב קָרְבָּנוֹ לְכָל נִדְרֵיהֶם וּלְכָל נִדְבוֹתָם אֲשֶׁר יַקְרִיבוּ לַיהוָה לְעֹלָה:

[ויקרא פרק-כג-לד]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר בַּחֲמִשָּׁה עָשָׂר יוֹם לַחֹדֶשׁ הַשְּׁבִיעִי הַזֶּה חַג הַסֻּכּוֹת שִׁבְעַת יָמִים לַיהוָֹה:

[ויקרא פרק-כה-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם כִּי תָבֹאוּ אֶל הָאָרֶץ אֲשֶׁר אֲנִי נֹתֵן לָכֶם וְשָׁבְתָה הָאָרֶץ שַׁבָּת לַיהוָה:

[ויקרא פרק-כז-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם אִישׁ כִּי יַפְלִא נֶדֶר בְּעֶרְכְּךָ נְפָשֹׁת לַיהוָה:

[במדבר פרק-ו-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם אִישׁ אוֹ אִשָּׁה כִּי יַפְלִא לִנְדֹּר נֶדֶר נָזִיר לְהַזִּיר לַיהוָה:

[במדבר פרק-ט-י]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר אִישׁ אִישׁ כִּי יִהְיֶה טָמֵא לָנֶפֶשׁ אוֹ בְדֶרֶךְ רְחֹקָה לָכֶם אוֹ לְדֹרֹתֵיכֶם וְעָשָׂה פֶסַח לַיהוָה:

[במדבר פרק-טו-יח]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם בְּבֹאֲכֶם אֶל הָאָרֶץ אֲשֶׁר אֲנִי מֵבִיא אֶתְכֶם שָׁמָּה:

[יהושע פרק-כ-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר תְּנוּ לָכֶם אֶת עָרֵי הַמִּקְלָט אֲשֶׁר דִּבַּרְתִּי אֲלֵיכֶם בְּיַד מֹשֶׁה:

[ישעיה פרק-מ-ב]  דַּבְּרוּ עַל לֵב יְרוּשָׁלִַם וְקִרְאוּ אֵלֶיהָ כִּי מָלְאָה צְבָאָהּ כִּי נִרְצָה עֲוֹנָהּ כִּי לָקְחָה מִיַּד יְהוָה כִּפְלַיִם בְּכָל חַטֹּאתֶיהָ:

[ירמיה פרק-כב-טז]  דָּן דִּין עָנִי וְאֶבְיוֹן אָז טוֹב הֲלוֹא הִיא הַדַּעַת אֹתִי נְאֻם יְהוָה:

[יחזקאל פרק-כא-כה]  דֶּרֶךְ תָּשִׂים לָבוֹא חֶרֶב אֵת רַבַּת בְּנֵי עַמּוֹן וְאֶת יְהוּדָה בִירוּשָׁלִַם בְּצוּרָה:

[יחזקאל פרק-כז-כ]  דְּדָן רֹכַלְתֵּךְ בְבִגְדֵי חֹפֶשׁ לְרִכְבָּה:

[צפניה פרק-א-א]  דְּבַר יְהוָה אֲשֶׁר הָיָה אֶל צְפַנְיָה בֶּן כּוּשִׁי בֶן גְּדַלְיָה בֶּן אֲמַרְיָה בֶּן חִזְקִיָּה בִּימֵי יֹאשִׁיָּהוּ בֶן אָמוֹן מֶלֶךְ יְהוּדָה:

[משלי פרק-יב-כה]  דְּאָגָה בְלֶב אִישׁ יַשְׁחֶנָּה וְדָבָר טוֹב יְשַׂמְּחֶנָּה:

[משלי פרק-טו-יט]  דֶּרֶךְ עָצֵל כִּמְשֻׂכַת חָדֶק וְאֹרַח יְשָׁרִים סְלֻלָה:

[משלי פרק-כז-טו]  דֶּלֶף טוֹרֵד בְּיוֹם סַגְרִיר וְאֵשֶׁת (מדונים) מִדְיָנִים נִשְׁתָּוָה:

[משלי פרק-ל-יט]  דֶּרֶךְ הַנֶּשֶׁר בַּשָּׁמַיִם דֶּרֶךְ נָחָשׁ עֲלֵי צוּר דֶּרֶךְ אֳנִיָּה בְלֶב יָם וְדֶרֶךְ גֶּבֶר בְּעַלְמָה:

[משלי פרק-לא-יג]  דָּרְשָׁה צֶמֶר וּפִשְׁתִּים וַתַּעַשׂ בְּחֵפֶץ כַּפֶּיהָ:

[איוב פרק-מא-ו]  דַּלְתֵי פָנָיו מִי פִתֵּחַ סְבִיבוֹת שִׁנָּיו אֵימָה:

[שיר השירים פרק-ה-י]  דּוֹדִי צַח וְאָדוֹם דָּגוּל מֵרְבָבָה:

[איכה פרק-א-ד]  דַּרְכֵי צִיּוֹן אֲבֵלוֹת מִבְּלִי בָּאֵי מוֹעֵד כָּל שְׁעָרֶיהָ שׁוֹמֵמִין כֹּהֲנֶיהָ נֶאֱנָחִים בְּתוּלֹתֶיהָ נּוּגוֹת וְהִיא מַר לָהּ:

[דניאל פרק-ה-כו]  דְּנָה פְּשַׁר מִלְּתָא מְנֵא מְנָה אֱלָהָא מַלְכוּתָךְ וְהַשְׁלְמַהּ:

[נחמיה פרק-א-א]  דִּבְרֵי נְחֶמְיָה בֶּן חֲכַלְיָה וַיְהִי בְחֹדֶשׁ (כסלו) כִּסְלֵיו שְׁנַת עֶשְׂרִים וַאֲנִי הָיִיתִי בְּשׁוּשַׁן הַבִּירָה:


ד-ו


[שמות פרק-ח-כג]  דֶּרֶךְ שְׁלֹשֶׁת יָמִים נֵלֵךְ בַּמִּדְבָּר וְזָבַחְנוּ לַיהוָה אֱלֹהֵינוּ כַּאֲשֶׁר יֹאמַר אֵלֵינוּ:

[ויקרא פרק-ז-כג]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר כָּל חֵלֶב שׁוֹר וְכֶשֶׂב וָעֵז לֹא תֹאכֵלוּ:

[ויקרא פרק-ז-כט]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר הַמַּקְרִיב אֶת זֶבַח שְׁלָמָיו לַיהוָה יָבִיא אֶת קָרְבָּנוֹ לַיהוָה מִזֶּבַח שְׁלָמָיו:

[ויקרא פרק-כא-יז]  דַּבֵּר אֶל אַהֲרֹן לֵאמֹר אִישׁ מִזַּרְעֲךָ לְדֹרֹתָם אֲשֶׁר יִהְיֶה בוֹ מוּם לֹא יִקְרַב לְהַקְרִיב לֶחֶם אֱלֹהָיו:

[במדבר פרק-יז-יז]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְקַח מֵאִתָּם מַטֶּה מַטֶּה לְבֵית אָב מֵאֵת כָּל נְשִׂיאֵהֶם לְבֵית אֲבֹתָם שְׁנֵים עָשָׂר מַטּוֹת אִישׁ אֶת שְׁמוֹ תִּכְתֹּב עַל מַטֵּהוּ:

[מלכים ב פרק-י-י]  דְּעוּ אֵפוֹא כִּי לֹא יִפֹּל מִדְּבַר יְהוָה אַרְצָה אֲשֶׁר דִּבֶּר יְהוָה עַל בֵּית אַחְאָב וַיהוָה עָשָׂה אֵת אֲשֶׁר דִּבֶּר בְּיַד עַבְדּוֹ אֵלִיָּהוּ:

[ישעיה פרק-נז-יח]  דְּרָכָיו רָאִיתִי וְאֶרְפָּאֵהוּ וְאַנְחֵהוּ וַאֲשַׁלֵּם נִחֻמִים לוֹ וְלַאֲבֵלָיו:

[ירמיה פרק-כא-ב]  דְּרָשׁ נָא בַעֲדֵנוּ אֶת יְהוָה כִּי נְבוּכַדְרֶאצַּר מֶלֶךְ בָּבֶל נִלְחָם עָלֵינוּ אוּלַי יַעֲשֶׂה יְהוָה אוֹתָנוּ כְּכָל נִפְלְאֹתָיו וְיַעֲלֶה מֵעָלֵינוּ:

[תהילים פרק-ק-ג]  דְּעוּ כִּי יְהוָה הוּא אֱלֹהִים הוּא עָשָׂנוּ (ולא) וְלוֹ אֲנַחְנוּ עַמּוֹ וְצֹאן מַרְעִיתוֹ:

[תהילים פרק-קמה-ד]  דּוֹר לְדוֹר יְשַׁבַּח מַעֲשֶׂיךָ וּגְבוּרֹתֶיךָ יַגִּידוּ:

[משלי פרק-ד-יט]  דֶּרֶךְ רְשָׁעִים כָּאֲפֵלָה לֹא יָדְעוּ בַּמֶּה יִכָּשֵׁלוּ:

[משלי פרק-כה-טז]  דְּבַשׁ מָצָאתָ אֱכֹל דַּיֶּךָּ פֶּן תִּשְׂבָּעֶנּוּ וַהֲקֵאתוֹ:

[משלי פרק-ל-יג]  דּוֹר מָה רָמוּ עֵינָיו וְעַפְעַפָּיו יִנָּשֵׂאוּ:

[משלי פרק-לא-א]  דִּבְרֵי לְמוּאֵל מֶלֶךְ מַשָּׂא אֲשֶׁר יִסְּרַתּוּ אִמּוֹ:

[שיר השירים פרק-ה-ד]  דּוֹדִי שָׁלַח יָדוֹ מִן הַחֹר וּמֵעַי הָמוּ עָלָיו:

[איכה פרק-ב-ד]  דָּרַךְ קַשְׁתּוֹ כְּאוֹיֵב נִצָּב יְמִינוֹ כְּצָר וַיַּהֲרֹג כֹּל מַחֲמַדֵּי עָיִן בְּאֹהֶל בַּת צִיּוֹן שָׁפַךְ כָּאֵשׁ חֲמָתוֹ:

[קהלת פרק-י-יב]  דִּבְרֵי פִי חָכָם חֵן וְשִׂפְתוֹת כְּסִיל תְּבַלְּעֶנּוּ:


ד-י


[שמות פרק-כה-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְיִקְחוּ לִי תְּרוּמָה מֵאֵת כָּל אִישׁ אֲשֶׁר יִדְּבֶנּוּ לִבּוֹ תִּקְחוּ אֶת תְּרוּמָתִי:

[ויקרא פרק-כג-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם מוֹעֲדֵי יְהוָה אֲשֶׁר תִּקְרְאוּ אֹתָם מִקְרָאֵי קֹדֶשׁ אֵלֶּה הֵם מוֹעֲדָי:

[במדבר פרק-ב-כה]  דֶּגֶל מַחֲנֵה דָן צָפֹנָה לְצִבְאֹתָם וְנָשִׂיא לִבְנֵי דָן אֲחִיעֶזֶר בֶּן עַמִּישַׁדָּי:

[שופטים פרק-ט-ב]  דַּבְּרוּ נָא בְּאָזְנֵי כָל בַּעֲלֵי שְׁכֶם מַה טּוֹב לָכֶם הַמְשֹׁל בָּכֶם שִׁבְעִים אִישׁ כֹּל בְּנֵי יְרֻבַּעַל אִם מְשֹׁל בָּכֶם אִישׁ אֶחָד וּזְכַרְתֶּם כִּי עַצְמֵכֶם וּבְשַׂרְכֶם אָנִי:

[ישעיה פרק-לח-יב]  דּוֹרִי נִסַּע וְנִגְלָה מִנִּי כְּאֹהֶל רֹעִי קִפַּדְתִּי כָאֹרֵג חַיַּי מִדַּלָּה יְבַצְּעֵנִי מִיּוֹם עַד לַיְלָה תַּשְׁלִימֵנִי:

[ירמיה פרק-כב-כא]  דִּבַּרְתִּי אֵלַיִךְ בְּשַׁלְוֹתַיִךְ אָמַרְתְּ לֹא אֶשְׁמָע זֶה דַרְכֵּךְ מִנְּעוּרַיִךְ כִּי לֹא שָׁמַעַתְּ בְּקוֹלִי:

[יחזקאל פרק-כט-ג]  דַּבֵּר וְאָמַרְתָּ כֹּה אָמַר אֲדֹנָי יְהוִה הִנְנִי עָלֶיךָ פַּרְעֹה מֶלֶךְ מִצְרַיִם הַתַּנִּים הַגָּדוֹל הָרֹבֵץ בְּתוֹךְ יְאֹרָיו אֲשֶׁר אָמַר לִי יְאֹרִי וַאֲנִי עֲשִׂיתִנִי:

[יחזקאל פרק-לז-יט]  דַּבֵּר אֲלֵהֶם כֹּה אָמַר אֲדֹנָי יְהוִה הִנֵּה אֲנִי לֹקֵחַ אֶת עֵץ יוֹסֵף אֲשֶׁר בְּיַד אֶפְרַיִם וְשִׁבְטֵי יִשְׂרָאֵל (חברו) חֲבֵרָיו וְנָתַתִּי אוֹתָם עָלָיו אֶת עֵץ יְהוּדָה וַעֲשִׂיתִם לְעֵץ אֶחָד וְהָיוּ אֶחָד בְּיָדִי:

[הושע פרק-י-ד]  דִּבְּרוּ דְבָרִים אָלוֹת שָׁוְא כָּרֹת בְּרִית וּפָרַח כָּרֹאשׁ מִשְׁפָּט עַל תַּלְמֵי שָׂדָי:

[תהילים פרק-כה-ד]  דְּרָכֶיךָ יְהוָה הוֹדִיעֵנִי אֹרְחוֹתֶיךָ לַמְּדֵנִי:

[תהילים פרק-לד-ה]  דָּרַשְׁתִּי אֶת יְהוָה וְעָנָנִי וּמִכָּל מְגוּרוֹתַי הִצִּילָנִי:

[תהילים פרק-קיח-יג]  דַּחֹה דְחִיתַנִי לִנְפֹּל וַיהוָה עֲזָרָנִי:

[תהילים פרק-קיט-כט]  דֶּרֶךְ שֶׁקֶר הָסֵר מִמֶּנִּי וְתוֹרָתְךָ חָנֵּנִי:

[תהילים פרק-קיט-ל]  דֶּרֶךְ אֱמוּנָה בָחָרְתִּי מִשְׁפָּטֶיךָ שִׁוִּיתִי:

[תהילים פרק-קיט-לא]  דָּבַקְתִּי בְעֵדְוֹתֶיךָ יְהוָה אַל תְּבִישֵׁנִי:

[תהילים פרק-קיט-לב]  דֶּרֶךְ מִצְוֹתֶיךָ אָרוּץ כִּי תַרְחִיב לִבִּי:

[דניאל פרק-ב-ט]  דִּי הֵן חֶלְמָא לָא תְהוֹדְעֻנַּנִי חֲדָה הִיא דָתְכוֹן וּמִלָּה כִדְבָה וּשְׁחִיתָה (הזמנתון) הִזְדְּמִנְתּוּן לְמֵאמַר קָדָמַי עַד דִּי עִדָּנָא יִשְׁתַּנֵּא לָהֵן חֶלְמָא אֱמַרוּ לִי וְאִנְדַּע דִּי פִשְׁרֵהּ תְּהַחֲוֻנַּנִי:

[עזרא פרק-ו-י]  דִּי לֶהֱוֹן מְהַקְרְבִין נִיחוֹחִין לֶאֱלָהּ שְׁמַיָּא וּמְצַלַּיִן לְחַיֵּי מַלְכָּא וּבְנוֹהִי:


ד-כ


[ישעיה פרק-כג-ב]  דֹּמּוּ יֹשְׁבֵי אִי סֹחֵר צִידוֹן עֹבֵר יָם מִלְאוּךְ:

[ירמיה פרק-ד-יח]  דַּרְכֵּךְ וּמַעֲלָלַיִךְ עָשׂוֹ אֵלֶּה לָךְ זֹאת רָעָתֵךְ כִּי מָר כִּי נָגַע עַד לִבֵּךְ:

[תהילים פרק-מח-י]  דִּמִּינוּ אֱלֹהִים חַסְדֶּךָ בְּקֶרֶב הֵיכָלֶךָ:

[תהילים פרק-סג-ט]  דָּבְקָה נַפְשִׁי אַחֲרֶיךָ בִּי תָּמְכָה יְמִינֶךָ:

[תהילים פרק-קיט-כה]  דָּבְקָה לֶעָפָר נַפְשִׁי חַיֵּנִי כִּדְבָרֶךָ:

[תהילים פרק-קיט-כו]  דְּרָכַי סִפַּרְתִּי וַתַּעֲנֵנִי לַמְּדֵנִי חֻקֶּיךָ:

[תהילים פרק-קיט-כז]  דֶּרֶךְ פִּקּוּדֶיךָ הֲבִינֵנִי וְאָשִׂיחָה בְּנִפְלְאוֹתֶיךָ:

[תהילים פרק-קיט-כח]  דָּלְפָה נַפְשִׁי מִתּוּגָה קַיְּמֵנִי כִּדְבָרֶךָ:

[משלי פרק-ל-יא]  דּוֹר אָבִיו יְקַלֵּל וְאֶת אִמּוֹ לֹא יְבָרֵךְ:

[דניאל פרק-ד-טו]  דְּנָה חֶלְמָא חֲזֵית אֲנָה מַלְכָּא נְבוּכַדְנֶצַּר (ואנתה) וְאַנְתְּ בֵּלְטְשַׁאצַּר פִּשְׁרֵא אֱמַר כָּל קֳבֵל דִּי כָּל חַכִּימֵי מַלְכוּתִי לָא יָכְלִין פִּשְׁרָא לְהוֹדָעֻתַנִי (ואנתה) וְאַנְתְּ כָּהֵל דִּי רוּחַ אֱלָהִין קַדִּישִׁין בָּךְ:

[נחמיה פרק-י-ז]  דָּנִיֵּאל גִּנְּתוֹן בָּרוּךְ:


ד-ל


[בראשית פרק-מט-טז]  דָּן יָדִין עַמּוֹ כְּאַחַד שִׁבְטֵי יִשְׂרָאֵל:

[במדבר פרק-ה-יב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם אִישׁ אִישׁ כִּי תִשְׂטֶה אִשְׁתּוֹ וּמָעֲלָה בוֹ מָעַל:

[ישעיה פרק-ט-ז]  דָּבָר שָׁלַח אֲדֹנָי בְּיַעֲקֹב וְנָפַל בְּיִשְׂרָאֵל:

[הושע פרק-א-א]  דְּבַר יְהוָה אֲשֶׁר הָיָה אֶל הוֹשֵׁעַ בֶּן בְּאֵרִי בִּימֵי עֻזִּיָּה יוֹתָם אָחָז יְחִזְקִיָּה מַלְכֵי יְהוּדָה וּבִימֵי יָרָבְעָם בֶּן יוֹאָשׁ מֶלֶךְ יִשְׂרָאֵל:

[יואל פרק-א-א]  דְּבַר יְהוָה אֲשֶׁר הָיָה אֶל יוֹאֵל בֶּן פְּתוּאֵל:

[עמוס פרק-ה-ו]  דִּרְשׁוּ אֶת יְהוָה וִחְיוּ פֶּן יִצְלַח כָּאֵשׁ בֵּית יוֹסֵף וְאָכְלָה וְאֵין מְכַבֶּה לְבֵית אֵל:

[משלי פרק-ל-א]  דִּבְרֵי אָגוּר בִּן יָקֶה הַמַּשָּׂא נְאֻם הַגֶּבֶר לְאִיתִיאֵל לְאִיתִיאֵל וְאֻכָל:


ד-מ


[שמות פרק-יד-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְיָשֻׁבוּ וְיַחֲנוּ לִפְנֵי פִּי הַחִירֹת בֵּין מִגְדֹּל וּבֵין הַיָּם לִפְנֵי בַּעַל צְפֹן נִכְחוֹ תַחֲנוּ עַל הַיָּם:

[ויקרא פרק-א-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם אָדָם כִּי יַקְרִיב מִכֶּם קָרְבָּן לַיהוָה מִן הַבְּהֵמָה מִן הַבָּקָר וּמִן הַצֹּאן תַּקְרִיבוּ אֶת קָרְבַּנְכֶם:

[ויקרא פרק-יח-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם אֲנִי יְהוָה אֱלֹהֵיכֶם:

[ויקרא פרק-יט-ב]  דַּבֵּר אֶל כָּל עֲדַת בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם קְדֹשִׁים תִּהְיוּ כִּי קָדוֹשׁ אֲנִי יְהוָה אֱלֹהֵיכֶם:

[במדבר פרק-ו-כג]  דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו לֵאמֹר כֹּה תְבָרֲכוּ אֶת בְּנֵי יִשְׂרָאֵל אָמוֹר לָהֶם:

[במדבר פרק-טו-ב]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם כִּי תָבֹאוּ אֶל אֶרֶץ מוֹשְׁבֹתֵיכֶם אֲשֶׁר אֲנִי נֹתֵן לָכֶם:

[במדבר פרק-טז-כד]  דַּבֵּר אֶל הָעֵדָה לֵאמֹר הֵעָלוּ מִסָּבִיב לְמִשְׁכַּן קֹרַח דָּתָן וַאֲבִירָם:

[ישעיה פרק-נט-ח]  דֶּרֶךְ שָׁלוֹם לֹא יָדָעוּ וְאֵין מִשְׁפָּט בְּמַעְגְּלוֹתָם נְתִיבוֹתֵיהֶם עִקְּשׁוּ לָהֶם כֹּל דֹּרֵךְ בָּהּ לֹא יָדַע שָׁלוֹם:

[ירמיה פרק-מב-יט]  דִּבֶּר יְהוָה עֲלֵיכֶם שְׁאֵרִית יְהוּדָה אַל תָּבֹאוּ מִצְרָיִם יָדֹעַ תֵּדְעוּ כִּי הַעִידֹתִי בָכֶם הַיּוֹם:

[עמוס פרק-ה-יד]  דִּרְשׁוּ טוֹב וְאַל רָע לְמַעַן תִּחְיוּ וִיהִי כֵן יְהוָה אֱלֹהֵי צְבָאוֹת אִתְּכֶם כַּאֲשֶׁר אֲמַרְתֶּם:

[מיכה פרק-א-א]  דְּבַר יְהוָה אֲשֶׁר הָיָה אֶל מִיכָה הַמֹּרַשְׁתִּי בִּימֵי יוֹתָם אָחָז יְחִזְקִיָּה מַלְכֵי יְהוּדָה אֲשֶׁר חָזָה עַל שֹׁמְרוֹן וִירוּשָׁלִָם:

[חבקוק פרק-ג-טו]  דָּרַכְתָּ בַיָּם סוּסֶיךָ חֹמֶר מַיִם רַבִּים:

[תהילים פרק-יז-יב]  דִּמְיֹנוֹ כְּאַרְיֵה יִכְסוֹף לִטְרוֹף וְכִכְפִיר יֹשֵׁב בְּמִסְתָּרִים:

[תהילים פרק-מא-ט]  דְּבַר בְּלִיַּעַל יָצוּק בּוֹ וַאֲשֶׁר שָׁכַב לֹא יוֹסִיף לָקוּם:

[תהילים פרק-סה-ד]  דִּבְרֵי עֲוֹנֹת גָּבְרוּ מֶנִּי פְּשָׁעֵינוּ אַתָּה תְכַפְּרֵם:

[משלי פרק-ג-יז]  דְּרָכֶיהָ דַרְכֵי נֹעַם וְכָל נְתִיבוֹתֶיהָ שָׁלוֹם:

[משלי פרק-יב-ו]  דִּבְרֵי רְשָׁעִים אֱרָב דָּם וּפִי יְשָׁרִים יַצִּילֵם:

[משלי פרק-יב-טו]  דֶּרֶךְ אֱוִיל יָשָׁר בְּעֵינָיו וְשֹׁמֵעַ לְעֵצָה חָכָם:

[משלי פרק-כו-ז]  דַּלְיוּ שֹׁקַיִם מִפִּסֵּחַ וּמָשָׁל בְּפִי כְסִילִים:

[משלי פרק-ל-יד]  דּוֹר חֲרָבוֹת שִׁנָּיו וּמַאֲכָלוֹת מְתַלְּעֹתָיו לֶאֱכֹל עֲנִיִּים מֵאֶרֶץ וְאֶבְיוֹנִים מֵאָדָם:

[שיר השירים פרק-ב-ט]  דּוֹמֶה דוֹדִי לִצְבִי אוֹ לְעֹפֶר הָאַיָּלִים הִנֵּה זֶה עוֹמֵד אַחַר כָּתְלֵנוּ מַשְׁגִּיחַ מִן הַחֲלֹּנוֹת מֵצִיץ מִן הַחֲרַכִּים:

[שיר השירים פרק-ב-טז]  דּוֹדִי לִי וַאֲנִי לוֹ הָרֹעֶה בַּשּׁוֹשַׁנִּים:

[שיר השירים פרק-ו-ב]  דּוֹדִי יָרַד לְגַנּוֹ לַעֲרוּגוֹת הַבֹּשֶׂם לִרְעוֹת בַּגַּנִּים וְלִלְקֹט שׁוֹשַׁנִּים:

[איכה פרק-ג-י]  דֹּב אֹרֵב הוּא לִי (אריה) אֲרִי בְּמִסְתָּרִים:

[איכה פרק-ג-יא]  דְּרָכַי סוֹרֵר וַיְפַשְּׁחֵנִי שָׂמַנִי שֹׁמֵם:

[איכה פרק-ד-ד]  דָּבַק לְשׁוֹן יוֹנֵק אֶל חִכּוֹ בַּצָּמָא עוֹלָלִים שָׁאֲלוּ לֶחֶם פֹּרֵשׂ אֵין לָהֶם:

[קהלת פרק-א-א]  דִּבְרֵי קֹהֶלֶת בֶּן דָּוִד מֶלֶךְ בִּירוּשָׁלִָם:

[קהלת פרק-ט-יז]  דִּבְרֵי חֲכָמִים בְּנַחַת נִשְׁמָעִים מִזַּעֲקַת מוֹשֵׁל בַּכְּסִילִים:


ד-נ


[ויקרא פרק-כג-י]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם כִּי תָבֹאוּ אֶל הָאָרֶץ אֲשֶׁר אֲנִי נֹתֵן לָכֶם וּקְצַרְתֶּם אֶת קְצִירָהּ וַהֲבֵאתֶם אֶת עֹמֶר רֵאשִׁית קְצִירְכֶם אֶל הַכֹּהֵן:

[במדבר פרק-לג-נא]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם כִּי אַתֶּם עֹבְרִים אֶת הַיַּרְדֵּן אֶל אֶרֶץ כְּנָעַן:

[במדבר פרק-לה-י]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם כִּי אַתֶּם עֹבְרִים אֶת הַיַּרְדֵּן אַרְצָה כְּנָעַן:

[ישעיה פרק-לד-טז]  דִּרְשׁוּ מֵעַל סֵפֶר יְהוָה וּקְרָאוּ אַחַת מֵהֵנָּה לֹא נֶעְדָּרָה אִשָּׁה רְעוּתָהּ לֹא פָקָדוּ כִּי פִי הוּא צִוָּה וְרוּחוֹ הוּא קִבְּצָן:

[ירמיה פרק-א-א]  דִּבְרֵי יִרְמְיָהוּ בֶּן חִלְקִיָּהוּ מִן הַכֹּהֲנִים אֲשֶׁר בַּעֲנָתוֹת בְּאֶרֶץ בִּנְיָמִן:

[משלי פרק-יח-ח]  דִּבְרֵי נִרְגָּן כְּמִתְלַהֲמִים וְהֵם יָרְדוּ חַדְרֵי בָטֶן:

[משלי פרק-כו-כב]  דִּבְרֵי נִרְגָּן כְּמִתְלַהֲמִים וְהֵם יָרְדוּ חַדְרֵי בָטֶן:


ד-פ


[ירמיה פרק-ט-כא]  דַּבֵּר כֹּה נְאֻם יְהוָה וְנָפְלָה נִבְלַת הָאָדָם כְּדֹמֶן עַל פְּנֵי הַשָּׂדֶה וּכְעָמִיר מֵאַחֲרֵי הַקֹּצֵר וְאֵין מְאַסֵּף:

[איוב פרק-יט-ו]  דְּעוּ אֵפוֹ כִּי אֱלוֹהַּ עִוְּתָנִי וּמְצוּדוֹ עָלַי הִקִּיף:


ד-צ


[בראשית פרק-מב-ל]  דִּבֶּר הָאִישׁ אֲדֹנֵי הָאָרֶץ אִתָּנוּ קָשׁוֹת וַיִּתֵּן אֹתָנוּ כִּמְרַגְּלִים אֶת הָאָרֶץ:

[ויקרא פרק-יא-ב]  דַּבְּרוּ אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר זֹאת הַחַיָּה אֲשֶׁר תֹּאכְלוּ מִכָּל הַבְּהֵמָה אֲשֶׁר עַל הָאָרֶץ:

[משלי פרק-ל-יב]  דּוֹר טָהוֹר בְּעֵינָיו וּמִצֹּאָתוֹ לֹא רֻחָץ:

[איכה פרק-ג-יב]  דָּרַךְ קַשְׁתּוֹ וַיַּצִּיבֵנִי כַּמַּטָּרָא לַחֵץ:


ד-ר


[שמות פרק-א-ד]  דָּן וְנַפְתָּלִי גָּד וְאָשֵׁר:

[ויקרא פרק-יז-ב]  דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו וְאֶל כָּל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵיהֶם זֶה הַדָּבָר אֲשֶׁר צִוָּה יְהוָה לֵאמֹר:

[במדבר פרק-ב-י]  דֶּגֶל מַחֲנֵה רְאוּבֵן תֵּימָנָה לְצִבְאֹתָם וְנָשִׂיא לִבְנֵי רְאוּבֵן אֱלִיצוּר בֶּן שְׁדֵיאוּר:

[יחזקאל פרק-כז-יח]  דַּמֶּשֶׂק סֹחַרְתֵּךְ בְּרֹב מַעֲשַׂיִךְ מֵרֹב כָּל הוֹן בְּיֵין חֶלְבּוֹן וְצֶמֶר צָחַר:

[משלי פרק-יג-ה]  דְּבַר שֶׁקֶר יִשְׂנָא צַדִּיק וְרָשָׁע יַבְאִישׁ וְיַחְפִּיר:

[דברי הימים א פרק-ב-ב]  דָּן יוֹסֵף וּבִנְיָמִן נַפְתָּלִי גָּד וְאָשֵׁר:


ד-ש


[ויקרא פרק-כג-כד]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר בַּחֹדֶשׁ הַשְּׁבִיעִי בְּאֶחָד לַחֹדֶשׁ יִהְיֶה לָכֶם שַׁבָּתוֹן זִכְרוֹן תְּרוּעָה מִקְרָא קֹדֶשׁ:

[עמוס פרק-א-א]  דִּבְרֵי עָמוֹס אֲשֶׁר הָיָה בַנֹּקְדִים מִתְּקוֹעַ אֲשֶׁר חָזָה עַל יִשְׂרָאֵל בִּימֵי עֻזִּיָּה מֶלֶךְ יְהוּדָה וּבִימֵי יָרָבְעָם בֶּן יוֹאָשׁ מֶלֶךְ יִשְׂרָאֵל שְׁנָתַיִם לִפְנֵי הָרָעַשׁ:


ד-ת


[שמות פרק-יב-ג]  דַּבְּרוּ אֶל כָּל עֲדַת יִשְׂרָאֵל לֵאמֹר בֶּעָשֹׂר לַחֹדֶשׁ הַזֶּה וְיִקְחוּ לָהֶם אִישׁ שֶׂה לְבֵית אָבֹת שֶׂה לַבָּיִת:

[במדבר פרק-ח-ב]  דַּבֵּר אֶל אַהֲרֹן וְאָמַרְתָּ אֵלָיו בְּהַעֲלֹתְךָ אֶת הַנֵּרֹת אֶל מוּל פְּנֵי הַמְּנוֹרָה יָאִירוּ שִׁבְעַת הַנֵּרוֹת:

[במדבר פרק-טו-לח]  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵהֶם וְעָשׂוּ לָהֶם צִיצִת עַל כַּנְפֵי בִגְדֵיהֶם לְדֹרֹתָם וְנָתְנוּ עַל צִיצִת הַכָּנָף פְּתִיל תְּכֵלֶת:

[תהילים פרק-לז-ז]  דּוֹם לַיהוָה וְהִתְחוֹלֵל לוֹ אַל תִּתְחַר בְּמַצְלִיחַ דַּרְכּוֹ בְּאִישׁ עֹשֶׂה מְזִמּוֹת:

[תהילים פרק-קב-ז]  דָּמִיתִי לִקְאַת מִדְבָּר הָיִיתִי כְּכוֹס חֳרָבוֹת:

[משלי פרק-ז-כז]  דַּרְכֵי שְׁאוֹל בֵּיתָהּ יֹרְדוֹת אֶל חַדְרֵי מָוֶת:

[קהלת פרק-א-ד]  דּוֹר הֹלֵךְ וְדוֹר בָּא וְהָאָרֶץ לְעוֹלָם עֹמָדֶת:

[קהלת פרק-א-טז]  דִּבַּרְתִּי אֲנִי עִם לִבִּי לֵאמֹר אֲנִי הִנֵּה הִגְדַּלְתִּי וְהוֹסַפְתִּי חָכְמָה עַל כָּל אֲשֶׁר הָיָה לְפָנַי עַל יְרוּשָׁלִָם וְלִבִּי רָאָה הַרְבֵּה חָכְמָה וָדָעַת:

[עזרא פרק-ד-יא]  דְּנָה פַּרְשֶׁגֶן אִגַּרְתָּא דִּי שְׁלַחוּ עֲלוֹהִי עַל אַרְתַּחְשַׁשְׂתְּא מַלְכָּא עַבְדֶיךְ אֱנָשׁ עֲבַר נַהֲרָה וּכְעֶנֶת:

[עזרא פרק-ד-טו]  דִּי יְבַקַּר בִּסְפַר דָּכְרָנַיָּא דִּי אֲבָהָתָךְ וּתְהַשְׁכַּח בִּסְפַר דָּכְרָנַיָּא וְתִנְדַּע דִּי קִרְיְתָא דָךְ קִרְיָא מָרָדָא וּמְהַנְזְקַת מַלְכִין וּמְדִנָן וְאֶשְׁתַּדּוּר עָבְדִין בְּגַוַּהּ מִן יוֹמָת עָלְמָא עַל דְּנָה קִרְיְתָא דָךְ הָחָרְבַת:
