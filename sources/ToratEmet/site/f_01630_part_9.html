פרשת בהר



ויקרא פרק-כה

{א} וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה בְּהַר סִינַי לֵאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה בְּהַר סִינַי לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה בְּטּוּרָא דְסִינַי לְמֵימָר:

{ב} דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָֽמַרְתָּ אֲלֵהֶם כִּי תָבֹאוּ אֶל הָאָרֶץ אֲשֶׁר אֲנִי נֹתֵן לָכֶם וְשָֽׁבְתָה הָאָרֶץ שַׁבָּת לַֽיהוָֹֽה:  מ  דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאָֽמַרְתָּ אֲלֵהֶם כִּי תָבֹאוּ אֶל הָאָרֶץ אֲשֶׁר אֲנִי נֹתֵן לָכֶם וְשָֽׁבְתָה הָאָרֶץ שַׁבָּת לַֽיהוָֹֽה:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן אֲרֵי תֵעַלוּן לְאַרְעָא דִי אֲנָא יָהֵב לְכוֹן וְתַשְּׁמֵט אַרְעָא שְׁמִטְתָא קֳדָם יְיָ:

{ג} שֵׁשׁ שָׁנִים תִּזְרַע שָׂדֶךָ וְשֵׁשׁ שָׁנִים תִּזְמֹר כַּרְמֶךָ וְאָֽסַפְתָּ אֶת תְּבֽוּאָתָֽהּ:  מ  שֵׁשׁ שָׁנִים תִּזְרַע שָׂדֶךָ וְשֵׁשׁ שָׁנִים תִּזְמֹר כַּרְמֶךָ וְאָֽסַפְתָּ אֶת תְּבֽוּאָתָֽהּ:  ת  שִׁית שְׁנִין תִּזְרַע חַקְלָךְ וְשִׁית שְׁנִין תִכְסַח כַּרְמָךְ וְתִכְנוֹשׁ יָת עֲלַלְתָּהּ:

{ד} וּבַשָּׁנָה הַשְּׁבִיעִת שַׁבַּת שַׁבָּתוֹן יִהְיֶה לָאָרֶץ שַׁבָּת לַֽיהוָֹה שָֽׂדְךָ לֹא תִזְרָע וְכַרְמְךָ לֹא תִזְמֹֽר:  מ  וּבַשָּׁנָה הַשְּׁבִיעִת שַׁבַּת שַׁבָּתוֹן יִהְיֶה לָאָרֶץ שַׁבָּת לַֽיהוָֹה שָֽׂדְךָ לֹא תִזְרָע וְכַרְמְךָ לֹא תִזְמֹֽר:  ת  וּבְשַׁתָּא שְׁבִיעֲתָא נְיַח שְׁמִטְתָא יְהֵי לְאַרְעָא דְתַשְׁמֵט קֳדָם יְיָ חַקְלָךְ לָא תִזְרַע וְכַרְמָךְ לָא תִכְסָח:

{ה} אֵת סְפִיחַ קְצִֽירְךָ לֹא תִקְצוֹר וְאֶת עִנְּבֵי נְזִירֶךָ לֹא תִבְצֹר שְׁנַת שַׁבָּתוֹן יִהְיֶה לָאָֽרֶץ:  מ  אֵת סְפִיחַ קְצִֽירְךָ לֹא תִקְצוֹר וְאֶת עִנְּבֵי נְזִירֶךָ לֹא תִבְצֹר שְׁנַת שַׁבָּתוֹן יִהְיֶה לָאָֽרֶץ:  ת  יָת כַּתֵּי דַחֲצָדָךְ לָא תֶחֱצַד וְיָת עִנְבֵי שִׁבְקָךְ לָא תִקְטוֹף שְׁנַת שְׁמִטְתָא יְהֵי לְאַרְעָא:

{ו} וְֽהָֽיְתָה שַׁבַּת הָאָרֶץ לָכֶם לְאָכְלָה לְךָ וּלְעַבְדְּךָ וְלַֽאֲמָתֶךָ וְלִשְׂכִֽירְךָ וּלְתוֹשָׁבְךָ הַגָּרִים עִמָּֽךְ:  מ  וְֽהָֽיְתָה שַׁבַּת הָאָרֶץ לָכֶם לְאָכְלָה לְךָ וּלְעַבְדְּךָ וְלַֽאֲמָתֶךָ וְלִשְׂכִֽירְךָ וּלְתוֹשָׁבְךָ הַגָּרִים עִמָּֽךְ:  ת  וּתְהֵי שְׁמִטַת אַרְעָא לְכוֹן לְמֵיכַל לָךְ וּלְעַבְדָךְ וּלְאַמְתָךְ וְלַאֲגִירָךְ וּלְתוֹתָבָךְ דְדָיְרִין עִמָךְ:

{ז} וְלִֽבְהֶמְתְּךָ וְלַֽחַיָּה אֲשֶׁר בְּאַרְצֶךָ תִּהְיֶה כָל תְּבֽוּאָתָהּ לֶֽאֱכֹֽל:  מ  וְלִֽבְהֶמְתְּךָ וְלַֽחַיָּה אֲשֶׁר בְּאַרְצֶךָ תִּהְיֶה כָל תְּבֽוּאָתָהּ לֶֽאֱכֹֽל:  ת  וְלִבְעִירָךְ וּלְחַיְתָא דִי בְאַרְעָךְ תְּהֵי כָל עֲלַלְתַּהּ לְמֵיכָל: (ס)

{ח} וְסָֽפַרְתָּ לְךָ שֶׁבַע שַׁבְּתֹת שָׁנִים שֶׁבַע שָׁנִים שֶׁבַע פְּעָמִים וְהָיוּ לְךָ יְמֵי שֶׁבַע שַׁבְּתֹת הַשָּׁנִים תֵּשַׁע וְאַרְבָּעִים שָׁנָֽה:  מ  וְסָֽפַרְתָּ לְךָ שֶׁבַע שַׁבְּתֹת שָׁנִים שֶׁבַע שָׁנִים שֶׁבַע פְּעָמִים וְהָיוּ לְךָ יְמֵי שֶׁבַע שַׁבְּתֹת הַשָּׁנִים תֵּשַׁע וְאַרְבָּעִים שָׁנָֽה:  ת  וְתִמְנֵי לָךְ שְׁבַע שְׁמִיטִין דִשְׁנִין שְׁבַע שְׁנִין שְׁבַע זִמְנִין וִיהוֹן לָךְ יוֹמֵי שְּׁבַע שְׁמִטִין דִשְׁנִין אַרְבְּעִין וְתֵשַׁע שְׁנִין:

{ט} וְהַֽעֲבַרְתָּ שׁוֹפַר תְּרוּעָה בַּחֹדֶשׁ הַשְּׁבִעִי בֶּֽעָשׂוֹר לַחֹדֶשׁ בְּיוֹם הַכִּפֻּרִים תַּֽעֲבִירוּ שׁוֹפָר בְּכָֽל אַרְצְכֶֽם:  מ  וְהַֽעֲבַרְתָּ שׁוֹפַר תְּרוּעָה בַּחֹדֶשׁ הַשְּׁבִעִי בֶּֽעָשׂוֹר לַחֹדֶשׁ בְּיוֹם הַכִּפֻּרִים תַּֽעֲבִירוּ שׁוֹפָר בְּכָֽל אַרְצְכֶֽם:  ת  וְתַעֲבַר שׁוֹפָר יַבָּבָא בְּיַרְחָא שְׁבִיעָאָה בְּעַשְׂרָא לְיַרְחָא בְּיוֹמָא דְכִפּוּרַיָא תַּעַבְרוּן שׁוֹפָרָא בְּכָל אַרְעֲכוֹן:

{י} וְקִדַּשְׁתֶּם אֵת שְׁנַת הַֽחֲמִשִּׁים שָׁנָה וּקְרָאתֶם דְּרוֹר בָּאָרֶץ לְכָל יֹֽשְׁבֶיהָ יוֹבֵל הִוא תִּהְיֶה לָכֶם וְשַׁבְתֶּם אִישׁ אֶל אֲחֻזָּתוֹ וְאִישׁ אֶל מִשְׁפַּחְתּוֹ תָּשֻֽׁבוּ:  מ  וְקִדַּשְׁתֶּם אֵת שְׁנַת הַֽחֲמִשִּׁים שָׁנָה וּקְרָאתֶם דְּרוֹר בָּאָרֶץ לְכָל יֹֽשְׁבֶיהָ יוֹבֵל הִוא תִּהְיֶה לָכֶם וְשַׁבְתֶּם אִישׁ אֶל אֲחֻזָּתוֹ וְאִישׁ אֶל מִשְׁפַּחְתּוֹ תָּשֻֽׁבוּ:  ת  וּתְקַדְשׁוּן יָת שְׁנַת חַמְשִׁין שְׁנִין וְתַקְרוּן חֵרוּתָא בְּאַרְעָא לְכָל יָתְבָהָא יוֹבֵלָא הִיא תְּהֵי לְכוֹן וּתְתוּבוּן גְבַר לְאַחֲסַנְתֵּהּ וּגְבַר לְזַרְעִיתֵהּ תְּתוּבוּן:

{יא} יוֹבֵל הִוא שְׁנַת הַֽחֲמִשִּׁים שָׁנָה תִּהְיֶה לָכֶם לֹא תִזְרָעוּ וְלֹא תִקְצְרוּ אֶת סְפִיחֶיהָ וְלֹא תִבְצְרוּ אֶת נְזִרֶֽיהָ:  מ  יוֹבֵל הִוא שְׁנַת הַֽחֲמִשִּׁים שָׁנָה תִּהְיֶה לָכֶם לֹא תִזְרָעוּ וְלֹא תִקְצְרוּ אֶת סְפִיחֶיהָ וְלֹא תִבְצְרוּ אֶת נְזִרֶֽיהָ:  ת  יוֹבֵלָא הִיא שְׁנַת חַמְשִׁין שְׁנִין תְּהֵי לְכוֹן לָא תִזְרְעוּן וְלָא תַחְצְדוּן יַת כַּתָּהָא וְלָא תִקְטְפוּן יַת שִׁבְקָהָא:

{יב} כִּי יוֹבֵל הִוא קֹדֶשׁ תִּהְיֶה לָכֶם מִן הַשָּׂדֶה תֹּֽאכְלוּ אֶת תְּבֽוּאָתָֽהּ:  מ  כִּי יוֹבֵל הִוא קֹדֶשׁ תִּהְיֶה לָכֶם מִן הַשָּׂדֶה תֹּֽאכְלוּ אֶת תְּבֽוּאָתָֽהּ:  ת  אֲרֵי יוֹבֵלָא הִיא קוּדְשָׁא תְּהֵי לְכוֹן מִן חַקְלָא תֵּיכְלוּן יָת עֲלַלְתָּהּ:

{יג} בִּשְׁנַת הַיּוֹבֵל הַזֹּאת תָּשֻׁבוּ אִישׁ אֶל אֲחֻזָּתֽוֹ:  מ  בִּשְׁנַת הַיּוֹבֵל הַזֹּאת תָּשֻׁבוּ אִישׁ אֶל אֲחֻזָּתֽוֹ:  ת  בְּשַׁתָּא דְיוֹבֵלָא הָדָא תְּתוּבוּן גְבַר לְאַחֲסַנְתֵּהּ:

{יד}  שני  וְכִֽי תִמְכְּרוּ מִמְכָּר לַֽעֲמִיתֶךָ אוֹ קָנֹה מִיַּד עֲמִיתֶךָ אַל תּוֹנוּ אִישׁ אֶת אָחִֽיו:  מ  וְכִֽי תִמְכְּרוּ מִמְכָּר לַֽעֲמִיתֶךָ אוֹ קָנֹה מִיַּד עֲמִיתֶךָ אַל תּוֹנוּ אִישׁ אֶת אָחִֽיו:  ת  וַאֲרֵי תְזַבֵּן זְבִינִין לְחַבְרָךְ אוֹ תִזְבֵּן מִידָא דְחַבְרָךְ לָא תוֹנוּן גְבַר יָת אֲחוֹהִי:

{טו} בְּמִסְפַּר שָׁנִים אַחַר הַיּוֹבֵל תִּקְנֶה מֵאֵת עֲמִיתֶךָ בְּמִסְפַּר שְׁנֵֽי תְבוּאֹת יִמְכָּר לָֽךְ:  מ  בְּמִסְפַּר שָׁנִים אַחַר הַיּוֹבֵל תִּקְנֶה מֵאֵת עֲמִיתֶךָ בְּמִסְפַּר שְׁנֵֽי תְבוּאֹת יִמְכָּר לָֽךְ:  ת  בְּמִנְיַן שְׁנַיָא בָּתַר יוֹבֵלָא תִּזְבֵּן מִן חַבְרָךְ בְּמִנְיַן שְׁנֵי עֲלַלְתָּא יְזַבֶּן לָךְ:

{טז} לְפִי רֹב הַשָּׁנִים תַּרְבֶּה מִקְנָתוֹ וּלְפִי מְעֹט הַשָּׁנִים תַּמְעִיט מִקְנָתוֹ כִּי מִסְפַּר תְּבוּאֹת הוּא מֹכֵר לָֽךְ:  מ  לְפִי רֹב הַשָּׁנִים תַּרְבֶּה מִקְנָתוֹ וּלְפִי מְעֹט הַשָּׁנִים תַּמְעִיט מִקְנָתוֹ כִּי מִסְפַּר תְּבוּאֹת הוּא מֹכֵר לָֽךְ:  ת  לְפוּם סְגִיאוּת שְׁנַיָא תַּסְגֵי זְבִינוֹהִי וּלְפוּם זְעֵרוּת שְׁנַיָא תַּזְעֵר זְבִינוֹהִי אֲרֵי מִנְיַן עֲלַלְתָּא הוּא מְזַבֵּן לָךְ:

{יז} וְלֹא תוֹנוּ אִישׁ אֶת עֲמִיתוֹ וְיָרֵאתָ מֵֽאֱלֹהֶיךָ כִּי אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  מ  וְלֹא תוֹנוּ אִישׁ אֶת עֲמִיתוֹ וְיָרֵאתָ מֵֽאֱלֹהֶיךָ כִּי אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  ת  וְלָא תוֹנוּן גְבַר יָת חַבְרֵהּ וְתִדְחַל מֵאֱלָהָךְ אֲרֵי אֲנָא יְיָ אֱלָהֲכוֹן:

{יח} וַֽעֲשִׂיתֶם אֶת חֻקֹּתַי וְאֶת מִשְׁפָּטַי תִּשְׁמְרוּ וַֽעֲשִׂיתֶם אֹתָם וִֽישַׁבְתֶּם עַל הָאָרֶץ לָבֶֽטַח:  מ  וַֽעֲשִׂיתֶם אֶת חֻקֹּתַי וְאֶת מִשְׁפָּטַי תִּשְׁמְרוּ וַֽעֲשִׂיתֶם אֹתָם וִֽישַׁבְתֶּם עַל הָאָרֶץ לָבֶֽטַח:  ת  וְתַעְבְּדוּן יָת קְיָמַי וְיָת דִינַי תִּטְּרוּן וְתַעֲבְּדוּן יָתְהוֹן וְתֵיתְבוּן עַל אַרְעָא לְרָחֲצָן:

{יט}  שלישי - שני במחוברין  וְנָֽתְנָה הָאָרֶץ פִּרְיָהּ וַֽאֲכַלְתֶּם לָשֹׂבַע וִֽישַׁבְתֶּם לָבֶטַח עָלֶֽיהָ:  מ  וְנָֽתְנָה הָאָרֶץ פִּרְיָהּ וַֽאֲכַלְתֶּם לָשֹׂבַע וִֽישַׁבְתֶּם לָבֶטַח עָלֶֽיהָ:  ת  וְתִתֵּן אַרְעָא אִבַּהּ וְתֵיכְלוּן לְמִשְׂבָּע וְתֵיתְבוּן לְרָחֲצָן עֲלַהּ:

{כ} וְכִי תֹֽאמְרוּ מַה נֹּאכַל בַּשָּׁנָה הַשְּׁבִיעִת הֵן לֹא נִזְרָע וְלֹא נֶֽאֱסֹף אֶת תְּבֽוּאָתֵֽנוּ:  מ  וְכִי תֹֽאמְרוּ מַה נֹּאכַל בַּשָּׁנָה הַשְּׁבִיעִת הֵן לֹא נִזְרָע וְלֹא נֶֽאֱסֹף אֶת תְּבֽוּאָתֵֽנוּ:  ת  וַאֲרֵי תֵימְרוּן מָא נֵיכוּל בְּשַׁתָּא שְׁבִיעֵתָא הָא לָא נִזְרָע וְלָא נִכְנוֹשׁ יָת עֲלַלְתָּנָא:

{כא} וְצִוִּיתִי אֶת בִּרְכָתִי לָכֶם בַּשָּׁנָה הַשִּׁשִּׁית וְעָשָׂת אֶת הַתְּבוּאָה לִשְׁלֹשׁ הַשָּׁנִֽים:  מ  וְצִוִּיתִי אֶת בִּרְכָתִי לָכֶם בַּשָּׁנָה הַשִּׁשִּׁית וְעָשָׂת אֶת הַתְּבוּאָה לִשְׁלֹשׁ הַשָּׁנִֽים:  ת  וֶאֱפַקֵד יָת בִּרְכְּתִי לְכוֹן בְּשַׁתָּא שְׁתִיתֵתָא וְתַעְבֵּד יָת עֲלַלְתָּא לִתְלַת שְׁנִין:

{כב} וּזְרַעְתֶּם אֵת הַשָּׁנָה הַשְּׁמִינִת וַֽאֲכַלְתֶּם מִן הַתְּבוּאָה יָשָׁן עַד הַשָּׁנָה הַתְּשִׁיעִת עַד בּוֹא תְּבוּאָתָהּ תֹּֽאכְלוּ יָשָֽׁן:  מ  וּזְרַעְתֶּם אֵת הַשָּׁנָה הַשְּׁמִינִת וַֽאֲכַלְתֶּם מִן הַתְּבוּאָה יָשָׁן עַד הַשָּׁנָה הַתְּשִׁיעִת עַד בּוֹא תְּבוּאָתָהּ תֹּֽאכְלוּ יָשָֽׁן:  ת  וְתִזְרְעוּן יָת שַׁתָּא תְמִינֵתָא וְתֵיכְלוּן מִן עֲלַלְתָּא עַתִּיקָא עַד שַׁתָּא תְשִׁיעֵתָא עַד מֵעַל עֲלַלְתַּהּ תֵּיכְלוּן עַתִּיקָא:

{כג} וְהָאָרֶץ לֹא תִמָּכֵר לִצְמִתֻת כִּי לִי הָאָרֶץ כִּֽי גֵרִים וְתֽוֹשָׁבִים אַתֶּם עִמָּדִֽי:  מ  וְהָאָרֶץ לֹא תִמָּכֵר לִצְמִתֻת כִּי לִי הָאָרֶץ כִּֽי גֵרִים וְתֽוֹשָׁבִים אַתֶּם עִמָּדִֽי:  ת  וְאַרְעָא לָא תִזְדַבַּן לַחֲלוּטִין אֲרֵי דִילִי אַרְעָא אֲרֵי דַיָרִין וְתוֹתָבִין אַתּוּן קֳדָמַי:

{כד} וּבְכֹל אֶרֶץ אֲחֻזַּתְכֶם גְּאֻלָּה תִּתְּנוּ לָאָֽרֶץ:  מ  וּבְכֹל אֶרֶץ אֲחֻזַּתְכֶם גְּאֻלָּה תִּתְּנוּ לָאָֽרֶץ:  ת  וּבְכֹל אַרְעָא אַחֲסַנְתְּכוֹן פֻּרְקָנָא תִּתְּנוּן לְאַרְעָא: (ס)

{כה}  רביעי  כִּֽי יָמוּךְ אָחִיךָ וּמָכַר מֵֽאֲחֻזָּתוֹ וּבָא גֹֽאֲלוֹ הַקָּרֹב אֵלָיו וְגָאַל אֵת מִמְכַּר אָחִֽיו:  מ  כִּֽי יָמוּךְ אָחִיךָ וּמָכַר מֵֽאֲחֻזָּתוֹ וּבָא גֹֽאֲלוֹ הַקָּרֹב אֵלָיו וְגָאַל אֵת מִמְכַּר אָחִֽיו:  ת  אֲרֵי יִתְמַסְכַּן אָחוּךְ וִיזַבֵּן מֵאַחֲסַנְתֵּהּ וְיֵיתֵי פָרִיקֵהּ דְקָרִיב לֵהּ וְיִפְרוֹק יָת זְבִינֵי אֲחוּהִי:

{כו} וְאִישׁ כִּי לֹא יִֽהְיֶה לּוֹ גֹּאֵל וְהִשִּׂיגָה יָדוֹ וּמָצָא כְּדֵי גְאֻלָּתֽוֹ:  מ  וְאִישׁ כִּי לֹא יִֽהְיֶה לּוֹ גֹּאֵל וְהִשִּׂיגָה יָדוֹ וּמָצָא כְּדֵי גְאֻלָּתֽוֹ:  ת  וּגְבַר אֲרֵי לָא יְהֵי לֵהּ פָּרִיק וְתַדְבֵּק יְדֵהּ וְיַשְׁכַּח כְּמִסַת פֻּרְקָנֵהּ:

{כז} וְחִשַּׁב אֶת שְׁנֵי מִמְכָּרוֹ וְהֵשִׁיב אֶת הָעֹדֵף לָאִישׁ אֲשֶׁר מָֽכַר לוֹ וְשָׁב לַֽאֲחֻזָּתֽוֹ:  מ  וְחִשַּׁב אֶת שְׁנֵי מִמְכָּרוֹ וְהֵשִׁיב אֶת הָעֹדֵף לָאִישׁ אֲשֶׁר מָֽכַר לוֹ וְשָׁב לַֽאֲחֻזָּתֽוֹ:  ת  וִיחַשֵׁב יָת שְׁנֵי זְבִינוֹהִי וְיָתֵב יָת מוֹתָרָא לִגְבַר דִי זַבֶּן לֵהּ וִיתוּב לְאַחֲסַנְתֵּהּ:

{כח} וְאִם לֹא מָֽצְאָה יָדוֹ דֵּי הָשִׁיב לוֹ וְהָיָה מִמְכָּרוֹ בְּיַד הַקֹּנֶה אֹתוֹ עַד שְׁנַת הַיּוֹבֵל וְיָצָא בַּיֹּבֵל וְשָׁב לַֽאֲחֻזָּתֽוֹ:  מ  וְאִם לֹא מָֽצְאָה יָדוֹ דֵּי הָשִׁיב לוֹ וְהָיָה מִמְכָּרוֹ בְּיַד הַקֹּנֶה אֹתוֹ עַד שְׁנַת הַיּוֹבֵל וְיָצָא בַּיֹּבֵל וְשָׁב לַֽאֲחֻזָּתֽוֹ:  ת  וְאִם לָא תַשְׁכַּח יְדֵהּ כְּמִסַת אֲתֵב לֵהּ וִיהֵי זְבִינוֹהִי בִּידָא דִזְבַן יָתֵהּ עַד שַׁתָּא דְיוֹבֵלָא וְיִפּוֹק בְּיוֹבֵלָא וִיתוּב לְאַחֲסַנְתֵּהּ: (ס)

{כט}  חמישי - שלישי במחוברין  וְאִישׁ כִּֽי יִמְכֹּר בֵּית מוֹשַׁב עִיר חוֹמָה וְהָֽיְתָה גְּאֻלָּתוֹ עַד תֹּם שְׁנַת מִמְכָּרוֹ יָמִים תִּהְיֶה גְאֻלָּתֽוֹ:  מ  וְאִישׁ כִּֽי יִמְכֹּר בֵּית מוֹשַׁב עִיר חוֹמָה וְהָֽיְתָה גְּאֻלָּתוֹ עַד תֹּם שְׁנַת מִמְכָּרוֹ יָמִים תִּהְיֶה גְאֻלָּתֽוֹ:  ת  וּגְבַר אֲרֵי יְזַבֵּן בֵּית מוֹתַב קַרְתָּא מַקְפָא שׁוּר וּתְהֵי פוּרְקָנֵהּ עַד מִשְׁלַם שַׁתָּא דִזְבִינוֹהִי עִדָן בְּעִדָן תְּהֵי פוּרְקָנֵהּ:

{ל} וְאִם לֹֽא יִגָּאֵל עַד מְלֹאת לוֹ שָׁנָה תְמִימָה וְקָם הַבַּיִת אֲשֶׁר בָּעִיר אֲשֶׁר (לא) לוֹ חֹמָה לַצְּמִיתֻת לַקֹּנֶה אֹתוֹ לְדֹֽרֹתָיו לֹא יֵצֵא בַּיֹּבֵֽל:  מ  וְאִם לֹֽא יִגָּאֵל עַד מְלֹאת לוֹ שָׁנָה תְמִימָה וְקָם הַבַּיִת אֲשֶׁר בָּעִיר אֲשֶׁר (לא) לוֹ חֹמָה לַצְּמִיתֻת לַקֹּנֶה אֹתוֹ לְדֹֽרֹתָיו לֹא יֵצֵא בַּיֹּבֵֽל:  ת  וְאִם לָא יִתְפָּרַק עַד מִשְׁלַם לֵהּ שַׁתָּא שְׁלֶמְתָּא וִיקוּם בֵּיתָא דִי בְקַרְתָּא דִי לֵהּ שׁוּרָא לַחֲלוּטִין לְדִי זְבַן יָתֵהּ לְדָרוֹהִי לָא יִפּוֹק בְּיוֹבֵלָא:

{לא} וּבָתֵּי הַֽחֲצֵרִים אֲשֶׁר אֵין לָהֶם חֹמָה סָבִיב עַל שְׂדֵה הָאָרֶץ יֵֽחָשֵׁב גְּאֻלָּה תִּֽהְיֶה לּוֹ וּבַיֹּבֵל יֵצֵֽא:  מ  וּבָתֵּי הַֽחֲצֵרִים אֲשֶׁר אֵין לָהֶם חֹמָה סָבִיב עַל שְׂדֵה הָאָרֶץ יֵֽחָשֵׁב גְּאֻלָּה תִּֽהְיֶה לּוֹ וּבַיֹּבֵל יֵצֵֽא:  ת  וּבָתֵּי פַצְחַיָא דִי לֵית לְהוֹן שׁוּר מַקַף סְחוֹר סְחוֹר עַל חֲקַל אַרְעָא יִתְחַשָׁב פֻּרְקָנָא תְּהֵי לֵהּ וּבְיוֹבֵלָא יִפּוֹק:

{לב} וְעָרֵי הַֽלְוִיִּם בָּתֵּי עָרֵי אֲחֻזָּתָם גְּאֻלַּת עוֹלָם תִּהְיֶה לַֽלְוִיִּֽם:  מ  וְעָרֵי הַֽלְוִיִּם בָּתֵּי עָרֵי אֲחֻזָּתָם גְּאֻלַּת עוֹלָם תִּהְיֶה לַֽלְוִיִּֽם:  ת  וְקִרְוֵי לֵיוָאֵי בָּתֵּי קִרְוֵי אַחֲסַנְתְּהוֹן פּוּרְקַן עָלָם תְּהֵי לְלֵיוָאֵי:

{לג} וַֽאֲשֶׁר יִגְאַל מִן הַֽלְוִיִּם וְיָצָא מִמְכַּר בַּיִת וְעִיר אֲחֻזָּתוֹ בַּיֹּבֵל כִּי בָֽתֵּי עָרֵי הַֽלְוִיִּם הִוא אֲחֻזָּתָם בְּתוֹךְ בְּנֵי יִשְׂרָאֵֽל:  מ  וַֽאֲשֶׁר יִגְאַל מִן הַֽלְוִיִּם וְיָצָא מִמְכַּר בַּיִת וְעִיר אֲחֻזָּתוֹ בַּיֹּבֵל כִּי בָֽתֵּי עָרֵי הַֽלְוִיִּם הִוא אֲחֻזָּתָם בְּתוֹךְ בְּנֵי יִשְׂרָאֵֽל:  ת  וְדִי יִפְרוֹק מִן לֵיוָאֵי וְיִפּוֹק זְבִין בֵּיתָא וְקַרְתָּא אַחֲסַנְתֵּהּ בְּיוֹבֵלָא אֲרֵי בָתֵּי קִרְוֵי לֵיוָאֵי הִיא אַחְסַנְתְּהוֹן בְּגוֹ בְּנֵי יִשְׂרָאֵל:

{לד} וּֽשֲׂדֵה מִגְרַשׁ עָֽרֵיהֶם לֹא יִמָּכֵר כִּֽי אֲחֻזַּת עוֹלָם הוּא לָהֶֽם:  מ  וּֽשֲׂדֵה מִגְרַשׁ עָֽרֵיהֶם לֹא יִמָּכֵר כִּֽי אֲחֻזַּת עוֹלָם הוּא לָהֶֽם:  ת  וַחֲקַל רְוַח קִרְוֵיהוֹן לָא יִזְדַבָּן אֲרֵי אַחֲסָנַת עָלָם הוּא לְהוֹן: (ס)

{לה} וְכִֽי יָמוּךְ אָחִיךָ וּמָטָה יָדוֹ עִמָּךְ וְהֶֽחֱזַקְתָּ בּוֹ גֵּר וְתוֹשָׁב וָחַי עִמָּֽךְ:  מ  וְכִֽי יָמוּךְ אָחִיךָ וּמָטָה יָדוֹ עִמָּךְ וְהֶֽחֱזַקְתָּ בּוֹ גֵּר וְתוֹשָׁב וָחַי עִמָּֽךְ:  ת  וַאֲרֵי יִתְמַסְכַּן אָחוּךְ וּתְמוּט יְדֵהּ עִמָךְ וְתַתְקַף בֵּהּ דַיַר וְתוֹתָב וִיחֵי עִמָךְ:

{לו} אַל תִּקַּח מֵֽאִתּוֹ נֶשֶׁךְ וְתַרְבִּית וְיָרֵאתָ מֵֽאֱלֹהֶיךָ וְחֵי אָחִיךָ עִמָּֽךְ:  מ  אַל תִּקַּח מֵֽאִתּוֹ נֶשֶׁךְ וְתַרְבִּית וְיָרֵאתָ מֵֽאֱלֹהֶיךָ וְחֵי אָחִיךָ עִמָּֽךְ:  ת  לָא תִסַב מִנֵהּ חִבּוּלְיָא וְרִבִּיתָא וְתִדְחַל מֵאֱלָהָךְ וִיחֵי אָחוּךְ עִמָךְ:

{לז} אֶֽת כַּסְפְּךָ לֹֽא תִתֵּן לוֹ בְּנֶשֶׁךְ וּבְמַרְבִּית לֹֽא תִתֵּן אָכְלֶֽךָ:  מ  אֶֽת כַּסְפְּךָ לֹֽא תִתֵּן לוֹ בְּנֶשֶׁךְ וּבְמַרְבִּית לֹֽא תִתֵּן אָכְלֶֽךָ:  ת  יָת כַּסְפָּךְ לָא תִתֵּן לֵהּ בְּחִבּוּלְיָא וּבְרִבִּיתָא לָא תִתֵּן מֵיכְלָךְ:

{לח} אֲנִי יְהוָֹה אֱלֹהֵיכֶם אֲשֶׁר הוֹצֵאתִי אֶתְכֶם מֵאֶרֶץ מִצְרָיִם לָתֵת לָכֶם אֶת אֶרֶץ כְּנַעַן לִהְיוֹת לָכֶם לֵֽאלֹהִֽים:  מ  אֲנִי יְהוָֹה אֱלֹהֵיכֶם אֲשֶׁר הוֹצֵאתִי אֶתְכֶם מֵאֶרֶץ מִצְרָיִם לָתֵת לָכֶם אֶת אֶרֶץ כְּנַעַן לִהְיוֹת לָכֶם לֵֽאלֹהִֽים:  ת  אֲנָא יְיָ אֱלָהֲכוֹן דִי אַפֵּקִית יָתְכוֹן מֵאַרְעָא דְמִצְרָיִם לְמִתַּן לְכוֹן יָת אַרְעָא דִכְנַעַן לְמֶהֱוֵי לְכוֹן לֶאֱלָהּ: (ס)

{לט}  שישי - רביעי במחוברין  וְכִֽי יָמוּךְ אָחִיךָ עִמָּךְ וְנִמְכַּר לָךְ לֹא תַֽעֲבֹד בּוֹ עֲבֹדַת עָֽבֶד:  מ  וְכִֽי יָמוּךְ אָחִיךָ עִמָּךְ וְנִמְכַּר לָךְ לֹא תַֽעֲבֹד בּוֹ עֲבֹדַת עָֽבֶד:  ת  וַאֲרֵי יִתְמַסְכַּן אָחוּךְ עִמָךְ וְיִזְדַבַּן לָךְ לָא תִפְלַח בֵּהּ פָּלְחַן עַבְדִין:

{מ} כְּשָׂכִיר כְּתוֹשָׁב יִהְיֶה עִמָּךְ עַד שְׁנַת הַיֹּבֵל יַֽעֲבֹד עִמָּֽךְ:  מ  כְּשָׂכִיר כְּתוֹשָׁב יִהְיֶה עִמָּךְ עַד שְׁנַת הַיֹּבֵל יַֽעֲבֹד עִמָּֽךְ:  ת  כַּאֲגִירָא כְתוֹתָבָא יְהֵי עִמָךְ עַד שַׁתָּא דְיוֹבֵלָא יִפְלַח עִמָךְ:

{מא} וְיָצָא מֵֽעִמָּךְ הוּא וּבָנָיו עִמּוֹ וְשָׁב אֶל מִשְׁפַּחְתּוֹ וְאֶל אֲחֻזַּת אֲבֹתָיו יָשֽׁוּב:  מ  וְיָצָא מֵֽעִמָּךְ הוּא וּבָנָיו עִמּוֹ וְשָׁב אֶל מִשְׁפַּחְתּוֹ וְאֶל אֲחֻזַּת אֲבֹתָיו יָשֽׁוּב:  ת  וְיִפּוֹק מֵעִמָךְ הוּא וּבְנוֹהִי עִמֵהּ וִיתוּב לְזַרְעִיתֵהּ וּלְאַחֲסָנַת אֲבָהָתוֹהִי יְתוּב:

{מב} כִּֽי עֲבָדַי הֵם אֲשֶׁר הוֹצֵאתִי אֹתָם מֵאֶרֶץ מִצְרָיִם לֹא יִמָּֽכְרוּ מִמְכֶּרֶת עָֽבֶד:  מ  כִּֽי עֲבָדַי הֵם אֲשֶׁר הוֹצֵאתִי אֹתָם מֵאֶרֶץ מִצְרָיִם לֹא יִמָּֽכְרוּ מִמְכֶּרֶת עָֽבֶד:  ת  אֲרֵי עַבְדַי אִנוּן דִי אַפֵּקִית יָתְהוֹן מֵאַרְעָא דְמִצְרָיִם לָא יִזְדַבְּנוּן זְבוּן עַבְדִין:

{מג} לֹֽא תִרְדֶּה בוֹ בְּפָרֶךְ וְיָרֵאתָ מֵֽאֱלֹהֶֽיךָ:  מ  לֹֽא תִרְדֶּה בוֹ בְּפָרֶךְ וְיָרֵאתָ מֵֽאֱלֹהֶֽיךָ:  ת  לָא תִפְלָח בֵּהּ בְּקַשְׁיוּ וְתִדְחַל מֵאֱלָהָךְ:

{מד} וְעַבְדְּךָ וַֽאֲמָתְךָ אֲשֶׁר יִֽהְיוּ לָךְ מֵאֵת הַגּוֹיִם אֲשֶׁר סְבִיבֹתֵיכֶם מֵהֶם תִּקְנוּ עֶבֶד וְאָמָֽה:  מ  וְעַבְדְּךָ וַֽאֲמָתְךָ אֲשֶׁר יִֽהְיוּ לָךְ מֵאֵת הַגּוֹיִם אֲשֶׁר סְבִיבֹתֵיכֶם מֵהֶם תִּקְנוּ עֶבֶד וְאָמָֽה:  ת  וְעַבְדָךְ וְאַמְתָךְ דִי יְהוֹן לָךְ מִן עַמְמַיָא דִי בְסַחֲרָנֵיכוֹן מִנְהוֹן תִּקְנוּן עַבְדִין וְאַמְהָן:

{מה} וְגַם מִבְּנֵי הַתּֽוֹשָׁבִים הַגָּרִים עִמָּכֶם מֵהֶם תִּקְנוּ וּמִמִּשְׁפַּחְתָּם אֲשֶׁר עִמָּכֶם אֲשֶׁר הוֹלִידוּ בְּאַרְצְכֶם וְהָיוּ לָכֶם לַֽאֲחֻזָּֽה:  מ  וְגַם מִבְּנֵי הַתּֽוֹשָׁבִים הַגָּרִים עִמָּכֶם מֵהֶם תִּקְנוּ וּמִמִּשְׁפַּחְתָּם אֲשֶׁר עִמָּכֶם אֲשֶׁר הוֹלִידוּ בְּאַרְצְכֶם וְהָיוּ לָכֶם לַֽאֲחֻזָּֽה:  ת  וְאַף מִבְּנֵי תוֹתָבַיָא עָרְלַיָא דְדָיְרִין עִמְכוֹן מִנְהוֹן תִּקְנוּן וּמִזַרְעִיתְהוֹן דְעִמְכוֹן דִי אִתְיְלִידוּ בְּאַרְעֲכוֹן וִיהוֹן לְכוֹן לְאַחֲסָנָא:

{מו} וְהִתְנַֽחַלְתֶּם אֹתָם לִבְנֵיכֶם אַֽחֲרֵיכֶם לָרֶשֶׁת אֲחֻזָּה לְעֹלָם בָּהֶם תַּֽעֲבֹדוּ וּבְאַחֵיכֶם בְּנֵֽי יִשְׂרָאֵל אִישׁ בְּאָחִיו לֹֽא תִרְדֶּה בוֹ בְּפָֽרֶךְ:  מ  וְהִתְנַֽחַלְתֶּם אֹתָם לִבְנֵיכֶם אַֽחֲרֵיכֶם לָרֶשֶׁת אֲחֻזָּה לְעֹלָם בָּהֶם תַּֽעֲבֹדוּ וּבְאַחֵיכֶם בְּנֵֽי יִשְׂרָאֵל אִישׁ בְּאָחִיו לֹֽא תִרְדֶּה בוֹ בְּפָֽרֶךְ:  ת  וְתַחֲסוּן יָתְהוֹן לִבְנֵיכוֹן בַּתְרֵיכוֹן לִירוּתַת אַחֲסָנָא לְעָלָם בְּהוֹן תִּפְלְחוּן וּבְאֲחֵיכוֹן בְּנֵי יִשְׂרָאֵל גְבַר בַּאֲחוּהִי לָא תִפְלַח בֵּהּ בְּקַשְׁיוּ: (ס)

{מז}  שביעי  וְכִי תַשִּׂיג יַד גֵּר וְתוֹשָׁב עִמָּךְ וּמָךְ אָחִיךָ עִמּוֹ וְנִמְכַּר לְגֵר תּוֹשָׁב עִמָּךְ אוֹ לְעֵקֶר מִשְׁפַּחַת גֵּֽר:  מ  וְכִי תַשִּׂיג יַד גֵּר וְתוֹשָׁב עִמָּךְ וּמָךְ אָחִיךָ עִמּוֹ וְנִמְכַּר לְגֵר תּוֹשָׁב עִמָּךְ אוֹ לְעֵקֶר מִשְׁפַּחַת גֵּֽר:  ת  וַאֲרֵי תַדְבֵּק יַד עָרֵל וְתוֹתָב עִמָךְ וְיִתְמַסְכַּן אָחוּךְ עִמֵהּ וְיִזְדַבַּן לְעָרֵל תּוֹתָב עִמָךְ אוֹ לְאַרְמַי זַרְעִית גִיוֹרָא:

{מח} אַֽחֲרֵי נִמְכַּר גְּאֻלָּה תִּֽהְיֶה לּוֹ אֶחָד מֵֽאֶחָיו יִגְאָלֶֽנּוּ:  מ  אַֽחֲרֵי נִמְכַּר גְּאֻלָּה תִּֽהְיֶה לּוֹ אֶחָד מֵֽאֶחָיו יִגְאָלֶֽנּוּ:  ת  בָּתַר דְיִזְדַבַּן פֻּרְקָנָא תְּהֵי לֵהּ חַד מֵאֲחוֹהִי יִפְרְקִנֵהּ:

{מט} אֽוֹ דֹדוֹ אוֹ בֶן דֹּדוֹ יִגְאָלֶנּוּ אֽוֹ מִשְּׁאֵר בְּשָׂרוֹ מִמִּשְׁפַּחְתּוֹ יִגְאָלֶנּוּ אֽוֹ הִשִּׂיגָה יָדוֹ וְנִגְאָֽל:  מ  אֽוֹ דֹדוֹ אוֹ בֶן דֹּדוֹ יִגְאָלֶנּוּ אֽוֹ מִשְּׁאֵר בְּשָׂרוֹ מִמִּשְׁפַּחְתּוֹ יִגְאָלֶנּוּ אֽוֹ הִשִּׂיגָה יָדוֹ וְנִגְאָֽל:  ת  אוֹ אַח אֲבוּהִּי אוֹ בַר אַח אֲבוּהִי יִפְרְקִנֵהּ אוֹ מִקָרִיב בִּשְׂרֵהּ מִזַרְעִיתֵהּ יִפְרְקִנֵהּ אוֹ דְתַדְבֵּק יְדֵהּ וְיִתְפָּרָק:

{נ} וְחִשַּׁב עִם קֹנֵהוּ מִשְּׁנַת הִמָּכְרוֹ לוֹ עַד שְׁנַת הַיֹּבֵל וְהָיָה כֶּסֶף מִמְכָּרוֹ בְּמִסְפַּר שָׁנִים כִּימֵי שָׂכִיר יִהְיֶה עִמּֽוֹ:  מ  וְחִשַּׁב עִם קֹנֵהוּ מִשְּׁנַת הִמָּכְרוֹ לוֹ עַד שְׁנַת הַיֹּבֵל וְהָיָה כֶּסֶף מִמְכָּרוֹ בְּמִסְפַּר שָׁנִים כִּימֵי שָׂכִיר יִהְיֶה עִמּֽוֹ:  ת  וִיחַשֵׁב עִם זָבְנֵהּ מִשַׁתָּא דְאִזְדַבַּן לֵהּ עַד שַׁתָּא דְיוֹבֵלָא וִיהֵי כְּסַף זְבִינוֹהִי בְּמִנְיַן שְׁנַיָא כְּיוֹמֵי אֲגִירָא יְהֵי עִמֵהּ:

{נא} אִם עוֹד רַבּוֹת בַּשָּׁנִים לְפִיהֶן יָשִׁיב גְּאֻלָּתוֹ מִכֶּסֶף מִקְנָתֽוֹ:  מ  אִם עוֹד רַבּוֹת בַּשָּׁנִים לְפִיהֶן יָשִׁיב גְּאֻלָּתוֹ מִכֶּסֶף מִקְנָתֽוֹ:  ת  אִם עוֹד סְגִיוּת בִּשְׁנַיָא לְפוּמְהֵן יָתֵב פֻּרְקָנֵהּ מִכְּסַף זְבִינוֹהִי:

{נב} וְאִם מְעַט נִשְׁאַר בַּשָּׁנִים עַד שְׁנַת הַיֹּבֵל וְחִשַּׁב לוֹ כְּפִי שָׁנָיו יָשִׁיב אֶת גְּאֻלָּתֽוֹ:  מ  וְאִם מְעַט נִשְׁאַר בַּשָּׁנִים עַד שְׁנַת הַיֹּבֵל וְחִשַּׁב לוֹ כְּפִי שָׁנָיו יָשִׁיב אֶת גְּאֻלָּתֽוֹ:  ת  וְאִם זְעֵר יִשְׁתָּאַר בִּשְׁנַיָא עַד שַׁתָּא דְיוֹבֵלָא וִיחַשֶׁב לֵהּ כְּפוּם שְׁנוֹהִי יָתֵב יָת פּוּרְקָנֵהּ:

{נג} כִּשְׂכִיר שָׁנָה בְּשָׁנָה יִהְיֶה עִמּוֹ לֹֽא יִרְדֶּנּוּ בְּפֶרֶךְ לְעֵינֶֽיךָ:  מ  כִּשְׂכִיר שָׁנָה בְּשָׁנָה יִהְיֶה עִמּוֹ לֹֽא יִרְדֶּנּוּ בְּפֶרֶךְ לְעֵינֶֽיךָ:  ת  כַּאֲגִיר שְׁנָא בִּשְׁנָא יְהֵי עִמֵהּ לָא יִפְלַח בֵּהּ בְּקַשְׁיוּ לְעֵינָיךְ:

{נד} וְאִם לֹא יִגָּאֵל בְּאֵלֶּה וְיָצָא בִּשְׁנַת הַיֹּבֵל הוּא וּבָנָיו עִמּֽוֹ:  מ  וְאִם לֹא יִגָּאֵל בְּאֵלֶּה וְיָצָא בִּשְׁנַת הַיֹּבֵל הוּא וּבָנָיו עִמּֽוֹ:  ת  וְאִם לָא יִתְפָּרַק בְּאִלֵין וְיִפּוֹק בְּשַׁתָּא דְיוֹבֵלָא הוּא וּבְנוֹהִי עִמֵהּ:

{נה}  מפטיר  כִּי לִי בְנֵֽי יִשְׂרָאֵל עֲבָדִים עֲבָדַי הֵם אֲשֶׁר הוֹצֵאתִי אוֹתָם מֵאֶרֶץ מִצְרָיִם אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  מ  כִּי לִי בְנֵֽי יִשְׂרָאֵל עֲבָדִים עֲבָדַי הֵם אֲשֶׁר הוֹצֵאתִי אוֹתָם מֵאֶרֶץ מִצְרָיִם אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  ת  אֲרֵי דִילִי בְנֵי יִשְׂרָאֵל עַבְדִין עַבְדַי אִנוּן דִי אַפֵּקִית יָתְהוֹן מֵאַרְעָא דְמִצְרָיִם אֲנָא יְיָ אֱלָהֲכוֹן:




ויקרא פרק-כו

{א} לֹא תַֽעֲשׂוּ לָכֶם אֱלִילִם וּפֶסֶל וּמַצֵּבָה לֹֽא תָקִימוּ לָכֶם וְאֶבֶן מַשְׂכִּית לֹא תִתְּנוּ בְּאַרְצְכֶם לְהִֽשְׁתַּֽחֲוֹת עָלֶיהָ כִּי אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  מ  לֹא תַֽעֲשׂוּ לָכֶם אֱלִילִם וּפֶסֶל וּמַצֵּבָה לֹֽא תָקִימוּ לָכֶם וְאֶבֶן מַשְׂכִּית לֹא תִתְּנוּ בְּאַרְצְכֶם לְהִֽשְׁתַּֽחֲוֹת עָלֶיהָ כִּי אֲנִי יְהוָֹה אֱלֹֽהֵיכֶֽם:  ת  לָא תַעְבְּדוּן לְכוֹן טַעֲוָן וְצֶלֶם וְקָמָא לָא תְקִימוּן לְכוֹן וְאֶבֶן סָגְדָא לָא תִתְּנוּן בְּאַרְעֲכוֹן לְמִסְגֻד עֲלַהּ אֲרֵי אֲנָא יְיָ אֱלָהֲכוֹן:

{ב} אֶת שַׁבְּתֹתַי תִּשְׁמֹרוּ וּמִקְדָּשִׁי תִּירָאוּ אֲנִי יְהוָֹֽה:  מ  אֶת שַׁבְּתֹתַי תִּשְׁמֹרוּ וּמִקְדָּשִׁי תִּירָאוּ אֲנִי יְהוָֹֽה:  ת  יָת יוֹמֵי שַׁבַּיָא דִילִי תִטְרוּן וּלְבֵית מַקְדְשִׁי תְּהוֹן דָחֲלִין אֲנָא יְיָ: (פפפ)
