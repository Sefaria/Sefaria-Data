פרשת וארא


{ב} וַיְדַבֵּר אֱלֹהִים אֶל מֹשֶׁה וַיֹּאמֶר אֵלָיו אֲנִי יְהוָֹֽה:  מ  וַיְדַבֵּר אֱלֹהִים אֶל מֹשֶׁה וַיֹּאמֶר אֵלָיו אֲנִי יְהוָֹֽה:  ת  וּמַלִיל יְיָ עִם משֶׁה וַאֲמַר לֵהּ אֲנָא יְיָ:

{ג} וָֽאֵרָא אֶל אַבְרָהָם אֶל יִצְחָק וְאֶֽל יַֽעֲקֹב בְּאֵל שַׁדָּי וּשְׁמִי יְהֹוָה לֹא נוֹדַעְתִּי לָהֶֽם:  מ  וָֽאֵרָא אֶל אַבְרָהָם אֶל יִצְחָק וְאֶֽל יַֽעֲקֹב בְּאֵל שַׁדָּי וּשְׁמִי יְהֹוָה לֹא נוֹדַעְתִּי לָהֶֽם:  ת  וְאִתְגְלֵיתִי לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב בְּאֵל שַׁדָי וּשְׁמִי יְיָ לָא הוֹדָעִית לְהוֹן:

{ד} וְגַם הֲקִמֹתִי אֶת בְּרִיתִי אִתָּם לָתֵת לָהֶם אֶת אֶרֶץ כְּנָעַן אֵת אֶרֶץ מְגֻֽרֵיהֶם אֲשֶׁר גָּרוּ בָֽהּ:  מ  וְגַם הֲקִמֹתִי אֶת בְּרִיתִי אִתָּם לָתֵת לָהֶם אֶת אֶרֶץ כְּנָעַן אֵת אֶרֶץ מְגֻֽרֵיהֶם אֲשֶׁר גָּרוּ בָֽהּ:  ת  וְאַף אֲקֵימִית יָת קְיָמִי עִמְהוֹן לְמִתַּן לְהוֹן יָת אַרְעָא דִכְנָעַן יָת אַרְעָא תּוֹתָבוּתְהוֹן דְאִתּוֹתָבוּ בָהּ:

{ה} וְגַם אֲנִי שָׁמַעְתִּי אֶֽת נַֽאֲקַת בְּנֵי יִשְׂרָאֵל אֲשֶׁר מִצְרַיִם מַֽעֲבִדִים אֹתָם וָֽאֶזְכֹּר אֶת בְּרִיתִֽי:  מ  וְגַם אֲנִי שָׁמַעְתִּי אֶֽת נַֽאֲקַת בְּנֵי יִשְׂרָאֵל אֲשֶׁר מִצְרַיִם מַֽעֲבִדִים אֹתָם וָֽאֶזְכֹּר אֶת בְּרִיתִֽי:  ת  וְאַף קֳדָמַי שְׁמִיעַ יָת קְבֵילַת בְּנֵי יִשְׂרָאֵל דִי מִצְרָאֵי מַפְלְחִין בְּהוֹן וְדָכִירְנָא יָת קְיָמִי:

{ו} לָכֵן אֱמֹר לִבְנֵֽי יִשְׂרָאֵל אֲנִי יְהוָֹה וְהֽוֹצֵאתִי אֶתְכֶם מִתַּחַת סִבְלֹת מִצְרַיִם וְהִצַּלְתִּי אֶתְכֶם מֵֽעֲבֹֽדָתָם וְגָֽאַלְתִּי אֶתְכֶם בִּזְרוֹעַ נְטוּיָה וּבִשְׁפָטִים גְּדֹלִֽים:  מ  לָכֵן אֱמֹר לִבְנֵֽי יִשְׂרָאֵל אֲנִי יְהוָֹה וְהֽוֹצֵאתִי אֶתְכֶם מִתַּחַת סִבְלֹת מִצְרַיִם וְהִצַּלְתִּי אֶתְכֶם מֵֽעֲבֹֽדָתָם וְגָֽאַלְתִּי אֶתְכֶם בִּזְרוֹעַ נְטוּיָה וּבִשְׁפָטִים גְּדֹלִֽים:  ת  בְּכֵן אֱמַר לִבְנֵי יִשְׂרָאֵל אֲנָא יְיָ וְאַפֵּיק יָתְכוֹן מִגוֹ דְחוֹק פָּלְחַן מִצְרָאֵי וֶאֱשֵׁיזֵב יָתְכוֹן מִפָּלְחָנְהוֹן וְאֶפְרוֹק יָתְכוֹן בִּדְרָעָה מְרָמָא וּבְדִינִין רַבְרְבִין:

{ז} וְלָֽקַחְתִּי אֶתְכֶם לִי לְעָם וְהָיִיתִי לָכֶם לֵֽאלֹהִים וִֽידַעְתֶּם כִּי אֲנִי יְהוָֹה אֱלֹהֵיכֶם הַמּוֹצִיא אֶתְכֶם מִתַּחַת סִבְלוֹת מִצְרָֽיִם:  מ  וְלָֽקַחְתִּי אֶתְכֶם לִי לְעָם וְהָיִיתִי לָכֶם לֵֽאלֹהִים וִֽידַעְתֶּם כִּי אֲנִי יְהוָֹה אֱלֹהֵיכֶם הַמּוֹצִיא אֶתְכֶם מִתַּחַת סִבְלוֹת מִצְרָֽיִם:  ת  וֶאֱקָרֵיב יָתְכוֹן קדָמַי לְעַמָא וְאֶהֱוֵי לְכוֹן לֶאֱלָהּ וְתִדְעוּן אֲרֵי אֲנָא יְיָ אֱלָהָכוֹן דְאַפֵּיק יָתְכוֹן מִגוֹ דְחוֹק פָּלְחַן מִצְרָיִם:

{ח} וְהֵֽבֵאתִי אֶתְכֶם אֶל הָאָרֶץ אֲשֶׁר נָשָּׁאתִי אֶת יָדִי לָתֵת אֹתָהּ לְאַבְרָהָם לְיִצְחָק וּֽלְיַֽעֲקֹב וְנָֽתַתִּי אֹתָהּ לָכֶם מֽוֹרָשָׁה אֲנִי יְהוָֹֽה:  מ  וְהֵֽבֵאתִי אֶתְכֶם אֶל הָאָרֶץ אֲשֶׁר נָשָּׁאתִי אֶת יָדִי לָתֵת אֹתָהּ לְאַבְרָהָם לְיִצְחָק וּֽלְיַֽעֲקֹב וְנָֽתַתִּי אֹתָהּ לָכֶם מֽוֹרָשָׁה אֲנִי יְהוָֹֽה:  ת  וְאָעֵיל יָתְכוֹן לְאַרְעָא דִי קַיֵמִית בְּמֵימְרִי לְמִתַּן יָתַהּ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב וְאֶתֵּן יָתַהּ לְכוֹן יְרֻתָּא אֲנָא יְיָ:

{ט} וַיְדַבֵּר מֹשֶׁה כֵּן אֶל בְּנֵי יִשְׂרָאֵל וְלֹא שָֽׁמְעוּ אֶל מֹשֶׁה מִקֹּצֶר רוּחַ וּמֵֽעֲבֹדָה קָשָֽׁה:  מ  וַיְדַבֵּר מֹשֶׁה כֵּן אֶל בְּנֵי יִשְׂרָאֵל וְלֹא שָֽׁמְעוּ אֶל מֹשֶׁה מִקֹּצֶר רוּחַ וּמֵֽעֲבֹדָה קָשָֽׁה:  ת  וּמַלֵיל משֶׁה כֵּן לִבְנֵי יִשְׂרָאֵל וְלָא קַבִּילוּ מִן משֶׁה מִמַעְיַק רוּחָא וּמִפָּלְחָנָא דַהֲוָה קָשְׁיָא עֲלֵיהוֹן: (פ)

{י} וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{יא} בֹּא דַבֵּר אֶל פַּרְעֹה מֶלֶךְ מִצְרָיִם וִֽישַׁלַּח אֶת בְּנֵֽי יִשְׂרָאֵל מֵֽאַרְצֽוֹ:  מ  בֹּא דַבֵּר אֶל פַּרְעֹה מֶלֶךְ מִצְרָיִם וִֽישַׁלַּח אֶת בְּנֵֽי יִשְׂרָאֵל מֵֽאַרְצֽוֹ:  ת  עוּל מַלֵיל לְוָת פַּרְעֹה מַלְכָּא דְמִצְרָיִם וִישַׁלַח יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעֵהּ:

{יב} וַיְדַבֵּר מֹשֶׁה לִפְנֵי יְהוָֹה לֵאמֹר הֵן בְּנֵֽי יִשְׂרָאֵל לֹא שָֽׁמְעוּ אֵלַי וְאֵיךְ יִשְׁמָעֵנִי פַרְעֹה וַֽאֲנִי עֲרַל שְׂפָתָֽיִם:  מ  וַיְדַבֵּר מֹשֶׁה לִפְנֵי יְהוָֹה לֵאמֹר הֵן בְּנֵֽי יִשְׂרָאֵל לֹא שָֽׁמְעוּ אֵלַי וְאֵיךְ יִשְׁמָעֵנִי פַרְעֹה וַֽאֲנִי עֲרַל שְׂפָתָֽיִם:  ת  וּמַלִיל משֶׁה קֳדָם יְיָ לְמֵימָר הָא בְנֵי יִשְׂרָאֵל לָא קַבִּילוּ מִנִי וְאֵכְדֵין יְקַבֵּל מִנִי פַרְעֹה וַאֲנָא יַקִיר מַמְלָל: (פ)

{יג} וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן וַיְצַוֵּם אֶל בְּנֵי יִשְׂרָאֵל וְאֶל פַּרְעֹה מֶלֶךְ מִצְרָיִם לְהוֹצִיא אֶת בְּנֵֽי יִשְׂרָאֵל מֵאֶרֶץ מִצְרָֽיִם:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן וַיְצַוֵּם אֶל בְּנֵי יִשְׂרָאֵל וְאֶל פַּרְעֹה מֶלֶךְ מִצְרָיִם לְהוֹצִיא אֶת בְּנֵֽי יִשְׂרָאֵל מֵאֶרֶץ מִצְרָֽיִם:  ת  וּמַלִיל יְיָ עִם משֶׁה וְעִם אַהֲרֹן וּפַקֵדִנוּן לְוָת בְּנֵי יִשְׂרָאֵל וּלְוָת פַּרְעֹה מַלְכָּא דְמִצְרָיִם לְאַפָּקָא יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעָא דְמִצְרָיִם: (ס)

{יד}  שני  אֵלֶּה רָאשֵׁי בֵית אֲבֹתָם בְּנֵי רְאוּבֵן בְּכֹר יִשְׂרָאֵל חֲנוֹךְ וּפַלּוּא חֶצְרֹן וְכַרְמִי אֵלֶּה מִשְׁפְּחֹת רְאוּבֵֽן:  מ  אֵלֶּה רָאשֵׁי בֵית אֲבֹתָם בְּנֵי רְאוּבֵן בְּכֹר יִשְׂרָאֵל חֲנוֹךְ וּפַלּוּא חֶצְרֹן וְכַרְמִי אֵלֶּה מִשְׁפְּחֹת רְאוּבֵֽן:  ת  אִלֵין רֵישֵׁי בֵית אֲבָהָתְהוֹן בְּנֵי רְאוּבֵן בּוּכְרָא דְיִשְׂרָאֵל חֲנוֹךְ וּפַלוּא חֶצְרוֹן וְכַרְמִי אִלֵין זַרְעֲיַת רְאוּבֵן:

{טו} וּבְנֵי שִׁמְעוֹן יְמוּאֵל וְיָמִין וְאֹהַד וְיָכִין וְצֹחַר וְשָׁאוּל בֶּן הַֽכְּנַֽעֲנִית אֵלֶּה מִשְׁפְּחֹת שִׁמְעֽוֹן:  מ  וּבְנֵי שִׁמְעוֹן יְמוּאֵל וְיָמִין וְאֹהַד וְיָכִין וְצֹחַר וְשָׁאוּל בֶּן הַֽכְּנַֽעֲנִית אֵלֶּה מִשְׁפְּחֹת שִׁמְעֽוֹן:  ת  וּבְנֵי שִׁמְעוֹן יְמוּאֵל וְיָמִין וְאֹהַד וְיָכִין וְצֹחַר וְשָׁאוּל בַּר כְּנַעֲנֵיתָא אִלֵין זַרְעֲיַת שִׁמְעוֹן:

{טז} וְאֵלֶּה שְׁמוֹת בְּנֵֽי לֵוִי לְתֹלְדֹתָם גֵּֽרְשׁוֹן וּקְהָת וּמְרָרִי וּשְׁנֵי חַיֵּי לֵוִי שֶׁבַע וּשְׁלֹשִׁים וּמְאַת שָׁנָֽה:  מ  וְאֵלֶּה שְׁמוֹת בְּנֵֽי לֵוִי לְתֹלְדֹתָם גֵּֽרְשׁוֹן וּקְהָת וּמְרָרִי וּשְׁנֵי חַיֵּי לֵוִי שֶׁבַע וּשְׁלֹשִׁים וּמְאַת שָׁנָֽה:  ת  וְאִלֵין שְׁמָהַת בְּנֵי לֵוִי לְתוּלְדָתְהוֹן גֵרְשׁוֹן וּקְהָת וּמְרָרִי וּשְׁנֵי חַיֵי לֵוִי מְאָה וּתְלָתִין וּשְׁבַע שְׁנִין:

{יז} בְּנֵי גֵֽרְשׁוֹן לִבְנִי וְשִׁמְעִי לְמִשְׁפְּחֹתָֽם:  מ  בְּנֵי גֵֽרְשׁוֹן לִבְנִי וְשִׁמְעִי לְמִשְׁפְּחֹתָֽם:  ת  בְּנֵי גֵרְשׁוֹן לִבְנִי וְשִׁמְעִי לְזַרְעֲיָתְהוֹן:

{יח} וּבְנֵי קְהָת עַמְרָם וְיִצְהָר וְחֶבְרוֹן וְעֻזִּיאֵל וּשְׁנֵי חַיֵּי קְהָת שָׁלֹשׁ וּשְׁלֹשִׁים וּמְאַת שָׁנָֽה:  מ  וּבְנֵי קְהָת עַמְרָם וְיִצְהָר וְחֶבְרוֹן וְעֻזִּיאֵל וּשְׁנֵי חַיֵּי קְהָת שָׁלֹשׁ וּשְׁלֹשִׁים וּמְאַת שָׁנָֽה:  ת  וּבְנֵי קְהָת עַמְרָם וְיִצְהָר וְחֶבְרוֹן וְעֻזִיאֵל וּשְׁנֵי חַיֵי קְהָת מְאָה וּתְלָתִין וּתְלַת שְׁנִין:

{יט} וּבְנֵי מְרָרִי מַחְלִי וּמוּשִׁי אֵלֶּה מִשְׁפְּחֹת הַלֵּוִי לְתֹֽלְדֹתָֽם:  מ  וּבְנֵי מְרָרִי מַחְלִי וּמוּשִׁי אֵלֶּה מִשְׁפְּחֹת הַלֵּוִי לְתֹֽלְדֹתָֽם:  ת  וּבְנֵי מְרָרִי מַחְלִי וּמוּשִׁי אִלֵין זַרְעֲיַת לֵוִי לְתוּלְדָתְהוֹן:

{כ} וַיִּקַּח עַמְרָם אֶת יוֹכֶבֶד דֹּֽדָתוֹ לוֹ לְאִשָּׁה וַתֵּלֶד לוֹ אֶֽת אַֽהֲרֹן וְאֶת מֹשֶׁה וּשְׁנֵי חַיֵּי עַמְרָם שֶׁבַע וּשְׁלשִׁים וּמְאַת שָׁנָֽה:  מ  וַיִּקַּח עַמְרָם אֶת יוֹכֶבֶד דֹּֽדָתוֹ לוֹ לְאִשָּׁה וַתֵּלֶד לוֹ אֶֽת אַֽהֲרֹן וְאֶת מֹשֶׁה וּשְׁנֵי חַיֵּי עַמְרָם שֶׁבַע וּשְׁלשִׁים וּמְאַת שָׁנָֽה:  ת  וּנְסִיב עַמְרָם יָת יוֹכֶבֶד אֲחַת אֲבוּהִי לֵהּ לְאִנְתּוּ וִילִידַת לֵהּ יָת אַהֲרֹן וְיָת משֶׁה וּשְׁנֵי חַיֵי עַמְרָם מְאָה וּתְלָתִין וּשְׁבַע שְׁנִין:

{כא} וּבְנֵי יִצְהָר קֹרַח וָנֶפֶג וְזִכְרִֽי:  מ  וּבְנֵי יִצְהָר קֹרַח וָנֶפֶג וְזִכְרִֽי:  ת  וּבְנֵי יִצְהָר קֹרַח וָנֶפֶג וְזִכְרִי:

{כב} וּבְנֵי עֻזִּיאֵל מִֽישָׁאֵל וְאֶלְצָפָן וְסִתְרִֽי:  מ  וּבְנֵי עֻזִּיאֵל מִֽישָׁאֵל וְאֶלְצָפָן וְסִתְרִֽי:  ת  וּבְנֵי עֻזִיאֵל מִישָׁאֵל וְאֶלְצָפָן וְסִתְרִי:

{כג} וַיִּקַּח אַֽהֲרֹן אֶת אֱלִישֶׁבַע בַּת עַמִּֽינָדָב אֲחוֹת נַחְשׁוֹן לוֹ לְאִשָּׁה וַתֵּלֶד לוֹ אֶת נָדָב וְאֶת אֲבִיהוּא אֶת אֶלְעָזָר וְאֶת אִֽיתָמָֽר:  מ  וַיִּקַּח אַֽהֲרֹן אֶת אֱלִישֶׁבַע בַּת עַמִּֽינָדָב אֲחוֹת נַחְשׁוֹן לוֹ לְאִשָּׁה וַתֵּלֶד לוֹ אֶת נָדָב וְאֶת אֲבִיהוּא אֶת אֶלְעָזָר וְאֶת אִֽיתָמָֽר:  ת  וּנְסִיב אַהֲרֹן יָת אֱלִישֶׁבַע בַּת עַמִינָדָב אַחָתֵהּ דְנַחְשׁוֹן לֵהּ לְאִנְתּוּ וִילִידַת לֵהּ יָת נָדָב וְיָת אֲבִיהוּא יָת אֶלְעָזָר וְיָת אִיתָמָר:

{כד} וּבְנֵי קֹרַח אַסִּיר וְאֶלְקָנָה וַֽאֲבִֽיאָסָף אֵלֶּה מִשְׁפְּחֹת הַקָּרְחִֽי:  מ  וּבְנֵי קֹרַח אַסִּיר וְאֶלְקָנָה וַֽאֲבִֽיאָסָף אֵלֶּה מִשְׁפְּחֹת הַקָּרְחִֽי:  ת  וּבְנֵי קֹרַח אַסִיר וְאֶלְקָנָה וַאֲבִיאָסָף אִלֵין זַרְעֲיַת קֹרַח:

{כה} וְאֶלְעָזָר בֶּן אַֽהֲרֹן לָֽקַח לוֹ מִבְּנוֹת פּֽוּטִיאֵל לוֹ לְאִשָּׁה וַתֵּלֶד לוֹ אֶת פִּֽינְחָס אֵלֶּה רָאשֵׁי אֲבוֹת הַֽלְוִיִּם לְמִשְׁפְּחֹתָֽם:  מ  וְאֶלְעָזָר בֶּן אַֽהֲרֹן לָֽקַח לוֹ מִבְּנוֹת פּֽוּטִיאֵל לוֹ לְאִשָּׁה וַתֵּלֶד לוֹ אֶת פִּֽינְחָס אֵלֶּה רָאשֵׁי אֲבוֹת הַֽלְוִיִּם לְמִשְׁפְּחֹתָֽם:  ת  וְאֶלְעָזָר בַּר אַהֲרֹן נְסִיב לֵהּ מִבְּנַת פּוּטִיאֵל לֵהּ לְאִנְתּוּ וִילִידַת לֵהּ יָת פִּינְחָס אִלֵין רֵישֵׁי אֲבָהַת לֵיוָאֵי לְזַרְעֲיָתְהוֹן:

{כו} הוּא אַֽהֲרֹן וּמֹשֶׁה אֲשֶׁר אָמַר יְהוָֹה לָהֶם הוֹצִיאוּ אֶת בְּנֵי יִשְׂרָאֵל מֵאֶרֶץ מִצְרַיִם עַל צִבְאֹתָֽם:  מ  הוּא אַֽהֲרֹן וּמֹשֶׁה אֲשֶׁר אָמַר יְהוָֹה לָהֶם הוֹצִיאוּ אֶת בְּנֵי יִשְׂרָאֵל מֵאֶרֶץ מִצְרַיִם עַל צִבְאֹתָֽם:  ת  הוּא אַהֲרֹן וּמשֶׁה דִי אֲמַר יְיָ לְהוֹן אַפִּיקוּ יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעָא דְמִצְרַיִם עַל חֵילֵיהוֹן:

{כז} הֵם הַֽמְדַבְּרִים אֶל פַּרְעֹה מֶֽלֶךְ מִצְרַיִם לְהוֹצִיא אֶת בְּנֵֽי יִשְׂרָאֵל מִמִּצְרָיִם הוּא מֹשֶׁה וְאַֽהֲרֹֽן:  מ  הֵם הַֽמְדַבְּרִים אֶל פַּרְעֹה מֶֽלֶךְ מִצְרַיִם לְהוֹצִיא אֶת בְּנֵֽי יִשְׂרָאֵל מִמִּצְרָיִם הוּא מֹשֶׁה וְאַֽהֲרֹֽן:  ת  אִנוּן דְמַלְלִין עִם פַּרְעֹה מַלְכָּא דְמִצְרַיִם לְאַפָּקָא יָת בְּנֵי יִשְׂרָאֵל מִמִצְרָיִם הוּא משֶׁה וְאַהֲרֹן:

{כח} וַיְהִי בְּיוֹם דִּבֶּר יְהוָֹה אֶל מֹשֶׁה בְּאֶרֶץ מִצְרָֽיִם:  מ  וַיְהִי בְּיוֹם דִּבֶּר יְהוָֹה אֶל מֹשֶׁה בְּאֶרֶץ מִצְרָֽיִם:  ת  וַהֲוָה בְּיוֹמָא דְמַלִיל יְיָ עִם משֶׁה בְּאַרְעָא דְמִצְרָיִם: (ס)

{כט}  שלישי  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹר אֲנִי יְהוָֹה דַּבֵּר אֶל פַּרְעֹה מֶלֶךְ מִצְרַיִם אֵת כָּל אֲשֶׁר אֲנִי דֹּבֵר אֵלֶֽיךָ:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה לֵּאמֹר אֲנִי יְהוָֹה דַּבֵּר אֶל פַּרְעֹה מֶלֶךְ מִצְרַיִם אֵת כָּל אֲשֶׁר אֲנִי דֹּבֵר אֵלֶֽיךָ:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימַר אֲנָא יְיָ מַלֵל עִם פַּרְעֹה מַלְכָּא דְמִצְרַיִם יָת כָּל דִי אֲנָא מְמַלֵל עִמָךְ:

{ל} וַיֹּאמֶר מֹשֶׁה לִפְנֵי יְהוָֹה הֵן אֲנִי עֲרַל שְׂפָתַיִם וְאֵיךְ יִשְׁמַע אֵלַי פַּרְעֹֽה:  מ  וַיֹּאמֶר מֹשֶׁה לִפְנֵי יְהוָֹה הֵן אֲנִי עֲרַל שְׂפָתַיִם וְאֵיךְ יִשְׁמַע אֵלַי פַּרְעֹֽה:  ת  וַאֲמַר משֶׁה קֳדָם יְיָ הָא אֲנָא יַקִיר מַמְלָל וְאֶכְדֵין יְקַבֵּל מִנִי פַּרְעֹה: (פ)




שמות פרק-ז

{א} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה רְאֵה נְתַתִּיךָ אֱלֹהִים לְפַרְעֹה וְאַֽהֲרֹן אָחִיךָ יִהְיֶה נְבִיאֶֽךָ:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה רְאֵה נְתַתִּיךָ אֱלֹהִים לְפַרְעֹה וְאַֽהֲרֹן אָחִיךָ יִהְיֶה נְבִיאֶֽךָ:  ת  וַאֲמַר יְיָ לְמשֶׁה חֲזֵי מַנֵיתָךְ רַב לְפַרְעֹה וְאַהֲרֹן אָחוּךְ יְהֵי מְתֻרְגְמָנָךְ:

{ב} אַתָּה תְדַבֵּר אֵת כָּל אֲשֶׁר אֲצַוֶּךָּ וְאַֽהֲרֹן אָחִיךָ יְדַבֵּר אֶל פַּרְעֹה וְשִׁלַּח אֶת בְּנֵֽי יִשְׂרָאֵל מֵֽאַרְצֽוֹ:  מ  אַתָּה תְדַבֵּר אֵת כָּל אֲשֶׁר אֲצַוֶּךָּ וְאַֽהֲרֹן אָחִיךָ יְדַבֵּר אֶל פַּרְעֹה וְשִׁלַּח אֶת בְּנֵֽי יִשְׂרָאֵל מֵֽאַרְצֽוֹ:  ת  אַתְּ תְּמַלֵל יָת כָּל דִי אֲפַקְדִנָךְ וְאַהֲרֹן אָחוּךְ יְמַלֵל עִם פַּרְעֹה וִישַׁלַח יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעֵהּ:

{ג} וַֽאֲנִי אַקְשֶׁה אֶת לֵב פַּרְעֹה וְהִרְבֵּיתִי אֶת אֹֽתֹתַי וְאֶת מֽוֹפְתַי בְּאֶרֶץ מִצְרָֽיִם:  מ  וַֽאֲנִי אַקְשֶׁה אֶת לֵב פַּרְעֹה וְהִרְבֵּיתִי אֶת אֹֽתֹתַי וְאֶת מֽוֹפְתַי בְּאֶרֶץ מִצְרָֽיִם:  ת  וַאֲנָא אַקְשֵׁי יָת לִבָּא דְפַרְעֹה וְאַסְגֵיתִי יָת אַתְוָתַי וְיָת מוֹפְתַי בְּאַרְעָא דְמִצְרָיִם:

{ד} וְלֹֽא יִשְׁמַע אֲלֵכֶם פַּרְעֹה וְנָֽתַתִּי אֶת יָדִי בְּמִצְרָיִם וְהֽוֹצֵאתִי אֶת צִבְאֹתַי אֶת עַמִּי בְנֵֽי יִשְׂרָאֵל מֵאֶרֶץ מִצְרַיִם בִּשְׁפָטִים גְּדֹלִֽים:  מ  וְלֹֽא יִשְׁמַע אֲלֵכֶם פַּרְעֹה וְנָֽתַתִּי אֶת יָדִי בְּמִצְרָיִם וְהֽוֹצֵאתִי אֶת צִבְאֹתַי אֶת עַמִּי בְנֵֽי יִשְׂרָאֵל מֵאֶרֶץ מִצְרַיִם בִּשְׁפָטִים גְּדֹלִֽים:  ת  וְלָא יְקַבֵּל מִנְכוֹן פַּרְעֹה וְאֶתֵּן יָת מָחַת גְבֻרְתִּי בְּמִצְרָיִם וְאַפֵּיק יָת חֵילִי יָת עַמִי בְנֵי יִשְׂרָאֵל מֵאַרְעָא דְמִצְרַיִם בְּדִינִין רַבְרְבִין:

{ה} וְיָֽדְעוּ מִצְרַיִם כִּֽי אֲנִי יְהֹוָה בִּנְטֹתִי אֶת יָדִי עַל מִצְרָיִם וְהֽוֹצֵאתִי אֶת בְּנֵֽי יִשְׂרָאֵל מִתּוֹכָֽם:  מ  וְיָֽדְעוּ מִצְרַיִם כִּֽי אֲנִי יְהֹוָה בִּנְטֹתִי אֶת יָדִי עַל מִצְרָיִם וְהֽוֹצֵאתִי אֶת בְּנֵֽי יִשְׂרָאֵל מִתּוֹכָֽם:  ת  וְיִדְעוּן מִצְרַיִם אֲרֵי אֲנָא יְיָ כַּד אָרֵים יָת מָחַת גְבֻרְתִּי עַל מִצְרָיִם וְאַפֵּיק יָת בְּנֵי יִשְׂרָאֵל מִבֵּינֵיהוֹן:

{ו} וַיַּעַשׂ מֹשֶׁה וְאַֽהֲרֹן כַּֽאֲשֶׁר צִוָּה יְהוָֹה אֹתָם כֵּן עָשֽׂוּ:  מ  וַיַּעַשׂ מֹשֶׁה וְאַֽהֲרֹן כַּֽאֲשֶׁר צִוָּה יְהוָֹה אֹתָם כֵּן עָשֽׂוּ:  ת  וַעֲבַד משֶׁה וְאַהֲרֹן כְּמָא דְפַקִיד יְיָ יָתְהוֹן כֵּן עֲבָדוּ:

{ז} וּמֹשֶׁה בֶּן שְׁמֹנִים שָׁנָה וְאַהֲרֹן בֶּן שָׁלשׁ וּשְׁמֹנִים שָׁנָה בְּדַבְּרָם אֶל פַּרְעֹֽה:  מ  וּמֹשֶׁה בֶּן שְׁמֹנִים שָׁנָה וְאַהֲרֹן בֶּן שָׁלשׁ וּשְׁמֹנִים שָׁנָה בְּדַבְּרָם אֶל פַּרְעֹֽה:  ת  וּמשֶׁה בַּר תְּמָנָן שְׁנִין וְאַהֲרֹן בַּר תְּמָנָן וּתְלָת שְׁנִין בְּמַלָלוּתְהוֹן עִם פַּרְעֹה: (פ)

{ח}  רביעי  וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן לֵאמֹֽר:  מ  וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן לֵאמֹֽר:  ת  וַאֲמַר יְיָ לְמשֶׁה וּלְאַהֲרֹן לְמֵימָר:

{ט} כִּי יְדַבֵּר אֲלֵכֶם פַּרְעֹה לֵאמֹר תְּנוּ לָכֶם מוֹפֵת וְאָֽמַרְתָּ אֶֽל אַֽהֲרֹן קַח אֶֽת מַטְּךָ וְהַשְׁלֵךְ לִפְנֵֽי פַרְעֹה יְהִי לְתַנִּֽין:  מ  כִּי יְדַבֵּר אֲלֵכֶם פַּרְעֹה לֵאמֹר תְּנוּ לָכֶם מוֹפֵת וְאָֽמַרְתָּ אֶֽל אַֽהֲרֹן קַח אֶֽת מַטְּךָ וְהַשְׁלֵךְ לִפְנֵֽי פַרְעֹה יְהִי לְתַנִּֽין:  ת  אֲרֵי יְמַלֵל לְוָתְכוֹן פַּרְעֹה לְמֵימַר הָבוּ לְכוֹן אָתָא וְתֵימַר לְאַהֲרֹן סַב יָת חוּטְרָךְ וּרְמֵי קֳדָם פַּרְעֹה יְהִי לְתַנִינָא:

{י} וַיָּבֹא מֹשֶׁה וְאַֽהֲרֹן אֶל פַּרְעֹה וַיַּֽעֲשׂוּ כֵן כַּֽאֲשֶׁר צִוָּה יְהוָֹה וַיַּשְׁלֵךְ אַֽהֲרֹן אֶת מַטֵּהוּ לִפְנֵי פַרְעֹה וְלִפְנֵי עֲבָדָיו וַיְהִי לְתַנִּֽין:  מ  וַיָּבֹא מֹשֶׁה וְאַֽהֲרֹן אֶל פַּרְעֹה וַיַּֽעֲשׂוּ כֵן כַּֽאֲשֶׁר צִוָּה יְהוָֹה וַיַּשְׁלֵךְ אַֽהֲרֹן אֶת מַטֵּהוּ לִפְנֵי פַרְעֹה וְלִפְנֵי עֲבָדָיו וַיְהִי לְתַנִּֽין:  ת  וְעַל משֶׁה וְאַהֲרֹן לְוָת פַּרְעֹה וַעֲבָדוּ כֵן כְּמָא דְפַקֵיד יְיָ וּרְמָא אַהֲרֹן יָת חוּטְרֵהּ קֳדָם פַּרְעֹה וְקָדָם עַבְדוֹהִי וַהֲוָה לְתַנִינָא:

{יא} וַיִּקְרָא גַּם פַּרְעֹה לַֽחֲכָמִים וְלַֽמְכַשְּׁפִים וַיַּֽעֲשׂוּ גַם הֵם חַרְטֻמֵּי מִצְרַיִם בְּלַֽהֲטֵיהֶם כֵּֽן:  מ  וַיִּקְרָא גַּם פַּרְעֹה לַֽחֲכָמִים וְלַֽמְכַשְּׁפִים וַיַּֽעֲשׂוּ גַם הֵם חַרְטֻמֵּי מִצְרַיִם בְּלַֽהֲטֵיהֶם כֵּֽן:  ת  וּקְרָא אַף פַּרְעֹה לְחַכִּימַיָא וּלְחָרָשַׁיָא וַעֲבָדוּ אַף אִנּוּן חָרָשֵׁי מִצְרַיִם בְּלַחֲשֵׁיהוֹן כֵּן:

{יב} וַיַּשְׁלִיכוּ אִישׁ מַטֵּהוּ וַיִּֽהְיוּ לְתַנִּינִם וַיִּבְלַע מַטֵּֽה אַֽהֲרֹן אֶת מַטֹּתָֽם:  מ  וַיַּשְׁלִיכוּ אִישׁ מַטֵּהוּ וַיִּֽהְיוּ לְתַנִּינִם וַיִּבְלַע מַטֵּֽה אַֽהֲרֹן אֶת מַטֹּתָֽם:  ת  וּרְמוֹ גְבַר חוּטְרֵהּ וַהֲווֹ לְתַנִינַיָא וּבְלַע חֻטְרָא דְאַהֲרֹן יָת חֻטְרֵיהוֹן:

{יג} וַיֶּֽחֱזַק לֵב פַּרְעֹה וְלֹא שָׁמַע אֲלֵהֶם כַּֽאֲשֶׁר דִּבֶּר יְהוָֹֽה:  מ  וַיֶּֽחֱזַק לֵב פַּרְעֹה וְלֹא שָׁמַע אֲלֵהֶם כַּֽאֲשֶׁר דִּבֶּר יְהוָֹֽה:  ת  וְאִתַּקַף לִבָּא דְפַרְעֹה וְלָא קַבִּיל מִנְהוֹן כְּמָא דְמַלִיל יְיָ: (ס)

{יד} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה כָּבֵד לֵב פַּרְעֹה מֵאֵן לְשַׁלַּח הָעָֽם:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה כָּבֵד לֵב פַּרְעֹה מֵאֵן לְשַׁלַּח הָעָֽם:  ת  וַאֲמַר יְיָ לְמשׁה אִתְיַקַר (לפרש''י יַקִיר) לִבָּא דְפַרְעֹה סָרִיב לְשַׁלָחָא עַמָא:

{טו} לֵךְ אֶל פַּרְעֹה בַּבֹּקֶר הִנֵּה יֹצֵא הַמַּיְמָה וְנִצַּבְתָּ לִקְרָאתוֹ עַל שְׂפַת הַיְאֹר וְהַמַּטֶּה אֲשֶׁר נֶהְפַּךְ לְנָחָשׁ תִּקַּח בְּיָדֶֽךָ:  מ  לֵךְ אֶל פַּרְעֹה בַּבֹּקֶר הִנֵּה יֹצֵא הַמַּיְמָה וְנִצַּבְתָּ לִקְרָאתוֹ עַל שְׂפַת הַיְאֹר וְהַמַּטֶּה אֲשֶׁר נֶהְפַּךְ לְנָחָשׁ תִּקַּח בְּיָדֶֽךָ:  ת  אִזֵיל לְוָת פַּרְעֹה בְּצַפְרָא הָא נָפֵיק לְמַיָא וְתִתְעַתַּד לְקַדָמוּתֵהּ עַל כֵּיף נַהֲרָא וְחֻטְרָא דְאִתְהֲפִיךְ לְחִוְיָא תִּסַב בִּידָךְ:

{טז} וְאָֽמַרְתָּ אֵלָיו יְהֹוָה אֱלֹהֵי הָֽעִבְרִים שְׁלָחַנִי אֵלֶיךָ לֵאמֹר שַׁלַּח אֶת עַמִּי וְיַֽעַבְדֻנִי בַּמִּדְבָּר וְהִנֵּה לֹֽא שָׁמַעְתָּ עַד כֹּֽה:  מ  וְאָֽמַרְתָּ אֵלָיו יְהֹוָה אֱלֹהֵי הָֽעִבְרִים שְׁלָחַנִי אֵלֶיךָ לֵאמֹר שַׁלַּח אֶת עַמִּי וְיַֽעַבְדֻנִי בַּמִּדְבָּר וְהִנֵּה לֹֽא שָׁמַעְתָּ עַד כֹּֽה:  ת  וְתֵימַר לֵהּ יְיָ אֱלָהָא דִיהוּדָאֵי שַׁלְחַנִי לְוָתָךְ לְמֵימַר שַׁלַח יָת עַמִי וְיִפְלְחוּן קֳדָמַי בְּמַדְבְּרָא וְהָא לָא קַבִּלְתָּא עַד כְּעָן:

{יז} כֹּה אָמַר יְהֹוָה בְּזֹאת תֵּדַע כִּי אֲנִי יְהוָֹה הִנֵּה אָֽנֹכִי מַכֶּה בַּמַּטֶּה אֲשֶׁר בְּיָדִי עַל הַמַּיִם אֲשֶׁר בַּיְאֹר וְנֶֽהֶפְכוּ לְדָֽם:  מ  כֹּה אָמַר יְהֹוָה בְּזֹאת תֵּדַע כִּי אֲנִי יְהוָֹה הִנֵּה אָֽנֹכִי מַכֶּה בַּמַּטֶּה אֲשֶׁר בְּיָדִי עַל הַמַּיִם אֲשֶׁר בַּיְאֹר וְנֶֽהֶפְכוּ לְדָֽם:  ת  כִּדְנַן אֲמַר יְיָ בְּדָא תִדַע אֲרֵי אֲנָא יְיָ הָא אֲנָא מָחֵי בְּחֻטְרָא דִי בִידִי עַל מַיָא דִי בְנַהֲרָא וְיִתְהַפְּכוּן לִדְמָא:

{יח} וְהַדָּגָה אֲשֶׁר בַּיְאֹר תָּמוּת וּבָאַשׁ הַיְאֹר וְנִלְאוּ מִצְרַיִם לִשְׁתּוֹת מַיִם מִן הַיְאֹֽר:  מ  וְהַדָּגָה אֲשֶׁר בַּיְאֹר תָּמוּת וּבָאַשׁ הַיְאֹר וְנִלְאוּ מִצְרַיִם לִשְׁתּוֹת מַיִם מִן הַיְאֹֽר:  ת  וְנוּנֵי דִי בְנַהֲרָא תְּמוּת וּסְרִי נַהֲרָא וְיִלְאוּן מִצְרָאֵי לְמִשְׁתֵּי מַיָא מִן נַהֲרָא: (ס)

{יט} וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה אֱמֹר אֶֽל אַֽהֲרֹן קַח מַטְּךָ וּנְטֵה יָֽדְךָ עַל מֵימֵי מִצְרַיִם עַל נַֽהֲרֹתָם עַל יְאֹֽרֵיהֶם וְעַל אַגְמֵיהֶם וְעַל כָּל מִקְוֵה מֵֽימֵיהֶם וְיִֽהְיוּ דָם וְהָיָה דָם בְּכָל אֶרֶץ מִצְרַיִם וּבָֽעֵצִים וּבָֽאֲבָנִֽים:  מ  וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה אֱמֹר אֶֽל אַֽהֲרֹן קַח מַטְּךָ וּנְטֵה יָֽדְךָ עַל מֵימֵי מִצְרַיִם עַל נַֽהֲרֹתָם עַל יְאֹֽרֵיהֶם וְעַל אַגְמֵיהֶם וְעַל כָּל מִקְוֵה מֵֽימֵיהֶם וְיִֽהְיוּ דָם וְהָיָה דָם בְּכָל אֶרֶץ מִצְרַיִם וּבָֽעֵצִים וּבָֽאֲבָנִֽים:  ת  וַאֲמַר יְיָ לְמשֶׁה אֱמַר לְאַהֲרֹן טּוֹל חֻטְרָךְ וַאֲרֵים יְדָךְ עַל מַיָא דְמִצְרַיִם עַל נַהֲרֵיהוֹן עַל אֲרִתֵּיהוֹן וְעַל אַגְמֵיהוֹן וְעַל כָּל בֵּית כְּנִישׁוּת מֵימֵיהוֹן וִיהוֹן דְמָא וִיהֵי דְמָא בְּכָל אַרְעָא דְמִצְרַיִם וּבְמָנֵי אָעָא וּבְמָנֵי אַבְנָא:

{כ} וַיַּֽעֲשׂוּ כֵן מֹשֶׁה וְאַֽהֲרֹן כַּֽאֲשֶׁר צִוָּה יְהֹוָה וַיָּרֶם בַּמַּטֶּה וַיַּךְ אֶת הַמַּיִם אֲשֶׁר בַּיְאֹר לְעֵינֵי פַרְעֹה וּלְעֵינֵי עֲבָדָיו וַיֵּהָֽפְכוּ כָּל הַמַּיִם אֲשֶׁר בַּיְאֹר לְדָֽם:  מ  וַיַּֽעֲשׂוּ כֵן מֹשֶׁה וְאַֽהֲרֹן כַּֽאֲשֶׁר צִוָּה יְהֹוָה וַיָּרֶם בַּמַּטֶּה וַיַּךְ אֶת הַמַּיִם אֲשֶׁר בַּיְאֹר לְעֵינֵי פַרְעֹה וּלְעֵינֵי עֲבָדָיו וַיֵּהָֽפְכוּ כָּל הַמַּיִם אֲשֶׁר בַּיְאֹר לְדָֽם:  ת  וַעֲבָדוּ כֵן משֶׁה וְאַהֲרֹן כְּמָא דְפַקֵיד יְיָ וַאֲרֵים בְּחֻטְרָא וּמְחָא יָת מַיָא דִי בְנַהֲרָא לְעֵינֵי פַּרְעֹה וּלְעֵינֵי עַבְדוֹהִי וְאִתְהַפִיכוּ כָּל מַיָא דִי בְנַהֲרָא לִדְמָא:

{כא} וְהַדָּגָה אֲשֶׁר בַּיְאֹר מֵתָה וַיִּבְאַשׁ הַיְאֹר וְלֹא יָכְלוּ מִצְרַיִם לִשְׁתּוֹת מַיִם מִן הַיְאֹר וַיְהִי הַדָּם בְּכָל אֶרֶץ מִצְרָֽיִם:  מ  וְהַדָּגָה אֲשֶׁר בַּיְאֹר מֵתָה וַיִּבְאַשׁ הַיְאֹר וְלֹא יָכְלוּ מִצְרַיִם לִשְׁתּוֹת מַיִם מִן הַיְאֹר וַיְהִי הַדָּם בְּכָל אֶרֶץ מִצְרָֽיִם:  ת  וְנוּנֵי דִי בְנַהֲרָא מִיתוּ וּסְרִי נַהֲרָא וְלָא יְכִילוּ מִצְרָאֵי לְמִשְׁתֵּי מַיָא מִן נַהֲרָא וַהֲוָה דְמָא בְּכָל אַרְעָא דְמִצְרָיִם:

{כב} וַיַּֽעֲשׂוּ כֵן חַרְטֻמֵּי מִצְרַיִם בְּלָֽטֵיהֶם וַיֶּֽחֱזַק לֵב פַּרְעֹה וְלֹֽא שָׁמַע אֲלֵהֶם כַּֽאֲשֶׁר דִּבֶּר יְהוָֹֽה:  מ  וַיַּֽעֲשׂוּ כֵן חַרְטֻמֵּי מִצְרַיִם בְּלָֽטֵיהֶם וַיֶּֽחֱזַק לֵב פַּרְעֹה וְלֹֽא שָׁמַע אֲלֵהֶם כַּֽאֲשֶׁר דִּבֶּר יְהוָֹֽה:  ת  וַעֲבָדוּ כֵן חָרָשֵׁי מִצְרַיִם בְּלַחֲשֵׁיהוֹן וְאִתַּקַף לִבָּא דְפַרְעֹה וְלָא קַבֵּיל מִנְהוֹן כְּמָא דְמַלִיל יְיָ:

{כג} וַיִּפֶן פַּרְעֹה וַיָּבֹא אֶל בֵּיתוֹ וְלֹא שָׁת לִבּוֹ גַּם לָזֹֽאת:  מ  וַיִּפֶן פַּרְעֹה וַיָּבֹא אֶל בֵּיתוֹ וְלֹא שָׁת לִבּוֹ גַּם לָזֹֽאת:  ת  וְאִתְפְּנִי פַרְעֹה וְעַל לְבֵיתֵהּ וְלָא שַׁוִי לִבֵּהּ אַף לְדָא:

{כד} וַיַּחְפְּרוּ כָל מִצְרַיִם סְבִיבֹת הַיְאֹר מַיִם לִשְׁתּוֹת כִּי לֹא יָֽכְלוּ לִשְׁתֹּת מִמֵּימֵי הַיְאֹֽר:  מ  וַיַּחְפְּרוּ כָל מִצְרַיִם סְבִיבֹת הַיְאֹר מַיִם לִשְׁתּוֹת כִּי לֹא יָֽכְלוּ לִשְׁתֹּת מִמֵּימֵי הַיְאֹֽר:  ת  וַחֲפָרוּ כָל מִצְרָאֵי סַחֲרָנוּת נַהֲרָא מַיָא לְמִשְׁתֵּי אֲרֵי לָא יְכִילוּ לְמִשְׁתֵּי מִמַיָא דְנַהֲרָא:

{כה} וַיִּמָּלֵא שִׁבְעַת יָמִים אַֽחֲרֵי הַכּוֹת יְהוָֹה אֶת הַיְאֹֽר:  מ  וַיִּמָּלֵא שִׁבְעַת יָמִים אַֽחֲרֵי הַכּוֹת יְהוָֹה אֶת הַיְאֹֽר:  ת  וְאַשְׁלֵם שִׁבְעַת יוֹמִין בָּתַר דִמְחָא יְיָ יָת נַהֲרָא: (פ)

{כו} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה בֹּא אֶל פַּרְעֹה וְאָֽמַרְתָּ אֵלָיו כֹּה אָמַר יְהֹוָה שַׁלַּח אֶת עַמִּי וְיַֽעַבְדֻֽנִי:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה בֹּא אֶל פַּרְעֹה וְאָֽמַרְתָּ אֵלָיו כֹּה אָמַר יְהֹוָה שַׁלַּח אֶת עַמִּי וְיַֽעַבְדֻֽנִי:  ת  וַאֲמַר יְיָ לְמשֶׁה עוּל לְוָת פַּרְעֹה וְתֵימַר לֵהּ כִּדְנַן אֲמַר יְיָ שַׁלַח יָת עַמִי וְיִפְלְחוּן קֳדָמָי:

{כז} וְאִם מָאֵן אַתָּה לְשַׁלֵּחַ הִנֵּה אָֽנֹכִי נֹגֵף אֶת כָּל גְּבֽוּלְךָ בַּֽצְפַרְדְּעִֽים:  מ  וְאִם מָאֵן אַתָּה לְשַׁלֵּחַ הִנֵּה אָֽנֹכִי נֹגֵף אֶת כָּל גְּבֽוּלְךָ בַּֽצְפַרְדְּעִֽים:  ת  וְאִם סָרִיב אַתְּ לְשַׁלָחָא הָא אֲנָא מָחֵי יָת כָּל תְּחוּמָךְ בְּעֻרְדְעָנַיָא:

{כח} וְשָׁרַץ הַיְאֹר צְפַרְדְּעִים וְעָלוּ וּבָאוּ בְּבֵיתֶךָ וּבַֽחֲדַר מִשְׁכָּֽבְךָ וְעַל מִטָּתֶךָ וּבְבֵית עֲבָדֶיךָ וּבְעַמֶּךָ וּבְתַנּוּרֶיךָ וּבְמִשְׁאֲרוֹתֶֽיךָ:  מ  וְשָׁרַץ הַיְאֹר צְפַרְדְּעִים וְעָלוּ וּבָאוּ בְּבֵיתֶךָ וּבַֽחֲדַר מִשְׁכָּֽבְךָ וְעַל מִטָּתֶךָ וּבְבֵית עֲבָדֶיךָ וּבְעַמֶּךָ וּבְתַנּוּרֶיךָ וּבְמִשְׁאֲרוֹתֶֽיךָ:  ת  וִירַבֵּי נַהֲרָא עֻרְדְעָנַיָא וְיִסְקוּן וְיֶעֱלוּן בְּבֵיתָךְ וּבְאִדְרוֹן בֵּית מִשְׁכְּבָךְ וְעַל עַרְסָךְ וּבְבֵית עַבְדָיךְ וּבְעַמָךְ וּבְתַנוּרָךְ וּבְאַצְוָתָךְ:

{כט} וּבְכָה וּֽבְעַמְּךָ וּבְכָל עֲבָדֶיךָ יַֽעֲלוּ הַֽצְפַרְדְּעִֽים:  מ  וּבְכָה וּֽבְעַמְּךָ וּבְכָל עֲבָדֶיךָ יַֽעֲלוּ הַֽצְפַרְדְּעִֽים:  ת  וּבָךְ וּבְעַמָךְ וּבְכָל עַבְדָיךְ יִסְקוּן עֻרְדְעָנַיָא:




שמות פרק-ח

{א} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה אֱמֹר אֶֽל אַֽהֲרֹן נְטֵה אֶת יָֽדְךָ בְּמַטֶּךָ עַל הַנְּהָרֹת עַל הַיְאֹרִים וְעַל הָֽאֲגַמִּים וְהַעַל אֶת הַֽצְפַרְדְּעִים עַל אֶרֶץ מִצְרָֽיִם:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה אֱמֹר אֶֽל אַֽהֲרֹן נְטֵה אֶת יָֽדְךָ בְּמַטֶּךָ עַל הַנְּהָרֹת עַל הַיְאֹרִים וְעַל הָֽאֲגַמִּים וְהַעַל אֶת הַֽצְפַרְדְּעִים עַל אֶרֶץ מִצְרָֽיִם:  ת  וַאֲמַר יְיָ לְמשֶׁה אֱמַר לְאַהֲרֹן אֲרֵים יָת יְדָךְ בְּחֻטְּרָךְ עַל נַהֲרַיָא עַל אֲרִיתַּיָא וְעַל אַגְמַיָא וְאַסֵיק יָת עֻרְדְעָנַיָא עַל אַרְעָא דְמִצְרָיִם:

{ב} וַיֵּט אַֽהֲרֹן אֶת יָדוֹ עַל מֵימֵי מִצְרָיִם וַתַּעַל הַצְּפַרְדֵּעַ וַתְּכַס אֶת אֶרֶץ מִצְרָֽיִם:  מ  וַיֵּט אַֽהֲרֹן אֶת יָדוֹ עַל מֵימֵי מִצְרָיִם וַתַּעַל הַצְּפַרְדֵּעַ וַתְּכַס אֶת אֶרֶץ מִצְרָֽיִם:  ת  וַאֲרֵים אַהֲרֹן יָת יְדֵהּ עַל מַיָא דְמִצְרָאֵי וּסְלֵיקַת עֻרְדְעָנַיָא וַחֲפַת יָת אַרְעָא דְמִצְרָיִם:

{ג} וַיַּֽעֲשׂוּ כֵן הַֽחַרְטֻמִּים בְּלָֽטֵיהֶם וַיַּֽעֲלוּ אֶת הַֽצְפַרְדְּעִים עַל אֶרֶץ מִצְרָֽיִם:  מ  וַיַּֽעֲשׂוּ כֵן הַֽחַרְטֻמִּים בְּלָֽטֵיהֶם וַיַּֽעֲלוּ אֶת הַֽצְפַרְדְּעִים עַל אֶרֶץ מִצְרָֽיִם:  ת  וַעֲבָדוּ כֵן חָרָשַׁיָא בְּלַחֲשֵׁיהוֹן וְאַסִ יקוּ יָת עֻרְדְעָנַיָא עַל אַרְעָא דְמִצְרָיִם:

{ד} וַיִּקְרָא פַרְעֹה לְמֹשֶׁה וּֽלְאַֽהֲרֹן וַיֹּאמֶר הַעְתִּירוּ אֶל יְהֹוָה וְיָסֵר הַֽצְפַרְדְּעִים מִמֶּנִּי וּמֵֽעַמִּי וַֽאֲשַׁלְּחָה אֶת הָעָם וְיִזְבְּחוּ לַֽיהוָֹֽה:  מ  וַיִּקְרָא פַרְעֹה לְמֹשֶׁה וּֽלְאַֽהֲרֹן וַיֹּאמֶר הַעְתִּירוּ אֶל יְהֹוָה וְיָסֵר הַֽצְפַרְדְּעִים מִמֶּנִּי וּמֵֽעַמִּי וַֽאֲשַׁלְּחָה אֶת הָעָם וְיִזְבְּחוּ לַֽיהוָֹֽה:  ת  וּקְרָא פַרְעֹה לְמשֶׁה וּלְאַהֲרֹן וַאֲמַר צַלוֹ קֳדָם יְיָ וְיַעְדֵי עֻרְדְעָנַיָא מִנִי וּמֵעַמִי וְאֶשְׁלַח יָת עַמָא וְיִדְבְּחוּן קֳדָם יְיָ:

{ה} וַיֹּאמֶר מֹשֶׁה לְפַרְעֹה הִתְפָּאֵר עָלַי לְמָתַי אַעְתִּיר לְךָ וְלַֽעֲבָדֶיךָ וּֽלְעַמְּךָ לְהַכְרִית הַֽצְפַרְדְּעִים מִמְּךָ וּמִבָּתֶּיךָ רַק בַּיְאֹר תִּשָּׁאַֽרְנָה:  מ  וַיֹּאמֶר מֹשֶׁה לְפַרְעֹה הִתְפָּאֵר עָלַי לְמָתַי אַעְתִּיר לְךָ וְלַֽעֲבָדֶיךָ וּֽלְעַמְּךָ לְהַכְרִית הַֽצְפַרְדְּעִים מִמְּךָ וּמִבָּתֶּיךָ רַק בַּיְאֹר תִּשָּׁאַֽרְנָה:  ת  וַאֲמַר משֶׁה לְפַרְעֹה שְׁאַל לָךְ גְבוּרָא הַב לִי זְמַן לְאִמָתַי אֱצַלֵי עֲלָךְ וְעַל עַבְדָיךְ וְעַל עַמָךְ לְשֵׁיצָאָה עֻרְדְעָנַיָא מִנָךְ וּמִבָּתָּיךְ לְחוֹד בְּנַהֲרָא תִּשְׁתָּאָרָן:

{ו} וַיֹּאמֶר לְמָחָר וַיֹּאמֶר כִּדְבָרְךָ לְמַעַן תֵּדַע כִּי אֵין כַּֽיהוָֹה אֱלֹהֵֽינוּ:  מ  וַיֹּאמֶר לְמָחָר וַיֹּאמֶר כִּדְבָרְךָ לְמַעַן תֵּדַע כִּי אֵין כַּֽיהוָֹה אֱלֹהֵֽינוּ:  ת  וַאֲמַר לִמְחָר וַאֲמַר כְּפִתְגָמָךְ בְּדִיל דְתִדַע אֲרֵי לֵית כַּיְיָ אֱלָהָנָא:

{ז}  חמישי  וְסָרוּ הַֽצְפַרְדְּעִים מִמְּךָ וּמִבָּתֶּיךָ וּמֵֽעֲבָדֶיךָ וּמֵֽעַמֶּךָ רַק בַּיְאֹר תִּשָּׁאַֽרְנָה:  מ  וְסָרוּ הַֽצְפַרְדְּעִים מִמְּךָ וּמִבָּתֶּיךָ וּמֵֽעֲבָדֶיךָ וּמֵֽעַמֶּךָ רַק בַּיְאֹר תִּשָּׁאַֽרְנָה:  ת  וְיֶעְדוּן עֻרְדְעָנַיָא מִנָךְ וּמִבָּתָּיךְ וּמֵעַבְדָיךְ וּמֵעַמָךְ לְחוֹד בְּנַהֲרָא יִשְׁתָּאָרָן:

{ח} וַיֵּצֵא מֹשֶׁה וְאַֽהֲרֹן מֵעִם פַּרְעֹה וַיִּצְעַק מֹשֶׁה אֶל יְהֹוָה עַל דְּבַר הַֽצְפַרְדְּעִים אֲשֶׁר שָׂם לְפַרְעֹֽה:  מ  וַיֵּצֵא מֹשֶׁה וְאַֽהֲרֹן מֵעִם פַּרְעֹה וַיִּצְעַק מֹשֶׁה אֶל יְהֹוָה עַל דְּבַר הַֽצְפַרְדְּעִים אֲשֶׁר שָׂם לְפַרְעֹֽה:  ת  וּנְפַק משֶׁה וְאַהֲרֹן מִלְוָת פַרְעֹה וְצַלִי משֶׁה קֳדָם יְיָ עַל עֵסַק עֻרְדְעָנַיָא דִי שַׁוִי לְפַרְעֹה:

{ט} וַיַּעַשׂ יְהוָֹה כִּדְבַר מֹשֶׁה וַיָּמֻתוּ הַֽצְפַרְדְּעִים מִן הַבָּתִּים מִן הַֽחֲצֵרֹת וּמִן הַשָּׂדֹֽת:  מ  וַיַּעַשׂ יְהוָֹה כִּדְבַר מֹשֶׁה וַיָּמֻתוּ הַֽצְפַרְדְּעִים מִן הַבָּתִּים מִן הַֽחֲצֵרֹת וּמִן הַשָּׂדֹֽת:  ת  וַעֲבַד יְיָ כְּפִתְגָמָא דְמשֶׁה וּמִיתוּ עוּרְדְעָנַיָא מִן בָּתַּיָא מִן דָרָתָא וּמִן חַקְלָתָא:

{י} וַיִּצְבְּרוּ אֹתָם חֳמָרִם חֳמָרִם וַתִּבְאַשׁ הָאָֽרֶץ:  מ  וַיִּצְבְּרוּ אֹתָם חֳמָרִם חֳמָרִם וַתִּבְאַשׁ הָאָֽרֶץ:  ת  וּכְנָשׁוּ יָתְהוֹן דְגוֹרִין דְגוֹרִין וּסְרִיאוּ עַל אַרְעָא:

{יא} וַיַּרְא פַּרְעֹה כִּי הָֽיְתָה הָֽרְוָחָה וְהַכְבֵּד אֶת לִבּוֹ וְלֹא שָׁמַע אֲלֵהֶם כַּֽאֲשֶׁר דִּבֶּר יְהוָֹֽה:  מ  וַיַּרְא פַּרְעֹה כִּי הָֽיְתָה הָֽרְוָחָה וְהַכְבֵּד אֶת לִבּוֹ וְלֹא שָׁמַע אֲלֵהֶם כַּֽאֲשֶׁר דִּבֶּר יְהוָֹֽה:  ת  וַחֲזָא פַרְעֹה אֲרֵי הֲוַת רְוַחְתָּא וְיִתְיַקַר יָת לִבֵּהּ וְלָא קַבִּיל מִנְהוֹן כְּמָא דְמַלִיל יְיָ: (ס)

{יב} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה אֱמֹר אֶֽל אַֽהֲרֹן נְטֵה אֶֽת מַטְּךָ וְהַךְ אֶת עֲפַר הָאָרֶץ וְהָיָה לְכִנִּם בְּכָל אֶרֶץ מִצְרָֽיִם:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה אֱמֹר אֶֽל אַֽהֲרֹן נְטֵה אֶֽת מַטְּךָ וְהַךְ אֶת עֲפַר הָאָרֶץ וְהָיָה לְכִנִּם בְּכָל אֶרֶץ מִצְרָֽיִם:  ת  וַאֲמַר יְיָ לְמשֶׁה אֱמַר לְאַהֲרֹן אֲרֵים יָת חֻטְרָךְ וּמְחָא יָת עַפְרָא דְאַרְעָא וִיהֵא לְקַלְמְתָא בְּכָל אַרְעָא דְמִצְרָיִם:

{יג} וַיַּֽעֲשׂוּ כֵן וַיֵּט אַֽהֲרֹן אֶת יָדוֹ בְמַטֵּהוּ וַיַּךְ אֶת עֲפַר הָאָרֶץ וַתְּהִי הַכִּנָּם בָּֽאָדָם וּבַבְּהֵמָה כָּל עֲפַר הָאָרֶץ הָיָה כִנִּים בְּכָל אֶרֶץ מִצְרָֽיִם:  מ  וַיַּֽעֲשׂוּ כֵן וַיֵּט אַֽהֲרֹן אֶת יָדוֹ בְמַטֵּהוּ וַיַּךְ אֶת עֲפַר הָאָרֶץ וַתְּהִי הַכִּנָּם בָּֽאָדָם וּבַבְּהֵמָה כָּל עֲפַר הָאָרֶץ הָיָה כִנִּים בְּכָל אֶרֶץ מִצְרָֽיִם:  ת  וַעֲבָדוּ כֵן וַאֲרֵים אַהֲרֹן יָת יְדֵהּ בְחֻטְרֵהּ וּמְחָא יָת עַפְרָא דְאַרְעָא וַהֲוַת קַלְמְתָא בַּאֲנָשָׁא וּבִבְעִירָא כָּל עַפְרָא דְאַרְעָא הֲוַת קַלְמְתָא בְּכָל אַרְעָא דְמִצְרָיִם:

{יד} וַיַּֽעֲשׂוּ כֵן הַֽחַרְטֻמִּים בְּלָֽטֵיהֶם לְהוֹצִיא אֶת הַכִּנִּים וְלֹא יָכֹלוּ וַתְּהִי הַכִּנָּם בָּֽאָדָם וּבַבְּהֵמָֽה:  מ  וַיַּֽעֲשׂוּ כֵן הַֽחַרְטֻמִּים בְּלָֽטֵיהֶם לְהוֹצִיא אֶת הַכִּנִּים וְלֹא יָכֹלוּ וַתְּהִי הַכִּנָּם בָּֽאָדָם וּבַבְּהֵמָֽה:  ת  וַעֲבָדוּ כֵן חָרָשַׁיָא בְּלַחֲשֵׁיהוֹן לְאַפָּקָא יָת קַלְמְתָא וְלָא יָכִילוּ וַהֲוַת קַלְמְתָא בַּאֲנָשָׁא וּבִבְעִירָא:

{טו} וַיֹּֽאמְרוּ הַֽחַרְטֻמִּם אֶל פַּרְעֹה אֶצְבַּע אֱלֹהִים הִוא וַיֶּֽחֱזַק לֵב פַּרְעֹה וְלֹֽא שָׁמַע אֲלֵהֶם כַּֽאֲשֶׁר דִּבֶּר יְהוָֹֽה:  מ  וַיֹּֽאמְרוּ הַֽחַרְטֻמִּם אֶל פַּרְעֹה אֶצְבַּע אֱלֹהִים הִוא וַיֶּֽחֱזַק לֵב פַּרְעֹה וְלֹֽא שָׁמַע אֲלֵהֶם כַּֽאֲשֶׁר דִּבֶּר יְהוָֹֽה:  ת  וַאֲמָרוּ חָרָשַּׁיָא לְפַרְעֹה הָא מָחָא מִן קֳדָם יְיָ הִיא וְאִתַּקַף לִבָּא דְפַרְעֹה וְלָא קַבִּיל מִנְהוֹן כְּמָא דִמַלִיל יְיָ: (ס)

{טז} וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה הַשְׁכֵּם בַּבֹּקֶר וְהִתְיַצֵּב לִפְנֵי פַרְעֹה הִנֵּה יוֹצֵא הַמָּיְמָה וְאָֽמַרְתָּ אֵלָיו כֹּה אָמַר יְהֹוָה שַׁלַּח עַמִּי וְיַֽעַבְדֻֽנִי:  מ  וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה הַשְׁכֵּם בַּבֹּקֶר וְהִתְיַצֵּב לִפְנֵי פַרְעֹה הִנֵּה יוֹצֵא הַמָּיְמָה וְאָֽמַרְתָּ אֵלָיו כֹּה אָמַר יְהֹוָה שַׁלַּח עַמִּי וְיַֽעַבְדֻֽנִי:  ת  וַאֲמַר יְיָ לְמשֶׁה אַקְדֵם בְּצַפְרָא וְאִתְעַתַּד קֳדָם פַּרְעֹה הָא נָפֵק לְמַיָא וְתֵימַר לֵהּ כִּדְנַן אֲמַר יְיָ שַׁלַח עַמִי וְיִפְלְחוּן קֳדָמָי:

{יז} כִּי אִם אֵֽינְךָ מְשַׁלֵּחַ אֶת עַמִּי הִנְנִי מַשְׁלִיחַ בְּךָ וּבַֽעֲבָדֶיךָ וּֽבְעַמְּךָ וּבְבָתֶּיךָ אֶת הֶֽעָרֹב וּמָלְאוּ בָּתֵּי מִצְרַיִם אֶת הֶעָרֹב וְגַם הָֽאֲדָמָה אֲשֶׁר הֵם עָלֶֽיהָ:  מ  כִּי אִם אֵֽינְךָ מְשַׁלֵּחַ אֶת עַמִּי הִנְנִי מַשְׁלִיחַ בְּךָ וּבַֽעֲבָדֶיךָ וּֽבְעַמְּךָ וּבְבָתֶּיךָ אֶת הֶֽעָרֹב וּמָלְאוּ בָּתֵּי מִצְרַיִם אֶת הֶעָרֹב וְגַם הָֽאֲדָמָה אֲשֶׁר הֵם עָלֶֽיהָ:  ת  אֲרֵי אִם לֵיתָךְ מְשַׁלַח יָת עַמִי הָא אֲנָא מַשְׁלַח בָּךְ וּבְעַבְדָיךְ וּבְעַמָךְ וּבְבָתָּיךְ יָת עָרוֹבָא וְיִמְלוּן בָּתֵּי מִצְרַיִם יָת עָרוֹבָא וְאַף אַרְעָא דְאִנוּן עֲלַהּ:

{יח} וְהִפְלֵיתִי בַיּוֹם הַהוּא אֶת אֶרֶץ גֹּשֶׁן אֲשֶׁר עַמִּי עֹמֵד עָלֶיהָ לְבִלְתִּי הֱיוֹת שָׁם עָרֹב לְמַעַן תֵּדַע כִּי אֲנִי יְהֹוָה בְּקֶרֶב הָאָֽרֶץ:  מ  וְהִפְלֵיתִי בַיּוֹם הַהוּא אֶת אֶרֶץ גֹּשֶׁן אֲשֶׁר עַמִּי עֹמֵד עָלֶיהָ לְבִלְתִּי הֱיוֹת שָׁם עָרֹב לְמַעַן תֵּדַע כִּי אֲנִי יְהֹוָה בְּקֶרֶב הָאָֽרֶץ:  ת  וְאַפְרֵישׁ בְּיוֹמָא הַהוּא יָת אַרְעָא דְגשֶׁן דְעַמִי קָאִים עֲלַהּ בְּדִיל דְלָא לְמֶהֱוֵי תַמָן עָרוֹבָא בְּדִיל דְתִדַע אֲרֵי אֲנָא יְיָ שַׁלִיטּ בְּגוֹ אַרְעָא:

{יט}  שישי  וְשַׂמְתִּי פְדֻת בֵּין עַמִּי וּבֵין עַמֶּךָ לְמָחָר יִֽהְיֶה הָאֹת הַזֶּֽה:  מ  וְשַׂמְתִּי פְדֻת בֵּין עַמִּי וּבֵין עַמֶּךָ לְמָחָר יִֽהְיֶה הָאֹת הַזֶּֽה:  ת  וֶאֱשַׁוֵי פוּרְקַן לְעַמִי וְעַל עַמָךְ אַיְתִי מָחָא לִמְחָר יְהֵא אָתָא הָדֵין:

{כ} וַיַּעַשׂ יְהוָֹה כֵּן וַיָּבֹא עָרֹב כָּבֵד בֵּיתָה פַרְעֹה וּבֵית עֲבָדָיו וּבְכָל אֶרֶץ מִצְרַיִם תִּשָּׁחֵת הָאָרֶץ מִפְּנֵי הֶעָרֹֽב:  מ  וַיַּעַשׂ יְהוָֹה כֵּן וַיָּבֹא עָרֹב כָּבֵד בֵּיתָה פַרְעֹה וּבֵית עֲבָדָיו וּבְכָל אֶרֶץ מִצְרַיִם תִּשָּׁחֵת הָאָרֶץ מִפְּנֵי הֶעָרֹֽב:  ת  וַעֲבַד יְיָ כֵּן וַאֲתָא עָרוֹבָא תַקִיף לְבֵית פַּרְעֹה וּלְבֵית עַבְּדוֹהִי וּבְכָל אַרְעָא דְמִצְרַיִם אִתְחַבָּלַת אַרְעָא מִן קֳדָם עָרוֹבָא:

{כא} וַיִּקְרָא פַרְעֹה אֶל מֹשֶׁה וּֽלְאַֽהֲרֹן וַיֹּאמֶר לְכוּ זִבְחוּ לֵאלֹֽהֵיכֶם בָּאָֽרֶץ:  מ  וַיִּקְרָא פַרְעֹה אֶל מֹשֶׁה וּֽלְאַֽהֲרֹן וַיֹּאמֶר לְכוּ זִבְחוּ לֵאלֹֽהֵיכֶם בָּאָֽרֶץ:  ת  וּקְרָא פַרְעֹה לְמשֶׁה וּלְאַהֲרֹן וַאֲמַר אֱזִילוּ דַבָּחוּ קֳדָם אֱלָהָכוֹן בְּאַרְעָא:

{כב} וַיֹּאמֶר מֹשֶׁה לֹא נָכוֹן לַֽעֲשׂוֹת כֵּן כִּי תּֽוֹעֲבַת מִצְרַיִם נִזְבַּח לַֽיהוָֹה אֱלֹהֵינוּ הֵן נִזְבַּח אֶת תּֽוֹעֲבַת מִצְרַיִם לְעֵֽינֵיהֶם וְלֹא יִסְקְלֻֽנוּ:  מ  וַיֹּאמֶר מֹשֶׁה לֹא נָכוֹן לַֽעֲשׂוֹת כֵּן כִּי תּֽוֹעֲבַת מִצְרַיִם נִזְבַּח לַֽיהוָֹה אֱלֹהֵינוּ הֵן נִזְבַּח אֶת תּֽוֹעֲבַת מִצְרַיִם לְעֵֽינֵיהֶם וְלֹא יִסְקְלֻֽנוּ:  ת  וַאֲמַר משֶׁה לָא תַקֵן לְמֶעְבַּד כֵּן אֲרֵי בְעִירָא דְמִצְרָאֵי דָחֲלִין לֵהּ מִנֵהּ אֲנַחְנָא נָסְבִין לְדַבָּחָא קֳדָם יְיָ אֱלָהָנָא הָא נְדַבַּח יָת בְּעִירָא דְמִצְרָאֵי דָחֲלִין לֵהּ וְאִנוּן יְהוֹן חָזָן הֲלָא יֵימְרוּן לְמִרְגְמָנָא:

{כג} דֶּרֶךְ שְׁלֹשֶׁת יָמִים נֵלֵךְ בַּמִּדְבָּר וְזָבַחְנוּ לַֽיהוָֹה אֱלֹהֵינוּ כַּֽאֲשֶׁר יֹאמַר אֵלֵֽינוּ:  מ  דֶּרֶךְ שְׁלֹשֶׁת יָמִים נֵלֵךְ בַּמִּדְבָּר וְזָבַחְנוּ לַֽיהוָֹה אֱלֹהֵינוּ כַּֽאֲשֶׁר יֹאמַר אֵלֵֽינוּ:  ת  מַהֲלַךְ תְּלָתָא יוֹמִין נֵיזִיל בְּמַדְבְּרָא וּנְדַבַּח קֳדָם יְיָ אֱלָהָנָא כְּמָא דְיֵימַר לָנָא:

{כד} וַיֹּאמֶר פַּרְעֹה אָֽנֹכִי אֲשַׁלַּח אֶתְכֶם וּזְבַחְתֶּם לַֽיהוָֹה אֱלֹֽהֵיכֶם בַּמִּדְבָּר רַק הַרְחֵק לֹֽא תַרְחִיקוּ לָלֶכֶת הַעְתִּירוּ בַּֽעֲדִֽי:  מ  וַיֹּאמֶר פַּרְעֹה אָֽנֹכִי אֲשַׁלַּח אֶתְכֶם וּזְבַחְתֶּם לַֽיהוָֹה אֱלֹֽהֵיכֶם בַּמִּדְבָּר רַק הַרְחֵק לֹֽא תַרְחִיקוּ לָלֶכֶת הַעְתִּירוּ בַּֽעֲדִֽי:  ת  וַאֲמַר פַּרְעֹה אֲנָא אֲשַׁלַח יָתְכוֹן וְתִדְבְּחוּן קֳדָם יְיָ אֱלָהָכוֹן בְּמַדְבְּרָא לְחוֹד אַרְחָקָא לָא תְרַחֲקוּן לְמֵיזַל צַלוֹ אַף עָלָי:

{כה} וַיֹּאמֶר מֹשֶׁה הִנֵּה אָֽנֹכִי יוֹצֵא מֵֽעִמָּךְ וְהַעְתַּרְתִּי אֶל יְהֹוָה וְסָר הֶֽעָרֹב מִפַּרְעֹה מֵֽעֲבָדָיו וּמֵֽעַמּוֹ מָחָר רַק אַל יֹסֵף פַּרְעֹה הָתֵל לְבִלְתִּי שַׁלַּח אֶת הָעָם לִזְבֹּחַ לַֽיהוָֹֽה:  מ  וַיֹּאמֶר מֹשֶׁה הִנֵּה אָֽנֹכִי יוֹצֵא מֵֽעִמָּךְ וְהַעְתַּרְתִּי אֶל יְהֹוָה וְסָר הֶֽעָרֹב מִפַּרְעֹה מֵֽעֲבָדָיו וּמֵֽעַמּוֹ מָחָר רַק אַל יֹסֵף פַּרְעֹה הָתֵל לְבִלְתִּי שַׁלַּח אֶת הָעָם לִזְבֹּחַ לַֽיהוָֹֽה:  ת  וַאֲמַר משֶׁה הָא אֲנָא נָפֵק מֵעִמָךְ וֶאֱצַלֵי קֳדָם יְיָ וְיַעְדֵי עָרוֹבָא מִפַּרְעֹה מֵעַבְדוֹהִי וּמֵעַמֵהּ מְחָר לְחוֹד לָא יוֹסֵף פַּרְעֹה לְשַׁקָרָא בְּדִיל דְלָא לְשַׁלָחָא יָת עַמָא לְדַבָּחָא קֳדָם יְיָ:

{כו} וַיֵּצֵא מֹשֶׁה מֵעִם פַּרְעֹה וַיֶּעְתַּר אֶל יְהוָֹֽה:  מ  וַיֵּצֵא מֹשֶׁה מֵעִם פַּרְעֹה וַיֶּעְתַּר אֶל יְהוָֹֽה:  ת  וּנְפַק משֶׁה מִן קֳדָם פַּרְעֹה וְצַלִי קֳדָם יְיָ:

{כז} וַיַּעַשׂ יְהוָֹה כִּדְבַר מֹשֶׁה וַיָּסַר הֶֽעָרֹב מִפַּרְעֹה מֵֽעֲבָדָיו וּמֵֽעַמּוֹ לֹא נִשְׁאַר אֶחָֽד:  מ  וַיַּעַשׂ יְהוָֹה כִּדְבַר מֹשֶׁה וַיָּסַר הֶֽעָרֹב מִפַּרְעֹה מֵֽעֲבָדָיו וּמֵֽעַמּוֹ לֹא נִשְׁאַר אֶחָֽד:  ת  וַעֲבַד יְיָ כְּפִתְגָמָא דְמשֶׁה וְאַעְדִי עָרוֹבָא מִפַּרְעֹה מֵעַבְדוֹהִי וּמֵעַמֵהּ לָא אִשְׁתָּאַר חָד:

{כח} וַיַּכְבֵּד פַּרְעֹה אֶת לִבּוֹ גַּם בַּפַּעַם הַזֹּאת וְלֹא שִׁלַּח אֶת הָעָֽם:  מ  וַיַּכְבֵּד פַּרְעֹה אֶת לִבּוֹ גַּם בַּפַּעַם הַזֹּאת וְלֹא שִׁלַּח אֶת הָעָֽם:  ת  וְיַקַר פַּרְעֹה יָת לִבֵּהּ אַף בְּזִמְנָא הָדָא וְלָא שַׁלַח יָת עַמָא: (פ)




שמות פרק-ט

{א} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה בֹּא אֶל פַּרְעֹה וְדִבַּרְתָּ אֵלָיו כֹּֽה אָמַר יְהוָֹה אֱלֹהֵי הָֽעִבְרִים שַׁלַּח אֶת עַמִּי וְיַֽעַבְדֻֽנִי:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה בֹּא אֶל פַּרְעֹה וְדִבַּרְתָּ אֵלָיו כֹּֽה אָמַר יְהוָֹה אֱלֹהֵי הָֽעִבְרִים שַׁלַּח אֶת עַמִּי וְיַֽעַבְדֻֽנִי:  ת  וַאֲמַר יְיָ לְמשֶׁה עוּל לְוָת פַּרְעֹה וּתְמַלֵל עִמֵהּ כִּדְנַן אֲמַר יְיָ אֱלָהָא דִיהוּדָאֵי שַׁלַח יָת עַמִי וְיִפְלְחוּן קֳדָמָי:

{ב} כִּי אִם מָאֵן אַתָּה לְשַׁלֵּחַ וְעֽוֹדְךָ מַֽחֲזִיק בָּֽם:  מ  כִּי אִם מָאֵן אַתָּה לְשַׁלֵּחַ וְעֽוֹדְךָ מַֽחֲזִיק בָּֽם:  ת  אֲרֵי אִם סָרִיב אַתְּ לְשַׁלָחָא וְעַד כְּעַן אַתְּ מַתְקֵף בְּהוֹן:

{ג} הִנֵּה יַד יְהֹוָה הוֹיָה בְּמִקְנְךָ אֲשֶׁר בַּשָּׂדֶה בַּסּוּסִים בַּֽחֲמֹרִים בַּגְּמַלִּים בַּבָּקָר וּבַצֹּאן דֶּבֶר כָּבֵד מְאֹֽד:  מ  הִנֵּה יַד יְהֹוָה הוֹיָה בְּמִקְנְךָ אֲשֶׁר בַּשָּׂדֶה בַּסּוּסִים בַּֽחֲמֹרִים בַּגְּמַלִּים בַּבָּקָר וּבַצֹּאן דֶּבֶר כָּבֵד מְאֹֽד:  ת  הָא מָחָא מִן קֳדָם יְיָ הַוְיָא בִּבְעִירָךְ דִי בְחַקְלָא בְּסוּסְוָתָא בַּחֲמָרֵי בְּגַמְלֵי בְּתוֹרֵי וּבְעָנָא מוֹתָא תַּקִיף לַחֲדָא:

{ד} וְהִפְלָה יְהֹוָה בֵּין מִקְנֵה יִשְׂרָאֵל וּבֵין מִקְנֵה מִצְרָיִם וְלֹא יָמוּת מִכָּל לִבְנֵי יִשְׂרָאֵל דָּבָֽר:  מ  וְהִפְלָה יְהֹוָה בֵּין מִקְנֵה יִשְׂרָאֵל וּבֵין מִקְנֵה מִצְרָיִם וְלֹא יָמוּת מִכָּל לִבְנֵי יִשְׂרָאֵל דָּבָֽר:  ת  וְאַפְרֵישׁ יְיָ בֵּין בְּעִירָא דְיִשְׂרָאֵל וּבֵין בְּעִירָא דְמִצְרָיִם וְלָא יְמוּת מִכָּל לִבְנֵי יִשְׂרָאֵל מִדָעַם:

{ה} וַיָּשֶׂם יְהוָֹה מוֹעֵד לֵאמֹר מָחָר יַֽעֲשֶׂה יְהוָֹה הַדָּבָר הַזֶּה בָּאָֽרֶץ:  מ  וַיָּשֶׂם יְהוָֹה מוֹעֵד לֵאמֹר מָחָר יַֽעֲשֶׂה יְהוָֹה הַדָּבָר הַזֶּה בָּאָֽרֶץ:  ת  וְשַׁוִי יְיָ זִמְנָא לְמֵימָר מְחַר יַעְבֵּד יְיָ פִּתְגָמָא הָדֵין בְּאַרְעָא:

{ו} וַיַּעַשׂ יְהֹוָה אֶת הַדָּבָר הַזֶּה מִֽמָּחֳרָת וַיָּמָת כֹּל מִקְנֵה מִצְרָיִם וּמִמִּקְנֵה בְנֵֽי יִשְׂרָאֵל לֹא מֵת אֶחָֽד:  מ  וַיַּעַשׂ יְהֹוָה אֶת הַדָּבָר הַזֶּה מִֽמָּחֳרָת וַיָּמָת כֹּל מִקְנֵה מִצְרָיִם וּמִמִּקְנֵה בְנֵֽי יִשְׂרָאֵל לֹא מֵת אֶחָֽד:  ת  וַעֲבַד יְיָ יָת פִּתְגָמָא הָדֵין בְּיוֹמָא דְבַתְרוֹהִי וּמִית כֹּל בְּעִירָא דְמִצְרָיִם וּמִבְּעִירָא דִבְנֵי יִשְׂרָאֵל לָא מִית חָד:

{ז} וַיִּשְׁלַח פַּרְעֹה וְהִנֵּה לֹא מֵת מִמִּקְנֵה יִשְׂרָאֵל עַד אֶחָד וַיִּכְבַּד לֵב פַּרְעֹה וְלֹא שִׁלַּח אֶת הָעָֽם:  מ  וַיִּשְׁלַח פַּרְעֹה וְהִנֵּה לֹא מֵת מִמִּקְנֵה יִשְׂרָאֵל עַד אֶחָד וַיִּכְבַּד לֵב פַּרְעֹה וְלֹא שִׁלַּח אֶת הָעָֽם:  ת  וּשְׁלַח פַּרְעֹה וְהָא לָא מִית מִבְּעִירָא דְיִשְׂרָאֵל עַד חָד וְאִתְיַקַר לִבָּא דְפַרְעֹה וְלָא שַׁלַח יָת עַמָא: (פ)

{ח} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן קְחוּ לָכֶם מְלֹא חָפְנֵיכֶם פִּיחַ כִּבְשָׁן וּזְרָקוֹ מֹשֶׁה הַשָּׁמַיְמָה לְעֵינֵי פַרְעֹֽה:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן קְחוּ לָכֶם מְלֹא חָפְנֵיכֶם פִּיחַ כִּבְשָׁן וּזְרָקוֹ מֹשֶׁה הַשָּׁמַיְמָה לְעֵינֵי פַרְעֹֽה:  ת  וַאֲמַר יְיָ לְמשֶׁה וּלְאַהֲרֹן סִיבוּ לְכוֹן מְלֵי חָפְנֵיכוֹן פִּיחָא דְאַתּוּנָא וְיִזְרְקִנֵהּ משֶׁה לְצֵית שְׁמַיָא לְעֵינֵי פַרְעֹה:

{ט} וְהָיָה לְאָבָק עַל כָּל אֶרֶץ מִצְרָיִם וְהָיָה עַל הָֽאָדָם וְעַל הַבְּהֵמָה לִשְׁחִין פֹּרֵחַ אֲבַעְבֻּעֹת בְּכָל אֶרֶץ מִצְרָֽיִם:  מ  וְהָיָה לְאָבָק עַל כָּל אֶרֶץ מִצְרָיִם וְהָיָה עַל הָֽאָדָם וְעַל הַבְּהֵמָה לִשְׁחִין פֹּרֵחַ אֲבַעְבֻּעֹת בְּכָל אֶרֶץ מִצְרָֽיִם:  ת  וִיהֵי לְאַבְקָא עַל כָּל אַרְעָא דְמִצְרָיִם וִיהֵי עַל אֲנָשָׁא וְעַל בְּעִירָא לְשִׁיחֲנָא סַגִי אֲבַעְבֻּעִין בְּכָל אַרְעָא דְמִצְרָיִם:

{י} וַיִּקְחוּ אֶת פִּיחַ הַכִּבְשָׁן וַיַּֽעַמְדוּ לִפְנֵי פַרְעֹה וַיִּזְרֹק אֹתוֹ מֹשֶׁה הַשָּׁמָיְמָה וַיְהִי שְׁחִין אֲבַעְבֻּעֹת פֹּרֵחַ בָּֽאָדָם וּבַבְּהֵמָֽה:  מ  וַיִּקְחוּ אֶת פִּיחַ הַכִּבְשָׁן וַיַּֽעַמְדוּ לִפְנֵי פַרְעֹה וַיִּזְרֹק אֹתוֹ מֹשֶׁה הַשָּׁמָיְמָה וַיְהִי שְׁחִין אֲבַעְבֻּעֹת פֹּרֵחַ בָּֽאָדָם וּבַבְּהֵמָֽה:  ת  וּנְסִיבוּ יָת פִּיחָא דְאַתּוּנָא וְקָמוּ קֳדָם פַּרְעֹה וּזְרַק יָתֵהּ משֶׁה לְצֵית שְׁמַיָא וַהֲוָה שִׁיחֲנָא אֲבַעְבֻּעִין סָגֵי בַּאֲנָשָׁא וּבִבְעִירָא:

{יא} וְלֹא יָֽכְלוּ הַֽחַרְטֻמִּים לַֽעֲמֹד לִפְנֵי מֹשֶׁה מִפְּנֵי הַשְּׁחִין כִּֽי הָיָה הַשְּׁחִין בַּֽחַרְטֻמִּם וּבְכָל מִצְרָֽיִם:  מ  וְלֹא יָֽכְלוּ הַֽחַרְטֻמִּים לַֽעֲמֹד לִפְנֵי מֹשֶׁה מִפְּנֵי הַשְּׁחִין כִּֽי הָיָה הַשְּׁחִין בַּֽחַרְטֻמִּם וּבְכָל מִצְרָֽיִם:  ת  וְלָא יָכִילוּ חָרָשַׁיָא לְמֵיקַם קֳדָם משֶׁה מִן קֳדָם שִׁיחֲנָא אֲרֵי הֲוָה שִׁיחֲנָא בְּחָרָשַׁיָא וּבְכָל מִצְרָאֵי:

{יב} וַיְחַזֵּק יְהוָֹה אֶת לֵב פַּרְעֹה וְלֹא שָׁמַע אֲלֵהֶם כַּֽאֲשֶׁר דִּבֶּר יְהוָֹה אֶל מֹשֶֽׁה:  מ  וַיְחַזֵּק יְהוָֹה אֶת לֵב פַּרְעֹה וְלֹא שָׁמַע אֲלֵהֶם כַּֽאֲשֶׁר דִּבֶּר יְהוָֹה אֶל מֹשֶֽׁה:  ת  וְתַקִיף יְיָ יָת לִבָּא דְפַרְעֹה וְלָא קַבִּיל מִנְהוֹן כְּמָא דְמַלֵיל יְיָ עִם משֶׁה: (ס)

{יג} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה הַשְׁכֵּם בַּבֹּקֶר וְהִתְיַצֵּב לִפְנֵי פַרְעֹה וְאָֽמַרְתָּ אֵלָיו כֹּֽה אָמַר יְהוָֹה אֱלֹהֵי הָֽעִבְרִים שַׁלַּח אֶת עַמִּי וְיַֽעַבְדֻֽנִי:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה הַשְׁכֵּם בַּבֹּקֶר וְהִתְיַצֵּב לִפְנֵי פַרְעֹה וְאָֽמַרְתָּ אֵלָיו כֹּֽה אָמַר יְהוָֹה אֱלֹהֵי הָֽעִבְרִים שַׁלַּח אֶת עַמִּי וְיַֽעַבְדֻֽנִי:  ת  וַאֲמַר יְיָ לְמשֶׁה אַקְדֵם בְּצַפְרָא וְאִתְעַתַּד קֳדָם פַּרְעֹה וְתֵימַר לֵהּ כִּדְנַן אֲמַר יְיָ אֱלָהָא דִיהוּדָאֵי שַׁלַח יָת עַמִי וְיִפְלְחוּן קֳדָמָי:

{יד} כִּי בַּפַּעַם הַזֹּאת אֲנִי שֹׁלֵחַ אֶת כָּל מַגֵּֽפֹתַי אֶֽל לִבְּךָ וּבַֽעֲבָדֶיךָ וּבְעַמֶּךָ בַּֽעֲבוּר תֵּדַע כִּי אֵין כָּמֹנִי בְּכָל הָאָֽרֶץ:  מ  כִּי בַּפַּעַם הַזֹּאת אֲנִי שֹׁלֵחַ אֶת כָּל מַגֵּֽפֹתַי אֶֽל לִבְּךָ וּבַֽעֲבָדֶיךָ וּבְעַמֶּךָ בַּֽעֲבוּר תֵּדַע כִּי אֵין כָּמֹנִי בְּכָל הָאָֽרֶץ:  ת  אֲרֵי בְּזִמְנָא הָדָא אֲנָא שָׁלַח יָת כָּל מְחָתַי לְלִבָּךְ וּבְעַבְדָיךְ וּבְעַמָךְ בְּדִיל דְתִדַע אֲרֵי לֵית דִכְוָתִי שַׁלִיט בְּכָל אַרְעָא:

{טו} כִּי עַתָּה שָׁלַחְתִּי אֶת יָדִי וָאַךְ אֽוֹתְךָ וְאֶֽת עַמְּךָ בַּדָּבֶר וַתִּכָּחֵד מִן הָאָֽרֶץ:  מ  כִּי עַתָּה שָׁלַחְתִּי אֶת יָדִי וָאַךְ אֽוֹתְךָ וְאֶֽת עַמְּךָ בַּדָּבֶר וַתִּכָּחֵד מִן הָאָֽרֶץ:  ת  אֲרֵי כְעַן קְרֵיב קֳדָמַי דְאֶשְׁלַח פּוֹן יָת מַחַת גְבֻרְתִּי וְאֶמְחֵי יָתָךְ וְיָת עַמָךְ בְּמוֹתָא וְשֵׁיצִיאַת מִן אַרְעָא:

{טז} וְאוּלָם בַּֽעֲבוּר זֹאת הֶֽעֱמַדְתִּיךָ בַּֽעֲבוּר הַרְאֹֽתְךָ אֶת כֹּחִי וּלְמַעַן סַפֵּר שְׁמִי בְּכָל הָאָֽרֶץ:  מ  וְאוּלָם בַּֽעֲבוּר זֹאת הֶֽעֱמַדְתִּיךָ בַּֽעֲבוּר הַרְאֹֽתְךָ אֶת כֹּחִי וּלְמַעַן סַפֵּר שְׁמִי בְּכָל הָאָֽרֶץ:  ת  וּבְרַם בְּדִיל דָא קַיֵמְתָּךְ בְּדִיל לְאַחֲזָיוּתָךְ יָת חֵילִי וּבְדִיל דִיהוֹן מִשְׁתָּעַן גְבוּרַת שְׁמִי בְּכָל אַרְעָא:

{יז}  שביעי  עֽוֹדְךָ מִסְתּוֹלֵל בְּעַמִּי לְבִלְתִּי שַׁלְּחָֽם:  מ  עֽוֹדְךָ מִסְתּוֹלֵל בְּעַמִּי לְבִלְתִּי שַׁלְּחָֽם:  ת  עַד כְּעַן אַתְּ כְּבֵישַׁת בֵּהּ בְּעַמִי בְּדִיל דְלָא לְשַׁלָחוּתְהוֹן:

{יח} הִנְנִי מַמְטִיר כָּעֵת מָחָר בָּרָד כָּבֵד מְאֹד אֲשֶׁר לֹֽא הָיָה כָמֹהוּ בְּמִצְרַיִם לְמִן הַיּוֹם הִוָּֽסְדָה וְעַד עָֽתָּה:  מ  הִנְנִי מַמְטִיר כָּעֵת מָחָר בָּרָד כָּבֵד מְאֹד אֲשֶׁר לֹֽא הָיָה כָמֹהוּ בְּמִצְרַיִם לְמִן הַיּוֹם הִוָּֽסְדָה וְעַד עָֽתָּה:  ת  הָא אֲנָא מָחֵית מִטְרָא בְּעִדָנָא הָדֵין מְחַר בַּרְדָא תַּקִיף לַחֲדָא דִי לָא הֲוָה דִכְוָתֵהּ בְּמִצְרַיִם לְמִן יוֹמָא דְאִשְׁתַּכְלָלַת וְעַד כְּעָן:

{יט} וְעַתָּה שְׁלַח הָעֵז אֶֽת מִקְנְךָ וְאֵת כָּל אֲשֶׁר לְךָ בַּשָּׂדֶה כָּל הָֽאָדָם וְהַבְּהֵמָה אֲשֶׁר יִמָּצֵא בַשָּׂדֶה וְלֹא יֵֽאָסֵף הַבַּיְתָה וְיָרַד עֲלֵהֶם הַבָּרָד וָמֵֽתוּ:  מ  וְעַתָּה שְׁלַח הָעֵז אֶֽת מִקְנְךָ וְאֵת כָּל אֲשֶׁר לְךָ בַּשָּׂדֶה כָּל הָֽאָדָם וְהַבְּהֵמָה אֲשֶׁר יִמָּצֵא בַשָּׂדֶה וְלֹא יֵֽאָסֵף הַבַּיְתָה וְיָרַד עֲלֵהֶם הַבָּרָד וָמֵֽתוּ:  ת  וּכְעַן שְׁלַח כְּנוֹשׁ יָת בְּעִירָךְ וְיָת כָּל דִי לָךְ בְּחַקְלָא כָּל אֲנָשָׁא וּבְעִירָא דְאִשְׁתְּכַח בְּחַקְלָא וְלָא יִתְכְּנֵשׁ לְבֵיתָא וְיֵחוֹת עֲלֵיהוֹן בַּרְדָא וִימוּתוּן:

{כ} הַיָּרֵא אֶת דְּבַר יְהֹוָה מֵֽעַבְדֵי פַּרְעֹה הֵנִיס אֶת עֲבָדָיו וְאֶת מִקְנֵהוּ אֶל הַבָּתִּֽים:  מ  הַיָּרֵא אֶת דְּבַר יְהֹוָה מֵֽעַבְדֵי פַּרְעֹה הֵנִיס אֶת עֲבָדָיו וְאֶת מִקְנֵהוּ אֶל הַבָּתִּֽים:  ת  דְדָחִיל מִפִּתְגָמָא דַיְיָ מֵעַבְדֵי פַּרְעֹה כְּנַשׁ יָת עַבְדוֹהִי וְיָת גֵיתוֹהִי לְבָתַּיָא:

{כא} וַֽאֲשֶׁר לֹא שָׂם לִבּוֹ אֶל דְּבַר יְהוָֹה וַֽיַּֽעֲזֹב אֶת עֲבָדָיו וְאֶת מִקְנֵהוּ בַּשָּׂדֶֽה:  מ  וַֽאֲשֶׁר לֹא שָׂם לִבּוֹ אֶל דְּבַר יְהוָֹה וַֽיַּֽעֲזֹב אֶת עֲבָדָיו וְאֶת מִקְנֵהוּ בַּשָּׂדֶֽה:  ת  וְדִי לָא שַׁוִי לִבֵּהּ לְפִתְגָמָא דַיְיָ שְׁבַק יָת עַבְדוֹהִי וְיָת גֵיתוֹהִי בְּחַקְלָא: (פ)

{כב} וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה נְטֵה אֶת יָֽדְךָ עַל הַשָּׁמַיִם וִיהִי בָרָד בְּכָל אֶרֶץ מִצְרָיִם עַל הָֽאָדָם וְעַל הַבְּהֵמָה וְעַל כָּל עֵשֶׂב הַשָּׂדֶה בְּאֶרֶץ מִצְרָֽיִם:  מ  וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה נְטֵה אֶת יָֽדְךָ עַל הַשָּׁמַיִם וִיהִי בָרָד בְּכָל אֶרֶץ מִצְרָיִם עַל הָֽאָדָם וְעַל הַבְּהֵמָה וְעַל כָּל עֵשֶׂב הַשָּׂדֶה בְּאֶרֶץ מִצְרָֽיִם:  ת  וַאֲמַר יְיָ לְמשֶׁה אֲרֵים יָת יְדָךְ עַל צֵית שְׁמַיָא וִיהֵי בַרְדָא בְּכָל אַרְעָא דְמִצְרָאֵי עַל אֲנָשָׁא וְעַל בְּעִירָא וְעַל כָּל עֶסְבָּא דְחַקְלָא בְּאַרְעָא דְמִצְרָיִם:

{כג} וַיֵּט מֹשֶׁה אֶת מַטֵּהוּ עַל הַשָּׁמַיִם וַֽיהֹוָה נָתַן קֹלֹת וּבָרָד וַתִּֽהֲלַךְ אֵשׁ אָרְצָה וַיַּמְטֵר יְהוָֹה בָּרָד עַל אֶרֶץ מִצְרָֽיִם:  מ  וַיֵּט מֹשֶׁה אֶת מַטֵּהוּ עַל הַשָּׁמַיִם וַֽיהֹוָה נָתַן קֹלֹת וּבָרָד וַתִּֽהֲלַךְ אֵשׁ אָרְצָה וַיַּמְטֵר יְהוָֹה בָּרָד עַל אֶרֶץ מִצְרָֽיִם:  ת  וַאֲרֵים משֶׁה יָת חֻטְרֵהּ עַל צֵית שְׁמַיָא וַיְיָ יְהַב קָלִין וּבַרְדָא וּמְהַלְכָא אֶשָׁתָא עַל אַרְעָא וְאַמְטַר יְיָ בַּרְדָא עַל אַרְעָא דְמִצְרָיִם:

{כד} וַיְהִי בָרָד וְאֵשׁ מִתְלַקַּחַת בְּתוֹךְ הַבָּרָד כָּבֵד מְאֹד אֲשֶׁר לֹֽא הָיָה כָמֹהוּ בְּכָל אֶרֶץ מִצְרַיִם מֵאָז הָֽיְתָה לְגֽוֹי:  מ  וַיְהִי בָרָד וְאֵשׁ מִתְלַקַּחַת בְּתוֹךְ הַבָּרָד כָּבֵד מְאֹד אֲשֶׁר לֹֽא הָיָה כָמֹהוּ בְּכָל אֶרֶץ מִצְרַיִם מֵאָז הָֽיְתָה לְגֽוֹי:  ת  וַהֲוָה בַרְדָא וְאֶשָׁתָא מִשְׁתַּלְהָבַת בְּגוֹ בַרְדָא תַּקִיף לַחֲדָא דִי לָא הֲוָה דִכְוָתֵהּ בְּכָל אַרְעָא דְמִצְרַיִם מֵעִדַן דַהֲוַת לְעָם:

{כה} וַיַּךְ הַבָּרָד בְּכָל אֶרֶץ מִצְרַיִם אֵת כָּל אֲשֶׁר בַּשָּׂדֶה מֵֽאָדָם וְעַד בְּהֵמָה וְאֵת כָּל עֵשֶׂב הַשָּׂדֶה הִכָּה הַבָּרָד וְאֶת כָּל עֵץ הַשָּׂדֶה שִׁבֵּֽר:  מ  וַיַּךְ הַבָּרָד בְּכָל אֶרֶץ מִצְרַיִם אֵת כָּל אֲשֶׁר בַּשָּׂדֶה מֵֽאָדָם וְעַד בְּהֵמָה וְאֵת כָּל עֵשֶׂב הַשָּׂדֶה הִכָּה הַבָּרָד וְאֶת כָּל עֵץ הַשָּׂדֶה שִׁבֵּֽר:  ת  וּמְחָא בַרְדָא בְּכָל אַרְעָא דְמִצְרַיִם יָת כָּל דִי בְחַקְלָא מֵאֲנָשָׁא וְעַד בְּעִירָא וְיָת כָּל עֶסְבָּא דְחַקְלָא מְחָא בַרְדָא וְיָת כָּל אִילַן דִבְחַקְלָא תְּבַר:

{כו} רַק בְּאֶרֶץ גּשֶׁן אֲשֶׁר שָׁם בְּנֵי יִשְׂרָאֵל לֹא הָיָה בָּרָֽד:  מ  רַק בְּאֶרֶץ גּשֶׁן אֲשֶׁר שָׁם בְּנֵי יִשְׂרָאֵל לֹא הָיָה בָּרָֽד:  ת  לְחוֹד בְּאַרְעָא דְגשֶׁן דִי תַמָן בְּנֵי יִשְׂרָאֵל לָא הֲוָה בַּרְדָא:

{כז} וַיִּשְׁלַח פַּרְעֹה וַיִּקְרָא לְמֹשֶׁה וּֽלְאַֽהֲרֹן וַיֹּאמֶר אֲלֵהֶם חָטָאתִי הַפָּעַם יְהוָֹה הַצַּדִּיק וַֽאֲנִי וְעַמִּי הָֽרְשָׁעִֽים:  מ  וַיִּשְׁלַח פַּרְעֹה וַיִּקְרָא לְמֹשֶׁה וּֽלְאַֽהֲרֹן וַיֹּאמֶר אֲלֵהֶם חָטָאתִי הַפָּעַם יְהוָֹה הַצַּדִּיק וַֽאֲנִי וְעַמִּי הָֽרְשָׁעִֽים:  ת  וּשְׁלַח פַּרְעֹה וּקְרָא לְמשֶׁה וּלְאַהֲרֹן וַאֲמַר לְהוֹן חָבִית זִמְנָא הָדָא יְיָ זַכָּאָה וַאֲנָא וְעַמִי חַיָבִין:

{כח} הַעְתִּירוּ אֶל יְהֹוָה וְרַב מִֽהְיֹת קֹלֹת אֱלֹהִים וּבָרָד וַֽאֲשַׁלְּחָה אֶתְכֶם וְלֹא תֹֽסִפוּן לַֽעֲמֹֽד:  מ  הַעְתִּירוּ אֶל יְהֹוָה וְרַב מִֽהְיֹת קֹלֹת אֱלֹהִים וּבָרָד וַֽאֲשַׁלְּחָה אֶתְכֶם וְלֹא תֹֽסִפוּן לַֽעֲמֹֽד:  ת  צַלוֹ קֳדָם יְיָ וְסַגִי קֳדָמוֹהִי רְוַח דְלָא יְהוֹן עֲלָנָא קָלִין דִלְוַט כְּאִלֵין מִן קֳדָם יְיָ וּבַרְדָא וְאֶשְׁלַח יָתְכוֹן וְלָא תוֹסְפוּן לְאִתְעַכָּבָא:

{כט} וַיֹּאמֶר אֵלָיו מֹשֶׁה כְּצֵאתִי אֶת הָעִיר אֶפְרֹשׂ אֶת כַּפַּי אֶל יְהוָֹה הַקֹּלוֹת יֶחְדָּלוּן וְהַבָּרָד לֹא יִֽהְיֶה עוֹד לְמַעַן תֵּדַע כִּי לַֽיהוָֹה הָאָֽרֶץ:  מ  וַיֹּאמֶר אֵלָיו מֹשֶׁה כְּצֵאתִי אֶת הָעִיר אֶפְרֹשׂ אֶת כַּפַּי אֶל יְהוָֹה הַקֹּלוֹת יֶחְדָּלוּן וְהַבָּרָד לֹא יִֽהְיֶה עוֹד לְמַעַן תֵּדַע כִּי לַֽיהוָֹה הָאָֽרֶץ:  ת  וַאֲמַר לֵהּ משֶׁה כְּמִפְּקִי מִן קַרְתָּא אֶפְרוֹשׂ יְדַי בִּצְלוֹ קֳדָם יְיָ קָלַיָא יִתְמַנְעוּן וּבַרְדָא לָא יְהֵי עוֹד בְּדִיל דְתִדַע אֲרֵי דַיְיָ אַרְעָא:

{ל} וְאַתָּה וַֽעֲבָדֶיךָ יָדַעְתִּי כִּי טֶרֶם תִּֽירְאוּן מִפְּנֵי יְהוָֹה אֱלֹהִֽים:  מ  וְאַתָּה וַֽעֲבָדֶיךָ יָדַעְתִּי כִּי טֶרֶם תִּֽירְאוּן מִפְּנֵי יְהוָֹה אֱלֹהִֽים:  ת  וְאַתְּ וְעַבְדָיךְ יְדַעְנָא אֲרֵי עַד כְּעַן לָא אִתְכְּנַעְתּוּן מִן קֳדָם יְיָ אֱלֹהִים:

{לא} וְהַפִּשְׁתָּה וְהַשְּׂעֹרָה נֻכָּתָה כִּי הַשְּׂעֹרָה אָבִיב וְהַפִּשְׁתָּה גִּבְעֹֽל:  מ  וְהַפִּשְׁתָּה וְהַשְּׂעֹרָה נֻכָּתָה כִּי הַשְּׂעֹרָה אָבִיב וְהַפִּשְׁתָּה גִּבְעֹֽל:  ת  וְכִתָּנָא וְסַעֲרֵי לְקוֹ אֲרֵי סַעְרַנְיָא אַבִּיבִין וְכִתָּנָא גַבְעוֹלִין:

{לב} וְהַֽחִטָּה וְהַכֻּסֶּמֶת לֹא נֻכּוּ כִּי אֲפִילֹת הֵֽנָּה:  מ  וְהַֽחִטָּה וְהַכֻּסֶּמֶת לֹא נֻכּוּ כִּי אֲפִילֹת הֵֽנָּה:  ת  וְחִטָאָה וְכֻנְתַּיָא לָא לְקוֹ אֲרֵי אֲפֵילָתָא אִנוּן:

{לג}  מפטיר  וַיֵּצֵא מֹשֶׁה מֵעִם פַּרְעֹה אֶת הָעִיר וַיִּפְרֹשׂ כַּפָּיו אֶל יְהוָֹה וַיַּחְדְּלוּ הַקֹּלוֹת וְהַבָּרָד וּמָטָר לֹֽא נִתַּךְ אָֽרְצָה:  מ  וַיֵּצֵא מֹשֶׁה מֵעִם פַּרְעֹה אֶת הָעִיר וַיִּפְרֹשׂ כַּפָּיו אֶל יְהוָֹה וַיַּחְדְּלוּ הַקֹּלוֹת וְהַבָּרָד וּמָטָר לֹֽא נִתַּךְ אָֽרְצָה:  ת  וּנְפַק משֶׁה מִלְוַת פַּרְעֹה יָת קַרְתָּא וּפְרַשׂ יְדוֹהִי בִּצְלוֹ קֳדָם יְיָ וַיִתְמַנְעוּן קָלַיָא וּבַרְדָא וּמִטְרָא דַהֲוָה נָחִית לָא מְטָא עַל אַרְעָא:

{לד} וַיַּרְא פַּרְעֹה כִּֽי חָדַל הַמָּטָר וְהַבָּרָד וְהַקֹּלֹת וַיֹּסֶף לַֽחֲטֹא וַיַּכְבֵּד לִבּוֹ הוּא וַֽעֲבָדָֽיו:  מ  וַיַּרְא פַּרְעֹה כִּֽי חָדַל הַמָּטָר וְהַבָּרָד וְהַקֹּלֹת וַיֹּסֶף לַֽחֲטֹא וַיַּכְבֵּד לִבּוֹ הוּא וַֽעֲבָדָֽיו:  ת  וַחֲזָא פַרְעֹה אֲרֵי אִתְמְנַע מִטְרָא וּבַרְדָא וְקָלַיָא וְאוֹסֵף לְמֶחֱטֵי וְיַקְרֵהּ לְלִבֵּהּ הוּא וְעַבְדוֹהִי:

{לה} וַיֶּֽחֱזַק לֵב פַּרְעֹה וְלֹא שִׁלַּח אֶת בְּנֵי יִשְׂרָאֵל כַּֽאֲשֶׁר דִּבֶּר יְהוָֹה בְּיַד מֹשֶֽׁה:  מ  וַיֶּֽחֱזַק לֵב פַּרְעֹה וְלֹא שִׁלַּח אֶת בְּנֵי יִשְׂרָאֵל כַּֽאֲשֶׁר דִּבֶּר יְהוָֹה בְּיַד מֹשֶֽׁה:  ת  וְאִתַּקַף לִבָּא דְפַרְעֹה וְלָא שַׁלַח יָת בְּנֵי יִשְׂרָאֵל כְּמָא דִי מַלֵיל יְיָ בִּידָא דְמשֶׁה: (פפפ)
