פ-א


[ויקרא פרק-ב-ו]  פָּתוֹת אֹתָהּ פִּתִּים וְיָצַקְתָּ עָלֶיהָ שָׁמֶן מִנְחָה הִוא:

[דברים פרק-ז-כה]  פְּסִילֵי אֱלֹהֵיהֶם תִּשְׂרְפוּן בָּאֵשׁ לֹא תַחְמֹד כֶּסֶף וְזָהָב עֲלֵיהֶם וְלָקַחְתָּ לָךְ פֶּן תִּוָּקֵשׁ בּוֹ כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ הוּא:

[ירמיה פרק-נא-ח]  פִּתְאֹם נָפְלָה בָבֶל וַתִּשָּׁבֵר הֵילִילוּ עָלֶיהָ קְחוּ צֳרִי לְמַכְאוֹבָהּ אוּלַי תֵּרָפֵא:

[הושע פרק-ב-ה]  פֶּן אַפְשִׁיטֶנָּה עֲרֻמָּה וְהִצַּגְתִּיהָ כְּיוֹם הִוָּלְדָהּ וְשַׂמְתִּיהָ כַמִּדְבָּר וְשַׁתִּהָ כְּאֶרֶץ צִיָּה וַהֲמִתִּיהָ בַּצָּמָא:

[עזרא פרק-ה-ו]  פַּרְשֶׁגֶן אִגַּרְתָּא דִּי שְׁלַח תַּתְּנַי פַּחַת עֲבַר נַהֲרָה וּשְׁתַר בּוֹזְנַי וּכְנָוָתֵהּ אֲפַרְסְכָיֵא דִּי בַּעֲבַר נַהֲרָה עַל דָּרְיָוֶשׁ מַלְכָּא:

[עזרא פרק-ה-ז]  פִּתְגָמָא שְׁלַחוּ עֲלוֹהִי וְכִדְנָה כְּתִיב בְּגַוֵּהּ לְדָרְיָוֶשׁ מַלְכָּא שְׁלָמָא כֹלָּא:


פ-ב


[שמות פרק-כח-לד]  פַּעֲמֹן זָהָב וְרִמּוֹן פַּעֲמֹן זָהָב וְרִמּוֹן עַל שׁוּלֵי הַמְּעִיל סָבִיב:

[משלי פרק-ז-יב]  פַּעַם בַּחוּץ פַּעַם בָּרְחֹבוֹת וְאֵצֶל כָּל פִּנָּה תֶאֱרֹב:

[משלי פרק-טו-לב]  פּוֹרֵעַ מוּסָר מוֹאֵס נַפְשׁוֹ וְשׁוֹמֵעַ תּוֹכַחַת קוֹנֶה לֵּב:

[משלי פרק-כה-י]  פֶּן יְחַסֶּדְךָ שֹׁמֵעַ וְדִבָּתְךָ לֹא תָשׁוּב:

[משלי פרק-כו-כא]  פֶּחָם לְגֶחָלִים וְעֵצִים לְאֵשׁ וְאִישׁ (מדונים) מִדְיָנִים לְחַרְחַר רִיב:


פ-ג


[תהילים פרק-כב-יד]  פָּצוּ עָלַי פִּיהֶם אַרְיֵה טֹרֵף וְשֹׁאֵג:


פ-ד


[ישעיה פרק-מה-כב]  פְּנוּ אֵלַי וְהִוָּשְׁעוּ כָּל אַפְסֵי אָרֶץ כִּי אֲנִי אֵל וְאֵין עוֹד:

[יחזקאל פרק-מח-לד]  פְּאַת יָמָּה חֲמֵשׁ מֵאוֹת וְאַרְבַּעַת אֲלָפִים שַׁעֲרֵיהֶם שְׁלֹשָׁה שַׁעַר גָּד אֶחָד שַׁעַר אָשֵׁר אֶחָד שַׁעַר נַפְתָּלִי אֶחָד:

[תהילים פרק-קיב-ט]  פִּזַּר נָתַן לָאֶבְיוֹנִים צִדְקָתוֹ עֹמֶדֶת לָעַד קַרְנוֹ תָּרוּם בְּכָבוֹד:

[איוב פרק-ד-יד]  פַּחַד קְרָאַנִי וּרְעָדָה וְרֹב עַצְמוֹתַי הִפְחִיד:


פ-ה


[בראשית פרק-מט-ד]  פַּחַז כַּמַּיִם אַל תּוֹתַר כִּי עָלִיתָ מִשְׁכְּבֵי אָבִיךָ אָז חִלַּלְתָּ יְצוּעִי עָלָה:

[שמות פרק-לט-כו]  פַּעֲמֹן וְרִמֹּן פַּעֲמֹן וְרִמֹּן עַל שׁוּלֵי הַמְּעִיל סָבִיב לְשָׁרֵת כַּאֲשֶׁר צִוָּה יְהוָה אֶת מֹשֶׁה:

[במדבר פרק-ז-טו]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-כא]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-כז]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-לג]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-לט]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-מה]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-נא]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-נז]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-סג]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-סט]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-עה]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-ז-פא]  פַּר אֶחָד בֶּן בָּקָר אַיִל אֶחָד כֶּבֶשׂ אֶחָד בֶּן שְׁנָתוֹ לְעֹלָה:

[במדבר פרק-יב-ח]  פֶּה אֶל פֶּה אֲדַבֶּר בּוֹ וּמַרְאֶה וְלֹא בְחִידֹת וּתְמֻנַת יְהוָה יַבִּיט וּמַדּוּעַ לֹא יְרֵאתֶם לְדַבֵּר בְּעַבְדִּי בְמֹשֶׁה:

[דברים פרק-ד-טז]  פֶּן תַּשְׁחִתוּן וַעֲשִׂיתֶם לָכֶם פֶּסֶל תְּמוּנַת כָּל סָמֶל תַּבְנִית זָכָר אוֹ נְקֵבָה:

[דברים פרק-כט-יז]  פֶּן יֵשׁ בָּכֶם אִישׁ אוֹ אִשָּׁה אוֹ מִשְׁפָּחָה אוֹ שֵׁבֶט אֲשֶׁר לְבָבוֹ פֹנֶה הַיּוֹם מֵעִם יְהוָה אֱלֹהֵינוּ לָלֶכֶת לַעֲבֹד אֶת אֱלֹהֵי הַגּוֹיִם הָהֵם פֶּן יֵשׁ בָּכֶם שֹׁרֶשׁ פֹּרֶה רֹאשׁ וְלַעֲנָה:

[מלכים א פרק-ט-טז]  פַּרְעֹה מֶלֶךְ מִצְרַיִם עָלָה וַיִּלְכֹּד אֶת גֶּזֶר וַיִּשְׂרְפָהּ בָּאֵשׁ וְאֶת הַכְּנַעֲנִי הַיֹּשֵׁב בָּעִיר הָרָג וַיִּתְּנָהּ שִׁלֻּחִים לְבִתּוֹ אֵשֶׁת שְׁלֹמֹה:

[ירמיה פרק-ב-כד]  פֶּרֶה לִמֻּד מִדְבָּר בְּאַוַּת (נפשו) נַפְשָׁהּ שָׁאֲפָה רוּחַ תַּאֲנָתָהּ מִי יְשִׁיבֶנָּה כָּל מְבַקְשֶׁיהָ לֹא יִיעָפוּ בְּחָדְשָׁהּ יִמְצָאוּנְהָ:

[ירמיה פרק-מח-מג]  פַּחַד וָפַחַת וָפָח עָלֶיךָ יוֹשֵׁב מוֹאָב נְאֻם יְהוָה:

[יחזקאל פרק-כט-כ]  פְּעֻלָּתוֹ אֲשֶׁר עָבַד בָּהּ נָתַתִּי לוֹ אֶת אֶרֶץ מִצְרָיִם אֲשֶׁר עָשׂוּ לִי נְאֻם אֲדֹנָי יְהוִה:

[תהילים פרק-נ-יט]  פִּיךָ שָׁלַחְתָּ בְרָעָה וּלְשׁוֹנְךָ תַּצְמִיד מִרְמָה:

[תהילים פרק-סה-י]  פָּקַדְתָּ הָאָרֶץ וַתְּשֹׁקְקֶהָ רַבַּת תַּעְשְׁרֶנָּה פֶּלֶג אֱלֹהִים מָלֵא מָיִם תָּכִין דְּגָנָם כִּי כֵן תְּכִינֶהָ:

[תהילים פרק-פט-מא]  פָּרַצְתָּ כָל גְּדֵרֹתָיו שַׂמְתָּ מִבְצָרָיו מְחִתָּה:

[תהילים פרק-קה-לט]  פָּרַשׂ עָנָן לְמָסָךְ וְאֵשׁ לְהָאִיר לָיְלָה:

[תהילים פרק-קיח-יט]  פִּתְחוּ לִי שַׁעֲרֵי צֶדֶק אָבֹא בָם אוֹדֶה יָהּ:

[תהילים פרק-קלט-ו]  (פלאיה) פְּלִיאָה דַעַת מִמֶּנִּי נִשְׂגְּבָה לֹא אוּכַל לָהּ:

[תהילים פרק-קמג-ו]  פֵּרַשְׂתִּי יָדַי אֵלֶיךָ נַפְשִׁי כְּאֶרֶץ עֲיֵפָה לְךָ סֶלָה:

[משלי פרק-לא-כו]  פִּיהָ פָּתְחָה בְחָכְמָה וְתוֹרַת חֶסֶד עַל לְשׁוֹנָהּ:

[איוב פרק-כא-ה]  פְּנוּ אֵלַי וְהָשַׁמּוּ וְשִׂימוּ יָד עַל פֶּה:

[איוב פרק-לג-כח]  פָּדָה (נפשי) נַפְשׁוֹ מֵעֲבֹר בַּשָּׁחַת (וחיתי) וְחַיָּתוֹ בָּאוֹר תִּרְאֶה:

[אסתר פרק-ג-יד]  פַּתְשֶׁגֶן הַכְּתָב לְהִנָּתֵן דָּת בְּכָל מְדִינָה וּמְדִינָה גָּלוּי לְכָל הָעַמִּים לִהְיוֹת עֲתִדִים לַיּוֹם הַזֶּה:

[נחמיה פרק-י-ד]  פַּשְׁחוּר אֲמַרְיָה מַלְכִּיָּה:

[נחמיה פרק-י-כג]  פְּלַטְיָה חָנָן עֲנָיָה:

[דברי הימים א פרק-ב-מח]  פִּלֶגֶשׁ כָּלֵב מַעֲכָה יָלַד שֶׁבֶר וְאֶת תִּרְחֲנָה:


פ-ו


[שמות פרק-לד-טו]  פֶּן תִּכְרֹת בְּרִית לְיוֹשֵׁב הָאָרֶץ וְזָנוּ אַחֲרֵי אֱלֹהֵיהֶם וְזָבְחוּ לֵאלֹהֵיהֶם וְקָרָא לְךָ וְאָכַלְתָּ מִזִּבְחוֹ:

[ישעיה פרק-לה-ב]  פָּרֹחַ תִּפְרַח וְתָגֵל אַף גִּילַת וְרַנֵּן כְּבוֹד הַלְּבָנוֹן נִתַּן לָהּ הֲדַר הַכַּרְמֶל וְהַשָּׁרוֹן הֵמָּה יִרְאוּ כְבוֹד יְהוָה הֲדַר אֱלֹהֵינוּ:

[חגי פרק-א-ט]  פָּנֹה אֶל הַרְבֵּה וְהִנֵּה לִמְעָט וַהֲבֵאתֶם הַבַּיִת וְנָפַחְתִּי בוֹ יַעַן מֶה נְאֻם יְהוָה צְבָאוֹת יַעַן בֵּיתִי אֲשֶׁר הוּא חָרֵב וְאַתֶּם רָצִים אִישׁ לְבֵיתוֹ:

[תהילים פרק-כה-כב]  פְּדֵה אֱלֹהִים אֶת יִשְׂרָאֵל מִכֹּל צָרוֹתָיו:

[תהילים פרק-לד-כג]  פּוֹדֶה יְהוָה נֶפֶשׁ עֲבָדָיו וְלֹא יֶאְשְׁמוּ כָּל הַחֹסִים בּוֹ:

[תהילים פרק-פב-ד]  פַּלְּטוּ דַל וְאֶבְיוֹן מִיַּד רְשָׁעִים הַצִּילוּ:

[תהילים פרק-קיא-ט]  פְּדוּת שָׁלַח לְעַמּוֹ צִוָּה לְעוֹלָם בְּרִיתוֹ קָדוֹשׁ וְנוֹרָא שְׁמוֹ:

[תהילים פרק-קטו-ה]  פֶּה לָהֶם וְלֹא יְדַבֵּרוּ עֵינַיִם לָהֶם וְלֹא יִרְאוּ:

[תהילים פרק-קלה-טז]  פֶּה לָהֶם וְלֹא יְדַבֵּרוּ עֵינַיִם לָהֶם וְלֹא יִרְאוּ:

[משלי פרק-ד-כו]  פַּלֵּס מַעְגַּל רַגְלֶךָ וְכָל דְּרָכֶיךָ יִכֹּנוּ:

[משלי פרק-יד-טו]  פֶּתִי יַאֲמִין לְכָל דָּבָר וְעָרוּם יָבִין לַאֲשֻׁרוֹ:

[משלי פרק-יז-יב]  פָּגוֹשׁ דֹּב שַׁכּוּל בְּאִישׁ וְאַל כְּסִיל בְּאִוַּלְתּוֹ:

[משלי פרק-יח-ז]  פִּי כְסִיל מְחִתָּה לוֹ וּשְׂפָתָיו מוֹקֵשׁ נַפְשׁוֹ:

[משלי פרק-כא-א]  פַּלְגֵי מַיִם לֶב מֶלֶךְ בְּיַד יְהוָה עַל כָּל אֲשֶׁר יַחְפֹּץ יַטֶּנּוּ:

[משלי פרק-כד-יח]  פֶּן יִרְאֶה יְהוָה וְרַע בְּעֵינָיו וְהֵשִׁיב מֵעָלָיו אַפּוֹ:

[איוב פרק-כח-ד]  פָּרַץ נַחַל מֵעִם גָּר הַנִּשְׁכָּחִים מִנִּי רָגֶל דַּלּוּ מֵאֱנוֹשׁ נָעוּ:

[איכה פרק-ב-טז]  פָּצוּ עָלַיִךְ פִּיהֶם כָּל אוֹיְבַיִךְ שָׁרְקוּ וַיַּחַרְקוּ שֵׁן אָמְרוּ בִּלָּעְנוּ אַךְ זֶה הַיּוֹם שֶׁקִּוִּינֻהוּ מָצָאנוּ רָאִינוּ:

[איכה פרק-ג-מו]  פָּצוּ עָלֵינוּ פִּיהֶם כָּל אֹיְבֵינוּ:

[איכה פרק-ד-טז]  פְּנֵי יְהוָה חִלְּקָם לֹא יוֹסִיף לְהַבִּיטָם פְּנֵי כֹהֲנִים לֹא נָשָׂאוּ (זקנים) וּזְקֵנִים לֹא חָנָנוּ:


פ-ט


[תהילים פרק-יג-ה]  פֶּן יֹאמַר אֹיְבִי יְכָלְתִּיו צָרַי יָגִילוּ כִּי אֶמּוֹט:

[תהילים פרק-לז-ל]  פִּי צַדִּיק יֶהְגֶּה חָכְמָה וּלְשׁוֹנוֹ תְּדַבֵּר מִשְׁפָּט:


פ-י


[במדבר פרק-כה-יא]  פִּינְחָס בֶּן אֶלְעָזָר בֶּן אַהֲרֹן הַכֹּהֵן הֵשִׁיב אֶת חֲמָתִי מֵעַל בְּנֵי יִשְׂרָאֵל בְּקַנְאוֹ אֶת קִנְאָתִי בְּתוֹכָם וְלֹא כִלִּיתִי אֶת בְּנֵי יִשְׂרָאֵל בְּקִנְאָתִי:

[ישעיה פרק-סג-ג]  פּוּרָה דָּרַכְתִּי לְבַדִּי וּמֵעַמִּים אֵין אִישׁ אִתִּי וְאֶדְרְכֵם בְּאַפִּי וְאֶרְמְסֵם בַּחֲמָתִי וְיֵז נִצְחָם עַל בְּגָדַי וְכָל מַלְבּוּשַׁי אֶגְאָלְתִּי:

[ירמיה פרק-כ-ז]  פִּתִּיתַנִי יְהוָה וָאֶפָּת חֲזַקְתַּנִי וַתּוּכָל הָיִיתִי לִשְׂחוֹק כָּל הַיּוֹם כֻּלֹּה לֹעֵג לִי:

[תהילים פרק-כה-טז]  פְּנֵה אֵלַי וְחָנֵּנִי כִּי יָחִיד וְעָנִי אָנִי:

[תהילים פרק-נה-יט]  פָּדָה בְשָׁלוֹם נַפְשִׁי מִקֲּרָב לִי כִּי בְרַבִּים הָיוּ עִמָּדִי:

[תהילים פרק-קיט-קכט]  פְּלָאוֹת עֵדְוֹתֶיךָ עַל כֵּן נְצָרָתַם נַפְשִׁי:

[תהילים פרק-קיט-קלא]  פִּי פָעַרְתִּי וָאֶשְׁאָפָה כִּי לְמִצְוֹתֶיךָ יָאָבְתִּי:

[משלי פרק-ה-ט]  פֶּן תִּתֵּן לַאֲחֵרִים הוֹדֶךָ וּשְׁנֹתֶיךָ לְאַכְזָרִי:

[משלי פרק-ה-י]  פֶּן יִשְׂבְּעוּ זָרִים כֹּחֶךָ וַעֲצָבֶיךָ בְּבֵית נָכְרִי:

[משלי פרק-ל-ט]  פֶּן אֶשְׂבַּע וְכִחַשְׁתִּי וְאָמַרְתִּי מִי יְהוָה וּפֶן אִוָּרֵשׁ וְגָנַבְתִּי וְתָפַשְׂתִּי שֵׁם אֱלֹהָי:

[משלי פרק-לא-ה]  פֶּן יִשְׁתֶּה וְיִשְׁכַּח מְחֻקָּק וִישַׁנֶּה דִּין כָּל בְּנֵי עֹנִי:

[שיר השירים פרק-ה-ו]  פָּתַחְתִּי אֲנִי לְדוֹדִי וְדוֹדִי חָמַק עָבָר נַפְשִׁי יָצְאָה בְדַבְּרוֹ בִּקַּשְׁתִּיהוּ וְלֹא מְצָאתִיהוּ קְרָאתִיו וְלֹא עָנָנִי:

[איכה פרק-ג-מח]  פַּלְגֵי מַיִם תֵּרַד עֵינִי עַל שֶׁבֶר בַּת עַמִּי:


פ-כ


[יחזקאל פרק-כז-י]  פָּרַס וְלוּד וּפוּט הָיוּ בְחֵילֵךְ אַנְשֵׁי מִלְחַמְתֵּךְ מָגֵן וְכוֹבַע תִּלּוּ בָךְ הֵמָּה נָתְנוּ הֲדָרֵךְ:

[זכריה פרק-יא-א]  פְּתַח לְבָנוֹן דְּלָתֶיךָ וְתֹאכַל אֵשׁ בַּאֲרָזֶיךָ:

[תהילים פרק-פו-טז]  פְּנֵה אֵלַי וְחָנֵּנִי תְּנָה עֻזְּךָ לְעַבְדֶּךָ וְהוֹשִׁיעָה לְבֶן אֲמָתֶךָ:

[תהילים פרק-קיט-קלב]  פְּנֵה אֵלַי וְחָנֵּנִי כְּמִשְׁפָּט לְאֹהֲבֵי שְׁמֶךָ:

[תהילים פרק-קיט-קלד]  פְּדֵנִי מֵעֹשֶׁק אָדָם וְאֶשְׁמְרָה פִּקּוּדֶיךָ:

[תהילים פרק-קיט-קלה]  פָּנֶיךָ הָאֵר בְּעַבְדֶּךָ וְלַמְּדֵנִי אֶת חֻקֶּיךָ:

[תהילים פרק-קיט-קלו]  פַּלְגֵי מַיִם יָרְדוּ עֵינָי עַל לֹא שָׁמְרוּ תוֹרָתֶךָ:

[משלי פרק-כב-כה]  פֶּן תֶּאֱלַף (ארחתו) אֹרְחֹתָיו וְלָקַחְתָּ מוֹקֵשׁ לְנַפְשֶׁךָ:


פ-ל


[תהילים פרק-ז-ג]  פֶּן יִטְרֹף כְּאַרְיֵה נַפְשִׁי פֹּרֵק וְאֵין מַצִּיל:


פ-מ


[בראשית פרק-מא-י]  פַּרְעֹה קָצַף עַל עֲבָדָיו וַיִּתֵּן אֹתִי בְּמִשְׁמַר בֵּית שַׂר הַטַּבָּחִים אֹתִי וְאֵת שַׂר הָאֹפִים:

[במדבר פרק-א-כה]  פְּקֻדֵיהֶם לְמַטֵּה גָד חֲמִשָּׁה וְאַרְבָּעִים אֶלֶף וְשֵׁשׁ מֵאוֹת וַחֲמִשִּׁים:

[במדבר פרק-א-לה]  פְּקֻדֵיהֶם לְמַטֵּה מְנַשֶּׁה שְׁנַיִם וּשְׁלֹשִׁים אֶלֶף וּמָאתָיִם:

[במדבר פרק-ג-טו]  פְּקֹד אֶת בְּנֵי לֵוִי לְבֵית אֲבֹתָם לְמִשְׁפְּחֹתָם כָּל זָכָר מִבֶּן חֹדֶשׁ וָמַעְלָה תִּפְקְדֵם:

[דברים פרק-יט-ו]  פֶּן יִרְדֹּף גֹּאֵל הַדָּם אַחֲרֵי הָרֹצֵחַ כִּי יֵחַם לְבָבוֹ וְהִשִּׂיגוֹ כִּי יִרְבֶּה הַדֶּרֶךְ וְהִכָּהוּ נָפֶשׁ וְלוֹ אֵין מִשְׁפַּט מָוֶת כִּי לֹא שֹׂנֵא הוּא לוֹ מִתְּמוֹל שִׁלְשׁוֹם:

[דברים פרק-כח-לג]  פְּרִי אַדְמָתְךָ וְכָל יְגִיעֲךָ יֹאכַל עַם אֲשֶׁר לֹא יָדָעְתָּ וְהָיִיתָ רַק עָשׁוּק וְרָצוּץ כָּל הַיָּמִים:

[מלכים א פרק-ו-ח]  פֶּתַח הַצֵּלָע הַתִּיכֹנָה אֶל כֶּתֶף הַבַּיִת הַיְמָנִית וּבְלוּלִּים יַעֲלוּ עַל הַתִּיכֹנָה וּמִן הַתִּיכֹנָה אֶל הַשְּׁלִשִׁים:

[ישעיה פרק-כו-ב]  פִּתְחוּ שְׁעָרִים וְיָבֹא גוֹי צַדִּיק שֹׁמֵר אֱמֻנִים:

[ישעיה פרק-לג-יד]  פָּחֲדוּ בְצִיּוֹן חַטָּאִים אָחֲזָה רְעָדָה חֲנֵפִים מִי יָגוּר לָנוּ אֵשׁ אוֹכֵלָה מִי יָגוּר לָנוּ מוֹקְדֵי עוֹלָם:

[ישעיה פרק-נב-ט]  פִּצְחוּ רַנְּנוּ יַחְדָּו חָרְבוֹת יְרוּשָׁלִָם כִּי נִחַם יְהוָה עַמּוֹ גָּאַל יְרוּשָׁלִָם:

[ישעיה פרק-סה-ב]  פֵּרַשְׂתִּי יָדַי כָּל הַיּוֹם אֶל עַם סוֹרֵר הַהֹלְכִים הַדֶּרֶךְ לֹא טוֹב אַחַר מַחְשְׁבֹתֵיהֶם:

[ירמיה פרק-נ-כה]  פָּתַח יְהוָה אֶת אוֹצָרוֹ וַיּוֹצֵא אֶת כְּלֵי זַעְמוֹ כִּי מְלָאכָה הִיא לַאדֹנָי יְהוִה צְבָאוֹת בְּאֶרֶץ כַּשְׂדִּים:

[ירמיה פרק-נא-נ]  פְּלֵטִים מֵחֶרֶב הִלְכוּ אַל תַּעֲמֹדוּ זִכְרוּ מֵרָחוֹק אֶת יְהוָה וִירוּשָׁלִַם תַּעֲלֶה עַל לְבַבְכֶם:

[נחום פרק-ג-ג]  פָּרָשׁ מַעֲלֶה וְלַהַב חֶרֶב וּבְרַק חֲנִית וְרֹב חָלָל וְכֹבֶד פָּגֶר וְאֵין קֵצֶה לַגְּוִיָּה (יכשלו) וְכָשְׁלוּ בִּגְוִיָּתָם:

[תהילים פרק-יט-ט]  פִּקּוּדֵי יְהוָה יְשָׁרִים מְשַׂמְּחֵי לֵב מִצְוַת יְהוָה בָּרָה מְאִירַת עֵינָיִם:

[תהילים פרק-כא-יא]  פִּרְיָמוֹ מֵאֶרֶץ תְּאַבֵּד וְזַרְעָם מִבְּנֵי אָדָם:

[תהילים פרק-לד-יז]  פְּנֵי יְהוָה בְּעֹשֵׂי רָע לְהַכְרִית מֵאֶרֶץ זִכְרָם:

[תהילים פרק-קב-יח]  פָּנָה אֶל תְּפִלַּת הָעַרְעָר וְלֹא בָזָה אֶת תְּפִלָּתָם:

[תהילים פרק-קו-מג]  פְּעָמִים רַבּוֹת יַצִּילֵם וְהֵמָּה יַמְרוּ בַעֲצָתָם וַיָּמֹכּוּ בַּעֲוֹנָם:

[תהילים פרק-קיט-קל]  פֵּתַח דְּבָרֶיךָ יָאִיר מֵבִין פְּתָיִים:

[משלי פרק-יא-ל]  פְּרִי צַדִּיק עֵץ חַיִּים וְלֹקֵחַ נְפָשׂוֹת חָכָם:

[משלי פרק-כג-ח]  פִּתְּךָ אָכַלְתָּ תְקִיאֶנָּה וְשִׁחַתָּ דְּבָרֶיךָ הַנְּעִימִים:

[שיר השירים פרק-ה-ג]  פָּשַׁטְתִּי אֶת כֻּתָּנְתִּי אֵיכָכָה אֶלְבָּשֶׁנָּה רָחַצְתִּי אֶת רַגְלַי אֵיכָכָה אֲטַנְּפֵם:

[איכה פרק-א-יז]  פֵּרְשָׂה צִיּוֹן בְּיָדֶיהָ אֵין מְנַחֵם לָהּ צִוָּה יְהוָה לְיַעֲקֹב סְבִיבָיו צָרָיו הָיְתָה יְרוּשָׁלִַם לְנִדָּה בֵּינֵיהֶם:

[אסתר פרק-ח-יג]  פַּתְשֶׁגֶן הַכְּתָב לְהִנָּתֵן דָּת בְּכָל מְדִינָה וּמְדִינָה גָּלוּי לְכָל הָעַמִּים וְלִהְיוֹת (היהודיים) הַיְּהוּדִים (עתודים) עֲתִידִים לַיּוֹם הַזֶּה לְהִנָּקֵם מֵאֹיְבֵיהֶם:


פ-נ


[תהילים פרק-קיט-קלג]  פְּעָמַי הָכֵן בְּאִמְרָתֶךָ וְאַל תַּשְׁלֶט בִּי כָל אָוֶן:

[תהילים פרק-קמה-טז]  פּוֹתֵחַ אֶת יָדֶךָ וּמַשְׂבִּיעַ לְכָל חַי רָצוֹן:

[משלי פרק-לא-ט]  פְּתַח פִּיךָ שְׁפָט צֶדֶק וְדִין עָנִי וְאֶבְיוֹן:

[איוב פרק-טז-י]  פָּעֲרוּ עָלַי בְּפִיהֶם בְּחֶרְפָּה הִכּוּ לְחָיָי יַחַד עָלַי יִתְמַלָּאוּן:


פ-ס


[משלי פרק-טז-יא]  פֶּלֶס וּמֹאזְנֵי מִשְׁפָּט לַיהוָה מַעֲשֵׂהוּ כָּל אַבְנֵי כִיס:

[דניאל פרק-ה-כח]  פְּרֵס פְּרִיסַת מַלְכוּתָךְ וִיהִיבַת לְמָדַי וּפָרָס:


פ-ע


[ישעיה פרק-סד-ד]  פָּגַעְתָּ אֶת שָׂשׂ וְעֹשֵׂה צֶדֶק בִּדְרָכֶיךָ יִזְכְּרוּךָ הֵן אַתָּה קָצַפְתָּ וַנֶּחֱטָא בָּהֶם עוֹלָם וְנִוָּשֵׁעַ:

[יחזקאל פרק-לח-ה]  פָּרַס כּוּשׁ וּפוּט אִתָּם כֻּלָּם מָגֵן וְכוֹבָע:

[יחזקאל פרק-מד-יח]  פַּאֲרֵי פִשְׁתִּים יִהְיוּ עַל רֹאשָׁם וּמִכְנְסֵי פִשְׁתִּים יִהְיוּ עַל מָתְנֵיהֶם לֹא יַחְגְּרוּ בַּיָּזַע:


פ-פ


[משלי פרק-לא-ח]  פְּתַח פִּיךָ לְאִלֵּם אֶל דִּין כָּל בְּנֵי חֲלוֹף:


פ-צ


[ישעיה פרק-כד-יז]  פַּחַד וָפַחַת וָפָח עָלֶיךָ יוֹשֵׁב הָאָרֶץ:

[תהילים פרק-פ-י]  פִּנִּיתָ לְפָנֶיהָ וַתַּשְׁרֵשׁ שָׁרָשֶׁיהָ וַתְּמַלֵּא אָרֶץ:


פ-ר


[דברים פרק-ט-כח]  פֶּן יֹאמְרוּ הָאָרֶץ אֲשֶׁר הוֹצֵאתָנוּ מִשָּׁם מִבְּלִי יְכֹלֶת יְהוָה לַהֲבִיאָם אֶל הָאָרֶץ אֲשֶׁר דִּבֶּר לָהֶם וּמִשִּׂנְאָתוֹ אוֹתָם הוֹצִיאָם לַהֲמִתָם בַּמִּדְבָּר:

[ישעיה פרק-לו-יח]  פֶּן יַסִּית אֶתְכֶם חִזְקִיָּהוּ לֵאמֹר יְהוָה יַצִּילֵנוּ הַהִצִּילוּ אֱלֹהֵי הַגּוֹיִם אִישׁ אֶת אַרְצוֹ מִיַּד מֶלֶךְ אַשּׁוּר:

[ישעיה פרק-נט-יג]  פָּשֹׁעַ וְכַחֵשׁ בַּיהוָה וְנָסוֹג מֵאַחַר אֱלֹהֵינוּ דַּבֶּר עֹשֶׁק וְסָרָה הֹרוֹ וְהֹגוֹ מִלֵּב דִּבְרֵי שָׁקֶר:

[תהילים פרק-קה-מא]  פָּתַח צוּר וַיָּזוּבוּ מָיִם הָלְכוּ בַּצִּיּוֹת נָהָר:

[תהילים פרק-קמד-יא]  פְּצֵנִי וְהַצִּילֵנִי מִיַּד בְּנֵי נֵכָר אֲשֶׁר פִּיהֶם דִּבֶּר שָׁוְא וִימִינָם יְמִין שָׁקֶר:

[משלי פרק-ד-טו]  פְּרָעֵהוּ אַל תַּעֲבָר בּוֹ שְׂטֵה מֵעָלָיו וַעֲבוֹר:

[איכה פרק-ג-מז]  פַּחַד וָפַחַת הָיָה לָנוּ הַשֵּׁאת וְהַשָּׁבֶר:


פ-ש


[דברים פרק-ה-ד]  פָּנִים בְּפָנִים דִּבֶּר יְהוָה עִמָּכֶם בָּהָר מִתּוֹךְ הָאֵשׁ:

[משלי פרק-יז-יד]  פּוֹטֵר מַיִם רֵאשִׁית מָדוֹן וְלִפְנֵי הִתְגַּלַּע הָרִיב נְטוֹשׁ:

[איוב פרק-לב-יג]  פֶּן תֹּאמְרוּ מָצָאנוּ חָכְמָה אֵל יִדְּפֶנּוּ לֹא אִישׁ:

[נחמיה פרק-ג-כה]  פָּלָל בֶּן אוּזַי מִנֶּגֶד הַמִּקְצוֹעַ וְהַמִּגְדָּל הַיּוֹצֵא מִבֵּית הַמֶּלֶךְ הָעֶלְיוֹן אֲשֶׁר לַחֲצַר הַמַּטָּרָה אַחֲרָיו פְּדָיָה בֶן פַּרְעֹשׁ:


פ-ת


[במדבר פרק-א-כא]  פְּקֻדֵיהֶם לְמַטֵּה רְאוּבֵן שִׁשָּׁה וְאַרְבָּעִים אֶלֶף וַחֲמֵשׁ מֵאוֹת:

[במדבר פרק-א-כג]  פְּקֻדֵיהֶם לְמַטֵּה שִׁמְעוֹן תִּשְׁעָה וַחֲמִשִּׁים אֶלֶף וּשְׁלֹשׁ מֵאוֹת:

[במדבר פרק-א-כז]  פְּקֻדֵיהֶם לְמַטֵּה יְהוּדָה אַרְבָּעָה וְשִׁבְעִים אֶלֶף וְשֵׁשׁ מֵאוֹת:

[במדבר פרק-א-כט]  פְּקֻדֵיהֶם לְמַטֵּה יִשָּׂשכָר אַרְבָּעָה וַחֲמִשִּׁים אֶלֶף וְאַרְבַּע מֵאוֹת:

[במדבר פרק-א-לא]  פְּקֻדֵיהֶם לְמַטֵּה זְבוּלֻן שִׁבְעָה וַחֲמִשִּׁים אֶלֶף וְאַרְבַּע מֵאוֹת:

[במדבר פרק-א-לג]  פְּקֻדֵיהֶם לְמַטֵּה אֶפְרָיִם אַרְבָּעִים אֶלֶף וַחֲמֵשׁ מֵאוֹת:

[במדבר פרק-א-לז]  פְּקֻדֵיהֶם לְמַטֵּה בִנְיָמִן חֲמִשָּׁה וּשְׁלֹשִׁים אֶלֶף וְאַרְבַּע מֵאוֹת:

[במדבר פרק-א-לט]  פְּקֻדֵיהֶם לְמַטֵּה דָן שְׁנַיִם וְשִׁשִּׁים אֶלֶף וּשְׁבַע מֵאוֹת:

[במדבר פרק-א-מא]  פְּקֻדֵיהֶם לְמַטֵּה אָשֵׁר אֶחָד וְאַרְבָּעִים אֶלֶף וַחֲמֵשׁ מֵאוֹת:

[במדבר פרק-א-מג]  פְּקֻדֵיהֶם לְמַטֵּה נַפְתָּלִי שְׁלֹשָׁה וַחֲמִשִּׁים אֶלֶף וְאַרְבַּע מֵאוֹת:

[במדבר פרק-ג-כב]  פְּקֻדֵיהֶם בְּמִסְפַּר כָּל זָכָר מִבֶּן חֹדֶשׁ וָמָעְלָה פְּקֻדֵיהֶם שִׁבְעַת אֲלָפִים וַחֲמֵשׁ מֵאוֹת:

[דברים פרק-א-ז]  פְּנוּ וּסְעוּ לָכֶם וּבֹאוּ הַר הָאֱמֹרִי וְאֶל כָּל שְׁכֵנָיו בָּעֲרָבָה בָהָר וּבַשְּׁפֵלָה וּבַנֶּגֶב וּבְחוֹף הַיָּם אֶרֶץ הַכְּנַעֲנִי וְהַלְּבָנוֹן עַד הַנָּהָר הַגָּדֹל נְהַר פְּרָת:

[דברים פרק-ח-יב]  פֶּן תֹּאכַל וְשָׂבָעְתָּ וּבָתִּים טוֹבִים תִּבְנֶה וְיָשָׁבְתָּ:

[תהילים פרק-מט-ד]  פִּי יְדַבֵּר חָכְמוֹת וְהָגוּת לִבִּי תְבוּנוֹת:

[תהילים פרק-עא-טו]  פִּי יְסַפֵּר צִדְקָתֶךָ כָּל הַיּוֹם תְּשׁוּעָתֶךָ כִּי לֹא יָדַעְתִּי סְפֹרוֹת:

[משלי פרק-י-טז]  פְּעֻלַּת צַדִּיק לְחַיִּים תְּבוּאַת רָשָׁע לְחַטָּאת:

[משלי פרק-י-לא]  פִּי צַדִּיק יָנוּב חָכְמָה וּלְשׁוֹן תַּהְפֻּכוֹת תִּכָּרֵת:

[משלי פרק-כא-ו]  פֹּעַל אוֹצָרוֹת בִּלְשׁוֹן שָׁקֶר הֶבֶל נִדָּף מְבַקְשֵׁי מָוֶת:

[איוב פרק-טז-טז]  פָּנַי (חמרמרה) חֳמַרְמְרוּ מִנִּי בֶכִי וְעַל עַפְעַפַּי צַלְמָוֶת:

[עזרא פרק-ד-יז]  פִּתְגָמָא שְׁלַח מַלְכָּא עַל רְחוּם בְּעֵל טְעֵם וְשִׁמְשַׁי סָפְרָא וּשְׁאָר כְּנָוָתְהוֹן דִּי יָתְבִין בְּשָׁמְרָיִן וּשְׁאָר עֲבַר נַהֲרָה שְׁלָם וּכְעֶת:
