פרשת שמיני



ויקרא פרק-ט

{א} וַֽיְהִי בַּיּוֹם הַשְּׁמִינִי קָרָא מֹשֶׁה לְאַֽהֲרֹן וּלְבָנָיו וּלְזִקְנֵי יִשְׂרָאֵֽל:  מ  וַֽיְהִי בַּיּוֹם הַשְּׁמִינִי קָרָא מֹשֶׁה לְאַֽהֲרֹן וּלְבָנָיו וּלְזִקְנֵי יִשְׂרָאֵֽל:  ת  וַהֲוָה בְּיוֹמָא תְמִינָאָה קְרָא מֹשֶׁה לְאַהֲרֹן וְלִבְנוֹהִי וּלְסָבֵי יִשְׂרָאֵל:

{ב} וַיֹּאמֶר אֶֽל אַֽהֲרֹן קַח לְךָ עֵגֶל בֶּן בָּקָר לְחַטָּאת וְאַיִל לְעֹלָה תְּמִימִם וְהַקְרֵב לִפְנֵי יְהוָֹֽה:  מ  וַיֹּאמֶר אֶֽל אַֽהֲרֹן קַח לְךָ עֵגֶל בֶּן בָּקָר לְחַטָּאת וְאַיִל לְעֹלָה תְּמִימִם וְהַקְרֵב לִפְנֵי יְהוָֹֽה:  ת  וַאֲמַר לְאַהֲרֹן סַב לָךְ עֵגֶל בַּר תּוֹרֵי לְחַטָּאתָא וּדְכַר לַעֲלָתָא שַׁלְמִין וְקָרֵב קֳדָם יְיָ:

{ג} וְאֶל בְּנֵי יִשְׂרָאֵל תְּדַבֵּר לֵאמֹר קְחוּ שְׂעִיר עִזִּים לְחַטָּאת וְעֵגֶל וָכֶבֶשׂ בְּנֵֽי שָׁנָה תְּמִימִם לְעֹלָֽה:  מ  וְאֶל בְּנֵי יִשְׂרָאֵל תְּדַבֵּר לֵאמֹר קְחוּ שְׂעִיר עִזִּים לְחַטָּאת וְעֵגֶל וָכֶבֶשׂ בְּנֵֽי שָׁנָה תְּמִימִם לְעֹלָֽה:  ת  וְעִם בְּנֵי יִשְׂרָאֵל תְּמַלֵל לְמֵימָר סִבוּ צְפִיר בַּר עִזִין לְחַטָאתָא וְעֵגֶל וְאִמָר בְּנֵי שְׁנָא שַׁלְמִין לַעֲלָתָא:

{ד} וְשׁוֹר וָאַיִל לִשְׁלָמִים לִזְבֹּחַ לִפְנֵי יְהֹוָה וּמִנְחָה בְּלוּלָה בַשָּׁמֶן כִּי הַיּוֹם יְהוָֹה נִרְאָה אֲלֵיכֶֽם:  מ  וְשׁוֹר וָאַיִל לִשְׁלָמִים לִזְבֹּחַ לִפְנֵי יְהֹוָה וּמִנְחָה בְּלוּלָה בַשָּׁמֶן כִּי הַיּוֹם יְהוָֹה נִרְאָה אֲלֵיכֶֽם:  ת  וְתוֹר וּדְכַר לְנִכְסַת קוּדְשַׁיָא לְדַבָּחָא קֳדָם יְיָ וּמִנְחָתָא דְפִילָא בִמְשָׁח אֲרֵי יוֹמָא דֵין יְקָרָא דַיְיָ מִתְגְלִי לְכוֹן:

{ה} וַיִּקְחוּ אֵת אֲשֶׁר צִוָּה מֹשֶׁה אֶל פְּנֵי אֹהֶל מוֹעֵד וַֽיִּקְרְבוּ כָּל הָעֵדָה וַיַּֽעַמְדוּ לִפְנֵי יְהוָֹֽה:  מ  וַיִּקְחוּ אֵת אֲשֶׁר צִוָּה מֹשֶׁה אֶל פְּנֵי אֹהֶל מוֹעֵד וַֽיִּקְרְבוּ כָּל הָעֵדָה וַיַּֽעַמְדוּ לִפְנֵי יְהוָֹֽה:  ת  וּנְסִיבוּ יָת דִי פַקִיד מֹשֶׁה לָקֳדָם מַשְׁכַּן זִמְנָא וּקְרִיבוּ כָּל כְּנִישְׁתָּא וְקָמוּ קֳדָם יְיָ:

{ו} וַיֹּאמֶר מֹשֶׁה זֶה הַדָּבָר אֲשֶׁר צִוָּה יְהוָֹה תַּֽעֲשׂוּ וְיֵרָא אֲלֵיכֶם כְּבוֹד יְהוָֹֽה:  מ  וַיֹּאמֶר מֹשֶׁה זֶה הַדָּבָר אֲשֶׁר צִוָּה יְהוָֹה תַּֽעֲשׂוּ וְיֵרָא אֲלֵיכֶם כְּבוֹד יְהוָֹֽה:  ת  וַאֲמַר מֹשֶׁה דֵין פִּתְגָמָא דִי פַקִיד יְיָ תַּעְבְּדוּן וְיִתְגְלִי לְכוֹן יְקָרָא דַיְיָ:

{ז} וַיֹּאמֶר מֹשֶׁה אֶֽל אַֽהֲרֹן קְרַב אֶל הַמִּזְבֵּחַ וַֽעֲשֵׂה אֶת חַטָּֽאתְךָ וְאֶת עֹלָתֶךָ וְכַפֵּר בַּֽעַדְךָ וּבְעַד הָעָם וַֽעֲשֵׂה אֶת קָרְבַּן הָעָם וְכַפֵּר בַּֽעֲדָם כַּֽאֲשֶׁר צִוָּה יְהוָֹֽה:  מ  וַיֹּאמֶר מֹשֶׁה אֶֽל אַֽהֲרֹן קְרַב אֶל הַמִּזְבֵּחַ וַֽעֲשֵׂה אֶת חַטָּֽאתְךָ וְאֶת עֹלָתֶךָ וְכַפֵּר בַּֽעַדְךָ וּבְעַד הָעָם וַֽעֲשֵׂה אֶת קָרְבַּן הָעָם וְכַפֵּר בַּֽעֲדָם כַּֽאֲשֶׁר צִוָּה יְהוָֹֽה:  ת  וַאֲמַר מֹשֶׁה לְאַהֲרֹן קְרַב לְמַדְבְּחָא וְעִבֵד יָת חַטָאתָךְ וְיָת עֲלָתָךְ וְכַפֵּר עֲלָךְ וְעַל עַמָא וְעִבֵד יָת קֻרְבַּן עַמָא וְכַפֵּר עֲלֵיהוֹן כְּמָא דִי פַקִיד יְיָ:

{ח} וַיִּקְרַב אַֽהֲרֹן אֶל הַמִּזְבֵּחַ וַיִּשְׁחַט אֶת עֵגֶל הַֽחַטָּאת אֲשֶׁר לֽוֹ:  מ  וַיִּקְרַב אַֽהֲרֹן אֶל הַמִּזְבֵּחַ וַיִּשְׁחַט אֶת עֵגֶל הַֽחַטָּאת אֲשֶׁר לֽוֹ:  ת  וּקְרַב אַהֲרֹן לְמַדְבְּחָא וּנְכֵס יָת עֶגְלָא דְחַטָאתָא דִי לֵהּ:

{ט} וַיַּקְרִבוּ בְּנֵי אַֽהֲרֹן אֶת הַדָּם אֵלָיו וַיִּטְבֹּל אֶצְבָּעוֹ בַּדָּם וַיִּתֵּן עַל קַרְנוֹת הַמִּזְבֵּחַ וְאֶת הַדָּם יָצַק אֶל יְסוֹד הַמִּזְבֵּֽחַ:  מ  וַיַּקְרִבוּ בְּנֵי אַֽהֲרֹן אֶת הַדָּם אֵלָיו וַיִּטְבֹּל אֶצְבָּעוֹ בַּדָּם וַיִּתֵּן עַל קַרְנוֹת הַמִּזְבֵּחַ וְאֶת הַדָּם יָצַק אֶל יְסוֹד הַמִּזְבֵּֽחַ:  ת  וְקָרִיבוּ בְּנֵי אַהֲרֹן יָת דְמָא לֵהּ וּטְבַל אֶצְבְּעֵהּ בִּדְמָא וִיהַב עַל קַרְנַת מַדְבְּחָא וְיָת דְמָא אֲרִיק לִיסוֹדָא דְמַדְבְּחָא:

{י} וְאֶת הַחֵלֶב וְאֶת הַכְּלָיֹת וְאֶת הַיֹּתֶרֶת מִן הַכָּבֵד מִן הַחַטָּאת הִקְטִיר הַמִּזְבֵּחָה כַּֽאֲשֶׁר צִוָּה יְהוָֹה אֶת מֹשֶֽׁה:  מ  וְאֶת הַחֵלֶב וְאֶת הַכְּלָיֹת וְאֶת הַיֹּתֶרֶת מִן הַכָּבֵד מִן הַחַטָּאת הִקְטִיר הַמִּזְבֵּחָה כַּֽאֲשֶׁר צִוָּה יְהוָֹה אֶת מֹשֶֽׁה:  ת  וְיָת תַּרְבָּא וְיָת כָּלְיֵתָא וְיָת חִצְרָא מִן כַּבְדָא מִן חַטָאתָא אַסֵק לְמַדְבְּחָא כְּמָא דִי פַקִיד יְיָ יָת מֹשֶׁה:

{יא} וְאֶת הַבָּשָׂר וְאֶת הָעוֹר שָׂרַף בָּאֵשׁ מִחוּץ לַֽמַּֽחֲנֶֽה:  מ  וְאֶת הַבָּשָׂר וְאֶת הָעוֹר שָׂרַף בָּאֵשׁ מִחוּץ לַֽמַּֽחֲנֶֽה:  ת  וְיָת בִּשְׂרָא וְיָת מַשְׁכָא אוֹקִיד בְּנוּרָא מִבָּרָא לְמַשְׁרִיתָא:

{יב} וַיִּשְׁחַט אֶת הָֽעֹלָה וַיַּמְצִאוּ בְּנֵי אַֽהֲרֹן אֵלָיו אֶת הַדָּם וַיִּזְרְקֵהוּ עַל הַמִּזְבֵּחַ סָבִֽיב:  מ  וַיִּשְׁחַט אֶת הָֽעֹלָה וַיַּמְצִאוּ בְּנֵי אַֽהֲרֹן אֵלָיו אֶת הַדָּם וַיִּזְרְקֵהוּ עַל הַמִּזְבֵּחַ סָבִֽיב:  ת  וּנְכֵס יָת עֲלָתָא וְאַמְטִיאוּ בְּנֵי אַהֲרֹן לֵּהּ יָת דְמָא וּזְרָקֵהּ עַל מַדְבְּחָא סְחוֹר סְחוֹר:

{יג} וְאֶת הָֽעֹלָה הִמְצִיאוּ אֵלָיו לִנְתָחֶיהָ וְאֶת הָרֹאשׁ וַיַּקְטֵר עַל הַמִּזְבֵּֽחַ:  מ  וְאֶת הָֽעֹלָה הִמְצִיאוּ אֵלָיו לִנְתָחֶיהָ וְאֶת הָרֹאשׁ וַיַּקְטֵר עַל הַמִּזְבֵּֽחַ:  ת  וְיָת עֲלָתָא אַמְטִיאוּ לֵהּ לְאֵבָרָהָא וְיָת רֵישָׁא וְאַסֵק עַל מַדְבְּחָא:

{יד} וַיִּרְחַץ אֶת הַקֶּרֶב וְאֶת הַכְּרָעָיִם וַיַּקְטֵר עַל הָֽעֹלָה הַמִּזְבֵּֽחָה:  מ  וַיִּרְחַץ אֶת הַקֶּרֶב וְאֶת הַכְּרָעָיִם וַיַּקְטֵר עַל הָֽעֹלָה הַמִּזְבֵּֽחָה:  ת  וְחַלִיל יָת גַוָא וְיָת כְּרָעַיָא וְאַסֵק עַל עֲלָתָא לְמַדְבְּחָא:

{טו} וַיַּקְרֵב אֵת קָרְבַּן הָעָם וַיִּקַּח אֶת שְׂעִיר הַֽחַטָּאת אֲשֶׁר לָעָם וַיִּשְׁחָטֵהוּ וַֽיְחַטְּאֵהוּ כָּֽרִאשֽׁוֹן:  מ  וַיַּקְרֵב אֵת קָרְבַּן הָעָם וַיִּקַּח אֶת שְׂעִיר הַֽחַטָּאת אֲשֶׁר לָעָם וַיִּשְׁחָטֵהוּ וַֽיְחַטְּאֵהוּ כָּֽרִאשֽׁוֹן:  ת  וְקָרֵב יָת קֻרְבַּן עַמָא וּנְסִב יָת צְפִירָא דְחַטָאתָא דִי לְעַמָא וְנָכְסֵהּ וְכַפֵּר בִּדְמֵהּ כְּקַדְמָאָה:

{טז} וַיַּקְרֵב אֶת הָֽעֹלָה וַיַּֽעֲשֶׂהָ כַּמִּשְׁפָּֽט:  מ  וַיַּקְרֵב אֶת הָֽעֹלָה וַיַּֽעֲשֶׂהָ כַּמִּשְׁפָּֽט:  ת  וְקָרֵב יָת עֲלָתָא וְעָבְדַהּ כִּדְחָזֵי:

{יז}  שני  וַיַּקְרֵב אֶת הַמִּנְחָה וַיְמַלֵּא כַפּוֹ מִמֶּנָּה וַיַּקְטֵר עַל הַמִּזְבֵּחַ מִלְּבַד עֹלַת הַבֹּֽקֶר:  מ  וַיַּקְרֵב אֶת הַמִּנְחָה וַיְמַלֵּא כַפּוֹ מִמֶּנָּה וַיַּקְטֵר עַל הַמִּזְבֵּחַ מִלְּבַד עֹלַת הַבֹּֽקֶר:  ת  וְקָרִיב יָת מִנְחָתָא וּמְלָא יְדֵהּ מִנַהּ וְאַסֵק עַל מַדְבְּחָא בַּר מֵעֲלַת צַפְרָא:

{יח} וַיִּשְׁחַט אֶת הַשּׁוֹר וְאֶת הָאַיִל זֶבַח הַשְּׁלָמִים אֲשֶׁר לָעָם וַיַּמְצִאוּ בְּנֵי אַֽהֲרֹן אֶת הַדָּם אֵלָיו וַיִּזְרְקֵהוּ עַל הַמִּזְבֵּחַ סָבִֽיב:  מ  וַיִּשְׁחַט אֶת הַשּׁוֹר וְאֶת הָאַיִל זֶבַח הַשְּׁלָמִים אֲשֶׁר לָעָם וַיַּמְצִאוּ בְּנֵי אַֽהֲרֹן אֶת הַדָּם אֵלָיו וַיִּזְרְקֵהוּ עַל הַמִּזְבֵּחַ סָבִֽיב:  ת  וּנְכֵס יָת תּוֹרָא וְיָת דִכְרָא נִכְסַת קוּדְשַׁיָא דִי לְעַמָא וְאַמְטִיאוּ בְּנֵי אַהֲרֹן יָת דְמָא לֵהּ וּזְרָקֵהּ עַל מַדְבְּחָא סְחוֹר סְחוֹר:

{יט} וְאֶת הַֽחֲלָבִים מִן הַשּׁוֹר וּמִן הָאַיִל הָֽאַלְיָה וְהַֽמְכַסֶּה וְהַכְּלָיֹת וְיֹתֶרֶת הַכָּבֵֽד:  מ  וְאֶת הַֽחֲלָבִים מִן הַשּׁוֹר וּמִן הָאַיִל הָֽאַלְיָה וְהַֽמְכַסֶּה וְהַכְּלָיֹת וְיֹתֶרֶת הַכָּבֵֽד:  ת  וְיָת תַּרְבַּיָא מִן תּוֹרָא וּמִן דִכְרָא אֲלִיתָא וְחָפֵי גַוָא וְכָלְיֵתָא וַחֲצַר כַּבְדָא:

{כ} וַיָּשִׂימוּ אֶת הַֽחֲלָבִים עַל הֶֽחָזוֹת וַיַּקְטֵר הַֽחֲלָבִים הַמִּזְבֵּֽחָה:  מ  וַיָּשִׂימוּ אֶת הַֽחֲלָבִים עַל הֶֽחָזוֹת וַיַּקְטֵר הַֽחֲלָבִים הַמִּזְבֵּֽחָה:  ת  וְשַׁוִיאוּ יָת תַּרְבַּיָא עַל חֶדְוָתָא וְאַסֵק תַּרְבַּיָא לְמַדְבְּחָא:

{כא} וְאֵת הֶֽחָזוֹת וְאֵת שׁוֹק הַיָּמִין הֵנִיף אַֽהֲרֹן תְּנוּפָה לִפְנֵי יְהוָֹה כַּֽאֲשֶׁר צִוָּה מֹשֶֽׁה:  מ  וְאֵת הֶֽחָזוֹת וְאֵת שׁוֹק הַיָּמִין הֵנִיף אַֽהֲרֹן תְּנוּפָה לִפְנֵי יְהוָֹה כַּֽאֲשֶׁר צִוָּה מֹשֶֽׁה:  ת  וְיָת חֶדְוָתָא וְיָת שׁוֹקָא דְיַמִינָא אֲרֵים אַהֲרֹן אֲרָמָא קֳדָם יְיָ כְּמָא דִי פַקִיד מֹשֶׁה:

{כב} וַיִּשָּׂא אַֽהֲרֹן אֶת (ידו) יָדָיו אֶל הָעָם וַֽיְבָֽרֲכֵם וַיֵּרֶד מֵֽעֲשֹׂת הַֽחַטָּאת וְהָֽעֹלָה וְהַשְּׁלָמִֽים:  מ  וַיִּשָּׂא אַֽהֲרֹן אֶת (ידו) יָדָיו אֶל הָעָם וַֽיְבָֽרֲכֵם וַיֵּרֶד מֵֽעֲשֹׂת הַֽחַטָּאת וְהָֽעֹלָה וְהַשְּׁלָמִֽים:  ת  וַאֲרֵם אַהֲרֹן יָת יְדוֹהִי עַל עַמָא וּבָרֵכִנוּן וּנְחַת מִלְמֶעְבַּד חַטָאתָא וַעֲלָתָא וְנִכְסַת קוּדְשַׁיָא:

{כג} וַיָּבֹא מֹשֶׁה וְאַֽהֲרֹן אֶל אֹהֶל מוֹעֵד וַיֵּצְאוּ וַֽיְבָֽרֲכוּ אֶת הָעָם וַיֵּרָא כְבוֹד יְהוָֹה אֶל כָּל הָעָֽם:  מ  וַיָּבֹא מֹשֶׁה וְאַֽהֲרֹן אֶל אֹהֶל מוֹעֵד וַיֵּצְאוּ וַֽיְבָֽרֲכוּ אֶת הָעָם וַיֵּרָא כְבוֹד יְהוָֹה אֶל כָּל הָעָֽם:  ת  וְעַל מֹשֶׁה וְאַהֲרֹן לְמַשְׁכַּן זִמְנָא וּנְפָקוּ וּבָרִיכוּ יָת עַמָא וְאִתְגְלִי יְקָרָא דַיְיָ לְכָל עַמָא:

{כד}  שלישי  וַתֵּצֵא אֵשׁ מִלִּפְנֵי יְהֹוָה וַתֹּאכַל עַל הַמִּזְבֵּחַ אֶת הָֽעֹלָה וְאֶת הַֽחֲלָבִים וַיַּרְא כָּל הָעָם וַיָּרֹנּוּ וַֽיִּפְּלוּ עַל פְּנֵיהֶֽם:  מ  וַתֵּצֵא אֵשׁ מִלִּפְנֵי יְהֹוָה וַתֹּאכַל עַל הַמִּזְבֵּחַ אֶת הָֽעֹלָה וְאֶת הַֽחֲלָבִים וַיַּרְא כָּל הָעָם וַיָּרֹנּוּ וַֽיִּפְּלוּ עַל פְּנֵיהֶֽם:  ת  וּנְפָקַת אֶשָׁתָא מִן קֳדָם יְיָ וַאֲכָלַת עַל מַדְבְּחָא יָת עֲלָתָא וְיָת תַּרְבַּיָא וַחֲזָא כָּל עַמָא וְשַׁבָּחוּ וּנְפָלוּ עַל אַפֵּיהוֹן:




ויקרא פרק-י

{א} וַיִּקְחוּ בְנֵֽי אַֽהֲרֹן נָדָב וַֽאֲבִיהוּא אִישׁ מַחְתָּתוֹ וַיִּתְּנוּ בָהֵן אֵשׁ וַיָּשִׂימוּ עָלֶיהָ קְטֹרֶת וַיַּקְרִיבוּ לִפְנֵי יְהוָֹה אֵשׁ זָרָה אֲשֶׁר לֹא צִוָּה אֹתָֽם:  מ  וַיִּקְחוּ בְנֵֽי אַֽהֲרֹן נָדָב וַֽאֲבִיהוּא אִישׁ מַחְתָּתוֹ וַיִּתְּנוּ בָהֵן אֵשׁ וַיָּשִׂימוּ עָלֶיהָ קְטֹרֶת וַיַּקְרִיבוּ לִפְנֵי יְהוָֹה אֵשׁ זָרָה אֲשֶׁר לֹא צִוָּה אֹתָֽם:  ת  וּנְסִיבוּ בְנֵי אַהֲרֹן נָדָב וַאֲבִיהוּא גְבַר מַחְתִּיתֵהּ וִיהָבוּ בְהוֹן אֶשָׁתָא וְשַׁוִיאוּ עֲלַהּ קְטוֹרֶת וְקָרִיבוּ קֳדָם יְיָ אֶשָׁתָא נוּכְרֵיתָא דִי לָא פַקִיד יָתְהוֹן:

{ב} וַתֵּצֵא אֵשׁ מִלִּפְנֵי יְהוָֹה וַתֹּאכַל אוֹתָם וַיָּמֻתוּ לִפְנֵי יְהוָֹֽה:  מ  וַתֵּצֵא אֵשׁ מִלִּפְנֵי יְהוָֹה וַתֹּאכַל אוֹתָם וַיָּמֻתוּ לִפְנֵי יְהוָֹֽה:  ת  וּנְפָקַת אֶשָׁתָא מִן קֳדָם יְיָ וַאֲכָלַת יָתְהוֹן וּמִיתוּ קֳדָם יְיָ:

{ג} וַיֹּאמֶר מֹשֶׁה אֶֽל אַֽהֲרֹן הוּא אֲשֶׁר דִּבֶּר יְהוָֹה לֵאמֹר בִּקְרֹבַי אֶקָּדֵשׁ וְעַל פְּנֵי כָל הָעָם אֶכָּבֵד וַיִּדֹּם אַֽהֲרֹֽן:  מ  וַיֹּאמֶר מֹשֶׁה אֶֽל אַֽהֲרֹן הוּא אֲשֶׁר דִּבֶּר יְהוָֹה לֵאמֹר בִּקְרֹבַי אֶקָּדֵשׁ וְעַל פְּנֵי כָל הָעָם אֶכָּבֵד וַיִּדֹּם אַֽהֲרֹֽן:  ת  וַאֲמַר מֹשֶׁה לְאַהֲרֹן הוּא דִי מַלִיל יְיָ לְמֵימַר בְּקָרִיבַי אִתְקַדַשׁ וְעַל אַפֵּי כָל עַמָא אִתְיַקָר וּשְׁתִיק אַהֲרֹן:

{ד} וַיִּקְרָא מֹשֶׁה אֶל מִֽישָׁאֵל וְאֶל אֶלְצָפָן בְּנֵי עֻזִּיאֵל דֹּד אַֽהֲרֹן וַיֹּאמֶר אֲלֵהֶם קִרְבוּ שְׂאוּ אֶת אֲחֵיכֶם מֵאֵת פְּנֵֽי הַקֹּדֶשׁ אֶל מִחוּץ לַֽמַּֽחֲנֶֽה:  מ  וַיִּקְרָא מֹשֶׁה אֶל מִֽישָׁאֵל וְאֶל אֶלְצָפָן בְּנֵי עֻזִּיאֵל דֹּד אַֽהֲרֹן וַיֹּאמֶר אֲלֵהֶם קִרְבוּ שְׂאוּ אֶת אֲחֵיכֶם מֵאֵת פְּנֵֽי הַקֹּדֶשׁ אֶל מִחוּץ לַֽמַּֽחֲנֶֽה:  ת  וּקְרָא מֹשֶׁה לְמִישָׁאֵל וּלְאֶלְצָפָן בְּנֵי עֻזִיאֵל אַח אֲבוּהִי דְאַהֲרֹן וַאֲמַר לְהוֹן קְרִיבוּ טוּלוּ יָת אֲחֵיכוֹן מִן קֳדָם אַפֵּי קוּדְשָׁא לְמִבָּרָא לְּמַשְׁרִיתָא:

{ה} וַֽיִּקְרְבוּ וַיִּשָּׂאֻם בְּכֻתֳּנֹתָם אֶל מִחוּץ לַֽמַּֽחֲנֶה כַּֽאֲשֶׁר דִּבֶּר מֹשֶֽׁה:  מ  וַֽיִּקְרְבוּ וַיִּשָּׂאֻם בְּכֻתֳּנֹתָם אֶל מִחוּץ לַֽמַּֽחֲנֶה כַּֽאֲשֶׁר דִּבֶּר מֹשֶֽׁה:  ת  וּקְרִיבוּ וּנְטָלֻּנוּן בְּכִתֻּנֵיהוֹן לְמִבָּרָא לְמַשְׁרִיתָא כְּמָא דִי מַלִיל מֹשֶׁה:

{ו} וַיֹּאמֶר מֹשֶׁה אֶֽל אַֽהֲרֹן וּלְאֶלְעָזָר וּלְאִֽיתָמָר בָּנָיו רָֽאשֵׁיכֶם אַל תִּפְרָעוּ וּבִגְדֵיכֶם לֹֽא תִפְרֹמוּ וְלֹא תָמֻתוּ וְעַל כָּל הָֽעֵדָה יִקְצֹף וַֽאֲחֵיכֶם כָּל בֵּית יִשְׂרָאֵל יִבְכּוּ אֶת הַשְּׂרֵפָה אֲשֶׁר שָׂרַף יְהוָֹֽה:  מ  וַיֹּאמֶר מֹשֶׁה אֶֽל אַֽהֲרֹן וּלְאֶלְעָזָר וּלְאִֽיתָמָר בָּנָיו רָֽאשֵׁיכֶם אַל תִּפְרָעוּ וּבִגְדֵיכֶם לֹֽא תִפְרֹמוּ וְלֹא תָמֻתוּ וְעַל כָּל הָֽעֵדָה יִקְצֹף וַֽאֲחֵיכֶם כָּל בֵּית יִשְׂרָאֵל יִבְכּוּ אֶת הַשְּׂרֵפָה אֲשֶׁר שָׂרַף יְהוָֹֽה:  ת  וַאֲמַר מֹשֶׁה לְאַהֲרֹן וּלְאֶלְעָזָר וּלְאִיתָמָר בְּנוֹהִי רֵישֵׁיכוֹן לָא תְרַבּוּן פֵּרוּעַ וּלְבוּשֵׁיכוֹן לָא תְבַזְעוּן וְלָא תְמוּתוּן וְעַל כָּל כְּנִשְׁתָּא יְהֵא רוּגְזָא וַאֲחֵיכוֹן כָּל בֵּית יִשְׂרָאֵל יִבְכּוּן יָת יְקִדְתָּא דִי אוֹקִיד יְיָ:

{ז} וּמִפֶּתַח אֹהֶל מוֹעֵד לֹא תֵֽצְאוּ פֶּן תָּמֻתוּ כִּי שֶׁמֶן מִשְׁחַת יְהוָֹה עֲלֵיכֶם וַיַּֽעֲשׂוּ כִּדְבַר מֹשֶֽׁה:  מ  וּמִפֶּתַח אֹהֶל מוֹעֵד לֹא תֵֽצְאוּ פֶּן תָּמֻתוּ כִּי שֶׁמֶן מִשְׁחַת יְהוָֹה עֲלֵיכֶם וַיַּֽעֲשׂוּ כִּדְבַר מֹשֶֽׁה:  ת  וּמִתְּרַע מַשְׁכַּן זִמְנָא לָא תִפְקוּן דִלְמָא תְמוּתוּן אֲרֵי מְשַׁח רְבוּתָא דַיְיָ עֲלֵיכוֹן וַעֲבָדוּ כְּפִתְגָמָא דְמֹשֶׁה: (פ)

{ח} וַיְדַבֵּר יְהֹוָה אֶֽל אַֽהֲרֹן לֵאמֹֽר:  מ  וַיְדַבֵּר יְהֹוָה אֶֽל אַֽהֲרֹן לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם אַהֲרֹן לְמֵימָר:

{ט} יַיִן וְשֵׁכָר אַל תֵּשְׁתְּ אַתָּה וּבָנֶיךָ אִתָּךְ בְּבֹֽאֲכֶם אֶל אֹהֶל מוֹעֵד וְלֹא תָמֻתוּ חֻקַּת עוֹלָם לְדֹרֹֽתֵיכֶֽם:  מ  יַיִן וְשֵׁכָר אַל תֵּשְׁתְּ אַתָּה וּבָנֶיךָ אִתָּךְ בְּבֹֽאֲכֶם אֶל אֹהֶל מוֹעֵד וְלֹא תָמֻתוּ חֻקַּת עוֹלָם לְדֹרֹֽתֵיכֶֽם:  ת  חֲמַר וּמְרַוֵי לָא תִשְׁתֵּי אַתְּ וּבְנָיךְ עִמָךְ בְּמֵעַלְכוֹן לְמַשְׁכַּן זִמְנָא וְלָא תְמוּתוּן קְיַם עָלָם לְדָרֵיכוֹן:

{י} וּֽלְהַבְדִּיל בֵּין הַקֹּדֶשׁ וּבֵין הַחֹל וּבֵין הַטָּמֵא וּבֵין הַטָּהֽוֹר:  מ  וּֽלְהַבְדִּיל בֵּין הַקֹּדֶשׁ וּבֵין הַחֹל וּבֵין הַטָּמֵא וּבֵין הַטָּהֽוֹר:  ת  וּלְאַפְרָשָׁא בֵּין קוּדְשָׁא וּבֵין חֻלָא וּבֵין מְסָאֲבָא וּבֵין דַכְיָא:

{יא} וּלְהוֹרֹת אֶת בְּנֵי יִשְׂרָאֵל אֵת כָּל הַחֻקִּים אֲשֶׁר דִּבֶּר יְהוָֹה אֲלֵיהֶם בְּיַד מֹשֶֽׁה:  מ  וּלְהוֹרֹת אֶת בְּנֵי יִשְׂרָאֵל אֵת כָּל הַחֻקִּים אֲשֶׁר דִּבֶּר יְהוָֹה אֲלֵיהֶם בְּיַד מֹשֶֽׁה:  ת  וּלְאַלָפָא יָת בְּנֵי יִשְׂרָאֵל יָת כָּל קְיָמַיָא דִי מַלִיל יְיָ לְהוֹן בִּידָא דְמֹשֶׁה: (פ)

{יב}  רביעי  וַיְדַבֵּר מֹשֶׁה אֶֽל אַֽהֲרֹן וְאֶל אֶלְעָזָר וְאֶל אִיתָמָר בָּנָיו הַנּֽוֹתָרִים קְחוּ אֶת הַמִּנְחָה הַנּוֹתֶרֶת מֵֽאִשֵּׁי יְהֹוָה וְאִכְלוּהָ מַצּוֹת אֵצֶל הַמִּזְבֵּחַ כִּי קֹדֶשׁ קָֽדָשִׁים הִֽוא:  מ  וַיְדַבֵּר מֹשֶׁה אֶֽל אַֽהֲרֹן וְאֶל אֶלְעָזָר וְאֶל אִיתָמָר בָּנָיו הַנּֽוֹתָרִים קְחוּ אֶת הַמִּנְחָה הַנּוֹתֶרֶת מֵֽאִשֵּׁי יְהֹוָה וְאִכְלוּהָ מַצּוֹת אֵצֶל הַמִּזְבֵּחַ כִּי קֹדֶשׁ קָֽדָשִׁים הִֽוא:  ת  וּמַלִיל מֹשֶׁה עִם אַהֲרֹן וְעִם אֶלְעָזָר וְעִם אִיתָמָר בְּנוֹהִי דְאִשְׁתָּאֲרוּ סִבוּ יָת מִנְחָתָא דְאִשְׁתָּאָרָת מִקֻרְבָּנַיָא דַיְיָ וְאִכְלוּהָ פַטִיר בִּסְטַר מַדְבְּחָא אֲרֵי קֹדֶשׁ קוּדְשִׁין הִיא:

{יג} וַֽאֲכַלְתֶּם אֹתָהּ בְּמָקוֹם קָדוֹשׁ כִּי חָקְךָ וְחָק בָּנֶיךָ הִוא מֵֽאִשֵּׁי יְהוָֹה כִּי כֵן צֻוֵּֽיתִי:  מ  וַֽאֲכַלְתֶּם אֹתָהּ בְּמָקוֹם קָדוֹשׁ כִּי חָקְךָ וְחָק בָּנֶיךָ הִוא מֵֽאִשֵּׁי יְהוָֹה כִּי כֵן צֻוֵּֽיתִי:  ת  וְתֵיכְלוּן יָתַהּ בַּאֲתַר קַדִישׁ אֲרֵי חֳלָקָךְ וָחֳלַק בְּנָיךְ הִיא מִקֻרְבָּנַיָא דַיְיָ אֲרֵי כֵן אִתְפַּקָדִית:

{יד} וְאֵת חֲזֵה הַתְּנוּפָה וְאֵת שׁוֹק הַתְּרוּמָה תֹּֽאכְלוּ בְּמָקוֹם טָהוֹר אַתָּה וּבָנֶיךָ וּבְנֹתֶיךָ אִתָּךְ כִּֽי חָקְךָ וְחָק בָּנֶיךָ נִתְּנוּ מִזִּבְחֵי שַׁלְמֵי בְּנֵי יִשְׂרָאֵֽל:  מ  וְאֵת חֲזֵה הַתְּנוּפָה וְאֵת שׁוֹק הַתְּרוּמָה תֹּֽאכְלוּ בְּמָקוֹם טָהוֹר אַתָּה וּבָנֶיךָ וּבְנֹתֶיךָ אִתָּךְ כִּֽי חָקְךָ וְחָק בָּנֶיךָ נִתְּנוּ מִזִּבְחֵי שַׁלְמֵי בְּנֵי יִשְׂרָאֵֽל:  ת  וְיָת חֶדְיָא דַאֲרָמוּתָא וְיָת שׁוֹקָא דְאַפְרָשׁוּתָא תֵּיכְלוּן בַּאֲתַר דְכֵי אַתְּ וּבְנָיךְ וּבְנָתָיךְ עִמָךְ אֲרֵי חֳלָקָךְ וָחֳלַק בְּנָיךְ אִתְיְהִיבוּ מִנִכְסַת קוּדְשַׁיָא דִבְנֵי יִשְׂרָאֵל:

{טו} שׁוֹק הַתְּרוּמָה וַֽחֲזֵה הַתְּנוּפָה עַל אִשֵּׁי הַֽחֲלָבִים יָבִיאוּ לְהָנִיף תְּנוּפָה לִפְנֵי יְהוָֹה וְהָיָה לְךָ וּלְבָנֶיךָ אִתְּךָ לְחָק עוֹלָם כַּֽאֲשֶׁר צִוָּה יְהוָֹֽה:  מ  שׁוֹק הַתְּרוּמָה וַֽחֲזֵה הַתְּנוּפָה עַל אִשֵּׁי הַֽחֲלָבִים יָבִיאוּ לְהָנִיף תְּנוּפָה לִפְנֵי יְהוָֹה וְהָיָה לְךָ וּלְבָנֶיךָ אִתְּךָ לְחָק עוֹלָם כַּֽאֲשֶׁר צִוָּה יְהוָֹֽה:  ת  שׁוֹקָא דְאַפְרָשׁוּתָא וְחֶדְיָא דַאֲרָמוּתָא עַל קֻרְבָּנֵי תַרְבַּיָא יַיְתוּן לַאֲרָמָא אֲרָמָא קֳדָם יְיָ וִיהֵי לָךְ וְלִבְנָיךְ עִמָךְ לִקְיַם עָלָם כְּמָא דִי פַקִיד יְיָ:

{טז}  חמישי  וְאֵת שְׂעִיר הַֽחַטָּאת דָּרֹשׁ דָּרַשׁ מֹשֶׁה וְהִנֵּה שׂרָף וַיִּקְצֹף עַל אֶלְעָזָר וְעַל אִֽיתָמָר בְּנֵי אַֽהֲרֹן הַנּֽוֹתָרִם לֵאמֹר:  מ  וְאֵת שְׂעִיר הַֽחַטָּאת דָּרֹשׁ דָּרַשׁ מֹשֶׁה וְהִנֵּה שׂרָף וַיִּקְצֹף עַל אֶלְעָזָר וְעַל אִֽיתָמָר בְּנֵי אַֽהֲרֹן הַנּֽוֹתָרִם לֵאמֹר:  ת  וְיָת צְפִירָא דְחַטָאתָא מִתְבַּע תָּבְעֵי מֹשֶׁה וְהָא אִתּוֹקָד וּרְגֵז עַל אֶלְעָזָר וְעַל אִיתָמָר בְּנֵי אַהֲרֹן דְאִשְׁתָּאֲרוּ לְמֵימָר:

{יז} מַדּוּעַ לֹֽא אֲכַלְתֶּם אֶת הַֽחַטָּאת בִּמְקוֹם הַקֹּדֶשׁ כִּי קֹדֶשׁ קָֽדָשִׁים הִוא וְאֹתָהּ נָתַן לָכֶם לָשֵׂאת אֶת עֲוֹן הָֽעֵדָה לְכַפֵּר עֲלֵיהֶם לִפְנֵי יְהוָֹֽה:  מ  מַדּוּעַ לֹֽא אֲכַלְתֶּם אֶת הַֽחַטָּאת בִּמְקוֹם הַקֹּדֶשׁ כִּי קֹדֶשׁ קָֽדָשִׁים הִוא וְאֹתָהּ נָתַן לָכֶם לָשֵׂאת אֶת עֲוֹן הָֽעֵדָה לְכַפֵּר עֲלֵיהֶם לִפְנֵי יְהוָֹֽה:  ת  מָדֵין לָא אֲכַלְתּוּן יָת חַטָאתָא בַּאֲתַר קַדִישׁ אֲרֵי קֹדֶשׁ קוּדְשִׁין הִיא וְיָתַהּ יְהַב לְכוֹן לְסַלָחָא עַל חוֹבֵי כְנִשְׁתָּא לְכַפָּרָא עֲלֵיהוֹן קֳדָם יְיָ:

{יח} הֵן לֹֽא הוּבָא אֶת דָּמָהּ אֶל הַקֹּדֶשׁ פְּנִימָה אָכוֹל תֹּֽאכְלוּ אֹתָהּ בַּקֹּדֶשׁ כַּֽאֲשֶׁר צִוֵּֽיתִי:  מ  הֵן לֹֽא הוּבָא אֶת דָּמָהּ אֶל הַקֹּדֶשׁ פְּנִימָה אָכוֹל תֹּֽאכְלוּ אֹתָהּ בַּקֹּדֶשׁ כַּֽאֲשֶׁר צִוֵּֽיתִי:  ת  הָא לָא אִתָּעַל יָת דְמַהּ לְבֵית קֻדְשָׁא גַוָאָה מֵיכַל תֵּכְלוּן יָתַהּ בְּקֻדְשָׁא כְּמָא דִי פַקֵדִית:

{יט} וַיְדַבֵּר אַֽהֲרֹן אֶל מֹשֶׁה הֵן הַיּוֹם הִקְרִיבוּ אֶת חַטָּאתָם וְאֶת עֹֽלָתָם לִפְנֵי יְהֹוָה וַתִּקְרֶאנָה אֹתִי כָּאֵלֶּה וְאָכַלְתִּי חַטָּאת הַיּוֹם הַיִּיטַב בְּעֵינֵי יְהוָֹֽה:  מ  וַיְדַבֵּר אַֽהֲרֹן אֶל מֹשֶׁה הֵן הַיּוֹם הִקְרִיבוּ אֶת חַטָּאתָם וְאֶת עֹֽלָתָם לִפְנֵי יְהֹוָה וַתִּקְרֶאנָה אֹתִי כָּאֵלֶּה וְאָכַלְתִּי חַטָּאת הַיּוֹם הַיִּיטַב בְּעֵינֵי יְהוָֹֽה:  ת  וּמַלִיל אַהֲרֹן עִם מֹשֶׁה הָא יוֹמָא דֵין קְרִיבוּ יָת חַטָּאתְהוֹן וְיָת עֲלָתְהוֹן קֳדָם יְיָ וַעֲרָעָא יָתִי עָקָן כְּאִלֵין וְאִלוּפוֹן אֲכָלִית חַטָאתָא יוֹמָא דֵין הֲתַקֵן קֳדָם יְיָ:

{כ} וַיִּשְׁמַע מֹשֶׁה וַיִּיטַב בְּעֵינָֽיו:  מ  וַיִּשְׁמַע מֹשֶׁה וַיִּיטַב בְּעֵינָֽיו:  ת  וּשְׁמַע מֹשֶׁה וּשְׁפַר בְּעֵינוֹהִי: (פ)




ויקרא פרק-יא

{א}  שישי  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן לֵאמֹר אֲלֵהֶֽם:  מ  וַיְדַבֵּר יְהוָֹה אֶל מֹשֶׁה וְאֶֽל אַֽהֲרֹן לֵאמֹר אֲלֵהֶֽם:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה וְעִם אַהֲרֹן לְמֵימַר לְהוֹן:

{ב} דַּבְּרוּ אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר זֹאת הַֽחַיָּה אֲשֶׁר תֹּֽאכְלוּ מִכָּל הַבְּהֵמָה אֲשֶׁר עַל הָאָֽרֶץ:  מ  דַּבְּרוּ אֶל בְּנֵי יִשְׂרָאֵל לֵאמֹר זֹאת הַֽחַיָּה אֲשֶׁר תֹּֽאכְלוּ מִכָּל הַבְּהֵמָה אֲשֶׁר עַל הָאָֽרֶץ:  ת  מַלִילוּ עִם בְּנֵי יִשְׂרָאֵל לְמֵימַר דָא חַיְתָא דִי תֵיכְלוּן מִכָּל בְּעִירָא דִי עַל אַרְעָא:

{ג} כֹּל מַפְרֶסֶת פַּרְסָה וְשֹׁסַעַת שֶׁסַע פְּרָסֹת מַֽעֲלַת גֵּרָה בַּבְּהֵמָה אֹתָהּ תֹּאכֵֽלוּ:  מ  כֹּל מַפְרֶסֶת פַּרְסָה וְשֹׁסַעַת שֶׁסַע פְּרָסֹת מַֽעֲלַת גֵּרָה בַּבְּהֵמָה אֹתָהּ תֹּאכֵֽלוּ:  ת  כֹּל דִי סְדִיקָן פַּרְסָתָא וּמַטִלְפָן טִלְפִין פַּרְסָתָא מַסְקָּא פִשְׁרָא בִּבְעִירָא יָתַהּ תֵּיכְלוּן:

{ד} אַךְ אֶת זֶה לֹא תֹֽאכְלוּ מִמַּֽעֲלֵי הַגֵּרָה וּמִמַּפְרִסֵי הַפַּרְסָה אֶת הַגָּמָל כִּי מַֽעֲלֵה גֵרָה הוּא וּפַרְסָה אֵינֶנּוּ מַפְרִיס טָמֵא הוּא לָכֶֽם:  מ  אַךְ אֶת זֶה לֹא תֹֽאכְלוּ מִמַּֽעֲלֵי הַגֵּרָה וּמִמַּפְרִסֵי הַפַּרְסָה אֶת הַגָּמָל כִּי מַֽעֲלֵה גֵרָה הוּא וּפַרְסָה אֵינֶנּוּ מַפְרִיס טָמֵא הוּא לָכֶֽם:  ת  בְּרָם יָת דֵין לָא תֵיכְלוּן מִמַסְקֵי פִשְׁרָא וּמִמַסְדִיקֵי פַּרְסָתָא יָת גַמְלָא אֲרֵי מַסִיק פִּשְׁרָא הוּא וּפַרְסָתָא לָא סְדִיקָא מְסָאָב הוּא לְכוֹן:

{ה} וְאֶת הַשָּׁפָן כִּי מַֽעֲלֵה גֵרָה הוּא וּפַרְסָה לֹא יַפְרִיס טָמֵא הוּא לָכֶֽם:  מ  וְאֶת הַשָּׁפָן כִּי מַֽעֲלֵה גֵרָה הוּא וּפַרְסָה לֹא יַפְרִיס טָמֵא הוּא לָכֶֽם:  ת  וְיָת טַפְזָא אֲרֵי מַסִיק פִּשְׁרָא הוּא וּפַרְסָתָא לָא סְדִיקָא מְסָאָב הוּא לְכוֹן:

{ו} וְאֶת הָֽאַרְנֶבֶת כִּי מַֽעֲלַת גֵּרָה הִוא וּפַרְסָה לֹא הִפְרִיסָה טְמֵאָה הִוא לָכֶֽם:  מ  וְאֶת הָֽאַרְנֶבֶת כִּי מַֽעֲלַת גֵּרָה הִוא וּפַרְסָה לֹא הִפְרִיסָה טְמֵאָה הִוא לָכֶֽם:  ת  וְיָת אַרְנְבָא אֲרֵי מַסְקָא פִשְׁרָא הִיא וּפַרְסָתָא לָא סְדִיקָא מְסָאָבָא הִיא לְכוֹן:

{ז} וְאֶֽת הַֽחֲזִיר כִּֽי מַפְרִיס פַּרְסָה הוּא וְשֹׁסַע שֶׁסַע פַּרְסָה וְהוּא גֵּרָה לֹֽא יִגָּר טָמֵא הוּא לָכֶֽם:  מ  וְאֶֽת הַֽחֲזִיר כִּֽי מַפְרִיס פַּרְסָה הוּא וְשֹׁסַע שֶׁסַע פַּרְסָה וְהוּא גֵּרָה לֹֽא יִגָּר טָמֵא הוּא לָכֶֽם:  ת  וְיָת חֲזִירָא אֲרֵי סְדִיק פַּרְסָתָא הוּא וּמַטִּלְפָן טִּלְפִין פַרְסָתֵהּ וְהוּא פִּשְׁרָא לָא פָשָׁר מְסָאָב הוּא לְכוֹן:

{ח} מִבְּשָׂרָם לֹא תֹאכֵלוּ וּבְנִבְלָתָם לֹא תִגָּעוּ טְמֵאִים הֵם לָכֶֽם:  מ  מִבְּשָׂרָם לֹא תֹאכֵלוּ וּבְנִבְלָתָם לֹא תִגָּעוּ טְמֵאִים הֵם לָכֶֽם:  ת  מִבִּסְרְהוֹן לָא תֵיכְלוּן וּבִנְבִלְתְּהוֹן לָא תִקְרְבוּן מְסָאֲבִין אִנוּן לְכוֹן:

{ט} אֶת זֶה תֹּֽאכְלוּ מִכֹּל אֲשֶׁר בַּמָּיִם כֹּל אֲשֶׁר לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת בַּמַּיִם בַּיַּמִּים וּבַנְּחָלִים אֹתָם תֹּאכֵֽלוּ:  מ  אֶת זֶה תֹּֽאכְלוּ מִכֹּל אֲשֶׁר בַּמָּיִם כֹּל אֲשֶׁר לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת בַּמַּיִם בַּיַּמִּים וּבַנְּחָלִים אֹתָם תֹּאכֵֽלוּ:  ת  יָת דֵין תֵּיכְלוּן מִכֹּל דִי בְמַיָא כֹּל דִי לֵהּ צִיצִין וְקַלְפִין בְּמַיָא בְּיַמְמַיָא וּבְנַחֲלַיָא יַתְהוֹן תֵּיכְלוּן:

{י} וְכֹל אֲשֶׁר אֵין לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת בַּיַּמִּים וּבַנְּחָלִים מִכֹּל שֶׁרֶץ הַמַּיִם וּמִכֹּל נֶפֶשׁ הַֽחַיָּה אֲשֶׁר בַּמָּיִם שֶׁקֶץ הֵם לָכֶֽם:  מ  וְכֹל אֲשֶׁר אֵין לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת בַּיַּמִּים וּבַנְּחָלִים מִכֹּל שֶׁרֶץ הַמַּיִם וּמִכֹּל נֶפֶשׁ הַֽחַיָּה אֲשֶׁר בַּמָּיִם שֶׁקֶץ הֵם לָכֶֽם:  ת  וְכֹל דִי לֵית לֵהּ צִיצִין וְקַלְפִין בְּיַמְמַיָא וּבְנַחֲלַיָא מִכֹּל רִחֲשָׁא דְמַיָא וּמִכֹּל נַפְשָׁא חַיְתָא דִי בְמַיָא שִׁקְצָא אִנוּן לְכוֹן:

{יא} וְשֶׁקֶץ יִהְיוּ לָכֶם מִבְּשָׂרָם לֹא תֹאכֵלוּ וְאֶת נִבְלָתָם תְּשַׁקֵּֽצוּ:  מ  וְשֶׁקֶץ יִהְיוּ לָכֶם מִבְּשָׂרָם לֹא תֹאכֵלוּ וְאֶת נִבְלָתָם תְּשַׁקֵּֽצוּ:  ת  וְשִׁקְצָא יְהוֹן לְכוֹן מִבִּסְרְהוֹן לָא תֵיכְלוּן וְיָת נְבִלְתְּהוֹן תְּשַׁקְצוּן:

{יב} כֹּל אֲשֶׁר אֵין לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת בַּמָּיִם שֶׁקֶץ הוּא לָכֶֽם:  מ  כֹּל אֲשֶׁר אֵין לוֹ סְנַפִּיר וְקַשְׂקֶשֶׂת בַּמָּיִם שֶׁקֶץ הוּא לָכֶֽם:  ת  כֹּל דִי לֵית לֵהּ צִיצִין וְקַלְפִין בְּמַיָא שִׁקְצָא הוּא לְכוֹן:

{יג} וְאֶת אֵלֶּה תְּשַׁקְּצוּ מִן הָעוֹף לֹא יֵאָֽכְלוּ שֶׁקֶץ הֵם אֶת הַנֶּשֶׁר וְאֶת הַפֶּרֶס וְאֵת הָֽעָזְנִיָּֽה:  מ  וְאֶת אֵלֶּה תְּשַׁקְּצוּ מִן הָעוֹף לֹא יֵאָֽכְלוּ שֶׁקֶץ הֵם אֶת הַנֶּשֶׁר וְאֶת הַפֶּרֶס וְאֵת הָֽעָזְנִיָּֽה:  ת  וְיָת אִלֵין תְּשַׁקְצוּן מִן עוֹפָא לָא יִתְאָכְלוּן שִׁקְצָא אִנוּן נִשְׁרָא וְעֲר וְעַזְיָא:

{יד} וְאֶת הַדָּאָה וְאֶת הָֽאַיָּה לְמִינָֽהּ:  מ  וְאֶת הַדָּאָה וְאֶת הָֽאַיָּה לְמִינָֽהּ:  ת  וְדַיְתָא וְטָרָפִיתָא לִזְנַהּ:

{טו} אֵת כָּל עֹרֵב לְמִינֽוֹ:  מ  אֵת כָּל עֹרֵב לְמִינֽוֹ:  ת  יָת כָּל עוּרְבָא לִזְנֵהּ:

{טז} וְאֵת בַּת הַיַּֽעֲנָה וְאֶת הַתַּחְמָס וְאֶת הַשָּׁחַף וְאֶת הַנֵּץ לְמִינֵֽהוּ:  מ  וְאֵת בַּת הַיַּֽעֲנָה וְאֶת הַתַּחְמָס וְאֶת הַשָּׁחַף וְאֶת הַנֵּץ לְמִינֵֽהוּ:  ת  וְיָת בַּת נַעֲמִיתָא וְצִיצָא וְצִפַּר שַׁחְפָא וְנָצָא לִזְנוֹהִי:

{יז} וְאֶת הַכּוֹס וְאֶת הַשָּׁלָךְ וְאֶת הַיַּנְשֽׁוּף:  מ  וְאֶת הַכּוֹס וְאֶת הַשָּׁלָךְ וְאֶת הַיַּנְשֽׁוּף:  ת  וְקַרְיָא וְשָׁלֵינוּנָא וְקִפּוֹפָא:

{יח} וְאֶת הַתִּנְשֶׁמֶת וְאֶת הַקָּאָת וְאֶת הָֽרָחָֽם:  מ  וְאֶת הַתִּנְשֶׁמֶת וְאֶת הַקָּאָת וְאֶת הָֽרָחָֽם:  ת  וּבַוְתָא וְקָתָא וִירַקְרֵקָא:

{יט} וְאֵת הַֽחֲסִידָה הָֽאֲנָפָה לְמִינָהּ וְאֶת הַדּֽוּכִיפַת וְאֶת הָֽעֲטַלֵּֽף:  מ  וְאֵת הַֽחֲסִידָה הָֽאֲנָפָה לְמִינָהּ וְאֶת הַדּֽוּכִיפַת וְאֶת הָֽעֲטַלֵּֽף:  ת  וְחַוָרִיתָא וְאִבּוּ לִזְנַהּ וְנַגַר טוּרָא וַעֲטַלֵפָא:

{כ} כֹּל שֶׁרֶץ הָעוֹף הַֽהֹלֵךְ עַל אַרְבַּע שֶׁקֶץ הוּא לָכֶֽם:  מ  כֹּל שֶׁרֶץ הָעוֹף הַֽהֹלֵךְ עַל אַרְבַּע שֶׁקֶץ הוּא לָכֶֽם:  ת  כֹּל רִחֲשָׁא דְעוֹפָא דִמְהַלֵךְ עַל אַרְבַּע שִׁקְצָא הוּא לְכוֹן:

{כא} אַךְ אֶת זֶה תֹּֽאכְלוּ מִכֹּל שֶׁרֶץ הָעוֹף הַֽהֹלֵךְ עַל אַרְבַּע אֲשֶׁר (לא) לוֹ כְרָעַיִם מִמַּעַל לְרַגְלָיו לְנַתֵּר בָּהֵן עַל הָאָֽרֶץ:  מ  אַךְ אֶת זֶה תֹּֽאכְלוּ מִכֹּל שֶׁרֶץ הָעוֹף הַֽהֹלֵךְ עַל אַרְבַּע אֲשֶׁר (לא) לוֹ כְרָעַיִם מִמַּעַל לְרַגְלָיו לְנַתֵּר בָּהֵן עַל הָאָֽרֶץ:  ת  בְּרַם יָת דֵין תֵּיכְלוּן מִכֹּל רִחֲשָׁא דְעוֹפָא דִמְהַלֵךְ עַל אַרְבַּע דִי לֵהּ קַרְצוּלִין מֵעִלַוֵי רַגְלוֹהִי לְקַפָּצָא בְהוֹן עַל אַרְעָא:

{כב} אֶת אֵלֶּה מֵהֶם תֹּאכֵלוּ אֶת הָֽאַרְבֶּה לְמִינוֹ וְאֶת הַסָּלְעָם לְמִינֵהוּ וְאֶת הַֽחַרְגֹּל לְמִינֵהוּ וְאֶת הֶֽחָגָב לְמִינֵֽהוּ:  מ  אֶת אֵלֶּה מֵהֶם תֹּאכֵלוּ אֶת הָֽאַרְבֶּה לְמִינוֹ וְאֶת הַסָּלְעָם לְמִינֵהוּ וְאֶת הַֽחַרְגֹּל לְמִינֵהוּ וְאֶת הֶֽחָגָב לְמִינֵֽהוּ:  ת  יָת אִלֵין מִנְהוֹן תֵּיכְלוּן יָת גוֹבָא לִזְנֵהּ וְיָת רָשׁוּנָא לִזְנוֹהִי וְיָת חַרְגוֹלָא לִזְנוֹהִי וְיָת חַגָבָא לִזְנוֹהִי:

{כג} וְכֹל שֶׁרֶץ הָעוֹף אֲשֶׁר לוֹ אַרְבַּע רַגְלָיִם שֶׁקֶץ הוּא לָכֶֽם:  מ  וְכֹל שֶׁרֶץ הָעוֹף אֲשֶׁר לוֹ אַרְבַּע רַגְלָיִם שֶׁקֶץ הוּא לָכֶֽם:  ת  וְכֹל רִחֲשָׁא דְעוֹפָא דִי לֵהּ אַרְבַּע רַגְלִין שִׁקְצָא הוּא לְכוֹן:

{כד} וּלְאֵלֶּה תִּטַּמָּאוּ כָּל הַנֹּגֵעַ בְּנִבְלָתָם יִטְמָא עַד הָעָֽרֶב:  מ  וּלְאֵלֶּה תִּטַּמָּאוּ כָּל הַנֹּגֵעַ בְּנִבְלָתָם יִטְמָא עַד הָעָֽרֶב:  ת  וּלְאִלֵין תִּסְתָּאָבוּן כָּל דְיִקְרַב בִּנְבִלְתְּהוֹן יְהֵי מְסָאָב עַד רַמשָׁא:

{כה} וְכָל הַנֹּשֵׂא מִנִּבְלָתָם יְכַבֵּס בְּגָדָיו וְטָמֵא עַד הָעָֽרֶב:  מ  וְכָל הַנֹּשֵׂא מִנִּבְלָתָם יְכַבֵּס בְּגָדָיו וְטָמֵא עַד הָעָֽרֶב:  ת  וְכָל דְיִטוֹל:מִנְבִלְתְּהוֹן יְצַבַּע לְבוּשׁוֹהִי וִיהֵי מְסָאָב עַד רַמְשָׁא:

{כו} לְכָל הַבְּהֵמָה אֲשֶׁר הִוא מַפְרֶסֶת פַּרְסָה וְשֶׁסַע אֵינֶנָּה שֹׁסַעַת וְגֵרָה אֵינֶנָּה מַֽעֲלָה טְמֵאִים הֵם לָכֶם כָּל הַנֹּגֵעַ בָּהֶם יִטְמָֽא:  מ  לְכָל הַבְּהֵמָה אֲשֶׁר הִוא מַפְרֶסֶת פַּרְסָה וְשֶׁסַע אֵינֶנָּה שֹׁסַעַת וְגֵרָה אֵינֶנָּה מַֽעֲלָה טְמֵאִים הֵם לָכֶם כָּל הַנֹּגֵעַ בָּהֶם יִטְמָֽא:  ת  לְכָל בְּעִירָא דִי הִיא סְדִיקָא פַרְסָתָא וְטִלְפִין לֵיתָהָא מַטִלְפָא וּפִשְׁרָא לֵיתָהָא מַסְקָא מְסָאֲבִין אִנוּן לְכוֹן כָּל דְיִקְרַב בְּהוֹן יְהֵי מְסָאָב:

{כז} וְכֹל הוֹלֵךְ עַל כַּפָּיו בְּכָל הַֽחַיָּה הַהֹלֶכֶת עַל אַרְבַּע טְמֵאִים הֵם לָכֶם כָּל הַנֹּגֵעַ בְּנִבְלָתָם יִטְמָא עַד הָעָֽרֶב:  מ  וְכֹל הוֹלֵךְ עַל כַּפָּיו בְּכָל הַֽחַיָּה הַהֹלֶכֶת עַל אַרְבַּע טְמֵאִים הֵם לָכֶם כָּל הַנֹּגֵעַ בְּנִבְלָתָם יִטְמָא עַד הָעָֽרֶב:  ת  וְכֹל דִמְהַלֵךְ עַל יְדוֹהִי בְּכָל חַיְתָא דִמְהַלְכָא עַל אַרְבַּע מְסַאֲבִין אִנוּן לְכוֹן כָּל דְיִקְרַב בִּנְבִלְתְּהוֹן יְהֵא מְסָאָב עַד רַמְשָׁא:

{כח} וְהַנֹּשֵׂא אֶת נִבְלָתָם יְכַבֵּס בְּגָדָיו וְטָמֵא עַד הָעָרֶב טְמֵאִים הֵמָּה לָכֶֽם:  מ  וְהַנֹּשֵׂא אֶת נִבְלָתָם יְכַבֵּס בְּגָדָיו וְטָמֵא עַד הָעָרֶב טְמֵאִים הֵמָּה לָכֶֽם:  ת  וּדְיִטוֹל יָת נְבִלְתְּהוֹן יְצַבַּע לְבוּשׁוֹהִי וִיהֵי מְסָאָב עַד רַמְשָׁא מְסַאֲבִין אִנוּן לְכוֹן: (ס)

{כט} וְזֶה לָכֶם הַטָּמֵא בַּשֶּׁרֶץ הַשֹּׁרֵץ עַל הָאָרֶץ הַחֹלֶד וְהָֽעַכְבָּר וְהַצָּב לְמִינֵֽהוּ:  מ  וְזֶה לָכֶם הַטָּמֵא בַּשֶּׁרֶץ הַשֹּׁרֵץ עַל הָאָרֶץ הַחֹלֶד וְהָֽעַכְבָּר וְהַצָּב לְמִינֵֽהוּ:  ת  וְדֵין לְכוֹן מְסַאָבָא דְּרִחֲשָׁא דְרָחֵשׁ עַל אַרְעָא חֻלְדָא וְעַכְבָּרָא וְצָבָא לִזְנוֹהִי:

{ל} וְהָֽאֲנָקָה וְהַכֹּחַ וְהַלְּטָאָה וְהַחֹמֶט וְהַתִּנְשָֽׁמֶת:  מ  וְהָֽאֲנָקָה וְהַכֹּחַ וְהַלְּטָאָה וְהַחֹמֶט וְהַתִּנְשָֽׁמֶת:  ת  וְיָלָא וְכוֹחָא וְחֻלְטֵיתָא וְחֻמְטָא וְאָשׁוּתָא:

{לא} אֵלֶּה הַטְּמֵאִים לָכֶם בְּכָל הַשָּׁרֶץ כָּל הַנֹּגֵעַ בָּהֶם בְּמֹתָם יִטְמָא עַד הָעָֽרֶב:  מ  אֵלֶּה הַטְּמֵאִים לָכֶם בְּכָל הַשָּׁרֶץ כָּל הַנֹּגֵעַ בָּהֶם בְּמֹתָם יִטְמָא עַד הָעָֽרֶב:  ת  אִלֵין דִמְסַאֲבִין לְכוֹן בְּכָל רִחֲשָׁא כֹּל דְיִקְרַב בְּהוֹן בְּמוֹתְהוֹן יְהֵי מְסָאָב עַד רַמְשָׁא:

{לב} וְכֹל אֲשֶׁר יִפֹּל עָלָיו מֵהֶם בְּמֹתָם יִטְמָא מִכָּל כְּלִי עֵץ אוֹ בֶגֶד אוֹ עוֹר אוֹ שָׂק כָּל כְּלִי אֲשֶׁר יֵֽעָשֶׂה מְלָאכָה בָּהֶם בַּמַּיִם יוּבָא וְטָמֵא עַד הָעֶרֶב וְטָהֵֽר:  מ  וְכֹל אֲשֶׁר יִפֹּל עָלָיו מֵהֶם בְּמֹתָם יִטְמָא מִכָּל כְּלִי עֵץ אוֹ בֶגֶד אוֹ עוֹר אוֹ שָׂק כָּל כְּלִי אֲשֶׁר יֵֽעָשֶׂה מְלָאכָה בָּהֶם בַּמַּיִם יוּבָא וְטָמֵא עַד הָעֶרֶב וְטָהֵֽר:  ת  וְכָל דִי יִפֵּל עֲלוֹהִי מִנְהוֹן בְּמוֹתְהוֹן יְהֵי מְסָאָב מִכָּל מָאן דְעָא אוֹ לְבוּשׁ אוֹ מְשַׁךְ אוֹ שָׂק כָּל מָאן דִי תִתְעֲבֵד עִבִידָא בְּהוֹן בְּמַיָא יִתָּעַל וִיהֵי מְסָאָב עַד רַמְשָׁא וְיִדְכֵּי:

{לג}  שביעי  וְכָל כְּלִי חֶרֶשׂ אֲשֶׁר יִפֹּל מֵהֶם אֶל תּוֹכוֹ כֹּל אֲשֶׁר בְּתוֹכוֹ יִטְמָא וְאֹתוֹ תִשְׁבֹּֽרוּ:  מ  וְכָל כְּלִי חֶרֶשׂ אֲשֶׁר יִפֹּל מֵהֶם אֶל תּוֹכוֹ כֹּל אֲשֶׁר בְּתוֹכוֹ יִטְמָא וְאֹתוֹ תִשְׁבֹּֽרוּ:  ת  וְכָל מָאן דַחֲסַף דִי יִפֵּל מִנְהוֹן לְגַוֵהּ כֹּל דִי בְגַוֵהּ יִסְתָּאַב וְיָתֵהּ תְּתַבְּרוּן:

{לד} מִכָּל הָאֹכֶל אֲשֶׁר יֵֽאָכֵל אֲשֶׁר יָבוֹא עָלָיו מַיִם יִטְמָא וְכָל מַשְׁקֶה אֲשֶׁר יִשָּׁתֶה בְּכָל כְּלִי יִטְמָֽא:  מ  מִכָּל הָאֹכֶל אֲשֶׁר יֵֽאָכֵל אֲשֶׁר יָבוֹא עָלָיו מַיִם יִטְמָא וְכָל מַשְׁקֶה אֲשֶׁר יִשָּׁתֶה בְּכָל כְּלִי יִטְמָֽא:  ת  מִכָּל מֵיכְלָא דְמִתְאֲכֵל דִי יַעֲלוּן עֲלוֹהִי מַיָא יְהֵי מְסָאָב וְכָל מַשְׁקֵה דִי יִשְׁתֵּתֵי בְּכָל מָאן יְהֵי מְסָאָב:

{לה} וְכֹל אֲשֶׁר יִפֹּל מִנִּבְלָתָם עָלָיו יִטְמָא תַּנּוּר וְכִירַיִם יֻתָּץ טְמֵאִים הֵם וּטְמֵאִים יִֽהְיוּ לָכֶֽם:  מ  וְכֹל אֲשֶׁר יִפֹּל מִנִּבְלָתָם עָלָיו יִטְמָא תַּנּוּר וְכִירַיִם יֻתָּץ טְמֵאִים הֵם וּטְמֵאִים יִֽהְיוּ לָכֶֽם:  ת  וְכֹל דִי יִפֹּל מִנְבִלְתְּהוֹן עֲלוֹהִי יְהֵי מְסָאָב תַּנוּר וְכִירַיִם יִתָּרְעוּן מְסַאֲבִין אִנוּן וּמְסַאֲבִין יְהוֹן לְכוֹן:

{לו} אַךְ מַעְיָן וּבוֹר מִקְוֵה מַיִם יִֽהְיֶה טָהוֹר וְנֹגֵעַ בְּנִבְלָתָם יִטְמָֽא:  מ  אַךְ מַעְיָן וּבוֹר מִקְוֵה מַיִם יִֽהְיֶה טָהוֹר וְנֹגֵעַ בְּנִבְלָתָם יִטְמָֽא:  ת  בְּרַם מַעְיָן וְגוֹב בֵּית כְּנִישׁוּת מַיָא יְהֵי דְכֵי וּדְיִקְרַב בִּנְבִלְתְּהוֹן יְהֵי מְסָאָב:

{לז} וְכִי יִפֹּל מִנִּבְלָתָם עַל כָּל זֶרַע זֵרוּעַ אֲשֶׁר יִזָּרֵעַ טָהוֹר הֽוּא:  מ  וְכִי יִפֹּל מִנִּבְלָתָם עַל כָּל זֶרַע זֵרוּעַ אֲשֶׁר יִזָּרֵעַ טָהוֹר הֽוּא:  ת  וַאֲרֵי יִפֵּל מִנְבִלְתְּהוֹן עַל כָּל בַּר זְרַע זֵרוּעַ דִי יִזְדְרָע דְכֵי הוּא:

{לח} וְכִי יֻתַּן מַיִם עַל זֶרַע וְנָפַל מִנִּבְלָתָם עָלָיו טָמֵא הוּא לָכֶֽם:  מ  וְכִי יֻתַּן מַיִם עַל זֶרַע וְנָפַל מִנִּבְלָתָם עָלָיו טָמֵא הוּא לָכֶֽם:  ת  וַאֲרֵי יִתְיַהֲבוּן מַיָא עַל בַּר זַרְעָא וְיִפֵּל מִנְבִלְתְּהוֹן עֲלוֹהִי מְסָאָב הוּא לְכוֹן: (ס)

{לט} וְכִי יָמוּת מִן הַבְּהֵמָה אֲשֶׁר הִיא לָכֶם לְאָכְלָה הַנֹּגֵעַ בְּנִבְלָתָהּ יִטְמָא עַד הָעָֽרֶב:  מ  וְכִי יָמוּת מִן הַבְּהֵמָה אֲשֶׁר הִיא לָכֶם לְאָכְלָה הַנֹּגֵעַ בְּנִבְלָתָהּ יִטְמָא עַד הָעָֽרֶב:  ת  וַאֲרֵי יְמוּת מִן בְּעִירָא דִי הִיא לְכוֹן לְמֵיכָל דְיִקְרַב בִּנְבִלְתַּהּ יְהֵי מְסָאָב עַד רַמְשָׁא:

{מ} וְהָֽאֹכֵל מִנִּבְלָתָהּ יְכַבֵּס בְּגָדָיו וְטָמֵא עַד הָעָרֶב וְהַנֹּשֵׂא אֶת נִבְלָתָהּ יְכַבֵּס בְּגָדָיו וְטָמֵא עַד הָעָֽרֶב:  מ  וְהָֽאֹכֵל מִנִּבְלָתָהּ יְכַבֵּס בְּגָדָיו וְטָמֵא עַד הָעָרֶב וְהַנֹּשֵׂא אֶת נִבְלָתָהּ יְכַבֵּס בְּגָדָיו וְטָמֵא עַד הָעָֽרֶב:  ת  וּדְיֵכוּל מִנְבִלְתַּהּ יְצַבַּע לְבוּשׁוֹהִי וִיהֵי מְסָאָב עַד רַמְשָׁא וּדְיִטוֹל יָת נְבִלְתַּהּ יְצַבַּע לְבוּשׁוֹהִי וִיהֵי מְסָאָב עַד רַמְשָׁא:

{מא} וְכָל הַשֶּׁרֶץ הַשֹּׁרֵץ עַל הָאָרֶץ שֶׁקֶץ הוּא לֹא יֵֽאָכֵֽל:  מ  וְכָל הַשֶּׁרֶץ הַשֹּׁרֵץ עַל הָאָרֶץ שֶׁקֶץ הוּא לֹא יֵֽאָכֵֽל:  ת  וְכָל רִחֲשָׁא דְרָחֵשׁ עַל אַרְעָא שִׁקְצָא הוּא לָא יִתְאָכֵל:

{מב} כֹּל הוֹלֵךְ עַל גָּחוֹן וְכֹל הוֹלֵךְ עַל אַרְבַּע עַד כָּל מַרְבֵּה רַגְלַיִם לְכָל הַשֶּׁרֶץ הַשֹּׁרֵץ עַל הָאָרֶץ לֹא תֹֽאכְלוּם כִּי שֶׁקֶץ הֵֽם:  מ  כֹּל הוֹלֵךְ עַל גָּחוֹן וְכֹל הוֹלֵךְ עַל אַרְבַּע עַד כָּל מַרְבֵּה רַגְלַיִם לְכָל הַשֶּׁרֶץ הַשֹּׁרֵץ עַל הָאָרֶץ לֹא תֹֽאכְלוּם כִּי שֶׁקֶץ הֵֽם:  ת  כֹּל דִמְהַלֵךְ עַל מְעוֹהִי וְכֹל דִמְהַלֵךְ עַל אַרְבַּע עַד כָּל סְגִיאוּת רַגְלָאִין לְכָל רִחֲשָׁא דְרָחֵשׁ עַל אַרְעָא לָא תֵיכְלֻנוּן אֲרֵי שִׁקְצָא אִנוּן:

{מג} אַל תְּשַׁקְּצוּ אֶת נַפְשֹׁתֵיכֶם בְּכָל הַשֶּׁרֶץ הַשֹּׁרֵץ וְלֹא תִֽטַּמְּאוּ בָּהֶם וְנִטְמֵתֶם בָּֽם:  מ  אַל תְּשַׁקְּצוּ אֶת נַפְשֹׁתֵיכֶם בְּכָל הַשֶּׁרֶץ הַשֹּׁרֵץ וְלֹא תִֽטַּמְּאוּ בָּהֶם וְנִטְמֵתֶם בָּֽם:  ת  לָא תְשַׁקְצוּ יָת נַפְשָׁתֵיכוֹן בְּכָל רִחֲשָׁא דְרָחֵשׁ וְלָא תִסְתַּאֲבוּן בְּהוֹן וְתִסְתַּאֲבוּן פּוֹן בְּהוֹן:

{מד} כִּי אֲנִי יְהוָֹה אֱלֹֽהֵיכֶם וְהִֽתְקַדִּשְׁתֶּם וִֽהְיִיתֶם קְדֹשִׁים כִּי קָדוֹשׁ אָנִי וְלֹא תְטַמְּאוּ אֶת נַפְשֹׁתֵיכֶם בְּכָל הַשֶּׁרֶץ הָרֹמֵשׂ עַל הָאָֽרֶץ:  מ  כִּי אֲנִי יְהוָֹה אֱלֹֽהֵיכֶם וְהִֽתְקַדִּשְׁתֶּם וִֽהְיִיתֶם קְדֹשִׁים כִּי קָדוֹשׁ אָנִי וְלֹא תְטַמְּאוּ אֶת נַפְשֹׁתֵיכֶם בְּכָל הַשֶּׁרֶץ הָרֹמֵשׂ עַל הָאָֽרֶץ:  ת  אֲרֵי אֲנָא יְיָ אֶלָהָכוֹן וְתִתְקַדְשׁוּן וּתְהוֹן קַדִישִׁין אֲרֵי קַדִישׁ אֲנָא וְלָא תְסַאֲבוּן יָת נַפְשָׁתֵיכוֹן בְּכָל רִחֲשָׁא דְרָחֵשׁ עַל אַרְעָא:

{מה}  מפטיר  כִּי אֲנִי יְהֹוָה הַמַּֽעֲלֶה אֶתְכֶם מֵאֶרֶץ מִצְרַיִם לִֽהְיֹת לָכֶם לֵֽאלֹהִים וִֽהְיִיתֶם קְדֹשִׁים כִּי קָדוֹשׁ אָֽנִי:  מ  כִּי אֲנִי יְהֹוָה הַמַּֽעֲלֶה אֶתְכֶם מֵאֶרֶץ מִצְרַיִם לִֽהְיֹת לָכֶם לֵֽאלֹהִים וִֽהְיִיתֶם קְדֹשִׁים כִּי קָדוֹשׁ אָֽנִי:  ת  אֲרֵי אֲנָא יְיָ דְאַסֵק יָתְכוֹן מֵאַרְעָא דְמִצְרַיִם לְמֶהֱוֵי לְכוֹן לֶאֱלָהּ וּתְהוֹן קַדִישִׁין אֲרֵי קַדִישׁ אֲנָא:

{מו} זֹאת תּוֹרַת הַבְּהֵמָה וְהָעוֹף וְכֹל נֶפֶשׁ הַֽחַיָּה הָֽרֹמֶשֶׂת בַּמָּיִם וּלְכָל נֶפֶשׁ הַשֹּׁרֶצֶת עַל הָאָֽרֶץ:  מ  זֹאת תּוֹרַת הַבְּהֵמָה וְהָעוֹף וְכֹל נֶפֶשׁ הַֽחַיָּה הָֽרֹמֶשֶׂת בַּמָּיִם וּלְכָל נֶפֶשׁ הַשֹּׁרֶצֶת עַל הָאָֽרֶץ:  ת  דָא אוֹרַיְתָא דִבְעִירָא וּדְעוֹפָא וּדְכָל נַפְשָׁתָא חַיְתָא דִרְחֵשָׁא בְּמַיָא וּלְכָל נַפְשָׁא דִרְחֵשָׁא עַל אַרְעָא:

{מז} לְהַבְדִּיל בֵּין הַטָּמֵא וּבֵין הַטָּהֹר וּבֵין הַֽחַיָּה הַֽנֶּֽאֱכֶלֶת וּבֵין הַֽחַיָּה אֲשֶׁר לֹא תֵֽאָכֵֽל:  מ  לְהַבְדִּיל בֵּין הַטָּמֵא וּבֵין הַטָּהֹר וּבֵין הַֽחַיָּה הַֽנֶּֽאֱכֶלֶת וּבֵין הַֽחַיָּה אֲשֶׁר לֹא תֵֽאָכֵֽל:  ת  לְאַפְרָשָׁא בֵּין מְסָאָבָא וּבֵין דַכְיָא וּבֵין חַיְתָא דְמִתְאַכְלָא וּבֵין חַיְתָא דִי לָא מִתְאַכְלָא: (פפפ)
