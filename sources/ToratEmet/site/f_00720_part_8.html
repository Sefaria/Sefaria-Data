ח-א


[ישעיה פרק-א-יד]  חָדְשֵׁיכֶם וּמוֹעֲדֵיכֶם שָׂנְאָה נַפְשִׁי הָיוּ עָלַי לָטֹרַח נִלְאֵיתִי נְשֹׂא:

[ישעיה פרק-ב-כב]  חִדְלוּ לָכֶם מִן הָאָדָם אֲשֶׁר נְשָׁמָה בְּאַפּוֹ כִּי בַמֶּה נֶחְשָׁב הוּא:

[ירמיה פרק-מו-יח]  חַי אָנִי נְאֻם הַמֶּלֶךְ יְהוָה צְבָאוֹת שְׁמוֹ כִּי כְּתָבוֹר בֶּהָרִים וּכְכַרְמֶל בַּיָּם יָבוֹא:

[איוב פרק-ז-כ]  חָטָאתִי מָה אֶפְעַל לָךְ נֹצֵר הָאָדָם לָמָה שַׂמְתַּנִי לְמִפְגָּע לָךְ וָאֶהְיֶה עָלַי לְמַשָּׂא:

[דניאל פרק-ז-יא]  חָזֵה הֲוֵית בֵּאדַיִן מִן קָל מִלַּיָּא רַבְרְבָתָא דִּי קַרְנָא מְמַלֱּלָה חָזֵה הֲוֵית עַד דִּי קְטִילַת חֵיוְתָא וְהוּבַד גִּשְׁמַהּ וִיהִיבַת לִיקֵדַת אֶשָּׁא:


ח-ב


[בראשית פרק-מט-יב]  חַכְלִילִי עֵינַיִם מִיָּיִן וּלְבֶן שִׁנַּיִם מֵחָלָב:

[ישעיה פרק-ז-טו]  חֶמְאָה וּדְבַשׁ יֹאכֵל לְדַעְתּוֹ מָאוֹס בָּרָע וּבָחוֹר בַּטּוֹב:

[משלי פרק-יג-כא]  חַטָּאִים תְּרַדֵּף רָעָה וְאֶת צַדִּיקִים יְשַׁלֶּם טוֹב:

[משלי פרק-כט-כה]  חֶרְדַּת אָדָם יִתֵּן מוֹקֵשׁ וּבוֹטֵחַ בַּיהוָה יְשֻׂגָּב:


ח-ד


[ישעיה פרק-כז-ד]  חֵמָה אֵין לִי מִי יִתְּנֵנִי שָׁמִיר שַׁיִת בַּמִּלְחָמָה אֶפְשְׂעָה בָהּ אֲצִיתֶנָּה יָּחַד:

[תהילים פרק-כא-ה]  חַיִּים שָׁאַל מִמְּךָ נָתַתָּה לּוֹ אֹרֶךְ יָמִים עוֹלָם וָעֶד:

[תהילים פרק-קמה-ח]  חַנּוּן וְרַחוּם יְהוָה אֶרֶךְ אַפַּיִם וּגְדָל חָסֶד:

[משלי פרק-כט-כד]  חוֹלֵק עִם גַּנָּב שׂוֹנֵא נַפְשׁוֹ אָלָה יִשְׁמַע וְלֹא יַגִּיד:


ח-ה


[בראשית פרק-כה-טו]  חֲדַד וְתֵימָא יְטוּר נָפִישׁ וָקֵדְמָה:

[שמות פרק-כו-ג]  חֲמֵשׁ הַיְרִיעֹת תִּהְיֶיןָ חֹבְרֹת אִשָּׁה אֶל אֲחֹתָהּ וְחָמֵשׁ יְרִיעֹת חֹבְרֹת אִשָּׁה אֶל אֲחֹתָהּ:

[שמות פרק-כו-ה]  חֲמִשִּׁים לֻלָאֹת תַּעֲשֶׂה בַּיְרִיעָה הָאֶחָת וַחֲמִשִּׁים לֻלָאֹת תַּעֲשֶׂה בִּקְצֵה הַיְרִיעָה אֲשֶׁר בַּמַּחְבֶּרֶת הַשֵּׁנִית מַקְבִּילֹת הַלֻּלָאֹת אִשָּׁה אֶל אֲחֹתָהּ:

[שופטים פרק-ה-כט]  חַכְמוֹת שָׂרוֹתֶיהָ תַּעֲנֶינָּה אַף הִיא תָּשִׁיב אֲמָרֶיהָ לָהּ:

[מלכים א פרק-יח-י]  חַי יְהוָה אֱלֹהֶיךָ אִם יֶשׁ גּוֹי וּמַמְלָכָה אֲשֶׁר לֹא שָׁלַח אֲדֹנִי שָׁם לְבַקֶּשְׁךָ וְאָמְרוּ אָיִן וְהִשְׁבִּיעַ אֶת הַמַּמְלָכָה וְאֶת הַגּוֹי כִּי לֹא יִמְצָאֶכָּה:

[ישעיה פרק-א-א]  חֲזוֹן יְשַׁעְיָהוּ בֶן אָמוֹץ אֲשֶׁר חָזָה עַל יְהוּדָה וִירוּשָׁלִָם בִּימֵי עֻזִּיָּהוּ יוֹתָם אָחָז יְחִזְקִיָּהוּ מַלְכֵי יְהוּדָה:

[ירמיה פרק-נ-לה]  חֶרֶב עַל כַּשְׂדִּים נְאֻם יְהוָה וְאֶל יֹשְׁבֵי בָבֶל וְאֶל שָׂרֶיהָ וְאֶל חֲכָמֶיהָ:

[ירמיה פרק-נא-ל]  חָדְלוּ גִבּוֹרֵי בָבֶל לְהִלָּחֵם יָשְׁבוּ בַּמְּצָדוֹת נָשְׁתָה גְבוּרָתָם הָיוּ לְנָשִׁים הִצִּיתוּ מִשְׁכְּנֹתֶיהָ נִשְׁבְּרוּ בְרִיחֶיהָ:

[יחזקאל פרק-יא-ח]  חֶרֶב יְרֵאתֶם וְחֶרֶב אָבִיא עֲלֵיכֶם נְאֻם אֲדֹנָי יְהוִה:

[עובדיה פרק-א-א]  חֲזוֹן עֹבַדְיָה כֹּה אָמַר אֲדֹנָי יְהוִה לֶאֱדוֹם שְׁמוּעָה שָׁמַעְנוּ מֵאֵת יְהוָה וְצִיר בַּגּוֹיִם שֻׁלָּח קוּמוּ וְנָקוּמָה עָלֶיהָ לַמִּלְחָמָה:

[תהילים פרק-כה-ז]  חַטֹּאות נְעוּרַי וּפְשָׁעַי אַל תִּזְכֹּר כְּחַסְדְּךָ זְכָר לִי אַתָּה לְמַעַן טוּבְךָ יְהוָה:

[תהילים פרק-לא-כה]  חִזְקוּ וְיַאֲמֵץ לְבַבְכֶם כָּל הַמְיַחֲלִים לַיהוָה:

[תהילים פרק-לב-ה]  חַטָּאתִי אוֹדִיעֲךָ וַעֲוֹנִי לֹא כִסִּיתִי אָמַרְתִּי אוֹדֶה עֲלֵי פְשָׁעַי לַיהוָה וְאַתָּה נָשָׂאתָ עֲוֹן חַטָּאתִי סֶלָה:

[תהילים פרק-לז-טו]  חַרְבָּם תָּבוֹא בְלִבָּם וְקַשְּׁתוֹתָם תִּשָּׁבַרְנָה:

[משלי פרק-א-כ]  חָכְמוֹת בַּחוּץ תָּרֹנָּה בָּרְחֹבוֹת תִּתֵּן קוֹלָהּ:

[משלי פרק-ט-א]  חָכְמוֹת בָּנְתָה בֵיתָהּ חָצְבָה עַמּוּדֶיהָ שִׁבְעָה:

[משלי פרק-י-יד]  חֲכָמִים יִצְפְּנוּ דָעַת וּפִי אֱוִיל מְחִתָּה קְרֹבָה:

[משלי פרק-יד-ח]  חָכְמַת עָרוּם הָבִין דַּרְכּוֹ וְאִוֶּלֶת כְּסִילִים מִרְמָה:

[משלי פרק-יד-ל]  חַיֵּי בְשָׂרִים לֵב מַרְפֵּא וּרְקַב עֲצָמוֹת קִנְאָה:

[משלי פרק-טז-יד]  חֲמַת מֶלֶךְ מַלְאֲכֵי מָוֶת וְאִישׁ חָכָם יְכַפְּרֶנָּה:

[משלי פרק-יז-כז]  חוֹשֵׂךְ אֲמָרָיו יוֹדֵעַ דָּעַת (וקר) יְקַר רוּחַ אִישׁ תְּבוּנָה:

[משלי פרק-כב-ו]  חֲנֹךְ לַנַּעַר עַל פִּי דַרְכּוֹ גַּם כִּי יַזְקִין לֹא יָסוּר מִמֶּנָּה:

[משלי פרק-לא-יז]  חָגְרָה בְעוֹז מָתְנֶיהָ וַתְּאַמֵּץ זְרֹעוֹתֶיהָ:

[איוב פרק-כד-ג]  חֲמוֹר יְתוֹמִים יִנְהָגוּ יַחְבְּלוּ שׁוֹר אַלְמָנָה:

[נחמיה פרק-י-ו]  חָרִם מְרֵמוֹת עֹבַדְיָה:

[דברי הימים א פרק-יט-יג]  חֲזַק וְנִתְחַזְּקָה בְּעַד עַמֵּנוּ וּבְעַד עָרֵי אֱלֹהֵינוּ וַיהוָה הַטּוֹב בְּעֵינָיו יַעֲשֶׂה:


ח-ו


[ויקרא פרק-ג-יז]  חֻקַּת עוֹלָם לְדֹרֹתֵיכֶם בְּכֹל מוֹשְׁבֹתֵיכֶם כָּל חֵלֶב וְכָל דָּם לֹא תֹאכֵלוּ:

[יהושע פרק-כב-כט]  חָלִילָה לָּנוּ מִמֶּנּוּ לִמְרֹד בַּיהוָה וְלָשׁוּב הַיּוֹם מֵאַחֲרֵי יְהוָה לִבְנוֹת מִזְבֵּחַ לְעֹלָה לְמִנְחָה וּלְזָבַח מִלְּבַד מִזְבַּח יְהוָה אֱלֹהֵינוּ אֲשֶׁר לִפְנֵי מִשְׁכָּנוֹ:

[שמואל א פרק-כו-יא]  חָלִילָה לִּי מֵיהוָה מִשְּׁלֹחַ יָדִי בִּמְשִׁיחַ יְהוָה וְעַתָּה קַח נָא אֶת הַחֲנִית אֲשֶׁר (מראשתו) מְרַאֲשֹׁתָיו וְאֶת צַפַּחַת הַמַּיִם וְנֵלֲכָה לָּנוּ:

[שמואל ב פרק-י-יב]  חֲזַק וְנִתְחַזַּק בְּעַד עַמֵּנוּ וּבְעַד עָרֵי אֱלֹהֵינוּ וַיהוָה יַעֲשֶׂה הַטּוֹב בְּעֵינָיו:

[ישעיה פרק-לג-כ]  חֲזֵה צִיּוֹן קִרְיַת מוֹעֲדֵנוּ עֵינֶיךָ תִרְאֶינָה יְרוּשָׁלִַם נָוֶה שַׁאֲנָן אֹהֶל בַּל יִצְעָן בַּל יִסַּע יְתֵדֹתָיו לָנֶצַח וְכָל חֲבָלָיו בַּל יִנָּתֵקוּ:

[ישעיה פרק-לה-ג]  חַזְּקוּ יָדַיִם רָפוֹת וּבִרְכַּיִם כֹּשְׁלוֹת אַמֵּצוּ:

[ישעיה פרק-נב-י]  חָשַׂף יְהוָה אֶת זְרוֹעַ קָדְשׁוֹ לְעֵינֵי כָּל הַגּוֹיִם וְרָאוּ כָּל אַפְסֵי אָרֶץ אֵת יְשׁוּעַת אֱלֹהֵינוּ:

[ישעיה פרק-סג-ז]  חַסְדֵי יְהוָה אַזְכִּיר תְּהִלֹּת יְהוָה כְּעַל כֹּל אֲשֶׁר גְּמָלָנוּ יְהוָה וְרַב טוּב לְבֵית יִשְׂרָאֵל אֲשֶׁר גְּמָלָם כְּרַחֲמָיו וּכְרֹב חֲסָדָיו:

[ירמיה פרק-ט-ז]  חֵץ (שוחט) שָׁחוּט לְשׁוֹנָם מִרְמָה דִבֵּר בְּפִיו שָׁלוֹם אֶת רֵעֵהוּ יְדַבֵּר וּבְקִרְבּוֹ יָשִׂים אָרְבּוֹ:

[ירמיה פרק-נ-לו]  חֶרֶב אֶל הַבַּדִּים וְנֹאָלוּ חֶרֶב אֶל גִּבּוֹרֶיהָ וָחָתּוּ:

[ירמיה פרק-נ-לז]  חֶרֶב אֶל סוּסָיו וְאֶל רִכְבּוֹ וְאֶל כָּל הָעֶרֶב אֲשֶׁר בְּתוֹכָהּ וְהָיוּ לְנָשִׁים חֶרֶב אֶל אוֹצְרֹתֶיהָ וּבֻזָּזוּ:

[ירמיה פרק-נ-לח]  חֹרֶב אֶל מֵימֶיהָ וְיָבֵשׁוּ כִּי אֶרֶץ פְּסִלִים הִיא וּבָאֵימִים יִתְהֹלָלוּ:

[יחזקאל פרק-א-ט]  חֹבְרֹת אִשָּׁה אֶל אֲחוֹתָהּ כַּנְפֵיהֶם לֹא יִסַּבּוּ בְלֶכְתָּן אִישׁ אֶל עֵבֶר פָּנָיו יֵלֵכוּ:

[יחזקאל פרק-לד-ח]  חַי אָנִי נְאֻם אֲדֹנָי יְהוִה אִם לֹא יַעַן הֱיוֹת צֹאנִי לָבַז וַתִּהְיֶינָה צֹאנִי לְאָכְלָה לְכָל חַיַּת הַשָּׂדֶה מֵאֵין רֹעֶה וְלֹא דָרְשׁוּ רֹעַי אֶת צֹאנִי וַיִּרְעוּ הָרֹעִים אוֹתָם וְאֶת צֹאנִי לֹא רָעוּ:

[הושע פרק-ד-ח]  חַטַּאת עַמִּי יֹאכֵלוּ וְאֶל עֲוֹנָם יִשְׂאוּ נַפְשׁוֹ:

[הושע פרק-ד-יז]  חֲבוּר עֲצַבִּים אֶפְרָיִם הַנַּח לוֹ:

[תהילים פרק-נח-ה]  חֲמַת לָמוֹ כִּדְמוּת חֲמַת נָחָשׁ כְּמוֹ פֶתֶן חֵרֵשׁ יַאְטֵם אָזְנוֹ:

[תהילים פרק-נט-יג]  חַטַּאת פִּימוֹ דְּבַר שְׂפָתֵימוֹ וְיִלָּכְדוּ בִגְאוֹנָם וּמֵאָלָה וּמִכַּחַשׁ יְסַפֵּרוּ:

[תהילים פרק-פה-יא]  חֶסֶד וֶאֱמֶת נִפְגָּשׁוּ צֶדֶק וְשָׁלוֹם נָשָׁקוּ:

[תהילים פרק-קו-ו]  חָטָאנוּ עִם אֲבוֹתֵינוּ הֶעֱוִינוּ הִרְשָׁעְנוּ:

[משלי פרק-יד-א]  חַכְמוֹת נָשִׁים בָּנְתָה בֵיתָהּ וְאִוֶּלֶת בְּיָדֶיהָ תֶהֶרְסֶנּוּ:

[משלי פרק-כ-כח]  חֶסֶד וֶאֱמֶת יִצְּרוּ מֶלֶךְ וְסָעַד בַּחֶסֶד כִּסְאוֹ:

[משלי פרק-כח-יא]  חָכָם בְּעֵינָיו אִישׁ עָשִׁיר וְדַל מֵבִין יַחְקְרֶנּוּ:

[משלי פרק-כט-כ]  חָזִיתָ אִישׁ אָץ בִּדְבָרָיו תִּקְוָה לִכְסִיל מִמֶּנּוּ:

[איכה פרק-ב-ח]  חָשַׁב יְהוָה לְהַשְׁחִית חוֹמַת בַּת צִיּוֹן נָטָה קָו לֹא הֵשִׁיב יָדוֹ מִבַּלֵּעַ וַיַּאֲבֶל חֵל וְחוֹמָה יַחְדָּו אֻמְלָלוּ:

[איכה פרק-ג-כב]  חַסְדֵי יְהוָה כִּי לֹא תָמְנוּ כִּי לֹא כָלוּ רַחֲמָיו:

[איכה פרק-ג-כד]  חֶלְקִי יְהוָה אָמְרָה נַפְשִׁי עַל כֵּן אוֹחִיל לוֹ:

[דברי הימים ב פרק-לב-ז]  חִזְקוּ וְאִמְצוּ אַל תִּירְאוּ וְאַל תֵּחַתּוּ מִפְּנֵי מֶלֶךְ אַשּׁוּר וּמִלִּפְנֵי כָּל הֶהָמוֹן אֲשֶׁר עִמּוֹ כִּי עִמָּנוּ רַב מֵעִמּוֹ:


ח-ז


[תהילים פרק-קכג-ג]  חָנֵּנוּ יְהוָה חָנֵּנוּ כִּי רַב שָׂבַעְנוּ בוּז:


ח-ח


[משלי פרק-יד-טז]  חָכָם יָרֵא וְסָר מֵרָע וּכְסִיל מִתְעַבֵּר וּבוֹטֵחַ:


ח-ט


[בראשית פרק-יח-כה]  חָלִלָה לְּךָ מֵעֲשֹׂת כַּדָּבָר הַזֶּה לְהָמִית צַדִּיק עִם רָשָׁע וְהָיָה כַצַּדִּיק כָּרָשָׁע חָלִלָה לָּךְ הֲשֹׁפֵט כָּל הָאָרֶץ לֹא יַעֲשֶׂה מִשְׁפָּט:

[משלי פרק-י-ח]  חֲכַם לֵב יִקַּח מִצְוֹת וֶאֱוִיל שְׂפָתַיִם יִלָּבֵט:

[נחמיה פרק-יא-לד]  חָדִיד צְבֹעִים נְבַלָּט:

[דברי הימים א פרק-טז-ל]  חִילוּ מִלְּפָנָיו כָּל הָאָרֶץ אַף תִּכּוֹן תֵּבֵל בַּל תִּמּוֹט:


ח-י


[שמואל א פרק-כא-טז]  חֲסַר מְשֻׁגָּעִים אָנִי כִּי הֲבֵאתֶם אֶת זֶה לְהִשְׁתַּגֵּעַ עָלָי הֲזֶה יָבוֹא אֶל בֵּיתִי:

[שמואל ב פרק-כב-מז]  חַי יְהוָה וּבָרוּךְ צוּרִי וְיָרֻם אֱלֹהֵי צוּר יִשְׁעִי:

[שמואל ב פרק-כג-כו]  חֶלֶץ הַפַּלְטִי עִירָא בֶן עִקֵּשׁ הַתְּקוֹעִי:

[שמואל ב פרק-כג-לה]  (חצרו) חֶצְרַי הַכַּרְמְלִי פַּעֲרַי הָאַרְבִּי:

[ישעיה פרק-כא-ב]  חָזוּת קָשָׁה הֻגַּד לִי הַבּוֹגֵד בּוֹגֵד וְהַשּׁוֹדֵד שׁוֹדֵד עֲלִי עֵילָם צוּרִי מָדַי כָּל אַנְחָתָה הִשְׁבַּתִּי:

[תהילים פרק-ו-ג]  חָנֵּנִי יְהוָה כִּי אֻמְלַל אָנִי רְפָאֵנִי יְהוָה כִּי נִבְהֲלוּ עֲצָמָי:

[תהילים פרק-טז-ו]  חֲבָלִים נָפְלוּ לִי בַּנְּעִמִים אַף נַחֲלָת שָׁפְרָה עָלָי:

[תהילים פרק-יח-מז]  חַי יְהוָה וּבָרוּךְ צוּרִי וְיָרוּם אֱלוֹהֵי יִשְׁעִי:

[תהילים פרק-לא-י]  חָנֵּנִי יְהוָה כִּי צַר לִי עָשְׁשָׁה בְכַעַס עֵינִי נַפְשִׁי וּבִטְנִי:

[תהילים פרק-לח-כג]  חוּשָׁה לְעֶזְרָתִי אֲדֹנָי תְּשׁוּעָתִי:

[תהילים פרק-לט-ד]  חַם לִבִּי בְּקִרְבִּי בַּהֲגִיגִי תִבְעַר אֵשׁ דִּבַּרְתִּי בִּלְשׁוֹנִי:

[תהילים פרק-נא-ג]  חָנֵּנִי אֱלֹהִים כְּחַסְדֶּךָ כְּרֹב רַחֲמֶיךָ מְחֵה פְשָׁעָי:

[תהילים פרק-נו-ב]  חָנֵּנִי אֱלֹהִים כִּי שְׁאָפַנִי אֱנוֹשׁ כָּל הַיּוֹם לֹחֵם יִלְחָצֵנִי:

[תהילים פרק-סט-כא]  חֶרְפָּה שָׁבְרָה לִבִּי וָאָנוּשָׁה וָאֲקַוֶּה לָנוּד וָאַיִן וְלַמְנַחֲמִים וְלֹא מָצָאתִי:

[תהילים פרק-פט-ב]  חַסְדֵי יְהוָה עוֹלָם אָשִׁירָה לְדֹר וָדֹר אוֹדִיעַ אֱמוּנָתְךָ בְּפִי:

[תהילים פרק-קיט-סא]  חֶבְלֵי רְשָׁעִים עִוְּדֻנִי תּוֹרָתְךָ לֹא שָׁכָחְתִּי:

[תהילים פרק-קיט-סד]  חַסְדְּךָ יְהוָה מָלְאָה הָאָרֶץ חֻקֶּיךָ לַמְּדֵנִי:

[תהילים פרק-קלט-כג]  חָקְרֵנִי אֵל וְדַע לְבָבִי בְּחָנֵנִי וְדַע שַׂרְעַפָּי:

[תהילים פרק-קמ-ב]  חַלְּצֵנִי יְהוָה מֵאָדָם רָע מֵאִישׁ חֲמָסִים תִּנְצְרֵנִי:

[תהילים פרק-קמד-ב]  חַסְדִּי וּמְצוּדָתִי מִשְׂגַּבִּי וּמְפַלְטִי לִי מָגִנִּי וּבוֹ חָסִיתִי הָרוֹדֵד עַמִּי תַחְתָּי:

[איוב פרק-י-יב]  חַיִּים וָחֶסֶד עָשִׂיתָ עִמָּדִי וּפְקֻדָּתְךָ שָׁמְרָה רוּחִי:

[איוב פרק-יד-יז]  חָתֻם בִּצְרוֹר פִּשְׁעִי וַתִּטְפֹּל עַל עֲוֹנִי:

[איוב פרק-יט-יד]  חָדְלוּ קְרוֹבָי וּמְיֻדָּעַי שְׁכֵחוּנִי:

[איוב פרק-יט-כא]  חָנֻּנִי חָנֻּנִי אַתֶּם רֵעָי כִּי יַד אֱלוֹהַּ נָגְעָה בִּי:

[איוב פרק-כז-ב]  חַי אֵל הֵסִיר מִשְׁפָּטִי וְשַׁדַּי הֵמַר נַפְשִׁי:

[איוב פרק-כז-ה]  חָלִילָה לִּי אִם אַצְדִּיק אֶתְכֶם עַד אֶגְוָע לֹא אָסִיר תֻּמָּתִי מִמֶּנִּי:

[דניאל פרק-ד-ב]  חֵלֶם חֲזֵית וִידַחֲלִנַּנִי וְהַרְהֹרִין עַל מִשְׁכְּבִי וְחֶזְוֵי רֵאשִׁי יְבַהֲלֻנַּנִי:

[דניאל פרק-ז-יג]  חָזֵה הֲוֵית בְּחֶזְוֵי לֵילְיָא וַאֲרוּ עִם עֲנָנֵי שְׁמַיָּא כְּבַר אֱנָשׁ אָתֵה הֲוָה וְעַד עַתִּיק יוֹמַיָּא מְטָה וּקְדָמוֹהִי הַקְרְבוּהִי:

[נחמיה פרק-י-כ]  חָרִיף עֲנָתוֹת (נובי) נֵיבָי:

[דברי הימים א פרק-יא-לב]  חוּרַי מִנַּחֲלֵי גָעַשׁ אֲבִיאֵל הָעַרְבָתִי:

[דברי הימים א פרק-יא-לו]  חֵפֶר הַמְּכֵרָתִי אֲחִיָּה הַפְּלֹנִי:

[דברי הימים א פרק-יא-לז]  חֶצְרוֹ הַכַּרְמְלִי נַעֲרַי בֶּן אֶזְבָּי:

[דברי הימים א פרק-יא-מג]  חָנָן בֶּן מַעֲכָה וְיוֹשָׁפָט הַמִּתְנִי:


ח-כ


[דברים פרק-טז-יג]  חַג הַסֻּכֹּת תַּעֲשֶׂה לְךָ שִׁבְעַת יָמִים בְּאָסְפְּךָ מִגָּרְנְךָ וּמִיִּקְבֶךָ:

[דברים פרק-לא-ו]  חִזְקוּ וְאִמְצוּ אַל תִּירְאוּ וְאַל תַּעַרְצוּ מִפְּנֵיהֶם כִּי יְהוָה אֱלֹהֶיךָ הוּא הַהֹלֵךְ עִמָּךְ לֹא יַרְפְּךָ וְלֹא יַעַזְבֶךָּ:

[ישעיה פרק-לח-יט]  חַי חַי הוּא יוֹדֶךָ כָּמוֹנִי הַיּוֹם אָב לְבָנִים יוֹדִיעַ אֶל אֲמִתֶּךָ:

[ירמיה פרק-טו-יג]  חֵילְךָ וְאוֹצְרוֹתֶיךָ לָבַז אֶתֵּן לֹא בִמְחִיר וּבְכָל חַטֹּאותֶיךָ וּבְכָל גְּבוּלֶיךָ:

[ירמיה פרק-כב-כד]  חַי אָנִי נְאֻם יְהוָה כִּי אִם יִהְיֶה כָּנְיָהוּ בֶן יְהוֹיָקִים מֶלֶךְ יְהוּדָה חוֹתָם עַל יַד יְמִינִי כִּי מִשָּׁם אֶתְּקֶנְךָּ:

[יחזקאל פרק-טז-מח]  חַי אָנִי נְאֻם אֲדֹנָי יְהוִה אִם עָשְׂתָה סְדֹם אֲחוֹתֵךְ הִיא וּבְנוֹתֶיהָ כַּאֲשֶׁר עָשִׂית אַתְּ וּבְנוֹתָיִךְ:

[יחזקאל פרק-כז-כג]  חָרָן וְכַנֵּה וָעֶדֶן רֹכְלֵי שְׁבָא אַשּׁוּר כִּלְמַד רֹכַלְתֵּךְ:

[הושע פרק-י-יג]  חֲרַשְׁתֶּם רֶשַׁע עַוְלָתָה קְצַרְתֶּם אֲכַלְתֶּם פְּרִי כָחַשׁ כִּי בָטַחְתָּ בְדַרְכְּךָ בְּרֹב גִּבּוֹרֶיךָ:

[יואל פרק-א-יג]  חִגְרוּ וְסִפְדוּ הַכֹּהֲנִים הֵילִילוּ מְשָׁרְתֵי מִזְבֵּחַ בֹּאוּ לִינוּ בַשַּׂקִּים מְשָׁרְתֵי אֱלֹהָי כִּי נִמְנַע מִבֵּית אֱלֹהֵיכֶם מִנְחָה וָנָסֶךְ:

[מיכה פרק-ד-י]  חוּלִי וָגֹחִי בַּת צִיּוֹן כַּיּוֹלֵדָה כִּי עַתָּה תֵצְאִי מִקִּרְיָה וְשָׁכַנְתְּ בַּשָּׂדֶה וּבָאת עַד בָּבֶל שָׁם תִּנָּצֵלִי שָׁם יִגְאָלֵךְ יְהוָה מִכַּף אֹיְבָיִךְ:

[מלאכי פרק-ג-יג]  חָזְקוּ עָלַי דִּבְרֵיכֶם אָמַר יְהוָה וַאֲמַרְתֶּם מַה נִּדְבַּרְנוּ עָלֶיךָ:

[תהילים פרק-לז-יד]  חֶרֶב פָּתְחוּ רְשָׁעִים וְדָרְכוּ קַשְׁתָּם לְהַפִּיל עָנִי וְאֶבְיוֹן לִטְבוֹחַ יִשְׁרֵי דָרֶךְ:

[תהילים פרק-מה-ד]  חֲגוֹר חַרְבְּךָ עַל יָרֵךְ גִּבּוֹר הוֹדְךָ וַהֲדָרֶךָ:

[תהילים פרק-מה-ו]  חִצֶּיךָ שְׁנוּנִים עַמִּים תַּחְתֶּיךָ יִפְּלוּ בְּלֵב אוֹיְבֵי הַמֶּלֶךְ:

[תהילים פרק-קיט-נז]  חֶלְקִי יְהוָה אָמַרְתִּי לִשְׁמֹר דְּבָרֶיךָ:

[תהילים פרק-קיט-נח]  חִלִּיתִי פָנֶיךָ בְכָל לֵב חָנֵּנִי כְּאִמְרָתֶךָ:

[תהילים פרק-קיט-נט]  חִשַּׁבְתִּי דְרָכָי וָאָשִׁיבָה רַגְלַי אֶל עֵדֹתֶיךָ:

[תהילים פרק-קיט-ס]  חַשְׁתִּי וְלֹא הִתְמַהְמָהְתִּי לִשְׁמֹר מִצְוֹתֶיךָ:

[תהילים פרק-קיט-סב]  חֲצוֹת לַיְלָה אָקוּם לְהוֹדוֹת לָךְ עַל מִשְׁפְּטֵי צִדְקֶךָ:

[תהילים פרק-קיט-סג]  חָבֵר אָנִי לְכָל אֲשֶׁר יְרֵאוּךָ וּלְשֹׁמְרֵי פִּקּוּדֶיךָ:

[משלי פרק-ג-ג]  חֶסֶד וֶאֱמֶת אַל יַעַזְבֻךָ קָשְׁרֵם עַל גַּרְגְּרוֹתֶיךָ כָּתְבֵם עַל לוּחַ לִבֶּךָ:

[איוב פרק-כו-י]  חֹק חָג עַל פְּנֵי מָיִם עַד תַּכְלִית אוֹר עִם חֹשֶׁךְ:

[איכה פרק-ג-כג]  חֲדָשִׁים לַבְּקָרִים רַבָּה אֱמוּנָתֶךָ:

[דניאל פרק-ט-ה]  חָטָאנוּ וְעָוִינוּ (והרשענו) הִרְשַׁעְנוּ וּמָרָדְנוּ וְסוֹר מִמִּצְוֹתֶךָ וּמִמִּשְׁפָּטֶיךָ:

[נחמיה פרק-א-ז]  חֲבֹל חָבַלְנוּ לָךְ וְלֹא שָׁמַרְנוּ אֶת הַמִּצְוֹת וְאֶת הַחֻקִּים וְאֶת הַמִּשְׁפָּטִים אֲשֶׁר צִוִּיתָ אֶת מֹשֶׁה עַבְדֶּךָ:

[נחמיה פרק-י-ה]  חַטּוּשׁ שְׁבַנְיָה מַלּוּךְ:

[דברי הימים א פרק-א-ג]  חֲנוֹךְ מְתוּשֶׁלַח לָמֶךְ:


ח-ל


[שופטים פרק-ה-ז]  חָדְלוּ פְרָזוֹן בְּיִשְׂרָאֵל חָדֵלּוּ עַד שַׁקַּמְתִּי דְּבוֹרָה שַׁקַּמְתִּי אֵם בְּיִשְׂרָאֵל:

[מלכים א פרק-ט-יא]  חִירָם מֶלֶךְ צֹר נִשָּׂא אֶת שְׁלֹמֹה בַּעֲצֵי אֲרָזִים וּבַעֲצֵי בְרוֹשִׁים וּבַזָּהָב לְכָל חֶפְצוֹ אָז יִתֵּן הַמֶּלֶךְ שְׁלֹמֹה לְחִירָם עֶשְׂרִים עִיר בְּאֶרֶץ הַגָּלִיל:

[יחזקאל פרק-יח-ג]  חַי אָנִי נְאֻם אֲדֹנָי יְהוִה אִם יִהְיֶה לָכֶם עוֹד מְשֹׁל הַמָּשָׁל הַזֶּה בְּיִשְׂרָאֵל:

[איוב פרק-ט-כו]  חָלְפוּ עִם אֳנִיּוֹת אֵבֶה כְּנֶשֶׁר יָטוּשׂ עֲלֵי אֹכֶל:

[איוב פרק-כ-טו]  חַיִל בָּלַע וַיְקִאֶנּוּ מִבִּטְנוֹ יוֹרִשֶׁנּוּ אֵל:


ח-מ


[בראשית פרק-ז-כ]  חֲמֵשׁ עֶשְׂרֵה אַמָּה מִלְמַעְלָה גָּבְרוּ הַמָּיִם וַיְכֻסּוּ הֶהָרִים:

[יהושע פרק-א-ו]  חֲזַק וֶאֱמָץ כִּי אַתָּה תַּנְחִיל אֶת הָעָם הַזֶּה אֶת הָאָרֶץ אֲשֶׁר נִשְׁבַּעְתִּי לַאֲבוֹתָם לָתֵת לָהֶם:

[יהושע פרק-ז-יא]  חָטָא יִשְׂרָאֵל וְגַם עָבְרוּ אֶת בְּרִיתִי אֲשֶׁר צִוִּיתִי אוֹתָם וְגַם לָקְחוּ מִן הַחֵרֶם וְגַם גָּנְבוּ וְגַם כִּחֲשׁוּ וְגַם שָׂמוּ בִכְלֵיהֶם:

[ישעיה פרק-לב-יא]  חִרְדוּ שַׁאֲנַנּוֹת רְגָזָה בֹּטְחוֹת פְּשֹׁטָה וְעֹרָה וַחֲגוֹרָה עַל חֲלָצָיִם:

[ישעיה פרק-לד-ו]  חֶרֶב לַיהוָה מָלְאָה דָם הֻדַּשְׁנָה מֵחֵלֶב מִדַּם כָּרִים וְעַתּוּדִים מֵחֵלֶב כִּלְיוֹת אֵילִים כִּי זֶבַח לַיהוָה בְּבָצְרָה וְטֶבַח גָּדוֹל בְּאֶרֶץ אֱדוֹם:

[ירמיה פרק-יז-א]  חַטַּאת יְהוּדָה כְּתוּבָה בְּעֵט בַּרְזֶל בְּצִפֹּרֶן שָׁמִיר חֲרוּשָׁה עַל לוּחַ לִבָּם וּלְקַרְנוֹת מִזְבְּחוֹתֵיכֶם:

[ירמיה פרק-נ-כז]  חִרְבוּ כָּל פָּרֶיהָ יֵרְדוּ לַטָּבַח הוֹי עֲלֵיהֶם כִּי בָא יוֹמָם עֵת פְּקֻדָּתָם:

[ירמיה פרק-נא-לה]  חֲמָסִי וּשְׁאֵרִי עַל בָּבֶל תֹּאמַר יֹשֶׁבֶת צִיּוֹן וְדָמִי אֶל יֹשְׁבֵי כַשְׂדִּים תֹּאמַר יְרוּשָׁלִָם:

[יחזקאל פרק-כ-לג]  חַי אָנִי נְאֻם אֲדֹנָי יְהוִה אִם לֹא בְּיָד חֲזָקָה וּבִזְרוֹעַ נְטוּיָה וּבְחֵמָה שְׁפוּכָה אֶמְלוֹךְ עֲלֵיכֶם:

[יחזקאל פרק-כג-טו]  חֲגוֹרֵי אֵזוֹר בְּמָתְנֵיהֶם סְרוּחֵי טְבוּלִים בְּרָאשֵׁיהֶם מַרְאֵה שָׁלִשִׁים כֻּלָּם דְּמוּת בְּנֵי בָבֶל כַּשְׂדִּים אֶרֶץ מוֹלַדְתָּם:

[הושע פרק-י-ב]  חָלַק לִבָּם עַתָּה יֶאְשָׁמוּ הוּא יַעֲרֹף מִזְבְּחוֹתָם יְשֹׁדֵד מַצֵּבוֹתָם:

[הושע פרק-יג-יג]  חֶבְלֵי יוֹלֵדָה יָבֹאוּ לוֹ הוּא בֵן לֹא חָכָם כִּי עֵת לֹא יַעֲמֹד בְּמִשְׁבַּר בָּנִים:

[זכריה פרק-יג-ז]  חֶרֶב עוּרִי עַל רֹעִי וְעַל גֶּבֶר עֲמִיתִי נְאֻם יְהוָה צְבָאוֹת הַךְ אֶת הָרֹעֶה וּתְפוּצֶיןָ הַצֹּאן וַהֲשִׁבֹתִי יָדִי עַל הַצֹּעֲרִים:

[תהילים פרק-לד-ח]  חֹנֶה מַלְאַךְ יְהוָה סָבִיב לִירֵאָיו וַיְחַלְּצֵם:

[תהילים פרק-סח-יא]  חַיָּתְךָ יָשְׁבוּ בָהּ תָּכִין בְּטוֹבָתְךָ לֶעָנִי אֱלֹהִים:

[תהילים פרק-עז-ו]  חִשַּׁבְתִּי יָמִים מִקֶּדֶם שְׁנוֹת עוֹלָמִים:

[תהילים פרק-פו-ג]  חָנֵּנִי אֲדֹנָי כִּי אֵלֶיךָ אֶקְרָא כָּל הַיּוֹם:

[תהילים פרק-קטז-ה]  חַנּוּן יְהוָֹה וְצַדִּיק וֵאלֹהֵינוּ מְרַחֵם:

[תהילים פרק-קכ-ד]  חִצֵּי גִבּוֹר שְׁנוּנִים עִם גַּחֲלֵי רְתָמִים:

[משלי פרק-כב-כט]  חָזִיתָ אִישׁ מָהִיר בִּמְלַאכְתּוֹ לִפְנֵי מְלָכִים יִתְיַצָּב בַּל יִתְיַצֵּב לִפְנֵי חֲשֻׁכִּים:

[משלי פרק-כו-ט]  חוֹחַ עָלָה בְיַד שִׁכּוֹר וּמָשָׁל בְּפִי כְסִילִים:

[משלי פרק-כו-טז]  חָכָם עָצֵל בְּעֵינָיו מִשִּׁבְעָה מְשִׁיבֵי טָעַם:

[איוב פרק-ט-ד]  חֲכַם לֵבָב וְאַמִּיץ כֹּחַ מִי הִקְשָׁה אֵלָיו וַיִּשְׁלָם:

[איוב פרק-לב-טו]  חַתּוּ לֹא עָנוּ עוֹד הֶעְתִּיקוּ מֵהֶם מִלִּים:

[שיר השירים פרק-ה-טז]  חִכּוֹ מַמְתַקִּים וְכֻלּוֹ מַחֲמַדִּים זֶה דוֹדִי וְזֶה רֵעִי בְּנוֹת יְרוּשָׁלִָם:

[נחמיה פרק-יא-לג]  חָצוֹר רָמָה גִּתָּיִם:


ח-נ


[יהושע פרק-יג-יז]  חֶשְׁבּוֹן וְכָל עָרֶיהָ אֲשֶׁר בַּמִּישֹׁר דִּיבוֹן וּבָמוֹת בַּעַל וּבֵית בַּעַל מְעוֹן:

[שמואל א פרק-כה-טז]  חוֹמָה הָיוּ עָלֵינוּ גַּם לַיְלָה גַּם יוֹמָם כָּל יְמֵי הֱיוֹתֵנוּ עִמָּם רֹעִים הַצֹּאן:

[שמואל ב פרק-כג-כט]  חֵלֶב בֶּן בַּעֲנָה הַנְּטֹפָתִי אִתַּי בֶּן רִיבַי מִגִּבְעַת בְּנֵי בִנְיָמִן:

[יחזקאל פרק-מז-טז]  חֲמָת בֵּרוֹתָה סִבְרַיִם אֲשֶׁר בֵּין גְּבוּל דַּמֶּשֶׂק וּבֵין גְּבוּל חֲמָת חָצֵר הַתִּיכוֹן אֲשֶׁר אֶל גְּבוּל חַוְרָן:

[משלי פרק-יב-יב]  חָמַד רָשָׁע מְצוֹד רָעִים וְשֹׁרֶשׁ צַדִּיקִים יִתֵּן:

[משלי פרק-כ-ל]  חַבֻּרוֹת פֶּצַע (תמריק) תַּמְרוּק בְּרָע וּמַכּוֹת חַדְרֵי בָטֶן:

[דניאל פרק-ב-לד]  חָזֵה הֲוַיְתָ עַד דִּי הִתְגְּזֶרֶת אֶבֶן דִּי לָא בִידַיִן וּמְחָת לְצַלְמָא עַל רַגְלוֹהִי דִּי פַרְזְלָא וְחַסְפָּא וְהַדֵּקֶת הִמּוֹן:

[דניאל פרק-ז-כא]  חָזֵה הֲוֵית וְקַרְנָא דִכֵּן עָבְדָה קְרָב עִם קַדִּישִׁין וְיָכְלָה לְהוֹן:


ח-ס


[ישעיה פרק-לד-יב]  חֹרֶיהָ וְאֵין שָׁם מְלוּכָה יִקְרָאוּ וְכָל שָׂרֶיהָ יִהְיוּ אָפֶס:


ח-פ


[ישעיה פרק-מד-יב]  חָרַשׁ בַּרְזֶל מַעֲצָד וּפָעַל בַּפֶּחָם וּבַמַּקָּבוֹת יִצְּרֵהוּ וַיִּפְעָלֵהוּ בִּזְרוֹעַ כֹּחוֹ גַּם רָעֵב וְאֵין כֹּחַ לֹא שָׁתָה מַיִם וַיִּיעָף:


ח-צ


[איכה פרק-ד-ח]  חָשַׁךְ מִשְּׁחוֹר תָּאֳרָם לֹא נִכְּרוּ בַּחוּצוֹת צָפַד עוֹרָם עַל עַצְמָם יָבֵשׁ הָיָה כָעֵץ:


ח-ק


[דניאל פרק-ז-ט]  חָזֵה הֲוֵית עַד דִּי כָרְסָוָן רְמִיו וְעַתִּיק יוֹמִין יְתִב לְבוּשֵׁהּ כִּתְלַג חִוָּר וּשְׂעַר רֵאשֵׁהּ כַּעֲמַר נְקֵא כָּרְסְיֵהּ שְׁבִיבִין דִּי נוּר גַּלְגִּלּוֹהִי נוּר דָּלִק:


ח-ר


[דברים פרק-לב-יד]  חֶמְאַת בָּקָר וַחֲלֵב צֹאן עִם חֵלֶב כָּרִים וְאֵילִים בְּנֵי בָשָׁן וְעַתּוּדִים עִם חֵלֶב כִּלְיוֹת חִטָּה וְדַם עֵנָב תִּשְׁתֶּה חָמֶר:

[דברים פרק-לב-לג]  חֲמַת תַּנִּינִם יֵינָם וְרֹאשׁ פְּתָנִים אַכְזָר:

[יהושע פרק-טו-נח]  חַלְחוּל בֵּית צוּר וּגְדוֹר:

[ישעיה פרק-מד-טז]  חֶצְיוֹ שָׂרַף בְּמוֹ אֵשׁ עַל חֶצְיוֹ בָּשָׂר יֹאכֵל יִצְלֶה צָלִי וְיִשְׂבָּע אַף יָחֹם וְיֹאמַר הֶאָח חַמּוֹתִי רָאִיתִי אוּר:

[יחזקאל פרק-יג-ו]  חָזוּ שָׁוְא וְקֶסֶם כָּזָב הָאֹמְרִים נְאֻם יְהוָה וַיהוָה לֹא שְׁלָחָם וְיִחֲלוּ לְקַיֵּם דָּבָר:

[משלי פרק-יג-כד]  חוֹשֵׂךְ שִׁבְטוֹ שׂוֹנֵא בְנוֹ וְאֹהֲבוֹ שִׁחֲרוֹ מוּסָר:

[משלי פרק-כז-יא]  חֲכַם בְּנִי וְשַׂמַּח לִבִּי וְאָשִׁיבָה חֹרְפִי דָבָר:

[איוב פרק-כד-טז]  חָתַר בַּחֹשֶׁךְ בָּתִּים יוֹמָם חִתְּמוּ לָמוֹ לֹא יָדְעוּ אוֹר:

[איכה פרק-א-ח]  חֵטְא חָטְאָה יְרוּשָׁלִַם עַל כֵּן לְנִידָה הָיָתָה כָּל מְכַבְּדֶיהָ הִזִּילוּהָ כִּי רָאוּ עֶרְוָתָהּ גַּם הִיא נֶאֶנְחָה וַתָּשָׁב אָחוֹר:

[דברי הימים א פרק-כו-יא]  חִלְקִיָּהוּ הַשֵּׁנִי טְבַלְיָהוּ הַשְּׁלִשִׁי זְכַרְיָהוּ הָרְבִעִי כָּל בָּנִים וְאַחִים לְחֹסָה שְׁלֹשָׁה עָשָׂר:


ח-ש


[קהלת פרק-י-ח]  חֹפֵר גּוּמָּץ בּוֹ יִפּוֹל וּפֹרֵץ גָּדֵר יִשְּׁכֶנּוּ נָחָשׁ:


ח-ת


[שמות פרק-לו-יב]  חֲמִשִּׁים לֻלָאֹת עָשָׂה בַּיְרִיעָה הָאֶחָת וַחֲמִשִּׁים לֻלָאֹת עָשָׂה בִּקְצֵה הַיְרִיעָה אֲשֶׁר בַּמַּחְבֶּרֶת הַשֵּׁנִית מַקְבִּילֹת הַלֻּלָאֹת אַחַת אֶל אֶחָת:

[דברים פרק-יח-ח]  חֵלֶק כְּחֵלֶק יֹאכֵלוּ לְבַד מִמְכָּרָיו עַל הָאָבוֹת:

[שופטים פרק-ג-ג]  חֲמֵשֶׁת סַרְנֵי פְלִשְׁתִּים וְכָל הַכְּנַעֲנִי וְהַצִּידֹנִי וְהַחִוִּי יֹשֵׁב הַר הַלְּבָנוֹן מֵהַר בַּעַל חֶרְמוֹן עַד לְבוֹא חֲמָת:

[שמואל ב פרק-כב-ו]  חֶבְלֵי שְׁאוֹל סַבֻּנִי קִדְּמֻנִי מֹקְשֵׁי מָוֶת:

[ישעיה פרק-מד-יג]  חָרַשׁ עֵצִים נָטָה קָו יְתָאֲרֵהוּ בַשֶּׂרֶד יַעֲשֵׂהוּ בַּמַּקְצֻעוֹת וּבַמְּחוּגָה יְתָאֳרֵהוּ וַיַּעֲשֵׂהוּ כְּתַבְנִית אִישׁ כְּתִפְאֶרֶת אָדָם לָשֶׁבֶת בָּיִת:

[יחזקאל פרק-יז-טז]  חַי אָנִי נְאֻם אֲדֹנָי יְהוִה אִם לֹא בִּמְקוֹם הַמֶּלֶךְ הַמַּמְלִיךְ אֹתוֹ אֲשֶׁר בָּזָה אֶת אָלָתוֹ וַאֲשֶׁר הֵפֵר אֶת בְּרִיתוֹ אִתּוֹ בְתוֹךְ בָּבֶל יָמוּת:

[יחזקאל פרק-לג-טו]  חֲבֹל יָשִׁיב רָשָׁע גְּזֵלָה יְשַׁלֵּם בְּחֻקּוֹת הַחַיִּים הָלַךְ לְבִלְתִּי עֲשׂוֹת עָוֶל חָיוֹ יִחְיֶה לֹא יָמוּת:

[תהילים פרק-ט-יד]  חָנְנֵנִי יְהוָה רְאֵה עָנְיִי מִשֹּׂנְאָי מְרוֹמְמִי מִשַּׁעֲרֵי מָוֶת:

[תהילים פרק-יז-י]  חֶלְבָּמוֹ סָּגְרוּ פִּימוֹ דִּבְּרוּ בְגֵאוּת:

[תהילים פרק-יח-ו]  חֶבְלֵי שְׁאוֹל סְבָבוּנִי קִדְּמוּנִי מוֹקְשֵׁי מָוֶת:

[תהילים פרק-נה-כב]  חָלְקוּ מַחְמָאֹת פִּיו וּקֲרָב לִבּוֹ רַכּוּ דְבָרָיו מִשֶּׁמֶן וְהֵמָּה פְתִחוֹת:

[תהילים פרק-נז-ב]  חָנֵּנִי אֱלֹהִים חָנֵּנִי כִּי בְךָ חָסָיָה נַפְשִׁי וּבְצֵל כְּנָפֶיךָ אֶחְסֶה עַד יַעֲבֹר הַוּוֹת:

[משלי פרק-יט-כז]  חַדַל בְּנִי לִשְׁמֹעַ מוּסָר לִשְׁגוֹת מֵאִמְרֵי דָעַת:

[אסתר פרק-א-ו]  חוּר כַּרְפַּס וּתְכֵלֶת אָחוּז בְּחַבְלֵי בוּץ וְאַרְגָּמָן עַל גְּלִילֵי כֶסֶף וְעַמּוּדֵי שֵׁשׁ מִטּוֹת זָהָב וָכֶסֶף עַל רִצְפַת בַּהַט וָשֵׁשׁ וְדַר וְסֹחָרֶת:

[דניאל פרק-ד-י]  חָזֵה הֲוֵית בְּחֶזְוֵי רֵאשִׁי עַל מִשְׁכְּבִי וַאֲלוּ עִיר וְקַדִּישׁ מִן שְׁמַיָּא נָחִת:
