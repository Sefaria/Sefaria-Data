שמות פרק-א

{א} וְאֵלֶּה שְׁמוֹת בְּנֵי יִשְׂרָאֵל הַבָּאִים מִצְרָיְמָה אֵת יַֽעֲקֹב אִישׁ וּבֵיתוֹ בָּֽאוּ:  מ  וְאֵלֶּה שְׁמוֹת בְּנֵי יִשְׂרָאֵל הַבָּאִים מִצְרָיְמָה אֵת יַֽעֲקֹב אִישׁ וּבֵיתוֹ בָּֽאוּ:  ת  וְאִלֵין שְׁמָהַת בְּנֵי יִשְׂרָאֵל דְעַלוּ לְמִצְרָיִם עִם יַעֲקֹב גְבַר וֶאֱנַשׁ בֵּיתֵהּ עַלוּ:

{ב} רְאוּבֵן שִׁמְעוֹן לֵוִי וִֽיהוּדָֽה:  מ  רְאוּבֵן שִׁמְעוֹן לֵוִי וִֽיהוּדָֽה:  ת  רְאוּבֵן שִׁמְעוֹן לֵוִי וִיהוּדָה:

{ג} יִשָּׂשׂכָר זְבוּלֻן וּבִנְיָמִֽן:  מ  יִשָּׂשׂכָר זְבוּלֻן וּבִנְיָמִֽן:  ת  יִשָׂשׂכָר זְבוּלֻן וּבִנְיָמִין:

{ד} דָּן וְנַפְתָּלִי גָּד וְאָשֵֽׁר:  מ  דָּן וְנַפְתָּלִי גָּד וְאָשֵֽׁר:  ת  דָן וְנַפְתָּלִי גָד וְאָשֵׁר:

{ה} וַיְהִי כָּל נֶפֶשׁ יֹֽצְאֵי יֶֽרֶךְ יַֽעֲקֹב שִׁבְעִים נָפֶשׁ וְיוֹסֵף הָיָה בְמִצְרָֽיִם:  מ  וַיְהִי כָּל נֶפֶשׁ יֹֽצְאֵי יֶֽרֶךְ יַֽעֲקֹב שִׁבְעִים נָפֶשׁ וְיוֹסֵף הָיָה בְמִצְרָֽיִם:  ת  וַהֲוָה כָּל נַפְשָׁתָא נַפְקֵי יַרְכָּא דְיַעֲקֹב שַׁבְעִין נַפְשָׁן וְיוֹסֵף דַהֲוָה בְּמִצְרָיִם:

{ו} וַיָּמָת יוֹסֵף וְכָל אֶחָיו וְכֹל הַדּוֹר הַהֽוּא:  מ  וַיָּמָת יוֹסֵף וְכָל אֶחָיו וְכֹל הַדּוֹר הַהֽוּא:  ת  וּמִית יוֹסֵף וְכָל אַחוֹהִי וְכֹל דָרָא הַהוּא:

{ז} וּבְנֵי יִשְׂרָאֵל פָּרוּ וַֽיִּשְׁרְצוּ וַיִּרְבּוּ וַיַּֽעַצְמוּ בִּמְאֹד מְאֹד וַתִּמָּלֵא הָאָרֶץ אֹתָֽם:  מ  וּבְנֵי יִשְׂרָאֵל פָּרוּ וַֽיִּשְׁרְצוּ וַיִּרְבּוּ וַיַּֽעַצְמוּ בִּמְאֹד מְאֹד וַתִּמָּלֵא הָאָרֶץ אֹתָֽם:  ת  וּבְנֵי יִשְׂרָאֵל נְפִישׁוּ וְאִתְיַלָדוּ וּסְגִיאוּ וּתְקִיפוּ לַחֲדָא לַחֲדָא וְאִתְמְלִיאַת אַרְעָא מִנְהוֹן: (פ)

{ח} וַיָּקָם מֶֽלֶךְ חָדָשׁ עַל מִצְרָיִם אֲשֶׁר לֹֽא יָדַע אֶת יוֹסֵֽף:  מ  וַיָּקָם מֶֽלֶךְ חָדָשׁ עַל מִצְרָיִם אֲשֶׁר לֹֽא יָדַע אֶת יוֹסֵֽף:  ת  וְקָם מַלְכָּא חַדְתָּא עַל מִצְרָיִם דְלָא מְקַיֵם גְזֵרַת יוֹסֵף:

{ט} וַיֹּאמֶר אֶל עַמּוֹ הִנֵּה עַם בְּנֵי יִשְׂרָאֵל רַב וְעָצוּם מִמֶּֽנּוּ:  מ  וַיֹּאמֶר אֶל עַמּוֹ הִנֵּה עַם בְּנֵי יִשְׂרָאֵל רַב וְעָצוּם מִמֶּֽנּוּ:  ת  וַאֲמַר לְעַמֵהּ הָא עַמָא בְּנֵי יִשְׂרָאֵל סָגָן וְתַקִיפִין מִנָנָא:

{י} הָבָה נִּֽתְחַכְּמָה לוֹ פֶּן יִרְבֶּה וְהָיָה כִּֽי תִקְרֶאנָה מִלְחָמָה וְנוֹסַף גַּם הוּא עַל שׂנְאֵינוּ וְנִלְחַם בָּנוּ וְעָלָה מִן הָאָֽרֶץ:  מ  הָבָה נִּֽתְחַכְּמָה לוֹ פֶּן יִרְבֶּה וְהָיָה כִּֽי תִקְרֶאנָה מִלְחָמָה וְנוֹסַף גַּם הוּא עַל שׂנְאֵינוּ וְנִלְחַם בָּנוּ וְעָלָה מִן הָאָֽרֶץ:  ת  הָבוּ נִתְחַכַּם לְהוֹן דִלְמָא יִסְגוּן וִיהֵי אֲרֵי יְעַרְעִנָנָא קְרָב וְיִתּוֹסְפוּן אַף אִנוּן עַל סַנְאָנָא וִיגִיחוּן בָּנָא קְרָב וְיִסְקוּן מִן אַרְעָא:

{יא} וַיָּשִׂימוּ עָלָיו שָׂרֵי מִסִּים לְמַעַן עַנֹּתוֹ בְּסִבְלֹתָם וַיִּבֶן עָרֵי מִסְכְּנוֹת לְפַרְעֹה אֶת פִּתֹם וְאֶת רַֽעַמְסֵֽס:  מ  וַיָּשִׂימוּ עָלָיו שָׂרֵי מִסִּים לְמַעַן עַנֹּתוֹ בְּסִבְלֹתָם וַיִּבֶן עָרֵי מִסְכְּנוֹת לְפַרְעֹה אֶת פִּתֹם וְאֶת רַֽעַמְסֵֽס:  ת  וּמַנִיאוּ עֲלֵיהוֹן שִׁלְטוֹנִין מַבְּאִישִׁין בְּדִיל לְעַנוֹאֵהוֹן בְּפָלְחָנְהוֹן וּבְנוֹ קִרְוֵי בֵית אוֹצָרָא לְפַרְעֹה יָת פִּיתוֹם וְיָת רַעַמְסֵס:

{יב} וְכַֽאֲשֶׁר יְעַנּוּ אֹתוֹ כֵּן יִרְבֶּה וְכֵן יִפְרֹץ וַיָּקֻצוּ מִפְּנֵי בְּנֵי יִשְׂרָאֵֽל:  מ  וְכַֽאֲשֶׁר יְעַנּוּ אֹתוֹ כֵּן יִרְבֶּה וְכֵן יִפְרֹץ וַיָּקֻצוּ מִפְּנֵי בְּנֵי יִשְׂרָאֵֽל:  ת  וּכְמָא דִמְעַנַן לְהוֹן כֵּן סָגָן וְכֵן תַּקְפִין וַעֲקַת לְמִצְרָאֵי מִן קֳדָם בְּנֵי יִשְׂרָאֵל:

{יג} וַיַּֽעֲבִדוּ מִצְרַיִם אֶת בְּנֵי יִשְׂרָאֵל בְּפָֽרֶךְ:  מ  וַיַּֽעֲבִדוּ מִצְרַיִם אֶת בְּנֵי יִשְׂרָאֵל בְּפָֽרֶךְ:  ת  וְאַפְלָחוּ מִצְרָאֵי יָת בְּנֵי יִשְׂרָאֵל בְּקַשְׁיוּ:

{יד} וַיְמָֽרֲרוּ אֶת חַיֵּיהֶם בַּֽעֲבֹדָה קָשָׁה בְּחֹמֶר וּבִלְבֵנִים וּבְכָל עֲבֹדָה בַּשָּׂדֶה אֵת כָּל עֲבֹדָתָם אֲשֶׁר עָֽבְדוּ בָהֶם בְּפָֽרֶךְ:  מ  וַיְמָֽרֲרוּ אֶת חַיֵּיהֶם בַּֽעֲבֹדָה קָשָׁה בְּחֹמֶר וּבִלְבֵנִים וּבְכָל עֲבֹדָה בַּשָּׂדֶה אֵת כָּל עֲבֹדָתָם אֲשֶׁר עָֽבְדוּ בָהֶם בְּפָֽרֶךְ:  ת  וְאַמְרָרוּ יָת חַיֵיהוֹן בְּפָלְחָנָא קָשְׁיָא בְּטִּינָא וּבְלִבְנִין וּבְכָל פָלְחָנָא בְּחַקְלָא יָת כָּל פָּלְחָנְהוֹן דְאַפְלָחוּ בְהוֹן בְּקַשְׁיוּ:

{טו} וַיֹּאמֶר מֶלֶךְ מִצְרַיִם לַֽמְיַלְּדֹת הָֽעִבְרִיֹּת אֲשֶׁר שֵׁם הָֽאַחַת שִׁפְרָה וְשֵׁם הַשֵּׁנִית פּוּעָֽה:  מ  וַיֹּאמֶר מֶלֶךְ מִצְרַיִם לַֽמְיַלְּדֹת הָֽעִבְרִיֹּת אֲשֶׁר שֵׁם הָֽאַחַת שִׁפְרָה וְשֵׁם הַשֵּׁנִית פּוּעָֽה:  ת  וַאֲמַר מַלְכָּא דְמִצְרַיִם לְחַיָתָא יְהוּדַיָתָא דְשׁוּם חָדָא שִׁפְרָה וְשּׁוּם תִּנְיֵתָא פּוּעָה:

{טז} וַיֹּאמֶר בְּיַלֶּדְכֶן אֶת הָעִבְרִיּוֹת וּרְאִיתֶן עַל הָֽאָבְנָיִם אִם בֵּן הוּא וַֽהֲמִתֶּן אֹתוֹ וְאִם בַּת הִוא וָחָֽיָה:  מ  וַיֹּאמֶר בְּיַלֶּדְכֶן אֶת הָעִבְרִיּוֹת וּרְאִיתֶן עַל הָֽאָבְנָיִם אִם בֵּן הוּא וַֽהֲמִתֶּן אֹתוֹ וְאִם בַּת הִוא וָחָֽיָה:  ת  וַאֲמַר כַּד תֶּהֶוְיָן מוֹלְדָן יָת יְהוּדַיָתָא וְתֶחְזְיָן עַל מַתְבְּרָא אִם בַּר הוּא תִּקְטְלַן יָתֵהּ וְאִם בְּרַתָּא הִיא ותְקַיְמִנַהּ:

{יז} וַתִּירֶאןָ הַֽמְיַלְּדֹת אֶת הָאֱלֹהִים וְלֹא עָשׂוּ כַּֽאֲשֶׁר דִּבֶּר אֲלֵיהֶן מֶלֶךְ מִצְרָיִם וַתְּחַיֶּיןָ אֶת הַיְלָדִֽים:  מ  וַתִּירֶאןָ הַֽמְיַלְּדֹת אֶת הָאֱלֹהִים וְלֹא עָשׂוּ כַּֽאֲשֶׁר דִּבֶּר אֲלֵיהֶן מֶלֶךְ מִצְרָיִם וַתְּחַיֶּיןָ אֶת הַיְלָדִֽים:  ת  וּדְחִילָא חַיָתָא מִן קֳדָם יְיָ וְלָא עֲבָדָא כְּמָא דְמַלִיל עִמְהֶן מַלְכָּא דְמִצְרָיִם וְקַיָמָא יָת בְּנַיָא:

{יח}  שני  וַיִּקְרָא מֶֽלֶךְ מִצְרַיִם לַֽמְיַלְּדֹת וַיֹּאמֶר לָהֶן מַדּוּעַ עֲשִׂיתֶן הַדָּבָר הַזֶּה וַתְּחַיֶּיןָ אֶת הַיְלָדִֽים:  מ  וַיִּקְרָא מֶֽלֶךְ מִצְרַיִם לַֽמְיַלְּדֹת וַיֹּאמֶר לָהֶן מַדּוּעַ עֲשִׂיתֶן הַדָּבָר הַזֶּה וַתְּחַיֶּיןָ אֶת הַיְלָדִֽים:  ת  וּקְרָא מַלְכָּא דְמִצְרַיִם לְחַיָתָא וַאֲמַר לְהֶן מָה דֵין עֲבַדְתּוּן יָת פִּתְגָמָא הָדֵין וְקַיֵמְתּוּן יָת בְּנַיָא:

{יט} וַתֹּאמַרְןָ הַֽמְיַלְּדֹת אֶל פַּרְעֹה כִּי לֹא כַנָּשִׁים הַמִּצְרִיֹּת הָֽעִבְרִיֹּת כִּֽי חָיוֹת הֵנָּה בְּטֶרֶם תָּבוֹא אֲלֵהֶן הַֽמְיַלֶּדֶת וְיָלָֽדוּ:  מ  וַתֹּאמַרְןָ הַֽמְיַלְּדֹת אֶל פַּרְעֹה כִּי לֹא כַנָּשִׁים הַמִּצְרִיֹּת הָֽעִבְרִיֹּת כִּֽי חָיוֹת הֵנָּה בְּטֶרֶם תָּבוֹא אֲלֵהֶן הַֽמְיַלֶּדֶת וְיָלָֽדוּ:  ת  וַאֲמָרָא חַיָתָא לְפַרְעֹה אֲרֵי לָא כִנְשַׁיָא מִצְרְיָתָא יְהוּדַיָתָא אֲרֵי חֲכִּימָן אִנוּן עַד לָא עַלַת לְוָתְהֶּן חַיָתָא וִילִידָן:

{כ} וַיֵּיטֶב אֱלֹהִים לַֽמְיַלְּדֹת וַיִּרֶב הָעָם וַיַּֽעַצְמוּ מְאֹֽד:  מ  וַיֵּיטֶב אֱלֹהִים לַֽמְיַלְּדֹת וַיִּרֶב הָעָם וַיַּֽעַצְמוּ מְאֹֽד:  ת  וְאוֹטֵּיב יְיָ לְחַיָתָא וּסְגִי עַמָא וּתְקִיפוּ לַחֲדָא:

{כא} וַיְהִי כִּֽי יָֽרְאוּ הַֽמְיַלְּדֹת אֶת הָֽאֱלֹהִים וַיַּעַשׂ לָהֶם בָּתִּֽים:  מ  וַיְהִי כִּֽי יָֽרְאוּ הַֽמְיַלְּדֹת אֶת הָֽאֱלֹהִים וַיַּעַשׂ לָהֶם בָּתִּֽים:  ת  וַהֲוָה כַּד דְחִילָא חַיָתָא מִן קֳדָם יְיָ וַעֲבַד לְהוֹן בָּתִּין:

{כב} וַיְצַו פַּרְעֹה לְכָל עַמּוֹ לֵאמֹר כָּל הַבֵּן הַיִּלּוֹד הַיְאֹרָה תַּשְׁלִיכֻהוּ וְכָל הַבַּת תְּחַיּֽוּן:  מ  וַיְצַו פַּרְעֹה לְכָל עַמּוֹ לֵאמֹר כָּל הַבֵּן הַיִּלּוֹד הַיְאֹרָה תַּשְׁלִיכֻהוּ וְכָל הַבַּת תְּחַיּֽוּן:  ת  וּפַקִיד פַּרְעֹה לְכָל עַמֵהּ לְמֵימָר כָּל בְּרָא דְיִתְיְלִּיד לִיהוּדָאֵי בְּנַהֲרָא תִּרְמֻנֵהּ וְכָל בְּרַתָּא תְּקַיְמוּן: (פ)




שמות פרק-ב

{א} וַיֵּלֶךְ אִישׁ מִבֵּית לֵוִי וַיִּקַּח אֶת בַּת לֵוִֽי:  מ  וַיֵּלֶךְ אִישׁ מִבֵּית לֵוִי וַיִּקַּח אֶת בַּת לֵוִֽי:  ת  וַאֲזַל גַבְרָא מִדְבֵית לֵוִי וּנְסִיב יָת בַּת לֵוִי:

{ב} וַתַּהַר הָֽאִשָּׁה וַתֵּלֶד בֵּן וַתֵּרֶא אֹתוֹ כִּי טוֹב הוּא וַֽתִּצְפְּנֵהוּ שְׁלֹשָׁה יְרָחִֽים:  מ  וַתַּהַר הָֽאִשָּׁה וַתֵּלֶד בֵּן וַתֵּרֶא אֹתוֹ כִּי טוֹב הוּא וַֽתִּצְפְּנֵהוּ שְׁלֹשָׁה יְרָחִֽים:  ת  וְעַדִיאַת אִתְּתָא וִילִידַת בָּר וַחֲזַת יָתֵהּ אֲרֵי טַּב הוּא וְאַטְמַרְתֵּהּ תְּלָתָא יַרְחִין:

{ג} וְלֹא יָֽכְלָה עוֹד הַצְּפִינוֹ וַתִּֽקַּֽח לוֹ תֵּבַת גֹּמֶא וַתַּחְמְרָה בַֽחֵמָר וּבַזָּפֶת וַתָּשֶׂם בָּהּ אֶת הַיֶּלֶד וַתָּשֶׂם בַּסּוּף עַל שְׂפַת הַיְאֹֽר:  מ  וְלֹא יָֽכְלָה עוֹד הַצְּפִינוֹ וַתִּֽקַּֽח לוֹ תֵּבַת גֹּמֶא וַתַּחְמְרָה בַֽחֵמָר וּבַזָּפֶת וַתָּשֶׂם בָּהּ אֶת הַיֶּלֶד וַתָּשֶׂם בַּסּוּף עַל שְׂפַת הַיְאֹֽר:  ת  וְלָא יְכֵילַת עוֹד לְאַטְמָרוּתֵהּ וּנְסֵיבַת לֵהּ תֵּבוּתָא דְגוֹמֶא וְחַפָּתַהּ בְּחֵמָרָא וּבְזִפְתָּא וְשַׁוִיאַת בַּהּ יָת רַבְיָא וְשַׁוִיתַהּ בְּיַעֲרָא עַל כֵּיף נַהֲרָא:

{ד} וַתֵּֽתַצַּב אֲחֹתוֹ מֵֽרָחֹק לְדֵעָה מַה יֵּֽעָשֶׂה לֽוֹ:  מ  וַתֵּֽתַצַּב אֲחֹתוֹ מֵֽרָחֹק לְדֵעָה מַה יֵּֽעָשֶׂה לֽוֹ:  ת  וְאִתְעַתָּדַת אֲחָתֵהּ מֵרָחִיק לְמִידַע מָה יִתְעֲבֵד לֵהּ:

{ה} וַתֵּרֶד בַּת פַּרְעֹה לִרְחֹץ עַל הַיְאֹר וְנַֽעֲרֹתֶיהָ הֹֽלְכֹת עַל יַד הַיְאֹר וַתֵּרֶא אֶת הַתֵּבָה בְּתוֹךְ הַסּוּף וַתִּשְׁלַח אֶת אֲמָתָהּ וַתִּקָּחֶֽהָ  מ  וַתֵּרֶד בַּת פַּרְעֹה לִרְחֹץ עַל הַיְאֹר וְנַֽעֲרֹתֶיהָ הֹֽלְכֹת עַל יַד הַיְאֹר וַתֵּרֶא אֶת הַתֵּבָה בְּתוֹךְ הַסּוּף וַתִּשְׁלַח אֶת אֲמָתָהּ וַתִּקָּחֶֽהָ  ת  וּנְחָתַת בַּת פַּרְעֹה לְמִסְחֵי עַל נַהֲרָא וְעוּלֵימָתָהָא מְהַלְכָן עַל כֵּיף נַהֲרָא וַחֲזַת יָת תֵּבוּתָא בְּגוֹ יַעֲרָא וְאוֹשֵׁטַת יָת אַמְתַהּ וּנְסֵבְתַהּ:

{ו} וַתִּפְתַּח וַתִּרְאֵהוּ אֶת הַיֶּלֶד וְהִנֵּה נַעַר בֹּכֶה וַתַּחְמֹל עָלָיו וַתֹּאמֶר מִיַּלְדֵי הָֽעִבְרִים זֶֽה:  מ  וַתִּפְתַּח וַתִּרְאֵהוּ אֶת הַיֶּלֶד וְהִנֵּה נַעַר בֹּכֶה וַתַּחְמֹל עָלָיו וַתֹּאמֶר מִיַּלְדֵי הָֽעִבְרִים זֶֽה:  ת  וּפְתָחַת וַחֲזַת יָת רַבְיָא וְהָא עוּלֵימָא בָּכֵי וְחָסַת עֲלוֹהִי וַאֲמֶרֶת מִבְּנֵי יְהוּדָאֵי הוּא דֵין:

{ז} וַתֹּאמֶר אֲחֹתוֹ אֶל בַּת פַּרְעֹה הַֽאֵלֵךְ וְקָרָאתִי לָךְ אִשָּׁה מֵינֶקֶת מִן הָֽעִבְרִיֹּת וְתֵינִק לָךְ אֶת הַיָּֽלֶד:  מ  וַתֹּאמֶר אֲחֹתוֹ אֶל בַּת פַּרְעֹה הַֽאֵלֵךְ וְקָרָאתִי לָךְ אִשָּׁה מֵינֶקֶת מִן הָֽעִבְרִיֹּת וְתֵינִק לָךְ אֶת הַיָּֽלֶד:  ת  וַאֲמֶרֶת אֲחָתֵהּ לְבַת פַרְעֹה הַאֵיזִיל וְאִקְרֵי לִיךְ אִתְּתָא מֵינִקְתָּא מִן יְהוּדַיָתָא וְתוֹנִיק לִיךְ יָת רַבְיָא:

{ח} וַתֹּֽאמֶר לָהּ בַּת פַּרְעֹה לֵכִי וַתֵּלֶךְ הָֽעַלְמָה וַתִּקְרָא אֶת אֵם הַיָּֽלֶד:  מ  וַתֹּֽאמֶר לָהּ בַּת פַּרְעֹה לֵכִי וַתֵּלֶךְ הָֽעַלְמָה וַתִּקְרָא אֶת אֵם הַיָּֽלֶד:  ת  וַאֲמֶרֶת לַהּ בַּת פַּרְעֹה אֵיזִילִי וַאֲזָלַת עוּלֶמְתָּא וּקְרַת יָת אִמֵהּ דְרַבְיָא:

{ט} וַתֹּאמֶר לָהּ בַּת פַּרְעֹה הֵילִיכִי אֶת הַיֶּלֶד הַזֶּה וְהֵֽינִקִהוּ לִי וַֽאֲנִי אֶתֵּן אֶת שְׂכָרֵךְ וַתִּקַּח הָֽאִשָּׁה הַיֶּלֶד וַתְּנִיקֵֽהוּ:  מ  וַתֹּאמֶר לָהּ בַּת פַּרְעֹה הֵילִיכִי אֶת הַיֶּלֶד הַזֶּה וְהֵֽינִקִהוּ לִי וַֽאֲנִי אֶתֵּן אֶת שְׂכָרֵךְ וַתִּקַּח הָֽאִשָּׁה הַיֶּלֶד וַתְּנִיקֵֽהוּ:  ת  וַאֲמֶרֶת לַהּ בַּת פַּרְעֹה אוֹבִילִי יָת רַבְיָא הָדֵין וְאוֹנְקִיהִי לִי וַאֲנָא אֶתֵּן יָת אַגְרָךְ וּנְסֵיבַת אִתְּתָא יָת רַבְיָא וְאוֹנִקְתֵּהּ:

{י} וַיִּגְדַּל הַיֶּלֶד וַתְּבִאֵהוּ לְבַת פַּרְעֹה וַֽיְהִי לָהּ לְבֵן וַתִּקְרָא שְׁמוֹ מֹשֶׁה וַתֹּאמֶר כִּי מִן הַמַּיִם מְשִׁיתִֽהוּ:  מ  וַיִּגְדַּל הַיֶּלֶד וַתְּבִאֵהוּ לְבַת פַּרְעֹה וַֽיְהִי לָהּ לְבֵן וַתִּקְרָא שְׁמוֹ מֹשֶׁה וַתֹּאמֶר כִּי מִן הַמַּיִם מְשִׁיתִֽהוּ:  ת  וּרְבָא רַבְיָא וְאַיְתִיתֵהּ לְבַת פַּרְעֹה וַהֲוָה לָהּ לְבָר וּקְרַת שְׁמֵהּ משֶׁה וַאֲמֶרֶת אֲרֵי מִן מַיָא שְׁחַלְתֵּהּ:

{יא}  שלישי  וַיְהִי בַּיָּמִים הָהֵם וַיִּגְדַּל מֹשֶׁה וַיֵּצֵא אֶל אֶחָיו וַיַּרְא בְּסִבְלֹתָם וַיַּרְא אִישׁ מִצְרִי מַכֶּה אִישׁ עִבְרִי מֵֽאֶחָֽיו:  מ  וַיְהִי בַּיָּמִים הָהֵם וַיִּגְדַּל מֹשֶׁה וַיֵּצֵא אֶל אֶחָיו וַיַּרְא בְּסִבְלֹתָם וַיַּרְא אִישׁ מִצְרִי מַכֶּה אִישׁ עִבְרִי מֵֽאֶחָֽיו:  ת  וַהֲוָה בְּיוֹמַיָא הָאִנוּן וּרְבָא משֶׁה וּנְפַק לְוָת אַחוֹהִי וַחֲזָא בְּפָלְחַנְהוֹן וַחֲזָא גְבַר מִצְרִי מָחֵי לִגְבַר יְהוּדִי מֵאַחוֹהִי:

{יב} וַיִּפֶן כֹּה וָכֹה וַיַּרְא כִּי אֵין אִישׁ וַיַּךְ אֶת הַמִּצְרִי וַֽיִּטְמְנֵהוּ בַּחֽוֹל:  מ  וַיִּפֶן כֹּה וָכֹה וַיַּרְא כִּי אֵין אִישׁ וַיַּךְ אֶת הַמִּצְרִי וַֽיִּטְמְנֵהוּ בַּחֽוֹל:  ת  וְאִתְפְּנֵי לְכָא וּלְכָא וַחֲזָא אֲרֵי לֵית גַבְרָא וּמְחָא יָת מִצְרָאָה וְטַמְרֵהּ בְּחָלָא:

{יג} וַיֵּצֵא בַּיּוֹם הַשֵּׁנִי וְהִנֵּה שְׁנֵֽי אֲנָשִׁים עִבְרִים נִצִּים וַיֹּאמֶר לָֽרָשָׁע לָמָּה תַכֶּה רֵעֶֽךָ:  מ  וַיֵּצֵא בַּיּוֹם הַשֵּׁנִי וְהִנֵּה שְׁנֵֽי אֲנָשִׁים עִבְרִים נִצִּים וַיֹּאמֶר לָֽרָשָׁע לָמָּה תַכֶּה רֵעֶֽךָ:  ת  וּנְפַק בְּיוֹמָא תִנְיָנָא וְהָא תְּרֵין גֻבְרִין יְהוּדָאִין נָצָן וַאֲמַר לְחַיָבָא לְמָא אַתְּ מָחֵי לְחַבְרָךְ:

{יד} וַיֹּאמֶר מִי שָֽׂמְךָ לְאִישׁ שַׂר וְשֹׁפֵט עָלֵינוּ הַֽלְהָרְגֵנִי אַתָּה אֹמֵר כַּֽאֲשֶׁר הָרַגְתָּ אֶת הַמִּצְרִי וַיִּירָא מֹשֶׁה וַיֹּאמַר אָכֵן נוֹדַע הַדָּבָֽר:  מ  וַיֹּאמֶר מִי שָֽׂמְךָ לְאִישׁ שַׂר וְשֹׁפֵט עָלֵינוּ הַֽלְהָרְגֵנִי אַתָּה אֹמֵר כַּֽאֲשֶׁר הָרַגְתָּ אֶת הַמִּצְרִי וַיִּירָא מֹשֶׁה וַיֹּאמַר אָכֵן נוֹדַע הַדָּבָֽר:  ת  וַאֲמַר מָן שַׁוְיָךְ לִגְבַר רַב וְדַיָן עֲלָנָא הַלְמִקְטְלִי אַתְּ אָמַר כְּמָא דִקְטַלְתָּ יָת מִצְרָאָה וּדְחִיל משֶׁה וַאֲמַר בְּקוּשְׁטָא אִתְיְדַע פִּתְגָמָא:

{טו} וַיִּשְׁמַע פַּרְעֹה אֶת הַדָּבָר הַזֶּה וַיְבַקֵּשׁ לַֽהֲרֹג אֶת מֹשֶׁה וַיִּבְרַח מֹשֶׁה מִפְּנֵי פַרְעֹה וַיֵּשֶׁב בְּאֶֽרֶץ מִדְיָן וַיֵּשֶׁב עַל הַבְּאֵֽר:  מ  וַיִּשְׁמַע פַּרְעֹה אֶת הַדָּבָר הַזֶּה וַיְבַקֵּשׁ לַֽהֲרֹג אֶת מֹשֶׁה וַיִּבְרַח מֹשֶׁה מִפְּנֵי פַרְעֹה וַיֵּשֶׁב בְּאֶֽרֶץ מִדְיָן וַיֵּשֶׁב עַל הַבְּאֵֽר:  ת  וּשְׁמַע פַּרְעֹה יָת פִּתְגָמָא הָדֵין וּבְעָא לְמִקְטַל יָת משֶׁה וַעֲרַק משֶׁה מִן קֳדָם פַּרְעֹה וִיתֵיב בְּאַרְעָא דְמִדְיָן וִיתֵיב עַל בֵּירָא:

{טז} וּלְכֹהֵן מִדְיָן שֶׁבַע בָּנוֹת וַתָּבֹאנָה וַתִּדְלֶנָה וַתְּמַלֶּאנָה אֶת הָרְהָטִים לְהַשְׁקוֹת צֹאן אֲבִיהֶֽן:  מ  וּלְכֹהֵן מִדְיָן שֶׁבַע בָּנוֹת וַתָּבֹאנָה וַתִּדְלֶנָה וַתְּמַלֶּאנָה אֶת הָרְהָטִים לְהַשְׁקוֹת צֹאן אֲבִיהֶֽן:  ת  וּלְרַבָא דְמִדְיָן שְׁבַע בְּנָן וְאָתָאָה וּדְלָאָה וּמְלָאָה יָת רְהָטַיָא לְאַשְׁקָאָה עֲנָא דַאֲבוּהֵן:

{יז} וַיָּבֹאוּ הָֽרֹעִים וַֽיְגָֽרְשׁוּם וַיָּקָם מֹשֶׁה וַיּוֹשִׁעָן וַיַּשְׁקְ אֶת צֹאנָֽם:  מ  וַיָּבֹאוּ הָֽרֹעִים וַֽיְגָֽרְשׁוּם וַיָּקָם מֹשֶׁה וַיּוֹשִׁעָן וַיַּשְׁקְ אֶת צֹאנָֽם:  ת  וַאֲתוֹ רָעַיָא וּטְרָדוּנוּן וְקָם משֶׁה וּפְרָקִנוּן וְאַשְׁקֵי יָת עַנְהֵן:

{יח} וַתָּבֹאנָה אֶל רְעוּאֵל אֲבִיהֶן וַיֹּאמֶר מַדּוּעַ מִֽהַרְתֶּן בֹּא הַיּֽוֹם:  מ  וַתָּבֹאנָה אֶל רְעוּאֵל אֲבִיהֶן וַיֹּאמֶר מַדּוּעַ מִֽהַרְתֶּן בֹּא הַיּֽוֹם:  ת  וַאֲתָאָה לְוַת רְעוּאֵל אֲבוּהֵן וַאֲמַר מָה דֵין אוֹחִיתוּן לְמֵיתֵי יוֹמָא דֵין:

{יט} וַתֹּאמַרְןָ אִישׁ מִצְרִי הִצִּילָנוּ מִיַּד הָֽרֹעִים וְגַם דָּלֹה דָלָה לָנוּ וַיַּשְׁקְ אֶת הַצֹּֽאן:  מ  וַתֹּאמַרְןָ אִישׁ מִצְרִי הִצִּילָנוּ מִיַּד הָֽרֹעִים וְגַם דָּלֹה דָלָה לָנוּ וַיַּשְׁקְ אֶת הַצֹּֽאן:  ת  וַאֲמָרָא גַבְרָא מִצְרָאָה שֵׁזְבָנָא מִיַד רָעַיָא וְאַף מִדְלָא דְלָא לָנָא וְאַשְׁקֵי יָת עָנָא:

{כ} וַיֹּאמֶר אֶל בְּנֹתָיו וְאַיּוֹ לָמָּה זֶּה עֲזַבְתֶּן אֶת הָאִישׁ קִרְאֶן לוֹ וְיֹאכַל לָֽחֶם:  מ  וַיֹּאמֶר אֶל בְּנֹתָיו וְאַיּוֹ לָמָּה זֶּה עֲזַבְתֶּן אֶת הָאִישׁ קִרְאֶן לוֹ וְיֹאכַל לָֽחֶם:  ת  וַאֲמַר לִבְנָתֵיהּ וְאָן הוּא לְמָא דְנַן שְׁבַקְתִּין יָת גַבְרָא קְרַן לֵהּ וְיֵכוּל לַחְמָא:

{כא} וַיּוֹאֶל מֹשֶׁה לָשֶׁבֶת אֶת הָאִישׁ וַיִּתֵּן אֶת צִפֹּרָה בִתּוֹ לְמֹשֶֽׁה:  מ  וַיּוֹאֶל מֹשֶׁה לָשֶׁבֶת אֶת הָאִישׁ וַיִּתֵּן אֶת צִפֹּרָה בִתּוֹ לְמֹשֶֽׁה:  ת  וּצְבִי משֶׁה לְמִיתַב עִם גַבְרָא וִיהַב יָת צִפּוֹרָה בְרַתֵּהּ לְמשֶׁה:

{כב} וַתֵּלֶד בֵּן וַיִּקְרָא אֶת שְׁמוֹ גֵּֽרְשֹׁם כִּי אָמַר גֵּר הָיִיתִי בְּאֶרֶץ נָכְרִיָּֽה:  מ  וַתֵּלֶד בֵּן וַיִּקְרָא אֶת שְׁמוֹ גֵּֽרְשֹׁם כִּי אָמַר גֵּר הָיִיתִי בְּאֶרֶץ נָכְרִיָּֽה:  ת  וִילֵדַת בַּר וּקְרָא יָת שְׁמֵהּ גֵרְשֹׁם אֲרֵי אֲמַר דַיָר הֲוֵיתִי בְּאַרַע נוּכְרָאָה: (פ)

{כג} וַיְהִי בַיָּמִים הָֽרַבִּים הָהֵם וַיָּמָת מֶלֶךְ מִצְרַיִם וַיֵּאָֽנְחוּ בְנֵֽי יִשְׂרָאֵל מִן הָֽעֲבֹדָה וַיִּזְעָקוּ וַתַּעַל שַׁוְעָתָם אֶל הָֽאֱלֹהִים מִן הָֽעֲבֹדָֽה:  מ  וַיְהִי בַיָּמִים הָֽרַבִּים הָהֵם וַיָּמָת מֶלֶךְ מִצְרַיִם וַיֵּאָֽנְחוּ בְנֵֽי יִשְׂרָאֵל מִן הָֽעֲבֹדָה וַיִּזְעָקוּ וַתַּעַל שַׁוְעָתָם אֶל הָֽאֱלֹהִים מִן הָֽעֲבֹדָֽה:  ת  וַהֲוָה בְּיוֹמַיָא סַגִיאַיָא הָאִנוּן וּמִית מַלְכָּא דְמִצְרַיִם וְאִתְנָחוּ בְנֵי יִשְׂרָאֵל מִן פָּלְחָנָא דַהֲוָה קְשֵׁי עֲלֵיהוֹן וּזְעִיקוּ וּסְלֵיקָת קְבֶלְתְּהוֹן לָקֳדָם יְיָ מִן פָּלְחָנָא:

{כד} וַיִּשְׁמַע אֱלֹהִים אֶת נַֽאֲקָתָם וַיִּזְכֹּר אֱלֹהִים אֶת בְּרִיתוֹ אֶת אַבְרָהָם אֶת יִצְחָק וְאֶֽת יַֽעֲקֹֽב:  מ  וַיִּשְׁמַע אֱלֹהִים אֶת נַֽאֲקָתָם וַיִּזְכֹּר אֱלֹהִים אֶת בְּרִיתוֹ אֶת אַבְרָהָם אֶת יִצְחָק וְאֶֽת יַֽעֲקֹֽב:  ת  וּשְׁמִיעַ קֳדָם יְיָ יָת קְבֶלְתְּהוֹן וּדְכִיר יְיָ יָת קַיָמֵהּ דְעִם אַבְרָהָם דְעִם יִצְחָק וּדְעִם יַעֲקֹב:

{כה} וַיַּרְא אֱלֹהִים אֶת בְּנֵי יִשְׂרָאֵל וַיֵּדַע אֱלֹהִֽים:  מ  וַיַּרְא אֱלֹהִים אֶת בְּנֵי יִשְׂרָאֵל וַיֵּדַע אֱלֹהִֽים:  ת  וּגְלִי קְדָם יְיָ שִׁעְבּוּדָא דִבְנֵי יִשְׂרָאֵל וַאֲמַר בְּמֵימְרֵהּ לְמִפְרַקְהוֹן יְיָ: (ס)




שמות פרק-ג

{א}  רביעי  וּמֹשֶׁה הָיָה רֹעֶה אֶת צֹאן יִתְרוֹ חֹֽתְנוֹ כֹּהֵן מִדְיָן וַיִּנְהַג אֶת הַצֹּאן אַחַר הַמִּדְבָּר וַיָּבֹא אֶל הַר הָֽאֱלֹהִים חֹרֵֽבָה:  מ  וּמֹשֶׁה הָיָה רֹעֶה אֶת צֹאן יִתְרוֹ חֹֽתְנוֹ כֹּהֵן מִדְיָן וַיִּנְהַג אֶת הַצֹּאן אַחַר הַמִּדְבָּר וַיָּבֹא אֶל הַר הָֽאֱלֹהִים חֹרֵֽבָה:  ת  וּמשֶׁה הֲּוָה רָעֵי יָת עָנָא דְיִתְרוֹ חֲמוֹהִי רַבָּא דְמִדְיָן וּדְבַר יָת עָנָא לְבָתַר שְׁפַר רַעֲיָא לְמַדְבְּרָא וְאָתָא לְטוּרָא דְאִתְגְלִי עֲלוֹהִי יְקָרָא דַיְיָ לְחוֹרֵב:

{ב} וַיֵּרָא מַלְאַךְ יְהוָֹה אֵלָיו בְּלַבַּת אֵשׁ מִתּוֹךְ הַסְּנֶה וַיַּרְא וְהִנֵּה הַסְּנֶה בֹּעֵר בָּאֵשׁ וְהַסְּנֶה אֵינֶנּוּ אֻכָּֽל:  מ  וַיֵּרָא מַלְאַךְ יְהוָֹה אֵלָיו בְּלַבַּת אֵשׁ מִתּוֹךְ הַסְּנֶה וַיַּרְא וְהִנֵּה הַסְּנֶה בֹּעֵר בָּאֵשׁ וְהַסְּנֶה אֵינֶנּוּ אֻכָּֽל:  ת  וְאִתְגְלִי מַלְאָכָא דַיְיָ לֵהּ בְּשַׁלְהוֹבִית אֶשָׁתָא מִגוֹ אַסָנָא וַחֲזָא וְהָא אַסָנָא בָּעֵר בְּאֶשָׁתָא וַאֲסָנָא לֵיתוֹהִי מִתְאָכִיל:

{ג} וַיֹּאמֶר מֹשֶׁה אָסֻֽרָה נָּא וְאֶרְאֶה אֶת הַמַּרְאֶה הַגָּדֹל הַזֶּה מַדּוּעַ לֹֽא יִבְעַר הַסְּנֶֽה:  מ  וַיֹּאמֶר מֹשֶׁה אָסֻֽרָה נָּא וְאֶרְאֶה אֶת הַמַּרְאֶה הַגָּדֹל הַזֶּה מַדּוּעַ לֹֽא יִבְעַר הַסְּנֶֽה:  ת  וַאֲמַר משֶׁה אִתְפְּנֵי כְעַן וְאֶחֱזֵי יָת חָזְוָנָא רַבָּא הָדֵין מָא דֵין לָא מִתּוֹקַד אַסָנָא:

{ד} וַיַּרְא יְהוָֹה כִּי סָר לִרְאוֹת וַיִּקְרָא אֵלָיו אֱלֹהִים מִתּוֹךְ הַסְּנֶה וַיֹּאמֶר מֹשֶׁה מֹשֶׁה וַיֹּאמֶר הִנֵּֽנִי:  מ  וַיַּרְא יְהוָֹה כִּי סָר לִרְאוֹת וַיִּקְרָא אֵלָיו אֱלֹהִים מִתּוֹךְ הַסְּנֶה וַיֹּאמֶר מֹשֶׁה מֹשֶׁה וַיֹּאמֶר הִנֵּֽנִי:  ת  וַחֲזָא יְיָ אֲרֵי אִתְפְּנֵי לְמֶחֱזֵי וּקְרָא לֵהּ יְיָ מִגוֹ אַסָנָא וַאֲמַר משֶׁה משֶׁה וַאֲמַר הָא אֲנָא:

{ה} וַיֹּאמֶר אַל תִּקְרַב הֲלֹם שַֽׁל נְעָלֶיךָ מֵעַל רַגְלֶיךָ כִּי הַמָּקוֹם אֲשֶׁר אַתָּה עוֹמֵד עָלָיו אַדְמַת קֹדֶשׁ הֽוּא:  מ  וַיֹּאמֶר אַל תִּקְרַב הֲלֹם שַֽׁל נְעָלֶיךָ מֵעַל רַגְלֶיךָ כִּי הַמָּקוֹם אֲשֶׁר אַתָּה עוֹמֵד עָלָיו אַדְמַת קֹדֶשׁ הֽוּא:  ת  וַאֲמַר לָא תִקְרַב הַלְכָא שְׁרִי סֵינָךְ מֵעַל רִגְלָךְ אֲרֵי אַתְרָא דְאַתְּ קָאִים עֲלוֹהִי אֲתַר קַדִישׁ הוּא:

{ו} וַיֹּאמֶר אָֽנֹכִי אֱלֹהֵי אָבִיךָ אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵֽאלֹהֵי יַֽעֲקֹב וַיַּסְתֵּר מֹשֶׁה פָּנָיו כִּי יָרֵא מֵֽהַבִּיט אֶל הָֽאֱלֹהִֽים:  מ  וַיֹּאמֶר אָֽנֹכִי אֱלֹהֵי אָבִיךָ אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵֽאלֹהֵי יַֽעֲקֹב וַיַּסְתֵּר מֹשֶׁה פָּנָיו כִּי יָרֵא מֵֽהַבִּיט אֶל הָֽאֱלֹהִֽים:  ת  וַאֲמַר אֲנָא אֱלָהָא דְאָבוּךְ אֱלָהָא דְאַבְרָהָם אֱלָהָא דְיִצְחָק וֵאלָהָא דְיַעֲקֹב וּכְבֵשִׁנּוּן משֶׁה לְאַפּוֹהִי אֲרֵי דָחִיל מִלְאִסְתַּכָּלָא בְּצֵית יְקָרָא דַיְיָ:

{ז} וַיֹּאמֶר יְהֹוָה רָאֹה רָאִיתִי אֶת עֳנִי עַמִּי אֲשֶׁר בְּמִצְרָיִם וְאֶת צַֽעֲקָתָם שָׁמַעְתִּי מִפְּנֵי נֹֽגְשָׂיו כִּי יָדַעְתִּי אֶת מַכְאֹבָֽיו:  מ  וַיֹּאמֶר יְהֹוָה רָאֹה רָאִיתִי אֶת עֳנִי עַמִּי אֲשֶׁר בְּמִצְרָיִם וְאֶת צַֽעֲקָתָם שָׁמַעְתִּי מִפְּנֵי נֹֽגְשָׂיו כִּי יָדַעְתִּי אֶת מַכְאֹבָֽיו:  ת  וַאֲמַר יְיָ מִגְלָא גְלִי קֳדָמַי יָת שִׁעְבּוּד עַמִי דִי בְמִצְרָיִם וְיָת קְבֶלְתְּהוֹן שְׁמִיעַ קֳדָמַי מִן קֳדָם מַפְלְחֵיהוֹן אֲרֵי גְלֵי קָדָמַי יָת כֵּיבֵיהוֹן:

{ח} וָֽאֵרֵד לְהַצִּילוֹ מִיַּד מִצְרַיִם וּֽלְהַֽעֲלֹתוֹ מִן הָאָרֶץ הַהִוא אֶל אֶרֶץ טוֹבָה וּרְחָבָה אֶל אֶרֶץ זָבַת חָלָב וּדְבָשׁ אֶל מְקוֹם הַֽכְּנַֽעֲנִי וְהַחִתִּי וְהָֽאֱמֹרִי וְהַפְּרִזִּי וְהַֽחִוִּי וְהַיְבוּסִֽי:  מ  וָֽאֵרֵד לְהַצִּילוֹ מִיַּד מִצְרַיִם וּֽלְהַֽעֲלֹתוֹ מִן הָאָרֶץ הַהִוא אֶל אֶרֶץ טוֹבָה וּרְחָבָה אֶל אֶרֶץ זָבַת חָלָב וּדְבָשׁ אֶל מְקוֹם הַֽכְּנַֽעֲנִי וְהַחִתִּי וְהָֽאֱמֹרִי וְהַפְּרִזִּי וְהַֽחִוִּי וְהַיְבוּסִֽי:  ת  וְאִתְגְלֵיתִי לְשֵׁזָבוּתְהוֹן מִידָא דְמִצְרָאֵי וּלְאַסָקוּתְהוֹן מִן אַרְעָא הַהִיא לְאַרְעָא טָבָא וּפַתְיָא לְאַרְעָא עָבְדָא חֲלָב וּדְבָשׁ לַאֲתַר כְּנַעֲנָאֵי וְחִתָּאֵי וֶאֱמוֹרָאֵי וּפְרִזָאֵי וְחִוָאֵי וִיבוּסָאֵי:

{ט} וְעַתָּה הִנֵּה צַֽעֲקַת בְּנֵֽי יִשְׂרָאֵל בָּאָה אֵלָי וְגַם רָאִיתִי אֶת הַלַּחַץ אֲשֶׁר מִצְרַיִם לֹֽחֲצִים אֹתָֽם:  מ  וְעַתָּה הִנֵּה צַֽעֲקַת בְּנֵֽי יִשְׂרָאֵל בָּאָה אֵלָי וְגַם רָאִיתִי אֶת הַלַּחַץ אֲשֶׁר מִצְרַיִם לֹֽחֲצִים אֹתָֽם:  ת  וּכְעַן הָא קְבֵילַת בְּנֵי יִשְׂרָאֵל עָלַת קֳדָמָי וְאַף גְלֵי קֳדָמַי יָת דוֹחֲקָא דִי מִצְרָאֵי דָחֲקִין לְהוֹן:

{י} וְעַתָּה לְכָה וְאֶֽשְׁלָֽחֲךָ אֶל פַּרְעֹה וְהוֹצֵא אֶת עַמִּי בְנֵֽי יִשְׂרָאֵל מִמִּצְרָֽיִם:  מ  וְעַתָּה לְכָה וְאֶֽשְׁלָֽחֲךָ אֶל פַּרְעֹה וְהוֹצֵא אֶת עַמִּי בְנֵֽי יִשְׂרָאֵל מִמִּצְרָֽיִם:  ת  וּכְעַן אִיתָא וְאֶשְׁלְחִנָּךְ לְוַת פַּרְעֹה וְאַפֵּיק יָת עַמִי בְנֵי יִשְׂרָאֵל מִמִצְרָיִם:

{יא} וַיֹּאמֶר מֹשֶׁה אֶל הָאֱלֹהִים מִי אָנֹכִי כִּי אֵלֵךְ אֶל פַּרְעֹה וְכִי אוֹצִיא אֶת בְּנֵי יִשְׂרָאֵל מִמִּצְרָֽיִם:  מ  וַיֹּאמֶר מֹשֶׁה אֶל הָאֱלֹהִים מִי אָנֹכִי כִּי אֵלֵךְ אֶל פַּרְעֹה וְכִי אוֹצִיא אֶת בְּנֵי יִשְׂרָאֵל מִמִּצְרָֽיִם:  ת  וַאֲמַר משֶׁה קֳדָם יְיָ מָן אֲנָא אֲרֵי אֵזִיל לְוָת פַּרְעֹה וַאֲרֵי אַפֵּיק יָת בְּנֵי יִשְׂרָאֵל מִמִצְרָיִם:

{יב} וַיֹּאמֶר כִּי אֶֽהְיֶה עִמָּךְ וְזֶה לְּךָ הָאוֹת כִּי אָֽנֹכִי שְׁלַחְתִּיךָ בְּהוֹצִֽיאֲךָ אֶת הָעָם מִמִּצְרַיִם תַּֽעַבְדוּן אֶת הָאֱלֹהִים עַל הָהָר הַזֶּֽה:  מ  וַיֹּאמֶר כִּי אֶֽהְיֶה עִמָּךְ וְזֶה לְּךָ הָאוֹת כִּי אָֽנֹכִי שְׁלַחְתִּיךָ בְּהוֹצִֽיאֲךָ אֶת הָעָם מִמִּצְרַיִם תַּֽעַבְדוּן אֶת הָאֱלֹהִים עַל הָהָר הַזֶּֽה:  ת  וַאֲמַר אֲרֵי יְהֵי מֵימְרִי בְסַעְדָךְ וְדֵין לָךְ אָתָא אֲרֵי אֲנָא שְׁלַחְתָּךְ בְּאַפָּקוּתָךְ יָת עַמָא מִמִצְרַיִם תִּפְלְחוּן קֳדָם יְיָ עַל טוּרָא הָדֵין:

{יג} וַיֹּאמֶר מֹשֶׁה אֶל הָֽאֱלֹהִים הִנֵּה אָֽנֹכִי בָא אֶל בְּנֵי יִשְׂרָאֵל וְאָֽמַרְתִּי לָהֶם אֱלֹהֵי אֲבֽוֹתֵיכֶם שְׁלָחַנִי אֲלֵיכֶם וְאָֽמְרוּ לִי מַה שְּׁמוֹ מָה אֹמַר אֲלֵהֶֽם:  מ  וַיֹּאמֶר מֹשֶׁה אֶל הָֽאֱלֹהִים הִנֵּה אָֽנֹכִי בָא אֶל בְּנֵי יִשְׂרָאֵל וְאָֽמַרְתִּי לָהֶם אֱלֹהֵי אֲבֽוֹתֵיכֶם שְׁלָחַנִי אֲלֵיכֶם וְאָֽמְרוּ לִי מַה שְּׁמוֹ מָה אֹמַר אֲלֵהֶֽם:  ת  וַאֲמַר משֶׁה קֳדָם יְיָ הָא אֲנָא אָתֵי לְוָת בְּנֵי יִשְׂרָאֵל וְאֵימַר לְהוֹן אֱלָהָא דְאַבְהַתְכוֹן שַׁלְחַנִי לְוָתְכוֹן וְיֵימְרוּן לִי מָא שְׁמֵהּ מָא אֵימַר לְהוֹן:

{יד} וַיֹּאמֶר אֱלֹהִים אֶל מֹשֶׁה אֶֽהְיֶה אֲשֶׁר אֶֽהְיֶה וַיֹּאמֶר כֹּה תֹאמַר לִבְנֵי יִשְׂרָאֵל אֶֽהְיֶה שְׁלָחַנִי אֲלֵיכֶֽם:  מ  וַיֹּאמֶר אֱלֹהִים אֶל מֹשֶׁה אֶֽהְיֶה אֲשֶׁר אֶֽהְיֶה וַיֹּאמֶר כֹּה תֹאמַר לִבְנֵי יִשְׂרָאֵל אֶֽהְיֶה שְׁלָחַנִי אֲלֵיכֶֽם:  ת  וַאֲמַר יְיָ לְמשֶׁה אֶהְיֶה אֲשֶׁר אֶהְיֶה וַאֲמַר כִּדְנַן תֵּימַר לִבְנֵי יִשְׂרָאֵל אֶהְיֶה שַׁלְחַנִי לְוָתְכוֹן:

{טו} וַיֹּאמֶר עוֹד אֱלֹהִים אֶל מֹשֶׁה כֹּה תֹאמַר אֶל בְּנֵי יִשְׂרָאֵל יְהֹוָה אֱלֹהֵי אֲבֹֽתֵיכֶם אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵֽאלֹהֵי יַֽעֲקֹב שְׁלָחַנִי אֲלֵיכֶם זֶה שְּׁמִי לְעֹלָם וְזֶה זִכְרִי לְדֹר דֹּֽר:  מ  וַיֹּאמֶר עוֹד אֱלֹהִים אֶל מֹשֶׁה כֹּה תֹאמַר אֶל בְּנֵי יִשְׂרָאֵל יְהֹוָה אֱלֹהֵי אֲבֹֽתֵיכֶם אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵֽאלֹהֵי יַֽעֲקֹב שְׁלָחַנִי אֲלֵיכֶם זֶה שְּׁמִי לְעֹלָם וְזֶה זִכְרִי לְדֹר דֹּֽר:  ת  וַאֲמַר עוֹד יְיָ לְמשֶׁה כִּדְנַן תֵּימַר לִבְנֵי יִשְׂרָאֵל יְיָ אֱלָהָא דְאַבְהַתְכוֹן אֱלָהֵהּ דְאַבְרָהָם אֱלָהֵהּ דְיִצְחָק וֵאלָהֵהּ דְיַעֲקֹב שַׁלְחַנִי לְוָתְכוֹן דֵין שְׁמִי לְעָלַם וְדֵין דָכְרָנִי לְכָל דָר וְדָר:

{טז}  חמישי  לֵךְ וְאָֽסַפְתָּ אֶת זִקְנֵי יִשְׂרָאֵל וְאָֽמַרְתָּ אֲלֵהֶם יְהֹוָה אֱלֹהֵי אֲבֹֽתֵיכֶם נִרְאָה אֵלַי אֱלֹהֵי אַבְרָהָם יִצְחָק וְיַֽעֲקֹב לֵאמֹר פָּקֹד פָּקַדְתִּי אֶתְכֶם וְאֶת הֶֽעָשׂוּי לָכֶם בְּמִצְרָֽיִם:  מ  לֵךְ וְאָֽסַפְתָּ אֶת זִקְנֵי יִשְׂרָאֵל וְאָֽמַרְתָּ אֲלֵהֶם יְהֹוָה אֱלֹהֵי אֲבֹֽתֵיכֶם נִרְאָה אֵלַי אֱלֹהֵי אַבְרָהָם יִצְחָק וְיַֽעֲקֹב לֵאמֹר פָּקֹד פָּקַדְתִּי אֶתְכֶם וְאֶת הֶֽעָשׂוּי לָכֶם בְּמִצְרָֽיִם:  ת  אִזֵיל וְתִכְנוֹשׁ יָת סָבֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן יְיָ אֱלָהָא דְאַבְהַתְכוֹן אִתְגְלֵי לִי אֱלָהֵהּ דְאַבְרָהָם יִצְחָק וְיַעֲקֹב לְמֵימָר מִדְכַּר דְכִירְנָא יַתְכוֹן וְיָת דְאִתְעֲבַד לְכוֹן בְּמִצְרַיִם:

{יז} וָֽאֹמַר אַֽעֲלֶה אֶתְכֶם מֵֽעֳנִי מִצְרַיִם אֶל אֶרֶץ הַֽכְּנַֽעֲנִי וְהַחִתִּי וְהָֽאֱמֹרִי וְהַפְּרִזִּי וְהַֽחִוִּי וְהַיְבוּסִי אֶל אֶרֶץ זָבַת חָלָב וּדְבָֽשׁ:  מ  וָֽאֹמַר אַֽעֲלֶה אֶתְכֶם מֵֽעֳנִי מִצְרַיִם אֶל אֶרֶץ הַֽכְּנַֽעֲנִי וְהַחִתִּי וְהָֽאֱמֹרִי וְהַפְּרִזִּי וְהַֽחִוִּי וְהַיְבוּסִי אֶל אֶרֶץ זָבַת חָלָב וּדְבָֽשׁ:  ת  וַאֲמָרִית אַסֵיק יָתְכוֹן מִשִׁעְבּוּד מִצְרָאֵי לַאֲרַע כְּנַעֲנָאֵי וְחִתָּאֵי וֶאֱמוֹרָאֵי וּפְרִזָאֵי וְחִוָאֵי וִיבוּסָאֵי לְאַרְעָא עַבְדָא חֲלָב וּדְבָשׁ:

{יח} וְשָֽׁמְעוּ לְקֹלֶךָ וּבָאתָ אַתָּה וְזִקְנֵי יִשְׂרָאֵל אֶל מֶלֶךְ מִצְרַיִם וַֽאֲמַרְתֶּם אֵלָיו יְהֹוָה אֱלֹהֵי הָֽעִבְרִיִּים נִקְרָה עָלֵינוּ וְעַתָּה נֵֽלְכָה נָּא דֶּרֶךְ שְׁלֹשֶׁת יָמִים בַּמִּדְבָּר וְנִזְבְּחָה לַֽיהוָֹה אֱלֹהֵֽינוּ:  מ  וְשָֽׁמְעוּ לְקֹלֶךָ וּבָאתָ אַתָּה וְזִקְנֵי יִשְׂרָאֵל אֶל מֶלֶךְ מִצְרַיִם וַֽאֲמַרְתֶּם אֵלָיו יְהֹוָה אֱלֹהֵי הָֽעִבְרִיִּים נִקְרָה עָלֵינוּ וְעַתָּה נֵֽלְכָה נָּא דֶּרֶךְ שְׁלֹשֶׁת יָמִים בַּמִּדְבָּר וְנִזְבְּחָה לַֽיהוָֹה אֱלֹהֵֽינוּ:  ת  וִיקַבְּלוּן לְמֵימְרָךְ וְתֵיתֵי אַתְּ וְסָבֵי יִשְׂרָאֵל לְוָת מַלְכָּא דְמִצְרַיִם וְתֵימְרוּן לֵהּ יְיָ אֱלָהָא דִיהוּדָאֵי אִתְקְרֵי עֲלָנָא וּכְעַן נֵזֵיל כְּעַן מַהֲלַךְ תְּלָתָא יוֹמִין בְּמַדְבְּרָא וּנְדַבַּח קֳדָם יְיָ אֱלָהָנָא:

{יט} וַֽאֲנִי יָדַעְתִּי כִּי לֹֽא יִתֵּן אֶתְכֶם מֶלֶךְ מִצְרַיִם לַֽהֲלֹךְ וְלֹא בְּיָד חֲזָקָֽה:  מ  וַֽאֲנִי יָדַעְתִּי כִּי לֹֽא יִתֵּן אֶתְכֶם מֶלֶךְ מִצְרַיִם לַֽהֲלֹךְ וְלֹא בְּיָד חֲזָקָֽה:  ת  וּקְדָמַי גְלֵי אֲרֵי לָא יִשְׁבּוֹק יָתְכוֹן מַלְכָּא דְמִצְרַיִם לְמֵיזָל וְלָא מִן קֳדָם דְחֵילֵהּ תַּקִיף:

{כ} וְשָֽׁלַחְתִּי אֶת יָדִי וְהִכֵּיתִי אֶת מִצְרַיִם בְּכֹל נִפְלְאֹתָי אֲשֶׁר אֶעֱשֶׂה בְּקִרְבּוֹ וְאַֽחֲרֵי כֵן יְשַׁלַּח אֶתְכֶֽם:  מ  וְשָֽׁלַחְתִּי אֶת יָדִי וְהִכֵּיתִי אֶת מִצְרַיִם בְּכֹל נִפְלְאֹתָי אֲשֶׁר אֶעֱשֶׂה בְּקִרְבּוֹ וְאַֽחֲרֵי כֵן יְשַׁלַּח אֶתְכֶֽם:  ת  וָאֲשַׁלַח פּוֹן יָת מְחַת גְבֻרְתִּי וְאֶמְחֵי יָת מִצְרָאֵי בְּכֹל פְּרִשְׁוָתִי דִי אַעְבֵּד בֵּינֵיהוֹן וּבָתַר כֵּן יְשַׁלַח יָתְכוֹן:

{כא} וְנָֽתַתִּי אֶת חֵן הָֽעָם הַזֶּה בְּעֵינֵי מִצְרָיִם וְהָיָה כִּי תֵֽלֵכוּן לֹא תֵֽלְכוּ רֵיקָֽם:  מ  וְנָֽתַתִּי אֶת חֵן הָֽעָם הַזֶּה בְּעֵינֵי מִצְרָיִם וְהָיָה כִּי תֵֽלֵכוּן לֹא תֵֽלְכוּ רֵיקָֽם:  ת  וְאֶתֵּן יָת עַמָא הָדֵין לְרַחֲמִין בְּעֵינֵי מִצְרָאֵי וִיהֵי אֲרֵי תְהָכוּן לָא תְהָכוּן רֵיקָנִין:

{כב} וְשָֽׁאֲלָה אִשָּׁה מִשְּׁכֶנְתָּהּ וּמִגָּרַת בֵּיתָהּ כְּלֵי כֶסֶף וּכְלֵי זָהָב וּשְׂמָלֹת וְשַׂמְתֶּם עַל בְּנֵיכֶם וְעַל בְּנֹתֵיכֶם וְנִצַּלְתֶּם אֶת מִצְרָֽיִם:  מ  וְשָֽׁאֲלָה אִשָּׁה מִשְּׁכֶנְתָּהּ וּמִגָּרַת בֵּיתָהּ כְּלֵי כֶסֶף וּכְלֵי זָהָב וּשְׂמָלֹת וְשַׂמְתֶּם עַל בְּנֵיכֶם וְעַל בְּנֹתֵיכֶם וְנִצַּלְתֶּם אֶת מִצְרָֽיִם:  ת  וְתִשְׁאַל אִתְּתָא מִשְׁבַבְתַּהּ וּמִקְרִיבַת בֵּיתַהּ מָנִין דִכְסַף וּמָנִין דִדְהַב וּלְבוּשִׁין וּתְשַׁווּן עַל בְּנֵיכוֹן וְעַל בְּנָתֵיכוֹן וּתְרוֹקִנוּן יָת מִצְרָיִם:




שמות פרק-ד

{א} וַיַּעַן מֹשֶׁה וַיֹּאמֶר וְהֵן לֹא יַֽאֲמִינוּ לִי וְלֹא יִשְׁמְעוּ בְּקֹלִי כִּי יֹֽאמְרוּ לֹֽא נִרְאָה אֵלֶיךָ יְהוָֹֽה:  מ  וַיַּעַן מֹשֶׁה וַיֹּאמֶר וְהֵן לֹא יַֽאֲמִינוּ לִי וְלֹא יִשְׁמְעוּ בְּקֹלִי כִּי יֹֽאמְרוּ לֹֽא נִרְאָה אֵלֶיךָ יְהוָֹֽה:  ת  וַאֲתֵיב משֶׁה וַאֲמַר וְהָא לָא יְהֵמְנוּן לִי וְלָא יְקַבְּלוּן מִנִי אֲרֵי יֵימְרוּן לָא אִתְגְלֵי לָךְ יְיָ:

{ב} וַיֹּאמֶר אֵלָיו יְהוָֹה (מזה) מַה זֶּה בְיָדֶךָ וַיֹּאמֶר מַטֶּֽה:  מ  וַיֹּאמֶר אֵלָיו יְהוָֹה (מזה) מַה זֶּה בְיָדֶךָ וַיֹּאמֶר מַטֶּֽה:  ת  וַאֲמַר לֵהּ יְיָ מָא דֵין בִּידָךְ וַאֲמַר חוּטְרָא:

{ג} וַיֹּאמֶר הַשְׁלִיכֵהוּ אַרְצָה וַיַּשְׁלִכֵהוּ אַרְצָה וַיְהִי לְנָחָשׁ וַיָּנָס מֹשֶׁה מִפָּנָֽיו:  מ  וַיֹּאמֶר הַשְׁלִיכֵהוּ אַרְצָה וַיַּשְׁלִכֵהוּ אַרְצָה וַיְהִי לְנָחָשׁ וַיָּנָס מֹשֶׁה מִפָּנָֽיו:  ת  וַאֲמַר רְמִיהִי לְאַרְעָא וּרְמָהִי לְאַרְעָא וַהֲוָה לְחִוֵי וַעֲרַק משֶׁה מִן קֳדָמוֹהִי:

{ד} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה שְׁלַח יָֽדְךָ וֶֽאֱחֹז בִּזְנָבוֹ וַיִּשְׁלַח יָדוֹ וַיַּֽחֲזֶק בּוֹ וַיְהִי לְמַטֶּה בְּכַפּֽוֹ:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה שְׁלַח יָֽדְךָ וֶֽאֱחֹז בִּזְנָבוֹ וַיִּשְׁלַח יָדוֹ וַיַּֽחֲזֶק בּוֹ וַיְהִי לְמַטֶּה בְּכַפּֽוֹ:  ת  וַאֲמַר יְיָ לְמשֶׁה אוֹשִׁט יְדָךְ וְאֵחוֹד בְּזַנְבֵהּ וְאוֹשֵׁיט יְדֵהּ וְאַתְקֵף בֵּהּ וַהֲוָה לְחוּטְּרָא בִּידֵהּ:

{ה} לְמַעַן יַֽאֲמִינוּ כִּֽי נִרְאָה אֵלֶיךָ יְהוָֹה אֱלֹהֵי אֲבֹתָם אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵֽאלֹהֵי יַֽעֲקֹֽב:  מ  לְמַעַן יַֽאֲמִינוּ כִּֽי נִרְאָה אֵלֶיךָ יְהוָֹה אֱלֹהֵי אֲבֹתָם אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵֽאלֹהֵי יַֽעֲקֹֽב:  ת  בְּדִיל דִיהֵמְנוּן אֲרֵי אִתְגְלִי לָךְ יְיָ אֱלָהָא דְאַבְהַתְהוֹן אֱלָהֵהּ דְאַבְרָהָם אֱלָהֵהּ דְיִצְחָק וֵאֵלָהֵהּ דְיַעֲקֹב:

{ו} וַיֹּאמֶר יְהֹוָה לוֹ עוֹד הָֽבֵא נָא יָֽדְךָ בְּחֵיקֶךָ וַיָּבֵא יָדוֹ בְּחֵיקוֹ וַיּוֹצִאָהּ וְהִנֵּה יָדוֹ מְצֹרַעַת כַּשָּֽׁלֶג:  מ  וַיֹּאמֶר יְהֹוָה לוֹ עוֹד הָֽבֵא נָא יָֽדְךָ בְּחֵיקֶךָ וַיָּבֵא יָדוֹ בְּחֵיקוֹ וַיּוֹצִאָהּ וְהִנֵּה יָדוֹ מְצֹרַעַת כַּשָּֽׁלֶג:  ת  וַאֲמַר יְיָ לֵהּ עוֹד אָעֵיל כְּעַן יְדָךְ בְּעִטְּפָךְ וְאָעֵיל יְדֵהּ בְּעִטְפֵהּ וְאַפְקַהּ וְהָא יְדֵהּ חַוְרָא כְתַלְגָא:

{ז} וַיֹּאמֶר הָשֵׁב יָֽדְךָ אֶל חֵיקֶךָ וַיָּשֶׁב יָדוֹ אֶל חֵיקוֹ וַיּֽוֹצִאָהּ מֵֽחֵיקוֹ וְהִנֵּה שָׁבָה כִּבְשָׂרֽוֹ:  מ  וַיֹּאמֶר הָשֵׁב יָֽדְךָ אֶל חֵיקֶךָ וַיָּשֶׁב יָדוֹ אֶל חֵיקוֹ וַיּֽוֹצִאָהּ מֵֽחֵיקוֹ וְהִנֵּה שָׁבָה כִּבְשָׂרֽוֹ:  ת  וַאֲמַר אָתֵיב יְדָךְ לְעִטְפָךְ וַאֲתֵיב יְדֵהּ לְעִטְּפֵהּ וְאַפְקַהּ מֵעִטְפֵהּ וְהָא תָבַת הֲוַת כְּבִשְׂרֵהּ:

{ח} וְהָיָה אִם לֹא יַֽאֲמִינוּ לָךְ וְלֹא יִשְׁמְעוּ לְקֹל הָאֹת הָֽרִאשׁוֹן וְהֶֽאֱמִינוּ לְקֹל הָאֹת הָאַֽחֲרֽוֹן:  מ  וְהָיָה אִם לֹא יַֽאֲמִינוּ לָךְ וְלֹא יִשְׁמְעוּ לְקֹל הָאֹת הָֽרִאשׁוֹן וְהֶֽאֱמִינוּ לְקֹל הָאֹת הָאַֽחֲרֽוֹן:  ת  וִיהֵי אִם לָא יְהֵמְנוּן לָךְ וְלָא יְקַבְּלוּן לְקַל אָתָא קַדְמָאָה וִיהֵמְנוּן לְקַל אָתָא בַתְרָאָה:

{ט} וְהָיָה אִם לֹא יַֽאֲמִינוּ גַּם לִשְׁנֵי הָֽאֹתוֹת הָאֵלֶּה וְלֹא יִשְׁמְעוּן לְקֹלֶךָ וְלָֽקַחְתָּ מִמֵּימֵי הַיְאֹר וְשָֽׁפַכְתָּ הַיַּבָּשָׁה וְהָיוּ הַמַּיִם אֲשֶׁר תִּקַּח מִן הַיְאֹר וְהָיוּ לְדָם בַּיַּבָּֽשֶׁת:  מ  וְהָיָה אִם לֹא יַֽאֲמִינוּ גַּם לִשְׁנֵי הָֽאֹתוֹת הָאֵלֶּה וְלֹא יִשְׁמְעוּן לְקֹלֶךָ וְלָֽקַחְתָּ מִמֵּימֵי הַיְאֹר וְשָֽׁפַכְתָּ הַיַּבָּשָׁה וְהָיוּ הַמַּיִם אֲשֶׁר תִּקַּח מִן הַיְאֹר וְהָיוּ לְדָם בַּיַּבָּֽשֶׁת:  ת  וִיהֵי אִם לָא יְהֵמְנוּן אַף לִתְרֵין אָתַיָא הָאִלֵּין וְלָא יְקַבְּלוּן מִנָךְ וְתִסַב מִמַיָא דִבְנַהֲרָא וְתֵשׁוֹד לְּיַבֶּשְׁתָּא וִיהוֹן מַיָא דִי תִסַב מִן נַהֲרָא וִיהוֹן לִדְמָא בְּיַבֶּשְׁתָּא:

{י} וַיֹּאמֶר מֹשֶׁה אֶל יְהֹוָה בִּי אֲדֹנָי לֹא אִישׁ דְּבָרִים אָנֹכִי גַּם מִתְּמוֹל גַּם מִשִּׁלְשֹׁם גַּם מֵאָז דַּבֶּרְךָ אֶל עַבְדֶּךָ כִּי כְבַד פֶּה וּכְבַד לָשׁוֹן אָנֹֽכִי:  מ  וַיֹּאמֶר מֹשֶׁה אֶל יְהֹוָה בִּי אֲדֹנָי לֹא אִישׁ דְּבָרִים אָנֹכִי גַּם מִתְּמוֹל גַּם מִשִּׁלְשֹׁם גַּם מֵאָז דַּבֶּרְךָ אֶל עַבְדֶּךָ כִּי כְבַד פֶּה וּכְבַד לָשׁוֹן אָנֹֽכִי:  ת  וַאֲמַר משֶׁה קֳדָם יְיָ בְּבָעוּ יְיָ לָא גְבַר דְמִלוּל אֲנָא אַף מִתְּמַלִי אַף מִדְקָמוֹהִי אַף מֵעִדַן דְמַלֶלְתָּא עִם עַבְדָךְ אֲרֵי יַקִיר מַמְלָל וְעַמִיק לִישָׁן אֲנָא:

{יא} וַיֹּאמֶר יְהֹוָה אֵלָיו מִי שָׂם פֶּה לָֽאָדָם אוֹ מִֽי יָשׂוּם אִלֵּם אוֹ חֵרֵשׁ אוֹ פִקֵּחַ אוֹ עִוֵּר הֲלֹא אָֽנֹכִי יְהוָֹֽה:  מ  וַיֹּאמֶר יְהֹוָה אֵלָיו מִי שָׂם פֶּה לָֽאָדָם אוֹ מִֽי יָשׂוּם אִלֵּם אוֹ חֵרֵשׁ אוֹ פִקֵּחַ אוֹ עִוֵּר הֲלֹא אָֽנֹכִי יְהוָֹֽה:  ת  וַאֲמַר יְיָ לֵהּ מָן שַׁוִי פּוּמָא לַאֲנָשָׁא אוֹ מָן שַׁוִי אִלְמָא אוֹ חֵרְשָּׁא אוֹ פִקְחָא אוֹ עַוִירָא הֲלָא אֲנָא יְיָ:

{יב} וְעַתָּה לֵךְ וְאָֽנֹכִי אֶהְיֶה עִם פִּיךָ וְהֽוֹרֵיתִיךָ אֲשֶׁר תְּדַבֵּֽר:  מ  וְעַתָּה לֵךְ וְאָֽנֹכִי אֶהְיֶה עִם פִּיךָ וְהֽוֹרֵיתִיךָ אֲשֶׁר תְּדַבֵּֽר:  ת  וּכְעַן אֱזֵיל וּמֵימְרִי יְהֵא עִם פּוּמָךְ וְאַלֵפִנָךְ דִי תְמַלֵל:

{יג} וַיֹּאמֶר בִּי אֲדֹנָי שְׁלַח נָא בְּיַד תִּשְׁלָֽח:  מ  וַיֹּאמֶר בִּי אֲדֹנָי שְׁלַח נָא בְּיַד תִּשְׁלָֽח:  ת  וַאֲמַר בְּבָעוּ יְיָ שְׁלַח כְּעַן בְּיַד מָן דְכָשַׁר דְתִשְׁלַח:

{יד} וַיִּֽחַר אַף יְהֹוָה בְּמֹשֶׁה וַיֹּאמֶר הֲלֹא אַֽהֲרֹן אָחִיךָ הַלֵּוִי יָדַעְתִּי כִּֽי דַבֵּר יְדַבֵּר הוּא וְגַם הִנֵּה הוּא יֹצֵא לִקְרָאתֶךָ וְרָֽאֲךָ וְשָׂמַח בְּלִבּֽוֹ:  מ  וַיִּֽחַר אַף יְהֹוָה בְּמֹשֶׁה וַיֹּאמֶר הֲלֹא אַֽהֲרֹן אָחִיךָ הַלֵּוִי יָדַעְתִּי כִּֽי דַבֵּר יְדַבֵּר הוּא וְגַם הִנֵּה הוּא יֹצֵא לִקְרָאתֶךָ וְרָֽאֲךָ וְשָׂמַח בְּלִבּֽוֹ:  ת  וּתְקֵיף רוּגְזָא דַיְיָ בְּמשֶׁה וַאֲמַר הֲלָא אַהֲרֹן אָחוּךְ לֵוָאֵי גְלֵי קֳדָמַי אֲרֵי מַלָלָא יְמַלֵל הוּא וְאַף הָא הוּא נָפֵק לְקַדָמוּתָךְ וְיֶחֱזִנָךְ וְיֶחְדֵי בְלִבֵּהּ:

{טו} וְדִבַּרְתָּ אֵלָיו וְשַׂמְתָּ אֶת הַדְּבָרִים בְּפִיו וְאָֽנֹכִי אֶהְיֶה עִם פִּיךָ וְעִם פִּיהוּ וְהֽוֹרֵיתִי אֶתְכֶם אֵת אֲשֶׁר תַּֽעֲשֽׂוּן:  מ  וְדִבַּרְתָּ אֵלָיו וְשַׂמְתָּ אֶת הַדְּבָרִים בְּפִיו וְאָֽנֹכִי אֶהְיֶה עִם פִּיךָ וְעִם פִּיהוּ וְהֽוֹרֵיתִי אֶתְכֶם אֵת אֲשֶׁר תַּֽעֲשֽׂוּן:  ת  וּתְמַלֵל עִמֵהּ וּתְשַׁוִי יָת פִּתְגָמַיָא בְּפוּמֵהּ וּמֵימְרִי יְהֵי עִם פּוּמָךְ וְעִם פּוּמֵהּ וְאַלֵיף יָתְכוֹן יָת דִי תַעְבְּדוּן:

{טז} וְדִבֶּר הוּא לְךָ אֶל הָעָם וְהָיָה הוּא יִֽהְיֶה לְּךָ לְפֶה וְאַתָּה תִּֽהְיֶה לּוֹ לֵֽאלֹהִֽים:  מ  וְדִבֶּר הוּא לְךָ אֶל הָעָם וְהָיָה הוּא יִֽהְיֶה לְּךָ לְפֶה וְאַתָּה תִּֽהְיֶה לּוֹ לֵֽאלֹהִֽים:  ת  וִימַלֵל הוּא לָךְ לְעַמָא וִיהֵי הוּא יְהֵי לָךְ לִמְתֻרְגְמָן וְאַתְּ תְּהֵי לֵהּ לְרָב:

{יז} וְאֶת הַמַּטֶּה הַזֶּה תִּקַּח בְּיָדֶךָ אֲשֶׁר תַּֽעֲשֶׂה בּוֹ אֶת הָֽאֹתֹֽת:  מ  וְאֶת הַמַּטֶּה הַזֶּה תִּקַּח בְּיָדֶךָ אֲשֶׁר תַּֽעֲשֶׂה בּוֹ אֶת הָֽאֹתֹֽת:  ת  וְיָת חֻטְרָא הָדֵין תִּסַב בִּידָךְ דְתַעְבֵּד בֵּהּ יָת אָתַיָא: (פ)

{יח}  שישי  וַיֵּלֶךְ מֹשֶׁה וַיָּשָׁב אֶל יֶתֶר חֹֽתְנוֹ וַיֹּאמֶר לוֹ אֵלְכָה נָּא וְאָשׁוּבָה אֶל אַחַי אֲשֶׁר בְּמִצְרַיִם וְאֶרְאֶה הַֽעוֹדָם חַיִּים וַיֹּאמֶר יִתְרוֹ לְמֹשֶׁה לֵךְ לְשָׁלֽוֹם:  מ  וַיֵּלֶךְ מֹשֶׁה וַיָּשָׁב אֶל יֶתֶר חֹֽתְנוֹ וַיֹּאמֶר לוֹ אֵלְכָה נָּא וְאָשׁוּבָה אֶל אַחַי אֲשֶׁר בְּמִצְרַיִם וְאֶרְאֶה הַֽעוֹדָם חַיִּים וַיֹּאמֶר יִתְרוֹ לְמֹשֶׁה לֵךְ לְשָׁלֽוֹם:  ת  וַאֲזַל משֶׁה וְתָב לְוָת יֶתֶר חֲמוֹהִי וַאֲמַר לֵהּ אֵיזִיל כְּעַן וְאֵיתוּב לְוָת אַחַי דִי בְמִצְרַיִם וְאֶחֱזֵי הַעַד כְּעַן קַיָמִין וַאֲמַר יִתְרוֹ לְמשֶׁה אֱזֵיל לִשְׁלָם:

{יט} וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה בְּמִדְיָן לֵךְ שֻׁב מִצְרָיִם כִּי מֵתוּ כָּל הָאֲנָשִׁים הַֽמְבַקְשִׁים אֶת נַפְשֶֽׁךָ:  מ  וַיֹּאמֶר יְהוָֹה אֶל מֹשֶׁה בְּמִדְיָן לֵךְ שֻׁב מִצְרָיִם כִּי מֵתוּ כָּל הָאֲנָשִׁים הַֽמְבַקְשִׁים אֶת נַפְשֶֽׁךָ:  ת  וַאֲמַר יְיָ לְמשֶׁה בְּמִדְיָן אֱזֵיל תּוּב לְמִצְרָיִם אֲרֵי מִיתוּ כָּל גֻבְרַיָא דִבְעוֹ לְמִקְטְלָךְ:

{כ} וַיִּקַּח מֹשֶׁה אֶת אִשְׁתּוֹ וְאֶת בָּנָיו וַיַּרְכִּבֵם עַֽל הַֽחֲמֹר וַיָּשָׁב אַרְצָה מִצְרָיִם וַיִּקַּח מֹשֶׁה אֶת מַטֵּה הָֽאֱלֹהִים בְּיָדֽוֹ:  מ  וַיִּקַּח מֹשֶׁה אֶת אִשְׁתּוֹ וְאֶת בָּנָיו וַיַּרְכִּבֵם עַֽל הַֽחֲמֹר וַיָּשָׁב אַרְצָה מִצְרָיִם וַיִּקַּח מֹשֶׁה אֶת מַטֵּה הָֽאֱלֹהִים בְּיָדֽוֹ:  ת  וּדְבַר משֶׁה יָת אִתְּתֵהּ וְיָת בְּנוֹהִי וְאַרְכְּבִנוּן עַל חֲמָרָא וְתָב לְאַרְעָא דְמִצְרָיִם וּנְסִיב משֶׁה יָת חוּטְרָא דְאִתְעֲבִידוּ בֵּהּ נִסִין מִן קֳדָם יְיָ בִּידֵהּ:

{כא} וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה בְּלֶכְתְּךָ לָשׁוּב מִצְרַיְמָה רְאֵה כָּל הַמֹּֽפְתִים אֲשֶׁר שַׂמְתִּי בְיָדֶךָ וַֽעֲשִׂיתָם לִפְנֵי פַרְעֹה וַֽאֲנִי אֲחַזֵּק אֶת לִבּוֹ וְלֹא יְשַׁלַּח אֶת הָעָֽם:  מ  וַיֹּאמֶר יְהֹוָה אֶל מֹשֶׁה בְּלֶכְתְּךָ לָשׁוּב מִצְרַיְמָה רְאֵה כָּל הַמֹּֽפְתִים אֲשֶׁר שַׂמְתִּי בְיָדֶךָ וַֽעֲשִׂיתָם לִפְנֵי פַרְעֹה וַֽאֲנִי אֲחַזֵּק אֶת לִבּוֹ וְלֹא יְשַׁלַּח אֶת הָעָֽם:  ת  וַאֲמַר יְיָ לְמשֶׁה בִּמְהָכָךְ לְמִתּוּב לְמִצְרַיִם חֲזֵי כָּל מוֹפְתַיָא דִי שַׁוֵיתִי בִידָךְ וְתַעְבְּדִנוּן קֳדָם פַּרְעֹה וַאֲנָא אַתְקֵיף יָת לִבֵּהּ וְלָא יְשַּׁלַח יָת עַמָא:

{כב} וְאָֽמַרְתָּ אֶל פַּרְעֹה כֹּה אָמַר יְהֹוָה בְּנִי בְכֹרִי יִשְׂרָאֵֽל:  מ  וְאָֽמַרְתָּ אֶל פַּרְעֹה כֹּה אָמַר יְהֹוָה בְּנִי בְכֹרִי יִשְׂרָאֵֽל:  ת  וְתֵימַר לְוָת פַּרְעֹה כִּדְנַן אֲמַר יְיָ בְּרִי בוּכְרִי יִשְׂרָאֵל:

{כג} וָֽאֹמַר אֵלֶיךָ שַׁלַּח אֶת בְּנִי וְיַעַבְדֵנִי וַתְּמָאֵן לְשַׁלְּחוֹ הִנֵּה אָֽנֹכִי הֹרֵג אֶת בִּנְךָ בְּכֹרֶֽךָ:  מ  וָֽאֹמַר אֵלֶיךָ שַׁלַּח אֶת בְּנִי וְיַעַבְדֵנִי וַתְּמָאֵן לְשַׁלְּחוֹ הִנֵּה אָֽנֹכִי הֹרֵג אֶת בִּנְךָ בְּכֹרֶֽךָ:  ת  וַאֲמָרִית לָךְ שַׁלַח יָת בְּרִי וְיִפְלַח קֳדָמַי וְאִם סָרִיב אַתְּ לְשַׁלָחוּתֵהּ הָא אֲנָא קָטִיל יָת בְּרָךְ בּוּכְרָךְ:

{כד} וַיְהִי בַדֶּרֶךְ בַּמָּלוֹן וַיִּפְגְּשֵׁהוּ יְהֹוָה וַיְבַקֵּשׁ הֲמִיתֽוֹ:  מ  וַיְהִי בַדֶּרֶךְ בַּמָּלוֹן וַיִּפְגְּשֵׁהוּ יְהֹוָה וַיְבַקֵּשׁ הֲמִיתֽוֹ:  ת  וַהֲוָה בְאָרְחָא בְּבֵית מְבָתָא וַעֲרַע בֵּהּ מַלְאֲכָא דַיְיָ וּבְעָא לְמִקְטְלֵהּ:

{כה} וַתִּקַּח צִפֹּרָה צֹר וַתִּכְרֹת אֶת עָרְלַת בְּנָהּ וַתַּגַּע לְרַגְלָיו וַתֹּאמֶר כִּי חֲתַן דָּמִים אַתָּה לִֽי:  מ  וַתִּקַּח צִפֹּרָה צֹר וַתִּכְרֹת אֶת עָרְלַת בְּנָהּ וַתַּגַּע לְרַגְלָיו וַתֹּאמֶר כִּי חֲתַן דָּמִים אַתָּה לִֽי:  ת  וּנְסֵיבַת צִפּוֹרָה טִנָרָא וּגְזָרַת יָת עָרְלַת בְּרַהּ וּקְרִיבַת לָקֳדָמוֹהִי וַאֲמֶ רֶת אֲרֵי בִדְמָא דִמְהֻלְתָּא הָדֵין אִתְיְהֵב חַתְנָא לָנָא:

{כו} וַיִּרֶף מִמֶּנּוּ אָז אָֽמְרָה חֲתַן דָּמִים לַמּוּלֹֽת:  מ  וַיִּרֶף מִמֶּנּוּ אָז אָֽמְרָה חֲתַן דָּמִים לַמּוּלֹֽת:  ת  וְנָח מִנֵהּ בְּכֵן אֲמֶרֶת אִלוּלֵי דְמָא דִמְהֻלְתָּא הָדֵין אִתְחַיַב חַתְנָא קְטוֹל: (פ)

{כז} וַיֹּאמֶר יְהוָֹה אֶֽל אַֽהֲרֹן לֵךְ לִקְרַאת מֹשֶׁה הַמִּדְבָּרָה וַיֵּלֶךְ וַֽיִּפְגְּשֵׁהוּ בְּהַר הָֽאֱלֹהִים וַיִּשַּׁק לֽוֹ:  מ  וַיֹּאמֶר יְהוָֹה אֶֽל אַֽהֲרֹן לֵךְ לִקְרַאת מֹשֶׁה הַמִּדְבָּרָה וַיֵּלֶךְ וַֽיִּפְגְּשֵׁהוּ בְּהַר הָֽאֱלֹהִים וַיִּשַּׁק לֽוֹ:  ת  וַאֲמַר יְיָ לְאַהֲרֹן אֱזֵיל לְקַדָמוּת משֶׁה לְמַדְבְּרָא וַאֲזַל וְעַרְעֵהּ בְּטוּרָא דְאִתְגְלֵי עֲלוֹהִי יְקָרָא דַיְיָ וּנְשַׁק לֵהּ:

{כח} וַיַּגֵּד מֹשֶׁה לְאַהֲרֹן אֵת כָּל דִּבְרֵי יְהוָֹה אֲשֶׁר שְׁלָחוֹ וְאֵת כָּל הָֽאֹתֹת אֲשֶׁר צִוָּֽהוּ:  מ  וַיַּגֵּד מֹשֶׁה לְאַהֲרֹן אֵת כָּל דִּבְרֵי יְהוָֹה אֲשֶׁר שְׁלָחוֹ וְאֵת כָּל הָֽאֹתֹת אֲשֶׁר צִוָּֽהוּ:  ת  וְחַוִי משֶׁה לְאַהֲרֹן יָת כָּל פִּתְגָמַיָא דַיְיָ דִי שַׁלְחֵהּ וְיָת כָּל אָתַיָא דִי פַקְדֵהּ:

{כט} וַיֵּלֶךְ מֹשֶׁה וְאַֽהֲרֹן וַיַּאַסְפוּ אֶת כָּל זִקְנֵי בְּנֵי יִשְׂרָאֵֽל:  מ  וַיֵּלֶךְ מֹשֶׁה וְאַֽהֲרֹן וַיַּאַסְפוּ אֶת כָּל זִקְנֵי בְּנֵי יִשְׂרָאֵֽל:  ת  וַאֲזַל משֶׁה וְאַהֲרֹן וּכְנָשׁוּ יָת כָּל סָבֵי בְּנֵי יִשְׂרָאֵל:

{ל} וַיְדַבֵּר אַֽהֲרֹן אֵת כָּל הַדְּבָרִים אֲשֶׁר דִּבֶּר יְהוָֹה אֶל מֹשֶׁה וַיַּעַשׂ הָֽאֹתֹת לְעֵינֵי הָעָֽם:  מ  וַיְדַבֵּר אַֽהֲרֹן אֵת כָּל הַדְּבָרִים אֲשֶׁר דִּבֶּר יְהוָֹה אֶל מֹשֶׁה וַיַּעַשׂ הָֽאֹתֹת לְעֵינֵי הָעָֽם:  ת  וּמַלִיל אַהֲרֹן יָת כָּל פִּתְגָמַיָא דִי מַלִיל יְיָ עִם משֶׁה וַעֲבַד אָתַיָא לְעֵינֵי עַמָא:

{לא} וַֽיַּֽאֲמֵן הָעָם וַֽיִּשְׁמְעוּ כִּֽי פָקַד יְהֹוָה אֶת בְּנֵי יִשְׂרָאֵל וְכִי רָאָה אֶת עָנְיָם וַֽיִּקְּדוּ וַיִּֽשְׁתַּֽחֲוֽוּ:  מ  וַֽיַּֽאֲמֵן הָעָם וַֽיִּשְׁמְעוּ כִּֽי פָקַד יְהֹוָה אֶת בְּנֵי יִשְׂרָאֵל וְכִי רָאָה אֶת עָנְיָם וַֽיִּקְּדוּ וַיִּֽשְׁתַּֽחֲוֽוּ:  ת  וְהֵימִן עַמָא וּשְׁמָעוּ אֲרֵי דְכִיר יְיָ יָת בְּנֵי יִשְׂרָאֵל וַאֲרֵי גְלֵי קֳדָמוֹהִי יָת שִׁעְבּוּדְהוֹן וּכְרָעוּ וּסְגִידוּ:




שמות פרק-ה

{א}  שביעי  וְאַחַר בָּאוּ מֹשֶׁה וְאַֽהֲרֹן וַיֹּֽאמְרוּ אֶל פַּרְעֹה כֹּֽה אָמַר יְהוָֹה אֱלֹהֵי יִשְׂרָאֵל שַׁלַּח אֶת עַמִּי וְיָחֹגּוּ לִי בַּמִּדְבָּֽר:  מ  וְאַחַר בָּאוּ מֹשֶׁה וְאַֽהֲרֹן וַיֹּֽאמְרוּ אֶל פַּרְעֹה כֹּֽה אָמַר יְהוָֹה אֱלֹהֵי יִשְׂרָאֵל שַׁלַּח אֶת עַמִּי וְיָחֹגּוּ לִי בַּמִּדְבָּֽר:  ת  וּבָתַר כֵּן עַלוּ משֶׁה וְאַהֲרֹן וַאֲמָרוּ לְפַרְעֹה כִּדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל שַׁלַח יָת עַמִי וִיחַגוּן קֳדָמַי בְּמַדְבְּרָא:

{ב} וַיֹּאמֶר פַּרְעֹה מִי יְהוָֹה אֲשֶׁר אֶשְׁמַע בְּקֹלוֹ לְשַׁלַּח אֶת יִשְׂרָאֵל לֹא יָדַעְתִּי אֶת יְהֹוָה וְגַם אֶת יִשְׂרָאֵל לֹא אֲשַׁלֵּֽחַ:  מ  וַיֹּאמֶר פַּרְעֹה מִי יְהוָֹה אֲשֶׁר אֶשְׁמַע בְּקֹלוֹ לְשַׁלַּח אֶת יִשְׂרָאֵל לֹא יָדַעְתִּי אֶת יְהֹוָה וְגַם אֶת יִשְׂרָאֵל לֹא אֲשַׁלֵּֽחַ:  ת  וַאֲמַר פַּרְעֹה שְׁמָא דַיְיָ לָא אִתְגְלִי לִי דִי אֲקַבֵּל לְמֵמְרֵהּ לְשַׁלָחָא יָת יִשְׂרָאֵל לָא אִתְגְלִי לִי שְׁמָא דַיְיָ וְאַף יָת יִשְׂרָאֵל לָא אֲשַׁלַח:

{ג} וַיֹּאמְרוּ אֱלֹהֵי הָֽעִבְרִים נִקְרָא עָלֵינוּ נֵֽלְכָה נָּא דֶּרֶךְ שְׁלשֶׁת יָמִים בַּמִּדְבָּר וְנִזְבְּחָה לַֽיהוָֹה אֱלֹהֵינוּ פֶּן יִפְגָּעֵנוּ בַּדֶּבֶר אוֹ בֶחָֽרֶב:  מ  וַיֹּאמְרוּ אֱלֹהֵי הָֽעִבְרִים נִקְרָא עָלֵינוּ נֵֽלְכָה נָּא דֶּרֶךְ שְׁלשֶׁת יָמִים בַּמִּדְבָּר וְנִזְבְּחָה לַֽיהוָֹה אֱלֹהֵינוּ פֶּן יִפְגָּעֵנוּ בַּדֶּבֶר אוֹ בֶחָֽרֶב:  ת  וַאֲמָרוּ אֱלָהָא דִיהוּדָאֵי אִתְגְלִי עֲלָנָא נֵיזֵיל כְּעַן מַהֲלַךְ תְּלָתָא יוֹמִין בְּמַדְבְּרָא וּנְדַבַּח קֳדָם יְיָ אֱלָהָנָא דִלְמָא יְעַרְעִנָנָא בְּמוֹתָא אוֹ בִקְטוֹל:

{ד} וַיֹּאמֶר אֲלֵהֶם מֶלֶךְ מִצְרַיִם לָמָּה מֹשֶׁה וְאַֽהֲרֹן תַּפְרִיעוּ אֶת הָעָם מִֽמַּֽעֲשָׂיו לְכוּ לְסִבְלֹֽתֵיכֶֽם:  מ  וַיֹּאמֶר אֲלֵהֶם מֶלֶךְ מִצְרַיִם לָמָּה מֹשֶׁה וְאַֽהֲרֹן תַּפְרִיעוּ אֶת הָעָם מִֽמַּֽעֲשָׂיו לְכוּ לְסִבְלֹֽתֵיכֶֽם:  ת  וַאֲמַר לְהוֹן מַלְכָּא דְמִצְרַיִם לְמָא משֶׁה וְאַהֲרֹן תְּבַטְלוּן יָת עַמָא מֵעוֹבָדֵהוֹן אֱזִילוּ לְפָלְחָנְכוֹן:

{ה} וַיֹּאמֶר פַּרְעֹה הֵן רַבִּים עַתָּה עַם הָאָרֶץ וְהִשְׁבַּתֶּם אֹתָם מִסִּבְלֹתָֽם:  מ  וַיֹּאמֶר פַּרְעֹה הֵן רַבִּים עַתָּה עַם הָאָרֶץ וְהִשְׁבַּתֶּם אֹתָם מִסִּבְלֹתָֽם:  ת  וַאֲמַר פַּרְעֹה הָא סַגִיאִין כְּעַן עַמָא דְאַרְעָא וּתְבַטְלוּן יָתְהוֹן מִפָּלְחָנְהוֹן:

{ו} וַיְצַו פַּרְעֹה בַּיּוֹם הַהוּא אֶת הַנֹּֽגְשִׂים בָּעָם וְאֶת שֹֽׁטְרָיו לֵאמֹֽר:  מ  וַיְצַו פַּרְעֹה בַּיּוֹם הַהוּא אֶת הַנֹּֽגְשִׂים בָּעָם וְאֶת שֹֽׁטְרָיו לֵאמֹֽר:  ת  וּפַקִיד פַּרְעֹה בְּיוֹמָא הַהוּא יָת שִׁלְטוֹנִין דְעַמָא וְיָת סָרְכוֹהִי לְמֵימָר:

{ז} לֹא תֹֽאסִפוּן לָתֵת תֶּבֶן לָעָם לִלְבֹּן הַלְּבֵנִים כִּתְמוֹל שִׁלְשֹׁם הֵם יֵֽלְכוּ וְקֽשְׁשׁוּ לָהֶם תֶּֽבֶן:  מ  לֹא תֹֽאסִפוּן לָתֵת תֶּבֶן לָעָם לִלְבֹּן הַלְּבֵנִים כִּתְמוֹל שִׁלְשֹׁם הֵם יֵֽלְכוּ וְקֽשְׁשׁוּ לָהֶם תֶּֽבֶן:  ת  לָא תוֹסְפוּן לְמִיתַּן תִּבְנָא לְעַמָא לְמִרְמֵי לִבְנִין כְּמִתְּמַלֵי וּמִדְקָמוֹהִי אִנוּן יְהָכוּן וִיגַבְּבוּן לְהוֹן תִּבְנָא:

{ח} וְאֶת מַתְכֹּנֶת הַלְּבֵנִים אֲשֶׁר הֵם עֹשִׂים תְּמוֹל שִׁלְשֹׁם תָּשִׂימוּ עֲלֵיהֶם לֹא תִגְרְעוּ מִמֶּנּוּ כִּֽי נִרְפִּים הֵם עַל כֵּן הֵם צֹֽעֲקִים לֵאמֹר נֵֽלְכָה נִזְבְּחָה לֵֽאלֹהֵֽינוּ:  מ  וְאֶת מַתְכֹּנֶת הַלְּבֵנִים אֲשֶׁר הֵם עֹשִׂים תְּמוֹל שִׁלְשֹׁם תָּשִׂימוּ עֲלֵיהֶם לֹא תִגְרְעוּ מִמֶּנּוּ כִּֽי נִרְפִּים הֵם עַל כֵּן הֵם צֹֽעֲקִים לֵאמֹר נֵֽלְכָה נִזְבְּחָה לֵֽאלֹהֵֽינוּ:  ת  וְיָת סְכוֹם לְבֵנַיָא דִי אִנוּן עָבְדִין מִתְּמַלֵי וּמִדְקָמוֹהִי תְּמַנוּן עֲלֵיהוֹן לָא תִמְנְעוּן מִנֵהּ אֲרֵי בַטְלָנִין אִנוּן עַל כֵּן אִנוּן צָוְחִין לְמֵימַר נֵיזֵיל נְדַבַּח קֳדָם אֱלָהָנָא:

{ט} תִּכְבַּד הָֽעֲבֹדָה עַל הָֽאֲנָשִׁים וְיַֽעֲשׂוּ בָהּ וְאַל יִשְׁעוּ בְּדִבְרֵי שָֽׁקֶר:  מ  תִּכְבַּד הָֽעֲבֹדָה עַל הָֽאֲנָשִׁים וְיַֽעֲשׂוּ בָהּ וְאַל יִשְׁעוּ בְּדִבְרֵי שָֽׁקֶר:  ת  תִּתָּקַף פָּלְחָנָא עַל גֻבְרַיָא וְיִתְעַסְקוּן בַּהּ וְלָא יִתְעַסְקוּן בְּפִתְגָמִ ין בְּטֵלִין:

{י} וַיֵּצְאוּ נֹֽגְשֵׂי הָעָם וְשֹׁטְרָיו וַיֹּֽאמְרוּ אֶל הָעָם לֵאמֹר כֹּה אָמַר פַּרְעֹה אֵינֶנִּי נֹתֵן לָכֶם תֶּֽבֶן:  מ  וַיֵּצְאוּ נֹֽגְשֵׂי הָעָם וְשֹׁטְרָיו וַיֹּֽאמְרוּ אֶל הָעָם לֵאמֹר כֹּה אָמַר פַּרְעֹה אֵינֶנִּי נֹתֵן לָכֶם תֶּֽבֶן:  ת  וּנְפָקוּ שִׁלְטוֹנֵי עַמָא וְסָרְכוֹהִי וַאֲמָרוּ לְעַמָא לְמֵימָר כִּדְנַן אֲמַר פַּרְעֹה לֵית אֲנָא יָהֵב לְכוֹן תִּבְנָא:

{יא} אַתֶּם לְכוּ קְחוּ לָכֶם תֶּבֶן מֵֽאֲשֶׁר תִּמְצָאוּ כִּי אֵין נִגְרָע מֵֽעֲבֹֽדַתְכֶם דָּבָֽר:  מ  אַתֶּם לְכוּ קְחוּ לָכֶם תֶּבֶן מֵֽאֲשֶׁר תִּמְצָאוּ כִּי אֵין נִגְרָע מֵֽעֲבֹֽדַתְכֶם דָּבָֽר:  ת  אַתּוּן אֱזִילוּ סִיבוּ לְכוֹן תִּבְנָא מֵאֲתַר דְתַשְׁכְּחוּן אֲרֵי לָא יִתִּמְנַע מִפָּלְחָנְכוֹן מִדָעַם:

{יב} וַיָּפֶץ הָעָם בְּכָל אֶרֶץ מִצְרָיִם לְקשֵׁשׁ קַשׁ לַתֶּֽבֶן:  מ  וַיָּפֶץ הָעָם בְּכָל אֶרֶץ מִצְרָיִם לְקשֵׁשׁ קַשׁ לַתֶּֽבֶן:  ת  וְאִתְבַּדַר עַמָא בְּכָל אַרְעָא דְמִצְרָיִם לְגַבָּבָא גִילֵי לְתִבְנָא:

{יג} וְהַנֹּֽגְשִׂים אָצִים לֵאמֹר כַּלּוּ מַֽעֲשֵׂיכֶם דְּבַר יוֹם בְּיוֹמוֹ כַּֽאֲשֶׁר בִּהְיוֹת הַתֶּֽבֶן:  מ  וְהַנֹּֽגְשִׂים אָצִים לֵאמֹר כַּלּוּ מַֽעֲשֵׂיכֶם דְּבַר יוֹם בְּיוֹמוֹ כַּֽאֲשֶׁר בִּהְיוֹת הַתֶּֽבֶן:  ת  וְשִׁלְטוֹנַיָא דָחֲקִין לְמֵימָר אַשְׁלִימוּ עוֹבָדֵיכוֹן פִּתְגַם יוֹם בְּיוֹמֵהּ כְּמָא דַהֲוֵיתוּן עָבְדִין כַּד אִתְיְהֵב לְכוֹן תִּבְנָא:

{יד} וַיֻּכּוּ שֹֽׁטְרֵי בְּנֵי יִשְׂרָאֵל אֲשֶׁר שָׂמוּ עֲלֵהֶם נֹֽגְשֵׂי פַרְעֹה לֵאמֹר מַדּוּעַ לֹא כִלִּיתֶם חָקְכֶם לִלְבֹּן כִּתְמוֹל שִׁלְשֹׁם גַּם תְּמוֹל גַּם הַיּֽוֹם:  מ  וַיֻּכּוּ שֹֽׁטְרֵי בְּנֵי יִשְׂרָאֵל אֲשֶׁר שָׂמוּ עֲלֵהֶם נֹֽגְשֵׂי פַרְעֹה לֵאמֹר מַדּוּעַ לֹא כִלִּיתֶם חָקְכֶם לִלְבֹּן כִּתְמוֹל שִׁלְשֹׁם גַּם תְּמוֹל גַּם הַיּֽוֹם:  ת  וּלְקוֹ סָרְכֵי בְּנֵי יִשְׂרָאֵל דִי מַנִיאוּ עֲלֵיהוֹן שִׁלְטוֹנֵי פַרְעֹה לְמֵימָר מָא דֵין לָא אַשְׁלֶמְתּוּן גְזֵרָתְכוֹן לְמִרְמֵי (לִבְנִין) כְּמֵאִתְמַלֵי וּמִדְקָמוֹהִי אַף תְּמַלֵי אַף יוֹמָא דֵין:

{טו} וַיָּבֹאוּ שֹֽׁטְרֵי בְּנֵי יִשְׂרָאֵל וַיִּצְעֲקוּ אֶל פַּרְעֹה לֵאמֹר לָמָּה תַֽעֲשֶׂה כֹה לַֽעֲבָדֶֽיךָ:  מ  וַיָּבֹאוּ שֹֽׁטְרֵי בְּנֵי יִשְׂרָאֵל וַיִּצְעֲקוּ אֶל פַּרְעֹה לֵאמֹר לָמָּה תַֽעֲשֶׂה כֹה לַֽעֲבָדֶֽיךָ:  ת  וַאֲתוֹ סָרְכֵי בְּנֵי יִשְׂרָאֵל וּצְוָחוּ (קֳדָם) לְפַרְעֹה לְמֵימָר לְמָא תַעְבֵּד כְּדֵין לְעַבְדָיךְ:

{טז} תֶּבֶן אֵין נִתָּן לַֽעֲבָדֶיךָ וּלְבֵנִים אֹֽמְרִים לָנוּ עֲשּׁוּ וְהִנֵּה עֲבָדֶיךָ מֻכִּים וְחָטָאת עַמֶּֽךָ:  מ  תֶּבֶן אֵין נִתָּן לַֽעֲבָדֶיךָ וּלְבֵנִים אֹֽמְרִים לָנוּ עֲשּׁוּ וְהִנֵּה עֲבָדֶיךָ מֻכִּים וְחָטָאת עַמֶּֽךָ:  ת  תִּבְנָא לָא מִתְיְהֵב לְעַבְדָיךְ וּלְבֵנַיָא אָמְרִין לָנָא עִבִידוּ וְהָא עַבְדָיךְ לָקָן וְחָטָן עֲלֵיהוֹן עַמָךְ:

{יז} וַיֹּאמֶר נִרְפִּים אַתֶּם נִרְפִּים עַל כֵּן אַתֶּם אֹֽמְרִים נֵֽלְכָה נִזְבְּחָה לַֽיהוָֹֽה:  מ  וַיֹּאמֶר נִרְפִּים אַתֶּם נִרְפִּים עַל כֵּן אַתֶּם אֹֽמְרִים נֵֽלְכָה נִזְבְּחָה לַֽיהוָֹֽה:  ת  וַאֲמַר בַּטְלָנִין אַתּוּן בַּטְלָנִין עַל כֵּן אַתּוּן אָמְרִין נֵזֵיל נְדַבַּח קֳדָם יְיָ:

{יח} וְעַתָּה לְכוּ עִבְדוּ וְתֶבֶן לֹֽא יִנָּתֵן לָכֶם וְתֹכֶן לְבֵנִים תִּתֵּֽנוּ:  מ  וְעַתָּה לְכוּ עִבְדוּ וְתֶבֶן לֹֽא יִנָּתֵן לָכֶם וְתֹכֶן לְבֵנִים תִּתֵּֽנוּ:  ת  וּכְעַן אֱזִילוּ פְלָחוּ וְתִבְנָא לָא יִתְיְהֵב לְכוֹן וּסְכוֹם לְבֵינַיָא תִּתְּנוּן:

{יט} וַיִּרְאוּ שֹֽׁטְרֵי בְנֵֽי יִשְׂרָאֵל אֹתָם בְּרָע לֵאמֹר לֹֽא תִגְרְעוּ מִלִּבְנֵיכֶם דְּבַר יוֹם בְּיוֹמֽוֹ:  מ  וַיִּרְאוּ שֹֽׁטְרֵי בְנֵֽי יִשְׂרָאֵל אֹתָם בְּרָע לֵאמֹר לֹֽא תִגְרְעוּ מִלִּבְנֵיכֶם דְּבַר יוֹם בְּיוֹמֽוֹ:  ת  וַחֲזוֹ סָרְכֵי בְנֵי יִשְׂרָאֵל יָתְהוֹן בְּבִישׁ לְמֵימָר לָא תִמְנְעוּן מִלִבְנֵיכוֹן פִּתְגַם יוֹם בְּיוֹמֵהּ:

{כ} וַֽיִּפְגְּעוּ אֶת מֹשֶׁה וְאֶֽת אַֽהֲרֹן נִצָּבִים לִקְרָאתָם בְּצֵאתָם מֵאֵת פַּרְעֹֽה:  מ  וַֽיִּפְגְּעוּ אֶת מֹשֶׁה וְאֶֽת אַֽהֲרֹן נִצָּבִים לִקְרָאתָם בְּצֵאתָם מֵאֵת פַּרְעֹֽה:  ת  וַעֲרָעוּ יָת משֶׁה וְיָת אַהֲּרֹן קָיְמִין לְקַדָמוּתְהוֹן בְּמִפַּקְהוֹן מִקֳדָם פַּרְעֹה:

{כא} וַיֹּֽאמְרוּ אֲלֵהֶם יֵרֶא יְהוָֹה עֲלֵיכֶם וְיִשְׁפֹּט אֲשֶׁר הִבְאַשְׁתֶּם אֶת רֵיחֵנוּ בְּעֵינֵי פַרְעֹה וּבְעֵינֵי עֲבָדָיו לָֽתֶת חֶרֶב בְּיָדָם לְהָרְגֵֽנוּ:  מ  וַיֹּֽאמְרוּ אֲלֵהֶם יֵרֶא יְהוָֹה עֲלֵיכֶם וְיִשְׁפֹּט אֲשֶׁר הִבְאַשְׁתֶּם אֶת רֵיחֵנוּ בְּעֵינֵי פַרְעֹה וּבְעֵינֵי עֲבָדָיו לָֽתֶת חֶרֶב בְּיָדָם לְהָרְגֵֽנוּ:  ת  וַאֲמָרוּ לְהוֹן יִתְגְלֵי יְיָ עֲלֵיכוֹן וְיִתְפְּרַע דִי אַבְאֶשְׁתּוּן יָת רֵיחָנָא בְּעֵינֵי פַרְעֹה וּבְעֵינֵי עַבְדוֹהִי לְמִתַּן חַרְבָּא בִּידֵיהוֹן לְקַטָלוּתָנָא:

{כב}  מפטיר  וַיָּשָׁב מֹשֶׁה אֶל יְהוָֹה וַיֹּאמַר אֲדֹנָי לָמָה הֲרֵעֹתָה לָעָם הַזֶּה לָמָּה זֶּה שְׁלַחְתָּֽנִי:  מ  וַיָּשָׁב מֹשֶׁה אֶל יְהוָֹה וַיֹּאמַר אֲדֹנָי לָמָה הֲרֵעֹתָה לָעָם הַזֶּה לָמָּה זֶּה שְׁלַחְתָּֽנִי:  ת  וְתָב משֶׁה קֳדָם יְיָ וַאֲמַר יְיָ לְמָא אַבְאֵשְׁתָּא לְעַמָא הָדֵין וּלְמָא דְנַן שְׁלַחְתָּנִי:

{כג} וּמֵאָז בָּאתִי אֶל פַּרְעֹה לְדַבֵּר בִּשְׁמֶךָ הֵרַע לָעָם הַזֶּה וְהַצֵּל לֹֽא הִצַּלְתָּ אֶת עַמֶּֽךָ:  מ  וּמֵאָז בָּאתִי אֶל פַּרְעֹה לְדַבֵּר בִּשְׁמֶךָ הֵרַע לָעָם הַזֶּה וְהַצֵּל לֹֽא הִצַּלְתָּ אֶת עַמֶּֽךָ:  ת  וּמֵעִדַן עַלִית לְוָת פַּרְעֹה לְמַלָלָא בִשְׁמָךְ אַבְאֵשׁ לְעַמָּא הָדֵין וְשֵׁזָבָא לָא שֵׁזַבְתָּא יָת עַמָךְ:
