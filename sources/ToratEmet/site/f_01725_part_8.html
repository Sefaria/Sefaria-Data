פרשת פינחס


{י} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{יא} פִּֽינְחָ֨ס בֶּן-אֶלְעָזָ֜ר בֶּן-אַֽהֲרֹ֣ן הַכֹּהֵ֗ן הֵשִׁ֤יב אֶת-חֲמָתִי֙ מֵעַ֣ל בְּנֵֽי-יִשְׂרָאֵ֔ל בְּקַנְא֥וֹ אֶת-קִנְאָתִ֖י בְּתוֹכָ֑ם וְלֹֽא-כִלִּ֥יתִי אֶת-בְּנֵֽי-יִשְׂרָאֵ֖ל בְּקִנְאָתִֽי:  מ  פִּֽינְחָ֨ס בֶּן-אֶלְעָזָ֜ר בֶּן-אַֽהֲרֹ֣ן הַכֹּהֵ֗ן הֵשִׁ֤יב אֶת-חֲמָתִי֙ מֵעַ֣ל בְּנֵֽי-יִשְׂרָאֵ֔ל בְּקַנְא֥וֹ אֶת-קִנְאָתִ֖י בְּתוֹכָ֑ם וְלֹֽא-כִלִּ֥יתִי אֶת-בְּנֵֽי-יִשְׂרָאֵ֖ל בְּקִנְאָתִֽי:  ת  פִּינְחָס בַּר אֶלְעָזָר בַּר אַהֲרֹן כַּהֲנָא אֲתֵיב יַת חֵמְתִי מֵעַל בְּנֵי יִשְׂרָאֵל בִּדְקַנֵי יַת קִנְאָתִי בֵּינֵיהוֹן וְלָא שֵׁיצֵיתִי יַת בְּנֵי יִשְׂרָאֵל בְּקִנְאָתִי:

{יב} לָכֵ֖ן אֱמֹ֑ר הִנְנִ֨י נֹתֵ֥ן ל֛וֹ אֶת-בְּרִיתִ֖י שָׁלֽם:  מ  לָכֵ֖ן אֱמֹ֑ר הִנְנִ֨י נֹתֵ֥ן ל֛וֹ אֶת-בְּרִיתִ֖י שָׁלֽם:  ת  בְּכֵן אֵימַר הָא אֲנָא גְזַר לֵיהּ יַת קְיָמִי שְׁלָם:

{יג} וְהָ֤יְתָה לּוֹ֙ וּלְזַרְע֣וֹ אַֽחֲרָ֔יו בְּרִ֖ית כְּהֻנַּ֣ת עוֹלָ֑ם תַּ֗חַת אֲשֶׁ֤ר קִנֵּא֙ לֵֽאלֹהָ֔יו וַיְכַפֵּ֖ר עַל-בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וְהָ֤יְתָה לּוֹ֙ וּלְזַרְע֣וֹ אַֽחֲרָ֔יו בְּרִ֖ית כְּהֻנַּ֣ת עוֹלָ֑ם תַּ֗חַת אֲשֶׁ֤ר קִנֵּא֙ לֵֽאלֹהָ֔יו וַיְכַפֵּ֖ר עַל-בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וּתְהֵי לֵיהּ וְלִבְנוֹהִי בַּתְרוֹהִי קְיַם כְּהֻנַת עֲלָם חֲלַף דִי קַנֵי קֳדָם אֱלָהֵהּ וְכַפֵּר עַל בְּנֵי יִשְׂרָאֵל:

{יד} וְשֵׁם֩ אִ֨ישׁ יִשְׂרָאֵ֜ל הַמֻּכֶּ֗ה אֲשֶׁ֤ר הֻכָּה֙ אֶת-הַמִּדְיָנִ֔ית זִמְרִ֖י בֶּן-סָל֑וּא נְשִׂ֥יא בֵֽית-אָ֖ב לַשִּׁמְעֹנִֽי:  מ  וְשֵׁם֩ אִ֨ישׁ יִשְׂרָאֵ֜ל הַמֻּכֶּ֗ה אֲשֶׁ֤ר הֻכָּה֙ אֶת-הַמִּדְיָנִ֔ית זִמְרִ֖י בֶּן-סָל֑וּא נְשִׂ֥יא בֵֽית-אָ֖ב לַשִּׁמְעֹנִֽי:  ת  וְשׁוּם גַבְרָא בַּר יִשְרָאֵל קְטִילָא דִי אִתְקְטֵל עִם מִדְיָנֵיתָא זִמְרִי בַּר סָלוּא רַב בֵּית אַבָּא לְבֵית שִׁמְעוֹן:

{טו} וְשֵׁ֨ם הָֽאִשָּׁ֧ה הַמֻּכָּ֛ה הַמִּדְיָנִ֖ית כָּזְבִּ֣י בַת-צ֑וּר רֹ֣אשׁ אֻמּ֥וֹת בֵּֽית-אָ֛ב בְּמִדְיָ֖ן הֽוּא:  מ  וְשֵׁ֨ם הָֽאִשָּׁ֧ה הַמֻּכָּ֛ה הַמִּדְיָנִ֖ית כָּזְבִּ֣י בַת-צ֑וּר רֹ֣אשׁ אֻמּ֥וֹת בֵּֽית-אָ֛ב בְּמִדְיָ֖ן הֽוּא:  ת  וְשׁוּם אִתְּתָא דְאִתְקְטִילַת מִדְיָנֵיתָא כָּזְבִּי בַּת צוּר רֵישׁ אוּמֵי בֵּית אַבָּא בְּמִדְיָן הוּא: (פ)

{טז} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{יז} צָר֖וֹר אֶת-הַמִּדְיָנִ֑ים וְהִכִּיתֶ֖ם אוֹתָֽם:  מ  צָר֖וֹר אֶת-הַמִּדְיָנִ֑ים וְהִכִּיתֶ֖ם אוֹתָֽם:  ת  אָעֵק יַת מִדְיָנָאֵי וְתִקְטְלוּן יָתְהוֹן:

{יח} כִּ֣י צֹֽרְרִ֥ים הֵם֙ לָכֶ֔ם בְּנִכְלֵיהֶ֛ם אֲשֶׁר-נִכְּל֥וּ לָכֶ֖ם עַל-דְּבַר-פְּע֑וֹר וְעַל-דְּבַ֞ר כָּזְבִּ֨י בַת-נְשִׂ֤יא מִדְיָן֙ אֲחֹתָ֔ם הַמֻּכָּ֥ה בְיוֹם-הַמַּגֵּפָ֖ה עַל-דְּבַר-פְּעֽוֹר:  מ  כִּ֣י צֹֽרְרִ֥ים הֵם֙ לָכֶ֔ם בְּנִכְלֵיהֶ֛ם אֲשֶׁר-נִכְּל֥וּ לָכֶ֖ם עַל-דְּבַר-פְּע֑וֹר וְעַל-דְּבַ֞ר כָּזְבִּ֨י בַת-נְשִׂ֤יא מִדְיָן֙ אֲחֹתָ֔ם הַמֻּכָּ֥ה בְיוֹם-הַמַּגֵּפָ֖ה עַל-דְּבַר-פְּעֽוֹר:  ת  אֲרֵי מְעִיקִין אִינוּן לְכוֹן בְּנִכְלֵיהוֹן דִי נְכִילוּ לְכוֹן עַל עֵיסַק פְּעוֹר וְעַל עֵיסַק כָּזְבִּי בַּת רַבָּא דְמִדְיָן אֲחַתְהוֹן דְאִתְקְטִילַת בְּיוֹמָא דְמוֹתָנָא עַל עֵיסַק פְּעוֹר:




במדבר פרק-כו

{א}  וַיְהִ֖י אַֽחֲרֵ֣י הַמַּגֵּפָ֑ה וַיֹּ֤אמֶר יְהֹוָה֙ אֶל מֹשֶׁ֔ה וְאֶ֧ל אֶלְעָזָ֛ר בֶּן אַֽהֲרֹ֥ן הַכֹּהֵ֖ן לֵאמֹֽר:  מ   וַיְהִ֖י אַֽחֲרֵ֣י הַמַּגֵּפָ֑ה וַיֹּ֤אמֶר יְהֹוָה֙ אֶל מֹשֶׁ֔ה וְאֶ֧ל אֶלְעָזָ֛ר בֶּן אַֽהֲרֹ֥ן הַכֹּהֵ֖ן לֵאמֹֽר:  ת   וַהֲוָה בָּתַר מוֹתָנָא וַאֲמַר יְיָ לְמשֶׁה וּלְאֶלְעָזָר בַּר אַהֲרֹן כַּהֲנָא לְמֵימָר:

{ב} שְׂא֞וּ אֶת-רֹ֣אשׁ | כָּל-עֲדַ֣ת בְּנֵֽי-יִשְׂרָאֵ֗ל מִבֶּ֨ן עֶשְׂרִ֥ים שָׁנָ֛ה וָמַ֖עְלָה לְבֵ֣ית אֲבֹתָ֑ם כָּל-יֹצֵ֥א צָבָ֖א בְּיִשְׂרָאֵֽל:  מ  שְׂא֞וּ אֶת-רֹ֣אשׁ | כָּל-עֲדַ֣ת בְּנֵֽי-יִשְׂרָאֵ֗ל מִבֶּ֨ן עֶשְׂרִ֥ים שָׁנָ֛ה וָמַ֖עְלָה לְבֵ֣ית אֲבֹתָ֑ם כָּל-יֹצֵ֥א צָבָ֖א בְּיִשְׂרָאֵֽל:  ת  קַבִּילוּ יַת חוּשְׁבַּן כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל מִבַּר עַשְׂרִין שְׁנִין וּלְעֵילָא לְבֵית אֲבָהַתְהוֹן כָּל נָפֵיק חֵילָא בְּיִשְׂרָאֵל:

{ג} וַיְדַבֵּ֨ר מֹשֶׁ֜ה וְאֶלְעָזָ֧ר הַכֹּהֵ֛ן אֹתָ֖ם בְּעַֽרְבֹ֣ת מוֹאָ֑ב עַל-יַרְדֵּ֥ן יְרֵח֖וֹ לֵאמֹֽר:  מ  וַיְדַבֵּ֨ר מֹשֶׁ֜ה וְאֶלְעָזָ֧ר הַכֹּהֵ֛ן אֹתָ֖ם בְּעַֽרְבֹ֣ת מוֹאָ֑ב עַל-יַרְדֵּ֥ן יְרֵח֖וֹ לֵאמֹֽר:  ת  וּמַלִיל משֶׁה וְאֶלְעָזָר כַּהֲנָא וַאֲמָרוּ לְמִמְנֵי יָתְהוֹן בְּמֵישְׁרַיָא דְמוֹאָב עַל יַרְדְנָא דִירִיחוֹ לְמֵימָר:

{ד} מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה כַּֽאֲשֶׁר֩ צִוָּ֨ה יְהֹוָ֤ה אֶת-מֹשֶׁה֙ וּבְנֵ֣י יִשְׂרָאֵ֔ל הַיֹּֽצְאִ֖ים מֵאֶ֥רֶץ מִצְרָֽיִם:  מ  מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה כַּֽאֲשֶׁר֩ צִוָּ֨ה יְהֹוָ֤ה אֶת-מֹשֶׁה֙ וּבְנֵ֣י יִשְׂרָאֵ֔ל הַיֹּֽצְאִ֖ים מֵאֶ֥רֶץ מִצְרָֽיִם:  ת  מִבַּר עַשְׂרִין שְׁנִין וּלְעֵילָא כְּמָא דְפַקֵיד יְיָ יָת משֶׁה וּבְנֵי יִשְׂרָאֵל דִנְפָקוּ מֵאַרְעָא דְמִצְרָיִם:

{ה}  שני  רְאוּבֵ֖ן בְּכ֣וֹר יִשְׂרָאֵ֑ל בְּנֵ֣י רְאוּבֵ֗ן חֲנוֹךְ֙ מִשְׁפַּ֣חַת הַֽחֲנֹכִ֔י לְפַלּ֕וּא מִשְׁפַּ֖חַת הַפַּלֻּאִֽי:  מ  רְאוּבֵ֖ן בְּכ֣וֹר יִשְׂרָאֵ֑ל בְּנֵ֣י רְאוּבֵ֗ן חֲנוֹךְ֙ מִשְׁפַּ֣חַת הַֽחֲנֹכִ֔י לְפַלּ֕וּא מִשְׁפַּ֖חַת הַפַּלֻּאִֽי:  ת  רְאוּבֵן בּוּכְרָא דְיִשְׂרָאֵל בְּנֵי רְאוּבֵן חֲנוֹךְ זַרְעִית חֲנוֹךְ לְפַלוּא זַרְעִית פַּלוּא:

{ו} לְחֶצְרֹ֕ן מִשְׁפַּ֖חַת הַֽחֶצְרוֹנִ֑י לְכַרְמִ֕י מִשְׁפַּ֖חַת הַכַּרְמִֽי:  מ  לְחֶצְרֹ֕ן מִשְׁפַּ֖חַת הַֽחֶצְרוֹנִ֑י לְכַרְמִ֕י מִשְׁפַּ֖חַת הַכַּרְמִֽי:  ת  לְחֶצְרוֹן זַרְעִית חֶצְרוֹן לְכַרְמִי זַרְעִית כַּרְמִי:

{ז} אֵ֖לֶּה מִשְׁפְּחֹ֣ת הָרֽאוּבֵנִ֑י וַיִּֽהְי֣וּ פְקֻֽדֵיהֶ֗ם שְׁלֹשָׁ֤ה וְאַרְבָּעִים֙ אֶ֔לֶף וּשְׁבַ֥ע מֵא֖וֹת וּשְׁלֹשִֽׁים:  מ  אֵ֖לֶּה מִשְׁפְּחֹ֣ת הָרֽאוּבֵנִ֑י וַיִּֽהְי֣וּ פְקֻֽדֵיהֶ֗ם שְׁלֹשָׁ֤ה וְאַרְבָּעִים֙ אֶ֔לֶף וּשְׁבַ֥ע מֵא֖וֹת וּשְׁלֹשִֽׁים:  ת  אִלֵין זַרְעֲיַת רְאוּבֵן וַהֲווֹ מִנְיָנֵיהוֹן אַרְבְּעִין וּתְלָתָא אַלְפִין וּשְׁבַע מְאָה וּתְּלָתִין:

{ח} וּבְנֵ֥י פַלּ֖וּא אֱלִיאָֽב:  מ  וּבְנֵ֥י פַלּ֖וּא אֱלִיאָֽב:  ת  וּבְנֵי פַלוּא אֱלִיאָב:

{ט} וּבְנֵ֣י אֱלִיאָ֔ב נְמוּאֵ֖ל וְדָתָ֣ן וַֽאֲבִירָ֑ם הֽוּא-דָתָ֨ן וַֽאֲבִירָ֜ם (קרואי) קְרִיאֵ֣י הָֽעֵדָ֗ה אֲשֶׁ֨ר הִצּ֜וּ עַל-מֹשֶׁ֤ה וְעַֽל-אַהֲרֹן֙ בַּֽעֲדַת-קֹ֔רַח בְּהַצֹּתָ֖ם עַל-יְהֹוָֽה:  מ  וּבְנֵ֣י אֱלִיאָ֔ב נְמוּאֵ֖ל וְדָתָ֣ן וַֽאֲבִירָ֑ם הֽוּא-דָתָ֨ן וַֽאֲבִירָ֜ם (קרואי) קְרִיאֵ֣י הָֽעֵדָ֗ה אֲשֶׁ֨ר הִצּ֜וּ עַל-מֹשֶׁ֤ה וְעַֽל-אַהֲרֹן֙ בַּֽעֲדַת-קֹ֔רַח בְּהַצֹּתָ֖ם עַל-יְהֹוָֽה:  ת  וּבְנֵי אֱלִיאָב נְמוּאֵל וְדָתָן וַאֲבִירָם הוּא דָתָן וַאֲבִירָם מְעַרְעֵי כְּנִשְׁתָּא דְאִתְכְּנָשׁוּ עַל משֶׁה וְעַל אַהֲרֹן בִּכְנִשְׁתָּא דְקֹרַח בְּאִתְכַּנוֹשֵׁיהוֹן עַל יְיָ:

{י} וַתִּפְתַּ֨ח הָאָ֜רֶץ אֶת-פִּ֗יהָ וַתִּבְלַ֥ע אֹתָ֛ם וְאֶת-קֹ֖רַח בְּמ֣וֹת הָֽעֵדָ֑ה בַּֽאֲכֹ֣ל הָאֵ֗שׁ אֵ֣ת חֲמִשִּׁ֤ים וּמָאתַ֨יִם֙ אִ֔ישׁ וַיִּֽהְי֖וּ לְנֵֽס:  מ  וַתִּפְתַּ֨ח הָאָ֜רֶץ אֶת-פִּ֗יהָ וַתִּבְלַ֥ע אֹתָ֛ם וְאֶת-קֹ֖רַח בְּמ֣וֹת הָֽעֵדָ֑ה בַּֽאֲכֹ֣ל הָאֵ֗שׁ אֵ֣ת חֲמִשִּׁ֤ים וּמָאתַ֨יִם֙ אִ֔ישׁ וַיִּֽהְי֖וּ לְנֵֽס:  ת  וּפְתָחַת אַרְעָא יַת פּוּמָהּ וּבְלָעַת יָתְהוֹן וְיָת קֹרַח בְּמוֹתָא דִכְנִשְׁתָּא כַּד אֲכָלַת אֵשָׁתָא יַת מָאתָן וְחַמְשִׁין גַבְרָא וַהֲווֹ לְאָת:

{יא} וּבְנֵי-קֹ֖רַח לֹא-מֵֽתוּ:  מ  וּבְנֵי-קֹ֖רַח לֹא-מֵֽתוּ:  ת  וּבְנֵי קֹרַח לָא מִיתוּ: (ס)

{יב} בְּנֵ֣י שִׁמְעוֹן֘ לְמִשְׁפְּחֹתָם֒ לִנְמוּאֵ֗ל מִשְׁפַּ֨חַת֙ הַנְּמ֣וּאֵלִ֔י לְיָמִ֕ין מִשְׁפַּ֖חַת הַיָּֽמִינִ֑י לְיָכִ֕ין מִשְׁפַּ֖חַת הַיָּֽכִינִֽי:  מ  בְּנֵ֣י שִׁמְעוֹן֘ לְמִשְׁפְּחֹתָם֒ לִנְמוּאֵ֗ל מִשְׁפַּ֨חַת֙ הַנְּמ֣וּאֵלִ֔י לְיָמִ֕ין מִשְׁפַּ֖חַת הַיָּֽמִינִ֑י לְיָכִ֕ין מִשְׁפַּ֖חַת הַיָּֽכִינִֽי:  ת  בְּנֵי שִׁמְעוֹן לְזַרְעֲיַתְהוֹן לִנְמוּאֵל זַרְעִית נְמוּאֵל לְיָמִין זַרְעִית יָמִין לְיָכִין זַרְעִית יָכִין:

{יג} לְזֶ֕רַח מִשְׁפַּ֖חַת הַזַּרְחִ֑י לְשָׁא֕וּל מִשְׁפַּ֖חַת הַשָּֽׁאוּלִֽי:  מ  לְזֶ֕רַח מִשְׁפַּ֖חַת הַזַּרְחִ֑י לְשָׁא֕וּל מִשְׁפַּ֖חַת הַשָּֽׁאוּלִֽי:  ת  לְזֶרַח זַרְעִית זֶרַח לְשָׁאוּל זַרְעִית שָׁאוּל:

{יד} אֵ֖לֶּה מִשְׁפְּחֹ֣ת הַשִּׁמְעֹנִ֑י שְׁנַ֧יִם וְעֶשְׂרִ֛ים אֶ֖לֶף וּמָאתָֽיִם:  מ  אֵ֖לֶּה מִשְׁפְּחֹ֣ת הַשִּׁמְעֹנִ֑י שְׁנַ֧יִם וְעֶשְׂרִ֛ים אֶ֖לֶף וּמָאתָֽיִם:  ת  אִלֵין זַרְעֲיַת שִׁמְעוֹן עַשְׂרִין וּתְרֵין אַלְפִין וּמָאתָן: (ס)

{טו} בְּנֵ֣י גָד֘ לְמִשְׁפְּחֹתָם֒ לִצְפ֗וֹן מִשְׁפַּ֨חַת֙ הַצְּפוֹנִ֔י לְחַגִּ֕י מִשְׁפַּ֖חַת הַֽחַגִּ֑י לְשׁוּנִ֕י מִשְׁפַּ֖חַת הַשּׁוּנִֽי:  מ  בְּנֵ֣י גָד֘ לְמִשְׁפְּחֹתָם֒ לִצְפ֗וֹן מִשְׁפַּ֨חַת֙ הַצְּפוֹנִ֔י לְחַגִּ֕י מִשְׁפַּ֖חַת הַֽחַגִּ֑י לְשׁוּנִ֕י מִשְׁפַּ֖חַת הַשּׁוּנִֽי:  ת  בְּנֵי גָד לְזַרְעֲיַתְהוֹן לִצְפוֹן זַרְעִית צְפוֹן לְחַגִי זַרְעִית חַגִי לְשׁוּנִי זַרְעִית שׁוּנִי:

{טז} לְאָזְנִ֕י מִשְׁפַּ֖חַת הָֽאָזְנִ֑י לְעֵרִ֕י מִשְׁפַּ֖חַת הָֽעֵרִֽי:  מ  לְאָזְנִ֕י מִשְׁפַּ֖חַת הָֽאָזְנִ֑י לְעֵרִ֕י מִשְׁפַּ֖חַת הָֽעֵרִֽי:  ת  לְאָזְנִי זַרְעִית אָזְנִי לְעֵרִי זַרְעִית עֵרִי:

{יז} לַֽאֲר֕וֹד מִשְׁפַּ֖חַת הָֽאֲרוֹדִ֑י לְאַ֨רְאֵלִ֔י מִשְׁפַּ֖חַת הָֽאַרְאֵלִֽי:  מ  לַֽאֲר֕וֹד מִשְׁפַּ֖חַת הָֽאֲרוֹדִ֑י לְאַ֨רְאֵלִ֔י מִשְׁפַּ֖חַת הָֽאַרְאֵלִֽי:  ת  לַאֲרוֹד זַרְעִית אֲרוֹדִי לְאַרְאֵלִי זַרְעִית אַרְאֵלִי:

{יח} אֵ֛לֶּה מִשְׁפְּחֹ֥ת בְּנֵֽי-גָ֖ד לִפְקֻֽדֵיהֶ֑ם אַרְבָּעִ֥ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת:  מ  אֵ֛לֶּה מִשְׁפְּחֹ֥ת בְּנֵֽי-גָ֖ד לִפְקֻֽדֵיהֶ֑ם אַרְבָּעִ֥ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת:  ת  אִלֵין זַרְעֲיַת בְּנֵי גָד לְמִנְיָנֵהוֹן אַרְבְּעִין אַלְפִין וַחֲמֵשׁ מְאָה: (ס)

{יט} בְּנֵ֥י יְהוּדָ֖ה עֵ֣ר וְאוֹנָ֑ן וַיָּ֥מָת עֵ֛ר וְאוֹנָ֖ן בְּאֶ֥רֶץ כְּנָֽעַן:  מ  בְּנֵ֥י יְהוּדָ֖ה עֵ֣ר וְאוֹנָ֑ן וַיָּ֥מָת עֵ֛ר וְאוֹנָ֖ן בְּאֶ֥רֶץ כְּנָֽעַן:  ת  בְּנֵי יְהוּדָה עֵר וְאוֹנָן וּמִית עֵר וְאוֹנָן בְּאַרְעָא דִכְנָעַן:

{כ} וַיִּֽהְי֣וּ בְנֵֽי-יְהוּדָה֘ לְמִשְׁפְּחֹתָם֒ לְשֵׁלָ֗ה מִשְׁפַּ֨חַת֙ הַשֵּׁ֣לָנִ֔י לְפֶ֕רֶץ מִשְׁפַּ֖חַת הַפַּרְצִ֑י לְזֶ֕רַח מִשְׁפַּ֖חַת הַזַּרְחִֽי:  מ  וַיִּֽהְי֣וּ בְנֵֽי-יְהוּדָה֘ לְמִשְׁפְּחֹתָם֒ לְשֵׁלָ֗ה מִשְׁפַּ֨חַת֙ הַשֵּׁ֣לָנִ֔י לְפֶ֕רֶץ מִשְׁפַּ֖חַת הַפַּרְצִ֑י לְזֶ֕רַח מִשְׁפַּ֖חַת הַזַּרְחִֽי:  ת  וַהֲווֹ בְנֵי יְהוּדָה לְזַרְעֲיַתְהוֹן לְשֵׁלָה זַרְעִית שֵׁלָה לְפֶרֶץ זַרְעִית פֶּרֶץ לְזֶרַח זַרְעִית זָרַח:

{כא} וַיִּֽהְי֣וּ בְנֵי-פֶ֔רֶץ לְחֶצְרֹ֕ן מִשְׁפַּ֖חַת הַֽחֶצְרֹנִ֑י לְחָמ֕וּל מִשְׁפַּ֖חַת הֶחָֽמוּלִֽי:  מ  וַיִּֽהְי֣וּ בְנֵי-פֶ֔רֶץ לְחֶצְרֹ֕ן מִשְׁפַּ֖חַת הַֽחֶצְרֹנִ֑י לְחָמ֕וּל מִשְׁפַּ֖חַת הֶחָֽמוּלִֽי:  ת  וַהֲווֹ בְנֵי פֶרֶץ לְחֶצְרוֹן זַרְעִית חֶצְרוֹן לְחָמוּל זַרְעִית חָמוּל:

{כב} אֵ֛לֶּה מִשְׁפְּחֹ֥ת יְהוּדָ֖ה לִפְקֻֽדֵיהֶ֑ם שִׁשָּׁ֧ה וְשִׁבְעִ֛ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת:  מ  אֵ֛לֶּה מִשְׁפְּחֹ֥ת יְהוּדָ֖ה לִפְקֻֽדֵיהֶ֑ם שִׁשָּׁ֧ה וְשִׁבְעִ֛ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת:  ת  אִלֵין זַרְעֲיַת יְהוּדָה לְמִנְיָנֵיהוֹן שִׁבְעִין וְשִׁתָּא אַלְפִין וַחֲמֵשׁ מְאָה: (ס)

{כג} בְּנֵ֤י יִשָּׂשכָר֙ לְמִשְׁפְּחֹתָ֔ם תּוֹלָ֕ע מִשְׁפַּ֖חַת הַתּֽוֹלָעִ֑י לְפֻוָ֕ה מִשְׁפַּ֖חַת הַפּוּנִֽי:  מ  בְּנֵ֤י יִשָּׂשכָר֙ לְמִשְׁפְּחֹתָ֔ם תּוֹלָ֕ע מִשְׁפַּ֖חַת הַתּֽוֹלָעִ֑י לְפֻוָ֕ה מִשְׁפַּ֖חַת הַפּוּנִֽי:  ת  בְּנֵי יִשָשׂכָר לְזַרְעֲיַתְהוֹן תּוֹלָע זַרְעִית תּוֹלָע לְפֻוָה זַרְעִית פּוּנִי:

{כד} לְיָשׁ֕וּב מִשְׁפַּ֖חַת הַיָּֽשֻׁבִ֑י לְשִׁמְרֹ֕ן מִשְׁפַּ֖חַת הַשִּׁמְרֹנִֽי:  מ  לְיָשׁ֕וּב מִשְׁפַּ֖חַת הַיָּֽשֻׁבִ֑י לְשִׁמְרֹ֕ן מִשְׁפַּ֖חַת הַשִּׁמְרֹנִֽי:  ת  לְיָשׁוּב זַרְעִית יָשׁוּב לְשִׁמְרוֹן זַרְעִית שִׁמְרוֹן:

{כה} אֵ֛לֶּה מִשְׁפְּחֹ֥ת יִשָּׂשכָ֖ר לִפְקֻֽדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁשִּׁ֛ים אֶ֖לֶף וּשְׁלֹ֥שׁ מֵאֽוֹת:  מ  אֵ֛לֶּה מִשְׁפְּחֹ֥ת יִשָּׂשכָ֖ר לִפְקֻֽדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁשִּׁ֛ים אֶ֖לֶף וּשְׁלֹ֥שׁ מֵאֽוֹת:  ת  אִלֵין זַרְעֲיַת יִשָׂשׂכָר לְמִנְיָנֵיהוֹן שִׁתִּין וְאַרְבְּעָה אַלפִין וּתְלַת מְאָה: (ס)

{כו} בְּנֵ֣י זְבוּלֻן֘ לְמִשְׁפְּחֹתָם֒ לְסֶ֗רֶד מִשְׁפַּ֨חַת֙ הַסַּרְדִּ֔י לְאֵל֕וֹן מִשְׁפַּ֖חַת הָאֵֽלֹנִ֑י לְיַ֨חְלְאֵ֔ל מִשְׁפַּ֖חַת הַיַּחְלְאֵלִֽי:  מ  בְּנֵ֣י זְבוּלֻן֘ לְמִשְׁפְּחֹתָם֒ לְסֶ֗רֶד מִשְׁפַּ֨חַת֙ הַסַּרְדִּ֔י לְאֵל֕וֹן מִשְׁפַּ֖חַת הָאֵֽלֹנִ֑י לְיַ֨חְלְאֵ֔ל מִשְׁפַּ֖חַת הַיַּחְלְאֵלִֽי:  ת  בְּנֵי זְבוּלוּן לְזַרְעֲיַתְהוֹן לְסֶרֶד זַרְעִית סֶרֶד לְאֵלוֹן זַרְעִית אֵלוֹן לְיַחְלְאֵל זַרְעִית יַחְלְאֵל:

{כז} אֵ֛לֶּה מִשְׁפְּחֹ֥ת הַזְּבֽוּלֹנִ֖י לִפְקֻֽדֵיהֶ֑ם שִׁשִּׁ֥ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת:  מ  אֵ֛לֶּה מִשְׁפְּחֹ֥ת הַזְּבֽוּלֹנִ֖י לִפְקֻֽדֵיהֶ֑ם שִׁשִּׁ֥ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת:  ת  אִלֵין זַרְעֲיַת זְבוּלוּן לְמִנְיָנֵיהוֹן שִׁתִּין אַלְפִין וְחֲמֵשׁ מְאָה: (ס)

{כח} בְּנֵ֥י יוֹסֵ֖ף לְמִשְׁפְּחֹתָ֑ם מְנַשֶּׁ֖ה וְאֶפְרָֽיִם:  מ  בְּנֵ֥י יוֹסֵ֖ף לְמִשְׁפְּחֹתָ֑ם מְנַשֶּׁ֖ה וְאֶפְרָֽיִם:  ת  בְּנֵי יוֹסֵף לְזַרְעֲיַתְהוֹן מְנַשֶׁה וְאֶפְרָיִם:

{כט} בְּנֵ֣י מְנַשֶּׁ֗ה לְמָכִיר֙ מִשְׁפַּ֣חַת הַמָּֽכִירִ֔י וּמָכִ֖יר הוֹלִ֣יד אֶת-גִּלְעָ֑ד לְגִלְעָ֕ד מִשְׁפַּ֖חַת הַגִּלְעָדִֽי:  מ  בְּנֵ֣י מְנַשֶּׁ֗ה לְמָכִיר֙ מִשְׁפַּ֣חַת הַמָּֽכִירִ֔י וּמָכִ֖יר הוֹלִ֣יד אֶת-גִּלְעָ֑ד לְגִלְעָ֕ד מִשְׁפַּ֖חַת הַגִּלְעָדִֽי:  ת  בְּנֵי מְנַשֶׁה לְמָכִיר זַרְעִית מָכִיר וּמָכִיר אוֹלִיד יַת גִלְעָד לְגִלְעָד זַרְעִית גִלְעָד:

{ל} אֵ֚לֶּה בְּנֵ֣י גִלְעָ֔ד אִיעֶ֕זֶר מִשְׁפַּ֖חַת הָאִֽיעֶזְרִ֑י לְחֵ֕לֶק מִשְׁפַּ֖חַת הַֽחֶלְקִֽי:  מ  אֵ֚לֶּה בְּנֵ֣י גִלְעָ֔ד אִיעֶ֕זֶר מִשְׁפַּ֖חַת הָאִֽיעֶזְרִ֑י לְחֵ֕לֶק מִשְׁפַּ֖חַת הַֽחֶלְקִֽי:  ת  אִלֵין בְּנֵי גִלְעָד אִיעֶזֶר זַרְעִית אִיעֶזֶר לְחֵלֶק זַרְעִית חֵלֶק:

{לא} וְאַ֨שְׂרִיאֵ֔ל מִשְׁפַּ֖חַת הָֽאַשְׂרִֽאֵלִ֑י וְשֶׁ֕כֶם מִשְׁפַּ֖חַת הַשִּׁכְמִֽי:  מ  וְאַ֨שְׂרִיאֵ֔ל מִשְׁפַּ֖חַת הָֽאַשְׂרִֽאֵלִ֑י וְשֶׁ֕כֶם מִשְׁפַּ֖חַת הַשִּׁכְמִֽי:  ת  וְאַשְׂרִיאֵל זַרְעִית אַשְׂרִיאֵלִי וְשֶׁכֶם זַרְעִית שֶׁכֶם:

{לב} וּשְׁמִידָ֕ע מִשְׁפַּ֖חַת הַשְּׁמִֽידָעִ֑י וְחֵ֕פֶר מִשְׁפַּ֖חַת הַֽחֶפְרִֽי:  מ  וּשְׁמִידָ֕ע מִשְׁפַּ֖חַת הַשְּׁמִֽידָעִ֑י וְחֵ֕פֶר מִשְׁפַּ֖חַת הַֽחֶפְרִֽי:  ת  וּשְׁמִידָע זַרְעִית שְׁמִידָע וְחֵפֶר זַרְעִית חֵפֶר:

{לג} וּצְלָפְחָ֣ד בֶּן-חֵ֗פֶר לֹא-הָ֥יוּ ל֛וֹ בָּנִ֖ים כִּ֣י אִם-בָּנ֑וֹת וְשֵׁם֙ בְּנ֣וֹת צְלָפְחָ֔ד מַחְלָ֣ה וְנֹעָ֔ה חָגְלָ֥ה מִלְכָּ֖ה וְתִרְצָֽה:  מ  וּצְלָפְחָ֣ד בֶּן-חֵ֗פֶר לֹא-הָ֥יוּ ל֛וֹ בָּנִ֖ים כִּ֣י אִם-בָּנ֑וֹת וְשֵׁם֙ בְּנ֣וֹת צְלָפְחָ֔ד מַחְלָ֣ה וְנֹעָ֔ה חָגְלָ֥ה מִלְכָּ֖ה וְתִרְצָֽה:  ת  וּצְלָפְחָד בַּר חֵפֶר לָא הֲווֹ לֵיהּ בְּנִין אֱלָהֵין בְּנָן וְשׁוּם בְּנַת צְלָפְחָד מַחְלָה וְנוֹעָה חָגְלָה מִלְכָּה וְתִרְצָה:

{לד} אֵ֖לֶּה מִשְׁפְּחֹ֣ת מְנַשֶּׁ֑ה וּפְקֻ֣דֵיהֶ֔ם שְׁנַ֧יִם וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וּשְׁבַ֥ע מֵאֽוֹת:  מ  אֵ֖לֶּה מִשְׁפְּחֹ֣ת מְנַשֶּׁ֑ה וּפְקֻ֣דֵיהֶ֔ם שְׁנַ֧יִם וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וּשְׁבַ֥ע מֵאֽוֹת:  ת  אִלֵין זַרְעֲיַת מְנַשֶׁה וּמִנְיָנֵיהוֹן חַמְשִׁין וּתְרֵין אַלְפִין וּשְׁבַע מְאָה: (ס)

{לה} אֵ֣לֶּה בְנֵֽי-אֶפְרַ֘יִם֘ לְמִשְׁפְּחֹתָם֒ לְשׁוּתֶ֗לַח מִשְׁפַּ֨חַת֙ הַשֻּׁ֣תַלְחִ֔י לְבֶ֕כֶר מִשְׁפַּ֖חַת הַבַּכְרִ֑י לְתַ֕חַן מִשְׁפַּ֖חַת הַֽתַּחֲנִֽי:  מ  אֵ֣לֶּה בְנֵֽי-אֶפְרַ֘יִם֘ לְמִשְׁפְּחֹתָם֒ לְשׁוּתֶ֗לַח מִשְׁפַּ֨חַת֙ הַשֻּׁ֣תַלְחִ֔י לְבֶ֕כֶר מִשְׁפַּ֖חַת הַבַּכְרִ֑י לְתַ֕חַן מִשְׁפַּ֖חַת הַֽתַּחֲנִֽי:  ת  אִלֵין בְּנֵי אֶפְרַיִם לְזַרְעֲיַתְהוֹן לְשׁוּתֶלַח זַרְעִית שׁוּתֶלַח לְבֶכֶר זַרְעִית בֶּכֶר לְתַחַן זַרעִית תַּחַן:

{לו} וְאֵ֖לֶּה בְּנֵ֣י שׁוּתָ֑לַח לְעֵרָ֕ן מִשְׁפַּ֖חַת הָעֵֽרָנִֽי:  מ  וְאֵ֖לֶּה בְּנֵ֣י שׁוּתָ֑לַח לְעֵרָ֕ן מִשְׁפַּ֖חַת הָעֵֽרָנִֽי:  ת  וְאִלֵין בְּנֵי שׁוּתָלַח לְעֵרָן זַרְעִית עֵרָן:

{לז} אֵ֣לֶּה מִשְׁפְּחֹ֤ת בְּנֵֽי-אֶפְרַ֨יִם֙ לִפְקֻ֣דֵיהֶ֔ם שְׁנַ֧יִם וּשְׁלֹשִׁ֛ים אֶ֖לֶף וַֽחֲמֵ֣שׁ מֵא֑וֹת אֵ֥לֶּה בְנֵֽי-יוֹסֵ֖ף לְמִשְׁפְּחֹתָֽם:  מ  אֵ֣לֶּה מִשְׁפְּחֹ֤ת בְּנֵֽי-אֶפְרַ֨יִם֙ לִפְקֻ֣דֵיהֶ֔ם שְׁנַ֧יִם וּשְׁלֹשִׁ֛ים אֶ֖לֶף וַֽחֲמֵ֣שׁ מֵא֑וֹת אֵ֥לֶּה בְנֵֽי-יוֹסֵ֖ף לְמִשְׁפְּחֹתָֽם:  ת  אִלֵין זַרְעֲיַת בְּנֵי אֶפְרַיִם לְמִנְיָנֵיהּוֹן תְּלָתִין וּתְרֵין אַלְפִין וַחֲמֵשׁ מְאָה אִלֵין בְּנֵי יוֹסֵף לְזַרְעֲיַתְהּוֹן: (ס)

{לח} בְּנֵ֣י בִנְיָמִן֘ לְמִשְׁפְּחֹתָם֒ לְבֶ֗לַע מִשְׁפַּ֨חַת֙ הַבַּלְעִ֔י לְאַשְׁבֵּ֕ל מִשְׁפַּ֖חַת הָֽאַשְׁבֵּלִ֑י לַֽאֲחִירָ֕ם מִשְׁפַּ֖חַת הָאֲחִֽירָמִֽי:  מ  בְּנֵ֣י בִנְיָמִן֘ לְמִשְׁפְּחֹתָם֒ לְבֶ֗לַע מִשְׁפַּ֨חַת֙ הַבַּלְעִ֔י לְאַשְׁבֵּ֕ל מִשְׁפַּ֖חַת הָֽאַשְׁבֵּלִ֑י לַֽאֲחִירָ֕ם מִשְׁפַּ֖חַת הָאֲחִֽירָמִֽי:  ת  בְּנֵי בִנְיָמִין לְזַרְעֲיַתְהוֹן לְבֶלַע זַרְעִית בֶּלַע לְאַשְׁבֵּל זַרְעִית אַשְׁבֵּל לַאֲחִירָם זַרְעִית אֲחִירָם:

{לט} לִשְׁפוּפָ֕ם מִשְׁפַּ֖חַת הַשּֽׁוּפָמִ֑י לְחוּפָ֕ם מִשְׁפַּ֖חַת הַחֽוּפָמִֽי:  מ  לִשְׁפוּפָ֕ם מִשְׁפַּ֖חַת הַשּֽׁוּפָמִ֑י לְחוּפָ֕ם מִשְׁפַּ֖חַת הַחֽוּפָמִֽי:  ת  לִשְׁפוּפָם זַרְעִית שְׁפוּפָם לְחוּפָם זַרְעִית חוּפָם:

{מ} וַיִּֽהְי֥וּ בְנֵי-בֶ֖לַע אַ֣רְדְּ וְנַֽעֲמָ֑ן מִשְׁפַּ֨חַת֙ הָֽאַרְדִּ֔י לְנַֽעֲמָ֕ן מִשְׁפַּ֖חַת הַֽנַּעֲמִֽי:  מ  וַיִּֽהְי֥וּ בְנֵי-בֶ֖לַע אַ֣רְדְּ וְנַֽעֲמָ֑ן מִשְׁפַּ֨חַת֙ הָֽאַרְדִּ֔י לְנַֽעֲמָ֕ן מִשְׁפַּ֖חַת הַֽנַּעֲמִֽי:  ת  וַהֲווֹ בְנֵי בֶלַע אַרְדְ וְנַעֲמָן זַרְעִית הָאַרְדִי לְנַעֲמָן זַרְעִית נַעֲמִי:

{מא} אֵ֥לֶּה בְנֵֽי-בִנְיָמִ֖ן לְמִשְׁפְּחֹתָ֑ם וּפְקֻ֣דֵיהֶ֔ם חֲמִשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת:  מ  אֵ֥לֶּה בְנֵֽי-בִנְיָמִ֖ן לְמִשְׁפְּחֹתָ֑ם וּפְקֻ֣דֵיהֶ֔ם חֲמִשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת:  ת  אִלֵין בְּנֵי בִנְיָמִן לְזַרְעֲיַתְהוֹן וּמִנְיָנֵיהוֹן אַרְבְּעִין וְחַמְשָׁא אַלְפִין וְשִׁית מְאָה: (ס)

{מב} אֵ֤לֶּה בְנֵי-דָן֙ לְמִשְׁפְּחֹתָ֔ם לְשׁוּחָ֕ם מִשְׁפַּ֖חַת הַשּֽׁוּחָמִ֑י אֵ֛לֶּה מִשְׁפְּחֹ֥ת דָּ֖ן לְמִשְׁפְּחֹתָֽם:  מ  אֵ֤לֶּה בְנֵי-דָן֙ לְמִשְׁפְּחֹתָ֔ם לְשׁוּחָ֕ם מִשְׁפַּ֖חַת הַשּֽׁוּחָמִ֑י אֵ֛לֶּה מִשְׁפְּחֹ֥ת דָּ֖ן לְמִשְׁפְּחֹתָֽם:  ת  אִלֵין בְּנֵי דָן לְזַרְעֲיַתְהוֹן לְשׁוּחָם זַרְעִית שׁוּחָם אִלֵין זַרְעֲיַת דָן לְזַרְעֲיַתְהוֹן:

{מג} כָּל-מִשְׁפְּחֹ֥ת הַשּֽׁוּחָמִ֖י לִפְקֻֽדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  מ  כָּל-מִשְׁפְּחֹ֥ת הַשּֽׁוּחָמִ֖י לִפְקֻֽדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  ת  כָל זַרְעֲיַת שׁוּחָם לְמִנְיָנֵיהוֹן שִׁתִּין וְאַרְבַּע אַלְפִין וְאַרְבַּע מְאָה: (ס)

{מד} בְּנֵ֣י אָשֵׁר֘ לְמִשְׁפְּחֹתָם֒ לְיִמְנָ֗ה מִשְׁפַּ֨חַת֙ הַיִּמְנָ֔ה לְיִשְׁוִ֕י מִשְׁפַּ֖חַת הַיִּשְׁוִ֑י לִבְרִיעָ֕ה מִשְׁפַּ֖חַת הַבְּרִיעִֽי:  מ  בְּנֵ֣י אָשֵׁר֘ לְמִשְׁפְּחֹתָם֒ לְיִמְנָ֗ה מִשְׁפַּ֨חַת֙ הַיִּמְנָ֔ה לְיִשְׁוִ֕י מִשְׁפַּ֖חַת הַיִּשְׁוִ֑י לִבְרִיעָ֕ה מִשְׁפַּ֖חַת הַבְּרִיעִֽי:  ת  בְּנֵי אָשֵׁר לְזַרְעֲיַתְהוֹן לְיִמְנָה זַרְעִית יִמְנָה לְיִשְׁוִי זַרְעִית יִשְׁוִי לִבְרִיעָה זַרְעִית בְּרִיעָה:

{מה} לִבְנֵ֣י בְרִיעָ֔ה לְחֵ֕בֶר מִשְׁפַּ֖חַת הַֽחֶבְרִ֑י לְמַ֨לְכִּיאֵ֔ל מִשְׁפַּ֖חַת הַמַּלְכִּֽיאֵלִֽי:  מ  לִבְנֵ֣י בְרִיעָ֔ה לְחֵ֕בֶר מִשְׁפַּ֖חַת הַֽחֶבְרִ֑י לְמַ֨לְכִּיאֵ֔ל מִשְׁפַּ֖חַת הַמַּלְכִּֽיאֵלִֽי:  ת  לִבְנֵי בְרִיעָה לְחֵבֶר זַרְעִית חֵבֶר לְמַלְכִּיאֵל זַרעִית מַלְכִּיאֵל:

{מו} וְשֵׁ֥ם בַּת-אָשֵׁ֖ר שָֽׂרַח:  מ  וְשֵׁ֥ם בַּת-אָשֵׁ֖ר שָֽׂרַח:  ת  וְשׁוּם בַּת אָשֵׁר סָרַח:

{מז} אֵ֛לֶּה מִשְׁפְּחֹ֥ת בְּנֵֽי-אָשֵׁ֖ר לִפְקֻֽדֵיהֶ֑ם שְׁלֹשָׁ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  מ  אֵ֛לֶּה מִשְׁפְּחֹ֥ת בְּנֵֽי-אָשֵׁ֖ר לִפְקֻֽדֵיהֶ֑ם שְׁלֹשָׁ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  ת  אִלֵין זַרְעֲיַת בְּנֵי אָשֵׁר לְמִנְיָנֵיהוֹן חַמְשִׁין וּתְלָתָא אַלְפִין ואַרְבַּע מְאָה: (ס)

{מח} בְּנֵ֤י נַפְתָּלִי֙ לְמִשְׁפְּחֹתָ֔ם לְיַ֨חְצְאֵ֔ל מִשְׁפַּ֖חַת הַיַּחְצְאֵלִ֑י לְגוּנִ֕י מִשְׁפַּ֖חַת הַגּוּנִֽי:  מ  בְּנֵ֤י נַפְתָּלִי֙ לְמִשְׁפְּחֹתָ֔ם לְיַ֨חְצְאֵ֔ל מִשְׁפַּ֖חַת הַיַּחְצְאֵלִ֑י לְגוּנִ֕י מִשְׁפַּ֖חַת הַגּוּנִֽי:  ת  בְּנֵי נַפְתָּלִי לְזַרְעֲיַתְהוֹן לְיַחְצְאֵל זַרְעִית יַחְצְאֵל לְגוּנִי זַרְעִית גוּנִי:

{מט} לְיֵ֕צֶר מִשְׁפַּ֖חַת הַיִּצְרִ֑י לְשִׁלֵּ֕ם מִשְׁפַּ֖חַת הַשִּׁלֵּמִֽי:  מ  לְיֵ֕צֶר מִשְׁפַּ֖חַת הַיִּצְרִ֑י לְשִׁלֵּ֕ם מִשְׁפַּ֖חַת הַשִּׁלֵּמִֽי:  ת  לְיֵצֶר זַרְעִית יֵצֶר לְשִׁלֵם זַרְעִית שִׁלֵם:

{נ} אֵ֛לֶּה מִשְׁפְּחֹ֥ת נַפְתָּלִ֖י לְמִשְׁפְּחֹתָ֑ם וּפְקֻ֣דֵיהֶ֔ם חֲמִשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  מ  אֵ֛לֶּה מִשְׁפְּחֹ֥ת נַפְתָּלִ֖י לְמִשְׁפְּחֹתָ֑ם וּפְקֻ֣דֵיהֶ֔ם חֲמִשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת:  ת  אִלֵין זַרְעֲיַת נַפְתָּלִי לְזַרְעֲיַתְהוֹן וּמִנְיָנֵיהוֹן אַרְבְּעִין וְחַמְשָׁא אַלְפִין וְאַרְבַּע מֵאָה:

{נא} אֵ֗לֶּה פְּקוּדֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל שֵׁשׁ-מֵא֥וֹת אֶ֖לֶף וָאָ֑לֶף שְׁבַ֥ע מֵא֖וֹת וּשְׁלֹשִֽׁים:  מ  אֵ֗לֶּה פְּקוּדֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל שֵׁשׁ-מֵא֥וֹת אֶ֖לֶף וָאָ֑לֶף שְׁבַ֥ע מֵא֖וֹת וּשְׁלֹשִֽׁים:  ת  אִלֵין מִנְיָנֵי בְּנֵי יִשְׂרָאֵל שִׁית מְאָה וְחַד אַלְפִין שְׁבַע מְאָה וּתְלָתִין: (ס)

{נב}  שלישי  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר:

{נג} לָאֵ֗לֶּה תֵּֽחָלֵ֥ק הָאָ֛רֶץ בְּנַֽחֲלָ֖ה בְּמִסְפַּ֥ר שֵׁמֽוֹת:  מ  לָאֵ֗לֶּה תֵּֽחָלֵ֥ק הָאָ֛רֶץ בְּנַֽחֲלָ֖ה בְּמִסְפַּ֥ר שֵׁמֽוֹת:  ת  לְאִלֵין תִּתְפְּלֵיג אַרְעָא בְּאַחֲסָנָא בְּמִנְיַן שְׁמָהָן:

{נד} לָרַ֗ב תַּרְבֶּה֙ נַֽחֲלָת֔וֹ וְלַמְעַ֕ט תַּמְעִ֖יט נַֽחֲלָת֑וֹ אִ֚ישׁ לְפִ֣י פְקֻדָ֔יו יֻתַּ֖ן נַֽחֲלָתֽוֹ:  מ  לָרַ֗ב תַּרְבֶּה֙ נַֽחֲלָת֔וֹ וְלַמְעַ֕ט תַּמְעִ֖יט נַֽחֲלָת֑וֹ אִ֚ישׁ לְפִ֣י פְקֻדָ֔יו יֻתַּ֖ן נַֽחֲלָתֽוֹ:  ת  לְסַגִיאֵי תִּסְגוּן אַחֲסַנְתְּהוֹן וְלִזְעֵירֵי תַּזְעֲרוּן אַחֲסַנְתְּהוֹן גְבַר לְפוּם מִנְיָנוֹהִי תִּתְיְהֵב אַחֲסַנְתֵּיהּ:

{נה} אַךְ-בְּגוֹרָ֕ל יֵֽחָלֵ֖ק אֶת-הָאָ֑רֶץ לִשְׁמ֥וֹת מַטּוֹת-אֲבֹתָ֖ם יִנְחָֽלוּ:  מ  אַךְ-בְּגוֹרָ֕ל יֵֽחָלֵ֖ק אֶת-הָאָ֑רֶץ לִשְׁמ֥וֹת מַטּוֹת-אֲבֹתָ֖ם יִנְחָֽלוּ:  ת  בְּרַם בְּעַדְבָא תִּתְפְּלֵיג יַת אַרְעָא לִשְׁמָהָן שִׁבְטֵי אֲבָהָתְהוֹן יַחְסְנוּן:

{נו} עַל-פִּי֙ הַגּוֹרָ֔ל תֵּֽחָלֵ֖ק נַֽחֲלָת֑וֹ בֵּ֥ין רַ֖ב לִמְעָֽט:  מ  עַל-פִּי֙ הַגּוֹרָ֔ל תֵּֽחָלֵ֖ק נַֽחֲלָת֑וֹ בֵּ֥ין רַ֖ב לִמְעָֽט:  ת  עַל פּוּם עַדְבָא תִּתְפְּלֵיג אַחֲסַנְתְּהוֹן בֵּין סַגִיאֵי לִזְעֵירֵי: (ס)

{נז} וְאֵ֨לֶּה פְקוּדֵ֣י הַלֵּוִי֘ לְמִשְׁפְּחֹתָם֒ לְגֵֽרְשׁ֗וֹן מִשְׁפַּ֨חַת֙ הַגֵּ֣רְשֻׁנִּ֔י לִקְהָ֕ת מִשְׁפַּ֖חַת הַקְּהָתִ֑י לִמְרָרִ֕י מִשְׁפַּ֖חַת הַמְּרָרִֽי:  מ  וְאֵ֨לֶּה פְקוּדֵ֣י הַלֵּוִי֘ לְמִשְׁפְּחֹתָם֒ לְגֵֽרְשׁ֗וֹן מִשְׁפַּ֨חַת֙ הַגֵּ֣רְשֻׁנִּ֔י לִקְהָ֕ת מִשְׁפַּ֖חַת הַקְּהָתִ֑י לִמְרָרִ֕י מִשְׁפַּ֖חַת הַמְּרָרִֽי:  ת  וְאִלֵין מִנְיָנֵי לֵוָאֵי לְזַרְעֲיַתְהוֹן לְגֵרְשׁוֹן זַרְעִית גֵרְשׁוֹן לִּקְהָת זַרְעִית קְהָת לִמְרָרִי זַרְעִית מְרָרִי:

{נח} אֵ֣לֶּה | מִשְׁפְּחֹ֣ת לֵוִ֗י מִשְׁפַּ֨חַת הַלִּבְנִ֜י מִשְׁפַּ֤חַת הַֽחֶבְרֹנִי֙ מִשְׁפַּ֤חַת הַמַּחְלִי֙ מִשְׁפַּ֣חַת הַמּוּשִׁ֔י מִשְׁפַּ֖חַת הַקָּרְחִ֑י וּקְהָ֖ת הוֹלִ֥ד אֶת-עַמְרָֽם:  מ  אֵ֣לֶּה | מִשְׁפְּחֹ֣ת לֵוִ֗י מִשְׁפַּ֨חַת הַלִּבְנִ֜י מִשְׁפַּ֤חַת הַֽחֶבְרֹנִי֙ מִשְׁפַּ֤חַת הַמַּחְלִי֙ מִשְׁפַּ֣חַת הַמּוּשִׁ֔י מִשְׁפַּ֖חַת הַקָּרְחִ֑י וּקְהָ֖ת הוֹלִ֥ד אֶת-עַמְרָֽם:  ת  אִלֵין זַרְעֲיַת לֵוִי זַרְעִית לִבְנִי זַרְעִית חֶבְרוֹן זַרְעִית מַחְלִי זַרְעִית מוּשִׁי זַרְעִית קֹרַח וּקְהָת אוֹלִיד יָת עַמְרָם:

{ס} וַיִּוָּלֵ֣ד לְאַֽהֲרֹ֔ן אֶת-נָדָ֖ב וְאֶת-אֲבִיה֑וּא אֶת-אֶלְעָזָ֖ר וְאֶת-אִֽיתָמָֽר:  מ  וַיִּוָּלֵ֣ד לְאַֽהֲרֹ֔ן אֶת-נָדָ֖ב וְאֶת-אֲבִיה֑וּא אֶת-אֶלְעָזָ֖ר וְאֶת-אִֽיתָמָֽר:  ת  וְאִתְיְלִיד לְאַהֲרֹן יַת נָדָב וְיַת אֲבִיהוּא יַת אֶלְעָזָר וְיַת אִיתָמָר:

{סא} וַיָּ֥מָת נָדָ֖ב וַֽאֲבִיה֑וּא בְּהַקְרִיבָ֥ם אֵשׁ-זָרָ֖ה לִפְנֵ֥י יְהֹוָֽה:  מ  וַיָּ֥מָת נָדָ֖ב וַֽאֲבִיה֑וּא בְּהַקְרִיבָ֥ם אֵשׁ-זָרָ֖ה לִפְנֵ֥י יְהֹוָֽה:  ת  וּמִית נָדָב וַאֲבִיהוּא בְּקָרוֹבֵיהוֹן אֶשָׁתָא נוּכְרֵיתָא קֳדָם יְיָ:

{סב} וַיִּֽהְי֣וּ פְקֻֽדֵיהֶ֗ם שְׁלֹשָׁ֤ה וְעֶשְׂרִים֙ אֶ֔לֶף כָּל-זָכָ֖ר מִבֶּן-חֹ֣דֶשׁ וָמָ֑עְלָה כִּ֣י | לֹ֣א הָתְפָּֽקְד֗וּ בְּתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כִּ֠י לֹֽא-נִתַּ֤ן לָהֶם֙ נַֽחֲלָ֔ה בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וַיִּֽהְי֣וּ פְקֻֽדֵיהֶ֗ם שְׁלֹשָׁ֤ה וְעֶשְׂרִים֙ אֶ֔לֶף כָּל-זָכָ֖ר מִבֶּן-חֹ֣דֶשׁ וָמָ֑עְלָה כִּ֣י | לֹ֣א הָתְפָּֽקְד֗וּ בְּתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כִּ֠י לֹֽא-נִתַּ֤ן לָהֶם֙ נַֽחֲלָ֔ה בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וַהֲווֹ מִנְיָנֵיהוֹן עַשְׂרִין וּתְלָתָא אַלְפִין כָּל דְכוּרָא מִבַּר יַרְחָא וּלְעֵילָא אֲרֵי לָא אִתְמְנִיאוּ בְּגוֹ בְּנֵי יִשְׂרָאֵל אֲרֵי לָא אִתְיְהִיבַת לְהוֹן אַחֲסָנָא בְּגוֹ בְּנֵי יִשְׂרָאֵל:

{סג} אֵ֚לֶּה פְּקוּדֵ֣י מֹשֶׁ֔ה וְאֶלְעָזָ֖ר הַכֹּהֵ֑ן אֲשֶׁ֨ר פָּֽקְד֜וּ אֶת-בְּנֵ֤י יִשְׂרָאֵל֙ בְּעַֽרְבֹ֣ת מוֹאָ֔ב עַ֖ל יַרְדֵּ֥ן יְרֵחֽוֹ:  מ  אֵ֚לֶּה פְּקוּדֵ֣י מֹשֶׁ֔ה וְאֶלְעָזָ֖ר הַכֹּהֵ֑ן אֲשֶׁ֨ר פָּֽקְד֜וּ אֶת-בְּנֵ֤י יִשְׂרָאֵל֙ בְּעַֽרְבֹ֣ת מוֹאָ֔ב עַ֖ל יַרְדֵּ֥ן יְרֵחֽוֹ:  ת  אִלֵין מִנְיָנֵי משֶׁה וְאֶלְעָזָר כַּהֲנָא דִי מְנוֹ יַת בְּנֵי יִשְׂרָאֵל בְּמֵישְׁרַיָא דְמוֹאָב עַל יַרְדְנָא דִירֵיחוֹ:

{סד} וּבְאֵ֨לֶּה֙ לֹא-הָ֣יָה אִ֔ישׁ מִפְּקוּדֵ֣י מֹשֶׁ֔ה וְאַֽהֲרֹ֖ן הַכֹּהֵ֑ן אֲשֶׁ֥ר פָּֽקְד֛וּ אֶת-בְּנֵ֥י יִשְׂרָאֵ֖ל בְּמִדְבַּ֥ר סִינָֽי:  מ  וּבְאֵ֨לֶּה֙ לֹא-הָ֣יָה אִ֔ישׁ מִפְּקוּדֵ֣י מֹשֶׁ֔ה וְאַֽהֲרֹ֖ן הַכֹּהֵ֑ן אֲשֶׁ֥ר פָּֽקְד֛וּ אֶת-בְּנֵ֥י יִשְׂרָאֵ֖ל בְּמִדְבַּ֥ר סִינָֽי:  ת  וּבְאִלֵין לָא הֲוָה גְבַר מִמִנְיָנֵי משֶׁה וְאַהֲרֹן כַּהֲנָא דִי מְנוֹ יָת בְּנֵי יִשְׂרָאֵל בְּמַדְבְּרָא דְסִינָי:

{סה} כִּֽי-אָמַ֤ר יְהֹוָה֙ לָהֶ֔ם מ֥וֹת יָמֻ֖תוּ בַּמִּדְבָּ֑ר וְלֹֽא-נוֹתַ֤ר מֵהֶם֙ אִ֔ישׁ כִּ֚י אִם-כָּלֵ֣ב בֶּן-יְפֻנֶּ֔ה וִֽיהוֹשֻׁ֖עַ בִּן-נֽוּן:  מ  כִּֽי-אָמַ֤ר יְהֹוָה֙ לָהֶ֔ם מ֥וֹת יָמֻ֖תוּ בַּמִּדְבָּ֑ר וְלֹֽא-נוֹתַ֤ר מֵהֶם֙ אִ֔ישׁ כִּ֚י אִם-כָּלֵ֣ב בֶּן-יְפֻנֶּ֔ה וִֽיהוֹשֻׁ֖עַ בִּן-נֽוּן:  ת  אֲרֵי אֲמַר יְיָ לְהוֹן מֵימַת יְמוּתוּן בְּמַדְבְּרָא וְלָא אִשְׁתְּאַר מִנְהוֹן אֱנַשׁ אֶלָהֵן כָּלֵב בַּר יְפֻנֶה וִיהוֹשֻׁעַ בַּר נוּן: (ס)




במדבר פרק-כז

{א} וַתִּקְרַ֜בְנָה בְּנ֣וֹת צְלָפְחָ֗ד בֶּן-חֵ֤פֶר בֶּן-גִּלְעָד֙ בֶּן-מָכִ֣יר בֶּן-מְנַשֶּׁ֔ה לְמִשְׁפְּחֹ֖ת מְנַשֶּׁ֣ה בֶן-יוֹסֵ֑ף וְאֵ֨לֶּה֙ שְׁמ֣וֹת בְּנֹתָ֔יו מַחְלָ֣ה נֹעָ֔ה וְחָגְלָ֥ה וּמִלְכָּ֖ה וְתִרְצָֽה:  מ  וַתִּקְרַ֜בְנָה בְּנ֣וֹת צְלָפְחָ֗ד בֶּן-חֵ֤פֶר בֶּן-גִּלְעָד֙ בֶּן-מָכִ֣יר בֶּן-מְנַשֶּׁ֔ה לְמִשְׁפְּחֹ֖ת מְנַשֶּׁ֣ה בֶן-יוֹסֵ֑ף וְאֵ֨לֶּה֙ שְׁמ֣וֹת בְּנֹתָ֔יו מַחְלָ֣ה נֹעָ֔ה וְחָגְלָ֥ה וּמִלְכָּ֖ה וְתִרְצָֽה:  ת  וְקָרֵיבָא בְּנַת צְלָפְחָד בַּר חֵפֶר בַּר גִלְעָד בַּר מָכִיר בַּר מְנַשֶׁה לְזַרְעֲיַת מְנַשֶׁה בַּר יוֹסֵף וְאִלֵין שְׁמָהַת בְּנָתֵיהּ מַחְלָה נוֹעָה וְחָגְלָה וּמִלְכָּה וְתִרְצָה:

{ב} וַֽתַּעֲמֹ֜דְנָה לִפְנֵ֣י מֹשֶׁ֗ה וְלִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֥י הַנְּשִׂיאִ֖ם וְכָל-הָֽעֵדָ֑ה פֶּ֥תַח אֹֽהֶל-מוֹעֵ֖ד לֵאמֹֽר:  מ  וַֽתַּעֲמֹ֜דְנָה לִפְנֵ֣י מֹשֶׁ֗ה וְלִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֥י הַנְּשִׂיאִ֖ם וְכָל-הָֽעֵדָ֑ה פֶּ֥תַח אֹֽהֶל-מוֹעֵ֖ד לֵאמֹֽר:  ת  וְקָמָן קֳדָם משֶׁה וְקֳדָם אֶלְעָזָר כַּהֲנָא וְקֳדָם רַבְרְבַיָא וְכָל כְּנִשְׁתָּא בִּתְרַע מַשְׁכַּן זִמְנָא לְמֵימָר:

{ג} אָבִ֘ינוּ֘ מֵ֣ת בַּמִּדְבָּר֒ וְה֨וּא לֹֽא-הָיָ֜ה בְּת֣וֹךְ הָֽעֵ֗דָה הַנּֽוֹעָדִ֛ים עַל-יְהֹוָ֖ה בַּֽעֲדַת-קֹ֑רַח כִּֽי-בְחֶטְא֣וֹ מֵ֔ת וּבָנִ֖ים לֹא-הָ֥יוּ לֽוֹ:  מ  אָבִ֘ינוּ֘ מֵ֣ת בַּמִּדְבָּר֒ וְה֨וּא לֹֽא-הָיָ֜ה בְּת֣וֹךְ הָֽעֵ֗דָה הַנּֽוֹעָדִ֛ים עַל-יְהֹוָ֖ה בַּֽעֲדַת-קֹ֑רַח כִּֽי-בְחֶטְא֣וֹ מֵ֔ת וּבָנִ֖ים לֹא-הָ֥יוּ לֽוֹ:  ת  אָבוּנָא מִית בְּמַדְבְּרָא וְהוּא לָא הֲוָה בְּגוֹ כְנִשְׁתָּא דְאִזְדַמָנוּ עַל יְיָ בְּכְנִשְׁתָּא דְקֹרַח אֲרֵי בְחוֹבֵיהּ מִית וּבְנִין לָא הֲווֹ לֵיהּ:

{ד} לָ֣מָּה יִגָּרַ֤ע שֵׁם-אָבִ֨ינוּ֙ מִתּ֣וֹךְ מִשְׁפַּחְתּ֔וֹ כִּ֛י אֵ֥ין ל֖וֹ בֵּ֑ן תְּנָה-לָּ֣נוּ אֲחֻזָּ֔ה בְּת֖וֹךְ אֲחֵ֥י אָבִֽינוּ:  מ  לָ֣מָּה יִגָּרַ֤ע שֵׁם-אָבִ֨ינוּ֙ מִתּ֣וֹךְ מִשְׁפַּחְתּ֔וֹ כִּ֛י אֵ֥ין ל֖וֹ בֵּ֑ן תְּנָה-לָּ֣נוּ אֲחֻזָּ֔ה בְּת֖וֹךְ אֲחֵ֥י אָבִֽינוּ:  ת  לְמָא יִתְמְנַע שְׁמָא דַאֲבוּנָא מִגוֹ זַרְעִיתֵיהּ אֲרֵי לֵית לֵיהּ בָּר הַב לָנָא אַחֲסָנָא בְּגוֹ אֲחֵי אָבוּנָא:

{ה} וַיַּקְרֵ֥ב מֹשֶׁ֛ה אֶת-מִשְׁפָּטָ֖ן לִפְנֵ֥י יְהֹוָֽה:  מ  וַיַּקְרֵ֥ב מֹשֶׁ֛ה אֶת-מִשְׁפָּטָ֖ן לִפְנֵ֥י יְהֹוָֽה:  ת  וְקָרֵב משֶׁה יָת דִינְהוֹן קֳדָם יְיָ: (פ)

{ו}  רביעי  וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וַאֲמַר יְיָ לְמשֶׁה לְמֵימָר:

{ז} כֵּ֗ן בְּנ֣וֹת צְלָפְחָד֘ דֹּֽבְרֹת֒ נָתֹ֨ן תִּתֵּ֤ן לָהֶם֙ אֲחֻזַּ֣ת נַֽחֲלָ֔ה בְּת֖וֹךְ אֲחֵ֣י אֲבִיהֶ֑ם וְהַֽעֲבַרְתָּ֛ אֶת-נַֽחֲלַ֥ת אֲבִיהֶ֖ן לָהֶֽן:  מ  כֵּ֗ן בְּנ֣וֹת צְלָפְחָד֘ דֹּֽבְרֹת֒ נָתֹ֨ן תִּתֵּ֤ן לָהֶם֙ אֲחֻזַּ֣ת נַֽחֲלָ֔ה בְּת֖וֹךְ אֲחֵ֣י אֲבִיהֶ֑ם וְהַֽעֲבַרְתָּ֛ אֶת-נַֽחֲלַ֥ת אֲבִיהֶ֖ן לָהֶֽן:  ת  יָאוּת בְּנַת צְלָפְחָד מְמַלְלָן מִתַּן תִּתֵּן לְהוֹן אֲחוּדַת אַחֲסָנָא בְּגוֹ אֲחֵי אֲבוּהֶן וְתַעֲבַר יַת אַחֲסָנַת אֲבוּהֶן לָהֶן:

{ח} וְאֶל-בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר אִ֣ישׁ כִּֽי-יָמ֗וּת וּבֵן֙ אֵ֣ין ל֔וֹ וְהַֽעֲבַרְתֶּ֥ם אֶת-נַֽחֲלָת֖וֹ לְבִתּֽוֹ:  מ  וְאֶל-בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר אִ֣ישׁ כִּֽי-יָמ֗וּת וּבֵן֙ אֵ֣ין ל֔וֹ וְהַֽעֲבַרְתֶּ֥ם אֶת-נַֽחֲלָת֖וֹ לְבִתּֽוֹ:  ת  וְעִם בְּנֵי יִשְׂרָאֵל תְּמַלֵיל לְמֵימָר גְבַר אֲרֵי יְמוּת וּבַר לֵית לֵיהּ וְתַעַבְרוּן יָת אַחֲסַנְתֵּיהּ לִּבְרַתֵּיהּ:

{ט} וְאִם-אֵ֥ין ל֖וֹ בַּ֑ת וּנְתַתֶּ֥ם אֶת-נַֽחֲלָת֖וֹ לְאֶחָֽיו:  מ  וְאִם-אֵ֥ין ל֖וֹ בַּ֑ת וּנְתַתֶּ֥ם אֶת-נַֽחֲלָת֖וֹ לְאֶחָֽיו:  ת  וְאִם לֵית לֵיהּ בְּרַתָּא וְתִתְּנוּן יָת אַחֲסַנְתֵּיה לַאֲחוֹהִי:

{י} וְאִם-אֵ֥ין ל֖וֹ אַחִ֑ים וּנְתַתֶּ֥ם אֶת-נַֽחֲלָת֖וֹ לַֽאֲחֵ֥י אָבִֽיו:  מ  וְאִם-אֵ֥ין ל֖וֹ אַחִ֑ים וּנְתַתֶּ֥ם אֶת-נַֽחֲלָת֖וֹ לַֽאֲחֵ֥י אָבִֽיו:  ת  וְאִם לֵית לֵיהּ אַחִין וְתִתְּנוּן יָת אַחֲסַנְתֵּיהּ לְאֲחֵי אֲבוּהִי:

{יא} וְאִם-אֵ֣ין אַחִים֘ לְאָבִיו֒ וּנְתַתֶּ֣ם אֶת-נַֽחֲלָת֗וֹ לִשְׁאֵר֞וֹ הַקָּרֹ֥ב אֵלָ֛יו מִמִּשְׁפַּחְתּ֖וֹ וְיָרַ֣שׁ אֹתָ֑הּ וְֽהָ֨יְתָ֜ה לִבְנֵ֤י יִשְׂרָאֵל֙ לְחֻקַּ֣ת מִשְׁפָּ֔ט כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת-מֹשֶֽׁה:  מ  וְאִם-אֵ֣ין אַחִים֘ לְאָבִיו֒ וּנְתַתֶּ֣ם אֶת-נַֽחֲלָת֗וֹ לִשְׁאֵר֞וֹ הַקָּרֹ֥ב אֵלָ֛יו מִמִּשְׁפַּחְתּ֖וֹ וְיָרַ֣שׁ אֹתָ֑הּ וְֽהָ֨יְתָ֜ה לִבְנֵ֤י יִשְׂרָאֵל֙ לְחֻקַּ֣ת מִשְׁפָּ֔ט כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת-מֹשֶֽׁה:  ת  וְאִם לֵית אַחִין לַאֲבוּהִי וְתִתְּנוּן יָת אַחֲסַנְתֵּיהּ לְקָרִיבֵיהּ דְקָרֵיבּ לֵיהּ מִזַרְעִיתֵיהּ וְיֵרַת יָתַהּ וּתְהֵי לִבְנֵי יִשְׂרָאֵל לִגְזֵירַת דִין כְּמָא דִי פַקִיד יְיָ יָת משֶׁה: (פ)

{יב} וַיֹּ֤אמֶר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה עֲלֵ֛ה אֶל-הַ֥ר הָֽעֲבָרִ֖ים הַזֶּ֑ה וּרְאֵה֙ אֶת-הָאָ֔רֶץ אֲשֶׁ֥ר נָתַ֖תִּי לִבְנֵ֥י יִשְׂרָאֵֽל:  מ  וַיֹּ֤אמֶר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה עֲלֵ֛ה אֶל-הַ֥ר הָֽעֲבָרִ֖ים הַזֶּ֑ה וּרְאֵה֙ אֶת-הָאָ֔רֶץ אֲשֶׁ֥ר נָתַ֖תִּי לִבְנֵ֥י יִשְׂרָאֵֽל:  ת  וַאֲמַר יְיָ לְמשֶׁה סַק לְטוּרָא דַעֲבָרָאֵי הָדֵין וַחֲזֵי יָת אַרְעָא דִי יְהָבִית לִבְנֵי יִשִׂרָאֵל:

{יג} וְרָאִ֣יתָה אֹתָ֔הּ וְנֶֽאֱסַפְתָּ֥ אֶל-עַמֶּ֖יךָ גַּם-אָ֑תָּה כַּֽאֲשֶׁ֥ר נֶֽאֱסַ֖ף אַֽהֲרֹ֥ן אָחִֽיךָ:  מ  וְרָאִ֣יתָה אֹתָ֔הּ וְנֶֽאֱסַפְתָּ֥ אֶל-עַמֶּ֖יךָ גַּם-אָ֑תָּה כַּֽאֲשֶׁ֥ר נֶֽאֱסַ֖ף אַֽהֲרֹ֥ן אָחִֽיךָ:  ת  וְתֶחֱזֵי יָתָהּ וְתִתְכְנֵּשׁ לְעַמָיךְ אַף אָתְּ כְּמָא דִי אִתְכְּנִישׁ אַהֲרֹן אָחוּךְ:

{יד} כַּֽאֲשֶׁר֩ מְרִיתֶ֨ם פִּ֜י בְּמִדְבַּר-צִ֗ן בִּמְרִיבַת֙ הָֽעֵדָ֔ה לְהַקְדִּישֵׁ֥נִי בַמַּ֖יִם לְעֵֽינֵיהֶ֑ם הֵ֛ם מֵֽי-מְרִיבַ֥ת קָדֵ֖שׁ מִדְבַּר-צִֽן:  מ  כַּֽאֲשֶׁר֩ מְרִיתֶ֨ם פִּ֜י בְּמִדְבַּר-צִ֗ן בִּמְרִיבַת֙ הָֽעֵדָ֔ה לְהַקְדִּישֵׁ֥נִי בַמַּ֖יִם לְעֵֽינֵיהֶ֑ם הֵ֛ם מֵֽי-מְרִיבַ֥ת קָדֵ֖שׁ מִדְבַּר-צִֽן:  ת  כְּמָא דִסָרֵיבְתּוּן עַל מֵימְרִי בְּמַדְבְּרָא דְצִן בְּמַצוּת כְּנִשְׁתָּא לְקַדָשׁוּתִי בְמַיָא לְעֵינֵיהוֹן אִנוּן מֵי מַצוּת רְקָם מַדְבְּרָא דְצִן: (ס)

{טו} וַיְדַבֵּ֣ר מֹשֶׁ֔ה אֶל-יְהֹוָ֖ה לֵאמֹֽר:  מ  וַיְדַבֵּ֣ר מֹשֶׁ֔ה אֶל-יְהֹוָ֖ה לֵאמֹֽר:  ת  וּמַלִיל משֶׁה קֳדָם יְיָ לְמֵימָר:

{טז} יִפְקֹ֣ד יְהֹוָ֔ה אֱלֹהֵ֥י הָֽרוּחֹ֖ת לְכָל-בָּשָׂ֑ר אִ֖ישׁ עַל-הָֽעֵדָֽה:  מ  יִפְקֹ֣ד יְהֹוָ֔ה אֱלֹהֵ֥י הָֽרוּחֹ֖ת לְכָל-בָּשָׂ֑ר אִ֖ישׁ עַל-הָֽעֵדָֽה:  ת  יְמַנֵי יְיָ אֱלָהָא רוּחַיָא לְכָל בִּשְׂרָא גְבַר עַל כְּנִשְׁתָּא:

{יז} אֲשֶׁר-יֵצֵ֣א לִפְנֵיהֶ֗ם וַֽאֲשֶׁ֤ר יָבֹא֙ לִפְנֵיהֶ֔ם וַֽאֲשֶׁ֥ר יֽוֹצִיאֵ֖ם וַֽאֲשֶׁ֣ר יְבִיאֵ֑ם וְלֹ֤א תִֽהְיֶה֙ עֲדַ֣ת יְהֹוָ֔ה כַּצֹּ֕אן אֲשֶׁ֥ר אֵין-לָהֶ֖ם רֹעֶֽה:  מ  אֲשֶׁר-יֵצֵ֣א לִפְנֵיהֶ֗ם וַֽאֲשֶׁ֤ר יָבֹא֙ לִפְנֵיהֶ֔ם וַֽאֲשֶׁ֥ר יֽוֹצִיאֵ֖ם וַֽאֲשֶׁ֣ר יְבִיאֵ֑ם וְלֹ֤א תִֽהְיֶה֙ עֲדַ֣ת יְהֹוָ֔ה כַּצֹּ֕אן אֲשֶׁ֥ר אֵין-לָהֶ֖ם רֹעֶֽה:  ת  דִי יִפּוֹק קֳדָמֵיהוֹן וְדִי יֵעוֹל קֳדָמֵיהוֹן וְדִי יַפֵּקִנוּן וְדִי יָעֵלִינוּן וְלָא תְהֵי כְּנִשְׁתָּא דַיְיָ כְּעָנָא דִי לֵית לְהֵן רָעֵי:

{יח} וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל-מֹשֶׁ֗ה קַח-לְךָ֙ אֶת-יְהוֹשֻׁ֣עַ בִּן-נ֔וּן אִ֖ישׁ אֲשֶׁר-ר֣וּחַ בּ֑וֹ וְסָֽמַכְתָּ֥ אֶת-יָֽדְךָ֖ עָלָֽיו:  מ  וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל-מֹשֶׁ֗ה קַח-לְךָ֙ אֶת-יְהוֹשֻׁ֣עַ בִּן-נ֔וּן אִ֖ישׁ אֲשֶׁר-ר֣וּחַ בּ֑וֹ וְסָֽמַכְתָּ֥ אֶת-יָֽדְךָ֖ עָלָֽיו:  ת  וַאֲמַר יְיָ לְמשֶׁה דְבַר לָךְ יָת יְהוֹשֻׁעַ בַּר נוּן גְבַר דִי רוּחַ נְבוּאָה בֵּיהּ וְתִסְמוֹךְ יָת יְדָךְ עֲלוֹהִי:

{יט} וְהַֽעֲמַדְתָּ֣ אֹת֗וֹ לִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֖י כָּל-הָֽעֵדָ֑ה וְצִוִּיתָ֥ה אֹת֖וֹ לְעֵֽינֵיהֶֽם:  מ  וְהַֽעֲמַדְתָּ֣ אֹת֗וֹ לִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֖י כָּל-הָֽעֵדָ֑ה וְצִוִּיתָ֥ה אֹת֖וֹ לְעֵֽינֵיהֶֽם:  ת  וּתְקֵים יָתֵיהּ לָקֳדָם אֶלְעָזָר כַּהֲנָא וְלָקֳדָם כָּל כְּנִשְׁתָּא וּתְפַקֵד יָתֵיהּ לְעֵינֵיהוֹן:

{כ} וְנָֽתַתָּ֥ה מֵהֽוֹדְךָ֖ עָלָ֑יו לְמַ֣עַן יִשְׁמְע֔וּ כָּל-עֲדַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וְנָֽתַתָּ֥ה מֵהֽוֹדְךָ֖ עָלָ֑יו לְמַ֣עַן יִשְׁמְע֔וּ כָּל-עֲדַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וְתִתֵּן מִזִיוָךְ עֲלוֹהִי בְּדִיל דִי יְקַבְּלוּן מִנֵיהּ כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל:

{כא} וְלִפְנֵ֨י אֶלְעָזָ֤ר הַכֹּהֵן֙ יַֽעֲמֹ֔ד וְשָׁ֥אַל ל֛וֹ בְּמִשְׁפַּ֥ט הָֽאוּרִ֖ים לִפְנֵ֣י יְהֹוָ֑ה עַל-פִּ֨יו יֵֽצְא֜וּ וְעַל-פִּ֣יו יָבֹ֗אוּ ה֛וּא וְכָל-בְּנֵֽי-יִשְׂרָאֵ֥ל אִתּ֖וֹ וְכָל-הָֽעֵדָֽה:  מ  וְלִפְנֵ֨י אֶלְעָזָ֤ר הַכֹּהֵן֙ יַֽעֲמֹ֔ד וְשָׁ֥אַל ל֛וֹ בְּמִשְׁפַּ֥ט הָֽאוּרִ֖ים לִפְנֵ֣י יְהֹוָ֑ה עַל-פִּ֨יו יֵֽצְא֜וּ וְעַל-פִּ֣יו יָבֹ֗אוּ ה֛וּא וְכָל-בְּנֵֽי-יִשְׂרָאֵ֥ל אִתּ֖וֹ וְכָל-הָֽעֵדָֽה:  ת  וְלִקֳדָם אֶלְעָזָר כַּהֲנָא יְקוּם וְיִשְׁאַל לֵיהּ בְּדִין אוּרַיָא קֳדָם יְיָ עַל מֵימְרֵיהּ יְהוֹן נַפְקִין וְעַל מֵימְרֵא יְהוֹן עָאלִין הוּא וְכָל בְּנֵי יִשְׂרָאֵל עִמֵיהּ וְכָל כְּנִשְׁתָּא:

{כב} וַיַּ֣עַשׂ מֹשֶׁ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֹת֑וֹ וַיִּ֣קַּח אֶת-יְהוֹשֻׁ֗עַ וַיַּֽעֲמִדֵ֨הוּ֙ לִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֖י כָּל-הָֽעֵדָֽה:  מ  וַיַּ֣עַשׂ מֹשֶׁ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֹת֑וֹ וַיִּ֣קַּח אֶת-יְהוֹשֻׁ֗עַ וַיַּֽעֲמִדֵ֨הוּ֙ לִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֖י כָּל-הָֽעֵדָֽה:  ת  וַעֲבַד משֶׁה כְּמָא דִי פַקֵיד יְיָ יָתֵיהּ וּדְבַר יָת יְהוֹשֻׁע וַאֲקִימֵיהּ קֳדָם אֶלְעָזָר כַּהֲנָא וּקְדָם כָּל כְּנִשְׁתָּא:

{כג} וַיִּסְמֹ֧ךְ אֶת-יָדָ֛יו עָלָ֖יו וַיְצַוֵּ֑הוּ כַּֽאֲשֶׁ֛ר דִּבֶּ֥ר יְהֹוָ֖ה בְּיַד-מֹשֶֽׁה:  מ  וַיִּסְמֹ֧ךְ אֶת-יָדָ֛יו עָלָ֖יו וַיְצַוֵּ֑הוּ כַּֽאֲשֶׁ֛ר דִּבֶּ֥ר יְהֹוָ֖ה בְּיַד-מֹשֶֽׁה:  ת  וּסְמַךְ יָת יְדוֹהִי עֲלוֹהִי וּפַקְדֵיהּ כְּמָא דִי מַלִיל יְיָ בִּידָא דְמשֶׁה: (פ)




במדבר פרק-כח

{א}  חמישי  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר:

{ב} צַ֚ו אֶת-בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם אֶת-קָרְבָּנִ֨י לַחְמִ֜י לְאִשַּׁ֗י רֵ֚יחַ נִֽיחֹחִ֔י תִּשְׁמְר֕וּ לְהַקְרִ֥יב לִ֖י בְּמֽוֹעֲדֽוֹ:  מ  צַ֚ו אֶת-בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם אֶת-קָרְבָּנִ֨י לַחְמִ֜י לְאִשַּׁ֗י רֵ֚יחַ נִֽיחֹחִ֔י תִּשְׁמְר֕וּ לְהַקְרִ֥יב לִ֖י בְּמֽוֹעֲדֽוֹ:  ת  פַקֵיד יָת בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן יָת קוּרְבָּנִי לְחֵם סִדוּר לְקוּרְבָּנַי לְאִתְקַבָּלָא בְּרַעֲוָא תִּטְרוּן לְקָרָבָא קֳדָמַי בְּזִמְנֵיהּ:

{ג} וְאָֽמַרְתָּ֣ לָהֶ֔ם זֶ֚ה הָֽאִשֶּׁ֔ה אֲשֶׁ֥ר תַּקְרִ֖יבוּ לַֽיהֹוָ֑ה כְּבָשִׂ֨ים בְּנֵֽי-שָׁנָ֧ה תְמִימִ֛ם שְׁנַ֥יִם לַיּ֖וֹם עֹלָ֥ה תָמִֽיד:  מ  וְאָֽמַרְתָּ֣ לָהֶ֔ם זֶ֚ה הָֽאִשֶּׁ֔ה אֲשֶׁ֥ר תַּקְרִ֖יבוּ לַֽיהֹוָ֑ה כְּבָשִׂ֨ים בְּנֵֽי-שָׁנָ֧ה תְמִימִ֛ם שְׁנַ֥יִם לַיּ֖וֹם עֹלָ֥ה תָמִֽיד:  ת  וְתֵימַר לְהוֹן דֵין קוּרְבָּנָא דִי תְקַרְבוּן קֳדָם יְיָ אִימְּרִין בְּנֵי שְׁנָא שַׁלְמִין תְּרֵין לְּיוֹמָא עֲלָתָא תְּדִירָא:

{ד} אֶת-הַכֶּ֥בֶשׂ אֶחָ֖ד תַּֽעֲשֶׂ֣ה בַבֹּ֑קֶר וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּֽעֲשֶׂ֖ה בֵּ֥ין הָֽעַרְבָּֽיִם:  מ  אֶת-הַכֶּ֥בֶשׂ אֶחָ֖ד תַּֽעֲשֶׂ֣ה בַבֹּ֑קֶר וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּֽעֲשֶׂ֖ה בֵּ֥ין הָֽעַרְבָּֽיִם:  ת  יָת אִמְרָא חָד תַּעֲבֵד בְּצַפְרָא וְיָת אִמְרָא תִנְיָנָא תַּעֲבֵד בֵּין שִׁמְשַׁיָא:

{ה} וַֽעֲשִׂירִ֧ית הָֽאֵיפָ֛ה סֹ֖לֶת לְמִנְחָ֑ה בְּלוּלָ֛ה בְּשֶׁ֥מֶן כָּתִ֖ית רְבִיעִ֥ת הַהִֽין:  מ  וַֽעֲשִׂירִ֧ית הָֽאֵיפָ֛ה סֹ֖לֶת לְמִנְחָ֑ה בְּלוּלָ֛ה בְּשֶׁ֥מֶן כָּתִ֖ית רְבִיעִ֥ת הַהִֽין:  ת  וְחָד מִן עַשְׂרָא בִּתְלַת סָאִין סוּלְתָּא לְמִנְחָתָא דְפִילָא בִּמְשַׁח כְּתִישָׁא רַבְעוּת הִינָא:

{ו} עֹלַ֖ת תָּמִ֑יד הָֽעֲשֻׂיָה֙ בְּהַ֣ר סִינַ֔י לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַֽיהֹוָֽה:  מ  עֹלַ֖ת תָּמִ֑יד הָֽעֲשֻׂיָה֙ בְּהַ֣ר סִינַ֔י לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַֽיהֹוָֽה:  ת  עֲלָתָא תְּדִירָא דְאִתְעֲבִידָא בְּטוּרָא דְסִינַי לְאִתְקַבָּלָא בְרַעֲוָא קוּרְבָּנָא קֳדָם יְיָ:

{ז} וְנִסְכּוֹ֙ רְבִיעִ֣ת הַהִ֔ין לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד בַּקֹּ֗דֶשׁ הַסֵּ֛ךְ נֶ֥סֶךְ שֵׁכָ֖ר לַֽיהֹוָֽה:  מ  וְנִסְכּוֹ֙ רְבִיעִ֣ת הַהִ֔ין לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד בַּקֹּ֗דֶשׁ הַסֵּ֛ךְ נֶ֥סֶךְ שֵׁכָ֖ר לַֽיהֹוָֽה:  ת  וְנִסְכֵּיהּ רַבְעוּת הִינָא לְאִמְרָא חָד בְּקוּדְשָׁא יִתְנַסַךְ נִסוּךְ דַחֲמַר עַתִּיק קֳדָם יְיָ:

{ח} וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּֽעֲשֶׂ֖ה בֵּ֣ין הָֽעַרְבָּ֑יִם כְּמִנְחַ֨ת הַבֹּ֤קֶר וּכְנִסְכּוֹ֙ תַּֽעֲשֶׂ֔ה אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַֽיהֹוָֽה:  מ  וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּֽעֲשֶׂ֖ה בֵּ֣ין הָֽעַרְבָּ֑יִם כְּמִנְחַ֨ת הַבֹּ֤קֶר וּכְנִסְכּוֹ֙ תַּֽעֲשֶׂ֔ה אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַֽיהֹוָֽה:  ת  וְיָת אִמְרָא תִנְיָנָא תַּעֲבֵיד בֵּין שִׁמְשַׁיָא כְּמִנְחַת צַפְרָא וּכְנִסְכֵּיהּ תַּעְבֵיד קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ: (פ)

{ט} וּבְיוֹם֙ הַשַּׁבָּ֔ת שְׁנֵֽי-כְבָשִׂ֥ים בְּנֵֽי-שָׁנָ֖ה תְּמִימִ֑ם וּשְׁנֵ֣י עֶשְׂרֹנִ֗ים סֹ֧לֶת מִנְחָ֛ה בְּלוּלָ֥ה בַשֶּׁ֖מֶן וְנִסְכּֽוֹ:  מ  וּבְיוֹם֙ הַשַּׁבָּ֔ת שְׁנֵֽי-כְבָשִׂ֥ים בְּנֵֽי-שָׁנָ֖ה תְּמִימִ֑ם וּשְׁנֵ֣י עֶשְׂרֹנִ֗ים סֹ֧לֶת מִנְחָ֛ה בְּלוּלָ֥ה בַשֶּׁ֖מֶן וְנִסְכּֽוֹ:  ת  וּבְיוֹמָא דְשַׁבַּתָּא תְּרֵין אִימְרִין בְּנֵי שְׁנָא שַׁלְמִין וּתְרֵין עִשְׂרוֹנִין סוּלְתָּא דְמִנְחָתָא דְפִילָא בִּמְשַׁח וְנִסְכֵּיהּ:

{י} עֹלַ֥ת שַׁבַּ֖ת בְּשַׁבַּתּ֑וֹ עַל-עֹלַ֥ת הַתָּמִ֖יד וְנִסְכָּֽהּ:  מ  עֹלַ֥ת שַׁבַּ֖ת בְּשַׁבַּתּ֑וֹ עַל-עֹלַ֥ת הַתָּמִ֖יד וְנִסְכָּֽהּ:  ת  עֲלַת שַׁבְּתָא דְתִּתְעֲבֵיד בְּשַׁבַּתָּא עַל עֲלָתָא תְּדִירָא וְנִסְכַּהּ: (פ)

{יא} וּבְרָאשֵׁי֙ חָדְשֵׁיכֶ֔ם תַּקְרִ֥יבוּ עֹלָ֖ה לַֽיהֹוָ֑ה פָּרִ֨ים בְּנֵֽי-בָקָ֤ר שְׁנַ֨יִם֙ וְאַ֣יִל אֶחָ֔ד כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם:  מ  וּבְרָאשֵׁי֙ חָדְשֵׁיכֶ֔ם תַּקְרִ֥יבוּ עֹלָ֖ה לַֽיהֹוָ֑ה פָּרִ֨ים בְּנֵֽי-בָקָ֤ר שְׁנַ֨יִם֙ וְאַ֣יִל אֶחָ֔ד כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם:  ת  וּבְרֵישֵׁי יַרְחֵיכוֹן תְּקַרְבוּן עֲלָתָא קֳדָם יְיָ תּוֹרִין בְּנֵי תּוֹרֵי תְּרֵין וּדְכַר חָד אִימְרִין בְּנֵי שְׁנָא שַׁבְעָא שַׁלְמִין:

{יב} וּשְׁלֹשָׁ֣ה עֶשְׂרֹנִ֗ים סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לַפָּ֖ר הָֽאֶחָ֑ד וּשְׁנֵ֣י עֶשְׂרֹנִ֗ים סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לָאַ֖יִל הָֽאֶחָֽד:  מ  וּשְׁלֹשָׁ֣ה עֶשְׂרֹנִ֗ים סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לַפָּ֖ר הָֽאֶחָ֑ד וּשְׁנֵ֣י עֶשְׂרֹנִ֗ים סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לָאַ֖יִל הָֽאֶחָֽד:  ת  וּתְלָתָא עִשְׂרוֹנִין סוּלְתָּא מִנְחָתָא דְפִילָא בִמְשַׁח לְתוֹרָא חָד וּתְרֵין עִשְׂרוֹנִין סֻלְתָּא מִנְחָתָא דְפִילָא בִמְשַׁח לְדִכְרָא חָד:

{יג} וְעִשָּׂרֹ֣ן עִשָּׂר֗וֹן סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד עֹלָה֙ רֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַֽיהֹוָֽה:  מ  וְעִשָּׂרֹ֣ן עִשָּׂר֗וֹן סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד עֹלָה֙ רֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַֽיהֹוָֽה:  ת  וְעִשְׂרוֹנָא עִשְׂרוֹנָא סֻלְתָּא מִנְחָתָא דְפִילָא בִמְשַׁח לְאִמְרָא חָד עֲלָתָא לְאִתְקַבָּלָא בְּרַעֲוָא קוּרְבָּנָא קֳדָם יְיָ:

{יד} וְנִסְכֵּיהֶ֗ם חֲצִ֣י הַהִין֩ יִֽהְיֶ֨ה לַפָּ֜ר וּשְׁלִישִׁ֧ת הַהִ֣ין לָאַ֗יִל וּרְבִיעִ֥ת הַהִ֛ין לַכֶּ֖בֶשׂ יָ֑יִן זֹ֣את עֹלַ֥ת חֹ֨דֶשׁ֙ בְּחָדְשׁ֔וֹ לְחָדְשֵׁ֖י הַשָּׁנָֽה:  מ  וְנִסְכֵּיהֶ֗ם חֲצִ֣י הַהִין֩ יִֽהְיֶ֨ה לַפָּ֜ר וּשְׁלִישִׁ֧ת הַהִ֣ין לָאַ֗יִל וּרְבִיעִ֥ת הַהִ֛ין לַכֶּ֖בֶשׂ יָ֑יִן זֹ֣את עֹלַ֥ת חֹ֨דֶשׁ֙ בְּחָדְשׁ֔וֹ לְחָדְשֵׁ֖י הַשָּׁנָֽה:  ת  וְנִסְכֵּיהוֹן פַּלְגוּת הִינָא יְהֵי לְתוֹרָא וְתַלְתּוּת הִינָא לְדִכְרָא וְרַבְעוּת הִינָא לְאִמְרָא חַמְרָא דָא עֲלַת רֵישׁ יַרְחָא בְּאִתְחַדָתוּתֵיהּ כֵּן לְכָל רֵישֵׁי יַרְחֵי שַׁתָּא:

{טו} וּשְׂעִ֨יר עִזִּ֥ים אֶחָ֛ד לְחַטָּ֖את לַֽיהֹוָ֑ה עַל-עֹלַ֧ת הַתָּמִ֛יד יֵֽעָשֶׂ֖ה וְנִסְכּֽוֹ:  מ  וּשְׂעִ֨יר עִזִּ֥ים אֶחָ֛ד לְחַטָּ֖את לַֽיהֹוָ֑ה עַל-עֹלַ֧ת הַתָּמִ֛יד יֵֽעָשֶׂ֖ה וְנִסְכּֽוֹ:  ת  וּצְפִיר בַּר עִזִין חָד לְחַטָאתָא קֳדָם יְיָ עַל עֲלָתָא תְּדִירָא יִתְעֲבֵד וְנִסְכֵּיהּ: (ס)

{טז}  שישי  וּבַחֹ֣דֶשׁ הָֽרִאשׁ֗וֹן בְּאַרְבָּעָ֥ה עָשָׂ֛ר י֖וֹם לַחֹ֑דֶשׁ פֶּ֖סַח לַֽיהֹוָֽה:  מ  וּבַחֹ֣דֶשׁ הָֽרִאשׁ֗וֹן בְּאַרְבָּעָ֥ה עָשָׂ֛ר י֖וֹם לַחֹ֑דֶשׁ פֶּ֖סַח לַֽיהֹוָֽה:  ת  וּבְיַרְחָא קַדְמָאָה בְּאַרְבַּעַת עַשְׂרָא יוֹמָא לְיַרְחָא פִּסְחָא קֳדָם יְיָ:

{יז} וּבַֽחֲמִשָּׁ֨ה עָשָׂ֥ר י֛וֹם לַחֹ֥דֶשׁ הַזֶּ֖ה חָ֑ג שִׁבְעַ֣ת יָמִ֔ים מַצּ֖וֹת יֵֽאָכֵֽל:  מ  וּבַֽחֲמִשָּׁ֨ה עָשָׂ֥ר י֛וֹם לַחֹ֥דֶשׁ הַזֶּ֖ה חָ֑ג שִׁבְעַ֣ת יָמִ֔ים מַצּ֖וֹת יֵֽאָכֵֽל:  ת  וּבְחַמְשַׁת עַשְׂרָא יוֹמָא לְיַרְחָא הָדֵין חַגָא שַׁבְעָא יוֹמִין פַּטִיר יִתְאָכֵל:

{יח} בַּיּ֥וֹם הָֽרִאשׁ֖וֹן מִקְרָא-קֹ֑דֶשׁ כָּל-מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ:  מ  בַּיּ֥וֹם הָֽרִאשׁ֖וֹן מִקְרָא-קֹ֑דֶשׁ כָּל-מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ:  ת  בְּיוֹמָא קַדְמָאָה מְעָרַע קַדִישׁ כָּל עֲבִידַת פּוּלְחַן לָא תַעְבְדוּן:

{יט} וְהִקְרַבְתֶּ֨ם אִשֶּׁ֤ה עֹלָה֙ לַֽיהֹוָ֔ה פָּרִ֧ים בְּנֵֽי-בָקָ֛ר שְׁנַ֖יִם וְאַ֣יִל אֶחָ֑ד וְשִׁבְעָ֤ה כְבָשִׂים֙ בְּנֵ֣י שָׁנָ֔ה תְּמִימִ֖ם יִֽהְי֥וּ לָכֶֽם:  מ  וְהִקְרַבְתֶּ֨ם אִשֶּׁ֤ה עֹלָה֙ לַֽיהֹוָ֔ה פָּרִ֧ים בְּנֵֽי-בָקָ֛ר שְׁנַ֖יִם וְאַ֣יִל אֶחָ֑ד וְשִׁבְעָ֤ה כְבָשִׂים֙ בְּנֵ֣י שָׁנָ֔ה תְּמִימִ֖ם יִֽהְי֥וּ לָכֶֽם:  ת  וּתְקַרְבוּן קוּרְבָּנָא עֲלָתָא קֳדָם יְיָ תּוֹרִין בְּנֵי תוֹרֵי תְּרֵין וּדְכַר חָד וְשַׁבְעָא אִימְרִין בְּנֵי שְׁנָא שַׁלְמִין יְהוֹן לְכוֹן:

{כ} וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֨ה עֶשְׂרֹנִ֜ים לַפָּ֗ר וּשְׁנֵ֧י עֶשְׂרֹנִ֑ים לָאַ֖יִל תַּֽעֲשֽׂוּ:  מ  וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֨ה עֶשְׂרֹנִ֜ים לַפָּ֗ר וּשְׁנֵ֧י עֶשְׂרֹנִ֑ים לָאַ֖יִל תַּֽעֲשֽׂוּ:  ת  וּמִנְחַתְהוֹן סוּלְתָּא דְפִילָא בִּמְשַׁח תְּלָתָא עִשְׂרוֹנִין לְתוֹרָא וּתְרֵין עִשְׂרוֹנִין לְדִכְרָא תַּעְבְּדוּן:

{כא} עִשָּׂר֤וֹן עִשָּׂרוֹן֙ תַּֽעֲשֶׂ֔ה לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים:  מ  עִשָּׂר֤וֹן עִשָּׂרוֹן֙ תַּֽעֲשֶׂ֔ה לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים:  ת  עִשְׂרוֹנָא עִשְׂרוֹנָא תַּעֲבֵיד לְאִמְרָא חָד כֵּן לְשַׁבְעָא אִימְרִין:

{כב} וּשְׂעִ֥יר חַטָּ֖את אֶחָ֑ד לְכַפֵּ֖ר עֲלֵיכֶֽם:  מ  וּשְׂעִ֥יר חַטָּ֖את אֶחָ֑ד לְכַפֵּ֖ר עֲלֵיכֶֽם:  ת  וּצְפִירָא דְחַטָאתָא חָד לְכַפָּרָא עֲלֵיכוֹן:

{כג} מִלְּבַד֙ עֹלַ֣ת הַבֹּ֔קֶר אֲשֶׁ֖ר לְעֹלַ֣ת הַתָּמִ֑יד תַּֽעֲשׂ֖וּ אֶת-אֵֽלֶּה:  מ  מִלְּבַד֙ עֹלַ֣ת הַבֹּ֔קֶר אֲשֶׁ֖ר לְעֹלַ֣ת הַתָּמִ֑יד תַּֽעֲשׂ֖וּ אֶת-אֵֽלֶּה:  ת  בַּר מֵעֲלַת צַפְרָא דְהִּיא עֲלַת תְּדִירָא תַּעְבְּדוּן יָת אִלֵין:

{כד} כָּאֵ֜לֶּה תַּֽעֲשׂ֤וּ לַיּוֹם֙ שִׁבְעַ֣ת יָמִ֔ים לֶ֛חֶם אִשֵּׁ֥ה רֵֽיחַ-נִיחֹ֖חַ לַֽיהֹוָ֑ה עַל-עוֹלַ֧ת הַתָּמִ֛יד יֵֽעָשֶׂ֖ה וְנִסְכּֽוֹ:  מ  כָּאֵ֜לֶּה תַּֽעֲשׂ֤וּ לַיּוֹם֙ שִׁבְעַ֣ת יָמִ֔ים לֶ֛חֶם אִשֵּׁ֥ה רֵֽיחַ-נִיחֹ֖חַ לַֽיהֹוָ֑ה עַל-עוֹלַ֧ת הַתָּמִ֛יד יֵֽעָשֶׂ֖ה וְנִסְכּֽוֹ:  ת  כְּאִלֵין תַּעַבְדוּן לְיוֹמָא שַׁבְעָא יוֹמִין לְחֵם קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ עַל עֲלָתָא תְדִירָא יִתְעֲבֵיד וִנִסִכֵּיהּ:

{כז} וְהִקְרַבְתֶּ֨ם עוֹלָ֜ה לְרֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה פָּרִ֧ים בְּנֵֽי-בָקָ֛ר שְׁנַ֖יִם אַ֣יִל אֶחָ֑ד שִׁבְעָ֥ה כְבָשִׂ֖ים בְּנֵ֥י שָׁנָֽה:  מ  וְהִקְרַבְתֶּ֨ם עוֹלָ֜ה לְרֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה פָּרִ֧ים בְּנֵֽי-בָקָ֛ר שְׁנַ֖יִם אַ֣יִל אֶחָ֑ד שִׁבְעָ֥ה כְבָשִׂ֖ים בְּנֵ֥י שָׁנָֽה:  ת  וּתְקַרְבוּן עֲלָתָא לְאִתְקַבָּלָא בְרַעֲוָא קֳדָם יְיָ תּוֹרִין בְּנֵי תוֹרֵי תְּרֵין וּדְכַר חָד שַׁבְעָא אִימְרִין בְּנֵי שְׁנָא:

{כח} וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֣ר הָֽאֶחָ֔ד שְׁנֵי֙ עֶשְׂרֹנִ֔ים לָאַ֖יִל הָֽאֶחָֽד:  מ  וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֣ר הָֽאֶחָ֔ד שְׁנֵי֙ עֶשְׂרֹנִ֔ים לָאַ֖יִל הָֽאֶחָֽד:  ת  וּמִנְחַתְהוֹן סֻלְתָּא דְפִילָא בִמְשָׁח תְּלָתָא עִשְׂרוֹנִין לְתוֹרָא חָד תְּרֵין עִשְׂרוֹנִין לְדִכְרָא חָד:

{כט} עִשָּׂרוֹן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים:  מ  עִשָּׂרוֹן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים:  ת  עִשְׂרוֹנָא עִשְׂרוֹנָא לְאִמְרָא חָד לְשַׁבְעָא אִימְרִין:

{ל} שְׂעִ֥יר עִזִּ֖ים אֶחָ֑ד לְכַפֵּ֖ר עֲלֵיכֶֽם:  מ  שְׂעִ֥יר עִזִּ֖ים אֶחָ֑ד לְכַפֵּ֖ר עֲלֵיכֶֽם:  ת  צְפִיר בַּר עִזִין חָד לְכַפָּרָא עֲלֵיכוֹן:

{לא} מִלְּבַ֞ד עֹלַ֧ת הַתָּמִ֛יד וּמִנְחָת֖וֹ תַּֽעֲשׂ֑וּ תְּמִימִ֥ם יִֽהְיוּ-לָכֶ֖ם וְנִסְכֵּיהֶֽם:  מ  מִלְּבַ֞ד עֹלַ֧ת הַתָּמִ֛יד וּמִנְחָת֖וֹ תַּֽעֲשׂ֑וּ תְּמִימִ֥ם יִֽהְיוּ-לָכֶ֖ם וְנִסְכֵּיהֶֽם:  ת  בַּר מֵעֲלַת תְּדִירָא וּמִנְחָתֵהּ תַּעְבְּדוּן שַׁלְמִין יְהוֹן לְכוֹן וְנִסְכֵּיהוֹן: (פ)




במדבר פרק-כט

{א} וּבַחֹ֨דֶשׁ הַשְּׁבִיעִ֜י בְּאֶחָ֣ד לַחֹ֗דֶשׁ מִֽקְרָא-קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם כָּל-מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַֽעֲשׂ֑וּ י֥וֹם תְּרוּעָ֖ה יִֽהְיֶ֥ה לָכֶֽם:  מ  וּבַחֹ֨דֶשׁ הַשְּׁבִיעִ֜י בְּאֶחָ֣ד לַחֹ֗דֶשׁ מִֽקְרָא-קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם כָּל-מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַֽעֲשׂ֑וּ י֥וֹם תְּרוּעָ֖ה יִֽהְיֶ֥ה לָכֶֽם:  ת  וּבְיַרְחָא שְׁבִיעָאָה בְּחָד לְיַרְחָא מְעָרַע קַדִישׁ יְהֵי לְכוֹן כָּל עִבִידַת פּוּלְחַן לָא תַעַבְדוּן יוֹם יַבָּבָא יְהֵי לְכוֹן:

{ב} וַֽעֲשִׂיתֶ֨ם עֹלָ֜ה לְרֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה פַּ֧ר בֶּן-בָּקָ֛ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם:  מ  וַֽעֲשִׂיתֶ֨ם עֹלָ֜ה לְרֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה פַּ֧ר בֶּן-בָּקָ֛ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם:  ת  וְתַעַבְדוּן עֲלָתָא לְאִתְקַבָּלָא בְּרַעֲוָא קֳדָם יְיָ תּוֹר בַּר תּוֹרֵי חָד דְכַר חָד אִימְרִין בְּנֵי שְׁנָא שַׁבְעָא שַׁלְמִין:

{ג} וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֔ר שְׁנֵ֥י עֶשְׂרֹנִ֖ים לָאָֽיִל:  מ  וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֔ר שְׁנֵ֥י עֶשְׂרֹנִ֖ים לָאָֽיִל:  ת  וּמִנְחַתְהוֹן סוּלְתָּא דְפִילָא בִמְשַׁח תְּלָתָא עִשְׂרוֹנִין לְתוֹרָא תְּרֵין עִשְׂרוֹנִין לְדִכְרָא:

{ד} וְעִשָּׂר֣וֹן אֶחָ֔ד לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים:  מ  וְעִשָּׂר֣וֹן אֶחָ֔ד לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים:  ת  וְעִשְׂרוֹנָא חָד לְאִימְרָא חָד לְשַׁבְעָא אִימְרִין:

{ה} וּשְׂעִיר-עִזִּ֥ים אֶחָ֖ד חַטָּ֑את לְכַפֵּ֖ר עֲלֵיכֶֽם:  מ  וּשְׂעִיר-עִזִּ֥ים אֶחָ֖ד חַטָּ֑את לְכַפֵּ֖ר עֲלֵיכֶֽם:  ת  וּצְפִיר בַּר עִזִין חָד לְחַטָאתָא לְכַפָּרָא עֲלֵיכוֹן:

{ו} מִלְּבַד֩ עֹלַ֨ת הַחֹ֜דֶשׁ וּמִנְחָתָ֗הּ וְעֹלַ֤ת הַתָּמִיד֙ וּמִנְחָתָ֔הּ וְנִסְכֵּיהֶ֖ם כְּמִשְׁפָּטָ֑ם לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַיֽהֹוָֽה:  מ  מִלְּבַד֩ עֹלַ֨ת הַחֹ֜דֶשׁ וּמִנְחָתָ֗הּ וְעֹלַ֤ת הַתָּמִיד֙ וּמִנְחָתָ֔הּ וְנִסְכֵּיהֶ֖ם כְּמִשְׁפָּטָ֑ם לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַיֽהֹוָֽה:  ת  בַּר מֵעֲלַת יַרְחָא וּמִנְחָתַהּ וַעֲלַת תְּדִירָא וּמִנְחָתַהּ וְנִסְכֵּיהוֹן כִּדְחָזֵי לְהוֹן לְאִתְקַבָּלָא בְרַעֲוָא קוּרְבָּנָא קֳדָם יְיָ: (ס)

{ז} וּבֶֽעָשׂוֹר֩ לַחֹ֨דֶשׁ הַשְּׁבִיעִ֜י הַזֶּ֗ה מִֽקְרָא-קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם וְעִנִּיתֶ֖ם אֶת-נַפְשֹֽׁתֵיכֶ֑ם כָּל-מְלָאכָ֖ה לֹ֥א תַֽעֲשֽׂוּ:  מ  וּבֶֽעָשׂוֹר֩ לַחֹ֨דֶשׁ הַשְּׁבִיעִ֜י הַזֶּ֗ה מִֽקְרָא-קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם וְעִנִּיתֶ֖ם אֶת-נַפְשֹֽׁתֵיכֶ֑ם כָּל-מְלָאכָ֖ה לֹ֥א תַֽעֲשֽׂוּ:  ת  וּבְעַשְׂרָא לְיַרְחָא שְׁבִיעָאָה הָדֵין מְעָרַע קַדִישׁ יְהֵי לְכוֹן וּתְעַנוּן יָת נַפְשָׁתֵיכוֹן כָּל עִבִידְתָּא לָא תַעַבְדוּן:

{ח} וְהִקְרַבְתֶּ֨ם עֹלָ֤ה לַֽיהֹוָה֙ רֵ֣יחַ נִיחֹ֔חַ פַּ֧ר בֶּן-בָּקָ֛ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֤ים בְּנֵֽי-שָׁנָה֙ שִׁבְעָ֔ה תְּמִימִ֖ם יִֽהְי֥וּ לָכֶֽם:  מ  וְהִקְרַבְתֶּ֨ם עֹלָ֤ה לַֽיהֹוָה֙ רֵ֣יחַ נִיחֹ֔חַ פַּ֧ר בֶּן-בָּקָ֛ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֤ים בְּנֵֽי-שָׁנָה֙ שִׁבְעָ֔ה תְּמִימִ֖ם יִֽהְי֥וּ לָכֶֽם:  ת  וּתְקָרְבוּן עֲלָתָא קֳדָם יְיָ לְאִתְקַבָּלָא בְרַעֲוָא תּוֹר בַּר תּוֹרֵי חָד דְכַר חָד אִימְרִין בְּנֵי שְׁנָא שַׁבְעָא שַׁלְמִין יְהוֹן לְכוֹן:

{ט} וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֔ר שְׁנֵי֙ עֶשְׂרֹנִ֔ים לָאַ֖יִל הָֽאֶחָֽד:  מ  וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֔ר שְׁנֵי֙ עֶשְׂרֹנִ֔ים לָאַ֖יִל הָֽאֶחָֽד:  ת  וּמִנְחַתְהוֹן סֻלְתָּא דְפִילָא בִמְשָׁח תְּלָתָא עִשְׂרוֹנִין לְתוֹרָא תְּרֵין עִשְׂרוֹנִין לְדִכְרָא חָד:

{י} עִשָּׁרוֹן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים:  מ  עִשָּׁרוֹן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים:  ת  עִשְׂרוֹנָא עִשְׂרוֹנָא לְאִימְרָא חָד לְשַׁבְעָא אִימְרִין:

{יא} שְׂעִיר-עִזִּ֥ים אֶחָ֖ד חַטָּ֑את מִלְּבַ֞ד חַטַּ֤את הַכִּפֻּרִים֙ וְעֹלַ֣ת הַתָּמִ֔יד וּמִנְחָתָ֖הּ וְנִסְכֵּיהֶֽם:  מ  שְׂעִיר-עִזִּ֥ים אֶחָ֖ד חַטָּ֑את מִלְּבַ֞ד חַטַּ֤את הַכִּפֻּרִים֙ וְעֹלַ֣ת הַתָּמִ֔יד וּמִנְחָתָ֖הּ וְנִסְכֵּיהֶֽם:  ת  צְפִיר בַּר עִזִין חָד חַטָּאתָא בַּר מֵחַטָאתָא דְכִפוּרַיָא וַעֲלַת תְּדִירָא וּמִנְחָתַהּ וְנִסְכֵּיהוֹן: (ס)

{יב}  שביעי  וּבַֽחֲמִשָּׁה֩ עָשָׂ֨ר י֜וֹם לַחֹ֣דֶשׁ הַשְּׁבִיעִ֗י מִֽקְרָא-קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם כָּל-מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַֽעֲשׂ֑וּ וְחַגֹּתֶ֥ם חַ֛ג לַֽיהֹוָ֖ה שִׁבְעַ֥ת יָמִֽים:  מ  וּבַֽחֲמִשָּׁה֩ עָשָׂ֨ר י֜וֹם לַחֹ֣דֶשׁ הַשְּׁבִיעִ֗י מִֽקְרָא-קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם כָּל-מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַֽעֲשׂ֑וּ וְחַגֹּתֶ֥ם חַ֛ג לַֽיהֹוָ֖ה שִׁבְעַ֥ת יָמִֽים:  ת  וּבְחַמְשַׁת עַשְׂרָא יוֹמָא לְיַרְחָא שְׁבִיעָאָה מְעָרַע קַדִישׁ יְהֵי לְכוֹן כָּל עִבִידַת פּוּלְחַן לָא תַעַבְדוּן וּתְחַגוּן חַגָא קֳדָם יְיָ שַׁבְעָא יוֹמִין:

{יג} וְהִקְרַבְתֶּ֨ם עֹלָ֜ה אִשֵּׁ֨ה רֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה פָּרִ֧ים בְּנֵֽי-בָקָ֛ר שְׁלֹשָׁ֥ה עָשָׂ֖ר אֵילִ֣ם שְׁנָ֑יִם כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה אַרְבָּעָ֥ה עָשָׂ֖ר תְּמִימִ֥ם יִֽהְיֽוּ:  מ  וְהִקְרַבְתֶּ֨ם עֹלָ֜ה אִשֵּׁ֨ה רֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה פָּרִ֧ים בְּנֵֽי-בָקָ֛ר שְׁלֹשָׁ֥ה עָשָׂ֖ר אֵילִ֣ם שְׁנָ֑יִם כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה אַרְבָּעָ֥ה עָשָׂ֖ר תְּמִימִ֥ם יִֽהְיֽוּ:  ת  וּתְקַרְבוּן עֲלָתָא קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ תּוֹרִין בְּנֵי תוֹרֵי תְּלַת עֲשַׂר דִכְרִין תְּרֵין אִימְרִין בְּנֵי שְׁנָא אַרְבְּעָא עֲשַׂר שַׁלְמִין יְהוֹן:

{יד} וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֨ה עֶשְׂרֹנִ֜ים לַפָּ֣ר הָֽאֶחָ֗ד לִשְׁלֹשָׁ֤ה עָשָׂר֙ פָּרִ֔ים שְׁנֵ֤י עֶשְׂרֹנִים֙ לָאַ֣יִל הָֽאֶחָ֔ד לִשְׁנֵ֖י הָֽאֵילִֽם:  מ  וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֨ה עֶשְׂרֹנִ֜ים לַפָּ֣ר הָֽאֶחָ֗ד לִשְׁלֹשָׁ֤ה עָשָׂר֙ פָּרִ֔ים שְׁנֵ֤י עֶשְׂרֹנִים֙ לָאַ֣יִל הָֽאֶחָ֔ד לִשְׁנֵ֖י הָֽאֵילִֽם:  ת  וּמִנְחַתְהוֹן סוּלְתָּא דְפִילָא בִמְשַׁח תְּלָתָא עִשְׂרוֹנִין לְתוֹרָא חָד לִתְלַת עֲשַׂר תּוֹרִין תְּרֵין עִשְׂרוֹנִין לְדִכְרָא חָד לִתְרֵין דִכְרִין:

{טו} וְעִשָּׂרוֹן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְאַרְבָּעָ֥ה עָשָׂ֖ר כְּבָשִֽׂים:  מ  וְעִשָּׂרוֹן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְאַרְבָּעָ֥ה עָשָׂ֖ר כְּבָשִֽׂים:  ת  וְעִשְׂרוֹנָא עִשְׂרוֹנָא לְאִימְרָא חָד לְאַרְבְּעָא עֲשַׂר אִימְרִין:

{טז} וּשְׂעִיר-עִזִּ֥ים אֶחָ֖ד חַטָּ֑את מִלְּבַד֙ עֹלַ֣ת הַתָּמִ֔יד מִנְחָתָ֖הּ וְנִסְכָּֽהּ:  מ  וּשְׂעִיר-עִזִּ֥ים אֶחָ֖ד חַטָּ֑את מִלְּבַד֙ עֹלַ֣ת הַתָּמִ֔יד מִנְחָתָ֖הּ וְנִסְכָּֽהּ:  ת  וּצְפִיר בַּר עִזִין חָד חַטָאתָא בַּר מֵעֲלַת תְּדִירָא מִנְחָתַהּ וְנִסְכַּהּ: (ס)

{יז} וּבַיּ֣וֹם הַשֵּׁנִ֗י פָּרִ֧ים בְּנֵֽי-בָקָ֛ר שְׁנֵ֥ים עָשָׂ֖ר אֵילִ֣ם שְׁנָ֑יִם כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה אַרְבָּעָ֥ה עָשָׂ֖ר תְּמִימִֽם:  מ  וּבַיּ֣וֹם הַשֵּׁנִ֗י פָּרִ֧ים בְּנֵֽי-בָקָ֛ר שְׁנֵ֥ים עָשָׂ֖ר אֵילִ֣ם שְׁנָ֑יִם כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה אַרְבָּעָ֥ה עָשָׂ֖ר תְּמִימִֽם:  ת  וּבְיוֹמָא תִנְיָנָא תּוֹרִין בְּנֵי תוֹרֵי תְּרֵי עֲשַׂר דִכְרִין תְּרֵין אִימְרִין בְּנֵי שְׁנָא אַרְבְּעָא עֲשַׂר שַׁלְמִין:

{יח} וּמִנְחָתָ֣ם וְנִסְכֵּיהֶ֡ם לַ֠פָּרִ֠ים לָֽאֵילִ֧ם וְלַכְּבָשִׂ֛ים בְּמִסְפָּרָ֖ם כַּמִּשְׁפָּֽט:  מ  וּמִנְחָתָ֣ם וְנִסְכֵּיהֶ֡ם לַ֠פָּרִ֠ים לָֽאֵילִ֧ם וְלַכְּבָשִׂ֛ים בְּמִסְפָּרָ֖ם כַּמִּשְׁפָּֽט:  ת  וּמִנְחַתְהוֹן וְנִסְכֵּיהוֹן לְתוֹרִין לְדִכְרִין וּלְאִימְרִין בְּמִנְיָנֵיהוֹן כִּדְחָזֵי:

{לה}  מפטיר  בַּיּוֹם֙ הַשְּׁמִינִ֔י עֲצֶ֖רֶת תִּֽהְיֶ֣ה לָכֶ֑ם כָּל-מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ:  מ  בַּיּוֹם֙ הַשְּׁמִינִ֔י עֲצֶ֖רֶת תִּֽהְיֶ֣ה לָכֶ֑ם כָּל-מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ:  ת  בְּיוֹמָא תְּמִינָאָה כְּנִישִׁין תְּהֵא לְכוֹן כָּל עִבִידַת פּוּלְחַן לָא תַעְבְדוּן:

{לו} וְהִקְרַבְתֶּ֨ם עֹלָ֜ה אִשֵּׁ֨ה רֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה פַּ֥ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם:  מ  וְהִקְרַבְתֶּ֨ם עֹלָ֜ה אִשֵּׁ֨ה רֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהֹוָ֔ה פַּ֥ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם:  ת  וּתְקַרְבוּן עֲלָתָא קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ תּוֹר חָד דְכַר חָד אִמְרִין בְּנֵי שְׁנָא שַׁבְעָא שַׁלְמִין:

{לז} מִנְחָתָ֣ם וְנִסְכֵּיהֶ֗ם לַפָּ֨ר לָאַ֧יִל וְלַכְּבָשִׂ֛ים בְּמִסְפָּרָ֖ם כַּמִּשְׁפָּֽט:  מ  מִנְחָתָ֣ם וְנִסְכֵּיהֶ֗ם לַפָּ֨ר לָאַ֧יִל וְלַכְּבָשִׂ֛ים בְּמִסְפָּרָ֖ם כַּמִּשְׁפָּֽט:  ת  מִנְחַתְהוֹן וְנִסְכֵּיהוֹן לְתוֹרָא לְדִכְרָא וּלְאִימְרִין בְּמִנְיָנֵיהוֹן כִּדְחָזֵי:

{לח} וּשְׂעִ֥יר חַטָּ֖את אֶחָ֑ד מִלְּבַד֙ עֹלַ֣ת הַתָּמִ֔יד וּמִנְחָתָ֖הּ וְנִסְכָּֽהּ:  מ  וּשְׂעִ֥יר חַטָּ֖את אֶחָ֑ד מִלְּבַד֙ עֹלַ֣ת הַתָּמִ֔יד וּמִנְחָתָ֖הּ וְנִסְכָּֽהּ:  ת  וּצְפִירָא דְחַטָאתָא חָד בַּר מֵעֲלַת תְּדִירָא וּמִנְחָתַהּ וְנִסְכַּהּ:

{לט} אֵ֛לֶּה תַּֽעֲשׂ֥וּ לַֽיהֹוָ֖ה בְּמֽוֹעֲדֵיכֶ֑ם לְבַ֨ד מִנִּדְרֵיכֶ֜ם וְנִדְבֹֽתֵיכֶ֗ם לְעֹלֹֽתֵיכֶם֙ וּלְמִנְחֹ֣תֵיכֶ֔ם וּלְנִסְכֵּיכֶ֖ם וּלְשַׁלְמֵיכֶֽם:  מ  אֵ֛לֶּה תַּֽעֲשׂ֥וּ לַֽיהֹוָ֖ה בְּמֽוֹעֲדֵיכֶ֑ם לְבַ֨ד מִנִּדְרֵיכֶ֜ם וְנִדְבֹֽתֵיכֶ֗ם לְעֹלֹֽתֵיכֶם֙ וּלְמִנְחֹ֣תֵיכֶ֔ם וּלְנִסְכֵּיכֶ֖ם וּלְשַׁלְמֵיכֶֽם:  ת  אִלֵין תַּעְבְדוּן קֳדָם יְיָ בְּמוֹעֲדֵיכוֹן בַּר מִנִדְרֵיכוֹן וְנִדְבוֹתֵיכוֹן לְעֲלְוַתְכוֹן וּלְמִנְחַתְכוֹן וּלְנִסְכֵּיכוֹן וּלְנִכְסַת קוּדְשֵׁיכוֹן:
