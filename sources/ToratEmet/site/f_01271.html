פרק א - משנה א

{א} שֶׁל תָּלוּשׁ: {ב} מִשֶּׁנִּתְלְשׁוּ. דְּאִם לֹא כֵן, אֵין לְךָ דָּבָר שֶׁלֹּא בָא עָלָיו מַיִם בִּמְחֻבָּר. רַשִׁ"י בַחֻמָּשׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:


פרק א - משנה ב

{ג} עַיֵּן מַה שֶּׁכָּתַבְנוּ בְּפֶרֶק ד' מִשְׁנָה ג' וּבְהָרַ"ב שָׁם מִשְׁנָה ז': {ד} אֵינָן כוּ'. וְאַף עַל פִּי שֶׁנָּשְׁרוּ מִמָּקוֹם לְמָקוֹם. הָרַמְבַּ"ם:


פרק א - משנה ג

{ה} סוֹכָה. עָנָף שֶׁל אִילָן. הָרַ"שׁ. וְהָרַמְבַּ"ם פֵּרֵשׁ סֻכָּה מַמָּשׁ: {ו} טָמֵא. רוֹצֶה לוֹמַר מֻכְשָׁר. וְהַרְבֵּה הֵם בְּזֹאת הַמַּסֶּכֶת. וְכֵן טָהוֹר עַל הַדָּבָר שֶׁאֵינוֹ מֻכְשָׁר. הָרַמְבַּ"ם:


פרק א - משנה ד

{ז} וְיֵרְדוּ מִצַּד הָעֶלְיוֹן לַצַּד הַתַּחְתּוֹן מִקֶּלַח לְקֶלַח. בֵּית שַׁמַּאי אוֹמְרִים כֵּיוָן דְּבִשְׁעַת נִעוּר נִיחָא לֵיהּ שֶׁיֵּרְדוּ מַיִם מִצַּד הָעֶלְיוֹן לַתַּחְתּוֹן וּמִתּוֹךְ כָּךְ יִפְּלוּ הַמַּיִם מִן הָאֲגֻדָּה, חָשׁוּב הֶכְשֵׁר. וּבֵית הִלֵּל אוֹמְרִים כֵּיוָן דְּהָיָה חָפֵץ שֶׁלֹּא יִהְיוּ מַיִם עַל זֹאת הָאֲגֻדָּה בַּצַּד הָעֶלְיוֹן, נְהִי דְבַצַּד הַתַּחְתּוֹן נִיחָא לֵיהּ שֶׁיֵּרְדוּ דֶרֶךְ שָׁם, מִכָּל מָקוֹם כָּל הָאֲגֻדָּה אַחַת הִיא וְתַחְתּוֹנָה שָׁוָה לָעֶלְיוֹנָה. אָמְרוּ לָהֶם בֵּית הִלֵּל לְבֵית שַׁמַּאי וַהֲרֵי כוּ', וְכִי חוֹשְׁשִׁין שֶׁמָּא יֵצְאוּ מֵעָלֶה לְעָלֶה, בִּתְמִיהָה, הָכָא נַמִּי לֹא נֵחוּשׁ. אָמְרוּ לָהֶם, שֶׁהַקֶּלַח אֶחָד, וְהַסְּבָרָא נוֹתֶנֶת לַעֲשׂוֹת כָּל עָלִין שֶׁבּוֹ דִּין אֶחָד לָהֶם, אֲבָל אֵין סְבָרָא לַעֲשׂוֹת כָּל קְלָחִין שֶׁבָּאֲגֻדָּה אֶחָד. מַהֲרַ"ם: {ח} אֲבָל. כְּלוֹמַר בְּהָא מוֹדִינָא. כְּמוֹ אֲבָל שָׂרָה אִשְׁתֶּךָ. מַהֲרַ"ם: {ט} טָהוֹר. שֶׁלֹּא נִתְכַּוֵּן שֶׁיִּבָּלֵל הַשַּׂק הַתַּחְתּוֹן, אֲבָל נִתְכַּוֵּן לְהָגֵן בּוֹ הַשַּׂק הָעֶלְיוֹן שֶׁלֹּא יִבָּלֵל, וְנִמְצָא בְלִילַת הַתַּחְתּוֹן שֶׁלֹּא לְרָצוֹן. וַהֲלָכָה כְּרַבִּי יוֹסֵי. הָרַמְבַּ"ם:


פרק א - משנה ה

{י} הַכְּרֵישָׁה. וְהוּא מְחֻבָּר לַקַּרְקַע. אוֹ נָפַל עָלָיו שֶׁלֹּא לְרָצוֹן. הָרַמְבַּ"ם: {יא} וְהָרַמְבַּ"ם גָּרַס כְּגִרְסַת הַסֵּפֶר. וּמְפָרֵשׁ שֶׁסּוֹחֵט שְׂעָרוֹ בִּבְגָדָיו לְנַגְּבוֹ: {יב} בְּכִי יֻתַּן. אַף עַל גַּב דְּשֶׁלֹּא יִלְקֶה הַכֹּתֶל אֵינָן בְּכִי יֻתַּן [לְהַלָּן פּ"ד מ"ג], הָתָם מִשּׁוּם דְּלֹא נִיחָא לֵיהּ דַּאֲתוּ, וְהָכָא נַמִּי אַף עַל גַּב דְּלֹא נִיחָא לֵיהּ מֵעִקָּרָא, מִכָּל מָקוֹם כֵּיוָן שֶׁהוּא עַצְמוֹ דּוֹחֲקוֹ לְהוֹצִיא מֵהֶן הַמַּיִם, אַחְשְׁבִינְהוּ. תּוֹסָפוֹת:


פרק א - משנה ו

{יג} לְבָדְקָן כוּ'. שֶׁרוֹאִין אִם נִטְפַּח וְנִתְלַחְלַח כְּשֶׁנּוֹפְחִים בּוֹ, נוֹדָע שֶׁיִּתְבַּשֵּׁל מַהֵר, כִּי הֱיוֹתוֹ מְקַבֵּל הַלֵּחוֹת בִּמְהֵרָה אוֹת אֶל פְּתִיחַת נַקְבִּיבוּתוֹ. הָרַמְבַּ"ם. וְלִדְבָרָיו לָאו בָּרֹק הַיּוֹצֵא מִפִּיו פְּלִיגֵי, אֶלָּא בַּלֵּחוֹת וְזִיעַ שֶׁמַּזִּיעִים הָעֲדָשִׁים בְּסִבַּת הַנְּפִיחָה פְּלִיגֵי. וְדִבְרֵי הָרַ"ב מִפֵּרוּשׁ הָרַ"שׁ הוּא:

פרק ב - משנה א

{א} בָּתִּים כוּ' טְהוֹרָה. וַאֲפִלּוּ הֵן טְמֵאִים. הָרַמְבַּ"ם: {ב} וְנִרְאֶה לִי דְּמַיְרֵי שֶׁלֹּא בָא בְתוֹכָם לִטְבֹּל. דְּלֹא תִקְשֶׁה מִמַּתְנִיתִין דְּרֵישׁ פֶּרֶק ה':


פרק ב - משנה ב

{ג} הָרַ"שׁ. וְקָשֶׁה לִי, דְּאַף עַל גַּב דִּשְׁאוּבִים הֵם, לֹא מִפְּנֵי כֵן הֵן מְטַמְּאִים אַף עַל פִּי שֶׁמַּכְשִׁירִים. וְצָרִיךְ לוֹמַר דִּשְׁאוּבִים וּטְמֵאִים הֵם. וְהָרַמְבַּ"ם מְפָרֵשׁ, דְּמֶרְחָץ שֶׁזֵּעָתוֹ הוּא מִן הַמַּיִם שֶׁבְּתוֹכָהּ וְאִם הָיְתָה הַמֶּרְחָץ טְמֵאָה זֵעָתָהּ טְמֵאָה כוּ':


פרק ב - משנה ג

{ד} טָמֵא. שֶׁמַּשְׁקִין טְמֵאִין שֶׁיָּרְדוּ לִטְהוֹרִים, מְטַמְּאִים בְּכָל שֶׁהֵן. הָרַמְבַּ"ם. וְקָשֶׁה, דְּאַדְּרַבָּה הֲוָה לֵיהּ לְמֵימַר קַמָּא קַמָּא בָטֵיל. [וְעַיֵּן פֶּרֶק ה' דַּעֲבוֹדָה זָרָה מִשְׁנָה ח' בְּהָרַ"ב] כוּ'. כֶּסֶף מִשְׁנֶה. וּלְשׁוֹן הָרַאֲבַ"ד, טָמֵא, וְאוּלַי מִפְּנֵי שֶׁמִּשָּׁעָה שֶׁנָּפְלוּ בַתְּחִלָּה, הֻחְזְקוּ גַם הֵם בִּשְׁפִיכוּת שֶׁכְּבָר נִשְׁכְּחָה נְפִילָתָם. וְכָתַב הַכֶּסֶף מִשְׁנֶה, וְאִם תֹּאמַר, לְפִי זֶה אֲפִלּוּ לֹא נָפְלוּ מֵי שְׁפִיכוּת נַמִּי. וְיֵשׁ לוֹמַר, דְּלֹא גַזּוּר אֶלָּא כְשֶׁיֵּשׁ שָׁם מֵי שְׁפִיכוּת:


פרק ב - משנה ו

{ה} קָרוֹב. כְּדֵי שֶׁיֵּלֵךְ הַנָּכְרִי וְיִתְלֹשׁ וְיִגְמֹר הַמְּלָאכָה וְיַחֲזֹר לְכָאן. וְשִׁעוּר הַתְּלִישָׁה אֵין צָרִיךְ לְהַזְכִּיר, שֶׁהִיא נַעֲשֵׂית בְּשָׁעָה מֻעֶטֶת. הָרַ"ן: {ו} וּבְפֶרֶק כ"ג דְּשַׁבָּת מִשְׁנָה ד', אֶלָּא אִם כֵּן בָּאוּ מִמָּקוֹם קָרוֹב, פֵּרֵשׁ הָרַ"ב אֶלָּא אִם כֵּן נוֹדַע בְּבֵרוּר שֶׁבָּאוּ מִתּוֹךְ הַתְּחוּם, כָּתְבוּ הַתּוֹסָפוֹת, דְּמֵת שָׁאנֵי דְּאָוְשִׁי מִלְּתָא. וּלְמַה שֶּׁכָּתַב הָרַ"ן כוּ', יֵשׁ לוֹמַר דְּמָקוֹם קָרוֹב דְּהָכָא אֵינוֹ כְמָקוֹם קָרוֹב דְּהָתָם, אֶלָּא לַאֲפוֹקֵי מָקוֹם רָחוֹק, וּלְעוֹלָם בְּחוּץ לַתְּחוּם, וְצָרִיךְ לְהַמְתִּין עַד שֶׁיָּבֹאוּ מִמֶּנּוּ, אֲבָל אֵין חוֹשְׁשִׁין לַמָּקוֹם הַיּוֹתֵר רָחוֹק מִזֶּה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} פֵּרוּשׁ, לֵךְ וּלְקֹט:


פרק ב - משנה ז

{ח} וְכָל שֶׁכֵּן לְעִנְיַן אֲבֵדָה וְלִנְזָקִין. רַשִׁ"י. וּבַגְּמָרָא, דִּלְעִנְיַן פִּקּוּחַ נֶפֶשׁ אֵינוֹ כֵן: {ט} וְאַף עַל גַּב דְּקַיְמָא לָן אֵין הוֹלְכִין בְּמָמוֹן אַחַר הָרֹב, הָנֵי מִלֵּי כְּשֶׁבָּא לְיָדוֹ הַמָּמוֹן בְּהֶתֵּר וּמִדַּעַת נְתָנָם לוֹ. הָרַ"שׁ וְתוֹסָפוֹת. וּלְיוֹחֲסִין לֹא מָצֵי לְאוֹקְמֵי, דְּבָעִינַן תְּרֵי רֻבָּא: {י} דִּלְהַחֲזִיר אֲבֵדָה אוֹמֵר לוֹ אַיְתֵּי רְאָיָה דְיִשְׂרָאֵל אַתְּ. תּוֹסָפוֹת: {יא} תֵּימַהּ, דַּהֲוָה מָצֵי לְמִתְנֵי מֶחֱצָה עַל מֶחֱצָה נָכְרִים, וּמְנַגַּח תּוֹרָא דִידָן לְתוֹרָא דִידֵיהּ. תּוֹסָפוֹת:


פרק ב - משנה ח

{יב} עִסָּה. נִרְאֶה לִי, דְּפַת עִסָּה הָיָה מִנְהָגָם דְּבַעֲלֵי בָתִּים אוֹפִין בְּבָתֵּיהֶם יוֹתֵר מִן הַנַּחְתּוֹמִין, וּלְפִיכָךְ אֵין הוֹלְכִין בּוֹ אַחַר רֹב נַחְתּוֹמִין. וְכֵן בְּפַת קִבָּר לְרַבִּי יְהוּדָה. וְנִרְאֶה לִי, דְּתַנָּא קַמָּא וְרַבִּי יְהוּדָה לֹא פְלִיגֵי, דְּמַר כִּי אַתְרֵיהּ כוּ', דִּבְאַתְרֵיהּ דְּתַנָּא קַמָּא הַנַּחְתּוֹמִים אוֹפִים פַּת קִבָּר, וּבְאַתְרֵיהּ דְּרַבִּי יְהוּדָה בַּעֲלֵי בָתִּים אוֹפִין פַּת קִבָּר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:


פרק ב - משנה ט

{יג} וְזֶה דְּלֹא כִדְמַשְׁמַע מִלְּשׁוֹנוֹ בְּפֶרֶק ז' דִּשְׁקָלִים מִשְׁנָה ג':
