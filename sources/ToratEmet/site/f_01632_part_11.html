פרשת וזאת הברכה



דברים פרק-לג

{א} וְזֹאת הַבְּרָכָה אֲשֶׁר בֵּרַךְ מֹשֶׁה אִישׁ הָֽאֱלֹהִים אֶת בְּנֵי יִשְׂרָאֵל לִפְנֵי מוֹתֽוֹ:  מ  וְזֹאת הַבְּרָכָה אֲשֶׁר בֵּרַךְ מֹשֶׁה אִישׁ הָֽאֱלֹהִים אֶת בְּנֵי יִשְׂרָאֵל לִפְנֵי מוֹתֽוֹ:  ת  וְדָא בִרְכְּתָא דִי בְרִיךְ מֹשֶׁה נְבִיָא דַיְיָ יָת בְּנֵי יִשְׂרָאֵל קֳדָם מוֹתֵהּ:

{ב} וַיֹּאמַר יְהֹוָה מִסִּינַי בָּא וְזָרַח מִשֵּׂעִיר לָמוֹ הוֹפִיעַ מֵהַר פָּארָן וְאָתָה מֵרִֽבְבֹת קֹדֶשׁ מִֽימִינוֹ (אשדת) אֵשׁ דָּת לָֽמוֹ:  מ  וַיֹּאמַר יְהֹוָה מִסִּינַי בָּא וְזָרַח מִשֵּׂעִיר לָמוֹ הוֹפִיעַ מֵהַר פָּארָן וְאָתָה מֵרִֽבְבֹת קֹדֶשׁ מִֽימִינוֹ (אשדת) אֵשׁ דָּת לָֽמוֹ:  ת  וַאֲמַר יְיָ מִסִינַי אִתְגְלִי וְזֵהוֹר יְקָרֵהּ מִשֵׂעִיר אִתְחֲזֵי לָנָא אִתְגְלִי בִגְבֻרְתֵּהּ מִטוּרָא דְפָארָן וְעִמֵהּ רִבְבַת קַדִישִׁין כְּתַב יַמִינֵהּ מִגוֹ אֶשָׁתָא אוֹרַיְתָא יְהַב לָנָא:

{ג} אַף חֹבֵב עַמִּים כָּל קְדֹשָׁיו בְּיָדֶךָ וְהֵם תֻּכּוּ לְרַגְלֶךָ יִשָּׂא מִדַּבְּרֹתֶֽיךָ:  מ  אַף חֹבֵב עַמִּים כָּל קְדֹשָׁיו בְּיָדֶךָ וְהֵם תֻּכּוּ לְרַגְלֶךָ יִשָּׂא מִדַּבְּרֹתֶֽיךָ:  ת  אַף חַבִּיבִנוּן לְשִׁבְטַיָא כָּל קַדִישׁוֹהִי בֵית יִשְׁרָאֵל בִּגְבוּרָא אַפֵּקִנוּן מִמִצְרָיִם וְאִנוּן מִדַבְּרִין תִּחוֹת עֲנָנָךְ נָטִלִין עַל מֵימְרָךְ:

{ד} תּוֹרָה צִוָּה לָנוּ מֹשֶׁה מֽוֹרָשָׁה קְהִלַּת יַֽעֲקֹֽב:  מ  תּוֹרָה צִוָּה לָנוּ מֹשֶׁה מֽוֹרָשָׁה קְהִלַּת יַֽעֲקֹֽב:  ת  אוֹרַיְתָא יְהַב לָנָא מֹשֶׁה מְסָרַהּ יְרֻתָּא לְכִנְשַׁת יַעֲקֹב:

{ה} וַיְהִי בִֽישֻׁרוּן מֶלֶךְ בְּהִתְאַסֵּף רָאשֵׁי עָם יַחַד שִׁבְטֵי יִשְׂרָאֵֽל:  מ  וַיְהִי בִֽישֻׁרוּן מֶלֶךְ בְּהִתְאַסֵּף רָאשֵׁי עָם יַחַד שִׁבְטֵי יִשְׂרָאֵֽל:  ת  וַהֲוָה בְיִשְׂרָאֵל מַלְכָּא בְּאִתְכַּנָשׁוּת רֵישֵׁי עַמָא כַּחֲדָא שִׁבְטַיָא דְיִשְׂרָאֵל:

{ו} יְחִי רְאוּבֵן וְאַל יָמֹת וִיהִי מְתָיו מִסְפָּֽר:  מ  יְחִי רְאוּבֵן וְאַל יָמֹת וִיהִי מְתָיו מִסְפָּֽר:  ת  יְחִי רְאוּבֵן לְחַיֵי עַלְמָא וּמוֹתָא תִנְיָנָא לָא יְמוּת וִיקַבְּלוּן בְּנוֹהִי אַחֲסַנְתְּהוֹן בְּמִנְיָנְהוֹן: (ס)

{ז} וְזֹאת לִֽיהוּדָה וַיֹּאמַר שְׁמַע יְהֹוָה קוֹל יְהוּדָה וְאֶל עַמּוֹ תְּבִיאֶנּוּ יָדָיו רָב לוֹ וְעֵזֶר מִצָּרָיו תִּֽהְיֶֽה:  מ  וְזֹאת לִֽיהוּדָה וַיֹּאמַר שְׁמַע יְהֹוָה קוֹל יְהוּדָה וְאֶל עַמּוֹ תְּבִיאֶנּוּ יָדָיו רָב לוֹ וְעֵזֶר מִצָּרָיו תִּֽהְיֶֽה:  ת  וְדָא לִיהוּדָה וַאֲמַר קַבֵּל יְיָ צְלוֹתֵהּ דִיהוּדָא בְּמִפְּקֵהּ לְאַגָחָא קְרָבָא וּלְעַמֵהּ תָּתֵיבִנֵהּ לִשְׁלָם יְדוֹהִי יַעְבְּדָן לֵהּ פֻּרְעֲנוּתָא מִסָנְאוֹהִי וְסָעִיד מִבַּעֲלֵי דְבָבוֹהִי הֲוֵי לֵהּ: (פ)

{ח}  שני  וּלְלֵוִי אָמַר תֻּמֶּיךָ וְאוּרֶיךָ לְאִישׁ חֲסִידֶךָ אֲשֶׁר נִסִּיתוֹ בְּמַסָּה תְּרִיבֵהוּ עַל מֵי מְרִיבָֽה:  מ  וּלְלֵוִי אָמַר תֻּמֶּיךָ וְאוּרֶיךָ לְאִישׁ חֲסִידֶךָ אֲשֶׁר נִסִּיתוֹ בְּמַסָּה תְּרִיבֵהוּ עַל מֵי מְרִיבָֽה:  ת  וּלְלֵוִי אֲמַר תּוּמַיָא וְאוּרַיָא אַלְבֶּשְׁתָּא לִגְבַר דְאִשְׁתְּכַח חֲסִידָא קֳדָמָךְ דִי נַסִיתוֹהִי בְּנִסֵתָא וַהֲוָה שְׁלִים בְּחַנְתּוֹהִי עַל מֵי מַצוּתָא וְאִשְׁתְּכַח מְהֵימָן:

{ט} הָֽאֹמֵר לְאָבִיו וּלְאִמּוֹ לֹא רְאִיתִיו וְאֶת אֶחָיו לֹא הִכִּיר וְאֶת (בנו) בָּנָיו לֹא יָדָע כִּי שָֽׁמְרוּ אִמְרָתֶךָ וּבְרִֽיתְךָ יִנְצֹֽרוּ:  מ  הָֽאֹמֵר לְאָבִיו וּלְאִמּוֹ לֹא רְאִיתִיו וְאֶת אֶחָיו לֹא הִכִּיר וְאֶת (בנו) בָּנָיו לֹא יָדָע כִּי שָֽׁמְרוּ אִמְרָתֶךָ וּבְרִֽיתְךָ יִנְצֹֽרוּ:  ת  דְעַל אֲבוֹהִי וְעַל אִמֵהּ לָא רַחֵם כַּד חָבוּ מִן דִינָא וְאַפֵּי אֲחוֹהִי וּבְנוֹהִי לָא נְסִיב אֲרֵי נְטָרוּ מַטְרַת מֵימְרָךְ וּקְיָמָךְ לָא אַשְׁנִיוּ:

{י} יוֹרוּ מִשְׁפָּטֶיךָ לְיַֽעֲקֹב וְתוֹרָֽתְךָ לְיִשְׂרָאֵל יָשִׂימוּ קְטוֹרָה בְּאַפֶּךָ וְכָלִיל עַל מִזְבְּחֶֽךָ:  מ  יוֹרוּ מִשְׁפָּטֶיךָ לְיַֽעֲקֹב וְתוֹרָֽתְךָ לְיִשְׂרָאֵל יָשִׂימוּ קְטוֹרָה בְּאַפֶּךָ וְכָלִיל עַל מִזְבְּחֶֽךָ:  ת  כָּשְׁרִין אִלֵין דְיַלְפוּן דִינָיךְ לְיַעֲקֹב וְאוֹרַיְתָךְ לְיִשְׂרָאֵל יְשַׁוּוּן קְטוֹרֶת בּוּסְמִין קֳדָמָךְ וּגְמִיר לְרַעֲוָא עַל מַדְבְּחָךְ:

{יא} בָּרֵךְ יְהֹוָה חֵילוֹ וּפֹעַל יָדָיו תִּרְצֶה מְחַץ מָתְנַיִם קָמָיו וּמְשַׂנְאָיו מִן יְקוּמֽוּן:  מ  בָּרֵךְ יְהֹוָה חֵילוֹ וּפֹעַל יָדָיו תִּרְצֶה מְחַץ מָתְנַיִם קָמָיו וּמְשַׂנְאָיו מִן יְקוּמֽוּן:  ת  בָּרֵךְ יְיָ נִכְסוֹהִי וְקוּרְבַּן יְדוֹהִי תְּקַבֵּל בְּרַעֲוָא תְּבַר חַרְצָא דְסָנְאוֹהִי וּדְבֵעֲלֵי דְבָבוֹהִי דְלָא יְקוּמוּן: (ס)

{יב} לְבִנְיָמִן אָמַר יְדִיד יְהֹוָה יִשְׁכֹּן לָבֶטַח עָלָיו חֹפֵף עָלָיו כָּל הַיּוֹם וּבֵין כְּתֵפָיו שָׁכֵֽן:  מ  לְבִנְיָמִן אָמַר יְדִיד יְהֹוָה יִשְׁכֹּן לָבֶטַח עָלָיו חֹפֵף עָלָיו כָּל הַיּוֹם וּבֵין כְּתֵפָיו שָׁכֵֽן:  ת  לְבִנְיָמִן אֲמַר רְחִימָא דַיְיָ יִשְׁרֵי לְרָחְצַן עֲלוֹהִי יְהֵי מָגֵן עֲלוֹהִי כָּל יוֹמַיָא וּבְאַרְעֵהּ תִּשְׁרֵי שְׁכִנְתָּא: (ס)

{יג}  שלישי  וּלְיוֹסֵף אָמַר מְבֹרֶכֶת יְהֹוָה אַרְצוֹ מִמֶּגֶד שָׁמַיִם מִטָּל וּמִתְּהוֹם רֹבֶצֶת תָּֽחַת:  מ  וּלְיוֹסֵף אָמַר מְבֹרֶכֶת יְהֹוָה אַרְצוֹ מִמֶּגֶד שָׁמַיִם מִטָּל וּמִתְּהוֹם רֹבֶצֶת תָּֽחַת:  ת  וּלְיוֹסֵף אֲמַר מְבָרָכָא מִן קֳדָם יְיָ אַרְעֵהּ עָבְדָא מִגְדָנִין מִטַלָא דִשְׁמַיָא מִלְעֵלָא וּמִמַבּוּעֵי עֵינָוָן וּתְהוֹמִין דְנָגְדָן מִמַעֲמַקֵי אַרְעָא מִלְרָע:

{יד} וּמִמֶּגֶד תְּבוּאֹת שָׁמֶשׁ וּמִמֶּגֶד גֶּרֶשׁ יְרָחִֽים:  מ  וּמִמֶּגֶד תְּבוּאֹת שָׁמֶשׁ וּמִמֶּגֶד גֶּרֶשׁ יְרָחִֽים:  ת  וְעָבְדָא מִגְדָנִין וַעֲלָלָן מִיבוּל שִׁמְשָׁא וְעָבְדָא מִגְדָנִין מֵרֵישׁ יְרַח בִּירַח:

{טו} וּמֵרֹאשׁ הַֽרֲרֵי קֶדֶם וּמִמֶּגֶד גִּבְעוֹת עוֹלָֽם:  מ  וּמֵרֹאשׁ הַֽרֲרֵי קֶדֶם וּמִמֶּגֶד גִּבְעוֹת עוֹלָֽם:  ת  וּמֵרֵיש טוּרַיָא בַכִּירַיָא וּמִטוּב רָמָן דְלָא פָסְקָן:

{טז} וּמִמֶּגֶד אֶרֶץ וּמְלֹאָהּ וּרְצוֹן שֹֽׁכְנִי סְנֶה תָּבוֹאתָה לְרֹאשׁ יוֹסֵף וּלְקָדְקֹד נְזִיר אֶחָֽיו:  מ  וּמִמֶּגֶד אֶרֶץ וּמְלֹאָהּ וּרְצוֹן שֹֽׁכְנִי סְנֶה תָּבוֹאתָה לְרֹאשׁ יוֹסֵף וּלְקָדְקֹד נְזִיר אֶחָֽיו:  ת  וּמִטוּב אַרְעָא וּמָלָאַהּ וְרָעֵי לֵהּ דִשְׁכִנְתֵּהּ בִּשְׁמַיָא וְעַל מֹשֶׁה אִתְגְלִי בַאֲסָנָא יֵיתוּן כָּל אִלֵין לְרֵישָׁא דְיוֹסֵף וּלְגַבְרָא פָּרִישָׁא דַאֲחוֹהִי:

{יז} בְּכוֹר שׁוֹרוֹ הָדָר לוֹ וְקַרְנֵי רְאֵם קַרְנָיו בָּהֶם עַמִּים יְנַגַּח יַחְדָּו אַפְסֵי אָרֶץ וְהֵם רִבְבוֹת אֶפְרַיִם וְהֵם אַלְפֵי מְנַשֶּֽׁה:  מ  בְּכוֹר שׁוֹרוֹ הָדָר לוֹ וְקַרְנֵי רְאֵם קַרְנָיו בָּהֶם עַמִּים יְנַגַּח יַחְדָּו אַפְסֵי אָרֶץ וְהֵם רִבְבוֹת אֶפְרַיִם וְהֵם אַלְפֵי מְנַשֶּֽׁה:  ת  רַבָּא דִבְנוֹהִי זִיוָא לֵהּ וּגְבוּרָן דְאִתְעֲבִידָא לֵהּ מִקֳדָם דְתוּקְפָא וְרוּמָא דִילֵהּ בִּגְבֻרְתֵּהּ עַמְמַיָא יִקְטֵל כַּחֲדָא עַד סְיָפֵי אַרְעָא וְאִנוּן רִבְוָתָא דְבֵית אֶפְרַיִם וְאִנוּן אַלְפַיָא דְבֵית מְנַשֶׁה: (ס)

{יח}  רביעי  וְלִזְבוּלֻן אָמַר שְׂמַח זְבוּלֻן בְּצֵאתֶךָ וְיִשָּׂשׂכָר בְּאֹֽהָלֶֽיךָ:  מ  וְלִזְבוּלֻן אָמַר שְׂמַח זְבוּלֻן בְּצֵאתֶךָ וְיִשָּׂשׂכָר בְּאֹֽהָלֶֽיךָ:  ת  וְלִזְבוּלֻן אֲמַר חֲדֵי זְבוּלֻן בְּמִפְּקָךְ לַאֲגָחָא קְרָבָא עַל בַּעֲלֵי דְבָבָךְ וְיִשָׂשׂכָר בִּמְהָכָךְ לְמֶעְבַּד זִמְנֵי מוֹעֲדַיָא בִּירוּשְׁלֵם:

{יט} עַמִּים הַר יִקְרָאוּ שָׁם יִזְבְּחוּ זִבְחֵי צֶדֶק כִּי שֶׁפַע יַמִּים יִינָקוּ וּשְׂפֻנֵי טְמוּנֵי חֽוֹל:  מ  עַמִּים הַר יִקְרָאוּ שָׁם יִזְבְּחוּ זִבְחֵי צֶדֶק כִּי שֶׁפַע יַמִּים יִינָקוּ וּשְׂפֻנֵי טְמוּנֵי חֽוֹל:  ת  שִׁבְטַּיָא דְיִשְׂרָאֵל לְטוּר בֵּית מַקְדְשָׁא יִתְכַּנְשׁוּן תַּמָן יִכְסוּן נִכְסַת קוּדְשִׁין לְרַעֲוָא אֲרֵי נִכְסֵי עַמְמַיָא יֵיכְלוּן וְסִימָא דְמִטַמְרָא בְחָלָא מִתְגַלְיָא לְהוֹן: (ס)

{כ} וּלְגָד אָמַר בָּרוּךְ מַרְחִיב גָּד כְּלָבִיא שָׁכֵן וְטָרַף זְרוֹעַ אַף קָדְקֹֽד:  מ  וּלְגָד אָמַר בָּרוּךְ מַרְחִיב גָּד כְּלָבִיא שָׁכֵן וְטָרַף זְרוֹעַ אַף קָדְקֹֽד:  ת  וּלְגָד אֲמַר בְּרִיךְ דְאַפְתֵּי לְגָד כְּלֵיתָא שָׁרֵי וְיִקְטוֹל שִׁלְטוֹנִין עִם מַלְכִין:

{כא} וַיַּרְא רֵאשִׁית לוֹ כִּי שָׁם חֶלְקַת מְחֹקֵק סָפוּן וַיֵּתֵא רָאשֵׁי עָם צִדְקַת יְהֹוָה עָשָׂה וּמִשְׁפָּטָיו עִם יִשְׂרָאֵֽל:  מ  וַיַּרְא רֵאשִׁית לוֹ כִּי שָׁם חֶלְקַת מְחֹקֵק סָפוּן וַיֵּתֵא רָאשֵׁי עָם צִדְקַת יְהֹוָה עָשָׂה וּמִשְׁפָּטָיו עִם יִשְׂרָאֵֽל:  ת  וְאִתְקַבֵּל בְּקַדְמֵתָא דִילֵהּ אֲרֵי תַמָן בְּאַחֲסַנְתֵּהּ מֹשֶׁה סָפְרָא רַבָּא דְיִשְׂרָאֵל קְבִיר וְהוּא נָפֵק וְעַל בְּרֵישׁ עַמָא זַכְוָן קֳדָם יְיָ עֲבַד וְדִינוֹהִי עִם יִשְׂרָאֵל: (ס)

{כב}  חמישי  וּלְדָן אָמַר דָּן גּוּר אַרְיֵה יְזַנֵּק מִן הַבָּשָֽׁן:  מ  וּלְדָן אָמַר דָּן גּוּר אַרְיֵה יְזַנֵּק מִן הַבָּשָֽׁן:  ת  וּלְדָן אֲמַר דָן תַּקִיף כְּגוּר אַרְיְוָן אַרְעֵהּ שָׁתְיָא מִן נַחֲלַיָא דְנָגְדָן מִן מַתְנָן:

{כג} וּלְנַפְתָּלִי אָמַר נַפְתָּלִי שְׂבַע רָצוֹן וּמָלֵא בִּרְכַּת יְהֹוָה יָם וְדָרוֹם יְרָֽשָׁה:  מ  וּלְנַפְתָּלִי אָמַר נַפְתָּלִי שְׂבַע רָצוֹן וּמָלֵא בִּרְכַּת יְהֹוָה יָם וְדָרוֹם יְרָֽשָׁה:  ת  וּלְנַפְתָּלִי אֲמַר נַפְתָּלִי שְׂבַע רַעֲוָא וּמְלֵי בִּרְכָן מִן קֳדָם יְיָ מַעֲרָב יָם גִנוֹסַר וְדָרוֹמָא יֵרָת: (ס)

{כד} וּלְאָשֵׁר אָמַר בָּרוּךְ מִבָּנִים אָשֵׁר יְהִי רְצוּי אֶחָיו וְטֹבֵל בַּשֶּׁמֶן רַגְלֽוֹ:  מ  וּלְאָשֵׁר אָמַר בָּרוּךְ מִבָּנִים אָשֵׁר יְהִי רְצוּי אֶחָיו וְטֹבֵל בַּשֶּׁמֶן רַגְלֽוֹ:  ת  וּלְאָשֵׁר אֲמַר בְּרִיךְ מִבִּרְכַּת בְּנַיָא אָשֵׁר יְהֵי רַעֲוָא לְאַחוֹהִי וְיִתְרַבֵּי בְּתַפְנוּקֵי מַלְכִין:

{כה} בַּרְזֶל וּנְחשֶׁת מִנְעָלֶךָ וּכְיָמֶיךָ דָּבְאֶֽךָ:  מ  בַּרְזֶל וּנְחשֶׁת מִנְעָלֶךָ וּכְיָמֶיךָ דָּבְאֶֽךָ:  ת  תַּקִיף כְּפַרְזְלָא וּנְחָשָׁא מוֹתָבָךְ וּכְיוֹמֵי עוּלֵמָתָךְ תָּקְפָּךְ:

{כו} אֵין כָּאֵל יְשֻׁרוּן רֹכֵב שָׁמַיִם בְּעֶזְרֶךָ וּבְגַֽאֲוָתוֹ שְׁחָקִֽים:  מ  אֵין כָּאֵל יְשֻׁרוּן רֹכֵב שָׁמַיִם בְּעֶזְרֶךָ וּבְגַֽאֲוָתוֹ שְׁחָקִֽים:  ת  לֵית אֱלָהּ כֶּאֱלָהָא דְיִשְׂרָאֵל דִשְׁכִנְתֵּהּ בִּשְׁמַ יָא בְּסַעְדָךְ וְתָקְפֵּהּ בִּשְׁמֵי שְׁמַיָא:

{כז}  שישי  מְעֹנָה אֱלֹהֵי קֶדֶם וּמִתַּחַת זְרֹעֹת עוֹלָם וַיְגָרֶשׁ מִפָּנֶיךָ אוֹיֵב וַיֹּאמֶר הַשְׁמֵֽד:  מ  מְעֹנָה אֱלֹהֵי קֶדֶם וּמִתַּחַת זְרֹעֹת עוֹלָם וַיְגָרֶשׁ מִפָּנֶיךָ אוֹיֵב וַיֹּאמֶר הַשְׁמֵֽד:  ת  מְדוֹר אֱלָהָא דִי מִלְקַדְמִין בְּמֵימְרֵהּ אִתְעֲבֵד עָלְמָא וְתָרִיךְ מִקֳדָמָךְ סָנְאָה וַאֲמַר שֵׁצִי:

{כח} וַיִּשְׁכֹּן יִשְׂרָאֵל בֶּטַח בָּדָד עֵין יַֽעֲקֹב אֶל אֶרֶץ דָּגָן וְתִירוֹשׁ אַף שָׁמָיו יַֽעַרְפוּ טָֽל:  מ  וַיִּשְׁכֹּן יִשְׂרָאֵל בֶּטַח בָּדָד עֵין יַֽעֲקֹב אֶל אֶרֶץ דָּגָן וְתִירוֹשׁ אַף שָׁמָיו יַֽעַרְפוּ טָֽל:  ת  וּשְׁרָא יִשְׂרָאֵל לְרָחְצָן בִּלְחוֹדוֹהִי כְּעֵין בִּרְכְּתָא דְבָרֵכִנוּן יַעֲקֹב אֲבוּהוֹן לְאַרְעָא עָבְדָא עִבוּר וַחֲמָר אַף שְׁמַיָא דְעִלָוֵיהוֹן יְשַׁ מְשֻׁנוּן בְּטַלָא:

{כט} אַשְׁרֶיךָ יִשְׂרָאֵל מִי כָמוֹךָ עַם נוֹשַׁע בַּֽיהֹוָה מָגֵן עֶזְרֶךָ וַֽאֲשֶׁר חֶרֶב גַּֽאֲוָתֶךָ וְיִכָּֽחֲשׁוּ אֹֽיְבֶיךָ לָךְ וְאַתָּה עַל בָּֽמוֹתֵימוֹ תִדְרֹֽךְ:  מ  אַשְׁרֶיךָ יִשְׂרָאֵל מִי כָמוֹךָ עַם נוֹשַׁע בַּֽיהֹוָה מָגֵן עֶזְרֶךָ וַֽאֲשֶׁר חֶרֶב גַּֽאֲוָתֶךָ וְיִכָּֽחֲשׁוּ אֹֽיְבֶיךָ לָךְ וְאַתָּה עַל בָּֽמוֹתֵימוֹ תִדְרֹֽךְ:  ת  טוּבָּךְ יִשְׂרָאֵל לֵית דִכְוָתָךְ עַמָא דְפֻרְקָנֵהּ מִן קֳדָם יְיָ תַּקִיף בְּסַעְדָךְ וּדְמִן קֳדָמוֹהִי נִצְחָן גִבַּרְוָתָךְ וִיכַדְבוּן סָנְאָךְ לָךְ וְאַתְּ עַל פְּרִיקַת צַוָארֵי מַלְכֵיהוֹן תִּדְרֹךְ: (ס)




דברים פרק-לד

{א} וַיַּעַל מֹשֶׁה מֵֽעַרְבֹת מוֹאָב אֶל הַר נְבוֹ רֹאשׁ הַפִּסְגָּה אֲשֶׁר עַל פְּנֵי יְרֵחוֹ וַיַּרְאֵהוּ יְהֹוָה אֶת כָּל הָאָרֶץ אֶת הַגִּלְעָד עַד דָּֽן:  מ  וַיַּעַל מֹשֶׁה מֵֽעַרְבֹת מוֹאָב אֶל הַר נְבוֹ רֹאשׁ הַפִּסְגָּה אֲשֶׁר עַל פְּנֵי יְרֵחוֹ וַיַּרְאֵהוּ יְהֹוָה אֶת כָּל הָאָרֶץ אֶת הַגִּלְעָד עַד דָּֽן:  ת  וּסְלֵיק מֹשֶׁה מִמֵישְׁרָא דְמוֹאָב לְטוּרָא דִנְבוֹ רֵישׁ רָמָתָא דִי עַל אַפֵּי יְרֵחוֹ וְאַחְזְיֵהּ יְיָ יָת כָּל אַרְעָא יָת גִלְעָד עַד דָן:

{ב} וְאֵת כָּל נַפְתָּלִי וְאֶת אֶרֶץ אֶפְרַיִם וּמְנַשֶּׁה וְאֵת כָּל אֶרֶץ יְהוּדָה עַד הַיָּם הָאַֽחֲרֽוֹן:  מ  וְאֵת כָּל נַפְתָּלִי וְאֶת אֶרֶץ אֶפְרַיִם וּמְנַשֶּׁה וְאֵת כָּל אֶרֶץ יְהוּדָה עַד הַיָּם הָאַֽחֲרֽוֹן:  ת  וְיָת כָּל נַפְתָּלִי וְיָת אַרְעָא דְאֶפְרַיִם וּמְנַשֶׁה וְיָת כָּל אַרְעָא דִיהוּדָה עַד יַמָא מַעַרְבָא:

{ג} וְאֶת הַנֶּגֶב וְאֶת הַכִּכָּר בִּקְעַת יְרֵחוֹ עִיר הַתְּמָרִים עַד צֹֽעַר:  מ  וְאֶת הַנֶּגֶב וְאֶת הַכִּכָּר בִּקְעַת יְרֵחוֹ עִיר הַתְּמָרִים עַד צֹֽעַר:  ת  וְיָת דָרוֹמָא וְיָת מֵישְׁרָא בִּקְעֲתָא דִירֵחוֹ קַרְתָּא דְדִקְלַיָא עַד צוֹעַר:

{ד} וַיֹּאמֶר יְהֹוָה אֵלָיו זֹאת הָאָרֶץ אֲשֶׁר נִשְׁבַּעְתִּי לְאַבְרָהָם לְיִצְחָק וּֽלְיַֽעֲקֹב לֵאמֹר לְזַֽרְעֲךָ אֶתְּנֶנָּה הֶרְאִיתִיךָ בְעֵינֶיךָ וְשָׁמָּה לֹא תַֽעֲבֹֽר:  מ  וַיֹּאמֶר יְהֹוָה אֵלָיו זֹאת הָאָרֶץ אֲשֶׁר נִשְׁבַּעְתִּי לְאַבְרָהָם לְיִצְחָק וּֽלְיַֽעֲקֹב לֵאמֹר לְזַֽרְעֲךָ אֶתְּנֶנָּה הֶרְאִיתִיךָ בְעֵינֶיךָ וְשָׁמָּה לֹא תַֽעֲבֹֽר:  ת  וַאֲמַר יְיָ לֵהּ דָא אַרְעָא דִי קַיֵמִית לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לְמֵימָר לִבְנָיךְ אֶתְּנִנַהּ אַחֲזִיתָךְ בְּעֵינָיךְ וּלְתַמָן לָא תְעִבַּר:

{ה} וַיָּמָת שָׁם מֹשֶׁה עֶֽבֶד יְהֹוָה בְּאֶרֶץ מוֹאָב עַל פִּי יְהֹוָֽה:  מ  וַיָּמָת שָׁם מֹשֶׁה עֶֽבֶד יְהֹוָה בְּאֶרֶץ מוֹאָב עַל פִּי יְהֹוָֽה:  ת  וּמִית תַּמָן מֹשֶׁה עַבְדָא דַיְיָ בְּאַרְעָא דְמוֹאָב עַל מֵימְרָא דַיְיָ:

{ו} וַיִּקְבֹּר אֹתוֹ בַגַּי בְּאֶרֶץ מוֹאָב מוּל בֵּית פְּעוֹר וְלֹֽא יָדַע אִישׁ אֶת קְבֻרָתוֹ עַד הַיּוֹם הַזֶּֽה:  מ  וַיִּקְבֹּר אֹתוֹ בַגַּי בְּאֶרֶץ מוֹאָב מוּל בֵּית פְּעוֹר וְלֹֽא יָדַע אִישׁ אֶת קְבֻרָתוֹ עַד הַיּוֹם הַזֶּֽה:  ת  וּקְבַר יָתֵהּ בְּחֵילָתָא בְּאַרְעָא דְמוֹאָב לְקֳבֵל בֵּית פְּעוֹר וְלָא יְדַע אֱנַשׁ יָת קְבֻרְתֵּה עַד יוֹמָא הָדֵין:

{ז} וּמֹשֶׁה בֶּן מֵאָה וְעֶשְׂרִים שָׁנָה בְּמֹתוֹ לֹא כָֽהֲתָה עֵינוֹ וְלֹא נָס לֵחֹֽה:  מ  וּמֹשֶׁה בֶּן מֵאָה וְעֶשְׂרִים שָׁנָה בְּמֹתוֹ לֹא כָֽהֲתָה עֵינוֹ וְלֹא נָס לֵחֹֽה:  ת  וּמֹשֶׁה בַּר מְאָה וְעַשְׂרִין שְׁנִין כַּד מִית לָא כְהַת עֵינוֹהִי וְלָא שְׁנָא זִיו יְקָרָא דְאַפּוֹהִי:

{ח} וַיִּבְכּוּ בְנֵי יִשְׂרָאֵל אֶת מֹשֶׁה בְּעַֽרְבֹת מוֹאָב שְׁלֹשִׁים יוֹם וַֽיִּתְּמוּ יְמֵי בְכִי אֵבֶל מֹשֶֽׁה:  מ  וַיִּבְכּוּ בְנֵי יִשְׂרָאֵל אֶת מֹשֶׁה בְּעַֽרְבֹת מוֹאָב שְׁלֹשִׁים יוֹם וַֽיִּתְּמוּ יְמֵי בְכִי אֵבֶל מֹשֶֽׁה:  ת  וּבְכוֹ בְנֵי יִשְׂרָאֵל יָת מֹשֶׁה בְּמֵישְׁרַיָא דְמוֹאָב תְּלָתִין יוֹמִין וּשְׁלִימוּ יוֹמֵי בָכִיתָא אֶבְלָא דְמֹשֶׁה:

{ט} וִֽיהוֹשֻׁעַ בִּן נוּן מָלֵא רוּחַ חָכְמָה כִּֽי סָמַךְ מֹשֶׁה אֶת יָדָיו עָלָיו וַיִּשְׁמְעוּ אֵלָיו בְּנֵֽי יִשְׂרָאֵל וַֽיַּֽעֲשׂוּ כַּֽאֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶֽׁה:  מ  וִֽיהוֹשֻׁעַ בִּן נוּן מָלֵא רוּחַ חָכְמָה כִּֽי סָמַךְ מֹשֶׁה אֶת יָדָיו עָלָיו וַיִּשְׁמְעוּ אֵלָיו בְּנֵֽי יִשְׂרָאֵל וַֽיַּֽעֲשׂוּ כַּֽאֲשֶׁר צִוָּה יְהֹוָה אֶת מֹשֶֽׁה:  ת  וִיהוֹשֻׁעַ בַּר נוּן מְלֵי רוּחַ חָכְמְתָא אֲרֵי סְמַךְ מֹשֶׁה יָת יְדוֹהִי עֲלוֹהִי וְקַבִּילוּ מִנֵהּ בְּנֵי יִשְׂרָאֵל וַעֲבָדוּ כְּמָא דִי פַּקִיד יְיָ יָת מֹשֶׁה:

{י} וְלֹֽא קָם נָבִיא עוֹד בְּיִשְׂרָאֵל כְּמֹשֶׁה אֲשֶׁר יְדָעוֹ יְהֹוָה פָּנִים אֶל פָּנִֽים:  מ  וְלֹֽא קָם נָבִיא עוֹד בְּיִשְׂרָאֵל כְּמֹשֶׁה אֲשֶׁר יְדָעוֹ יְהֹוָה פָּנִים אֶל פָּנִֽים:  ת  וְלָא קָם נְבִיָא עוֹד בְּיִשְׂרָאֵל כְּמֹשֶׁה דִי אִתְגְלִי לֵהּ יְיָ אַפִּין בְּאַפִּין:

{יא} לְכָל הָֽאֹתֹת וְהַמּֽוֹפְתִים אֲשֶׁר שְׁלָחוֹ יְהֹוָה לַֽעֲשׂוֹת בְּאֶרֶץ מִצְרָיִם לְפַרְעֹה וּלְכָל עֲבָדָיו וּלְכָל אַרְצֽוֹ:  מ  לְכָל הָֽאֹתֹת וְהַמּֽוֹפְתִים אֲשֶׁר שְׁלָחוֹ יְהֹוָה לַֽעֲשׂוֹת בְּאֶרֶץ מִצְרָיִם לְפַרְעֹה וּלְכָל עֲבָדָיו וּלְכָל אַרְצֽוֹ:  ת  לְכָל אָתַיָא וּמוֹפְתַיָא דִי שָׁלְחֵהּ יְיָ לְמֶעְבַּד בְּאַרְעָא דְמִצְרָיִם לְפַרְעֹה וּלְכָל עַבְדוֹהִי וּלְכָל אַרְעֵהּ:

{יב} וּלְכֹל הַיָּד הַֽחֲזָקָה וּלְכֹל הַמּוֹרָא הַגָּדוֹל אֲשֶׁר עָשָׂה מֹשֶׁה לְעֵינֵי כָּל יִשְׂרָאֵֽל:  מ  וּלְכֹל הַיָּד הַֽחֲזָקָה וּלְכֹל הַמּוֹרָא הַגָּדוֹל אֲשֶׁר עָשָׂה מֹשֶׁה לְעֵינֵי כָּל יִשְׂרָאֵֽל:  ת  וּלְכֹל יְדָא תַקֶפְתָּא וּלְכֹל חֵזְוָנָא רַבָּא דִי עֲבַד מֹשֶׁה לְעֵינֵי כָּל יִשְׂרָאֵל: (חזק)
