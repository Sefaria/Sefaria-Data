ע-א


[בראשית פרק-כט-ט]  עוֹדֶנּוּ מְדַבֵּר עִמָּם וְרָחֵל בָּאָה עִם הַצֹּאן אֲשֶׁר לְאָבִיהָ כִּי רֹעָה הִוא:

[ויקרא פרק-יח-ח]  עֶרְוַת אֵשֶׁת אָבִיךָ לֹא תְגַלֵּה עֶרְוַת אָבִיךָ הִוא:

[ויקרא פרק-יח-יב]  עֶרְוַת אֲחוֹת אָבִיךָ לֹא תְגַלֵּה שְׁאֵר אָבִיךָ הִוא:

[ויקרא פרק-יח-יג]  עֶרְוַת אֲחוֹת אִמְּךָ לֹא תְגַלֵּה כִּי שְׁאֵר אִמְּךָ הִוא:

[ויקרא פרק-יח-יד]  עֶרְוַת אֲחִי אָבִיךָ לֹא תְגַלֵּה אֶל אִשְׁתּוֹ לֹא תִקְרָב דֹּדָתְךָ הִוא:

[ויקרא פרק-יח-טז]  עֶרְוַת אֵשֶׁת אָחִיךָ לֹא תְגַלֵּה עֶרְוַת אָחִיךָ הִוא:

[ויקרא פרק-יח-יז]  עֶרְוַת אִשָּׁה וּבִתָּהּ לֹא תְגַלֵּה אֶת בַּת בְּנָהּ וְאֶת בַּת בִּתָּהּ לֹא תִקַּח לְגַלּוֹת עֶרְוָתָהּ שַׁאֲרָה הֵנָּה זִמָּה הִוא:

[יהושע פרק-יד-יא]  עוֹדֶנִּי הַיּוֹם חָזָק כַּאֲשֶׁר בְּיוֹם שְׁלֹחַ אוֹתִי מֹשֶׁה כְּכֹחִי אָז וּכְכֹחִי עָתָּה לַמִּלְחָמָה וְלָצֵאת וְלָבוֹא:

[ישעיה פרק-ב-יא]  עֵינֵי גַּבְהוּת אָדָם שָׁפֵל וְשַׁח רוּם אֲנָשִׁים וְנִשְׂגַּב יְהוָה לְבַדּוֹ בַּיּוֹם הַהוּא:

[ישעיה פרק-לג-י]  עַתָּה אָקוּם יֹאמַר יְהוָה עַתָּה אֵרוֹמָם עַתָּה אֶנָּשֵׂא:

[ישעיה פרק-נב-א]  עוּרִי עוּרִי לִבְשִׁי עֻזֵּךְ צִיּוֹן לִבְשִׁי בִּגְדֵי תִפְאַרְתֵּךְ יְרוּשָׁלִַם עִיר הַקֹּדֶשׁ כִּי לֹא יוֹסִיף יָבֹא בָךְ עוֹד עָרֵל וְטָמֵא:

[ירמיה פרק-מו-כ]  עֶגְלָה יְפֵה פִיָּה מִצְרָיִם קֶרֶץ מִצָּפוֹן בָּא בָא:

[יחזקאל פרק-יח-טו]  עַל הֶהָרִים לֹא אָכָל וְעֵינָיו לֹא נָשָׂא אֶל גִּלּוּלֵי בֵּית יִשְׂרָאֵל אֶת אֵשֶׁת רֵעֵהוּ לֹא טִמֵּא:

[הושע פרק-ו-ה]  עַל כֵּן חָצַבְתִּי בַּנְּבִיאִים הֲרַגְתִּים בְּאִמְרֵי פִי וּמִשְׁפָּטֶיךָ אוֹר יֵצֵא:

[משלי פרק-ו-טו]  עַל כֵּן פִּתְאֹם יָבוֹא אֵידוֹ פֶּתַע יִשָּׁבֵר וְאֵין מַרְפֵּא:

[משלי פרק-ז-כג]  עַד יְפַלַּח חֵץ כְּבֵדוֹ כְּמַהֵר צִפּוֹר אֶל פָּח וְלֹא יָדַע כִּי בְנַפְשׁוֹ הוּא:

[משלי פרק-טז-לא]  עֲטֶרֶת תִּפְאֶרֶת שֵׂיבָה בְּדֶרֶךְ צְדָקָה תִּמָּצֵא:

[דניאל פרק-ב-ח]  עָנֵה מַלְכָּא וְאָמַר מִן יַצִּיב יָדַע אֲנָה דִּי עִדָּנָא אַנְתּוּן זָבְנִין כָּל קֳבֵל דִּי חֲזֵיתוֹן דִּי אַזְדָּא מִנִּי מִלְּתָא:

[דניאל פרק-ב-כ]  עָנֵה דָנִיֵּאל וְאָמַר לֶהֱוֵא שְׁמֵהּ דִּי אֱלָהָא מְבָרַךְ מִן עָלְמָא וְעַד עָלְמָא דִּי חָכְמְתָא וּגְבוּרְתָא דִּי לֵהּ הִיא:

[דניאל פרק-ב-כז]  עָנֵה דָנִיֵּאל קֳדָם מַלְכָּא וְאָמַר רָזָה דִּי מַלְכָּא שָׁאֵל לָא חַכִּימִין אָשְׁפִין חַרְטֻמִּין גָּזְרִין יָכְלִין לְהַחֲוָיָה לְמַלְכָּא:

[דניאל פרק-ד-ט]  עָפְיֵהּ שַׁפִּיר וְאִנְבֵּהּ שַׂגִּיא וּמָזוֹן לְכֹלָּא בֵהּ תְּחֹתוֹהִי תַּטְלֵל חֵיוַת בָּרָא וּבְעַנְפוֹהִי (ידרון) יְדוּרָן צִפֲּרֵי שְׁמַיָּא וּמִנֵּהּ יִתְּזִין כָּל בִּשְׂרָא:

[דניאל פרק-ז-ב]  עָנֵה דָנִיֵּאל וְאָמַר חָזֵה הֲוֵית בְּחֶזְוִי עִם לֵילְיָא וַאֲרוּ אַרְבַּע רוּחֵי שְׁמַיָּא מְגִיחָן לְיַמָּא רַבָּא:


ע-ב


[ישעיה פרק-לב-טו]  עַד יֵעָרֶה עָלֵינוּ רוּחַ מִמָּרוֹם וְהָיָה מִדְבָּר לַכַּרְמֶל (וכרמל) וְהַכַּרְמֶל לַיַּעַר יֵחָשֵׁב:

[ירמיה פרק-ב-טו]  עָלָיו יִשְׁאֲגוּ כְפִרִים נָתְנוּ קוֹלָם וַיָּשִׁיתוּ אַרְצוֹ לְשַׁמָּה עָרָיו (נצתה) נִצְּתוּ מִבְּלִי יֹשֵׁב:

[ירמיה פרק-ד-ז]  עָלָה אַרְיֵה מִסֻּבְּכוֹ וּמַשְׁחִית גּוֹיִם נָסַע יָצָא מִמְּקֹמוֹ לָשׂוּם אַרְצֵךְ לְשַׁמָּה עָרַיִךְ תִּצֶּינָה מֵאֵין יוֹשֵׁב:

[ירמיה פרק-ט-כה]  עַל מִצְרַיִם וְעַל יְהוּדָה וְעַל אֱדוֹם וְעַל בְּנֵי עַמּוֹן וְעַל מוֹאָב וְעַל כָּל קְצוּצֵי פֵאָה הַיֹּשְׁבִים בַּמִּדְבָּר כִּי כָל הַגּוֹיִם עֲרֵלִים וְכָל בֵּית יִשְׂרָאֵל עַרְלֵי לֵב:

[תהילים פרק-קלב-ה]  עַד אֶמְצָא מָקוֹם לַיהוָה מִשְׁכָּנוֹת לַאֲבִיר יַעֲקֹב:

[משלי פרק-יא-כט]  עוֹכֵר בֵּיתוֹ יִנְחַל רוּחַ וְעֶבֶד אֱוִיל לַחֲכַם לֵב:

[משלי פרק-יב-יא]  עֹבֵד אַדְמָתוֹ יִשְׂבַּע לָחֶם וּמְרַדֵּף רֵיקִים חֲסַר לֵב:

[משלי פרק-יט-טו]  עַצְלָה תַּפִּיל תַּרְדֵּמָה וְנֶפֶשׁ רְמִיָּה תִרְעָב:

[משלי פרק-כד-ל]  עַל שְׂדֵה אִישׁ עָצֵל עָבַרְתִּי וְעַל כֶּרֶם אָדָם חֲסַר לֵב:

[איוב פרק-כ-יא]  עַצְמוֹתָיו מָלְאוּ (עלומו) עֲלוּמָיו וְעִמּוֹ עַל עָפָר תִּשְׁכָּב:

[איוב פרק-ל-ל]  עוֹרִי שָׁחַר מֵעָלָי וְעַצְמִי חָרָה מִנִּי חֹרֶב:

[איכה פרק-א-טז]  עַל אֵלֶּה אֲנִי בוֹכִיָּה עֵינִי עֵינִי יֹרְדָה מַּיִם כִּי רָחַק מִמֶּנִּי מְנַחֵם מֵשִׁיב נַפְשִׁי הָיוּ בָנַי שׁוֹמֵמִים כִּי גָבַר אוֹיֵב:

[איכה פרק-ה-י]  עוֹרֵנוּ כְּתַנּוּר נִכְמָרוּ מִפְּנֵי זַלְעֲפוֹת רָעָב:

[עזרא פרק-ז-כב]  עַד כְּסַף כַּכְּרִין מְאָה וְעַד חִנְטִין כֹּרִין מְאָה וְעַד חֲמַר בַּתִּין מְאָה וְעַד בַּתִּין מְשַׁח מְאָה וּמְלַח דִּי לָא כְתָב:


ע-ד


[בראשית פרק-ב-כד]  עַל כֵּן יַעֲזָב אִישׁ אֶת אָבִיו וְאֶת אִמּוֹ וְדָבַק בְּאִשְׁתּוֹ וְהָיוּ לְבָשָׂר אֶחָד:

[בראשית פרק-טז-יד]  עַל כֵּן קָרָא לַבְּאֵר בְּאֵר לַחַי רֹאִי הִנֵּה בֵין קָדֵשׁ וּבֵין בָּרֶד:

[בראשית פרק-כז-מה]  עַד שׁוּב אַף אָחִיךָ מִמְּךָ וְשָׁכַח אֵת אֲשֶׁר עָשִׂיתָ לּוֹ וְשָׁלַחְתִּי וּלְקַחְתִּיךָ מִשָּׁם לָמָה אֶשְׁכַּל גַּם שְׁנֵיכֶם יוֹם אֶחָד:

[שמות פרק-כו-טז]  עֶשֶׂר אַמּוֹת אֹרֶךְ הַקָּרֶשׁ וְאַמָּה וַחֲצִי הָאַמָּה רֹחַב הַקֶּרֶשׁ הָאֶחָד:

[שמות פרק-לו-כא]  עֶשֶׂר אַמֹּת אֹרֶךְ הַקָּרֶשׁ וְאַמָּה וַחֲצִי הָאַמָּה רֹחַב הַקֶּרֶשׁ הָאֶחָד:

[ויקרא פרק-כד-ד]  עַל הַמְּנֹרָה הַטְּהֹרָה יַעֲרֹךְ אֶת הַנֵּרוֹת לִפְנֵי יְהוָה תָּמִיד:

[דברים פרק-ב-יג]  עַתָּה קֻמוּ וְעִבְרוּ לָכֶם אֶת נַחַל זָרֶד וַנַּעֲבֹר אֶת נַחַל זָרֶד:

[דברים פרק-יז-ו]  עַל פִּי שְׁנַיִם עֵדִים אוֹ שְׁלֹשָׁה עֵדִים יוּמַת הַמֵּת לֹא יוּמַת עַל פִּי עֵד אֶחָד:

[מלכים ב פרק-ו-לג]  עוֹדֶנּוּ מְדַבֵּר עִמָּם וְהִנֵּה הַמַּלְאָךְ יֹרֵד אֵלָיו וַיֹּאמֶר הִנֵּה זֹאת הָרָעָה מֵאֵת יְהוָה מָה אוֹחִיל לַיהוָה עוֹד:

[ישעיה פרק-יז-ב]  עֲזֻבוֹת עָרֵי עֲרֹעֵר לַעֲדָרִים תִּהְיֶינָה וְרָבְצוּ וְאֵין מַחֲרִיד:

[ישעיה פרק-כג-י]  עִבְרִי אַרְצֵךְ כַּיְאֹר בַּת תַּרְשִׁישׁ אֵין מֵזַח עוֹד:

[נחום פרק-ב-ב]  עָלָה מֵפִיץ עַל פָּנַיִךְ נָצוֹר מְצֻרָה צַפֵּה דֶרֶךְ חַזֵּק מָתְנַיִם אַמֵּץ כֹּחַ מְאֹד:

[תהילים פרק-עא-ו]  עָלֶיךָ נִסְמַכְתִּי מִבֶּטֶן מִמְּעֵי אִמִּי אַתָּה גוֹזִי בְּךָ תְהִלָּתִי תָמִיד:

[משלי פרק-ז-ח]  עֹבֵר בַּשּׁוּק אֵצֶל פִּנָּהּ וְדֶרֶךְ בֵּיתָהּ יִצְעָד:

[משלי פרק-יט-ט]  עֵד שְׁקָרִים לֹא יִנָּקֶה וְיָפִיחַ כְּזָבִים יֹאבֵד:

[משלי פרק-כב-יב]  עֵינֵי יְהוָה נָצְרוּ דָעַת וַיְסַלֵּף דִּבְרֵי בֹגֵד:

[קהלת פרק-ג-ד]  עֵת לִבְכּוֹת וְעֵת לִשְׂחוֹק עֵת סְפוֹד וְעֵת רְקוֹד:


ע-ה


[בראשית פרק-לא-נב]  עֵד הַגַּל הַזֶּה וְעֵדָה הַמַּצֵּבָה אִם אָנִי לֹא אֶעֱבֹר אֵלֶיךָ אֶת הַגַּל הַזֶּה וְאִם אַתָּה לֹא תַעֲבֹר אֵלַי אֶת הַגַּל הַזֶּה וְאֶת הַמַּצֵּבָה הַזֹּאת לְרָעָה:

[בראשית פרק-לב-לג]  עַל כֵּן לֹא יֹאכְלוּ בְנֵי יִשְׂרָאֵל אֶת גִּיד הַנָּשֶׁה אֲשֶׁר עַל כַּף הַיָּרֵךְ עַד הַיּוֹם הַזֶּה כִּי נָגַע בְּכַף יֶרֶךְ יַעֲקֹב בְּגִיד הַנָּשֶׁה:

[ויקרא פרק-ו-יד]  עַל מַחֲבַת בַּשֶּׁמֶן תֵּעָשֶׂה מֻרְבֶּכֶת תְּבִיאֶנָּה תֻּפִינֵי מִנְחַת פִּתִּים תַּקְרִיב רֵיחַ נִיחֹחַ לַיהוָה:

[ויקרא פרק-יח-ז]  עֶרְוַת אָבִיךָ וְעֶרְוַת אִמְּךָ לֹא תְגַלֵּה אִמְּךָ הִוא לֹא תְגַלֶּה עֶרְוָתָהּ:

[ויקרא פרק-יח-י]  עֶרְוַת בַּת בִּנְךָ אוֹ בַת בִּתְּךָ לֹא תְגַלֶּה עֶרְוָתָן כִּי עֶרְוָתְךָ הֵנָּה:

[ויקרא פרק-יח-יא]  עֶרְוַת בַּת אֵשֶׁת אָבִיךָ מוֹלֶדֶת אָבִיךָ אֲחוֹתְךָ הִוא לֹא תְגַלֶּה עֶרְוָתָהּ:

[ויקרא פרק-יח-טו]  עֶרְוַת כַּלָּתְךָ לֹא תְגַלֵּה אֵשֶׁת בִּנְךָ הִוא לֹא תְגַלֶּה עֶרְוָתָהּ:

[ויקרא פרק-כב-כב]  עַוֶּרֶת אוֹ שָׁבוּר אוֹ חָרוּץ אוֹ יַבֶּלֶת אוֹ גָרָב אוֹ יַלֶּפֶת לֹא תַקְרִיבוּ אֵלֶּה לַיהוָה וְאִשֶּׁה לֹא תִתְּנוּ מֵהֶם עַל הַמִּזְבֵּחַ לַיהוָה:

[ויקרא פרק-כג-טז]  עַד מִמָּחֳרַת הַשַּׁבָּת הַשְּׁבִיעִת תִּסְפְּרוּ חֲמִשִּׁים יוֹם וְהִקְרַבְתֶּם מִנְחָה חֲדָשָׁה לַיהוָה:

[במדבר פרק-ד-מט]  עַל פִּי יְהוָה פָּקַד אוֹתָם בְּיַד מֹשֶׁה אִישׁ אִישׁ עַל עֲבֹדָתוֹ וְעַל מַשָּׂאוֹ וּפְקֻדָיו אֲשֶׁר צִוָּה יְהוָה אֶת מֹשֶׁה:

[במדבר פרק-ט-כג]  עַל פִּי יְהוָה יַחֲנוּ וְעַל פִּי יְהוָה יִסָּעוּ אֶת מִשְׁמֶרֶת יְהוָה שָׁמָרוּ עַל פִּי יְהוָה בְּיַד מֹשֶׁה:

[במדבר פרק-כח-ו]  עֹלַת תָּמִיד הָעֲשֻׂיָה בְּהַר סִינַי לְרֵיחַ נִיחֹחַ אִשֶּׁה לַיהוָה:

[במדבר פרק-כח-י]  עֹלַת שַׁבַּת בְּשַׁבַּתּוֹ עַל עֹלַת הַתָּמִיד וְנִסְכָּהּ:

[דברים פרק-ג-כז]  עֲלֵה רֹאשׁ הַפִּסְגָּה וְשָׂא עֵינֶיךָ יָמָּה וְצָפֹנָה וְתֵימָנָה וּמִזְרָחָה וּרְאֵה בְעֵינֶיךָ כִּי לֹא תַעֲבֹר אֶת הַיַּרְדֵּן הַזֶּה:

[דברים פרק-י-יח]  עֹשֶׂה מִשְׁפַּט יָתוֹם וְאַלְמָנָה וְאֹהֵב גֵּר לָתֶת לוֹ לֶחֶם וְשִׂמְלָה:

[דברים פרק-יד-כב]  עַשֵּׂר תְּעַשֵּׂר אֵת כָּל תְּבוּאַת זַרְעֶךָ הַיֹּצֵא הַשָּׂדֶה שָׁנָה שָׁנָה:

[דברים פרק-לב-מט]  עֲלֵה אֶל הַר הָעֲבָרִים הַזֶּה הַר נְבוֹ אֲשֶׁר בְּאֶרֶץ מוֹאָב אֲשֶׁר עַל פְּנֵי יְרֵחוֹ וּרְאֵה אֶת אֶרֶץ כְּנַעַן אֲשֶׁר אֲנִי נֹתֵן לִבְנֵי יִשְׂרָאֵל לַאֲחֻזָּה:

[יהושע פרק-א-יא]  עִבְרוּ בְּקֶרֶב הַמַּחֲנֶה וְצַוּוּ אֶת הָעָם לֵאמֹר הָכִינוּ לָכֶם צֵידָה כִּי בְּעוֹד שְׁלֹשֶׁת יָמִים אַתֶּם עֹבְרִים אֶת הַיַּרְדֵּן הַזֶּה לָבוֹא לָרֶשֶׁת אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֵיכֶם נֹתֵן לָכֶם לְרִשְׁתָּהּ:

[יהושע פרק-יג-ח]  עִמּוֹ הָראוּבֵנִי וְהַגָּדִי לָקְחוּ נַחֲלָתָם אֲשֶׁר נָתַן לָהֶם מֹשֶׁה בְּעֵבֶר הַיַּרְדֵּן מִזְרָחָה כַּאֲשֶׁר נָתַן לָהֶם מֹשֶׁה עֶבֶד יְהוָה:

[יהושע פרק-טו-מה]  עֶקְרוֹן וּבְנֹתֶיהָ וַחֲצֵרֶיהָ:

[יהושע פרק-יט-נ]  עַל פִּי יְהוָה נָתְנוּ לוֹ אֶת הָעִיר אֲשֶׁר שָׁאָל אֶת תִּמְנַת סֶרַח בְּהַר אֶפְרָיִם וַיִּבְנֶה אֶת הָעִיר וַיֵּשֶׁב בָּהּ:

[שמואל א פרק-ה-ה]  עַל כֵּן לֹא יִדְרְכוּ כֹהֲנֵי דָגוֹן וְכָל הַבָּאִים בֵּית דָּגוֹן עַל מִפְתַּן דָּגוֹן בְּאַשְׁדּוֹד עַד הַיּוֹם הַזֶּה:

[מלכים א פרק-ז-כה]  עֹמֵד עַל שְׁנֵי עָשָׂר בָּקָר שְׁלֹשָׁה פֹנִים צָפוֹנָה וּשְׁלֹשָׁה פֹנִים יָמָּה וּשְׁלֹשָׁה פֹּנִים נֶגְבָּה וּשְׁלֹשָׁה פֹּנִים מִזְרָחָה וְהַיָּם עֲלֵיהֶם מִלְמָעְלָה וְכָל אֲחֹרֵיהֶם בָּיְתָה:

[מלכים ב פרק-יז-ג]  עָלָיו עָלָה שַׁלְמַנְאֶסֶר מֶלֶךְ אַשּׁוּר וַיְהִי לוֹ הוֹשֵׁעַ עֶבֶד וַיָּשֶׁב לוֹ מִנְחָה:

[מלכים ב פרק-יז-כג]  עַד אֲשֶׁר הֵסִיר יְהוָה אֶת יִשְׂרָאֵל מֵעַל פָּנָיו כַּאֲשֶׁר דִּבֶּר בְּיַד כָּל עֲבָדָיו הַנְּבִיאִים וַיִּגֶל יִשְׂרָאֵל מֵעַל אַדְמָתוֹ אַשּׁוּרָה עַד הַיּוֹם הַזֶּה:

[מלכים ב פרק-יח-כה]  עַתָּה הֲמִבַּלְעֲדֵי יְהוָה עָלִיתִי עַל הַמָּקוֹם הַזֶּה לְהַשְׁחִתוֹ יְהוָה אָמַר אֵלַי עֲלֵה עַל הָאָרֶץ הַזֹּאת וְהַשְׁחִיתָהּ:

[ישעיה פרק-ה-כה]  עַל כֵּן חָרָה אַף יְהוָה בְּעַמּוֹ וַיֵּט יָדוֹ עָלָיו וַיַּכֵּהוּ וַיִּרְגְּזוּ הֶהָרִים וַתְּהִי נִבְלָתָם כַּסּוּחָה בְּקֶרֶב חוּצוֹת בְּכָל זֹאת לֹא שָׁב אַפּוֹ וְעוֹד יָדוֹ נְטוּיָה:

[ישעיה פרק-ט-טז]  עַל כֵּן עַל בַּחוּרָיו לֹא יִשְׂמַח אֲדֹנָי וְאֶת יְתֹמָיו וְאֶת אַלְמְנֹתָיו לֹא יְרַחֵם כִּי כֻלּוֹ חָנֵף וּמֵרַע וְכָל פֶּה דֹּבֵר נְבָלָה בְּכָל זֹאת לֹא שָׁב אַפּוֹ וְעוֹד יָדוֹ נְטוּיָה:

[ישעיה פרק-י-כט]  עָבְרוּ מַעְבָּרָה גֶּבַע מָלוֹן לָנוּ חָרְדָה הָרָמָה גִּבְעַת שָׁאוּל נָסָה:

[ישעיה פרק-טו-ב]  עָלָה הַבַּיִת וְדִיבֹן הַבָּמוֹת לְבֶכִי עַל נְבוֹ וְעַל מֵידְבָא מוֹאָב יְיֵלִיל בְּכָל רֹאשָׁיו קָרְחָה כָּל זָקָן גְּרוּעָה:

[ישעיה פרק-לב-יב]  עַל שָׁדַיִם סֹפְדִים עַל שְׂדֵי חֶמֶד עַל גֶּפֶן פֹּרִיָּה:

[ישעיה פרק-לב-יג]  עַל אַדְמַת עַמִּי קוֹץ שָׁמִיר תַּעֲלֶה כִּי עַל כָּל בָּתֵּי מָשׂוֹשׂ קִרְיָה עַלִּיזָה:

[ישעיה פרק-מט-כ]  עוֹד יֹאמְרוּ בְאָזְנַיִךְ בְּנֵי שִׁכֻּלָיִךְ צַר לִי הַמָּקוֹם גְּשָׁה לִּי וְאֵשֵׁבָה:

[ישעיה פרק-סד-ט]  עָרֵי קָדְשְׁךָ הָיוּ מִדְבָּר צִיּוֹן מִדְבָּר הָיָתָה יְרוּשָׁלִַם שְׁמָמָה:

[ירמיה פרק-ד-כח]  עַל זֹאת תֶּאֱבַל הָאָרֶץ וְקָדְרוּ הַשָּׁמַיִם מִמָּעַל עַל כִּי דִבַּרְתִּי זַמֹּתִי וְלֹא נִחַמְתִּי וְלֹא אָשׁוּב מִמֶּנָּה:

[ירמיה פרק-ה-י]  עֲלוּ בְשָׁרוֹתֶיהָ וְשַׁחֵתוּ וְכָלָה אַל תַּעֲשׂוּ הָסִירוּ נְטִישׁוֹתֶיהָ כִּי לוֹא לַיהוָה הֵמָּה:

[ירמיה פרק-ז-ב]  עֲמֹד בְּשַׁעַר בֵּית יְהוָה וְקָרָאתָ שָּׁם אֶת הַדָּבָר הַזֶּה וְאָמַרְתָּ שִׁמְעוּ דְבַר יְהוָה כָּל יְהוּדָה הַבָּאִים בַּשְּׁעָרִים הָאֵלֶּה לְהִשְׁתַּחֲוֹת לַיהוָה:

[ירמיה פרק-ח-יד]  עַל מָה אֲנַחְנוּ יֹשְׁבִים הֵאָסְפוּ וְנָבוֹא אֶל עָרֵי הַמִּבְצָר וְנִדְּמָה שָּׁם כִּי יְהוָה אֱלֹהֵינוּ הֲדִמָּנוּ וַיַּשְׁקֵנוּ מֵי רֹאשׁ כִּי חָטָאנוּ לַיהוָה:

[ירמיה פרק-יב-ז]  עָזַבְתִּי אֶת בֵּיתִי נָטַשְׁתִּי אֶת נַחֲלָתִי נָתַתִּי אֶת יְדִדוּת נַפְשִׁי בְּכַף אֹיְבֶיהָ:

[ירמיה פרק-מו-ג]  עִרְכוּ מָגֵן וְצִנָּה וּגְשׁוּ לַמִּלְחָמָה:

[ירמיה פרק-מח-לא]  עַל כֵּן עַל מוֹאָב אֲיֵלִיל וּלְמוֹאָב כֻּלֹּה אֶזְעָק אֶל אַנְשֵׁי קִיר חֶרֶשׂ יֶהְגֶּה:

[ירמיה פרק-מח-לח]  עַל כָּל גַּגּוֹת מוֹאָב וּבִרְחֹבֹתֶיהָ כֻּלֹּה מִסְפֵּד כִּי שָׁבַרְתִּי אֶת מוֹאָב כִּכְלִי אֵין חֵפֶץ בּוֹ נְאֻם יְהוָה:

[ירמיה פרק-נ-יד]  עִרְכוּ עַל בָּבֶל סָבִיב כָּל דֹּרְכֵי קֶשֶׁת יְדוּ אֵלֶיהָ אַל תַּחְמְלוּ אֶל חֵץ כִּי לַיהוָה חָטָאָה:

[ירמיה פרק-נא-מב]  עָלָה עַל בָּבֶל הַיָּם בַּהֲמוֹן גַּלָּיו נִכְסָתָה:

[יחזקאל פרק-יח-יב]  עָנִי וְאֶבְיוֹן הוֹנָה גְּזֵלוֹת גָּזָל חֲבֹל לֹא יָשִׁיב וְאֶל הַגִּלּוּלִים נָשָׂא עֵינָיו תּוֹעֵבָה עָשָׂה:

[יחזקאל פרק-לא-יג]  עַל מַפַּלְתּוֹ יִשְׁכְּנוּ כָּל עוֹף הַשָּׁמָיִם וְאֶל פֹּארֹתָיו הָיוּ כֹּל חַיַּת הַשָּׂדֶה:

[יחזקאל פרק-לה-ד]  עָרֶיךָ חָרְבָּה אָשִׂים וְאַתָּה שְׁמָמָה תִהְיֶה וְיָדַעְתָּ כִּי אֲנִי יְהוָה:

[יחזקאל פרק-לט-ד]  עַל הָרֵי יִשְׂרָאֵל תִּפּוֹל אַתָּה וְכָל אֲגַפֶּיךָ וְעַמִּים אֲשֶׁר אִתָּךְ לְעֵיט צִפּוֹר כָּל כָּנָף וְחַיַּת הַשָּׂדֶה נְתַתִּיךָ לְאָכְלָה:

[יחזקאל פרק-לט-ה]  עַל פְּנֵי הַשָּׂדֶה תִּפּוֹל כִּי אֲנִי דִבַּרְתִּי נְאֻם אֲדֹנָי יְהוִה:

[הושע פרק-ד-יג]  עַל רָאשֵׁי הֶהָרִים יְזַבֵּחוּ וְעַל הַגְּבָעוֹת יְקַטֵּרוּ תַּחַת אַלּוֹן וְלִבְנֶה וְאֵלָה כִּי טוֹב צִלָּהּ עַל כֵּן תִּזְנֶינָה בְּנוֹתֵיכֶם וְכַלּוֹתֵיכֶם תְּנָאַפְנָה:

[מיכה פרק-א-ח]  עַל זֹאת אֶסְפְּדָה וְאֵילִילָה אֵילְכָה (שילל) שׁוֹלָל וְעָרוֹם אֶעֱשֶׂה מִסְפֵּד כַּתַּנִּים וְאֵבֶל כִּבְנוֹת יַעֲנָה:

[מיכה פרק-ד-ט]  עַתָּה לָמָּה תָרִיעִי רֵעַ הֲמֶלֶךְ אֵין בָּךְ אִם יוֹעֲצֵךְ אָבָד כִּי הֶחֱזִיקֵךְ חִיל כַּיּוֹלֵדָה:

[מיכה פרק-ו-ה]  עַמִּי זְכָר נָא מַה יָּעַץ בָּלָק מֶלֶךְ מוֹאָב וּמֶה עָנָה אֹתוֹ בִּלְעָם בֶּן בְּעוֹר מִן הַשִּׁטִּים עַד הַגִּלְגָּל לְמַעַן דַּעַת צִדְקוֹת יְהוָה:

[מיכה פרק-ו-י]  עוֹד הַאִשׁ בֵּית רָשָׁע אֹצְרוֹת רֶשַׁע וְאֵיפַת רָזוֹן זְעוּמָה:

[מיכה פרק-ז-ג]  עַל הָרַע כַּפַּיִם לְהֵיטִיב הַשַּׂר שֹׁאֵל וְהַשֹּׁפֵט בַּשִּׁלּוּם וְהַגָּדוֹל דֹּבֵר הַוַּת נַפְשׁוֹ הוּא וַיְעַבְּתוּהָ:

[חבקוק פרק-א-טז]  עַל כֵּן יְזַבֵּחַ לְחֶרְמוֹ וִיקַטֵּר לְמִכְמַרְתּוֹ כִּי בָהֵמָּה שָׁמֵן חֶלְקוֹ וּמַאֲכָלוֹ בְּרִאָה:

[חגי פרק-א-ח]  עֲלוּ הָהָר וַהֲבֵאתֶם עֵץ וּבְנוּ הַבָּיִת וְאֶרְצֶה בּוֹ (ואכבד) וְאֶכָּבְדָה אָמַר יְהוָה:

[חגי פרק-א-י]  עַל כֵּן עֲלֵיכֶם כָּלְאוּ שָׁמַיִם מִטָּל וְהָאָרֶץ כָּלְאָה יְבוּלָהּ:

[זכריה פרק-י-ג]  עַל הָרֹעִים חָרָה אַפִּי וְעַל הָעַתּוּדִים אֶפְקוֹד כִּי פָקַד יְהוָה צְבָאוֹת אֶת עֶדְרוֹ אֶת בֵּית יְהוּדָה וְשָׂם אוֹתָם כְּסוּס הוֹדוֹ בַּמִּלְחָמָה:

[תהילים פרק-ב-יא]  עִבְדוּ אֶת יְהוָה בְּיִרְאָה וְגִילוּ בִּרְעָדָה:

[תהילים פרק-יח-נ]  עַל כֵּן אוֹדְךָ בַגּוֹיִם יְהוָה וּלְשִׁמְךָ אֲזַמֵּרָה:

[תהילים פרק-כב-יא]  עָלֶיךָ הָשְׁלַכְתִּי מֵרָחֶם מִבֶּטֶן אִמִּי אֵלִי אָתָּה:

[תהילים פרק-סב-ד]  עַד אָנָה תְּהוֹתְתוּ עַל אִישׁ תְּרָצְּחוּ כֻלְּכֶם כְּקִיר נָטוּי גָּדֵר הַדְּחוּיָה:

[תהילים פרק-סו-טו]  עֹלוֹת מֵחִים אַעֲלֶה לָּךְ עִם קְטֹרֶת אֵילִים אֶעֱשֶׂה בָקָר עִם עַתּוּדִים סֶלָה:

[תהילים פרק-פב-ב]  עַד מָתַי תִּשְׁפְּטוּ עָוֶל וּפְנֵי רְשָׁעִים תִּשְׂאוּ סֶלָה:

[תהילים פרק-פד-ז]  עֹבְרֵי בְּעֵמֶק הַבָּכָא מַעְיָן יְשִׁיתוּהוּ גַּם בְּרָכוֹת יַעְטֶה מוֹרֶה:

[תהילים פרק-פח-ח]  עָלַי סָמְכָה חֲמָתֶךָ וְכָל מִשְׁבָּרֶיךָ עִנִּיתָ סֶּלָה:

[תהילים פרק-פח-טז]  עָנִי אֲנִי וְגוֵֹעַ מִנֹּעַר נָשָׂאתִי אֵמֶיךָ אָפוּנָה:

[תהילים פרק-פט-ה]  עַד עוֹלָם אָכִין זַרְעֶךָ וּבָנִיתִי לְדֹר וָדוֹר כִּסְאֲךָ סֶלָה:

[תהילים פרק-ק-ב]  עִבְדוּ אֶת יְהוָה בְּשִׂמְחָה בֹּאוּ לְפָנָיו בִּרְנָנָה:

[תהילים פרק-קד-ב]  עֹטֶה אוֹר כַּשַּׂלְמָה נוֹטֶה שָׁמַיִם כַּיְרִיעָה:

[תהילים פרק-קיח-יד]  עָזִּי וְזִמְרָת יָהּ וַיְהִי לִי לִישׁוּעָה:

[משלי פרק-ח-יח]  עֹשֶׁר וְכָבוֹד אִתִּי הוֹן עָתֵק וּצְדָקָה:

[משלי פרק-ט-ו]  עִזְבוּ פְתָאיִם וִחְיוּ וְאִשְׁרוּ בְּדֶרֶךְ בִּינָה:

[משלי פרק-טו-כז]  עֹכֵר בֵּיתוֹ בּוֹצֵעַ בָּצַע וְשׂוֹנֵא מַתָּנֹת יִחְיֶה:

[משלי פרק-טז-ל]  עֹצֶה עֵינָיו לַחְשֹׁב תַּהְפֻּכוֹת קֹרֵץ שְׂפָתָיו כִּלָּה רָעָה:

[משלי פרק-יז-ב]  עֶבֶד מַשְׂכִּיל יִמְשֹׁל בְּבֵן מֵבִישׁ וּבְתוֹךְ אַחִים יַחֲלֹק נַחֲלָה:

[משלי פרק-יז-כ]  עִקֶּשׁ לֵב לֹא יִמְצָא טוֹב וְנֶהְפָּךְ בִּלְשׁוֹנוֹ יִפּוֹל בְּרָעָה:

[משלי פרק-כא-כב]  עִיר גִּבֹּרִים עָלָה חָכָם וַיֹּרֶד עֹז מִבְטֶחָה:

[משלי פרק-כב-ב]  עָשִׁיר וָרָשׁ נִפְגָּשׁוּ עֹשֵׂה כֻלָּם יְהוָה:

[משלי פרק-כב-ז]  עָשִׁיר בְּרָשִׁים יִמְשׁוֹל וְעֶבֶד לוֶֹה לְאִישׁ מַלְוֶה:

[איוב פרק-ח-יז]  עַל גַּל שָׁרָשָׁיו יְסֻבָּכוּ בֵּית אֲבָנִים יֶחֱזֶה:

[איוב פרק-ח-כא]  עַד יְמַלֵּה שְׂחוֹק פִּיךָ וּשְׂפָתֶיךָ תְרוּעָה:

[איוב פרק-יב-יג]  עִמּוֹ חָכְמָה וּגְבוּרָה לוֹ עֵצָה וּתְבוּנָה:

[איוב פרק-יב-טז]  עִמּוֹ עֹז וְתוּשִׁיָּה לוֹ שֹׁגֵג וּמַשְׁגֶּה:

[איוב פרק-טז-יז]  עַל לֹא חָמָס בְּכַפָּי וּתְפִלָּתִי זַכָּה:

[איוב פרק-כא-כד]  עֲטִינָיו מָלְאוּ חָלָב וּמֹחַ עַצְמוֹתָיו יְשֻׁקֶּה:

[איוב פרק-כד-ז]  עָרוֹם יָלִינוּ מִבְּלִי לְבוּשׁ וְאֵין כְּסוּת בַּקָּרָה:

[איוב פרק-לח-ו]  עַל מָה אֲדָנֶיהָ הָטְבָּעוּ אוֹ מִי יָרָה אֶבֶן פִּנָּתָהּ:

[שיר השירים פרק-ד-ו]  עַד שֶׁיָּפוּחַ הַיּוֹם וְנָסוּ הַצְּלָלִים אֵלֶךְ לִי אֶל הַר הַמּוֹר וְאֶל גִּבְעַת הַלְּבוֹנָה:

[דניאל פרק-ב-ז]  עֲנוֹ תִנְיָנוּת וְאָמְרִין מַלְכָּא חֶלְמָא יֵאמַר לְעַבְדוֹהִי וּפִשְׁרָה נְהַחֲוֵה:

[דניאל פרק-ב-כו]  עָנֵה מַלְכָּא וְאָמַר לְדָנִיֵּאל דִּי שְׁמֵהּ בֵּלְטְשַׁאצַּר (האיתיך) הַאִיתָךְ כָּהֵל לְהוֹדָעֻתַנִי חֶלְמָא דִי חֲזֵית וּפִשְׁרֵהּ:

[דניאל פרק-ב-מז]  עָנֵה מַלְכָּא לְדָנִיֵּאל וְאָמַר מִן קְשֹׁט דִּי אֱלָהֲכוֹן הוּא אֱלָהּ אֱלָהִין וּמָרֵא מַלְכִין וְגָלֵה רָזִין דִּי יְכֵלְתָּ לְמִגְלֵא רָזָה דְנָה:

[עזרא פרק-י-מא]  עֲזַרְאֵל וְשֶׁלֶמְיָהוּ שְׁמַרְיָה:

[נחמיה פרק-ג-ח]  עַל יָדוֹ הֶחֱזִיק עֻזִּיאֵל בֶּן חַרְהֲיָה צוֹרְפִים וְעַל יָדוֹ הֶחֱזִיק חֲנַנְיָה בֶּן הָרַקָּחִים וַיַּעַזְבוּ יְרוּשָׁלִַם עַד הַחוֹמָה הָרְחָבָה:

[נחמיה פרק-יא-לב]  עֲנָתוֹת נֹב עֲנָנְיָה:

[נחמיה פרק-יב-ד]  עִדּוֹא גִנְּתוֹי אֲבִיָּה:

[דברי הימים א פרק-ט-ד]  עוּתַי בֶּן עַמִּיהוּד בֶּן עָמְרִי בֶּן אִמְרִי בֶן (בנימן) בָּנִי מִן בְּנֵי פֶרֶץ בֶּן יְהוּדָה:

[דברי הימים א פרק-כב-יט]  עַתָּה תְּנוּ לְבַבְכֶם וְנַפְשְׁכֶם לִדְרוֹשׁ לַיהוָה אֱלֹהֵיכֶם וְקוּמוּ וּבְנוּ אֶת מִקְדַּשׁ יְהוָה הָאֱלֹהִים לְהָבִיא אֶת אֲרוֹן בְּרִית יְהוָה וּכְלֵי קֹדֶשׁ הָאֱלֹהִים לַבַּיִת הַנִּבְנֶה לְשֵׁם יְהוָה:

[דברי הימים ב פרק-ד-ד]  עוֹמֵד עַל שְׁנֵים עָשָׂר בָּקָר שְׁלֹשָׁה פֹנִים צָפוֹנָה וּשְׁלוֹשָׁה פֹנִים יָמָּה וּשְׁלֹשָׁה פֹּנִים נֶגְבָּה וּשְׁלֹשָׁה פֹּנִים מִזְרָחָה וְהַיָּם עֲלֵיהֶם מִלְמָעְלָה וְכָל אֲחֹרֵיהֶם בָּיְתָה:

[דברי הימים ב פרק-ו-מ]  עַתָּה אֱלֹהַי יִהְיוּ נָא עֵינֶיךָ פְּתֻחוֹת וְאָזְנֶיךָ קַשֻּׁבוֹת לִתְפִלַּת הַמָּקוֹם הַזֶּה:

[דברי הימים ב פרק-ז-טו]  עַתָּה עֵינַי יִהְיוּ פְתֻחוֹת וְאָזְנַי קַשֻּׁבוֹת לִתְפִלַּת הַמָּקוֹם הַזֶּה:

[דברי הימים ב פרק-לב-ח]  עִמּוֹ זְרוֹעַ בָּשָׂר וְעִמָּנוּ יְהוָה אֱלֹהֵינוּ לְעָזְרֵנוּ וּלְהִלָּחֵם מִלְחֲמֹתֵנוּ וַיִּסָּמְכוּ הָעָם עַל דִּבְרֵי יְחִזְקִיָּהוּ מֶלֶךְ יְהוּדָה:

[דברי הימים ב פרק-לו-ו]  עָלָיו עָלָה נְבוּכַדְנֶאצַּר מֶלֶךְ בָּבֶל וַיַּאַסְרֵהוּ בַּנְחֻשְׁתַּיִם לְהֹלִיכוֹ בָּבֶלָה:


ע-ו


[בראשית פרק-ח-כב]  עֹד כָּל יְמֵי הָאָרֶץ זֶרַע וְקָצִיר וְקֹר וָחֹם וְקַיִץ וָחֹרֶף וְיוֹם וָלַיְלָה לֹא יִשְׁבֹּתוּ:

[שמות פרק-טו-ב]  עָזִּי וְזִמְרָת יָהּ וַיְהִי לִי לִישׁוּעָה זֶה אֵלִי וְאַנְוֵהוּ אֱלֹהֵי אָבִי וַאֲרֹמְמֶנְהוּ:

[שמות פרק-כב-ח]  עַל כָּל דְּבַר פֶּשַׁע עַל שׁוֹר עַל חֲמוֹר עַל שֶׂה עַל שַׂלְמָה עַל כָּל אֲבֵדָה אֲשֶׁר יֹאמַר כִּי הוּא זֶה עַד הָאֱלֹהִים יָבֹא דְּבַר שְׁנֵיהֶם אֲשֶׁר יַרְשִׁיעֻן אֱלֹהִים יְשַׁלֵּם שְׁנַיִם לְרֵעֵהוּ:

[ויקרא פרק-ז-יג]  עַל חַלֹּת לֶחֶם חָמֵץ יַקְרִיב קָרְבָּנוֹ עַל זֶבַח תּוֹדַת שְׁלָמָיו:

[במדבר פרק-ט-יח]  עַל פִּי יְהוָה יִסְעוּ בְּנֵי יִשְׂרָאֵל וְעַל פִּי יְהוָה יַחֲנוּ כָּל יְמֵי אֲשֶׁר יִשְׁכֹּן הֶעָנָן עַל הַמִּשְׁכָּן יַחֲנוּ:

[דברים פרק-י-ט]  עַל כֵּן לֹא הָיָה לְלֵוִי חֵלֶק וְנַחֲלָה עִם אֶחָיו יְהוָה הוּא נַחֲלָתוֹ כַּאֲשֶׁר דִּבֶּר יְהוָה אֱלֹהֶיךָ לוֹ:

[דברים פרק-כג-יז]  עִמְּךָ יֵשֵׁב בְּקִרְבְּךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר בְּאַחַד שְׁעָרֶיךָ בַּטּוֹב לוֹ לֹא תּוֹנֶנּוּ:

[שמואל ב פרק-ז-כב]  עַל כֵּן גָּדַלְתָּ אֲדֹנָי יְהוִה כִּי אֵין כָּמוֹךָ וְאֵין אֱלֹהִים זוּלָתֶךָ בְּכֹל אֲשֶׁר שָׁמַעְנוּ בְּאָזְנֵינוּ:

[שמואל ב פרק-כב-ט]  עָלָה עָשָׁן בְּאַפּוֹ וְאֵשׁ מִפִּיו תֹּאכֵל גֶּחָלִים בָּעֲרוּ מִמֶּנּוּ:

[מלכים ב פרק-יח-יב]  עַל אֲשֶׁר לֹא שָׁמְעוּ בְּקוֹל יְהוָה אֱלֹהֵיהֶם וַיַּעַבְרוּ אֶת בְּרִיתוֹ אֵת כָּל אֲשֶׁר צִוָּה מֹשֶׁה עֶבֶד יְהוָה וְלֹא שָׁמְעוּ וְלֹא עָשׂוּ:

[מלכים ב פרק-יח-כא]  עַתָּה הִנֵּה בָטַחְתָּ לְּךָ עַל מִשְׁעֶנֶת הַקָּנֶה הָרָצוּץ הַזֶּה עַל מִצְרַיִם אֲשֶׁר יִסָּמֵךְ אִישׁ עָלָיו וּבָא בְכַפּוֹ וּנְקָבָהּ כֵּן פַּרְעֹה מֶלֶךְ מִצְרַיִם לְכָל הַבֹּטְחִים עָלָיו:

[מלכים ב פרק-יח-לב]  עַד בֹּאִי וְלָקַחְתִּי אֶתְכֶם אֶל אֶרֶץ כְּאַרְצְכֶם אֶרֶץ דָּגָן וְתִירוֹשׁ אֶרֶץ לֶחֶם וּכְרָמִים אֶרֶץ זֵית יִצְהָר וּדְבַשׁ וִחְיוּ וְלֹא תָמֻתוּ וְאַל תִּשְׁמְעוּ אֶל חִזְקִיָּהוּ כִּי יַסִּית אֶתְכֶם לֵאמֹר יְהוָה יַצִּילֵנוּ:

[ישעיה פרק-ג-יב]  עַמִּי נֹגְשָׂיו מְעוֹלֵל וְנָשִׁים מָשְׁלוּ בוֹ עַמִּי מְאַשְּׁרֶיךָ מַתְעִים וְדֶרֶךְ אֹרְחֹתֶיךָ בִּלֵּעוּ:

[ישעיה פרק-יג-יג]  עַל כֵּן שָׁמַיִם אַרְגִּיז וְתִרְעַשׁ הָאָרֶץ מִמְּקוֹמָהּ בְּעֶבְרַת יְהוָה צְבָאוֹת וּבְיוֹם חֲרוֹן אַפּוֹ:

[ישעיה פרק-יט-ז]  עָרוֹת עַל יְאוֹר עַל פִּי יְאוֹר וְכֹל מִזְרַע יְאוֹר יִיבַשׁ נִדַּף וְאֵינֶנּוּ:

[ישעיה פרק-מג-כא]  עַם זוּ יָצַרְתִּי לִי תְּהִלָּתִי יְסַפֵּרוּ:

[ירמיה פרק-ד-ח]  עַל זֹאת חִגְרוּ שַׂקִּים סִפְדוּ וְהֵילִילוּ כִּי לֹא שָׁב חֲרוֹן אַף יְהוָֹה מִמֶּנּוּ:

[ירמיה פרק-ו-י]  עַל מִי אֲדַבְּרָה וְאָעִידָה וְיִשְׁמָעוּ הִנֵּה עֲרֵלָה אָזְנָם וְלֹא יוּכְלוּ לְהַקְשִׁיב הִנֵּה דְבַר יְהוָה הָיָה לָהֶם לְחֶרְפָּה לֹא יַחְפְּצוּ בוֹ:

[ירמיה פרק-ח-כ]  עָבַר קָצִיר כָּלָה קָיִץ וַאֲנַחְנוּ לוֹא נוֹשָׁעְנוּ:

[ירמיה פרק-ט-ט]  עַל הֶהָרִים אֶשָּׂא בְכִי וָנֶהִי וְעַל נְאוֹת מִדְבָּר קִינָה כִּי נִצְּתוּ מִבְּלִי אִישׁ עֹבֵר וְלֹא שָׁמְעוּ קוֹל מִקְנֶה מֵעוֹף הַשָּׁמַיִם וְעַד בְּהֵמָה נָדְדוּ הָלָכוּ:

[ירמיה פרק-יב-ד]  עַד מָתַי תֶּאֱבַל הָאָרֶץ וְעֵשֶׂב כָּל הַשָּׂדֶה יִיבָשׁ מֵרָעַת יֹשְׁבֵי בָהּ סָפְתָה בְהֵמוֹת וָעוֹף כִּי אָמְרוּ לֹא יִרְאֶה אֶת אַחֲרִיתֵנוּ:

[ירמיה פרק-יז-ט]  עָקֹב הַלֵּב מִכֹּל וְאָנֻשׁ הוּא מִי יֵדָעֶנּוּ:

[ירמיה פרק-כה-לח]  עָזַב כַּכְּפִיר סֻכּוֹ כִּי הָיְתָה אַרְצָם לְשַׁמָּה מִפְּנֵי חֲרוֹן הַיּוֹנָה וּמִפְּנֵי חֲרוֹן אַפּוֹ:

[ירמיה פרק-לא-ה]  עוֹד תִּטְּעִי כְרָמִים בְּהָרֵי שֹׁמְרוֹן נָטְעוּ נֹטְעִים וְחִלֵּלוּ:

[ירמיה פרק-לב-יח]  עֹשֶׂה חֶסֶד לַאֲלָפִים וּמְשַׁלֵּם עֲוֹן אָבוֹת אֶל חֵיק בְּנֵיהֶם אַחֲרֵיהֶם הָאֵל הַגָּדוֹל הַגִּבּוֹר יְהוָה צְבָאוֹת שְׁמוֹ:

[ירמיה פרק-מח-לו]  עַל כֵּן לִבִּי לְמוֹאָב כַּחֲלִלִים יֶהֱמֶה וְלִבִּי אֶל אַנְשֵׁי קִיר חֶרֶשׂ כַּחֲלִילִים יֶהֱמֶה עַל כֵּן יִתְרַת עָשָׂה אָבָדוּ:

[ירמיה פרק-מט-יא]  עָזְבָה יְתֹמֶיךָ אֲנִי אֲחַיֶּה וְאַלְמְנֹתֶיךָ עָלַי תִּבְטָחוּ:

[יחזקאל פרק-כא-לב]  עַוָּה עַוָּה עַוָּה אֲשִׂימֶנָּה גַּם זֹאת לֹא הָיָה עַד בֹּא אֲשֶׁר לוֹ הַמִּשְׁפָּט וּנְתַתִּיו:

[יחזקאל פרק-כג-לח]  עוֹד זֹאת עָשׂוּ לִי טִמְּאוּ אֶת מִקְדָּשִׁי בַּיּוֹם הַהוּא וְאֶת שַׁבְּתוֹתַי חִלֵּלוּ:

[יחזקאל פרק-כז-לד]  עֵת נִשְׁבֶּרֶת מִיַּמִּים בְּמַעֲמַקֵּי מָיִם מַעֲרָבֵךְ וְכָל קְהָלֵךְ בְּתוֹכֵךְ נָפָלוּ:

[יחזקאל פרק-לא-ה]  עַל כֵּן גָּבְהָא קֹמָתוֹ מִכֹּל עֲצֵי הַשָּׂדֶה וַתִּרְבֶּינָה סַרְעַפֹּתָיו וַתֶּאֱרַכְנָה (פארתו) פֹארֹתָיו מִמַּיִם רַבִּים בְּשַׁלְּחוֹ:

[יחזקאל פרק-לג-כו]  עֲמַדְתֶּם עַל חַרְבְּכֶם עֲשִׂיתֶן תּוֹעֵבָה וְאִישׁ אֶת אֵשֶׁת רֵעֵהוּ טִמֵּאתֶם וְהָאָרֶץ תִּירָשׁוּ:

[הושע פרק-ד-ג]  עַל כֵּן תֶּאֱבַל הָאָרֶץ וְאֻמְלַל כָּל יוֹשֵׁב בָּהּ בְּחַיַּת הַשָּׂדֶה וּבְעוֹף הַשָּׁמָיִם וְגַם דְּגֵי הַיָּם יֵאָסֵפוּ:

[הושע פרק-ה-יא]  עָשׁוּק אֶפְרַיִם רְצוּץ מִשְׁפָּט כִּי הוֹאִיל הָלַךְ אַחֲרֵי צָו:

[עמוס פרק-ה-ח]  עֹשֵׂה כִימָה וּכְסִיל וְהֹפֵךְ לַבֹּקֶר צַלְמָוֶת וְיוֹם לַיְלָה הֶחְשִׁיךְ הַקּוֹרֵא לְמֵי הַיָּם וַיִּשְׁפְּכֵם עַל פְּנֵי הָאָרֶץ יְהוָה שְׁמוֹ:

[עובדיה פרק-א-ז]  עַד הַגְּבוּל שִׁלְּחוּךָ כֹּל אַנְשֵׁי בְרִיתֶךָ הִשִּׁיאוּךָ יָכְלוּ לְךָ אַנְשֵׁי שְׁלֹמֶךָ לַחְמְךָ יָשִׂימוּ מָזוֹר תַּחְתֶּיךָ אֵין תְּבוּנָה בּוֹ:

[מיכה פרק-א-יא]  עִבְרִי לָכֶם יוֹשֶׁבֶת שָׁפִיר עֶרְיָה בֹשֶׁת לֹא יָצְאָה יוֹשֶׁבֶת צַאֲנָן מִסְפַּד בֵּית הָאֵצֶל יִקַּח מִכֶּם עֶמְדָּתוֹ:

[חבקוק פרק-ג-ו]  עָמַד וַיְמֹדֶד אֶרֶץ רָאָה וַיַּתֵּר גּוֹיִם וַיִּתְפֹּצְצוּ הַרְרֵי עַד שַׁחוּ גִּבְעוֹת עוֹלָם הֲלִיכוֹת עוֹלָם לוֹ:

[תהילים פרק-יד-ו]  עֲצַת עָנִי תָבִישׁוּ כִּי יְהוָה מַחְסֵהוּ:

[תהילים פרק-יח-ט]  עָלָה עָשָׁן בְּאַפּוֹ וְאֵשׁ מִפִּיו תֹּאכֵל גֶּחָלִים בָּעֲרוּ מִמֶּנּוּ:

[תהילים פרק-כ-ז]  עַתָּה יָדַעְתִּי כִּי הוֹשִׁיעַ יְהוָה מְשִׁיחוֹ יַעֲנֵהוּ מִשְּׁמֵי קָדְשׁוֹ בִּגְבֻרוֹת יֵשַׁע יְמִינוֹ:

[תהילים פרק-לב-ו]  עַל זֹאת יִתְפַּלֵּל כָּל חָסִיד אֵלֶיךָ לְעֵת מְצֹא רַק לְשֵׁטֶף מַיִם רַבִּים אֵלָיו לֹא יַגִּיעוּ:

[תהילים פרק-צב-טו]  עוֹד יְנוּבוּן בְּשֵׂיבָה דְּשֵׁנִים וְרַעֲנַנִּים יִהְיוּ:

[תהילים פרק-צד-ג]  עַד מָתַי רְשָׁעִים יְהוָה עַד מָתַי רְשָׁעִים יַעֲלֹזוּ:

[תהילים פרק-צד-ה]  עַמְּךָ יְהוָה יְדַכְּאוּ וְנַחֲלָתְךָ יְעַנּוּ:

[תהילים פרק-צז-ב]  עָנָן וַעֲרָפֶל סְבִיבָיו צֶדֶק וּמִשְׁפָּט מְכוֹן כִּסְאוֹ:

[תהילים פרק-קד-יט]  עָשָׂה יָרֵחַ לְמוֹעֲדִים שֶׁמֶשׁ יָדַע מְבוֹאוֹ:

[תהילים פרק-קה-יח]  עִנּוּ בַכֶּבֶל (רגליו) רַגְלוֹ בַּרְזֶל בָּאָה נַפְשׁוֹ:

[תהילים פרק-קה-יט]  עַד עֵת בֹּא דְבָרוֹ אִמְרַת יְהוָה צְרָפָתְהוּ:

[תהילים פרק-קלז-ב]  עַל עֲרָבִים בְּתוֹכָהּ תָּלִינוּ כִּנֹּרוֹתֵינוּ:

[תהילים פרק-קמה-טו]  עֵינֵי כֹל אֵלֶיךָ יְשַׂבֵּרוּ וְאַתָּה נוֹתֵן לָהֶם אֶת אָכְלָם בְּעִתּוֹ:

[משלי פרק-כב-ג]  עָרוּם רָאָה רָעָה (ויסתר) וְנִסְתָּר וּפְתָיִים עָבְרוּ וְנֶעֱנָשׁוּ:

[משלי פרק-כה-כח]  עִיר פְּרוּצָה אֵין חוֹמָה אִישׁ אֲשֶׁר אֵין מַעְצָר לְרוּחוֹ:

[משלי פרק-כו-ה]  עֲנֵה כְסִיל כְּאִוַּלְתּוֹ פֶּן יִהְיֶה חָכָם בְּעֵינָיו:

[משלי פרק-כז-יב]  עָרוּם רָאָה רָעָה נִסְתָּר פְּתָאיִם עָבְרוּ נֶעֱנָשׁוּ:

[איוב פרק-ג-יד]  עִם מְלָכִים וְיֹעֲצֵי אָרֶץ הַבֹּנִים חֳרָבוֹת לָמוֹ:

[איוב פרק-כ-ט]  עַיִן שְׁזָפַתּוּ וְלֹא תוֹסִיף וְלֹא עוֹד תְּשׁוּרֶנּוּ מְקוֹמוֹ:

[איוב פרק-כג-טו]  עַל כֵּן מִפָּנָיו אֶבָּהֵל אֶתְבּוֹנֵן וְאֶפְחַד מִמֶּנּוּ:

[איוב פרק-כו-יא]  עַמּוּדֵי שָׁמַיִם יְרוֹפָפוּ וְיִתְמְהוּ מִגַּעֲרָתוֹ:

[איוב פרק-כז-יט]  עָשִׁיר יִשְׁכַּב וְלֹא יֵאָסֵף עֵינָיו פָּקַח וְאֵינֶנּוּ:

[שיר השירים פרק-א-יב]  עַד שֶׁהַמֶּלֶךְ בִּמְסִבּוֹ נִרְדִּי נָתַן רֵיחוֹ:

[שיר השירים פרק-ג-א]  עַל מִשְׁכָּבִי בַּלֵּילוֹת בִּקַּשְׁתִּי אֵת שֶׁאָהֲבָה נַפְשִׁי בִּקַּשְׁתִּיו וְלֹא מְצָאתִיו:

[שיר השירים פרק-ד-טז]  עוּרִי צָפוֹן וּבוֹאִי תֵימָן הָפִיחִי גַנִּי יִזְּלוּ בְשָׂמָיו יָבֹא דוֹדִי לְגַנּוֹ וְיֹאכַל פְּרִי מְגָדָיו:

[איכה פרק-ה-ה]  עַל צַוָּארֵנוּ נִרְדָּפְנוּ יָגַעְנוּ (לא) וְלֹא הוּנַח לָנוּ:

[איכה פרק-ה-יז]  עַל זֶה הָיָה דָוֶה לִבֵּנוּ עַל אֵלֶּה חָשְׁכוּ עֵינֵינוּ:

[איכה פרק-ה-יח]  עַל הַר צִיּוֹן שֶׁשָּׁמֵם שׁוּעָלִים הִלְּכוּ בוֹ:

[אסתר פרק-ט-יט]  עַל כֵּן הַיְּהוּדִים (הפרוזים) הַפְּרָזִים הַיֹּשְׁבִים בְּעָרֵי הַפְּרָזוֹת עֹשִׂים אֵת יוֹם אַרְבָּעָה עָשָׂר לְחֹדֶשׁ אֲדָר שִׂמְחָה וּמִשְׁתֶּה וְיוֹם טוֹב וּמִשְׁלוֹחַ מָנוֹת אִישׁ לְרֵעֵהוּ:

[דברי הימים א פרק-א-כה]  עֵבֶר פֶּלֶג רְעוּ:

[דברי הימים ב פרק-כט-י]  עַתָּה עִם לְבָבִי לִכְרוֹת בְּרִית לַיהוָה אֱלֹהֵי יִשְׂרָאֵל וְיָשֹׁב מִמֶּנּוּ חֲרוֹן אַפּוֹ:

[דברי הימים ב פרק-ל-ח]  עַתָּה אַל תַּקְשׁוּ עָרְפְּכֶם כַּאֲבוֹתֵיכֶם תְּנוּ יָד לַיהוָה וּבֹאוּ לְמִקְדָּשׁוֹ אֲשֶׁר הִקְדִּישׁ לְעוֹלָם וְעִבְדוּ אֶת יְהוָה אֱלֹהֵיכֶם וְיָשֹׁב מִכֶּם חֲרוֹן אַפּוֹ:


ע-ז


[תהילים פרק-קיט-קכז]  עַל כֵּן אָהַבְתִּי מִצְוֹתֶיךָ מִזָּהָב וּמִפָּז:


ע-ח


[ישעיה פרק-נז-ז]  עַל הַר גָּבֹהַּ וְנִשָּׂא שַׂמְתְּ מִשְׁכָּבֵךְ גַּם שָׁם עָלִית לִזְבֹּחַ זָבַח:

[תהילים פרק-מד-כד]  עוּרָה לָמָּה תִישַׁן אֲדֹנָי הָקִיצָה אַל תִּזְנַח לָנֶצַח:

[תהילים פרק-עד-י]  עַד מָתַי אֱלֹהִים יְחָרֶף צָר יְנָאֵץ אוֹיֵב שִׁמְךָ לָנֶצַח:

[משלי פרק-כא-ג]  עֲשֹׂה צְדָקָה וּמִשְׁפָּט נִבְחָר לַיהוָה מִזָּבַח:


ע-ט


[במדבר פרק-כו-נו]  עַל פִּי הַגּוֹרָל תֵּחָלֵק נַחֲלָתוֹ בֵּין רַב לִמְעָט:

[יחזקאל פרק-כב-כט]  עַם הָאָרֶץ עָשְׁקוּ עֹשֶׁק וְגָזְלוּ גָּזֵל וְעָנִי וְאֶבְיוֹן הוֹנוּ וְאֶת הַגֵּר עָשְׁקוּ בְּלֹא מִשְׁפָּט:

[תהילים פרק-קד-ד]  עֹשֶׂה מַלְאָכָיו רוּחוֹת מְשָׁרְתָיו אֵשׁ לֹהֵט:

[משלי פרק-יט-ה]  עֵד שְׁקָרִים לֹא יִנָּקֶה וְיָפִיחַ כְּזָבִים לֹא יִמָּלֵט:


ע-י


[בראשית פרק-כו-ה]  עֵקֶב אֲשֶׁר שָׁמַע אַבְרָהָם בְּקֹלִי וַיִּשְׁמֹר מִשְׁמַרְתִּי מִצְוֹתַי חֻקּוֹתַי וְתוֹרֹתָי:

[בראשית פרק-לו-ב]  עֵשָׂו לָקַח אֶת נָשָׁיו מִבְּנוֹת כְּנָעַן אֶת עָדָה בַּת אֵילוֹן הַחִתִּי וְאֶת אָהֳלִיבָמָה בַּת עֲנָה בַּת צִבְעוֹן הַחִוִּי:

[במדבר פרק-יד-כז]  עַד מָתַי לָעֵדָה הָרָעָה הַזֹּאת אֲשֶׁר הֵמָּה מַלִּינִים עָלָי אֶת תְּלֻנּוֹת בְּנֵי יִשְׂרָאֵל אֲשֶׁר הֵמָּה מַלִּינִים עָלַי שָׁמָעְתִּי:

[שמואל ב פרק-כג-לח]  עִירָא הַיִּתְרִי גָּרֵב הַיִּתְרִי:

[מלכים א פרק-ה-כג]  עֲבָדַי יֹרִדוּ מִן הַלְּבָנוֹן יָמָּה וַאֲנִי אֲשִׂימֵם דֹּבְרוֹת בַּיָּם עַד הַמָּקוֹם אֲשֶׁר תִּשְׁלַח אֵלַי וְנִפַּצְתִּים שָׁם וְאַתָּה תִשָּׂא וְאַתָּה תַּעֲשֶׂה אֶת חֶפְצִי לָתֵת לֶחֶם בֵּיתִי:

[ישעיה פרק-א-ה]  עַל מֶה תֻכּוּ עוֹד תּוֹסִיפוּ סָרָה כָּל רֹאשׁ לָחֳלִי וְכָל לֵבָב דַּוָּי:

[ישעיה פרק-כב-ד]  עַל כֵּן אָמַרְתִּי שְׁעוּ מִנִּי אֲמָרֵר בַּבֶּכִי אַל תָּאִיצוּ לְנַחֲמֵנִי עַל שֹׁד בַּת עַמִּי:

[ישעיה פרק-כג-ו]  עִבְרוּ תַּרְשִׁישָׁה הֵילִילוּ יֹשְׁבֵי אִי:

[ישעיה פרק-מז-יב]  עִמְדִי נָא בַחֲבָרַיִךְ וּבְרֹב כְּשָׁפַיִךְ בַּאֲשֶׁר יָגַעַתְּ מִנְּעוּרָיִךְ אוּלַי תּוּכְלִי הוֹעִיל אוּלַי תַּעֲרוֹצִי:

[ירמיה פרק-ח-כא]  עַל שֶׁבֶר בַּת עַמִּי הָשְׁבָּרְתִּי קָדַרְתִּי שַׁמָּה הֶחֱזִקָתְנִי:

[ירמיה פרק-לא-כו]  עַל זֹאת הֱקִיצֹתִי וָאֶרְאֶה וּשְׁנָתִי עָרְבָה לִּי:

[מיכה פרק-ו-ג]  עַמִּי מֶה עָשִׂיתִי לְךָ וּמָה הֶלְאֵתִיךָ עֲנֵה בִי:

[חבקוק פרק-ב-א]  עַל מִשְׁמַרְתִּי אֶעֱמֹדָה וְאֶתְיַצְּבָה עַל מָצוֹר וַאֲצַפֶּה לִרְאוֹת מַה יְדַבֶּר בִּי וּמָה אָשִׁיב עַל תּוֹכַחְתִּי:

[תהילים פרק-ו-ח]  עָשְׁשָׁה מִכַּעַס עֵינִי עָתְקָה בְּכָל צוֹרְרָי:

[תהילים פרק-יג-ב]  עַד אָנָה יְהוָה תִּשְׁכָּחֵנִי נֶצַח עַד אָנָה תַּסְתִּיר אֶת פָּנֶיךָ מִמֶּנִּי:

[תהילים פרק-יג-ג]  עַד אָנָה אָשִׁית עֵצוֹת בְּנַפְשִׁי יָגוֹן בִּלְבָבִי יוֹמָם עַד אָנָה יָרוּם אֹיְבִי עָלָי:

[תהילים פרק-כה-טו]  עֵינַי תָּמִיד אֶל יְהוָה כִּי הוּא יוֹצִיא מֵרֶשֶׁת רַגְלָי:

[תהילים פרק-נה-יח]  עֶרֶב וָבֹקֶר וְצָהֳרַיִם אָשִׂיחָה וְאֶהֱמֶה וַיִּשְׁמַע קוֹלִי:

[תהילים פרק-נט-י]  עֻזּוֹ אֵלֶיךָ אֶשְׁמֹרָה כִּי אֱלֹהִים מִשְׂגַּבִּי:

[תהילים פרק-נט-יח]  עֻזִּי אֵלֶיךָ אֲזַמֵּרָה כִּי אֱלֹהִים מִשְׂגַּבִּי אֱלֹהֵי חַסְדִּי:

[תהילים פרק-סט-יז]  עֲנֵנִי יְהוָה כִּי טוֹב חַסְדֶּךָ כְּרֹב רַחֲמֶיךָ פְּנֵה אֵלָי:

[תהילים פרק-פו-יז]  עֲשֵׂה עִמִּי אוֹת לְטוֹבָה וְיִרְאוּ שֹׂנְאַי וְיֵבֹשׁוּ כִּי אַתָּה יְהוָה עֲזַרְתַּנִי וְנִחַמְתָּנִי:

[תהילים פרק-פח-י]  עֵינִי דָאֲבָה מִנִּי עֹנִי קְרָאתִיךָ יְהוָה בְּכָל יוֹם שִׁטַּחְתִּי אֵלֶיךָ כַפָּי:

[תהילים פרק-פח-יז]  עָלַי עָבְרוּ חֲרוֹנֶיךָ בִּעוּתֶיךָ צִמְּתוּתֻנִי:

[תהילים פרק-קא-ו]  עֵינַי בְּנֶאֶמְנֵי אֶרֶץ לָשֶׁבֶת עִמָּדִי הֹלֵךְ בְּדֶרֶךְ תָּמִים הוּא יְשָׁרְתֵנִי:

[תהילים פרק-קב-כד]  עִנָּה בַדֶּרֶךְ (כחו) כֹּחִי קִצַּר יָמָי:

[תהילים פרק-קיט-קכא]  עָשִׂיתִי מִשְׁפָּט וָצֶדֶק בַּל תַּנִּיחֵנִי לְעֹשְׁקָי:

[תהילים פרק-קיט-קכד]  עֲשֵׂה עִם עַבְדְּךָ כְחַסְדֶּךָ וְחֻקֶּיךָ לַמְּדֵנִי:

[תהילים פרק-קיט-קכח]  עַל כֵּן כָּל פִּקּוּדֵי כֹל יִשָּׁרְתִּי כָּל אֹרַח שֶׁקֶר שָׂנֵאתִי:

[משלי פרק-ו-יז]  עֵינַיִם רָמוֹת לְשׁוֹן שָׁקֶר וְיָדַיִם שֹׁפְכוֹת דָּם נָקִי:

[איוב פרק-י-יא]  עוֹר וּבָשָׂר תַּלְבִּישֵׁנִי וּבַעֲצָמוֹת וְגִידִים תְּסֹכְכֵנִי:

[איוב פרק-יג-יד]  עַל מָה אֶשָּׂא בְשָׂרִי בְשִׁנָּי וְנַפְשִׁי אָשִׂים בְּכַפִּי:

[איוב פרק-כט-טו]  עֵינַיִם הָיִיתִי לַעִוֵּר וְרַגְלַיִם לַפִּסֵּחַ אָנִי:

[איכה פרק-ג-נא]  עֵינִי עוֹלְלָה לְנַפְשִׁי מִכֹּל בְּנוֹת עִירִי:

[קהלת פרק-ב-ה]  עָשִׂיתִי לִי גַּנּוֹת וּפַרְדֵּסִים וְנָטַעְתִּי בָהֶם עֵץ כָּל פֶּרִי:

[דניאל פרק-ב-י]  עֲנוֹ (כשדיא) כַשְׂדָּאֵי קֳדָם מַלְכָּא וְאָמְרִין לָא אִיתַי אֲנָשׁ עַל יַבֶּשְׁתָּא דִּי מִלַּת מַלְכָּא יוּכַל לְהַחֲוָיָה כָּל קֳבֵל דִּי כָּל מֶלֶךְ רַב וְשַׁלִּיט מִלָּה כִדְנָה לָא שְׁאֵל לְכָל חַרְטֹּם וְאָשַׁף וְכַשְׂדָּי:

[דניאל פרק-ג-ט]  עֲנוֹ וְאָמְרִין לִנְבוּכַדְנֶצַּר מַלְכָּא מַלְכָּא לְעָלְמִין חֱיִי:

[דניאל פרק-ד-כז]  עָנֵה מַלְכָּא וְאָמַר הֲלָא דָא הִיא בָּבֶל רַבְּתָא דִּי אֲנָה בֱנַיְתַהּ לְבֵית מַלְכוּ בִּתְקַף חִסְנִי וְלִיקָר הַדְרִי:

[דברי הימים א פרק-יא-כח]  עִירָא בֶן עִקֵּשׁ הַתְּקוֹעִי אֲבִיעֶזֶר הָעֲנְּתוֹתִי:

[דברי הימים א פרק-יא-לג]  עַזְמָוֶת הַבַּחֲרוּמִי אֶלְיַחְבָּא הַשַּׁעַלְבֹנִי:

[דברי הימים א פרק-יא-מ]  עִירָא הַיִּתְרִי גָּרֵב הַיִּתְרִי:

[דברי הימים א פרק-יא-מד]  עֻזִיָּא הָעֲשְׁתְּרָתִי שָׁמָע (ויעואל) וִיעִיאֵל בְּנֵי חוֹתָם הָעֲרֹעֵרִי:

[דברי הימים א פרק-יב-י]  עֵזֶר הָרֹאשׁ עֹבַדְיָה הַשֵּׁנִי אֱלִיאָב הַשְּׁלִשִׁי:

[דברי הימים א פרק-יב-יב]  עַתַּי הַשִּׁשִּׁי אֱלִיאֵל הַשְּׁבִעִי:

[דברי הימים א פרק-כו-ג]  עֵילָם הַחֲמִישִׁי יְהוֹחָנָן הַשִּׁשִּׁי אֶלְיְהוֹעֵינַי הַשְּׁבִיעִי:


ע-כ


[דברים פרק-ד-ג]  עֵינֵיכֶם הָרֹאֹת אֵת אֲשֶׁר עָשָׂה יְהוָה בְּבַעַל פְּעוֹר כִּי כָל הָאִישׁ אֲשֶׁר הָלַךְ אַחֲרֵי בַעַל פְּעוֹר הִשְׁמִידוֹ יְהוָה אֱלֹהֶיךָ מִקִּרְבֶּךָ:

[דברים פרק-יט-ז]  עַל כֵּן אָנֹכִי מְצַוְּךָ לֵאמֹר שָׁלֹשׁ עָרִים תַּבְדִּיל לָךְ:

[דברים פרק-כג-ה]  עַל דְּבַר אֲשֶׁר לֹא קִדְּמוּ אֶתְכֶם בַּלֶּחֶם וּבַמַּיִם בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם וַאֲשֶׁר שָׂכַר עָלֶיךָ אֶת בִּלְעָם בֶּן בְּעוֹר מִפְּתוֹר אֲרַם נַהֲרַיִם לְקַלְלֶךָּ:

[ישעיה פרק-כה-ג]  עַל כֵּן יְכַבְּדוּךָ עַם עָז קִרְיַת גּוֹיִם עָרִיצִים יִירָאוּךָ:

[ישעיה פרק-נט-ט]  עַל כֵּן רָחַק מִשְׁפָּט מִמֶּנּוּ וְלֹא תַשִּׂיגֵנוּ צְדָקָה נְקַוֶּה לָאוֹר וְהִנֵּה חֹשֶׁךְ לִנְגֹהוֹת בָּאֲפֵלוֹת נְהַלֵּךְ:

[ירמיה פרק-כב-כ]  עֲלִי הַלְּבָנוֹן וּצְעָקִי וּבַבָּשָׁן תְּנִי קוֹלֵךְ וְצַעֲקִי מֵעֲבָרִים כִּי נִשְׁבְּרוּ כָּל מְאַהֲבָיִךְ:

[ירמיה פרק-מו-יא]  עֲלִי גִלְעָד וּקְחִי צֳרִי בְּתוּלַת בַּת מִצְרָיִם לַשָּׁוְא (הרביתי) הִרְבֵּית רְפֻאוֹת תְּעָלָה אֵין לָךְ:

[ירמיה פרק-נ-כא]  עַל הָאָרֶץ מְרָתַיִם עֲלֵה עָלֶיהָ וְאֶל יוֹשְׁבֵי פְּקוֹד חֲרֹב וְהַחֲרֵם אַחֲרֵיהֶם נְאֻם יְהוָה וַעֲשֵׂה כְּכֹל אֲשֶׁר צִוִּיתִיךָ:

[יחזקאל פרק-ז-ג]  עַתָּה הַקֵּץ עָלַיִךְ וְשִׁלַּחְתִּי אַפִּי בָּךְ וּשְׁפַטְתִּיךְ כִּדְרָכָיִךְ וְנָתַתִּי עָלַיִךְ אֵת כָּל תּוֹעֲבֹתָיִךְ:

[יחזקאל פרק-ז-ח]  עַתָּה מִקָּרוֹב אֶשְׁפּוֹךְ חֲמָתִי עָלַיִךְ וְכִלֵּיתִי אַפִּי בָּךְ וּשְׁפַטְתִּיךְ כִּדְרָכָיִךְ וְנָתַתִּי עָלַיִךְ אֵת כָּל תּוֹעֲבוֹתָיִךְ:

[יחזקאל פרק-כב-י]  עֶרְוַת אָב גִּלָּה בָךְ טְמֵאַת הַנִּדָּה עִנּוּ בָךְ:

[יחזקאל פרק-כו-יח]  עַתָּה יֶחְרְדוּ הָאִיִּן יוֹם מַפַּלְתֵּךְ וְנִבְהֲלוּ הָאִיִּים אֲשֶׁר בַּיָּם מִצֵּאתֵךְ:

[יחזקאל פרק-כז-כא]  עֲרַב וְכָל נְשִׂיאֵי קֵדָר הֵמָּה סֹחֲרֵי יָדֵךְ בְּכָרִים וְאֵילִים וְעַתּוּדִים בָּם סֹחֲרָיִךְ:

[יואל פרק-ד-יא]  עוּשׁוּ וָבֹאוּ כָל הַגּוֹיִם מִסָּבִיב וְנִקְבָּצוּ שָׁמָּה הַנְחַת יְהוָה גִּבּוֹרֶיךָ:

[תהילים פרק-נו-יג]  עָלַי אֱלֹהִים נְדָרֶיךָ אֲשַׁלֵּם תּוֹדֹת לָךְ:

[תהילים פרק-עט-ה]  עַד מָה יְהוָה תֶּאֱנַף לָנֶצַח תִּבְעַר כְּמוֹ אֵשׁ קִנְאָתֶךָ:

[תהילים פרק-עט-ט]  עָזְרֵנוּ אֱלֹהֵי יִשְׁעֵנוּ עַל דְּבַר כְּבוֹד שְׁמֶךָ וְהַצִּילֵנוּ וְכַפֵּר עַל חַטֹּאתֵינוּ לְמַעַן שְׁמֶךָ:

[תהילים פרק-פג-ד]  עַל עַמְּךָ יַעֲרִימוּ סוֹד וְיִתְיָעֲצוּ עַל צְפוּנֶיךָ:

[תהילים פרק-פט-מז]  עַד מָה יְהוָה תִּסָּתֵר לָנֶצַח תִּבְעַר כְּמוֹ אֵשׁ חֲמָתֶךָ:

[תהילים פרק-צא-יב]  עַל כַּפַּיִם יִשָּׂאוּנְךָ פֶּן תִּגֹּף בָּאֶבֶן רַגְלֶךָ:

[תהילים פרק-קט-כו]  עָזְרֵנִי יְהוָה אֱלֹהָי הוֹשִׁיעֵנִי כְחַסְדֶּךָ:

[תהילים פרק-קי-ג]  עַמְּךָ נְדָבֹת בְּיוֹם חֵילֶךָ בְּהַדְרֵי קֹדֶשׁ מֵרֶחֶם מִשְׁחָר לְךָ טַל יַלְדֻתֶיךָ:

[תהילים פרק-קיט-קכג]  עֵינַי כָּלוּ לִישׁוּעָתֶךָ וּלְאִמְרַת צִדְקֶךָ:

[תהילים פרק-קיט-קכה]  עַבְדְּךָ אָנִי הֲבִינֵנִי וְאֵדְעָה עֵדֹתֶיךָ:

[תהילים פרק-קיט-קכו]  עֵת לַעֲשׂוֹת לַיהוָה הֵפֵרוּ תּוֹרָתֶךָ:

[משלי פרק-ד-כה]  עֵינֶיךָ לְנֹכַח יַבִּיטוּ וְעַפְעַפֶּיךָ יַיְשִׁרוּ נֶגְדֶּךָ:

[משלי פרק-ה-כב]  עַווֹנוֹתָיו יִלְכְּדֻנוֹ אֶת הָרָשָׁע וּבְחַבְלֵי חַטָּאתוֹ יִתָּמֵךְ:

[משלי פרק-ו-ג]  עֲשֵׂה זֹאת אֵפוֹא בְּנִי וְהִנָּצֵל כִּי בָאתָ בְכַף רֵעֶךָ לֵךְ הִתְרַפֵּס וּרְהַב רֵעֶיךָ:

[משלי פרק-ו-ט]  עַד מָתַי עָצֵל תִּשְׁכָּב מָתַי תָּקוּם מִשְּׁנָתֶךָ:

[משלי פרק-ז-טו]  עַל כֵּן יָצָאתִי לִקְרָאתֶךָ לְשַׁחֵר פָּנֶיךָ וָאֶמְצָאֶךָּ:

[איוב פרק-א-טז]  עוֹד זֶה מְדַבֵּר וְזֶה בָּא וַיֹּאמַר אֵשׁ אֱלֹהִים נָפְלָה מִן הַשָּׁמַיִם וַתִּבְעַר בַּצֹּאן וּבַנְּעָרִים וַתֹּאכְלֵם וָאִמָּלְטָה רַק אֲנִי לְבַדִּי לְהַגִּיד לָךְ:

[איוב פרק-א-יז]  עוֹד זֶה מְדַבֵּר וְזֶה בָּא וַיֹּאמַר כַּשְׂדִּים שָׂמוּ שְׁלֹשָׁה רָאשִׁים וַיִּפְשְׁטוּ עַל הַגְּמַלִּים וַיִּקָּחוּם וְאֶת הַנְּעָרִים הִכּוּ לְפִי חָרֶב וָאִמָּלְטָה רַק אֲנִי לְבַדִּי לְהַגִּיד לָךְ:

[איוב פרק-ח-ב]  עַד אָן תְּמַלֶּל אֵלֶּה וְרוּחַ כַּבִּיר אִמְרֵי פִיךָ:

[איוב פרק-כב-יד]  עָבִים סֵתֶר לוֹ וְלֹא יִרְאֶה וְחוּג שָׁמַיִם יִתְהַלָּךְ:

[שיר השירים פרק-ב-י]  עָנָה דוֹדִי וְאָמַר לִי קוּמִי לָךְ רַעְיָתִי יָפָתִי וּלְכִי לָךְ:

[איכה פרק-ב-יז]  עָשָׂה יְהוָה אֲשֶׁר זָמָם בִּצַּע אֶמְרָתוֹ אֲשֶׁר צִוָּה מִימֵי קֶדֶם הָרַס וְלֹא חָמָל וַיְשַׂמַּח עָלַיִךְ אוֹיֵב הֵרִים קֶרֶן צָרָיִךְ:

[קהלת פרק-ג-ו]  עֵת לְבַקֵּשׁ וְעֵת לְאַבֵּד עֵת לִשְׁמוֹר וְעֵת לְהַשְׁלִיךְ:

[דניאל פרק-ג-טז]  עֲנוֹ שַׁדְרַךְ מֵישַׁךְ וַעֲבֵד נְגוֹ וְאָמְרִין לְמַלְכָּא נְבוּכַדְנֶצַּר לָא חַשְׁחִין אֲנַחְנָה עַל דְּנָה פִּתְגָם לַהֲתָבוּתָךְ:

[דניאל פרק-ד-כח]  עוֹד מִלְּתָא בְּפֻם מַלְכָּא קָל מִן שְׁמַיָּא נְפַל לָךְ אָמְרִין נְבוּכַדְנֶצַּר מַלְכָּא מַלְכוּתָה עֲדָת מִנָּךְ:

[דברי הימים א פרק-כב-יא]  עַתָּה בְנִי יְהִי יְהוָה עִמָּךְ וְהִצְלַחְתָּ וּבָנִיתָ בֵּית יְהוָה אֱלֹהֶיךָ כַּאֲשֶׁר דִּבֶּר עָלֶיךָ:


ע-ל


[שמות פרק-כא-כד]  עַיִן תַּחַת עַיִן שֵׁן תַּחַת שֵׁן יָד תַּחַת יָד רֶגֶל תַּחַת רָגֶל:

[דברים פרק-יז-יא]  עַל פִּי הַתּוֹרָה אֲשֶׁר יוֹרוּךָ וְעַל הַמִּשְׁפָּט אֲשֶׁר יֹאמְרוּ לְךָ תַּעֲשֶׂה לֹא תָסוּר מִן הַדָּבָר אֲשֶׁר יַגִּידוּ לְךָ יָמִין וּשְׂמֹאל:

[דברים פרק-לב-נא]  עַל אֲשֶׁר מְעַלְתֶּם בִּי בְּתוֹךְ בְּנֵי יִשְׂרָאֵל בְּמֵי מְרִיבַת קָדֵשׁ מִדְבַּר צִן עַל אֲשֶׁר לֹא קִדַּשְׁתֶּם אוֹתִי בְּתוֹךְ בְּנֵי יִשְׂרָאֵל:

[דברים פרק-לג-יט]  עַמִּים הַר יִקְרָאוּ שָׁם יִזְבְּחוּ זִבְחֵי צֶדֶק כִּי שֶׁפַע יַמִּים יִינָקוּ וּשְׂפוּנֵי טְמוּנֵי חוֹל:

[בראשית פרק-כא-לא]  עַל כֵּן קָרָא לַמָּקוֹם הַהוּא בְּאֵר שָׁבַע כִּי שָׁם נִשְׁבְּעוּ שְׁנֵיהֶם:

[בראשית פרק-לא-לב]  עִם אֲשֶׁר תִּמְצָא אֶת אֱלֹהֶיךָ לֹא יִחְיֶה נֶגֶד אַחֵינוּ הַכֶּר לְךָ מָה עִמָּדִי וְקַח לָךְ וְלֹא יָדַע יַעֲקֹב כִּי רָחֵל גְּנָבָתַם:

[בראשית פרק-לב-טו]  עִזִּים מָאתַיִם וּתְיָשִׁים עֶשְׂרִים רְחֵלִים מָאתַיִם וְאֵילִים עֶשְׂרִים:

[שמות פרק-ט-יז]  עוֹדְךָ מִסְתּוֹלֵל בְּעַמִּי לְבִלְתִּי שַׁלְּחָם:

[שמות פרק-יח-יא]  עַתָּה יָדַעְתִּי כִּי גָדוֹל יְהוָה מִכָּל הָאֱלֹהִים כִּי בַדָּבָר אֲשֶׁר זָדוּ עֲלֵיהֶם:

[שמות פרק-יח-יט]  עַתָּה שְׁמַע בְּקֹלִי אִיעָצְךָ וִיהִי אֱלֹהִים עִמָּךְ הֱיֵה אַתָּה לָעָם מוּל הָאֱלֹהִים וְהֵבֵאתָ אַתָּה אֶת הַדְּבָרִים אֶל הָאֱלֹהִים:

[שמות פרק-כט-מב]  עֹלַת תָּמִיד לְדֹרֹתֵיכֶם פֶּתַח אֹהֶל מוֹעֵד לִפְנֵי יְהוָה אֲשֶׁר אִוָּעֵד לָכֶם שָׁמָּה לְדַבֵּר אֵלֶיךָ שָׁם:

[שמות פרק-ל-לב]  עַל בְּשַׂר אָדָם לֹא יִיסָךְ וּבְמַתְכֻּנְתּוֹ לֹא תַעֲשׂוּ כָּמֹהוּ קֹדֶשׁ הוּא קֹדֶשׁ יִהְיֶה לָכֶם:

[ויקרא פרק-יז-יב]  עַל כֵּן אָמַרְתִּי לִבְנֵי יִשְׂרָאֵל כָּל נֶפֶשׁ מִכֶּם לֹא תֹאכַל דָּם וְהַגֵּר הַגָּר בְּתוֹכְכֶם לֹא יֹאכַל דָּם:

[במדבר פרק-ד-כז]  עַל פִּי אַהֲרֹן וּבָנָיו תִּהְיֶה כָּל עֲבֹדַת בְּנֵי הַגֵּרְשֻׁנִּי לְכָל מַשָּׂאָם וּלְכֹל עֲבֹדָתָם וּפְקַדְתֶּם עֲלֵהֶם בְּמִשְׁמֶרֶת אֵת כָּל מַשָּׂאָם:

[במדבר פרק-כח-כא]  עִשָּׂרוֹן עִשָּׂרוֹן תַּעֲשֶׂה לַכֶּבֶשׂ הָאֶחָד לְשִׁבְעַת הַכְּבָשִׂים:

[במדבר פרק-כח-כט]  עִשָּׂרוֹן עִשָּׂרוֹן לַכֶּבֶשׂ הָאֶחָד לְשִׁבְעַת הַכְּבָשִׂים:

[במדבר פרק-כט-י]  עִשָּׂרוֹן עִשָּׂרוֹן לַכֶּבֶשׂ הָאֶחָד לְשִׁבְעַת הַכְּבָשִׂים:

[דברים פרק-ב-כא]  עַם גָּדוֹל וְרַב וָרָם כָּעֲנָקִים וַיַּשְׁמִידֵם יְהוָה מִפְּנֵיהֶם וַיִּירָשֻׁם וַיֵּשְׁבוּ תַחְתָּם:

[דברים פרק-ג-כ]  עַד אֲשֶׁר יָנִיחַ יְהוָה לַאֲחֵיכֶם כָּכֶם וְיָרְשׁוּ גַם הֵם אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֵיכֶם נֹתֵן לָהֶם בְּעֵבֶר הַיַּרְדֵּן וְשַׁבְתֶּם אִישׁ לִירֻשָּׁתוֹ אֲשֶׁר נָתַתִּי לָכֶם:

[שופטים פרק-ה-יב]  עוּרִי עוּרִי דְּבוֹרָה עוּרִי עוּרִי דַּבְּרִי שִׁיר קוּם בָּרָק וּשֲׁבֵה שֶׁבְיְךָ בֶּן אֲבִינֹעַם:

[שמואל ב פרק-כב-כו]  עִם חָסִיד תִּתְחַסָּד עִם גִּבּוֹר תָּמִים תִּתַּמָּם:

[שמואל ב פרק-כג-כד]  עֲשָׂה אֵל אֲחִי יוֹאָב בַּשְּׁלֹשִׁים אֶלְחָנָן בֶּן דֹּדוֹ בֵּית לָחֶם:

[מלכים א פרק-ה-ג]  עֲשָׂרָה בָקָר בְּרִאִים וְעֶשְׂרִים בָּקָר רְעִי וּמֵאָה צֹאן לְבַד מֵאַיָּל וּצְבִי וְיַחְמוּר וּבַרְבֻּרִים אֲבוּסִים:

[מלכים א פרק-ז-מא]  עַמֻּדִים שְׁנַיִם וְגֻלֹּת הַכֹּתָרֹת אֲשֶׁר עַל רֹאשׁ הָעַמֻּדִים שְׁתָּיִם וְהַשְּׂבָכוֹת שְׁתַּיִם לְכַסּוֹת אֶת שְׁתֵּי גֻּלֹּת הַכֹּתָרֹת אֲשֶׁר עַל רֹאשׁ הָעַמּוּדִים:

[מלכים ב פרק-ד-כו]  עַתָּה רוּץ נָא לִקְרָאתָהּ וֶאֱמָר לָהּ הֲשָׁלוֹם לָךְ הֲשָׁלוֹם לְאִישֵׁךְ הֲשָׁלוֹם לַיָּלֶד וַתֹּאמֶר שָׁלוֹם:

[מלכים ב פרק-כב-ד]  עֲלֵה אֶל חִלְקִיָּהוּ הַכֹּהֵן הַגָּדוֹל וְיַתֵּם אֶת הַכֶּסֶף הַמּוּבָא בֵּית יְהוָה אֲשֶׁר אָסְפוּ שֹׁמְרֵי הַסַּף מֵאֵת הָעָם:

[ישעיה פרק-י-לב]  עוֹד הַיּוֹם בְּנֹב לַעֲמֹד יְנֹפֵף יָדוֹ הַר (בית) בַּת צִיּוֹן גִּבְעַת יְרוּשָׁלִָם:

[ישעיה פרק-טו-ז]  עַל כֵּן יִתְרָה עָשָׂה וּפְקֻדָּתָם עַל נַחַל הָעֲרָבִים יִשָּׂאוּם:

[ישעיה פרק-ל-ח]  עַתָּה בּוֹא כָתְבָהּ עַל לוּחַ אִתָּם וְעַל סֵפֶר חֻקָּהּ וּתְהִי לְיוֹם אַחֲרוֹן לָעַד עַד עוֹלָם:

[ישעיה פרק-לו-יז]  עַד בֹּאִי וְלָקַחְתִּי אֶתְכֶם אֶל אֶרֶץ כְּאַרְצְכֶם אֶרֶץ דָּגָן וְתִירוֹשׁ אֶרֶץ לֶחֶם וּכְרָמִים:

[ישעיה פרק-מ-ט]  עַל הַר גָּבֹהַ עֲלִי לָךְ מְבַשֶּׂרֶת צִיּוֹן הָרִימִי בַכֹּחַ קוֹלֵךְ מְבַשֶּׂרֶת יְרוּשָׁלִָם הָרִימִי אַל תִּירָאִי אִמְרִי לְעָרֵי יְהוּדָה הִנֵּה אֱלֹהֵיכֶם:

[ישעיה פרק-נד-יא]  עֲנִיָּה סֹעֲרָה לֹא נֻחָמָה הִנֵּה אָנֹכִי מַרְבִּיץ בַּפּוּךְ אֲבָנַיִךְ וִיסַדְתִּיךְ בַּסַּפִּירִים:

[ישעיה פרק-סב-ו]  עַל חוֹמֹתַיִךְ יְרוּשָׁלִַם הִפְקַדְתִּי שֹׁמְרִים כָּל הַיּוֹם וְכָל הַלַּיְלָה תָּמִיד לֹא יֶחֱשׁוּ הַמַּזְכִּרִים אֶת יְהוָה אַל דֳּמִי לָכֶם:

[ישעיה פרק-סב-י]  עִבְרוּ עִבְרוּ בַּשְּׁעָרִים פַּנּוּ דֶּרֶךְ הָעָם סֹלּוּ סֹלּוּ הַמְסִלָּה סַקְּלוּ מֵאֶבֶן הָרִימוּ נֵס עַל הָעַמִּים:

[ישעיה פרק-סה-ז]  עֲוֹנֹתֵיכֶם וַעֲוֹנֹת אֲבוֹתֵיכֶם יַחְדָּו אָמַר יְהוָה אֲשֶׁר קִטְּרוּ עַל הֶהָרִים וְעַל הַגְּבָעוֹת חֵרְפוּנִי וּמַדֹּתִי פְעֻלָּתָם רִאשֹׁנָה (על) אֶל חֵיקָם:

[ירמיה פרק-ה-ו]  עַל כֵּן הִכָּם אַרְיֵה מִיַּעַר זְאֵב עֲרָבוֹת יְשָׁדְדֵם נָמֵר שֹׁקֵד עַל עָרֵיהֶם כָּל הַיּוֹצֵא מֵהֵנָּה יִטָּרֵף כִּי רַבּוּ פִּשְׁעֵיהֶם עָצְמוּ (משבותיהם) מְשׁוּבוֹתֵיהֶם:

[ירמיה פרק-ה-כה]  עֲוֹנוֹתֵיכֶם הִטּוּ אֵלֶּה וְחַטֹּאותֵיכֶם מָנְעוּ הַטּוֹב מִכֶּם:

[ירמיה פרק-י-יב]  עֹשֵׂה אֶרֶץ בְּכֹחוֹ מֵכִין תֵּבֵל בְּחָכְמָתוֹ וּבִתְבוּנָתוֹ נָטָה שָׁמָיִם:

[ירמיה פרק-יג-יט]  עָרֵי הַנֶּגֶב סֻגְּרוּ וְאֵין פֹּתֵחַ הָגְלָת יְהוּדָה כֻּלָּהּ הָגְלָת שְׁלוֹמִים:

[ירמיה פרק-כג-כו]  עַד מָתַי הֲיֵשׁ בְּלֵב הַנְּבִאִים נִבְּאֵי הַשָּׁקֶר וּנְבִיאֵי תַּרְמִת לִבָּם:

[ירמיה פרק-לא-ד]  עוֹד אֶבְנֵךְ וְנִבְנֵית בְּתוּלַת יִשְׂרָאֵל עוֹד תַּעְדִּי תֻפַּיִךְ וְיָצָאת בִּמְחוֹל מְשַׂחֲקִים:

[ירמיה פרק-לב-לב]  עַל כָּל רָעַת בְּנֵי יִשְׂרָאֵל וּבְנֵי יְהוּדָה אֲשֶׁר עָשׂוּ לְהַכְעִסֵנִי הֵמָּה מַלְכֵיהֶם שָׂרֵיהֶם כֹּהֲנֵיהֶם וּנְבִיאֵיהֶם וְאִישׁ יְהוּדָה וְיֹשְׁבֵי יְרוּשָׁלִָם:

[ירמיה פרק-נא-טו]  עֹשֵׂה אֶרֶץ בְּכֹחוֹ מֵכִין תֵּבֵל בְּחָכְמָתוֹ וּבִתְבוּנָתוֹ נָטָה שָׁמָיִם:

[יחזקאל פרק-א-כ]  עַל אֲשֶׁר יִהְיֶה שָּׁם הָרוּחַ לָלֶכֶת יֵלֵכוּ שָׁמָּה הָרוּחַ לָלֶכֶת וְהָאוֹפַנִּים יִנָּשְׂאוּ לְעֻמָּתָם כִּי רוּחַ הַחַיָּה בָּאוֹפַנִּים:

[יחזקאל פרק-כג-ל]  עָשֹׂה אֵלֶּה לָךְ בִּזְנוֹתֵךְ אַחֲרֵי גוֹיִם עַל אֲשֶׁר נִטְמֵאת בְּגִלּוּלֵיהֶם:

[יחזקאל פרק-מג-ט]  עַתָּה יְרַחֲקוּ אֶת זְנוּתָם וּפִגְרֵי מַלְכֵיהֶם מִמֶּנִּי וְשָׁכַנְתִּי בְתוֹכָם לְעוֹלָם:

[הושע פרק-ד-יב]  עַמִּי בְּעֵצוֹ יִשְׁאָל וּמַקְלוֹ יַגִּיד לוֹ כִּי רוּחַ זְנוּנִים הִתְעָה וַיִּזְנוּ מִתַּחַת אֱלֹהֵיהֶם:

[עמוס פרק-ו-ב]  עִבְרוּ כַלְנֵה וּרְאוּ וּלְכוּ מִשָּׁם חֲמַת רַבָּה וּרְדוּ גַת פְּלִשְׁתּים הֲטוֹבִים מִן הַמַּמְלָכוֹת הָאֵלֶּה אִם רַב גְּבוּלָם מִגְּבֻלְכֶם:

[מיכה פרק-ב-יג]  עָלָה הַפֹּרֵץ לִפְנֵיהֶם פָּרְצוּ וַיַּעֲבֹרוּ שַׁעַר וַיֵּצְאוּ בוֹ וַיַּעֲבֹר מַלְכָּם לִפְנֵיהֶם וַיהוָה בְּרֹאשָׁם:

[זכריה פרק-א-יז]  עוֹד קְרָא לֵאמֹר כֹּה אָמַר יְהוָה צְבָאוֹת עוֹד תְּפוּצֶינָה עָרַי מִטּוֹב וְנִחַם יְהוָה עוֹד אֶת צִיּוֹן וּבָחַר עוֹד בִּירוּשָׁלִָם:

[תהילים פרק-א-ה]  עַל כֵּן לֹא יָקֻמוּ רְשָׁעִים בַּמִּשְׁפָּט וְחַטָּאִים בַּעֲדַת צַדִּיקִים:

[תהילים פרק-יח-כו]  עִם חָסִיד תִּתְחַסָּד עִם גְּבַר תָּמִים תִּתַּמָּם:

[תהילים פרק-לד-טז]  עֵינֵי יְהוָה אֶל צַדִּיקִים וְאָזְנָיו אֶל שַׁוְעָתָם:

[תהילים פרק-מו-ג]  עַל כֵּן לֹא נִירָא בְּהָמִיר אָרֶץ וּבְמוֹט הָרִים בְּלֵב יַמִּים:

[תהילים פרק-נו-ח]  עַל אָוֶן פַּלֶּט לָמוֹ בְּאַף עַמִּים הוֹרֵד אֱלֹהִים:

[תהילים פרק-סב-ח]  עַל אֱלֹהִים יִשְׁעִי וּכְבוֹדִי צוּר עֻזִּי מַחְסִי בֵּאלֹהִים:

[תהילים פרק-סח-יט]  עָלִיתָ לַמָּרוֹם שָׁבִיתָ שֶּׁבִי לָקַחְתָּ מַתָּנוֹת בָּאָדָם וְאַף סוֹרְרִים לִשְׁכֹּן יָהּ אֱלֹהִים:

[תהילים פרק-עג-יז]  עַד אָבוֹא אֶל מִקְדְּשֵׁי אֵל אָבִינָה לְאַחֲרִיתָם:

[תהילים פרק-צג-ה]  עֵדֹתֶיךָ נֶאֶמְנוּ מְאֹד לְבֵיתְךָ נַאֲוָה קֹדֶשׁ יְהוָה לְאֹרֶךְ יָמִים:

[תהילים פרק-קג-ו]  עֹשֵׂה צְדָקוֹת יְהוָה וּמִשְׁפָּטִים לְכָל עֲשׁוּקִים:

[תהילים פרק-קטו-ד]  עֲצַבֵּיהֶם כֶּסֶף וְזָהָב מַעֲשֵׂה יְדֵי אָדָם:

[תהילים פרק-קיט-קכב]  עֲרֹב עַבְדְּךָ לְטוֹב אַל יַעַשְׁקֻנִי זֵדִים:

[תהילים פרק-קכב-ב]  עֹמְדוֹת הָיוּ רַגְלֵינוּ בִּשְׁעָרַיִךְ יְרוּשָׁלִָם:

[תהילים פרק-קכט-ג]  עַל גַּבִּי חָרְשׁוּ חֹרְשִׁים הֶאֱרִיכוּ (למענותם) לְמַעֲנִיתָם:

[תהילים פרק-קלה-טו]  עֲצַבֵּי הַגּוֹיִם כֶּסֶף וְזָהָב מַעֲשֵׂה יְדֵי אָדָם:

[תהילים פרק-קמו-ו]  עֹשֶׂה שָׁמַיִם וָאָרֶץ אֶת הַיָּם וְאֶת כָּל אֲשֶׁר בָּם הַשֹּׁמֵר אֱמֶת לְעוֹלָם:

[תהילים פרק-קמו-ז]  עֹשֶׂה מִשְׁפָּט לָעֲשׁוּקִים נֹתֵן לֶחֶם לָרְעֵבִים יְהוָה מַתִּיר אֲסוּרִים:

[משלי פרק-יז-ו]  עֲטֶרֶת זְקֵנִים בְּנֵי בָנִים וְתִפְאֶרֶת בָּנִים אֲבוֹתָם:

[משלי פרק-כב-ד]  עֵקֶב עֲנָוָה יִרְאַת יְהוָה עֹשֶׁר וְכָבוֹד וְחַיִּים:

[משלי פרק-כח-ד]  עֹזְבֵי תוֹרָה יְהַלְלוּ רָשָׁע וְשֹׁמְרֵי תוֹרָה יִתְגָּרוּ בָם:

[איוב פרק-יט-ב]  עַד אָנָה תּוֹגְיוּן נַפְשִׁי וּתְדַכְּאוּנַנִי בְמִלִּים:

[איוב פרק-כב-י]  עַל כֵּן סְבִיבוֹתֶיךָ פַחִים וִיבַהֶלְךָ פַּחַד פִּתְאֹם:

[איוב פרק-ל-יב]  עַל יָמִין פִּרְחַח יָקוּמוּ רַגְלַי שִׁלֵּחוּ וַיָּסֹלּוּ עָלַי אָרְחוֹת אֵידָם:

[שיר השירים פרק-ג-י]  עַמּוּדָיו עָשָׂה כֶסֶף רְפִידָתוֹ זָהָב מֶרְכָּבוֹ אַרְגָּמָן תּוֹכוֹ רָצוּף אַהֲבָה מִבְּנוֹת יְרוּשָׁלִָם:

[רות פרק-ב-ט]  עֵינַיִךְ בַּשָּׂדֶה אֲשֶׁר יִקְצֹרוּן וְהָלַכְתְּ אַחֲרֵיהֶן הֲלוֹא צִוִּיתִי אֶת הַנְּעָרִים לְבִלְתִּי נָגְעֵךְ וְצָמִת וְהָלַכְתְּ אֶל הַכֵּלִים וְשָׁתִית מֵאֲשֶׁר יִשְׁאֲבוּן הַנְּעָרִים:

[איכה פרק-ג-נ]  עַד יַשְׁקִיף וְיֵרֶא יְהוָה מִשָּׁמָיִם:

[איכה פרק-ה-ח]  עֲבָדִים מָשְׁלוּ בָנוּ פֹּרֵק אֵין מִיָּדָם:

[קהלת פרק-ב-ו]  עָשִׂיתִי לִי בְּרֵכוֹת מָיִם לְהַשְׁקוֹת מֵהֶם יַעַר צוֹמֵחַ עֵצִים:

[קהלת פרק-ג-ח]  עֵת לֶאֱהֹב וְעֵת לִשְׂנֹא עֵת מִלְחָמָה וְעֵת שָׁלוֹם:

[קהלת פרק-ט-יד]  עִיר קְטַנָּה וַאֲנָשִׁים בָּהּ מְעָט וּבָא אֵלֶיהָ מֶלֶךְ גָּדוֹל וְסָבַב אֹתָהּ וּבָנָה עָלֶיהָ מְצוֹדִים גְּדֹלִים:

[קהלת פרק-יב-ב]  עַד אֲשֶׁר לֹא תֶחְשַׁךְ הַשֶּׁמֶשׁ וְהָאוֹר וְהַיָּרֵחַ וְהַכּוֹכָבִים וְשָׁבוּ הֶעָבִים אַחַר הַגָּשֶׁם:

[אסתר פרק-ט-י]  עֲשֶׂרֶת בְּנֵי הָמָן בֶּן הַמְּדָתָא צֹרֵר הַיְּהוּדִים הָרָגוּ וּבַבִּזָּה לֹא שָׁלְחוּ אֶת יָדָם:

[אסתר פרק-ט-כו]  עַל כֵּן קָרְאוּ לַיָּמִים הָאֵלֶּה פוּרִים עַל שֵׁם הַפּוּר עַל כֵּן עַל כָּל דִּבְרֵי הָאִגֶּרֶת הַזֹּאת וּמָה רָאוּ עַל כָּכָה וּמָה הִגִּיעַ אֲלֵיהֶם:

[דברי הימים א פרק-יא-מב]  עֲדִינָא בֶן שִׁיזָא הָראוּבֵנִי רֹאשׁ לָראוּבֵנִי וְעָלָיו שְׁלוֹשִׁים:

[דברי הימים א פרק-כו-ה]  עַמִּיאֵל הַשִּׁשִּׁי יִשָׂשכָר הַשְּׁבִיעִי פְּעֻלְּתַי הַשְּׁמִינִי כִּי בֵרֲכוֹ אֱלֹהִים:

[דברי הימים ב פרק-ד-יב]  עַמּוּדִים שְׁנַיִם וְהַגֻּלּוֹת וְהַכֹּתָרוֹת עַל רֹאשׁ הָעַמּוּדִים שְׁתָּיִם וְהַשְּׂבָכוֹת שְׁתַּיִם לְכַסּוֹת אֶת שְׁתֵּי גֻּלּוֹת הַכֹּתָרוֹת אֲשֶׁר עַל רֹאשׁ הָעַמּוּדִים:


ע-נ


[ויקרא פרק-יח-ט]  עֶרְוַת אֲחוֹתְךָ בַת אָבִיךָ אוֹ בַת אִמֶּךָ מוֹלֶדֶת בַּיִת אוֹ מוֹלֶדֶת חוּץ לֹא תְגַלֶּה עֶרְוָתָן:

[במדבר פרק-יג-כט]  עֲמָלֵק יוֹשֵׁב בְּאֶרֶץ הַנֶּגֶב וְהַחִתִּי וְהַיְבוּסִי וְהָאֱמֹרִי יוֹשֵׁב בָּהָר וְהַכְּנַעֲנִי יֹשֵׁב עַל הַיָּם וְעַל יַד הַיַּרְדֵּן:

[במדבר פרק-כא-יד]  עַל כֵּן יֵאָמַר בְּסֵפֶר מִלְחֲמֹת יְהוָה אֶת וָהֵב בְּסוּפָה וְאֶת הַנְּחָלִים אַרְנוֹן:

[במדבר פרק-כא-כז]  עַל כֵּן יֹאמְרוּ הַמֹּשְׁלִים בֹּאוּ חֶשְׁבּוֹן תִּבָּנֶה וְתִכּוֹנֵן עִיר סִיחוֹן:

[במדבר פרק-לב-ג]  עֲטָרוֹת וְדִיבֹן וְיַעְזֵר וְנִמְרָה וְחֶשְׁבּוֹן וְאֶלְעָלֵה וּשְׂבָם וּנְבוֹ וּבְעֹן:

[יהושע פרק-יט-ז]  עַיִן רִמּוֹן וָעֶתֶר וְעָשָׁן עָרִים אַרְבַּע וְחַצְרֵיהֶן:

[ישעיה פרק-כא-ה]  עָרֹךְ הַשֻּׁלְחָן צָפֹה הַצָּפִית אָכוֹל שָׁתֹה קוּמוּ הַשָּׂרִים מִשְׁחוּ מָגֵן:

[ישעיה פרק-מח-ז]  עַתָּה נִבְרְאוּ וְלֹא מֵאָז וְלִפְנֵי יוֹם וְלֹא שְׁמַעְתָּם פֶּן תֹּאמַר הִנֵּה יְדַעְתִּין:

[ישעיה פרק-נא-ט]  עוּרִי עוּרִי לִבְשִׁי עֹז זְרוֹעַ יְהוָה עוּרִי כִּימֵי קֶדֶם דֹּרוֹת עוֹלָמִים הֲלוֹא אַתְּ הִיא הַמַּחְצֶבֶת רַהַב מְחוֹלֶלֶת תַּנִּין:

[יחזקאל פרק-א-יז]  עַל אַרְבַּעַת רִבְעֵיהֶן בְּלֶכְתָּם יֵלֵכוּ לֹא יִסַּבּוּ בְּלֶכְתָּן:

[יואל פרק-א-יז]  עָבְשׁוּ פְרֻדוֹת תַּחַת מֶגְרְפֹתֵיהֶם נָשַׁמּוּ אֹצָרוֹת נֶהֶרְסוּ מַמְּגֻרוֹת כִּי הֹבִישׁ דָּגָן:

[תהילים פרק-סה-יב]  עִטַּרְתָּ שְׁנַת טוֹבָתֶךָ וּמַעְגָּלֶיךָ יִרְעֲפוּן דָּשֶׁן:

[תהילים פרק-פג-י]  עֲשֵׂה לָהֶם כְּמִדְיָן כְּסִיסְרָא כְיָבִין בְּנַחַל קִישׁוֹן:

[תהילים פרק-צא-יג]  עַל שַׁחַל וָפֶתֶן תִּדְרֹךְ תִּרְמֹס כְּפִיר וְתַנִּין:

[תהילים פרק-קלז-א]  עַל נַהֲרוֹת בָּבֶל שָׁם יָשַׁבְנוּ גַּם בָּכִינוּ בְּזָכְרֵנוּ אֶת צִיּוֹן:

[משלי פרק-יד-לא]  עֹשֵׁק דָּל חֵרֵף עֹשֵׂהוּ וּמְכַבְּדוֹ חֹנֵן אֶבְיוֹן:

[משלי פרק-יט-כח]  עֵד בְּלִיַּעַל יָלִיץ מִשְׁפָּט וּפִי רְשָׁעִים יְבַלַּע אָוֶן:

[משלי פרק-לא-כה]  עֹז וְהָדָר לְבוּשָׁהּ וַתִּשְׂחַק לְיוֹם אַחֲרוֹן:

[איוב פרק-ט-ט]  עֹשֶׂה עָשׁ כְּסִיל וְכִימָה וְחַדְרֵי תֵמָן:

[איוב פרק-כו-ו]  עָרוֹם שְׁאוֹל נֶגְדּוֹ וְאֵין כְּסוּת לָאֲבַדּוֹן:

[איוב פרק-לט-כג]  עָלָיו תִּרְנֶה אַשְׁפָּה לַהַב חֲנִית וְכִידוֹן:

[דניאל פרק-ב-ה]  עָנֵה מַלְכָּא וְאָמַר (לכשדיא) לְכַשְׂדָּאֵי מִלְּתָא מִנִּי אַזְדָּא הֵן לָא תְהוֹדְעוּנַּנִי חֶלְמָא וּפִשְׁרֵהּ הַדָּמִין תִּתְעַבְדוּן וּבָתֵּיכוֹן נְוָלִי יִתְּשָׂמוּן:

[דניאל פרק-ג-יד]  עָנֵה נְבֻכַדְנֶצַּר וְאָמַר לְהוֹן הַצְדָּא שַׁדְרַךְ מֵישַׁךְ וַעֲבֵד נְגוֹ לֵאלָהַי לָא אִיתֵיכוֹן פָּלְחִין וּלְצֶלֶם דַּהֲבָא דִּי הֲקֵימֶת לָא סָגְדִין:

[דניאל פרק-ג-כה]  עָנֵה וְאָמַר הָא אֲנָה חָזֵה גֻּבְרִין אַרְבְּעָה שְׁרַיִן מַהְלְכִין בְּגוֹא נוּרָא וַחֲבָל לָא אִיתַי בְּהוֹן וְרֵוֵהּ דִּי (רביעיא) רְבִיעָאָה דָּמֵה לְבַר אֱלָהִין:

[דניאל פרק-ג-כח]  עָנֵה נְבוּכַדְנֶצַּר וְאָמַר בְּרִיךְ אֱלָהֲהוֹן דִּי שַׁדְרַךְ מֵישַׁךְ וַעֲבֵד נְגוֹ דִּי שְׁלַח מַלְאֲכֵהּ וְשֵׁיזִב לְעַבְדוֹהִי דִּי הִתְרְחִצוּ עֲלוֹהִי וּמִלַּת מַלְכָּא שַׁנִּיו וִיהַבוּ (גשמיהון) גֶשְׁמְהוֹן דִּי לָא יִפְלְחוּן וְלָא יִסְגְּדוּן לְכָל אֱלָהּ לָהֵן לֵאלָהֲהוֹן:

[דניאל פרק-ז-כב]  עַד דִּי אֲתָה עַתִּיק יוֹמַיָּא וְדִינָא יְהִב לְקַדִּישֵׁי עֶלְיוֹנִין וְזִמְנָא מְטָה וּמַלְכוּתָא הֶחֱסִנוּ קַדִּישִׁין:


ע-ס


[ישעיה פרק-יג-ז]  עַל כֵּן כָּל יָדַיִם תִּרְפֶּינָה וְכָל לְבַב אֱנוֹשׁ יִמָּס:

[יחזקאל פרק-ז-כג]  עֲשֵׂה הָרַתּוֹק כִּי הָאָרֶץ מָלְאָה מִשְׁפַּט דָּמִים וְהָעִיר מָלְאָה חָמָס:


ע-ע


[חבקוק פרק-א-ב]  עַד אָנָה יְהוָה שִׁוַּעְתִּי וְלֹא תִשְׁמָע אֶזְעַק אֵלֶיךָ חָמָס וְלֹא תוֹשִׁיעַ:

[תהילים פרק-פא-ו]  עֵדוּת בִּיהוֹסֵף שָׂמוֹ בְּצֵאתוֹ עַל אֶרֶץ מִצְרָיִם שְׂפַת לֹא יָדַעְתִּי אֶשְׁמָע:

[איוב פרק-לד-ו]  עַל מִשְׁפָּטִי אֲכַזֵּב אָנוּשׁ חִצִּי בְלִי פָשַׁע:

[איוב פרק-לו-לב]  עַל כַּפַּיִם כִּסָּה אוֹר וַיְצַו עָלֶיהָ בְמַפְגִּיעַ:

[איכה פרק-ד-יז]  (עודינה) עוֹדֵינוּ תִּכְלֶינָה עֵינֵינוּ אֶל עֶזְרָתֵנוּ הָבֶל בְּצִפִּיָּתֵנוּ צִפִּינוּ אֶל גּוֹי לֹא יוֹשִׁעַ:

[קהלת פרק-ג-ב]  עֵת לָלֶדֶת וְעֵת לָמוּת עֵת לָטַעַת וְעֵת לַעֲקוֹר נָטוּעַ:


ע-פ


[שמות פרק-לח-י]  עַמּוּדֵיהֶם עֶשְׂרִים וְאַדְנֵיהֶם עֶשְׂרִים נְחֹשֶׁת וָוֵי הָעַמֻּדִים וַחֲשֻׁקֵיהֶם כָּסֶף:

[דברי הימים א פרק-כז-ב]  עַל הַמַּחֲלֹקֶת הָרִאשׁוֹנָה לַחֹדֶשׁ הָרִאשׁוֹן יָשָׁבְעָם בֶּן זַבְדִּיאֵל וְעַל מַחֲלֻקְתּוֹ עֶשְׂרִים וְאַרְבָּעָה אָלֶף:

[דברי הימים א פרק-כז-יד]  עַשְׁתֵּי עָשָׂר לְעַשְׁתֵּי עָשָׂר הַחֹדֶשׁ בְּנָיָה הַפִּרְעָתוֹנִי מִן בְּנֵי אֶפְרָיִם וְעַל מַחֲלֻקְתּוֹ עֶשְׂרִים וְאַרְבָּעָה אָלֶף:


ע-צ


[בראשית פרק-יא-ט]  עַל כֵּן קָרָא שְׁמָהּ בָּבֶל כִּי שָׁם בָּלַל יְהוָה שְׂפַת כָּל הָאָרֶץ וּמִשָּׁם הֱפִיצָם יְהוָה עַל פְּנֵי כָּל הָאָרֶץ:

[חבקוק פרק-ג-ט]  עֶרְיָה תֵעוֹר קַשְׁתֶּךָ שְׁבֻעוֹת מַטּוֹת אֹמֶר סֶלָה נְהָרוֹת תְּבַקַּע אָרֶץ:

[תהילים פרק-קכא-ב]  עֶזְרִי מֵעִם יְהוָה עֹשֵׂה שָׁמַיִם וָאָרֶץ:

[תהילים פרק-קכד-ח]  עֶזְרֵנוּ בְּשֵׁם יְהוָה עֹשֵׂה שָׁמַיִם וָאָרֶץ:

[משלי פרק-כ-יז]  עָרֵב לָאִישׁ לֶחֶם שָׁקֶר וְאַחַר יִמָּלֵא פִיהוּ חָצָץ:

[דברי הימים ב פרק-א-ט]  עַתָּה יְהוָה אֱלֹהִים יֵאָמֵן דְּבָרְךָ עִם דָּוִיד אָבִי כִּי אַתָּה הִמְלַכְתַּנִי עַל עַם רַב כַּעֲפַר הָאָרֶץ:


ע-ק


[דברים פרק-ט-ב]  עַם גָּדוֹל וָרָם בְּנֵי עֲנָקִים אֲשֶׁר אַתָּה יָדַעְתָּ וְאַתָּה שָׁמַעְתָּ מִי יִתְיַצֵּב לִפְנֵי בְּנֵי עֲנָק:

[קהלת פרק-ג-ה]  עֵת לְהַשְׁלִיךְ אֲבָנִים וְעֵת כְּנוֹס אֲבָנִים עֵת לַחֲבוֹק וְעֵת לִרְחֹק מֵחַבֵּק:


ע-ר


[בראשית פרק-ו-יד]  עֲשֵׂה לְךָ תֵּבַת עֲצֵי גֹפֶר קִנִּים תַּעֲשֶׂה אֶת הַתֵּבָה וְכָפַרְתָּ אֹתָהּ מִבַּיִת וּמִחוּץ בַּכֹּפֶר:

[בראשית פרק-יד-ב]  עָשׂוּ מִלְחָמָה אֶת בֶּרַע מֶלֶךְ סְדֹם וְאֶת בִּרְשַׁע מֶלֶךְ עֲמֹרָה שִׁנְאָב מֶלֶךְ אַדְמָה וְשֶׁמְאֵבֶר מֶלֶךְ (צביים) צְבוֹיִים וּמֶלֶךְ בֶּלַע הִיא צֹעַר:

[שמואל א פרק-טו-ג]  עַתָּה לֵךְ וְהִכִּיתָה אֶת עֲמָלֵק וְהַחֲרַמְתֶּם אֶת כָּל אֲשֶׁר לוֹ וְלֹא תַחְמֹל עָלָיו וְהֵמַתָּה מֵאִישׁ עַד אִשָּׁה מֵעֹלֵל וְעַד יוֹנֵק מִשּׁוֹר וְעַד שֶׂה מִגָּמָל וְעַד חֲמוֹר:

[שמואל ב פרק-כב-נ]  עַל כֵּן אוֹדְךָ יְהוָה בַּגּוֹיִם וּלְשִׁמְךָ אֲזַמֵּר:

[מלכים א פרק-א-מב]  עוֹדֶנּוּ מְדַבֵּר וְהִנֵּה יוֹנָתָן בֶּן אֶבְיָתָר הַכֹּהֵן בָּא וַיֹּאמֶר אֲדֹנִיָּהוּ בֹּא כִּי אִישׁ חַיִל אַתָּה וְטוֹב תְּבַשֵּׂר:

[ישעיה פרק-כד-ו]  עַל כֵּן אָלָה אָכְלָה אֶרֶץ וַיֶּאְשְׁמוּ יֹשְׁבֵי בָהּ עַל כֵּן חָרוּ יֹשְׁבֵי אֶרֶץ וְנִשְׁאַר אֱנוֹשׁ מִזְעָר:

[ישעיה פרק-נז-ד]  עַל מִי תִּתְעַנָּגוּ עַל מִי תַּרְחִיבוּ פֶה תַּאֲרִיכוּ לָשׁוֹן הֲלוֹא אַתֶּם יִלְדֵי פֶשַׁע זֶרַע שָׁקֶר:

[ירמיה פרק-ד-כא]  עַד מָתַי אֶרְאֶה נֵּס אֶשְׁמְעָה קוֹל שׁוֹפָר:

[ירמיה פרק-יב-יב]  עַל כָּל שְׁפָיִם בַּמִּדְבָּר בָּאוּ שֹׁדְדִים כִּי חֶרֶב לַיהוָה אֹכְלָה מִקְצֵה אֶרֶץ וְעַד קְצֵה הָאָרֶץ אֵין שָׁלוֹם לְכָל בָּשָׂר:

[ירמיה פרק-לא-כב]  עַד מָתַי תִּתְחַמָּקִין הַבַּת הַשּׁוֹבֵבָה כִּי בָרָא יְהוָה חֲדָשָׁה בָּאָרֶץ נְקֵבָה תְּסוֹבֵב גָּבֶר:

[ירמיה פרק-מז-ד]  עַל הַיּוֹם הַבָּא לִשְׁדוֹד אֶת כָּל פְּלִשְׁתִּים לְהַכְרִית לְצֹר וּלְצִידוֹן כֹּל שָׂרִיד עֹזֵר כִּי שֹׁדֵד יְהוָה אֶת פְּלִשְׁתִּים שְׁאֵרִית אִי כַפְתּוֹר:

[יואל פרק-א-ג]  עָלֶיהָ לִבְנֵיכֶם סַפֵּרוּ וּבְנֵיכֶם לִבְנֵיהֶם וּבְנֵיהֶם לְדוֹר אַחֵר:

[תהילים פרק-לג-יא]  עֲצַת יְהוָה לְעוֹלָם תַּעֲמֹד מַחְשְׁבוֹת לִבּוֹ לְדֹר וָדֹר:

[תהילים פרק-מז-ו]  עָלָה אֱלֹהִים בִּתְרוּעָה יְהוָֹה בְּקוֹל שׁוֹפָר:

[תהילים פרק-נז-ט]  עוּרָה כְבוֹדִי עוּרָה הַנֵּבֶל וְכִנּוֹר אָעִירָה שָּׁחַר:

[תהילים פרק-צב-ד]  עֲלֵי עָשׂוֹר וַעֲלֵי נָבֶל עֲלֵי הִגָּיוֹן בְּכִנּוֹר:

[תהילים פרק-קח-ג]  עוּרָה הַנֵּבֶל וְכִנּוֹר אָעִירָה שָּׁחַר:

[תהילים פרק-קמז-ז]  עֱנוּ לַיהוָה בְּתוֹדָה זַמְּרוּ לֵאלֹהֵינוּ בְכִנּוֹר:

[משלי פרק-ג-יח]  עֵץ חַיִּים הִיא לַמַּחֲזִיקִים בָּהּ וְתֹמְכֶיהָ מְאֻשָּׁר:

[משלי פרק-יד-ה]  עֵד אֱמוּנִים לֹא יְכַזֵּב וְיָפִיחַ כְּזָבִים עֵד שָׁקֶר:

[משלי פרק-כא-כח]  עֵד כְּזָבִים יֹאבֵד וְאִישׁ שׁוֹמֵעַ לָנֶצַח יְדַבֵּר:

[משלי פרק-כב-טז]  עֹשֵׁק דָּל לְהַרְבּוֹת לוֹ נֹתֵן לְעָשִׁיר אַךְ לְמַחְסוֹר:

[משלי פרק-ל-יז]  עַיִן תִּלְעַג לְאָב וְתָבוּז לִיקֲּהַת אֵם יִקְּרוּהָ עֹרְבֵי נַחַל וְיֹאכְלוּהָ בְנֵי נָשֶׁר:

[איוב פרק-א-יח]  עַד זֶה מְדַבֵּר וְזֶה בָּא וַיֹּאמַר בָּנֶיךָ וּבְנוֹתֶיךָ אֹכְלִים וְשֹׁתִים יַיִן בְּבֵית אֲחִיהֶם הַבְּכוֹר:

[איוב פרק-ה-ט]  עֹשֶׂה גְדֹלוֹת וְאֵין חֵקֶר נִפְלָאוֹת עַד אֵין מִסְפָּר:

[איוב פרק-ט-י]  עֹשֶׂה גְדֹלוֹת עַד אֵין חֵקֶר וְנִפְלָאוֹת עַד אֵין מִסְפָּר:

[איוב פרק-יח-ב]  עַד אָנָה תְּשִׂימוּן קִנְצֵי לְמִלִּין תָּבִינוּ וְאַחַר נְדַבֵּר:

[איוב פרק-יח-כ]  עַל יוֹמוֹ נָשַׁמּוּ אַחֲרֹנִים וְקַדְמֹנִים אָחֲזוּ שָׂעַר:

[איוב פרק-כד-י]  עָרוֹם הִלְּכוּ בְּלִי לְבוּשׁ וּרְעֵבִים נָשְׂאוּ עֹמֶר:

[איוב פרק-מא-י]  עֲטִישֹׁתָיו תָּהֶל אוֹר וְעֵינָיו כְּעַפְעַפֵּי שָׁחַר:

[איוב פרק-מב-ו]  עַל כֵּן אֶמְאַס וְנִחַמְתִּי עַל עָפָר וָאֵפֶר:

[שיר השירים פרק-ב-יז]  עַד שֶׁיָּפוּחַ הַיּוֹם וְנָסוּ הַצְּלָלִים סֹב דְּמֵה לְךָ דוֹדִי לִצְבִי אוֹ לְעֹפֶר הָאַיָּלִים עַל הָרֵי בָתֶר:

[קהלת פרק-ג-ז]  עֵת לִקְרוֹעַ וְעֵת לִתְפּוֹר עֵת לַחֲשׁוֹת וְעֵת לְדַבֵּר:

[קהלת פרק-י-טו]  עֲמַל הַכְּסִילִים תְּיַגְּעֶנּוּ אֲשֶׁר לֹא יָדַע לָלֶכֶת אֶל עִיר:

[קהלת פרק-יב-ו]  עַד אֲשֶׁר לֹא (ירחק) יֵרָתֵק חֶבֶל הַכֶּסֶף וְתָרֻץ גֻּלַּת הַזָּהָב וְתִשָּׁבֶר כַּד עַל הַמַּבּוּעַ וְנָרֹץ הַגַּלְגַּל אֶל הַבּוֹר:

[אסתר פרק-ו-יד]  עוֹדָם מְדַבְּרִים עִמּוֹ וְסָרִיסֵי הַמֶּלֶךְ הִגִּיעוּ וַיַּבְהִלוּ לְהָבִיא אֶת הָמָן אֶל הַמִּשְׁתֶּה אֲשֶׁר עָשְׂתָה אֶסְתֵּר:

[דברי הימים א פרק-כה-יח]  עַשְׁתֵּי עָשָׂר עֲזַרְאֵל בָּנָיו וְאֶחָיו שְׁנֵים עָשָׂר:


ע-ש


[יהושע פרק-א-טו]  עַד אֲשֶׁר יָנִיחַ יְהוָה לַאֲחֵיכֶם כָּכֶם וְיָרְשׁוּ גַם הֵמָּה אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֵיכֶם נֹתֵן לָהֶם וְשַׁבְתֶּם לְאֶרֶץ יְרֻשַּׁתְכֶם וִירִשְׁתֶּם אוֹתָהּ אֲשֶׁר נָתַן לָכֶם מֹשֶׁה עֶבֶד יְהוָה בְּעֵבֶר הַיַּרְדֵּן מִזְרַח הַשָּׁמֶשׁ:

[ישעיה פרק-טז-יא]  עַל כֵּן מֵעַי לְמוֹאָב כַּכִּנּוֹר יֶהֱמוּ וְקִרְבִּי לְקִיר חָרֶשׂ:

[תהילים פרק-י-יג]  עַל מֶה נִאֵץ רָשָׁע אֱלֹהִים אָמַר בְּלִבּוֹ לֹא תִּדְרֹשׁ:

[משלי פרק-כח-יט]  עֹבֵד אַדְמָתוֹ יִשְׂבַּע לָחֶם וּמְרַדֵּף רֵקִים יִשְׂבַּע רִישׁ:

[איוב פרק-ח-יב]  עֹדֶנּוּ בְאִבּוֹ לֹא יִקָּטֵף וְלִפְנֵי כָל חָצִיר יִיבָשׁ:

[איוב פרק-מ-י]  עֲדֵה נָא גָאוֹן וָגֹבַהּ וְהוֹד וְהָדָר תִּלְבָּשׁ:


ע-ת


[במדבר פרק-י-ב]  עֲשֵׂה לְךָ שְׁתֵּי חֲצוֹצְרֹת כֶּסֶף מִקְשָׁה תַּעֲשֶׂה אֹתָם וְהָיוּ לְךָ לְמִקְרָא הָעֵדָה וּלְמַסַּע אֶת הַמַּחֲנוֹת:

[מלכים א פרק-יח-לז]  עֲנֵנִי יְהוָה עֲנֵנִי וְיֵדְעוּ הָעָם הַזֶּה כִּי אַתָּה יְהוָה הָאֱלֹהִים וְאַתָּה הֲסִבֹּתָ אֶת לִבָּם אֲחֹרַנִּית:

[ישעיה פרק-כא-ג]  עַל כֵּן מָלְאוּ מָתְנַי חַלְחָלָה צִירִים אֲחָזוּנִי כְּצִירֵי יוֹלֵדָה נַעֲוֵיתִי מִשְּׁמֹעַ נִבְהַלְתִּי מֵרְאוֹת:

[ירמיה פרק-טו-ח]  עָצְמוּ לִי אַלְמְנֹתָו מֵחוֹל יַמִּים הֵבֵאתִי לָהֶם עַל אֵם בָּחוּר שֹׁדֵד בַּצָּהֳרָיִם הִפַּלְתִּי עָלֶיהָ פִּתְאֹם עִיר וּבֶהָלוֹת:

[ירמיה פרק-מו-ט]  עֲלוּ הַסּוּסִים וְהִתְהֹלְלוּ הָרֶכֶב וְיֵצְאוּ הַגִּבּוֹרִים כּוּשׁ וּפוּט תֹּפְשֵׂי מָגֵן וְלוּדִים תֹּפְשֵׂי דֹּרְכֵי קָשֶׁת:

[ירמיה פרק-מח-כח]  עִזְבוּ עָרִים וְשִׁכְנוּ בַּסֶּלַע יֹשְׁבֵי מוֹאָב וִהְיוּ כְיוֹנָה תְּקַנֵּן בְּעֶבְרֵי פִי פָחַת:

[יחזקאל פרק-מא-יז]  עַל מֵעַל הַפֶּתַח וְעַד הַבַּיִת הַפְּנִימִי וְלַחוּץ וְאֶל כָּל הַקִּיר סָבִיב סָבִיב בַּפְּנִימִי וּבַחִיצוֹן מִדּוֹת:

[משלי פרק-א-כב]  עַד מָתַי פְּתָיִם תְּאֵהֲבוּ פֶתִי וְלֵצִים לָצוֹן חָמְדוּ לָהֶם וּכְסִילִים יִשְׂנְאוּ דָעַת:

[משלי פרק-יד-כד]  עֲטֶרֶת חֲכָמִים עָשְׁרָם אִוֶּלֶת כְּסִילִים אִוֶּלֶת:

[משלי פרק-כג-לג]  עֵינֶיךָ יִרְאוּ זָרוֹת וְלִבְּךָ יְדַבֵּר תַּהְפֻּכוֹת:

[שיר השירים פרק-ה-יב]  עֵינָיו כְּיוֹנִים עַל אֲפִיקֵי מָיִם רֹחֲצוֹת בֶּחָלָב יֹשְׁבוֹת עַל מִלֵּאת:

[איכה פרק-ג-מט]  עֵינִי נִגְּרָה וְלֹא תִדְמֶה מֵאֵין הֲפֻגוֹת:

[קהלת פרק-ג-ג]  עֵת לַהֲרוֹג וְעֵת לִרְפּוֹא עֵת לִפְרוֹץ וְעֵת לִבְנוֹת:

[דניאל פרק-ז-כח]  עַד כָּה סוֹפָא דִי מִלְּתָא אֲנָה דָנִיֵּאל שַׂגִּיא רַעְיוֹנַי יְבַהֲלֻנַּנִי וְזִיוַי יִשְׁתַּנּוֹן עֲלַי וּמִלְּתָא בְּלִבִּי נִטְרֵת:

[דברי הימים א פרק-כט-ל]  עִם כָּל מַלְכוּתוֹ וּגְבוּרָתוֹ וְהָעִתִּים אֲשֶׁר עָבְרוּ עָלָיו וְעַל יִשְׂרָאֵל וְעַל כָּל מַמְלְכוֹת הָאֲרָצוֹת:
