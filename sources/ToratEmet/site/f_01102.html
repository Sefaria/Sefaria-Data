פרק א - משנה א

מַשְׁקִין בֵּית הַשְּׁלָחִין בַּמּוֹעֵד {ב} וּבַשְּׁבִיעִית, בֵּין מִמַּעְיָן שֶׁיָּצָא בַתְּחִלָּה, בֵּין מִמַּעְיָן שֶׁלֹּא {ו} יָצָא בַתְּחִלָּה. אֲבָל אֵין מַשְׁקִין {ז} לֹא מִמֵּי הַגְּשָׁמִים וְלֹא מִמֵּי הַקִּילוֹן {ט}. וְאֵין עוֹשִׂין {י} עוּגִיּוֹת לַגְּפָנִים:

 ר"ע מברטנורה  מַשְׁקִין בֵּית הַשְּׁלָחִין. אֶרֶץ עֲיֵפָה וּצְמֵאָה לְמַיִם. תַּרְגּוּם עָיֵף וְיָגֵעַ, מְשַׁלְהֵי וְלָאֵי {א}: מַשְׁקִין בַּמּוֹעֵד. דְּדָבָר הָאָבֵד מֻתָּר בַּמּוֹעֵד {ג}, וּשְׂדֵה בֵּית הַשְּׁלָחִין מִשָּׁעָה שֶׁמַּתְחִיל לְהַשְׁקוֹתוֹ אִם אֵינוֹ מַשְׁקֶה אוֹתוֹ תָּדִיר מִיָּד נֶאֱבָד, כְּדִתְנַן בְּסוֹף מַתְנִיתִין, זְרָעִים שֶׁלֹּא שָׁתוּ לִפְנֵי הַמּוֹעֵד לֹא יַשְׁקֵם בַּמּוֹעֵד דְּלֹא פָּסְדֵי, אֲבָל שָׁתוּ לִפְנֵי הַמּוֹעֵד יַשְׁקֵם בַּמּוֹעֵד. וְדַוְקָא שְׂדֵה לָבָן שֶׁל תְּבוּאָה שֶׁהוּא שְׁלָחִין מַשְׁקִין בַּמּוֹעֵד, אֲבָל שְׂדֵה שְׁלָחִין שֶׁל אִילָן לֹא שָׁרֵי בַּמּוֹעֵד, דְּלֹא פָּסְדֵי. וּבַשְּׁבִיעִית שָׁרֵי לְהַשְׁקוֹת בֵּין שְׂדֵה בֵּית הַשְּׁלָחִין בֵּין שְׂדֵה בֵּית הַבַּעַל. וּבֵית הַשְּׁלָחִין, מִשּׁוּם מוֹעֵד בִּלְבָד נָקַט לַהּ {ד}: שֶׁיָּצָא בַתְּחִלָּה. שֶׁיּוֹצֵא עַכְשָׁיו מִתְּחִלָּה. וְאֵין חוֹשְׁשִׁין הוֹאִיל וְחָדָשׁ הוּא שֶׁמָּא יִפְּלוּ גְּדוֹתָיו וְאָתֵי לְתַקֵּן בַּמּוֹעֵד וְאִיכָּא טִרְחָא {ה}: שֶׁלֹּא יָצָא בַתְּחִלָּה. שֶׁהוּא יָשָׁן: אֲבָל לֹא מִמֵּי גְּשָׁמִים. גְּזֵרָה מִשּׁוּם מֵי קִילוֹן. קִילוֹן הֵם מַיִם שֶׁל בּוֹר עָמֹק וּבוֹ מֵי גְּשָׁמִים מְכֻנָּסִים, דְּבֵיהּ אִיכָּא טִרְחָא יְתֵרָה לִדְלוֹת {ח}: עוּגִיּוֹת. חֲרִיצִים {יא} שֶׁעוֹשִׂים בְּעִקְּרֵי הַזֵּיתִים וּבְעִקְּרֵי הַגְּפָנִים {יב} כְּדֵי שֶׁיִּתְמַלְּאוּ מַיִם: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} וְהֵ"א מִתְחַלֵּף בְּחֵי"ת. רַשִׁ"י. שֶׁשְּׁנֵיהֶם מִמּוֹצָא הַגָּרוֹן. וְעָיֵף הַיְנוּ לְמַיִם, כְּדִכְתִיב [מִשְׁלֵי כה] מַיִם קָרִים עַל נֶפֶשׁ עֲיֵפָה: {ב} בַּמּוֹעֵד. הַמִּשְׁנָה תִּקְרָא [מוֹעֵד, מַה שֶּׁאָנוּ קוֹרִים] חֻלּוֹ שֶׁל מוֹעֵד, [וְהֵם] הַיָּמִים שֶׁבֵּין מוֹעֵד לְמוֹעֵד. הָרַמְבַּ"ם: {ג} רַשִׁ"י. וְהָרִי"ף וְהָרַמְבַּ"ם וְהָרֹא"שׁ וְהַטּוּר פֶּה אֶחָד הִסְכִּימוּ דְּאֵין לְחֻלּוֹ שֶׁל מוֹעֵד עִקָּר מִן הַתּוֹרָה. וּבְרַיְתָא דַּחֲגִיגָה דְּלַמְּדוּהָ מִקְּרָא אֵינוֹ אֶלָּא אַסְמַכְתָּא. וְהַבֵּית יוֹסֵף כָּתַב שֶׁנִּרְאֶה לוֹ מִזֹּאת הַבְּרַיְתָא שֶׁיֵּשׁ לוֹ עִקָּר מִן הַתּוֹרָה, וְהַכָּתוּב מְסָרָהּ לַחֲכָמִים לֶאֱסֹר וּלְהַתִּיר כְּפִי שֶׁנִּרְאֶה לָהֶם: {ד} אֲבָל הָרַמְבַּ"ם נִרְאֶה שֶׁאֵין סוֹבֵר כָּךְ, אֶלָּא דְּבֵית הַבַּעַל נַמִּי אָסוּר בִּשְׁבִיעִית. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} כְּלוֹמַר שֶׁאֲפִלּוּ דָּבָר הָאָבֵד לֹא הִתִּירוּ כְּשֶׁיֵּשׁ טִרְחָא: {ו} שֶׁלֹּא. וְאִצְטְרִיךְ לְמִתְנֵי לְמִעוּטֵי שְׂדֵה הַבַּעַל, שֶׁאֲפִלּוּ מִמַּעְיָן שֶׁלֹּא יָצָא בִּתְחִלָּה אָסוּר. גְּמָרָא: {ז} אֵין מַשְׁקִין. אַמּוֹעֵד קָאֵי, דְּגַבֵּי שְׁבִיעִית לֹא חָיְישִׁינַן לְטִרְחָא יְתֵרָא, וְטַעְמָא מְפָרֵשׁ בַּיְרוּשַׁלְמִי דִּשְׁבִיעִית עַל יְדֵי שֶׁהוּא מֻתָּר בִּמְלָאכָה שֶׁאֵינָהּ דְּקַרְקַע הִתִּירוּ אֲפִלּוּ בְּדָבָר שֶׁל טֹרַח, אִי נַמִּי שְׁבִיעִית עַל יְדֵי שֶׁזְּמַנָּהּ מְרֻבֶּה כוּ' וְלֹא מִסְתַּבֵּר לַעֲשׂוֹת שִׁבְעָה יָמִים הָאַחֲרוֹנִים כְּשִׁבְעָה דְּרֶגֶל: {ח} וּמִלְּשׁוֹן קוּלְתָּא שֶׁהוּא תִּרְגּוּם כַּדָּהּ עַל שִׁכְמָה, נִקְרֵאת קִילוֹן. הָרַ"ן: {ט} הַקִּילוֹן. כֵּיוָן דְּמֵי הַגְּשָׁמִים לֹא נֶאֶסְרוּ אֶלָּא מִשּׁוּם גְּזֵרָה דְּמֵי הַקִּילוֹן, הִלְכָּךְ שַׁפִּיר נַקְטֵיהּ אַחַר מֵי גְּשָׁמִים. הָרַ"ן: {י} וְאֵין עוֹשִׂין כוּ'. מִדְּרֵישָׁא דְּהַךְ אֲבָל אֵין מַשְׁקִין לֹא קָאֵי אַשְּׁבִיעִית, הַךְ סֵיפָא נַמִּי לֹא קָאֵי אַשְּׁבִיעִית. הָרַמְבַּ"ם: {יא} כְּדֵי שֶׁיִּתְעַכְּבוּ שָׁם הַמַּיִם, מִלְּשׁוֹן [רוּת א] הֲלָהֶן תֵּעָגֵנָה, שֶׁהוּא לְשׁוֹן עִכּוּב. הָרַ"ן: {יב} לִישְׁנָא דִּבְרַיְתָא, וּבְאַתְרֵיהּ דְּתַנָּא דְּמַתְנִיתִין שְׁכִיחֵי גְּפָנִים וְלֹא זֵיתִים: (עיקר תוי"ט).

פרק א - משנה ב

רַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה אוֹמֵר, אֵין עוֹשִׂין אֶת הָאַמָּה בַתְּחִלָּה בַּמּוֹעֵד וּבַשְּׁבִיעִית, וַחֲכָמִים אוֹמְרִים {יג}, עוֹשִׂין אֶת הָאַמָּה בַּתְּחִלָּה בַּשְּׁבִיעִית, וּמְתַקְּנִין אֶת הַמְּקֻלְקָלוֹת {יד} בַּמּוֹעֵד. וּמְתַקְּנִין אֶת קִלְקוּלֵי הַמַּיִם שֶׁבִּרְשׁוּת הָרַבִּים {טז} וְחוֹטְטִין אוֹתָן. וּמְתַקְּנִין אֶת הַדְּרָכִים וְאֶת הָרְחוֹבוֹת {יז} וְאֶת מִקְווֹת הַמַּיִם, וְעוֹשִׂין {יח} כָּל צָרְכֵי הָרַבִּים, וּמְצַיְּנִין אֶת הַקְּבָרוֹת, וְיוֹצְאִין {יט} אַף עַל הַכִּלְאָיִם:

 ר"ע מברטנורה  הָאַמָּה. חֲרִיצִים שֶׁעוֹשִׂים בַּקַּרְקַע שֶׁבָּהֶן הַמַּיִם הוֹלְכִים סְבִיב הַשָּׂדֶה וּמִשָּׂדֶה לְשָׂדֶה, קְרוּיִין אַמָּה, לְפִי שֶׁהֵן רֹחַב אַמָּה בְּרוּם אַמָּה: בַּתְּחִלָּה. אִם לֹא הָיְתָה מֵעוֹלָם אֵין עוֹשִׂין אוֹתָהּ בַּתְּחִלָּה בַּמּוֹעֵד, מִשּׁוּם טִרְחָא: וּבַשְּׁבִיעִית. מִפְּנֵי שֶׁנִּרְאֶה כְּחוֹפֵר בַּקַּרְקַע בַּשְּׁבִיעִית: וּמְתַקְּנִים אֶת הַמְּקֻלְקֶלֶת. אִם נָפַל לְתוֹכָהּ עָפָר וְאֵינָהּ מְקַלַּחַת, מְתַקְּנִים אוֹתָהּ {טו}, אֲבָל אֵין עוֹשִׂים אוֹתָהּ בַּתְּחִלָּה בַּמּוֹעֵד. וַהֲלָכָה כַּחֲכָמִים: וּמְתַקְּנִים קִלְקוּלֵי הַמַּיִם. מַיִם הַמְּכֻנָּסִים בַּגֻּמּוֹת לִשְׁתּוֹת מֵהֶן: וְחוֹטְטִין. מְנַקִּין וּמְנַקְּרִין וּמוֹצִיאִין מִתּוֹכָן צְרוֹרוֹת וְקִסְמִין וְעַפְרוּרִית שֶׁנָּפַל בָּהֶן: וּמְצַיְּנִין. עוֹשִׂין צִיּוּנִין עַל הַקְּבָרוֹת. שֶׁהָיוּ מְמַחִין סִיד וְשׁוֹפְכִין עַל הַקֶּבֶר, וְהוּא סִימָן לְהוֹלְכֵי דְּרָכִים שֶׁלֹּא יַעַבְרוּ בִּמְקוֹם טֻמְאָה: וְיוֹצְאִין. שְׁלוּחֵי בֵּית דִּין בַּמּוֹעֵד לְעַיֵּן הַשָּׂדוֹת שֶׁנִּזְרְעוּ כִּלְאַיִם, לַעֲקֹר אוֹתָם. לְפִי שֶׁנּוֹטְלִין שָׂכָר עַל כָּךְ מִתְּרוּמַת הַלִּשְׁכָּה {כ}, וּבִימֵי הַמּוֹעֵד נִשְׂכָּרִים בְּזוֹל, מִפְּנֵי שֶׁהֵן בְּטֵלִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {יג} וַחֲכָמִים אוֹמְרִים כוּ'. דְּכָל שֶׁנִּרְאֶה כְּחוֹפֵר לֹא הֲוֵי עֲבוֹדָה, דְּעוֹדֵר לֹא הֲוֵי כֵּיוָן דְּמַשְׁלִיךְ אֶת הֶעָפָר לְמֵרָחוֹק, וְהָעוֹדֵר מֵנִיחַ בִּמְקוֹמוֹ. הָרַ"ן. וּבָרַמְבַּ"ם מַשְׁמָע דְּמַתִּיר אֲפִלּוּ אֵינוֹ מַשְׁלִיךְ לְמֵרָחוֹק, וְעַיֵּן שָׁם הַטַּעַם: {יד} הַמְקֻלְקָלוֹת. סִיּוּמָא דְּמִלְּתָא דְּרַבָּנָן הוּא: {טו} וְלֹא דָּמִי לְמַעְיָן שֶׁיָּצָא בַּתְּחִלָּה כְּדִלְעֵיל דְּחַיְּשִׁינַן שֶׁיִּפְּלוּ גְּדוֹתָיו וְאָתֵי לְתַקֵּן, דְּתִקּוּנָא דְּהָתָם עֲשִׂיָּה בַּתְּחִלָּה הִיא כִּדְקָתָנֵי שֶׁיָּצָא בַּתְּחִלָּה: {טז} שֶׁבִּרְשׁוּת הָרַבִּים. אֲפִלּוּ אֵין רַבִּים צְרִיכִים לָהֶם. וְטַעְמָא כָּתַב הָרַ"ן בְּשֵׁם הָרַאבַ"ד לְפִי שֶׁצָּרְכֵי רַבִּים אֵינָם נִגְמָרִים אֶלָּא בִּשְׁעַת הַבַּטָּלָה, מִשּׁוּם דְּדָמִי לִקְדֵרָה דְּבֵי שֻׁתְּפֵי כוּ', וּמַסִּיק דְּכִי הִתִּירוּ בִּמְלֶאכֶת הֶדְיוֹט [וּכְדִלְקָמָּן]. וְנִרְאֶה לִי דְּהַיְנוּ דְּתָנֵי וְחוֹטְטִין [דְּהַיְנוּ שֶׁהוּא מְלֶאכֶת הֶדְיוֹט] וּפֵרוּשֵׁי קָמְפָרֵשׁ דְּאִי לֹא תֵּימָא הָכִי לִישָׁנָא יְתֵרָא הוּא דְּהַיְנוּ מְתַקְּנִין: {יז} וּמְתַקְּנִין כוּ'. בְּרֵישׁ שְׁקָלִים תְּנַן דִּבְט"ו בַּאֲדָר מְתַקְּנִין. וּלְפִי שֶׁהַקִּלְקוּלִים מְצוּיִין בַּדְּרָכִים לְכָךְ הֻצְרַךְ תּוּ לְתַקֵּן בְּחֹל הַמּוֹעֵד: {יח} וְעוֹשִׂין כוּ'. לַאֲתוֹיֵי בּוֹרוֹת שִׁיחִין וּמְעָרוֹת כְּשֶׁרַבִּים צְרִיכִין לָהֶם עַתָּה. אֲבָל שֶׁלֹּא לְצֹרֶךְ הַמּוֹעֵד לֹא הִתִּירוּ, מִפְּנֵי שֶׁהֵן מְלֶאכֶת אֻמָּן, וְאֵינוֹ יָכֹל לְשַׁנּוֹת בָּהֶן, וְאִם יְשַׁנֶּה בָּהֶן יַפְסִיד. הָרֹא"שׁ. וְעַיֵּן פֵּרוּשׁ הָרַ"ב בְּרֵישׁ שְׁקָלִים, וְהוּא מֵהַיְרוּשַׁלְמִי. וּגְמָרָא דִּידָן לֹא סְבִירָא לֵיהּ הַאי דִּיּוּקָא, מִשּׁוּם דְּלֹא דָּמִי לְהָנָךְ פְּרָטֵי דְּמַתְנִיתִין: {יט} וְיוֹצְאִין כוּ'. בַּגְּמָרָא רָמֵי וְהָתְנַן בְּרֵישׁ שְׁקָלִים בְּא' בַּאֲדָר כוּ', וּמְשָׁנֵי כָּאן בְּבָכִיר כָּאן בְּאָפִיל: {כ} וְלֹא הֲוֵי מְעִילָה, דְּלֵב בֵּית דִּין מַתְנֶה עֲלֵיהֶן: (עיקר תוי"ט).

פרק א - משנה ג

רַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר, מוֹשְׁכִים אֶת הַמַּיִם מֵאִילָן לְאִילָן, וּבִלְבַד שֶׁלֹּא יַשְׁקֶה אֶת כָּל הַשָּׂדֶה. זְרָעִים שֶׁלֹּא שָׁתוּ לִפְנֵי הַמּוֹעֵד, לֹא יַשְׁקֵם בַּמּוֹעֵד. וַחֲכָמִים מַתִּירִין בָּזֶה וּבָזֶה:

 ר"ע מברטנורה  מוֹשְׁכִים אֶת הַמַּיִם. שֶׁתַּחַת אִילָן זֶה, לְאִילָן אַחֵר, דְּלֵיכָּא טִרְחָא. אֲבָל לֹא יַשְׁקֶה כָּל הַשָּׂדֶה. וּבִשְׂדֵה בֵּית הַבַּעַל אָיְרֵי {כא}, דְּלֹא פָּסְדֵי {כב}, שֶׁמֵּי גְּשָׁמִים מַסְפִּיקִין לוֹ, אֶלָּא שֶׁכְּשֶׁמַּשְׁקִין אוֹתוֹ הֲוֵי הַרְוָחָה טְפֵי: שֶׁלֹּא שָׁתוּ. דְּלֹא פָּסְדֵי: בָּזֶה וּבָזֶה. אֲפִלּוּ לֹא שָׁתוּ, וַאֲפִלּוּ שְׂדֵה הַבַּעַל, דְּמִשּׁוּם הַרְוָחָה שָׁרוּ חֲכָמִים. וַהֲלָכָה כְּרַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב, דִּסְתָם מַתְנִיתִין מַשְׁקִין בֵּית הַשְּׁלָחִין דִּלְעֵיל כְּוָתֵיהּ. וּמִיהוּ שָׂדֶה לַחָה שֶׁהַקַּרְקַע שֶׁלָּהּ כְּמוֹ טִיט מֻתָּר לְהַשְׁקוֹתָהּ בַּמּוֹעֵד אֲפִלּוּ לְרַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב, דְּכִזְרָעִים שֶׁשָּׁתוּ לִפְנֵי הַמּוֹעֵד דָּמוּ: (ר"ע מברטנורה).

 עיקר תוי"ט  {כא} דְּאִי בְּבֵית הַשְּׁלָחִין הָתְנַן בְּרֵישׁ פִּרְקִין דְּמַשְׁקִין, וְהַיְינוּ כָּל הַשָּׂדֶה. אֲבָל תֵּימַהּ מַה דּוֹחְקֵיהּ לְאוֹקְמֵיהּ בִּשְׂדֵה הַבַּעַל, לוֹקְמֵיהּ בִּשְׁלָחִין, וְאִלּוּ בֵּית הַבַּעַל מַאן דְּכַר שְׁמֵיהּ כוּ', וְאִי קַשְׁיָא מַשְׁקִין דִּלְעֵיל הֲרֵי כָּתַב בְּעַצְמוֹ לְעֵיל דְּשֶׁל אִילָן לֹא שָׁרֵי בַּמּוֹעֵד דְּלֹא פָּסְדֵי, וּמַתְנִיתִין הָכִי דַּיְיקָא דִּמְשִׁיכָה מֵאִילָן לְאִילָן הוּא דְּמֻתָּר הָא הַשְׁקָאָה לֹא. וְכֵן דִּבְרֵי הָרַ"ן לְחַלֵּק בְּכָךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כב} סְבָרָתוֹ בָּזֶה כְּפֵרוּשׁ רַשִׁ"י דִּמְפָרֵשׁ זְרָעִים שֶׁלֹּא שָׁתוּ לִפְנֵי הַמּוֹעֵד שֶׁלֹּא הָיוּ מְלֻמָּדִים לִשְׁתּוֹת תָּמִיד לִפְנֵי הַמּוֹעֵד לֹא יַשְׁקֶה בַּמּוֹעֵד דְּהוֹאִיל וְלֹא מַשְׁקֶה לְהוּ תָּדִיר לִפְנֵי הַמּוֹעֵד לֹא הֲוֵי לְהוּ פְּסִידָא אִי לֹא מַשְׁקֶה לְהוּ בַּמּוֹעֵד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק א - משנה ד

צָדִין אֶת הָאִישׁוּת וְאֶת הָעַכְבָּרִים מִשְּׂדֵה הָאִילָן וּמִשְּׂדֵה הַלָּבָן, שֶׁלֹּא כְדַרְכּוֹ, בַּמּוֹעֵד וּבַשְּׁבִיעִית {כד}. וַחֲכָמִים אוֹמְרִים, מִשְּׂדֵה הָאִילָן כְּדַרְכּוֹ, וּמִשְּׂדֵה הַלָּבָן שֶׁלֹּא כְדַרְכּוֹ. וּמְקָרִין {כה} אֶת הַפִּרְצָה בַּמּוֹעֵד, וּבַשְּׁבִיעִית בּוֹנֶה כְדַרְכּוֹ:

 ר"ע מברטנורה  הָאִישׁוּת. תִּנְשֶׁמֶת. וְהִיא בְּרִיָּה שֶׁאֵין לָהּ עֵינַיִם, וּמַפְסֶדֶת אֶת הַשָּׂדוֹת: וּבִשְׂדֵה הַלָּבָן. שֶׁחוֹרְשִׁין וְזוֹרְעִין אוֹתוֹ לִתְבוּאָה וְאֵין בּוֹ אִילָנוֹת: בִּשְׂדֵה הָאִילָן כְּדַרְכּוֹ. דְּמַפְסְדֵי טְפֵי, וּלְהֶפְסֵד מְרֻבֶּה חָשְׁשׁוּ. לְפִיכָךְ חוֹפֵר גֻּמָּא כְּדַרְכּוֹ וְתוֹלֶה בָּהּ מְצוּדָה {כג}: שֶׁלֹּא כְדַרְכּוֹ. נוֹעֵץ שִׁפּוּד בַּקַּרְקַע בְּחֹזֶק וּמְנַעֲנֵעַ אֵילָךְ וָאֵילָךְ וּמוֹצִיא, וְנִמְצֵאת גֻּמָּא נַעֲשֵׂית מֵאֵלֶיהָ. אֲבָל לֹא עַל יְדֵי חֲפִירָה. וַהֲלָכָה כַּחֲכָמִים. וּבִשְׂדֵה לָבָן הַסְּמוּכָה לִשְׂדֵה אִילָן, חוֹפֵר כְּדַרְכּוֹ, שֶׁמָּא יֵצְאוּ מִשְּׂדֵה לָבָן וְיַחְרִיבוּ הָאִילָנוֹת: וּמְקָרִין אֶת הַפִּרְצָה. מְסַדְּרִין אֶת הָאֲבָנִים זוֹ עַל זוֹ כְּדֶרֶךְ בִּנְיָן, וְאֵינוֹ טָח בְּטִיט. וּבְפִרְצָה שֶׁל גִּנָּה מַיְרֵי, אֲבָל כֹּתֶל חָצֵר שֶׁנָּפַל, בּוֹנֶה כְּדַרְכּוֹ, מִפְּנֵי הַגַּנָּבִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {כג} וְהַגֻּמָּא אֵינוֹ אֶלָּא כְּדֵי שֶׁיִּתְקַבְּצוּ שָׁמָּה אִישׁוּת הַרְבֵּה, וְאֶפְשָׁר שֶׁנּוֹתֵן שָׁם אֳכָלִין כְּדֵי שֶׁיָּבֹאוּ וְיֵאָחְזוּ בַּמְּצוּדָה. הָרַ"ן: {כד} וּבַשְּׁבִיעִית. לְשׁוֹן רַשִׁ"י אַף עַל גַּב דִּמְתַקֵּן אֶת הַשָּׂדֶה. וּלְשׁוֹן הָרַ"ן דְּלֹא גָּזְרִינַן דִּלְמָא מֵחֲזֵי כַּחֲרִישָׁה כְּשֶׁאָדָם חוֹפֵר וְסוֹתֵם אֶת חֹרֵיהֶם: {כה} וּמְקָרִין. נִרְאֶה לִי שֶׁהוּא נִגְזַר מִשֵּׁם קִיר. וְלֹא תָּנֵי מְקִירִין, שֶׁלֹּא יִתְעָרֵב בְּמִלַּת מֵקִיר שֶׁעִנְיָנוֹ קְרִירוּת: (עיקר תוי"ט).

פרק א - משנה ה

רַבִּי מֵאִיר אוֹמֵר, רוֹאִין אֶת הַנְּגָעִים בַּתְּחִלָּה {כו} לְהָקֵל, אֲבָל לֹא לְהַחְמִיר. וַחֲכָמִים אוֹמְרִים, לֹא לְהָקֵל {כח} וְלֹא לְהַחְמִיר. וְעוֹד {כט} אָמַר רַבִּי מֵאִיר, מְלַקֵּט אָדָם עַצְמוֹת אָבִיו וְאִמּוֹ, מִפְּנֵי שֶׁשִּׂמְחָה הִיא לוֹ. רַבִּי יוֹסֵי אוֹמֵר, אֵבֶל הוּא לוֹ. לֹא יְעוֹרֵר אָדָם עַל מֵתוֹ וְלֹא יַסְפִּידֶנּוּ קֹדֶם לָרֶגֶל שְׁלֹשִׁים יוֹם:

 ר"ע מברטנורה  רוֹאִין אֶת הַנְּגָעִים בַּמּוֹעֵד לְהָקֵל. שֶׁאִם טָהוֹר הוּא אוֹמֵר לוֹ הַכֹּהֵן טָהוֹר אַתָּה, שֶׁשִּׂמְחָה הִיא לוֹ: אֲבָל לֹא לְהַחְמִיר. שֶׁאִם טָמֵא הוּא, יִשְׁתֹּק, וְלֹא יְטַמְּאֶנּוּ וְיַצְרִיכֶנּוּ לָצֵאת חוּץ לַמַּחֲנֶה {כז}: לֹא לְהָקֵל וְלֹא לְהַחְמִיר. מִתּוֹךְ שֶׁנִּזְקַק לִרְאוֹתוֹ אִם הוּא טָהוֹר כְּדֵי לְהָקֵל עָלָיו, הוּא נִזְקָק לְהַחְמִיר, שֶׁאִם הוּא טָמֵא צָרִיךְ לוֹמַר לוֹ טָמֵא אַתָּה, כְּדִכְתִיב (וַיִּקְרָא יג) לְטַהֲרוֹ אוֹ לְטַמְּאוֹ, שֶׁאֵין כֹּהֵן רַשַּׁאי לִשְׁתֹּק, וּמוּטָב לוֹ שֶׁלֹּא יִרְאֵהוּ הַכֹּהֵן כְּלָל: מְלַקֵּט אָדָם עַצְמוֹת אָבִיו וְאִמּוֹ. בַּמּוֹעֵד, כְּדֵי לְקָבְרָם בְּמָקוֹם הָגוּן: שֶׁשִּׂמְחָה הִיא לוֹ. כְּשֶׁרוֹאֶה אוֹתָם שֶׁקּוֹבְרָן בְּקִבְרֵי אֲבוֹתָיו. וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר בִּשְׁתֵּיהֶן: לֹא יְעַרְעֵר עַל מֵתוֹ. לֹא יָבִיא סַפְדָּן לְעַרְעֵר עַל מֵתוֹ שֶׁמֵּת זֶה יָמִים רַבִּים, לַחְזֹר עַל קְרוֹבֵי הַמֵּת כְּדֶרֶךְ שֶׁחוֹזֵר הַסַּפְדָּן שֶׁצּוֹעֵק וְאוֹמֵר בֹּאוּ בְּכוּ עִמִּי כָּל מָרֵי לֵב, וּמִי שֶׁלִּבּוֹ דָּוֶה עָלָיו הוֹלֵךְ וּמַסְפִּיד עַל קְרוֹבוֹ: לֹא יַסְפִּידֶנּוּ. לֹא יִשְׂכֹּר סַפְדָּן עַל מֵתוֹ שֶׁמֵּת לוֹ בְּאוֹתָהּ שָׁעָה: קֹדֶם לָרֶגֶל שְׁלֹשִׁים יוֹם. תָּמִיד {ל}. דְּאָמְרֵי טַעְמָא בַּגְּמָרָא שֶׁדֶּרֶךְ בְּנֵי אָדָם לִכְנֹס מָעוֹת לְצֹרֶךְ הָרֶגֶל קֹדֶם לָרֶגֶל שְׁלֹשִׁים יוֹם מִכִּי שָׁמַע שֶׁדּוֹרְשִׁים בְּהִלְכוֹת הָרֶגֶל, וְשֶׁמָּא יִתֵּן הַמָּעוֹת שֶׁכִּנֵּס לְצֹרֶךְ הָרֶגֶל לַסַּפְדָּן וְיִמָּנַע מִשִּׂמְחַת הָרֶגֶל. וְאִית דְּאָמְרֵי שֶׁאֵין הַמֵּת מִשְׁתַּכֵּחַ מִן הַלֵּב אֶלָּא לְאַחַר שְׁלֹשִׁים יוֹם אַחַר הֶסְפֵּדוֹ: (ר"ע מברטנורה).

 עיקר תוי"ט  {כו} בַּתְּחִלָּה. אֲפִלּוּ בַּתְּחִלָּה שֶׁעֲדַיִן לֹא נִזְקָק לְהֶסְגֵּר, דְּמִכָּל מָקוֹם שִׂמְחָה הִיא לוֹ כְּשֶׁאוֹמֵר לוֹ אֵינְךָ צָרִיךְ לַהֶסְגֵּר. וּמִלְּשׁוֹן רַשִׁ"י וְהָרַ"ב נִרְאֶה דְּלֹא גָּרְסִינַן בַּתְּחִלָּה. וְכֵן מַשְׁמָע בַּגְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כז} דְּאַף הַמֻּסְגָּר מִשְׁתַּלֵּחַ, וְהָא וַדַּאי דְּכֵיוָן דְּמִטַּמֵּא נַמִּי דְּגַם הוּא צַעַר לוֹ, מִכָּל מָקוֹם צַוְתָּא דְּאִינְשֵׁי עִקָּר צַעֲרָא הוּא: {כח} לֹא לְהָקֵל כוּ'. סְבִירָא לְהוּ נַמִּי דִּבְכֹהֵן תַּלְיָא מִלְּתָא, אֶלָּא מִדְּרָשָׁא דִּלְטַהֲרוֹ אוֹ לְטַמְּאוֹ נָפֵיק לְהוּ דְּמִכֵּיוָן שֶׁנִּזְקַק לִרְאוֹתוֹ אֵינוֹ רַשַּׁאי שֶׁיִּשְׁתֹּק: {כט} וְעוֹד. כְּלוֹמַר וְעוֹד מֵקֵל: {ל} וּרְצוֹנוֹ לוֹמַר בֵּין לְעַרְעֵר בֵּין לִסְפֹּד. וְלָשׁוֹן רַשִׁ"י אַתַּרְוַיְיהוּ קָאֵי. וְיוֹתֵר נִרְאֶה לִי שֶׁתֵּבַת תָּמִיד צָרִיךְ לוֹמַר אִית: (עיקר תוי"ט).

פרק א - משנה ו

אֵין חוֹפְרִין כּוּכִין וּקְבָרוֹת בַּמּוֹעֵד, אֲבָל מְחַנְּכִים אֶת הַכּוּכִין {לא} בַּמּוֹעֵד. וְעוֹשִׂין נִבְרֶכֶת בַּמּוֹעֵד, וְאָרוֹן עִם הַמֵּת בֶּחָצֵר. רַבִּי יְהוּדָה אוֹסֵר, אֶלָּא אִם כֵּן יֵשׁ עִמּוֹ נְסָרִים {לד}:

 ר"ע מברטנורה  אֵין חוֹפְרִין כּוּכִין וּקְבָרוֹת בַּמּוֹעֵד. לְהִקָּבֵר בּוֹ מֵת שֶׁיָּמוּת לְאַחַר הַמּוֹעֵד. כּוּכִין, בִּמְעָרָה שֶׁתַּחַת הַקַּרְקַע. וּקְבָרוֹת, בְּבִנְיָן: אֲבָל מְחַנְּכִין. שֶׁאִם הָיָה אָרֹךְ מְקַצְּרוֹ. וְאִם הָיָה קָצָר מַאֲרִיךְ וּמַרְחִיב בּוֹ: נִבְרֶכֶת. שֶׁל כּוֹבְסִין {לב}, מָקוֹם מְתֻקָּן לְכַבֵּס בּוֹ הַבְּגָדִים, וְלֵית בֵּיהּ טִרְחָא כֻּלֵּי הַאי: וְאָרוֹן עִם הַמֵּת בֶּחָצֵר. שֶׁמֵּת שָׁם, מֻתָּר לְנַסֵּר הַנְּסָרִים וְלַעֲשׂוֹת לוֹ אָרוֹן. אֲבָל בְּחָצֵר אַחֶרֶת לֹא, שֶׁלֹּא יֹאמְרוּ מְלָאכָה אַחֶרֶת הוּא עוֹשֶׂה {לג}. וְלֹא שָׁנוּ אֶלָּא בְּאָדָם שֶׁאֵינוֹ מְפֻרְסָם, אֲבָל אָדָם מְפֻרְסָם עוֹשִׂין לוֹ אָרוֹן וַאֲפִלּוּ בַּשּׁוּק. וּלְדִידָן שֶׁאָנוּ מְתֵי מִסְפָּר, כָּל אָדָם הֲוֵי מְפֻרְסָם: רַבִּי יְהוּדָה אוֹסֵר. לְהָבִיא עֵצִים לַעֲשׂוֹת נְסָרִים בַּתְּחִלָּה לְצֹרֶךְ הָאָרוֹן, אֶלָּא אִם כֵּן מְנֻסָּרוֹת קֹדֶם יוֹם טוֹב. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {לא} הַכּוּכִין. וְהוּא הַדִּין לְקֶבֶר. וְאַף עַל פִּי שֶׁאֵין צְרִיכִין לָהֶם עַכְשָׁיו עֳשָׂאוּם כְּצָרְכֵי רַבִּים, רַבִּים שָׁתוּ וְרַבִּים יִשְׁתּוּ, וַהֲרֵי הֵן כַּחֲטִיטַת בּוֹרוֹת שִׁיחִין וּמְעָרוֹת שֶׁאֵין צְרִיכִין לָהֶם עַכְשָׁיו שֶׁהִתִּירוּ בָּהֶן לְסוֹפָן דְּהַיְנוּ חֲטִיטָה אֲבָל לֹא תְּחִלָּתָם דְּהַיְינוּ חֲפִירָה כָּל שֶׁאֵין צְרִיכִין לָהֶם עַכְשָׁיו. הָרֹא"שׁ וְהָרַ"ן. וְשֵׁם כּוּךְ אֲרַמִּית, דְּכֶלַח דִּכְתִּיב בְּאִיּוֹב מְתַרְגֵּם כּוּךְ: {לב} לְפִי שֶׁיֵּשׁ בָּהּ צֹרֶךְ לְחֹל הַמּוֹעֵד כְּדִתְנַן וְאֵלּוּ מְכַבְּסִין בְּמוֹעֵד: {לג} לְפִי שֶׁאֵין דֶּרֶךְ לַעֲשׂוֹת אֲרוֹנוֹת לְרֹב מֵתִים. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לד} נְסָרִים. דְּטֹרַח גָּדוֹל לַעֲשׂוֹתָן בַּמּוֹעֵד. רַשִׁ"י: (עיקר תוי"ט).

פרק א - משנה ז

אֵין נוֹשְׂאִין נָשִׁים בַּמּוֹעֵד, לֹא בְתוּלוֹת, וְלֹא אַלְמָנוֹת, וְלֹא מְיַבְּמִין, מִפְּנֵי שֶׁשִּׂמְחָה הִיא לוֹ. אֲבָל מַחֲזִיר הוּא אֶת גְּרוּשָׁתוֹ. וְעוֹשָׂה אִשָּׁה תַּכְשִׁיטֶיהָ בַמּוֹעֵד. רַבִּי יְהוּדָה אוֹמֵר, לֹא תָּסוּד מִפְּנֵי שֶׁנִּוּוּל הוּא לָהּ:

 ר"ע מברטנורה  מִפְּנֵי שֶׁשִּׂמְחָה הִיא לוֹ. וְאֵין מְעָרְבִים שִׂמְחָה אַחֶרֶת בְּשִׂמְחַת הָרֶגֶל, דִּכְתִיב וְשָׂמַחְתָּ בְּחַגֶּךָ (דְּבָרִים טז), בְּחַגֶּךָ וְלֹא בְּאִשְׁתְּךָ {לה}: אֲבָל מַחְזִיר הוּא אֶת גְּרוּשָׁתוֹ. דְּאֵינָהּ שִׂמְחָה כָּל כָּךְ אֶלָּא בְּאִשָּׁה חֲדָשָׁה שֶׁלֹּא הָיְתָה לוֹ מֵעוֹלָם: וְעוֹשָׂה אִשָּׁה תַּכְשִׁיטֶיהָ. כְּגוֹן לָתֵת כָּחָל בְּעֵינֶיהָ, וּלְהַשְׁווֹת שְׂעַר רֹאשָׁהּ שֶׁלֹּא יִתְפַּזֵּר, וּמַעֲבֶרֶת צֶבַע אָדֹם עַל פָּנֶיהָ, וּמַעֲבֶרֶת סַכִּין עַל פָּנֶיהָ שֶׁל מַטָּה, וְכֵן כָּל כַּיּוֹצֵא בָּזֶה: שֶׁנִּוּוּל הוּא לָהּ. צַעַר הוּא לָהּ {לו}. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {לה} וּלְשׁוֹן מְעָרְבִין לֹא נִיחָא לְפוּם כֵּן, דַּהֲוָה לֵיהּ לְמֵימַר אֵין עוֹשִׂין שִׂמְחָה אַחֶרֶת בָּרֶגֶל כוּ'. וּבַגְּמָרָא אִיכָּא מַאן דְּאָמַר מִפְּנֵי שֶׁמַּנִּיחַ שִׂמְחַת הָרֶגֶל וְעוֹסֵק בְּשִׂמְחַת אִשְׁתּוֹ, וְאִיכָּא מַאן דְּאָמַר דְּאֵין מְעָרְבִין כוּ', וְלָמְדוּ מִקְּרָא אַחְרִינָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לו} כְּלוֹמַר שֶׁמִּצְטַעֶרֶת עַל הַנִּוּוּל שֶׁבּוֹ הִיא מִתְנַוֶּלֶת: (עיקר תוי"ט).

פרק א - משנה ח

הַהֶדְיוֹט תּוֹפֵר כְּדַרְכּוֹ {לז}, וְהָאֻמָּן מַכְלִיב. וּמְסָרְגִין אֶת הַמִּטּוֹת. רַבִּי יוֹסֵי אוֹמֵר, אַף מְמַתְּחִין:

 ר"ע מברטנורה  הַהֶדְיוֹט תּוֹפֵר כְּדַרְכּוֹ. הֶדְיוֹט מִקְּרֵי כָּל שֶׁאֵינוֹ יוֹדֵעַ לְכַוֵּן שָׂפָה לַבֶּגֶד שֶׁתְּהֵא שָׁוָה וּמְכֻוֶּנֶת בּוֹ, אֶלָּא מְעַקֵּם לְכָאן וּלְכָאן וְעוֹשֵׂהוּ רָחָב בְּמָקוֹם זֶה וְקָצָר בְּמָקוֹם אַחֵר: מַכְלִיב. עוֹשֶׂה תְּפִירוֹת כְּעֵין שִׁנֵּי הַכֶּלֶב, שֶׁאֵינָן שָׁוִין זוֹ כְּנֶגֶד זוֹ אֶלָּא אַחַת לְמַעְלָה וְאַחַת לְמַטָּה: מְסָרְגִין אֶת הַמִּטּוֹת. אוֹרְגִין אוֹתָן בַּחֲבָלִין שְׁתִי וָעֵרֶב: מְמַתְּחִין. אִם הָיָה אָרוּג זֶה יָמִים רַבִּים וְרָפוּ הַחֲבָלִים, מְמַתְּחִין, אֲבָל לֹא מְסָרְגִין. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: (ר"ע מברטנורה).

 עיקר תוי"ט  {לז} כְּדַרְכּוֹ. לְצֹרֶךְ הַמּוֹעֵד, דִּבְחִנָּם לָמָּה יַתִּירוּ אֲפִלּוּ עַל יְדֵי שִׁנּוּי. תּוֹסָפוֹת: (עיקר תוי"ט).

פרק א - משנה ט

מַעֲמִידִין תַּנּוּר וְכִירַיִם וְרֵחַיִם בַּמּוֹעֵד. רַבִּי יְהוּדָה אוֹמֵר, אֵין מְכַבְּשִׁין אֶת הָרֵחַיִם בַּתְּחִלָּה {מ}:

 ר"ע מברטנורה  מַעֲמִידִין. בּוֹנִין וּמוֹשִׁיבִין וּמְתַקְּנִים תַּנּוּר {לח} שֶׁצָּרִיךְ לוֹ בַּמּוֹעֵד. וּלְפִי שֶׁתַּנּוּרִים שֶׁלָּהֶם מִטַּלְטְלִין הָיוּ מִמָּקוֹם לְמָקוֹם, לְהָכִי תָּנֵי מַעֲמִידִין: אֵין מְכַבְּשִׁין. מְנַקְּרִין {לט} וּמַכִּין בְּפַטִּישׁ עַל הָרֵחַיִם כְּשֶׁהֵן חֲלָקוֹת יוֹתֵר מִדַּאי וְאֵין טוֹחֲנוֹת יָפֶה. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {לח} לְשׁוֹן הָרַמְבַּ"ם, כְּלוֹמַר בּוֹנִין עַל חֶרֶס שֶׁל הַתַּנּוּר וְהַכִּירָה הַטְּפֵלָה שֶׁלָּהֶם. וּלְשׁוֹן הָרֹא"שׁ, הַיְינוּ שֶׁמְּתַקְּנִין אוֹתוֹ יָפֶה עַל עָמְדוֹ כְּדֵי לְהִשְׁתַּמֵּשׁ בָּהֶם. וְהָרַ"ב מְפָרֵשׁ לִשְׁנֵיהֶם, דִּלְדִינָא שְׁנֵיהֶם אֱמֶת: {לט} וְנִרְאֶה לִי שֶׁהוּא מִלָּה אֲרַמִּית, בַּמְּסִלָּה נַעֲלֶה [בְּמִדְבָּר כ] תִּרְגֵּם בְּאֹרַח כְּבִישָׁא נִיסַק: {מ} בַּתְּחִלָּה. פֵּרוּשׁ, חֲדָשָׁה. אֲבָל יְשָׁנָה מְכַבְּשִׁין. הָרֹא"שׁ: (עיקר תוי"ט).

פרק א - משנה י

עוֹשִׂין מַעֲקֶה לְגַג וּלְמִרְפֶּסֶת מַעֲשֵׂה הֶדְיוֹט {מא}, אֲבָל לֹא מַעֲשֵׂה אֻמָּן. שָׁפִין אֶת הַסְּדָקִין וּמַעְגִּילִין אוֹתָן בַּמַּעְגִּילָה בַּיָּד וּבָרֶגֶל, אֲבָל לֹא בַמַּחֲלָצָיִם. הַצִּיר וְהַצִּנּוֹר וְהַקּוֹרָה וְהַמַּנְעוּל וְהַמַּפְתֵּחַ שֶׁנִּשְׁבְּרוּ, מְתַקְּנָן בַּמּוֹעֵד {מג}, וּבִלְבַד שֶׁלֹּא יְכַוֵּן מְלַאכְתּוֹ בַּמּוֹעֵד, וְכָל כְּבָשִׁין שֶׁהוּא יָכוֹל לֶאֱכֹל מֵהֶן בַּמּוֹעֵד {מד}, כּוֹבְשָׁן:

 ר"ע מברטנורה  וּלְמִרְפֶּסֶת. הִלּוּךְ שֶׁלִּפְנֵי עֲלִיָּה שֶׁבְּנֵי הָעֲלִיָּה הוֹלְכִים בּוֹ וְיוֹרְדִים דֶּרֶךְ סֻלָּם לֶחָצֵר: וְשָׁפִין אֶת הַסְּדָקִין. גַּגּוֹת שֶׁאֵינָן מְשֻׁפָּעִין טוֹחִין אוֹתָן בַּטִּיט, וּכְשֶׁיֵּשׁ סְדָקִים בִּטְחִיַּת הַטִּיט הַדֶּלֶף יוֹרֵד בַּבַּיִת וְאִיכָּא פְּסֵידָא. שָׁפִין, טָחִין: מַעֲגִילָה. עֵץ עָגֹל שֶׁבּוֹ מַחֲלִיקִין הַגַּגּוֹת {מב}: בְּמַחֲלָצָיִם. כְּלִי אֻמָּנוּת שֶׁל בַּרְזֶל רְחָבָה כְּעֵין כַּף, וְלָהּ בֵּית יָד: הַצִּיר. רֶגֶל הַדֶּלֶת הַסּוֹבֶבֶת בְּתוֹךְ חֹר הָאַסְקֻפָּה הַתַּחְתּוֹנָה שֶׁל פֶּתַח: צִנּוֹר. הַחֹר שֶׁבָּאַסְקֻפָּה: וְהַקּוֹרָה. קוֹרוֹת הַבַּיִת מַמָּשׁ שֶׁנִּשְׁבְּרוּ אֲפִלּוּ קֹדֶם יוֹם טוֹב: וּבִלְבַד שֶׁלֹּא יְכַוֵּן מְלַאכְתּוֹ בַּמּוֹעֵד. שֶׁלֹּא יֹאמַר אַמְתִּין עַד הַמּוֹעֵד שֶׁאֵין לִי מְלָאכָה אַחֶרֶת וְאֶעֱשֶׂה אֶת זֹאת: כְּבָשִׁים. כְּגוֹן דָּגִים וִירָקוֹת וְכַיּוֹצֵא בָּהֵן שֶׁכּוֹבְשִׁין בְּמֶלַח וְחֹמֶץ: שֶׁיָּכוֹל לְאָכְלָן בַּמּוֹעֵד. שֶׁיְּהוּ נִכְבָּשִׁין מַהֵר וּרְאוּיִין לֶאֱכֹל מִיָּד: (ר"ע מברטנורה).

 עיקר תוי"ט  {מא} מַעֲשֵׂה הֶדְיוֹט. כְּמוֹ שֶׁקָּדַם זִכְרוֹ בְּמִשְׁנָה ד' שֶׁיִּתֵּן הָאֲבָנִים זוֹ עַל גַּב זוֹ בְּלֹא טִיט. הָרַמְבַּ"ם: {מב} וּבַגְּמָרָא, הַשְׁתָּא בַּמַּעְגִּילָה אָמַרְתְּ שָׁרֵי, בַּיָּד וּבָרֶגֶל מִבָּעֲיָא. הָכִי קָאָמַר, וּמַעֲגִילִין אוֹתָן כְּעֵין מַעְגִּילָה בַּיָּד וּבָרֶגֶל. וּפֵרֵשׁ רַשִׁ"י וְלֹא מַעְגִּילָה מַמָּשׁ אֶלָּא כְּעֵין כוּ': {מג} מְתַקְּנָן בַּמּוֹעֵד. מַסְקִינַן בַּגְּמָרָא כְּרַבִּי יוֹסֵי דְּרֵישׁ פֶּרֶק ב' (וּכְרַבִּי יְהוּדָה בְּמַעֲשֵׂר שֵׁנִי פ"ה מט"ו), וּלְכָךְ מְתַקְּנוֹ כְּדַרְכּוֹ בֵּין בְּשֶׁל עֵץ בֵּין בְּשֶׁל בַּרְזֶל. וְכָתַב הָרַמְבַּ"ם שֶׁזֶּה הֶפְסֵד גָּדוֹל הוּא, שֶׁאִם יַנִּיחַ הַפֶּתַח פָּתוּחַ וּדְלָתוֹת שְׁבוּרוֹת נִמְצָא מְאַבֵּד כָּל מַה שֶּׁבַּבַּיִת: {מד} בַּמּוֹעֵד. דְּאִי אֵין יָכֹל לְאָכְלָן לֹא, דְּכוֹבֵשׁ אָסוּר לְהָרַמְבַּ"ם מִפְּנֵי שֶׁהוּא כִּמְבַשֵּׁל. וּלְרַשִׁ"י וְהָרַ"ב מִשּׁוּם מְעַבֵּד. נִרְאֶה לִי. תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ב - משנה א

מִי שֶׁהָפַךְ אֶת זֵיתָיו וְאֵרְעוֹ אֵבֶל אוֹ אֹנֶס, אוֹ שֶׁהִטְעוּהוּ פוֹעֲלִים, טוֹעֵן קוֹרָה רִאשׁוֹנָה וּמַנִּיחָהּ לְאַחַר הַמּוֹעֵד. דִּבְרֵי רַבִּי יְהוּדָה. רַבִּי יוֹסֵי אוֹמֵר, זוֹלֵף וְגוֹמֵר וְגָף כְּדַרְכּוֹ:

 ר"ע מברטנורה  מִי שֶׁהָפַךְ. דֶּרֶךְ בְּנֵי אָדָם שֶׁמְּהַפְּכִין הַזֵּיתִים בְּתוֹךְ הַמַּעֲטָן שֶׁהֵם צְבוּרִים בּוֹ, כְּשֶׁרוֹצִים לְהַכְנִיסָן לְבֵית הַבַּד. וְאִי לֹא מַסִּיק לְהוּ לַבַּד לִטְחֹן לְאַחַר שֶׁהֲפָכָן פָּסְדֵי: וְאֵרְעוֹ אֵבֶל. שֶׁאָסוּר בִּמְלָאכָה: אוֹ שֶׁהִטְעוּהוּ פוֹעֲלִים. שֶׁלֹּא בָּאוּ לִמְלַאכְתּוֹ. וְהִגִּיעַ הַמּוֹעֵד: טוֹעֵן. עַל הַזֵּיתִים קוֹרָה שֶׁל בֵּית הַבַּד פַּעַם רִאשׁוֹנָה וּמַכְבִּיד, כְּדֵי שֶׁיָּזוּבוּ קְצָת וְלֹא יַפְסִידוּ, אֲבָל לֹא בְּפַעַם שְׁנִיָּה מִשּׁוּם דְּתוּ לֹא פָּסְדֵי. וּמֵהָכָא שָׁמְעִינַן דִּדְבָרִים הַמֻּתָּרִים בַּמּוֹעֵד אָסוּר לָאָבֵל לַעֲשׂוֹתָם בְּיָדוֹ בִּימֵי אֶבְלוֹ, מִדְּקָתָנֵי וְאֵרְעוֹ אֵבֶל וּמֵחֲמַת הָאֵבֶל עִכֵּב עַד הַמּוֹעֵד {א} טוֹעֵן קוֹרָה רִאשׁוֹנָה בַּמּוֹעֵד, אֲבָל בִּימֵי אֶבְלוֹ הָיָה אָסוּר לִטְעֹן בְּיָדוֹ אֲפִלּוּ קוֹרָה רִאשׁוֹנָה. אֲבָל אֲחֵרִים טוֹעֲנִים לוֹ: זוֹלֵף. שׁוֹפֵךְ הַזֵּיתִים עַל גַּבֵּי בֵּית הַבַּד. וְכָל מִידֵי דְּשָׁפִיךְ קָרוּי זִלּוּף: וְגוֹמֵר. סְחִיטָתָם וְעִצּוּרָם בְּבֵית הַבַּד: וְגָף. סוֹתֵם פִּי הֶחָבִיּוֹת {ב}. וַהֲלָכָה כְּרַבִּי יוֹסֵי: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} כְּלוֹמַר דְּפָתַח [בְּמַתְנִיתִין] בְּאָבֵל, וְלֹא פֵּרֵשׁ אִם יָכֹל לִטְעֹן קוֹרָה רִאשׁוֹנָה בִּימֵי אֶבְלוֹ, אֶלָּא מִלֵּי דְּמוֹעֵד קָא פָּרֵישׁ, דְּאִם אֵרַע לוֹ אֹנֶס קֹדֶם הַמּוֹעֵד דְּטוֹעֵן קוֹרָה רִאשׁוֹנָה. גְּמָרָא: {ב} וְהוּא מִלְּשׁוֹן מְגֻפָה. וְכָתְבוּ הַתּוֹסָפוֹת יֵשׁ לוֹמַר דְּצָרִיךְ לְכַסּוֹתוֹ מִפְּנֵי הַשְּׁרָצִים. וּמִיהוּ מְגֻפָה לֹא הָיָה צָרִיךְ, דְּדַיּוֹ בְּכִסּוּי מֻעָט, וְשֶׁמָּא חָדָשׁ מִתְקַלְקֵל: (עיקר תוי"ט).

פרק ב - משנה ב

וְכֵן מִי שֶׁהָיָה יֵינוֹ {ג} בְּתוֹךְ הַבּוֹר וְאֵרְעוֹ אֵבֶל אוֹ אֹנֶס, אוֹ שֶׁהִטְעוּהוּ פוֹעֲלִים, זוֹלֵף {ד} וְגוֹמֵר וְגָף כְּדַרְכּוֹ, דִּבְרֵי רַבִּי יוֹסֵי. רַבִּי יְהוּדָה אוֹמֵר, עוֹשֶׂה לוֹ לִמּוּדִים, בִּשְׁבִיל שֶׁלֹּא יַחְמִיץ:

 ר"ע מברטנורה  בְּתוֹךְ הַבּוֹר. חֲפִירָה שֶׁתַּחַת הַגַּת מְסֻיֶּדֶת בְּסִיד לְקַבֵּל הַיַּיִן: לִמּוּדִים. נְסָרִים שֶׁל עֵץ. כְּלוֹמַר מְכַסֶּה אוֹתוֹ שָׁם בִּנְסָרִים שֶׁלֹּא יַחְמִיץ, וְלֹא יִשְׁלֵהוּ מִן הַבּוֹר. וַהֲלָכָה כְּרַבִּי יוֹסֵי: (ר"ע מברטנורה).

 עיקר תוי"ט  {ג} יֵינוֹ. וּצְרִיכֵי, דְּאִי אַשְׁמוֹעִינַן קַמַּיְתָא, בְּהַהוּא קָאָמַר רַבִּי יוֹסֵי מִשּׁוּם דְּמִשְׁחָא יַקִּיר וְאִיכָּא פְּסֵידָא יְתֵירָא, אֲבָל חַמְרָא דְּלֹא נָפִישׁ פְּסֵידֵיהּ אֵימָא מוֹדֶה לְרַבִּי יְהוּדָה. וְאִי אַשְׁמוֹעִינַן בַּתְרַיְתָא, בְּהַהוּא קָאָמַר רַבִּי יְהוּדָה כוּ'. גְּמָרָא: {ד} זוֹלֵף. כְּלוֹמַר מֵרִיק אוֹתוֹ בֶּחָבִיוֹת. רַשִׁ"י: (עיקר תוי"ט).

פרק ב - משנה ג

מַכְנִיס אָדָם פֵּרוֹתָיו מִפְּנֵי הַגַּנָּבִים, וְשׁוֹלֶה פִשְׁתָּנוֹ מִן הַמִּשְׁרָה בִּשְׁבִיל שֶׁלֹּא תֹאבַד, וּבִלְבַד {ה} שֶׁלֹּא יְכַוֵּן אֶת מְלַאכְתּוֹ בַמּוֹעֵד. וְכֻלָּן אִם כִּוְּנוּ מְלַאכְתָּן בַּמּוֹעֵד, יֹאבֵדוּ:

 ר"ע מברטנורה  וְשׁוֹלֶה פִשְׁתָּנוֹ. כָּל מַה שֶּׁמַּעֲלֶה מִן הַמַּיִם קָרוּי שׁוֹלֶה: אִם כִּוְּנוּ מְלַאכְתָּן בַּמּוֹעֵד יֹאבֵדוּ. בֵּית דִּין מְאַבְּדִים אוֹתוֹ מָמוֹן, וְעוֹשִׂים אוֹתוֹ הֶפְקֵר {ו}: (ר"ע מברטנורה).

 עיקר תוי"ט  {ה} וּבִלְבַד כוּ'. שֶׁיֵּשׁ לוֹ פְּנַאי בִּשְׁאָר יְמוֹת הַשָּׁנָה וְהוּא מַשְׁהֶה לַמּוֹעֵד. רַשִׁ"י: {ו} וְכֵן פֵּרֵשׁ הָרַמְבַּ"ם. וּלְשׁוֹן רַשִׁ"י שֶׁאָסוּר לֶהֱנוֹת מֵהֶן. וְהַגְּאוֹנִים מְפָרְשִׁים מַנִּיחִין אוֹתוֹ לֵאָבֵד וּמוֹנְעִין אוֹתוֹ מִלַּעֲשׂוֹת אוֹתָהּ מְלָאכָה, אֲבָל לְהַפְקִירָהּ לַכֹּל וּלְאַבְּדָהּ בַּיָּדַיִם לֹא קָנְסוּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב שֶׁהֵבִיא רְאָיָה לְפֵרוּשׁ הָרַמְבַּ"ם: (עיקר תוי"ט).

אֵין לוֹקְחִין בָּתִּים, עֲבָדִים וּבְהֵמָה, אֶלָּא לְצֹרֶךְ הַמּוֹעֵד {ז}, אוֹ לְצֹרֶךְ הַמּוֹכֵר, שֶׁאֵין לוֹ מַה יֹּאכָל {ח}. אֵין מְפַנִּין מִבַּיִת לְבַיִת, אֲבָל מְפַנֶּה הוּא לַחֲצֵרוֹ. אֵין מְבִיאִין כֵּלִים מִבֵּית הָאֻמָּן {יא}. וְאִם חוֹשֵׁשׁ לָהֶם, מְפַנָּן לְחָצֵר אַחֶרֶת:

 ר"ע מברטנורה  אֶלָּא לְצֹרֶךְ הַמּוֹעֵד. שֶׁהַקּוֹנֶה צָרִיךְ לָהֶם בַּמּוֹעֵד, בָּתִּים לֵישֵׁב בָּהֶם, עֲבָדִים לְשַׁמְּשׁוֹ, בְּהֵמָה לִשְׁחִיטָה: אוֹ לְצֹרֶךְ הַמּוֹכֵר. שֶׁצָּרִיךְ לְמָעוֹת לְהוֹצִיאָן לְצֹרֶךְ יוֹם טוֹב: אֵין מְפַנִּים. כֵּלִים אוֹ תְּבוּאָה {ט}: מִבַּיִת. זֶה לְבַיִת אַחֵר רָחוֹק מִמֶּנּוּ, מִפְּנֵי הַטֹּרַח: אֲבָל מְפַנֶּה לַחֲצֵרוֹ. לְבַיִת אַחֵר שֶׁבְּאוֹתוֹ חָצֵר, דְּצִנְעָה הֲוֵי. אֲבָל מִבַּיִת לְבַיִת דֶּרֶךְ רְשׁוּת הָרַבִּים לֹא {י}: וְאֵין מְבִיאִין כֵּלִים מִבֵּית הָאֻמָּן. וְהַנֵּי מִילֵי כֵּלִים שֶׁאֵינָן לְצֹרֶךְ הַמּוֹעֵד, אֲבָל כֵּלִים שֶׁהֵן לְצֹרֶךְ הַמּוֹעֵד כְּגוֹן כָּרִים וּכְסָתוֹת וְכוֹסוֹת וְקִיתוֹנִיּוֹת שָׁרֵי: וְאִם חוֹשֵׁשׁ לָהֶם. שֶׁמָּא יִגָּנְבוּ מִבֵּית הָאֻמָּן, אוֹ אִם צָרִיךְ הָאֻמָּן לִשְׂכָרוֹ וְאֵין בַּעַל הַכְּלִי מַאֲמִינוֹ כִּי יָרֵא שֶׁמָּא יִשְׁאַל מִמֶּנּוּ שְׂכַר פְּעֻלָּתוֹ פַּעַם שְׁנִיָּה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ז} הַמּוֹעֵד. פֵּרְשׁוּ הַתּוֹסְפוֹת דִּרְבוּתָא קָא מַשְׁמָע לָן, דַּאֲפִלּוּ הָנֵי דַּהֲוְיַן בְּפַרְהֶסְיָא שָׁרֵי לְצֹרֶךְ הַמּוֹעֵד, אֲבָל כָּל שֶׁאֵינוֹ לְצֹרֶךְ הַמּוֹעֵד הוּא הַדִּין שְׁאָר דְּבָרִים דְּאָסוּר, דְּהָא בַּמִּשְׁנָה דִּלְקַמָּן אָסְרוּ לִמְכֹּר פֵּרוֹת אֶלָּא לְצֹרֶךְ הַמּוֹעֵד. וְצָרִיךְ עִיּוּן אִי צָרִיךְ בְּהַאי צִנְעָא כְּמוֹ בְּהַהִיא דְּפֵרוֹת דִּלְקַמָּן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} מַה יֹּאכַל. מָצִינַן לְמֵימַר דְּפֵרוּשֵׁי קָא מְפָרֵשׁ. וּמִיהוּ בַּגְּמָרָא דִּיֵּק דְּהָכִי קָאָמַר, אִי נַמִּי מִי שֶׁאֵין לוֹ מַה יֹּאכַל מֻתָּר שֶׁיַּשְׂכִּיר עַצְמוֹ אַף עַל פִּי שֶׁאֵין הַמְּלָאכָה לְצֹרֶךְ הַמּוֹעֵד, כְּדֵי לְהַחְיוֹת אֶת נַפְשׁוֹ. הָרַ"ן: {ט} שֶׁאֵינָן לְצֹרֶךְ הַמּוֹעֵד. טוּר. וְטַעְמָא נִרְאֶה מִשּׁוּם טִרְחָא וְאָתֵי לְאִמְנוּעֵי מִשִּׂמְחַת יוֹם טוֹב: {י} וְהַיְנוּ רֵישָׁא, וּכְבָר פֵּרְשָׁהּ, אֶלָּא דִּלְעֵיל נָקַט סְתָם רָחוֹק, כִּלְשׁוֹן הָרַ"ן בַּמִּשְׁנָה, וְהָכָא נָקַט דֶּרֶךְ רְשׁוּת הָרַבִּים, כְּמוֹ שֶׁמְּבָאֵר הָרַ"ן בַּגְּמָרָא: {יא} אֵין מְבִיאִין כוּ'. שֶׁיַּחְשְׁבוּ הָרוֹאִים שֶׁבַּעַל הַבַּיִת נְתָנָם לְתַקֵּן אוֹתָם בַּמּוֹעֵד, וְיֹאמְרוּ שֶׁהָאֻמָּן תִּקְּנָם בַּמּוֹעֵד. הָרַ"ן: (עיקר תוי"ט).

מְחַפִּין אֶת הַקְּצִיעוֹת בְּקַשׁ. רַבִּי יְהוּדָה אוֹמֵר, אַף מְעַבִּין. מוֹכְרֵי פֵרוֹת, כְּסוּת וְכֵלִים, מוֹכְרִים בְּצִנְעָה {יב} לְצֹרֶךְ הַמּוֹעֵד. הַצַּיָּדִין {יג} וְהַדָּשׁוֹשׁוֹת וְהַגָּרוֹסוֹת, עוֹשִׂין בְּצִנְעָה {יד} לְצֹרֶךְ הַמּוֹעֵד. רַבִּי יוֹסֵי אוֹמֵר, הֵם הֶחְמִירוּ עַל עַצְמָן:

 ר"ע מברטנורה  מְחַפִּין אֶת הַקְּצִיעוֹת. תְּאֵנִים שְׁטוּחִין בַּשָּׂדֶה לְיַבְּשָׁן, מְחַפִּין וּמְכַסִּין אוֹתָן בְּקַשׁ מִפְּנֵי הַגְּשָׁמִים: אַף מְעַבִּין. שֶׁנּוֹתְנִין אוֹתָן זוֹ עַל גַּבֵּי זוֹ וְעוֹשִׂין אוֹתָן כְּמִין כְּרִי, שֶׁיִּהְיוּ הָעֶלְיוֹנוֹת מְגִנּוֹת עַל הַתַּחְתּוֹנוֹת: הַדָּשׁוֹשׁוֹת. שֶׁדָּשִׁים וְכוֹתְשִׁים חִטִּים לְדַיְסָא: הַגָּרוֹסוֹת. שֶׁעוֹשִׂים גְּרִיסִין שֶׁל פּוֹל: הֵם. הָאֻמָּנִין: הֶחְמִירוּ עַל עַצְמָן. וְאֵין עוֹשִׂין אֲפִלּוּ בְּצִנְעָה. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: (ר"ע מברטנורה).

 עיקר תוי"ט  {יב} בְּצִנְעָה. וְהַטַּעַם שֶׁכָּל שֶׁאֻמָּנוּתוֹ בְּכָךְ כָּל הַשָּׁנָה, בָּעֵי צִנְעָה אַף עַל פִּי שֶׁהוּא לְצֹרֶךְ הַמּוֹעֵד. וּפֵרֵשׁ בְּצִנְעָה בַּגְּמָרָא, אִם הָיָה חֲנוּתוֹ פָּתוּחַ לִרְשׁוּת הָרַבִּים פּוֹתֵחַ חֶצְיוֹ וּמַנִּיחַ חֶצְיוֹ נָעוּל. וְאִם פָּתוּחַ לִסְטָיו פּוֹתֵחַ כְּדַרְכּוֹ כוּ': {יג} הַצַּיָּדִין. צַיָּדֵי [חַיָּה] עוֹפוֹת וְדָגִים. הָרַ"ן: {יד} בְּצִנְעָה. שֶׁמָּא יֹאמְרוּ זֶה לוֹקֵחַ שֶׁלֹּא לְצֹרֶךְ הַמּוֹעֵד. רַשִׁ"י. וְקָשֶׁה לָמָּה נָקְטִינַן אֵלּוּ. וְהָרַמְבַּ"ם לֹא הִצְרִיךְ לַעֲשׂוֹתָן בְּצִנְעָה אֶלָּא הָאֻמָּנִים שֶׁעוֹשִׂין מְלָאכוֹת לָרַבִּים, וּפֵרֵשׁ הַבֵּית יוֹסֵף דְּהַטַּעַם מִפְּנֵי שֶׁהֵן צְרִיכִין לַעֲשׂוֹת הַרְבֵּה בְּיַחַד וְנִרְאֶה כְּאִלּוּ הֵן עוֹשִׂין אוֹתָהּ לְצֹרֶךְ חֹל: (עיקר תוי"ט).

פרק ג - משנה א

וְאֵלּוּ מְגַלְּחִין בַּמּוֹעֵד, הַבָּא מִמְּדִינַת הַיָּם, וּמִבֵּית הַשִּׁבְיָה, וְהַיּוֹצֵא מִבֵּית הָאֲסוּרִין, וְהַמְנֻדֶּה שֶׁהִתִּירוּ לוֹ חֲכָמִים, וְכֵן מִי שֶׁנִּשְׁאַל לְחָכָם וְהֻתַּר, וְהַנָּזִיר, וְהַמְּצֹרָע הָעוֹלֶה מִטֻּמְאָתוֹ לְטָהֳרָתוֹ:

 ר"ע מברטנורה  וְאֵלּוּ מְגַלְּחִין. הַבָּא מִמְּדִינַת הַיָּם. בַּמּוֹעֵד. שֶׁלֹּא הָיָה לוֹ פְּנַאי לְגַלֵּחַ קֹדֶם הַמּוֹעֵד {א}. וְהוּא שֶׁפָּרַשׂ לִסְחוֹרָה אוֹ לְדָבָר הַצָּרִיךְ, אֲבָל אִם לֹא פָּרַשׂ אֶלָּא לְטִיּוּל בְּעָלְמָא אָסוּר: וְהַיּוֹצֵא מִבֵּית הָאֲסוּרִין. וַאֲפִלּוּ חָבוּשׁ בְּיַד יִשְׂרָאֵל, שֶׁהָיָה מַנִּיחוֹ לְגַלֵּחַ, לְפִי שֶׁהָיָה בְּצַעַר: וּמְנֻדֶּה. שֶׁהִתִּירוּ לוֹ נִדּוּיוֹ בַּמּוֹעֵד. וְקֹדֶם הַמּוֹעֵד לֹא הָיָה יָכֹל לְגַלֵּחַ, דִּמְנֻדֶּה אָסוּר בְּתִסְפֹּרֶת: וּמִי שֶׁנָּדַר. שֶׁלֹּא לְגַלֵּחַ, וְלֹא מָצָא חָכָם שֶׁיַּתִּיר לוֹ נִדְרוֹ קֹדֶם הַמּוֹעֵד. אִי נַמִּי לֹא מָצָא פֶּתַח לַחֲרָטָה אֶלָּא בַּמּוֹעֵד {ב}: וְהַנָּזִיר. שֶׁהִשְׁלִים נְזִירוּתוֹ בַּמּוֹעֵד: הָכִי גָּרְסִינַן, וְהַמְּצֹרָע הָעוֹלֶה מִטֻּמְאָתוֹ לְטָהֳרָתוֹ. אִם חָל שְׁבִיעִי שֶׁלּוֹ בַּמּוֹעֵד מֻתָּר בְּתִסְפֹּרֶת, דִּכְתִיב (וַיִּקְרָא יד) בַּיּוֹם הַשְּׁבִיעִי יְגַלַּח אֶת כָּל שְׂעָרוֹ {ג}. וְגִלּוּחַ דְּנֶתֶק לֹא חָשִׁיב, שֶׁאֵינוֹ אֶלָּא דָּבָר מֻעָט שֶׁמְּגַלֵּחַ סְבִיב הַנֶּתֶק. וְטַעְמָא דְּאָסוּר לְכָל אָדָם לְגַלֵּחַ בַּמּוֹעֵד חוּץ מֵאֵלּוּ הַשְּׁנוּיִים בַּמִּשְׁנָה, כְּדֵי שֶׁלֹּא יְכַוְּנוּ {ד} לְגַלֵּחַ בְּתוֹךְ הַמּוֹעֵד שֶׁהֵם בְּטֵלִים בּוֹ מִמְּלָאכָה, וְיָבֹא יוֹם טוֹב רִאשׁוֹן שֶׁל רֶגֶל כְּשֶׁהֵם מְנֻוָּלִים. וְהַיְינוּ טַעְמָא נַמִּי דְּאָסְרוּ הַכִּבּוּס בַּמּוֹעֵד: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} כְּגוֹן שֶׁלֹּא בָּא לַיִּשּׁוּב עַד חֹל הַמּוֹעֵד, אוֹ שֶׁלֹּא הָיָה לוֹ שְׁהוּת לְגַלֵּחַ קֹדֶם הָרֶגֶל כִּי כְּשֶׁבָּא לֹא הָיָה לוֹ שְׁהוּת בַּיּוֹם. בֵּית יוֹסֵף: {ב} וּבְרֵישׁ פֶּרֶק ט' דִּנְדָרִים כָּתַב דְּקָיְמָא לָן דְּאֵינוֹ צָרִיךְ לִפְתֹּחַ כְּשֶׁמִּתְחָרֵט מֵעַצְמוֹ, וְכֵיוָן שֶׁכֵּן נִמְצָא כִּי לֹא נִתְחָרֵט קֹדֶם הָרֶגֶל דִּבְמִתְכַּוֵּן לֹא גִּלֵּחַ וְלִתְּסַר, וְאִם כֵּן לֹא הֲוָה לֵיהּ לְהָרַ"ב לִכְתֹּב טַעַם זֶה. וּמִדַּעַת הָרִי"ף וְהָרַמְבַּ"ם נִרְאֶה דְּכָל שֶׁהָיָה בְּדַעְתּוֹ לְהַשְׁלִים נִדְרוֹ וְנִמְלַךְ בָּרֶגֶל וְהִתִּירוּ לוֹ חָשׁוּב אֹנֶס: {ג} וְאִם תֹּאמַר פְּשִׁיטָא. וְיֵשׁ לוֹמַר דְּלֹא גָּזְרִינַן דִּלְמָא אָתֵי לְשַׁהוֹיֵי קָרְבְּנוֹתָיו עַד הָרֶגֶל שֶׁהוּא בָּטֵל מִמְּלָאכָה. הָרַ"ן. אֲבָל בַּגְּמָרָא מַסִּיק דְּנָזִיר וּמְצֹרָע אַף עַל פִּי שֶׁהָיָה לָהֶם פְּנַאי מֻתָּרִין: {ד} דְּאִי לָאו הָכִי אַף עַל גַּב דְּגִלּוּחַ מְלָאכָה הוּא הָיָה לָנוּ לְהַתִּירָהּ לְצֹרֶךְ הַמּוֹעֵד. תּוֹסָפוֹת: (עיקר תוי"ט).

פרק ג - משנה ב

וְאֵלּוּ מְכַבְּסִין בַּמּוֹעֵד, הַבָּא מִמְּדִינַת הַיָּם, וּמִבֵּית הַשִּׁבְיָה, וְהַיּוֹצֵא מִבֵּית הָאֲסוּרִים, וְהַמְנֻדֶּה {ה} שֶׁהִתִּירוּ לוֹ חֲכָמִים, וְכֵן מִי שֶׁנִּשְׁאַל לְחָכָם וְהֻתַּר, מִטְפְּחוֹת הַיָּדַיִם וּמִטְפְּחוֹת הַסַּפָּרִים וּמִטְפְּחוֹת הַסְּפָג, הַזָּבִין וְהַזָּבוֹת וְהַנִּדּוֹת וְהַיּוֹלְדוֹת, וְכָל הָעוֹלִין {ו} מִטֻּמְאָה לְטָהֳרָה, הֲרֵי אֵלּוּ מֻתָּרִין. וּשְׁאָר כָּל אָדָם, אֲסוּרִין:

 ר"ע מברטנורה  וּמִי שֶׁנִּשְׁאַל לְחָכָם. שֶׁנָּדַר שֶׁלֹּא לְכַבֵּס בְּגָדָיו, וְנִשְׁאַל לְחָכָם בַּמּוֹעֵד וְהִתִּיר לוֹ נִדְרוֹ: וּמִטְפְּחוֹת יָדַיִם. שֶׁמְּנַגְּבִים בָּהֶם יָדַיִם בִּשְׁעַת אֲכִילָה: וּמִטְפְּחוֹת סַפָּרִים. כְּסוּת שֶׁנּוֹתְנִים הַסַּפָּרִים לַמִּסְתַּפֵּר בֵּין כְּתֵפָיו מִפְּנֵי הַשֵּׂעָר. וּכְשֶׁבָּא לְגַלֵּחַ לְאֵלּוּ הַשְּׁנוּיִים בַּמִּשְׁנָה שֶׁהֵן מֻתָּרִין לְגַלֵּחַ בַּמּוֹעֵד צָרִיךְ לְכַבְּסָהּ תָּמִיד: מִטְפְּחוֹת הַסַּפָּגִין. שֶׁמְּנַגְּבִין בָּהֵן כְּשֶׁיּוֹצְאִין מִן הַמֶּרְחָץ: וְכָל הָעוֹלִים מִטֻּמְאָה לְטָהֳרָה. בַּמּוֹעֵד, מֻתָּרִים לְכַבֵּס בִּגְדֵיהֶם. וּכְלֵי פִּשְׁתָּן אֲפִלּוּ שֶׁל כָּל אָדָם מֻתָּר לְכַבְּסָן בַּמּוֹעֵד, מִפְּנֵי שֶׁהֵן צְרִיכִין כִּבּוּס תָּדִיר, וַאֲפִלּוּ הַמְּכֻבָּסִים עֶרֶב הָרֶגֶל מִטַּנְפֵי מִיָּד וּצְרִיכִים כִּבּוּס בְּתוֹךְ הָרֶגֶל, הִלְכָּךְ לֹא גָּזְרוּ בְּהוּ. וּמִי שֶׁאֵין לוֹ אֶלָּא חָלוּק אֶחָד, אֲפִלּוּ שֶׁאֵינוֹ שֶׁל פִּשְׁתָּן מֻתָּר לְכַבֵּס בַּמּוֹעֵד, וְהוּא שֶׁעוֹמֵד עָרוּם בִּשְׁעַת כְּבִיסָתוֹ עִם אֵזוֹר בְּמָתְנָיו בִּלְבַד לְכַסּוֹת בְּשַׂר עֶרְוָה, שֶׁזֶּה מוֹכִיחַ שֶׁאֵין לוֹ אֶלָּא הֶחָלוּק הַזֶּה שֶׁהוּא מְכַבֵּס בִּלְבַד: (ר"ע מברטנורה).

 עיקר תוי"ט  {ה} וְהַמְנֻדֶּה. דְּקָיְמָא לָן שֶׁאָסוּר בְּתִכְבֹּסֶת כְּמוֹ הָאָבֵל. הָרַ"ן: {ו} וְכָל הָעוֹלִין כוּ'. כְּגוֹן מְצֹרָע וּטְמֵא מֵת שֶׁצְּרִיכִים לְהַחְלִיף בִּגְדֵיהֶם אוֹ לְכַבְּסָן, כָּל אֵלּוּ מֻתָּרִים בַּמּוֹעֵד, שֶׁלֹּא הָיָה אֶפְשָׁר לָהֶם לְכַבֵּס קֹדֶם הָרֶגֶל. הָרַ"ן. וְהַבֵּית יוֹסֵף כָּתַב לְהָבִיא בַּעַל קֶרִי שֶׁאֵרַע לוֹ קֶרִי בַּמּוֹעֵד. וּלְאָתוֹיֵי נַמִּי בַּעֲלַת הַכֶּתֶם שֶׁנִּמְצָא בַּמּוֹעֵד: (עיקר תוי"ט).

פרק ג - משנה ג

וְאֵלּוּ כּוֹתְבִין בַּמּוֹעֵד, קִדּוּשֵׁי נָשִׁים, גִּטִּין וְשׁוֹבָרִין, דְּיָתֵיקֵי, מַתָּנָה וּפְרוֹזְבּוּלִין, אִגְּרוֹת שׁוּם וְאִגְּרוֹת מָזוֹן, שִׁטְרֵי חֲלִיצָה וּמֵאוּנִים, וְשִׁטְרֵי בֵרוּרִין, וּגְזֵרוֹת בֵּית דִּין, וְאִגְּרוֹת שֶׁל רָשׁוּת:

 ר"ע מברטנורה  קִדּוּשֵׁי אִשָּׁה. כּוֹתֵב עַל הַנְּיָר אוֹ עַל הַחֶרֶס הֲרֵי אַתְּ מְקֻדֶּשֶׁת לִי, וּמְקַדֵּשׁ בּוֹ הָאִשָּׁה אַף עַל פִּי שֶׁאֵין בּוֹ שְׁוֵה פְּרוּטָה. וּלְפִיכָךְ מֻתָּר לִכְתֹּב בַּמּוֹעֵד, שֶׁמִּתְיָרֵא שֶׁמָּא יְקַדְּמֶנּוּ אַחֵר, וַהֲוָה לֵיהּ דָּבָר הָאָבֵד: גִּטִין. שֶׁרוֹצֶה לָצֵאת בְּשַׁיָּרָא, וְאִם אֵינוֹ כּוֹתֵב עַכְשָׁיו תִּשָּׁאֵר זֹאת עֲגוּנָה: וְשׁוֹבָרִים. שֶׁאִם יֹאמַר הַלֹּוֶה לֹא אֶפְרַע חוֹבְךָ אִם אֵין לִי שׁוֹבֵר מִמְּךָ, שׁוֹמְעִין לוֹ {ז}, וְיֵלֵךְ לְדַרְכּוֹ, וְנִמְצָא זֶה מַפְסִיד אֶת מְעוֹתָיו: דְּיָתֵיקֵי. מַתְּנַת שְׁכִיב מֵרַע. וּפֵרוּשׁ דְּיָתֵיקֵי, דָּא תְּהֵא לְמֵיקָם וְלִהְיוֹת, שֶׁדִּבְרֵי שְׁכִיב מֵרַע כִּכְתוּבִין וְכִמְסוּרִין דָּמוּ: מַתָּנָה. מַתְּנַת בָּרִיא. וְאִי לֹא כָּתִיב לְהוּ מַפְסִיד הַמְקַבֵּל, דְּשֶׁמָּא יַחְזֹר בּוֹ הַנּוֹתֵן {ח}: וּפְרוֹזְבּוּלִין. שֶׁלֹּא תִּשְׁמֹט שְׁבִיעִית וְיַפְסִיד אֶת מְעוֹתָיו {ט}. פְּרוֹזְבּוּל, פְּרוֹז בּוּלֵי וּפְרוֹז בּוּטֵי, כְּלוֹמַר תַּקָּנָה לָעֲשִׁירִים שֶׁלֹּא יַעַבְרוּ עַל מַה שֶּׁכָּתוּב בַּתּוֹרָה (דְּבָרִים טו) הִשָּׁמֵר לְךָ פֶּן יִהְיֶה דָּבָר עִם לְבָבְךָ בְּלִיַּעַל, וְתַקָּנָה לָעֲנִיִּים שֶׁיִּמְצְאוּ לִלְווֹת. בּוּלֵי עֲשִׁירִים. בּוּטֵי עֲנִיִּים. וְהִלֵּל תִּקֵּן פְּרוֹזְבּוּל, שְׁטָר שֶׁכּוֹתְבִין בּוֹ מוֹסֵר אֲנִי לָכֶם פְּלוֹנִי וּפְלוֹנִי הַדַּיָּנִין שֶׁכָּל שְׁטָר שֶׁיֵּשׁ לִי עַל פְּלוֹנִי אֶגְבֶנּוּ כָּל זְמַן שֶׁאֶרְצֶה, וְשׁוּב אֵין שְׁבִיעִית מְשַׁמֶּטֶת הַלְוָאָתוֹ, דַּהֲוֵי כְּאִלּוּ גָּבוּ בֵּית דִּין אֶת חוֹבוֹ, וְשׁוּב לֹא קָרִינַן בֵּיהּ לֹא יִגֹּשׂ אֶת רֵעֵהוּ: אִגְּרוֹת שׁוּם. שֶׁשָּׁמוּ בֵּית דִּין נִכְסֵי לֹוֶה וְנָתְנוּ לַמַּלְוֶה {י}: אִגְּרוֹת מָזוֹן. שֶׁמָּכְרוּ בֵּית דִּין אֶת הַקַּרְקַע לִמְזוֹן הָאִשָּׁה וְהַבָּנוֹת, וְכוֹתְבִין מַעֲשֵׂה בֵּית דִּין עַל זֶה. אִי נַמִּי מִי שֶׁקִּבֵּל עָלָיו לָזוּן אֶת בַּת אִשְׁתּוֹ: מֵאוּנִים. קְטַנָּה שֶׁהִשִּׂיאוּהָ אִמָּהּ וְאַחֶיהָ, יְכוֹלָה לְמָאֵן בְּבַעֲלָהּ וְלוֹמַר אִי אֶפְשִׁי בִּפְלוֹנִי בַּעֲלִי, וְיֹצְאָה בְּלֹא גֵּט. וּשְׁטָר שֶׁכּוֹתְבִין לְעֵדוּת דָּבָר זֶה, נִקְרָא שְׁטַר מֵאוּנִים {יא}: שִׁטְרֵי בֵרוּרִין. שֶׁבָּרְרוּ לָהֶם דַּיָּנִין, זֶה בּוֹרֵר לוֹ אֶחָד וְזֶה בּוֹרֵר לוֹ אֶחָד, וְכוֹתְבִין שְׁטָר, שֶׁלֹּא יַחְזְרוּ בָּהֶן הַבַּעֲלֵי דִּינִים {יב}: וּגְזֵרוֹת בֵּית דִּין. פְּסַק דִּין שֶׁפָּסְקוּ הַדַּיָּנִים {יג}: אִגְּרוֹת שֶׁל רָשׁוּת. צַוֵּי הַשִּׁלְטוֹן {יד}. כְּמוֹ וְאַל תִּתְוַדַּע לָרָשׁוּת, הֱוּוּ זְהִירִים בָּרָשׁוּת. וְיֵשׁ מְפָרְשִׁין אִגְּרוֹת שֶׁל שְׁאִילַת שָׁלוֹם שֶׁכּוֹתְבִים בֵּין אָדָם לַחֲבֵרוֹ {טו}: (ר"ע מברטנורה).

 עיקר תוי"ט  {ז} פֵּרוּשׁ הַבֵּית דִּין. וּמַיְרֵי שֶׁאִבֵּד הַמַּלְוֶה שְׁטַר חוֹב כִּדְכָתַב. הָרַ"ן: {ח} לָאו חֲזָרָה מַמָּשׁ דְּהָא מַתְּנַת בָּרִיא בְּלֹא קִנְיָן לֹא סַגִּי. אֶלָּא שֶׁאִם לֹא נִכְתַּב עֵדוּתָן יַכְחִישׁ הַנּוֹתֵן לֹא כָּךְ הָיָה. הָרַ"ן: {ט} וְאִם לֹא יִכְתְּבֶנּוּ עַכְשָׁיו וְהוּא רוֹצֶה לֵילֵךְ יַעֲבֹר שְׁבִיעִית וְיַפְסִיד חוֹבוֹ. הָרַ"ן: {י} וְקוֹרְעִין לוֹ שְׁטַר חוֹב, כִּדְאָמְרִינַן כָּל שׁוּמָא דְּלֹא כְּתִיב בֵּיהּ קְרַעְנָא לִשְׁטָרָא דְּמַלְוֶה לָאו שׁוּמָא הוּא, וְאִם הָעֵדִים רוֹצִים לָלֶכֶת יִהְיֶה הַמַּלְוֶה קֵרֵחַ מִכָּאן וּמִכָּאן (כָּךְ נִרְאֶה לִי). וְרַשִׁ"י פֵּרֵשׁ אִגְּרוֹת שׁוּם, בְּכָךְ שָׁמוּ בֵּית דִּין שָׂדֶה זוֹ וְנָתְנוּ לָאַחִים כָּךְ וְכָךְ נֶגְדּוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} וְאִם לֹא יִכְתְּבוּ הַשְּׁטָר עַכְשָׁיו שֶׁמָּא תָּבִיא שְׁתֵּי שְׂעָרוֹת לְאַחַר הַמּוֹעֵד וְשׁוּב אֵינָהּ יְכוֹלָה לְמָאֵן, וְתֵשֵׁב עִמּוֹ שֶׁאֵינָהּ חֲפֵצָה בְּאוֹתוֹ בַּעַל. הָרַ"ן: {יב} וִיבֻטַּל הַשָּׁלוֹם: {יג} שֶׁאִם לֹא יִכְתְּבוּ עַכְשָׁיו שֶׁמָּא יִשְׁכְּחוּ הַדָּבָר, וְלֹא יִפְסְקוּ פַּעַם אַחֶרֶת כֵּן. הָרַ"ן: {יד} וְעִנְיָנוֹ שֶׁהַנָּשִׂיא הָיָה נוֹתֵן רְשׁוּת לָזֶה לָדוּן וְלוֹמַר לָעַם הוֹרָאוֹתֵיהֶן, וְנִקְרָאִין פִּתְקִין דְּרַבָּנוּתָא, וְשׁוּב אִם טָעָה לֹא הָיָה מְשַׁלֵּם. וְאִם אֵין לוֹ שְׁטָר יְשַׁלֵּם מִבֵּיתוֹ. הָרַ"ן: {טו} בַּיְרוּשַׁלְמִי, וְטַעְמָא מִפְּנֵי שֶׁאֵין אָדָם נִזְהָר בְּתִקּוּנָן מְאֹד וַהֲוָה לֵיהּ כְּמַעֲשֵׂה הֶדְיוֹט. הָרַמְבַּ"ם. וְלִדְבָרָיו צָרִיךְ שֶׁיְּהֵא לְצֹרֶךְ הַמּוֹעֵד כוּ'. וְהָרַאבַ"ד כָּתַב דְּשֶׁמָּא לֹא יִזְדַּמֵּן לוֹ מוֹלִיךְ הַכְּתָב, כְּלוֹמַר וְהָוֵי דָּבָר הָאָבֵד: (עיקר תוי"ט).

פרק ג - משנה ד

אֵין כּוֹתְבִין שִׁטְרֵי חוֹב בַּמּוֹעֵד. וְאִם אֵינוֹ מַאֲמִינוֹ אוֹ שֶׁאֵין לוֹ מַה יֹּאכַל, הֲרֵי זֶה יִכְתֹּב. אֵין כּוֹתְבִין סְפָרִים, תְּפִלִּין {יז} וּמְזוּזוֹת, בַּמּוֹעֵד, וְאֵין מַגִּיהִין אוֹת אַחַת, אֲפִלּוּ בְּסֵפֶר (הָעֲזָרָה) עֶזְרָא. רַבִּי יְהוּדָה אוֹמֵר, כּוֹתֵב אָדָם תְּפִלִּין וּמְזוּזוֹת לְעַצְמוֹ, וְטוֹוֶה עַל יְרֵכוֹ {יט} תְּכֵלֶת לְצִיצִיתוֹ:

 ר"ע מברטנורה  אֵין כּוֹתְבִין שִׁטְרֵי חוֹב בַּמּוֹעֵד. הוֹאִיל וְיָכֹל לְכָתְבוֹ לְאַחַר הַמּוֹעֵד {טז}: וְאִם אֵינוֹ מַאֲמִינוֹ. וְאֵינוֹ רוֹצֶה לְהַלְווֹתוֹ בְּלֹא שְׁטָר, וְלֹוֶה צָרִיךְ לְמָעוֹת, כּוֹתְבִין: אוֹ שֶׁאֵין לוֹ. לַסּוֹפֵר מַה יֹּאכַל, הֲרֵי זֶה יִכְתֹּב וְיִטֹּל שְׂכָרוֹ, דִּשְׂכַר פְּעֻלָּה לְמִי שֶׁאֵין לוֹ מַה יֹּאכַל שָׁרֵי: וַאֲפִלּוּ בְּסֵפֶר הָעֲזָרָה. סֵפֶר תּוֹרָה שֶׁכֹּהֵן גָּדוֹל קוֹרֵא בּוֹ בְּיוֹם הַכִּפּוּרִים, וְאַף עַל גַּב דְּצָרִיךְ לָרַבִּים {יח}: לְעַצְמוֹ. לְקַיֵּם הַמִּצְוָה. אֲבָל לֹא לִמְכֹּר אוֹ לְהַשְׂכִּיר: וְטוֹוֶה עַל יְרֵכוֹ תְּכֵלֶת. שֶׁנּוֹתֵן הַחוּט עַל יְרֵכוֹ וּמְשַׁפְשֵׁף בְּיָדוֹ וְהוּא נִטְוֶה מֵאֵלָיו. אֲבָל לֹא בְּיָדוֹ בֵּין אֶצְבְּעוֹתָיו, וְלֹא בַּפֶּלֶךְ כְּדֶרֶךְ שֶׁהוּא עוֹשֶׂה בְּחֹל. וַהֲלָכָה, טוֹוֶה אָדָם תְּכֵלֶת לְבִגְדּוֹ בֵּין בְּפֶלֶךְ בֵּין בְּאֶבֶן {כ}: (ר"ע מברטנורה).

 עיקר תוי"ט  {טז} דְּהָכָא מִסְּתָמָא לֵיכָּא לְמֵיחַשׁ שֶׁמָּא יֹאמַר לֹא כָּךְ הָיָה כְּמוֹ בְּמַתָּנָה, כֵּיוָן שֶׁהִלְוָהוּ לֹא יִהְיֶה לֹוֶה רָשָׁע וְלֹא יְשַׁלֵּם: {יז} תְּפִלִּין כוּ'. כָּתַב סְמַ"ג בְּשֵׁם הַיְרוּשַׁלְמִי, בַּמֶּה דְּבָרִים אֲמוּרִים, לְהַנִּיחַ, פֵּרוּשׁ לְאַחַר הַמּוֹעֵד, אֲבָל לַהֲנִיחָן בַּמּוֹעֵד אֲפִלּוּ לְהָנִיחַ לַאֲחֵרִים יָכֹל לְכָתְבָן. וּבֵית יוֹסֵף נִדְחַק בְּפֵרוּשׁוֹ לְמַאן דִּסְבִירָא לֵיהּ דְּלֹא מַנַח תְּפִלִּין בַּמּוֹעֵד: {יח} יֵשׁ לְפָרֵשׁ בְּיוֹם הַכִּפּוּרִים צְרִיכִים לוֹ רַבִּים, אֲבָל לֹא עַכְשָׁיו, הָא אִם צָרִיךְ לִקְרוֹת בּוֹ מַגִּיהִין בּוֹ וּכְדַעַת א"ח. אֲבָל הַמַּגִּיד כָּתַב דִּבְסֵפֶר תּוֹרָה אֵין לְהַגִּיהַּ, דְּלֹא הִתִּירוּ אֶלָּא תְּפִלִּין דְּאֶפְשָׁר לְכָתְבָן כֻּלּוֹ בַּמּוֹעֵד, מַה שֶּׁאֵין כֵּן סֵפֶר תּוֹרָה שֶׁאִי אֶפְשָׁר לַעֲשׂוֹתָהּ כֻּלָּהּ בְּחֹל הַמּוֹעֵד. וְגִרְסַת הַסֵּפֶר בְּאָלֶ"ף, וּפֵרֵשׁ רַשִׁ"י סֵפֶר תּוֹרָה שֶׁל עֶזְרָא: {יט} עַל יְרֵכוֹ. אֲבָל בִּתְפִלִּין אִי אֶפְשָׁר עַל יְדֵי שִׁנּוּי, שֶׁצָּרִיךְ כְּתִיבָה תַּמָּה. תּוֹסָפוֹת: {כ} שֶׁתּוֹלֶה בַּחוּט כְּדֵי שֶׁיּוּכַל לְשָׁזְרוֹ יָפֶה. רַשִׁ"י: (עיקר תוי"ט).

פרק ג - משנה ה

הַקּוֹבֵר {כא} אֶת מֵתוֹ שְׁלֹשָׁה יָמִים {כב} קֹדֶם לָרֶגֶל, בָּטְלָה הֵימֶנּוּ גְּזֵרַת שִׁבְעָה {כג}. שְׁמֹנָה, בָּטְלָה הֵימֶנּוּ גְּזֵרַת שְׁלֹשִׁים {כה}, מִפְּנֵי שֶׁאָמְרוּ, שַׁבָּת עוֹלָה {כו} וְאֵינָהּ מַפְסֶקֶת, רְגָלִים מַפְסִיקִין וְאֵינָן עוֹלִין:

 ר"ע מברטנורה  הַקּוֹבֵר אֶת מֵתוֹ. שֶׁמֵּת לוֹ מֵת קֹדֶם הָרֶגֶל, וְנָהַג אֲבֵלוּת קֹדֶם הָרֶגֶל שְׁלֹשָׁה יָמִים: בָּטְלָה מִמֶּנּוּ גְּזֵרַת שִׁבְעָה. אֲבָל לֹא בָּטְלָה הֵימֶנּוּ גְּזֵרַת שְׁלֹשִׁים, דְּהַיְנוּ אִסּוּר תִּסְפֹּרֶת, וּמוֹנֶה אַחַר הָרֶגֶל שְׁלֹשִׁים יוֹם עִם הַשְּׁלֹשָׁה שֶׁעָבְרוּ כְּבָר מִן הַשְּׁלֹשִׁים. וּפְסַק הֲלָכָה, שֶׁהַקּוֹבֵר אֶת מֵתוֹ אֲפִלּוּ שָׁעָה אַחַת קֹדֶם הָרֶגֶל בָּטְלָה מִמֶּנּוּ גְּזֵרַת שִׁבְעָה: שְׁמֹנָה. יָמִים קֹדֶם הָרֶגֶל, הוֹאִיל וְנִכְנַס אֶחָד מִימֵי הַגִּלּוּחַ קֹדֶם לָרֶגֶל, אָתֵי רֶגֶל וּמְבַטֵּל אֶת הַשְּׁאָר לְגַמְרֵי {כד}: שַׁבָּת עוֹלָה. שַׁבָּת שֶׁבִּימֵי אֶבְלוֹ עוֹלָה לְמִנְיַן שִׁבְעָה: וְאֵינָהּ מַפְסֶקֶת. שֶׁאֵינָהּ מְבַטֶּלֶת אֶת הַשְּׁאָר, אֶלָּא יוֹשֵׁב וּמִתְאַבֵּל אַחַר שַׁבָּת: הָרְגָלִים מַפְסִיקִין וְאֵינָם עוֹלִים. אִם נָהַג אֲבֵלוּת קֹדֶם הָרֶגֶל, הָרֶגֶל מַפְסִיק וּמְבַטֵּל גְּזֵרַת שְׁלֹשִׁים. וְאִם לֹא נָהַג אֲבֵלוּת קֹדֶם הָרֶגֶל אֶלָּא הִתְחִיל אֲבֵלוּתוֹ בָּרֶגֶל, אֵין יְמֵי הָרֶגֶל עוֹלִים לְמִנְיַן שִׁבְעָה, אֲבָל עוֹלִין לְמִנְיַן שְׁלֹשִׁים. וְטַעְמָא דְּשַׁבָּת עוֹלָה לְחֶשְׁבּוֹן שִׁבְעַת יְמֵי אֲבֵלוּת, מִשּׁוּם דִּדְבָרִים שֶׁבְּצִנְעָה נוֹהֵג בָּהּ, כְּגוֹן תַּשְׁמִישׁ הַמִּטָּה, וּפְרִיעַת הָרֹאשׁ, שֶׁאָסוּר בְּגִלּוּי הָרֹאשׁ, וּרְחִיצָה בְּחַמִּין, שֶׁהָאָבֵל אָסוּר בָּהֵן בְּשַׁבָּת. אֲבָל הָרֶגֶל אֵין שׁוּם דִּין מִדִּינֵי אֲבֵלוּת נוֹהֵג בּוֹ {כז}, לְפִיכָךְ מַפְסִיק לְגַמְרֵי. וְאִם לֹא חָל אֲבֵלוּת כְּלָל קֹדֶם הָרֶגֶל, כְּגוֹן שֶׁנִּקְבַּר הַמֵּת בְּתוֹךְ הָרֶגֶל, אֵינוֹ עוֹלֶה לְמִנְיַן שִׁבְעָה, וּמַתְחִיל לִמְנוֹת שִׁבְעָה אַחַר הָרֶגֶל: (ר"ע מברטנורה).

 עיקר תוי"ט  {כא} הַקּוֹבֵר. לְהָכִי נָקַט קוֹבֵר, דַּאֲבֵלוּת מַתְחֶלֶת בִּשְׁעַת סְתִימַת הַגּוֹלֵל כִּדְאִיתָא בַּגְּמָרָא. תּוֹסָפוֹת: {כב} שְׁלֹשָׁה יָמִים. כָּךְ הוּא עִקַּר אֲבֵלוּת לְכַמָּה מִלֵּי, דְּאָמְרִינַן בַּגְּמָרָא שְׁלֹשָׁה יָמִים לִבְכִי. תּוֹסָפוֹת: {כג} שִׁבְעָה. אַסְמָכוּהוּ אַקְּרָא דִּכְתִיב [עָמוֹס ח] וְהָפַכְתִּי חַגֵּיכֶם לְאֵבֶל, מַה חַג שִׁבְעָה אַף אֲבֵלוּת שִׁבְעָה. גְּמָרָא: {כד} וְטַעֲמָא דְּסַגִּי בְּחַד יוֹמָא גַּבֵּי שְׁלֹשִׁים, וְגַבֵּי שִׁבְעָה בָּעִינַן שְׁלֹשָׁה, מִשּׁוּם דְּדִין כָּל שְׁלֹשִׁים שָׁוֶה, מַה שֶּׁאֵין כֵּן בְּשִׁבְעָה. הָרַ"ן: {כה} גְּזֵרַת שְׁלֹשִׁים. יָלֵיף פֶּרַע פֶּרַע מִנָּזִיר. כְּתִיב הָכָא [וַיִּקְרָא י] רָאשֵׁיכֶם אַל תִּפְרָעוּ, דִּשְׁמַע מִינַהּ שֶׁשְּׁאָר אָבֵל חַיָּב לִפְרֹעַ, וּכְתִיב הָתָם גָּדֵל פֶּרַע שְׂעַר רֹאשׁוֹ, מַה לְּהַלָּן שְׁלֹשִׁים אַף כָּאן שְׁלֹשִׁים. גְּמָרָא. וְהָרַמְבַּ"ם כָּתַב מִדִּכְתִיב [דְּבָרִים כא] וּבָכְתָה אֶת אָבִיהָ וְאֶת אִמָּהּ יֶרַח יָמִים, מִכְּלָל שֶׁהָאָבֵל מִצְטַעֵר כָּל שְׁלֹשִׁים: {כו} מִפְּנֵי כוּ'. כְּלוֹמַר וְאִם תֹּאמַר מַה טַּעַם לֹא אָמְרוּ חֲכָמִים הַקּוֹבֵר אֶת מֵתוֹ קֹדֶם שַׁבָּת בָּטְלָה מִמֶּנּוּ גְּזֵרַת שִׁבְעָה, מִפְּנֵי שֶׁאָמְרוּ שַׁבָּת עוֹלָה מִשּׁוּם דְּלֹא כְּתִיב בֵּיהּ שִׂמְחָה אֶלָּא עֹנֶג וּמִקְצָת אֲבֵלוּת נוֹהֵג בּוֹ וְכוּ'. הָרַ"ן. וְהַטּוּר כָּתַב דְּאִי אָמַרְתְּ דְּשַׁבָּת מַפְסִיק לֹא יִהְיֶה לְעוֹלָם שִׁבְעַת יָמִים. וְכָתַב בֵּית יוֹסֵף שֶׁהוּא מֵהַיְרוּשַׁלְמִי: {כז} וְזֶה תָּמוּהַּ, דְּהָא קַיְמָא לָן אַף בָּרֶגֶל דְּבָרִים שֶׁבְּצִנְעָה נוֹהֵג כְּמוֹ בְּשַׁבָּת. וְהַטַּעַם מִשּׁוּם דְּשַׁבָּת לֹא כְּתִיב בֵּיהּ שִׂמְחָה, כָּךְ נִרְאֶה לִי. וְכֵן כָּתַב בְּסֵפֶר מִשְׁנַת חֲכָמִים: (עיקר תוי"ט).

פרק ג - משנה ו

רַבִּי אֱלִיעֶזֶר אוֹמֵר, מִשֶּׁחָרַב בֵּית הַמִּקְדָּשׁ, עֲצֶרֶת כְּשַׁבָּת {כח}. רַבָּן גַּמְלִיאֵל אוֹמֵר, רֹאשׁ הַשָּׁנָה וְיוֹם הַכִּפּוּרִים, כָּרְגָלִים {כט}. וַחֲכָמִים אוֹמְרִים, לֹא כְדִבְרֵי זֶה וְלֹא כְדִבְרֵי זֶה, אֶלָּא עֲצֶרֶת כָּרְגָלִים, רֹאשׁ הַשָּׁנָה וְיוֹם הַכִּפּוּרִים כְּשַׁבָּת:

 ר"ע מברטנורה  מִשֶּׁחָרַב בֵּית הַמִּקְדָּשׁ. דְּאֵין לָעֲצֶרֶת תַּשְׁלוּמִין כָּל שִׁבְעָה לְקָרְבָּן, בָּטְלָה לֵיהּ תּוֹרַת הָרֶגֶל, וְהָוֵי כְּשַׁבָּת שֶׁאֵינָהּ מַפְסֶקֶת. אֲבָל בִּזְמַן הַמִּקְדָּשׁ שֶׁמִּי שֶׁלֹּא חָג בְּיוֹם רִאשׁוֹן שֶׁל עֲצֶרֶת הָיָה חוֹגֵג וְהוֹלֵךְ כָּל שִׁבְעָה, כְּחַג הַמַּצּוֹת, הָיָה עֲצֶרֶת כְּחַג הַמַּצּוֹת אַף לְעִנְיַן אֲבֵלוּת. וּמַסְקָנָא דְּמִלְתָא לְעִנְיַן פְּסַק הַהֲלָכָה, דַּעֲצֶרֶת בַּזְּמַן הַזֶּה כָּרְגָלִים, וְרֹאשׁ הַשָּׁנָה וְיוֹם הַכִּפּוּרִים כָּרְגָלִים. וּמִי שֶׁמֵּת לוֹ מֵת קֹדֶם אֶחָד מֵאֵלּוּ אֲפִלּוּ שָׁעָה אֶחָת, בָּטְלָה מִמֶּנּוּ גְּזֵרַת שִׁבְעָה. וְאִם מֵת שִׁבְעָה יָמִים קֹדֶם אֶחָד מִיָּמִים טוֹבִים אֵלּוּ {ל}, בָּטְלָה מִמֶּנּוּ גְּזֵרַת שְׁלֹשִׁים. וּמִי שֶׁמֵּת לוֹ מֵת שָׁעָה אַחַת קֹדֶם חַג הַסֻּכּוֹת בָּטְלָה מִמֶּנּוּ גְּזֵרַת שִׁבְעָה כִּדְאָמְרָן, וְשִׁבְעַת יְמֵי הָרֶגֶל עוֹלִין לְמִנְיַן שְׁלֹשִׁים. וְאַף עַל גַּב דִּתְנַן רְגָלִים מַפְסִיקִין וְאֵינָן עוֹלִין, אֵין עוֹלִין לְמִנְיַן שִׁבְעָה, אֲבָל עוֹלִין לְמִנְיַן שְׁלֹשִׁים, הֲרֵי אַרְבָּעָה עָשָׂר יוֹם, וּשְׁמִינִי עֲצֶרֶת הוּא רֶגֶל בִּפְנֵי עַצְמוֹ {לא}, וְנֶחְשָׁב כְּאִלּוּ הוּא שִׁבְעָה יָמִים, הֲרֵי עֶשְׂרִים וְאֶחָד יוֹם, מוֹנֶה עוֹד תִּשְׁעָה יָמִים עַד שְׁלֹשִׁים וְדַיּוֹ. וּגְזֵרַת שִׁבְעָה דְּאָמְרָן הוּא שֶׁאָבֵל כָּל שִׁבְעַת הַיָּמִים אָסוּר בִּרְחִיצָה בְּחַמִּין אֲפִלּוּ מִקְצָת גּוּפוֹ, וּבְצוֹנֵן כָּל גּוּפוֹ, וְאָסוּר בְּכִבּוּס בְּגָדָיו {לב}, וּבְסִיכָה, וּבִנְעִילַת הַסַּנְדָּל, וּבְתַשְׁמִישׁ הַמִּטָּה, וּבַעֲשִׂיַּת מְלָאכָה, וּבִשְׁאִילַת שָׁלוֹם. וְחַיָּב בַּעֲטִיפַת הָרֹאשׁ, וּבִכְפִיַּת הַמִּטָּה, שֶׁתִּהְיֶינָה כָּל הַמִּטּוֹת שֶׁבְּבֵיתוֹ כְּפוּיוֹת עַל גַּבֵּי קַרְקַע וְיִישַׁן בָּהֶם וְלֹא עַל מִטָּה זְקוּפָה, וְאָסוּר לִטֹּל צִפָּרְנָיו בִּכְלִי, וְלֹא שְׂעַר שְׂפָמוֹ וַאֲפִלּוּ מְעַכֵּב עָלָיו אֶת הָאֲכִילָה, וְאָסוּר לִקְרוֹת בַּתּוֹרָה, בֵּין בַּתּוֹרָה שֶׁבִּכְתָב בֵּין בַּתּוֹרָה שֶׁבְּעַל פֶּה, וְלֹא יִשְׁלֹל {לג} הַקֶּרַע שֶׁעָלָיו. וּגְזֵרַת שְׁלֹשִׁים הִיא שֶׁמִּשִּׁבְעָה וְעַד תַּשְׁלוּם שְׁלֹשִׁים אָסוּר לִלְבֹּשׁ בֶּגֶד חָדָשׁ, אוֹ בֶּגֶד לָבָן מְגֹהָץ, וְאָסוּר בְּתִסְפֹּרֶת, וּבִסְעוּדַת הָרְשׁוּת, וְלָלֶכֶת בִּסְחוֹרָה עִם חֲבֵרָיו, וְלִשָּׂא אִשָּׁה אַחֶרֶת אִם יֵשׁ לוֹ אִשָּׁה וּבָנִים, וְלֹא יְאַחֶה הַקֶּרַע שֶׁעָלָיו: (ר"ע מברטנורה).

 עיקר תוי"ט  {כח} עֲצֶרֶת. וְכָל שֶׁכֵּן רֹאשׁ הַשָּׁנָה וְיוֹם הַכִּפּוּרִים. תּוֹסָפוֹת. וְכָתַב הָרַ"ן דְּטַעְמָא דְּרַבִּי אֱלִיעֶזֶר דִּסְבִירָא לֵיהּ דְּטַעְמָא [דִּרְגָלִים] מַפְסִיקִין, שֶׁאִם אַתָּה אוֹמֵר יַעֲלוּ לְמִנְיַן שִׁבְעָה הֲרֵי הִפְסִיקוּ הַכֹּל, וְכֵן עֲצֶרֶת בִּזְמַן הַמִּקְדָּשׁ, וְכֵיוָן שֶׁאֵין עוֹלִין אָמְרוּ מַפְסִיקִין, וְעָשׂוּ דִּין שְׁלֹשִׁים כְּדִין שִׁבְעָה, אֲבָל מִשֶּׁחָרַב בֵּית הַמִּקְדָּשׁ כוּ': {כט} כָּרְגָלִים. מִפְּנֵי שֶׁהֻקְשׁוּ כָּל הַמּוֹעֲדוֹת זֶה לָזֶה, דִּכְתִיב [וְיִּקְרָא כג] אֵלֶּה מוֹעֲדֵי ה'. הָרַמְבַּ"ן: {ל} כְּאַבָּא שָׁאוּל דְּפָלֵיג אַמַּתְנִיתִין דִּלְעֵיל דְּתָנֵי שְׁמֹנָה יָמִים. וְטַעֲמוֹ דְּסוֹבֵר מִקְצַת הַיּוֹם כְּכֻלּוֹ וּשְׁבִיעִי עוֹלֶה לְכָאן וּלְכָאן: {לא} וְכֵן יוֹם אֶחָד לִפְנֵי שָׁבוּעוֹת וְלִפְנֵי רֹאשׁ הַשָּׁנָה וְלִפְנֵי יוֹם הַכִּפּוּרִים: {לב} בַּגְּמָרָא דַּף ט"ו יָלֵיף לְהוּ מִקְּרָאֵי. וּנְטִילַת צִפָּרְנַיִם בִּכְלָל גִּלּוּחַ הוּא. נִרְאֶה לִי: {לג} פֵּרֵשׁ בָּעָרוּךְ שֶׁהוּא כְּמוֹ מַכְלִיב דְּמִשְׁנָה ח' פֶּרֶק קַמָּא. וְנִרְאֶה לִי שֶׁהוּא מִלְּשׁוֹן שָׁלָל וּבִיזָה, שֶׁכֵּן הַשָּׁלָל אַחַת הֵנָּה וְאַחַת הֵנָּה וְאֵינוֹ מֻנָּח וְצָבוּר: (עיקר תוי"ט).

פרק ג - משנה ז

אֵין קוֹרְעִין {לד}, וְלֹא חוֹלְצִין, וְאֵין מַבְרִין, אֶלָּא קְרוֹבָיו שֶׁל מֵת, וְאֵין מַבְרִין אֶלָּא עַל מִטָּה זְקוּפָה. אֵין מוֹלִיכִין לְבֵית הָאֵבֶל לֹא בְּטַבְלָא וְלֹא בְאִסְקוּטְלָא וְלֹא בְקָנוֹן, אֶלָּא בְסַלִּים. וְאֵין אוֹמְרִים בִּרְכַּת אֲבֵלִים בַּמּוֹעֵד, אֲבָל עוֹמְדִין בְּשׁוּרָה וּמְנַחֲמִין וּפוֹטְרִין אֶת הָרַבִּים:

 ר"ע מברטנורה  אֵין קוֹרְעִין. בַּמּוֹעֵד, אֶלָּא קְרוֹבָיו שֶׁל מֵת. לְאִמּוֹ וּלְאָבִיו וְלִבְנוֹ וּלְבִתּוֹ וּלְאָחִיו וְלַאֲחוֹתוֹ וּלְאִשְׁתּוֹ, אֵלּוּ שִׁבְעָה מֵתִים שֶׁחַיָּב לְהִתְאַבֵּל עֲלֵיהֶם {לה}. וְעַל שְׁאָר קְרוֹבִים שֶׁאֵינוֹ חַיָּב לְהִתְאַבֵּל עֲלֵיהֶם אֵין קוֹרְעִין עֲלֵיהֶם בַּמּוֹעֵד. וְלֹא אָמְרָן אֶלָּא שֶׁאֵינוֹ חָכָם אוֹ אָדָם כָּשֵׁר וּבַעַל מַעֲשִׂים. אֲבָל חָכָם שֶׁמֵּת, הַכֹּל קְרוֹבָיו, וְהַכֹּל קוֹרְעִין עָלָיו אֲפִלּוּ בַּמּוֹעֵד. וְכֵן עַל אָדָם הַמֻּחְזָק בְּכַשְׁרוּת וַחֲסִידוּת. וְכָל הָעוֹמֵד בִּשְׁעַת יְצִיאַת נְשָׁמָה עַל כָּל אָדָם מִיִּשְׂרָאֵל, חַיָּב לִקְרֹעַ וַאֲפִלּוּ בַּמּוֹעֵד. וּקְרִיעָה עַל אָבִיו וְעַל אִמּוֹ הִיא בְּכָל בְּגָדָיו עַד שֶׁמְּגַלֶּה אֶת לִבּוֹ, וְקוֹרֵעַ בְּיָדוֹ וּמַבְדִּיל שְׂפַת הַבֶּגֶד הָעֶלְיוֹנָה, וְקוֹרֵעַ מִבַּחוּץ, שֶׁאֵינוֹ מַכְנִיס יָדוֹ תַּחַת בְּגָדָיו כְּשֶׁהוּא קוֹרֵעַ, וְשׁוֹלֵל לְאַחַר שְׁלֹשִׁים יוֹם, וְאֵינוֹ מְאַחֶה לְעוֹלָם. וְכֵן עַל רַבּוֹ שֶׁלִּמְּדוֹ תּוֹרָה. אֲבָל עַל כָּל שְׁאָר קְרוֹבָיו קוֹרֵעַ טֶפַח מִן הַבֶּגֶד הָעֶלְיוֹן לְבַד, וְקוֹרֵעַ בִּכְלִי אִם יִרְצֶה, וְאֵינוֹ צָרִיךְ לְהַבְדִּיל הַשָּׂפָה, וְיֵשׁ לוֹ לְהַכְנִיס יָדוֹ תַּחַת בְּגָדָיו כְּשֶׁהוּא קוֹרֵעַ, וְשׁוֹלֵל לְאַחַר שִׁבְעָה, וּמְאַחֶה לְאַחַר שְׁלֹשִׁים. וְאֵין קְרִיעָה אֶלָּא מְעֻמָּד: וְלֹא חוֹלְצִים בַּמּוֹעֵד. חֲלִיצַת כָּתֵף, שֶׁזְּרוֹעוֹתֵיהֶן וְכִתְפֵיהֶן מְגֻלּוֹת: וְלֹא מַבְרִין. סְעוּדָה רִאשׁוֹנָה, שֶׁאָבֵל אָסוּר לֶאֱכֹל מִשֶּׁלּוֹ, וּרְגִילִים לְהַבְרוֹתוֹ בִּרְחוֹבָהּ שֶׁל עִיר, וּבַמּוֹעֵד אֵין מַבְרִין אֶלָּא קְרוֹבָיו בְּתוֹךְ בֵּיתוֹ: וְאֵין מַבְרִין אֶלָּא עַל מִטּוֹת זְקוּפוֹת. וַאֲפִלּוּ בְּתוֹךְ בֵּיתוֹ אֵין מַבְרִין אוֹתוֹ עַל מִטּוֹת כְּפוּיוֹת, שֶׁרְגִילִין בִּשְׁאָר יְמוֹת הַשָּׁנָה כָּל קְרוֹבָיו וּמִי שֶׁלִּבּוֹ גַּס בּוֹ לֶאֱכֹל עִמּוֹ עַל מִטָּה כְּפוּיָה לָאָרֶץ, וּבַמּוֹעֵד אֵין מַבְרִין אֶלָּא עַל מִטּוֹת זְקוּפוֹת: וְאֵין מוֹלִיכִין. הַסְּעוּדָה לְבֵית הָאָבֵל: לֹא בְּטַבְלָא. דֶּרֶךְ כָּבוֹד: וְלֹא בְאִסְקוּטְלָא. כְּמִין שֻׁלְחָן קָטָן שֶׁל כֶּסֶף אוֹ שֶׁל זָהָב אוֹ שֶׁל זְכוּכִית. וַאֲנִי שָׁמַעְתִּי קְעָרָה שֶׁל כֶּסֶף, שֶׁכֵּן בְּלַעַ"ז קוֹרִין לִקְעָרָה שקודיל"א: אֶלָּא בְסַלִּין. שֶׁל עֲרָבָה קְלוּפָה. שֶׁלֹּא לְבַיֵּשׁ הָעֲנִיִּים שֶׁבָּאִין לְהַבְרוֹת שֶׁאֵין לָהֶם טַבְלָא וְאִסְקוּטְלָא {לו}: וּפוֹטְרִין אֶת הָרַבִּים. נוֹתְנִים לָהֶם רְשׁוּת לֵילֵךְ מִיָּד: (ר"ע מברטנורה).

 עיקר תוי"ט  {לד} אֵין כוּ'. דִּין הַקְּרִיעָה יָלֵיף בַּגְּמָרָא שָׁם. וְדִמְעֻמָּד בַּגְּמָרָא דַּף כ"א, וּדְבַעֲיָא טֶפַח דַּף כ"ב. וּפֵרוּשׁ מְאַחֶה תּוֹפֵר מַמָּשׁ, שֶׁכֵּן וְעֵת לִתְפֹּר מְתַרְגֵּם וְעִדַּן בְּחִיר לַאֲחָאָה: {לה} שֶׁהַכֹּהֵן מִטַּמֵּא לָהֶם: {לו} גְּמָרָא. וַאֲפִלּוּ בְּחֹל קָאָמַר, וְאַשְׁמְעִינַן הָא בְּחֹל הַמּוֹעֵד, דְּמַהוּ דְּתֵימָא כָּל הֵיכָא דְּלֹא יַרְאֶה דֶּרֶךְ אֲבֵלוּת אֶלָּא דּוֹרוֹן וַחֲשִׁיבוּת עָדִיף טְפֵי, קָא מַשְׁמַע לָן. הָרַ"ן: (עיקר תוי"ט).

פרק ג - משנה ח

אֵין מַנִּיחִין אֶת הַמִּטָּה בָּרְחוֹב, שֶׁלֹּא לְהַרְגִּיל אֶת הַהֶסְפֵּד {לז}, וְלֹא שֶׁל נָשִׁים לְעוֹלָם, מִפְּנֵי הַכָּבוֹד {לח}. נָשִׁים בַּמּוֹעֵד מְעַנּוֹת, אֲבָל לֹא מְטַפְּחוֹת. רַבִּי יִשְׁמָעֵאל אוֹמֵר, הַסְּמוּכוֹת לַמִּטָּה, מְטַפְּחוֹת:

 ר"ע מברטנורה  אֵין מַנִּיחִין אֶת הַמִּטָּה. שֶׁל מֵת: בָּרְחוֹב. בַּמּוֹעֵד: לְעוֹלָם. אֲפִלּוּ בְּחֹל, דִּכְתִיב (בְּמִדְבָּר כ) וַתָּמָת שָׁם מִרְיָם וַתִּקָּבֵר שָׁם, סָמוּךְ לְמִיתָה קְבוּרָה: מְטַפְּחוֹת. מְסַפְּקוֹת כַּף אֶל כַּף: רַבִּי שִׁמְעוֹן אוֹמֵר כו'. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: (ר"ע מברטנורה).

 עיקר תוי"ט  {לז} שֶׁלֹּא כוּ'. שֶׁהַמּוֹעֵד אָסוּר בְּהֶסְפֵּד וּבְתַעֲנִית. הָרַמְבַּ"ם: {לח} מִפְּנֵי הַכָּבוֹד. שֶׁשּׁוֹפַעַת זִיבָה. רַשִׁ"י: (עיקר תוי"ט).

פרק ג - משנה ט

בְּרָאשֵׁי חֳדָשִׁים, בַּחֲנֻכָּה וּבְפוּרִים, מְעַנּוֹת, וּמְטַפְּחוֹת בָּזֶה וּבָזֶה, אֲבָל לֹא מְקוֹנְנוֹת. נִקְבַּר הַמֵּת {לט}, לֹא מְעַנּוֹת וְלֹא מְטַפְּחוֹת. אֵיזֶהוּ עִנּוּי, שֶׁכֻּלָּן עוֹנוֹת כְּאֶחָת. קִינָה, שֶׁאַחַת מְדַבֶּרֶת וְכֻלָּן עוֹנוֹת אַחֲרֶיהָ, שֶׁנֶּאֱמַר (ירמיה ט), וְלַמֵּדְנָה בְנֹתֵיכֶם נֶהִי, וְאִשָּׁה רְעוּתָהּ קִינָה. אֲבָל לֶעָתִיד לָבֹא הוּא אוֹמֵר (ישעיה כה), בִּלַּע הַמָּוֶת לָנֶצַח, וּמָחָה ה' אֱלֹהִים דִּמְעָה מֵעַל כָּל פָּנִים וְגוֹ':

 ר"ע מברטנורה  וְאִשָּׁה רְעוּתָהּ קִינָה. וּמִדִּכְתִיב וְלַמֵּדְנָה, אַלְמָא קִינָה, אַחַת אוֹמֶרֶת וַחֲבֶרְתָּהּ עוֹנָה: אֲבָל לֶעָתִיד לָבֹא אוֹמֵר בִּלַּע הַמָּוֶת לָנֶצַח. שֶׁיְּבַלַּע הַקָּדוֹשׁ בָּרוּךְ הוּא אֶת הַמָּוֶת, וְכֻלָּן עוֹנוֹת שִׁירָה, שֶׁאֵין מִיתָה וְאֵין דִּמְעָה. וְנָקְטִינַן לְהַאי קְרָא הָכָא, מִשּׁוּם אַל תַּעֲמֹד בְּדָבָר רָע [לָכֵן רָאָה לְסַיֵּם בְּטוֹב]: (ר"ע מברטנורה).

 עיקר תוי"ט  {לט} נִקְבַּר הַמֵּת. אֲפִלּוּ בְּרֹאשׁ חֹדֶשׁ חֲנֻכָּה וּפוּרִים, וְכָל שֶׁכֵּן בְּחֹל הַמּוֹעֵד דְּחָמִיר טְפֵי: (עיקר תוי"ט).
