פרשת תולדות


{יט} וְאֵלֶּה תּֽוֹלְדֹת יִצְחָק בֶּן אַבְרָהָם אַבְרָהָם הוֹלִיד אֶת יִצְחָֽק:  מ  וְאֵלֶּה תּֽוֹלְדֹת יִצְחָק בֶּן אַבְרָהָם אַבְרָהָם הוֹלִיד אֶת יִצְחָֽק:  ת  וְאִלֵּין תּוּלְדַת יִצְחָק בַּר אַבְרָהָם אַבְרָהָם אוֹלִיד יָת יִצְחָק:

{כ} וַיְהִי יִצְחָק בֶּן אַרְבָּעִים שָׁנָה בְּקַחְתּוֹ אֶת רִבְקָה בַּת בְּתוּאֵל הָֽאֲרַמִּי מִפַּדַּן אֲרָם אֲחוֹת לָבָן הָֽאֲרַמִּי לוֹ לְאִשָּֽׁה:  מ  וַיְהִי יִצְחָק בֶּן אַרְבָּעִים שָׁנָה בְּקַחְתּוֹ אֶת רִבְקָה בַּת בְּתוּאֵל הָֽאֲרַמִּי מִפַּדַּן אֲרָם אֲחוֹת לָבָן הָֽאֲרַמִּי לוֹ לְאִשָּֽׁה:  ת  וַהֲוָה יִצְחָק בַּר אַרְבְּעִין שְׁנִין כַּד נְסֵיב יָת רִבְקָה בַּת בְּתוּאֵל אֲרַמָּאָה מִפַּדַּן אֲרָם אֲחָתֵהּ דְלָבָן אֲרַמָּאָה לֵהּ לְאִנְתּוּ:

{כא} וַיֶּעְתַּר יִצְחָק לַֽיהֹוָה לְנֹכַח אִשְׁתּוֹ כִּי עֲקָרָה הִוא וַיֵּעָתֶר לוֹ יְהֹוָה וַתַּהַר רִבְקָה אִשְׁתּֽוֹ:  מ  וַיֶּעְתַּר יִצְחָק לַֽיהֹוָה לְנֹכַח אִשְׁתּוֹ כִּי עֲקָרָה הִוא וַיֵּעָתֶר לוֹ יְהֹוָה וַתַּהַר רִבְקָה אִשְׁתּֽוֹ:  ת  וְצַלִּי יִצְחָק קֳדָם יְיָ לָקֳבֵל אִתְּתֵהּ אֲרֵי עֲקָרָה הִיא וְקַבֵּל צְלוֹתֵהּ יְיָ וְעַדִּיאַת רִבְקָה אִתְּתֵהּ:

{כב} וַיִּתְרֹֽצֲצוּ הַבָּנִים בְּקִרְבָּהּ וַתֹּאמֶר אִם כֵּן לָמָּה זֶּה אָנֹכִי וַתֵּלֶךְ לִדְרשׁ אֶת יְהֹוָֽה:  מ  וַיִּתְרֹֽצֲצוּ הַבָּנִים בְּקִרְבָּהּ וַתֹּאמֶר אִם כֵּן לָמָּה זֶּה אָנֹכִי וַתֵּלֶךְ לִדְרשׁ אֶת יְהֹוָֽה:  ת  וְדַחֲקִין בְּנַיָּא בִּמְעָהָא וַאֲמֶרֶת אִם כֵּן לְמָה דְּנַן אֲנָא וַאֲזָלַת לְמִתְבַּע אוּלְפַן מִן קֳדָם יְיָ:

{כג} וַיֹּאמֶר יְהֹוָה לָהּ שְׁנֵי (גיים) גוֹיִם בְּבִטְנֵךְ וּשְׁנֵי לְאֻמִּים מִמֵּעַיִךְ יִפָּרֵדוּ וּלְאֹם מִלְאֹם יֶֽאֱמָץ וְרַב יַֽעֲבֹד צָעִֽיר:  מ  וַיֹּאמֶר יְהֹוָה לָהּ שְׁנֵי (גיים) גוֹיִם בְּבִטְנֵךְ וּשְׁנֵי לְאֻמִּים מִמֵּעַיִךְ יִפָּרֵדוּ וּלְאֹם מִלְאֹם יֶֽאֱמָץ וְרַב יַֽעֲבֹד צָעִֽיר:  ת  וַאֲמַר יְיָ לַהּ תְּרֵין עַמְמִין בִּמְעַיְכִי וְתַרְתֵּין מַלְכְּוָן מִמֵעַיְכִי יִתְפָּרְשָׁן וּמַלְכוּ מִמַלְכוּ יִתְקַף וְרַבָּא יִשְׁתַּעֲבֵּיד לִזְעֵירָא:

{כד} וַיִּמְלְאוּ יָמֶיהָ לָלֶדֶת וְהִנֵּה תוֹמִם בְּבִטְנָֽהּ:  מ  וַיִּמְלְאוּ יָמֶיהָ לָלֶדֶת וְהִנֵּה תוֹמִם בְּבִטְנָֽהּ:  ת  וּשְׁלִימוּ יוֹמָהָא לְמֵילָד וְהָא תְיוּמִין בִּמְעָהָא:

{כה} וַיֵּצֵא הָֽרִאשׁוֹן אַדְמוֹנִי כֻּלּוֹ כְּאַדֶּרֶת שֵׂעָר וַיִּקְרְאוּ שְׁמוֹ עֵשָֽׂו:  מ  וַיֵּצֵא הָֽרִאשׁוֹן אַדְמוֹנִי כֻּלּוֹ כְּאַדֶּרֶת שֵׂעָר וַיִּקְרְאוּ שְׁמוֹ עֵשָֽׂו:  ת  וּנְפַק קַדְמָאָה סִמוֹק כֻּלֵּהּ כִּגְלִים (נ''י כִּכְלָן) דִּשְׂעָר וּקְרוֹ שְׁמֵהּ עֵשָׂו:

{כו} וְאַֽחֲרֵי כֵן יָצָא אָחִיו וְיָדוֹ אֹחֶזֶת בַּֽעֲקֵב עֵשָׂו וַיִּקְרָא שְׁמוֹ יַֽעֲקֹב וְיִצְחָק בֶּן שִׁשִּׁים שָׁנָה בְּלֶדֶת אֹתָֽם:  מ  וְאַֽחֲרֵי כֵן יָצָא אָחִיו וְיָדוֹ אֹחֶזֶת בַּֽעֲקֵב עֵשָׂו וַיִּקְרָא שְׁמוֹ יַֽעֲקֹב וְיִצְחָק בֶּן שִׁשִּׁים שָׁנָה בְּלֶדֶת אֹתָֽם:  ת  וּבָתַר כֵּן נְפַק אֲחוּהִי וִידֵהּ אֲחִידָא בְּעִקְבָא דְעֵשָׂו וּקְרָא שְׁמֵהּ יַעֲקֹב וְיִצְחָק בַּר שִׁתִּין שְׁנִין כַּד יְלִידַת יָתְהוֹן:

{כז} וַֽיִּגְדְּלוּ הַנְּעָרִים וַיְהִי עֵשָׂו אִישׁ יֹדֵעַ צַיִד אִישׁ שָׂדֶה וְיַֽעֲקֹב אִישׁ תָּם ישֵׁב אֹֽהָלִֽים:  מ  וַֽיִּגְדְּלוּ הַנְּעָרִים וַיְהִי עֵשָׂו אִישׁ יֹדֵעַ צַיִד אִישׁ שָׂדֶה וְיַֽעֲקֹב אִישׁ תָּם ישֵׁב אֹֽהָלִֽים:  ת  וּרְבִיוּ עוּלֵמַיָּא וַהֲוָה עֵשָׂו גְבַר נַחְשִׁירְכָן גְבַר נָפֵק לְחַקְלָא וְיַעֲקֹב גְבַר שְׁלִים מְשַׁמֵשׁ בֵּית אוּלְפָנָא:

{כח} וַיֶּֽאֱהַב יִצְחָק אֶת עֵשָׂו כִּי צַיִד בְּפִיו וְרִבְקָה אֹהֶבֶת אֶת יַֽעֲקֹֽב:  מ  וַיֶּֽאֱהַב יִצְחָק אֶת עֵשָׂו כִּי צַיִד בְּפִיו וְרִבְקָה אֹהֶבֶת אֶת יַֽעֲקֹֽב:  ת  וּרְחֵם יִצְחָק יָת עֵשָׂו אֲרֵי מִצֵידֵהּ הֲוָה אָכִיל וְרִבְקָה רְחִימַת יָת יַעֲקֹב:

{כט} וַיָּזֶד יַֽעֲקֹב נָזִיד וַיָּבֹא עֵשָׂו מִן הַשָּׂדֶה וְהוּא עָיֵֽף:  מ  וַיָּזֶד יַֽעֲקֹב נָזִיד וַיָּבֹא עֵשָׂו מִן הַשָּׂדֶה וְהוּא עָיֵֽף:  ת  וּבַשִּׁיל יַעֲקֹב תַּבְשִׁילָא וַאֲתָא (נ''י וְעַל) עֵשָׂו מִן חַקְלָא וְהוּא מְשַׁלְהֵי:

{ל} וַיֹּאמֶר עֵשָׂו אֶֽל יַֽעֲקֹב הַלְעִיטֵנִי נָא מִן הָֽאָדֹם הָֽאָדֹם הַזֶּה כִּי עָיֵף אָנֹכִי עַל כֵּן קָֽרָא שְׁמוֹ אֱדֽוֹם:  מ  וַיֹּאמֶר עֵשָׂו אֶֽל יַֽעֲקֹב הַלְעִיטֵנִי נָא מִן הָֽאָדֹם הָֽאָדֹם הַזֶּה כִּי עָיֵף אָנֹכִי עַל כֵּן קָֽרָא שְׁמוֹ אֱדֽוֹם:  ת  וַאֲמַר עֵשָׂו לְיַעֲקֹב אַטְעֵמְנִי כְעַן מִן סִמּוֹקָא סִמּוֹקָא הָדֵין אֲרֵי מְשַׁלְהֵי אֲנָא עַל כֵּן קְרָא שְׁמֵהּ אֱדוֹם:

{לא} וַיֹּאמֶר יַֽעֲקֹב מִכְרָה כַיּוֹם אֶת בְּכֹרָֽתְךָ לִֽי:  מ  וַיֹּאמֶר יַֽעֲקֹב מִכְרָה כַיּוֹם אֶת בְּכֹרָֽתְךָ לִֽי:  ת  וַאֲמַר יַעֲקֹב זַבִּין כְּיוֹם דִּלְהֵן יָת בְּכֵרוּתָךְ לִי:

{לב} וַיֹּאמֶר עֵשָׂו הִנֵּה אָֽנֹכִי הוֹלֵךְ לָמוּת וְלָֽמָּה זֶּה לִי בְּכֹרָֽה:  מ  וַיֹּאמֶר עֵשָׂו הִנֵּה אָֽנֹכִי הוֹלֵךְ לָמוּת וְלָֽמָּה זֶּה לִי בְּכֹרָֽה:  ת  וַאֲמַר עֵשָׂו הָא אֲנָא אָזֵל לִמְמָת וּלְמָה דְּנָן לִי בְּכֵרוּתָא:

{לג} וַיֹּאמֶר יַֽעֲקֹב הִשָּׁבְעָה לִּי כַּיּוֹם וַיִּשָּׁבַע לוֹ וַיִּמְכֹּר אֶת בְּכֹֽרָתוֹ לְיַֽעֲקֹֽב:  מ  וַיֹּאמֶר יַֽעֲקֹב הִשָּׁבְעָה לִּי כַּיּוֹם וַיִּשָּׁבַע לוֹ וַיִּמְכֹּר אֶת בְּכֹֽרָתוֹ לְיַֽעֲקֹֽב:  ת  וַאֲמַר יַעֲקֹב קַיֵּם לִי כְּיוֹם דִּלְהֵן וְקַיִּים לֵהּ וְזַבִּין יָת בְּכֵרוּתֵהּ לְיַעֲקֹב:

{לד} וְיַֽעֲקֹב נָתַן לְעֵשָׂו לֶחֶם וּנְזִיד עֲדָשִׁים וַיֹּאכַל וַיֵּשְׁתְּ וַיָּקָם וַיֵּלַךְ וַיִּבֶז עֵשָׂו אֶת הַבְּכֹרָֽה:  מ  וְיַֽעֲקֹב נָתַן לְעֵשָׂו לֶחֶם וּנְזִיד עֲדָשִׁים וַיֹּאכַל וַיֵּשְׁתְּ וַיָּקָם וַיֵּלַךְ וַיִּבֶז עֵשָׂו אֶת הַבְּכֹרָֽה:  ת  וְיַעֲקֹב יְהַב לְעֵשָׂו לְחֵם וְתַבְשִׁיל דִּטְלוֹפְחִין וַאֲכַל וּשְׁתִי וְקָם וַאֲזָל וְשָׁט עֵשָׂו יָת בְּכֵרוּתָא: (פ)




בראשית פרק-כו

{א} וַיְהִי רָעָב בָּאָרֶץ מִלְּבַד הָֽרָעָב הָֽרִאשׁוֹן אֲשֶׁר הָיָה בִּימֵי אַבְרָהָם וַיֵּלֶךְ יִצְחָק אֶל אֲבִימֶלֶךְ מֶֽלֶךְ פְּלִשְׁתִּים גְּרָֽרָה:  מ  וַיְהִי רָעָב בָּאָרֶץ מִלְּבַד הָֽרָעָב הָֽרִאשׁוֹן אֲשֶׁר הָיָה בִּימֵי אַבְרָהָם וַיֵּלֶךְ יִצְחָק אֶל אֲבִימֶלֶךְ מֶֽלֶךְ פְּלִשְׁתִּים גְּרָֽרָה:  ת  וַהֲוָה כַפְנָא בְּאַרְעָא בַּר מִכַּפְנָא קַדְמָאָה דִּי הֲוָה בְּיוֹמֵי אַבְרָהָם וַאֲזַל יִצְחָק לְוַת אֲבִימֶלֶךְ מַלְכָּא דִּפְלִשְׁתָּאֵי לִגְרָר:

{ב} וַיֵּרָא אֵלָיו יְהֹוָה וַיֹּאמֶר אַל תֵּרֵד מִצְרָיְמָה שְׁכֹן בָּאָרֶץ אֲשֶׁר אֹמַר אֵלֶֽיךָ:  מ  וַיֵּרָא אֵלָיו יְהֹוָה וַיֹּאמֶר אַל תֵּרֵד מִצְרָיְמָה שְׁכֹן בָּאָרֶץ אֲשֶׁר אֹמַר אֵלֶֽיךָ:  ת  וְאִתְגְּלִי לֵהּ יְיָ וַאֲמַר לָא תֵחוֹת לְמִצְרָיִם שְׁרֵי בְאַרְעָא דִּי אֵימַר לָךְ:

{ג} גּוּר בָּאָרֶץ הַזֹּאת וְאֶֽהְיֶה עִמְּךָ וַֽאֲבָֽרֲכֶךָּ כִּֽי לְךָ וּֽלְזַרְעֲךָ אֶתֵּן אֶת כָּל הָֽאֲרָצֹת הָאֵל וַֽהֲקִֽמֹתִי אֶת הַשְּׁבֻעָה אֲשֶׁר נִשְׁבַּעְתִּי לְאַבְרָהָם אָבִֽיךָ:  מ  גּוּר בָּאָרֶץ הַזֹּאת וְאֶֽהְיֶה עִמְּךָ וַֽאֲבָֽרֲכֶךָּ כִּֽי לְךָ וּֽלְזַרְעֲךָ אֶתֵּן אֶת כָּל הָֽאֲרָצֹת הָאֵל וַֽהֲקִֽמֹתִי אֶת הַשְּׁבֻעָה אֲשֶׁר נִשְׁבַּעְתִּי לְאַבְרָהָם אָבִֽיךָ:  ת  דּוּר בְּאַרְעָא הָדָא וִיהֵי מֵימְרִי בְסַעְדָךְ וְאֵבָרְכִנָךְ אֲרֵי לָךְ וְלִבְנָיךְ אֶתֵּן יָת כָּל אַרְעָתָא הָאִלֵּין וְאָקֵם יָת קְיָמָא דִּי קַיֵּמִית לְאַבְרָהָם אֲבוּךְ:

{ד} וְהִרְבֵּיתִי אֶת זַרְעֲךָ כְּכֽוֹכְבֵי הַשָּׁמַיִם וְנָֽתַתִּי לְזַרְעֲךָ אֵת כָּל הָֽאֲרָצֹת הָאֵל וְהִתְבָּֽרֲכוּ בְזַרְעֲךָ כֹּל גּוֹיֵי הָאָֽרֶץ:  מ  וְהִרְבֵּיתִי אֶת זַרְעֲךָ כְּכֽוֹכְבֵי הַשָּׁמַיִם וְנָֽתַתִּי לְזַרְעֲךָ אֵת כָּל הָֽאֲרָצֹת הָאֵל וְהִתְבָּֽרֲכוּ בְזַרְעֲךָ כֹּל גּוֹיֵי הָאָֽרֶץ:  ת  וְאַסְגֵּי יָת בְּנָיךְ סַגִּיאִין כְּכוֹכְבֵי שְׁמַיָּא וְאֶתֵּן לִבְנָיךְ יָת כָּל אַרְעָתָא הָאִלֵּין וְיִתְבָּרְכוּן בְּדִיל בְּנָךְ כֹּל עַמְמֵי אַרְעָא:

{ה} עֵקֶב אֲשֶׁר שָׁמַע אַבְרָהָם בְּקֹלִי וַיִּשְׁמֹר מִשְׁמַרְתִּי מִצְוֹתַי חֻקּוֹתַי וְתֽוֹרֹתָֽי:  מ  עֵקֶב אֲשֶׁר שָׁמַע אַבְרָהָם בְּקֹלִי וַיִּשְׁמֹר מִשְׁמַרְתִּי מִצְוֹתַי חֻקּוֹתַי וְתֽוֹרֹתָֽי:  ת  חֳלַף דִּי קַבִּיל אַבְרָהָם בְּמֵימְרִי וּנְטַר מַטְּרַת מֵימְרִי פִּקּוֹדַי קְיָמַי וְאוֹרָיָתָי:

{ו}  שני  וַיֵּשֶׁב יִצְחָק בִּגְרָֽר:  מ  וַיֵּשֶׁב יִצְחָק בִּגְרָֽר:  ת  וִיתֵב יִצְחָק בִּגְרָר:

{ז} וַֽיִּשְׁאֲלוּ אַנְשֵׁי הַמָּקוֹם לְאִשְׁתּוֹ וַיֹּאמֶר אֲחֹתִי הִוא כִּי יָרֵא לֵאמֹר אִשְׁתִּי פֶּן יַֽהַרְגֻנִי אַנְשֵׁי הַמָּקוֹם עַל רִבְקָה כִּֽי טוֹבַת מַרְאֶה הִֽוא:  מ  וַֽיִּשְׁאֲלוּ אַנְשֵׁי הַמָּקוֹם לְאִשְׁתּוֹ וַיֹּאמֶר אֲחֹתִי הִוא כִּי יָרֵא לֵאמֹר אִשְׁתִּי פֶּן יַֽהַרְגֻנִי אַנְשֵׁי הַמָּקוֹם עַל רִבְקָה כִּֽי טוֹבַת מַרְאֶה הִֽוא:  ת  וּשְׁאִילוּ אֲנָשֵׁי אַתְרָא (עַל עֵיסַק אִתְּתֵהּ) לְאִתְּתֵהּ וַאֲמַר אֲחָתִי הִיא אֲרֵי דְחִיל לְמֵימַר אִתְּתִי דִּלְמָא יִקְטְּלֻנַּנִי אֲנָשֵׁי אַתְרָא עַל רִבְקָה אֲרֵי שַׁפִּירַת חֵיזוּ הִיא:

{ח} וַיְהִי כִּי אָֽרְכוּ לוֹ שָׁם הַיָּמִים וַיַּשְׁקֵף אֲבִימֶלֶךְ מֶלֶךְ פְּלִשְׁתִּים בְּעַד הַֽחַלּוֹן וַיַּרְא וְהִנֵּה יִצְחָק מְצַחֵק אֵת רִבְקָה אִשְׁתּֽוֹ:  מ  וַיְהִי כִּי אָֽרְכוּ לוֹ שָׁם הַיָּמִים וַיַּשְׁקֵף אֲבִימֶלֶךְ מֶלֶךְ פְּלִשְׁתִּים בְּעַד הַֽחַלּוֹן וַיַּרְא וְהִנֵּה יִצְחָק מְצַחֵק אֵת רִבְקָה אִשְׁתּֽוֹ:  ת  וַהֲוָה כַּד סְגִיאוּ לֵהּ תַּמָּן יוֹמַיָּא וְאִסְתְּכִי אֲבִימֶלֶךְ מַלְכָּא דִּפְלִשְׁתָּאֵי מִן חֲרַכָּא וַחֲזָא וְהָא יִצְחָק מְחַיִּיךְ עִם רִבְקָה אִתְּתֵהּ:

{ט} וַיִּקְרָא אֲבִימֶלֶךְ לְיִצְחָק וַיֹּאמֶר אַךְ הִנֵּה אִשְׁתְּךָ הִוא וְאֵיךְ אָמַרְתָּ אֲחֹתִי הִוא וַיֹּאמֶר אֵלָיו יִצְחָק כִּי אָמַרְתִּי פֶּן אָמוּת עָלֶֽיהָ:  מ  וַיִּקְרָא אֲבִימֶלֶךְ לְיִצְחָק וַיֹּאמֶר אַךְ הִנֵּה אִשְׁתְּךָ הִוא וְאֵיךְ אָמַרְתָּ אֲחֹתִי הִוא וַיֹּאמֶר אֵלָיו יִצְחָק כִּי אָמַרְתִּי פֶּן אָמוּת עָלֶֽיהָ:  ת  וּקְרָא אֲבִימֶלֶךְ לְיִצְחָק וַאֲמַר בְּרַם הָא אִתְּתָךְ הִיא וְאֶכְדֵּין אֲמַרְתָּ אֲחָתִי הִיא וַאֲמַר לֵהּ יִצְחָק אֲרֵי אֲמָרִית דִּלְמָא אֵימוּת עֲלַהּ:

{י} וַיֹּאמֶר אֲבִימֶלֶךְ מַה זֹּאת עָשִׂיתָ לָּנוּ כִּמְעַט שָׁכַב אַחַד הָעָם אֶת אִשְׁתֶּךָ וְהֵֽבֵאתָ עָלֵינוּ אָשָֽׁם:  מ  וַיֹּאמֶר אֲבִימֶלֶךְ מַה זֹּאת עָשִׂיתָ לָּנוּ כִּמְעַט שָׁכַב אַחַד הָעָם אֶת אִשְׁתֶּךָ וְהֵֽבֵאתָ עָלֵינוּ אָשָֽׁם:  ת  וַאֲמַר אֲבִימֶלֶךְ מָה דָא עֲבַדְתָּ לָנָא כִּזְעֵיר פּוֹן שָׁכִיב דִּמְיָחֵד בְּעַמָּא עִם אִתְּתָךְ וְאַיְתֵיתָא עֲלָנָא חוֹבָא:

{יא} וַיְצַו אֲבִימֶלֶךְ אֶת כָּל הָעָם לֵאמֹר הַנֹּגֵעַ בָּאִישׁ הַזֶּה וּבְאִשְׁתּוֹ מוֹת יוּמָֽת:  מ  וַיְצַו אֲבִימֶלֶךְ אֶת כָּל הָעָם לֵאמֹר הַנֹּגֵעַ בָּאִישׁ הַזֶּה וּבְאִשְׁתּוֹ מוֹת יוּמָֽת:  ת  וּפַקִּיד אֲבִימֶלֶךְ יָת כָּל עַמָּא לְמֵימָר דְּיַנְזֵק לְגַבְרָא הָדֵין וּבְאִתְּתֵהּ אִתְקְטָלָא יִתְקְטֵּל:

{יב} וַיִּזְרַע יִצְחָק בָּאָרֶץ הַהִוא וַיִּמְצָא בַּשָּׁנָה הַהִוא מֵאָה שְׁעָרִים וַֽיְבָֽרְכֵהוּ יְהֹוָֽה:  מ  וַיִּזְרַע יִצְחָק בָּאָרֶץ הַהִוא וַיִּמְצָא בַּשָּׁנָה הַהִוא מֵאָה שְׁעָרִים וַֽיְבָֽרְכֵהוּ יְהֹוָֽה:  ת  וּזְרַע יִצְחָק בְּאַרְעָא הַהִיא וְאַשְׁכַּח בְּשַׁתָּא הַהִיא עַל חַד מְאָה בִדְשַׁעֲרוֹהִי וּבָרְכֵהּ יְיָ:

{יג}  שלישי  וַיִּגְדַּל הָאִישׁ וַיֵּלֶךְ הָלוֹךְ וְגָדֵל עַד כִּֽי גָדַל מְאֹֽד:  מ  וַיִּגְדַּל הָאִישׁ וַיֵּלֶךְ הָלוֹךְ וְגָדֵל עַד כִּֽי גָדַל מְאֹֽד:  ת  וּרְבָא גַּבְרָא וַאֲזַל אֲזִיל (נ''י סָגֵי) וְרָבֵי עַד דִּי רְבָא לַחֲדָא:

{יד} וַֽיְהִי לוֹ מִקְנֵה צֹאן וּמִקְנֵה בָקָר וַֽעֲבֻדָּה רַבָּה וַיְקַנְאוּ אֹתוֹ פְּלִשְׁתִּֽים:  מ  וַֽיְהִי לוֹ מִקְנֵה צֹאן וּמִקְנֵה בָקָר וַֽעֲבֻדָּה רַבָּה וַיְקַנְאוּ אֹתוֹ פְּלִשְׁתִּֽים:  ת  וַהֲוָה לֵהּ גֵּיתֵי עָנָא וְגֵיתֵי תוֹרִין וּפָלְחָנָא (נ''י וַעֲבוּדָה) סַגְיָא וְקַנִּיאוּ בֵהּ פְּלִשְׁתָּאֵי:

{טו} וְכָל הַבְּאֵרֹת אֲשֶׁר חָֽפְרוּ עַבְדֵי אָבִיו בִּימֵי אַבְרָהָם אָבִיו סִתְּמוּם פְּלִשְׁתִּים וַֽיְמַלְאוּם עָפָֽר:  מ  וְכָל הַבְּאֵרֹת אֲשֶׁר חָֽפְרוּ עַבְדֵי אָבִיו בִּימֵי אַבְרָהָם אָבִיו סִתְּמוּם פְּלִשְׁתִּים וַֽיְמַלְאוּם עָפָֽר:  ת  וְכָל בֵּירִין דִּי חֲפָרוּ עַבְדֵי אֲבוּהִי בְּיוֹמֵי אַבְרָהָם אֲבוּהִי טַמּוֹנוּן פְּלִשְׁתָּאֵי וּמְלוֹנוּן עַפְרָא:

{טז} וַיֹּאמֶר אֲבִימֶלֶךְ אֶל יִצְחָק לֵךְ מֵֽעִמָּנוּ כִּֽי עָצַמְתָּ מִמֶּנּוּ מְאֹֽד:  מ  וַיֹּאמֶר אֲבִימֶלֶךְ אֶל יִצְחָק לֵךְ מֵֽעִמָּנוּ כִּֽי עָצַמְתָּ מִמֶּנּוּ מְאֹֽד:  ת  וַאֲמַר אֲבִימֶלֶךְ לְיִצְחָק אִזְּל מֵעִמָּנָא אֲרֵי תְקֵפְתָּא מִנָּנָא לַחֲדָא:

{יז} וַיֵּלֶךְ מִשָּׁם יִצְחָק וַיִּחַן בְּנַֽחַל גְּרָר וַיֵּשֶׁב שָֽׁם:  מ  וַיֵּלֶךְ מִשָּׁם יִצְחָק וַיִּחַן בְּנַֽחַל גְּרָר וַיֵּשֶׁב שָֽׁם:  ת  וַאֲזַל מִתַּמָּן יִצְחָק וּשְׁרָא בְנַחֲלָא דִּגְרָר וִיתֵב תַּמָּן:

{יח} וַיָּשָׁב יִצְחָק וַיַּחְפֹּר אֶת בְּאֵרֹת הַמַּיִם אֲשֶׁר חָֽפְרוּ בִּימֵי אַבְרָהָם אָבִיו וַיְסַתְּמוּם פְּלִשְׁתִּים אַֽחֲרֵי מוֹת אַבְרָהָם וַיִּקְרָא לָהֶן שֵׁמוֹת כַּשֵּׁמֹת אֲשֶׁר קָרָא לָהֶן אָבִֽיו:  מ  וַיָּשָׁב יִצְחָק וַיַּחְפֹּר אֶת בְּאֵרֹת הַמַּיִם אֲשֶׁר חָֽפְרוּ בִּימֵי אַבְרָהָם אָבִיו וַיְסַתְּמוּם פְּלִשְׁתִּים אַֽחֲרֵי מוֹת אַבְרָהָם וַיִּקְרָא לָהֶן שֵׁמוֹת כַּשֵּׁמֹת אֲשֶׁר קָרָא לָהֶן אָבִֽיו:  ת  וְתָב יִצְחָק וְחָפַר יָת בֵּירֵי דְמַיָּא דִּי חֲפָרוּ בְּיוֹמֵי אַבְרָהָם אֲבוּהִי וְטַּמּוֹנוּן פְּלִשְׁתָּאֵי בָּתַר דְּמִית אַבְרָהָם וּקְרָא לְהֵן שְׁמָהָן כִּשְׁמָהָן דִּי הֲוָה קָרֵי לְהֵן אֲבוּהִי:

{יט} וַיַּחְפְּרוּ עַבְדֵֽי יִצְחָק בַּנָּחַל וַיִּמְצְאוּ שָׁם בְּאֵר מַיִם חַיִּֽים:  מ  וַיַּחְפְּרוּ עַבְדֵֽי יִצְחָק בַּנָּחַל וַיִּמְצְאוּ שָׁם בְּאֵר מַיִם חַיִּֽים:  ת  וַחֲפָרוּ עַבְדֵי יִצְחָק בְּנַחֲלָא וְאַשְׁכָּחוּ תַמָּן בֵּירָא דְּמַיִּין נָבְעִין:

{כ} וַיָּרִיבוּ רֹעֵי גְרָר עִם רֹעֵי יִצְחָק לֵאמֹר לָנוּ הַמָּיִם וַיִּקְרָא שֵׁם הַבְּאֵר עֵשֶׂק כִּי הִֽתְעַשְּׂקוּ עִמּֽוֹ:  מ  וַיָּרִיבוּ רֹעֵי גְרָר עִם רֹעֵי יִצְחָק לֵאמֹר לָנוּ הַמָּיִם וַיִּקְרָא שֵׁם הַבְּאֵר עֵשֶׂק כִּי הִֽתְעַשְּׂקוּ עִמּֽוֹ:  ת  וּנְצוֹ רַעֲוָתָא דִּגְרָר עִם רַעֲוָתָא דְיִצְחָק לְמֵימַר דִּי לָנָא מַיָּא וּקְרָא שְׁמָא דְבֵירָא עִסְקָא אֲרֵי אִתְעַסִּיקוּ עִמֵּהּ:

{כא} וַֽיַּחְפְּרוּ בְּאֵר אַחֶרֶת וַיָּרִיבוּ גַּם עָלֶיהָ וַיִּקְרָא שְׁמָהּ שִׂטְנָֽה:  מ  וַֽיַּחְפְּרוּ בְּאֵר אַחֶרֶת וַיָּרִיבוּ גַּם עָלֶיהָ וַיִּקְרָא שְׁמָהּ שִׂטְנָֽה:  ת  וַחֲפָרוּ בֵּירָא אָחֳרִי וּנְצוֹ אַף עֲלַהּ וּקְרָא שְׁמַהּ שִׂטְנָא:

{כב} וַיַּעְתֵּק מִשָּׁם וַיַּחְפֹּר בְּאֵר אַחֶרֶת וְלֹא רָבוּ עָלֶיהָ וַיִּקְרָא שְׁמָהּ רְחֹבוֹת וַיֹּאמֶר כִּֽי עַתָּה הִרְחִיב יְהֹוָה לָנוּ וּפָרִינוּ בָאָֽרֶץ:  מ  וַיַּעְתֵּק מִשָּׁם וַיַּחְפֹּר בְּאֵר אַחֶרֶת וְלֹא רָבוּ עָלֶיהָ וַיִּקְרָא שְׁמָהּ רְחֹבוֹת וַיֹּאמֶר כִּֽי עַתָּה הִרְחִיב יְהֹוָה לָנוּ וּפָרִינוּ בָאָֽרֶץ:  ת  וְאִסְתַּלַּק מִתַּמָּן וַחֲפַר בֵּירָא אָחֳרִי וְלָא נְצוֹ עֲלַהּ וּקְרָא שְׁמַהּ רְחֹבוֹת וַאֲמַר אֲרֵי כְעַן אַפְתֵּי יְיָ לָנָא וְנִיפוּשׁ בְּאַרְעָא:

{כג}  רביעי  וַיַּעַל מִשָּׁם בְּאֵר שָֽׁבַע:  מ  וַיַּעַל מִשָּׁם בְּאֵר שָֽׁבַע:  ת  וְאִסְתַּלַּק מִתַּמָּן לִבְּאֵר שָׁבַע:

{כד} וַיֵּרָא אֵלָיו יְהֹוָה בַּלַּיְלָה הַהוּא וַיֹּאמֶר אָֽנֹכִי אֱלֹהֵי אַבְרָהָם אָבִיךָ אַל תִּירָא כִּֽי אִתְּךָ אָנֹכִי וּבֵֽרַכְתִּיךָ וְהִרְבֵּיתִי אֶת זַרְעֲךָ בַּֽעֲבוּר אַבְרָהָם עַבְדִּֽי:  מ  וַיֵּרָא אֵלָיו יְהֹוָה בַּלַּיְלָה הַהוּא וַיֹּאמֶר אָֽנֹכִי אֱלֹהֵי אַבְרָהָם אָבִיךָ אַל תִּירָא כִּֽי אִתְּךָ אָנֹכִי וּבֵֽרַכְתִּיךָ וְהִרְבֵּיתִי אֶת זַרְעֲךָ בַּֽעֲבוּר אַבְרָהָם עַבְדִּֽי:  ת  וְאִתְגְּלִי לֵהּ יְיָ בְּלֵילְיָא הַהוּא וַאֲמַר אֲנָא אֱלָהֵהּ דְּאַבְרָהָם אֲבוּךְ לָא תִדְחַל אֲרֵי בְסַעְדָּךְ מֵימְרִי וֶאֱבָרֵכִנָּךְ וְאַסְגֵּי יָת בְּנָיךְ בְּדִיל אַבְרָהָם עַבְדִּי:

{כה} וַיִּבֶן שָׁם מִזְבֵּחַ וַיִּקְרָא בְּשֵׁם יְהֹוָה וַיֶּט שָׁם אָֽהֳלוֹ וַיִּכְרוּ שָׁם עַבְדֵֽי יִצְחָק בְּאֵֽר:  מ  וַיִּבֶן שָׁם מִזְבֵּחַ וַיִּקְרָא בְּשֵׁם יְהֹוָה וַיֶּט שָׁם אָֽהֳלוֹ וַיִּכְרוּ שָׁם עַבְדֵֽי יִצְחָק בְּאֵֽר:  ת  וּבְנָא תַמָּן מַדְבְּחָא וְצַלִּי בִּשְׁמָא דַּיְיָ וּפְרַס תַּמָּן מַשְׁכְּנֵהּ וּכְרוֹ תַמָּן עַבְדֵי יִצְחָק בֵּירָא:

{כו} וַֽאֲבִימֶלֶךְ הָלַךְ אֵלָיו מִגְּרָר וַֽאֲחֻזַּת מֵֽרֵעֵהוּ וּפִיכֹל שַׂר צְבָאֽוֹ:  מ  וַֽאֲבִימֶלֶךְ הָלַךְ אֵלָיו מִגְּרָר וַֽאֲחֻזַּת מֵֽרֵעֵהוּ וּפִיכֹל שַׂר צְבָאֽוֹ:  ת  וַאֲבִימֶלֶךְ אֲזַל לְוָתֵהּ מִגְּרָר וְסִיעַת מֵרַחֲמוֹהִי וּפִיכֹל רַב חֵילֵהּ:

{כז} וַיֹּאמֶר אֲלֵהֶם יִצְחָק מַדּוּעַ בָּאתֶם אֵלָי וְאַתֶּם שְׂנֵאתֶם אֹתִי וַֽתְּשַׁלְּחוּנִי מֵֽאִתְּכֶֽם:  מ  וַיֹּאמֶר אֲלֵהֶם יִצְחָק מַדּוּעַ בָּאתֶם אֵלָי וְאַתֶּם שְׂנֵאתֶם אֹתִי וַֽתְּשַׁלְּחוּנִי מֵֽאִתְּכֶֽם:  ת  וַאֲמַר לְהוֹן יִצְחָק מָה דֵּין אֲתֵיתוּן לְוָתִי וְאַתּוּן סְנֵיתוּן יָתִי וְשַׁלַּחְתּוּנִי מִלְּוָתְכוֹן:

{כח} וַיֹּֽאמְרוּ רָאוֹ רָאִינוּ כִּֽי הָיָה יְהֹוָה עִמָּךְ וַנֹּאמֶר תְּהִי נָא אָלָה בֵּֽינוֹתֵינוּ בֵּינֵינוּ וּבֵינֶךָ וְנִכְרְתָה בְרִית עִמָּֽךְ:  מ  וַיֹּֽאמְרוּ רָאוֹ רָאִינוּ כִּֽי הָיָה יְהֹוָה עִמָּךְ וַנֹּאמֶר תְּהִי נָא אָלָה בֵּֽינוֹתֵינוּ בֵּינֵינוּ וּבֵינֶךָ וְנִכְרְתָה בְרִית עִמָּֽךְ:  ת  וַאֲמָרוּ מֶחֱזָא חֲזֵינָא אֲרֵי הֲוָה מֵימְרָא דַּיְיָ בִּסַעְדָּךְ וַאֲמַרְנָא תִּתְקַיַּם כְּעַן מוֹמָתָא דִּהֲוָה בֵין אֲבָהָתָנָא בֵּינָנָא וּבֵינָךְ וְנִגְזַר קְיָם עִמָּךְ:

{כט} אִם תַּֽעֲשֵׂה עִמָּנוּ רָעָה כַּֽאֲשֶׁר לֹא נְגַֽעֲנוּךָ וְכַֽאֲשֶׁר עָשִׂינוּ עִמְּךָ רַק טוֹב וַֽנְּשַׁלֵּֽחֲךָ בְּשָׁלוֹם אַתָּה עַתָּה בְּרוּךְ יְהֹוָֽה:  מ  אִם תַּֽעֲשֵׂה עִמָּנוּ רָעָה כַּֽאֲשֶׁר לֹא נְגַֽעֲנוּךָ וְכַֽאֲשֶׁר עָשִׂינוּ עִמְּךָ רַק טוֹב וַֽנְּשַׁלֵּֽחֲךָ בְּשָׁלוֹם אַתָּה עַתָּה בְּרוּךְ יְהֹוָֽה:  ת  אִם תַּעְבֵּד עִמָּנָא בִּישָׁא כְּמָא דִּי לָא אַנְזִיקְנָךְ וּכְמָא דִּי עֲבַדְנָא עִמָךְ לְחוֹד טָב וַנְשַׁלְּחִנָּךְ בִּשְׁלָם אַתְּ כְּעַן בְּרִיכָא דַּיְיָ:

{ל}  חמישי  וַיַּעַשׂ לָהֶם מִשְׁתֶּה וַיֹּֽאכְלוּ וַיִּשְׁתּֽוּ:  מ  וַיַּעַשׂ לָהֶם מִשְׁתֶּה וַיֹּֽאכְלוּ וַיִּשְׁתּֽוּ:  ת  וַעֲבַד לְהוֹן מִשְׁתְּיָא וַאֲכָלוּ וּשְׁתִיאוּ:

{לא} וַיַּשְׁכִּימוּ בַבֹּקֶר וַיִּשָּֽׁבְעוּ אִישׁ לְאָחִיו וַיְשַׁלְּחֵם יִצְחָק וַיֵּֽלְכוּ מֵֽאִתּוֹ בְּשָׁלֽוֹם:  מ  וַיַּשְׁכִּימוּ בַבֹּקֶר וַיִּשָּֽׁבְעוּ אִישׁ לְאָחִיו וַיְשַׁלְּחֵם יִצְחָק וַיֵּֽלְכוּ מֵֽאִתּוֹ בְּשָׁלֽוֹם:  ת  וְאַקְדִימוּ בְצַפְרָא וְקַיִּימוּ גְּבַר לְאָחוּהִי וְשַׁלָּחִנוּן יִצְחָק וַאֲזָלוּ מִלְוָתֵהּ בִּשְׁלָם:

{לב} וַיְהִי בַּיּוֹם הַהוּא וַיָּבֹאוּ עַבְדֵי יִצְחָק וַיַּגִּדוּ לוֹ עַל אֹדוֹת הַבְּאֵר אֲשֶׁר חָפָרוּ וַיֹּאמְרוּ לוֹ מָצָאנוּ מָֽיִם:  מ  וַיְהִי בַּיּוֹם הַהוּא וַיָּבֹאוּ עַבְדֵי יִצְחָק וַיַּגִּדוּ לוֹ עַל אֹדוֹת הַבְּאֵר אֲשֶׁר חָפָרוּ וַיֹּאמְרוּ לוֹ מָצָאנוּ מָֽיִם:  ת  וַהֲוָה בְּיוֹמָא הַהוּא וַאֲתוֹ עַבְדֵי יִצְחָק וְחַוִּיאוּ לֵהּ עַל עֵיסַק בֵּירָא דִּי חֲפָרוּ וַאֲמָרוּ לֵהּ אַשְׁכַּחְנָא מַיָּא:

{לג} וַיִּקְרָא אֹתָהּ שִׁבְעָה עַל כֵּן שֵׁם הָעִיר בְּאֵר שֶׁבַע עַד הַיּוֹם הַזֶּֽה:  מ  וַיִּקְרָא אֹתָהּ שִׁבְעָה עַל כֵּן שֵׁם הָעִיר בְּאֵר שֶׁבַע עַד הַיּוֹם הַזֶּֽה:  ת  וּקְרָא יָתַהּ שִׁבְעָה עַל כֵּן שְׁמָא דְקַרְתָּא בְּאֵרָא דְשֶׁבַע (נ''י בְּאֵר שֶׁבַע) עַד יוֹמָא הָדֵין: (ס)

{לד} וַיְהִי עֵשָׂו בֶּן אַרְבָּעִים שָׁנָה וַיִּקַּח אִשָּׁה אֶת יְהוּדִית בַּת בְּאֵרִי הַֽחִתִּי וְאֶת בָּשְׂמַת בַּת אֵילֹן הַֽחִתִּֽי:  מ  וַיְהִי עֵשָׂו בֶּן אַרְבָּעִים שָׁנָה וַיִּקַּח אִשָּׁה אֶת יְהוּדִית בַּת בְּאֵרִי הַֽחִתִּי וְאֶת בָּשְׂמַת בַּת אֵילֹן הַֽחִתִּֽי:  ת  וַהֲוָה עֵשָׂו בַּר אַרְבְּעִין שְׁנִין וּנְסִיב אִתְּתָא יָת יְהוּדִית בַּת בְּאֵרִי חִתָּאָה וְיָת בָּשְׂמַת בַּת אֵילוֹן חִתָּאָה:

{לה} וַתִּֽהְיֶיןָ מֹרַת רוּחַ לְיִצְחָק וּלְרִבְקָֽה:  מ  וַתִּֽהְיֶיןָ מֹרַת רוּחַ לְיִצְחָק וּלְרִבְקָֽה:  ת  וַהֲוָאָה מְסָרְבָן וּמְרַגְּזָן עַל מֵימַר יִצְחָק וְרִבְקָה: (ס)




בראשית פרק-כז

{א} וַֽיְהִי כִּֽי זָקֵן יִצְחָק וַתִּכְהֶיןָ עֵינָיו מֵֽרְאֹת וַיִּקְרָא אֶת עֵשָׂו בְּנוֹ הַגָּדֹל וַיֹּאמֶר אֵלָיו בְּנִי וַיֹּאמֶר אֵלָיו הִנֵּֽנִי:  מ  וַֽיְהִי כִּֽי זָקֵן יִצְחָק וַתִּכְהֶיןָ עֵינָיו מֵֽרְאֹת וַיִּקְרָא אֶת עֵשָׂו בְּנוֹ הַגָּדֹל וַיֹּאמֶר אֵלָיו בְּנִי וַיֹּאמֶר אֵלָיו הִנֵּֽנִי:  ת  וַהֲוָה כַּד סִיב יִצְחָק וְכָהֲיָּא עֵינוֹהִי מִלְּמֶחֱזֵי וּקְרָא יַת עֵשָׂו בְּרֵהּ רַבָּה וַאֲמַר לֵהּ בְּרִי וַאֲמַר לֵהּ הָא אָנָא:

{ב} וַיֹּאמֶר הִנֵּה נָא זָקַנְתִּי לֹא יָדַעְתִּי יוֹם מוֹתִֽי:  מ  וַיֹּאמֶר הִנֵּה נָא זָקַנְתִּי לֹא יָדַעְתִּי יוֹם מוֹתִֽי:  ת  וַאֲמַר הָא כְעַן סֵיבִית לֵית אֲנָא יְדַע יוֹמָא דְאֵימוּת:

{ג} וְעַתָּה שָׂא נָא כֵלֶיךָ תֶּלְיְךָ וְקַשְׁתֶּךָ וְצֵא הַשָּׂדֶה וְצוּדָה לִּי (צידה) צָיִד:  מ  וְעַתָּה שָׂא נָא כֵלֶיךָ תֶּלְיְךָ וְקַשְׁתֶּךָ וְצֵא הַשָּׂדֶה וְצוּדָה לִּי (צידה) צָיִד:  ת  וּכְעַן סַב כְּעַן זֵינָךְ סַיְפָךְ וְקַשְׁתָּךְ וּפוּק לְחַקְלָא וְצוּד לִי צֵידָא:

{ד} וַֽעֲשֵׂה לִי מַטְעַמִּים כַּֽאֲשֶׁר אָהַבְתִּי וְהָבִיאָה לִּי וְאֹכֵלָה בַּֽעֲבוּר תְּבָֽרֶכְךָ נַפְשִׁי בְּטֶרֶם אָמֽוּת:  מ  וַֽעֲשֵׂה לִי מַטְעַמִּים כַּֽאֲשֶׁר אָהַבְתִּי וְהָבִיאָה לִּי וְאֹכֵלָה בַּֽעֲבוּר תְּבָֽרֶכְךָ נַפְשִׁי בְּטֶרֶם אָמֽוּת:  ת  וְעִבֵיד לִי תַבְשִׁילִין כְּמָא דִרְחֵימִית וְאָעֵל לִי וְאֵיכוּל בְּדִיל דִּי תְבָרֵכִנָּךְ נַפְשִׁי עַד לָא אֵימוּת:

{ה} וְרִבְקָה שֹׁמַעַת בְּדַבֵּר יִצְחָק אֶל עֵשָׂו בְּנוֹ וַיֵּלֶךְ עֵשָׂו הַשָּׂדֶה לָצוּד צַיִד לְהָבִֽיא:  מ  וְרִבְקָה שֹׁמַעַת בְּדַבֵּר יִצְחָק אֶל עֵשָׂו בְּנוֹ וַיֵּלֶךְ עֵשָׂו הַשָּׂדֶה לָצוּד צַיִד לְהָבִֽיא:  ת  וְרִבְקָה שְׁמַעַת כַּד מַלִּיל יִצְחָק לְוַת עֵשָׂו בְּרֵהּ וַאֲזַל עֵשָׂו לְחַקְלָא לְמֵיצַד צֵידָא לְאַיְתָאָה:

{ו} וְרִבְקָה אָֽמְרָה אֶל יַֽעֲקֹב בְּנָהּ לֵאמֹר הִנֵּה שָׁמַעְתִּי אֶת אָבִיךָ מְדַבֵּר אֶל עֵשָׂו אָחִיךָ לֵאמֹֽר:  מ  וְרִבְקָה אָֽמְרָה אֶל יַֽעֲקֹב בְּנָהּ לֵאמֹר הִנֵּה שָׁמַעְתִּי אֶת אָבִיךָ מְדַבֵּר אֶל עֵשָׂו אָחִיךָ לֵאמֹֽר:  ת  וְרִבְקָה אֲמָרַת לְוַת יַעֲקֹב בְּרַהּ לְמֵימָר הָא שְׁמָעִית יַת אֲבוּךְ מְמַלֵּל עִם עֵשָׂו אַחוּךְ לְמֵימָר:

{ז} הָבִיאָה לִּי צַיִד וַֽעֲשֵׂה לִי מַטְעַמִּים וְאֹכֵלָה וַֽאֲבָֽרֶכְכָה לִפְנֵי יְהֹוָה לִפְנֵי מוֹתִֽי:  מ  הָבִיאָה לִּי צַיִד וַֽעֲשֵׂה לִי מַטְעַמִּים וְאֹכֵלָה וַֽאֲבָֽרֶכְכָה לִפְנֵי יְהֹוָה לִפְנֵי מוֹתִֽי:  ת  אַיְתִי לִי צֵידָא וְעִבֵּיד לִי תַבְשִׁילִין וְאֵיכוּל וֶאֱבָרֵכִנָּךְ קֳדָם יְיָ קֳדָם מוֹתִי:

{ח} וְעַתָּה בְנִי שְׁמַע בְּקֹלִי לַֽאֲשֶׁר אֲנִי מְצַוָּה אֹתָֽךְ:  מ  וְעַתָּה בְנִי שְׁמַע בְּקֹלִי לַֽאֲשֶׁר אֲנִי מְצַוָּה אֹתָֽךְ:  ת  וּכְעַן בְּרִי קַבֵּל מִנִי לְמָא דִי אֲנָא מפַקֵד יָתָךְ:

{ט} לֶךְ נָא אֶל הַצֹּאן וְקַח לִי מִשָּׁם שְׁנֵי גְּדָיֵי עִזִּים טֹבִים וְאֶֽעֱשֶׂה אֹתָם מַטְעַמִּים לְאָבִיךָ כַּֽאֲשֶׁר אָהֵֽב:  מ  לֶךְ נָא אֶל הַצֹּאן וְקַח לִי מִשָּׁם שְׁנֵי גְּדָיֵי עִזִּים טֹבִים וְאֶֽעֱשֶׂה אֹתָם מַטְעַמִּים לְאָבִיךָ כַּֽאֲשֶׁר אָהֵֽב:  ת  אִזֵיל כְּעַן לְוַת עָנָא וְסַב לִי מִתַּמָּן תְּרֵין גַדְיֵי (בַר) עִזְּין טָבָן וְאֶעְבֵּד יָתְהוֹן תַּבְשִׁילִין לְאָבוּךְ כְּמָא דִי רְחֵם:

{י} וְהֵֽבֵאתָ לְאָבִיךָ וְאָכָל בַּֽעֲבֻר אֲשֶׁר יְבָֽרֶכְךָ לִפְנֵי מוֹתֽוֹ:  מ  וְהֵֽבֵאתָ לְאָבִיךָ וְאָכָל בַּֽעֲבֻר אֲשֶׁר יְבָֽרֶכְךָ לִפְנֵי מוֹתֽוֹ:  ת  וְתַיְתִי (נ''י וְתָעֵל) לְאָבוּךְ וְיֵכוּל בְּדִיל דִּיְבָרְכִנָּךְ קֳדָם מוֹתֵהּ:

{יא} וַיֹּאמֶר יַֽעֲקֹב אֶל רִבְקָה אִמּוֹ הֵן עֵשָׂו אָחִי אִישׁ שָׂעִר וְאָֽנֹכִי אִישׁ חָלָֽק:  מ  וַיֹּאמֶר יַֽעֲקֹב אֶל רִבְקָה אִמּוֹ הֵן עֵשָׂו אָחִי אִישׁ שָׂעִר וְאָֽנֹכִי אִישׁ חָלָֽק:  ת  וַאֲמַר יַעֲקֹב לְרִבְקָה אִמֵּהּ הָא עֵשָׂו אָחִי גְּבַר שַׂעֲרָן וַאֲנָא גְּבַר שְׁעִיעַ:

{יב} אוּלַי יְמֻשֵּׁנִי אָבִי וְהָיִיתִי בְעֵינָיו כִּמְתַעְתֵּעַ וְהֵֽבֵאתִי עָלַי קְלָלָה וְלֹא בְרָכָֽה:  מ  אוּלַי יְמֻשֵּׁנִי אָבִי וְהָיִיתִי בְעֵינָיו כִּמְתַעְתֵּעַ וְהֵֽבֵאתִי עָלַי קְלָלָה וְלֹא בְרָכָֽה:  ת  מָאִים יְמֻשִּׁנַּנִי אַבָּא וְאֵהֵי בְעֵינוֹהִי כְּמִתְלָעָב וְאֵהֵי מַיְתֵי (נ''י וָאֱהִי מַיְתֵי) עֲלַי לְוָטִין וְלָא בִרְכָן:

{יג} וַתֹּאמֶר לוֹ אִמּוֹ עָלַי קִלְלָֽתְךָ בְּנִי אַךְ שְׁמַע בְּקֹלִי וְלֵךְ קַח לִֽי:  מ  וַתֹּאמֶר לוֹ אִמּוֹ עָלַי קִלְלָֽתְךָ בְּנִי אַךְ שְׁמַע בְּקֹלִי וְלֵךְ קַח לִֽי:  ת  וַאֲמֶרֶת לֵהּ אִמֵּהּ עֲלַי אִתְאַמַּר בִּנְבוּאָה דְּלָא יֵיתוּן לְוָטַּיָּא עֲלָךְ בְּרִי בְּרַם קַבֵּל מִנִּי וְאִזְּיל סַב לִי:

{יד} וַיֵּלֶךְ וַיִּקַּח וַיָּבֵא לְאִמּוֹ וַתַּעַשׂ אִמּוֹ מַטְעַמִּים כַּֽאֲשֶׁר אָהֵב אָבִֽיו:  מ  וַיֵּלֶךְ וַיִּקַּח וַיָּבֵא לְאִמּוֹ וַתַּעַשׂ אִמּוֹ מַטְעַמִּים כַּֽאֲשֶׁר אָהֵב אָבִֽיו:  ת  וַאֲזַל וּנְסִיב וְאַיְתִי לְאִמֵּהּ וַעֲבָדַת אִמֵּהּ תַּבְשִׁילִין כְּמָא דִּרְחֵם אֲבוּהִי:

{טו} וַתִּקַּח רִבְקָה אֶת בִּגְדֵי עֵשָׂו בְּנָהּ הַגָּדֹל הַֽחֲמֻדֹת אֲשֶׁר אִתָּהּ בַּבָּיִת וַתַּלְבֵּשׁ אֶֽת יַֽעֲקֹב בְּנָהּ הַקָּטָֽן:  מ  וַתִּקַּח רִבְקָה אֶת בִּגְדֵי עֵשָׂו בְּנָהּ הַגָּדֹל הַֽחֲמֻדֹת אֲשֶׁר אִתָּהּ בַּבָּיִת וַתַּלְבֵּשׁ אֶֽת יַֽעֲקֹב בְּנָהּ הַקָּטָֽן:  ת  וּנְסִיבַת רִבְקָה יָת לְבוּשֵׁי עֵשָׂו בְּרַהּ רַבָּא דָּכְיָתָא דִּי עִמַּהּ בְּבֵיתָא וְאַלְבִּישַׁת יָת יַעֲקֹב בְּרַהּ זְעֵירָא:

{טז} וְאֵת עֹרֹת גְּדָיֵי הָֽעִזִּים הִלְבִּישָׁה עַל יָדָיו וְעַל חֶלְקַת צַוָּארָֽיו:  מ  וְאֵת עֹרֹת גְּדָיֵי הָֽעִזִּים הִלְבִּישָׁה עַל יָדָיו וְעַל חֶלְקַת צַוָּארָֽיו:  ת  וְיָת מַשְׁכֵי גַּדְיֵי בַּר עִזֵּי אַלְבִּישַׁת עַל יְדוֹהִי וְעַל שְׁעִיעוּת צַוְרֵהּ:

{יז} וַתִּתֵּן אֶת הַמַּטְעַמִּים וְאֶת הַלֶּחֶם אֲשֶׁר עָשָׂתָה בְּיַד יַֽעֲקֹב בְּנָֽהּ:  מ  וַתִּתֵּן אֶת הַמַּטְעַמִּים וְאֶת הַלֶּחֶם אֲשֶׁר עָשָׂתָה בְּיַד יַֽעֲקֹב בְּנָֽהּ:  ת  וִיהָבַת יָת תַּבְשִׁילַיָּא וְיָת לַחְמָא דִּי עֲבָדַת בִּידָא דְיַעֲקֹב בְּרַהּ:

{יח} וַיָּבֹא אֶל אָבִיו וַיֹּאמֶר אָבִי וַיֹּאמֶר הִנֶּנִּי מִי אַתָּה בְּנִֽי:  מ  וַיָּבֹא אֶל אָבִיו וַיֹּאמֶר אָבִי וַיֹּאמֶר הִנֶּנִּי מִי אַתָּה בְּנִֽי:  ת  וְעַל לְוַת אֲבוּהִי וַאֲמַר אַבָּא וַאֲמַר הָא אֲנָא מָן אַתְּ בְּרִי:

{יט} וַיֹּאמֶר יַֽעֲקֹב אֶל אָבִיו אָֽנֹכִי עֵשָׂו בְּכֹרֶךָ עָשִׂיתִי כַּֽאֲשֶׁר דִּבַּרְתָּ אֵלָי קוּם נָא שְׁבָה וְאָכְלָה מִצֵּידִי בַּֽעֲבוּר תְּבָֽרֲכַנִּי נַפְשֶֽׁךָ:  מ  וַיֹּאמֶר יַֽעֲקֹב אֶל אָבִיו אָֽנֹכִי עֵשָׂו בְּכֹרֶךָ עָשִׂיתִי כַּֽאֲשֶׁר דִּבַּרְתָּ אֵלָי קוּם נָא שְׁבָה וְאָכְלָה מִצֵּידִי בַּֽעֲבוּר תְּבָֽרֲכַנִּי נַפְשֶֽׁךָ:  ת  וַאֲמַר יַעֲקֹב לַאֲבוּהִי אֲנָא עֵשָׂו בּוּכְרָךְ עֲבָדִית כּמָא דִי מַלֵּילְתָּא עִמִּי (נ''י:לִי) קוּם כְּעַן אִסְתַּחַר וֶאֱכוּל מִצֵידִי בְּדִיל דִּי תְבָרְכִנַּנִּי נַפְשָׁךְ:

{כ} וַיֹּאמֶר יִצְחָק אֶל בְּנוֹ מַה זֶּה מִהַרְתָּ לִמְצֹא בְּנִי וַיֹּאמֶר כִּי הִקְרָה יְהֹוָה אֱלֹהֶיךָ לְפָנָֽי:  מ  וַיֹּאמֶר יִצְחָק אֶל בְּנוֹ מַה זֶּה מִהַרְתָּ לִמְצֹא בְּנִי וַיֹּאמֶר כִּי הִקְרָה יְהֹוָה אֱלֹהֶיךָ לְפָנָֽי:  ת  וַאֲמַר יִצְחָק לִבְרֵהּ מָה דֵּין אוֹחֵיתָא לְאַשְׁכָּחָא בְּרִי וַאֲמַר אֲרֵי זַמִּין יְיָ אֱלָהָךְ קֳדָמַי:

{כא} וַיֹּאמֶר יִצְחָק אֶֽל יַֽעֲקֹב גְּשָׁה נָּא וַֽאֲמֻֽשְׁךָ בְּנִי הַֽאַתָּה זֶה בְּנִי עֵשָׂו אִם לֹֽא:  מ  וַיֹּאמֶר יִצְחָק אֶֽל יַֽעֲקֹב גְּשָׁה נָּא וַֽאֲמֻֽשְׁךָ בְּנִי הַֽאַתָּה זֶה בְּנִי עֵשָׂו אִם לֹֽא:  ת  וַאֲמַר יִצְחָק לְיַעֲקֹב קְרֵיב כְּעַן וֶאֱמֻשִּׁנָּךְ בְּרִי הַאַתְּ דֵּין בְּרִי עֵשָׂו אִם לָא:

{כב} וַיִּגַּשׁ יַֽעֲקֹב אֶל יִצְחָק אָבִיו וַיְמֻשֵּׁהוּ וַיֹּאמֶר הַקֹּל קוֹל יַֽעֲקֹב וְהַיָּדַיִם יְדֵי עֵשָֽׂו:  מ  וַיִּגַּשׁ יַֽעֲקֹב אֶל יִצְחָק אָבִיו וַיְמֻשֵּׁהוּ וַיֹּאמֶר הַקֹּל קוֹל יַֽעֲקֹב וְהַיָּדַיִם יְדֵי עֵשָֽׂו:  ת  וּקְרֵיב יַעֲקֹב לְוַת יִצְחָק אֲבוּהִי וּמָשְׁיֵהּ וַאֲמַר קָלָא קָלָא דְיַעֲקֹב וִידַיָּא יְדוֹהִי דְעֵשָׂו:

{כג} וְלֹא הִכִּירוֹ כִּֽי הָיוּ יָדָיו כִּידֵי עֵשָׂו אָחִיו שְׂעִרֹת וַֽיְבָֽרֲכֵֽהוּ:  מ  וְלֹא הִכִּירוֹ כִּֽי הָיוּ יָדָיו כִּידֵי עֵשָׂו אָחִיו שְׂעִרֹת וַֽיְבָֽרֲכֵֽהוּ:  ת  וְלָא אִשְׁתְּמוֹדְעֵהּ אֲרֵי הֲוָאָה יְדוֹהִי כִּידֵי עֵשָׂו אֲחוּהִי שְׂעִירָן (נ''י שַׂעֲרַנְיָן) וּבָרְכֵהּ:

{כד} וַיֹּאמֶר אַתָּה זֶה בְּנִי עֵשָׂו וַיֹּאמֶר אָֽנִי:  מ  וַיֹּאמֶר אַתָּה זֶה בְּנִי עֵשָׂו וַיֹּאמֶר אָֽנִי:  ת  וַאֲמַר אַתְּ דֵּין בְּרִי עֵשָׂו וַאֲמַר הָא אֲנָא:

{כה} וַיֹּאמֶר הַגִּשָׁה לִּי וְאֹֽכְלָה מִצֵּיד בְּנִי לְמַעַן תְּבָֽרֶכְךָ נַפְשִׁי וַיַּגֶּשׁ לוֹ וַיֹּאכַל וַיָּבֵא לוֹ יַיִן וַיֵּֽשְׁתְּ:  מ  וַיֹּאמֶר הַגִּשָׁה לִּי וְאֹֽכְלָה מִצֵּיד בְּנִי לְמַעַן תְּבָֽרֶכְךָ נַפְשִׁי וַיַּגֶּשׁ לוֹ וַיֹּאכַל וַיָּבֵא לוֹ יַיִן וַיֵּֽשְׁתְּ:  ת  וַאֲמַר קָרֵיב קֳדָמַי וְאֵיכוּל מִצֵידָא דִּבְרִי בְּדִיל דִּי תְבָרְכִנָּךְ נַפְשִׁי וְקָרֵיב לֵהּ וַאֲכַל וְאַיְתִי (נ''י וְאָעֵל) לֵהּ חַמְרָא וּשְׁתִי:

{כו} וַיֹּאמֶר אֵלָיו יִצְחָק אָבִיו גְּשָׁה נָּא וּֽשֲׁקָה לִּי בְּנִֽי:  מ  וַיֹּאמֶר אֵלָיו יִצְחָק אָבִיו גְּשָׁה נָּא וּֽשֲׁקָה לִּי בְּנִֽי:  ת  וַאֲמַר לֵהּ יִצְחָק אֲבוּהִי קְרֵיב כְּעַן וּנְשַׁק לִי (נ''י וְשַׁק לִי) בְּרִי:

{כז} וַיִּגַּשׁ וַיִּֽשַּׁק לוֹ וַיָּרַח אֶת רֵיחַ בְּגָדָיו וַֽיְבָֽרֲכֵהוּ וַיֹּאמֶר רְאֵה רֵיחַ בְּנִי כְּרֵיחַ שָׂדֶה אֲשֶׁר בֵּֽרֲכוֹ יְהֹוָֽה:  מ  וַיִּגַּשׁ וַיִּֽשַּׁק לוֹ וַיָּרַח אֶת רֵיחַ בְּגָדָיו וַֽיְבָֽרֲכֵהוּ וַיֹּאמֶר רְאֵה רֵיחַ בְּנִי כְּרֵיחַ שָׂדֶה אֲשֶׁר בֵּֽרֲכוֹ יְהֹוָֽה:  ת  וּקְרֵב וּנְשַׁק לֵהּ וַאֲרַח יָת רֵיחָא דִּלְבוּשׁוֹהִי וּבָרְכֵהּ וַאֲמַר חֲזֵי רֵיחָא דִּבְרִי כְּרֵיחָא דְחַקְלָא דִּי בָרֲכֵהּ יְיָ:

{כח}  שישי  וְיִֽתֶּן לְךָ הָֽאֱלֹהִים מִטַּל הַשָּׁמַיִם וּמִשְׁמַנֵּי הָאָרֶץ וְרֹב דָּגָן וְתִירֽשׁ:  מ  וְיִֽתֶּן לְךָ הָֽאֱלֹהִים מִטַּל הַשָּׁמַיִם וּמִשְׁמַנֵּי הָאָרֶץ וְרֹב דָּגָן וְתִירֽשׁ:  ת  וְיִתֶּן לָךְ יְיָ מִטַּלָּא דִּשְׁמַיָּא וּמִטּוּבָא דְאַרְעָא וּסְגִיאוּת (נ''י וְסַגְיוּת) עִיבוּר וַחֲמָר:

{כט} יַֽעַבְדוּךָ עַמִּים (וישתחו) וְיִשְׁתַּחֲווּ לְךָ לְאֻמִּים הֱוֵה גְבִיר לְאַחֶיךָ וְיִשְׁתַּֽחֲווּ לְךָ בְּנֵי אִמֶּךָ אֹֽרֲרֶיךָ אָרוּר וּֽמְבָֽרֲכֶיךָ בָּרֽוּךְ:  מ  יַֽעַבְדוּךָ עַמִּים (וישתחו) וְיִשְׁתַּחֲווּ לְךָ לְאֻמִּים הֱוֵה גְבִיר לְאַחֶיךָ וְיִשְׁתַּֽחֲווּ לְךָ בְּנֵי אִמֶּךָ אֹֽרֲרֶיךָ אָרוּר וּֽמְבָֽרֲכֶיךָ בָּרֽוּךְ:  ת  יִפְלְחֻנָּךְ עַמְמִין וְיִשְׁתַּעְבְּדוּן לָךְ מַלְכְּוָן הֱוֵי רַב לְאַחָיךְ וְיִסְגְּדוּן לָךְ בְּנֵי אִמָּךְ מְלַטְטָיךְ יְהוֹן לִיטִין וּמְבָרֲכָיךְ יְהוֹן בְּרִיכִין:

{ל} וַיְהִי כַּֽאֲשֶׁר כִּלָּה יִצְחָק לְבָרֵךְ אֶת יַֽעֲקֹב וַיְהִי אַךְ יָצֹא יָצָא יַֽעֲקֹב מֵאֵת פְּנֵי יִצְחָק אָבִיו וְעֵשָׂו אָחִיו בָּא מִצֵּידֽוֹ:  מ  וַיְהִי כַּֽאֲשֶׁר כִּלָּה יִצְחָק לְבָרֵךְ אֶת יַֽעֲקֹב וַיְהִי אַךְ יָצֹא יָצָא יַֽעֲקֹב מֵאֵת פְּנֵי יִצְחָק אָבִיו וְעֵשָׂו אָחִיו בָּא מִצֵּידֽוֹ:  ת  וַהֲוָה כַּד שֵׁיצֵי יִצְחָק לְבָרָכָא יָת יַעֲקֹב וַהֲוָה בְּרַם מִפַּק נְפַק יַעֲקֹב מִלְּוַת אַפֵּי יִצְחָק אֲבוּהִי וְעֵשָׂו אֲחוּהִי אָתָא (נ''י עַל) מִצֵידֵהּ:

{לא} וַיַּעַשׂ גַּם הוּא מַטְעַמִּים וַיָּבֵא לְאָבִיו וַיֹּאמֶר לְאָבִיו יָקֻם אָבִי וְיֹאכַל מִצֵּיד בְּנוֹ בַּֽעֲבֻר תְּבָֽרֲכַנִּי נַפְשֶֽׁךָ:  מ  וַיַּעַשׂ גַּם הוּא מַטְעַמִּים וַיָּבֵא לְאָבִיו וַיֹּאמֶר לְאָבִיו יָקֻם אָבִי וְיֹאכַל מִצֵּיד בְּנוֹ בַּֽעֲבֻר תְּבָֽרֲכַנִּי נַפְשֶֽׁךָ:  ת  וַעֲבַד אַף הוּא תַּבְשִׁילִין וְאָעֵיל לְוַת אֲבוּהִי וַאֲמַר לַאֲבוּהִי יָקוּם אַבָּא וְיֵכוּל מִצֵידָא דִּבְרֵהּ בְּדִיל דִּי תְבָרְכִנַּנִּי נַפְשָׁךְ:

{לב} וַיֹּאמֶר לוֹ יִצְחָק אָבִיו מִי אָתָּה וַיֹּאמֶר אֲנִי בִּנְךָ בְכֹֽרְךָ עֵשָֽׂו:  מ  וַיֹּאמֶר לוֹ יִצְחָק אָבִיו מִי אָתָּה וַיֹּאמֶר אֲנִי בִּנְךָ בְכֹֽרְךָ עֵשָֽׂו:  ת  וַאֲמַר לֵהּ יִצְחָק אֲבוּהִי מָן אָתְּ וַאֲמַר אֲנָא בְּרָךְ בּוּכְרָךְ עֵשָׂו:

{לג} וַיֶּֽחֱרַד יִצְחָק חֲרָדָה גְּדֹלָה עַד מְאֹד וַיֹּאמֶר מִֽי אֵפוֹא הוּא הַצָּֽד צַיִד וַיָּבֵא לִי וָֽאֹכַל מִכֹּל בְּטֶרֶם תָּבוֹא וָֽאֲבָֽרֲכֵהוּ גַּם בָּרוּךְ יִֽהְיֶֽה:  מ  וַיֶּֽחֱרַד יִצְחָק חֲרָדָה גְּדֹלָה עַד מְאֹד וַיֹּאמֶר מִֽי אֵפוֹא הוּא הַצָּֽד צַיִד וַיָּבֵא לִי וָֽאֹכַל מִכֹּל בְּטֶרֶם תָּבוֹא וָֽאֲבָֽרֲכֵהוּ גַּם בָּרוּךְ יִֽהְיֶֽה:  ת  וּתְוָה יִצְחָק תִּוְהָא רַבָּא עַד לַחֲדָא וַאֲמַר מָן הוּא דֵּיכִי דְצָד צֵידָא וְאָעֵיל לִי וַאֲכָלִית מִכֹּלָּא עַד לָא תֵיעוֹל וּבָרֵכִתֵּהּ אַף בְּרִיךְ יְהֵי:

{לד} כִּשְׁמֹעַ עֵשָׂו אֶת דִּבְרֵי אָבִיו וַיִּצְעַק צְעָקָה גְּדֹלָה וּמָרָה עַד מְאֹד וַיֹּאמֶר לְאָבִיו בָּֽרֲכֵנִי גַם אָנִי אָבִֽי:  מ  כִּשְׁמֹעַ עֵשָׂו אֶת דִּבְרֵי אָבִיו וַיִּצְעַק צְעָקָה גְּדֹלָה וּמָרָה עַד מְאֹד וַיֹּאמֶר לְאָבִיו בָּֽרֲכֵנִי גַם אָנִי אָבִֽי:  ת  כַּד שְׁמַע עֵשָׂו יָת פִּתְגָּמֵי אֲבוּהִי וּצְוַח צִוְחָא רַבָּא וּמְרִירָא עַד לַחֲדָא וַאֲמַר לַאֲבוּהִי בָּרֵכְנִי אַף אֲנָא (נ''י לִי) אַבָּא:

{לה} וַיֹּאמֶר בָּא אָחִיךָ בְּמִרְמָה וַיִּקַּח בִּרְכָתֶֽךָ:  מ  וַיֹּאמֶר בָּא אָחִיךָ בְּמִרְמָה וַיִּקַּח בִּרְכָתֶֽךָ:  ת  וַאֲמַר עַל אָחוּךְ בְּחָכְמְתָא וְקַבִּיל בִּרְכְּתָךְ:

{לו} וַיֹּאמֶר הֲכִי קָרָא שְׁמוֹ יַֽעֲקֹב וַֽיַּעְקְבֵנִי זֶה פַֽעֲמַיִם אֶת בְּֽכֹרָתִי לָקָח וְהִנֵּה עַתָּה לָקַח בִּרְכָתִי וַיֹּאמַר הֲלֹֽא אָצַלְתָּ לִּי בְּרָכָֽה:  מ  וַיֹּאמֶר הֲכִי קָרָא שְׁמוֹ יַֽעֲקֹב וַֽיַּעְקְבֵנִי זֶה פַֽעֲמַיִם אֶת בְּֽכֹרָתִי לָקָח וְהִנֵּה עַתָּה לָקַח בִּרְכָתִי וַיֹּאמַר הֲלֹֽא אָצַלְתָּ לִּי בְּרָכָֽה:  ת  וַאֲמַר יָאוּת קְרָא שְׁמֵהּ יַעֲקֹב וְכָמַנִי דְּנָן תַּרְתֵּין זִמְנִין יָת בְּכֵירוּתִי נְסִיב וְהָא כְּעַן קַבִּיל בִּרְכְּתִי וַאֲמַר הֲלָא שְׁבַקְתָּ לִי בִּרְכְּתָא:

{לז} וַיַּעַן יִצְחָק וַיֹּאמֶר לְעֵשָׂו הֵן גְּבִיר שַׂמְתִּיו לָךְ וְאֶת כָּל אֶחָיו נָתַתִּי לוֹ לַֽעֲבָדִים וְדָגָן וְתִירשׁ סְמַכְתִּיו וּלְכָה אֵפוֹא מָה אֶֽעֱשֶׂה בְּנִֽי:  מ  וַיַּעַן יִצְחָק וַיֹּאמֶר לְעֵשָׂו הֵן גְּבִיר שַׂמְתִּיו לָךְ וְאֶת כָּל אֶחָיו נָתַתִּי לוֹ לַֽעֲבָדִים וְדָגָן וְתִירשׁ סְמַכְתִּיו וּלְכָה אֵפוֹא מָה אֶֽעֱשֶׂה בְּנִֽי:  ת  וַאֲתֵיב יִצְחָק וַאֲמַר לְעֵשָׂו הָא רַב שַׁוִּתֵּהּ לָךְ (נ''י עִלַּוָּךְ) וְיָת כָּל אֲחוֹהִי יְהָבִית לֵהּ לְעַבְדִּין וְעִיבוּר וַחֲמָר סְעִדְתֵּהּ וְלָךְ הָכָא מָה אֶעְבֵּיד בְּרִי:

{לח} וַיֹּאמֶר עֵשָׂו אֶל אָבִיו הַֽבֲרָכָה אַחַת הִֽוא לְךָ אָבִי בָּֽרֲכֵנִי גַם אָנִי אָבִי וַיִּשָּׂא עֵשָׂו קֹלוֹ וַיֵּֽבְךְּ:  מ  וַיֹּאמֶר עֵשָׂו אֶל אָבִיו הַֽבֲרָכָה אַחַת הִֽוא לְךָ אָבִי בָּֽרֲכֵנִי גַם אָנִי אָבִי וַיִּשָּׂא עֵשָׂו קֹלוֹ וַיֵּֽבְךְּ:  ת  וַאֲמַר עֵשָׂו לַאֲבוּהִי הֲבִרְכְּתָא חֲדָא הִיא לָךְ אַבָּא בָּרֵךְ לִי אַף אֲנָא אַבָּא וַאֲרֵים עֵשָׂו קָלֵהּ וּבְכָא:

{לט} וַיַּעַן יִצְחָק אָבִיו וַיֹּאמֶר אֵלָיו הִנֵּה מִשְׁמַנֵּי הָאָרֶץ יִֽהְיֶה מֽוֹשָׁבֶךָ וּמִטַּל הַשָּׁמַיִם מֵעָֽל:  מ  וַיַּעַן יִצְחָק אָבִיו וַיֹּאמֶר אֵלָיו הִנֵּה מִשְׁמַנֵּי הָאָרֶץ יִֽהְיֶה מֽוֹשָׁבֶךָ וּמִטַּל הַשָּׁמַיִם מֵעָֽל:  ת  וַאֲתֵיב יִצְחָק אֲבוּהִי וַאֲמַר לֵהּ הָא מִטּוּבָא דְאַרְעָא יְהֵא מוֹתְבָךְ וּמִטַּלָּא דִּשְׁמַיָּא מִלְּעֵלָא:

{מ} וְעַל חַרְבְּךָ תִֽחְיֶה וְאֶת אָחִיךָ תַּֽעֲבֹד וְהָיָה כַּֽאֲשֶׁר תָּרִיד וּפָֽרַקְתָּ עֻלּוֹ מֵעַל צַוָּארֶֽךָ:  מ  וְעַל חַרְבְּךָ תִֽחְיֶה וְאֶת אָחִיךָ תַּֽעֲבֹד וְהָיָה כַּֽאֲשֶׁר תָּרִיד וּפָֽרַקְתָּ עֻלּוֹ מֵעַל צַוָּארֶֽךָ:  ת  וְעַל חַרְבָּךְ תֵּיחֵי וְיָת אָחוּךְ תִּפְלָח וִיהֵי כַּד יַעְבְּרוּן בְּנוֹהִי עַל פִּתְגָּמֵי אוֹרַיְתָא וְתַעְדִּי נִירֵהּ מֵעַל צַוְרָךְ:

{מא} וַיִּשְׂטֹם עֵשָׂו אֶת יַֽעֲקֹב עַל הַבְּרָכָה אֲשֶׁר בֵּֽרֲכוֹ אָבִיו וַיֹּאמֶר עֵשָׂו בְּלִבּוֹ יִקְרְבוּ יְמֵי אֵבֶל אָבִי וְאַֽהַרְגָה אֶת יַֽעֲקֹב אָחִֽי:  מ  וַיִּשְׂטֹם עֵשָׂו אֶת יַֽעֲקֹב עַל הַבְּרָכָה אֲשֶׁר בֵּֽרֲכוֹ אָבִיו וַיֹּאמֶר עֵשָׂו בְּלִבּוֹ יִקְרְבוּ יְמֵי אֵבֶל אָבִי וְאַֽהַרְגָה אֶת יַֽעֲקֹב אָחִֽי:  ת  וּנְטַר עֵשָׂו דְּבָבוּ לְיַעֲקֹב עַל בִּרְכְּתָא דִּי בָרְכֵהּ אֲבוּהִי וַאֲמַר עֵשָׂו בְּלִבֵּהּ יִקְרְבוּן יוֹמֵי אֶבְלֵי דְאַבָּא וְאֶקְטוֹל יָת יַעֲקֹב אָחִי:

{מב} וַיֻּגַּד לְרִבְקָה אֶת דִּבְרֵי עֵשָׂו בְּנָהּ הַגָּדֹל וַתִּשְׁלַח וַתִּקְרָא לְיַֽעֲקֹב בְּנָהּ הַקָּטָן וַתֹּאמֶר אֵלָיו הִנֵּה עֵשָׂו אָחִיךָ מִתְנַחֵם לְךָ לְהָרְגֶֽךָ:  מ  וַיֻּגַּד לְרִבְקָה אֶת דִּבְרֵי עֵשָׂו בְּנָהּ הַגָּדֹל וַתִּשְׁלַח וַתִּקְרָא לְיַֽעֲקֹב בְּנָהּ הַקָּטָן וַתֹּאמֶר אֵלָיו הִנֵּה עֵשָׂו אָחִיךָ מִתְנַחֵם לְךָ לְהָרְגֶֽךָ:  ת  וְאִתְחַוָּה לְרִבְקָה יָת פִּתְגָּמֵי עֵשָׂו בְּרַהּ רַבָּא וּשְׁלָחַת וּקְרַת לְיַעֲקֹב בְּרַהּ זְעֵירָא וַאֲמֶרֶת לֵהּ הָא עֵשָׂו אָחוּךְ כָּמִן לָךְ לְמִקְטְלָךְ:

{מג} וְעַתָּה בְנִי שְׁמַע בְּקֹלִי וְקוּם בְּרַח לְךָ אֶל לָבָן אָחִי חָרָֽנָה:  מ  וְעַתָּה בְנִי שְׁמַע בְּקֹלִי וְקוּם בְּרַח לְךָ אֶל לָבָן אָחִי חָרָֽנָה:  ת  וּכְעַן בְּרִי קַבֵּל מִנִּי וְקוּם אִזֵּיל לָךְ לְוַת לָבָן אָחִי לְחָרָן:

{מד} וְיָֽשַׁבְתָּ עִמּוֹ יָמִים אֲחָדִים עַד אֲשֶׁר תָּשׁוּב חֲמַת אָחִֽיךָ:  מ  וְיָֽשַׁבְתָּ עִמּוֹ יָמִים אֲחָדִים עַד אֲשֶׁר תָּשׁוּב חֲמַת אָחִֽיךָ:  ת  וְתֵתֵיב עִמֵּהּ יוֹמִין זְעֵירִין עַד דִּיתוּב רוּגְזָא דְאָחוּךְ:

{מה} עַד שׁוּב אַף אָחִיךָ מִמְּךָ וְשָׁכַח אֵת אֲשֶׁר עָשִׂיתָ לּוֹ וְשָֽׁלַחְתִּי וּלְקַחְתִּיךָ מִשָּׁם לָמָה אֶשְׁכַּל גַּם שְׁנֵיכֶם יוֹם אֶחָֽד:  מ  עַד שׁוּב אַף אָחִיךָ מִמְּךָ וְשָׁכַח אֵת אֲשֶׁר עָשִׂיתָ לּוֹ וְשָֽׁלַחְתִּי וּלְקַחְתִּיךָ מִשָּׁם לָמָה אֶשְׁכַּל גַּם שְׁנֵיכֶם יוֹם אֶחָֽד:  ת  עַד דִּיתוּב רוּגְזָא דְאָחוּךְ מִנָּךְ וְיִתְנְשֵׁי יָת דִּי עֲבַדְתָּ לֵהּ וְאֶשְׁלַח וְאֶדְבְּרִנָּךְ מִתַּמָּן לְמָה אֶתְכַּל אַף תַּרְוֵיכוֹן יוֹמָא חָד:

{מו} וַתֹּאמֶר רִבְקָה אֶל יִצְחָק קַצְתִּי בְחַיַּי מִפְּנֵי בְּנוֹת חֵת אִם לֹקֵחַ יַֽעֲקֹב אִשָּׁה מִבְּנוֹת חֵת כָּאֵלֶּה מִבְּנוֹת הָאָרֶץ לָמָּה לִּי חַיִּֽים:  מ  וַתֹּאמֶר רִבְקָה אֶל יִצְחָק קַצְתִּי בְחַיַּי מִפְּנֵי בְּנוֹת חֵת אִם לֹקֵחַ יַֽעֲקֹב אִשָּׁה מִבְּנוֹת חֵת כָּאֵלֶּה מִבְּנוֹת הָאָרֶץ לָמָּה לִּי חַיִּֽים:  ת  וַאֲמֶרֶת רִבְקָה לְיִצְחָק עָקִית בְּחַיַּי מִן קֳדָם בְּנַת חִתָּאָה אִם נְסִיב יַעֲקֹב אִתְּתָא מִבְּנַת חִתָּאָה כְאִלֵּין מִבְּנַת אַרְעָא לְמָה לִי חַיִּים:




בראשית פרק-כח

{א} וַיִּקְרָא יִצְחָק אֶֽל יַֽעֲקֹב וַיְבָרֶךְ אֹתוֹ וַיְצַוֵּהוּ וַיֹּאמֶר לוֹ לֹֽא תִקַּח אִשָּׁה מִבְּנוֹת כְּנָֽעַן:  מ  וַיִּקְרָא יִצְחָק אֶֽל יַֽעֲקֹב וַיְבָרֶךְ אֹתוֹ וַיְצַוֵּהוּ וַיֹּאמֶר לוֹ לֹֽא תִקַּח אִשָּׁה מִבְּנוֹת כְּנָֽעַן:  ת  וּקְרָא יִצְחָק לְיַעֲקֹב וּבָרִיךְ יָתֵהּ וּפַקְדֵהּ וַאֲמַר לֵהּ לָא תִסַּב אִתְּתָא מִבְּנַת כְּנָעַן:

{ב} קוּם לֵךְ פַּדֶּנָֽה אֲרָם בֵּיתָה בְתוּאֵל אֲבִי אִמֶּךָ וְקַח לְךָ מִשָּׁם אִשָּׁה מִבְּנוֹת לָבָן אֲחִי אִמֶּֽךָ:  מ  קוּם לֵךְ פַּדֶּנָֽה אֲרָם בֵּיתָה בְתוּאֵל אֲבִי אִמֶּךָ וְקַח לְךָ מִשָּׁם אִשָּׁה מִבְּנוֹת לָבָן אֲחִי אִמֶּֽךָ:  ת  קוּם אִזֵּיל לְפַדַּן אֲרָם לְבֵית בְּתוּאֵל אֲבוּהָא דְאִמָּךְ וְסַב לָךְ מִתַּמָּן אִתְּתָא מִבְּנַת לָבָן אֲחוּהָא דְאִמָּךְ:

{ג} וְאֵל שַׁדַּי יְבָרֵךְ אֹֽתְךָ וְיַפְרְךָ וְיַרְבֶּךָ וְהָיִיתָ לִקְהַל עַמִּֽים:  מ  וְאֵל שַׁדַּי יְבָרֵךְ אֹֽתְךָ וְיַפְרְךָ וְיַרְבֶּךָ וְהָיִיתָ לִקְהַל עַמִּֽים:  ת  וְאֵל שַׁדַּי יְבָרֵךְ יָתָךְ וְיַפְּשָׁךְ וְיַסְגִּנָּךְ וּתְהֵי לְכִנְשַׁת שִׁבְטִין:

{ד} וְיִֽתֶּן לְךָ אֶת בִּרְכַּת אַבְרָהָם לְךָ וּלְזַרְעֲךָ אִתָּךְ לְרִשְׁתְּךָ אֶת אֶרֶץ מְגֻרֶיךָ אֲשֶׁר נָתַן אֱלֹהִים לְאַבְרָהָֽם:  מ  וְיִֽתֶּן לְךָ אֶת בִּרְכַּת אַבְרָהָם לְךָ וּלְזַרְעֲךָ אִתָּךְ לְרִשְׁתְּךָ אֶת אֶרֶץ מְגֻרֶיךָ אֲשֶׁר נָתַן אֱלֹהִים לְאַבְרָהָֽם:  ת  וְיִתֶּן לָךְ יָת בִּרְכְּתָא דְאַבְרָהָם לָךְ וְלִבְנָיךְ עִמָּךְ לְמֵירְתָךְ יָת אֲרַע תּוֹתָבוּתָךְ דִּי יְהַב יְיָ לְאַבְרָהָם:

{ה}  שביעי  וַיִּשְׁלַח יִצְחָק אֶֽת יַֽעֲקֹב וַיֵּלֶךְ פַּדֶּנָֽה אֲרָם אֶל לָבָן בֶּן בְּתוּאֵל הָֽאֲרַמִּי אֲחִי רִבְקָה אֵם יַֽעֲקֹב וְעֵשָֽׂו:  מ  וַיִּשְׁלַח יִצְחָק אֶֽת יַֽעֲקֹב וַיֵּלֶךְ פַּדֶּנָֽה אֲרָם אֶל לָבָן בֶּן בְּתוּאֵל הָֽאֲרַמִּי אֲחִי רִבְקָה אֵם יַֽעֲקֹב וְעֵשָֽׂו:  ת  וּשְׁלַח יִצְחָק יָת יַעֲקֹב וַאֲזַל לְפַדַּן אֲרָם לְוַת לָבָן בַּר בְּתוּאֵל אֲרַמָּאָה אֲחוּהָא דְרִבְקָה אִמֵּהּ דְּיַעֲקֹב וְעֵשָׂו:

{ו} וַיַּרְא עֵשָׂו כִּֽי בֵרַךְ יִצְחָק אֶת יַֽעֲקֹב וְשִׁלַּח אֹתוֹ פַּדֶּנָֽה אֲרָם לָקַֽחַת לוֹ מִשָּׁם אִשָּׁה בְּבָֽרֲכוֹ אֹתוֹ וַיְצַו עָלָיו לֵאמֹר לֹֽא תִקַּח אִשָּׁה מִבְּנוֹת כְּנָֽעַן:  מ  וַיַּרְא עֵשָׂו כִּֽי בֵרַךְ יִצְחָק אֶת יַֽעֲקֹב וְשִׁלַּח אֹתוֹ פַּדֶּנָֽה אֲרָם לָקַֽחַת לוֹ מִשָּׁם אִשָּׁה בְּבָֽרֲכוֹ אֹתוֹ וַיְצַו עָלָיו לֵאמֹר לֹֽא תִקַּח אִשָּׁה מִבְּנוֹת כְּנָֽעַן:  ת  וַחֲזָא עֵשָׂו אֲרֵי בָרִיךְ יִצְחָק יָת יַעֲקֹב וְשַׁלַּח יָתֵהּ לְפַדַּן אֲרָם לְמִסַּב לֵהּ מִתַּמָּן אִתְּתָא כַּד בָּרִיךְ יָתֵהּ וּפַקֵּיד עֲלוֹהִי לְמֵימַר לָא תִסַּב אִתְּתָא מִבְּנַת כְּנָעַן:

{ז}  מפטיר  וַיִּשְׁמַע יַֽעֲקֹב אֶל אָבִיו וְאֶל אִמּוֹ וַיֵּלֶךְ פַּדֶּנָֽה אֲרָֽם:  מ  וַיִּשְׁמַע יַֽעֲקֹב אֶל אָבִיו וְאֶל אִמּוֹ וַיֵּלֶךְ פַּדֶּנָֽה אֲרָֽם:  ת  וְקַבִּיל יַעֲקֹב מִן אֲבוּהִי וּמִן אִמֵּהּ וַאֲזַל לְפַדַּן אֲרָם:
