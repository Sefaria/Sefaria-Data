פרשת בחקתי


{ג} אִם-בְּחֻקֹּתַ֖י תֵּלֵ֑כוּ וְאֶת-מִצְוֹתַ֣י תִּשְׁמְר֔וּ וַֽעֲשִׂיתֶ֖ם אֹתָֽם:  מ  אִם-בְּחֻקֹּתַ֖י תֵּלֵ֑כוּ וְאֶת-מִצְוֹתַ֣י תִּשְׁמְר֔וּ וַֽעֲשִׂיתֶ֖ם אֹתָֽם:  ת  אִם בִּקְיָמַי תְּהָכוּן וְיָת פִּקוֹדַי תִּטְרוּן וְתַעֲבְּדוּן יָתְהוֹן:

{ד} וְנָֽתַתִּ֥י גִשְׁמֵיכֶ֖ם בְּעִתָּ֑ם וְנָֽתְנָ֤ה הָאָ֨רֶץ֙ יְבוּלָ֔הּ וְעֵ֥ץ הַשָּׂדֶ֖ה יִתֵּ֥ן פִּרְיֽוֹ:  מ  וְנָֽתַתִּ֥י גִשְׁמֵיכֶ֖ם בְּעִתָּ֑ם וְנָֽתְנָ֤ה הָאָ֨רֶץ֙ יְבוּלָ֔הּ וְעֵ֥ץ הַשָּׂדֶ֖ה יִתֵּ֥ן פִּרְיֽוֹ:  ת  וְאֶתֵּן מַטְּרְכוֹן בְּעִדָנְהוֹן וְתִתֵּן אַרְעָא עֲלַלְתָּהּ וְאִילַן חַקְלָא יִתֵּן אִבֵּהּ:

{ה} וְהִשִּׂ֨יג לָכֶ֥ם דַּ֨יִשׁ֙ אֶת-בָּצִ֔יר וּבָצִ֖יר יַשִּׂ֣יג אֶת-זָ֑רַע וַֽאֲכַלְתֶּ֤ם לַחְמְכֶם֙ לָשֹׂ֔בַע וִֽישַׁבְתֶּ֥ם לָבֶ֖טַח בְּאַרְצְכֶֽם:  מ  וְהִשִּׂ֨יג לָכֶ֥ם דַּ֨יִשׁ֙ אֶת-בָּצִ֔יר וּבָצִ֖יר יַשִּׂ֣יג אֶת-זָ֑רַע וַֽאֲכַלְתֶּ֤ם לַחְמְכֶם֙ לָשֹׂ֔בַע וִֽישַׁבְתֶּ֥ם לָבֶ֖טַח בְּאַרְצְכֶֽם:  ת  וִיעָרַע לְכוֹן דְיָשָׁא יָת קְטָפָא וּקְטָפָא יְעָרַע לְאַפּוֹקֵי בַר זַרְעָא וְתֵיכְלוּן לַחְמְכוֹן לְמִשְׂבַּע וְתֵיתְבוּן לְרָחְצָן בְּאַרְעַכוֹן:

{ו}  שני  וְנָֽתַתִּ֤י שָׁלוֹם֙ בָּאָ֔רֶץ וּשְׁכַבְתֶּ֖ם וְאֵ֣ין מַֽחֲרִ֑יד וְהִשְׁבַּתִּ֞י חַיָּ֤ה רָעָה֙ מִן-הָאָ֔רֶץ וְחֶ֖רֶב לֹא-תַֽעֲבֹ֥ר בְּאַרְצְכֶֽם:  מ  וְנָֽתַתִּ֤י שָׁלוֹם֙ בָּאָ֔רֶץ וּשְׁכַבְתֶּ֖ם וְאֵ֣ין מַֽחֲרִ֑יד וְהִשְׁבַּתִּ֞י חַיָּ֤ה רָעָה֙ מִן-הָאָ֔רֶץ וְחֶ֖רֶב לֹא-תַֽעֲבֹ֥ר בְּאַרְצְכֶֽם:  ת  וְאֶתֵּן שְׁלָמָא בְּאַרְעָא וְתִשְׁרוּן וְלֵית דְמָנִיד וֶאֱבַטֵל חַיְתָא בִשְׁתָּא מִן אַרְעָא וּדְקַטְלִין בְּחַרְבָּא לָא יְעִדוּן בְּאַרְעֲכוֹן:

{ז} וּרְדַפְתֶּ֖ם אֶת-אֹֽיְבֵיכֶ֑ם וְנָֽפְל֥וּ לִפְנֵיכֶ֖ם לֶחָֽרֶב:  מ  וּרְדַפְתֶּ֖ם אֶת-אֹֽיְבֵיכֶ֑ם וְנָֽפְל֥וּ לִפְנֵיכֶ֖ם לֶחָֽרֶב:  ת  וְתִרְדְפוּן יָת בַּעֲלֵי דְבָבֵיכוֹן וְיִפְּלוּן קֳדָמֵיכוֹן לְחַרְבָּא:

{ח} וְרָֽדְפ֨וּ מִכֶּ֤ם חֲמִשָּׁה֙ מֵאָ֔ה וּמֵאָ֥ה מִכֶּ֖ם רְבָבָ֣ה יִרְדֹּ֑פוּ וְנָֽפְל֧וּ אֹֽיְבֵיכֶ֛ם לִפְנֵיכֶ֖ם לֶחָֽרֶב:  מ  וְרָֽדְפ֨וּ מִכֶּ֤ם חֲמִשָּׁה֙ מֵאָ֔ה וּמֵאָ֥ה מִכֶּ֖ם רְבָבָ֣ה יִרְדֹּ֑פוּ וְנָֽפְל֧וּ אֹֽיְבֵיכֶ֛ם לִפְנֵיכֶ֖ם לֶחָֽרֶב:  ת  וְיִרְדְפוּן מִנְכוֹן חַמְשָׁא מְאָה וּמְאָה מִנְכוֹן לְרִבּוֹתָא יֵיעִירְקוּן וְיִפְּלוּן בַּעֲלֵי דְבָבֵיכוֹן קֳדָמֵיכוֹן לְחַרְבָּא:

{ט} וּפָנִ֣יתִי אֲלֵיכֶ֔ם וְהִפְרֵיתִ֣י אֶתְכֶ֔ם וְהִרְבֵּיתִ֖י אֶתְכֶ֑ם וַֽהֲקִֽימֹתִ֥י אֶת-בְּרִיתִ֖י אִתְּכֶֽם:  מ  וּפָנִ֣יתִי אֲלֵיכֶ֔ם וְהִפְרֵיתִ֣י אֶתְכֶ֔ם וְהִרְבֵּיתִ֖י אֶתְכֶ֑ם וַֽהֲקִֽימֹתִ֥י אֶת-בְּרִיתִ֖י אִתְּכֶֽם:  ת  וְאֶתְפְּנֵי בְּמֵימְרִי לְאוֹטָבָא לְכוֹן וְאַפֵּשׁ יָתְכוֹן וְאַסְגֵי יָתְכוֹן וַאֲקֵם יָת קְיָמִי עִמְכוֹן:

{י}  שלישי - חמישי במחוברין  וַֽאֲכַלְתֶּ֥ם יָשָׁ֖ן נוֹשָׁ֑ן וְיָשָׁ֕ן מִפְּנֵ֥י חָדָ֖שׁ תּוֹצִֽיאוּ:  מ  וַֽאֲכַלְתֶּ֥ם יָשָׁ֖ן נוֹשָׁ֑ן וְיָשָׁ֕ן מִפְּנֵ֥י חָדָ֖שׁ תּוֹצִֽיאוּ:  ת  וְתֵיכְלוּן עַתִּיקָא דְעַתִּיק וְעַתִּיקָא מִן קֳדָם חֲדַתָּא תְּפַנוּן:

{יא} וְנָֽתַתִּ֥י מִשְׁכָּנִ֖י בְּתֽוֹכֲכֶ֑ם וְלֹֽא-תִגְעַ֥ל נַפְשִׁ֖י אֶתְכֶֽם:  מ  וְנָֽתַתִּ֥י מִשְׁכָּנִ֖י בְּתֽוֹכֲכֶ֑ם וְלֹֽא-תִגְעַ֥ל נַפְשִׁ֖י אֶתְכֶֽם:  ת  וְאֶתֵּן מַשְׁכְּנִי בֵּינֵיכוֹן וְלָא יְרַחֵק מֵימְרִי יָתְכוֹן:

{יב} וְהִתְהַלַּכְתִּי֙ בְּת֣וֹכֲכֶ֔ם וְהָיִ֥יתִי לָכֶ֖ם לֵֽאלֹהִ֑ים וְאַתֶּ֖ם תִּֽהְיוּ-לִ֥י לְעָֽם:  מ  וְהִתְהַלַּכְתִּי֙ בְּת֣וֹכֲכֶ֔ם וְהָיִ֥יתִי לָכֶ֖ם לֵֽאלֹהִ֑ים וְאַתֶּ֖ם תִּֽהְיוּ-לִ֥י לְעָֽם:  ת  וְאַשְׁרֵי שְׁכִינְתִּי בֵּינֵיכוֹן וְאֶהֱוֵי לְכוֹן לֶאֱלָהּ וְאַתּוּן תְּהוֹן קֳדָמַי לְעָם:

{יג} אֲנִ֞י יְהוָֹ֣ה אֱלֹֽהֵיכֶ֗ם אֲשֶׁ֨ר הוֹצֵ֤אתִי אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם מִהְיֹ֥ת לָהֶ֖ם עֲבָדִ֑ים וָֽאֶשְׁבֹּר֙ מֹטֹ֣ת עֻלְּכֶ֔ם וָֽאוֹלֵ֥ךְ אֶתְכֶ֖ם קֽוֹמְמִיּֽוּת:  מ  אֲנִ֞י יְהוָֹ֣ה אֱלֹֽהֵיכֶ֗ם אֲשֶׁ֨ר הוֹצֵ֤אתִי אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם מִהְיֹ֥ת לָהֶ֖ם עֲבָדִ֑ים וָֽאֶשְׁבֹּר֙ מֹטֹ֣ת עֻלְּכֶ֔ם וָֽאוֹלֵ֥ךְ אֶתְכֶ֖ם קֽוֹמְמִיּֽוּת:  ת  אֲנָא יְיָ אֱלָהֲכוֹן דִי אַפֵּקִית יָתְכוֹן מֵאַרְעָא דְמִצְרַיִם מִלְמֶהֱוֵי לְהוֹן עַבְדִין וְתַבָּרִית נִיר עַמְמַיָא מִנְכוֹן וְדַבָּרִית יָתְכוֹן לְחֵרוּתָא: (פ)

{יד} וְאִם-לֹ֥א תִשְׁמְע֖וּ לִ֑י וְלֹ֣א תַֽעֲשׂ֔וּ אֵ֥ת כָּל-הַמִּצְוֹ֖ת הָאֵֽלֶּה:  מ  וְאִם-לֹ֥א תִשְׁמְע֖וּ לִ֑י וְלֹ֣א תַֽעֲשׂ֔וּ אֵ֥ת כָּל-הַמִּצְוֹ֖ת הָאֵֽלֶּה:  ת  וְאִם לָא תְקַבְּלוּן לְמֵימְרִי וְלָא תַעְבְּדוּן יָת כָּל פִּקוּדַיָא הָאִלֵין:

{טו} וְאִם-בְּחֻקֹּתַ֣י תִּמְאָ֔סוּ וְאִ֥ם אֶת-מִשְׁפָּטַ֖י תִּגְעַ֣ל נַפְשְׁכֶ֑ם לְבִלְתִּ֤י עֲשׂוֹת֙ אֶת-כָּל-מִצְוֹתַ֔י לְהַפְרְכֶ֖ם אֶת-בְּרִיתִֽי:  מ  וְאִם-בְּחֻקֹּתַ֣י תִּמְאָ֔סוּ וְאִ֥ם אֶת-מִשְׁפָּטַ֖י תִּגְעַ֣ל נַפְשְׁכֶ֑ם לְבִלְתִּ֤י עֲשׂוֹת֙ אֶת-כָּל-מִצְוֹתַ֔י לְהַפְרְכֶ֖ם אֶת-בְּרִיתִֽי:  ת  וְאִם בִּקְיָמַי תְּקוֹצוּן וְאִם יָת דִינַי תְּרַחֵק נַפְשְׁכוֹן בְּדִיל דְלָא לְמֶעְבַּד יָת כָּל פִּקוֹדַי לְאַשְׁנָיוּתְכוֹן יָת קְיָמִי:

{טז} אַף-אֲנִ֞י אֶֽעֱשֶׂה-זֹּ֣את לָכֶ֗ם וְהִפְקַדְתִּ֨י עֲלֵיכֶ֤ם בֶּֽהָלָה֙ אֶת-הַשַּׁחֶ֣פֶת וְאֶת-הַקַּדַּ֔חַת מְכַלּ֥וֹת עֵינַ֖יִם וּמְדִיבֹ֣ת נָ֑פֶשׁ וּזְרַעְתֶּ֤ם לָרִיק֙ זַֽרְעֲכֶ֔ם וַֽאֲכָלֻ֖הוּ אֹֽיְבֵיכֶֽם:  מ  אַף-אֲנִ֞י אֶֽעֱשֶׂה-זֹּ֣את לָכֶ֗ם וְהִפְקַדְתִּ֨י עֲלֵיכֶ֤ם בֶּֽהָלָה֙ אֶת-הַשַּׁחֶ֣פֶת וְאֶת-הַקַּדַּ֔חַת מְכַלּ֥וֹת עֵינַ֖יִם וּמְדִיבֹ֣ת נָ֑פֶשׁ וּזְרַעְתֶּ֤ם לָרִיק֙ זַֽרְעֲכֶ֔ם וַֽאֲכָלֻ֖הוּ אֹֽיְבֵיכֶֽם:  ת  אַף אֲנָא אֶעְבֵּד דָא לְכוֹן וְאַסְעַר עֲלֵיכוֹן בַּהֶלְתָּא יָת שַׁחֶפְתָּא וְיָת קַדַחְתָּא מְחַשְׁכָן עַיְנִין וּמַפְּחַן נְפָשׁ וְתִזְרְעוּן לְרֵקָנוּן זַרְעֲכוֹן וְיֵיכְלֻנֵהּ בַּעֲלֵי דְבָבֵיכוֹן:

{יז} וְנָֽתַתִּ֤י פָנַי֙ בָּכֶ֔ם וְנִגַּפְתֶּ֖ם לִפְנֵ֣י אֹֽיְבֵיכֶ֑ם וְרָד֤וּ בָכֶם֙ שֹֽׂנְאֵיכֶ֔ם וְנַסְתֶּ֖ם וְאֵין-רֹדֵ֥ף אֶתְכֶֽם:  מ  וְנָֽתַתִּ֤י פָנַי֙ בָּכֶ֔ם וְנִגַּפְתֶּ֖ם לִפְנֵ֣י אֹֽיְבֵיכֶ֑ם וְרָד֤וּ בָכֶם֙ שֹֽׂנְאֵיכֶ֔ם וְנַסְתֶּ֖ם וְאֵין-רֹדֵ֥ף אֶתְכֶֽם:  ת  וְאֶתֵּן רוּגְזִי בְּכוֹן וְתִתַּבְּרוּן קֳדָם בַּעֲלֵי דְבָבֵיכוֹן וְיִרְדוּן בְּכוֹן סַנְאֵכוֹן וְתֵעִירְקוּן וְלֵית דְרָדִיף יָתְכוֹן:

{יח} וְאִ֨ם-עַד-אֵ֔לֶּה לֹ֥א תִשְׁמְע֖וּ לִ֑י וְיָֽסַפְתִּי֙ לְיַסְּרָ֣ה אֶתְכֶ֔ם שֶׁ֖בַע עַל-חַטֹּֽאתֵיכֶֽם:  מ  וְאִ֨ם-עַד-אֵ֔לֶּה לֹ֥א תִשְׁמְע֖וּ לִ֑י וְיָֽסַפְתִּי֙ לְיַסְּרָ֣ה אֶתְכֶ֔ם שֶׁ֖בַע עַל-חַטֹּֽאתֵיכֶֽם:  ת  וְאִם עַד אִלֵין לָא תְקַבְּלוּן לְמֵימְרִי וְאוֹסֵף לְמִרְדֵי יָתְכוֹן שְׁבַע עַל חוֹבֵיכוֹן:

{יט} וְשָֽׁבַרְתִּ֖י אֶת-גְּא֣וֹן עֻזְּכֶ֑ם וְנָֽתַתִּ֤י אֶת-שְׁמֵיכֶם֙ כַּבַּרְזֶ֔ל וְאֶֽת-אַרְצְכֶ֖ם כַּנְּחֻשָֽׁה:  מ  וְשָֽׁבַרְתִּ֖י אֶת-גְּא֣וֹן עֻזְּכֶ֑ם וְנָֽתַתִּ֤י אֶת-שְׁמֵיכֶם֙ כַּבַּרְזֶ֔ל וְאֶֽת-אַרְצְכֶ֖ם כַּנְּחֻשָֽׁה:  ת  וְאֶתְבַּר יָת יְקַר תָּקְפְּכוֹן וְאֶתֵּן יָת שְׁמַיָא דִי עִלָוֵיכוֹן תַּקִיפִין כְּפַרְזְלָא מִלַאֲחָתָא מִטְרָא וְאַרְעָא דִתְחוֹתֵיכוֹן חֲסִינָא כִּנְחָשָׁא מִלְמֶעְבַּד פֵּירִין:

{כ} וְתַ֥ם לָרִ֖יק כֹּֽחֲכֶ֑ם וְלֹֽא-תִתֵּ֤ן אַרְצְכֶם֙ אֶת-יְבוּלָ֔הּ וְעֵ֣ץ הָאָ֔רֶץ לֹ֥א יִתֵּ֖ן פִּרְיֽוֹ:  מ  וְתַ֥ם לָרִ֖יק כֹּֽחֲכֶ֑ם וְלֹֽא-תִתֵּ֤ן אַרְצְכֶם֙ אֶת-יְבוּלָ֔הּ וְעֵ֣ץ הָאָ֔רֶץ לֹ֥א יִתֵּ֖ן פִּרְיֽוֹ:  ת  וִיסוּפוּן לְרֵיקָנוּ חֵילֵיכוֹן וְלָא תִתֵּן אַרְעֲכוֹן יָת עֲלַלְתָּהּ וְאִילַן אַרְעָא לָא יִתֵּן אִבֵּהּ:

{כא} וְאִם-תֵּֽלְכ֤וּ עִמִּי֙ קֶ֔רִי וְלֹ֥א תֹאב֖וּ לִשְׁמֹ֣עַֽ לִ֑י וְיָֽסַפְתִּ֤י עֲלֵיכֶם֙ מַכָּ֔ה שֶׁ֖בַע כְּחַטֹּֽאתֵיכֶֽם:  מ  וְאִם-תֵּֽלְכ֤וּ עִמִּי֙ קֶ֔רִי וְלֹ֥א תֹאב֖וּ לִשְׁמֹ֣עַֽ לִ֑י וְיָֽסַפְתִּ֤י עֲלֵיכֶם֙ מַכָּ֔ה שֶׁ֖בַע כְּחַטֹּֽאתֵיכֶֽם:  ת  וְאִם תְּהָכוּן קֳדָמַי בְּקַשְׁיוּ וְלָא תֵיבוּן לְקַבָּלָא לְמֵימְרִי וְאוֹסֵף לֶאֱתָאָה עֲלֵיכוֹן מָחָא שְׁבַע כְּחוֹבֵיכוֹן:

{כב} וְהִשְׁלַחְתִּ֨י בָכֶ֜ם אֶת-חַיַּ֤ת הַשָּׂדֶה֙ וְשִׁכְּלָ֣ה אֶתְכֶ֔ם וְהִכְרִ֨יתָה֙ אֶת-בְּהֶמְתְּכֶ֔ם וְהִמְעִ֖יטָה אֶתְכֶ֑ם וְנָשַׁ֖מּוּ דַּרְכֵיכֶֽם:  מ  וְהִשְׁלַחְתִּ֨י בָכֶ֜ם אֶת-חַיַּ֤ת הַשָּׂדֶה֙ וְשִׁכְּלָ֣ה אֶתְכֶ֔ם וְהִכְרִ֨יתָה֙ אֶת-בְּהֶמְתְּכֶ֔ם וְהִמְעִ֖יטָה אֶתְכֶ֑ם וְנָשַׁ֖מּוּ דַּרְכֵיכֶֽם:  ת  וֶאֶגָרֵי בְכוֹן יָת חֵוַת בָּרָא וּתְתַכֵּל יָתְכוֹן וּתְשֵׁיצֵי יָת בְּעִירְכוֹן וְתַזְעֵר יָתְכוֹן וִיצַדְיָן אוֹרְחָתְכוֹן:

{כג} וְאִ֨ם-בְּאֵ֔לֶּה לֹ֥א תִוָּֽסְר֖וּ לִ֑י וַֽהֲלַכְתֶּ֥ם עִמִּ֖י קֶֽרִי:  מ  וְאִ֨ם-בְּאֵ֔לֶּה לֹ֥א תִוָּֽסְר֖וּ לִ֑י וַֽהֲלַכְתֶּ֥ם עִמִּ֖י קֶֽרִי:  ת  וְאִם בְּאִלֵין לָא תִתְרְדוּן לְמֵימְרִי וּתְהָכוּן קֳדָמַי בְּקַשְׁיוּ:

{כד} וְהָֽלַכְתִּ֧י אַף-אֲנִ֛י עִמָּכֶ֖ם בְּקֶ֑רִי וְהִכֵּיתִ֤י אֶתְכֶם֙ גַּם-אָ֔נִי שֶׁ֖בַע עַל-חַטֹּֽאתֵיכֶֽם:  מ  וְהָֽלַכְתִּ֧י אַף-אֲנִ֛י עִמָּכֶ֖ם בְּקֶ֑רִי וְהִכֵּיתִ֤י אֶתְכֶם֙ גַּם-אָ֔נִי שֶׁ֖בַע עַל-חַטֹּֽאתֵיכֶֽם:  ת  וְאֵהַךְ אַף אֲנָא עִמְכוֹן בְּקַשְׁיוּ וְאַלְקֵי יָתְכוֹן אַף אֲנָא שְׁבַע עַל חוֹבֵיכוֹן:

{כה} וְהֵֽבֵאתִ֨י עֲלֵיכֶ֜ם חֶ֗רֶב נֹקֶ֨מֶת֙ נְקַם-בְּרִ֔ית וְנֶֽאֱסַפְתֶּ֖ם אֶל-עָֽרֵיכֶ֑ם וְשִׁלַּ֤חְתִּי דֶ֨בֶר֙ בְּת֣וֹכֲכֶ֔ם וְנִתַּתֶּ֖ם בְּיַד-אוֹיֵֽב:  מ  וְהֵֽבֵאתִ֨י עֲלֵיכֶ֜ם חֶ֗רֶב נֹקֶ֨מֶת֙ נְקַם-בְּרִ֔ית וְנֶֽאֱסַפְתֶּ֖ם אֶל-עָֽרֵיכֶ֑ם וְשִׁלַּ֤חְתִּי דֶ֨בֶר֙ בְּת֣וֹכֲכֶ֔ם וְנִתַּתֶּ֖ם בְּיַד-אוֹיֵֽב:  ת  וְאַיְתִי עֲלֵיכוֹן דְקָטְלִין בְּחַרְבָּא וְיִתְפָּרְעוּן מִנְכוֹן פּוּרְעֲנוּתָא עַל דַעֲבַרְתּוּן עַל פִתְגָמֵי אוֹרַיְתָא וְתִתְכַּנְשׁוּן לְקִרְוֵיכוֹן וֶאֱגָרֵי מוֹתָנָא בֵּינֵיכוֹן וְתִתְמַסְרוּן בִּידָא דְסָנְאָה:

{כו} בְּשִׁבְרִ֣י לָכֶם֘ מַטֵּה-לֶ֒חֶם֒ וְ֠אָפ֠וּ עֶ֣שֶׂר נָשִׁ֤ים לַחְמְכֶם֙ בְּתַנּ֣וּר אֶחָ֔ד וְהֵשִׁ֥יבוּ לַחְמְכֶ֖ם בַּמִּשְׁקָ֑ל וַֽאֲכַלְתֶּ֖ם וְלֹ֥א תִשְׂבָּֽעוּ:  מ  בְּשִׁבְרִ֣י לָכֶם֘ מַטֵּה-לֶ֒חֶם֒ וְ֠אָפ֠וּ עֶ֣שֶׂר נָשִׁ֤ים לַחְמְכֶם֙ בְּתַנּ֣וּר אֶחָ֔ד וְהֵשִׁ֥יבוּ לַחְמְכֶ֖ם בַּמִּשְׁקָ֑ל וַֽאֲכַלְתֶּ֖ם וְלֹ֥א תִשְׂבָּֽעוּ:  ת  בִּדְאִתְּבַר לְכוֹן סָעִיד מֵיכְלָא וְיָפְיָן עֲסַר נְשִׁין לַחְמְכוֹן בְּתַנוּרָא חַד וִיתִיבוּן לַחְמְכוֹן בְּמַתְקְלָא וְתוֹכְלוּן וְלָא תִשְׂבְּעוּן: (ס)

{כז} וְאִ֨ם-בְּזֹ֔את לֹ֥א תִשְׁמְע֖וּ לִ֑י וַֽהֲלַכְתֶּ֥ם עִמִּ֖י בְּקֶֽרִי:  מ  וְאִ֨ם-בְּזֹ֔את לֹ֥א תִשְׁמְע֖וּ לִ֑י וַֽהֲלַכְתֶּ֥ם עִמִּ֖י בְּקֶֽרִי:  ת  וְאִם בְּדָא לָא תְקַבְּלוּן לְמֵימְרִי וּתְהָכוּן קֳדָמַי בְּקַשְׁיוּ:

{כח} וְהָֽלַכְתִּ֥י עִמָּכֶ֖ם בַּֽחֲמַת-קֶ֑רִי וְיִסַּרְתִּ֤י אֶתְכֶם֙ אַף-אָ֔נִי שֶׁ֖בַע עַל-חַטֹּֽאתֵיכֶֽם:  מ  וְהָֽלַכְתִּ֥י עִמָּכֶ֖ם בַּֽחֲמַת-קֶ֑רִי וְיִסַּרְתִּ֤י אֶתְכֶם֙ אַף-אָ֔נִי שֶׁ֖בַע עַל-חַטֹּֽאתֵיכֶֽם:  ת  וְאֵהַךְ עִמְכוֹן בִּתְקוֹף רְגַז וְאִרְדֵי יָתְכוֹן אַף אֲנָא שְׁבַע עַל חוֹבֵיכוֹן:

{כט} וַֽאֲכַלְתֶּ֖ם בְּשַׂ֣ר בְּנֵיכֶ֑ם וּבְשַׂ֥ר בְּנֹֽתֵיכֶ֖ם תֹּאכֵֽלוּ:  מ  וַֽאֲכַלְתֶּ֖ם בְּשַׂ֣ר בְּנֵיכֶ֑ם וּבְשַׂ֥ר בְּנֹֽתֵיכֶ֖ם תֹּאכֵֽלוּ:  ת  וְתֵיכְלוּן בְּשַׂר בְּנֵיכוֹן וּבְשַׂר בְּנָתֵיכוֹן תֵּיכְלוּן:

{ל} וְהִשְׁמַדְתִּ֞י אֶת-בָּמֹֽתֵיכֶ֗ם וְהִכְרַתִּי֙ אֶת-חַמָּ֣נֵיכֶ֔ם וְנָֽתַתִּי֙ אֶת-פִּגְרֵיכֶ֔ם עַל-פִּגְרֵ֖י גִּלּֽוּלֵיכֶ֑ם וְגָֽעֲלָ֥ה נַפְשִׁ֖י אֶתְכֶֽם:  מ  וְהִשְׁמַדְתִּ֞י אֶת-בָּמֹֽתֵיכֶ֗ם וְהִכְרַתִּי֙ אֶת-חַמָּ֣נֵיכֶ֔ם וְנָֽתַתִּי֙ אֶת-פִּגְרֵיכֶ֔ם עַל-פִּגְרֵ֖י גִּלּֽוּלֵיכֶ֑ם וְגָֽעֲלָ֥ה נַפְשִׁ֖י אֶתְכֶֽם:  ת  וֶאֱשֵׁיצֵי יָת בָּמָתֵיכוֹן וֶאֱקַצֵץ יָת חֲנִיסְנְסֵיכוֹן וְאֶתֵּן יָת פִּגְרֵיכוֹן עַל פִּגוּר טַעֲוָתְכוֹן וִירַחֵק מֵימְרִי יָתְכוֹן:

{לא} וְנָֽתַתִּ֤י אֶת-עָֽרֵיכֶם֙ חָרְבָּ֔ה וַֽהֲשִׁמּוֹתִ֖י אֶת-מִקְדְּשֵׁיכֶ֑ם וְלֹ֣א אָרִ֔יחַ בְּרֵ֖יחַ נִיחֹֽחֲכֶֽם:  מ  וְנָֽתַתִּ֤י אֶת-עָֽרֵיכֶם֙ חָרְבָּ֔ה וַֽהֲשִׁמּוֹתִ֖י אֶת-מִקְדְּשֵׁיכֶ֑ם וְלֹ֣א אָרִ֔יחַ בְּרֵ֖יחַ נִיחֹֽחֲכֶֽם:  ת  וְאֶתֵּן יָת קִרְוֵיכוֹן צָדָא וְאֶצְדֵי יָת מַקְדְשֵׁיכוֹן וְלָא אֲקַבֵּל בְּרַעֲוָא קֻרְבַּן כְּנִשָׁתְכוֹן:

{לב} וַֽהֲשִׁמֹּתִ֥י אֲנִ֖י אֶת-הָאָ֑רֶץ וְשָֽׁמֲמ֤וּ עָלֶ֨יהָ֙ אֹֽיְבֵיכֶ֔ם הַיּֽשְׁבִ֖ים בָּֽהּ:  מ  וַֽהֲשִׁמֹּתִ֥י אֲנִ֖י אֶת-הָאָ֑רֶץ וְשָֽׁמֲמ֤וּ עָלֶ֨יהָ֙ אֹֽיְבֵיכֶ֔ם הַיּֽשְׁבִ֖ים בָּֽהּ:  ת  וְאֶצְדֵי אֲנָא יָת אַרְעָא וִיצָדְיוּן עֲלַהּ בַּעֲלֵי דְבָבֵיכוֹן דְיָתְבוּן בַּהּ:

{לג} וְאֶתְכֶם֙ אֱזָרֶ֣ה בַגּוֹיִ֔ם וַֽהֲרִֽיקֹתִ֥י אַֽחֲרֵיכֶ֖ם חָ֑רֶב וְהָֽיְתָ֤ה אַרְצְכֶם֙ שְׁמָמָ֔ה וְעָֽרֵיכֶ֖ם יִהְי֥וּ חָרְבָּֽה:  מ  וְאֶתְכֶם֙ אֱזָרֶ֣ה בַגּוֹיִ֔ם וַֽהֲרִֽיקֹתִ֥י אַֽחֲרֵיכֶ֖ם חָ֑רֶב וְהָֽיְתָ֤ה אַרְצְכֶם֙ שְׁמָמָ֔ה וְעָֽרֵיכֶ֖ם יִהְי֥וּ חָרְבָּֽה:  ת  וְיָתְכוֹן אֲבַדַר בֵּינֵי עַמְמַיָא וֶאֱגָרֵי בַתְרֵיכוֹן דְקַטְלִין בְּחַרְבָּא וּתְהֵי אַרְעֲכוֹן צָדְיָא וְקִרְוֵיכוֹן יְהוֹן חָרְבָּא:

{לד} אָז֩ תִּרְצֶ֨ה הָאָ֜רֶץ אֶת-שַׁבְּתֹתֶ֗יהָ כֹּ֚ל יְמֵ֣י הָשַּׁמָּ֔ה וְאַתֶּ֖ם בְּאֶ֣רֶץ אֹֽיְבֵיכֶ֑ם אָ֚ז תִּשְׁבַּ֣ת הָאָ֔רֶץ וְהִרְצָ֖ת אֶת-שַׁבְּתֹתֶֽיהָ:  מ  אָז֩ תִּרְצֶ֨ה הָאָ֜רֶץ אֶת-שַׁבְּתֹתֶ֗יהָ כֹּ֚ל יְמֵ֣י הָשַּׁמָּ֔ה וְאַתֶּ֖ם בְּאֶ֣רֶץ אֹֽיְבֵיכֶ֑ם אָ֚ז תִּשְׁבַּ֣ת הָאָ֔רֶץ וְהִרְצָ֖ת אֶת-שַׁבְּתֹתֶֽיהָ:  ת  בְּכֵן תַּרְעֵי אַרְעָא יָת שְׁמִטָהָא כֹּל יוֹמִין דִי צְדִיאַת וְאַתּוּן בְּאַרַע בַּעֲלֵי דְבָבֵיכוֹן בְּכֵן תַּשְׁמֵט אַרְעָא וְתִרְעֵי יָת שְׁמִטָּהָא:

{לה} כָּל-יְמֵ֥י הָשַּׁמָּ֖ה תִּשְׁבֹּ֑ת אֵ֣ת אֲשֶׁ֧ר לֹא-שָֽׁבְתָ֛ה בְּשַׁבְּתֹֽתֵיכֶ֖ם בְּשִׁבְתְּכֶ֥ם עָלֶֽיהָ:  מ  כָּל-יְמֵ֥י הָשַּׁמָּ֖ה תִּשְׁבֹּ֑ת אֵ֣ת אֲשֶׁ֧ר לֹא-שָֽׁבְתָ֛ה בְּשַׁבְּתֹֽתֵיכֶ֖ם בְּשִׁבְתְּכֶ֥ם עָלֶֽיהָ:  ת  כָּל יוֹמִין דִי צְדִיאַת תַּשְׁמֵט יָת דִי לָא שְׁמֵטַת בִּשְׁמִטֵיכוֹן כַּד הֲוֵיתוּן יָתְבִין עֲלַהּ:

{לו} וְהַנִּשְׁאָרִ֣ים בָּכֶ֗ם וְהֵבֵ֤אתִי מֹ֨רֶךְ֙ בִּלְבָבָ֔ם בְּאַרְצֹ֖ת אֹֽיְבֵיהֶ֑ם וְרָדַ֣ף אֹתָ֗ם ק֚וֹל עָלֶ֣ה נִדָּ֔ף וְנָס֧וּ מְנֻֽסַת-חֶ֛רֶב וְנָֽפְל֖וּ וְאֵ֥ין רֹדֵֽף:  מ  וְהַנִּשְׁאָרִ֣ים בָּכֶ֗ם וְהֵבֵ֤אתִי מֹ֨רֶךְ֙ בִּלְבָבָ֔ם בְּאַרְצֹ֖ת אֹֽיְבֵיהֶ֑ם וְרָדַ֣ף אֹתָ֗ם ק֚וֹל עָלֶ֣ה נִדָּ֔ף וְנָס֧וּ מְנֻֽסַת-חֶ֛רֶב וְנָֽפְל֖וּ וְאֵ֥ין רֹדֵֽף:  ת  וּדְיִשְׁתַּאֲרוּן בְּכוֹן וְאָעֵל תַּבְרָא בְּלִבְּהוֹן בְּאַרְעֲתָא דְסַנְאֵיהוֹן וְיִרְדוֹף יָתְהוֹן קַל טַרְפָּא דְשַׁקִיף וְיֵעַרְקֻן כְּמֵעִירוֹק מִן קֳדָם דְקַטְלִין בְּחַרְבָּא וְיִפְּלוּן וְלֵית דְרָדִיף:

{לז} וְכָֽשְׁל֧וּ אִישׁ-בְּאָחִ֛יו כְּמִפְּנֵי-חֶ֖רֶב וְרֹדֵ֣ף אָ֑יִן וְלֹא-תִֽהְיֶ֤ה לָכֶם֙ תְּקוּמָ֔ה לִפְנֵ֖י אֹֽיְבֵיכֶֽם:  מ  וְכָֽשְׁל֧וּ אִישׁ-בְּאָחִ֛יו כְּמִפְּנֵי-חֶ֖רֶב וְרֹדֵ֣ף אָ֑יִן וְלֹא-תִֽהְיֶ֤ה לָכֶם֙ תְּקוּמָ֔ה לִפְנֵ֖י אֹֽיְבֵיכֶֽם:  ת  וְיִתַּקְלוּן גְבַר בְּאָחוּהִי כְּמִקֳדָם דְקָטְלִין בְּחַרְבָּא וְרָדִיף לָיִת וְלָא תְהֵי לְכוֹן תְּקוּמָה קֳדָם בַּעֲלֵי דְבָבֵיכוֹן:

{לח} וַֽאֲבַדְתֶּ֖ם בַּגּוֹיִ֑ם וְאָֽכְלָ֣ה אֶתְכֶ֔ם אֶ֖רֶץ אֹֽיְבֵיכֶֽם:  מ  וַֽאֲבַדְתֶּ֖ם בַּגּוֹיִ֑ם וְאָֽכְלָ֣ה אֶתְכֶ֔ם אֶ֖רֶץ אֹֽיְבֵיכֶֽם:  ת  וְתֵבְדוּן בֵּינֵי עַמְמַיָא וּתְגַמַר יָתְכוֹן אֲרַע בַּעֲלֵי דְבָבֵיכוֹן:

{לט} וְהַנִּשְׁאָרִ֣ים בָּכֶ֗ם יִמַּ֨קּוּ֙ בַּֽעֲוֹנָ֔ם בְּאַרְצֹ֖ת אֹֽיְבֵיכֶ֑ם וְאַ֛ף בַּֽעֲוֹנֹ֥ת אֲבֹתָ֖ם אִתָּ֥ם יִמָּֽקּוּ:  מ  וְהַנִּשְׁאָרִ֣ים בָּכֶ֗ם יִמַּ֨קּוּ֙ בַּֽעֲוֹנָ֔ם בְּאַרְצֹ֖ת אֹֽיְבֵיכֶ֑ם וְאַ֛ף בַּֽעֲוֹנֹ֥ת אֲבֹתָ֖ם אִתָּ֥ם יִמָּֽקּוּ:  ת  וּדְיִשְׁתַּאֲרוּן בְּכוֹן יִתִּמְסוּן בְּחוֹבֵיהוֹן בְּאַרְעֲתָא בַּעֲלֵי דְבָבֵיכוֹן וְאַף בְּחוֹבֵי אֲבָהַתְהוֹן בִּישַׁיָא דַאֲחִידִין בִּידֵיהוֹן יִתִּמְסוּן:

{מ} וְהִתְוַדּ֤וּ אֶת-עֲוֹנָם֙ וְאֶת-עֲוֹ֣ן אֲבֹתָ֔ם בְּמַֽעֲלָ֖ם אֲשֶׁ֣ר מָֽעֲלוּ-בִ֑י וְאַ֕ף אֲשֶׁר-הָֽלְכ֥וּ עִמִּ֖י בְּקֶֽרִי:  מ  וְהִתְוַדּ֤וּ אֶת-עֲוֹנָם֙ וְאֶת-עֲוֹ֣ן אֲבֹתָ֔ם בְּמַֽעֲלָ֖ם אֲשֶׁ֣ר מָֽעֲלוּ-בִ֑י וְאַ֕ף אֲשֶׁר-הָֽלְכ֥וּ עִמִּ֖י בְּקֶֽרִי:  ת  וִיוַדוּן יָת חוֹבֵיהוֹן וְיָת חוֹבֵי אֲבָהַתְהוֹן בְּשִׁקְרְהוֹן דְשַׁקָרוּ קֳדָמַי וְאַף דִי הַלִיכוּ קֳדָמַי בְּקַשְׁיוּ:

{מא} אַף-אֲנִ֞י אֵלֵ֤ךְ עִמָּם֙ בְּקֶ֔רִי וְהֵֽבֵאתִ֣י אֹתָ֔ם בְּאֶ֖רֶץ אֹֽיְבֵיהֶ֑ם אוֹ-אָ֣ז יִכָּנַ֗ע לְבָבָם֙ הֶֽעָרֵ֔ל וְאָ֖ז יִרְצ֥וּ אֶת-עֲוֹנָֽם:  מ  אַף-אֲנִ֞י אֵלֵ֤ךְ עִמָּם֙ בְּקֶ֔רִי וְהֵֽבֵאתִ֣י אֹתָ֔ם בְּאֶ֖רֶץ אֹֽיְבֵיהֶ֑ם אוֹ-אָ֣ז יִכָּנַ֗ע לְבָבָם֙ הֶֽעָרֵ֔ל וְאָ֖ז יִרְצ֥וּ אֶת-עֲוֹנָֽם:  ת  אַף אֲנָא אֱהַךְ עִמְהוֹן בְּקַשְׁיוּ וְאָעֵיל יָתְהוֹן בְּאַרַע בַּעֲלֵי דְבָבֵיהוֹן אוֹ בְּכֵן יִתָּבַר לִבְּהוֹן טַפְשָׁא וּבְכֵן יִרְעוּן יָת חוֹבֵיהוֹן:

{מב} וְזָֽכַרְתִּ֖י אֶת-בְּרִיתִ֣י יַֽעֲק֑וֹב וְאַף֩ אֶת-בְּרִיתִ֨י יִצְחָ֜ק וְאַ֨ף אֶת-בְּרִיתִ֧י אַבְרָהָ֛ם אֶזְכֹּ֖ר וְהָאָ֥רֶץ אֶזְכֹּֽר:  מ  וְזָֽכַרְתִּ֖י אֶת-בְּרִיתִ֣י יַֽעֲק֑וֹב וְאַף֩ אֶת-בְּרִיתִ֨י יִצְחָ֜ק וְאַ֨ף אֶת-בְּרִיתִ֧י אַבְרָהָ֛ם אֶזְכֹּ֖ר וְהָאָ֥רֶץ אֶזְכֹּֽר:  ת  וְדָכִירְנָא יָת קְיָמִי דְעִם יַעֲקוֹב וְאַף יָת קְיָמִי דְעִם יִצְחָק וְאַף יָת קְיָמִי דְעִם אַבְרָהָם אֲנָא דָכִיר וְאַרְעָא אֲנָא דָכִיר:

{מג} וְהָאָרֶץ֩ תֵּֽעָזֵ֨ב מֵהֶ֜ם וְתִ֣רֶץ אֶת-שַׁבְּתֹתֶ֗יהָ בָּהְשַׁמָּה֙ מֵהֶ֔ם וְהֵ֖ם יִרְצ֣וּ אֶת-עֲוֹנָ֑ם יַ֣עַן וּבְיַ֔עַן בְּמִשְׁפָּטַ֣י מָאָ֔סוּ וְאֶת-חֻקֹּתַ֖י גָּֽעֲלָ֥ה נַפְשָֽׁם:  מ  וְהָאָרֶץ֩ תֵּֽעָזֵ֨ב מֵהֶ֜ם וְתִ֣רֶץ אֶת-שַׁבְּתֹתֶ֗יהָ בָּהְשַׁמָּה֙ מֵהֶ֔ם וְהֵ֖ם יִרְצ֣וּ אֶת-עֲוֹנָ֑ם יַ֣עַן וּבְיַ֔עַן בְּמִשְׁפָּטַ֣י מָאָ֔סוּ וְאֶת-חֻקֹּתַ֖י גָּֽעֲלָ֥ה נַפְשָֽׁם:  ת  וְאַרְעָא תִתְרְטֵשׁ מִנְהוֹן וְתַרְעֵי יָת שְׁמִיטָהָא בְּדִצְדִיאַת מִנְהוֹן וְאִנוּן יִרְעוּן יָת חוֹבֵיהוֹן לְוָטִין חֳלַף בִּרְכָן אַיְתִי עֲלֵיהוֹן בְּדִיל דִבְדִינַי קָצוּ וְיָת קְיָמַי רְחֵקַת נַפְשְׁהוֹן:

{מד} וְאַ֣ף-גַּם-זֹ֠את בִּֽהְיוֹתָ֞ם בְּאֶ֣רֶץ אֹֽיְבֵיהֶ֗ם לֹֽא-מְאַסְתִּ֤ים וְלֹֽא-גְעַלְתִּים֙ לְכַלֹּתָ֔ם לְהָפֵ֥ר בְּרִיתִ֖י אִתָּ֑ם כִּ֛י אֲנִ֥י יְהוָֹ֖ה אֱלֹֽהֵיהֶֽם:  מ  וְאַ֣ף-גַּם-זֹ֠את בִּֽהְיוֹתָ֞ם בְּאֶ֣רֶץ אֹֽיְבֵיהֶ֗ם לֹֽא-מְאַסְתִּ֤ים וְלֹֽא-גְעַלְתִּים֙ לְכַלֹּתָ֔ם לְהָפֵ֥ר בְּרִיתִ֖י אִתָּ֑ם כִּ֛י אֲנִ֥י יְהוָֹ֖ה אֱלֹֽהֵיהֶֽם:  ת  וְאַף בְּרַם דָא בְּמֶהֱוֵיהוֹן בְּאַרַע בַּעֲלֵי דְבָבֵיהוֹן לָא אַרְטֵשִׁנוּן וְלָא אַרְחֵקִנוּן לְשֵׁיצָיוּתְהוֹן לְאַשְׁנָאָה קְיָמִי עִמְהוֹן אֲרֵי אֲנָא יְיָ אֱלָהֲהוֹן:

{מה} וְזָֽכַרְתִּ֥י לָהֶ֖ם בְּרִ֣ית רִֽאשֹׁנִ֑ים אֲשֶׁ֣ר הוֹצֵֽאתִי-אֹתָם֩ מֵאֶ֨רֶץ מִצְרַ֜יִם לְעֵינֵ֣י הַגּוֹיִ֗ם לִהְי֥וֹת לָהֶ֛ם לֵֽאלֹהִ֖ים אֲנִ֥י יְהוָֹֽה:  מ  וְזָֽכַרְתִּ֥י לָהֶ֖ם בְּרִ֣ית רִֽאשֹׁנִ֑ים אֲשֶׁ֣ר הוֹצֵֽאתִי-אֹתָם֩ מֵאֶ֨רֶץ מִצְרַ֜יִם לְעֵינֵ֣י הַגּוֹיִ֗ם לִהְי֥וֹת לָהֶ֛ם לֵֽאלֹהִ֖ים אֲנִ֥י יְהוָֹֽה:  ת  וְדָכִירְנָא לְהוֹן קְיַם קַדְמָאֵי דִי אַפֵּקִית יָתְהוֹן מֵאַרְעָא דְמִצְרַיִם לְעֵינֵי עַמְמַיָא לְמֶהֱוֵי לְהוֹן לֵאלָהָא אֲנָא יְיָ:

{מו} אֵ֠לֶּה הַֽחֻקִּ֣ים וְהַמִּשְׁפָּטִים֘ וְהַתּוֹרֹת֒ אֲשֶׁר֙ נָתַ֣ן יְהֹוָ֔ה בֵּינ֕וֹ וּבֵ֖ין בְּנֵ֣י יִשְׂרָאֵ֑ל בְּהַ֥ר סִינַ֖י בְּיַד-מֹשֶֽׁה:  מ  אֵ֠לֶּה הַֽחֻקִּ֣ים וְהַמִּשְׁפָּטִים֘ וְהַתּוֹרֹת֒ אֲשֶׁר֙ נָתַ֣ן יְהֹוָ֔ה בֵּינ֕וֹ וּבֵ֖ין בְּנֵ֣י יִשְׂרָאֵ֑ל בְּהַ֥ר סִינַ֖י בְּיַד-מֹשֶֽׁה:  ת  אִלֵין קְיָמַיָא וְדִינַיָא וְאוֹרָיָתָא דִי יְהַב יְיָ בֵּין מֵימְרֵהּ וּבֵין בְּנֵי יִשְׂרָאֵל בְּטוּרָא דְסִינַי בִּידָא דְמֹשֶׁה: (פ)




ויקרא פרק-כז

{א}  רביעי - ששי במחוברין  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר:

{ב} דַּבֵּ֞ר אֶל-בְּנֵ֤י יִשְׂרָאֵל֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם אִ֕ישׁ כִּ֥י יַפְלִ֖א נֶ֑דֶר בְּעֶרְכְּךָ֥ נְפָשֹׁ֖ת לַֽיהוָֹֽה:  מ  דַּבֵּ֞ר אֶל-בְּנֵ֤י יִשְׂרָאֵל֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם אִ֕ישׁ כִּ֥י יַפְלִ֖א נֶ֑דֶר בְּעֶרְכְּךָ֥ נְפָשֹׁ֖ת לַֽיהוָֹֽה:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן גְבַר אֲרֵי יַפְרֵשׁ נְדַר בְּפֻרְסַן נַפְשָׁתָא קֳדָם יְיָ:

{ג} וְהָיָ֤ה עֶרְכְּךָ֙ הַזָּכָ֔ר מִבֶּן֙ עֶשְׂרִ֣ים שָׁנָ֔ה וְעַ֖ד בֶּן-שִׁשִּׁ֣ים שָׁנָ֑ה וְהָיָ֣ה עֶרְכְּךָ֗ חֲמִשִּׁ֛ים שֶׁ֥קֶל כֶּ֖סֶף בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ:  מ  וְהָיָ֤ה עֶרְכְּךָ֙ הַזָּכָ֔ר מִבֶּן֙ עֶשְׂרִ֣ים שָׁנָ֔ה וְעַ֖ד בֶּן-שִׁשִּׁ֣ים שָׁנָ֑ה וְהָיָ֣ה עֶרְכְּךָ֗ חֲמִשִּׁ֛ים שֶׁ֥קֶל כֶּ֖סֶף בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ:  ת  וִיהֵי פֻרְסָנֵהּ דְכוּרָא מִבַּר עֶסְרִין שְׁנִין וְעַד בַּר שִׁתִּין שְׁנִין וִיהֵי פֻרְסָנֵהּ חַמְשִׁין סִלְעִין דִכְסַף בְּסִלְעֵי קוּדְשָׁא:

{ד} וְאִם-נְקֵבָ֖ה הִ֑וא וְהָיָ֥ה עֶרְכְּךָ֖ שְׁלֹשִׁ֥ים שָֽׁקֶל:  מ  וְאִם-נְקֵבָ֖ה הִ֑וא וְהָיָ֥ה עֶרְכְּךָ֖ שְׁלֹשִׁ֥ים שָֽׁקֶל:  ת  וְאִם נְקוּבְתָא הִיא וִיהֵי פֻרְסָנֵהּ תְּלָתִין סִלְעִין:

{ה} וְאִ֨ם מִבֶּן-חָמֵ֜שׁ שָׁנִ֗ים וְעַד֙ בֶּן-עֶשְׂרִ֣ים שָׁנָ֔ה וְהָיָ֧ה עֶרְכְּךָ֛ הַזָּכָ֖ר עֶשְׂרִ֣ים שְׁקָלִ֑ים וְלַנְּקֵבָ֖ה עֲשֶׂ֥רֶת שְׁקָלִֽים:  מ  וְאִ֨ם מִבֶּן-חָמֵ֜שׁ שָׁנִ֗ים וְעַד֙ בֶּן-עֶשְׂרִ֣ים שָׁנָ֔ה וְהָיָ֧ה עֶרְכְּךָ֛ הַזָּכָ֖ר עֶשְׂרִ֣ים שְׁקָלִ֑ים וְלַנְּקֵבָ֖ה עֲשֶׂ֥רֶת שְׁקָלִֽים:  ת  וְאִם מִבַּר חֲמֵשׁ שְׁנִין וְעַד בַּר עֶסְרִין שְׁנִין וִיהֵי פֻרְסָנֵהּ דְכוּרָא עֶסְרִין סִלְעִין וְלִנְקוּבְתָא עֲסַר סִלְעִין:

{ו} וְאִ֣ם מִבֶּן-חֹ֗דֶשׁ וְעַד֙ בֶּן-חָמֵ֣שׁ שָׁנִ֔ים וְהָיָ֤ה עֶרְכְּךָ֙ הַזָּכָ֔ר חֲמִשָּׁ֥ה שְׁקָלִ֖ים כָּ֑סֶף וְלַנְּקֵבָ֣ה עֶרְכְּךָ֔ שְׁלֹ֥שֶׁת שְׁקָלִ֖ים כָּֽסֶף:  מ  וְאִ֣ם מִבֶּן-חֹ֗דֶשׁ וְעַד֙ בֶּן-חָמֵ֣שׁ שָׁנִ֔ים וְהָיָ֤ה עֶרְכְּךָ֙ הַזָּכָ֔ר חֲמִשָּׁ֥ה שְׁקָלִ֖ים כָּ֑סֶף וְלַנְּקֵבָ֣ה עֶרְכְּךָ֔ שְׁלֹ֥שֶׁת שְׁקָלִ֖ים כָּֽסֶף:  ת  וְאִם מִבַּר יַרְחָא וְעַד בַּר חֲמֵשׁ שְׁנִין וִיהֵי פֻרְסָנֵהּ דְכוּרָא חֲמֵשׁ סִלְעִין דִכְסָף וְלִנְקוּבְתָא פֻרְסָנֵהּ תְּלַת סִלְעִין דִכְסָף:

{ז} וְ֠אִם מִבֶּן-שִׁשִּׁ֨ים שָׁנָ֤ה וָמַ֨עְלָה֙ אִם-זָכָ֔ר וְהָיָ֣ה עֶרְכְּךָ֔ חֲמִשָּׁ֥ה עָשָׂ֖ר שָׁ֑קֶל וְלַנְּקֵבָ֖ה עֲשָׂרָ֥ה שְׁקָלִֽים:  מ  וְ֠אִם מִבֶּן-שִׁשִּׁ֨ים שָׁנָ֤ה וָמַ֨עְלָה֙ אִם-זָכָ֔ר וְהָיָ֣ה עֶרְכְּךָ֔ חֲמִשָּׁ֥ה עָשָׂ֖ר שָׁ֑קֶל וְלַנְּקֵבָ֖ה עֲשָׂרָ֥ה שְׁקָלִֽים:  ת  וְאִם מִבַּר שִׁתִּין שְׁנִין וּלְעֵלָא אִם דְכוּרָא וִיהֵי פֻרְסָנֵהּ חֲמֵשׁ עֶסְרֵי סִלְעִין וְלִנְקוּבְתָא עֲסַר סִלְעִין:

{ח} וְאִם-מָ֥ךְ הוּא֙ מֵֽעֶרְכֶּ֔ךָ וְהֶֽעֱמִידוֹ֙ לִפְנֵ֣י הַכֹּהֵ֔ן וְהֶֽעֱרִ֥יךְ אֹת֖וֹ הַכֹּהֵ֑ן עַל-פִּ֗י אֲשֶׁ֤ר תַּשִּׂיג֙ יַ֣ד הַנֹּדֵ֔ר יַֽעֲרִיכֶ֖נּוּ הַכֹּהֵֽן:  מ  וְאִם-מָ֥ךְ הוּא֙ מֵֽעֶרְכֶּ֔ךָ וְהֶֽעֱמִידוֹ֙ לִפְנֵ֣י הַכֹּהֵ֔ן וְהֶֽעֱרִ֥יךְ אֹת֖וֹ הַכֹּהֵ֑ן עַל-פִּ֗י אֲשֶׁ֤ר תַּשִּׂיג֙ יַ֣ד הַנֹּדֵ֔ר יַֽעֲרִיכֶ֖נּוּ הַכֹּהֵֽן:  ת  וְאִם מִסְכֵּן הוּא מִפֻּרְסָנֵהּ וִיקִימִנֵהּ קֳדָם כַּהֲנָא וְיִפְרוֹס יָתֵהּ כַּהֲנָא עַל מֵימַר דִי תַדְבֵּק יְדָא דְנוֹדֵרָא יִפְרְסִנֵהּ כַּהֲנָא: (ס)

{ט} וְאִ֨ם-בְּהֵמָ֔ה אֲשֶׁ֨ר יַקְרִ֧יבוּ מִמֶּ֛נָּה קָרְבָּ֖ן לַֽיהוָֹ֑ה כֹּל֩ אֲשֶׁ֨ר יִתֵּ֥ן מִמֶּ֛נּוּ לַֽיהוָֹ֖ה יִֽהְיֶה-קֹּֽדֶשׁ:  מ  וְאִ֨ם-בְּהֵמָ֔ה אֲשֶׁ֨ר יַקְרִ֧יבוּ מִמֶּ֛נָּה קָרְבָּ֖ן לַֽיהוָֹ֑ה כֹּל֩ אֲשֶׁ֨ר יִתֵּ֥ן מִמֶּ֛נּוּ לַֽיהוָֹ֖ה יִֽהְיֶה-קֹּֽדֶשׁ:  ת  וְאִם בְּעִירָא דִי יְקָרְבוּן מִנַהּ קֻרְבָּנָא קֳדָם יְיָ כָּל דִי יִתֵּן מִנֵהּ קֳדָם יְיָ יְהֵי קוּדְשָׁא:

{י} לֹ֣א יַֽחֲלִיפֶ֔נּוּ וְלֹֽא-יָמִ֥יר אֹת֛וֹ ט֥וֹב בְּרָ֖ע אוֹ-רַ֣ע בְּט֑וֹב וְאִם-הָמֵ֨ר יָמִ֤יר בְּהֵמָה֙ בִּבְהֵמָ֔ה וְהָֽיָה-ה֥וּא וּתְמֽוּרָת֖וֹ יִֽהְיֶה-קֹּֽדֶשׁ:  מ  לֹ֣א יַֽחֲלִיפֶ֔נּוּ וְלֹֽא-יָמִ֥יר אֹת֛וֹ ט֥וֹב בְּרָ֖ע אוֹ-רַ֣ע בְּט֑וֹב וְאִם-הָמֵ֨ר יָמִ֤יר בְּהֵמָה֙ בִּבְהֵמָ֔ה וְהָֽיָה-ה֥וּא וּתְמֽוּרָת֖וֹ יִֽהְיֶה-קֹּֽדֶשׁ:  ת  לָא יְחַלְפִנֵהּ וְלָא יְעִבַּר יָתֵהּ טַּב בְּבִישׁ אוֹ בִישׁ בְּטָב וְאִם חַלָפָא יְחַלֵף בְּעִירָא בִּבְעִירָא וִיהֵי הוּא וְחִלוּפֵי יְהֵי קַדִישׁ:

{יא} וְאִם֙ כָּל-בְּהֵמָ֣ה טְמֵאָ֔ה אֲ֠שֶׁ֠ר לֹֽא-יַקְרִ֧יבוּ מִמֶּ֛נָּה קָרְבָּ֖ן לַֽיהוָֹ֑ה וְהֶֽעֱמִ֥יד אֶת-הַבְּהֵמָ֖ה לִפְנֵ֥י הַכֹּהֵֽן:  מ  וְאִם֙ כָּל-בְּהֵמָ֣ה טְמֵאָ֔ה אֲ֠שֶׁ֠ר לֹֽא-יַקְרִ֧יבוּ מִמֶּ֛נָּה קָרְבָּ֖ן לַֽיהוָֹ֑ה וְהֶֽעֱמִ֥יד אֶת-הַבְּהֵמָ֖ה לִפְנֵ֥י הַכֹּהֵֽן:  ת  וְאִם כָּל בְּעִירָא מְסָאָבָא דִי לָא יְקָרְבוּן מִנַהּ קֻרְבָּנָא קֳדָם יְיָ וִיקִים יָת בְּעִירָא קֳדָם כַּהֲנָא:

{יב} וְהֶֽעֱרִ֤יךְ הַכֹּהֵן֙ אֹתָ֔הּ בֵּ֥ין ט֖וֹב וּבֵ֣ין רָ֑ע כְּעֶרְכְּךָ֥ הַכֹּהֵ֖ן כֵּ֥ן יִֽהְיֶֽה:  מ  וְהֶֽעֱרִ֤יךְ הַכֹּהֵן֙ אֹתָ֔הּ בֵּ֥ין ט֖וֹב וּבֵ֣ין רָ֑ע כְּעֶרְכְּךָ֥ הַכֹּהֵ֖ן כֵּ֥ן יִֽהְיֶֽה:  ת  וְיִפְרוֹס כַּהֲנָא יָתַהּ בֵּין טַב וּבֵין בִּישׁ כְּפֻרְסָנָא דְכַהֲנָא כֵּן יְהֵי:

{יג} וְאִם-גָּאֹ֖ל יִגְאָלֶ֑נָּה וְיָסַ֥ף חֲמִֽישִׁת֖וֹ עַל-עֶרְכֶּֽךָ:  מ  וְאִם-גָּאֹ֖ל יִגְאָלֶ֑נָּה וְיָסַ֥ף חֲמִֽישִׁת֖וֹ עַל-עֶרְכֶּֽךָ:  ת  וְאִם מִפְרַק יִפְרְקֻנֵהּ וְיוֹסֵף חֻמְשֵׁהּ עַל פֻּרְסָנֵהּ:

{יד} וְאִ֗ישׁ כִּֽי-יַקְדִּ֨שׁ אֶת-בֵּית֥וֹ קֹ֨דֶשׁ֙ לַֽיהֹוָ֔ה וְהֶֽעֱרִיכוֹ֙ הַכֹּהֵ֔ן בֵּ֥ין ט֖וֹב וּבֵ֥ין רָ֑ע כַּֽאֲשֶׁ֨ר יַֽעֲרִ֥יךְ אֹת֛וֹ הַכֹּהֵ֖ן כֵּ֥ן יָקֽוּם:  מ  וְאִ֗ישׁ כִּֽי-יַקְדִּ֨שׁ אֶת-בֵּית֥וֹ קֹ֨דֶשׁ֙ לַֽיהֹוָ֔ה וְהֶֽעֱרִיכוֹ֙ הַכֹּהֵ֔ן בֵּ֥ין ט֖וֹב וּבֵ֥ין רָ֑ע כַּֽאֲשֶׁ֨ר יַֽעֲרִ֥יךְ אֹת֛וֹ הַכֹּהֵ֖ן כֵּ֥ן יָקֽוּם:  ת  וּגְבַר אֲרֵי יַקְדֵשׁ יָת בֵּיתֵהּ קוּדְשָׁא קֳדָם יְיָ וְיִפְרְסִנֵהּ כַּהֲנָא בֵּין טַב וּבֵין בִּישׁ כְּמָא דִי יִפְרוֹס יָתֵהּ כַּהֲנָא כֵּן יְקוּם:

{טו} וְאִ֨ם-הַמַּקְדִּ֔ישׁ יִגְאַ֖ל אֶת-בֵּית֑וֹ וְ֠יָסַ֠ף חֲמִישִׁ֧ית כֶּֽסֶף-עֶרְכְּךָ֛ עָלָ֖יו וְהָ֥יָה לֽוֹ:  מ  וְאִ֨ם-הַמַּקְדִּ֔ישׁ יִגְאַ֖ל אֶת-בֵּית֑וֹ וְ֠יָסַ֠ף חֲמִישִׁ֧ית כֶּֽסֶף-עֶרְכְּךָ֛ עָלָ֖יו וְהָ֥יָה לֽוֹ:  ת  וְאִם דְאַקְדֵשׁ יִפְרוֹק יָת בֵּיתֵהּ וְיוֹסֵף חוֹמֶשׁ כְּסַף פֻּרְסָנֵהּ עֲלוֹהִי וִיהֵי לֵהּ:

{טז}  חמישי - שביעי במחוברין  וְאִ֣ם | מִשְּׂדֵ֣ה אֲחֻזָּת֗וֹ יַקְדִּ֥ישׁ אִישׁ֙ לַֽיהֹוָ֔ה וְהָיָ֥ה עֶרְכְּךָ֖ לְפִ֣י זַרְע֑וֹ זֶ֚רַע חֹ֣מֶר שְׂעֹרִ֔ים בַּֽחֲמִשִּׁ֖ים שֶׁ֥קֶל כָּֽסֶף:  מ  וְאִ֣ם | מִשְּׂדֵ֣ה אֲחֻזָּת֗וֹ יַקְדִּ֥ישׁ אִישׁ֙ לַֽיהֹוָ֔ה וְהָיָ֥ה עֶרְכְּךָ֖ לְפִ֣י זַרְע֑וֹ זֶ֚רַע חֹ֣מֶר שְׂעֹרִ֔ים בַּֽחֲמִשִּׁ֖ים שֶׁ֥קֶל כָּֽסֶף:  ת  וְאִם מֵחֲקַל אַחֲסַנְתֵּהּ יַקְדִישׁ גְבַר קֳדָם יְיָ וִיהֵי פֻרְסָנֵהּ לְפוּם זַרְעֵהּ בַּר זְרַע כּוֹר שְׂעוֹרִין בְּחַמְשִׁין סִלְעִין דִכְסָף:

{יז} אִם-מִשְּׁנַ֥ת הַיֹּבֵ֖ל יַקְדִּ֣ישׁ שָׂדֵ֑הוּ כְּעֶרְכְּךָ֖ יָקֽוּם:  מ  אִם-מִשְּׁנַ֥ת הַיֹּבֵ֖ל יַקְדִּ֣ישׁ שָׂדֵ֑הוּ כְּעֶרְכְּךָ֖ יָקֽוּם:  ת  אִם מִשַּׁתָּא דְיוֹבֵלָא יַקְדֵשׁ חַקְלֵהּ כְּפֻרְסָנֵהּ יְקוּם:

{יח} וְאִם-אַחַ֣ר הַיֹּבֵל֘ יַקְדִּ֣ישׁ שָׂדֵ֒הוּ֒ וְחִשַּׁב-ל֨וֹ הַכֹּהֵ֜ן אֶת-הַכֶּ֗סֶף עַל-פִּ֤י הַשָּׁנִים֙ הַנּ֣וֹתָרֹ֔ת עַ֖ד שְׁנַ֣ת הַיֹּבֵ֑ל וְנִגְרַ֖ע מֵֽעֶרְכֶּֽךָ:  מ  וְאִם-אַחַ֣ר הַיֹּבֵל֘ יַקְדִּ֣ישׁ שָׂדֵ֒הוּ֒ וְחִשַּׁב-ל֨וֹ הַכֹּהֵ֜ן אֶת-הַכֶּ֗סֶף עַל-פִּ֤י הַשָּׁנִים֙ הַנּ֣וֹתָרֹ֔ת עַ֖ד שְׁנַ֣ת הַיֹּבֵ֑ל וְנִגְרַ֖ע מֵֽעֶרְכֶּֽךָ:  ת  וְאִם בָּתַר יוֹבֵלָא יַקְדֵשׁ חַקְלֵהּ וִיחַשֵׁב לֵהּ כַּהֲנָא יָת כַּסְפָּא עַל מֵימַר שְׁנַיָא דְאִשְׁתָּאָרָן עַד שַׁתָּא דְיוֹבֵלָא וְיִתִּמְנַע מִפֻּרְסָנֵהּ:

{יט} וְאִם-גָּאֹ֤ל יִגְאַל֙ אֶת-הַשָּׂדֶ֔ה הַמַּקְדִּ֖ישׁ אֹת֑וֹ וְ֠יָסַ֠ף חֲמִשִׁ֧ית כֶּֽסֶף-עֶרְכְּךָ֛ עָלָ֖יו וְקָ֥ם לֽוֹ:  מ  וְאִם-גָּאֹ֤ל יִגְאַל֙ אֶת-הַשָּׂדֶ֔ה הַמַּקְדִּ֖ישׁ אֹת֑וֹ וְ֠יָסַ֠ף חֲמִשִׁ֧ית כֶּֽסֶף-עֶרְכְּךָ֛ עָלָ֖יו וְקָ֥ם לֽוֹ:  ת  וְאִם מִפְרַק יִפְרוֹק יָת חַקְלָא דְאַקְדֵשׁ יָתֵהּ וְיוֹסֵף חוֹמֶשׁ כְּסַף פֻּרְסָנֵהּ עֲלוֹהִי וִיקוּם לֵהּ:

{כ} וְאִם-לֹ֤א יִגְאַל֙ אֶת-הַשָּׂדֶ֔ה וְאִם-מָכַ֥ר אֶת-הַשָּׂדֶ֖ה לְאִ֣ישׁ אַחֵ֑ר לֹֽא יִגָּאֵ֖ל עֽוֹד:  מ  וְאִם-לֹ֤א יִגְאַל֙ אֶת-הַשָּׂדֶ֔ה וְאִם-מָכַ֥ר אֶת-הַשָּׂדֶ֖ה לְאִ֣ישׁ אַחֵ֑ר לֹֽא יִגָּאֵ֖ל עֽוֹד:  ת  וְאִם לָא יִפְרוֹק יָת חַקְלָא וְאִם זַבִּין יָת חַקְלָא לִגְבַר אָחֳרָן לָא יִתְפָּרַק עוֹד:

{כא} וְהָיָ֨ה הַשָּׂדֶ֜ה בְּצֵאת֣וֹ בַיֹּבֵ֗ל קֹ֛דֶשׁ לַֽיהוָֹ֖ה כִּשְׂדֵ֣ה הַחֵ֑רֶם לַכֹּהֵ֖ן תִּהְיֶ֥ה אֲחֻזָּתֽוֹ:  מ  וְהָיָ֨ה הַשָּׂדֶ֜ה בְּצֵאת֣וֹ בַיֹּבֵ֗ל קֹ֛דֶשׁ לַֽיהוָֹ֖ה כִּשְׂדֵ֣ה הַחֵ֑רֶם לַכֹּהֵ֖ן תִּהְיֶ֥ה אֲחֻזָּתֽוֹ:  ת  וִיהֵי חַקְלָא בְּמִפְּקֵהּ בְּיוֹבֵלָא קוּדְשָׁא קֳדָם יְיָ כַּחֲקַל חֶרְמָא לְכַהֲנָא תְּהֵי אַחֲסַנְתֵּהּ:

{כב}  שישי  וְאִם֙ אֶת-שְׂדֵ֣ה מִקְנָת֔וֹ אֲשֶׁ֕ר לֹ֖א מִשְּׂדֵ֣ה אֲחֻזָּת֑וֹ יַקְדִּ֖ישׁ לַֽיהוָֹֽה:  מ  וְאִם֙ אֶת-שְׂדֵ֣ה מִקְנָת֔וֹ אֲשֶׁ֕ר לֹ֖א מִשְּׂדֵ֣ה אֲחֻזָּת֑וֹ יַקְדִּ֖ישׁ לַֽיהוָֹֽה:  ת  וְאִם יָת חֲקַל זְבִינוֹהִי דִי לָא מֵחֲקַל אַחֲסַנְתֵּהּ יַקְדֵשׁ קֳדָם יְיָ:

{כג} וְחִשַּׁב-ל֣וֹ הַכֹּהֵ֗ן אֵ֚ת מִכְסַ֣ת הָֽעֶרְכְּךָ֔ עַ֖ד שְׁנַ֣ת הַיֹּבֵ֑ל וְנָתַ֤ן אֶת-הָֽעֶרְכְּךָ֙ בַּיּ֣וֹם הַה֔וּא קֹ֖דֶשׁ לַֽיהוָֹֽה:  מ  וְחִשַּׁב-ל֣וֹ הַכֹּהֵ֗ן אֵ֚ת מִכְסַ֣ת הָֽעֶרְכְּךָ֔ עַ֖ד שְׁנַ֣ת הַיֹּבֵ֑ל וְנָתַ֤ן אֶת-הָֽעֶרְכְּךָ֙ בַּיּ֣וֹם הַה֔וּא קֹ֖דֶשׁ לַֽיהוָֹֽה:  ת  וִיחַשֶׁב לֵהּ כַּהֲנָא יָת מִנְיַן פֻּרְסָנֵהּ עַד שַׁתָּא דְיוֹבֵלָא וְיִתֵּן יָת פֻרְסָנֵהּ בְּיוֹמָא הַהוּא קוּדְשָׁא קֳדָם יְיָ:

{כד} בִּשְׁנַ֤ת הַיּוֹבֵל֙ יָשׁ֣וּב הַשָּׂדֶ֔ה לַֽאֲשֶׁ֥ר קָנָ֖הוּ מֵֽאִתּ֑וֹ לַֽאֲשֶׁר-ל֖וֹ אֲחֻזַּ֥ת הָאָֽרֶץ:  מ  בִּשְׁנַ֤ת הַיּוֹבֵל֙ יָשׁ֣וּב הַשָּׂדֶ֔ה לַֽאֲשֶׁ֥ר קָנָ֖הוּ מֵֽאִתּ֑וֹ לַֽאֲשֶׁר-ל֖וֹ אֲחֻזַּ֥ת הָאָֽרֶץ:  ת  בְּשַׁתָּא דְיוֹבֵלָא יְתוּב חַקְלָא לִדְזַבְנֵהּ מִנֵהּ לִדְדִילֵהּ אַחֲסָנַת אַרְעָא:

{כה} וְכָ֨ל-עֶרְכְּךָ֔ יִֽהְיֶ֖ה בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ עֶשְׂרִ֥ים גֵּרָ֖ה יִהְיֶ֥ה הַשָּֽׁקֶל:  מ  וְכָ֨ל-עֶרְכְּךָ֔ יִֽהְיֶ֖ה בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ עֶשְׂרִ֥ים גֵּרָ֖ה יִהְיֶ֥ה הַשָּֽׁקֶל:  ת  וְכָל פֻּרְסָנֵּה יְהֵי בְּסִלְעֵי קוּדְשָׁא עֶסְרִין מָעִין יְהֵי סִלְעָא:

{כו} אַךְ-בְּכ֞וֹר אֲשֶׁ֨ר-יְבֻכַּ֤ר לַֽיהוָֹה֙ בִּבְהֵמָ֔ה לֹֽא-יַקְדִּ֥ישׁ אִ֖ישׁ אֹת֑וֹ אִם-שׁ֣וֹר אִם-שֶׂ֔ה לַֽיהוָֹ֖ה הֽוּא:  מ  אַךְ-בְּכ֞וֹר אֲשֶׁ֨ר-יְבֻכַּ֤ר לַֽיהוָֹה֙ בִּבְהֵמָ֔ה לֹֽא-יַקְדִּ֥ישׁ אִ֖ישׁ אֹת֑וֹ אִם-שׁ֣וֹר אִם-שֶׂ֔ה לַֽיהוָֹ֖ה הֽוּא:  ת  בְּרַם בּוּכְרָא דִי יִתְבַּכַּר קֳדָם יְיָ בִּבְעִירָא לָא יַקְדֵשׁ גְבַר יָתֵהּ אִם תּוֹר אִם אִמַר דַיְיָ הוּא:

{כז} וְאִ֨ם בַּבְּהֵמָ֤ה הַטְּמֵאָה֙ וּפָדָ֣ה בְעֶרְכֶּ֔ךָ וְיָסַ֥ף חֲמִֽשִׁת֖וֹ עָלָ֑יו וְאִם-לֹ֥א יִגָּאֵ֖ל וְנִמְכַּ֥ר בְּעֶרְכֶּֽךָ:  מ  וְאִ֨ם בַּבְּהֵמָ֤ה הַטְּמֵאָה֙ וּפָדָ֣ה בְעֶרְכֶּ֔ךָ וְיָסַ֥ף חֲמִֽשִׁת֖וֹ עָלָ֑יו וְאִם-לֹ֥א יִגָּאֵ֖ל וְנִמְכַּ֥ר בְּעֶרְכֶּֽךָ:  ת  וְאִם בִּבְעִירָא מְסָאָבָא וְיִפְרוֹק בְּפֻרְסָנֵהּ וְיוֹסֵף חֻמְשֵׁהּ עֲלוֹהִי וְאִם לָא יִתְפָּרַק וְיִזְדַבַּן בְּפֻרְסָנֵהּ:

{כח} אַ֣ךְ-כָּל-חֵ֡רֶם אֲשֶׁ֣ר יַֽחֲרִם֩ אִ֨ישׁ לַֽיהֹוָ֜ה מִכָּל-אֲשֶׁר-ל֗וֹ מֵֽאָדָ֤ם וּבְהֵמָה֙ וּמִשְּׂדֵ֣ה אֲחֻזָּת֔וֹ לֹ֥א יִמָּכֵ֖ר וְלֹ֣א יִגָּאֵ֑ל כָּל-חֵ֕רֶם קֹֽדֶשׁ-קָֽדָשִׁ֥ים ה֖וּא לַֽיהוָֹֽה:  מ  אַ֣ךְ-כָּל-חֵ֡רֶם אֲשֶׁ֣ר יַֽחֲרִם֩ אִ֨ישׁ לַֽיהֹוָ֜ה מִכָּל-אֲשֶׁר-ל֗וֹ מֵֽאָדָ֤ם וּבְהֵמָה֙ וּמִשְּׂדֵ֣ה אֲחֻזָּת֔וֹ לֹ֥א יִמָּכֵ֖ר וְלֹ֣א יִגָּאֵ֑ל כָּל-חֵ֕רֶם קֹֽדֶשׁ-קָֽדָשִׁ֥ים ה֖וּא לַֽיהוָֹֽה:  ת  בְּרַם כָּל חֶרְמָא דִי יַחְרֵם גְבַר קֳדָם יְיָ מִכָּל דִי לֵהּ מֵאֲנָשָׁא וּבְעִירָא וּמֵחֲקַל אַחֲסַנְתֵּהּ לָא יִזְדַבַּן וְלָא יִתְפָּרָק כָּל חֶרְמָא קֹדֶשׁ קוּדְשִׁין הוּא קֳדָם יְיָ:

{כט}  שביעי  כָּל-חֵ֗רֶם אֲשֶׁ֧ר יָֽחֳרַ֛ם מִן-הָֽאָדָ֖ם לֹ֣א יִפָּדֶ֑ה מ֖וֹת יוּמָֽת:  מ  כָּל-חֵ֗רֶם אֲשֶׁ֧ר יָֽחֳרַ֛ם מִן-הָֽאָדָ֖ם לֹ֣א יִפָּדֶ֑ה מ֖וֹת יוּמָֽת:  ת  כָּל חֶרְמָא דִי יִתַּחְרַם מִן אֱנָשָׁא לָא יִתְפָּרָק אִתְקְטָלָא יִתְקְטֵל:

{ל} וְכָל-מַעְשַׂ֨ר הָאָ֜רֶץ מִזֶּ֤רַע הָאָ֨רֶץ֙ מִפְּרִ֣י הָעֵ֔ץ לַֽיהוָֹ֖ה ה֑וּא קֹ֖דֶשׁ לַֽיהוָֹֽה:  מ  וְכָל-מַעְשַׂ֨ר הָאָ֜רֶץ מִזֶּ֤רַע הָאָ֨רֶץ֙ מִפְּרִ֣י הָעֵ֔ץ לַֽיהוָֹ֖ה ה֑וּא קֹ֖דֶשׁ לַֽיהוָֹֽה:  ת  וְכָל מַעְשְׂרָא דְאַרְעָא מִזַרְעָא דְאַרְעָא מִפֵּירֵי אִילָנָא דַיְיָ הוּא קוּדְשָׁא קֳדָם יְיָ:

{לא} וְאִם-גָּאֹ֥ל יִגְאַ֛ל אִ֖ישׁ מִמַּֽעַשְׂר֑וֹ חֲמִֽשִׁית֖וֹ יֹסֵ֥ף עָלָֽיו:  מ  וְאִם-גָּאֹ֥ל יִגְאַ֛ל אִ֖ישׁ מִמַּֽעַשְׂר֑וֹ חֲמִֽשִׁית֖וֹ יֹסֵ֥ף עָלָֽיו:  ת  וְאִם מִפְרַק יִפְרוֹק גְבַר מִמַעַשְׂרֵהּ חֻמְשֵׁהּ יוֹסֵף עֲלוֹהִי:

{לב}  מפטיר  וְכָל-מַעְשַׂ֤ר בָּקָר֙ וָצֹ֔אן כֹּ֥ל אֲשֶֽׁר-יַֽעֲבֹ֖ר תַּ֣חַת הַשָּׁ֑בֶט הָֽעֲשִׂירִ֕י יִֽהְיֶה-קֹּ֖דֶשׁ לַֽיהוָֹֽה:  מ  וְכָל-מַעְשַׂ֤ר בָּקָר֙ וָצֹ֔אן כֹּ֥ל אֲשֶֽׁר-יַֽעֲבֹ֖ר תַּ֣חַת הַשָּׁ֑בֶט הָֽעֲשִׂירִ֕י יִֽהְיֶה-קֹּ֖דֶשׁ לַֽיהוָֹֽה:  ת  וְכָל מַעֲשַׂר תּוֹרִין וְעָן כֹּל דְיֵעִבַּר תְּחוֹת חֻטְרָא עֲשִׂירָאָה יְהֵי קַדִישׁ קֳדָם יְיָ:

{לג} לֹ֧א יְבַקֵּ֛ר בֵּין-ט֥וֹב לָרַ֖ע וְלֹ֣א יְמִירֶ֑נּוּ וְאִם-הָמֵ֣ר יְמִירֶ֔נּוּ וְהָֽיָה-ה֧וּא וּתְמֽוּרָת֛וֹ יִֽהְיֶה-קֹּ֖דֶשׁ לֹ֥א יִגָּאֵֽל:  מ  לֹ֧א יְבַקֵּ֛ר בֵּין-ט֥וֹב לָרַ֖ע וְלֹ֣א יְמִירֶ֑נּוּ וְאִם-הָמֵ֣ר יְמִירֶ֔נּוּ וְהָֽיָה-ה֧וּא וּתְמֽוּרָת֛וֹ יִֽהְיֶה-קֹּ֖דֶשׁ לֹ֥א יִגָּאֵֽל:  ת  לָא יְבַקַר בֵּין טַב לְבִישׁ וְלָא יְחַלְפִנֵהּ וְאִם חַלָפָא יְחַלְפִנֵהּ וִיהֵי הוּא וְחִלוּפֵהּ יְהֵי קוּדְשָׁא לָא יִתְפָּרָק:

{לד} אֵ֣לֶּה הַמִּצְוֹ֗ת אֲשֶׁ֨ר צִוָּ֧ה יְהוָֹ֛ה אֶת-מֹשֶׁ֖ה אֶל-בְּנֵ֣י יִשְׂרָאֵ֑ל בְּהַ֖ר סִינָֽי:  מ  אֵ֣לֶּה הַמִּצְוֹ֗ת אֲשֶׁ֨ר צִוָּ֧ה יְהוָֹ֛ה אֶת-מֹשֶׁ֖ה אֶל-בְּנֵ֣י יִשְׂרָאֵ֑ל בְּהַ֖ר סִינָֽי:  ת  אִלֵין פִּקוֹדַיָא דִי פַקִיד יְיָ יָת מֹשֶׁה לְוָת בְּנֵי יִשְׂרָאֵל בְּטוּרָא דְסִינָי: (חזק)
