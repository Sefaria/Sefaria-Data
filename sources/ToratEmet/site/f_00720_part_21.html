ש-א


[בראשית פרק-מא-כו]  שֶׁבַע פָּרֹת הַטֹּבֹת שֶׁבַע שָׁנִים הֵנָּה וְשֶׁבַע הַשִּׁבֳּלִים הַטֹּבֹת שֶׁבַע שָׁנִים הֵנָּה חֲלוֹם אֶחָד הוּא:

[דברים פרק-יז-טו]  שׂוֹם תָּשִׂים עָלֶיךָ מֶלֶךְ אֲשֶׁר יִבְחַר יְהוָה אֱלֹהֶיךָ בּוֹ מִקֶּרֶב אַחֶיךָ תָּשִׂים עָלֶיךָ מֶלֶךְ לֹא תוּכַל לָתֵת עָלֶיךָ אִישׁ נָכְרִי אֲשֶׁר לֹא אָחִיךָ הוּא:

[תהילים פרק-י-טו]  שְׁבֹר זְרוֹעַ רָשָׁע וָרָע תִּדְרוֹשׁ רִשְׁעוֹ בַל תִּמְצָא:

[תהילים פרק-פו-ד]  שַׂמֵּחַ נֶפֶשׁ עַבְדֶּךָ כִּי אֵלֶיךָ אֲדֹנָי נַפְשִׁי אֶשָּׂא:

[תהילים פרק-קכז-ב]  שָׁוְא לָכֶם מַשְׁכִּימֵי קוּם מְאַחֲרֵי שֶׁבֶת אֹכְלֵי לֶחֶם הָעֲצָבִים כֵּן יִתֵּן לִידִידוֹ שֵׁנָא:

[משלי פרק-יח-ו]  שִׂפְתֵי כְסִיל יָבֹאוּ בְרִיב וּפִיו לְמַהֲלֻמוֹת יִקְרָא:

[איוב פרק-לז-ב]  שִׁמְעוּ שָׁמוֹעַ בְּרֹגֶז קֹלוֹ וְהֶגֶה מִפִּיו יֵצֵא:

[דניאל פרק-ו-ב]  שְׁפַר קֳדָם דָּרְיָוֶשׁ וַהֲקִים עַל מַלְכוּתָא לַאֲחַשְׁדַּרְפְּנַיָּא מְאָה וְעֶשְׂרִין דִּי לֶהֱוֹן בְּכָל מַלְכוּתָא:


ש-ב


[בראשית פרק-ב-יא]  שֵׁם הָאֶחָד פִּישׁוֹן הוּא הַסֹּבֵב אֵת כָּל אֶרֶץ הַחֲוִילָה אֲשֶׁר שָׁם הַזָּהָב:

[ירמיה פרק-יב-יא]  שָׂמָהּ לִשְׁמָמָה אָבְלָה עָלַי שְׁמֵמָה נָשַׁמָּה כָּל הָאָרֶץ כִּי אֵין אִישׁ שָׂם עַל לֵב:

[יחזקאל פרק-לב-כב]  שָׁם אַשּׁוּר וְכָל קְהָלָהּ סְבִיבוֹתָיו קִבְרֹתָיו כֻּלָּם חֲלָלִים הַנֹּפְלִים בֶּחָרֶב:

[תהילים פרק-כו-ה]  שָׂנֵאתִי קְהַל מְרֵעִים וְעִם רְשָׁעִים לֹא אֵשֵׁב:

[תהילים פרק-לב-יא]  שִׂמְחוּ בַיהוָה וְגִילוּ צַדִּיקִים וְהַרְנִינוּ כָּל יִשְׁרֵי לֵב:

[תהילים פרק-קיט-קסב]  שָׂשׂ אָנֹכִי עַל אִמְרָתֶךָ כְּמוֹצֵא שָׁלָל רָב:

[תהילים פרק-קכה-א]  שִׁיר הַמַּעֲלוֹת הַבֹּטְחִים בַּיהוָה כְּהַר צִיּוֹן לֹא יִמּוֹט לְעוֹלָם יֵשֵׁב:

[תהילים פרק-קכו-ד]  שׁוּבָה יְהוָה אֶת (שבותנו) שְׁבִיתֵנוּ כַּאֲפִיקִים בַּנֶּגֶב:

[משלי פרק-טו-כג]  שִׂמְחָה לָאִישׁ בְּמַעֲנֵה פִיו וְדָבָר בְּעִתּוֹ מַה טּוֹב:


ש-ג


[נחום פרק-ב-ז]  שַׁעֲרֵי הַנְּהָרוֹת נִפְתָּחוּ וְהַהֵיכָל נָמוֹג:

[תהילים פרק-קב-ח]  שָׁקַדְתִּי וָאֶהְיֶה כְּצִפּוֹר בּוֹדֵד עַל גָּג:

[איוב פרק-כא-ג]  שָׂאוּנִי וְאָנֹכִי אֲדַבֵּר וְאַחַר דַּבְּרִי תַלְעִיג:


ש-ד


[דברים פרק-ו-ד]  שְׁמַע יִשְׂרָאֵל יְהוָה אֱלֹהֵינוּ יְהוָה אֶחָד:

[שמואל א פרק-כה-ח]  שְׁאַל אֶת נְעָרֶיךָ וְיַגִּידוּ לָךְ וְיִמְצְאוּ הַנְּעָרִים חֵן בְּעֵינֶיךָ כִּי עַל יוֹם טוֹב בָּנוּ תְּנָה נָּא אֵת אֲשֶׁר תִּמְצָא יָדְךָ לַעֲבָדֶיךָ וּלְבִנְךָ לְדָוִד:

[צפניה פרק-ג-יג]  שְׁאֵרִית יִשְׂרָאֵל לֹא יַעֲשׂוּ עַוְלָה וְלֹא יְדַבְּרוּ כָזָב וְלֹא יִמָּצֵא בְּפִיהֶם לְשׁוֹן תַּרְמִית כִּי הֵמָּה יִרְעוּ וְרָבְצוּ וְאֵין מַחֲרִיד:

[תהילים פרק-כד-ז]  שְׂאוּ שְׁעָרִים רָאשֵׁיכֶם וְהִנָּשְׂאוּ פִּתְחֵי עוֹלָם וְיָבוֹא מֶלֶךְ הַכָּבוֹד:

[תהילים פרק-כד-ט]  שְׂאוּ שְׁעָרִים רָאשֵׁיכֶם וּשְׂאוּ פִּתְחֵי עוֹלָם וְיָבֹא מֶלֶךְ הַכָּבוֹד:

[תהילים פרק-מט-ב]  שִׁמְעוּ זֹאת כָּל הָעַמִּים הַאֲזִינוּ כָּל יֹשְׁבֵי חָלֶד:

[תהילים פרק-קח-א]  שִׁיר מִזְמוֹר לְדָוִד:

[תהילים פרק-קיט-קסז]  שָׁמְרָה נַפְשִׁי עֵדֹתֶיךָ וָאֹהֲבֵם מְאֹד:

[תהילים פרק-קלג-א]  שִׁיר הַמַּעֲלוֹת לְדָוִד הִנֵּה מַה טּוֹב וּמַה נָּעִים שֶׁבֶת אַחִים גַּם יָחַד:

[תהילים פרק-קמה-כ]  שׁוֹמֵר יְהוָה אֶת כָּל אֹהֲבָיו וְאֵת כָּל הָרְשָׁעִים יַשְׁמִיד:


ש-ה


[בראשית פרק-מט-לא]  שָׁמָּה קָבְרוּ אֶת אַבְרָהָם וְאֵת שָׂרָה אִשְׁתּוֹ שָׁמָּה קָבְרוּ אֶת יִצְחָק וְאֵת רִבְקָה אִשְׁתּוֹ וְשָׁמָּה קָבַרְתִּי אֶת לֵאָה:

[שמות פרק-יג-ו]  שִׁבְעַת יָמִים תֹּאכַל מַצֹּת וּבַיּוֹם הַשְּׁבִיעִי חַג לַיהוָה:

[שמות פרק-כג-יד]  שָׁלֹשׁ רְגָלִים תָּחֹג לִי בַּשָּׁנָה:

[שמות פרק-כג-יז]  שָׁלֹשׁ פְּעָמִים בַּשָּׁנָה יֵרָאֶה כָּל זְכוּרְךָ אֶל פְּנֵי הָאָדֹן יְהוָה:

[שמות פרק-כה-לג]  שְׁלֹשָׁה גְבִעִים מְשֻׁקָּדִים בַּקָּנֶה הָאֶחָד כַּפְתֹּר וָפֶרַח וּשְׁלֹשָׁה גְבִעִים מְשֻׁקָּדִים בַּקָּנֶה הָאֶחָד כַּפְתֹּר וָפָרַח כֵּן לְשֵׁשֶׁת הַקָּנִים הַיֹּצְאִים מִן הַמְּנֹרָה:

[שמות פרק-לז-יט]  שְׁלֹשָׁה גְבִעִים מְשֻׁקָּדִים בַּקָּנֶה הָאֶחָד כַּפְתֹּר וָפֶרַח וּשְׁלֹשָׁה גְבִעִים מְשֻׁקָּדִים בְּקָנֶה אֶחָד כַּפְתֹּר וָפָרַח כֵּן לְשֵׁשֶׁת הַקָּנִים הַיֹּצְאִים מִן הַמְּנֹרָה:

[ויקרא פרק-י-טו]  שׁוֹק הַתְּרוּמָה וַחֲזֵה הַתְּנוּפָה עַל אִשֵּׁי הַחֲלָבִים יָבִיאוּ לְהָנִיף תְּנוּפָה לִפְנֵי יְהוָה וְהָיָה לְךָ וּלְבָנֶיךָ אִתְּךָ לְחָק עוֹלָם כַּאֲשֶׁר צִוָּה יְהוָה:

[ויקרא פרק-כב-כז]  שׁוֹר אוֹ כֶשֶׂב אוֹ עֵז כִּי יִוָּלֵד וְהָיָה שִׁבְעַת יָמִים תַּחַת אִמּוֹ וּמִיּוֹם הַשְּׁמִינִי וָהָלְאָה יֵרָצֶה לְקָרְבַּן אִשֶּׁה לַיהוָה:

[ויקרא פרק-כה-ג]  שֵׁשׁ שָׁנִים תִּזְרַע שָׂדֶךָ וְשֵׁשׁ שָׁנִים תִּזְמֹר כַּרְמֶךָ וְאָסַפְתָּ אֶת תְּבוּאָתָהּ:

[במדבר פרק-לא-כו]  שָׂא אֵת רֹאשׁ מַלְקוֹחַ הַשְּׁבִי בָּאָדָם וּבַבְּהֵמָה אַתָּה וְאֶלְעָזָר הַכֹּהֵן וְרָאשֵׁי אֲבוֹת הָעֵדָה:

[במדבר פרק-לד-טו]  שְׁנֵי הַמַּטּוֹת וַחֲצִי הַמַּטֶּה לָקְחוּ נַחֲלָתָם מֵעֵבֶר לְיַרְדֵּן יְרֵחוֹ קֵדְמָה מִזְרָחָה:

[דברים פרק-ח-ד]  שִׂמְלָתְךָ לֹא בָלְתָה מֵעָלֶיךָ וְרַגְלְךָ לֹא בָצֵקָה זֶה אַרְבָּעִים שָׁנָה:

[דברים פרק-טז-א]  שָׁמוֹר אֶת חֹדֶשׁ הָאָבִיב וְעָשִׂיתָ פֶּסַח לַיהוָה אֱלֹהֶיךָ כִּי בְּחֹדֶשׁ הָאָבִיב הוֹצִיאֲךָ יְהוָה אֱלֹהֶיךָ מִמִּצְרַיִם לָיְלָה:

[דברים פרק-טז-ח]  שֵׁשֶׁת יָמִים תֹּאכַל מַצּוֹת וּבַיּוֹם הַשְּׁבִיעִי עֲצֶרֶת לַיהוָה אֱלֹהֶיךָ לֹא תַעֲשֶׂה מְלָאכָה:

[דברים פרק-יט-ב]  שָׁלוֹשׁ עָרִים תַּבְדִּיל לָךְ בְּתוֹךְ אַרְצְךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לְרִשְׁתָּהּ:

[שמואל א פרק-ב-ה]  שְׂבֵעִים בַּלֶּחֶם נִשְׂכָּרוּ וּרְעֵבִים חָדֵלּוּ עַד עֲקָרָה יָלְדָה שִׁבְעָה וְרַבַּת בָּנִים אֻמְלָלָה:

[מלכים ב פרק-כ-ה]  שׁוּב וְאָמַרְתָּ אֶל חִזְקִיָּהוּ נְגִיד עַמִּי כֹּה אָמַר יְהוָה אֱלֹהֵי דָּוִד אָבִיךָ שָׁמַעְתִּי אֶת תְּפִלָּתֶךָ רָאִיתִי אֶת דִּמְעָתֶךָ הִנְנִי רֹפֶא לָךְ בַּיּוֹם הַשְּׁלִישִׁי תַּעֲלֶה בֵּית יְהוָה:

[מלכים ב פרק-כה-יז]  שְׁמֹנֶה עֶשְׂרֵה אַמָּה קוֹמַת הָעַמּוּד הָאֶחָד וְכֹתֶרֶת עָלָיו נְחֹשֶׁת וְקוֹמַת הַכֹּתֶרֶת שָׁלֹשׁ (אמה) אַמּוֹת וּשְׂבָכָה וְרִמֹּנִים עַל הַכֹּתֶרֶת סָבִיב הַכֹּל נְחֹשֶׁת וְכָאֵלֶּה לַעַמּוּד הַשֵּׁנִי עַל הַשְּׂבָכָה:

[ישעיה פרק-א-י]  שִׁמְעוּ דְבַר יְהוָה קְצִינֵי סְדֹם הַאֲזִינוּ תּוֹרַת אֱלֹהֵינוּ עַם עֲמֹרָה:

[ישעיה פרק-ז-יא]  שְׁאַל לְךָ אוֹת מֵעִם יְהוָה אֱלֹהֶיךָ הַעְמֵק שְׁאָלָה אוֹ הַגְבֵּהַּ לְמָעְלָה:

[ישעיה פרק-יד-יז]  שָׂם תֵּבֵל כַּמִּדְבָּר וְעָרָיו הָרָס אֲסִירָיו לֹא פָתַח בָּיְתָה:

[ישעיה פרק-לד-טו]  שָׁמָּה קִנְּנָה קִפּוֹז וַתְּמַלֵּט וּבָקְעָה וְדָגְרָה בְצִלָּהּ אַךְ שָׁם נִקְבְּצוּ דַיּוֹת אִשָּׁה רְעוּתָהּ:

[ישעיה פרק-מו-יב]  שִׁמְעוּ אֵלַי אַבִּירֵי לֵב הָרְחוֹקִים מִצְּדָקָה:

[ישעיה פרק-מח-א]  שִׁמְעוּ זֹאת בֵּית יַעֲקֹב הַנִּקְרָאִים בְּשֵׁם יִשְׂרָאֵל וּמִמֵּי יְהוּדָה יָצָאוּ הַנִּשְׁבָּעִים בְּשֵׁם יְהוָה וּבֵאלֹהֵי יִשְׂרָאֵל יַזְכִּירוּ לֹא בֶאֱמֶת וְלֹא בִצְדָקָה:

[ישעיה פרק-מט-יח]  שְׂאִי סָבִיב עֵינַיִךְ וּרְאִי כֻּלָּם נִקְבְּצוּ בָאוּ לָךְ חַי אָנִי נְאֻם יְהוָה כִּי כֻלָּם כָּעֲדִי תִלְבָּשִׁי וּתְקַשְּׁרִים כַּכַּלָּה:

[ישעיה פרק-ס-ד]  שְׂאִי סָבִיב עֵינַיִךְ וּרְאִי כֻּלָּם נִקְבְּצוּ בָאוּ לָךְ בָּנַיִךְ מֵרָחוֹק יָבֹאוּ וּבְנֹתַיִךְ עַל צַד תֵּאָמַנָה:

[ישעיה פרק-סא-י]  שׂוֹשׂ אָשִׂישׂ בַּיהוָה תָּגֵל נַפְשִׁי בֵּאלֹהַי כִּי הִלְבִּישַׁנִי בִּגְדֵי יֶשַׁע מְעִיל צְדָקָה יְעָטָנִי כֶּחָתָן יְכַהֵן פְּאֵר וְכַכַּלָּה תַּעְדֶּה כֵלֶיהָ:

[ישעיה פרק-סו-ג]  שׁוֹחֵט הַשּׁוֹר מַכֵּה אִישׁ זוֹבֵחַ הַשֶּׂה עֹרֵף כֶּלֶב מַעֲלֵה מִנְחָה דַּם חֲזִיר מַזְכִּיר לְבֹנָה מְבָרֵךְ אָוֶן גַּם הֵמָּה בָּחֲרוּ בְּדַרְכֵיהֶם וּבְשִׁקּוּצֵיהֶם נַפְשָׁם חָפֵצָה:

[ישעיה פרק-סו-י]  שִׂמְחוּ אֶת יְרוּשָׁלִַם וְגִילוּ בָהּ כָּל אֹהֲבֶיהָ שִׂישׂוּ אִתָּהּ מָשׂוֹשׂ כָּל הַמִּתְאַבְּלִים עָלֶיהָ:

[ירמיה פרק-ב-יב]  שֹׁמּוּ שָׁמַיִם עַל זֹאת וְשַׂעֲרוּ חָרְבוּ מְאֹד נְאֻם יְהוָה:

[ירמיה פרק-ה-א]  שׁוֹטְטוּ בְּחוּצוֹת יְרוּשָׁלִַם וּרְאוּ נָא וּדְעוּ וּבַקְשׁוּ בִרְחוֹבוֹתֶיהָ אִם תִּמְצְאוּ אִישׁ אִם יֵשׁ עֹשֶׂה מִשְׁפָּט מְבַקֵּשׁ אֱמוּנָה וְאֶסְלַח לָהּ:

[ירמיה פרק-ו-יט]  שִׁמְעִי הָאָרֶץ הִנֵּה אָנֹכִי מֵבִיא רָעָה אֶל הָעָם הַזֶּה פְּרִי מַחְשְׁבוֹתָם כִּי עַל דְּבָרַי לֹא הִקְשִׁיבוּ וְתוֹרָתִי וַיִּמְאֲסוּ בָהּ:

[ירמיה פרק-ו-כד]  שָׁמַעְנוּ אֶת שָׁמְעוֹ רָפוּ יָדֵינוּ צָרָה הֶחֱזִיקַתְנוּ חִיל כַּיּוֹלֵדָה:

[ירמיה פרק-ט-ה]  שִׁבְתְּךָ בְּתוֹךְ מִרְמָה בְּמִרְמָה מֵאֲנוּ דַעַת אוֹתִי נְאֻם יְהוָה:

[ירמיה פרק-לב-מד]  שָׂדוֹת בַּכֶּסֶף יִקְנוּ וְכָתוֹב בַּסֵּפֶר וְחָתוֹם וְהָעֵד עֵדִים בְּאֶרֶץ בִּנְיָמִן וּבִסְבִיבֵי יְרוּשָׁלִַם וּבְעָרֵי יְהוּדָה וּבְעָרֵי הָהָר וּבְעָרֵי הַשְּׁפֵלָה וּבְעָרֵי הַנֶּגֶב כִּי אָשִׁיב אֶת שְׁבוּתָם נְאֻם יְהוָה:

[ירמיה פרק-לו-כח]  שׁוּב קַח לְךָ מְגִלָּה אַחֶרֶת וּכְתֹב עָלֶיהָ אֵת כָּל הַדְּבָרִים הָרִאשֹׁנִים אֲשֶׁר הָיוּ עַל הַמְּגִלָּה הָרִאשֹׁנָה אֲשֶׁר שָׂרַף יְהוֹיָקִים מֶלֶךְ יְהוּדָה:

[ירמיה פרק-מט-יד]  שְׁמוּעָה שָׁמַעְתִּי מֵאֵת יְהוָה וְצִיר בַּגּוֹיִם שָׁלוּחַ הִתְקַבְּצוּ וּבֹאוּ עָלֶיהָ וְקוּמוּ לַמִּלְחָמָה:

[ירמיה פרק-נ-מג]  שָׁמַע מֶלֶךְ בָּבֶל אֶת שִׁמְעָם וְרָפוּ יָדָיו צָרָה הֶחֱזִיקַתְהוּ חִיל כַּיּוֹלֵדָה:

[יחזקאל פרק-יד-טז]  שְׁלֹשֶׁת הָאֲנָשִׁים הָאֵלֶּה בְּתוֹכָהּ חַי אָנִי נְאֻם אֲדֹנָי יְהוִה אִם בָּנִים וְאִם בָּנוֹת יַצִּילוּ הֵמָּה לְבַדָּם יִנָּצֵלוּ וְהָאָרֶץ תִּהְיֶה שְׁמָמָה:

[יחזקאל פרק-כב-יב]  שֹׁחַד לָקְחוּ בָךְ לְמַעַן שְׁפָךְ דָּם נֶשֶׁךְ וְתַרְבִּית לָקַחַתְּ וַתְּבַצְּעִי רֵעַיִךְ בַּעֹשֶׁק וְאֹתִי שָׁכַחַתְּ נְאֻם אֲדֹנָי יְהוִה:

[יחזקאל פרק-לה-ט]  שִׁמְמוֹת עוֹלָם אֶתֶּנְךָ וְעָרֶיךָ לֹא (תישבנה) תָשֹׁבְנָה וִידַעְתֶּם כִּי אֲנִי יְהוָה:

[יואל פרק-א-ז]  שָׂם גַּפְנִי לְשַׁמָּה וּתְאֵנָתִי לִקְצָפָה חָשֹׂף חֲשָׂפָהּ וְהִשְׁלִיךְ הִלְבִּינוּ שָׂרִיגֶיהָ:

[עמוס פרק-ד-א]  שִׁמְעוּ הַדָּבָר הַזֶּה פָּרוֹת הַבָּשָׁן אֲשֶׁר בְּהַר שֹׁמְרוֹן הָעֹשְׁקוֹת דַּלִּים הָרֹצְצוֹת אֶבְיוֹנִים הָאֹמְרֹת לַאֲדֹנֵיהֶם הָבִיאָה וְנִשְׁתֶּה:

[עמוס פרק-ד-י]  שִׁלַּחְתִּי בָכֶם דֶּבֶר בְּדֶרֶךְ מִצְרַיִם הָרַגְתִּי בַחֶרֶב בַּחוּרֵיכֶם עִם שְׁבִי סוּסֵיכֶם וָאַעֲלֶה בְּאֹשׁ מַחֲנֵיכֶם וּבְאַפְּכֶם וְלֹא שַׁבְתֶּם עָדַי נְאֻם יְהוָה:

[נחום פרק-ג-טו]  שָׁם תֹּאכְלֵךְ אֵשׁ תַּכְרִיתֵךְ חֶרֶב תֹּאכְלֵךְ כַּיָּלֶק הִתְכַּבֵּד כַּיֶּלֶק הִתְכַּבְּדִי כָּאַרְבֶּה:

[זכריה פרק-י-א]  שַׁאֲלוּ מֵיְהוָה מָטָר בְּעֵת מַלְקוֹשׁ יְהוָה עֹשֶׂה חֲזִיזִים וּמְטַר גֶּשֶׁם יִתֵּן לָהֶם לְאִישׁ עֵשֶׂב בַּשָּׂדֶה:

[תהילים פרק-ט-כא]  שִׁיתָה יְהוָה מוֹרָה לָהֶם יֵדְעוּ גוֹיִם אֱנוֹשׁ הֵמָּה סֶּלָה:

[תהילים פרק-לג-ג]  שִׁירוּ לוֹ שִׁיר חָדָשׁ הֵיטִיבוּ נַגֵּן בִּתְרוּעָה:

[תהילים פרק-לד-כא]  שֹׁמֵר כָּל עַצְמוֹתָיו אַחַת מֵהֵנָּה לֹא נִשְׁבָּרָה:

[תהילים פרק-עו-ד]  שָׁמָּה שִׁבַּר רִשְׁפֵי קָשֶׁת מָגֵן וְחֶרֶב וּמִלְחָמָה סֶלָה:

[תהילים פרק-צ-טו]  שַׂמְּחֵנוּ כִּימוֹת עִנִּיתָנוּ שְׁנוֹת רָאִינוּ רָעָה:

[תהילים פרק-צז-ח]  שָׁמְעָה וַתִּשְׂמַח צִיּוֹן וַתָּגֵלְנָה בְּנוֹת יְהוּדָה לְמַעַן מִשְׁפָּטֶיךָ יְהוָה:

[תהילים פרק-קכב-ד]  שֶׁשָּׁם עָלוּ שְׁבָטִים שִׁבְטֵי יָהּ עֵדוּת לְיִשְׂרָאֵל לְהֹדוֹת לְשֵׁם יְהוָה:

[תהילים פרק-קל-א]  שִׁיר הַמַּעֲלוֹת מִמַּעֲמַקִּים קְרָאתִיךָ יְהוָה:

[תהילים פרק-קלד-ב]  שְׂאוּ יְדֵכֶם קֹדֶשׁ וּבָרֲכוּ אֶת יְהוָה:

[תהילים פרק-קלה-ח]  שֶׁהִכָּה בְּכוֹרֵי מִצְרָיִם מֵאָדָם עַד בְּהֵמָה:

[תהילים פרק-קמ-ד]  שָׁנֲנוּ לְשׁוֹנָם כְּמוֹ נָחָשׁ חֲמַת עַכְשׁוּב תַּחַת שְׂפָתֵימוֹ סֶלָה:

[משלי פרק-ד-א]  שִׁמְעוּ בָנִים מוּסַר אָב וְהַקְשִׁיבוּ לָדַעַת בִּינָה:

[משלי פרק-י-יב]  שִׂנְאָה תְּעוֹרֵר מְדָנִים וְעַל כָּל פְּשָׁעִים תְּכַסֶּה אַהֲבָה:

[משלי פרק-כה-יט]  שֵׁן רֹעָה וְרֶגֶל מוּעָדֶת מִבְטָח בּוֹגֵד בְּיוֹם צָרָה:

[משלי פרק-כז-כ]  שְׁאוֹל (ואבדה) וַאֲבַדּוֹ לֹא תִשְׂבַּעְנָה וְעֵינֵי הָאָדָם לֹא תִשְׂבַּעְנָה:

[איוב פרק-ו-כט]  שֻׁבוּ נָא אַל תְּהִי עַוְלָה (ושבי) וְשׁוּבוּ עוֹד צִדְקִי בָהּ:

[איוב פרק-יב-כא]  שׁוֹפֵךְ בּוּז עַל נְדִיבִים וּמְזִיחַ אֲפִיקִים רִפָּה:

[איוב פרק-טז-יב]  שָׁלֵו הָיִיתִי וַיְפַרְפְּרֵנִי וְאָחַז בְּעָרְפִּי וַיְפַצְפְּצֵנִי וַיְקִימֵנִי לוֹ לְמַטָּרָה:

[איוב פרק-כג-ט]  שְׂמֹאול בַּעֲשֹׂתוֹ וְלֹא אָחַז יַעְטֹף יָמִין וְלֹא אֶרְאֶה:

[איוב פרק-כז-טו]  (שרידו) שְׂרִידָיו בַּמָּוֶת יִקָּבֵרוּ וְאַלְמְנֹתָיו לֹא תִבְכֶּינָה:

[איוב פרק-לז-כג]  שַׁדַּי לֹא מְצָאנֻהוּ שַׂגִּיא כֹחַ וּמִשְׁפָּט וְרֹב צְדָקָה לֹא יְעַנֶּה:

[שיר השירים פרק-א-א]  שִׁיר הַשִּׁירִים אֲשֶׁר לִשְׁלֹמֹה:

[שיר השירים פרק-א-ה]  שְׁחוֹרָה אֲנִי וְנָאוָה בְּנוֹת יְרוּשָׁלִָם כְּאָהֳלֵי קֵדָר כִּירִיעוֹת שְׁלֹמֹה:

[שיר השירים פרק-ז-ד]  שְׁנֵי שָׁדַיִךְ כִּשְׁנֵי עֳפָרִים תָּאֳמֵי צְבִיָּה:

[שיר השירים פרק-ח-ו]  שִׂימֵנִי כַחוֹתָם עַל לִבֶּךָ כַּחוֹתָם עַל זְרוֹעֶךָ כִּי עַזָּה כַמָּוֶת אַהֲבָה קָשָׁה כִשְׁאוֹל קִנְאָה רְשָׁפֶיהָ רִשְׁפֵּי אֵשׁ שַׁלְהֶבֶתְיָה:

[עזרא פרק-ו-ז]  שְׁבֻקוּ לַעֲבִידַת בֵּית אֱלָהָא דֵךְ פַּחַת יְהוּדָיֵא וּלְשָׂבֵי יְהוּדָיֵא בֵּית אֱלָהָא דֵךְ יִבְנוֹן עַל אַתְרֵהּ:

[עזרא פרק-ח-כט]  שִׁקְדוּ וְשִׁמְרוּ עַד תִּשְׁקְלוּ לִפְנֵי שָׂרֵי הַכֹּהֲנִים וְהַלְוִיִּם וְשָׂרֵי הָאָבוֹת לְיִשְׂרָאֵל בִּירוּשָׁלִָם הַלִּשְׁכוֹת בֵּית יְהוָה:

[נחמיה פרק-ג-לו]  שְׁמַע אֱלֹהֵינוּ כִּי הָיִינוּ בוּזָה וְהָשֵׁב חֶרְפָּתָם אֶל רֹאשָׁם וּתְנֵם לְבִזָּה בְּאֶרֶץ שִׁבְיָה:

[נחמיה פרק-י-ג]  שְׂרָיָה עֲזַרְיָה יִרְמְיָה:

[נחמיה פרק-יב-ו]  שְׁמַעְיָה וְיוֹיָרִיב יְדַעְיָה:


ש-ו


[בראשית פרק-יד-ד]  שְׁתֵּים עֶשְׂרֵה שָׁנָה עָבְדוּ אֶת כְּדָרְלָעֹמֶר וּשְׁלֹשׁ עֶשְׂרֵה שָׁנָה מָרָדוּ:

[שמות פרק-יב-ה]  שֶׂה תָמִים זָכָר בֶּן שָׁנָה יִהְיֶה לָכֶם מִן הַכְּבָשִׂים וּמִן הָעִזִּים תִּקָּחוּ:

[שמות פרק-טז-כו]  שֵׁשֶׁת יָמִים תִּלְקְטֻהוּ וּבַיּוֹם הַשְּׁבִיעִי שַׁבָּת לֹא יִהְיֶה בּוֹ:

[ויקרא פרק-כג-לו]  שִׁבְעַת יָמִים תַּקְרִיבוּ אִשֶּׁה לַיהוָה בַּיּוֹם הַשְּׁמִינִי מִקְרָא קֹדֶשׁ יִהְיֶה לָכֶם וְהִקְרַבְתֶּם אִשֶּׁה לַיהוָה עֲצֶרֶת הִוא כָּל מְלֶאכֶת עֲבֹדָה לֹא תַעֲשׂוּ:

[ויקרא פרק-כד-כ]  שֶׁבֶר תַּחַת שֶׁבֶר עַיִן תַּחַת עַיִן שֵׁן תַּחַת שֵׁן כַּאֲשֶׁר יִתֵּן מוּם בָּאָדָם כֵּן יִנָּתֶן בּוֹ:

[שמואל ב פרק-א-כג]  שָׁאוּל וִיהוֹנָתָן הַנֶּאֱהָבִים וְהַנְּעִימִם בְּחַיֵּיהֶם וּבְמוֹתָם לֹא נִפְרָדוּ מִנְּשָׁרִים קַלּוּ מֵאֲרָיוֹת גָּבֵרוּ:

[ישעיה פרק-טז-ו]  שָׁמַעְנוּ גְאוֹן מוֹאָב גֵּא מְאֹד גַּאֲוָתוֹ וּגְאוֹנוֹ וְעֶבְרָתוֹ לֹא כֵן בַּדָּיו:

[ישעיה פרק-נא-ז]  שִׁמְעוּ אֵלַי יֹדְעֵי צֶדֶק עַם תּוֹרָתִי בְלִבָּם אַל תִּירְאוּ חֶרְפַּת אֱנוֹשׁ וּמִגִּדֻּפֹתָם אַל תֵּחָתּוּ:

[ישעיה פרק-ס-ו]  שִׁפְעַת גְּמַלִּים תְּכַסֵּךְ בִּכְרֵי מִדְיָן וְעֵיפָה כֻּלָּם מִשְּׁבָא יָבֹאוּ זָהָב וּלְבוֹנָה יִשָּׂאוּ וּתְהִלֹּת יְהוָה יְבַשֵּׂרוּ:

[ישעיה פרק-סו-ה]  שִׁמְעוּ דְּבַר יְהוָה הַחֲרֵדִים אֶל דְּבָרוֹ אָמְרוּ אֲחֵיכֶם שֹׂנְאֵיכֶם מְנַדֵּיכֶם לְמַעַן שְׁמִי יִכְבַּד יְהוָה וְנִרְאֶה בְשִׂמְחַתְכֶם וְהֵם יֵבֹשׁוּ:

[ירמיה פרק-ג-כב]  שׁוּבוּ בָּנִים שׁוֹבָבִים אֶרְפָּה מְשׁוּבֹתֵיכֶם הִנְנוּ אָתָנוּ לָךְ כִּי אַתָּה יְהוָֹה אֱלֹהֵינוּ:

[ירמיה פרק-ה-כא]  שִׁמְעוּ נָא זֹאת עַם סָכָל וְאֵין לֵב עֵינַיִם לָהֶם וְלֹא יִרְאוּ אָזְנַיִם לָהֶם וְלֹא יִשְׁמָעוּ:

[ירמיה פרק-ה-כח]  שָׁמְנוּ עָשְׁתוּ גַּם עָבְרוּ דִבְרֵי רָע דִּין לֹא דָנוּ דִּין יָתוֹם וְיַצְלִיחוּ וּמִשְׁפַּט אֶבְיוֹנִים לֹא שָׁפָטוּ:

[ירמיה פרק-י-כה]  שְׁפֹךְ חֲמָתְךָ עַל הַגּוֹיִם אֲשֶׁר לֹא יְדָעוּךָ וְעַל מִשְׁפָּחוֹת אֲשֶׁר בְּשִׁמְךָ לֹא קָרָאוּ כִּי אָכְלוּ אֶת יַעֲקֹב וַאֲכָלֻהוּ וַיְכַלֻּהוּ וְאֶת נָוֵהוּ הֵשַׁמּוּ:

[ירמיה פרק-לא-י]  שִׁמְעוּ דְבַר יְהוָה גּוֹיִם וְהַגִּידוּ בָאִיִּים מִמֶּרְחָק וְאִמְרוּ מְזָרֵה יִשְׂרָאֵל יְקַבְּצֶנּוּ וּשְׁמָרוֹ כְּרֹעֶה עֶדְרוֹ:

[ירמיה פרק-מח-טו]  שֻׁדַּד מוֹאָב וְעָרֶיהָ עָלָה וּמִבְחַר בַּחוּרָיו יָרְדוּ לַטָּבַח נְאֻם הַמֶּלֶךְ יְהוָה צְבָאוֹת שְׁמוֹ:

[ירמיה פרק-מח-כט]  שָׁמַעְנוּ גְאוֹן מוֹאָב גֵּאֶה מְאֹד גָּבְהוֹ וּגְאוֹנוֹ וְגַאֲוָתוֹ וְרֻם לִבּוֹ:

[יחזקאל פרק-מג-כה]  שִׁבְעַת יָמִים תַּעֲשֶׂה שְׂעִיר חַטָּאת לַיּוֹם וּפַר בֶּן בָּקָר וְאַיִל מִן הַצֹּאן תְּמִימִים יַעֲשׂוּ:

[יחזקאל פרק-מג-כו]  שִׁבְעַת יָמִים יְכַפְּרוּ אֶת הַמִּזְבֵּחַ וְטִהֲרוּ אֹתוֹ וּמִלְאוּ (ידו) יָדָיו:

[עמוס פרק-ה-י]  שָׂנְאוּ בַשַּׁעַר מוֹכִיחַ וְדֹבֵר תָּמִים יְתָעֵבוּ:

[מיכה פרק-א-ב]  שִׁמְעוּ עַמִּים כֻּלָּם הַקְשִׁיבִי אֶרֶץ וּמְלֹאָהּ וִיהִי אֲדֹנָי יְהוִה בָּכֶם לְעֵד אֲדֹנָי מֵהֵיכַל קָדְשׁוֹ:

[מיכה פרק-ג-ט]  שִׁמְעוּ נָא זֹאת רָאשֵׁי בֵּית יַעֲקֹב וּקְצִינֵי בֵּית יִשְׂרָאֵל הַמֲתַעֲבִים מִשְׁפָּט וְאֵת כָּל הַיְשָׁרָה יְעַקֵּשׁוּ:

[חבקוק פרק-ג-טז]  שָׁמַעְתִּי וַתִּרְגַּז בִּטְנִי לְקוֹל צָלֲלוּ שְׂפָתַי יָבוֹא רָקָב בַּעֲצָמַי וְתַחְתַּי אֶרְגָּז אֲשֶׁר אָנוּחַ לְיוֹם צָרָה לַעֲלוֹת לְעַם יְגוּדֶנּוּ:

[תהילים פרק-יב-ג]  שָׁוְא יְדַבְּרוּ אִישׁ אֶת רֵעֵהוּ שְׂפַת חֲלָקוֹת בְּלֵב וָלֵב יְדַבֵּרוּ:

[תהילים פרק-נה-כא]  שָׁלַח יָדָיו בִּשְׁלֹמָיו חִלֵּל בְּרִיתוֹ:

[תהילים פרק-סה-ג]  שֹׁמֵעַ תְּפִלָּה עָדֶיךָ כָּל בָּשָׂר יָבֹאוּ:

[תהילים פרק-סח-ה]  שִׁירוּ לֵאלֹהִים זַמְּרוּ שְׁמוֹ סֹלּוּ לָרֹכֵב בָּעֲרָבוֹת בְּיָהּ שְׁמוֹ וְעִלְזוּ לְפָנָיו:

[תהילים פרק-עט-ו]  שְׁפֹךְ חֲמָתְךָ אֶל הַגּוֹיִם אֲשֶׁר לֹא יְדָעוּךָ וְעַל מַמְלָכוֹת אֲשֶׁר בְּשִׁמְךָ לֹא קָרָאוּ:

[תהילים פרק-פ-יז]  שְׂרֻפָה בָאֵשׁ כְּסוּחָה מִגַּעֲרַת פָּנֶיךָ יֹאבֵדוּ:

[תהילים פרק-פב-ג]  שִׁפְטוּ דַל וְיָתוֹם עָנִי וָרָשׁ הַצְדִּיקוּ:

[תהילים פרק-פג-יב]  שִׁיתֵמוֹ נְדִיבֵמוֹ כְּעֹרֵב וְכִזְאֵב וּכְזֶבַח וּכְצַלְמֻנָּע כָּל נְסִיכֵמוֹ:

[תהילים פרק-פה-ה]  שׁוּבֵנוּ אֱלֹהֵי יִשְׁעֵנוּ וְהָפֵר כַּעַסְךָ עִמָּנוּ:

[תהילים פרק-פט-מב]  שַׁסֻּהוּ כָּל עֹבְרֵי דָרֶךְ הָיָה חֶרְפָּה לִשְׁכֵנָיו:

[תהילים פרק-צ-יד]  שַׂבְּעֵנוּ בַבֹּקֶר חַסְדֶּךָ וּנְרַנְּנָה וְנִשְׂמְחָה בְּכָל יָמֵינוּ:

[תהילים פרק-צב-יד]  שְׁתוּלִים בְּבֵית יְהוָה בְּחַצְרוֹת אֱלֹהֵינוּ יַפְרִיחוּ:

[תהילים פרק-צו-ב]  שִׁירוּ לַיהוָה בָּרֲכוּ שְׁמוֹ בַּשְּׂרוּ מִיּוֹם לְיוֹם יְשׁוּעָתוֹ:

[תהילים פרק-צז-יב]  שִׂמְחוּ צַדִּיקִים בַּיהוָה וְהוֹדוּ לְזֵכֶר קָדְשׁוֹ:

[תהילים פרק-קד-כו]  שָׁם אֳנִיּוֹת יְהַלֵּכוּן לִוְיָתָן זֶה יָצַרְתָּ לְשַׂחֶק בּוֹ:

[תהילים פרק-קה-ב]  שִׁירוּ לוֹ זַמְּרוּ לוֹ שִׂיחוּ בְּכָל נִפְלְאוֹתָיו:

[תהילים פרק-קה-כ]  שָׁלַח מֶלֶךְ וַיַּתִּירֵהוּ מֹשֵׁל עַמִּים וַיְפַתְּחֵהוּ:

[תהילים פרק-קה-כא]  שָׂמוֹ אָדוֹן לְבֵיתוֹ וּמֹשֵׁל בְּכָל קִנְיָנוֹ:

[תהילים פרק-קה-כו]  שָׁלַח מֹשֶׁה עַבְדּוֹ אַהֲרֹן אֲשֶׁר בָּחַר בּוֹ:

[תהילים פרק-קה-כח]  שָׁלַח חֹשֶׁךְ וַיַּחְשִׁךְ וְלֹא מָרוּ אֶת (דברוו) דְּבָרוֹ:

[תהילים פרק-קכח-א]  שִׁיר הַמַּעֲלוֹת אַשְׁרֵי כָּל יְרֵא יְהוָה הַהֹלֵךְ בִּדְרָכָיו:

[תהילים פרק-קלב-א]  שִׁיר הַמַּעֲלוֹת זְכוֹר יְהוָה לְדָוִד אֵת כָּל עֻנּוֹתוֹ:

[תהילים פרק-קלה-ב]  שֶׁעֹמְדִים בְּבֵית יְהוָה בְּחַצְרוֹת בֵּית אֱלֹהֵינוּ:

[תהילים פרק-קלה-ט]  שָׁלַח אֹתוֹת וּמֹפְתִים בְּתוֹכֵכִי מִצְרָיִם בְּפַרְעֹה וּבְכָל עֲבָדָיו:

[תהילים פרק-קלו-כג]  שֶׁבְּשִׁפְלֵנוּ זָכַר לָנוּ כִּי לְעוֹלָם חַסְדּוֹ:

[משלי פרק-ו-טז]  שֶׁשׁ הֵנָּה שָׂנֵא יְהוָה וְשֶׁבַע (תועבות) תּוֹעֲבַת נַפְשׁוֹ:

[משלי פרק-ח-לג]  שִׁמְעוּ מוּסָר וַחֲכָמוּ וְאַל תִּפְרָעוּ:

[משלי פרק-י-כא]  שִׂפְתֵי צַדִּיק יִרְעוּ רַבִּים וֶאֱוִילִים בַּחֲסַר לֵב יָמוּתוּ:

[משלי פרק-יא-כז]  שֹׁחֵר טוֹב יְבַקֵּשׁ רָצוֹן וְדֹרֵשׁ רָעָה תְבוֹאֶנּוּ:

[משלי פרק-כא-כג]  שֹׁמֵר פִּיו וּלְשׁוֹנוֹ שֹׁמֵר מִצָּרוֹת נַפְשׁוֹ:

[משלי פרק-כט-טו]  שֵׁבֶט וְתוֹכַחַת יִתֵּן חָכְמָה וְנַעַר מְשֻׁלָּח מֵבִישׁ אִמּוֹ:

[איוב פרק-ד-י]  שַׁאֲגַת אַרְיֵה וְקוֹל שָׁחַל וְשִׁנֵּי כְפִירִים נִתָּעוּ:

[איוב פרק-ח-כב]  שֹׂנְאֶיךָ יִלְבְּשׁוּ בֹשֶׁת וְאֹהֶל רְשָׁעִים אֵינֶנּוּ:

[איוב פרק-יג-ו]  שִׁמְעוּ נָא תוֹכַחְתִּי וְרִבוֹת שְׂפָתַי הַקְשִׁיבוּ:

[איוב פרק-יד-ו]  שְׁעֵה מֵעָלָיו וְיֶחְדָּל עַד יִרְצֶה כְּשָׂכִיר יוֹמוֹ:

[איכה פרק-ה-יב]  שָׂרִים בְּיָדָם נִתְלוּ פְּנֵי זְקֵנִים לֹא נֶהְדָּרוּ:

[איכה פרק-ה-טו]  שָׁבַת מְשׂוֹשׂ לִבֵּנוּ נֶהְפַּךְ לְאֵבֶל מְחֹלֵנוּ:

[קהלת פרק-יא-א]  שַׁלַּח לַחְמְךָ עַל פְּנֵי הַמָּיִם כִּי בְרֹב הַיָּמִים תִּמְצָאֶנּוּ:

[דברי הימים א פרק-ד-כה]  שַׁלֻּם בְּנוֹ מִבְשָׂם בְּנוֹ מִשְׁמָע בְּנוֹ:

[דברי הימים א פרק-ו-טו]  שִׁמְעָא בְנוֹ חַגִּיָּה בְנוֹ עֲשָׂיָה בְנוֹ:

[דברי הימים א פרק-טז-ט]  שִׁירוּ לוֹ זַמְּרוּ לוֹ שִׂיחוּ בְּכָל נִפְלְאֹתָיו:

[דברי הימים א פרק-טז-כג]  שִׁירוּ לַיהוָה כָּל הָאָרֶץ בַּשְּׂרוּ מִיּוֹם אֶל יוֹם יְשׁוּעָתוֹ:


ש-ח


[בראשית פרק-ז-ט]  שְׁנַיִם שְׁנַיִם בָּאוּ אֶל נֹחַ אֶל הַתֵּבָה זָכָר וּנְקֵבָה כַּאֲשֶׁר צִוָּה אֱלֹהִים אֶת נֹחַ:

[דברים פרק-טז-טו]  שִׁבְעַת יָמִים תָּחֹג לַיהוָה אֱלֹהֶיךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר יְהוָה כִּי יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל תְּבוּאָתְךָ וּבְכֹל מַעֲשֵׂה יָדֶיךָ וְהָיִיתָ אַךְ שָׂמֵחַ:

[מיכה פרק-ו-ב]  שִׁמְעוּ הָרִים אֶת רִיב יְהוָה וְהָאֵתָנִים מֹסְדֵי אָרֶץ כִּי רִיב לַיהוָה עִם עַמּוֹ וְעִם יִשְׂרָאֵל יִתְוַכָּח:

[זכריה פרק-ג-ח]  שְׁמַע נָא יְהוֹשֻׁעַ הַכֹּהֵן הַגָּדוֹל אַתָּה וְרֵעֶיךָ הַיֹּשְׁבִים לְפָנֶיךָ כִּי אַנְשֵׁי מוֹפֵת הֵמָּה כִּי הִנְנִי מֵבִיא אֶת עַבְדִּי צֶמַח:

[תהילים פרק-ו-י]  שָׁמַע יְהוָה תְּחִנָּתִי יְהוָה תְּפִלָּתִי יִקָּח:

[תהילים פרק-מח-א]  שִׁיר מִזְמוֹר לִבְנֵי קֹרַח:

[איוב פרק-ג-יז]  שָׁם רְשָׁעִים חָדְלוּ רֹגֶז וְשָׁם יָנוּחוּ יְגִיעֵי כֹחַ:

[דברי הימים א פרק-א-כד]  שֵׁם אַרְפַּכְשַׁד שָׁלַח:

[דברי הימים א פרק-א-כו]  שְׂרוּג נָחוֹר תָּרַח:


ש-ט


[תהילים פרק-טז-ח]  שִׁוִּיתִי יְהוָה לְנֶגְדִּי תָמִיד כִּי מִימִינִי בַּל אֶמּוֹט:

[תהילים פרק-לג-יז]  שֶׁקֶר הַסּוּס לִתְשׁוּעָה וּבְרֹב חֵילוֹ לֹא יְמַלֵּט:

[משלי פרק-יז-כג]  שֹׁחַד מֵחֵיק רָשָׁע יִקָּח לְהַטּוֹת אָרְחוֹת מִשְׁפָּט:

[משלי פרק-יח-ה]  שְׂאֵת פְּנֵי רָשָׁע לֹא טוֹב לְהַטּוֹת צַדִּיק בַּמִּשְׁפָּט:

[משלי פרק-כא-ז]  שֹׁד רְשָׁעִים יְגוֹרֵם כִּי מֵאֲנוּ לַעֲשׂוֹת מִשְׁפָּט:

[קהלת פרק-יא-ט]  שְׂמַח בָּחוּר בְּיַלְדוּתֶיךָ וִיטִיבְךָ לִבְּךָ בִּימֵי בְחוּרוֹתֶךָ וְהַלֵּךְ בְּדַרְכֵי לִבְּךָ וּבְמַרְאֵי עֵינֶיךָ וְדָע כִּי עַל כָּל אֵלֶּה יְבִיאֲךָ הָאֱלֹהִים בַּמִּשְׁפָּט:


ש-י


[שמות פרק-יב-טו]  שִׁבְעַת יָמִים מַצּוֹת תֹּאכֵלוּ אַךְ בַּיּוֹם הָרִאשׁוֹן תַּשְׁבִּיתוּ שְּׂאֹר מִבָּתֵּיכֶם כִּי כָּל אֹכֵל חָמֵץ וְנִכְרְתָה הַנֶּפֶשׁ הַהִוא מִיִּשְׂרָאֵל מִיּוֹם הָרִאשֹׁן עַד יוֹם הַשְּׁבִעִי:

[שמות פרק-לד-יא]  שְׁמָר לְךָ אֵת אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם הִנְנִי גֹרֵשׁ מִפָּנֶיךָ אֶת הָאֱמֹרִי וְהַכְּנַעֲנִי וְהַחִתִּי וְהַפְּרִזִּי וְהַחִוִּי וְהַיְבוּסִי:

[שמואל א פרק-כב-כג]  שְׁבָה אִתִּי אַל תִּירָא כִּי אֲשֶׁר יְבַקֵּשׁ אֶת נַפְשִׁי יְבַקֵּשׁ אֶת נַפְשֶׁךָ כִּי מִשְׁמֶרֶת אַתָּה עִמָּדִי:

[שמואל ב פרק-כג-כה]  שַׁמָּה הַחֲרֹדִי אֱלִיקָא הַחֲרֹדִי:

[שמואל ב פרק-כג-לג]  שַׁמָּה הַהֲרָרִי אֲחִיאָם בֶּן שָׁרָר הָארָרִי:

[ישעיה פרק-א-ב]  שִׁמְעוּ שָׁמַיִם וְהַאֲזִינִי אֶרֶץ כִּי יְהוָה דִּבֵּר בָּנִים גִּדַּלְתִּי וְרוֹמַמְתִּי וְהֵם פָּשְׁעוּ בִי:

[ישעיה פרק-לג-יג]  שִׁמְעוּ רְחוֹקִים אֲשֶׁר עָשִׂיתִי וּדְעוּ קְרוֹבִים גְּבֻרָתִי:

[ישעיה פרק-לח-יג]  שִׁוִּיתִי עַד בֹּקֶר כָּאֲרִי כֵּן יְשַׁבֵּר כָּל עַצְמוֹתָי מִיּוֹם עַד לַיְלָה תַּשְׁלִימֵנִי:

[ישעיה פרק-מט-א]  שִׁמְעוּ אִיִּים אֵלַי וְהַקְשִׁיבוּ לְאֻמִּים מֵרָחוֹק יְהוָה מִבֶּטֶן קְרָאָנִי מִמְּעֵי אִמִּי הִזְכִּיר שְׁמִי:

[ירמיה פרק-ד-כ]  שֶׁבֶר עַל שֶׁבֶר נִקְרָא כִּי שֻׁדְּדָה כָּל הָאָרֶץ פִּתְאֹם שֻׁדְּדוּ אֹהָלַי רֶגַע יְרִיעֹתָי:

[ירמיה פרק-כג-כה]  שָׁמַעְתִּי אֵת אֲשֶׁר אָמְרוּ הַנְּבִאִים הַנִּבְּאִים בִּשְׁמִי שֶׁקֶר לֵאמֹר חָלַמְתִּי חָלָמְתִּי:

[ירמיה פרק-לא-יח]  שָׁמוֹעַ שָׁמַעְתִּי אֶפְרַיִם מִתְנוֹדֵד יִסַּרְתַּנִי וָאִוָּסֵר כְּעֵגֶל לֹא לֻמָּד הֲשִׁיבֵנִי וְאָשׁוּבָה כִּי אַתָּה יְהוָה אֱלֹהָי:

[תהילים פרק-ז-א]  שִׁגָּיוֹן לְדָוִד אֲשֶׁר שָׁר לַיהוָה עַל דִּבְרֵי כוּשׁ בֶּן יְמִינִי:

[תהילים פרק-יז-ח]  שָׁמְרֵנִי כְּאִישׁוֹן בַּת עָיִן בְּצֵל כְּנָפֶיךָ תַּסְתִּירֵנִי:

[תהילים פרק-יט-יג]  שְׁגִיאוֹת מִי יָבִין מִנִּסְתָּרוֹת נַקֵּנִי:

[תהילים פרק-כז-ז]  שְׁמַע יְהוָה קוֹלִי אֶקְרָא וְחָנֵּנִי וַעֲנֵנִי:

[תהילים פרק-ל-יא]  שְׁמַע יְהוָה וְחָנֵּנִי יְהוָה הֱיֵה עֹזֵר לִי:

[תהילים פרק-לא-ז]  שָׂנֵאתִי הַשֹּׁמְרִים הַבְלֵי שָׁוְא וַאֲנִי אֶל יְהוָה בָּטָחְתִּי:

[תהילים פרק-לה-כד]  שָׁפְטֵנִי כְצִדְקְךָ יְהוָה אֱלֹהָי וְאַל יִשְׂמְחוּ לִי:

[תהילים פרק-לט-יג]  שִׁמְעָה תְפִלָּתִי יְהוָה וְשַׁוְעָתִי הַאֲזִינָה אֶל דִּמְעָתִי אַל תֶּחֱרַשׁ כִּי גֵר אָנֹכִי עִמָּךְ תּוֹשָׁב כְּכָל אֲבוֹתָי:

[תהילים פרק-מג-א]  שָׁפְטֵנִי אֱלֹהִים וְרִיבָה רִיבִי מִגּוֹי לֹא חָסִיד מֵאִישׁ מִרְמָה וְעַוְלָה תְפַלְּטֵנִי:

[תהילים פרק-נ-ז]  שִׁמְעָה עַמִּי וַאֲדַבֵּרָה יִשְׂרָאֵל וְאָעִידָה בָּךְ אֱלֹהִים אֱלֹהֶיךָ אָנֹכִי:

[תהילים פרק-סא-ב]  שִׁמְעָה אֱלֹהִים רִנָּתִי הַקְשִׁיבָה תְּפִלָּתִי:

[תהילים פרק-סד-ב]  שְׁמַע אֱלֹהִים קוֹלִי בְשִׂיחִי מִפַּחַד אוֹיֵב תִּצֹּר חַיָּי:

[תהילים פרק-סח-כח]  שָׁם בִּנְיָמִן צָעִיר רֹדֵם שָׂרֵי יְהוּדָה רִגְמָתָם שָׂרֵי זְבֻלוּן שָׂרֵי נַפְתָּלִי:

[תהילים פרק-פא-ט]  שְׁמַע עַמִּי וְאָעִידָה בָּךְ יִשְׂרָאֵל אִם תִּשְׁמַע לִי:

[תהילים פרק-פח-א]  שִׁיר מִזְמוֹר לִבְנֵי קֹרַח לַמְנַצֵּחַ עַל מָחֲלַת לְעַנּוֹת מַשְׂכִּיל לְהֵימָן הָאֶזְרָחִי:

[תהילים פרק-קטז-ז]  שׁוּבִי נַפְשִׁי לִמְנוּחָיְכִי כִּי יְהוָה גָּמַל עָלָיְכִי:

[תהילים פרק-קיט-קסא]  שָׂרִים רְדָפוּנִי חִנָּם (ומדבריך) וּמִדְּבָרְךָ פָּחַד לִבִּי:

[תהילים פרק-קיט-קסג]  שֶׁקֶר שָׂנֵאתִי וַאֲתַעֵבָה תּוֹרָתְךָ אָהָבְתִּי:

[תהילים פרק-קיט-קסו]  שִׂבַּרְתִּי לִישׁוּעָתְךָ יְהוָה וּמִצְוֹתֶיךָ עָשִׂיתִי:

[תהילים פרק-קכ-א]  שִׁיר הַמַּעֲלוֹת אֶל יְהוָה בַּצָּרָתָה לִּי קָרָאתִי וַיַּעֲנֵנִי:

[תהילים פרק-קכא-א]  שִׁיר לַמַּעֲלוֹת אֶשָּׂא עֵינַי אֶל הֶהָרִים מֵאַיִן יָבֹא עֶזְרִי:

[תהילים פרק-קלא-א]  שִׁיר הַמַּעֲלוֹת לְדָוִד יְהוָה לֹא גָבַהּ לִבִּי וְלֹא רָמוּ עֵינַי וְלֹא הִלַּכְתִּי בִּגְדֹלוֹת וּבְנִפְלָאוֹת מִמֶּנִּי:

[תהילים פרק-קלב-יז]  שָׁם אַצְמִיחַ קֶרֶן לְדָוִד עָרַכְתִּי נֵר לִמְשִׁיחִי:

[תהילים פרק-קמ-ה]  שָׁמְרֵנִי יְהוָה מִידֵי רָשָׁע מֵאִישׁ חֲמָסִים תִּנְצְרֵנִי אֲשֶׁר חָשְׁבוּ לִדְחוֹת פְּעָמָי:

[תהילים פרק-קמא-ג]  שִׁיתָה יְהוָה שָׁמְרָה לְפִי נִצְּרָה עַל דַּל שְׂפָתָי:

[משלי פרק-ל-ח]  שָׁוְא וּדְבַר כָּזָב הַרְחֵק מִמֶּנִּי רֵאשׁ וָעֹשֶׁר אַל תִּתֶּן לִי הַטְרִיפֵנִי לֶחֶם חֻקִּי:

[איוב פרק-טז-טו]  שַׂק תָּפַרְתִּי עֲלֵי גִלְדִּי וְעֹלַלְתִּי בֶעָפָר קַרְנִי:

[איוב פרק-כג-ז]  שָׁם יָשָׁר נוֹכָח עִמּוֹ וַאֲפַלְּטָה לָנֶצַח מִשֹּׁפְטִי:

[איוב פרק-כט-יט]  שָׁרְשִׁי פָתוּחַ אֱלֵי מָיִם וְטַל יָלִין בִּקְצִירִי:

[איוב פרק-לד-ב]  שִׁמְעוּ חֲכָמִים מִלָּי וְיֹדְעִים הַאֲזִינוּ לִי:

[איוב פרק-מב-ד]  שְׁמַע נָא וְאָנֹכִי אֲדַבֵּר אֶשְׁאָלְךָ וְהוֹדִיעֵנִי:

[שיר השירים פרק-ב-ו]  שְׂמֹאלוֹ תַּחַת לְרֹאשִׁי וִימִינוֹ תְּחַבְּקֵנִי:

[שיר השירים פרק-ח-ג]  שְׂמֹאלוֹ תַּחַת רֹאשִׁי וִימִינוֹ תְּחַבְּקֵנִי:

[איכה פרק-א-כא]  שָׁמְעוּ כִּי נֶאֱנָחָה אָנִי אֵין מְנַחֵם לִי כָּל אֹיְבַי שָׁמְעוּ רָעָתִי שָׂשׂוּ כִּי אַתָּה עָשִׂיתָ הֵבֵאתָ יוֹם קָרָאתָ וְיִהְיוּ כָמוֹנִי:

[איכה פרק-ג-סא]  שָׁמַעְתָּ חֶרְפָּתָם יְהוָה כָּל מַחְשְׁבֹתָם עָלָי:

[איכה פרק-ד-כא]  שִׂישִׂי וְשִׂמְחִי בַּת אֱדוֹם (יושבתי) יוֹשֶׁבֶת בְּאֶרֶץ עוּץ גַּם עָלַיִךְ תַּעֲבָר כּוֹס תִּשְׁכְּרִי וְתִתְעָרִי:

[דברי הימים א פרק-יא-כז]  שַׁמּוֹת הַהֲרוֹרִי חֶלֶץ הַפְּלוֹנִי:


ש-כ


[בראשית פרק-כג-ו]  שְׁמָעֵנוּ אֲדֹנִי נְשִׂיא אֱלֹהִים אַתָּה בְּתוֹכֵנוּ בְּמִבְחַר קְבָרֵינוּ קְבֹר אֶת מֵתֶךָ אִישׁ מִמֶּנּוּ אֶת קִבְרוֹ לֹא יִכְלֶה מִמְּךָ מִקְּבֹר מֵתֶךָ:

[שמות פרק-כ-ט]  שֵׁשֶׁת יָמִים תַּעֲבֹד וְעָשִׂיתָ כָּל מְלַאכְתֶּךָ:

[דברים פרק-ה-יב]  שָׁמוֹר אֶת יוֹם הַשַּׁבָּת לְקַדְּשׁוֹ כַּאֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ:

[דברים פרק-ה-יג]  שֵׁשֶׁת יָמִים תַּעֲבֹד וְעָשִׂיתָ כָּֿל מְלַאכְתֶּךָ:

[דברים פרק-ו-יז]  שָׁמוֹר תִּשְׁמְרוּן אֶת מִצְוֹת יְהוָה אֱלֹהֵיכֶם וְעֵדֹתָיו וְחֻקָּיו אֲשֶׁר צִוָּךְ:

[דברים פרק-יב-כח]  שְׁמֹר וְשָׁמַעְתָּ אֵת כָּל הַדְּבָרִים הָאֵלֶּה אֲשֶׁר אָנֹכִי מְצַוֶּךָּ לְמַעַן יִיטַב לְךָ וּלְבָנֶיךָ אַחֲרֶיךָ עַד עוֹלָם כִּי תַעֲשֶׂה הַטּוֹב וְהַיָּשָׁר בְּעֵינֵי יְהוָה אֱלֹהֶיךָ:

[שמואל א פרק-כה-כח]  שָׂא נָא לְפֶשַׁע אֲמָתֶךָ כִּי עָשֹׂה יַעֲשֶׂה יְהוָה לַאדֹנִי בַּיִת נֶאֱמָן כִּי מִלְחֲמוֹת יְהוָה אֲדֹנִי נִלְחָם וְרָעָה לֹא תִמָּצֵא בְךָ מִיָּמֶיךָ:

[ישעיה פרק-נא-יט]  שְׁתַּיִם הֵנָּה קֹרְאֹתַיִךְ מִי יָנוּד לָךְ הַשֹּׁד וְהַשֶּׁבֶר וְהָרָעָב וְהַחֶרֶב מִי אֲנַחֲמֵךְ:

[ירמיה פרק-ג-ב]  שְׂאִי עֵינַיִךְ עַל שְׁפָיִם וּרְאִי אֵיפֹה לֹא (שגלת) שֻׁכַּבְּתְּ עַל דְּרָכִים יָשַׁבְתְּ לָהֶם כַּעֲרָבִי בַּמִּדְבָּר וַתַּחֲנִיפִי אֶרֶץ בִּזְנוּתַיִךְ וּבְרָעָתֵךְ:

[ירמיה פרק-יג-כ]  (שאי) שְׂאוּ עֵינֵיכֶם (וראי) וּרְאוּ הַבָּאִים מִצָּפוֹן אַיֵּה הָעֵדֶר נִתַּן לָךְ צֹאן תִּפְאַרְתֵּךְ:

[ירמיה פרק-נא-יג]  (שכנתי) שֹׁכַנְתְּ עַל מַיִם רַבִּים רַבַּת אוֹצָרֹת בָּא קִצֵּךְ אַמַּת בִּצְעֵךְ:

[יחזקאל פרק-כז-ז]  שֵׁשׁ בְּרִקְמָה מִמִּצְרַיִם הָיָה מִפְרָשֵׂךְ לִהְיוֹת לָךְ לְנֵס תְּכֵלֶת וְאַרְגָּמָן מֵאִיֵּי אֱלִישָׁה הָיָה מְכַסֵּךְ:

[הושע פרק-יג-ט]  שִׁחֶתְךָ יִשְׂרָאֵל כִּי בִי בְעֶזְרֶךָ:

[הושע פרק-יד-ב]  שׁוּבָה יִשְׂרָאֵל עַד יְהוָה אֱלֹהֶיךָ כִּי כָשַׁלְתָּ בַּעֲוֹנֶךָ:

[מיכה פרק-ו-א]  שִׁמְעוּ נָא אֵת אֲשֶׁר יְהוָה אֹמֵר קוּם רִיב אֶת הֶהָרִים וְתִשְׁמַעְנָה הַגְּבָעוֹת קוֹלֶךָ:

[חבקוק פרק-ב-טז]  שָׂבַעְתָּ קָלוֹן מִכָּבוֹד שְׁתֵה גַם אַתָּה וְהֵעָרֵל תִּסּוֹב עָלֶיךָ כּוֹס יְמִין יְהוָה וְקִיקָלוֹן עַל כְּבוֹדֶךָ:

[חבקוק פרק-ג-יא]  שֶׁמֶשׁ יָרֵחַ עָמַד זְבֻלָה לְאוֹר חִצֶּיךָ יְהַלֵּכוּ לְנֹגַהּ בְּרַק חֲנִיתֶךָ:

[זכריה פרק-ט-יב]  שׁוּבוּ לְבִצָּרוֹן אֲסִירֵי הַתִּקְוָה גַּם הַיּוֹם מַגִּיד מִשְׁנֶה אָשִׁיב לָךְ:

[תהילים פרק-ו-ה]  שׁוּבָה יְהוָה חַלְּצָה נַפְשִׁי הוֹשִׁיעֵנִי לְמַעַן חַסְדֶּךָ:

[תהילים פרק-כה-כ]  שָׁמְרָה נַפְשִׁי וְהַצִּילֵנִי אַל אֵבוֹשׁ כִּי חָסִיתִי בָךְ:

[תהילים פרק-כח-ב]  שְׁמַע קוֹל תַּחֲנוּנַי בְּשַׁוְּעִי אֵלֶיךָ בְּנָשְׂאִי יָדַי אֶל דְּבִיר קָדְשֶׁךָ:

[תהילים פרק-מג-ג]  שְׁלַח אוֹרְךָ וַאֲמִתְּךָ הֵמָּה יַנְחוּנִי יְבִיאוּנִי אֶל הַר קָדְשְׁךָ וְאֶל מִשְׁכְּנוֹתֶיךָ:

[תהילים פרק-מה-יא]  שִׁמְעִי בַת וּרְאִי וְהַטִּי אָזְנֵךְ וְשִׁכְחִי עַמֵּךְ וּבֵית אָבִיךְ:

[תהילים פרק-עד-ז]  שִׁלְחוּ בָאֵשׁ מִקְדָּשֶׁךָ לָאָרֶץ חִלְּלוּ מִשְׁכַּן שְׁמֶךָ:

[תהילים פרק-פו-ב]  שָׁמְרָה נַפְשִׁי כִּי חָסִיד אָנִי הוֹשַׁע עַבְדְּךָ אַתָּה אֱלֹהַי הַבּוֹטֵחַ אֵלֶיךָ:

[תהילים פרק-צ-ח]  (שת) שַׁתָּה עֲוֹנֹתֵינוּ לְנֶגְדֶּךָ עֲלֻמֵנוּ לִמְאוֹר פָּנֶיךָ:

[תהילים פרק-צ-יג]  שׁוּבָה יְהוָה עַד מָתָי וְהִנָּחֵם עַל עֲבָדֶיךָ:

[תהילים פרק-קז-מ]  שֹׁפֵךְ בּוּז עַל נְדִיבִים וַיַּתְעֵם בְּתֹהוּ לֹא דָרֶךְ:

[תהילים פרק-קיט-קסד]  שֶׁבַע בַּיּוֹם הִלַּלְתִּיךָ עַל מִשְׁפְּטֵי צִדְקֶךָ:

[תהילים פרק-קיט-קסח]  שָׁמַרְתִּי פִקּוּדֶיךָ וְעֵדֹתֶיךָ כִּי כָל דְּרָכַי נֶגְדֶּךָ:

[תהילים פרק-קכב-א]  שִׁיר הַמַּעֲלוֹת לְדָוִד שָׂמַחְתִּי בְּאֹמְרִים לִי בֵּית יְהוָה נֵלֵךְ:

[תהילים פרק-קכב-ו]  שַׁאֲלוּ שְׁלוֹם יְרוּשָׁלִָם יִשְׁלָיוּ אֹהֲבָיִךְ:

[משלי פרק-א-ח]  שְׁמַע בְּנִי מוּסַר אָבִיךָ וְאַל תִּטֹּשׁ תּוֹרַת אִמֶּךָ:

[משלי פרק-ה-טו]  שְׁתֵה מַיִם מִבּוֹרֶךָ וְנֹזְלִים מִתּוֹךְ בְּאֵרֶךָ:

[משלי פרק-ז-ב]  שְׁמֹר מִצְוֹתַי וֶחְיֵה וְתוֹרָתִי כְּאִישׁוֹן עֵינֶיךָ:

[משלי פרק-יט-כ]  שְׁמַע עֵצָה וְקַבֵּל מוּסָר לְמַעַן תֶּחְכַּם בְּאַחֲרִיתֶךָ:

[משלי פרק-כג-יט]  שְׁמַע אַתָּה בְנִי וַחֲכָם וְאַשֵּׁר בַּדֶּרֶךְ לִבֶּךָ:

[משלי פרק-כג-כב]  שְׁמַע לְאָבִיךָ זֶה יְלָדֶךָ וְאַל תָּבוּז כִּי זָקְנָה אִמֶּךָ:

[משלי פרק-ל-כח]  שְׂמָמִית בְּיָדַיִם תְּתַפֵּשׂ וְהִיא בְּהֵיכְלֵי מֶלֶךְ:


ש-ל


[שמות פרק-לד-כג]  שָׁלֹשׁ פְּעָמִים בַּשָּׁנָה יֵרָאֶה כָּל זְכוּרְךָ אֶת פְּנֵי הָאָדֹן יְהוָה אֱלֹהֵי יִשְׂרָאֵל:

[במדבר פרק-כו-ב]  שְׂאוּ אֶת רֹאשׁ כָּל עֲדַת בְּנֵי יִשְׂרָאֵל מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה לְבֵית אֲבֹתָם כָּל יֹצֵא צָבָא בְּיִשְׂרָאֵל:

[דברים פרק-לב-ה]  שִׁחֵת לוֹ לֹא בָּנָיו מוּמָם דּוֹר עִקֵּשׁ וּפְתַלְתֹּל:

[שופטים פרק-ה-ג]  שִׁמְעוּ מְלָכִים הַאֲזִינוּ רֹזְנִים אָנֹכִי לַיהוָה אָנֹכִי אָשִׁירָה אֲזַמֵּר לַיהוָה אֱלֹהֵי יִשְׂרָאֵל:

[ישעיה פרק-ה-כט]  שְׁאָגָה לוֹ כַּלָּבִיא (ושאג) יִשְׁאַג כַּכְּפִירִים וְיִנְהֹם וְיֹאחֵז טֶרֶף וְיַפְלִיט וְאֵין מַצִּיל:

[ישעיה פרק-לא-ו]  שׁוּבוּ לַאֲשֶׁר הֶעְמִיקוּ סָרָה בְּנֵי יִשְׂרָאֵל:

[ירמיה פרק-ב-ד]  שִׁמְעוּ דְבַר יְהוָה בֵּית יַעֲקֹב וְכָל מִשְׁפְּחוֹת בֵּית יִשְׂרָאֵל:

[ירמיה פרק-ד-ו]  שְׂאוּ נֵס צִיּוֹנָה הָעִיזוּ אַל תַּעֲמֹדוּ כִּי רָעָה אָנֹכִי מֵבִיא מִצָּפוֹן וְשֶׁבֶר גָּדוֹל:

[ירמיה פרק-י-א]  שִׁמְעוּ אֶת הַדָּבָר אֲשֶׁר דִּבֶּר יְהוָה עֲלֵיכֶם בֵּית יִשְׂרָאֵל:

[ירמיה פרק-לד-יט]  שָׂרֵי יְהוּדָה וְשָׂרֵי יְרוּשָׁלִַם הַסָּרִסִים וְהַכֹּהֲנִים וְכֹל עַם הָאָרֶץ הָעֹבְרִים בֵּין בִּתְרֵי הָעֵגֶל:

[ירמיה פרק-נ-יז]  שֶׂה פְזוּרָה יִשְׂרָאֵל אֲרָיוֹת הִדִּיחוּ הָרִאשׁוֹן אֲכָלוֹ מֶלֶךְ אַשּׁוּר וְזֶה הָאַחֲרוֹן עִצְּמוֹ נְבוּכַדְרֶאצַּר מֶלֶךְ בָּבֶל:

[תהילים פרק-קיט-קסה]  שָׁלוֹם רָב לְאֹהֲבֵי תוֹרָתֶךָ וְאֵין לָמוֹ מִכְשׁוֹל:

[תהילים פרק-קכד-א]  שִׁיר הַמַּעֲלוֹת לְדָוִד לוּלֵי יְהוָה שֶׁהָיָה לָנוּ יֹאמַר נָא יִשְׂרָאֵל:

[איוב פרק-כא-י]  שׁוֹרוֹ עִבַּר וְלֹא יַגְעִל תְּפַלֵּט פָּרָתוֹ וְלֹא תְשַׁכֵּל:

[בראשית פרק-ט-ו]  שֹׁפֵךְ דַּם הָאָדָם בָּאָדָם דָּמוֹ יִשָּׁפֵךְ כִּי בְּצֶלֶם אֱלֹהִים עָשָׂה אֶת הָאָדָם:

[בראשית פרק-מב-טז]  שִׁלְחוּ מִכֶּם אֶחָד וְיִקַּח אֶת אֲחִיכֶם וְאַתֶּם הֵאָסְרוּ וְיִבָּחֲנוּ דִּבְרֵיכֶם הַאֱמֶת אִתְּכֶם וְאִם לֹא חֵי פַרְעֹה כִּי מְרַגְּלִים אַתֶּם:

[בראשית פרק-מט-ה]  שִׁמְעוֹן וְלֵוִי אַחִים כְּלֵי חָמָס מְכֵרֹתֵיהֶם:

[שמות פרק-טז-יב]  שָׁמַעְתִּי אֶת תְּלוּנֹּת בְּנֵי יִשְׂרָאֵל דַּבֵּר אֲלֵהֶם לֵאמֹר בֵּין הָעַרְבַּיִם תֹּאכְלוּ בָשָׂר וּבַבֹּקֶר תִּשְׂבְּעוּ לָחֶם וִידַעְתֶּם כִּי אֲנִי יְהוָה אֱלֹהֵיכֶם:

[שמות פרק-כב-י]  שְׁבֻעַת יְהוָה תִּהְיֶה בֵּין שְׁנֵיהֶם אִם לֹא שָׁלַח יָדוֹ בִּמְלֶאכֶת רֵעֵהוּ וְלָקַח בְּעָלָיו וְלֹא יְשַׁלֵּם:

[שמות פרק-כה-ו]  שֶׁמֶן לַמָּאֹר בְּשָׂמִים לְשֶׁמֶן הַמִּשְׁחָה וְלִקְטֹרֶת הַסַּמִּים:

[שמות פרק-כח-י]  שִׁשָּׁה מִשְּׁמֹתָם עַל הָאֶבֶן הָאֶחָת וְאֶת שְׁמוֹת הַשִּׁשָּׁה הַנּוֹתָרִים עַל הָאֶבֶן הַשֵּׁנִית כְּתוֹלְדֹתָם:

[ויקרא פרק-טז-לא]  שַׁבַּת שַׁבָּתוֹן הִיא לָכֶם וְעִנִּיתֶם אֶת נַפְשֹׁתֵיכֶם חֻקַּת עוֹלָם:

[ויקרא פרק-כג-ג]  שֵׁשֶׁת יָמִים תֵּעָשֶׂה מְלָאכָה וּבַיּוֹם הַשְּׁבִיעִי שַׁבַּת שַׁבָּתוֹן מִקְרָא קֹדֶשׁ כָּל מְלָאכָה לֹא תַעֲשׂוּ שַׁבָּת הִוא לַיהוָה בְּכֹל מוֹשְׁבֹתֵיכֶם:

[ויקרא פרק-כג-לב]  שַׁבַּת שַׁבָּתוֹן הוּא לָכֶם וְעִנִּיתֶם אֶת נַפְשֹׁתֵיכֶם בְּתִשְׁעָה לַחֹדֶשׁ בָּעֶרֶב מֵעֶרֶב עַד עֶרֶב תִּשְׁבְּתוּ שַׁבַּתְּכֶם:

[במדבר פרק-א-ב]  שְׂאוּ אֶת רֹאשׁ כָּל עֲדַת בְּנֵי יִשְׂרָאֵל לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמוֹת כָּל זָכָר לְגֻלְגְּלֹתָם:

[במדבר פרק-יג-ב]  שְׁלַח לְךָ אֲנָשִׁים וְיָתֻרוּ אֶת אֶרֶץ כְּנַעַן אֲשֶׁר אֲנִי נֹתֵן לִבְנֵי יִשְׂרָאֵל אִישׁ אֶחָד אִישׁ אֶחָד לְמַטֵּה אֲבֹתָיו תִּשְׁלָחוּ כֹּל נָשִׂיא בָהֶם:

[במדבר פרק-כח-ל]  שְׂעִיר עִזִּים אֶחָד לְכַפֵּר עֲלֵיכֶם:

[במדבר פרק-כט-יא]  שְׂעִיר עִזִּים אֶחָד חַטָּאת מִלְּבַד חַטַּאת הַכִּפֻּרִים וְעֹלַת הַתָּמִיד וּמִנְחָתָהּ וְנִסְכֵּיהֶם:

[דברים פרק-ט-א]  שְׁמַע יִשְׂרָאֵל אַתָּה עֹבֵר הַיּוֹם אֶת הַיַּרְדֵּן לָבֹא לָרֶשֶׁת גּוֹיִם גְּדֹלִים וַעֲצֻמִים מִמֶּךָּ עָרִים גְּדֹלֹת וּבְצֻרֹת בַּשָּׁמָיִם:

[דברים פרק-טז-טז]  שָׁלוֹשׁ פְּעָמִים בַּשָּׁנָה יֵרָאֶה כָל זְכוּרְךָ אֶת פְּנֵי יְהוָה אֱלֹהֶיךָ בַּמָּקוֹם אֲשֶׁר יִבְחָר בְּחַג הַמַּצּוֹת וּבְחַג הַשָּׁבֻעוֹת וּבְחַג הַסֻּכּוֹת וְלֹא יֵרָאֶה אֶת פְּנֵי יְהוָה רֵיקָם:

[דברים פרק-כב-ז]  שַׁלֵּחַ תְּשַׁלַּח אֶת הָאֵם וְאֶת הַבָּנִים תִּקַּח לָךְ לְמַעַן יִיטַב לָךְ וְהַאֲרַכְתָּ יָמִים:

[מלכים א פרק-טו-ב]  שָׁלֹשׁ שָׁנִים מָלַךְ בִּירוּשָׁלִָם וְשֵׁם אִמּוֹ מַעֲכָה בַּת אֲבִישָׁלוֹם:

[ישעיה פרק-א-כג]  שָׂרַיִךְ סוֹרְרִים וְחַבְרֵי גַּנָּבִים כֻּלּוֹ אֹהֵב שֹׁחַד וְרֹדֵף שַׁלְמֹנִים יָתוֹם לֹא יִשְׁפֹּטוּ וְרִיב אַלְמָנָה לֹא יָבוֹא אֲלֵיהֶם:

[ישעיה פרק-יד-ה]  שָׁבַר יְהוָה מַטֵּה רְשָׁעִים שֵׁבֶט מֹשְׁלִים:

[ישעיה פרק-יד-ט]  שְׁאוֹל מִתַּחַת רָגְזָה לְךָ לִקְרַאת בּוֹאֶךָ עוֹרֵר לְךָ רְפָאִים כָּל עַתּוּדֵי אָרֶץ הֵקִים מִכִּסְאוֹתָם כֹּל מַלְכֵי גוֹיִם:

[ישעיה פרק-מב-י]  שִׁירוּ לַיהוָה שִׁיר חָדָשׁ תְּהִלָּתוֹ מִקְצֵה הָאָרֶץ יוֹרְדֵי הַיָּם וּמְלֹאוֹ אִיִּים וְיֹשְׁבֵיהֶם:

[ישעיה פרק-מו-ג]  שִׁמְעוּ אֵלַי בֵּית יַעֲקֹב וְכָל שְׁאֵרִית בֵּית יִשְׂרָאֵל הַעֲמֻסִים מִנִּי בֶטֶן הַנְּשֻׂאִים מִנִּי רָחַם:

[ישעיה פרק-מח-ו]  שָׁמַעְתָּ חֲזֵה כֻּלָּהּ וְאַתֶּם הֲלוֹא תַגִּידוּ הִשְׁמַעְתִּיךָ חֲדָשׁוֹת מֵעַתָּה וּנְצֻרוֹת וְלֹא יְדַעְתָּם:

[ישעיה פרק-נא-א]  שִׁמְעוּ אֵלַי רֹדְפֵי צֶדֶק מְבַקְשֵׁי יְהוָה הַבִּיטוּ אֶל צוּר חֻצַּבְתֶּם וְאֶל מַקֶּבֶת בּוֹר נֻקַּרְתֶּם:

[ירמיה פרק-יא-ב]  שִׁמְעוּ אֶת דִּבְרֵי הַבְּרִית הַזֹּאת וְדִבַּרְתָּם אֶל אִישׁ יְהוּדָה וְעַל יֹשְׁבֵי יְרוּשָׁלִָם:

[ירמיה פרק-יא-י]  שָׁבוּ עַל עֲוֹנֹת אֲבוֹתָם הָרִאשֹׁנִים אֲשֶׁר מֵאֲנוּ לִשְׁמוֹעַ אֶת דְּבָרַי וְהֵמָּה הָלְכוּ אַחֲרֵי אֱלֹהִים אֲחֵרִים לְעָבְדָם הֵפֵרוּ בֵית יִשְׂרָאֵל וּבֵית יְהוּדָה אֶת בְּרִיתִי אֲשֶׁר כָּרַתִּי אֶת אֲבוֹתָם:

[ירמיה פרק-כ-יג]  שִׁירוּ לַיהוָה הַלְלוּ אֶת יְהוָה כִּי הִצִּיל אֶת נֶפֶשׁ אֶבְיוֹן מִיַּד מְרֵעִים:

[ירמיה פרק-מו-יב]  שָׁמְעוּ גוֹיִם קְלוֹנֵךְ וְצִוְחָתֵךְ מָלְאָה הָאָרֶץ כִּי גִבּוֹר בְּגִבּוֹר כָּשָׁלוּ יַחְדָּיו נָפְלוּ שְׁנֵיהֶם:

[יחזקאל פרק-ה-ב]  שְׁלִשִׁית בָּאוּר תַּבְעִיר בְּתוֹךְ הָעִיר כִּמְלֹאת יְמֵי הַמָּצוֹר וְלָקַחְתָּ אֶת הַשְּׁלִשִׁית תַּכֶּה בַחֶרֶב סְבִיבוֹתֶיהָ וְהַשְּׁלִשִׁית תִּזְרֶה לָרוּחַ וְחֶרֶב אָרִיק אַחֲרֵיהֶם:

[יחזקאל פרק-ה-יב]  שְׁלִשִׁתֵיךְ בַּדֶּבֶר יָמוּתוּ וּבָרָעָב יִכְלוּ בְתוֹכֵךְ וְהַשְּׁלִשִׁית בַּחֶרֶב יִפְּלוּ סְבִיבוֹתָיִךְ וְהַשְּׁלִישִׁית לְכָל רוּחַ אֱזָרֶה וְחֶרֶב אָרִיק אַחֲרֵיהֶם:

[יחזקאל פרק-לב-כו]  שָׁם מֶשֶׁךְ תֻּבַל וְכָל הֲמוֹנָהּ סְבִיבוֹתָיו קִבְרוֹתֶיהָ כֻּלָּם עֲרֵלִים מְחֻלְלֵי חֶרֶב כִּי נָתְנוּ חִתִּיתָם בְּאֶרֶץ חַיִּים:

[יואל פרק-א-ב]  שִׁמְעוּ זֹאת הַזְּקֵנִים וְהַאֲזִינוּ כֹּל יוֹשְׁבֵי הָאָרֶץ הֶהָיְתָה זֹּאת בִּימֵיכֶם וְאִם בִּימֵי אֲבֹתֵיכֶם:

[יואל פרק-ד-יג]  שִׁלְחוּ מַגָּל כִּי בָשַׁל קָצִיר בֹּאוּ רְדוּ כִּי מָלְאָה גַּת הֵשִׁיקוּ הַיְקָבִים כִּי רַבָּה רָעָתָם:

[יואל פרק-ד-טו]  שֶׁמֶשׁ וְיָרֵחַ קָדָרוּ וְכוֹכָבִים אָסְפוּ נָגְהָם:

[עמוס פרק-ה-כא]  שָׂנֵאתִי מָאַסְתִּי חַגֵּיכֶם וְלֹא אָרִיחַ בְּעַצְּרֹתֵיכֶם:

[מיכה פרק-ג-ב]  שֹׂנְאֵי טוֹב וְאֹהֲבֵי (רעה) רָע גֹּזְלֵי עוֹרָם מֵעֲלֵיהֶם וּשְׁאֵרָם מֵעַל עַצְמוֹתָם:

[צפניה פרק-ב-ח]  שָׁמַעְתִּי חֶרְפַּת מוֹאָב וְגִדּוּפֵי בְּנֵי עַמּוֹן אֲשֶׁר חֵרְפוּ אֶת עַמִּי וַיַּגְדִּילוּ עַל גְּבוּלָם:

[חגי פרק-ב-יח]  שִׂימוּ נָא לְבַבְכֶם מִן הַיּוֹם הַזֶּה וָמָעְלָה מִיּוֹם עֶשְׂרִים וְאַרְבָּעָה לַתְּשִׁיעִי לְמִן הַיּוֹם אֲשֶׁר יֻסַּד הֵיכַל יְהוָה שִׂימוּ לְבַבְכֶם:

[תהילים פרק-לו-יג]  שָׁם נָפְלוּ פֹּעֲלֵי אָוֶן דֹּחוּ וְלֹא יָכְלוּ קוּם:

[תהילים פרק-לז-לז]  שְׁמָר תָּם וּרְאֵה יָשָׁר כִּי אַחֲרִית לְאִישׁ שָׁלוֹם:

[תהילים פרק-נג-ו]  שָׁם פָּחֲדוּ פַחַד לֹא הָיָה פָחַד כִּי אֱלֹהִים פִּזַּר עַצְמוֹת חֹנָךְ הֱבִשֹׁתָה כִּי אֱלֹהִים מְאָסָם:

[תהילים פרק-נו-ג]  שָׁאֲפוּ שׁוֹרְרַי כָּל הַיּוֹם כִּי רַבִּים לֹחֲמִים לִי מָרוֹם:

[תהילים פרק-סט-כה]  שְׁפָךְ עֲלֵיהֶם זַעְמֶךָ וַחֲרוֹן אַפְּךָ יַשִּׂיגֵם:

[תהילים פרק-קה-כז]  שָׂמוּ בָם דִּבְרֵי אֹתוֹתָיו וּמֹפְתִים בְּאֶרֶץ חָם:

[תהילים פרק-קה-ל]  שָׁרַץ אַרְצָם צְפַרְדְּעִים בְּחַדְרֵי מַלְכֵיהֶם:

[תהילים פרק-קה-לח]  שָׂמַח מִצְרַיִם בְּצֵאתָם כִּי נָפַל פַּחְדָּם עֲלֵיהֶם:

[תהילים פרק-קה-מ]  שָׁאַל וַיָּבֵא שְׂלָו וְלֶחֶם שָׁמַיִם יַשְׂבִּיעֵם:

[תהילים פרק-קו-כא]  שָׁכְחוּ אֵל מוֹשִׁיעָם עֹשֶׂה גְדֹלוֹת בְּמִצְרָיִם:

[תהילים פרק-קכג-א]  שִׁיר הַמַּעֲלוֹת אֵלֶיךָ נָשָׂאתִי אֶת עֵינַי הַיֹּשְׁבִי בַּשָּׁמָיִם:

[תהילים פרק-קכו-א]  שִׁיר הַמַּעֲלוֹת בְּשׁוּב יְהוָה אֶת שִׁיבַת צִיּוֹן הָיִינוּ כְּחֹלְמִים:

[תהילים פרק-קלה-י]  שֶׁהִכָּה גּוֹיִם רַבִּים וְהָרַג מְלָכִים עֲצוּמִים:

[משלי פרק-ד-י]  שְׁמַע בְּנִי וְקַח אֲמָרָי וְיִרְבּוּ לְךָ שְׁנוֹת חַיִּים:

[משלי פרק-ח-ו]  שִׁמְעוּ כִּי נְגִידִים אֲדַבֵּר וּמִפְתַּח שְׂפָתַי מֵישָׁרִים:

[משלי פרק-טו-יא]  שְׁאוֹל וַאֲבַדּוֹן נֶגֶד יְהוָה אַף כִּי לִבּוֹת בְּנֵי אָדָם:

[משלי פרק-כב-יד]  שׁוּחָה עֲמֻקָּה פִּי זָרוֹת זְעוּם יְהוָה (יפול) יִפָּל שָׁם:

[משלי פרק-כד-כו]  שְׂפָתַיִם יִשָּׁק מֵשִׁיב דְּבָרִים נְכֹחִים:

[משלי פרק-כו-ג]  שׁוֹט לַסּוּס מֶתֶג לַחֲמוֹר וְשֵׁבֶט לְגֵו כְּסִילִים:

[משלי פרק-ל-יח]  שְׁלֹשָׁה הֵמָּה נִפְלְאוּ מִמֶּנִּי (וארבע) וְאַרְבָּעָה לֹא יְדַעְתִּים:

[משלי פרק-ל-כו]  שְׁפַנִּים עַם לֹא עָצוּם וַיָּשִׂימוּ בַסֶּלַע בֵּיתָם:

[איוב פרק-יב-ד]  שְׂחֹק לְרֵעֵהוּ אֶהְיֶה קֹרֵא לֶאֱלוֹהַּ וַיַּעֲנֵהוּ שְׂחוֹק צַדִּיק תָּמִים:

[איוב פרק-יג-יז]  שִׁמְעוּ שָׁמוֹעַ מִלָּתִי וְאַחֲוָתִי בְּאָזְנֵיכֶם:

[איוב פרק-טז-ב]  שָׁמַעְתִּי כְאֵלֶּה רַבּוֹת מְנַחֲמֵי עָמָל כֻּלְּכֶם:

[איוב פרק-כ-כה]  שָׁלַף וַיֵּצֵא מִגֵּוָה וּבָרָק מִמְּרֹרָתוֹ יַהֲלֹךְ עָלָיו אֵמִים:

[איוב פרק-כא-ב]  שִׁמְעוּ שָׁמוֹעַ מִלָּתִי וּתְהִי זֹאת תַּנְחוּמֹתֵיכֶם:

[איוב פרק-כט-ט]  שָׂרִים עָצְרוּ בְמִלִּים וְכַף יָשִׂימוּ לְפִיהֶם:

[איוב פרק-לה-יב]  שָׁם יִצְעֲקוּ וְלֹא יַעֲנֶה מִפְּנֵי גְּאוֹן רָעִים:

[שיר השירים פרק-ד-ב]  שִׁנַּיִךְ כְּעֵדֶר הַקְּצוּבוֹת שֶׁעָלוּ מִן הָרַחְצָה שֶׁכֻּלָּם מַתְאִימוֹת וְשַׁכֻּלָה אֵין בָּהֶם:

[שיר השירים פרק-ד-ה]  שְׁנֵי שָׁדַיִךְ כִּשְׁנֵי עֳפָרִים תְּאוֹמֵי צְבִיָּה הָרוֹעִים בַּשּׁוֹשַׁנִּים:

[שיר השירים פרק-ד-יג]  שְׁלָחַיִךְ פַּרְדֵּס רִמּוֹנִים עִם פְּרִי מְגָדִים כְּפָרִים עִם נְרָדִים:

[שיר השירים פרק-ה-טו]  שׁוֹקָיו עַמּוּדֵי שֵׁשׁ מְיֻסָּדִים עַל אַדְנֵי פָז מַרְאֵהוּ כַּלְּבָנוֹן בָּחוּר כָּאֲרָזִים:

[שיר השירים פרק-ו-ו]  שִׁנַּיִךְ כְּעֵדֶר הָרְחֵלִים שֶׁעָלוּ מִן הָרַחְצָה שֶׁכֻּלָּם מַתְאִימוֹת וְשַׁכֻּלָה אֵין בָּהֶם:

[שיר השירים פרק-ז-א]  שׁוּבִי שׁוּבִי הַשּׁוּלַמִּית שׁוּבִי שׁוּבִי וְנֶחֱזֶה בָּךְ מַה תֶּחֱזוּ בַּשּׁוּלַמִּית כִּמְחֹלַת הַמַּחֲנָיִם:

[שיר השירים פרק-ז-ג]  שָׁרְרֵךְ אַגַּן הַסַּהַר אַל יֶחְסַר הַמָּזֶג בִּטְנֵךְ עֲרֵמַת חִטִּים סוּגָה בַּשּׁוֹשַׁנִּים:

[רות פרק-א-יב]  שֹׁבְנָה בְנֹתַי לֵכְןָ כִּי זָקַנְתִּי מִהְיוֹת לְאִישׁ כִּי אָמַרְתִּי יֶשׁ לִי תִקְוָה גַּם הָיִיתִי הַלַּיְלָה לְאִישׁ וְגַם יָלַדְתִּי בָנִים:

[איכה פרק-ג-סב]  שִׂפְתֵי קָמַי וְהֶגְיוֹנָם עָלַי כָּל הַיּוֹם:

[איכה פרק-ג-סג]  שִׁבְתָּם וְקִימָתָם הַבִּיטָה אֲנִי מַנְגִּינָתָם:

[קהלת פרק-ח-ה]  שׁוֹמֵר מִצְוָה לֹא יֵדַע דָּבָר רָע וְעֵת וּמִשְׁפָּט יֵדַע לֵב חָכָם:

[קהלת פרק-ט-יא]  שַׁבְתִּי וְרָאֹה תַחַת הַשֶּׁמֶשׁ כִּי לֹא לַקַּלִּים הַמֵּרוֹץ וְלֹא לַגִּבּוֹרִים הַמִּלְחָמָה וְגַם לֹא לַחֲכָמִים לֶחֶם וְגַם לֹא לַנְּבֹנִים עֹשֶׁר וְגַם לֹא לַיֹּדְעִים חֵן כִּי עֵת וָפֶגַע יִקְרֶה אֶת כֻּלָּם:

[דניאל פרק-ט-כד]  שָׁבֻעִים שִׁבְעִים נֶחְתַּךְ עַל עַמְּךָ וְעַל עִיר קָדְשֶׁךָ לְכַלֵּא הַפֶּשַׁע (ולחתם) וּלְהָתֵם (חטאות) חַטָּאת וּלְכַפֵּר עָוֹן וּלְהָבִיא צֶדֶק עֹלָמִים וְלַחְתֹּם חָזוֹן וְנָבִיא וְלִמְשֹׁחַ קֹדֶשׁ קָדָשִׁים:

[נחמיה פרק-יא-יא]  שְׂרָיָה בֶן חִלְקִיָּה בֶּן מְשֻׁלָּם בֶּן צָדוֹק בֶּן מְרָיוֹת בֶּן אֲחִיטוּב נְגִד בֵּית הָאֱלֹהִים:

[דברי הימים א פרק-ג-ד]  שִׁשָּׁה נוֹלַד לוֹ בְחֶבְרוֹן וַיִּמְלָךְ שָׁם שֶׁבַע שָׁנִים וְשִׁשָּׁה חֳדָשִׁים וּשְׁלֹשִׁים וְשָׁלוֹשׁ שָׁנָה מָלַךְ בִּירוּשָׁלִָם:

[דברי הימים א פרק-כט-ד]  שְׁלֹשֶׁת אֲלָפִים כִּכְּרֵי זָהָב מִזְּהַב אוֹפִיר וְשִׁבְעַת אֲלָפִים כִּכַּר כֶּסֶף מְזֻקָּק לָטוּחַ קִירוֹת הַבָּתִּים:

[דברי הימים ב פרק-יג-ב]  שָׁלוֹשׁ שָׁנִים מָלַךְ בִּירוּשָׁלִַם וְשֵׁם אִמּוֹ מִיכָיָהוּ בַת אוּרִיאֵל מִן גִּבְעָה וּמִלְחָמָה הָיְתָה בֵּין אֲבִיָּה וּבֵין יָרָבְעָם:


ש-נ


[בראשית פרק-מב-לב]  שְׁנֵים עָשָׂר אֲנַחְנוּ אַחִים בְּנֵי אָבִינוּ הָאֶחָד אֵינֶנּוּ וְהַקָּטֹן הַיּוֹם אֶת אָבִינוּ בְּאֶרֶץ כְּנָעַן:

[שמות פרק-כו-יז]  שְׁתֵּי יָדוֹת לַקֶּרֶשׁ הָאֶחָד מְשֻׁלָּבֹת אִשָּׁה אֶל אֲחֹתָהּ כֵּן תַּעֲשֶׂה לְכֹל קַרְשֵׁי הַמִּשְׁכָּן:

[שמות פרק-לו-כב]  שְׁתֵּי יָדֹת לַקֶּרֶשׁ הָאֶחָד מְשֻׁלָּבֹת אַחַת אֶל אֶחָת כֵּן עָשָׂה לְכֹל קַרְשֵׁי הַמִּשְׁכָּן:

[במדבר פרק-יא-ח]  שָׁטוּ הָעָם וְלָקְטוּ וְטָחֲנוּ בָרֵחַיִם אוֹ דָכוּ בַּמְּדֹכָה וּבִשְּׁלוּ בַּפָּרוּר וְעָשׂוּ אֹתוֹ עֻגוֹת וְהָיָה טַעְמוֹ כְּטַעַם לְשַׁד הַשָּׁמֶן:

[מלכים א פרק-ד-יח]  שִׁמְעִי בֶן אֵלָא בְּבִנְיָמִן:

[מלכים ב פרק-טו-יג]  שַׁלּוּם בֶּן יָבֵישׁ מָלַךְ בִּשְׁנַת שְׁלֹשִׁים וָתֵשַׁע שָׁנָה לְעֻזִיָּה מֶלֶךְ יְהוּדָה וַיִּמְלֹךְ יֶרַח יָמִים בְּשֹׁמְרוֹן:

[ישעיה פרק-טז-א]  שִׁלְחוּ כַר מֹשֵׁל אֶרֶץ מִסֶּלַע מִדְבָּרָה אֶל הַר בַּת צִיּוֹן:

[ישעיה פרק-מח-יב]  שְׁמַע אֵלַי יַעֲקֹב וְיִשְׂרָאֵל מְקֹרָאִי אֲנִי הוּא אֲנִי רִאשׁוֹן אַף אֲנִי אַחֲרוֹן:

[ירמיה פרק-ג-יד]  שׁוּבוּ בָנִים שׁוֹבָבִים נְאֻם יְהוָה כִּי אָנֹכִי בָּעַלְתִּי בָכֶם וְלָקַחְתִּי אֶתְכֶם אֶחָד מֵעִיר וּשְׁנַיִם מִמִּשְׁפָּחָה וְהֵבֵאתִי אֶתְכֶם צִיּוֹן:

[ירמיה פרק-ל-ו]  שַׁאֲלוּ נָא וּרְאוּ אִם יֹלֵד זָכָר מַדּוּעַ רָאִיתִי כָל גֶּבֶר יָדָיו עַל חֲלָצָיו כַּיּוֹלֵדָה וְנֶהֶפְכוּ כָל פָּנִים לְיֵרָקוֹן:

[יחזקאל פרק-כג-לג]  שִׁכָּרוֹן וְיָגוֹן תִּמָּלֵאִי כּוֹס שַׁמָּה וּשְׁמָמָה כּוֹס אֲחוֹתֵךְ שֹׁמְרוֹן:

[תהילים פרק-מח-יד]  שִׁיתוּ לִבְּכֶם לְחֵילָה פַּסְּגוּ אַרְמְנוֹתֶיהָ לְמַעַן תְּסַפְּרוּ לְדוֹר אַחֲרוֹן:

[תהילים פרק-קמא-ט]  שָׁמְרֵנִי מִידֵי פַח יָקְשׁוּ לִי וּמֹקְשׁוֹת פֹּעֲלֵי אָוֶן:

[תהילים פרק-קמז-יב]  שַׁבְּחִי יְרוּשָׁלִַם אֶת יְהוָה הַלְלִי אֱלֹהַיִךְ צִיּוֹן:

[משלי פרק-יג-טו]  שֵׂכֶל טוֹב יִתֶּן חֵן וְדֶרֶךְ בֹּגְדִים אֵיתָן:

[משלי פרק-טו-ז]  שִׂפְתֵי חֲכָמִים יְזָרוּ דָעַת וְלֵב כְּסִילִים לֹא כֵן:

[משלי פרק-כא-טו]  שִׂמְחָה לַצַּדִּיק עֲשׂוֹת מִשְׁפָּט וּמְחִתָּה לְפֹעֲלֵי אָוֶן:

[משלי פרק-ל-טז]  שְׁאוֹל וְעֹצֶר רָחַם אֶרֶץ לֹא שָׂבְעָה מַּיִם וְאֵשׁ לֹא אָמְרָה הוֹן:


ש-ע


[דברים פרק-כח-לא]  שׁוֹרְךָ טָבוּחַ לְעֵינֶיךָ וְלֹא תֹאכַל מִמֶּנּוּ חֲמֹרְךָ גָּזוּל מִלְּפָנֶיךָ וְלֹא יָשׁוּב לָךְ צֹאנְךָ נְתֻנוֹת לְאֹיְבֶיךָ וְאֵין לְךָ מוֹשִׁיעַ:

[יחזקאל פרק-כב-כז]  שָׂרֶיהָ בְקִרְבָּהּ כִּזְאֵבִים טֹרְפֵי טָרֶף לִשְׁפָּךְ דָּם לְאַבֵּד נְפָשׁוֹת לְמַעַן בְּצֹעַ בָּצַע:

[תהילים פרק-קטז-ו]  שֹׁמֵר פְּתָאיִם יְהוָֹה דַּלּוֹתִי וְלִי יְהוֹשִׁיעַ:

[משלי פרק-יט-יא]  שֵׂכֶל אָדָם הֶאֱרִיךְ אַפּוֹ וְתִפאַרְתּוֹ עֲבֹר עַל פָּשַׁע:

[איוב פרק-יז-ג]  שִׂימָה נָּא עָרְבֵנִי עִמָּךְ מִי הוּא לְיָדִי יִתָּקֵעַ:

[קהלת פרק-ד-יז]  שְׁמֹר (רגליך) רַגְלְךָ כַּאֲשֶׁר תֵּלֵךְ אֶל בֵּית הָאֱלֹהִים וְקָרוֹב לִשְׁמֹעַ מִתֵּת הַכְּסִילִים זָבַח כִּי אֵינָם יוֹדְעִים לַעֲשׂוֹת רָע:


ש-פ


[ישעיה פרק-ו-ב]  שְׂרָפִים עֹמְדִים מִמַּעַל לוֹ שֵׁשׁ כְּנָפַיִם שֵׁשׁ כְּנָפַיִם לְאֶחָד בִּשְׁתַּיִם יְכַסֶּה פָנָיו וּבִשְׁתַּיִם יְכַסֶּה רַגְלָיו וּבִשְׁתַּיִם יְעוֹפֵף:

[עמוס פרק-ה-טו]  שִׂנְאוּ רָע וְאֶהֱבוּ טוֹב וְהַצִּיגוּ בַשַּׁעַר מִשְׁפָּט אוּלַי יֶחֱנַן יְהוָה אֱלֹהֵי צְבָאוֹת שְׁאֵרִית יוֹסֵף:

[תהילים פרק-פג-א]  שִׁיר מִזְמוֹר לְאָסָף:

[תהילים פרק-קה-יז]  שָׁלַח לִפְנֵיהֶם אִישׁ לְעֶבֶד נִמְכַּר יוֹסֵף:

[איוב פרק-מ-לב]  שִׂים עָלָיו כַּפֶּךָ זְכֹר מִלְחָמָה אַל תּוֹסַף:

[דניאל פרק-ב-לג]  שָׁקוֹהִי דִּי פַרְזֶל רַגְלוֹהִי (מנהון) מִנְּהֵין דִּי פַרְזֶל (ומנהון) וּמִנְּהֵין דִּי חֲסַף:

[עזרא פרק-י-מב]  שַׁלּוּם אֲמַרְיָה יוֹסֵף:

[דברי הימים א פרק-כז-ה]  שַׂר הַצָּבָא הַשְּׁלִישִׁי לַחֹדֶשׁ הַשְּׁלִישִׁי בְּנָיָהוּ בֶן יְהוֹיָדָע הַכֹּהֵן רֹאשׁ וְעַל מַחֲלֻקְתּוֹ עֶשְׂרִים וְאַרְבָּעָה אָלֶף:


ש-צ


[בראשית פרק-ט-יט]  שְׁלֹשָׁה אֵלֶּה בְּנֵי נֹחַ וּמֵאֵלֶּה נָפְצָה כָל הָאָרֶץ:

[שמות פרק-יב-יט]  שִׁבְעַת יָמִים שְׂאֹר לֹא יִמָּצֵא בְּבָתֵּיכֶם כִּי כָּל אֹכֵל מַחְמֶצֶת וְנִכְרְתָה הַנֶּפֶשׁ הַהִוא מֵעֲדַת יִשְׂרָאֵל בַּגֵּר וּבְאֶזְרַח הָאָרֶץ:

[ירמיה פרק-ה-ל]  שַׁמָּה וְשַׁעֲרוּרָה נִהְיְתָה בָּאָרֶץ:

[הושע פרק-ד-א]  שִׁמְעוּ דְבַר יְהוָה בְּנֵי יִשְׂרָאֵל כִּי רִיב לַיהוָה עִם יוֹשְׁבֵי הָאָרֶץ כִּי אֵין אֱמֶת וְאֵין חֶסֶד וְאֵין דַּעַת אֱלֹהִים בָּאָרֶץ:

[עמוס פרק-ח-ד]  שִׁמְעוּ זֹאת הַשֹּׁאֲפִים אֶבְיוֹן וְלַשְׁבִּית (ענוי) עֲנִיֵּי אָרֶץ:

[תהילים פרק-ב-ח]  שְׁאַל מִמֶּנִּי וְאֶתְּנָה גוֹיִם נַחֲלָתֶךָ וַאֲחֻזָּתְךָ אַפְסֵי אָרֶץ:

[תהילים פרק-עג-ט]  שַׁתּוּ בַשָּׁמַיִם פִּיהֶם וּלְשׁוֹנָם תִּהֲלַךְ בָּאָרֶץ:

[תהילים פרק-צו-א]  שִׁירוּ לַיהוָה שִׁיר חָדָשׁ שִׁירוּ לַיהוָה כָּל הָאָרֶץ:


ש-ק


[דברים פרק-טז-יח]  שֹׁפְטִים וְשֹׁטְרִים תִּתֶּן לְךָ בְּכָל שְׁעָרֶיךָ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ לִשְׁבָטֶיךָ וְשָׁפְטוּ אֶת הָעָם מִשְׁפַּט צֶדֶק:

[תהילים פרק-יד-ה]  שָׁם פָּחֲדוּ פָחַד כִּי אֱלֹהִים בְּדוֹר צַדִּיק:

[משלי פרק-יד-יט]  שַׁחוּ רָעִים לִפְנֵי טוֹבִים וּרְשָׁעִים עַל שַׁעֲרֵי צַדִּיק:


ש-ר


[שמות פרק-כג-יב]  שֵׁשֶׁת יָמִים תַּעֲשֶׂה מַעֲשֶׂיךָ וּבַיּוֹם הַשְּׁבִיעִי תִּשְׁבֹּת לְמַעַן יָנוּחַ שׁוֹרְךָ וַחֲמֹרֶךָ וְיִנָּפֵשׁ בֶּן אֲמָתְךָ וְהַגֵּר:

[שמות פרק-כח-ז]  שְׁתֵּי כְתֵפֹת חֹבְרֹת יִהְיֶה לּוֹ אֶל שְׁנֵי קְצוֹתָיו וְחֻבָּר:

[ישעיה פרק-י-כא]  שְׁאָר יָשׁוּב שְׁאָר יַעֲקֹב אֶל אֵל גִּבּוֹר:

[ישעיה פרק-כד-ח]  שָׁבַת מְשׂוֹשׂ תֻּפִּים חָדַל שְׁאוֹן עַלִּיזִים שָׁבַת מְשׂוֹשׂ כִּנּוֹר:

[ישעיה פרק-מ-כו]  שְׂאוּ מָרוֹם עֵינֵיכֶם וּרְאוּ מִי בָרָא אֵלֶּה הַמּוֹצִיא בְמִסְפָּר צְבָאָם לְכֻלָּם בְּשֵׁם יִקְרָא מֵרֹב אוֹנִים וְאַמִּיץ כֹּחַ אִישׁ לֹא נֶעְדָּר:

[ירמיה פרק-יג-טו]  שִׁמְעוּ וְהַאֲזִינוּ אַל תִּגְבָּהוּ כִּי יְהוָה דִּבֵּר:

[ירמיה פרק-כט-לא]  שְׁלַח עַל כָּל הַגּוֹלָה לֵאמֹר כֹּה אָמַר יְהוָה אֶל שְׁמַעְיָה הַנֶּחֱלָמִי יַעַן אֲשֶׁר נִבָּא לָכֶם שְׁמַעְיָה וַאֲנִי לֹא שְׁלַחְתִּיו וַיַּבְטַח אֶתְכֶם עַל שָׁקֶר:

[ירמיה פרק-מח-יא]  שַׁאֲנַן מוֹאָב מִנְּעוּרָיו וְשֹׁקֵט הוּא אֶל שְׁמָרָיו וְלֹא הוּרַק מִכְּלִי אֶל כֶּלִי וּבַגּוֹלָה לֹא הָלָךְ עַל כֵּן עָמַד טַעְמוֹ בּוֹ וְרֵיחוֹ לֹא נָמָר:

[ירמיה פרק-נא-כז]  שְׂאוּ נֵס בָּאָרֶץ תִּקְעוּ שׁוֹפָר בַּגּוֹיִם קַדְּשׁוּ עָלֶיהָ גּוֹיִם הַשְׁמִיעוּ עָלֶיהָ מַמְלְכוֹת אֲרָרַט מִנִּי וְאַשְׁכְּנָז פִּקְדוּ עָלֶיהָ טִפְסָר הַעֲלוּ סוּס כְּיֶלֶק סָמָר:

[יחזקאל פרק-לב-כד]  שָׁם עֵילָם וְכָל הֲמוֹנָהּ סְבִיבוֹת קְבֻרָתָהּ כֻּלָּם חֲלָלִים הַנֹּפְלִים בַּחֶרֶב אֲשֶׁר יָרְדוּ עֲרֵלִים אֶל אֶרֶץ תַּחְתִּיּוֹת אֲשֶׁר נָתְנוּ חִתִּיתָם בְּאֶרֶץ חַיִּים וַיִּשְׂאוּ כְלִמָּתָם אֶת יוֹרְדֵי בוֹר:

[יחזקאל פרק-לב-כט]  שָׁמָּה אֱדוֹם מְלָכֶיהָ וְכָל נְשִׂיאֶיהָ אֲשֶׁר נִתְּנוּ בִגְבוּרָתָם אֶת חַלְלֵי חָרֶב הֵמָּה אֶת עֲרֵלִים יִשְׁכָּבוּ וְאֶת יֹרְדֵי בוֹר:

[יחזקאל פרק-לב-ל]  שָׁמָּה נְסִיכֵי צָפוֹן כֻּלָּם וְכָל צִדֹנִי אֲשֶׁר יָרְדוּ אֶת חֲלָלִים בְּחִתִּיתָם מִגְבוּרָתָם בּוֹשִׁים וַיִּשְׁכְּבוּ עֲרֵלִים אֶת חַלְלֵי חֶרֶב וַיִּשְׂאוּ כְלִמָּתָם אֶת יוֹרְדֵי בוֹר:

[הושע פרק-ה-א]  שִׁמְעוּ זֹאת הַכֹּהֲנִים וְהַקְשִׁיבוּ בֵּית יִשְׂרָאֵל וּבֵית הַמֶּלֶךְ הַאֲזִינוּ כִּי לָכֶם הַמִּשְׁפָּט כִּי פַח הֱיִיתֶם לְמִצְפָּה וְרֶשֶׁת פְּרוּשָׂה עַל תָּבוֹר:

[יואל פרק-א-י]  שֻׁדַּד שָׂדֶה אָבְלָה אֲדָמָה כִּי שֻׁדַּד דָּגָן הוֹבִישׁ תִּירוֹשׁ אֻמְלַל יִצְהָר:

[עמוס פרק-ג-א]  שִׁמְעוּ אֶת הַדָּבָר הַזֶּה אֲשֶׁר דִּבֶּר יְהוָה עֲלֵיכֶם בְּנֵי יִשְׂרָאֵל עַל כָּל הַמִּשְׁפָּחָה אֲשֶׁר הֶעֱלֵיתִי מֵאֶרֶץ מִצְרַיִם לֵאמֹר:

[צפניה פרק-ג-ג]  שָׂרֶיהָ בְקִרְבָּהּ אֲרָיוֹת שֹׁאֲגִים שֹׁפְטֶיהָ זְאֵבֵי עֶרֶב לֹא גָרְמוּ לַבֹּקֶר:

[תהילים פרק-עט-ג]  שָׁפְכוּ דָמָם כַּמַּיִם סְבִיבוֹת יְרוּשָׁלִָם וְאֵין קוֹבֵר:

[תהילים פרק-קכז-א]  שִׁיר הַמַּעֲלוֹת לִשְׁלֹמֹה אִם יְהוָה לֹא יִבְנֶה בַיִת שָׁוְא עָמְלוּ בוֹנָיו בּוֹ אִם יְהוָה לֹא יִשְׁמָר עִיר שָׁוְא שָׁקַד שׁוֹמֵר:

[תהילים פרק-קכט-ז]  שֶׁלֹּא מִלֵּא כַפּוֹ קוֹצֵר וְחִצְנוֹ מְעַמֵּר:

[תהילים פרק-קמד-ז]  שְׁלַח יָדֶיךָ מִמָּרוֹם פְּצֵנִי וְהַצִּילֵנִי מִמַּיִם רַבִּים מִיַּד בְּנֵי נֵכָר:

[משלי פרק-יב-יט]  שְׂפַת אֱמֶת תִּכּוֹן לָעַד וְעַד אַרְגִּיעָה לְשׁוֹן שָׁקֶר:

[משלי פרק-כה-ג]  שָׁמַיִם לָרוּם וָאָרֶץ לָעֹמֶק וְלֵב מְלָכִים אֵין חֵקֶר:

[שיר השירים פרק-ו-ח]  שִׁשִּׁים הֵמָּה מְּלָכוֹת וּשְׁמֹנִים פִּילַגְשִׁים וַעֲלָמוֹת אֵין מִסְפָּר:

[קהלת פרק-יא-ד]  שֹׁמֵר רוּחַ לֹא יִזְרָע וְרֹאֶה בֶעָבִים לֹא יִקְצוֹר:

[דברי הימים א פרק-ב-נא]  שַׂלְמָא אֲבִי בֵית לָחֶם חָרֵף אֲבִי בֵית גָּדֵר:


ש-ש


[שמות פרק-כט-ל]  שִׁבְעַת יָמִים יִלְבָּשָׁם הַכֹּהֵן תַּחְתָּיו מִבָּנָיו אֲשֶׁר יָבֹא אֶל אֹהֶל מוֹעֵד לְשָׁרֵת בַּקֹּדֶשׁ:

[שמות פרק-כט-לז]  שִׁבְעַת יָמִים תְּכַפֵּר עַל הַמִּזְבֵּחַ וְקִדַּשְׁתָּ אֹתוֹ וְהָיָה הַמִּזְבֵּחַ קֹדֶשׁ קָדָשִׁים כָּל הַנֹּגֵעַ בַּמִּזְבֵּחַ יִקְדָּשׁ:

[במדבר פרק-ז-פה]  שְׁלֹשִׁים וּמֵאָה הַקְּעָרָה הָאַחַת כֶּסֶף וְשִׁבְעִים הַמִּזְרָק הָאֶחָד כֹּל כֶּסֶף הַכֵּלִים אַלְפַּיִם וְאַרְבַּע מֵאוֹת בְּשֶׁקֶל הַקֹּדֶשׁ:

[ישעיה פרק-ג-ג]  שַׂר חֲמִשִּׁים וּנְשׂוּא פָנִים וְיוֹעֵץ וַחֲכַם חֲרָשִׁים וּנְבוֹן לָחַשׁ:

[משלי פרק-כז-ט]  שֶׁמֶן וּקְטֹרֶת יְשַׂמַּח לֵב וּמֶתֶק רֵעֵהוּ מֵעֲצַת נָפֶשׁ:

[שמות פרק-לא-טו]  שֵׁשֶׁת יָמִים יֵעָשֶׂה מְלָאכָה וּבַיּוֹם הַשְּׁבִיעִי שַׁבַּת שַׁבָּתוֹן קֹדֶשׁ לַיהוָה כָּל הָעֹשֶׂה מְלָאכָה בְּיוֹם הַשַּׁבָּת מוֹת יוּמָת:

[שמות פרק-לד-כא]  שֵׁשֶׁת יָמִים תַּעֲבֹד וּבַיּוֹם הַשְּׁבִיעִי תִּשְׁבֹּת בֶּחָרִישׁ וּבַקָּצִיר תִּשְׁבֹּת:

[שמות פרק-לה-ב]  שֵׁשֶׁת יָמִים תֵּעָשֶׂה מְלָאכָה וּבַיּוֹם הַשְּׁבִיעִי יִהְיֶה לָכֶם קֹדֶשׁ שַׁבַּת שַׁבָּתוֹן לַיהוָה כָּל הָעֹשֶׂה בוֹ מְלָאכָה יוּמָת:

[במדבר פרק-ז-פב]  שְׂעִיר עִזִּים אֶחָד לְחַטָּאת:

[דברים פרק-טז-ט]  שִׁבְעָה שָׁבֻעֹת תִּסְפָּר לָךְ מֵהָחֵל חֶרְמֵשׁ בַּקָּמָה תָּחֵל לִסְפֹּר שִׁבְעָה שָׁבֻעוֹת:

[מלכים א פרק-ז-יז]  שְׂבָכִים מַעֲשֵׂה שְׂבָכָה גְּדִלִים מַעֲשֵׂה שַׁרְשְׁרוֹת לַכֹּתָרֹת אֲשֶׁר עַל רֹאשׁ הָעַמּוּדִים שִׁבְעָה לַכֹּתֶרֶת הָאֶחָת וְשִׁבְעָה לַכֹּתֶרֶת הַשֵּׁנִית:

[מלכים א פרק-י-יט]  שֵׁשׁ מַעֲלוֹת לַכִּסֵּה וְרֹאשׁ עָגֹל לַכִּסֵּה מֵאַחֲרָיו וְיָדֹת מִזֶּה וּמִזֶּה אֶל מְקוֹם הַשָּׁבֶת וּשְׁנַיִם אֲרָיוֹת עֹמְדִים אֵצֶל הַיָּדוֹת:

[ישעיה פרק-מז-ה]  שְׁבִי דוּמָם וּבֹאִי בַחֹשֶׁךְ בַּת כַּשְׂדִּים כִּי לֹא תוֹסִיפִי יִקְרְאוּ לָךְ גְּבֶרֶת מַמְלָכוֹת:

[ישעיה פרק-נא-ו]  שְׂאוּ לַשָּׁמַיִם עֵינֵיכֶם וְהַבִּיטוּ אֶל הָאָרֶץ מִתַּחַת כִּי שָׁמַיִם כֶּעָשָׁן נִמְלָחוּ וְהָאָרֶץ כַּבֶּגֶד תִּבְלֶה וְיֹשְׁבֶיהָ כְּמוֹ כֵן יְמוּתוּן וִישׁוּעָתִי לְעוֹלָם תִּהְיֶה וְצִדְקָתִי לֹא תֵחָת:

[עמוס פרק-ג-יג]  שִׁמְעוּ וְהָעִידוּ בְּבֵית יַעֲקֹב נְאֻם אֲדֹנָי יְהוִה אֱלֹהֵי הַצְּבָאוֹת:

[תהילים פרק-עד-ד]  שָׁאֲגוּ צֹרְרֶיךָ בְּקֶרֶב מוֹעֲדֶךָ שָׂמוּ אוֹתֹתָם אֹתוֹת:

[תהילים פרק-פח-ז]  שַׁתַּנִי בְּבוֹר תַּחְתִּיּוֹת בְּמַחֲשַׁכִּים בִּמְצֹלוֹת:

[תהילים פרק-קלד-א]  שִׁיר הַמַּעֲלוֹת הִנֵּה בָּרֲכוּ אֶת יְהוָה כָּל עַבְדֵי יְהוָה הָעֹמְדִים בְּבֵית יְהוָה בַּלֵּילוֹת:

[משלי פרק-ט-ג]  שָׁלְחָה נַעֲרֹתֶיהָ תִקְרָא עַל גַּפֵּי מְרֹמֵי קָרֶת:

[משלי פרק-י-לב]  שִׂפְתֵי צַדִּיק יֵדְעוּן רָצוֹן וּפִי רְשָׁעִים תַּהְפֻּכוֹת:

[משלי פרק-יט-טז]  שֹׁמֵר מִצְוָה שֹׁמֵר נַפְשׁוֹ בּוֹזֵה דְרָכָיו (יומת) יָמוּת:

[משלי פרק-ל-ז]  שְׁתַּיִם שָׁאַלְתִּי מֵאִתָּךְ אַל תִּמְנַע מִמֶּנִּי בְּטֶרֶם אָמוּת:

[משלי פרק-ל-כט]  שְׁלֹשָׁה הֵמָּה מֵיטִיבֵי צָעַד וְאַרְבָּעָה מֵיטִבֵי לָכֶת:

[איכה פרק-ב-כא]  שָׁכְבוּ לָאָרֶץ חוּצוֹת נַעַר וְזָקֵן בְּתוּלֹתַי וּבַחוּרַי נָפְלוּ בֶחָרֶב הָרַגְתָּ בְּיוֹם אַפֶּךָ טָבַחְתָּ לֹא חָמָלְתָּ:
