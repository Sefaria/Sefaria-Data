פרשת עקב


{יב} וְהָיָה עֵקֶב תִּשְׁמְעוּן אֵת הַמִּשְׁפָּטִים הָאֵלֶּה וּשְׁמַרְתֶּם וַֽעֲשִׂיתֶם אֹתָם וְשָׁמַר יְהֹוָה אֱלֹהֶיךָ לְךָ אֶֽת הַבְּרִית וְאֶת הַחֶסֶד אֲשֶׁר נִשְׁבַּע לַֽאֲבֹתֶֽיךָ:  מ  וְהָיָה עֵקֶב תִּשְׁמְעוּן אֵת הַמִּשְׁפָּטִים הָאֵלֶּה וּשְׁמַרְתֶּם וַֽעֲשִׂיתֶם אֹתָם וְשָׁמַר יְהֹוָה אֱלֹהֶיךָ לְךָ אֶֽת הַבְּרִית וְאֶת הַחֶסֶד אֲשֶׁר נִשְׁבַּע לַֽאֲבֹתֶֽיךָ:  ת  וִיהֵי חֳלַף דִי תְקַבְּלוּן יָת דִינַיָא הָאִלֵין וְתִטְרוּן וְתַעְבְּדוּן יָתְהוֹן וְיִטַר יְיָ אֱלָהָךְ לָךְ יָת קְיָמָא וְיָת חִסְדָא דִי קַיִים לַאֲבָהָתָךְ:

{יג} וַֽאֲהֵבְךָ וּבֵֽרַכְךָ וְהִרְבֶּךָ וּבֵרַךְ פְּרִֽי בִטְנְךָ וּפְרִֽי אַדְמָתֶךָ דְּגָנְךָ וְתִֽירשְׁךָ וְיִצְהָרֶךָ שְׁגַר אֲלָפֶיךָ וְעַשְׁתְּרֹת צֹאנֶךָ עַל הָֽאֲדָמָה אֲשֶׁר נִשְׁבַּע לַֽאֲבֹתֶיךָ לָתֶת לָֽךְ:  מ  וַֽאֲהֵבְךָ וּבֵֽרַכְךָ וְהִרְבֶּךָ וּבֵרַךְ פְּרִֽי בִטְנְךָ וּפְרִֽי אַדְמָתֶךָ דְּגָנְךָ וְתִֽירשְׁךָ וְיִצְהָרֶךָ שְׁגַר אֲלָפֶיךָ וְעַשְׁתְּרֹת צֹאנֶךָ עַל הָֽאֲדָמָה אֲשֶׁר נִשְׁבַּע לַֽאֲבֹתֶיךָ לָתֶת לָֽךְ:  ת  וְיִרְחֲמִנָךְ וִיבָרֵכִנָךְ וְיַסְגִינָךְ וִיבָרֵךְ וַלְדָא דִמְעָךְ וְאִבָּא דְאַרְעָךְ עִבוּרָךְ וְחַמְרָךְ וּמִשְׁחָךְ בַּקְרֵי תוֹרָיךְ וְעֶדְרֵי עָנָךְ עַל אַרְעָא דִי קַיִים לַאֲבָהָתָךְ לְמִתַּן לָךְ:

{יד} בָּרוּךְ תִּֽהְיֶה מִכָּל הָֽעַמִּים לֹא יִֽהְיֶה בְךָ עָקָר וַֽעֲקָרָה וּבִבְהֶמְתֶּֽךָ:  מ  בָּרוּךְ תִּֽהְיֶה מִכָּל הָֽעַמִּים לֹא יִֽהְיֶה בְךָ עָקָר וַֽעֲקָרָה וּבִבְהֶמְתֶּֽךָ:  ת  בְּרִיךְ תְּהֵי מִכָּל עַמְמַיָא לָא יְהֵי בָךְ עֲקַר וְעַקְרָא וּבִבְעִירָךְ:

{טו} וְהֵסִיר יְהֹוָה מִמְּךָ כָּל חֹלִי וְכָל מַדְוֵי מִצְרַיִם הָֽרָעִים אֲשֶׁר יָדַעְתָּ לֹא יְשִׂימָם בָּךְ וּנְתָנָם בְּכָל שֹֽׂנְאֶֽיךָ:  מ  וְהֵסִיר יְהֹוָה מִמְּךָ כָּל חֹלִי וְכָל מַדְוֵי מִצְרַיִם הָֽרָעִים אֲשֶׁר יָדַעְתָּ לֹא יְשִׂימָם בָּךְ וּנְתָנָם בְּכָל שֹֽׂנְאֶֽיךָ:  ת  וְיֶעְדֵי יְיָ מִנָךְ כָּל מַרְעִין וְכָל מַכְתְּשֵׁי מִצְרַיִם בִּישַׁיָא דִי יְדַעַתְּ לָא יְשַׁוִּנוּן בָּךְ וְיִתְּנִנוּן בְּכָל סָנְאָךְ:

{טז} וְאָֽכַלְתָּ אֶת כָּל הָֽעַמִּים אֲשֶׁר יְהֹוָה אֱלֹהֶיךָ נֹתֵן לָךְ לֹֽא תָחוֹס עֵֽינְךָ עֲלֵיהֶם וְלֹא תַֽעֲבֹד אֶת אֱלֹהֵיהֶם כִּֽי מוֹקֵשׁ הוּא לָֽךְ:  מ  וְאָֽכַלְתָּ אֶת כָּל הָֽעַמִּים אֲשֶׁר יְהֹוָה אֱלֹהֶיךָ נֹתֵן לָךְ לֹֽא תָחוֹס עֵֽינְךָ עֲלֵיהֶם וְלֹא תַֽעֲבֹד אֶת אֱלֹהֵיהֶם כִּֽי מוֹקֵשׁ הוּא לָֽךְ:  ת  וּתְגַמַר יָת כָּל עַמְמַיָא דִי יְיָ אֱלָהָךְ יָהֵב לָךְ לָא תְחוּס עֵינָךְ עֲלֵיהוֹן וְלָא תִפְלַח יָת טַּעֲוָתְהוֹן אֲרֵי לְתַקְלָא הוּא לָךְ: (ס)

{יז} כִּי תֹאמַר בִּלְבָבְךָ רַבִּים הַגּוֹיִם הָאֵלֶּה מִמֶּנִּי אֵיכָה אוּכַל לְהֽוֹרִישָֽׁם:  מ  כִּי תֹאמַר בִּלְבָבְךָ רַבִּים הַגּוֹיִם הָאֵלֶּה מִמֶּנִּי אֵיכָה אוּכַל לְהֽוֹרִישָֽׁם:  ת  דִילְמָא תֵימַר בְּלִבָּךְ סַגִיאִין עַמְמַיָא הָאִלֵין מִנִי אֶכְדֵין אִכּוּל לְתָרָכוּתְהוֹן:

{יח} לֹא תִירָא מֵהֶם זָכֹר תִּזְכֹּר אֵת אֲשֶׁר עָשָׂה יְהֹוָה אֱלֹהֶיךָ לְפַרְעֹה וּלְכָל מִצְרָֽיִם:  מ  לֹא תִירָא מֵהֶם זָכֹר תִּזְכֹּר אֵת אֲשֶׁר עָשָׂה יְהֹוָה אֱלֹהֶיךָ לְפַרְעֹה וּלְכָל מִצְרָֽיִם:  ת  לָא תִדְחַל מִנְהוֹן מִדְכַּר תִּדְכַּר יָת דִי עֲבַד יְיָ אֱלָהָךְ לְפַרְעֹה וּלְכָל מִצְרָיִם:

{יט} הַמַּסֹּת הַגְּדֹלֹת אֲשֶׁר רָאוּ עֵינֶיךָ וְהָֽאֹתֹת וְהַמֹּֽפְתִים וְהַיָּד הַֽחֲזָקָה וְהַזְּרֹעַ הַנְּטוּיָה אֲשֶׁר הוֹצִֽאֲךָ יְהֹוָה אֱלֹהֶיךָ כֵּן יַֽעֲשֶׂה יְהֹוָה אֱלֹהֶיךָ לְכָל הָעַמִּים אֲשֶׁר אַתָּה יָרֵא מִפְּנֵיהֶֽם:  מ  הַמַּסֹּת הַגְּדֹלֹת אֲשֶׁר רָאוּ עֵינֶיךָ וְהָֽאֹתֹת וְהַמֹּֽפְתִים וְהַיָּד הַֽחֲזָקָה וְהַזְּרֹעַ הַנְּטוּיָה אֲשֶׁר הוֹצִֽאֲךָ יְהֹוָה אֱלֹהֶיךָ כֵּן יַֽעֲשֶׂה יְהֹוָה אֱלֹהֶיךָ לְכָל הָעַמִּים אֲשֶׁר אַתָּה יָרֵא מִפְּנֵיהֶֽם:  ת  נִסִין רַבְרְבִין דִי חֲזָאָה עֵינָךְ וְאָתַיָא וּמוֹפְתַיָא וִידָא תַקֶפְתָּא וּדְרָעָא מְרָמְמָא דִי אַפְּקָךְ יְיָ אֱלָהָךְ כֵּן יַעְבֵּד יְיָ אֱלָהָךְ לְכָל עַמְמַיָא דִי אַתְּ דָחֵל מִקֳדָמֵיהוֹן:

{כ} וְגַם אֶת הַצִּרְעָה יְשַׁלַּח יְהֹוָה אֱלֹהֶיךָ בָּם עַד אֲבֹד הַנִּשְׁאָרִים וְהַנִּסְתָּרִים מִפָּנֶֽיךָ:  מ  וְגַם אֶת הַצִּרְעָה יְשַׁלַּח יְהֹוָה אֱלֹהֶיךָ בָּם עַד אֲבֹד הַנִּשְׁאָרִים וְהַנִּסְתָּרִים מִפָּנֶֽיךָ:  ת  וְאַף יָת עֲרָעִיתָא יְגָרֵי יְיָ אֱלָהָךְ בְּהוֹן עַד דְיֵבַדוּן דְאִשְׁתָּאָרוּ וּדְאִטַמָרוּ מֵקֳדָמָךְ:

{כא} לֹא תַֽעֲרֹץ מִפְּנֵיהֶם כִּֽי יְהֹוָה אֱלֹהֶיךָ בְּקִרְבֶּךָ אֵל גָּדוֹל וְנוֹרָֽא:  מ  לֹא תַֽעֲרֹץ מִפְּנֵיהֶם כִּֽי יְהֹוָה אֱלֹהֶיךָ בְּקִרְבֶּךָ אֵל גָּדוֹל וְנוֹרָֽא:  ת  לָא תִתְּבַר מִקֳדָמֵיהוֹן אֲרֵי יְיָ אֱלָהָךְ שְׁכִנְתֵּהּ בֵּינָךְ אֱלָהָא רַבָּא וּדְחִילָא:

{כב} וְנָשַׁל יְהֹוָה אֱלֹהֶיךָ אֶת הַגּוֹיִם הָאֵל מִפָּנֶיךָ מְעַט מְעָט לֹא תוּכַל כַּלֹּתָם מַהֵר פֶּן תִּרְבֶּה עָלֶיךָ חַיַּת הַשָּׂדֶֽה:  מ  וְנָשַׁל יְהֹוָה אֱלֹהֶיךָ אֶת הַגּוֹיִם הָאֵל מִפָּנֶיךָ מְעַט מְעָט לֹא תוּכַל כַּלֹּתָם מַהֵר פֶּן תִּרְבֶּה עָלֶיךָ חַיַּת הַשָּׂדֶֽה:  ת  וִיתָרַךְ יְיָ אֱלָהָךְ יָת עַמְמַיָא הָאִלֵין מִקֳדָמָךְ זְעֵר זְעֵר לָא תִכּוּל לְשֵׁצָי וּתְהוֹן בִּפְרִי עַ דִי לְמָא תִסְגֵּי עֲלָךְ חֵיוַת בָּרָא:

{כג} וּנְתָנָם יְהֹוָה אֱלֹהֶיךָ לְפָנֶיךָ וְהָמָם מְהוּמָה גְדֹלָה עַד הִשָּֽׁמְדָֽם:  מ  וּנְתָנָם יְהֹוָה אֱלֹהֶיךָ לְפָנֶיךָ וְהָמָם מְהוּמָה גְדֹלָה עַד הִשָּֽׁמְדָֽם:  ת  וְיִמְסְרִנוּן יְיָ אֱלָהָךְ קֳדָמָךְ וִישַׁגֵשִׁנוּן שִׁגוּשׁ רַב עַד דְיִשְׁתֵּצוּן:

{כד} וְנָתַן מַלְכֵיהֶם בְּיָדֶךָ וְהַֽאֲבַדְתָּ אֶת שְׁמָם מִתַּחַת הַשָּׁמָיִם לֹֽא יִתְיַצֵּב אִישׁ בְּפָנֶיךָ עַד הִשְׁמִֽדְךָ אֹתָֽם:  מ  וְנָתַן מַלְכֵיהֶם בְּיָדֶךָ וְהַֽאֲבַדְתָּ אֶת שְׁמָם מִתַּחַת הַשָּׁמָיִם לֹֽא יִתְיַצֵּב אִישׁ בְּפָנֶיךָ עַד הִשְׁמִֽדְךָ אֹתָֽם:  ת  וְיִמְסַר מַלְכֵיהוֹן בִּידָךְ וְתוֹבַד יָת שִׁמְהוֹן מִתְּחוֹת שְׁמַיָא לָא יִתְעַתַּד אֱנַשׁ קֳדָמָךְ עַד דִתְשֵׁיצֵי יָתְהוֹן:

{כה} פְּסִילֵי אֱלֹֽהֵיהֶם תִּשְׂרְפוּן בָּאֵשׁ לֹֽא תַחְמֹד כֶּסֶף וְזָהָב עֲלֵיהֶם וְלָֽקַחְתָּ לָךְ פֶּן תִּוָּקֵשׁ בּוֹ כִּי תֽוֹעֲבַת יְהֹוָה אֱלֹהֶיךָ הֽוּא:  מ  פְּסִילֵי אֱלֹֽהֵיהֶם תִּשְׂרְפוּן בָּאֵשׁ לֹֽא תַחְמֹד כֶּסֶף וְזָהָב עֲלֵיהֶם וְלָֽקַחְתָּ לָךְ פֶּן תִּוָּקֵשׁ בּוֹ כִּי תֽוֹעֲבַת יְהֹוָה אֱלֹהֶיךָ הֽוּא:  ת  צַלְמֵי טַעֲוָתְהוֹן תּוֹקְדוּן בְּנוּרָא לָא תַחְמֵד כַּסְפָּא וְדַהֲבָא דִי עֲלֵיהוֹן וְתִסַב לָךְ דִילְמָא תִּתָּקַל בֵּהּ אֲרֵי מְרָחָקָא דַיְיָ אֱלָהָךְ הוּא:

{כו} וְלֹֽא תָבִיא תֽוֹעֵבָה אֶל בֵּיתֶךָ וְהָיִיתָ חֵרֶם כָּמֹהוּ שַׁקֵּץ תְּשַׁקְּצֶנּוּ וְתַעֵב תְּֽתַעֲבֶנּוּ כִּי חֵרֶם הֽוּא:  מ  וְלֹֽא תָבִיא תֽוֹעֵבָה אֶל בֵּיתֶךָ וְהָיִיתָ חֵרֶם כָּמֹהוּ שַׁקֵּץ תְּשַׁקְּצֶנּוּ וְתַעֵב תְּֽתַעֲבֶנּוּ כִּי חֵרֶם הֽוּא:  ת  וְלָא תָעֵל דִמְרָחָק לְבֵיתָךְ וּתְהֵי חֶרְמָא כְּוָתֵהּ שַׁקָצָא תְשַׁקְצִנֵהּ וְרַחָקָא תְּרַחֲקִנֵהּ אֲרֵי חֶרְמָה הוּא: (פ)




דברים פרק-ח

{א} כָּל הַמִּצְוָה אֲשֶׁר אָֽנֹכִי מְצַוְּךָ הַיּוֹם תִּשְׁמְרוּן לַֽעֲשׂוֹת לְמַעַן תִּֽחְיוּן וּרְבִיתֶם וּבָאתֶם וִֽירִשְׁתֶּם אֶת הָאָרֶץ אֲשֶׁר נִשְׁבַּע יְהֹוָה לַֽאֲבֹֽתֵיכֶֽם:  מ  כָּל הַמִּצְוָה אֲשֶׁר אָֽנֹכִי מְצַוְּךָ הַיּוֹם תִּשְׁמְרוּן לַֽעֲשׂוֹת לְמַעַן תִּֽחְיוּן וּרְבִיתֶם וּבָאתֶם וִֽירִשְׁתֶּם אֶת הָאָרֶץ אֲשֶׁר נִשְׁבַּע יְהֹוָה לַֽאֲבֹֽתֵיכֶֽם:  ת  כָּל תַּפְקֶדְתָּא דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין תִּטְרוּן לְמֶעְבָּד בְּדִיל דְתֵיחוּן וְתִסְגוּן וְתֵיתוּן וְתֵירְתוּן יָת אַרְעָא דִי קַיִים יְיָ לַאֲבָהָתְכוֹן:

{ב} וְזָֽכַרְתָּ אֶת כָּל הַדֶּרֶךְ אֲשֶׁר הוֹלִֽיכְךָ יְהֹוָה אֱלֹהֶיךָ זֶה אַרְבָּעִים שָׁנָה בַּמִּדְבָּר לְמַעַן עַנֹּֽתְךָ לְנַסֹּֽתְךָ לָדַעַת אֶת אֲשֶׁר בִּלְבָֽבְךָ הֲתִשְׁמֹר (מצותו) מִצְוֹתָיו אִם לֹֽא:  מ  וְזָֽכַרְתָּ אֶת כָּל הַדֶּרֶךְ אֲשֶׁר הוֹלִֽיכְךָ יְהֹוָה אֱלֹהֶיךָ זֶה אַרְבָּעִים שָׁנָה בַּמִּדְבָּר לְמַעַן עַנֹּֽתְךָ לְנַסֹּֽתְךָ לָדַעַת אֶת אֲשֶׁר בִּלְבָֽבְךָ הֲתִשְׁמֹר (מצותו) מִצְוֹתָיו אִם לֹֽא:  ת  וְתִדְכַּר יָת כָּל אָרְחָא דִי דַבְּרָךְ יְיָ אֱלָהָךְ דְנַן אַרְבְּעִין שְׁנִין בְּמַדְבְּרָא בְּדִיל לְעַנָיוּתָךְ לְנַסָיוּתָךְ לְמִדַע יָת דִי בְלִבָּךְ הֲתִטַר פִּקוּדוֹהִי אִם לָא:

{ג} וַֽיְעַנְּךָ וַיַּרְעִבֶךָ וַיַּֽאֲכִֽלְךָ אֶת הַמָּן אֲשֶׁר לֹֽא יָדַעְתָּ וְלֹא יָֽדְעוּן אֲבֹתֶיךָ לְמַעַן הוֹדִֽיעֲךָ כִּי לֹא עַל הַלֶּחֶם לְבַדּוֹ יִחְיֶה הָֽאָדָם כִּי עַל כָּל מוֹצָא פִֽי יְהֹוָה יִחְיֶה הָֽאָדָֽם:  מ  וַֽיְעַנְּךָ וַיַּרְעִבֶךָ וַיַּֽאֲכִֽלְךָ אֶת הַמָּן אֲשֶׁר לֹֽא יָדַעְתָּ וְלֹא יָֽדְעוּן אֲבֹתֶיךָ לְמַעַן הוֹדִֽיעֲךָ כִּי לֹא עַל הַלֶּחֶם לְבַדּוֹ יִחְיֶה הָֽאָדָם כִּי עַל כָּל מוֹצָא פִֽי יְהֹוָה יִחְיֶה הָֽאָדָֽם:  ת  וְעַנְיָךְ וְאַכְפָּנָךְ וְאוֹכָלָךְ יָת מַנָא דִי לָא יְדַעְתְּ וְלָא יָדְעוּן אֲבָהָתָיךְ בְּדִיל לְאוֹדָעוּתָךְ אֲרֵי לָא עַל לַחְמָא בִלְחוֹדוֹהִי מִתְקַיַם אֱנָשָׁא אֲרֵי עַל כָּל אַפָּקוּת מֵימְרָא דַיְיָ יִתְקַיַם אֱנָשָׁא:

{ד} שִׂמְלָתְךָ לֹא בָֽלְתָה מֵֽעָלֶיךָ וְרַגְלְךָ לֹא בָצֵקָה זֶה אַרְבָּעִים שָׁנָֽה:  מ  שִׂמְלָתְךָ לֹא בָֽלְתָה מֵֽעָלֶיךָ וְרַגְלְךָ לֹא בָצֵקָה זֶה אַרְבָּעִים שָׁנָֽה:  ת  כְּסוּתָךְ לָא אִבְלִיאַת מִנָךְ וּמְסָנָךְ לָא יְחֵפוּ דְנַן אַרְבְּעִין שְׁנִין:

{ה} וְיָֽדַעְתָּ עִם לְבָבֶךָ כִּי כַּֽאֲשֶׁר יְיַסֵּר אִישׁ אֶת בְּנוֹ יְהֹוָה אֱלֹהֶיךָ מְיַסְּרֶֽךָּ:  מ  וְיָֽדַעְתָּ עִם לְבָבֶךָ כִּי כַּֽאֲשֶׁר יְיַסֵּר אִישׁ אֶת בְּנוֹ יְהֹוָה אֱלֹהֶיךָ מְיַסְּרֶֽךָּ:  ת  וְתִדַע עִם לִבָּךְ אֲרֵי כְּמָא דִי מְאַלֵף גַבְרָא יָת בְּרֵהּ יְיָ אֱלָהָךְ מְאַלֵף לָךְ:

{ו} וְשָׁמַרְתָּ אֶת מִצְוֹת יְהֹוָה אֱלֹהֶיךָ לָלֶכֶת בִּדְרָכָיו וּלְיִרְאָה אֹתֽוֹ:  מ  וְשָׁמַרְתָּ אֶת מִצְוֹת יְהֹוָה אֱלֹהֶיךָ לָלֶכֶת בִּדְרָכָיו וּלְיִרְאָה אֹתֽוֹ:  ת  וְתִטַר יָת פִּקוּדַיָא דַיְיָ אֱלָהָךְ לִמְהַךְ בְּאָרְחַן דְתַקְנָן קֳדָמוֹהִי וּלְמִדְחַל יָתֵהּ:

{ז} כִּי יְהֹוָה אֱלֹהֶיךָ מְבִֽיאֲךָ אֶל אֶרֶץ טוֹבָה אֶרֶץ נַחֲלֵי מָיִם עֲיָנֹת וּתְהֹמֹת יֹֽצְאִים בַּבִּקְעָה וּבָהָֽר:  מ  כִּי יְהֹוָה אֱלֹהֶיךָ מְבִֽיאֲךָ אֶל אֶרֶץ טוֹבָה אֶרֶץ נַחֲלֵי מָיִם עֲיָנֹת וּתְהֹמֹת יֹֽצְאִים בַּבִּקְעָה וּבָהָֽר:  ת  אֲרֵי יְיָ אֱלָהָךְ מַעֲלָךְ לְאַרְעָא טַבְתָא אַרְעָא נָגְדָא נַחֲלִין דְמַיִין מַבּוּעֵי עֵינַוָן וּתְהוֹמִין נָפְקִין בְּבִקְעָן וּבְטוּרִין:

{ח} אֶרֶץ חִטָּה וּשְׂעֹרָה וְגֶפֶן וּתְאֵנָה וְרִמּוֹן אֶֽרֶץ זֵית שֶׁמֶן וּדְבָֽשׁ:  מ  אֶרֶץ חִטָּה וּשְׂעֹרָה וְגֶפֶן וּתְאֵנָה וְרִמּוֹן אֶֽרֶץ זֵית שֶׁמֶן וּדְבָֽשׁ:  ת  אַרְעָא חִטִּין וְסַעֲרִין וְגוּפְנִין וּתְאֵנִין וְרִמוֹנִין אַרְעָא דְזֵיתָהָא עָבְדִין מִשְׁחָא וְהִיא עָבְדָא דְבָשׁ:

{ט} אֶרֶץ אֲשֶׁר לֹא בְמִסְכֵּנֻת תֹּֽאכַל בָּהּ לֶחֶם לֹֽא תֶחְסַר כֹּל בָּהּ אֶרֶץ אֲשֶׁר אֲבָנֶיהָ בַרְזֶל וּמֵֽהֲרָרֶיהָ תַּחְצֹב נְחֽשֶׁת:  מ  אֶרֶץ אֲשֶׁר לֹא בְמִסְכֵּנֻת תֹּֽאכַל בָּהּ לֶחֶם לֹֽא תֶחְסַר כֹּל בָּהּ אֶרֶץ אֲשֶׁר אֲבָנֶיהָ בַרְזֶל וּמֵֽהֲרָרֶיהָ תַּחְצֹב נְחֽשֶׁת:  ת  אַרְעָא דִי לָא בְעִצוּרִין (נ''א בְמִסְכֵּנוּ) תֵּיכוּל בַּהּ לַחֲמָא לָא תֶחְסַר כָּל מִדַעַם בַּהּ אַרְעָא דִי אַבְנָהָא פַרְזְלָא וּמִטוּרָהָא תִּפְסוּל נְחָשָׁא:

{י} וְאָֽכַלְתָּ וְשָׂבָעְתָּ וּבֵֽרַכְתָּ אֶת יְהֹוָה אֱלֹהֶיךָ עַל הָאָרֶץ הַטֹּבָה אֲשֶׁר נָֽתַן לָֽךְ:  מ  וְאָֽכַלְתָּ וְשָׂבָעְתָּ וּבֵֽרַכְתָּ אֶת יְהֹוָה אֱלֹהֶיךָ עַל הָאָרֶץ הַטֹּבָה אֲשֶׁר נָֽתַן לָֽךְ:  ת  וְתֵיכוּל וְתִשְׂבַּע וּתְבָרַךְ יָת יְיָ אֱלָהָךְ עַל אַרְעָא טַבְתָא דִיהַב לָךְ:

{יא}  שני  הִשָּׁמֶר לְךָ פֶּן תִּשְׁכַּח אֶת יְהֹוָה אֱלֹהֶיךָ לְבִלְתִּי שְׁמֹר מִצְוֹתָיו וּמִשְׁפָּטָיו וְחֻקֹּתָיו אֲשֶׁר אָֽנֹכִי מְצַוְּךָ הַיּֽוֹם:  מ  הִשָּׁמֶר לְךָ פֶּן תִּשְׁכַּח אֶת יְהֹוָה אֱלֹהֶיךָ לְבִלְתִּי שְׁמֹר מִצְוֹתָיו וּמִשְׁפָּטָיו וְחֻקֹּתָיו אֲשֶׁר אָֽנֹכִי מְצַוְּךָ הַיּֽוֹם:  ת  אִסְתַּמַר לָךְ דִילְמָא תִתְנְשֵׁי יָת דַחַלְתָּא דַיְיָ אֱלָהָךְ בְּדִיל דְלָא לְמִטַר פִּקּוֹדוֹהִי וְדִינוֹהִי וּקְיָמוֹהִי דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין:

{יב} פֶּן תֹּאכַל וְשָׂבָעְתָּ וּבָתִּים טֹבִים תִּבְנֶה וְיָשָֽׁבְתָּ:  מ  פֶּן תֹּאכַל וְשָׂבָעְתָּ וּבָתִּים טֹבִים תִּבְנֶה וְיָשָֽׁבְתָּ:  ת  דִילְמָא תֵיכוּל וְתִשְׂבַּע וּבָתִּין שַׁפִּירִין תִּבְנֵי וְתֵיתֵב:

{יג} וּבְקָֽרְךָ וְצֹֽאנְךָ יִרְבְּיֻן וְכֶסֶף וְזָהָב יִרְבֶּה לָּךְ וְכֹל אֲשֶׁר לְךָ יִרְבֶּֽה:  מ  וּבְקָֽרְךָ וְצֹֽאנְךָ יִרְבְּיֻן וְכֶסֶף וְזָהָב יִרְבֶּה לָּךְ וְכֹל אֲשֶׁר לְךָ יִרְבֶּֽה:  ת  וְתוֹרָךְ וְעָנָךְ יִסְגוּן וְכַסְפָּא וְדַהֲבָא יִסְגֵא לָךְ וְכֹל דִי לָךְ יִסְגֵא:

{יד} וְרָם לְבָבֶךָ וְשָֽׁכַחְתָּ אֶת יְהֹוָה אֱלֹהֶיךָ הַמּוֹצִֽיאֲךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִֽים:  מ  וְרָם לְבָבֶךָ וְשָֽׁכַחְתָּ אֶת יְהֹוָה אֱלֹהֶיךָ הַמּוֹצִֽיאֲךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִֽים:  ת  וִירִים לִבָּךְ וְתִנְשֵׁי יָת דַחַלְתָּא דַיְיָ אֱלָהָךְ דִי אַפְּקָךְ מֵאַרְעָא דְמִצְרַיִם מִבֵּית עַבְדוּתָא:

{טו} הַמּוֹלִיכֲךָ בַּמִּדְבָּר הַגָּדֹל וְהַנּוֹרָא נָחָשׁ שָׂרָף וְעַקְרָב וְצִמָּאוֹן אֲשֶׁר אֵין מָיִם הַמּוֹצִיא לְךָ מַיִם מִצּוּר הַֽחַלָּמִֽישׁ:  מ  הַמּוֹלִיכֲךָ בַּמִּדְבָּר הַגָּדֹל וְהַנּוֹרָא נָחָשׁ שָׂרָף וְעַקְרָב וְצִמָּאוֹן אֲשֶׁר אֵין מָיִם הַמּוֹצִיא לְךָ מַיִם מִצּוּר הַֽחַלָּמִֽישׁ:  ת  דְדַבְּרָךְ בְּמַדְבְּרָא רַבָּא וּדְחִילָא אֲתַר חֵיוָן קָלָן וְעַקְרַבִּין וּבֵית צַחֲוָנָא אֲתַר דִי לֵית מַיָא דְאַפֵּק לָךְ מַיָא מִטִּנָרָא תַּקִיפָא:

{טז} הַמַּֽאֲכִלְךָ מָן בַּמִּדְבָּר אֲשֶׁר לֹא יָֽדְעוּן אֲבֹתֶיךָ לְמַעַן עַנֹּֽתְךָ וּלְמַעַן נַסֹּתֶךָ לְהֵיטִֽבְךָ בְּאַֽחֲרִיתֶֽךָ:  מ  הַמַּֽאֲכִלְךָ מָן בַּמִּדְבָּר אֲשֶׁר לֹא יָֽדְעוּן אֲבֹתֶיךָ לְמַעַן עַנֹּֽתְךָ וּלְמַעַן נַסֹּתֶךָ לְהֵיטִֽבְךָ בְּאַֽחֲרִיתֶֽךָ:  ת  דְאוֹכָלָךְ מַנָא בְּמַדְבְּרָא דִי לָא יָדְעוּן אֲבָהָתָךְ בְּדִיל לְעַנָיוּתָךְ וּבְדִיל לְנַסָיוּתָךְ לְאוֹטָבָא לָךְ בְּסוֹפָךְ:

{יז} וְאָֽמַרְתָּ בִּלְבָבֶךָ כֹּחִי וְעֹצֶם יָדִי עָשָׂה לִי אֶת הַחַיִל הַזֶּֽה:  מ  וְאָֽמַרְתָּ בִּלְבָבֶךָ כֹּחִי וְעֹצֶם יָדִי עָשָׂה לִי אֶת הַחַיִל הַזֶּֽה:  ת  וְתֵימַר בְּלִבָּךְ חֵילִי וּתְקֵף יְדִי קְנוֹ לִי יָת נִכְסַיָא הָאִלֵין:

{יח} וְזָֽכַרְתָּ אֶת יְהֹוָה אֱלֹהֶיךָ כִּי הוּא הַנֹּתֵן לְךָ כֹּחַ לַֽעֲשׂוֹת חָיִל לְמַעַן הָקִים אֶת בְּרִיתוֹ אֲשֶׁר נִשְׁבַּע לַֽאֲבֹתֶיךָ כַּיּוֹם הַזֶּֽה:  מ  וְזָֽכַרְתָּ אֶת יְהֹוָה אֱלֹהֶיךָ כִּי הוּא הַנֹּתֵן לְךָ כֹּחַ לַֽעֲשׂוֹת חָיִל לְמַעַן הָקִים אֶת בְּרִיתוֹ אֲשֶׁר נִשְׁבַּע לַֽאֲבֹתֶיךָ כַּיּוֹם הַזֶּֽה:  ת  וְתִדְכַּר יָת יְיָ אֱלָהָךְ אֲרֵי הוּא יָהֵב לָךְ עֵצָה לְמִקְנֵי נִכְסִין בְּדִיל לְקַיָמָא יָת קְיָמֵהּ דִי קַיִים לְאֲבָהָתָךְ כְּיוֹמָא הָדֵין: (פ)

{יט} וְהָיָה אִם שָׁכֹחַ תִּשְׁכַּח אֶת יְהֹוָה אֱלֹהֶיךָ וְהָֽלַכְתָּ אַֽחֲרֵי אֱלֹהִים אֲחֵרִים וַֽעֲבַדְתָּם וְהִשְׁתַּֽחֲוִיתָ לָהֶם הַֽעִדֹתִי בָכֶם הַיּוֹם כִּי אָבֹד תֹּֽאבֵדֽוּן:  מ  וְהָיָה אִם שָׁכֹחַ תִּשְׁכַּח אֶת יְהֹוָה אֱלֹהֶיךָ וְהָֽלַכְתָּ אַֽחֲרֵי אֱלֹהִים אֲחֵרִים וַֽעֲבַדְתָּם וְהִשְׁתַּֽחֲוִיתָ לָהֶם הַֽעִדֹתִי בָכֶם הַיּוֹם כִּי אָבֹד תֹּֽאבֵדֽוּן:  ת  וִיהֵי אִם מִנְשָׁאָה תִנְשֵׁי יָת דַחַלְתָּא דַיְיָ אֱלָהָךְ וּתְהַךְ בָּתַר טַעְוַת עַמְמַיָא וְתִפְלְחִנוּן וְתִסְגוּד לְהוֹן אַסְהֵדִית בְּכוֹן יוֹמָא דֵין אֲרֵי מֵיבַד תֵּיבְדוּן:

{כ} כַּגּוֹיִם אֲשֶׁר יְהֹוָה מַֽאֲבִיד מִפְּנֵיכֶם כֵּן תֹּֽאבֵדוּן עֵקֶב לֹא תִשְׁמְעוּן בְּקוֹל יְהֹוָה אֱלֹֽהֵיכֶֽם:  מ  כַּגּוֹיִם אֲשֶׁר יְהֹוָה מַֽאֲבִיד מִפְּנֵיכֶם כֵּן תֹּֽאבֵדוּן עֵקֶב לֹא תִשְׁמְעוּן בְּקוֹל יְהֹוָה אֱלֹֽהֵיכֶֽם:  ת  כְּעַמְמַיָא דִי יְיָ מֹאבִיד מִקֳדָמֵיכוֹן כֵּן תֵּיבְדוּן חֳלַף דְלָא קַבֶּלְתּוּן בְּמֵימְרָא דַיְיָ אֱלָהֲכוֹן: (פ)




דברים פרק-ט

{א} שְׁמַע יִשְׂרָאֵל אַתָּה עֹבֵר הַיּוֹם אֶת הַיַּרְדֵּן לָבֹא לָרֶשֶׁת גּוֹיִם גְּדֹלִים וַֽעֲצֻמִים מִמֶּךָּ עָרִים גְּדֹלֹת וּבְצֻרֹת בַּשָּׁמָֽיִם:  מ  שְׁמַע יִשְׂרָאֵל אַתָּה עֹבֵר הַיּוֹם אֶת הַיַּרְדֵּן לָבֹא לָרֶשֶׁת גּוֹיִם גְּדֹלִים וַֽעֲצֻמִים מִמֶּךָּ עָרִים גְּדֹלֹת וּבְצֻרֹת בַּשָּׁמָֽיִם:  ת  שְׁמַע יִשְׂרָאֵל אַתְּ עָבֵר יוֹמָא דֵין יָת יַרְדְנָא לְמֵיעַל לְמֵירַת עַמְמִין רַבְרְבִין וְתַקִיפִין מִנָךְ קִרְוִין רַבְרְבָן וּכְרִיכָן עַד צֵית שְׁמַיָא:

{ב} עַם גָּדוֹל וָרָם בְּנֵי עֲנָקִים אֲשֶׁר אַתָּה יָדַעְתָּ וְאַתָּה שָׁמַעְתָּ מִי יִתְיַצֵּב לִפְנֵי בְּנֵֽי עֲנָֽק:  מ  עַם גָּדוֹל וָרָם בְּנֵי עֲנָקִים אֲשֶׁר אַתָּה יָדַעְתָּ וְאַתָּה שָׁמַעְתָּ מִי יִתְיַצֵּב לִפְנֵי בְּנֵֽי עֲנָֽק:  ת  עַם רַב וְתַקִיף בְּנֵי גִבָּרַיָא דִי אַתְּ יְדַעַתְּ וְאַתְּ שְׁמַעְתְּ מָן יִכּוּל לְמֵיקַם קֳדָם בְּנֵי גִבָּרַיָא:

{ג} וְיָֽדַעְתָּ הַיּוֹם כִּי יְהֹוָה אֱלֹהֶיךָ הֽוּא הָֽעֹבֵר לְפָנֶיךָ אֵשׁ אֹֽכְלָה הוּא יַשְׁמִידֵם וְהוּא יַכְנִיעֵם לְפָנֶיךָ וְהֽוֹרַשְׁתָּם וְהַֽאֲבַדְתָּם מַהֵר כַּֽאֲשֶׁר דִּבֶּר יְהֹוָה לָֽךְ:  מ  וְיָֽדַעְתָּ הַיּוֹם כִּי יְהֹוָה אֱלֹהֶיךָ הֽוּא הָֽעֹבֵר לְפָנֶיךָ אֵשׁ אֹֽכְלָה הוּא יַשְׁמִידֵם וְהוּא יַכְנִיעֵם לְפָנֶיךָ וְהֽוֹרַשְׁתָּם וְהַֽאֲבַדְתָּם מַהֵר כַּֽאֲשֶׁר דִּבֶּר יְהֹוָה לָֽךְ:  ת  וְתִדַע יוֹמָא דֵין אֲרֵי יְיָ אֱלָהָךְ הוּא דְעֲבַר קֳדָמָךְ מֵימְרֵהּ אֶשָׁא אָכְלָא הוּא יְשֵׁצִנוּן וְהוּא יְתָרֵכִנוּן קֳדָמָךְ וּתְתָרֵכִנוּן וְתוֹבְדִנוּן בִּפְרִיעַ כְּמָא דִי מַלִיל יְיָ לָךְ:

{ד}  שלישי  אַל תֹּאמַר בִּלְבָֽבְךָ בַּֽהֲדֹף יְהֹוָה אֱלֹהֶיךָ אֹתָם מִלְּפָנֶיךָ לֵאמֹר בְּצִדְקָתִי הֱבִיאַנִי יְהֹוָה לָרֶשֶׁת אֶת הָאָרֶץ הַזֹּאת וּבְרִשְׁעַת הַגּוֹיִם הָאֵלֶּה יְהֹוָה מֽוֹרִישָׁם מִפָּנֶֽיךָ:  מ  אַל תֹּאמַר בִּלְבָֽבְךָ בַּֽהֲדֹף יְהֹוָה אֱלֹהֶיךָ אֹתָם מִלְּפָנֶיךָ לֵאמֹר בְּצִדְקָתִי הֱבִיאַנִי יְהֹוָה לָרֶשֶׁת אֶת הָאָרֶץ הַזֹּאת וּבְרִשְׁעַת הַגּוֹיִם הָאֵלֶּה יְהֹוָה מֽוֹרִישָׁם מִפָּנֶֽיךָ:  ת  לָא תֵימַר בְּלִבָּךְ בִּדְיִתְּבַר יְיָ אֱלָהָךְ יָתְהוֹן מִקֳדָמָךְ לְמֵימַר בְּזָכוּתִי אַעֲלַנִי יְיָ לְמֵירַת יָת אַרְעָא הָדָא וּבְחוֹבֵי עַמְמַיָא הָאִלֵין יְיָ מְתָרֶכְהוֹן מֵקֳדָמָךְ:

{ה} לֹא בְצִדְקָֽתְךָ וּבְיֹשֶׁר לְבָבְךָ אַתָּה בָא לָרֶשֶׁת אֶת אַרְצָם כִּי בְּרִשְׁעַת הַגּוֹיִם הָאֵלֶּה יְהֹוָה אֱלֹהֶיךָ מֽוֹרִישָׁם מִפָּנֶיךָ וּלְמַעַן הָקִים אֶת הַדָּבָר אֲשֶׁר נִשְׁבַּע יְהֹוָה לַֽאֲבֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּֽלְיַֽעֲקֹֽב:  מ  לֹא בְצִדְקָֽתְךָ וּבְיֹשֶׁר לְבָבְךָ אַתָּה בָא לָרֶשֶׁת אֶת אַרְצָם כִּי בְּרִשְׁעַת הַגּוֹיִם הָאֵלֶּה יְהֹוָה אֱלֹהֶיךָ מֽוֹרִישָׁם מִפָּנֶיךָ וּלְמַעַן הָקִים אֶת הַדָּבָר אֲשֶׁר נִשְׁבַּע יְהֹוָה לַֽאֲבֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּֽלְיַֽעֲקֹֽב:  ת  לָא בְזָכוּתָךְ וּבְקַשִׁיטּוּת לִבָּךְ אַתְּ עָלֵל לְמֵירַת יָת אַרְעֲהוֹן אֲרֵי בְּחוֹבֵי עַמְמַיָא הָאִלֵין יְיָ אֱלָהָךְ מתָרֶכְהוֹן מִקָדָמָךְ וּבְדִיל לַאֲקָמָא יָת פִּתְגָמָא דִי קַיִים יְיָ לַאֲבָהָתָךְ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב:

{ו} וְיָֽדַעְתָּ כִּי לֹא בְצִדְקָֽתְךָ יְהֹוָה אֱלֹהֶיךָ נֹתֵן לְךָ אֶת הָאָרֶץ הַטּוֹבָה הַזֹּאת לְרִשְׁתָּהּ כִּי עַם קְשֵׁה עֹרֶף אָֽתָּה:  מ  וְיָֽדַעְתָּ כִּי לֹא בְצִדְקָֽתְךָ יְהֹוָה אֱלֹהֶיךָ נֹתֵן לְךָ אֶת הָאָרֶץ הַטּוֹבָה הַזֹּאת לְרִשְׁתָּהּ כִּי עַם קְשֵׁה עֹרֶף אָֽתָּה:  ת  וְתִדַע אֲרֵי לָא בְזָכוּתָךְ יְיָ אֱלָהָךְ יָהֵב לָךְ יָת אַרְעָא טַבְתָא הָדָא לְמֵירְתַהּ אֲרֵי עַם קְשֵׁי קְדַל אַתְּ:

{ז} זְכֹר אַל תִּשְׁכַּח אֵת אֲשֶׁר הִקְצַפְתָּ אֶת יְהֹוָה אֱלֹהֶיךָ בַּמִּדְבָּר לְמִן הַיּוֹם אֲשֶׁר יָצָאתָ מֵאֶרֶץ מִצְרַיִם עַד בֹּֽאֲכֶם עַד הַמָּקוֹם הַזֶּה מַמְרִים הֱיִיתֶם עִם יְהֹוָֽה:  מ  זְכֹר אַל תִּשְׁכַּח אֵת אֲשֶׁר הִקְצַפְתָּ אֶת יְהֹוָה אֱלֹהֶיךָ בַּמִּדְבָּר לְמִן הַיּוֹם אֲשֶׁר יָצָאתָ מֵאֶרֶץ מִצְרַיִם עַד בֹּֽאֲכֶם עַד הַמָּקוֹם הַזֶּה מַמְרִים הֱיִיתֶם עִם יְהֹוָֽה:  ת  הֱוֵי דְכִיר לָא תִנְשֵׁי יָת דִי אַרְגֶזְתָּא קֳדָם יְיָ אֱלָהָךְ בְּמַדְבְּרָא לְמִן יוֹמָא דִי נְפַקְתָּא מֵאַרְעָא דְמִצְרַיִם עַד מֵיתֵיכוֹן עַד אַתְרָא הָדֵין מְסָרְבִין הֲוֵיתוּן קֳדָם יְיָ:

{ח} וּבְחֹרֵב הִקְצַפְתֶּם אֶת יְהֹוָה וַיִּתְאַנַּף יְהֹוָה בָּכֶם לְהַשְׁמִיד אֶתְכֶֽם:  מ  וּבְחֹרֵב הִקְצַפְתֶּם אֶת יְהֹוָה וַיִּתְאַנַּף יְהֹוָה בָּכֶם לְהַשְׁמִיד אֶתְכֶֽם:  ת  וּבְחוֹרֵב אַרְגֶזְתּוּן קֳדָם יְיָ וַהֲוָה רְגַז מִן קֳדָם יְיָ בְּכוֹן לְשֵׁיצָאָה יַתְכוֹן:

{ט} בַּֽעֲלֹתִי הָהָרָה לָקַחַת לוּחֹת הָֽאֲבָנִים לוּחֹת הַבְּרִית אֲשֶׁר כָּרַת יְהֹוָה עִמָּכֶם וָֽאֵשֵׁב בָּהָר אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה לֶחֶם לֹא אָכַלְתִּי וּמַיִם לֹא שָׁתִֽיתִי:  מ  בַּֽעֲלֹתִי הָהָרָה לָקַחַת לוּחֹת הָֽאֲבָנִים לוּחֹת הַבְּרִית אֲשֶׁר כָּרַת יְהֹוָה עִמָּכֶם וָֽאֵשֵׁב בָּהָר אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה לֶחֶם לֹא אָכַלְתִּי וּמַיִם לֹא שָׁתִֽיתִי:  ת  בְּמִסְקִי לְטוּרָא לְמִסַב לוּחֵי אַבְנַיָא לוּחֵי קְיָמָא דִי גְזַר יְיָ עִמְכוֹן וִיתֵבִית בְּטוּרָא אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן לַחֲמָא לָא אֲכָלִית וּמַיָא לָא אִשְׁתֵּיתִי:

{י} וַיִּתֵּן יְהֹוָה אֵלַי אֶת שְׁנֵי לוּחֹת הָֽאֲבָנִים כְּתֻבִים בְּאֶצְבַּע אֱלֹהִים וַֽעֲלֵיהֶם כְּֽכָל הַדְּבָרִים אֲשֶׁר דִּבֶּר יְהֹוָה עִמָּכֶם בָּהָר מִתּוֹךְ הָאֵשׁ בְּיוֹם הַקָּהָֽל:  מ  וַיִּתֵּן יְהֹוָה אֵלַי אֶת שְׁנֵי לוּחֹת הָֽאֲבָנִים כְּתֻבִים בְּאֶצְבַּע אֱלֹהִים וַֽעֲלֵיהֶם כְּֽכָל הַדְּבָרִים אֲשֶׁר דִּבֶּר יְהֹוָה עִמָּכֶם בָּהָר מִתּוֹךְ הָאֵשׁ בְּיוֹם הַקָּהָֽל:  ת  וִיהַב יְיָ לִי יָת תְּרֵין לוּחֵי אַבְנַיָא כְּתִיבִין בְּאֶצְבְּעָא דַיְיָ וַעֲלֵיהוֹן כְּכָל פִּתְגָמַיָא דִי מַלִיל יְיָ עִמְכוֹן בְּטוּרָא מִגוֹ אֶשָׁתָא בְּיוֹמָא דִקְהָלָא:

{יא} וַיְהִי מִקֵּץ אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה נָתַן יְהֹוָה אֵלַי אֶת שְׁנֵי לֻחֹת הָֽאֲבָנִים לֻחוֹת הַבְּרִֽית:  מ  וַיְהִי מִקֵּץ אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה נָתַן יְהֹוָה אֵלַי אֶת שְׁנֵי לֻחֹת הָֽאֲבָנִים לֻחוֹת הַבְּרִֽית:  ת  וַהֲוָה מִסוֹף אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן יְהַב יְיָ לִי יָת תְּרֵין לוּחֵי אַבְנַיָא לוּחֵי קְיָמָא:

{יב} וַיֹּאמֶר יְהֹוָה אֵלַי קוּם רֵד מַהֵר מִזֶּה כִּי שִׁחֵת עַמְּךָ אֲשֶׁר הוֹצֵאתָ מִמִּצְרָיִם סָרוּ מַהֵר מִן הַדֶּרֶךְ אֲשֶׁר צִוִּיתִם עָשׂוּ לָהֶם מַסֵּכָֽה:  מ  וַיֹּאמֶר יְהֹוָה אֵלַי קוּם רֵד מַהֵר מִזֶּה כִּי שִׁחֵת עַמְּךָ אֲשֶׁר הוֹצֵאתָ מִמִּצְרָיִם סָרוּ מַהֵר מִן הַדֶּרֶךְ אֲשֶׁר צִוִּיתִם עָשׂוּ לָהֶם מַסֵּכָֽה:  ת  וַאֲמַר יְיָ לִי קוּם חוּת בִּפְרִיעַ מִכָּא אֲרֵי חַבִּיל עַמָךְ דִי אַפֶּקְתָּא מִמִצְרָיִם סָטוּ בִּפְרִיעַ מִן אָרְחָא דִי פַקֶדְתִּנוּן עֲבָדוּ לְהוֹן מַתְּכָא:

{יג} וַיֹּאמֶר יְהֹוָה אֵלַי לֵאמֹר רָאִיתִי אֶת הָעָם הַזֶּה וְהִנֵּה עַם קְשֵׁה עֹרֶף הֽוּא:  מ  וַיֹּאמֶר יְהֹוָה אֵלַי לֵאמֹר רָאִיתִי אֶת הָעָם הַזֶּה וְהִנֵּה עַם קְשֵׁה עֹרֶף הֽוּא:  ת  וַאֲמַר יְיָ לִי לְמֵימָר גְלֵי קֳדָמַי יָת עַמָא הָדֵין וְהָא עַם קְשֵׁי קְדַל הוּא:

{יד} הֶרֶף מִמֶּנִּי וְאַשְׁמִידֵם וְאֶמְחֶה אֶת שְׁמָם מִתַּחַת הַשָּׁמָיִם וְאֶֽעֱשֶׂה אֽוֹתְךָ לְגֽוֹי עָצוּם וָרָב מִמֶּֽנּוּ:  מ  הֶרֶף מִמֶּנִּי וְאַשְׁמִידֵם וְאֶמְחֶה אֶת שְׁמָם מִתַּחַת הַשָּׁמָיִם וְאֶֽעֱשֶׂה אֽוֹתְךָ לְגֽוֹי עָצוּם וָרָב מִמֶּֽנּוּ:  ת  אַנַח בָּעוּתָךְ מִקֳדָמַי וֶאֱשֵׁצִנוּן וְאֶמְחֵי יָת שִׁמְהוֹן מִתְּחוֹת שְׁמַיָא וְאֶעְבֵּד יָתָךְ לְעַם תַּקִיף וְסַגִי מִנְהוֹן:

{טו} וָאֵפֶן וָֽאֵרֵד מִן הָהָר וְהָהָר בֹּעֵר בָּאֵשׁ וּשְׁנֵי לוּחֹת הַבְּרִית עַל שְׁתֵּי יָדָֽי:  מ  וָאֵפֶן וָֽאֵרֵד מִן הָהָר וְהָהָר בֹּעֵר בָּאֵשׁ וּשְׁנֵי לוּחֹת הַבְּרִית עַל שְׁתֵּי יָדָֽי:  ת  וְאִתְפְּנֵיתִי וּנְחָתִית מִן טוּרָא וְטּוּרָא בָּעֵר בְּאֶשָׁתָא וּתְרֵין לוּחֵי קְיָמָא עַל תַּרְתֵּין יְדָי:

{טז} וָאֵרֶא וְהִנֵּה חֲטָאתֶם לַֽיהֹוָה אֱלֹֽהֵיכֶם עֲשִׂיתֶם לָכֶם עֵגֶל מַסֵּכָה סַרְתֶּם מַהֵר מִן הַדֶּרֶךְ אֲשֶׁר צִוָּה יְהֹוָה אֶתְכֶֽם:  מ  וָאֵרֶא וְהִנֵּה חֲטָאתֶם לַֽיהֹוָה אֱלֹֽהֵיכֶם עֲשִׂיתֶם לָכֶם עֵגֶל מַסֵּכָה סַרְתֶּם מַהֵר מִן הַדֶּרֶךְ אֲשֶׁר צִוָּה יְהֹוָה אֶתְכֶֽם:  ת  וַחֲזֵיתִי וְהָא חַבְתּוּן קֳדָם יְיָ אֱלָהֲכוֹן עֲבַדְתּוּן לְכוֹן עֵגֶל מַתְּכָא סְטֵיתוּן בִּפְרִיעַ מִן אָרְחָא דִי פַקִיד יְיָ יָתְכוֹן:

{יז} וָֽאֶתְפֹּשׂ בִּשְׁנֵי הַלֻּחֹת וָֽאַשְׁלִכֵם מֵעַל שְׁתֵּי יָדָי וָֽאֲשַׁבְּרֵם לְעֵֽינֵיכֶֽם:  מ  וָֽאֶתְפֹּשׂ בִּשְׁנֵי הַלֻּחֹת וָֽאַשְׁלִכֵם מֵעַל שְׁתֵּי יָדָי וָֽאֲשַׁבְּרֵם לְעֵֽינֵיכֶֽם:  ת  וַאֲחָדִית בִּתְרֵין לוּחַיָא וּרְמִיתִנוּן מֵעַל תַּרְתֵּין יְדָי וּתְבַרְתִּנוּן לְעֵינֵיכוֹן:

{יח} וָֽאֶתְנַפַּל לִפְנֵי יְהֹוָה כָּרִֽאשֹׁנָה אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה לֶחֶם לֹא אָכַלְתִּי וּמַיִם לֹא שָׁתִיתִי עַל כָּל חַטַּאתְכֶם אֲשֶׁר חֲטָאתֶם לַֽעֲשׂוֹת הָרַע בְּעֵינֵי יְהֹוָה לְהַכְעִיסֽוֹ:  מ  וָֽאֶתְנַפַּל לִפְנֵי יְהֹוָה כָּרִֽאשֹׁנָה אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה לֶחֶם לֹא אָכַלְתִּי וּמַיִם לֹא שָׁתִיתִי עַל כָּל חַטַּאתְכֶם אֲשֶׁר חֲטָאתֶם לַֽעֲשׂוֹת הָרַע בְּעֵינֵי יְהֹוָה לְהַכְעִיסֽוֹ:  ת  וְאִשְׁתַּטָחִית קֳדָם יְיָ כְּקַדְמֵיתָא אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן לַחְמָא לָא אֲכָלִית וּמַיָא לָא אִשְׁתֵּיתִי עַל כָּל חוֹבֵיכוֹן דִי חַבְתּוּן לְמֶעְבַּד דְבִישׁ קֳדָם יְיָ לְאַרְגָזָא קֳדָמוֹהִי:

{יט} כִּי יָגֹרְתִּי מִפְּנֵי הָאַף וְהַחֵמָה אֲשֶׁר קָצַף יְהֹוָה עֲלֵיכֶם לְהַשְׁמִיד אֶתְכֶם וַיִּשְׁמַע יְהֹוָה אֵלַי גַּם בַּפַּעַם הַהִֽוא:  מ  כִּי יָגֹרְתִּי מִפְּנֵי הָאַף וְהַחֵמָה אֲשֶׁר קָצַף יְהֹוָה עֲלֵיכֶם לְהַשְׁמִיד אֶתְכֶם וַיִּשְׁמַע יְהֹוָה אֵלַי גַּם בַּפַּעַם הַהִֽוא:  ת  אֲרֵי דְחֵלִית מִקָדָם רָגְזָא וְחֵמְתָא דִי רְגַז יְיָ עֲלֵיכוֹן לְשֵׁצָאָה יָתְכוֹן וְקַבִּיל יְיָ צְלוֹתִי אַף בְּזִמְנָא הַהִיא:

{כ} וּֽבְאַֽהֲרֹן הִתְאַנַּף יְהֹוָה מְאֹד לְהַשְׁמִידוֹ וָֽאֶתְפַּלֵּל גַּם בְּעַד אַֽהֲרֹן בָּעֵת הַהִֽוא:  מ  וּֽבְאַֽהֲרֹן הִתְאַנַּף יְהֹוָה מְאֹד לְהַשְׁמִידוֹ וָֽאֶתְפַּלֵּל גַּם בְּעַד אַֽהֲרֹן בָּעֵת הַהִֽוא:  ת  וְעַל אַהֲרֹן הֲוָה רְגַז מִן קֳדָם יְיָ לַחֲדָא לְשֵׁצָיוּתֵהּ וְצַלֵיתִי אַף עַל אַהֲרֹן בְּעִדָנָא הַהִיא:

{כא} וְֽאֶת חַטַּאתְכֶם אֲשֶׁר עֲשִׂיתֶם אֶת הָעֵגֶל לָקַחְתִּי וָֽאֶשְׂרֹף אֹתוֹ בָּאֵשׁ וָֽאֶכֹּת אֹתוֹ טָחוֹן הֵיטֵב עַד אֲשֶׁר דַּק לְעָפָר וָֽאַשְׁלִךְ אֶת עֲפָרוֹ אֶל הַנַּחַל הַיֹּרֵד מִן הָהָֽר:  מ  וְֽאֶת חַטַּאתְכֶם אֲשֶׁר עֲשִׂיתֶם אֶת הָעֵגֶל לָקַחְתִּי וָֽאֶשְׂרֹף אֹתוֹ בָּאֵשׁ וָֽאֶכֹּת אֹתוֹ טָחוֹן הֵיטֵב עַד אֲשֶׁר דַּק לְעָפָר וָֽאַשְׁלִךְ אֶת עֲפָרוֹ אֶל הַנַּחַל הַיֹּרֵד מִן הָהָֽר:  ת  וְיָת חוֹבָתְכוֹן דִי עֲבַדְתּוּן יָת עֶגְלָא נְסֵבִית וְאוֹקֵדִית יָתֵהּ בְּנוּרָא וְשָׁפִית יָתֵהּ בְּשׁוּפִינָא יָאוּת עַד דִי הֲוָה דַקִיק לְעַפְרָא וּרְמֵת יָת עַפְרֵהּ לְנַחֲלָא דְנָחֵת מִן טוּרָא:

{כב} וּֽבְתַבְעֵרָה וּבְמַסָּה וּבְקִבְרֹת הַֽתַּֽאֲוָה מַקְצִפִים הֱיִיתֶם אֶת יְהֹוָֽה:  מ  וּֽבְתַבְעֵרָה וּבְמַסָּה וּבְקִבְרֹת הַֽתַּֽאֲוָה מַקְצִפִים הֱיִיתֶם אֶת יְהֹוָֽה:  ת  וּבִדְלֶקְתָּא וּבְנִסֵתָא וּבְקִבְרֵי דִמְשַׁאֲלֵי מַרְגְזִין הֲוֵיתוּן קֳדָם יְיָ:

{כג} וּבִשְׁלֹחַ יְהֹוָה אֶתְכֶם מִקָּדֵשׁ בַּרְנֵעַ לֵאמֹר עֲלוּ וּרְשׁוּ אֶת הָאָרֶץ אֲשֶׁר נָתַתִּי לָכֶם וַתַּמְרוּ אֶת פִּי יְהֹוָה אֱלֹהֵיכֶם וְלֹא הֶֽאֱמַנְתֶּם לוֹ וְלֹא שְׁמַעְתֶּם בְּקֹלֽוֹ:  מ  וּבִשְׁלֹחַ יְהֹוָה אֶתְכֶם מִקָּדֵשׁ בַּרְנֵעַ לֵאמֹר עֲלוּ וּרְשׁוּ אֶת הָאָרֶץ אֲשֶׁר נָתַתִּי לָכֶם וַתַּמְרוּ אֶת פִּי יְהֹוָה אֱלֹהֵיכֶם וְלֹא הֶֽאֱמַנְתֶּם לוֹ וְלֹא שְׁמַעְתֶּם בְּקֹלֽוֹ:  ת  וְכַד שְׁלַח יְיָ יָתְכוֹן מֵרְקַם גֵיאָה לְמֵימַר סְקוֹ וְאַחֲסִינוּ יָת אַרְעָא דִי יְהָבִית לְכוֹן וְסָרֶבְתּוּן עַל גְזֵרַת מֵימְרָא דַיְיָ אֱלָהֲכוֹן וְלָא הֵימַנְתּוּן לֵהּ וְלָא קַבֶּלְתּוּן לְמֵימְרֵהּ:

{כד} מַמְרִים הֱיִיתֶם עִם יְהֹוָה מִיּוֹם דַּעְתִּי אֶתְכֶֽם:  מ  מַמְרִים הֱיִיתֶם עִם יְהֹוָה מִיּוֹם דַּעְתִּי אֶתְכֶֽם:  ת  מְסָרְבִין הֲוֵיתוּן קֳדָם יְיָ מִיוֹמָא דִידָעִית יָתְכוֹן:

{כה} וָֽאֶתְנַפַּל לִפְנֵי יְהֹוָה אֵת אַרְבָּעִים הַיּוֹם וְאֶת אַרְבָּעִים הַלַּיְלָה אֲשֶׁר הִתְנַפָּלְתִּי כִּֽי אָמַר יְהֹוָה לְהַשְׁמִיד אֶתְכֶֽם:  מ  וָֽאֶתְנַפַּל לִפְנֵי יְהֹוָה אֵת אַרְבָּעִים הַיּוֹם וְאֶת אַרְבָּעִים הַלַּיְלָה אֲשֶׁר הִתְנַפָּלְתִּי כִּֽי אָמַר יְהֹוָה לְהַשְׁמִיד אֶתְכֶֽם:  ת  וְאִשְׁתַּטָחִית קֳדָם יְיָ יָת אַרְבְּעִין יְמָמִין וְיָת אַרְבְּעִין לֵילָוָן דִי אִשְׁתַּטָחִית אֲרֵי אֲמַר יְיָ לְשֵׁצָאָה יָתְכוֹן:

{כו} וָֽאֶתְפַּלֵּל אֶל יְהֹוָה וָֽאֹמַר אֲדֹנָי יֱהֹוִה אַל תַּשְׁחֵת עַמְּךָ וְנַֽחֲלָתְךָ אֲשֶׁר פָּדִיתָ בְּגָדְלֶךָ אֲשֶׁר הוֹצֵאתָ מִמִּצְרַיִם בְּיָד חֲזָקָֽה:  מ  וָֽאֶתְפַּלֵּל אֶל יְהֹוָה וָֽאֹמַר אֲדֹנָי יֱהֹוִה אַל תַּשְׁחֵת עַמְּךָ וְנַֽחֲלָתְךָ אֲשֶׁר פָּדִיתָ בְּגָדְלֶךָ אֲשֶׁר הוֹצֵאתָ מִמִּצְרַיִם בְּיָד חֲזָקָֽה:  ת  וְצַלֵיתִי קֳדָם יְיָ וַאֲמָרִית יְיָ אֱלֹהִים לָא תְחַבֵּל עַמָךְ וְאַחֲסַנְתָּךְ דִי פְרַקְתָּא בְּתָקְפָּךְ דִי אַפֶּקְתָּא מִמִצְרַיִם בִּידָא תַקִיפָא:

{כז} זְכֹר לַֽעֲבָדֶיךָ לְאַבְרָהָם לְיִצְחָק וּֽלְיַֽעֲקֹב אַל תֵּפֶן אֶל קְשִׁי הָעָם הַזֶּה וְאֶל רִשְׁעוֹ וְאֶל חַטָּאתֽוֹ:  מ  זְכֹר לַֽעֲבָדֶיךָ לְאַבְרָהָם לְיִצְחָק וּֽלְיַֽעֲקֹב אַל תֵּפֶן אֶל קְשִׁי הָעָם הַזֶּה וְאֶל רִשְׁעוֹ וְאֶל חַטָּאתֽוֹ:  ת  אִדְכַר לְעַבְדָיךְ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לָא תִתְפְּנֵי לְקַשְׁיוּת עַמָא הָדֵין וּלְחוֹבֵיהוֹן וְלַחֲטָאֵיהוֹן:

{כח} פֶּן יֹֽאמְרוּ הָאָרֶץ אֲשֶׁר הֽוֹצֵאתָנוּ מִשָּׁם מִבְּלִי יְכֹלֶת יְהֹוָה לַֽהֲבִיאָם אֶל הָאָרֶץ אֲשֶׁר דִּבֶּר לָהֶם וּמִשִּׂנְאָתוֹ אוֹתָם הֽוֹצִיאָם לַֽהֲמִתָם בַּמִּדְבָּֽר:  מ  פֶּן יֹֽאמְרוּ הָאָרֶץ אֲשֶׁר הֽוֹצֵאתָנוּ מִשָּׁם מִבְּלִי יְכֹלֶת יְהֹוָה לַֽהֲבִיאָם אֶל הָאָרֶץ אֲשֶׁר דִּבֶּר לָהֶם וּמִשִּׂנְאָתוֹ אוֹתָם הֽוֹצִיאָם לַֽהֲמִתָם בַּמִּדְבָּֽר:  ת  דִילְמָא יֵימְרוּן דָיְרֵי אַרְעָא דִי אַפֶּקְתָּנָא מִתַּמָן מִדְלֵית יוּכְלָא דַיְיָ לְאָעָלוּתְהוֹן לְאַרְעָא דִי מַלִיל לְהוֹן וּמִדְסָנֵי יָתְהוֹן אַפֵּקִנוּן לְקַטָלוּתְהוֹן בְּמַדְבְּרָא:

{כט} וְהֵם עַמְּךָ וְנַֽחֲלָתֶךָ אֲשֶׁר הוֹצֵאתָ בְּכֹֽחֲךָ הַגָּדֹל וּבִֽזְרֹֽעֲךָ הַנְּטוּיָֽה:  מ  וְהֵם עַמְּךָ וְנַֽחֲלָתֶךָ אֲשֶׁר הוֹצֵאתָ בְּכֹֽחֲךָ הַגָּדֹל וּבִֽזְרֹֽעֲךָ הַנְּטוּיָֽה:  ת  וְאִנוּן עַמָךְ וְאַחֲסַנְתָּךְ דִי אַפֶּקְתָּא בְּחֵילָךְ רַבָּא וּבִדְרָעָךְ מְרָמְמָא: (פ)




דברים פרק-י

{א}  רביעי  בָּעֵת הַהִוא אָמַר יְהֹוָה אֵלַי פְּסָל לְךָ שְׁנֵֽי לוּחֹת אֲבָנִים כָּרִאשֹׁנִים וַֽעֲלֵה אֵלַי הָהָרָה וְעָשִׂיתָ לְּךָ אֲרוֹן עֵֽץ:  מ  בָּעֵת הַהִוא אָמַר יְהֹוָה אֵלַי פְּסָל לְךָ שְׁנֵֽי לוּחֹת אֲבָנִים כָּרִאשֹׁנִים וַֽעֲלֵה אֵלַי הָהָרָה וְעָשִׂיתָ לְּךָ אֲרוֹן עֵֽץ:  ת  בְּעִדָנָא הַהִיא אֲמַר יְיָ לִי פְּסַל לָךְ תְּרֵין לוּחֵי אַבְנַיָא כְּקַדְמָאֵי וְסַק לָקֳדָמַי לְטּוּרָא וְתַעְבֵּד לָךְ אֲרוֹנָא דְאָעָא:

{ב} וְאֶכְתֹּב עַל הַלֻּחֹת אֶת הַדְּבָרִים אֲשֶׁר הָיוּ עַל הַלֻּחֹת הָרִֽאשֹׁנִים אֲשֶׁר שִׁבַּרְתָּ וְשַׂמְתָּם בָּֽאָרֽוֹן:  מ  וְאֶכְתֹּב עַל הַלֻּחֹת אֶת הַדְּבָרִים אֲשֶׁר הָיוּ עַל הַלֻּחֹת הָרִֽאשֹׁנִים אֲשֶׁר שִׁבַּרְתָּ וְשַׂמְתָּם בָּֽאָרֽוֹן:  ת  וְאֶכְתּוֹב עַל לוּחַיָא יָת פִּתְגָמַיָא דִי הֲווֹ עַל לוּחַיָא קַדְמָאֵי דִי תְבַרְתָּא וּתְשַׁוִנוּן בַּאֲרוֹנָא:

{ג} וָאַעַשׂ אֲרוֹן עֲצֵי שִׁטִּים וָֽאֶפְסֹל שְׁנֵֽי לֻחֹת אֲבָנִים כָּרִֽאשֹׁנִים וָאַעַל הָהָרָה וּשְׁנֵי הַלֻּחֹת בְּיָדִֽי:  מ  וָאַעַשׂ אֲרוֹן עֲצֵי שִׁטִּים וָֽאֶפְסֹל שְׁנֵֽי לֻחֹת אֲבָנִים כָּרִֽאשֹׁנִים וָאַעַל הָהָרָה וּשְׁנֵי הַלֻּחֹת בְּיָדִֽי:  ת  וַעֲבָדִית אֲרוֹנָא דְאָעֵי שִׁטִּין וּפְסָלִית תְּרֵין לוּחֵי אַבְנַיָא כְּקַדְמָאֵי וּסְלֵקִית לְטוּרָא וּתְרֵין לוּחַיָא בִּידִי:

{ד} וַיִּכְתֹּב עַל הַלֻּחֹת כַּמִּכְתָּב הָֽרִאשׁוֹן אֵת עֲשֶׂרֶת הַדְּבָרִים אֲשֶׁר דִּבֶּר יְהֹוָה אֲלֵיכֶם בָּהָר מִתּוֹךְ הָאֵשׁ בְּיוֹם הַקָּהָל וַיִּתְּנֵם יְהֹוָה אֵלָֽי:  מ  וַיִּכְתֹּב עַל הַלֻּחֹת כַּמִּכְתָּב הָֽרִאשׁוֹן אֵת עֲשֶׂרֶת הַדְּבָרִים אֲשֶׁר דִּבֶּר יְהֹוָה אֲלֵיכֶם בָּהָר מִתּוֹךְ הָאֵשׁ בְּיוֹם הַקָּהָל וַיִּתְּנֵם יְהֹוָה אֵלָֽי:  ת  וּכְתַב עַל לוּחַיָא כִּכְתָבָא קַדְמָאָה יָת עַשְׂרָא פִתְגָמַיָא דִי מַלִיל יְיָ עִמְכוֹן בְּטוּרָא מִגוֹ אֶשְׁתָא בְּיוֹמָא דִקְהָלָא וִיהָבִנוּן יְיָ לִי:

{ה} וָאֵפֶן וָֽאֵרֵד מִן הָהָר וָֽאָשִׂם אֶת הַלֻּחֹת בָּֽאָרוֹן אֲשֶׁר עָשִׂיתִי וַיִּהְיוּ שָׁם כַּֽאֲשֶׁר צִוַּנִי יְהֹוָֽה:  מ  וָאֵפֶן וָֽאֵרֵד מִן הָהָר וָֽאָשִׂם אֶת הַלֻּחֹת בָּֽאָרוֹן אֲשֶׁר עָשִׂיתִי וַיִּהְיוּ שָׁם כַּֽאֲשֶׁר צִוַּנִי יְהֹוָֽה:  ת  וְאִתְפְּנֵיתִי וּנְחָתִית מִן טוּרָא וְשַׁוֵיתִי יָת לוּחַיָא בַּאֲרוֹנָא דִי עֲבָדִית וַהֲווֹ תַמָן כְּמָא דִי פַקְדַנִי יְיָ:

{ו} וּבְנֵי יִשְׂרָאֵל נָֽסְעוּ מִבְּאֵרֹת בְּנֵי יַֽעֲקָן מֽוֹסֵרָה שָׁם מֵת אַֽהֲרֹן וַיִּקָּבֵר שָׁם וַיְכַהֵן אֶלְעָזָר בְּנוֹ תַּחְתָּֽיו:  מ  וּבְנֵי יִשְׂרָאֵל נָֽסְעוּ מִבְּאֵרֹת בְּנֵי יַֽעֲקָן מֽוֹסֵרָה שָׁם מֵת אַֽהֲרֹן וַיִּקָּבֵר שָׁם וַיְכַהֵן אֶלְעָזָר בְּנוֹ תַּחְתָּֽיו:  ת  וּבְנֵי יִשְׂרָאֵל נְטָלוּ מִבְּאֵרוֹת בְּנֵי יַעֲקָן לְמוֹסֵרָה תַּמָן מִית אַהֲרֹן וְאִתְקְבַר תַּמָן וְשַׁמֵשׁ אֶלְעָזָר בְּרֵהּ תְּחוֹתוֹהִי:

{ז} מִשָּׁם נָֽסְעוּ הַגֻּדְגֹּדָה וּמִן הַגֻּדְגֹּדָה יָטְבָתָה אֶרֶץ נַֽחֲלֵי מָֽיִם:  מ  מִשָּׁם נָֽסְעוּ הַגֻּדְגֹּדָה וּמִן הַגֻּדְגֹּדָה יָטְבָתָה אֶרֶץ נַֽחֲלֵי מָֽיִם:  ת  מִתַּמָן נְטָּלוּ לְגֻדְגֹד וּמִן גֻדְגֹד לְיָטְבָת אַרְעָא נָגְדָא נַחֲלִין דְמַיִין:

{ח} בָּעֵת הַהִוא הִבְדִּיל יְהֹוָה אֶת שֵׁבֶט הַלֵּוִי לָשֵׂאת אֶת אֲרוֹן בְּרִית יְהֹוָה לַֽעֲמֹד לִפְנֵי יְהֹוָה לְשָֽׁרְתוֹ וּלְבָרֵךְ בִּשְׁמוֹ עַד הַיּוֹם הַזֶּֽה:  מ  בָּעֵת הַהִוא הִבְדִּיל יְהֹוָה אֶת שֵׁבֶט הַלֵּוִי לָשֵׂאת אֶת אֲרוֹן בְּרִית יְהֹוָה לַֽעֲמֹד לִפְנֵי יְהֹוָה לְשָֽׁרְתוֹ וּלְבָרֵךְ בִּשְׁמוֹ עַד הַיּוֹם הַזֶּֽה:  ת  בְּעִדָנָא הַהִיא אַפְרֵשׁ יְיָ יָת שִׁבְטָּא דְלֵוִי לְמִטַל יָת אֲרוֹן קְיָמָא דַיְיָ לְמֵיקַם קֳדָם יְיָ לְשַׁמָשׁוּתֵהּ וּלְבָרָכָא בִשְׁמֵהּ עַד יוֹמָא הָדֵין:

{ט} עַל כֵּן לֹֽא הָיָה לְלֵוִי חֵלֶק וְנַֽחֲלָה עִם אֶחָיו יְהֹוָה הוּא נַֽחֲלָתוֹ כַּֽאֲשֶׁר דִּבֶּר יְהֹוָה אֱלֹהֶיךָ לֽוֹ:  מ  עַל כֵּן לֹֽא הָיָה לְלֵוִי חֵלֶק וְנַֽחֲלָה עִם אֶחָיו יְהֹוָה הוּא נַֽחֲלָתוֹ כַּֽאֲשֶׁר דִּבֶּר יְהֹוָה אֱלֹהֶיךָ לֽוֹ:  ת  עַל כֵּן לָא הֲּוָה לְלֵוִי חֳלָק וְאַחֲסָנָא עִם אֲחוֹהִי מַתְּנָן דִיהַב לֵהּ יְיָ אִנוּן אַחֲסַנְתֵּהּ כְּמָא דִי מַלִיל יְיָ אֱלָהָךְ לֵהּ:

{י} וְאָֽנֹכִי עָמַדְתִּי בָהָר כַּיָּמִים הָרִאשֹׁנִים אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה וַיִּשְׁמַע יְהֹוָה אֵלַי גַּם בַּפַּעַם הַהִוא לֹֽא אָבָה יְהֹוָה הַשְׁחִיתֶֽךָ:  מ  וְאָֽנֹכִי עָמַדְתִּי בָהָר כַּיָּמִים הָרִאשֹׁנִים אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה וַיִּשְׁמַע יְהֹוָה אֵלַי גַּם בַּפַּעַם הַהִוא לֹֽא אָבָה יְהֹוָה הַשְׁחִיתֶֽךָ:  ת  וַאֲנָא הֲוֵיתִי קָאֵם בְּטוּרָא כְּיוֹמִין קַדְמָאִין אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן וְקַבִּיל יְיָ צְלוֹתִי אַף בְּזִמְנָא הַהוּא לָא אָבֵי יְיָ לְחַבָּלוּתָךְ:

{יא} וַיֹּאמֶר יְהֹוָה אֵלַי קוּם לֵךְ לְמַסַּע לִפְנֵי הָעָם וְיָבֹאוּ וְיִֽירְשׁוּ אֶת הָאָרֶץ אֲשֶׁר נִשְׁבַּעְתִּי לַֽאֲבֹתָם לָתֵת לָהֶֽם:  מ  וַיֹּאמֶר יְהֹוָה אֵלַי קוּם לֵךְ לְמַסַּע לִפְנֵי הָעָם וְיָבֹאוּ וְיִֽירְשׁוּ אֶת הָאָרֶץ אֲשֶׁר נִשְׁבַּעְתִּי לַֽאֲבֹתָם לָתֵת לָהֶֽם:  ת  וַאֲמַר יְיָ לִי קוּם זִיל לְמִטוּל קֳדָם עַמָא וְיֵעֲלוּן וְיֵרְתוּן יָת אַרְעָא דִי קַיֵמִית לַאֲבָהָתְהוֹן לְמִתַּן לְהוֹן: (פ)

{יב}  חמישי  וְעַתָּה יִשְׂרָאֵל מָה יְהֹוָה אֱלֹהֶיךָ שֹׁאֵל מֵֽעִמָּךְ כִּי אִם לְיִרְאָה אֶת יְהֹוָה אֱלֹהֶיךָ לָלֶכֶת בְּכָל דְּרָכָיו וּלְאַֽהֲבָה אֹתוֹ וְלַֽעֲבֹד אֶת יְהֹוָה אֱלֹהֶיךָ בְּכָל לְבָֽבְךָ וּבְכָל נַפְשֶֽׁךָ:  מ  וְעַתָּה יִשְׂרָאֵל מָה יְהֹוָה אֱלֹהֶיךָ שֹׁאֵל מֵֽעִמָּךְ כִּי אִם לְיִרְאָה אֶת יְהֹוָה אֱלֹהֶיךָ לָלֶכֶת בְּכָל דְּרָכָיו וּלְאַֽהֲבָה אֹתוֹ וְלַֽעֲבֹד אֶת יְהֹוָה אֱלֹהֶיךָ בְּכָל לְבָֽבְךָ וּבְכָל נַפְשֶֽׁךָ:  ת  וּכְעַן יִשְׂרָאֵל מָה יְיָ אֱלָהָךְ תָּבַע מִנָךְ אֱלָהֵן לְמִדְחַל קֳדָם יְיָ אֱלָהָךְ לִמְהַךְ בְּכָל אָרְחָן דְתָקְנָן קֳדָמוֹהִי וּלְמִרְחַם יָתֵהּ וּלְמִפְלַח קֳדָם יְיָ אֱלָהָךְ בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ:

{יג} לִשְׁמֹר אֶת מִצְוֹת יְהֹוָה וְאֶת חֻקֹּתָיו אֲשֶׁר אָֽנֹכִי מְצַוְּךָ הַיּוֹם לְטוֹב לָֽךְ:  מ  לִשְׁמֹר אֶת מִצְוֹת יְהֹוָה וְאֶת חֻקֹּתָיו אֲשֶׁר אָֽנֹכִי מְצַוְּךָ הַיּוֹם לְטוֹב לָֽךְ:  ת  לְמִטַר יָת פִּקוֹדַיָא דַיְיָ וְיָת קְיָמוֹהִי דַאֲנָא מְפַקְדָךְ יוֹמָא דֵין לִדְיֵיטַב לָךְ:

{יד} הֵן לַֽיהֹוָה אֱלֹהֶיךָ הַשָּׁמַיִם וּשְׁמֵי הַשָּׁמָיִם הָאָרֶץ וְכָל אֲשֶׁר בָּֽהּ:  מ  הֵן לַֽיהֹוָה אֱלֹהֶיךָ הַשָּׁמַיִם וּשְׁמֵי הַשָּׁמָיִם הָאָרֶץ וְכָל אֲשֶׁר בָּֽהּ:  ת  הָא דַיְיָ אֱלָהָךְ שְׁמַיָא וּשְׁמֵי שְׁמַיָא אַרְעָא וְכָל דִי בַהּ:

{טו} רַק בַּֽאֲבֹתֶיךָ חָשַׁק יְהֹוָה לְאַֽהֲבָה אוֹתָם וַיִּבְחַר בְּזַרְעָם אַֽחֲרֵיהֶם בָּכֶם מִכָּל הָֽעַמִּים כַּיּוֹם הַזֶּֽה:  מ  רַק בַּֽאֲבֹתֶיךָ חָשַׁק יְהֹוָה לְאַֽהֲבָה אוֹתָם וַיִּבְחַר בְּזַרְעָם אַֽחֲרֵיהֶם בָּכֶם מִכָּל הָֽעַמִּים כַּיּוֹם הַזֶּֽה:  ת  לְחוֹד בַּאֲבָהָתָךְ צָבֵי יְיָ לְמִרְחַם יָתְהוֹן וְאִתִּרְעֵי בִּבְנֵיהוֹן בַּתְרֵיהוֹן בְּכוֹן מִכָּל עַמְמַיָא כְּיוֹמָא הָדֵין:

{טז} וּמַלְתֶּם אֵת עָרְלַת לְבַבְכֶם וְעָרְפְּכֶם לֹא תַקְשׁוּ עֽוֹד:  מ  וּמַלְתֶּם אֵת עָרְלַת לְבַבְכֶם וְעָרְפְּכֶם לֹא תַקְשׁוּ עֽוֹד:  ת  וְתֶעְדוּן יָת טַפְשׁוּת לִבְּכוֹן וּקְדָלְכוֹן לָא תַקְשׁוּן עוֹד:

{יז} כִּי יְהֹוָה אֱלֹֽהֵיכֶם הוּא אֱלֹהֵי הָֽאֱלֹהִים וַֽאֲדֹנֵי הָֽאֲדֹנִים הָאֵל הַגָּדֹל הַגִּבֹּר וְהַנּוֹרָא אֲשֶׁר לֹֽא יִשָּׂא פָנִים וְלֹא יִקַּח שֹֽׁחַד:  מ  כִּי יְהֹוָה אֱלֹֽהֵיכֶם הוּא אֱלֹהֵי הָֽאֱלֹהִים וַֽאֲדֹנֵי הָֽאֲדֹנִים הָאֵל הַגָּדֹל הַגִּבֹּר וְהַנּוֹרָא אֲשֶׁר לֹֽא יִשָּׂא פָנִים וְלֹא יִקַּח שֹֽׁחַד:  ת  אֲרֵי יְיָ אֱלָהֲכוֹן הוּא אֱלָהָא דַיָנִין וּמָרֵי מַלְכִין אֱלָהָא רַבָּא גִבָּרָא וּדְחִילָא דִי לֵית קֳדָמוֹהִי מִסַב אַפִּין וְאַף לָא לְקַבָּלָא שָׁחְדָא:

{יח} עֹשֶׂה מִשְׁפַּט יָתוֹם וְאַלְמָנָה וְאֹהֵב גֵּר לָתֶת לוֹ לֶחֶם וְשִׂמְלָֽה:  מ  עֹשֶׂה מִשְׁפַּט יָתוֹם וְאַלְמָנָה וְאֹהֵב גֵּר לָתֶת לוֹ לֶחֶם וְשִׂמְלָֽה:  ת  עָבֵד דִין יִתַּם וְאַרְמְלָא וְרָחֵם גִיוֹרָא לְמִתַּן לֵהּ מְזוֹנָא וּכְסוּ:

{יט} וַֽאֲהַבְתֶּם אֶת הַגֵּר כִּֽי גֵרִים הֱיִיתֶם בְּאֶרֶץ מִצְרָֽיִם:  מ  וַֽאֲהַבְתֶּם אֶת הַגֵּר כִּֽי גֵרִים הֱיִיתֶם בְּאֶרֶץ מִצְרָֽיִם:  ת  וְתִרְחֲמוּן יָת גִיוֹרָא אֲרֵי דַיָרִין הֲוֵיתוּן בְּאַרְעָא דְמִצְרָיִם:

{כ} אֶת יְהֹוָה אֱלֹהֶיךָ תִּירָא אֹתוֹ תַֽעֲבֹד וּבוֹ תִדְבָּק וּבִשְׁמוֹ תִּשָּׁבֵֽעַ:  מ  אֶת יְהֹוָה אֱלֹהֶיךָ תִּירָא אֹתוֹ תַֽעֲבֹד וּבוֹ תִדְבָּק וּבִשְׁמוֹ תִּשָּׁבֵֽעַ:  ת  יָת יְיָ אֱלָהָךְ תִּדְחַל וְקָדָמוֹהִי תִפְלָח וּלְדַחַלְתֵּהּ תִּקְרַב וּבִשְׁמֵהּ תְּקַיַם:

{כא} הוּא תְהִלָּֽתְךָ וְהוּא אֱלֹהֶיךָ אֲשֶׁר עָשָׂה אִתְּךָ אֶת הַגְּדֹלֹת וְאֶת הַנּֽוֹרָאֹת הָאֵלֶּה אֲשֶׁר רָאוּ עֵינֶֽיךָ:  מ  הוּא תְהִלָּֽתְךָ וְהוּא אֱלֹהֶיךָ אֲשֶׁר עָשָׂה אִתְּךָ אֶת הַגְּדֹלֹת וְאֶת הַנּֽוֹרָאֹת הָאֵלֶּה אֲשֶׁר רָאוּ עֵינֶֽיךָ:  ת  הוּא תֻשְׁבַּחְתָּךְ וְהוּא אֱלָהָךְ דַעֲבַד עִמָךְ יָת רַבְרְבָתָא וְיָת חֲסִינָתָא הָאִלֵין דִי חֲזוֹ עֵינָיךְ:

{כב} בְּשִׁבְעִים נֶפֶשׁ יָֽרְדוּ אֲבֹתֶיךָ מִצְרָיְמָה וְעַתָּה שָֽׂמְךָ יְהֹוָה אֱלֹהֶיךָ כְּכֽוֹכְבֵי הַשָּׁמַיִם לָרֹֽב:  מ  בְּשִׁבְעִים נֶפֶשׁ יָֽרְדוּ אֲבֹתֶיךָ מִצְרָיְמָה וְעַתָּה שָֽׂמְךָ יְהֹוָה אֱלֹהֶיךָ כְּכֽוֹכְבֵי הַשָּׁמַיִם לָרֹֽב:  ת  בְּשַׁבְעִין נַפְשָׁן נְחָתוּ אֲבָהָתָךְ לְמִצְרָיִם וּכְעַן שַׁוְיָךְ יְיָ אֱלָהָךְ כְּכוֹכְבֵי שְׁמַיָא לְמִסְגֵי:




דברים פרק-יא

{א} וְאָהַבְתָּ אֵת יְהֹוָה אֱלֹהֶיךָ וְשָֽׁמַרְתָּ מִשְׁמַרְתּוֹ וְחֻקֹּתָיו וּמִשְׁפָּטָיו וּמִצְוֹתָיו כָּל הַיָּמִֽים:  מ  וְאָהַבְתָּ אֵת יְהֹוָה אֱלֹהֶיךָ וְשָֽׁמַרְתָּ מִשְׁמַרְתּוֹ וְחֻקֹּתָיו וּמִשְׁפָּטָיו וּמִצְוֹתָיו כָּל הַיָּמִֽים:  ת  וְתִרְחַם יָת יְיָ אֱלָהָךְ וְתִטַּר מַטְרַת מֵימְרֵהּ וּקְיָמוֹהִי וְדִינוֹהִי וּפִקוֹדוֹהִי כָּל יוֹמַיָא:

{ב} וִֽידַעְתֶּם הַיּוֹם כִּי לֹא אֶת בְּנֵיכֶם אֲשֶׁר לֹא יָֽדְעוּ וַֽאֲשֶׁר לֹֽא רָאוּ אֶת מוּסַר יְהֹוָה אֱלֹֽהֵיכֶם אֶת גָּדְלוֹ אֶת יָדוֹ הַֽחֲזָקָה וּזְרֹעוֹ הַנְּטוּיָֽה:  מ  וִֽידַעְתֶּם הַיּוֹם כִּי לֹא אֶת בְּנֵיכֶם אֲשֶׁר לֹא יָֽדְעוּ וַֽאֲשֶׁר לֹֽא רָאוּ אֶת מוּסַר יְהֹוָה אֱלֹֽהֵיכֶם אֶת גָּדְלוֹ אֶת יָדוֹ הַֽחֲזָקָה וּזְרֹעוֹ הַנְּטוּיָֽה:  ת  וְתִדְעוּן יוֹמָא דֵין אֲרֵי לָא יָת בְּנֵיכוֹן דִי לָא יְדָעוּ וְדִי לָא חֲזוֹ יָת אֻלְפָנָא דַיְיָ אֱלָהֲכוֹן יָת רְבוּתֵהּ יָת יְדֵהּ תַּקִיפָא וּדְרָעֵהּ מְרָמְמָא:

{ג} וְאֶת אֹֽתֹתָיו וְאֶֽת מַֽעֲשָׂיו אֲשֶׁר עָשָׂה בְּתוֹךְ מִצְרָיִם לְפַרְעֹה מֶֽלֶךְ מִצְרַיִם וּלְכָל אַרְצֽוֹ:  מ  וְאֶת אֹֽתֹתָיו וְאֶֽת מַֽעֲשָׂיו אֲשֶׁר עָשָׂה בְּתוֹךְ מִצְרָיִם לְפַרְעֹה מֶֽלֶךְ מִצְרַיִם וּלְכָל אַרְצֽוֹ:  ת  וְיָת אַתְוָתֵהּ וְיָת עוֹבָדוֹהִי דִי עֲבַד בְּגוֹ מִצְרָיִם לְפַרְעֹה מַלְכָּא דְמִצְרַיִם וּלְכָל אַרְעֵהּ:

{ד} וַֽאֲשֶׁר עָשָׂה לְחֵיל מִצְרַיִם לְסוּסָיו וּלְרִכְבּוֹ אֲשֶׁר הֵצִיף אֶת מֵי יַם סוּף עַל פְּנֵיהֶם בְּרָדְפָם אַֽחֲרֵיכֶם וַיְאַבְּדֵם יְהֹוָה עַד הַיּוֹם הַזֶּֽה:  מ  וַֽאֲשֶׁר עָשָׂה לְחֵיל מִצְרַיִם לְסוּסָיו וּלְרִכְבּוֹ אֲשֶׁר הֵצִיף אֶת מֵי יַם סוּף עַל פְּנֵיהֶם בְּרָדְפָם אַֽחֲרֵיכֶם וַיְאַבְּדֵם יְהֹוָה עַד הַיּוֹם הַזֶּֽה:  ת  וְדִי עֲבַד לְמַשִׁרְיַת מִצְרַיִם לְסוּסְוָתֵהּ וְלִרְתִיכוֹהִי דִי אַטִיף יָת מֵי יַמָא דְסוּף עַל אַפֵּיהוֹן בְּמִרְדָפְהוֹן בַּתְרֵיכוֹן וְאוֹבָדִנוּן יְיָ עַד יוֹמָא הָדֵין:

{ה} וַֽאֲשֶׁר עָשָׂה לָכֶם בַּמִּדְבָּר עַד בֹּֽאֲכֶם עַד הַמָּקוֹם הַזֶּֽה:  מ  וַֽאֲשֶׁר עָשָׂה לָכֶם בַּמִּדְבָּר עַד בֹּֽאֲכֶם עַד הַמָּקוֹם הַזֶּֽה:  ת  וְדִי עֲבַד לְכוֹן בְּמַדְבְּרָא עַד מֵיתֵיכוֹן עַד אַתְרָא הָדֵין:

{ו} וַֽאֲשֶׁר עָשָׂה לְדָתָן וְלַֽאֲבִירָם בְּנֵי אֱלִיאָב בֶּן רְאוּבֵן אֲשֶׁר פָּֽצְתָה הָאָרֶץ אֶת פִּיהָ וַתִּבְלָעֵם וְאֶת בָּֽתֵּיהֶם וְאֶת אָֽהֳלֵיהֶם וְאֵת כָּל הַיְקוּם אֲשֶׁר בְּרַגְלֵיהֶם בְּקֶרֶב כָּל יִשְׂרָאֵֽל:  מ  וַֽאֲשֶׁר עָשָׂה לְדָתָן וְלַֽאֲבִירָם בְּנֵי אֱלִיאָב בֶּן רְאוּבֵן אֲשֶׁר פָּֽצְתָה הָאָרֶץ אֶת פִּיהָ וַתִּבְלָעֵם וְאֶת בָּֽתֵּיהֶם וְאֶת אָֽהֳלֵיהֶם וְאֵת כָּל הַיְקוּם אֲשֶׁר בְּרַגְלֵיהֶם בְּקֶרֶב כָּל יִשְׂרָאֵֽל:  ת  וְדִי עֲבַד לְדָתָן וְלַאֲבִירָם בְּנֵי אֱלִיאָב בַּר רְאוּבֵן דִי פְתָחַת אַרְעָא יָת פּוּמַהּ וּבְלָעָתְנוּן וְיָת אֱנַשׁ בָּתֵּיהוֹן וְיָת מַשְׁכְּנֵיהוֹן וְיָת כָּל יְקוּמָא דִי עִמְהוֹן בְּגוֹ כָּל יִשְׂרָאֵל:

{ז} כִּי עֵֽינֵיכֶם הָֽרֹאֹת אֵת כָּל מַֽעֲשֵׂה יְהֹוָה הַגָּדֹל אֲשֶׁר עָשָֽׂה:  מ  כִּי עֵֽינֵיכֶם הָֽרֹאֹת אֵת כָּל מַֽעֲשֵׂה יְהֹוָה הַגָּדֹל אֲשֶׁר עָשָֽׂה:  ת  אֲרֵי עֵינֵיכוֹן חֲזָאָן יָת כָּל עוֹבָדָא דַיְיָ רַבָּא דִי עֲבָד:

{ח} וּשְׁמַרְתֶּם אֶת כָּל הַמִּצְוָה אֲשֶׁר אָֽנֹכִי מְצַוְּךָ הַיּוֹם לְמַעַן תֶּֽחֶזְקוּ וּבָאתֶם וִֽירִשְׁתֶּם אֶת הָאָרֶץ אֲשֶׁר אַתֶּם עֹֽבְרִים שָׁמָּה לְרִשְׁתָּֽהּ:  מ  וּשְׁמַרְתֶּם אֶת כָּל הַמִּצְוָה אֲשֶׁר אָֽנֹכִי מְצַוְּךָ הַיּוֹם לְמַעַן תֶּֽחֶזְקוּ וּבָאתֶם וִֽירִשְׁתֶּם אֶת הָאָרֶץ אֲשֶׁר אַתֶּם עֹֽבְרִים שָׁמָּה לְרִשְׁתָּֽהּ:  ת  וְתִטְרוּן יָת כָּל תַּפְקֶדְתָּא דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין בְּדִיל דְתִתְקְפוּן וְתֵעֲלוּן וְתֵירְתוּן יָת אַרְעָא דִי אַתּוּן עָבְרִין תַּמָן לְמֵירְתַהּ:

{ט} וּלְמַעַן תַּֽאֲרִיכוּ יָמִים עַל הָאֲדָמָה אֲשֶׁר נִשְׁבַּע יְהֹוָה לַֽאֲבֹֽתֵיכֶם לָתֵת לָהֶם וּלְזַרְעָם אֶרֶץ זָבַת חָלָב וּדְבָֽשׁ:  מ  וּלְמַעַן תַּֽאֲרִיכוּ יָמִים עַל הָאֲדָמָה אֲשֶׁר נִשְׁבַּע יְהֹוָה לַֽאֲבֹֽתֵיכֶם לָתֵת לָהֶם וּלְזַרְעָם אֶרֶץ זָבַת חָלָב וּדְבָֽשׁ:  ת  וּבְדִיל דְתוֹרְכוּן יוֹמִין עַל אַרְעָא דִי קַיִים יְיָ לַאֲבָהָתְכוֹן לְמִתַּן לְהוֹן וְלִבְנֵיהוֹן אַרְעָא עָבְדָא חֲלַב וּדְבָשׁ: (ס)

{י}  שישי  כִּי הָאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ לֹא כְאֶרֶץ מִצְרַיִם הִוא אֲשֶׁר יְצָאתֶם מִשָּׁם אֲשֶׁר תִּזְרַע אֶת זַֽרְעֲךָ וְהִשְׁקִיתָ בְרַגְלְךָ כְּגַן הַיָּרָֽק:  מ  כִּי הָאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ לֹא כְאֶרֶץ מִצְרַיִם הִוא אֲשֶׁר יְצָאתֶם מִשָּׁם אֲשֶׁר תִּזְרַע אֶת זַֽרְעֲךָ וְהִשְׁקִיתָ בְרַגְלְךָ כְּגַן הַיָּרָֽק:  ת  אֲרֵי אַרְעָא דִי אַתְּ עָלֵל לְתַמָן לְמֵירְתַהּ לָא כְאַרְעָא דְמִצְרַיִם הִיא דִי נְפַקְתּוּן מִתַּמָן דִי תִזְרַע יָת זַרְעָךְ וּמַשְׁקַת לֵהּ בְּרַגְלָךְ כְּגִנְתָא יַרְקָא:

{יא} וְהָאָרֶץ אֲשֶׁר אַתֶּם עֹֽבְרִים שָׁמָּה לְרִשְׁתָּהּ אֶרֶץ הָרִים וּבְקָעֹת לִמְטַר הַשָּׁמַיִם תִּשְׁתֶּה מָּֽיִם:  מ  וְהָאָרֶץ אֲשֶׁר אַתֶּם עֹֽבְרִים שָׁמָּה לְרִשְׁתָּהּ אֶרֶץ הָרִים וּבְקָעֹת לִמְטַר הַשָּׁמַיִם תִּשְׁתֶּה מָּֽיִם:  ת  וְאַרְעָא דִי אַתּוּן עָבְרִין תַּמָן לְמֵירְתַהּ אֲרַע טוּרִין וּבִקְעָן לִּמְטַּר שְׁמַיָא תִּשְׁתֵּי מַיָא:

{יב} אֶרֶץ אֲשֶׁר יְהֹוָה אֱלֹהֶיךָ דֹּרֵשׁ אֹתָהּ תָּמִיד עֵינֵי יְהֹוָה אֱלֹהֶיךָ בָּהּ מֵֽרֵשִׁית הַשָּׁנָה וְעַד אַֽחֲרִית שָׁנָֽה:  מ  אֶרֶץ אֲשֶׁר יְהֹוָה אֱלֹהֶיךָ דֹּרֵשׁ אֹתָהּ תָּמִיד עֵינֵי יְהֹוָה אֱלֹהֶיךָ בָּהּ מֵֽרֵשִׁית הַשָּׁנָה וְעַד אַֽחֲרִית שָׁנָֽה:  ת  אַרְעָא דַיְיָ אֱלָהָךְ תָּבַע יָתַהּ תְּדִירָא עֵינֵי דַיְיָ אֱלָהָךְ בַּהּ מֵרֵישָׁא דְשַׁתָּא וְעַד סוֹפָא דְשַׁתָּא: (ס)

{יג} וְהָיָה אִם שָׁמֹעַ תִּשְׁמְעוּ אֶל מִצְוֹתַי אֲשֶׁר אָֽנֹכִי מְצַוֶּה אֶתְכֶם הַיּוֹם לְאַֽהֲבָה אֶת יְהֹוָה אֱלֹֽהֵיכֶם וּלְעָבְדוֹ בְּכָל לְבַבְכֶם וּבְכָל נַפְשְׁכֶֽם:  מ  וְהָיָה אִם שָׁמֹעַ תִּשְׁמְעוּ אֶל מִצְוֹתַי אֲשֶׁר אָֽנֹכִי מְצַוֶּה אֶתְכֶם הַיּוֹם לְאַֽהֲבָה אֶת יְהֹוָה אֱלֹֽהֵיכֶם וּלְעָבְדוֹ בְּכָל לְבַבְכֶם וּבְכָל נַפְשְׁכֶֽם:  ת  וִיהֵי אִם קַבָּלָא תְקַבְּלוּן לְפִקוּדַי דִי אֲנָא מְפַקֵד יָתְכוֹן יוֹמָא דֵין לְמִרְחַם יָת יְיָ אֱלָהֲכוֹן וּלְמִפְלַח קֳדָמוֹהִי בְּכָל לִבְּכוֹן וּבְכָל נַפְשְׁכוֹן:

{יד} וְנָֽתַתִּי מְטַֽר אַרְצְכֶם בְּעִתּוֹ יוֹרֶה וּמַלְקוֹשׁ וְאָֽסַפְתָּ דְגָנֶךָ וְתִירֽשְׁךָ וְיִצְהָרֶֽךָ:  מ  וְנָֽתַתִּי מְטַֽר אַרְצְכֶם בְּעִתּוֹ יוֹרֶה וּמַלְקוֹשׁ וְאָֽסַפְתָּ דְגָנֶךָ וְתִירֽשְׁךָ וְיִצְהָרֶֽךָ:  ת  וְאֶתֵּן מְטַר אַרְעֲכוֹן בְּעִדָנֵהּ בַּכִּיר וְלַקִישׁ וְתִכְנוֹשׁ עִבוּרָךְ וְחַמְרָךְ וּמִשְׁחָךְ:

{טו} וְנָֽתַתִּי עֵשֶׂב בְּשָֽׂדְךָ לִבְהֶמְתֶּךָ וְאָֽכַלְתָּ וְשָׂבָֽעְתָּ:  מ  וְנָֽתַתִּי עֵשֶׂב בְּשָֽׂדְךָ לִבְהֶמְתֶּךָ וְאָֽכַלְתָּ וְשָׂבָֽעְתָּ:  ת  וְאֶתֵּן עִשְׂבָּא בְחַקְלָךְ לִבְעִירָךְ וְתֵיכוּל וְתִשְׂבָּע:

{טז} הִשָּֽׁמְרוּ לָכֶם פֶּן יִפְתֶּה לְבַבְכֶם וְסַרְתֶּם וַֽעֲבַדְתֶּם אֱלֹהִים אֲחֵרִים וְהִשְׁתַּֽחֲוִיתֶם לָהֶֽם:  מ  הִשָּֽׁמְרוּ לָכֶם פֶּן יִפְתֶּה לְבַבְכֶם וְסַרְתֶּם וַֽעֲבַדְתֶּם אֱלֹהִים אֲחֵרִים וְהִשְׁתַּֽחֲוִיתֶם לָהֶֽם:  ת  אִסְתַּמָרוּ לְכוֹן דִילְמָא יִטְעוּן לִבְּכוֹן וְתִסְטוּן וְתִפְלְחוּן לְטַעֲוַת עַמְמַיָא וְתִסְגְדוּן לְהוֹן:

{יז} וְחָרָה אַף יְהֹוָה בָּכֶם וְעָצַר אֶת הַשָּׁמַיִם וְלֹא יִֽהְיֶה מָטָר וְהָאֲדָמָה לֹא תִתֵּן אֶת יְבוּלָהּ וַֽאֲבַדְתֶּם מְהֵרָה מֵעַל הָאָרֶץ הַטֹּבָה אֲשֶׁר יְהֹוָה נֹתֵן לָכֶֽם:  מ  וְחָרָה אַף יְהֹוָה בָּכֶם וְעָצַר אֶת הַשָּׁמַיִם וְלֹא יִֽהְיֶה מָטָר וְהָאֲדָמָה לֹא תִתֵּן אֶת יְבוּלָהּ וַֽאֲבַדְתֶּם מְהֵרָה מֵעַל הָאָרֶץ הַטֹּבָה אֲשֶׁר יְהֹוָה נֹתֵן לָכֶֽם:  ת  וְיִתְקוֹף רָגְזָא דַיְיָ בְּכוֹן וְיֵחוֹד יָת שְׁמַיָא וְלָא יְהֵי מִטְרָא וְאַרְעָא לָא תִתֵּן יָת עֲלַלְתַּהּ וְתֵיבְדוּן בִּפְרִיעַ מֵעַל אַרְעָא טַבְתָא דַיְיָ יָהֵב לְכוֹן:

{יח} וְשַׂמְתֶּם אֶת דְּבָרַי אֵלֶּה עַל לְבַבְכֶם וְעַֽל נַפְשְׁכֶם וּקְשַׁרְתֶּם אֹתָם לְאוֹת עַל יֶדְכֶם וְהָיוּ לְטֽוֹטָפֹת בֵּין עֵֽינֵיכֶֽם:  מ  וְשַׂמְתֶּם אֶת דְּבָרַי אֵלֶּה עַל לְבַבְכֶם וְעַֽל נַפְשְׁכֶם וּקְשַׁרְתֶּם אֹתָם לְאוֹת עַל יֶדְכֶם וְהָיוּ לְטֽוֹטָפֹת בֵּין עֵֽינֵיכֶֽם:  ת  וּתְשַׁווּן יָת פִּתְגָמַי אִלֵין עַל לִבְּכוֹן וְעַל נַפְשְׁכוֹן וְתִקְטְרוּן יָתְהוֹן לְאָת עַל יֶדְכוֹן וִיהוֹן לִתְפִלִין בֵּין עֵינֵיכוֹן:

{יט} וְלִמַּדְתֶּם אֹתָם אֶת בְּנֵיכֶם לְדַבֵּר בָּם בְּשִׁבְתְּךָ בְּבֵיתֶךָ וּבְלֶכְתְּךָ בַדֶּרֶךְ וּֽבְשָׁכְבְּךָ וּבְקוּמֶֽךָ:  מ  וְלִמַּדְתֶּם אֹתָם אֶת בְּנֵיכֶם לְדַבֵּר בָּם בְּשִׁבְתְּךָ בְּבֵיתֶךָ וּבְלֶכְתְּךָ בַדֶּרֶךְ וּֽבְשָׁכְבְּךָ וּבְקוּמֶֽךָ:  ת  וְתַלְפוּן יָתְהוֹן יָת בְּנֵיכוֹן לְמַלָלָא בְהוֹן בְּמִתְּבָךְ בְּבֵיתָךְ וּבִמְהָכָךְ בְּאָרְחָא וּבְמִשְׁכְּבָךְ וּבִמְקִימָךְ:

{כ} וּכְתַבְתָּם עַל מְזוּזוֹת בֵּיתֶךָ וּבִשְׁעָרֶֽיךָ:  מ  וּכְתַבְתָּם עַל מְזוּזוֹת בֵּיתֶךָ וּבִשְׁעָרֶֽיךָ:  ת  וְתִכְתְּבִנוּן עַל מְזוּזְיָן וְתִקְבְּעִנוּן בְּסִפֵּי בֵיתָךְ וּבְתַרְעָיךְ:

{כא} לְמַעַן יִרְבּוּ יְמֵיכֶם וִימֵי בְנֵיכֶם עַל הָֽאֲדָמָה אֲשֶׁר נִשְׁבַּע יְהֹוָה לַֽאֲבֹֽתֵיכֶם לָתֵת לָהֶם כִּימֵי הַשָּׁמַיִם עַל הָאָֽרֶץ:  מ  לְמַעַן יִרְבּוּ יְמֵיכֶם וִימֵי בְנֵיכֶם עַל הָֽאֲדָמָה אֲשֶׁר נִשְׁבַּע יְהֹוָה לַֽאֲבֹֽתֵיכֶם לָתֵת לָהֶם כִּימֵי הַשָּׁמַיִם עַל הָאָֽרֶץ:  ת  בְּדִיל דְיִסְגוּן יוֹמֵיכוֹן וְיוֹמֵי בְנֵיכוֹן עַל אַרְעָא דִי קַיִים יְיָ לַאֲבָהָתְכוֹן לְמִתַּן לְהוֹן כְּיוֹמֵי שְׁמַיָא עַל אַרְעָא: (ס)

{כב}  שביעי - במחוברות מפטיר  כִּי אִם שָׁמֹר תִּשְׁמְרוּן אֶת כָּל הַמִּצְוָה הַזֹּאת אֲשֶׁר אָֽנֹכִי מְצַוֶּה אֶתְכֶם לַֽעֲשׂתָהּ לְאַֽהֲבָה אֶת יְהֹוָה אֱלֹֽהֵיכֶם לָלֶכֶת בְּכָל דְּרָכָיו וּֽלְדָבְקָה בֽוֹ:  מ  כִּי אִם שָׁמֹר תִּשְׁמְרוּן אֶת כָּל הַמִּצְוָה הַזֹּאת אֲשֶׁר אָֽנֹכִי מְצַוֶּה אֶתְכֶם לַֽעֲשׂתָהּ לְאַֽהֲבָה אֶת יְהֹוָה אֱלֹֽהֵיכֶם לָלֶכֶת בְּכָל דְּרָכָיו וּֽלְדָבְקָה בֽוֹ:  ת  אֲרֵי אִם מִטַר תִּטְרוּן יָת כָּל תַּפְקֶדְתָּא הָדָא דִי אֲנָא מְפַקֵד יָתְכוֹן לְמֶעְבְּדַהּ לְמִרְחַם יָת יְיָ אֱלָהֲכוֹן לִמְהַךְ בְּכָל אָרְחָן דְתַקְנָן קֳדָמוֹהִי וּלְאִתְקְרָבָא לְדַחַלְתֵּהּ:

{כג} וְהוֹרִישׁ יְהֹוָה אֶת כָּל הַגּוֹיִם הָאֵלֶּה מִלִּפְנֵיכֶם וִֽירִשְׁתֶּם גּוֹיִם גְּדֹלִים וַֽעֲצֻמִים מִכֶּֽם:  מ  וְהוֹרִישׁ יְהֹוָה אֶת כָּל הַגּוֹיִם הָאֵלֶּה מִלִּפְנֵיכֶם וִֽירִשְׁתֶּם גּוֹיִם גְּדֹלִים וַֽעֲצֻמִים מִכֶּֽם:  ת  וִיתָרֵךְ יְיָ יָת כָּל עַמְמַיָא הָאִלֵין מִקֳדָמֵיכוֹן וְתֵירְתוּן עַמְמִין רַבְרְבִין וְתַקִיפִין מִנְכוֹן:

{כד} כָּל הַמָּקוֹם אֲשֶׁר תִּדְרֹךְ כַּֽף רַגְלְכֶם בּוֹ לָכֶם יִֽהְיֶה מִן הַמִּדְבָּר וְהַלְּבָנוֹן מִן הַנָּהָר נְהַר פְּרָת וְעַד הַיָּם הָאַֽחֲרוֹן יִֽהְיֶה גְּבֻֽלְכֶֽם:  מ  כָּל הַמָּקוֹם אֲשֶׁר תִּדְרֹךְ כַּֽף רַגְלְכֶם בּוֹ לָכֶם יִֽהְיֶה מִן הַמִּדְבָּר וְהַלְּבָנוֹן מִן הַנָּהָר נְהַר פְּרָת וְעַד הַיָּם הָאַֽחֲרוֹן יִֽהְיֶה גְּבֻֽלְכֶֽם:  ת  כָּל אַתְרָא דִי תִדְרוֹךְ פַּרְסַת רַגְלְכוֹן בֵּהּ דִלְכוֹן יְהֵי מִן מַדְבְּרָא וְלִבְנָן מִן נַהֲרָא נְהַר פְּרָת וְעַד יַמָא מַעַרְבָא יְהֵי תְחוּמְכוֹן:

{כה} לֹֽא יִתְיַצֵּב אִישׁ בִּפְנֵיכֶם פַּחְדְּכֶם וּמוֹרַֽאֲכֶם יִתֵּן יְהֹוָה אֱלֹֽהֵיכֶם עַל פְּנֵי כָל הָאָרֶץ אֲשֶׁר תִּדְרְכוּ בָהּ כַּֽאֲשֶׁר דִּבֶּר לָכֶֽם:  מ  לֹֽא יִתְיַצֵּב אִישׁ בִּפְנֵיכֶם פַּחְדְּכֶם וּמוֹרַֽאֲכֶם יִתֵּן יְהֹוָה אֱלֹֽהֵיכֶם עַל פְּנֵי כָל הָאָרֶץ אֲשֶׁר תִּדְרְכוּ בָהּ כַּֽאֲשֶׁר דִּבֶּר לָכֶֽם:  ת  לָא יִתְעַתַּד אֱנָשׁ קֳדָמֵיכוֹן דַחֲלָתְכוֹן וְאֵימָתְכוֹן יִתֵּן יְיָ אֱלָהֲכוֹן עַל אַפֵּי כָל אַרְעָא דִי תִדְרְכוּן בַּהּ כְּמָה דִי מַלִיל לְכוֹן: ססס: (ססס)
