פרשת וירא




בראשית פרק-יח

{א} וַיֵּרָ֤א אֵלָיו֙ יְהֹוָ֔ה בְּאֵֽלֹנֵ֖י מַמְרֵ֑א וְה֛וּא יֹשֵׁ֥ב פֶּֽתַח-הָאֹ֖הֶל כְּחֹ֥ם הַיּֽוֹם:  מ  וַיֵּרָ֤א אֵלָיו֙ יְהֹוָ֔ה בְּאֵֽלֹנֵ֖י מַמְרֵ֑א וְה֛וּא יֹשֵׁ֥ב פֶּֽתַח-הָאֹ֖הֶל כְּחֹ֥ם הַיּֽוֹם:  ת  וְאִתְגְּלִי לֵהּ יְיָ בְּמֵישְׁרֵי מַמְרֵא וְהוּא יָתֵב בִּתְרַע מַשְׁכְּנָא כְּמֵיחַם יוֹמָא:

{ב} וַיִּשָּׂ֤א עֵינָיו֙ וַיַּ֔רְא וְהִנֵּה֙ שְׁלֹשָׁ֣ה אֲנָשִׁ֔ים נִצָּבִ֖ים עָלָ֑יו וַיַּ֗רְא וַיָּ֤רָץ לִקְרָאתָם֙ מִפֶּ֣תַח הָאֹ֔הֶל וַיִּשְׁתַּ֖חוּ אָֽרְצָה:  מ  וַיִּשָּׂ֤א עֵינָיו֙ וַיַּ֔רְא וְהִנֵּה֙ שְׁלֹשָׁ֣ה אֲנָשִׁ֔ים נִצָּבִ֖ים עָלָ֑יו וַיַּ֗רְא וַיָּ֤רָץ לִקְרָאתָם֙ מִפֶּ֣תַח הָאֹ֔הֶל וַיִּשְׁתַּ֖חוּ אָֽרְצָה:  ת  וּזְקַף עֵינוֹהִי וַחֲזָא וְהָא תְּלָתָא גֻּבְרִין (נ''י גַּבְרִין) קָיְמִין עִלָּווֹהִי וַחֲזָא וּרְהַט לְקַדָּמוּתְהוֹן מִתְּרַע מַשְׁכְּנָא וּסְגִיד עַל אַרְעָא:

{ג} וַיֹּאמַ֑ר אֲדֹנָ֗י אִם-נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵינֶ֔יךָ אַל-נָ֥א תַֽעֲבֹ֖ר מֵעַ֥ל עַבְדֶּֽךָ:  מ  וַיֹּאמַ֑ר אֲדֹנָ֗י אִם-נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵינֶ֔יךָ אַל-נָ֥א תַֽעֲבֹ֖ר מֵעַ֥ל עַבְדֶּֽךָ:  ת  וַאֲמָר יְיָ אִם כְּעַן אַשְׁכָּחִית רַחֲמִין קֳדָמָךְ (נ''י בְּעֵינָיךְ) לָא כְעַן תְּעִבַּר מֵעַל עַבְדָּךְ:

{ד} יֻקַּח-נָ֣א מְעַט-מַ֔יִם וְרַֽחֲצ֖וּ רַגְלֵיכֶ֑ם וְהִשָּֽׁעֲנ֖וּ תַּ֥חַת הָעֵֽץ:  מ  יֻקַּח-נָ֣א מְעַט-מַ֔יִם וְרַֽחֲצ֖וּ רַגְלֵיכֶ֑ם וְהִשָּֽׁעֲנ֖וּ תַּ֥חַת הָעֵֽץ:  ת  יִסְּבוּן כְּעַן זְעֵיר מַיָּא וְאַסְחוּ רַגְלֵיכוֹן וְאִסְתְּמִיכוּ תְּחוֹת אִילָנָא:

{ה} וְאֶקְחָ֨ה פַת-לֶ֜חֶם וְסַֽעֲד֤וּ לִבְּכֶם֙ אַחַ֣ר תַּֽעֲבֹ֔רוּ כִּֽי-עַל-כֵּ֥ן עֲבַרְתֶּ֖ם עַל-עַבְדְּכֶ֑ם וַיֹּ֣אמְר֔וּ כֵּ֥ן תַּֽעֲשֶׂ֖ה כַּֽאֲשֶׁ֥ר דִּבַּֽרְתָּ:  מ  וְאֶקְחָ֨ה פַת-לֶ֜חֶם וְסַֽעֲד֤וּ לִבְּכֶם֙ אַחַ֣ר תַּֽעֲבֹ֔רוּ כִּֽי-עַל-כֵּ֥ן עֲבַרְתֶּ֖ם עַל-עַבְדְּכֶ֑ם וַיֹּ֣אמְר֔וּ כֵּ֥ן תַּֽעֲשֶׂ֖ה כַּֽאֲשֶׁ֥ר דִּבַּֽרְתָּ:  ת  וְאֶסַּב פִּתָּא דְלַחְמָא וּסְעִידוּ לִבְּכוֹן בָּתַר כֵּן תְּעִבְּרוּן אֲרֵי עַל כֵּן עֲבַרְתּוּן עַל עַבְדְּכוֹן וַאֲמָרוּ כֵּן תַּעְבֵּיד כְּמָא דִּי מַלֵּלְתָּא:

{ו} וַיְמַהֵ֧ר אַבְרָהָ֛ם הָאֹ֖הֱלָה אֶל-שָׂרָ֑ה וַיֹּ֗אמֶר מַֽהֲרִ֞י שְׁלֹ֤שׁ סְאִים֙ קֶ֣מַח סֹ֔לֶת ל֖וּשִׁי וַֽעֲשִׂ֥י עֻגֽוֹת:  מ  וַיְמַהֵ֧ר אַבְרָהָ֛ם הָאֹ֖הֱלָה אֶל-שָׂרָ֑ה וַיֹּ֗אמֶר מַֽהֲרִ֞י שְׁלֹ֤שׁ סְאִים֙ קֶ֣מַח סֹ֔לֶת ל֖וּשִׁי וַֽעֲשִׂ֥י עֻגֽוֹת:  ת  וְאוֹחִי אַבְרָהָם לְמַשְׁכְּנָא לְוַת שָׂרָה וַאֲמַר אוֹחָא תְּלָת סְאִין קִמְחָא דְּסָלְתָּא לוּשִׁי וְעִבִידִי גְּרִיצָן:

{ז} וְאֶל-הַבָּקָ֖ר רָ֣ץ אַבְרָהָ֑ם וַיִּקַּ֨ח בֶּן-בָּקָ֜ר רַ֤ךְ וָטוֹב֙ וַיִּתֵּ֣ן אֶל-הַנַּ֔עַר וַיְמַהֵ֖ר לַֽעֲשׂ֥וֹת אֹתֽוֹ:  מ  וְאֶל-הַבָּקָ֖ר רָ֣ץ אַבְרָהָ֑ם וַיִּקַּ֨ח בֶּן-בָּקָ֜ר רַ֤ךְ וָטוֹב֙ וַיִּתֵּ֣ן אֶל-הַנַּ֔עַר וַיְמַהֵ֖ר לַֽעֲשׂ֥וֹת אֹתֽוֹ:  ת  וּלְוַת תּוֹרֵי רְהַט אַבְרָהָם וּדְבַר בַּר תּוֹרֵי רַכִּיךְ וְטַב וִיהַב לְעוּלֵמָא וְאוֹחִי לְמֶעְבַּד יָתֵהּ:

{ח} וַיִּקַּ֨ח חֶמְאָ֜ה וְחָלָ֗ב וּבֶן-הַבָּקָר֙ אֲשֶׁ֣ר עָשָׂ֔ה וַיִּתֵּ֖ן לִפְנֵיהֶ֑ם וְהֽוּא-עֹמֵ֧ד עֲלֵיהֶ֛ם תַּ֥חַת הָעֵ֖ץ וַיֹּאכֵֽלוּ:  מ  וַיִּקַּ֨ח חֶמְאָ֜ה וְחָלָ֗ב וּבֶן-הַבָּקָר֙ אֲשֶׁ֣ר עָשָׂ֔ה וַיִּתֵּ֖ן לִפְנֵיהֶ֑ם וְהֽוּא-עֹמֵ֧ד עֲלֵיהֶ֛ם תַּ֥חַת הָעֵ֖ץ וַיֹּאכֵֽלוּ:  ת  וּנְסִיב שְׁמַן וַחֲלָב וּבַר תּוֹרֵי דִּי עֲבַד וִיהַב קֳדָמֵיהוֹן וְהוּא מְשַׁמֵּשׁ עִלָּוֵיהוֹן תְּחוֹת אִילָנָא וַאֲכָלוּ:

{ט} וַיֹּֽאמְר֣וּ אֵ֗לָ֔י֗ו֗ אַיֵּ֖ה שָׂרָ֣ה אִשְׁתֶּ֑ךָ וַיֹּ֖אמֶר הִנֵּ֥ה בָאֹֽהֶל:  מ  וַיֹּֽאמְר֣וּ אֵ֗לָ֔י֗ו֗ אַיֵּ֖ה שָׂרָ֣ה אִשְׁתֶּ֑ךָ וַיֹּ֖אמֶר הִנֵּ֥ה בָאֹֽהֶל:  ת  וַאֲמָרוּ לֵהּ אָן שָׂרָה אִתְּתָךְ וַאֲמַר הָא בְמַשְׁכְּנָא:

{י} וַיֹּ֗אמֶר שׁ֣וֹב אָשׁ֤וּב אֵלֶ֨יךָ֙ כָּעֵ֣ת חַיָּ֔ה וְהִנֵּה-בֵ֖ן לְשָׂרָ֣ה אִשְׁתֶּ֑ךָ וְשָׂרָ֥ה שֹׁמַ֛עַת פֶּ֥תַח הָאֹ֖הֶל וְה֥וּא אַֽחֲרָֽיו:  מ  וַיֹּ֗אמֶר שׁ֣וֹב אָשׁ֤וּב אֵלֶ֨יךָ֙ כָּעֵ֣ת חַיָּ֔ה וְהִנֵּה-בֵ֖ן לְשָׂרָ֣ה אִשְׁתֶּ֑ךָ וְשָׂרָ֥ה שֹׁמַ֛עַת פֶּ֥תַח הָאֹ֖הֶל וְה֥וּא אַֽחֲרָֽיו:  ת  וַאֲמַר מִתַּב אִתּוּב לְוָתָךְ כְּעִדָּן דְּאַתּוּן קַיָּמִין וְהָא בַר לְשָׂרָה אִתְּתָךְ וְשָׂרָה שְמַעַת בִּתְרַע מַשְׁכְנָא וְהוּא אֲחוֹרוֹהִי:

{יא} וְאַבְרָהָ֤ם וְשָׂרָה֙ זְקֵנִ֔ים בָּאִ֖ים בַּיָּמִ֑ים חָדַל֙ לִֽהְי֣וֹת לְשָׂרָ֔ה אֹ֖רַח כַּנָּשִֽׁים:  מ  וְאַבְרָהָ֤ם וְשָׂרָה֙ זְקֵנִ֔ים בָּאִ֖ים בַּיָּמִ֑ים חָדַל֙ לִֽהְי֣וֹת לְשָׂרָ֔ה אֹ֖רַח כַּנָּשִֽׁים:  ת  וְאַבְרָהָם וְשָׂרָה סִיבוּ עַלּוּ בְיוֹמִין פְּסַק מִלְּמֶהֱוֵי לְשָׂרָה אוֹרַח כִּנְשַׁיָּא:

{יב} וַתִּצְחַ֥ק שָׂרָ֖ה בְּקִרְבָּ֣הּ לֵאמֹ֑ר אַֽחֲרֵ֤י בְלֹתִי֙ הָֽיְתָה-לִּ֣י עֶדְנָ֔ה וַֽאדֹנִ֖י זָקֵֽן:  מ  וַתִּצְחַ֥ק שָׂרָ֖ה בְּקִרְבָּ֣הּ לֵאמֹ֑ר אַֽחֲרֵ֤י בְלֹתִי֙ הָֽיְתָה-לִּ֣י עֶדְנָ֔ה וַֽאדֹנִ֖י זָקֵֽן:  ת  וְחַיְכַת שָׂרָה בִּמְעָהַהּּ לְמֵימָר בָּתַר דְּסֵיבִית הֲוַת לִי עוּלֵימוּ וְרִבּוֹנִי סִיב:

{יג} וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-אַבְרָהָ֑ם לָ֣מָּה זֶּה֩ צָֽחֲקָ֨ה שָׂרָ֜ה לֵאמֹ֗ר הַאַ֥ף אֻמְנָ֛ם אֵלֵ֖ד וַֽאֲנִ֥י זָקַֽנְתִּי:  מ  וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-אַבְרָהָ֑ם לָ֣מָּה זֶּה֩ צָֽחֲקָ֨ה שָׂרָ֜ה לֵאמֹ֗ר הַאַ֥ף אֻמְנָ֛ם אֵלֵ֖ד וַֽאֲנִ֥י זָקַֽנְתִּי:  ת  וַאֲמַר יְיָ לְאַבְרָהָם לְמָּא דְנָן חַיְכַת שָׂרָה לְמֵימַר הַבְרַם בְּקוּשְׁטָא אוֹלִיד וַאֲנָא סֵיבִית:

{יד} הֲיִפָּלֵ֥א מֵֽיהֹוָ֖ה דָּבָ֑ר לַמּוֹעֵ֞ד אָשׁ֥וּב אֵלֶ֛יךָ כָּעֵ֥ת חַיָּ֖ה וּלְשָׂרָ֥ה בֵֽן:  מ  הֲיִפָּלֵ֥א מֵֽיהֹוָ֖ה דָּבָ֑ר לַמּוֹעֵ֞ד אָשׁ֥וּב אֵלֶ֛יךָ כָּעֵ֥ת חַיָּ֖ה וּלְשָׂרָ֥ה בֵֽן:  ת  הַיִתְכַּסֵּי מִן קֳדָם יְיָ פִּתְגָּמָא לִזְמַן אִתּוּב לְוָתָךְ כְּעִדָּן דְּאַתּוּן קַיָּמִין וּלְשָׂרָה בָר:

{טו}  שני  וַתְּכַחֵ֨שׁ שָׂרָ֧ה | לֵאמֹ֛ר לֹ֥א צָחַ֖קְתִּי כִּ֣י | יָרֵ֑אָה וַיֹּ֥אמֶר | לֹ֖א כִּ֥י צָחָֽקְתְּ:  מ  וַתְּכַחֵ֨שׁ שָׂרָ֧ה | לֵאמֹ֛ר לֹ֥א צָחַ֖קְתִּי כִּ֣י | יָרֵ֑אָה וַיֹּ֥אמֶר | לֹ֖א כִּ֥י צָחָֽקְתְּ:  ת  וְכַדִּיבַת שָׂרָה לְמֵימָר לָא חַיֵּכִית אֲרֵי דְחֵלַת וַאֲמַר לָא בְּרַם חַיָּכְתְּ:

{טז} וַיָּקֻ֤מוּ מִשָּׁם֙ הָֽאֲנָשִׁ֔ים וַיַּשְׁקִ֖פוּ עַל-פְּנֵ֣י סְדֹ֑ם וְאַ֨בְרָהָ֔ם הֹלֵ֥ךְ עִמָּ֖ם לְשַׁלְּחָֽם:  מ  וַיָּקֻ֤מוּ מִשָּׁם֙ הָֽאֲנָשִׁ֔ים וַיַּשְׁקִ֖פוּ עַל-פְּנֵ֣י סְדֹ֑ם וְאַ֨בְרָהָ֔ם הֹלֵ֥ךְ עִמָּ֖ם לְשַׁלְּחָֽם:  ת  וְקָמוּ מִתַּמָּן גֻּבְרַיָּא וְאִסְתְּכִיאוּ עַל אַפֵּי סְדוֹם וְאַבְרָהָם אָזַל עִמְּהוֹן לְאַלְווֹאֵיהוֹן:

{יז} וַֽיהֹוָ֖ה אָמָ֑ר הַֽמְכַסֶּ֤ה אֲנִי֙ מֵֽאַבְרָהָ֔ם אֲשֶׁ֖ר אֲנִ֥י עֹשֶֽׂה:  מ  וַֽיהֹוָ֖ה אָמָ֑ר הַֽמְכַסֶּ֤ה אֲנִי֙ מֵֽאַבְרָהָ֔ם אֲשֶׁ֖ר אֲנִ֥י עֹשֶֽׂה:  ת  וַיְיָ אֲמָר הַמְכַסֵּי אֲנָא מֵאַבְרָהָם דִּי אֲנָא עָבֵד:

{יח} וְאַ֨בְרָהָ֔ם הָי֧וֹ יִֽהְיֶ֛ה לְג֥וֹי גָּד֖וֹל וְעָצ֑וּם וְנִ֨בְרְכוּ ב֔וֹ כֹּ֖ל גּוֹיֵ֥י הָאָֽרֶץ:  מ  וְאַ֨בְרָהָ֔ם הָי֧וֹ יִֽהְיֶ֛ה לְג֥וֹי גָּד֖וֹל וְעָצ֑וּם וְנִ֨בְרְכוּ ב֔וֹ כֹּ֖ל גּוֹיֵ֥י הָאָֽרֶץ:  ת  וְאַבְרָהָם מֵהֲוָה יֶהֱוֵי לְעַם סַגִּי וְתַקִּיף וְיִתְבָּרְכוּן בְּדִילֵהּ כֹּל עַמְמֵי אַרְעָא:

{יט} כִּ֣י יְדַעְתִּ֗יו לְמַ֩עַן֩ אֲשֶׁ֨ר יְצַוֶּ֜ה אֶת-בָּנָ֤יו וְאֶת-בֵּיתוֹ֙ אַֽחֲרָ֔יו וְשָֽׁמְרוּ֙ דֶּ֣רֶךְ יְהֹוָ֔ה לַֽעֲשׂ֥וֹת צְדָקָ֖ה וּמִשְׁפָּ֑ט לְמַ֗עַן הָבִ֤יא יְהֹוָה֙ עַל-אַבְרָהָ֔ם אֵ֥ת אֲשֶׁר-דִּבֶּ֖ר עָלָֽיו:  מ  כִּ֣י יְדַעְתִּ֗יו לְמַ֩עַן֩ אֲשֶׁ֨ר יְצַוֶּ֜ה אֶת-בָּנָ֤יו וְאֶת-בֵּיתוֹ֙ אַֽחֲרָ֔יו וְשָֽׁמְרוּ֙ דֶּ֣רֶךְ יְהֹוָ֔ה לַֽעֲשׂ֥וֹת צְדָקָ֖ה וּמִשְׁפָּ֑ט לְמַ֗עַן הָבִ֤יא יְהֹוָה֙ עַל-אַבְרָהָ֔ם אֵ֥ת אֲשֶׁר-דִּבֶּ֖ר עָלָֽיו:  ת  אֲרֵי גְּלִי קֳדָמַי (גי' רמב''ן יְדַעְתִּנֵּהּ) בְּדִיל דִּי יְפַקֵּד יָת בְּנוֹהִי וְיָת אֱנַשׁ בֵּיתֵהּ בַּתְרוֹהִי וְיִטְּרוּן אָרְחָן דְּתַקְּנָן קֳדָם יְיָ לְמֶעֱבַּד צְדַקְתָּא וְדִינָא בְּדִיל אַיְיתִי יְיָ עַל אַבְרָהָם יָת דְּמַלֵּל עֲלוֹהִי:

{כ} וַיֹּ֣אמֶר יְהֹוָ֔ה זַֽעֲקַ֛ת סְדֹ֥ם וַֽעֲמֹרָ֖ה כִּי-רָ֑בָּה וְחַ֨טָּאתָ֔ם כִּ֥י כָֽבְדָ֖ה מְאֹֽד:  מ  וַיֹּ֣אמֶר יְהֹוָ֔ה זַֽעֲקַ֛ת סְדֹ֥ם וַֽעֲמֹרָ֖ה כִּי-רָ֑בָּה וְחַ֨טָּאתָ֔ם כִּ֥י כָֽבְדָ֖ה מְאֹֽד:  ת  וַאֲמַר יְיָ קְבֵלַת סְדוֹם וַעֲמוֹרָה אֲרֵי סְגִיאַת וְחוֹבָתְהוֹן אֲרֵי תְקִיפַת לַחֲדָא:

{כא} אֵֽרְדָה-נָּ֣א וְאֶרְאֶ֔ה הַכְּצַֽעֲקָתָ֛הּ הַבָּ֥אָה אֵלַ֖י עָשׂ֣וּ | כָּלָ֑ה וְאִם-לֹ֖א אֵדָֽעָה:  מ  אֵֽרְדָה-נָּ֣א וְאֶרְאֶ֔ה הַכְּצַֽעֲקָתָ֛הּ הַבָּ֥אָה אֵלַ֖י עָשׂ֣וּ | כָּלָ֑ה וְאִם-לֹ֖א אֵדָֽעָה:  ת  אִתְגְּלֵי כְעַן וְאֶדּוּן הֲכִי קְבִלְתְּהוֹן דְּעַלַּת לָקֳדָמַי עֲבָדוּ אֶעְבֵּד עִמְּהוֹן גְּמִירָא אִם לָא תָיְבִין וְאִם תָּיְבִין לָא אֶתְפְּרָע:

{כב} וַיִּפְנ֤וּ מִשָּׁם֙ הָֽאֲנָשִׁ֔ים וַיֵּֽלְכ֖וּ סְדֹ֑מָה וְאַ֨בְרָהָ֔ם עוֹדֶ֥נּוּ עֹמֵ֖ד לִפְנֵ֥י יְהֹוָֽה:  מ  וַיִּפְנ֤וּ מִשָּׁם֙ הָֽאֲנָשִׁ֔ים וַיֵּֽלְכ֖וּ סְדֹ֑מָה וְאַ֨בְרָהָ֔ם עוֹדֶ֥נּוּ עֹמֵ֖ד לִפְנֵ֥י יְהֹוָֽה:  ת  וְאִתְפְּנִיאוּ מִתַּמָּן גֻּבְרַיָּא וַאֲזָלוּ לִסְדוֹם וְאַבְרָהָם עַד כְּעַן מְשַׁמֵּשׁ בִּצְלוֹ קֳדָם יְיָ:

{כג} וַיִּגַּ֥שׁ אַבְרָהָ֖ם וַיֹּאמַ֑ר הַאַ֣ף תִּסְפֶּ֔ה צַדִּ֖יק עִם-רָשָֽׁע:  מ  וַיִּגַּ֥שׁ אַבְרָהָ֖ם וַיֹּאמַ֑ר הַאַ֣ף תִּסְפֶּ֔ה צַדִּ֖יק עִם-רָשָֽׁע:  ת  וּקְרֵב אַבְרָהָם וַאֲמָר הֲבִרְגַז תְּשֵׁיצֵי זַכָּאָה עִם חַיָּבָא:

{כד} אוּלַ֥י יֵ֛שׁ חֲמִשִּׁ֥ים צַדִּיקִ֖ם בְּת֣וֹךְ הָעִ֑יר הַאַ֤ף תִּסְפֶּה֙ וְלֹֽא-תִשָּׂ֣א לַמָּק֔וֹם לְמַ֛עַן חֲמִשִּׁ֥ים הַצַּדִּיקִ֖ם אֲשֶׁ֥ר בְּקִרְבָּֽהּ:  מ  אוּלַ֥י יֵ֛שׁ חֲמִשִּׁ֥ים צַדִּיקִ֖ם בְּת֣וֹךְ הָעִ֑יר הַאַ֤ף תִּסְפֶּה֙ וְלֹֽא-תִשָּׂ֣א לַמָּק֔וֹם לְמַ֛עַן חֲמִשִּׁ֥ים הַצַּדִּיקִ֖ם אֲשֶׁ֥ר בְּקִרְבָּֽהּ:  ת  מָאִים אִית חַמְּשִׁין זַכָּאִין בְּגוֹ קַרְתָּא הֲבִרְגַז תְּשֵׁיצֵי וְלָא תִשְׁבּוֹק לְאַתְרָא בְּדִיל חַמְשִׁין זַכָּאִין דִּי בְגַוָּהּ:

{כה} חָלִ֨לָה לְּךָ֜ מֵֽעֲשׂ֣ת | כַּדָּבָ֣ר הַזֶּ֗ה לְהָמִ֤ית צַדִּיק֙ עִם-רָשָׁ֔ע וְהָיָ֥ה כַצַּדִּ֖יק כָּֽרָשָׁ֑ע חָלִ֣לָה לָּ֔ךְ הֲשֹׁפֵט֙ כָּל-הָאָ֔רֶץ לֹ֥א יַֽעֲשֶׂ֖ה מִשְׁפָּֽט:  מ  חָלִ֨לָה לְּךָ֜ מֵֽעֲשׂ֣ת | כַּדָּבָ֣ר הַזֶּ֗ה לְהָמִ֤ית צַדִּיק֙ עִם-רָשָׁ֔ע וְהָיָ֥ה כַצַּדִּ֖יק כָּֽרָשָׁ֑ע חָלִ֣לָה לָּ֔ךְ הֲשֹׁפֵט֙ כָּל-הָאָ֔רֶץ לֹ֥א יַֽעֲשֶׂ֖ה מִשְׁפָּֽט:  ת  קוּשְׁטָא אִנּוּן דִּינָךְ מִלְּמֶעְבַּד כְּפִתְגָּמָא הָדֵין לְקַטָּלָא זַכָּאָה עִם חַיָּבָא וִיהֵי זַכָּאָה כְּחַיָּבָא קוּשְׁטָא אִנּוּן דִּינָךְ דְּדָיִן כָּל אַרְעָא לָא יַעְבֵד דִּינָא:

{כו} וַיֹּ֣אמֶר יְהֹוָ֔ה אִם-אֶמְצָ֥א בִסְדֹ֛ם חֲמִשִּׁ֥ים צַדִּיקִ֖ם בְּת֣וֹךְ הָעִ֑יר וְנָשָׂ֥אתִי לְכָל-הַמָּק֖וֹם בַּֽעֲבוּרָֽם:  מ  וַיֹּ֣אמֶר יְהֹוָ֔ה אִם-אֶמְצָ֥א בִסְדֹ֛ם חֲמִשִּׁ֥ים צַדִּיקִ֖ם בְּת֣וֹךְ הָעִ֑יר וְנָשָׂ֥אתִי לְכָל-הַמָּק֖וֹם בַּֽעֲבוּרָֽם:  ת  וַאֲמַר יְיָ אִם אַשְׁכַּח בִּסְדוֹם חַמְשִׁין זַכָּאִין בְּגוֹ קַרְתָּא וְאֶשְׁבּוֹק לְכָל אַתְרָא בְּדִילְהוֹן:

{כז} וַיַּ֥עַן אַבְרָהָ֖ם וַיֹּאמַ֑ר הִנֵּה-נָ֤א הוֹאַ֨לְתִּי֙ לְדַבֵּ֣ר אֶל-אֲדֹנָ֔י וְאָֽנֹכִ֖י עָפָ֥ר וָאֵֽפֶר:  מ  וַיַּ֥עַן אַבְרָהָ֖ם וַיֹּאמַ֑ר הִנֵּה-נָ֤א הוֹאַ֨לְתִּי֙ לְדַבֵּ֣ר אֶל-אֲדֹנָ֔י וְאָֽנֹכִ֖י עָפָ֥ר וָאֵֽפֶר:  ת  וַאֲתֵב אַבְרָהָם וַאֲמַר הָא כְעַן שָׁרֵיתִי לְמַלָּלָא קֳדָם יְיָ וַאֲנָא עָפָר וּקְטָם:

{כח} א֠וּלַ֠י יַחְסְר֞וּן חֲמִשִּׁ֤ים הַצַּדִּיקִם֙ חֲמִשָּׁ֔ה הֲתַשְׁחִ֥ית בַּֽחֲמִשָּׁ֖ה אֶת-כָּל-הָעִ֑יר וַיֹּ֨אמֶר֙ לֹ֣א אַשְׁחִ֔ית אִם-אֶמְצָ֣א שָׁ֔ם אַרְבָּעִ֖ים וַֽחֲמִשָּֽׁה:  מ  א֠וּלַ֠י יַחְסְר֞וּן חֲמִשִּׁ֤ים הַצַּדִּיקִם֙ חֲמִשָּׁ֔ה הֲתַשְׁחִ֥ית בַּֽחֲמִשָּׁ֖ה אֶת-כָּל-הָעִ֑יר וַיֹּ֨אמֶר֙ לֹ֣א אַשְׁחִ֔ית אִם-אֶמְצָ֣א שָׁ֔ם אַרְבָּעִ֖ים וַֽחֲמִשָּֽׁה:  ת  מָאִים יַחְסְרוּן חַמְשִׁין זַכָּאִין חַמְשָׁא הַתְחַבֵּל בְּחַמְשָׁא יָת כָּל קַרְתָּא וַאֲמַר לָא אֲחַבֵּל אִם אַשְׁכַּח תַּמָּן אַרְבְּעִין וְחַמְשָׁא:

{כט} וַיֹּ֨סֶף ע֜וֹד לְדַבֵּ֤ר אֵלָיו֙ וַיֹּאמַ֔ר אוּלַ֛י יִמָּֽצְא֥וּן שָׁ֖ם אַרְבָּעִ֑ים וַיֹּ֨אמֶר֙ לֹ֣א אֶֽעֱשֶׂ֔ה בַּֽעֲב֖וּר הָֽאַרְבָּעִֽים:  מ  וַיֹּ֨סֶף ע֜וֹד לְדַבֵּ֤ר אֵלָיו֙ וַיֹּאמַ֔ר אוּלַ֛י יִמָּֽצְא֥וּן שָׁ֖ם אַרְבָּעִ֑ים וַיֹּ֨אמֶר֙ לֹ֣א אֶֽעֱשֶׂ֔ה בַּֽעֲב֖וּר הָֽאַרְבָּעִֽים:  ת  וְאוֹסִף עוֹד לְמַלָּלָא קֳדָמוֹהִי וַאֲמַר מָאִים יִשְׁתַּכְחוּן תַּמָּן אַרְבְּעִין וַאֲמַר לָא אֶעְבֵּד גְּמֵרָא בְּדִיל אַרְבְּעִין:

{ל} וַ֠יֹּ֠אמֶר אַל-נָ֞א יִ֤חַר לַֽאדֹנָי֙ וַֽאֲדַבֵּ֔רָה אוּלַ֛י יִמָּֽצְא֥וּן שָׁ֖ם שְׁלֹשִׁ֑ים וַיֹּ֨אמֶר֙ לֹ֣א אֶֽעֱשֶׂ֔ה אִם-אֶמְצָ֥א שָׁ֖ם שְׁלֹשִֽׁים:  מ  וַ֠יֹּ֠אמֶר אַל-נָ֞א יִ֤חַר לַֽאדֹנָי֙ וַֽאֲדַבֵּ֔רָה אוּלַ֛י יִמָּֽצְא֥וּן שָׁ֖ם שְׁלֹשִׁ֑ים וַיֹּ֨אמֶר֙ לֹ֣א אֶֽעֱשֶׂ֔ה אִם-אֶמְצָ֥א שָׁ֖ם שְׁלֹשִֽׁים:  ת  וַאֲמַר לָא כְעַן יִתְקֵף קֳדָם יְיָ (נ''י רוּגְזָא דַּיְיָ) וַאֲמַלֵּל מָאִים יִשְׁתַּכְּחוּן תַּמָּן תְּלָתִין וַאֲמַר לָא אֶעְבֵּד גְּמֵרָא אִם אַשְׁכַּח תַּמָּן תְּלָתִין:

{לא} וַיֹּ֗אמֶר הִנֵּה-נָ֤א הוֹאַ֨לְתִּי֙ לְדַבֵּ֣ר אֶל-אֲדֹנָ֔י אוּלַ֛י יִמָּֽצְא֥וּן שָׁ֖ם עֶשְׂרִ֑ים וַיֹּ֨אמֶר֙ לֹ֣א אַשְׁחִ֔ית בַּֽעֲב֖וּר הָֽעֶשְׂרִֽים:  מ  וַיֹּ֗אמֶר הִנֵּה-נָ֤א הוֹאַ֨לְתִּי֙ לְדַבֵּ֣ר אֶל-אֲדֹנָ֔י אוּלַ֛י יִמָּֽצְא֥וּן שָׁ֖ם עֶשְׂרִ֑ים וַיֹּ֨אמֶר֙ לֹ֣א אַשְׁחִ֔ית בַּֽעֲב֖וּר הָֽעֶשְׂרִֽים:  ת  וַאֲמַר הָא כְעַן שָׁרֵיתִי לְמַלָּלָא קֳדָם יְיָ מָאִים יִשְׁתַּכְּחוּן תַּמָּן עֶשְרִין וַאֲמַר לָא אֲחַבֵּל בְּדִיל עֶשְׁרִין:

{לב} וַ֠יֹּ֠אמֶר אַל-נָ֞א יִ֤חַר לַֽאדֹנָי֙ וַֽאֲדַבְּרָ֣ה אַךְ-הַפַּ֔עַם אוּלַ֛י יִמָּֽצְא֥וּן שָׁ֖ם עֲשָׂרָ֑ה וַיֹּ֨אמֶר֙ לֹ֣א אַשְׁחִ֔ית בַּֽעֲב֖וּר הָֽעֲשָׂרָֽה:  מ  וַ֠יֹּ֠אמֶר אַל-נָ֞א יִ֤חַר לַֽאדֹנָי֙ וַֽאֲדַבְּרָ֣ה אַךְ-הַפַּ֔עַם אוּלַ֛י יִמָּֽצְא֥וּן שָׁ֖ם עֲשָׂרָ֑ה וַיֹּ֨אמֶר֙ לֹ֣א אַשְׁחִ֔ית בַּֽעֲב֖וּר הָֽעֲשָׂרָֽה:  ת  וַאֲמַר לָא כְעַן יִתְקֵף קֳדָם יְיָ (נ''י רוּגְזָא דַּיְיָ) וֶאֱמַלֵּל בְּרַם זִמְנָא הָדָא מָאִים יִשְׁתַּכְּחוּן תַּמָּן עַשְׂרָא וַאֲמַר לָא אֲחַבֵּל בְּדִיל עַשְׂרָא:

{לג} וַיֵּ֣לֶךְ יְהֹוָ֔ה כַּֽאֲשֶׁ֣ר כִּלָּ֔ה לְדַבֵּ֖ר אֶל-אַבְרָהָ֑ם וְאַבְרָהָ֖ם שָׁ֥ב לִמְקֹמֽוֹ:  מ  וַיֵּ֣לֶךְ יְהֹוָ֔ה כַּֽאֲשֶׁ֣ר כִּלָּ֔ה לְדַבֵּ֖ר אֶל-אַבְרָהָ֑ם וְאַבְרָהָ֖ם שָׁ֥ב לִמְקֹמֽוֹ:  ת  וְאִסְתַּלַּק יְקָרָא דַּיְיָ כַּד שֵׁצִי לְמַלָּלָא עִם אַבְרָהָם וְאַבְרָהָם תָּב לְאַתְרֵהּ:




בראשית פרק-יט

{א}  שלישי  וַ֠יָּבֹ֠אוּ שְׁנֵ֨י הַמַּלְאָכִ֤ים סְדֹ֨מָה֙ בָּעֶ֔רֶב וְל֖וֹט יֹשֵׁ֣ב בְּשַֽׁעַר-סְדֹ֑ם וַיַּרְא-לוֹט֙ וַיָּ֣קָם לִקְרָאתָ֔ם וַיִּשְׁתַּ֥חוּ אַפַּ֖יִם אָֽרְצָה:  מ  וַ֠יָּבֹ֠אוּ שְׁנֵ֨י הַמַּלְאָכִ֤ים סְדֹ֨מָה֙ בָּעֶ֔רֶב וְל֖וֹט יֹשֵׁ֣ב בְּשַֽׁעַר-סְדֹ֑ם וַיַּרְא-לוֹט֙ וַיָּ֣קָם לִקְרָאתָ֔ם וַיִּשְׁתַּ֥חוּ אַפַּ֖יִם אָֽרְצָה:  ת  וְעַלּוּ תְּרֵין מַלְאָכַיָּא לִסְדוֹם בְּרַמְשָׁא וְלוֹט יָתֵב בְּתַרְעָא (נ''י בִּתְרַע) דִּסְדוֹם וַחֲזָא לוֹט וְקָם לְקַדָּמוּתְהוֹן וּסְגִיד עַל אַפּוֹהִי עַל אַרְעָא:

{ב} וַיֹּ֜אמֶר הִנֶּ֣ה נָּֽא-אֲדֹנַ֗י סוּרוּ נָ֠א אֶל-בֵּ֨ית עַבְדְּכֶ֤ם וְלִ֨ינוּ֙ וְרַֽחֲצ֣וּ רַגְלֵיכֶ֔ם וְהִשְׁכַּמְתֶּ֖ם וַֽהֲלַכְתֶּ֣ם לְדַרְכְּכֶ֑ם וַיֹּֽאמְר֣וּ לֹּ֔א כִּ֥י בָֽרְח֖וֹב נָלִֽין:  מ  וַיֹּ֜אמֶר הִנֶּ֣ה נָּֽא-אֲדֹנַ֗י סוּרוּ נָ֠א אֶל-בֵּ֨ית עַבְדְּכֶ֤ם וְלִ֨ינוּ֙ וְרַֽחֲצ֣וּ רַגְלֵיכֶ֔ם וְהִשְׁכַּמְתֶּ֖ם וַֽהֲלַכְתֶּ֣ם לְדַרְכְּכֶ֑ם וַיֹּֽאמְר֣וּ לֹּ֔א כִּ֥י בָֽרְח֖וֹב נָלִֽין:  ת  וַאֲמַר בְּבָעוּ כְעַן רִבּוֹנַי זוּרוּ כְעַן לְבֵית עַבְדְּכוֹן וּבִיתוּ וְאַסְחוּ רַגְלֵיכוֹן וּתְקַדְּמוּן וּתְהָכוּן לְאָרְחֲכוֹן וְאֲמָרוּ לָא אֶלָּהֵן בִּרְחוֹבָא נְבִית:

{ג} וַיִּפְצַר-בָּ֣ם מְאֹ֔ד וַיָּסֻ֣רוּ אֵלָ֔יו וַיָּבֹ֖אוּ אֶל-בֵּית֑וֹ וַיַּ֤עַשׂ לָהֶם֙ מִשְׁתֶּ֔ה וּמַצּ֥וֹת אָפָ֖ה וַיֹּאכֵֽלוּ:  מ  וַיִּפְצַר-בָּ֣ם מְאֹ֔ד וַיָּסֻ֣רוּ אֵלָ֔יו וַיָּבֹ֖אוּ אֶל-בֵּית֑וֹ וַיַּ֤עַשׂ לָהֶם֙ מִשְׁתֶּ֔ה וּמַצּ֥וֹת אָפָ֖ה וַיֹּאכֵֽלוּ:  ת  וְאַתְקַף בְּהוֹן לַחֲדָא וְזָרוּ לְוָתֵהּ וְעַלּוּ לְבֵיתֵהּ וַעֲבַד לְהוֹן מִשְׁתְּיָא וּפַטִּיר אֲפָא לְהוֹן וְאֲכָלוּ:

{ד} טֶ֘רֶם֘ יִשְׁכָּ֒בוּ֒ וְאַנְשֵׁ֨י הָעִ֜יר אַנְשֵׁ֤י סְדֹם֙ נָסַ֣בּוּ עַל-הַבַּ֔יִת מִנַּ֖עַר וְעַד-זָקֵ֑ן כָּל-הָעָ֖ם מִקָּצֶֽה:  מ  טֶ֘רֶם֘ יִשְׁכָּ֒בוּ֒ וְאַנְשֵׁ֨י הָעִ֜יר אַנְשֵׁ֤י סְדֹם֙ נָסַ֣בּוּ עַל-הַבַּ֔יִת מִנַּ֖עַר וְעַד-זָקֵ֑ן כָּל-הָעָ֖ם מִקָּצֶֽה:  ת  עַד לָא שְׁכִיבוּ וְאֲנָשֵׁי קַרְתָּא אֲנָשֵׁי סְדוֹם אַקִּיפוּ עַל בֵּיתָא מֵעוּלֵימָא וְעַד סָבָא כָּל עַמָּא מִסּוֹפֵהּ:

{ה} וַיִּקְרְא֤וּ אֶל-לוֹט֙ וַיֹּ֣אמְרוּ ל֔וֹ אַיֵּ֧ה הָֽאֲנָשִׁ֛ים אֲשֶׁר-בָּ֥אוּ אֵלֶ֖יךָ הַלָּ֑יְלָה הֽוֹצִיאֵ֣ם אֵלֵ֔ינוּ וְנֵֽדְעָ֖ה אֹתָֽם:  מ  וַיִּקְרְא֤וּ אֶל-לוֹט֙ וַיֹּ֣אמְרוּ ל֔וֹ אַיֵּ֧ה הָֽאֲנָשִׁ֛ים אֲשֶׁר-בָּ֥אוּ אֵלֶ֖יךָ הַלָּ֑יְלָה הֽוֹצִיאֵ֣ם אֵלֵ֔ינוּ וְנֵֽדְעָ֖ה אֹתָֽם:  ת  וּקְרוֹ לְלוֹט וַאֲמָרוּ לֵהּ אָן גֻּבְרַיָא דִּי אֲתוֹ לְוָתָךְ לֵילְיָא אַפֵּקִנּוּן לְוָתָנָא וְנִדַּע יָתְהוֹן:

{ו} וַיֵּצֵ֧א אֲלֵהֶ֛ם ל֖וֹט הַפֶּ֑תְחָה וְהַדֶּ֖לֶת סָגַ֥ר אַֽחֲרָֽיו:  מ  וַיֵּצֵ֧א אֲלֵהֶ֛ם ל֖וֹט הַפֶּ֑תְחָה וְהַדֶּ֖לֶת סָגַ֥ר אַֽחֲרָֽיו:  ת  וּנְפַק לְוָתְהוֹן לוֹט לְתַרְעָא וְדָשָׁא אֲחַד בַּתְרוֹהִי:

{ז} וַיֹּאמַ֑ר אַל-נָ֥א אַחַ֖י תָּרֵֽעוּ:  מ  וַיֹּאמַ֑ר אַל-נָ֥א אַחַ֖י תָּרֵֽעוּ:  ת  וַאֲמַר בְּבָעוּ כְעַן אַחַי לָא תַבְאִישוּן:

{ח} הִנֵּה-נָ֨א לִ֜י שְׁתֵּ֣י בָנ֗וֹת אֲשֶׁ֤ר לֹא-יָֽדְעוּ֙ אִ֔ישׁ אוֹצִֽיאָה-נָּ֤א אֶתְהֶן֙ אֲלֵיכֶ֔ם וַֽעֲשׂ֣וּ לָהֶ֔ן כַּטּ֖וֹב בְּעֵֽינֵיכֶ֑ם רַ֠ק לָֽאֲנָשִׁ֤ים הָאֵל֙ אַל-תַּֽעֲשׂ֣וּ דָבָ֔ר כִּֽי-עַל-כֵּ֥ן בָּ֖אוּ בְּצֵ֥ל קֹֽרָתִֽי:  מ  הִנֵּה-נָ֨א לִ֜י שְׁתֵּ֣י בָנ֗וֹת אֲשֶׁ֤ר לֹא-יָֽדְעוּ֙ אִ֔ישׁ אוֹצִֽיאָה-נָּ֤א אֶתְהֶן֙ אֲלֵיכֶ֔ם וַֽעֲשׂ֣וּ לָהֶ֔ן כַּטּ֖וֹב בְּעֵֽינֵיכֶ֑ם רַ֠ק לָֽאֲנָשִׁ֤ים הָאֵל֙ אַל-תַּֽעֲשׂ֣וּ דָבָ֔ר כִּֽי-עַל-כֵּ֥ן בָּ֖אוּ בְּצֵ֥ל קֹֽרָתִֽי:  ת  הָא כְעַן לִי תַּרְתֵּין בְּנָן דִּי לָא יְדָעִנּוּן גְּבַר אַפֵּק כְּעַן יַתְהוֹן לְוָתְכוֹן וְעִיבִידוּ לְהֵן כִּדְתַקֵּן בְּעֵינֵיכוֹן לְחוֹד לְגֻבְרַיָּא הָאִלֵּין לָא תַעְבְּדוּן מִדַּעַם אֲרֵי עַל כֵּן עַלּוּ בִּטְלַל שֵׁרוּתִי:

{ט} וַיֹּֽאמְר֣וּ | גֶּשׁ-הָ֗לְאָה וַיֹּֽאמְרוּ֙ הָֽאֶחָ֤ד בָּֽא-לָגוּר֙ וַיִּשְׁפֹּ֣ט שָׁפ֔וֹט עַתָּ֕ה נָרַ֥ע לְךָ֖ מֵהֶ֑ם וַיִּפְצְר֨וּ בָאִ֤ישׁ בְּלוֹט֙ מְאֹ֔ד וַיִּגְּשׁ֖וּ לִשְׁבֹּ֥ר הַדָּֽלֶת:  מ  וַיֹּֽאמְר֣וּ | גֶּשׁ-הָ֗לְאָה וַיֹּֽאמְרוּ֙ הָֽאֶחָ֤ד בָּֽא-לָגוּר֙ וַיִּשְׁפֹּ֣ט שָׁפ֔וֹט עַתָּ֕ה נָרַ֥ע לְךָ֖ מֵהֶ֑ם וַיִּפְצְר֨וּ בָאִ֤ישׁ בְּלוֹט֙ מְאֹ֔ד וַיִּגְּשׁ֖וּ לִשְׁבֹּ֥ר הַדָּֽלֶת:  ת  וַאֲמָרוּ קְרֵב לְהַלָּא וַאֲמָרוּ חַד אֲתָא לְאִתּוֹתָבָא והָא דָיִין דִּינָא כְּעַן נַבְאֵשׁ לָךְ מִדִּילְהוֹן וּתְקִיפוּ בְגַבְרָא בְלוֹט לַחֲדָא וּקְרִיבוּ לְמִתְּבַר דָּשָׁא:

{י} וַיִּשְׁלְח֤וּ הָֽאֲנָשִׁים֙ אֶת-יָדָ֔ם וַיָּבִ֧יאוּ אֶת-ל֛וֹט אֲלֵיהֶ֖ם הַבָּ֑יְתָה וְאֶת-הַדֶּ֖לֶת סָגָֽרוּ:  מ  וַיִּשְׁלְח֤וּ הָֽאֲנָשִׁים֙ אֶת-יָדָ֔ם וַיָּבִ֧יאוּ אֶת-ל֛וֹט אֲלֵיהֶ֖ם הַבָּ֑יְתָה וְאֶת-הַדֶּ֖לֶת סָגָֽרוּ:  ת  וְאוֹשִׁיטוּ גֻּבְרַיָּא יָת יְדֵיהוֹן וְאַיְתִיאוּ יָת לוֹט לְוָתְהוֹן לְבֵיתָא וְיָת דָּשָׁא אֲחָדוּ:

{יא} וְאֶת-הָֽאֲנָשִׁ֞ים אֲשֶׁר-פֶּ֣תַח הַבַּ֗יִת הִכּוּ֙ בַּסַּנְוֵרִ֔ים מִקָּטֹ֖ן וְעַד-גָּד֑וֹל וַיִּלְא֖וּ לִמְצֹ֥א הַפָּֽתַח:  מ  וְאֶת-הָֽאֲנָשִׁ֞ים אֲשֶׁר-פֶּ֣תַח הַבַּ֗יִת הִכּוּ֙ בַּסַּנְוֵרִ֔ים מִקָּטֹ֖ן וְעַד-גָּד֑וֹל וַיִּלְא֖וּ לִמְצֹ֥א הַפָּֽתַח:  ת  וְיָת גֻּבְרַיָּא דִּי בִתְרַע בֵּיתָא מְחוֹ בְּשַׁבְרִירַיָּא מִזְּעֵירָא וְעַד רַבָּא וּלְאִיוּ לְאַשְׁכָּחָא תַרְעָא:

{יב} וַיֹּֽאמְר֨וּ הָֽאֲנָשִׁ֜ים אֶל-ל֗וֹט עֹ֚ד מִֽי-לְךָ֣ פֹ֔ה חָתָן֙ וּבָנֶ֣יךָ וּבְנֹתֶ֔יךָ וְכֹ֥ל אֲשֶׁר-לְךָ֖ בָּעִ֑יר הוֹצֵ֖א מִן-הַמָּקֽוֹם:  מ  וַיֹּֽאמְר֨וּ הָֽאֲנָשִׁ֜ים אֶל-ל֗וֹט עֹ֚ד מִֽי-לְךָ֣ פֹ֔ה חָתָן֙ וּבָנֶ֣יךָ וּבְנֹתֶ֔יךָ וְכֹ֥ל אֲשֶׁר-לְךָ֖ בָּעִ֑יר הוֹצֵ֖א מִן-הַמָּקֽוֹם:  ת  וַאֲמָרוּ גֻּבְרַיָּא לְלוֹט עוֹד מָן לָךְ הָכָא חַתְנָא וּבְנָךְ וּבְנָתָךְ וְכֹל דִּי לָךְ בְּקַרְתָּא אַפֵּיק מִן אַתְרָא:

{יג} כִּֽי-מַשְׁחִתִ֣ים אֲנַ֔חְנוּ אֶת-הַמָּק֖וֹם הַזֶּ֑ה כִּי-גָֽדְלָ֤ה צַֽעֲקָתָם֙ אֶת-פְּנֵ֣י יְהֹוָ֔ה וַיְשַׁלְּחֵ֥נוּ יְהֹוָ֖ה לְשַֽׁחֲתָֽהּ:  מ  כִּֽי-מַשְׁחִתִ֣ים אֲנַ֔חְנוּ אֶת-הַמָּק֖וֹם הַזֶּ֑ה כִּי-גָֽדְלָ֤ה צַֽעֲקָתָם֙ אֶת-פְּנֵ֣י יְהֹוָ֔ה וַיְשַׁלְּחֵ֥נוּ יְהֹוָ֖ה לְשַֽׁחֲתָֽהּ:  ת  אֲרֵי מְחַבְּלִין אֲנַחְנָא יָת אַתְרָא הָדֵין אֲרֵי סְגִיאַת קְבִלְתְּהוֹן קֳדָם יְיָ וְשַׁלְּחָנָא יְיָ לְחַבָּלוּתַהּ:

{יד} וַיֵּצֵ֨א ל֜וֹט וַיְדַבֵּ֣ר | אֶל-חֲתָנָ֣יו | לֹֽקְחֵ֣י בְנֹתָ֗יו וַיֹּ֨אמֶר֙ ק֤וּמוּ צְּאוּ֙ מִן-הַמָּק֣וֹם הַזֶּ֔ה כִּֽי-מַשְׁחִ֥ית יְהֹוָ֖ה אֶת-הָעִ֑יר וַיְהִ֥י כִמְצַחֵ֖ק בְּעֵינֵ֥י חֲתָנָֽיו:  מ  וַיֵּצֵ֨א ל֜וֹט וַיְדַבֵּ֣ר | אֶל-חֲתָנָ֣יו | לֹֽקְחֵ֣י בְנֹתָ֗יו וַיֹּ֨אמֶר֙ ק֤וּמוּ צְּאוּ֙ מִן-הַמָּק֣וֹם הַזֶּ֔ה כִּֽי-מַשְׁחִ֥ית יְהֹוָ֖ה אֶת-הָעִ֑יר וַיְהִ֥י כִמְצַחֵ֖ק בְּעֵינֵ֥י חֲתָנָֽיו:  ת  וּנְפַק לוֹט וּמַלֵּיל עִם חַתְנוֹהִי נָסְבֵי בְנָתֵהּ וְאֲמַר קוּמוּ פּוּקוּ מִן אַתְרָא הָדֵין אֲרֵי מְחַבֵּל יְיָ יָת קַרְתָּא וַהֲוָה כִּמְחָיִךְ בְּעֵינֵי חַתְנוֹהִי:

{טו} וּכְמוֹ֙ הַשַּׁ֣חַר עָלָ֔ה וַיָּאִ֥יצוּ הַמַּלְאָכִ֖ים בְּל֣וֹט לֵאמֹ֑ר קוּם֩ קַ֨ח אֶת-אִשְׁתְּךָ֜ וְאֶת-שְׁתֵּ֤י בְנֹתֶ֨יךָ֙ הַנִּמְצָאֹ֔ת פֶּן-תִּסָּפֶ֖ה בַּֽעֲוֹ֥ן הָעִֽיר:  מ  וּכְמוֹ֙ הַשַּׁ֣חַר עָלָ֔ה וַיָּאִ֥יצוּ הַמַּלְאָכִ֖ים בְּל֣וֹט לֵאמֹ֑ר קוּם֩ קַ֨ח אֶת-אִשְׁתְּךָ֜ וְאֶת-שְׁתֵּ֤י בְנֹתֶ֨יךָ֙ הַנִּמְצָאֹ֔ת פֶּן-תִּסָּפֶ֖ה בַּֽעֲוֹ֥ן הָעִֽיר:  ת  וּכְמִסַּק צַפְרָא הֲוָה וּדְחִיקוּ מַלְאָכַיָּא בְּלוֹט לְמֵימָר קוּם דְּבַר יָת אִתְּתָךְ וְיָת תַּרְתֵּין בְּנָתָךְ דְּאִשְׁתְּכַחַן מְהֵימְנָא עִמָּךְ דִּילְמָא תִלְקֵי בְּחוֹבֵי קַרְתָּא:

{טז} וַיִּתְמַהְמָ֓הּ | וַיַּֽחֲזִ֨יקוּ הָֽאֲנָשִׁ֜ים בְּיָד֣וֹ וּבְיַד-אִשְׁתּ֗וֹ וּבְיַד֙ שְׁתֵּ֣י בְנֹתָ֔יו בְּחֶמְלַ֥ת יְהֹוָ֖ה עָלָ֑יו וַיֹּֽצִאֻ֥הוּ וַיַּנִּחֻ֖הוּ מִח֥וּץ לָעִֽיר:  מ  וַיִּתְמַהְמָ֓הּ | וַיַּֽחֲזִ֨יקוּ הָֽאֲנָשִׁ֜ים בְּיָד֣וֹ וּבְיַד-אִשְׁתּ֗וֹ וּבְיַד֙ שְׁתֵּ֣י בְנֹתָ֔יו בְּחֶמְלַ֥ת יְהֹוָ֖ה עָלָ֑יו וַיֹּֽצִאֻ֥הוּ וַיַּנִּחֻ֖הוּ מִח֥וּץ לָעִֽיר:  ת  וְאִתְעַכַּב וְאַתְקִיפוּ גֻּבְרַיָּא בִּידֵהּ וּבִידָא דְּאִתְּתֵהּ וּבִידָא דְתַּרְתֵּין בְּנָתֵהּ כִּרְחַם (נ''י כַּד חָס) יְיָ עֲלוֹהִי וְאַפְּקוּהִי וְאַשְׁרוּהי מִבָּרָא לְקַרְתָּא:

{יז} וַיְהִי֩ כְהֽוֹצִיאָ֨ם אֹתָ֜ם הַח֗וּצָה וַיֹּ֨אמֶר֙ הִמָּלֵ֣ט עַל-נַפְשֶׁ֔ךָ אַל-תַּבִּ֣יט אַֽחֲרֶ֔יךָ וְאַל-תַּֽעֲמֹ֖ד בְּכָל-הַכִּכָּ֑ר הָהָ֥רָה הִמָּלֵ֖ט פֶּן-תִּסָּפֶֽה:  מ  וַיְהִי֩ כְהֽוֹצִיאָ֨ם אֹתָ֜ם הַח֗וּצָה וַיֹּ֨אמֶר֙ הִמָּלֵ֣ט עַל-נַפְשֶׁ֔ךָ אַל-תַּבִּ֣יט אַֽחֲרֶ֔יךָ וְאַל-תַּֽעֲמֹ֖ד בְּכָל-הַכִּכָּ֑ר הָהָ֥רָה הִמָּלֵ֖ט פֶּן-תִּסָּפֶֽה:  ת  וַהֲוָה כַּד אַפִּיקוּ יָתְהוֹן לְבָרָא וַאֲמַר חוּס עַל נַפְשָׁךְ לָא תִּסְתְּכִי לַאֲחוֹרָךְ וְלָא תְקוּם בְּכָל מֵישְׁרָא לְטוּרָא אִשְׁתְּזֵב דִּלְמָא תִלְקֵי:

{יח} וַיֹּ֥אמֶר ל֖וֹט אֲלֵהֶ֑ם אַל-נָ֖א אֲדֹנָֽי:  מ  וַיֹּ֥אמֶר ל֖וֹט אֲלֵהֶ֑ם אַל-נָ֖א אֲדֹנָֽי:  ת  וַאֲמַר לוֹט לְהוֹן בְּבָעוּ כְעַן רִבּוֹנָי (נ''י יְיָ):

{יט} הִנֵּה-נָ֠א מָצָ֨א עַבְדְּךָ֣ חֵן֘ בְּעֵינֶ֒יךָ֒ וַתַּגְדֵּ֣ל חַסְדְּךָ֗ אֲשֶׁ֤ר עָשִׂ֨יתָ֙ עִמָּדִ֔י לְהַֽחֲי֖וֹת אֶת-נַפְשִׁ֑י וְאָֽנֹכִ֗י לֹ֤א אוּכַל֙ לְהִמָּלֵ֣ט הָהָ֔רָה פֶּן-תִּדְבָּקַ֥נִי הָֽרָעָ֖ה וָמַֽתִּי:  מ  הִנֵּה-נָ֠א מָצָ֨א עַבְדְּךָ֣ חֵן֘ בְּעֵינֶ֒יךָ֒ וַתַּגְדֵּ֣ל חַסְדְּךָ֗ אֲשֶׁ֤ר עָשִׂ֨יתָ֙ עִמָּדִ֔י לְהַֽחֲי֖וֹת אֶת-נַפְשִׁ֑י וְאָֽנֹכִ֗י לֹ֤א אוּכַל֙ לְהִמָּלֵ֣ט הָהָ֔רָה פֶּן-תִּדְבָּקַ֥נִי הָֽרָעָ֖ה וָמַֽתִּי:  ת  הָא כְעַן אַשְׁכַּח עַבְדָךְ רַחֲמִין קֳדָמָךְ וְאַסְגֵּיתָא טֵיבוּתָךְ דִּי עֲבַדְתָּ עִמִּי לְקַיָּמָא יָת נַפְשִׁי וַאֲנָא לֵית אֲנָא יָכִיל לְאִשְׁתֵּזָבָא לְטּוּרָא דִּילְמָא תְעַרְעִנַּנִי בִשְׁתָּא וְאֵימוּת:

{כ} הִנֵּה-נָ֠א הָעִ֨יר הַזֹּ֧את קְרֹבָ֛ה לָנ֥וּס שָׁ֖מָּה וְהִ֣וא מִצְעָ֑ר אִמָּֽלְטָ֨ה נָ֜א שָׁ֗מָּה הֲלֹ֥א מִצְעָ֛ר הִ֖וא וּתְחִ֥י נַפְשִֽׁי:  מ  הִנֵּה-נָ֠א הָעִ֨יר הַזֹּ֧את קְרֹבָ֛ה לָנ֥וּס שָׁ֖מָּה וְהִ֣וא מִצְעָ֑ר אִמָּֽלְטָ֨ה נָ֜א שָׁ֗מָּה הֲלֹ֥א מִצְעָ֛ר הִ֖וא וּתְחִ֥י נַפְשִֽׁי:  ת  הָא כְעַן קַרְתָּא הָדָא קְרִיבָא לְמֵעִירוֹק לְתַמָּן וְהִיא זְעֵירָא אִשְׁתְּזֵב כְּעַן תַּמָּן הֲלָא זְעֵירָא הִיא וְתִתְקַיַּם נַפְשִׁי:

{כא}  רביעי  וַיֹּ֣אמֶר אֵלָ֔יו הִנֵּה֙ נָשָׂ֣אתִי פָנֶ֔יךָ גַּ֖ם לַדָּבָ֣ר הַזֶּ֑ה לְבִלְתִּ֛י הָפְכִ֥י אֶת-הָעִ֖יר אֲשֶׁ֥ר דִּבַּֽרְתָּ:  מ  וַיֹּ֣אמֶר אֵלָ֔יו הִנֵּה֙ נָשָׂ֣אתִי פָנֶ֔יךָ גַּ֖ם לַדָּבָ֣ר הַזֶּ֑ה לְבִלְתִּ֛י הָפְכִ֥י אֶת-הָעִ֖יר אֲשֶׁ֥ר דִּבַּֽרְתָּ:  ת  וַאֲמַר לֵהּ הָא נְסֵיבִית אַפָּךְ אַף לְפִתְגָּמָא הָדֵין בְּדִיל דְּלָא לְמֵהְפַּךְ יָת קַרְתָּא דִּבְעִיתָא עֲלַהּ:

{כב} מַהֵר֙ הִמָּלֵ֣ט שָׁ֔מָּה כִּ֣י לֹ֤א אוּכַל֙ לַֽעֲשׂ֣וֹת דָּבָ֔ר עַד-בֹּֽאֲךָ֖ שָׁ֑מָּה עַל-כֵּ֛ן קָרָ֥א שֵׁם-הָעִ֖יר צֽוֹעַר:  מ  מַהֵר֙ הִמָּלֵ֣ט שָׁ֔מָּה כִּ֣י לֹ֤א אוּכַל֙ לַֽעֲשׂ֣וֹת דָּבָ֔ר עַד-בֹּֽאֲךָ֖ שָׁ֑מָּה עַל-כֵּ֛ן קָרָ֥א שֵׁם-הָעִ֖יר צֽוֹעַר:  ת  אוֹחִי לְאִשְׁתֵּזֵב תַּמָּן אֲרֵי לָא אִכּוּל לְמֶעְבַּד פִּתְגָּמָא עַד מֵיתָךְ לְתַמָּן עַל כֵּן קְרָא שְׁמָא דְּקַרְתָּא צוֹעַר:

{כג} הַשֶּׁ֖מֶשׁ יָצָ֣א עַל-הָאָ֑רֶץ וְל֖וֹט בָּ֥א צֹֽעֲרָה:  מ  הַשֶּׁ֖מֶשׁ יָצָ֣א עַל-הָאָ֑רֶץ וְל֖וֹט בָּ֥א צֹֽעֲרָה:  ת  שִׁמְשָׁא נְפַק עַל אַרְעָא וְלוֹט עַל לְצוֹעַר:

{כד} וַֽיהֹוָ֗ה הִמְטִ֧יר עַל-סְדֹ֛ם וְעַל-עֲמֹרָ֖ה גָּפְרִ֣ית וָאֵ֑שׁ מֵאֵ֥ת יְהֹוָ֖ה מִן-הַשָּׁמָֽיִם:  מ  וַֽיהֹוָ֗ה הִמְטִ֧יר עַל-סְדֹ֛ם וְעַל-עֲמֹרָ֖ה גָּפְרִ֣ית וָאֵ֑שׁ מֵאֵ֥ת יְהֹוָ֖ה מִן-הַשָּׁמָֽיִם:  ת  וַיְיָ אַמְטַר עַל סְדוֹם וְעַל עֲמוֹרָה גָּפְרֵיתָא וְאֶשָּׁתָא מִן קֳדָם יְיָ מִן שְׁמַיָּא:

{כה} וַיַּֽהֲפֹךְ֙ אֶת-הֶֽעָרִ֣ים הָאֵ֔ל וְאֵ֖ת כָּל-הַכִּכָּ֑ר וְאֵת֙ כָּל-יֹֽשְׁבֵ֣י הֶֽעָרִ֔ים וְצֶ֖מַח הָֽאֲדָמָֽה:  מ  וַיַּֽהֲפֹךְ֙ אֶת-הֶֽעָרִ֣ים הָאֵ֔ל וְאֵ֖ת כָּל-הַכִּכָּ֑ר וְאֵת֙ כָּל-יֹֽשְׁבֵ֣י הֶֽעָרִ֔ים וְצֶ֖מַח הָֽאֲדָמָֽה:  ת  וַהֲפַךְ יָת קִרְוַיָּא הָאִלֵּין וְיָת כָּל מֵישְׁרָא וְיָת כָּל יָתְבֵי קִרְוַיָּא וְצִמְחָא דְּאַרְעָא:

{כו} וַתַּבֵּ֥ט אִשְׁתּ֖וֹ מֵאַֽחֲרָ֑יו וַתְּהִ֖י נְצִ֥יב מֶֽלַח:  מ  וַתַּבֵּ֥ט אִשְׁתּ֖וֹ מֵאַֽחֲרָ֑יו וַתְּהִ֖י נְצִ֥יב מֶֽלַח:  ת  וְאִסְתְּכִיאַת אִתְּתֵהּ מִבַּתְרוֹהִי וַהֲוַת קָמָא דְּמִלְחָא:

{כז} וַיַּשְׁכֵּ֥ם אַבְרָהָ֖ם בַּבֹּ֑קֶר אֶ֨ל-הַמָּק֔וֹם אֲשֶׁר-עָ֥מַד שָׁ֖ם אֶת-פְּנֵ֥י יְהֹוָֽה:  מ  וַיַּשְׁכֵּ֥ם אַבְרָהָ֖ם בַּבֹּ֑קֶר אֶ֨ל-הַמָּק֔וֹם אֲשֶׁר-עָ֥מַד שָׁ֖ם אֶת-פְּנֵ֥י יְהֹוָֽה:  ת  וְאַקְדֵּים אַבְרָהָם בְּצַפְרָא לְאַתְרָא דִּשַׁמֵשׁ תַּמָּן בִּצְלוֹ קֳדָם יְיָ:

{כח} וַיַּשְׁקֵ֗ף עַל-פְּנֵ֤י סְדֹם֙ וַֽעֲמֹרָ֔ה וְעַ֥ל-כָּל-פְּנֵ֖י אֶ֣רֶץ הַכִּכָּ֑ר וַיַּ֗רְא וְהִנֵּ֤ה עָלָה֙ קִיטֹ֣ר הָאָ֔רֶץ כְּקִיטֹ֖ר הַכִּבְשָֽׁן:  מ  וַיַּשְׁקֵ֗ף עַל-פְּנֵ֤י סְדֹם֙ וַֽעֲמֹרָ֔ה וְעַ֥ל-כָּל-פְּנֵ֖י אֶ֣רֶץ הַכִּכָּ֑ר וַיַּ֗רְא וְהִנֵּ֤ה עָלָה֙ קִיטֹ֣ר הָאָ֔רֶץ כְּקִיטֹ֖ר הַכִּבְשָֽׁן:  ת  וְאִסְתְּכִי עַל אַפֵּי סְדוֹם וַעֲמוֹרָה וְעַל כָּל אַפֵּי אַרְעָא דְּמֵישְׁרָא וַחֲזָא וְהָא סְלִיק תְּנָנָא דְּאַרְעָא כִּתְנָנָא דְּאַתּוּנָא:

{כט} וַיְהִ֗י בְּשַׁחֵ֤ת אֱלֹהִים֙ אֶת-עָרֵ֣י הַכִּכָּ֔ר וַיִּזְכֹּ֥ר אֱלֹהִ֖ים אֶת-אַבְרָהָ֑ם וַיְשַׁלַּ֤ח אֶת-לוֹט֙ מִתּ֣וֹךְ הַֽהֲפֵכָ֔ה בַּֽהֲפֹךְ֙ אֶת-הֶ֣עָרִ֔ים אֲשֶׁר-יָשַׁ֥ב בָּהֵ֖ן לֽוֹט:  מ  וַיְהִ֗י בְּשַׁחֵ֤ת אֱלֹהִים֙ אֶת-עָרֵ֣י הַכִּכָּ֔ר וַיִּזְכֹּ֥ר אֱלֹהִ֖ים אֶת-אַבְרָהָ֑ם וַיְשַׁלַּ֤ח אֶת-לוֹט֙ מִתּ֣וֹךְ הַֽהֲפֵכָ֔ה בַּֽהֲפֹךְ֙ אֶת-הֶ֣עָרִ֔ים אֲשֶׁר-יָשַׁ֥ב בָּהֵ֖ן לֽוֹט:  ת  וַהֲוָה בְּחַבָּלוּת (נ''י כַּד חֲבַל) יְיָ יָת קִרְוֵי מֵישְׁרָא וּדְכִיר יְיָ יָת אַבְרָהָם וְשַׁלַּח יָת לוֹט מִגּוֹ הֲפֶכְתָּא כַּד הֲפַךְ יָת קִרְוַיָּא דִּי הֲוָה יָתֵב בְּהֵן לוֹט:

{ל} וַיַּ֩עַל֩ ל֨וֹט מִצּ֜וֹעַר וַיֵּ֣שֶׁב בָּהָ֗ר וּשְׁתֵּ֤י בְנֹתָיו֙ עִמּ֔וֹ כִּ֥י יָרֵ֖א לָשֶׁ֣בֶת בְּצ֑וֹעַר וַיֵּ֨שֶׁב֙ בַּמְּעָרָ֔ה ה֖וּא וּשְׁתֵּ֥י בְנֹתָֽיו:  מ  וַיַּ֩עַל֩ ל֨וֹט מִצּ֜וֹעַר וַיֵּ֣שֶׁב בָּהָ֗ר וּשְׁתֵּ֤י בְנֹתָיו֙ עִמּ֔וֹ כִּ֥י יָרֵ֖א לָשֶׁ֣בֶת בְּצ֑וֹעַר וַיֵּ֨שֶׁב֙ בַּמְּעָרָ֔ה ה֖וּא וּשְׁתֵּ֥י בְנֹתָֽיו:  ת  וּסְלֵק לוֹט מִצּוֹעַר וִיתֵב בְּטוּרָא וְתַרְתֵּין בְּנָתֵהּ עִמֵּהּ אֲרֵי דָּחִיל לְמִתַּב בְּצוֹעַר וִיתֵב בִּמְעַרְתָּא הוּא וְתַרְתֵּין בְּנָתֵהּ:

{לא} וַתֹּ֧אמֶר הַבְּכִירָ֛ה אֶל-הַצְּעִירָ֖ה אָבִ֣ינוּ זָקֵ֑ן וְאִ֨ישׁ אֵ֤ין בָּאָ֨רֶץ֙ לָב֣וֹא עָלֵ֔ינוּ כְּדֶ֖רֶךְ כָּל-הָאָֽרֶץ:  מ  וַתֹּ֧אמֶר הַבְּכִירָ֛ה אֶל-הַצְּעִירָ֖ה אָבִ֣ינוּ זָקֵ֑ן וְאִ֨ישׁ אֵ֤ין בָּאָ֨רֶץ֙ לָב֣וֹא עָלֵ֔ינוּ כְּדֶ֖רֶךְ כָּל-הָאָֽרֶץ:  ת  וַאֲמֶרֶת רַבְּתָא לִזְעֶרְתָּא אֲבוּנָא סִיב וּגְבַר לֵית בְּאַרְעָא לְמֵיעַל עֲלָנָא כְּאוֹרַח כָּל אַרְעָא:

{לב} לְכָ֨ה נַשְׁקֶ֧ה אֶת-אָבִ֛ינוּ יַ֖יִן וְנִשְׁכְּבָ֣ה עִמּ֑וֹ וּנְחַיֶּ֥ה מֵֽאָבִ֖ינוּ זָֽרַע:  מ  לְכָ֨ה נַשְׁקֶ֧ה אֶת-אָבִ֛ינוּ יַ֖יִן וְנִשְׁכְּבָ֣ה עִמּ֑וֹ וּנְחַיֶּ֥ה מֵֽאָבִ֖ינוּ זָֽרַע:  ת  אִיתָא נַשְׁקֵי יָת אֲבוּנָא חַמְרָא וְנִשְׁכּוּב עִמֵּהּ וּנְקַיֵּם מֵאֲבוּנָא בְּנִין:

{לג} וַתַּשְׁקֶ֧יןָ אֶת-אֲבִיהֶ֛ן יַ֖יִן בַּלַּ֣יְלָה ה֑וּא וַתָּבֹ֤א הַבְּכִירָה֙ וַתִּשְׁכַּ֣ב אֶת-אָבִ֔יהָ וְלֹֽא-יָדַ֥ע בְּשִׁכְבָ֖הּ וּבְקוּ֗מָֽהּ:  מ  וַתַּשְׁקֶ֧יןָ אֶת-אֲבִיהֶ֛ן יַ֖יִן בַּלַּ֣יְלָה ה֑וּא וַתָּבֹ֤א הַבְּכִירָה֙ וַתִּשְׁכַּ֣ב אֶת-אָבִ֔יהָ וְלֹֽא-יָדַ֥ע בְּשִׁכְבָ֖הּ וּבְקוּ֗מָֽהּ:  ת  וְאַשְׁקִיאָה יָת אֲבוּהֶן חַמְרָא בְּלֵילְיָא הוּא וְעַלַּת רַבְּתָא וּשְׁכִיבָא עִם אֲבוּהָא וְלָא יְדַע בְּמִשְׁכְּבַהּ וּבְקִימַהּ:

{לד} וַֽיְהִי֙ מִֽמָּחֳרָ֔ת וַתֹּ֤אמֶר הַבְּכִירָה֙ אֶל-הַצְּעִירָ֔ה הֵן-שָׁכַ֥בְתִּי אֶ֖מֶשׁ אֶת-אָבִ֑י נַשְׁקֶ֨נּוּ יַ֜יִן גַּ֣ם-הַלַּ֗יְלָה וּבֹ֨אִי֙ שִׁכְבִ֣י עִמּ֔וֹ וּנְחַיֶּ֥ה מֵֽאָבִ֖ינוּ זָֽרַע:  מ  וַֽיְהִי֙ מִֽמָּחֳרָ֔ת וַתֹּ֤אמֶר הַבְּכִירָה֙ אֶל-הַצְּעִירָ֔ה הֵן-שָׁכַ֥בְתִּי אֶ֖מֶשׁ אֶת-אָבִ֑י נַשְׁקֶ֨נּוּ יַ֜יִן גַּ֣ם-הַלַּ֗יְלָה וּבֹ֨אִי֙ שִׁכְבִ֣י עִמּ֔וֹ וּנְחַיֶּ֥ה מֵֽאָבִ֖ינוּ זָֽרַע:  ת  וַהֲוָה בְּיוֹמָא דְבַתְרוֹהִי וַאֲמֶרֶת רַבְּתָא לִזְעֶרְתָּא הָא שְׁכֵיבִית רַמְשָׁא עִם אַבָּא נַשְׁקִנֵּהּ חַמְרָא אַף בְּלֵילְיָא וְעוּלִי שְׁכִיבִי עִמֵּהּ וּנְקַיֵּם מֵאֲבוּנָא בְּנִין:

{לה} וַתַּשְׁקֶ֜יןָ גַּ֣ם בַּלַּ֧יְלָה הַה֛וּא אֶת-אֲבִיהֶ֖ן יָ֑יִן וַתָּ֤קָם הַצְּעִירָה֙ וַתִּשְׁכַּ֣ב עִמּ֔וֹ וְלֹֽא-יָדַ֥ע בְּשִׁכְבָ֖הּ וּבְקֻמָֽהּ:  מ  וַתַּשְׁקֶ֜יןָ גַּ֣ם בַּלַּ֧יְלָה הַה֛וּא אֶת-אֲבִיהֶ֖ן יָ֑יִן וַתָּ֤קָם הַצְּעִירָה֙ וַתִּשְׁכַּ֣ב עִמּ֔וֹ וְלֹֽא-יָדַ֥ע בְּשִׁכְבָ֖הּ וּבְקֻמָֽהּ:  ת  וְאַשְׁקִיאָה אַף בְּלֵילְיָא הַהוּא יָת אֲבוּהֶן חַמְרָא וְקָמַת זְעֶרְתָּא וּשְׁכִיבַת עִמֵּהּ וְלָא יְדַע בְּמִשְׁכְּבַהּ וּבְקִימַהּ:

{לו} וַתַּֽהֲרֶ֛יןָ שְׁתֵּ֥י בְנֽוֹת-ל֖וֹט מֵֽאֲבִיהֶֽן:  מ  וַתַּֽהֲרֶ֛יןָ שְׁתֵּ֥י בְנֽוֹת-ל֖וֹט מֵֽאֲבִיהֶֽן:  ת  וְעַדִּיאַן תַּרְתֵּין בְּנַת לוֹט מֵאֲבוּהֶן:

{לז} וַתֵּ֤לֶד הַבְּכִירָה֙ בֵּ֔ן וַתִּקְרָ֥א שְׁמ֖וֹ מוֹאָ֑ב ה֥וּא אֲבִֽי-מוֹאָ֖ב עַד-הַיּֽוֹם:  מ  וַתֵּ֤לֶד הַבְּכִירָה֙ בֵּ֔ן וַתִּקְרָ֥א שְׁמ֖וֹ מוֹאָ֑ב ה֥וּא אֲבִֽי-מוֹאָ֖ב עַד-הַיּֽוֹם:  ת  וִילִידַת רַבְּתָא בַּר וּקְרַת שְׁמֵהּ מוֹאָב הוּא אֲבוּהוֹן דְמוֹאָבָא עַד יוֹמָא דֵין:

{לח} וְהַצְּעִירָ֤ה גַם-הִוא֙ יָ֣לְדָה בֵּ֔ן וַתִּקְרָ֥א שְׁמ֖וֹ בֶּן-עַמִּ֑י ה֛וּא אֲבִ֥י בְנֵֽי-עַמּ֖וֹן עַד-הַיּֽוֹם:  מ  וְהַצְּעִירָ֤ה גַם-הִוא֙ יָ֣לְדָה בֵּ֔ן וַתִּקְרָ֥א שְׁמ֖וֹ בֶּן-עַמִּ֑י ה֛וּא אֲבִ֥י בְנֵֽי-עַמּ֖וֹן עַד-הַיּֽוֹם:  ת  וּזְעֶרְתָּא אַף הִיא יְלִידַת בַּר וּקְרַת שְׁמֵהּ בַּר עַמִּי הוּא אֲבוּהוֹן דִּבְנֵי עַמּוֹן עַד יוֹמָא דֵין: (ס)




בראשית פרק-כ

{א} וַיִּסַּ֨ע מִשָּׁ֤ם אַבְרָהָם֙ אַ֣רְצָה הַנֶּ֔גֶב וַיֵּ֥שֶׁב בֵּֽין-קָדֵ֖שׁ וּבֵ֣ין שׁ֑וּר וַיָּ֖גָר בִּגְרָֽר:  מ  וַיִּסַּ֨ע מִשָּׁ֤ם אַבְרָהָם֙ אַ֣רְצָה הַנֶּ֔גֶב וַיֵּ֥שֶׁב בֵּֽין-קָדֵ֖שׁ וּבֵ֣ין שׁ֑וּר וַיָּ֖גָר בִּגְרָֽר:  ת  וּנְטַל מִתַּמָּן אַבְרָהָם לְאַרְעָא דָּרוֹמָא וִיתֵב בֵּין רְקָם וּבֵין חַגְרָא וְאִתּוֹתַב בִּגְרָר:

{ב} וַיֹּ֧אמֶר אַבְרָהָ֛ם אֶל-שָׂרָ֥ה אִשְׁתּ֖וֹ אֲחֹ֣תִי הִ֑וא וַיִּשְׁלַ֗ח אֲבִימֶ֨לֶךְ֙ מֶ֣לֶךְ גְּרָ֔ר וַיִּקַּ֖ח אֶת-שָׂרָֽה:  מ  וַיֹּ֧אמֶר אַבְרָהָ֛ם אֶל-שָׂרָ֥ה אִשְׁתּ֖וֹ אֲחֹ֣תִי הִ֑וא וַיִּשְׁלַ֗ח אֲבִימֶ֨לֶךְ֙ מֶ֣לֶךְ גְּרָ֔ר וַיִּקַּ֖ח אֶת-שָׂרָֽה:  ת  וַאֲמַר אַבְרָהָם עַל שָׂרָה אִתְּתֵהּ אֲחָתִי הִיא וּשְׁלַח אֲבִימֶלֶךְ מַלְכָּא דִּגְרָר וּדְבַר יָת שָׂרָה:

{ג} וַיָּבֹ֧א אֱלֹהִ֛ים אֶל-אֲבִימֶ֖לֶךְ בַּֽחֲל֣וֹם הַלָּ֑יְלָה וַיֹּ֣אמֶר ל֗וֹ הִנְּךָ֥ מֵת֙ עַל-הָֽאִשָּׁ֣ה אֲשֶׁר-לָקַ֔חְתָּ וְהִ֖וא בְּעֻ֥לַת בָּֽעַל:  מ  וַיָּבֹ֧א אֱלֹהִ֛ים אֶל-אֲבִימֶ֖לֶךְ בַּֽחֲל֣וֹם הַלָּ֑יְלָה וַיֹּ֣אמֶר ל֗וֹ הִנְּךָ֥ מֵת֙ עַל-הָֽאִשָּׁ֣ה אֲשֶׁר-לָקַ֔חְתָּ וְהִ֖וא בְּעֻ֥לַת בָּֽעַל:  ת  וַאֲתָא מֵימַר מִן קָדָם יְיָ לְוַת אֲבִימֶלֶךְ בְּחֶלְמָא דְלֵילְיָא וַאֲמַר לֵהּ הָא אַתְּ מִית עַל עֵיסַק אִתְּתָא דִּדְבַרְתָּא וְהִיא אִתַּת גְּבָר:

{ד} וַֽאֲבִימֶ֕לֶךְ לֹ֥א קָרַ֖ב אֵלֶ֑יהָ וַיֹּאמַ֕ר אֲדֹנָ֕י הֲג֥וֹי גַּם-צַדִּ֖יק תַּֽהֲרֹֽג:  מ  וַֽאֲבִימֶ֕לֶךְ לֹ֥א קָרַ֖ב אֵלֶ֑יהָ וַיֹּאמַ֕ר אֲדֹנָ֕י הֲג֥וֹי גַּם-צַדִּ֖יק תַּֽהֲרֹֽג:  ת  וַאֲבִימֶלֶךְ לָא קְרֵב לְוָתַהּ וַאֲמַר יְיָ הָעָם אַף זַכַּאי תִּקְטוֹל:

{ה} הֲלֹ֨א ה֤וּא אָֽמַר-לִי֙ אֲחֹ֣תִי הִ֔וא וְהִֽיא-גַם-הִ֥וא אָֽמְרָ֖ה אָחִ֣י ה֑וּא בְּתָם-לְבָבִ֛י וּבְנִקְיֹ֥ן כַּפַּ֖י עָשִׂ֥יתִי זֹֽאת:  מ  הֲלֹ֨א ה֤וּא אָֽמַר-לִי֙ אֲחֹ֣תִי הִ֔וא וְהִֽיא-גַם-הִ֥וא אָֽמְרָ֖ה אָחִ֣י ה֑וּא בְּתָם-לְבָבִ֛י וּבְנִקְיֹ֥ן כַּפַּ֖י עָשִׂ֥יתִי זֹֽאת:  ת  הֲלָא הוּא אֲמַר לִי אֲחָתִי הִיא וְהִיא אַף הִיא אֲמָרַת אָחִי הוּא בְּקַשִּׁיטוּת לִבִּי וּבְזַכָּאוּת יָדַי עֲבָדִית דָּא:

{ו} וַיֹּ֩אמֶר֩ אֵלָ֨יו הָֽאֱלֹהִ֜ים בַּֽחֲלֹ֗ם גַּ֣ם אָֽנֹכִ֤י יָדַ֨עְתִּי֙ כִּ֤י בְתָם-לְבָֽבְךָ֙ עָשִׂ֣יתָ זֹּ֔את וָֽאֶחְשׂ֧ךְ גַּם-אָֽנֹכִ֛י אֽוֹתְךָ֖ מֵֽחֲטוֹ-לִ֑י עַל-כֵּ֥ן לֹֽא-נְתַתִּ֖יךָ לִנְגֹּ֥עַ אֵלֶֽיהָ:  מ  וַיֹּ֩אמֶר֩ אֵלָ֨יו הָֽאֱלֹהִ֜ים בַּֽחֲלֹ֗ם גַּ֣ם אָֽנֹכִ֤י יָדַ֨עְתִּי֙ כִּ֤י בְתָם-לְבָֽבְךָ֙ עָשִׂ֣יתָ זֹּ֔את וָֽאֶחְשׂ֧ךְ גַּם-אָֽנֹכִ֛י אֽוֹתְךָ֖ מֵֽחֲטוֹ-לִ֑י עַל-כֵּ֥ן לֹֽא-נְתַתִּ֖יךָ לִנְגֹּ֥עַ אֵלֶֽיהָ:  ת  וַאֲמַר לֵהּ מֵימַר מִן קֳדָם יְיָ בְּחֶלְמָא אַף קֳדָמַי גְּלִי אֲרֵי בְּקַשִּׁיטוּת לִבָּךְ עֲבַדְתָּ דָּא וּמְנָעִית אַף אֲנָא יָתָךְ מִלְּמֶחְטֵי קֳדָמָי עַל כֵּן לָא שְׁבָקְתָּךְ לְמִקְרַב לְוָתַהּ:

{ז} וְעַתָּ֗ה הָשֵׁ֤ב אֵֽשֶׁת-הָאִישׁ֙ כִּֽי-נָבִ֣יא ה֔וּא וְיִתְפַּלֵּ֥ל בַּֽעַדְךָ֖ וֶֽחְיֵ֑ה וְאִם-אֵֽינְ֣ךָ מֵשִׁ֔יב דַּ֚ע כִּי-מוֹת תָּמ֔וּת אַתָּ֖ה וְכָל-אֲשֶׁר-לָֽךְ:  מ  וְעַתָּ֗ה הָשֵׁ֤ב אֵֽשֶׁת-הָאִישׁ֙ כִּֽי-נָבִ֣יא ה֔וּא וְיִתְפַּלֵּ֥ל בַּֽעַדְךָ֖ וֶֽחְיֵ֑ה וְאִם-אֵֽינְ֣ךָ מֵשִׁ֔יב דַּ֚ע כִּי-מוֹת תָּמ֔וּת אַתָּ֖ה וְכָל-אֲשֶׁר-לָֽךְ:  ת  וּכְעַן אֲתֵיב אִתַּת גַּבְרָא אֲרֵי נְבִיָּא הוּא וִיצַלֵּי עֲלָךְ וּתְחֵי וְאִם לֵיתָךְ מֵתֵיב דַּע אֲרֵי מֵימַת תְּמוּת אַתְּ וְכָל דִּי לָךְ:

{ח} וַיַּשְׁכֵּ֨ם אֲבִימֶ֜לֶךְ בַּבֹּ֗קֶר וַיִּקְרָא֙ לְכָל-עֲבָדָ֔יו וַיְדַבֵּ֛ר אֶת-כָּל-הַדְּבָרִ֥ים הָאֵ֖לֶּה בְּאָזְנֵיהֶ֑ם וַיִּֽירְא֥וּ הָֽאֲנָשִׁ֖ים מְאֹֽד:  מ  וַיַּשְׁכֵּ֨ם אֲבִימֶ֜לֶךְ בַּבֹּ֗קֶר וַיִּקְרָא֙ לְכָל-עֲבָדָ֔יו וַיְדַבֵּ֛ר אֶת-כָּל-הַדְּבָרִ֥ים הָאֵ֖לֶּה בְּאָזְנֵיהֶ֑ם וַיִּֽירְא֥וּ הָֽאֲנָשִׁ֖ים מְאֹֽד:  ת  וְאַקְדֵּים אֲבִימֶלֶךְ בְּצַפְרָא וּקְרָא לְכָל עַבְדּוֹהִי וּמַלִּיל יַת כָּל פִּתְגָּמַיָּא הָאִלֵּין קֳדָמֵיהוֹן וּדְחִילוּ גּוּבְרַיָּא לַחֲדָא:

{ט} וַיִּקְרָ֨א אֲבִימֶ֜לֶךְ לְאַבְרָהָ֗ם וַיֹּ֨אמֶר ל֜וֹ מֶה-עָשִׂ֤יתָ לָּ֨נוּ֙ וּמֶה-חָטָ֣אתִי לָ֔ךְ כִּֽי-הֵבֵ֧אתָ עָלַ֛י וְעַל-מַמְלַכְתִּ֖י חֲטָאָ֣ה גְדֹלָ֑ה מַֽעֲשִׂים֙ אֲשֶׁ֣ר לֹא-יֵֽעָשׂ֔וּ עָשִׂ֖יתָ עִמָּדִֽי:  מ  וַיִּקְרָ֨א אֲבִימֶ֜לֶךְ לְאַבְרָהָ֗ם וַיֹּ֨אמֶר ל֜וֹ מֶה-עָשִׂ֤יתָ לָּ֨נוּ֙ וּמֶה-חָטָ֣אתִי לָ֔ךְ כִּֽי-הֵבֵ֧אתָ עָלַ֛י וְעַל-מַמְלַכְתִּ֖י חֲטָאָ֣ה גְדֹלָ֑ה מַֽעֲשִׂים֙ אֲשֶׁ֣ר לֹא-יֵֽעָשׂ֔וּ עָשִׂ֖יתָ עִמָּדִֽי:  ת  וּקְרָא אֲבִימֶלֶךְ לְאַבְרָהָם וַאֲמַר לֵהּ מָה עֲבַדְתָּ לָנָא וּמָה חָבִית (נ''י חָטֵית) לָךְ אֲרֵי אַיְתֵיתָא עֲלַי וְעַל מַלְכוּתִי חוֹבָא רַבָּא עוֹבָדִין דִּי לָא כָשְׁרִין לְאִתְעֲבָדָא עֲבַדְתָּ עִמִּי:

{י} וַיֹּ֥אמֶר אֲבִימֶ֖לֶךְ אֶל-אַבְרָהָ֑ם מָ֣ה רָאִ֔יתָ כִּ֥י עָשִׂ֖יתָ אֶת-הַדָּבָ֥ר הַזֶּֽה:  מ  וַיֹּ֥אמֶר אֲבִימֶ֖לֶךְ אֶל-אַבְרָהָ֑ם מָ֣ה רָאִ֔יתָ כִּ֥י עָשִׂ֖יתָ אֶת-הַדָּבָ֥ר הַזֶּֽה:  ת  וַאֲמַר אֲבִימֶלֶךְ לְאַבְרָהָם מָה חָזֵיתָ אֲרֵי עֲבַדְתָּ יָת פִתְגָּמָא הָדֵין:

{יא} וַיֹּ֨אמֶר֙ אַבְרָהָ֔ם כִּ֣י אָמַ֔רְתִּי רַ֚ק אֵין-יִרְאַ֣ת אֱלֹהִ֔ים בַּמָּק֖וֹם הַזֶּ֑ה וַֽהֲרָג֖וּנִי עַל-דְּבַ֥ר אִשְׁתִּֽי:  מ  וַיֹּ֨אמֶר֙ אַבְרָהָ֔ם כִּ֣י אָמַ֔רְתִּי רַ֚ק אֵין-יִרְאַ֣ת אֱלֹהִ֔ים בַּמָּק֖וֹם הַזֶּ֑ה וַֽהֲרָג֖וּנִי עַל-דְּבַ֥ר אִשְׁתִּֽי:  ת  וַאֲמַר אַבְרָהָם אֲרֵי אֲמָרִית לְחוֹד לֵית דַּחַלְתָּא דַּיְיָ בְּאַתְרָא הָדֵין וְיִקְטְלֻנַּנִי עַל עֵיסַק אִתְּתִי:

{יב} וְגַם-אָמְנָ֗ה אֲחֹתִ֤י בַת-אָבִי֙ הִ֔וא אַ֖ךְ לֹ֣א בַת-אִמִּ֑י וַתְּהִי-לִ֖י לְאִשָּֽׁה:  מ  וְגַם-אָמְנָ֗ה אֲחֹתִ֤י בַת-אָבִי֙ הִ֔וא אַ֖ךְ לֹ֣א בַת-אִמִּ֑י וַתְּהִי-לִ֖י לְאִשָּֽׁה:  ת  וּבְרַם בְּקוּשְׁטָא אֲחָתִי בַת אַבָּא הִיא בְּרַם לָא בַת אִמִּי וַהֲוַת לִי לְאִנְתּוּ:

{יג} וַיְהִ֞י כַּֽאֲשֶׁ֧ר הִתְע֣וּ אֹתִ֗י אֱלֹהִים֘ מִבֵּ֣ית אָבִי֒ וָֽאֹמַ֣ר לָ֔הּ זֶ֣ה חַסְדֵּ֔ךְ אֲשֶׁ֥ר תַּֽעֲשִׂ֖י עִמָּדִ֑י אֶ֤ל כָּל-הַמָּקוֹם֙ אֲשֶׁ֣ר נָב֣וֹא שָׁ֔מָּה אִמְרִי-לִ֖י אָחִ֥י הֽוּא:  מ  וַיְהִ֞י כַּֽאֲשֶׁ֧ר הִתְע֣וּ אֹתִ֗י אֱלֹהִים֘ מִבֵּ֣ית אָבִי֒ וָֽאֹמַ֣ר לָ֔הּ זֶ֣ה חַסְדֵּ֔ךְ אֲשֶׁ֥ר תַּֽעֲשִׂ֖י עִמָּדִ֑י אֶ֤ל כָּל-הַמָּקוֹם֙ אֲשֶׁ֣ר נָב֣וֹא שָׁ֔מָּה אִמְרִי-לִ֖י אָחִ֥י הֽוּא:  ת  וַהֲוָה כַּד טָעוּ עַמְמַיָּא בָּתַר עוֹבָדֵי יְדֵיהוֹן יָתִי קָרִיב יְיָ לְדַחַלְתֵּהּ מִבֵּית אַבָּא וַאֲמָרִית לַהּ דֵּין (נ''י דָּא) טֵיבוּתִיךְ דִּי תַעְבְּדִי עִמִּי לְכָל אַתְרָא דִּי נְהַךְ לְתַמָּן אֱמָרִי עֲלַי אָחִי הוּא:

{יד} וַיִּקַּ֨ח אֲבִימֶ֜לֶךְ צֹ֣אן וּבָקָ֗ר וַֽעֲבָדִים֙ וּשְׁפָחֹ֔ת וַיִּתֵּ֖ן לְאַבְרָהָ֑ם וַיָּ֣שֶׁב ל֔וֹ אֵ֖ת שָׂרָ֥ה אִשְׁתּֽוֹ:  מ  וַיִּקַּ֨ח אֲבִימֶ֜לֶךְ צֹ֣אן וּבָקָ֗ר וַֽעֲבָדִים֙ וּשְׁפָחֹ֔ת וַיִּתֵּ֖ן לְאַבְרָהָ֑ם וַיָּ֣שֶׁב ל֔וֹ אֵ֖ת שָׂרָ֥ה אִשְׁתּֽוֹ:  ת  וּדְבַר אֲבִימֶלֶךְ עָאן וְתוֹרִין וְעַבְדִין וְאַמְהָן וִיהַב לְאַבְרָהָם וַאֲתֵיב לֵהּ יָת שָׂרָה אִתְּתֵהּ:

{טו} וַיֹּ֣אמֶר אֲבִימֶ֔לֶךְ הִנֵּ֥ה אַרְצִ֖י לְפָנֶ֑יךָ בַּטּ֥וֹב בְּעֵינֶ֖יךָ שֵֽׁב:  מ  וַיֹּ֣אמֶר אֲבִימֶ֔לֶךְ הִנֵּ֥ה אַרְצִ֖י לְפָנֶ֑יךָ בַּטּ֥וֹב בְּעֵינֶ֖יךָ שֵֽׁב:  ת  וַאֲמַר אֲבִימֶלֶךְ הָא אַרְעִי קֳדָמָךְ בִּדְתַקִּין בְּעֵינָיךְ תִּיב:

{טז} וּלְשָׂרָ֣ה אָמַ֗ר הִנֵּ֨ה נָתַ֜תִּי אֶ֤לֶף כֶּ֨סֶף֙ לְאָחִ֔יךְ הִנֵּ֤ה הוּא-לָךְ֙ כְּס֣וּת עֵינַ֔יִם לְכֹ֖ל אֲשֶׁ֣ר אִתָּ֑ךְ וְאֵ֥ת כֹּ֖ל וְנֹכָֽחַת:  מ  וּלְשָׂרָ֣ה אָמַ֗ר הִנֵּ֨ה נָתַ֜תִּי אֶ֤לֶף כֶּ֨סֶף֙ לְאָחִ֔יךְ הִנֵּ֤ה הוּא-לָךְ֙ כְּס֣וּת עֵינַ֔יִם לְכֹ֖ל אֲשֶׁ֣ר אִתָּ֑ךְ וְאֵ֥ת כֹּ֖ל וְנֹכָֽחַת:  ת  וּלְשָׂרָה אֲמַר הָא יְהָבִית אֶלֶף סַלְעִין דִּכְסַף לְאָחִיךְ הָא הוּא לִיךְ כְּסוּת דִּיקָר עַיְנִין חֳלַף דִּשְׁלָחִית וּדְּבַרְתִּיךְ וַחֲזֵית יָתִיךְ וְיָת כָּל דְּעִמָּךְ וְעַל כָּל מָה דַּאֲמָרַת אִתּוֹכָחַת:

{יז} וַיִּתְפַּלֵּ֥ל אַבְרָהָ֖ם אֶל-הָֽאֱלֹהִ֑ים וַיִּרְפָּ֨א אֱלֹהִ֜ים אֶת-אֲבִימֶ֧לֶךְ וְאֶת-אִשְׁתּ֛וֹ וְאַמְהֹתָ֖יו וַיֵּלֵֽדוּ:  מ  וַיִּתְפַּלֵּ֥ל אַבְרָהָ֖ם אֶל-הָֽאֱלֹהִ֑ים וַיִּרְפָּ֨א אֱלֹהִ֜ים אֶת-אֲבִימֶ֧לֶךְ וְאֶת-אִשְׁתּ֛וֹ וְאַמְהֹתָ֖יו וַיֵּלֵֽדוּ:  ת  וְצַלִּי אַבְרָהָם קֳדָם יְיָ וְאַסִּי יְיָ יָת אֲבִימֶלֶךְ וְיָת אִתְּתֵהּ וְאַמְהָתֵהּ וְאִתְרְוָחוּ:

{יח} כִּֽי-עָצֹ֤ר עָצַר֙ יְהֹוָ֔ה בְּעַ֥ד כָּל-רֶ֖חֶם לְבֵ֣ית אֲבִימֶ֑לֶךְ עַל-דְּבַ֥ר שָׂרָ֖ה אֵ֥שֶׁת אַבְרָהָֽם:  מ  כִּֽי-עָצֹ֤ר עָצַר֙ יְהֹוָ֔ה בְּעַ֥ד כָּל-רֶ֖חֶם לְבֵ֣ית אֲבִימֶ֑לֶךְ עַל-דְּבַ֥ר שָׂרָ֖ה אֵ֥שֶׁת אַבְרָהָֽם:  ת  אֲרֵי מֵיחַד אֲחַד יְיָ בְּאַפֵּי כָל פָּתַח וַלְדָא לְבֵית אֲבִימֶלֶךְ עַל עֵיסַק שָׂרָה אִתַּת אַבְרָהָם: (ס)




בראשית פרק-כא

{א} וַֽיהוָֹ֛ה פָּקַ֥ד אֶת-שָׂרָ֖ה כַּֽאֲשֶׁ֣ר אָמָ֑ר וַיַּ֧עַשׂ יְהֹוָ֛ה לְשָׂרָ֖ה כַּֽאֲשֶׁ֥ר דִּבֵּֽר:  מ  וַֽיהוָֹ֛ה פָּקַ֥ד אֶת-שָׂרָ֖ה כַּֽאֲשֶׁ֣ר אָמָ֑ר וַיַּ֧עַשׂ יְהֹוָ֛ה לְשָׂרָ֖ה כַּֽאֲשֶׁ֥ר דִּבֵּֽר:  ת  וַיְיָ דְּכִיר יָת שָׂרָה כְּמָא דִּי אֲמַר וַעֲבַד יְיָ לְשָׂרָה כְּמָא דִּי מַלִּיל:

{ב} וַתַּ֩הַר֩ וַתֵּ֨לֶד שָׂרָ֧ה לְאַבְרָהָ֛ם בֵּ֖ן לִזְקֻנָ֑יו לַמּוֹעֵ֕ד אֲשֶׁר-דִּבֶּ֥ר אֹת֖וֹ אֱלֹהִֽים:  מ  וַתַּ֩הַר֩ וַתֵּ֨לֶד שָׂרָ֧ה לְאַבְרָהָ֛ם בֵּ֖ן לִזְקֻנָ֑יו לַמּוֹעֵ֕ד אֲשֶׁר-דִּבֶּ֥ר אֹת֖וֹ אֱלֹהִֽים:  ת  וְעַדִּיאַת וִילִידַת שָׂרָה לְאַבְרָהָם בַּר לְסִיבְתוֹהִי לִזְמַן דִּי מַלִּיל יָתֵהּ יְיָ:

{ג} וַיִּקְרָ֨א אַבְרָהָ֜ם אֶת-שֶׁם-בְּנ֧וֹ הַנּֽוֹלַד-ל֛וֹ אֲשֶׁר-יָֽלְדָה-לּ֥וֹ שָׂרָ֖ה יִצְחָֽק:  מ  וַיִּקְרָ֨א אַבְרָהָ֜ם אֶת-שֶׁם-בְּנ֧וֹ הַנּֽוֹלַד-ל֛וֹ אֲשֶׁר-יָֽלְדָה-לּ֥וֹ שָׂרָ֖ה יִצְחָֽק:  ת  וּקְרָא אַבְרָהָם יָת שׁוּם בְּרֵהּ דְאִתְיְלִיד לֵהּ דִּילִידַת לֵהּ שָׂרָה יִצְחָק:

{ד} וַיָּ֤מָל אַבְרָהָם֙ אֶת-יִצְחָ֣ק בְּנ֔וֹ בֶּן-שְׁמֹנַ֖ת יָמִ֑ים כַּֽאֲשֶׁ֛ר צִוָּ֥ה אֹת֖וֹ אֱלֹהִֽים:  מ  וַיָּ֤מָל אַבְרָהָם֙ אֶת-יִצְחָ֣ק בְּנ֔וֹ בֶּן-שְׁמֹנַ֖ת יָמִ֑ים כַּֽאֲשֶׁ֛ר צִוָּ֥ה אֹת֖וֹ אֱלֹהִֽים:  ת  וּגְזַר אַבְרָהָם יָת יִצְחָק בְּרֵהּ בַּר תְּמַנְיָא יוֹמִין כְּמָא דִּי פַקִּיד יָתֵהּ יְיָ:

{ה}  חמישי  וְאַבְרָהָ֖ם בֶּן-מְאַ֣ת שָׁנָ֑ה בְּהִוָּ֣לֶד ל֔וֹ אֵ֖ת יִצְחָ֥ק בְּנֽוֹ:  מ  וְאַבְרָהָ֖ם בֶּן-מְאַ֣ת שָׁנָ֑ה בְּהִוָּ֣לֶד ל֔וֹ אֵ֖ת יִצְחָ֥ק בְּנֽוֹ:  ת  וְאַבְרָהָם בַּר מְאָה שְׁנִין כַּד אִתְיְלִיד לֵהּ יָת יִצְחָק בְּרֵהּ:

{ו} וַתֹּ֣אמֶר שָׂרָ֔ה צְחֹ֕ק עָ֥שָׂה לִ֖י אֱלֹהִ֑ים כָּל-הַשֹּׁמֵ֖עַ יִֽצְחַק-לִֽי:  מ  וַתֹּ֣אמֶר שָׂרָ֔ה צְחֹ֕ק עָ֥שָׂה לִ֖י אֱלֹהִ֑ים כָּל-הַשֹּׁמֵ֖עַ יִֽצְחַק-לִֽי:  ת  וַאֲמֶרֶת שָׂרָה חֶדְוָא עֲבַד לִי יְיָ כָּל דִּשְׁמַע יֶחֱדֵּי לִי:

{ז} וַתֹּ֗אמֶר מִ֤י מִלֵּל֙ לְאַבְרָהָ֔ם הֵינִ֥יקָה בָנִ֖ים שָׂרָ֑ה כִּֽי-יָלַ֥דְתִּי בֵ֖ן לִזְקֻנָֽיו:  מ  וַתֹּ֗אמֶר מִ֤י מִלֵּל֙ לְאַבְרָהָ֔ם הֵינִ֥יקָה בָנִ֖ים שָׂרָ֑ה כִּֽי-יָלַ֥דְתִּי בֵ֖ן לִזְקֻנָֽיו:  ת  וַאֲמֶרֶת מָאן מְהֵימָן דַּאֲמַר לְאַבְרָהָם קַיָּם דְּתוֹנִיק בְּנִין שָׂרָה אֲרֵי יְלֵידִית בַּר לְסִיבְתוֹהִי:

{ח} וַיִּגְדַּ֥ל הַיֶּ֖לֶד וַיִּגָּמַ֑ל וַיַּ֤עַשׂ אַבְרָהָם֙ מִשְׁתֶּ֣ה גָד֔וֹל בְּי֖וֹם הִגָּמֵ֥ל אֶת-יִצְחָֽק:  מ  וַיִּגְדַּ֥ל הַיֶּ֖לֶד וַיִּגָּמַ֑ל וַיַּ֤עַשׂ אַבְרָהָם֙ מִשְׁתֶּ֣ה גָד֔וֹל בְּי֖וֹם הִגָּמֵ֥ל אֶת-יִצְחָֽק:  ת  וּרְבָא רַבְיָא וְאִתְחֲסִיל וַעֲבַד אַבְרָהָם מִשְׁתְּיָא רַבָּא בְּיוֹמָא דְּאִתְחֲסִיל יָת יִצְחָק:

{ט} וַתֵּ֨רֶא שָׂרָ֜ה אֶת-בֶּן-הָגָ֧ר הַמִּצְרִ֛ית אֲשֶׁר-יָֽלְדָ֥ה לְאַבְרָהָ֖ם מְצַחֵֽק:  מ  וַתֵּ֨רֶא שָׂרָ֜ה אֶת-בֶּן-הָגָ֧ר הַמִּצְרִ֛ית אֲשֶׁר-יָֽלְדָ֥ה לְאַבְרָהָ֖ם מְצַחֵֽק:  ת  וַחֲזַת שָׂרָה יָת בַּר הָגָר מִצְרֵיתָא דִּילִידַת לְאַבְרָהָם מְחָיִיךְ:

{י} וַתֹּ֨אמֶר֙ לְאַבְרָהָ֔ם גָּרֵ֛שׁ הָֽאָמָ֥ה הַזֹּ֖את וְאֶת-בְּנָ֑הּ כִּ֣י לֹ֤א יִירַשׁ֙ בֶּן-הָֽאָמָ֣ה הַזֹּ֔את עִם-בְּנִ֖י עִם-יִצְחָֽק:  מ  וַתֹּ֨אמֶר֙ לְאַבְרָהָ֔ם גָּרֵ֛שׁ הָֽאָמָ֥ה הַזֹּ֖את וְאֶת-בְּנָ֑הּ כִּ֣י לֹ֤א יִירַשׁ֙ בֶּן-הָֽאָמָ֣ה הַזֹּ֔את עִם-בְּנִ֖י עִם-יִצְחָֽק:  ת  וַאֲמֶרֶת לְאַבְרָהָם תָּרֵךְ אַמְתָא הָדָא וְיָת בְּרַהּ אֲרֵי לָא יֵרַת בַּר אַמְתָא הָדָא עִם בְּרִי עִם יִצְחָק:

{יא} וַיֵּ֧רַע הַדָּבָ֛ר מְאֹ֖ד בְּעֵינֵ֣י אַבְרָהָ֑ם עַ֖ל אוֹדֹ֥ת בְּנֽוֹ:  מ  וַיֵּ֧רַע הַדָּבָ֛ר מְאֹ֖ד בְּעֵינֵ֣י אַבְרָהָ֑ם עַ֖ל אוֹדֹ֥ת בְּנֽוֹ:  ת  וּבְאֵישׁ פִּתְגָּמָא לַחֲדָא בְּעֵינֵי אַבְרָהָם עַל עֵיסַק בְּרֵהּ:

{יב} וַיֹּ֨אמֶר אֱלֹהִ֜ים אֶל-אַבְרָהָ֗ם אַל-יֵרַ֤ע בְּעֵינֶ֨יךָ֙ עַל-הַנַּ֣עַר וְעַל-אֲמָתֶ֔ךָ כֹּ֩ל אֲשֶׁ֨ר תֹּאמַ֥ר אֵלֶ֛יךָ שָׂרָ֖ה שְׁמַ֣ע בְּקֹלָ֑הּ כִּ֣י בְיִצְחָ֔ק יִקָּרֵ֥א לְךָ֖ זָֽרַע:  מ  וַיֹּ֨אמֶר אֱלֹהִ֜ים אֶל-אַבְרָהָ֗ם אַל-יֵרַ֤ע בְּעֵינֶ֨יךָ֙ עַל-הַנַּ֣עַר וְעַל-אֲמָתֶ֔ךָ כֹּ֩ל אֲשֶׁ֨ר תֹּאמַ֥ר אֵלֶ֛יךָ שָׂרָ֖ה שְׁמַ֣ע בְּקֹלָ֑הּ כִּ֣י בְיִצְחָ֔ק יִקָּרֵ֥א לְךָ֖ זָֽרַע:  ת  וַאֲמַר יְיָ לְאַבְרָהָם לָא יַבְאֵשׁ בְּעֵינָיךְ עַל עוּלֵימָא וְעַל אַמְתָךְ כֹּל דִּי תֵימַר לָךְ שָׂרָה קַבֵּל מִנַּהּ אֲרֵי בְיִצְחָק יִתְקְרוּן לָךְ בְּנִין:

{יג} וְגַ֥ם אֶת-בֶּן-הָֽאָמָ֖ה לְג֣וֹי אֲשִׂימֶ֑נּוּ כִּ֥י זַרְעֲךָ֖ הֽוּא:  מ  וְגַ֥ם אֶת-בֶּן-הָֽאָמָ֖ה לְג֣וֹי אֲשִׂימֶ֑נּוּ כִּ֥י זַרְעֲךָ֖ הֽוּא:  ת  וְאַף יָת בַּר אַמְתָא לְעַם אֲשַׁוִּנֵּהּ אֲרֵי בְנָךְ הוּא:

{יד} וַיַּשְׁכֵּ֣ם אַבְרָהָ֣ם | בַּבֹּ֡קֶר וַיִּֽקַּח-לֶחֶם֩ וְחֵ֨מַת מַ֜יִם וַיִּתֵּ֣ן אֶל-הָ֠גָ֠ר שָׂ֧ם עַל-שִׁכְמָ֛הּ וְאֶת-הַיֶּ֖לֶד וַֽיְשַׁלְּחֶ֑הָ וַתֵּ֣לֶךְ וַתֵּ֔תַע בְּמִדְבַּ֖ר בְּאֵ֥ר שָֽׁבַע:  מ  וַיַּשְׁכֵּ֣ם אַבְרָהָ֣ם | בַּבֹּ֡קֶר וַיִּֽקַּח-לֶחֶם֩ וְחֵ֨מַת מַ֜יִם וַיִּתֵּ֣ן אֶל-הָ֠גָ֠ר שָׂ֧ם עַל-שִׁכְמָ֛הּ וְאֶת-הַיֶּ֖לֶד וַֽיְשַׁלְּחֶ֑הָ וַתֵּ֣לֶךְ וַתֵּ֔תַע בְּמִדְבַּ֖ר בְּאֵ֥ר שָֽׁבַע:  ת  וְאַקְדֵּים אַבְרָהָם בְּצַפְרָא וּנְסִיב לַחֲמָא וְרָקְבָּא דְמַיָּא וִיהַב לְהָּגָר שַׁוֵי עַל כַּתְפַּהּ וְיָת רַבְיָא וְשַׁלְחַהּ וַאֲזָלַת וְתָעַת בְּמַדְבְּרָא (נ''א בְּמַדְבֵּר) בְּאֵר שָׁבַע:

{טו} וַיִּכְל֥וּ הַמַּ֖יִם מִן-הַחֵ֑מֶת וַתַּשְׁלֵ֣ךְ אֶת-הַיֶּ֔לֶד תַּ֖חַת אַחַ֥ד הַשִּׂיחִֽם:  מ  וַיִּכְל֥וּ הַמַּ֖יִם מִן-הַחֵ֑מֶת וַתַּשְׁלֵ֣ךְ אֶת-הַיֶּ֔לֶד תַּ֖חַת אַחַ֥ד הַשִּׂיחִֽם:  ת  וּשְׁלִימוּ מַיָּא מִן רָקְבָּא וּרְמַת יָת רַבְיָא תְּחוֹת חַד מִן אִילָנַיָּא:

{טז} וַתֵּ֩לֶךְ֩ וַתֵּ֨שֶׁב לָ֜הּ מִנֶּ֗גֶד הַרְחֵק֙ כִּמְטַֽחֲוֵ֣י קֶ֔שֶׁת כִּ֣י אָֽמְרָ֔ה אַל-אֶרְאֶ֖ה בְּמ֣וֹת הַיָּ֑לֶד וַתֵּ֣שֶׁב מִנֶּ֔גֶד וַתִּשָּׂ֥א אֶת-קֹלָ֖הּ וַתֵּֽבְךְּ:  מ  וַתֵּ֩לֶךְ֩ וַתֵּ֨שֶׁב לָ֜הּ מִנֶּ֗גֶד הַרְחֵק֙ כִּמְטַֽחֲוֵ֣י קֶ֔שֶׁת כִּ֣י אָֽמְרָ֔ה אַל-אֶרְאֶ֖ה בְּמ֣וֹת הַיָּ֑לֶד וַתֵּ֣שֶׁב מִנֶּ֔גֶד וַתִּשָּׂ֥א אֶת-קֹלָ֖הּ וַתֵּֽבְךְּ:  ת  וַאֲזָלַת וִיתִיבַת לַהּ מִקֳּבֵל אַרְחֵיקַת (נ''י אַרְחִיק) כְּמֵיגַד קַשְׁתָּא אֲרֵי אֲמֶרֶת לָא אֶחֱזֵי בְּמוֹתָא דְרַבְיָא וִיתִיבַת מִקֳּבֵל וַאֲרֵימַת יָת קָלַהּ וּבְכָת:

{יז} וַיִּשְׁמַ֣ע אֱלֹהִים֘ אֶת-קוֹל הַנַּ֒עַר֒ וַיִּקְרָא֩ מַלְאַ֨ךְ אֱלֹהִ֤ים | אֶל-הָגָר֙ מִן-הַשָּׁמַ֔יִם וַיֹּ֥אמֶר לָ֖הּ מַה-לָּ֣ךְ הָגָ֑ר אַל-תִּ֣ירְאִ֔י כִּֽי-שָׁמַ֧ע אֱלֹהִ֛ים אֶל-ק֥וֹל הַנַּ֖עַר בַּֽאֲשֶׁ֥ר הוּא-שָֽׁם:  מ  וַיִּשְׁמַ֣ע אֱלֹהִים֘ אֶת-קוֹל הַנַּ֒עַר֒ וַיִּקְרָא֩ מַלְאַ֨ךְ אֱלֹהִ֤ים | אֶל-הָגָר֙ מִן-הַשָּׁמַ֔יִם וַיֹּ֥אמֶר לָ֖הּ מַה-לָּ֣ךְ הָגָ֑ר אַל-תִּ֣ירְאִ֔י כִּֽי-שָׁמַ֧ע אֱלֹהִ֛ים אֶל-ק֥וֹל הַנַּ֖עַר בַּֽאֲשֶׁ֥ר הוּא-שָֽׁם:  ת  וּשְׁמִיעַ קֳדָם יְיָ יָת קָלֵהּ דְּרַבְיָא וּקְרָא מַלְאֲכָא דַּיְיָ לְהָגָר מִן שְׁמַיָּא וַאֲמַר לַהּ מָה לָיךְ הָגָר לָא תִדְחֲלִי אֲרֵי שְׁמִיעַ קֳדָם יְיָ יָת קָלֵהּ דְּרַבְיָא בַּאֲתַר דְּהוּא תַמָּן:

{יח} ק֚וּמִי שְׂאִ֣י אֶת-הַנַּ֔עַר וְהַֽחֲזִ֥יקִי אֶת-יָדֵ֖ךְ בּ֑וֹ כִּֽי-לְג֥וֹי גָּד֖וֹל אֲשִׂימֶֽנּוּ:  מ  ק֚וּמִי שְׂאִ֣י אֶת-הַנַּ֔עַר וְהַֽחֲזִ֥יקִי אֶת-יָדֵ֖ךְ בּ֑וֹ כִּֽי-לְג֥וֹי גָּד֖וֹל אֲשִׂימֶֽנּוּ:  ת  קוּמִי טוּלִי יָת רַבְיָא וְאַתְקִיפִי יָת יְדֵךְ בֵּהּ אֲרֵי לְעַם סַגִּי אֲשַׁוִּנֵּהּ:

{יט} וַיִּפְקַ֤ח אֱלֹהִים֙ אֶת-עֵינֶ֔יהָ וַתֵּ֖רֶא בְּאֵ֣ר מָ֑יִם וַתֵּ֜לֶךְ וַתְּמַלֵּ֤א אֶת-הַחֵ֨מֶת֙ מַ֔יִם וַתַּ֖שְׁקְ אֶת-הַנָּֽעַר:  מ  וַיִּפְקַ֤ח אֱלֹהִים֙ אֶת-עֵינֶ֔יהָ וַתֵּ֖רֶא בְּאֵ֣ר מָ֑יִם וַתֵּ֜לֶךְ וַתְּמַלֵּ֤א אֶת-הַחֵ֨מֶת֙ מַ֔יִם וַתַּ֖שְׁקְ אֶת-הַנָּֽעַר:  ת  וּגְלָא יְיָ יָת עֵינָהַהּ וַחֲזַת בֵּירָא דְמַיָּא וַאֲזָלַת וּמְלַת יָת רָקְבָּא מַיָּא וְאַשְׁקִיאַת יָת רַבְיָא:

{כ} וַיְהִ֧י אֱלֹהִ֛ים אֶת-הַנַּ֖עַר וַיִּגְדָּ֑ל וַיֵּ֨שֶׁב֙ בַּמִּדְבָּ֔ר וַיְהִ֖י רֹבֶ֥ה קַשָּֽׁת:  מ  וַיְהִ֧י אֱלֹהִ֛ים אֶת-הַנַּ֖עַר וַיִּגְדָּ֑ל וַיֵּ֨שֶׁב֙ בַּמִּדְבָּ֔ר וַיְהִ֖י רֹבֶ֥ה קַשָּֽׁת:  ת  וַהֲוָה מֵימְרָא דַּיְיָ בְּסַעְדֵהּ דְּרַבְיָא וּרְבָא וִיתֵב בְּמַדְבְּרָא וַהֲוָה רָבֵי קַשְׁתָּא:

{כא} וַיֵּ֖שֶׁב בְּמִדְבַּ֣ר פָּארָ֑ן וַתִּֽקַּח-ל֥וֹ אִמּ֛וֹ אִשָּׁ֖ה מֵאֶ֥רֶץ מִצְרָֽיִם:  מ  וַיֵּ֖שֶׁב בְּמִדְבַּ֣ר פָּארָ֑ן וַתִּֽקַּח-ל֥וֹ אִמּ֛וֹ אִשָּׁ֖ה מֵאֶ֥רֶץ מִצְרָֽיִם:  ת  וִיתֵב בְּמַדְבְּרָא דְפָארָן וּנְסִיבַת לֵהּ אִמֵּהּ אִתְּתָא מֵאַרְעָא דְמִצְרָיִם: (פ)

{כב}  שישי  וַֽיְהִי֙ בָּעֵ֣ת הַהִ֔וא וַיֹּ֣אמֶר אֲבִימֶ֗לֶךְ וּפִיכֹל֙ שַׂר-צְבָא֔וֹ אֶל-אַבְרָהָ֖ם לֵאמֹ֑ר אֱלֹהִ֣ים עִמְּךָ֔ בְּכֹ֥ל אֲשֶׁר-אַתָּ֖ה עֹשֶֽׂה:  מ  וַֽיְהִי֙ בָּעֵ֣ת הַהִ֔וא וַיֹּ֣אמֶר אֲבִימֶ֗לֶךְ וּפִיכֹל֙ שַׂר-צְבָא֔וֹ אֶל-אַבְרָהָ֖ם לֵאמֹ֑ר אֱלֹהִ֣ים עִמְּךָ֔ בְּכֹ֥ל אֲשֶׁר-אַתָּ֖ה עֹשֶֽׂה:  ת  וַהֲוָה בְּעִדָּנָא הַהִיא וַאֲמַר אֲבִימֶלֶךְ וּפִיכֹל רַב חֵילֵהּ לְאַבְרָהָם לְמֵימָר מֵימְרָא דַּיְיָ בְּסַעְדָּךְ בְּכֹל דִּי אַתְּ עָבֵד:

{כג} וְעַתָּ֗ה הִשָּׁ֨בְעָה לִּ֤י בֵֽאלֹהִים֙ הֵ֔נָּה אִם-תִּשְׁקֹ֣ר לִ֔י וּלְנִינִ֖י וּלְנֶכְדִּ֑י כַּחֶ֜סֶד אֲשֶׁר-עָשִׂ֤יתִי עִמְּךָ֙ תַּֽעֲשֶׂ֣ה עִמָּדִ֔י וְעִם-הָאָ֖רֶץ אֲשֶׁר-גַּ֥רְתָּה בָּֽהּ:  מ  וְעַתָּ֗ה הִשָּׁ֨בְעָה לִּ֤י בֵֽאלֹהִים֙ הֵ֔נָּה אִם-תִּשְׁקֹ֣ר לִ֔י וּלְנִינִ֖י וּלְנֶכְדִּ֑י כַּחֶ֜סֶד אֲשֶׁר-עָשִׂ֤יתִי עִמְּךָ֙ תַּֽעֲשֶׂ֣ה עִמָּדִ֔י וְעִם-הָאָ֖רֶץ אֲשֶׁר-גַּ֥רְתָּה בָּֽהּ:  ת  וּכְעַן קַיֵּם לִי בְּמֵימְרָא דַּיְיָ הָכָא דְּלָא תְשַׁקַּר בִּי וּבִבְרִי וּבְבַר בְּרִי כְּטִיבוֹתָא דִּי עֲבָדִית עִמָּךְ תַּעְבֵּד עִמִּי וְעִם אַרְעָא דְאִתּוֹתַבְתָּא בַהּ:

{כד} וַיֹּ֨אמֶר֙ אַבְרָהָ֔ם אָֽנֹכִ֖י אִשָּׁבֵֽעַ:  מ  וַיֹּ֨אמֶר֙ אַבְרָהָ֔ם אָֽנֹכִ֖י אִשָּׁבֵֽעַ:  ת  וַאֲמַר אַבְרָהָם אֲנָא אֲקַיֵּם:

{כה} וְהוֹכִ֥חַ אַבְרָהָ֖ם אֶת-אֲבִימֶ֑לֶךְ עַל-אֹדוֹת֙ בְּאֵ֣ר הַמַּ֔יִם אֲשֶׁ֥ר גָּֽזְל֖וּ עַבְדֵ֥י אֲבִימֶֽלֶךְ:  מ  וְהוֹכִ֥חַ אַבְרָהָ֖ם אֶת-אֲבִימֶ֑לֶךְ עַל-אֹדוֹת֙ בְּאֵ֣ר הַמַּ֔יִם אֲשֶׁ֥ר גָּֽזְל֖וּ עַבְדֵ֥י אֲבִימֶֽלֶךְ:  ת  וְאוֹכַח אַבְרָהָם יָת אֲבִימֶלֶךְ עַל עֵיסַק בֵּירָא דְמַיָּא דִּי אֲנִּיסוּ עַבְדֵי אֲבִימֶלֶךְ:

{כו} וַיֹּ֣אמֶר אֲבִימֶ֔לֶךְ לֹ֣א יָדַ֔עְתִּי מִ֥י עָשָׂ֖ה אֶת-הַדָּבָ֣ר הַזֶּ֑ה וְגַם-אַתָּ֞ה לֹֽא-הִגַּ֣דְתָּ לִּ֗י וְגַ֧ם אָֽנֹכִ֛י לֹ֥א שָׁמַ֖עְתִּי בִּלְתִּ֥י הַיּֽוֹם:  מ  וַיֹּ֣אמֶר אֲבִימֶ֔לֶךְ לֹ֣א יָדַ֔עְתִּי מִ֥י עָשָׂ֖ה אֶת-הַדָּבָ֣ר הַזֶּ֑ה וְגַם-אַתָּ֞ה לֹֽא-הִגַּ֣דְתָּ לִּ֗י וְגַ֧ם אָֽנֹכִ֛י לֹ֥א שָׁמַ֖עְתִּי בִּלְתִּ֥י הַיּֽוֹם:  ת  וַאֲמַר אֲבִימֶלֶךְ לָא יְדָעִית מָאן עֲבַד יָת פִּתְגָּמָא הָדֵין וְאַף אַתְּ לָא חַוֵּיתָ לִּי וְאַף אֲנָא לָא שְׁמָעִית אֱלָּהֵין יוֹמָא דֵין:

{כז} וַיִּקַּ֤ח אַבְרָהָם֙ צֹ֣אן וּבָקָ֔ר וַיִּתֵּ֖ן לַֽאֲבִימֶ֑לֶךְ וַיִּכְרְת֥וּ שְׁנֵיהֶ֖ם בְּרִֽית:  מ  וַיִּקַּ֤ח אַבְרָהָם֙ צֹ֣אן וּבָקָ֔ר וַיִּתֵּ֖ן לַֽאֲבִימֶ֑לֶךְ וַיִּכְרְת֥וּ שְׁנֵיהֶ֖ם בְּרִֽית:  ת  וּדְבַר אַבְרָהָם עָאן וְתוֹרִין וִיהַב לַאֲבִימֶלֶךְ וּגְזָרוּ תַרְוֵיהוֹן קְיָם:

{כח} וַיַּצֵּ֣ב אַבְרָהָ֗ם אֶת-שֶׁ֛בַע כִּבְשׂ֥ת הַצֹּ֖אן לְבַדְּהֶֽן:  מ  וַיַּצֵּ֣ב אַבְרָהָ֗ם אֶת-שֶׁ֛בַע כִּבְשׂ֥ת הַצֹּ֖אן לְבַדְּהֶֽן:  ת  וַאֲקִים אַבְרָהָם יָת שְׁבַע חוּרְפַן דְּעָאן בִּלְחוֹדֵיהֶן:

{כט} וַיֹּ֥אמֶר אֲבִימֶ֖לֶךְ אֶל-אַבְרָהָ֑ם מָ֣ה הֵ֗נָּה שֶׁ֤בַע כְּבָשׂת֙ הָאֵ֔לֶּה אֲשֶׁ֥ר הִצַּ֖בְתָּ לְבַדָּֽנָה:  מ  וַיֹּ֥אמֶר אֲבִימֶ֖לֶךְ אֶל-אַבְרָהָ֑ם מָ֣ה הֵ֗נָּה שֶׁ֤בַע כְּבָשׂת֙ הָאֵ֔לֶּה אֲשֶׁ֥ר הִצַּ֖בְתָּ לְבַדָּֽנָה:  ת  וַאֲמַר אֲבִימֶלֶךְ לְאַבְרָהָם מָה אִנּוּן שְׁבַע חוּרְפַן אִלֵּין דַּאֲקֵמְתָּא בִּלְחוֹדֵיהֶן:

{ל} וַיֹּ֕אמֶר כִּ֚י אֶת-שֶׁ֣בַע כְּבָשׂ֔ת תִּקַּ֖ח מִיָּדִ֑י בַּֽעֲבוּר֙ תִּֽהְיֶה-לִּ֣י לְעֵדָ֔ה כִּ֥י חָפַ֖רְתִּי אֶת-הַבְּאֵ֥ר הַזֹּֽאת:  מ  וַיֹּ֕אמֶר כִּ֚י אֶת-שֶׁ֣בַע כְּבָשׂ֔ת תִּקַּ֖ח מִיָּדִ֑י בַּֽעֲבוּר֙ תִּֽהְיֶה-לִּ֣י לְעֵדָ֔ה כִּ֥י חָפַ֖רְתִּי אֶת-הַבְּאֵ֥ר הַזֹּֽאת:  ת  וַאֲמַר אֲרֵי יָת שְׁבַע חוּרְפַן תְּקַבֵּל מִן יְדִי בְּדִיל דִּתְהֵי לִי לְסַהֲדוּ אֲרֵי חֲפָרִית יָת בֵּירָא הָדָא:

{לא} עַל-כֵּ֗ן קָרָ֛א לַמָּק֥וֹם הַה֖וּא בְּאֵ֣ר שָׁ֑בַע כִּ֛י שָׁ֥ם נִשְׁבְּע֖וּ שְׁנֵיהֶֽם:  מ  עַל-כֵּ֗ן קָרָ֛א לַמָּק֥וֹם הַה֖וּא בְּאֵ֣ר שָׁ֑בַע כִּ֛י שָׁ֥ם נִשְׁבְּע֖וּ שְׁנֵיהֶֽם:  ת  עַל כֵּן קְרָא לְאַתְרָא הַהוּא בְּאֵר שָׁבַע אֲרֵי תַמָּן קַיָּמוּ תַּרְוֵיהוֹן:

{לב} וַיִּכְרְת֥וּ בְרִ֖ית בִּבְאֵ֣ר שָׁ֑בַע וַיָּ֣קָם אֲבִימֶ֗לֶךְ וּפִיכֹל֙ שַׂר-צְבָא֔וֹ וַיָּשֻׁ֖בוּ אֶל-אֶ֥רֶץ פְּלִשְׁתִּֽים:  מ  וַיִּכְרְת֥וּ בְרִ֖ית בִּבְאֵ֣ר שָׁ֑בַע וַיָּ֣קָם אֲבִימֶ֗לֶךְ וּפִיכֹל֙ שַׂר-צְבָא֔וֹ וַיָּשֻׁ֖בוּ אֶל-אֶ֥רֶץ פְּלִשְׁתִּֽים:  ת  וּגְזָרוּ קְיָם בִּבְאֵר שָׁבַע וְקָם אֲבִימֶלֶךְ וּפִיכֹל רַב חֵילֵהּ וְתָבוּ לְאַרַע פְּלִשְׁתָּאֵי:

{לג} וַיִּטַּ֥ע אֵ֖שֶׁל בִּבְאֵ֣ר שָׁ֑בַע וַיִּ֨קְרָא-שָׁ֔ם בְּשֵׁ֥ם יְהֹוָ֖ה אֵ֥ל עוֹלָֽם:  מ  וַיִּטַּ֥ע אֵ֖שֶׁל בִּבְאֵ֣ר שָׁ֑בַע וַיִּ֨קְרָא-שָׁ֔ם בְּשֵׁ֥ם יְהֹוָ֖ה אֵ֥ל עוֹלָֽם:  ת  וּנְצִיב נִצְבָּא (נ''י אִילָנָא) בִּבְאֵר שָׁבַע וְצַלִּי תַמָּן בִּשְׁמָא דַּיְיָ אֱלָהָא דְעַלְמָא:

{לד} וַיָּ֧גָר אַבְרָהָ֛ם בְּאֶ֥רֶץ פְּלִשְׁתִּ֖ים יָמִ֥ים רַבִּֽים:  מ  וַיָּ֧גָר אַבְרָהָ֛ם בְּאֶ֥רֶץ פְּלִשְׁתִּ֖ים יָמִ֥ים רַבִּֽים:  ת  וְאִתּוֹתָב אַבְרָהָם בְּאַרַע פְּלִשְׁתָּאֵי יוֹמִין סַגִּיאִין: (פ)




בראשית פרק-כב

{א}  שביעי  וַיְהִ֗י אַחַר֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וְהָ֣אֱלֹהִ֔ים נִסָּ֖ה אֶת-אַבְרָהָ֑ם וַיֹּ֣אמֶר אֵלָ֔יו אַבְרָהָ֖ם וַיֹּ֥אמֶר הִנֵּֽנִי:  מ  וַיְהִ֗י אַחַר֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וְהָ֣אֱלֹהִ֔ים נִסָּ֖ה אֶת-אַבְרָהָ֑ם וַיֹּ֣אמֶר אֵלָ֔יו אַבְרָהָ֖ם וַיֹּ֥אמֶר הִנֵּֽנִי:  ת  וַהֲוָה בָּתַר פִּתְגָּמַיָּא הָאִלֵּין וַיְיָ נַסִּי יָת אַבְרָהָם וַאֲמַר לֵהּ אַבְרָהָם וַאֲמַר הָא אֲנָא:

{ב} וַיֹּ֡אמֶר קַח-נָ֠א אֶת-בִּנְךָ֙ אֶת-יְחִֽידְךָ֤ אֲשֶׁר-אָהַ֨בְתָּ֙ אֶת-יִצְחָ֔ק וְלֶ֨ךְ-לְךָ֔ אֶל-אֶ֖רֶץ הַמֹּֽרִיָּ֑ה וְהַֽעֲלֵ֤הוּ שָׁם֙ לְעֹלָ֔ה עַ֚ל אַחַ֣ד הֶֽהָרִ֔ים אֲשֶׁ֖ר אֹמַ֥ר אֵלֶֽיךָ:  מ  וַיֹּ֡אמֶר קַח-נָ֠א אֶת-בִּנְךָ֙ אֶת-יְחִֽידְךָ֤ אֲשֶׁר-אָהַ֨בְתָּ֙ אֶת-יִצְחָ֔ק וְלֶ֨ךְ-לְךָ֔ אֶל-אֶ֖רֶץ הַמֹּֽרִיָּ֑ה וְהַֽעֲלֵ֤הוּ שָׁם֙ לְעֹלָ֔ה עַ֚ל אַחַ֣ד הֶֽהָרִ֔ים אֲשֶׁ֖ר אֹמַ֥ר אֵלֶֽיךָ:  ת  וַאֲמַר דְּבַר כְּעַן יָת בְּרָךְ יָת יְחִידָךְ דִּי רְחַמְתָּ יָת יִצְחָק וְאִיזֵל לָךְ לְאַרְעָא פּוּלְחָנָא וְאַסְקֵהּ קֳדָמַי תַּמָּן לַעֲלָתָא עַל חַד מִן טוּרַיָּא דִּי אֵימַר לָךְ:

{ג} וַיַּשְׁכֵּ֨ם אַבְרָהָ֜ם בַּבֹּ֗קֶר וַיַּֽחֲבֹשׁ֙ אֶת-חֲמֹר֔וֹ וַיִּקַּ֞ח אֶת-שְׁנֵ֤י נְעָרָיו֙ אִתּ֔וֹ וְאֵ֖ת יִצְחָ֣ק בְּנ֑וֹ וַיְבַקַּע֙ עֲצֵ֣י עֹלָ֔ה וַיָּ֣קָם וַיֵּ֔לֶךְ אֶל-הַמָּק֖וֹם אֲשֶׁר-אָֽמַר-ל֥וֹ הָֽאֱלֹהִֽים:  מ  וַיַּשְׁכֵּ֨ם אַבְרָהָ֜ם בַּבֹּ֗קֶר וַיַּֽחֲבֹשׁ֙ אֶת-חֲמֹר֔וֹ וַיִּקַּ֞ח אֶת-שְׁנֵ֤י נְעָרָיו֙ אִתּ֔וֹ וְאֵ֖ת יִצְחָ֣ק בְּנ֑וֹ וַיְבַקַּע֙ עֲצֵ֣י עֹלָ֔ה וַיָּ֣קָם וַיֵּ֔לֶךְ אֶל-הַמָּק֖וֹם אֲשֶׁר-אָֽמַר-ל֥וֹ הָֽאֱלֹהִֽים:  ת  וְאַקְדֵּים אַבְרָהָם בְּצַפְרָא וְזָרֵז יָת חֲמָרֵהּ וּדְבַר יָת תְּרֵין עוּלֵימוֹהִי עִמֵּהּ וְיָת יִצְחָק בְּרֵהּ וְצַלַּח אָעֵי לַעֲלָתָא וְקָם וַאֲזַל לְאַתְרָא דִּי אֲמַר לֵהּ יְיָ:

{ד} בַּיּ֣וֹם הַשְּׁלִישִׁ֗י וַיִּשָּׂ֨א אַבְרָהָ֧ם אֶת-עֵינָ֛יו וַיַּ֥רְא אֶת-הַמָּק֖וֹם מֵֽרָחֹֽק:  מ  בַּיּ֣וֹם הַשְּׁלִישִׁ֗י וַיִּשָּׂ֨א אַבְרָהָ֧ם אֶת-עֵינָ֛יו וַיַּ֥רְא אֶת-הַמָּק֖וֹם מֵֽרָחֹֽק:  ת  בְּיוֹמָא תְלִיתָאָה וּזְקַף אַבְרָהָם יָת עֵינוֹהִי וַחֲזָא יָת אַתְרָא מֵרָחִיק:

{ה} וַיֹּ֨אמֶר אַבְרָהָ֜ם אֶל-נְעָרָ֗יו שְׁבֽוּ-לָכֶ֥ם פֹּה֙ עִֽם-הַֽחֲמ֔וֹר וַֽאֲנִ֣י וְהַנַּ֔עַר נֵֽלְכָ֖ה עַד-כֹּ֑ה וְנִשְׁתַּֽחֲוֶ֖ה וְנָשׁ֥וּבָה אֲלֵיכֶֽם:  מ  וַיֹּ֨אמֶר אַבְרָהָ֜ם אֶל-נְעָרָ֗יו שְׁבֽוּ-לָכֶ֥ם פֹּה֙ עִֽם-הַֽחֲמ֔וֹר וַֽאֲנִ֣י וְהַנַּ֔עַר נֵֽלְכָ֖ה עַד-כֹּ֑ה וְנִשְׁתַּֽחֲוֶ֖ה וְנָשׁ֥וּבָה אֲלֵיכֶֽם:  ת  וַאֲמַר אַבְרָהָם לְעוּלֵימוֹהִי אוֹרִיכוּ לְכוֹן הָכָא עִם חֲמָרָא וַאֲנָא וְעוּלֵימָא נִתְמְטֵי עַד כָּא וְנִסְגּוּד וּנְתוּב לְוַתְכוֹן:

{ו} וַיִּקַּ֨ח אַבְרָהָ֜ם אֶת-עֲצֵ֣י הָֽעֹלָ֗ה וַיָּ֨שֶׂם֙ עַל-יִצְחָ֣ק בְּנ֔וֹ וַיִּקַּ֣ח בְּיָד֔וֹ אֶת-הָאֵ֖שׁ וְאֶת-הַמַּֽאֲכֶ֑לֶת וַיֵּֽלְכ֥וּ שְׁנֵיהֶ֖ם יַחְדָּֽו:  מ  וַיִּקַּ֨ח אַבְרָהָ֜ם אֶת-עֲצֵ֣י הָֽעֹלָ֗ה וַיָּ֨שֶׂם֙ עַל-יִצְחָ֣ק בְּנ֔וֹ וַיִּקַּ֣ח בְּיָד֔וֹ אֶת-הָאֵ֖שׁ וְאֶת-הַמַּֽאֲכֶ֑לֶת וַיֵּֽלְכ֥וּ שְׁנֵיהֶ֖ם יַחְדָּֽו:  ת  וּנְסֵיב אַבְרָהָם יָת אָעֵי דַּעֲלָתָא וְשַׁוִּי עַל יִצְחָק בְּרֵהּ וּנְסֵיב בִּידֵהּ יָת אֶשָּׁתָא וְיָת סַכִּינָא וַאֲזָלוּ תַרְוֵיהוֹן כַּחֲדָא:

{ז} וַיֹּ֨אמֶר יִצְחָ֜ק אֶל-אַבְרָהָ֤ם אָבִיו֙ וַיֹּ֣אמֶר אָבִ֔י וַיֹּ֖אמֶר הִנֶּ֣נִּי בְנִ֑י וַיֹּ֗אמֶר הִנֵּ֤ה הָאֵשׁ֙ וְהָ֣עֵצִ֔ים וְאַיֵּ֥ה הַשֶּׂ֖ה לְעֹלָֽה:  מ  וַיֹּ֨אמֶר יִצְחָ֜ק אֶל-אַבְרָהָ֤ם אָבִיו֙ וַיֹּ֣אמֶר אָבִ֔י וַיֹּ֖אמֶר הִנֶּ֣נִּי בְנִ֑י וַיֹּ֗אמֶר הִנֵּ֤ה הָאֵשׁ֙ וְהָ֣עֵצִ֔ים וְאַיֵּ֥ה הַשֶּׂ֖ה לְעֹלָֽה:  ת  וַאֲמַר יִצְחָק לְאַבְרָהָם אֲבוּהִי וַאֲמַר אַבָּא וַאֲמַר הָא אֲנָא בְרִי וַאֲמַר הָא אֶשָּׁתָא וְאָעַיָּא וְאָן אִימְרָא לַעֲלָתָא:

{ח} וַיֹּ֨אמֶר֙ אַבְרָהָ֔ם אֱלֹהִ֞ים יִרְאֶה-לּ֥וֹ הַשֶּׂ֛ה לְעֹלָ֖ה בְּנִ֑י וַיֵּֽלְכ֥וּ שְׁנֵיהֶ֖ם יַחְדָּֽו:  מ  וַיֹּ֨אמֶר֙ אַבְרָהָ֔ם אֱלֹהִ֞ים יִרְאֶה-לּ֥וֹ הַשֶּׂ֛ה לְעֹלָ֖ה בְּנִ֑י וַיֵּֽלְכ֥וּ שְׁנֵיהֶ֖ם יַחְדָּֽו:  ת  וַאֲמַר אַבְרָהָם קֳדָם יְיָ גְּלֵי לֵהּ אִימְרָא לַעֲלָתָא בְּרִי וַאֲזָלוּ תַרְוֵיהוֹן כַּחֲדָא:

{ט} וַיָּבֹ֗אוּ אֶל-הַמָּקוֹם֘ אֲשֶׁ֣ר אָֽמַר-ל֣וֹ הָֽאֱלֹהִים֒ וַיִּ֨בֶן שָׁ֤ם אַבְרָהָם֙ אֶת-הַמִּזְבֵּ֔חַ וַיַּֽעֲרֹ֖ךְ אֶת-הָֽעֵצִ֑ים וַיַּֽעֲקֹד֙ אֶת-יִצְחָ֣ק בְּנ֔וֹ וַיָּ֤שֶׂם אֹתוֹ֙ עַל-הַמִּזְבֵּ֔חַ מִמַּ֖עַל לָֽעֵצִֽים:  מ  וַיָּבֹ֗אוּ אֶל-הַמָּקוֹם֘ אֲשֶׁ֣ר אָֽמַר-ל֣וֹ הָֽאֱלֹהִים֒ וַיִּ֨בֶן שָׁ֤ם אַבְרָהָם֙ אֶת-הַמִּזְבֵּ֔חַ וַיַּֽעֲרֹ֖ךְ אֶת-הָֽעֵצִ֑ים וַיַּֽעֲקֹד֙ אֶת-יִצְחָ֣ק בְּנ֔וֹ וַיָּ֤שֶׂם אֹתוֹ֙ עַל-הַמִּזְבֵּ֔חַ מִמַּ֖עַל לָֽעֵצִֽים:  ת  וְאָתוּ לְאַתְרָא דִּי אֲמַר לֵהּ יְיָ וּבְנָא תַמָּן אַבְרָהָם יָת מַדְבְּחָא וְסַדַּר יָת אָעַיָּא וַעֲקַד יָת יִצְחָק בְּרֵהּ וְשַׁוֵּי יָתֵהּ עַל מַדְבְּחָא עֵיל מִן אָעַיָּא:

{י} וַיִּשְׁלַ֤ח אַבְרָהָם֙ אֶת-יָד֔וֹ וַיִּקַּ֖ח אֶת-הַמַּֽאֲכֶ֑לֶת לִשְׁחֹ֖ט אֶת-בְּנֽוֹ:  מ  וַיִּשְׁלַ֤ח אַבְרָהָם֙ אֶת-יָד֔וֹ וַיִּקַּ֖ח אֶת-הַמַּֽאֲכֶ֑לֶת לִשְׁחֹ֖ט אֶת-בְּנֽוֹ:  ת  וְאוֹשִׁיט אַבְרָהָם יָת יְדֵהּ וּנְסֵיב יָת סַכִּינָא לְמֵיכַס יַת בְּרֵהּ:

{יא} וַיִּקְרָ֨א אֵלָ֜יו מַלְאַ֤ךְ יְהֹוָה֙ מִן-הַשָּׁמַ֔יִם וַיֹּ֖אמֶר אַבְרָהָ֣ם | אַבְרָהָ֑ם וַיֹּ֖אמֶר הִנֵּֽנִי:  מ  וַיִּקְרָ֨א אֵלָ֜יו מַלְאַ֤ךְ יְהֹוָה֙ מִן-הַשָּׁמַ֔יִם וַיֹּ֖אמֶר אַבְרָהָ֣ם | אַבְרָהָ֑ם וַיֹּ֖אמֶר הִנֵּֽנִי:  ת  וּקְרָא לֵהּ מַלְאָכָא דַּיְיָ מִן שְׁמַיָּא וַאֲמַר אַבְרָהָם אַבְרָהָם וַאֲמַר הָא אֲנָא:

{יב} וַיֹּ֗אמֶר אַל-תִּשְׁלַ֤ח יָֽדְךָ֙ אֶל-הַנַּ֔עַר וְאַל-תַּ֥עַשׂ ל֖וֹ מְא֑וּמָה כִּ֣י | עַתָּ֣ה יָדַ֗עְתִּי כִּֽי-יְרֵ֤א אֱלֹהִים֙ אַ֔תָּה וְלֹ֥א חָשַׂ֛כְתָּ אֶת-בִּנְךָ֥ אֶת-יְחִֽידְךָ֖ מִמֶּֽנִּי:  מ  וַיֹּ֗אמֶר אַל-תִּשְׁלַ֤ח יָֽדְךָ֙ אֶל-הַנַּ֔עַר וְאַל-תַּ֥עַשׂ ל֖וֹ מְא֑וּמָה כִּ֣י | עַתָּ֣ה יָדַ֗עְתִּי כִּֽי-יְרֵ֤א אֱלֹהִים֙ אַ֔תָּה וְלֹ֥א חָשַׂ֛כְתָּ אֶת-בִּנְךָ֥ אֶת-יְחִֽידְךָ֖ מִמֶּֽנִּי:  ת  וַאֲמַר לָא תוֹשִׁיט יְדָךְ לְעוּלֵימָא וְלָא תַעְבֵּד לֵהּ מִדָּעָם אֲרֵי כְּעַן יְדַעְנָא (נ''י יְדָעִית) אֲרֵי דַּחֲלָא דַּיְיָ אַתְּ וְלָא מְנַעְתָּ יָת בְּרָךְ יָת יְחִידָךְ מִנִּי:

{יג} וַיִּשָּׂ֨א אַבְרָהָ֜ם אֶת-עֵינָ֗יו וַיַּרְא֙ וְהִנֵּה-אַ֔יִל אַחַ֕ר נֶֽאֱחַ֥ז בַּסְּבַ֖ךְ בְּקַרְנָ֑יו וַיֵּ֤לֶךְ אַבְרָהָם֙ וַיִּקַּ֣ח אֶת-הָאַ֔יִל וַיַּֽעֲלֵ֥הוּ לְעֹלָ֖ה תַּ֥חַת בְּנֽוֹ:  מ  וַיִּשָּׂ֨א אַבְרָהָ֜ם אֶת-עֵינָ֗יו וַיַּרְא֙ וְהִנֵּה-אַ֔יִל אַחַ֕ר נֶֽאֱחַ֥ז בַּסְּבַ֖ךְ בְּקַרְנָ֑יו וַיֵּ֤לֶךְ אַבְרָהָם֙ וַיִּקַּ֣ח אֶת-הָאַ֔יִל וַיַּֽעֲלֵ֥הוּ לְעֹלָ֖ה תַּ֥חַת בְּנֽוֹ:  ת  וּזְקַף אַבְרָהָם יָת עֵינוֹהִי בָּתַר אִלֵּין וַחֲזָא וְהָא דִּכְרָא בָּתַר אֲחִיד בְּאִילָנָא בְּקַרְנוֹהִי וַאֲזַל אַבְרָהָם וּנְסֵיב יָת דִּכְרָא וְאַסְקֵהּ לַעֲלָתָא חֲלַף בְּרֵהּ:

{יד} וַיִּקְרָ֧א אַבְרָהָ֛ם שֵֽׁם-הַמָּק֥וֹם הַה֖וּא יְהֹוָ֣ה | יִרְאֶ֑ה אֲשֶׁר֙ יֵֽאָמֵ֣ר הַיּ֔וֹם בְּהַ֥ר יְהֹוָ֖ה יֵֽרָאֶֽה:  מ  וַיִּקְרָ֧א אַבְרָהָ֛ם שֵֽׁם-הַמָּק֥וֹם הַה֖וּא יְהֹוָ֣ה | יִרְאֶ֑ה אֲשֶׁר֙ יֵֽאָמֵ֣ר הַיּ֔וֹם בְּהַ֥ר יְהֹוָ֖ה יֵֽרָאֶֽה:  ת  וּפְלַח וְצַלִּי אַבְרָהָם תַּמָּן בְּאַתְרָא הַהוּא וַאֲמַר קֳדָם יְיָ הָכָא יְהוֹן פַּלְחַן דָּרַיָּא בְּכֵן יִתְאֲמַר בְּיוֹמָא הָדֵין בְּטוּרָא הָדֵין אַבְרָהָם קֳדָם יְיָ פְּלָח:

{טו} וַיִּקְרָ֛א מַלְאַ֥ךְ יְהֹוָ֖ה אֶל-אַבְרָהָ֑ם שֵׁנִ֖ית מִן-הַשָּׁמָֽיִם:  מ  וַיִּקְרָ֛א מַלְאַ֥ךְ יְהֹוָ֖ה אֶל-אַבְרָהָ֑ם שֵׁנִ֖ית מִן-הַשָּׁמָֽיִם:  ת  וּקְרָא מַלְאָכָא דַּיְיָ לְאַבְרָהָם תִּנְיָנוּת מִן שְׁמַיָּא:

{טז} וַיֹּ֕אמֶר בִּ֥י נִשְׁבַּ֖עְתִּי נְאֻם-יְהֹוָ֑ה כִּ֗י יַ֚עַן אֲשֶׁ֤ר עָשִׂ֨יתָ֙ אֶת-הַדָּבָ֣ר הַזֶּ֔ה וְלֹ֥א חָשַׂ֖כְתָּ אֶת-בִּנְךָ֥ אֶת-יְחִידֶֽךָ:  מ  וַיֹּ֕אמֶר בִּ֥י נִשְׁבַּ֖עְתִּי נְאֻם-יְהֹוָ֑ה כִּ֗י יַ֚עַן אֲשֶׁ֤ר עָשִׂ֨יתָ֙ אֶת-הַדָּבָ֣ר הַזֶּ֔ה וְלֹ֥א חָשַׂ֖כְתָּ אֶת-בִּנְךָ֥ אֶת-יְחִידֶֽךָ:  ת  וַאֲמַר בְּמֵימְרִי קַיֵּמִית אֲמַר יְיָ אֲרֵי חֲלַף דִּי עֲבַדְתָּא יָת פִּתְגָּמָא הָדֵין וְלָא מְנַעְתָּא יָת בְּרָךְ יָת יְחִידָךְ:

{יז} כִּֽי-בָרֵ֣ךְ אֲבָֽרֶכְךָ֗ וְהַרְבָּ֨ה אַרְבֶּ֤ה אֶת-זַרְעֲךָ֙ כְּכֽוֹכְבֵ֣י הַשָּׁמַ֔יִם וְכַח֕וֹל אֲשֶׁ֖ר עַל-שְׂפַ֣ת הַיָּ֑ם וְיִרַ֣שׁ זַרְעֲךָ֔ אֵ֖ת שַׁ֥עַר אֹֽיְבָֽיו:  מ  כִּֽי-בָרֵ֣ךְ אֲבָֽרֶכְךָ֗ וְהַרְבָּ֨ה אַרְבֶּ֤ה אֶת-זַרְעֲךָ֙ כְּכֽוֹכְבֵ֣י הַשָּׁמַ֔יִם וְכַח֕וֹל אֲשֶׁ֖ר עַל-שְׂפַ֣ת הַיָּ֑ם וְיִרַ֣שׁ זַרְעֲךָ֔ אֵ֖ת שַׁ֥עַר אֹֽיְבָֽיו:  ת  אֲרֵי בָרָכָא אֲבָרְכִנָּךְ וְאַסְגָּאָה אַסְגֵּי יָת בְּנָיךְ כְּכוֹכְבֵי שְׁמַיָּא וּכְחָלָא דִּי עַל כֵּיף יַמָּא וְיִרְתוּן בְּנָיךְ יָת קִרְוֵי סַנְאֵיהוֹן:

{יח} וְהִתְבָּֽרֲכ֣וּ בְזַרְעֲךָ֔ כֹּ֖ל גּוֹיֵ֣י הָאָ֑רֶץ עֵ֕קֶב אֲשֶׁ֥ר שָׁמַ֖עְתָּ בְּקֹלִֽי:  מ  וְהִתְבָּֽרֲכ֣וּ בְזַרְעֲךָ֔ כֹּ֖ל גּוֹיֵ֣י הָאָ֑רֶץ עֵ֕קֶב אֲשֶׁ֥ר שָׁמַ֖עְתָּ בְּקֹלִֽי:  ת  וְיִתְבָּרְכוּן בְּדִיל בְּנָיךְ כֹּל עַמְמַיָּא דְאַרְעָא חֲלַף דִּי קַבֵּלְתָּא לְמֵימְרִי:

{יט} וַיָּ֤שָׁב אַבְרָהָם֙ אֶל-נְעָרָ֔יו וַיָּקֻ֛מוּ וַיֵּֽלְכ֥וּ יַחְדָּ֖ו אֶל-בְּאֵ֣ר שָׁ֑בַע וַיֵּ֥שֶׁב אַבְרָהָ֖ם בִּבְאֵ֥ר שָֽׁבַע:  מ  וַיָּ֤שָׁב אַבְרָהָם֙ אֶל-נְעָרָ֔יו וַיָּקֻ֛מוּ וַיֵּֽלְכ֥וּ יַחְדָּ֖ו אֶל-בְּאֵ֣ר שָׁ֑בַע וַיֵּ֥שֶׁב אַבְרָהָ֖ם בִּבְאֵ֥ר שָֽׁבַע:  ת  וְתַב אַבְרָהָם לְוָת עוּלֵימוֹהִי וְקָמוּ וַאֲזָלוּ כַחֲדָא לִבְאֵר שָׁבַע וִיתֵיב אַבְרָהָם בִּבְאֵר שָׁבַע: (פ)

{כ}  מפטיר  וַיְהִ֗י אַֽחֲרֵי֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וַיֻּגַּ֥ד לְאַבְרָהָ֖ם לֵאמֹ֑ר הִ֠נֵּ֠ה יָֽלְדָ֨ה מִלְכָּ֥ה גַם-הִ֛וא בָּנִ֖ים לְנָח֥וֹר אָחִֽיךָ:  מ  וַיְהִ֗י אַֽחֲרֵי֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וַיֻּגַּ֥ד לְאַבְרָהָ֖ם לֵאמֹ֑ר הִ֠נֵּ֠ה יָֽלְדָ֨ה מִלְכָּ֥ה גַם-הִ֛וא בָּנִ֖ים לְנָח֥וֹר אָחִֽיךָ:  ת  וַהֲוָה בָּתַר פִּתְגָּמַיָּא הָאִלֵּין וְאִתְחֲוָא לְאַבְרָהָם לְמֵימָר הָא יְלֵידַת מִלְכָּה אַף הִיא בְּנִין לְנָחוֹר אָחוּךְ:

{כא} אֶת-ע֥וּץ בְּכֹר֖וֹ וְאֶת-בּוּז אָחִ֑יו וְאֶת-קְמוּאֵ֖ל אֲבִ֥י אֲרָֽם:  מ  אֶת-ע֥וּץ בְּכֹר֖וֹ וְאֶת-בּוּז אָחִ֑יו וְאֶת-קְמוּאֵ֖ל אֲבִ֥י אֲרָֽם:  ת  יָת עוּץ בּוּכְרֵהּ וְיָת בּוּז אָחוּהִי וְיָת קְמוּאֵל אֲבוּהִי דַּאֲרָם:

{כב} וְאֶת-כֶּ֣שֶׂד וְאֶת-חֲז֔וֹ וְאֶת-פִּלְדָּ֖שׁ וְאֶת-יִדְלָ֑ף וְאֵ֖ת בְּתוּאֵֽל:  מ  וְאֶת-כֶּ֣שֶׂד וְאֶת-חֲז֔וֹ וְאֶת-פִּלְדָּ֖שׁ וְאֶת-יִדְלָ֑ף וְאֵ֖ת בְּתוּאֵֽל:  ת  וְיָת כֶּשֶׂד וְיָת חֲזוֹ וְיָת פִּלְדָּשׁ וְיָת יִדְלָף ויָת בְּתוּאֵל:

{כג} וּבְתוּאֵ֖ל יָלַ֣ד אֶת-רִבְקָ֑ה שְׁמֹנָ֥ה אֵ֨לֶּה֙ יָֽלְדָ֣ה מִלְכָּ֔ה לְנָח֖וֹר אֲחִ֥י אַבְרָהָֽם:  מ  וּבְתוּאֵ֖ל יָלַ֣ד אֶת-רִבְקָ֑ה שְׁמֹנָ֥ה אֵ֨לֶּה֙ יָֽלְדָ֣ה מִלְכָּ֔ה לְנָח֖וֹר אֲחִ֥י אַבְרָהָֽם:  ת  וּבְתוּאֵל:אוֹלִיד יָת רִבְקָה תְּמַנְיָא אִלֵּין יְלֵידַת מִלְכָּה לְנָחוֹר אֲחוּהִי דְאַבְרָהָם:

{כד} וּפִֽילַגְשׁ֖וֹ וּשְׁמָ֣הּ רְאוּמָ֑ה וַתֵּ֤לֶד גַּם-הִוא֙ אֶת-טֶ֣בַח וְאֶת-גַּ֔חַם וְאֶת-תַּ֖חַשׁ וְאֶת-מַֽעֲכָֽה:  מ  וּפִֽילַגְשׁ֖וֹ וּשְׁמָ֣הּ רְאוּמָ֑ה וַתֵּ֤לֶד גַּם-הִוא֙ אֶת-טֶ֣בַח וְאֶת-גַּ֔חַם וְאֶת-תַּ֖חַשׁ וְאֶת-מַֽעֲכָֽה:  ת  וּלְחֵנָתֵהּ וּשְׁמַהּ רְאוּמָה וִילֵידַת אַף הִיא יָת טֶבַח וְיָת גַּחַם וְיָת תַּחַשׁ וְיָת מַעֲכָה: (פפפ)
