פרק א - משנה א

חַטַּאת הָעוֹף נַעֲשֵׂית לְמַטָּה, וְחַטַּאת בְּהֵמָה לְמַעְלָה. עוֹלַת הָעוֹף נַעֲשֵׂית לְמַעְלָה, וְעוֹלַת הַבְּהֵמָה לְמַטָּה. אִם שִׁנָּה בָּזֶה וּבָזֶה, פָּסוּל. סֵדֶר קִנִּים {ג} {ד} כָּךְ הוּא. הַחוֹבָה, אֶחָד חַטָּאת וְאֶחָד עוֹלָה. בִּנְדָרִים וּנְדָבוֹת, כֻּלָּן עוֹלוֹת. אֵיזֶהוּ נֶדֶר, הָאוֹמֵר הֲרֵי עָלַי עוֹלָה. וְאֵיזוֹ הִיא נְדָבָה, הָאוֹמֵר הֲרֵי זוֹ עוֹלָה. מַה בֵּין נְדָרִים לִנְדָבוֹת. אֶלָּא שֶׁהַנְּדָרִים, מֵתוּ אוֹ נִגְנְבוּ, חַיָּבִים בְּאַחֲרָיוּתָם. וּנְדָבוֹת, מֵתוּ אוֹ נִגְנְבוּ, אֵין חַיָּבִים בְּאַחֲרָיוּתָן:

 ר"ע מברטנורה  חַטַּאת הָעוֹף נַעֲשֵׂית לְמַטָּה. מִחוּט הַסִּקְרָא {א}. דִּכְתִיב (וַיִּקְרָא ה) וְהִזָּה מִדַּם הַחַטָּאת עַל קִיר הַמִּזְבֵּחַ וְהַנִּשְׁאָר בַּדָּם יִמָּצֶה אֶל יְסוֹד הַמִּזְבֵּחַ, קִיר שֶׁהַשְּׁיָרִים שֶׁלּוֹ מִתְמַצִּין לַיְסוֹד, וְזֶהוּ קִיר הַתַּחְתּוֹן לְמַטָּה מִחוּט הַסִּקְרָא. דְּאִי קִיר הָעֶלְיוֹן דְּהַיְנוּ מִן הַחוּט וּלְמַעְלָה, פְּעָמִים שֶׁהוּא מִתְמַצֶּה לַסּוֹבֵב, כְּגוֹן שֶׁעוֹשֵׂהוּ לְמַעְלָה מִן הַסּוֹבֵב, שֶׁהַחוּט לְמַטָּה מִן הַסּוֹבֵב אַמָּה: וְחַטַּאת בְּהֵמָה לְמַעְלָה. דִּכְתִיב קַרְנוֹת, בְּגוּפָהּ שֶׁל קֶרֶן: עוֹלַת הָעוֹף לְמַעְלָה. דִּכְתִיב (שָׁם א) וּמָלַק וְהִקְטִיר וְנִמְצָה דָמוֹ, מַה הַקְטָרָה בְּרֹאשׁוֹ שֶׁל מִזְבֵּחַ, אַף מְלִיקָה וּמִצּוּי בְּרֹאשׁוֹ שֶׁל מִזְבֵּחַ: וְשֶׁל בְּהֵמָה לְמַטָּה. דִּכְתִיב בָּהּ יְסוֹד: וְאִם שִׁנָּה בָּזֶה וּבָזֶה פָּסוּל. הַאי דְּפָסוּל אִם עָשָׂה מַעֲשֵׂה חַטַּאת הָעוֹף לְמַעְלָה, דַּוְקָא בַהַזָּאָה {ב}. אֲבָל הַמְּלִיקָה שֶׁל חַטָּאת, אַף לְמַעְלָה כְּשֵׁרָה, דִּמְלִיקָה בְּכָל מָקוֹם בַּמִּזְבֵּחַ כְּשֵׁרָה. וְעוֹלַת הָעוֹף לֵית בָּהּ הַזָּאָה כִּי אִם מִצּוּי, וְאִם עֲשָׂאָהּ לְמַטָּה, פְּסוּלָה: סֵדֶר קִנִּין שֶׁל חוֹבָה אֶחָד חַטָּאת וְאֶחָד עוֹלָה. כְּגוֹן זָב וְזָבָה וְיוֹלֶדֶת, וְטֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו, וּשְׁמִיעַת קוֹל אָלָה, וּבִטּוּי שְׂפָתַיִם, דְּכֻלָּן מְבִיאִין קֵן בְּדַלּוּת, אֶלָּא שֶׁזָּב וְזָבָה מְבִיאִין קֵן בֵּין בְּדַלּוּת בֵּין בַּעֲשִׁירוּת. וְאַף עַל גַּב דְּאִיכָּא קִנִּין שֶׁל גֵּר דְּהֵן חוֹבָה וּשְׁנֵיהֶן עוֹלוֹת, לֹא חַיֵשׁ תַּנָּא עֲלַיְהוּ, מִשּׁוּם דְּלֹא שְׁכִיחֵי כֻּלֵּי הַאי {ה}: בִּנְדָרִים וּנְדָבוֹת כֻּלָּן עוֹלוֹת. לְפִי שֶׁאֵין בָּאִין בְּנֶדֶר וּנְדָבָה אֶלָּא עוֹלוֹת וּשְׁלָמִים בִּלְבַד, וְאֵין עוֹפוֹת בָּאִים שְׁלָמִים, הִלְכָּךְ נְדָרִים וּנְדָבוֹת שֶׁבְּעוֹפוֹת כֻּלָּן עוֹלוֹת: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} שֶׁאֵין בַּמִּזְבֵּחַ כִּי אִם שְׁנֵי מְקוֹמוֹת. תּוֹסָפוֹת: {ב} וּבְמִצּוּי פְּלוּגְתָּא בִּמְעִילָה דַּף ח'. דְּאִיכָּא לְמַאן דְּאָמַר דְּלֹא מְעַכֵּב. מְפָרֵשׁ: {ג} קִנִּים. קֵן קָרוּי שְׁתֵּי תוֹרִים אוֹ שְׁנֵי בְנֵי יוֹנָה. פְּרִידָה קְרוּיָה יוֹנָה אַחַת אוֹ תּוֹר אֶחָד. מְפָרֵשׁ: {ד} קִנִּים. הַקּוֹ"ף בְּחִירִי"ק, כְּמוֹ חֵץ חִצִּים. וְקַן צִפּוֹר שֶׁבְּפַתַּ"ח, מִפְּנֵי הַסְּמִיכוּת: {ה} הָרַמְבַּ"ם. וְגַם כֵּן שֶׁאֵינוֹ מְפֹרָשׁ בַּתּוֹרָה. וְכָתַב עוֹד הָרַמְבַּ"ם, וּמַה שֶּׁאָמַרְנוּ שֶׁהַחוֹבָה אֶחָד עוֹלָה וְאֶחָד חַטָּאת הֲרֵי הוּא כְּפִי עִקַּר הַחִיּוּב הָרִאשׁוֹן. וְאִם כְּבָר הִקְרִיב מִקְצָת קָרְבְּנוֹתָיו עַד שֶׁיִּהְיֶה הָאִישׁ הַזֶּה דֶּרֶךְ מָשָׁל יֵשׁ עָלָיו עֶשְׂרִים עוֹלוֹת וְעֶשֶׂר חַטָּאוֹת כֻּלָּן חוֹבָה, אוֹ בְהֶפֶךְ, וּלְפִיכָךְ אָמַר בְּמַסֶּכֶת זוֹ מִנְיַן חַטָּאוֹת שֶׁבַּחוֹבָה וּמִנְיַן עוֹלוֹת שֶׁבַּחוֹבָה, וְלֹא אָמַר חֲצִי הַחוֹבָה: (עיקר תוי"ט).

פרק א - משנה ב

חַטָּאת שֶׁנִּתְעָרְבָה בְעוֹלָה וְעוֹלָה בְחַטָּאת, אֲפִלּוּ אֶחָד בְּרִבּוֹא, יָמוּתוּ כֻלָּם {ו}. חַטָּאת שֶׁנִּתְעָרְבָה בְחוֹבָה, אֵין כָּשֵׁר אֶלָּא כְמִנְיַן חַטָּאוֹת שֶׁבַּחוֹבָה. וְכֵן עוֹלָה שֶׁנִּתְעָרְבָה בְחוֹבָה, אֵין כָּשֵׁר אֶלָּא כְמִנְיַן עוֹלוֹת שֶׁבַּחוֹבָה, בֵּין שֶׁהַחוֹבָה מְרֻבָּה וְהַנְּדָבָה מְמֻעֶטֶת, בֵּין שֶׁהַנְּדָבָה מְרֻבָּה וְהַחוֹבָה מְמֻעֶטֶת, בֵּין שֶׁשְּׁתֵּיהֶן שָׁווֹת:

 ר"ע מברטנורה  אֲפִלּוּ אֶחָד בְּרִבּוֹא יָמוּתוּ כֻלָּן. וְלֹא בָטְלֵי בְרֻבָּא, דְּבַעֲלֵי חַיִּים חֲשִׁיבֵי וְלֹא בָטְלֵי: חַטָּאת שֶׁנִּתְעָרְבָה בְחוֹבָה. בֶּן יוֹנָה אוֹ תוֹר שֶׁל חַטָּאת שֶׁנִּתְעָרְבָה בִּשְׁנֵי קִנִּין שֶׁל חוֹבָה, דְּהַיְנוּ קִנִּין סְתָמָא שֶׁל יוֹלֶדֶת אוֹ שֶׁל זָבָה, שֶׁכָּל קֵן מֵהֶן הוּא אֶחָד עוֹלָה וְאֶחָד חַטָּאת, וְנִמְצָא עַתָּה חָמֵשׁ פְּרִידוֹת מְעֹרָבוֹת יַחַד: אֵין כָּשֵׁר אֶלָּא מִנְיַן חַטָּאוֹת שֶׁבַּחוֹבָה. וְאֵינוֹ יָכוֹל לְהָבִיא מֵאֵלּוּ הֶחָמֵשׁ פְּרִידוֹת רַק שְׁתֵּי חַטָּאוֹת בִּלְבַד, שֶׁזֶּהוּ מִנְיָן הַכָּשֵׁר בִּשְׁנֵי קִנִּין שֶׁל חוֹבָה. דְּאִי עֲבִיד שָׁלֹשׁ חַטָּאוֹת, שֶׁמָּא מִן הַשְּׁנֵי קִנִּין עֲבִיד לְהוּ וְלֹא מִן הַמְּעֹרָבוֹת בָּהֶן, וּמִשְּׁנֵי קִנִּין אֵין יְכוֹלִים לַעֲשׂוֹת רַק שְׁתֵּי חַטָּאוֹת. וְלֹא מָצֵי עֲבִיד אֲפִלּוּ עוֹלָה אַחַת, דְּשֶׁמָּא יִקַּח הַחַטָּאת הַמְּעֹרֶבֶת: וְכֵן עוֹלָה. בֶּן יוֹנָה אוֹ תוֹר שֶׁהֻפְרַשׁ לְשֵׁם עוֹלָה: שֶׁנִּתְעָרְבָה בְחוֹבָה. בִּשְׁנֵי קִנִּין סְתָמָא: אֵין כָּשֵׁר אֶלָּא מִנְיַן עוֹלוֹת שֶׁבַּחוֹבָה. כִּדְפָרֵישְׁנָא לְעֵיל גַּבֵּי חַטָּאת: בֵּין שֶׁהַחוֹבָה מְרֻבָּה וְהַנְּדָבָה מֻעֶטֶת. כְּגוֹן שֶׁקֵּן אֶחָד מְפֹרָשׁ שֶׁהֵן עוֹלוֹת, נִתְעָרֵב עִם כַּמָּה קִנִּין סְתָמָא שֶׁל יוֹלֶדֶת אוֹ שֶׁל זָבָה שֶׁכָּל קֵן מֵהֶן אֶחָד חַטָּאת וְאֶחָד עוֹלָה: בֵּין וְכוּ'. כְּגוֹן קֵן אֶחָד סְתוּמָה, שֶׁנִּתְעָרֵב בְּכַמָּה קִנִּין שֶׁל עוֹלוֹת. וְקָרִינַן לְעוֹלוֹת מְפֹרָשׁוֹת נְדָבָה, מִשּׁוּם דִּנְדָרִים וּנְדָבוֹת כֻּלָּן עוֹלוֹת כְּדִתְנַן בְּרֵישׁ פִּרְקִין {ז}: (ר"ע מברטנורה).

 עיקר תוי"ט  {ו} יָמוּתוּ כוּ'. דִּבְכָל אֶחָד אִיכָּא לְסַפּוֹקֵי שֶׁמָּא עוֹלָה וּפְסוּלָה לְמַטָּה, אוֹ שֶׁמָּא חַטָּאת וּפְסוּלָה לְמַעְלָה. רַשִׁ"י: {ז} וּנְדָבָה דְנָקַט, וְלֹא נָקַט חַטָּאת מֻעֶטֶת בַּחוֹבָה עוֹלָה מֻעֶטֶת בַּחוֹבָה, מִשּׁוּם דְּקָתָנֵי בַסֵּיפָא אֲבָל חוֹבָה שֶׁנִּתְעָרְבָה זוֹ בָזוֹ, לְהָכִי נָקַט נְדָבָה. מְפָרֵשׁ: (עיקר תוי"ט).

פרק א - משנה ג

בַּמֶּה דְבָרִים אֲמוּרִים, בְּחוֹבָה וּבִנְדָבָה. אֲבָל בְּחוֹבָה שֶׁנִּתְעָרְבָה זוֹ בָזוֹ, אַחַת לָזוֹ {ח} וְאַחַת לָזוֹ, שְׁתַּיִם לָזוֹ וּשְׁתַּיִם לָזוֹ, שָׁלשׁ לָזוֹ וְשָׁלשׁ לָזוֹ, מֶחֱצָה כָּשֵׁר {ט} וּמֶחֱצָה פָּסוּל. אַחַת לָזוֹ וּשְׁתַּיִם לָזוֹ, וְשָׁלשׁ לָזוֹ, וְעֶשֶׂר לָזוֹ, וּמֵאָה לָזוֹ, הַמֻּעָט כָּשֵׁר, בֵּין מִשֵּׁם אֶחָד, בֵּין מִשְּׁנֵי שֵׁמוֹת, בֵּין מֵאִשָּׁה אַחַת, בֵּין מִשְּׁתֵּי נָשִׁים:

 ר"ע מברטנורה  בְּחוֹבָה וּבִנְדָבָה. שֶׁנִּתְעָרְבוּ קִנֵּי חוֹבָה שֶׁמִּקְצָתָן חַטָּאוֹת וּמִקְצָתָן עוֹלוֹת עִם קִנֵּי נְדָבָה שֶׁכֻּלָּן עוֹלוֹת: אֲבָל בְּחוֹבָה שֶׁנִּתְעָרְבָה זוֹ בָזוֹ. קֵן סְתוּמָה שֶׁל יוֹלֶדֶת אוֹ שֶׁל זָבָה שֶׁנִּתְעָרֵב בְּקֵן שֶׁל יוֹלֶדֶת אוֹ שֶׁל זָבָה אַחֶרֶת: מֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל. וּמַקְרִיב מֵהָאַרְבָּעָה הַמְעֹרָבִים שְׁנַיִם בִּלְבַד, אֶחָד לְעוֹלָה וְאֶחָד לְחַטָּאת וְלֹא יוֹתֵר, דְּאִי עֲבִיד שְׁתֵּי עוֹלוֹת שֶׁמָּא יַעֲשֶׂה אוֹתָן מִקֵּן אֶחָד, וְכָל קֵן סְתוּמָה הוּא אֶחָד עוֹלָה וְאֶחָד חַטָּאת: הַמֻּעָט כָּשֵׁר. אִם נִתְעָרֵב קֵן אֶחָד סְתוּמָה בִּשְׁנֵי קִנִּים אֲחֵרִים אוֹ בִּשְׁלֹשָׁה, אֵינוֹ יָכוֹל לְהַקְרִיב אֶלָּא קֵן אֶחָד, פְּרִידָה אַחַת לְחַטָּאת וּפְרִידָה אַחַת לְעוֹלָה. אֲבָל שְׁתֵּי עוֹלוֹת לֹא מָצֵי עֲבִיד, שֶׁמָּא יִקַּח הַקֵּן שֶׁהוּא מְבֹרָר תְּחִלָּה לְיוֹלֶדֶת אַחַת שֶׁאֵינוֹ יָכוֹל לַעֲשׂוֹת מִמֶּנּוּ אֶלָּא אֶחָד חַטָּאת וְאֶחָד עוֹלָה. וְכֵן אִם נִתְעָרְבוּ עֲשָׂרָה קִנִּין בְּמֵאָה, אֵינוֹ מַקְרִיב מִכֻּלָּן אֶלָּא עֲשָׂרָה קִנִּין, מִקְצָתָן חַטָּאוֹת וּמִקְצָתָן עוֹלוֹת, וְהַשְּׁאָר פְּסוּלִין, וְזֶהוּ הַמֻּעָט כָּשֵׁר: בֵּין מִשֵּׁם אֶחָד. לֵידָה וְלֵידָה, זִיבָה וְזִיבָה, מִשְּׁתֵּי נָשִׁים אוֹ מֵאִשָּׁה אֶחָת: בֵּין מִשְּׁנֵי שֵׁמוֹת. לֵידָה וְזִיבָה, בֵּין מִשְּׁתֵּי נָשִׁים בֵּין מֵאִשָּׁה אַחַת. כְּדִמְפָרֵשׁ וְאָזֵיל: (ר"ע מברטנורה).

 עיקר תוי"ט  {ח} לָזוֹ כוּ'. נָתַן דֻּגְמָאוֹת לְכָל הַמַּסֶּכְתָּא בְּקָרְבְּנוֹת הַנָּשִׁים, לְפִי שֶׁהֵן יוֹתֵר חַיָּבוֹת בְּקִנִּים, מִפְּנֵי שֶׁחַיָּבוֹת בָּהֶן כְּמוֹ שֶׁחַיָּבִים הָאֲנָשִׁים, וּמוֹסִיף עַל הָאֲנָשִׁים יוֹלְדוֹת, וְגַם כֵּן שֶׁהַזִּיבוּת מָצוּי הַרְבֵּה בְנָשִׁים יוֹתֵר מִמַּה שֶׁהוּא מָצוּי בָּאֲנָשִׁים כְּמוֹ שֶׁהוּא מְפֻרְסָם וְנִגְלֶה לָעַיִן. הָרַמְבַּ"ם: {ט} מֶחֱצָה כָּשֵׁר. אַף עַל גַּב דְּאֶפְשָׁר שֶׁמַּקְרִיב מִשֶּׁל שְׁתֵּי נָשִׁים, הֲרֵי יָכוֹל לְהַקְרִיב לְשֵׁם מִי שֶׁהוּא, וַאֲנַן לֹא קָיְמִינַן אֶלָּא בַפְּרִידוֹת גּוּפַיְהוּ כַּמָּה כְשֵׁרִים וְכַמָּה פְסוּלִין מִצַּד עַצְמָן. וְעַיֵּן בְּסוֹף פִּרְקִין. וְכָל זֶה בְּכֹהֵן נִמְלָךְ כְּדִתְנַן בְּרֵישׁ פֶּרֶק ג': (עיקר תוי"ט).

פרק א - משנה ד

כֵּיצַד מִשֵּׁם אֶחָד, לֵידָה וְלֵידָה, זִיבָה וְזִיבָה, מִשֵּׁם אֶחָד. מִשְּׁנֵי שֵׁמוֹת, לֵידָה וְזִיבָה. כֵּיצַד שְׁתֵּי נָשִׁים, עַל זוֹ לֵידָה וְעַל זוֹ לֵידָה, עַל זוֹ זִיבָה וְעַל זוֹ זִיבָה, מִשֵּׁם אֶחָד. מִשְּׁנֵי שֵׁמוֹת, עַל זוֹ לֵידָה וְעַל זוֹ זִיבָה. רַבִּי יוֹסֵי אוֹמֵר, שְׁתֵּי נָשִׁים שֶׁלָּקְחוּ קִנֵּיהֶן בְּעֵרוּב, אוֹ שֶׁנָּתְנוּ דְמֵי קִנֵּיהֶן לַכֹּהֵן, לְאֵיזוֹ שֶׁיִּרְצֶה כֹהֵן יַקְרִיב חַטָּאת, וּלְאֵיזוֹ שֶׁיִּרְצֶה כֹהֵן יַקְרִיב עוֹלָה, בֵּין מִשֵּׁם אֶחָד, בֵּין מִשְּׁנֵי שֵׁמוֹת:

 ר"ע מברטנורה  רַבִּי יוֹסֵי אוֹמֵר כוּ'. לָאו מִשּׁוּם דְּסָבַר רַבִּי יוֹסֵי יֵשׁ בְּרֵרָה וְנֵימָא דִּכְשֶׁהִקְרִיב הַכֹּהֵן קֵן אֶחָד לְשֵׁם אַחַת מֵהֶן הֻבְרַר הַדָּבָר לְמַפְרֵעַ דְּמִתְּחִלָּה בִּשְׁעַת לְקִיחַת הַקֵּן הָיָה קֵן זֶה שֶׁלָּהּ {י}, אֶלָּא מְפָרֵשׁ בַּגְּמָרָא בְּפֶרֶק בַּכֹּל מְעָרְבִין [עֵרוּבִין דַּף לו] דְּרַבִּי יוֹסֵי מַיְרֵי בְּשֶׁהִתְנוּ בִשְׁעַת לְקִיחָה שֶׁיּוּכַל הַכֹּהֵן לַעֲשׂוֹת קֵן שֶׁיִּרְצֶה לְשֵׁם אֵיזוֹ אִשָּׁה שֶׁיִּרְצֶה. וַהֲלָכָה כְּרַבִּי יוֹסֵי: (ר"ע מברטנורה).

 עיקר תוי"ט  {י} דְּאִי אֵין בְּרֵרָה, אִיכָּא לְמֵימַר כָּל קַן רָחֵל הִקְרִיב לְמַעְלָה, וְכָל קַן לֵאָה הִקְרִיב לְמַטָּה, וְהִקְרִיב עוֹלָה לְמַטָּה וְחַטָּאת לְמַעְלָה כוּ', וְקָא מַשְׁמַע לָן כִּדְרַב חִסְדָּא דְּאָמַר אֵין הַקִּנִּים מִתְפָּרְשׁוֹת אֶלָּא אוֹ בִּלְקִיחַת בְּעָלִים אוֹ בַּעֲשִׂיַּת כֹּהֵן, דִּכְתִיב [וַיִּקְרָא יב] וְלָקַח וְגוֹ' אֶחָד לְעוֹלָה וְאֶחָד לְחַטָּאת, וּכְתִיב [שָׁם טו] וְעָשָׂה הַכֹּהֵן אֶת הָאֶחָד חַטָּאת וְאֶת הָאֶחָד עוֹלָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ב - משנה א

קֵן סְתוּמָה שֶׁפָּרַח מִמֶּנָּה גוֹזָל לָאֲוִיר, אוֹ שֶׁפָּרַח בֵּין הַמֵּתוֹת {ב}, אוֹ שֶׁמֵּת אַחַד מֵהֶן, יִקַּח זוּג לַשֵּׁנִי. פָּרַח לְבֵין הַקְּרֵבוֹת, פָּסוּל וּפוֹסֵל אֶחָד כְּנֶגְדּוֹ, שֶׁהַגּוֹזָל הַפּוֹרֵחַ, פָּסוּל וּפוֹסֵל אֶחָד כְּנֶגְדּוֹ:

 ר"ע מברטנורה  קֵן סְתוּמָה. שֶׁלֹּא פֵרֵשׁ זוֹ חַטָּאת וְזוֹ עוֹלָה. וְלֹא מִבַּעְיָא מְפֹרֶשֶׁת, דִּפְשִׁיטָא שֶׁאִם פָּרַח הָאֶחָד יִקַּח זוּג לַשֵּׁנִי כֵּיוָן שֶׁהֵן נִפְרָשִׁים זֶה מִזֶּה וְכָל אֶחָד שְׁמוֹ עָלָיו {א}. אֶלָּא אֲפִלּוּ סְתוּמָה שֶׁעֲדַיִן הֵם יַחַד וְעוֹמְדִים לְהִתְפָּרֵשׁ וְסָלְקָא דַּעְתָּךְ אֲמֵינָא שֶׁאִם פָּרַח הָאֶחָד יָמוּת הַשֵּׁנִי, קָמַשְׁמַע לָן: לְבֵין הַמֵּתוֹת. לְבֵין קִנִּין שֶׁמִּשְׁפָּטָן שֶׁיָּמוּתוּ כֻלָּם, כְּגוֹן חַטָּאת שֶׁנִּתְעָרְבָה בְעוֹלָה דִּתְנַן בְּהוּ בְּפֶרֶק קַמָּא יָמוּתוּ כֻלָּם: לְבֵין הַקְּרֵבוֹת. לְבֵין קִנִּין סְתוּמוֹת הָעוֹמְדוֹת לִקָּרֵב: פָּסוּל וּפוֹסֵל אֶחָד כְּנֶגְדּוֹ. לְפִי שֶׁהוּא סְתָם וְלֹא נִתְפָּרֵשׁ אִם חַטָּאת אִם עוֹלָה, לְפִיכָךְ אֵינוֹ פוֹסֵל אֶלָּא אֶחָד כְּנֶגְדּוֹ, כְּדִמְפָרֵשׁ וְאָזֵיל: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} מְפָרֵשׁ. וּקְצָת קָשֶׁה, דִּלְפִי זֶה רֵישָׁא נָקַט סְתוּמָה בְּלָאו דַּוְקָא, וּבַסֵּיפָא פָּרַח לְבֵין הַקְּרֵבוֹת אִי אֶפְשָׁר לְפָרֵשׁ אֶלָּא דַּוְקָא סְתוּמָה. וְהַתּוֹסָפוֹת כָּתְבוּ קֵן סְתוּמָה הוּא הַדִּין בְּקֵן מְפֹרָשׁ הָוֵי דִּין זֶה, אֶלָּא מִשּׁוּם דִּבְסֵיפָא קָאָמַר פָּרַח כוּ', דְּדַוְקָא פְרִידָה שֶׁל קֵן סְתוּמָה שֶׁפָּרְחָה וְנִתְעָרְבָה לְבֵין הַקְּרֵבוֹת, אֲבָל פְּרִידָה הַמְּפֹרֶשֶׁת לְעוֹלָה שֶׁפָּרְחָה בֵּין שְׁאָר קִנִּין אֵינוֹ יָכוֹל לְהַקְרִיב אֶלָּא כְמִנְיַן עוֹלוֹת שֶׁבַּקִּנִּים שֶׁפָּרַח לְשָׁם, כִּדְאָמַר סוֹף פֶּרֶק קַמָּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} הַמֵּתוֹת. לְהָכִי קָאָמַר בֵּין חַטָּאוֹת הַמֵּתוֹת, דְּאִלּוּ נִתְעָרְבָה בֵּין חַטָּאוֹת חַיּוֹת וְהָנָךְ חַטָּאוֹת דִּידֵיהּ נִינְהוּ, מָצֵי לְמִסַּב חַד מִנַּיְהוּ וַעֲבִיד לֵיהּ זוּג לְהָא, וְהָא דְּלֹא נִתְעָרֵב הֲוָה קָרֵי לֵיהּ עוֹלָה. גִּלְיוֹן רַשִׁ"י: (עיקר תוי"ט).

כְּלוּם הָיָה יָכוֹל לְהַקְרִיב בַּחֲמִישִׁית שָׁלֹשׁ עוֹלוֹת וְשָׁלֹשׁ חַטָּאוֹת, דַּהֲרֵי אוֹתָן שֶׁהָלְכוּ בָרְבִיעִית אֵינָן קְרֵבוֹת, וּלְפִי זֶה שֶׁהַחֲמִישִׁית אֵין לָהּ אֶלָּא אַחַת הָיָה יָכוֹל לְהַקְרִיב בָּרְבִיעִית שְׁתֵּי עוֹלוֹת וּשְׁתֵּי חַטָּאוֹת. אֶלָּא גָּזְרִינַן בְּכָל פְּרִיחָה וַחֲזָרָה שְׁנֵי קִנִּין [גְּזֵרָה בָזוֹ אַטּוּ בָזוֹ]. וְכֵן כִּי פָרַח וְחָזַר פַּעַם שְׁלִישִׁית אֵין לָהֶם כְּלוּם לְבַד הַשְּׁבִיעִית שֶׁיֵּשׁ לָהּ אַרְבַּע, כְּמוֹ כֵן גָּזְרִינַן בְּכָל פְּרִיחָה וַחֲזָרָה שְׁנֵי קִנִּין: וְיֵשׁ אוֹמְרִים הַשְּׁבִיעִית לֹא הִפְסִידָה כְלוּם. פֵּרוּשׁ, בִּפְרִיחָה וַחֲזָרָה שְׁלִישִׁית. אֶלָּא לְעוֹלָם יַקְרִיבוּ מִמֶּנָּה חֲמִשָּׁה קִנִּין {ה}, דְּשָׁלֹשׁ פְּרִידוֹת פָּרְחוּ מִמֶּנָּה לְכָל הַיּוֹתֵר וְנִשְׁאֲרוּ בָהּ רַק אַחַת עֶשְׂרֵה פְרִידוֹת, וְיַעֲשֶׂה מֵהֶם חָמֵשׁ חַטָּאוֹת וְחָמֵשׁ עוֹלוֹת, כֵּיוָן דְּאַף הַשִּׁשִּׁית אֵין לָהּ כְּלוּם. וְהָכָא לֹא גָזְרִינַן בִּפְרִיחָה זוֹ כְּמוֹ בִּשְׁאָר קִנִּין, דְּכֵיוָן דְּאֵינָהּ מַפְסֶדֶת בַּפְּרִיחָה בְּשׁוּם פַּעַם, לֹא גָזְרוּ עַל הַחֲזָרָה. וְאֵין הֲלָכָה כְּיֵשׁ אוֹמְרִים: (ר"ע מברטנורה).

 עיקר תוי"ט  {ה} אֲבָל אֵין לְפָרֵשׁ לֹא הִפְסִידָה כְּלוּם וְיַקְרִיבוּ בָּהּ שֶׁבַע חַטָּאוֹת וְשֶׁבַע עוֹלוֹת דְּמָצֵי עֲבִיד לְהוּ כֵּיוָן דִּבְקִנִּין אֲחֵרִים אֵין בָּהֶם כְּלוּם וְאוֹתָן שָׁלֹשׁ שֶׁנִּתְעָרְבוּ בָהּ מֵאֲחֵרִים רְאוּיִין הֵם לְהַקְרִיב כֻּלָּם בֵּין עוֹלוֹת בֵּין חַטָּאוֹת, שֶׁבְּכָל פַּעַם שֶׁבָּא לָהּ אֶחָד הָיָה בָא מִמָּקוֹם שֶׁרָאוּי לְהַקְרִיב בֵּין עוֹלָה בֵּין חַטָּאת, דְּאֵין זֶה הַלָּשׁוֹן, דַּהֲוָה לֵיהּ לְמֵימַר נִכְשְׁרָה [נִשְׂכְּרָה] שְׁבִיעִית בִּפְרִיחָה שְׁלִישִׁית, שֶׁעַל יְדֵי פְרִיחָה שְׁלִישִׁית הֻכְשְׁרוּ כָל הַקִּנִּין שֶׁבַּשְּׁבִיעִית, אֶלָּא הַלָּשׁוֹן מַשְׁמַע לֹא הִפְסִידָה כְלוּם בְּזֹאת הַפְּרִיחָה שְׁלִישִׁית. מְפָרֵשׁ: (עיקר תוי"ט).

פרק ב - משנה ד

קֵן סְתוּמָה וְקֵן מְפֹרֶשֶׁת, פָּרַח מִן הַסְּתוּמָה לַמְפֹרֶשֶׁת, יִקַּח זוּג לַשֵּׁנִי. חָזַר, אוֹ שֶׁפָּרַח מִן הַמְפֹרֶשֶׁת רִאשׁוֹן, הֲרֵי כֻלָּן יָמוּתוּ:

 ר"ע מברטנורה  קֵן סְתוּמָה. שֶׁלֹּא פֵרֵשׁ אֵיזוֹ חַטָּאת וְאֵיזוֹ עוֹלָה: וְקֵן מְפֹרֶשֶׁת. שֶׁיָּדוּעַ שֶׁזּוֹ חַטָּאת וְזוֹ עוֹלָה. וְהָכָא מַיְרֵי שֶׁנִּתְעָרְבוּ שְׁתֵּי הַפְּרִידוֹת יַחַד אַחַר שֶׁהָיוּ מְפֹרָשׁוֹת וְהַשְׁתָּא אֵין יָדוּעַ אֵיזוֹ חַטָּאת וְאֵיזוֹ עוֹלָה, וְהָכִי מוּכָח בַּסֵּיפָא דְּקָתָנֵי אוֹ שֶׁפָּרַח מִן הַמְפֹרֶשֶׁת רִאשׁוֹן לַסְּתוּמָה יָמוּתוּ כֻלָּם, וְאִם לֹא נִתְעָרְבוּ פְרִידוֹת הַמְפֹרָשׁוֹת זוֹ בָזוֹ אַמַּאי יָמוּתוּ, וְאִם הָעוֹלָה פָרְחָה, הַחַטָּאת הַנִּשְׁאֶרֶת תִּקָּרֵב, וְקֵן הַסְּתוּמָה יַקְרִיבוּ בָהּ עוֹלָה אַחַת מִפְּנֵי הָעוֹלָה שֶׁנִּתְעָרְבָה בָהּ. אֶלָּא מַיְרֵי שֶׁנִּתְעָרְבוּ פְרִידוֹת הַמְפֹרָשׁוֹת זוֹ בָזוֹ, וְאִם נִתְעָרְבָה פְרִידָה אַחַת מִן הַקֵּן הַסְּתוּמָה לְתוֹךְ הַמְפֹרֶשֶׁת, יִקַּח זוּג לַשֵּׁנִי וְיַפְרִישׁ אֶחָד חַטָּאת וְאֶחָד עוֹלָה, וְהַשְּׁלֹשָׁה שֶׁנִּתְעָרְבוּ יַחַד יָמוּתוּ כֻלָּן. חָזַר אֶחָד מִן הַשְּׁלֹשָׁה שֶׁנִּתְעָרְבוּ וְנִתְעָרֵב עִם הַיְחִידִי הַנִּשְׁאָר בַּסְּתוּמָה, אוֹ בִּתְחִלַּת הַפְּרִיחָה נִתְעָרֵב אֶחָד מִן הַמְפֹרָשׁוֹת לַסְּתוּמָה, וְהַיְנוּ רִאשׁוֹן דְּקָאָמַר, כְּלוֹמַר שֶׁבִּתְחִלַּת פְּרִיחָה הָלַךְ אֶחָד וְנִתְעָרֵב בַּסְּתוּמָה, יָמוּתוּ כֻלָּן, דְּכֵיוָן שֶׁפְּרִידוֹת הַמְפֹרָשׁוֹת נִתְעָרְבוּ יַחַד אֵינָן קְרֵבִין עוֹד, וְכִי פָרַח אֶחָד מֵהֶן בְּמָקוֹם אַחֵר הַיְנוּ אֶחָד מִן הַמֵּתוֹת שֶׁפָּרַח לְבֵין הַקְּרֵבוֹת, יָמוּתוּ כֻלָּן: (ר"ע מברטנורה).

פרק ב - משנה ה

חַטָּאת מִיכָּן וְעוֹלָה מִיכָּן וּסְתוּמָה בָאֶמְצַע, פָּרַח מִן הָאֶמְצַע לַצְּדָדִין, אֶחָד הֵלָךְ וְאֶחָד הֵלָךְ, לֹא הִפְסִיד כְּלוּם, אֶלָּא יֹאמַר, זֶה שֶׁהָלַךְ אֵצֶל חַטָּאוֹת, חַטָּאת. וְזֶה שֶׁהָלַךְ אֵצֶל עוֹלוֹת, עוֹלָה. חָזַר לָאֶמְצַע, הָאֶמְצָעִיִּים יָמוּתוּ, אֵלּוּ יִקְרְבוּ חַטָּאוֹת, וְאֵלּוּ יִקְרְבוּ עוֹלוֹת. חָזַר אוֹ שֶׁפָּרַח {ז} מִן הָאֶמְצַע לַצְּדָדִין, הֲרֵי כֻלָּן יָמוּתוּ. אֵין מְבִיאִין תּוֹרִין כְּנֶגֶד בְּנֵי יוֹנָה, וְלֹא בְנֵי יוֹנָה כְּנֶגֶד תּוֹרִין. כֵּיצַד. הָאִשָּׁה שֶׁהֵבִיאָה חַטָּאתָהּ תּוֹר, וְעוֹלָתָהּ בֶּן יוֹנָה, תִּכְפֹּל וְתָבִיא עוֹלָתָהּ תּוֹר. עוֹלָתָהּ תּוֹר, וְחַטָּאתָהּ בֶּן יוֹנָה, תִּכְפֹּל וְתָבִיא עוֹלָתָהּ בֶּן יוֹנָה. בֶּן עַזַּאי אוֹמֵר, הוֹלְכִין אַחַר הָרִאשׁוֹן. הָאִשָּׁה שֶׁהֵבִיאָה חַטָּאתָהּ וּמֵתָה, יָבִיאוּ הַיּוֹרְשִׁין {ט} עוֹלָתָהּ. עוֹלָתָהּ וּמֵתָה, לֹא יָבִיאוּ הַיּוֹרְשִׁין חַטָּאתָהּ:

 ר"ע מברטנורה  חַטָּאת מִכָּאן. פְּרִידָה הַמְפֹרֶשֶׁת לְחַטָּאת לְצַד אֶחָד, וּפְרִידָה הַמְפֹרֶשֶׁת לְעוֹלָה לַצַּד הַשֵּׁנִי, וּשְׁתֵּי פְרִידוֹת שֶׁל קֵן סְתוּמָה בָּאֶמְצַע: פָּרַח מִן הָאֶמְצַע. פְּרִידָה אַחַת לְיָמִין וּפְרִידָה אַחַת לִשְׂמֹאל: לֹא הִפְסִיד כְּלוּם. דְּאוֹתוֹ שֶׁמְּעֹרָב עִם הַחַטָּאת יַעֲשֶׂה חַטָּאת, וְאוֹתוֹ שֶׁעִם הָעוֹלָה יַעֲשֶׂה עוֹלָה, אֲבָל לֹא חַטָּאת, דְּשֶׁמָּא יַעֲשׂוּהוּ מִן הָעוֹלָה הַמְפֹרֶשֶׁת: חָזַר לָאֶמְצַע. חָזְרוּ מִן הַצְּדָדִים אֶחָד מִכָּאן וְאֶחָד מִכָּאן וְנִתְעָרְבוּ יַחַד, אֵלּוּ יָמוּתוּ, דְּחַטָּאת וְעוֹלָה נִתְעָרְבוּ יַחַד. אֲבָל אוֹתָן שֶׁנִּשְׁאֲרוּ כָּל אֶחָד לְבַדּוֹ, קְרֵבִים כָּל אֶחָד כְּדִינוֹ: חָזַר. מִן הָאֶמְצָעִיִּים לַצְּדָדִים, הֲרֵי חַטָּאוֹת וְעוֹלוֹת מְעֹרָבִים, וְיָמוּתוּ כֻלָּן {ו}: אֵין מְבִיאִים תּוֹרִים כְּנֶגֶד בְּנֵי יוֹנָה. אֶלָּא אוֹ שְׁנֵיהֶם תּוֹרִים אוֹ שְׁנֵיהֶן בְּנֵי יוֹנָה: תִּכְפֹּל וְתָבִיא עוֹלָתָהּ בֶּן יוֹנָה. כְּדִין הַחַטָּאת. שֶׁהַחַטָּאת הִיא עִקָּר {ח}, בֵּין שֶׁהִפְרִישַׁתּוּ תְחִלָּה בֵּין שֶׁהִפְרִישַׁתּוּ בַסּוֹף: הוֹלְכִין אַחַר הָרִאשׁוֹן. וְאֵינָהּ מְבִיאָה הַשֵּׁנִי אֶלָּא מִמִּין שֶׁהִפְרִישָׁה בָרִאשׁוֹן, בֵּין שֶׁהָיָה הָרִאשׁוֹן חַטָּאת בֵּין שֶׁהָיָה עוֹלָה. וַהֲלָכָה כְּתַנָּא קַמָּא: לֹא יָבִיאוּ הַיּוֹרְשִׁים חַטָּאתָהּ. דְּהַוְיָא חַטָּאת שֶׁמֵּתוּ בְעָלֶיהָ. וְאִי לֹא מֵהַאי טַעֲמָא, מְבִיאִים חַטָּאת אַף עַל פִּי שֶׁקָּרְבָה עוֹלָה תְּחִלָּה. וְאַף עַל גַּב דִּבְכָל מָקוֹם חַטָּאת קוֹדֶמֶת לְעוֹלָה, הַיְנוּ דַּוְקָא לְמִצְוָה אֲבָל לֹא לְעַכֵּב: (ר"ע מברטנורה).

נִמְלָךְ. וּמִיהוּ אַף בַּעֲשָׂרָה לָזוֹ וּמֵאָה לָזוֹ הַמְרֻבֶּה כָּשֵׁר, שֶׁאֵין פָּסוּל מֵהֶן אֶלָּא עֲשָׂרָה קִנִּים, דְּשֶׁמָּא כָּל הַקִּנִּים שֶׁל אִשָּׁה אַחַת לְמַעְלָה וְנִפְסְלוּ לְמַעְלָה. וְהוֹאִיל שֶׁקִּנִּים הָאֵלּוּ לִשְׁתֵּי נָשִׁים, יָבִיאוּ קֵן אֶחָד בְּשֻׁתָּפוּת וְיַתְנוּ בֵינֵיהֶן, אִם מִשֶּׁל רִאשׁוֹן עוֹלָה לְלֵאָה, תְּהֵא בָאַחֲרוֹן חַטָּאת לְלֵאָה, אוֹ אִפְּכָא. וְדַוְקָא עָשָׂה קֵן שָׁלֵם לְמַעְלָה, אֲבָל אִם חִלֵּק הַקִּנִּים פְּרִידָה אַחַת לְמַעְלָה וּפְרִידָה אַחַת לְמַטָּה, הַכֹּל כָּשֵׁר, שֶׁאֲנִי אוֹמֵר עוֹלָה לְמַעְלָה וְחַטָּאת לְמַטָּה, שֶׁהַקִּנִּים מִתְפָּרְשִׁים בַּעֲשִׂיַּת כֹּהֵן: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} וְכֵן כָּתַב הַמְּפָרֵשׁ. וּמִשּׁוּם מִשְׁנָה ג' הוּא דְּנָקְטֵי הָכִי. דְּהָא הָכָא בִּסְתוּמָה אֲנַן קָיְמִינַן: {ב} וְיַעֲשׂוּ כְּמוֹ שֶׁכָּתַב הָרַ"ב בְּסוֹף מַתְנִיתִין. וְכֻלֵּיהּ פִּרְקִין כְּמַאן דְּאָמַר בַּעֲלֵי חַיִּים אֵינָן נִדְחִין, וּמִשּׁוּם הָכִי כִּי אַקְרִיב אִרַצֵּי: (עיקר תוי"ט).

אַחַת לָזוֹ, וּשְׁתַּיִם לָזוֹ, וְשָׁלשׁ לָזוֹ, וְעֶשֶׂר לָזוֹ, וּמֵאָה לָזוֹ, עָשָׂה כֻלָּן לְמַעְלָה, מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. כֻּלָּן לְמַטָּן, מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. חֶצְיָן לְמַעְלָן וְחֶצְיָן לְמַטָּן, הַמְרֻבֶּה כָשֵׁר. זֶה הַכְּלָל, כָּל מָקוֹם שֶׁאַתָּה יָכוֹל לַחֲלֹק אֶת הַקִּנִּין וְלֹא יְהוּ מִשֶּׁל אִשָּׁה אַחַת, בֵּין מִלְמַעְלָן בֵּין מִלְּמַטָּן, מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. כָּל מָקוֹם שֶׁאֵין אַתָּה יָכוֹל לַחֲלֹק אֶת הַקִּנִּין עַד שֶׁיְּהוּ מִשֶּׁל אִשָּׁה אַחַת, בֵּין מִלְמַעְלָן בֵּין מִלְּמַטָּן, הַמְרֻבֶּה כָשֵׁר:

 ר"ע מברטנורה  זֶה הַכְּלָל כָּל מָקוֹם שֶׁאַתָּה יָכוֹל לַחֲלֹק כוּ'. כְּלוֹמַר, שֶׁקַּן אִשָּׁה אַחַת אֵינוֹ נֶחְלָק לִשְׁנַיִם אֶלָּא קַן אִשָּׁה אַחַת לְמַעְלָה וְקַן אִשָּׁה אַחֶרֶת לְמַטָּה, מֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל: וְכָל מָקוֹם כוּ'. כְּגוֹן אֶחָד לָזוֹ וּשְׁנַיִם לָזוֹ, שֶׁיֵּשׁ מֵחֵלֶק שֶׁל אִשָּׁה אַחַת לְמַעְלָה וּלְמַטָּה, הַמְרֻבֶּה כָּשֵׁר כִּדְפָרִישִׁית: (ר"ע מברטנורה).

פרק ג - משנה ג

חַטָּאת לָזוֹ, וְעוֹלָה לָזוֹ, עָשָׂה כֻלָּן לְמַעְלָן, מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. כֻּלָּן לְמַטָּן, מֱחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. חֶצְיָן לְמַעְלָן וְחֶצְיָן לְמַטָּן, שְׁתֵּיהֶן פָּסוּל, שֶׁאֲנִי אוֹמֵר, חַטָּאת קְרֵבָה לְמַעְלָן וְעוֹלָה לְמַטָּן:

 ר"ע מברטנורה  חַטָּאת לָזוֹ וְעוֹלָה לָזוֹ כוּ'. הָא נַמִּי קָאֵי אַדְּתָנֵי לְעֵיל בְּפֶרֶק קַמָּא, עוֹלָה שֶׁנִּתְעָרְבָה בְחַטָּאת אוֹ אִפְּכָא, כֻּלָּן יָמוּתוּ. וְהָנֵי מִלֵּי בְּנִמְלָךְ, אֲבָל בְּלֹא נִמְלָךְ עַל תַּעֲרָבְתּוֹ וְעָשָׂה קֵן {ג} לְמַטָּה וְקֵן לְמַעְלָה, מֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל. וְאַף עַל גַּב דְּאִי אָתֵי לְאִמְּלוֹכֵי אָמְרִינַן לֵיהּ לֹא תַקְרִיב, כִּדְתַנְיָא יָמוּתוּ כֻלָּן, הַשְׁתָּא דְּלֹא נִמְלַךְ אָמְרִינַן שֶׁכָּל אֶחָד נַעֲשָׂה כְהִלְכָתוֹ וְכָשֵׁר: (ר"ע מברטנורה).

 עיקר תוי"ט  {ג} רוֹצֶה לוֹמַר שְׁתֵּי פְרִידוֹת שֶׁהָאַחַת חַטָּאת וְאַחַת לְעוֹלָה: (עיקר תוי"ט).

פרק ג - משנה ד

חַטָּאת וְעוֹלָה וּסְתוּמָה וּמְפֹרֶשֶׁת, עָשָׂה כֻלָּן לְמַעְלָן, מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. כֻּלָּן לְמַטָּה, מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. חֶצְיָן לְמַעְלָן וְחֶצְיָן לְמַטָּן, אֵין כָּשֵׁר אֶלָּא סְתוּמָה, וְהִיא מִתְחַלֶּקֶת בֵּינֵיהֶן:

 ר"ע מברטנורה  חַטָּאת וְעוֹלָה וּסְתוּמָה וּמְפֹרֶשֶׁת. שְׁתֵּי נָשִׁים שֶׁלָּקְחוּ שְׁלֹשָׁה קִנִּין, זוֹ צְרִיכָה עוֹלָה וְקֵן שָׁלֵם, דְּהַיְנוּ שְׁתֵּי עוֹלוֹת וְחַטָּאת אַחַת, וְזוֹ צְרִיכָה חַטָּאת וְקֵן שָׁלֵם, דְּהַיְנוּ שְׁתֵּי חַטָּאוֹת וְעוֹלָה אַחַת, וּפֵרְשׁוּ אֶחָד מִן הַקִּנִּין פְּרִידָה עוֹלָה לָזוֹ וּפְרִידָה חַטָּאת לָזוֹ, וְאֶחָד מִן הַקִּנִּין הִנִּיחוּ סְתוּמָה שֶׁלֹּא פֵרְשׁוּ אֵיזוֹ עוֹלָה וְאֵיזוֹ חַטָּאת, וְהַשְּׁלִישִׁי פֵּרְשׁוּ אֵיזוֹ עוֹלָה וְאֵיזוֹ חַטָּאת, אֲבָל בְּעָלִים לֹא נִתְפָּרְשׁוּ, הֲרֵי חַטָּאת לָזוֹ וְעוֹלָה לָזוֹ וְקֵן סְתוּמָה וְקֵן מְפֹרֶשֶׁת, וְצָרִיךְ לְהַקְרִיב שְׁנֵי הַקִּנִּין הַלָּלוּ סְתָם עַל שְׁתֵּיהֶן, הָעוֹלוֹת לְמַעְלָה וְחַטָּאוֹת לְמַטָּה, וְעוֹלָה לָזוֹ וְחַטָּאת לָזוֹ צָרִיךְ לְהַקְרִיב לְשֵׁם בְּעָלִים {ד}. וְאִם נְתָנָם הַכֹּהֵן חֶצְיָן לְמַטָּה וְחֶצְיָן לְמַעְלָה, אֵין כְּשֵׁרָה אֶלָּא סְתוּמָה, דְּהֵי מִנַּיְהוּ דַּעֲבַד עוֹלָה וְהֵי מִנַּיְהוּ דַּעֲבַד חַטָּאת, כָּשֵׁר, כֵּיוָן שֶׁכָּל קֵן וְקֵן הָיָה בִּפְנֵי עַצְמוֹ: מִתְחַלֶּקֶת בֵּינֵיהֶם. הוֹאִיל וּבְעֵרוּב לְקָחוּם, הָאַחַת יָצְאָה יְדֵי חַטָּאת וְהָאַחַת יָצְאָה יְדֵי עוֹלָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ד} וּמִיהוּ כִּי אַקְרִיב בִּסְתָמָא, כָּל אַחַת עָלְתָה לְמִי שֶׁהִיא כוּ'. וּמִשּׁוּם הָכִי מַשְׁכַּחַת לְהוּ מֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל כְּדִתְנִינַן. וְכָתְבוּ הַתּוֹסָפוֹת דְּלֹא הֲוָה צָרִיךְ לְמִתְנֵי אֶלָּא סְתוּמָה וּמְפֹרֶשֶׁת, דִּבְחַטָּאת וְעוֹלָה אֵין מַרְוִיחַ כְּלוּם: (עיקר תוי"ט).

פרק ג - משנה ה

חַטָּאת שֶׁנִּתְעָרְבָה בְחוֹבָה, אֵין כָּשֵׁר אֶלָּא מִנְיַן חַטָּאת שֶׁבַּחוֹבָה. חוֹבָה שְׁנַיִם בְּחַטָּאת, מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. וְחַטָּאת שְׁנַיִם בְּחוֹבָה, הַמִּנְיָן שֶׁבַּחוֹבָה כָּשֵׁר. וְכֵן עוֹלָה שֶׁנִּתְעָרְבָה בְחוֹבָה, אֵין כָּשֵׁר אֶלָּא מִנְיַן עוֹלוֹת שֶׁבַּחוֹבָה. חוֹבָה שְׁנַיִם בְּעוֹלָה, מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. עוֹלָה שְׁנַיִם בְּחוֹבָה, הַמִּנְיָן שֶׁבַּחוֹבָה כָּשֵׁר:

 ר"ע מברטנורה  חַטָּאת שֶׁנִּתְעָרְבָה בְחוֹבָה. כְּלוֹמַר הָא דִתְנִינַן לְעֵיל בְּפֶרֶק קַמָּא חַטָּאת שֶׁנִּתְעָרְבָה בְחוֹבָה אֵין כָּשֵׁר אֶלָּא מִנְיַן חַטָּאוֹת שֶׁבַּחוֹבָה, פְּעָמִים שֶׁעוֹלֶה אוֹתוֹ מִנְיָן לְמֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל, וּפְעָמִים שֶׁעוֹלֶה לְפָחוֹת מִמֶּחֱצָה, וּמְפָרֵשׁ וְאָזֵיל כֵּיצַד: חוֹבָה שְׁנַיִם בְּחַטָּאת מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. פֵּרוּשׁ, שְׁנֵי קִנִּין חוֹבָה שֶׁיֵּשׁ בָּהֶן כִּפְלַיִם בַּחַטָּאת, כְּגוֹן שֶׁהִקְרִיב מֵהֶן עוֹלָה אַחַת וְנִשְׁתַּיְּרוּ שְׁתֵּי חַטָּאוֹת וְעוֹלָה אֶחָת: מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל. שֶׁעוֹלָה אַחַת הַנִּשְׁאֶרֶת אֵינוֹ יָכוֹל לְהַקְרִיב שֶׁמָּא זֶהוּ שֶׁנִּקְבַּע לְחַטָּאת {ה}, וְחַטָּאת הַמְעֹרֶבֶת אֵינוֹ יָכוֹל לְהַקְרִיב שֶׁמָּא הִיא הָעוֹלָה הַנִּשְׁאֶרֶת {ו}, הִלְכָּךְ אֵינוֹ מַקְרִיב אֶלָּא שְׁתֵּי הַחַטָּאוֹת, וְהַחַטָּאת הַמְּעֹרֶבֶת וְהָעוֹלָה הַנִּשְׁאֶרֶת פְּסוּלִין, וְהַיְנוּ מֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל: חַטָּאת שְׁנַיִם בְּחוֹבָה מִנְיָן שֶׁבַּחוֹבָה כָּשֵׁר. פֵּרוּשׁ, כְּגוֹן שֶׁקָּרְבָה חַטָּאת אַחַת מִן הַשְּׁנֵי קִנִּין וְנִשְׁתַּיְּרוּ שְׁתֵּי עוֹלוֹת וְחַטָּאת אֶחָת {ז}, הֲרֵי הֵן אַרְבַּע פְּרִידוֹת עִם חַטָּאת שֶׁנִּתְעָרְבָה בֵינֵיהֶן, וַהֲרֵי אֵינוֹ יָכוֹל לְהַקְרִיב שְׁתֵּי הָעוֹלוֹת {ח} שֶׁמָּא יִקַּח חַטָּאת הַמְעֹרֶבֶת, וְגַם שְׁתֵּי חַטָּאוֹת {ט} לֹא יַקְרִיב שֶׁמָּא יִקַּח אוֹתוֹ שֶׁנִּקְבַּע לְעוֹלָה, הִלְכָּךְ אֵינוֹ מַקְרִיב אֶלָּא חַטָּאת אַחַת, הֲרֵי עוֹלֶה אוֹתוֹ מִנְיָן הַפָּחוֹת שֶׁבַּחוֹבָה לְפָחוֹת מִמֶּחֱצָה כָּשֵׁר. וּכְהַאי גַּוְנָא מְפָרְשִׁינַן נַמִּי וְכֵן עוֹלָה שֶׁנִּתְעָרְבָה בְחוֹבָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ה} מְפָרֵשׁ. וְרוֹצֶה לוֹמַר שֶׁמָּא זוֹהִי אוֹתָהּ הַמְּעֹרֶבֶת שֶׁכְּבָר נִקְבְּעָה לְחַטָּאת: {ו} כְּלוֹמַר, אַחַר שֶׁהִקְרִיב שְׁתֵּי חַטָּאוֹת מֵהַתַּעֲרֹבוֹת שׁוּב לֹא יוּכַל לְהַקְרִיב עוֹד חַטָּאת עַל דַּעַת שֶׁהִיא זוֹ שֶׁנִּתְעָרְבָה, דְּשֶׁמָּא זוֹ הִיא עוֹלָה הַנִּשְׁאֶרֶת, דְּהוֹאִיל וְנִקְרְבוּ שְׁתֵּי חַטָּאוֹת מִן הַתַּעֲרֹבוֹת הֲרֵי הֻקְבְּעָה הַשְּׁלִישִׁית לְעוֹלָה וְשֶׁמָּא זוֹ הִיא, וְלֹא הִיא אוֹתָהּ הַחַטָּאת שֶׁנִּתְעָרְבָה: {ז} וְיִהְיֶה פִּי שְׁנַיִם בַּחוֹבָה, שֶׁיֵּשׁ בַּחוֹבָה כִּפְלַיִם עוֹלוֹת כְּנֶגֶד הַחַטָּאוֹת: {ח} רְצוֹנוֹ לוֹמַר, גַּם שְׁתֵּיהֶן, וַאֲפִלּוּ אַחַת מֵהֶן לֹא: {ט} רְצוֹנוֹ לוֹמַר, שְׁתַּיִם בְּיַחַד אֵינוֹ יָכוֹל לְהַקְרִיב, אֲבָל אַחַת מַקְרִיב. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ג - משנה ו

הָאִשָּׁה שֶׁאָמְרָה, הֲרֵי עָלַי קֵן כְּשֶׁאֵלֵד זָכָר, יָלְדָה זָכָר, מְבִיאָה שְׁתֵּי קִנִּים, אַחַת לְנִדְרָהּ וְאַחַת לְחוֹבָתָהּ. נְתָנָתַם לַכֹּהֵן, וְהַכֹּהֵן צָרִיךְ לַעֲשׂוֹת שָׁלשׁ פְּרִידִים מִלְמַעְלָן וְאַחַת מִלְּמַטָּן. לֹא עָשָׂה כֵן, אֶלָּא עָשָׂה שְׁתַּיִם לְמַעְלָן וּשְׁתַּיִם לְמַטָּן וְלֹא נִמְלַךְ, צְרִיכָה לְהָבִיא עוֹד פְּרִידָה אַחַת, וְיַקְרִיבֶנָּה לְמַעְלָן, מִמִּין אֶחָד. מִשְּׁנֵי מִינִין, תָּבִיא שְׁתַּיִם. פֵּרְשָׁה נִדְרָהּ, צְרִיכָה לְהָבִיא עוֹד שָׁלשׁ פְּרִידִים, מִמִּין אֶחָד. מִשְּׁנֵי מִינִין, תָּבִיא אַרְבַּע. קָבְעָה נִדְרָהּ, צְרִיכָה לְהָבִיא עוֹד חָמֵשׁ פְּרִידִים, מִמִּין אֶחָד. מִשְּׁנֵי מִינִין, תָּבִיא שֵׁשׁ. נְתָנָתַם לַכֹּהֵן וְאֵין יָדוּעַ מַה נָּתְנָה, הָלַךְ הַכֹּהֵן וְעָשָׂה וְאֵין יָדוּעַ מֶה עָשָׂה, צְרִיכָה לְהָבִיא עוֹד אַרְבַּע פְּרִידִים לְנִדְרָהּ, וּשְׁתַּיִם לְחוֹבָתָהּ, וְחַטָּאת אֶחָת. בֶּן עַזַּאי אוֹמֵר, שְׁתֵּי חַטָּאוֹת. אָמַר רַבִּי יְהוֹשֻׁעַ, זֶה הוּא שֶׁאָמְרוּ, כְּשֶׁהוּא חַי קוֹלוֹ אֶחָד, וּכְשֶׁהוּא מֵת קוֹלוֹ שִׁבְעָה. כֵּיצַד קוֹלוֹ שִׁבְעָה. שְׁתֵּי קַרְנָיו, שְׁתֵּי חֲצוֹצְרוֹת. שְׁתֵּי שׁוֹקָיו, שְׁנֵי חֲלִילִין. עוֹרוֹ, לְתֹף. מֵעָיו, לִנְבָלִים. בְּנֵי מֵעָיו {יח}, לְכִנּוֹרוֹת. וְיֵשׁ אוֹמְרִים, אַף צַמְרוֹ לִתְכֵלֶת. רַבִּי שִׁמְעוֹן בֶּן עֲקַשְׁיָא אוֹמֵר, זִקְנֵי עַם הָאָרֶץ, כָּל זְמַן שֶׁמַּזְקִינִין, דַּעְתָּן מִטָּרֶפֶת עֲלֵיהֶן, שֶׁנֶּאֱמַר (איוב יב), מֵסִיר שָׂפָה לְנֶאֱמָנִים וְטַעַם זְקֵנִים יִקָּח. אֲבָל זִקְנֵי תוֹרָה אֵינָן כֵן, אֶלָּא כָל זְמַן שֶׁמַּזְקִינִין, דַּעְתָּן מִתְיַשֶּׁבֶת עֲלֵיהֶן, שֶׁנֶּאֱמַר (שם), בִּישִׁישִׁים חָכְמָה וְאֹרֶךְ יָמִים תְּבוּנָה:

 ר"ע מברטנורה  הֲרֵי עָלַי קֵן לִכְשֶׁאֵלֵד זָכָר. בְּיוֹלֶדֶת עֲנִיָּה עָסְקִינַן. דְּאִלּוּ עֲשִׁירָה מְבִיאָה כֶבֶשׂ לְחוֹבָתָהּ: וְהַכֹּהֵן צָרִיךְ לַעֲשׂוֹת שָׁלֹשׁ לְמַעְלָה. שֶׁקַּן נְדָבָה שְׁתֵּיהֶן עוֹלוֹת וְקֵן שֶׁל חוֹבָה עוֹלָה לְמַעְלָה וְחַטָּאת לְמַטָּה, וְהוּא לֹא עָשָׂה כֵן, אֶלָּא שְׁתַּיִם לְמַעְלָה וּשְׁתַּיִם לְמַטָּה {י}, וְלֹא נִמְלַךְ בָּהּ לָדַעַת עַל מַה הֱבִיאָתָן, וְנִפְסְלָה פְּרִידָה אַחַת, הִלְכָּךְ צְרִיכָה שֶׁתָּבִיא פְּרִידָה אַחֶרֶת וְתַקְרִיבֶנָּהּ לְמַעְלָה: מִמִּין אֶחָד. כְּלוֹמַר, בַּמֶּה דְבָרִים אֲמוּרִים, כְּשֶׁהָיוּ מִמִּין אֶחָד. אֲבָל הָיוּ הַשְּׁנֵי קִנִּין שֶׁהֵבִיאָה מִשְּׁנֵי מִינִים, דְּהַיְנוּ קֵן אֶחָד תּוֹרִים וְקֵן אֶחָד בְּנֵי יוֹנָה, צְרִיכָה לְהָבִיא פְּרִידָה אַחַת תּוֹר וּפְרִידָה אַחַת מִבְּנֵי יוֹנָה, כְּדֵי לְהַשְׁלִים הַפְּרִידָה שֶׁנִּפְסְלָה, דְּלֹא יָדְעִינַן אִם מִן הַתּוֹרִים אִם מִן בְּנֵי יוֹנָה הָיְתָה, וּתְנַן לְעֵיל אֵין מְבִיאִים תּוֹרִים כְּנֶגֶד בְּנֵי יוֹנָה כוּ': פֵּרְשָׁה נִדְרָהּ. כְּשֶׁהֵבִיאָה שְׁנֵי קִנִּין לַכֹּהֵן פֵּרְשָׁה וְאָמְרָה אֵלּוּ לְנִדְרִי וְאֵלּוּ לְחוֹבָתִי: מִמִּין אֶחָד. כְּלוֹמַר אִם שְׁנֵי קִנִּין הַלָּלוּ הָיוּ מִמִּין אֶחָד דְּהַיְנוּ שְׁנֵיהֶן תּוֹרִים אוֹ שְׁנֵיהֶן בְּנֵי יוֹנָה, וְהִקְרִיב מֵהֶן הַכֹּהֵן שְׁתַּיִם לְמַעְלָה וּשְׁתַּיִם לְמַטָּה, אֵין כָּשֵׁר אֶלָּא פְּרִידָה אַחַת מִכָּל הָאַרְבַּע שֶׁהִקְרִיב, וְהִיא הָעוֹלָה שֶׁבַּחוֹבָה, מִידִי דַּהֲוֵי זְמַן שֶׁמַּזְקִינִים וּבָאִים לִידֵי תַּשּׁוּת כֹּחַ, הֵן מוֹסִיפִין חָכְמָה {יט}: שֶׁנֶּאֱמַר מֵסִיר שָׂפָה לְנֶאֱמָנִים. וּבְעַמֵּי הָאָרֶץ מִשְׁתַּעֵי קְרָא, דִּלְעֵיל מִנֵּיהּ כְּתִיב מֵסִיר לֵב רָאשֵׁי עַם הָאָרֶץ: בִּישִׁישִׁים חָכְמָה. הַיְנוּ תַּלְמִידֵי חֲכָמִים. דְּאִלּוּ עַמֵּי הָאָרֶץ, חַכְמַת מָה לָהֶם: (ר"ע מברטנורה).

 עיקר תוי"ט  {י} כְּסָבוּר שְׁנֵי הַקִּנִּין חוֹבָה עָלֶיהָ שֶׁמָּא יוֹלֶדֶת בְּזוֹב הָיְתָה וְהֻזְקְקָה לְקַן לֵידָה וְקַן זִיבָה. רַשִׁ"י: {יא} וְכֵן לְכָל הַחַטָּאוֹת דְּמַתְנִיתִין הַבָּאוֹת עַל הַסָּפֵק: {יב} הָרַמְבַּ"ם לֹא כָתַב שֶׁקָּבְעָה לְחוֹבָתָהּ. וְדוֹ"ק: {יג} לִרְבוּתָא כָּתַב כֵּן, דְּלֹא תֵימָא דְּמִכָּל מִין הִקְרִיב עוֹלָה אַחַת כוּ', קָא מַשְׁמַע לָן דְּלֹא, דְּהָא בְּפֵרְשָׁה וְקָבְעָה מַיְרֵי: {יד} לָאו דַּוְקָא. דִּיכוֹלָה לְהָבִיא שְׁתֵּי פְרִידוֹת מֵאֵיזֶה מִין שֶׁתִּרְצֶה אַחַת חַטָּאת וְאַחַת עוֹלָה, וּכְמוֹ שֶׁכָּתַבְתִּי בְּסָמוּךְ. וְכֵן כָּתַב הָרַמְבַּ"ם: {טו} מֵאֵיזֶה מִין שֶׁיִּהְיֶה לְהוֹסִיף עַל כֶּבֶשׂ לְעוֹלָה דַּעֲשִׁירוֹת, הוֹאִיל וְחוֹבָתָהּ גַּם כֵּן קְבוּעָה וְהִיא לֹא יָדְעָה מֵאֵיזֶה מִין. הָרַמְבַּ"ם. וּבֶן עַזַּאי מִסְתַּפֵּק עוֹד בְּאֵיזֶה מִין קָבְעָה עִם הַכֶּבֶשׂ, וּלְכָךְ אָמַר שְׁתֵּי, וְרוֹצֶה לוֹמַר תּוֹר וּבֶן יוֹנָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב שֶׁהֶאֱרִיךְ בְּפֵרוּשׁ הָרַמְבַּ"ם הַרְבֵּה, וְתָפַס עִקָּר לְפֵרוּשׁ רַשִׁ"י בִּזְבָחִים דַּף ס"ז. דּוֹק בָּהֶם אִם בַּעַל נֶפֶשׁ אָתָּה: {טז} רַשִׁ"י. וְלִרְבוּתָא נָקְטֵי מִנְיָן הַמְרֻבֶּה אַף עַל גַּב דְּהִנּוֹ לְבֶן עַזַּאי. אֲבָל הָיוּ יְכוֹלִים לְפָרֵשׁ אַלִּיבָּא דְתַנָּא קַמָּא, וְשָׁלֹשׁ לְחוֹבָתָהּ, וַהֲווּ מַמָּשׁ שֶׁבַע: {יז} דְּאִי חֲצוֹצְרוֹת מַמָּשׁ, הָא תְנַן בְּסוֹף תָּמִיד וּשְׁתֵּי חֲצוֹצְרוֹת שֶׁל כֶּסֶף בְּיָדָם. עוֹד יֵשׁ לוֹמַר, דְּהַהִיא דְּתָמִיד שֶׁל כֹּהֲנִים וְהָנֵי דְהָכָא שֶׁל לְוִיִּים לִכְלֵי שִׁיר, וּשְׁנֵי מִינֵי חֲצוֹצְרוֹת הָיוּ. תּוֹסָפוֹת: {יח} בְּנֵי כוּ'. מֵעַיִם הַדַּקִּים. כְּמוֹ הָדְרָא דְכַנְתָּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יט} שֶׁבְּשָׁעָה שֶׁהֵן מַזְקִינִין וְנֶחְלָשִׁים וְיַפְסִיד גּוּפָם, תִּרְבֶּה חָכְמָתָם וְיִתְחַזֵּק שִׂכְלָם וְיוֹסִיפוּ שְׁלֵמוּת. הָרַמְבַּ"ם: (עיקר תוי"ט).
