[איכה פרק-ג-נז]  קָרַבְתָּ בְּיוֹם אֶקְרָאֶךָּ אָמַרְתָּ אַל תִּירָא:

[ירמיה פרק-ו-ד]  קַדְּשׁוּ עָלֶיהָ מִלְחָמָה קוּמוּ וְנַעֲלֶה בַצָּהֳרָיִם אוֹי לָנוּ כִּי פָנָה הַיּוֹם כִּי יִנָּטוּ צִלְלֵי עָרֶב:

[משלי פרק-יט-ח]  קֹנֶה לֵּב אֹהֵב נַפְשׁוֹ שֹׁמֵר תְּבוּנָה לִמְצֹא טוֹב:

[ירמיה פרק-מח-טז]  קָרוֹב אֵיד מוֹאָב לָבוֹא וְרָעָתוֹ מִהֲרָה מְאֹד:

[תהילים פרק-כט-ט]  קוֹל יְהוָה יְחוֹלֵל אַיָּלוֹת וַיֶּחֱשֹׂף יְעָרוֹת וּבְהֵיכָלוֹ כֻּלּוֹ אֹמֵר כָּבוֹד:

[דברי הימים א פרק-א-ב]  קֵינָן מַהֲלַלְאֵל יָרֶד:

[שמות פרק-לח-יד]  קְלָעִים חֲמֵשׁ עֶשְׂרֵה אַמָּה אֶל הַכָּתֵף עַמּוּדֵיהֶם שְׁלֹשָׁה וְאַדְנֵיהֶם שְׁלֹשָׁה:

[במדבר פרק-ז-לז]  קָרְבָּנוֹ קַעֲרַת כֶּסֶף אַחַת שְׁלֹשִׁים וּמֵאָה מִשְׁקָלָהּ מִזְרָק אֶחָד כֶּסֶף שִׁבְעִים שֶׁקֶל בְּשֶׁקֶל הַקֹּדֶשׁ שְׁנֵיהֶם מְלֵאִים סֹלֶת בְּלוּלָה בַשֶּׁמֶן לְמִנְחָה:

[במדבר פרק-ז-מג]  קָרְבָּנוֹ קַעֲרַת כֶּסֶף אַחַת שְׁלֹשִׁים וּמֵאָה מִשְׁקָלָהּ מִזְרָק אֶחָד כֶּסֶף שִׁבְעִים שֶׁקֶל בְּשֶׁקֶל הַקֹּדֶשׁ שְׁנֵיהֶם מְלֵאִים סֹלֶת בְּלוּלָה בַשֶּׁמֶן לְמִנְחָה:

[במדבר פרק-ז-מט]  קָרְבָּנוֹ קַעֲרַת כֶּסֶף אַחַת שְׁלֹשִׁים וּמֵאָה מִשְׁקָלָהּ מִזְרָק אֶחָד כֶּסֶף שִׁבְעִים שֶׁקֶל בְּשֶׁקֶל הַקֹּדֶשׁ שְׁנֵיהֶם מְלֵאִים סֹלֶת בְּלוּלָה בַשֶּׁמֶן לְמִנְחָה:

[במדבר פרק-ז-נה]  קָרְבָּנוֹ קַעֲרַת כֶּסֶף אַחַת שְׁלֹשִׁים וּמֵאָה מִשְׁקָלָהּ מִזְרָק אֶחָד כֶּסֶף שִׁבְעִים שֶׁקֶל בְּשֶׁקֶל הַקֹּדֶשׁ שְׁנֵיהֶם מְלֵאִים סֹלֶת בְּלוּלָה בַשֶּׁמֶן לְמִנְחָה:

[במדבר פרק-ז-סא]  קָרְבָּנוֹ קַעֲרַת כֶּסֶף אַחַת שְׁלֹשִׁים וּמֵאָה מִשְׁקָלָהּ מִזְרָק אֶחָד כֶּסֶף שִׁבְעִים שֶׁקֶל בְּשֶׁקֶל הַקֹּדֶשׁ שְׁנֵיהֶם מְלֵאִים סֹלֶת בְּלוּלָה בַשֶּׁמֶן לְמִנְחָה:

[במדבר פרק-ז-סז]  קָרְבָּנוֹ קַעֲרַת כֶּסֶף אַחַת שְׁלֹשִׁים וּמֵאָה מִשְׁקָלָהּ מִזְרָק אֶחָד כֶּסֶף שִׁבְעִים שֶׁקֶל בְּשֶׁקֶל הַקֹּדֶשׁ שְׁנֵיהֶם מְלֵאִים סֹלֶת בְּלוּלָה בַשֶּׁמֶן לְמִנְחָה:

[במדבר פרק-ז-עג]  קָרְבָּנוֹ קַעֲרַת כֶּסֶף אַחַת שְׁלֹשִׁים וּמֵאָה מִשְׁקָלָהּ מִזְרָק אֶחָד כֶּסֶף שִׁבְעִים שֶׁקֶל בְּשֶׁקֶל הַקֹּדֶשׁ שְׁנֵיהֶם מְלֵאִים סֹלֶת בְּלוּלָה בַשֶּׁמֶן לְמִנְחָה:

[דברים פרק-ב-כד]  קוּמוּ סְּעוּ וְעִבְרוּ אֶת נַחַל אַרְנֹן רְאֵה נָתַתִּי בְיָדְךָ אֶת סִיחֹן מֶלֶךְ חֶשְׁבּוֹן הָאֱמֹרִי וְאֶת אַרְצוֹ הָחֵל רָשׁ וְהִתְגָּר בּוֹ מִלְחָמָה:

[ישעיה פרק-יג-ד]  קוֹל הָמוֹן בֶּהָרִים דְּמוּת עַם רָב קוֹל שְׁאוֹן מַמְלְכוֹת גּוֹיִם נֶאֱסָפִים יְהוָה צְבָאוֹת מְפַקֵּד צְבָא מִלְחָמָה:

[ישעיה פרק-לד-א]  קִרְבוּ גוֹיִם לִשְׁמֹעַ וּלְאֻמִּים הַקְשִׁיבוּ תִּשְׁמַע הָאָרֶץ וּמְלֹאָהּ תֵּבֵל וְכָל צֶאֱצָאֶיהָ:

[ישעיה פרק-מו-ב]  קָרְסוּ כָרְעוּ יַחְדָּו לֹא יָכְלוּ מַלֵּט מַשָּׂא וְנַפְשָׁם בַּשְּׁבִי הָלָכָה:

[ישעיה פרק-מו-יא]  קֹרֵא מִמִּזְרָח עַיִט מֵאֶרֶץ מֶרְחָק אִישׁ (עצתו) עֲצָתִי אַף דִּבַּרְתִּי אַף אֲבִיאֶנָּה יָצַרְתִּי אַף אֶעֱשֶׂנָּה:

[ירמיה פרק-ו-ה]  קוּמוּ וְנַעֲלֶה בַלָּיְלָה וְנַשְׁחִיתָה אַרְמְנוֹתֶיהָ:

[ירמיה פרק-לג-יא]  קוֹל שָׂשׂוֹן וְקוֹל שִׂמְחָה קוֹל חָתָן וְקוֹל כַּלָּה קוֹל אֹמְרִים הוֹדוּ אֶת יְהוָה צְבָאוֹת כִּי טוֹב יְהוָה כִּי לְעוֹלָם חַסְדּוֹ מְבִאִים תּוֹדָה בֵּית יְהוָה כִּי אָשִׁיב אֶת שְׁבוּת הָאָרֶץ כְּבָרִאשֹׁנָה אָמַר יְהוָה:

[ירמיה פרק-לו-ב]  קַח לְךָ מְגִלַּת סֵפֶר וְכָתַבְתָּ אֵלֶיהָ אֵת כָּל הַדְּבָרִים אֲשֶׁר דִּבַּרְתִּי אֵלֶיךָ עַל יִשְׂרָאֵל וְעַל יְהוּדָה וְעַל כָּל הַגּוֹיִם מִיּוֹם דִּבַּרְתִּי אֵלֶיךָ מִימֵי יֹאשִׁיָּהוּ וְעַד הַיּוֹם הַזֶּה:

[יחזקאל פרק-ז-ו]  קֵץ בָּא בָּא הַקֵּץ הֵקִיץ אֵלָיִךְ הִנֵּה בָּאָה:

[יחזקאל פרק-כב-כה]  קֶשֶׁר נְבִיאֶיהָ בְּתוֹכָהּ כַּאֲרִי שׁוֹאֵג טֹרֵף טָרֶף נֶפֶשׁ אָכָלוּ חֹסֶן וִיקָר יִקָּחוּ אַלְמְנוֹתֶיהָ הִרְבּוּ בְתוֹכָהּ:

[יחזקאל פרק-לב-טז]  קִינָה הִיא וְקוֹנְנוּהָ בְּנוֹת הַגּוֹיִם תְּקוֹנֵנָּה אוֹתָהּ עַל מִצְרַיִם וְעַל כָּל הֲמוֹנָהּ תְּקוֹנֵנָּה אוֹתָהּ נְאֻם אֲדֹנָי יְהוִה:

[יואל פרק-א-יד]  קַדְּשׁוּ צוֹם קִרְאוּ עֲצָרָה אִסְפוּ זְקֵנִים כֹּל יֹשְׁבֵי הָאָרֶץ בֵּית יְהוָה אֱלֹהֵיכֶם וְזַעֲקוּ אֶל יְהוָה:

[יואל פרק-ד-ט]  קִרְאוּ זֹאת בַּגּוֹיִם קַדְּשׁוּ מִלְחָמָה הָעִירוּ הַגִּבּוֹרִים יִגְּשׁוּ יַעֲלוּ כֹּל אַנְשֵׁי הַמִּלְחָמָה:

[מיכה פרק-ו-ט]  קוֹל יְהוָה לָעִיר יִקְרָא וְתוּשִׁיָּה יִרְאֶה שְׁמֶךָ שִׁמְעוּ מַטֶּה וּמִי יְעָדָהּ:

[תהילים פרק-ג-ה]  קוֹלִי אֶל יְהוָה אֶקְרָא וַיַּעֲנֵנִי מֵהַר קָדְשׁוֹ סֶלָה:

[תהילים פרק-כז-יד]  קַוֵּה אֶל יְהוָה חֲזַק וְיַאֲמֵץ לִבֶּךָ וְקַוֵּה אֶל יְהוָה:

[תהילים פרק-לז-לד]  קַוֵּה אֶל יְהוָה וּשְׁמֹר דַּרְכּוֹ וִירוֹמִמְךָ לָרֶשֶׁת אָרֶץ בְּהִכָּרֵת רְשָׁעִים תִּרְאֶה:

[תהילים פרק-קיט-קמה]  קָרָאתִי בְכָל לֵב עֲנֵנִי יְהוָה חֻקֶּיךָ אֶצֹּרָה:

[משלי פרק-לא-כח]  קָמוּ בָנֶיהָ וַיְאַשְּׁרוּהָ בַּעְלָהּ וַיְהַלְלָהּ:

[איוב פרק-ה-א]  קְרָא נָא הֲיֵשׁ עוֹנֶךָּ וְאֶל מִי מִקְּדֹשִׁים תִּפְנֶה:

[איוב פרק-כט-י]  קוֹל נְגִידִים נֶחְבָּאוּ וּלְשׁוֹנָם לְחִכָּם דָּבֵקָה:

[אסתר פרק-ט-כז]  קִיְּמוּ (וקבל) וְקִבְּלוּ הַיְּהוּדִים עֲלֵיהֶם וְעַל זַרְעָם וְעַל כָּל הַנִּלְוִים עֲלֵיהֶם וְלֹא יַעֲבוֹר לִהְיוֹת עֹשִׂים אֵת שְׁנֵי הַיָּמִים הָאֵלֶּה כִּכְתָבָם וְכִזְמַנָּם בְּכָל שָׁנָה וְשָׁנָה:

[דניאל פרק-ז-ד]  קַדְמָיְתָא כְאַרְיֵה וְגַפִּין דִּי נְשַׁר לַהּ חָזֵה הֲוֵית עַד דִּי מְּרִיטוּ גַפַּיהּ וּנְטִילַת מִן אַרְעָא וְעַל רַגְלַיִן כֶּאֱנָשׁ הֳקִימַת וּלְבַב אֱנָשׁ יְהִיב לַהּ:

[עזרא פרק-י-ד]  קוּם כִּי עָלֶיךָ הַדָּבָר וַאֲנַחְנוּ עִמָּךְ חֲזַק וַעֲשֵׂה:


ק-ו


[בראשית פרק-כא-יח]  קוּמִי שְׂאִי אֶת הַנַּעַר וְהַחֲזִיקִי אֶת יָדֵךְ בּוֹ כִּי לְגוֹי גָּדוֹל אֲשִׂימֶנּוּ:

[במדבר פרק-ז-ה]  קַח מֵאִתָּם וְהָיוּ לַעֲבֹד אֶת עֲבֹדַת אֹהֶל מוֹעֵד וְנָתַתָּה אוֹתָם אֶל הַלְוִיִּם אִישׁ כְּפִי עֲבֹדָתוֹ:

[דברים פרק-ה-כז]  קְרַב אַתָּה וּשֲׁמָע אֵת כָּל אֲשֶׁר יֹאמַר יְהוָה אֱלֹהֵינוּ וְאַתְּ תְּדַבֵּר אֵלֵינוּ אֵת כָּל אֲשֶׁר יְדַבֵּר יְהוָה אֱלֹהֵינוּ אֵלֶיךָ וְשָׁמַעְנוּ וְעָשִׂינוּ:

[מלכים א פרק-כא-יח]  קוּם רֵד לִקְרַאת אַחְאָב מֶלֶךְ יִשְׂרָאֵל אֲשֶׁר בְּשֹׁמְרוֹן הִנֵּה בְּכֶרֶם נָבוֹת אֲשֶׁר יָרַד שָׁם לְרִשְׁתּוֹ:

[ישעיה פרק-מ-ג]  קוֹל קוֹרֵא בַּמִּדְבָּר פַּנּוּ דֶּרֶךְ יְהוָה יַשְּׁרוּ בָּעֲרָבָה מְסִלָּה לֵאלֹהֵינוּ:

[ישעיה פרק-מח-טז]  קִרְבוּ אֵלַי שִׁמְעוּ זֹאת לֹא מֵרֹאשׁ בַּסֵּתֶר דִּבַּרְתִּי מֵעֵת הֱיוֹתָהּ שָׁם אָנִי וְעַתָּה אֲדֹנָי יְהוִה שְׁלָחַנִי וְרוּחוֹ:

[ישעיה פרק-סו-ו]  קוֹל שָׁאוֹן מֵעִיר קוֹל מֵהֵיכָל קוֹל יְהוָה מְשַׁלֵּם גְּמוּל לְאֹיְבָיו:

[ירמיה פרק-כט-ו]  קְחוּ נָשִׁים וְהוֹלִידוּ בָּנִים וּבָנוֹת וּקְחוּ לִבְנֵיכֶם נָשִׁים וְאֶת בְּנוֹתֵיכֶם תְּנוּ לַאֲנָשִׁים וְתֵלַדְנָה בָּנִים וּבָנוֹת וּרְבוּ שָׁם וְאַל תִּמְעָטוּ:

[ירמיה פרק-לט-יב]  קָחֶנּוּ וְעֵינֶיךָ שִׂים עָלָיו וְאַל תַּעַשׂ לוֹ מְאוּמָה רָּע כִּי (אם) כַּאֲשֶׁר יְדַבֵּר אֵלֶיךָ כֵּן עֲשֵׂה עִמּוֹ:

[ירמיה פרק-מט-לא]  קוּמוּ עֲלוּ אֶל גּוֹי שְׁלֵיו יוֹשֵׁב לָבֶטַח נְאֻם יְהוָה לֹא דְלָתַיִם וְלֹא בְרִיחַ לוֹ בָּדָד יִשְׁכֹּנוּ:

[ירמיה פרק-נ-כח]  קוֹל נָסִים וּפְלֵטִים מֵאֶרֶץ בָּבֶל לְהַגִּיד בְּצִיּוֹן אֶת נִקְמַת יְהוָה אֱלֹהֵינוּ נִקְמַת הֵיכָלוֹ:

[ירמיה פרק-נא-כח]  קַדְּשׁוּ עָלֶיהָ גוֹיִם אֶת מַלְכֵי מָדַי אֶת פַּחוֹתֶיהָ וְאֶת כָּל סְגָנֶיהָ וְאֵת כָּל אֶרֶץ מֶמְשַׁלְתּוֹ:

[הושע פרק-יד-ג]  קְחוּ עִמָּכֶם דְּבָרִים וְשׁוּבוּ אֶל יְהוָה אִמְרוּ אֵלָיו כָּל תִּשָּׂא עָוֹן וְקַח טוֹב וּנְשַׁלְּמָה פָרִים שְׂפָתֵינוּ:

[תהילים פרק-קיט-קנ]  קָרְבוּ רֹדְפֵי זִמָּה מִתּוֹרָתְךָ רָחָקוּ:

[משלי פרק-ו-יג]  קֹרֵץ בְּעֵינָיו מֹלֵל בְּרַגְלָו מֹרֶה בְּאֶצְבְּעֹתָיו:

[משלי פרק-טז-י]  קֶסֶם עַל שִׂפְתֵי מֶלֶךְ בְּמִשְׁפָּט לֹא יִמְעַל פִּיו:

[משלי פרק-כז-יג]  קַח בִּגְדוֹ כִּי עָרַב זָר וּבְעַד נָכְרִיָּה חַבְלֵהוּ:

[איוב פרק-ג-יט]  קָטֹן וְגָדוֹל שָׁם הוּא וְעֶבֶד חָפְשִׁי מֵאֲדֹנָיו:

[איוב פרק-טו-כא]  קוֹל פְּחָדִים בְּאָזְנָיו בַּשָּׁלוֹם שׁוֹדֵד יְבוֹאֶנּוּ:

[איכה פרק-ד-יט]  קַלִּים הָיוּ רֹדְפֵינוּ מִנִּשְׁרֵי שָׁמָיִם עַל הֶהָרִים דְּלָקֻנוּ בַּמִּדְבָּר אָרְבוּ לָנוּ:


ק-ח


[בראשית פרק-לג-יא]  קַח נָא אֶת בִּרְכָתִי אֲשֶׁר הֻבָאת לָךְ כִּי חַנַּנִי אֱלֹהִים וְכִי יֶשׁ לִי כֹל וַיִּפְצַר בּוֹ וַיִּקָּח:

[ויקרא פרק-ב-יב]  קָרְבַּן רֵאשִׁית תַּקְרִיבוּ אֹתָם לַיהוָה וְאֶל הַמִּזְבֵּחַ לֹא יַעֲלוּ לְרֵיחַ נִיחֹחַ:

[ישעיה פרק-ס-א]  קוּמִי אוֹרִי כִּי בָא אוֹרֵךְ וּכְבוֹד יְהוָה עָלַיִךְ זָרָח:


ק-ט


[יהושע פרק-ד-ב]  קְחוּ לָכֶם מִן הָעָם שְׁנֵים עָשָׂר אֲנָשִׁים אִישׁ אֶחָד אִישׁ אֶחָד מִשָּׁבֶט:

[ישעיה פרק-מב-ג]  קָנֶה רָצוּץ לֹא יִשְׁבּוֹר וּפִשְׁתָּה כֵהָה לֹא יְכַבֶּנָּה לֶאֱמֶת יוֹצִיא מִשְׁפָּט:

[משלי פרק-י-י]  קֹרֵץ עַיִן יִתֵּן עַצָּבֶת וֶאֱוִיל שְׂפָתַיִם יִלָּבֵט:

[דניאל פרק-ה-ז]  קָרֵא מַלְכָּא בְּחַיִל לְהֶעָלָה לְאָשְׁפַיָּא (כשדיא) כַּשְׂדָּאֵי וְגָזְרַיָּא עָנֵה מַלְכָּא וְאָמַר לְחַכִּימֵי בָבֶל דִּי כָל אֱנָשׁ דִּי יִקְרֵה כְּתָבָה דְנָה וּפִשְׁרֵהּ יְחַוִּנַּנִי אַרְגְּוָנָא יִלְבַּשׁ (והמונכא) וְהַמְנִיכָא דִי דַהֲבָא עַל צַוְּארֵהּ וְתַלְתִּי בְמַלְכוּתָא יִשְׁלַט:


ק-י


[מלכים א פרק-ו-כו]  קוֹמַת הַכְּרוּב הָאֶחָד עֶשֶׂר בָּאַמָּה וְכֵן הַכְּרוּב הַשֵּׁנִי:

[ישעיה פרק-כג-טז]  קְחִי כִנּוֹר סֹבִּי עִיר זוֹנָה נִשְׁכָּחָה הֵיטִיבִי נַגֵּן הַרְבִּי שִׁיר לְמַעַן תִּזָּכֵרִי:

[ישעיה פרק-מו-יג]  קֵרַבְתִּי צִדְקָתִי לֹא תִרְחָק וּתְשׁוּעָתִי לֹא תְאַחֵר וְנָתַתִּי בְצִיּוֹן תְּשׁוּעָה לְיִשְׂרָאֵל תִּפְאַרְתִּי:

[ישעיה פרק-נ-ח]  קָרוֹב מַצְדִּיקִי מִי יָרִיב אִתִּי נַעַמְדָה יָּחַד מִי בַעַל מִשְׁפָּטִי יִגַּשׁ אֵלָי:

[ירמיה פרק-יח-ב]  קוּם וְיָרַדְתָּ בֵּית הַיּוֹצֵר וְשָׁמָּה אַשְׁמִיעֲךָ אֶת דְּבָרָי:

[יונה פרק-א-ב]  קוּם לֵךְ אֶל נִינְוֵה הָעִיר הַגְּדוֹלָה וּקְרָא עָלֶיהָ כִּי עָלְתָה רָעָתָם לְפָנָי:

[תהילים פרק-מ-ב]  קַוֹּה קִוִּיתִי יְהוָה וַיֵּט אֵלַי וַיִּשְׁמַע שַׁוְעָתִי:

[תהילים פרק-סט-יט]  קָרְבָה אֶל נַפְשִׁי גְאָלָהּ לְמַעַן אֹיְבַי פְּדֵנִי:

[תהילים פרק-עז-ב]  קוֹלִי אֶל אֱלֹהִים וְאֶצְעָקָה קוֹלִי אֶל אֱלֹהִים וְהַאֲזִין אֵלָי:

[תהילים פרק-קיט-קמז]  קִדַּמְתִּי בַנֶּשֶׁף וָאֲשַׁוֵּעָה (לדבריך) לִדְבָרְךָ יִחָלְתִּי:

[תהילים פרק-קיט-קמט]  קוֹלִי שִׁמְעָה כְחַסְדֶּךָ יְהוָה כְּמִשְׁפָּטֶךָ חַיֵּנִי:

[תהילים פרק-קל-ה]  קִוִּיתִי יְהוָה קִוְּתָה נַפְשִׁי וְלִדְבָרוֹ הוֹחָלְתִּי:

[משלי פרק-ד-ה]  קְנֵה חָכְמָה קְנֵה בִינָה אַל תִּשְׁכַּח וְאַל תֵּט מֵאִמְרֵי פִי:

[איכה פרק-ג-נו]  קוֹלִי שָׁמָעְתָּ אַל תַּעְלֵם אָזְנְךָ לְרַוְחָתִי לְשַׁוְעָתִי:

[דניאל פרק-ד-יא]  קָרֵא בְחַיִל וְכֵן אָמַר גֹּדּוּ אִילָנָא וְקַצִּצוּ עַנְפוֹהִי אַתַּרוּ עָפְיֵהּ וּבַדַּרוּ אִנְבֵּהּ תְּנֻד חֵיוְתָא מִן תַּחְתּוֹהִי וְצִפְּרַיָּא מִן עַנְפוֹהִי:

[דניאל פרק-ז-טז]  קִרְבֵת עַל חַד מִן קָאֲמַיָּא וְיַצִּיבָא אֶבְעֵא מִנֵּהּ עַל כָּל דְּנָה וַאֲמַר לִי וּפְשַׁר מִלַּיָּא יְהוֹדְעִנַּנִי:


ק-כ


[בראשית פרק-כח-ב]  קוּם לֵךְ פַּדֶּנָה אֲרָם בֵּיתָה בְתוּאֵל אֲבִי אִמֶּךָ וְקַח לְךָ מִשָּׁם אִשָּׁה מִבְּנוֹת לָבָן אֲחִי אִמֶּךָ:

[מלכים א פרק-יז-ט]  קוּם לֵךְ צָרְפַתָה אֲשֶׁר לְצִידוֹן וְיָשַׁבְתָּ שָׁם הִנֵּה צִוִּיתִי שָׁם אִשָּׁה אַלְמָנָה לְכַלְכְּלֶךָ:

[יונה פרק-ג-ב]  קוּם לֵךְ אֶל נִינְוֵה הָעִיר הַגְּדוֹלָה וִּקְרָא אֵלֶיהָ אֶת הַקְּרִיאָה אֲשֶׁר אָנֹכִי דֹּבֵר אֵלֶיךָ:

[מיכה פרק-א-טז]  קָרְחִי וָגֹזִּי עַל בְּנֵי תַּעֲנוּגָיִךְ הַרְחִבִי קָרְחָתֵךְ כַּנֶּשֶׁר כִּי גָלוּ מִמֵּךְ:

[תהילים פרק-ט-כ]  קוּמָה יְהוָה אַל יָעֹז אֱנוֹשׁ יִשָּׁפְטוּ גוֹיִם עַל פָּנֶיךָ:

[תהילים פרק-יז-יג]  קוּמָה יְהוָה קַדְּמָה פָנָיו הַכְרִיעֵהוּ פַּלְּטָה נַפְשִׁי מֵרָשָׁע חַרְבֶּךָ:

[תהילים פרק-מד-כז]  קוּמָה עֶזְרָתָה לָּנוּ וּפְדֵנוּ לְמַעַן חַסְדֶּךָ:

[תהילים פרק-קיט-קמו]  קְרָאתִיךָ הוֹשִׁיעֵנִי וְאֶשְׁמְרָה עֵדֹתֶיךָ:

[תהילים פרק-קיט-קמח]  קִדְּמוּ עֵינַי אַשְׁמֻרוֹת לָשִׂיחַ בְּאִמְרָתֶךָ:

[תהילים פרק-קלב-ח]  קוּמָה יְהוָה לִמְנוּחָתֶךָ אַתָּה וַאֲרוֹן עֻזֶּךָ:

[משלי פרק-ו-כא]  קָשְׁרֵם עַל לִבְּךָ תָמִיד עָנְדֵם עַל גַּרְגְּרֹתֶךָ:

[משלי פרק-ז-ג]  קָשְׁרֵם עַל אֶצְבְּעֹתֶיךָ כָּתְבֵם עַל לוּחַ לִבֶּךָ:

[איוב פרק-כב-כב]  קַח נָא מִפִּיו תּוֹרָה וְשִׂים אֲמָרָיו בִּלְבָבֶךָ:


ק-ל


[שמואל א פרק-ב-ד]  קֶשֶׁת גִּבֹּרִים חַתִּים וְנִכְשָׁלִים אָזְרוּ חָיִל:

[ירמיה פרק-יז-יא]  קֹרֵא דָגַר וְלֹא יָלָד עֹשֶׂה עֹשֶׁר וְלֹא בְמִשְׁפָּט בַּחֲצִי (ימו) יָמָיו יַעַזְבֶנּוּ וּבְאַחֲרִיתוֹ יִהְיֶה נָבָל:

[ירמיה פרק-מח-ג]  קוֹל צְעָקָה מֵחֹרוֹנָיִם שֹׁד וָשֶׁבֶר גָּדוֹל:

[ירמיה פרק-נ-כב]  קוֹל מִלְחָמָה בָּאָרֶץ וְשֶׁבֶר גָּדוֹל:

[ירמיה פרק-נ-מב]  קֶשֶׁת וְכִידֹן יַחֲזִיקוּ אַכְזָרִי הֵמָּה וְלֹא יְרַחֵמוּ קוֹלָם כַּיָּם יֶהֱמֶה וְעַל סוּסִים יִרְכָּבוּ עָרוּךְ כְּאִישׁ לַמִּלְחָמָה עָלַיִךְ בַּת בָּבֶל:

[תהילים פרק-קיח-טו]  קוֹל רִנָּה וִישׁוּעָה בְּאָהֳלֵי צַדִּיקִים יְמִין יְהוָה עֹשָׂה חָיִל:

[שיר השירים פרק-ה-ה]  קַמְתִּי אֲנִי לִפְתֹּחַ לְדוֹדִי וְיָדַי נָטְפוּ מוֹר וְאֶצְבְּעֹתַי מוֹר עֹבֵר עַל כַּפּוֹת הַמַּנְעוּל:


ק-מ


[במדבר פרק-ח-ו]  קַח אֶת הַלְוִיִּם מִתּוֹךְ בְּנֵי יִשְׂרָאֵל וְטִהַרְתָּ אֹתָם:

[במדבר פרק-כ-ח]  קַח אֶת הַמַּטֶּה וְהַקְהֵל אֶת הָעֵדָה אַתָּה וְאַהֲרֹן אָחִיךָ וְדִבַּרְתֶּם אֶל הַסֶּלַע לְעֵינֵיהֶם וְנָתַן מֵימָיו וְהוֹצֵאתָ לָהֶם מַיִם מִן הַסֶּלַע וְהִשְׁקִיתָ אֶת הָעֵדָה וְאֶת בְּעִירָם:

[יהושע פרק-ז-יג]  קֻם קַדֵּשׁ אֶת הָעָם וְאָמַרְתָּ הִתְקַדְּשׁוּ לְמָחָר כִּי כֹה אָמַר יְהוָה אֱלֹהֵי יִשְׂרָאֵל חֵרֶם בְּקִרְבְּךָ יִשְׂרָאֵל לֹא תוּכַל לָקוּם לִפְנֵי אֹיְבֶיךָ עַד הֲסִירְכֶם הַחֵרֶם מִקִּרְבְּכֶם:

[מלכים א פרק-כא-יב]  קָרְאוּ צוֹם וְהֹשִׁיבוּ אֶת נָבוֹת בְּרֹאשׁ הָעָם:

[ישעיה פרק-נח-א]  קְרָא בְגָרוֹן אַל תַּחְשֹׂךְ כַּשּׁוֹפָר הָרֵם קוֹלֶךָ וְהַגֵּד לְעַמִּי פִּשְׁעָם וּלְבֵית יַעֲקֹב חַטֹּאתָם:

[ישעיה פרק-נט-ו]  קוּרֵיהֶם לֹא יִהְיוּ לְבֶגֶד וְלֹא יִתְכַּסּוּ בְּמַעֲשֵׂיהֶם מַעֲשֵׂיהֶם מַעֲשֵׂי אָוֶן וּפֹעַל חָמָס בְּכַפֵּיהֶם:

[ירמיה פרק-ג-כא]  קוֹל עַל שְׁפָיִים נִשְׁמָע בְּכִי תַחֲנוּנֵי בְּנֵי יִשְׂרָאֵל כִּי הֶעֱוּוּ אֶת דַּרְכָּם שָׁכְחוּ אֶת יְהוָֹה אֱלֹהֵיהֶם:

[ירמיה פרק-י-כב]  קוֹל שְׁמוּעָה הִנֵּה בָאָה וְרַעַשׁ גָּדוֹל מֵאֶרֶץ צָפוֹן לָשׂוּם אֶת עָרֵי יְהוּדָה שְׁמָמָה מְעוֹן תַּנִּים:

[ירמיה פרק-כב-יט]  קְבוּרַת חֲמוֹר יִקָּבֵר סָחוֹב וְהַשְׁלֵךְ מֵהָלְאָה לְשַׁעֲרֵי יְרוּשָׁלִָם:

[ירמיה פרק-כה-לו]  קוֹל צַעֲקַת הָרֹעִים וִילְלַת אַדִּירֵי הַצֹּאן כִּי שֹׁדֵד יְהוָה אֶת מַרְעִיתָם:

[ירמיה פרק-לג-ג]  קְרָא אֵלַי וְאֶעֱנֶךָּ וְאַגִּידָה לְּךָ גְּדֹלוֹת וּבְצֻרוֹת לֹא יְדַעְתָּם:

[ירמיה פרק-מג-ט]  קַח בְּיָדְךָ אֲבָנִים גְּדֹלוֹת וּטְמַנְתָּם בַּמֶּלֶט בַּמַּלְבֵּן אֲשֶׁר בְּפֶתַח בֵּית פַּרְעֹה בְּתַחְפַּנְחֵס לְעֵינֵי אֲנָשִׁים יְהוּדִים:

[ירמיה פרק-מו-כב]  קוֹלָהּ כַּנָּחָשׁ יֵלֵךְ כִּי בְחַיִל יֵלֵכוּ וּבְקַרְדֻּמּוֹת בָּאוּ לָהּ כְּחֹטְבֵי עֵצִים:

[ירמיה פרק-נא-נד]  קוֹל זְעָקָה מִבָּבֶל וְשֶׁבֶר גָּדוֹל מֵאֶרֶץ כַּשְׂדִּים:

[יחזקאל פרק-כב-כ]  קְבֻצַת כֶּסֶף וּנְחֹשֶׁת וּבַרְזֶל וְעוֹפֶרֶת וּבְדִיל אֶל תּוֹךְ כּוּר לָפַחַת עָלָיו אֵשׁ לְהַנְתִּיךְ כֵּן אֶקְבֹּץ בְּאַפִּי וּבַחֲמָתִי וְהִנַּחְתִּי וְהִתַּכְתִּי אֶתְכֶם:

[תהילים פרק-י-יב]  קוּמָה יְהוָה אֵל נְשָׂא יָדֶךָ אַל תִּשְׁכַּח (עניים) עֲנָוִים:

[תהילים פרק-כט-ג]  קוֹל יְהוָה עַל הַמָּיִם אֵל הַכָּבוֹד הִרְעִים יְהוָה עַל מַיִם רַבִּים:

[תהילים פרק-עד-כב]  קוּמָה אֱלֹהִים רִיבָה רִיבֶךָ זְכֹר חֶרְפָּתְךָ מִנִּי נָבָל כָּל הַיּוֹם:

[תהילים פרק-פב-ח]  קוּמָה אֱלֹהִים שָׁפְטָה הָאָרֶץ כִּי אַתָּה תִנְחַל בְּכָל הַגּוֹיִם:

[תהילים פרק-קיט-קנב]  קֶדֶם יָדַעְתִּי מֵעֵדֹתֶיךָ כִּי לְעוֹלָם יְסַדְתָּם:

[איוב פרק-כד-יח]  קַל הוּא עַל פְּנֵי מַיִם תְּקֻלַּל חֶלְקָתָם בָּאָרֶץ לֹא יִפְנֶה דֶּרֶךְ כְּרָמִים:

[שיר השירים פרק-א-יז]  קֹרוֹת בָּתֵּינוּ אֲרָזִים (רחיטנו) רַהִיטֵנוּ בְּרוֹתִים:

[איכה פרק-א-יט]  קָרָאתִי לַמְאַהֲבַי הֵמָּה רִמּוּנִי כֹּהֲנַי וּזְקֵנַי בָּעִיר גָּוָעוּ כִּי בִקְשׁוּ אֹכֶל לָמוֹ וְיָשִׁיבוּ אֶת נַפְשָׁם:

[קהלת פרק-ב-ז]  קָנִיתִי עֲבָדִים וּשְׁפָחוֹת וּבְנֵי בַיִת הָיָה לִי גַּם מִקְנֶה בָקָר וָצֹאן הַרְבֵּה הָיָה לִי מִכֹּל שֶׁהָיוּ לְפָנַי בִּירוּשָׁלִָם:


ק-נ


[יהושע פרק-טו-ס]  קִרְיַת בַּעַל הִיא קִרְיַת יְעָרִים וְהָרַבָּה עָרִים שְׁתַּיִם וְחַצְרֵיהֶן:

[ישעיה פרק-נא-ה]  קָרוֹב צִדְקִי יָצָא יִשְׁעִי וּזְרֹעַי עַמִּים יִשְׁפֹּטוּ אֵלַי אִיִּים יְקַוּוּ וְאֶל זְרֹעִי יְיַחֵלוּן:

[ישעיה פרק-נב-ח]  קוֹל צֹפַיִךְ נָשְׂאוּ קוֹל יַחְדָּו יְרַנֵּנוּ כִּי עַיִן בְּעַיִן יִרְאוּ בְּשׁוּב יְהוָה צִיּוֹן:

[ירמיה פרק-ו-כג]  קֶשֶׁת וְכִידוֹן יַחֲזִיקוּ אַכְזָרִי הוּא וְלֹא יְרַחֵמוּ קוֹלָם כַּיָּם יֶהֱמֶה וְעַל סוּסִים יִרְכָּבוּ עָרוּךְ כְּאִישׁ לַמִּלְחָמָה עָלַיִךְ בַּת צִיּוֹן:

[יחזקאל פרק-ז-כה]  קְפָדָה בָא וּבִקְשׁוּ שָׁלוֹם וָאָיִן:

[הושע פרק-יא-ב]  קָרְאוּ לָהֶם כֵּן הָלְכוּ מִפְּנֵיהֶם לַבְּעָלִים יְזַבֵּחוּ וְלַפְּסִלִים יְקַטֵּרוּן:

[זכריה פרק-יא-ג]  קוֹל יִלְלַת הָרֹעִים כִּי שֻׁדְּדָה אַדַּרְתָּם קוֹל שַׁאֲגַת כְּפִירִים כִּי שֻׁדַּד גְּאוֹן הַיַּרְדֵּן:

[תהילים פרק-כט-ה]  קוֹל יְהוָה שֹׁבֵר אֲרָזִים וַיְשַׁבֵּר יְהוָה אֶת אַרְזֵי הַלְּבָנוֹן:

[תהילים פרק-קמב-ב]  קוֹלִי אֶל יְהוָה אֶזְעָק קוֹלִי אֶל יְהוָה אֶתְחַנָּן:


ק-ע


[ירמיה פרק-יג-ד]  קַח אֶת הָאֵזוֹר אֲשֶׁר קָנִיתָ אֲשֶׁר עַל מָתְנֶיךָ וְקוּם לֵךְ פְּרָתָה וְטָמְנֵהוּ שָׁם בִּנְקִיק הַסָּלַע:

[תהילים פרק-לד-יט]  קָרוֹב יְהוָה לְנִשְׁבְּרֵי לֵב וְאֶת דַּכְּאֵי רוּחַ יוֹשִׁיעַ:

[איוב פרק-ל-כח]  קֹדֵר הִלַּכְתִּי בְּלֹא חַמָּה קַמְתִּי בַקָּהָל אֲשַׁוֵּעַ:


ק-פ


[זכריה פרק-א-ב]  קָצַף יְהוָה עַל אֲבוֹתֵיכֶם קָצֶף:

[משלי פרק-טז-טז]  קְנֹה חָכְמָה מַה טּוֹב מֵחָרוּץ וּקְנוֹת בִּינָה נִבְחָר מִכָּסֶף:


ק-צ


[מיכה פרק-ב-י]  קוּמוּ וּלְכוּ כִּי לֹא זֹאת הַמְּנוּחָה בַּעֲבוּר טָמְאָה תְּחַבֵּל וְחֶבֶל נִמְרָץ:

[מיכה פרק-ד-יג]  קוּמִי וָדוֹשִׁי בַת צִיּוֹן כִּי קַרְנֵךְ אָשִׂים בַּרְזֶל וּפַרְסֹתַיִךְ אָשִׂים נְחוּשָׁה וַהֲדִקּוֹת עַמִּים רַבִּים וְהַחֲרַמְתִּי לַיהוָה בִּצְעָם וְחֵילָם לַאֲדוֹן כָּל הָאָרֶץ:

[תהילים פרק-עז-יט]  קוֹל רַעַמְךָ בַּגַּלְגַּל הֵאִירוּ בְרָקִים תֵּבֵל רָגְזָה וַתִּרְעַשׁ הָאָרֶץ:


ק-ר


[במדבר פרק-כ-כה]  קַח אֶת אַהֲרֹן וְאֶת אֶלְעָזָר בְּנוֹ וְהַעַל אֹתָם הֹר הָהָר:

[צפניה פרק-א-יד]  קָרוֹב יוֹם יְהוָה הַגָּדוֹל קָרוֹב וּמַהֵר מְאֹד קוֹל יוֹם יְהוָה מַר צֹרֵחַ שָׁם גִּבּוֹר:

[תהילים פרק-כט-ד]  קוֹל יְהוָה בַּכֹּחַ קוֹל יְהוָה בֶּהָדָר:

[משלי פרק-ח-י]  קְחוּ מוּסָרִי וְאַל כָּסֶף וְדַעַת מֵחָרוּץ נִבְחָר:


ק-ש


[ויקרא פרק-כא-ו]  קְדֹשִׁים יִהְיוּ לֵאלֹהֵיהֶם וְלֹא יְחַלְּלוּ שֵׁם אֱלֹהֵיהֶם כִּי אֶת אִשֵּׁי יְהוָה לֶחֶם אֱלֹהֵיהֶם הֵם מַקְרִיבִם וְהָיוּ קֹדֶשׁ:

[יחזקאל פרק-מה-ד]  קֹדֶשׁ מִן הָאָרֶץ הוּא לַכֹּהֲנִים מְשָׁרְתֵי הַמִּקְדָּשׁ יִהְיֶה הַקְּרֵבִים לְשָׁרֵת אֶת יְהוָה וְהָיָה לָהֶם מָקוֹם לְבָתִּים וּמִקְדָּשׁ לַמִּקְדָּשׁ:

[תהילים פרק-כט-ז]  קוֹל יְהוָה חֹצֵב לַהֲבוֹת אֵשׁ:

[תהילים פרק-כט-ח]  קוֹל יְהוָה יָחִיל מִדְבָּר יָחִיל יְהוָה מִדְבַּר קָדֵשׁ:


ק-ת


[בראשית פרק-לב-יא]  קָטֹנְתִּי מִכֹּל הַחֲסָדִים וּמִכָּל הָאֱמֶת אֲשֶׁר עָשִׂיתָ אֶת עַבְדֶּךָ כִּי בְמַקְלִי עָבַרְתִּי אֶת הַיַּרְדֵּן הַזֶּה וְעַתָּה הָיִיתִי לִשְׁנֵי מַחֲנוֹת:

[שמות פרק-לה-ה]  קְחוּ מֵאִתְּכֶם תְּרוּמָה לַיהוָה כֹּל נְדִיב לִבּוֹ יְבִיאֶהָ אֵת תְּרוּמַת יְהוָה זָהָב וָכֶסֶף וּנְחֹשֶׁת:

[ויקרא פרק-ח-ב]  קַח אֶת אַהֲרֹן וְאֶת בָּנָיו אִתּוֹ וְאֵת הַבְּגָדִים וְאֵת שֶׁמֶן הַמִּשְׁחָה וְאֵת פַּר הַחַטָּאת וְאֵת שְׁנֵי הָאֵילִים וְאֵת סַל הַמַּצּוֹת:

[ישעיה פרק-מז-ב]  קְחִי רֵחַיִם וְטַחֲנִי קָמַח גַּלִּי צַמָּתֵךְ חֶשְׂפִּי שֹׁבֶל גַּלִּי שׁוֹק עִבְרִי נְהָרוֹת:

[יחזקאל פרק-כב-ח]  קָדָשַׁי בָּזִית וְאֶת שַׁבְּתֹתַי חִלָּלְתְּ:

[תהילים פרק-ג-ח]  קוּמָה יְהוָה הוֹשִׁיעֵנִי אֱלֹהַי כִּי הִכִּיתָ אֶת כָּל אֹיְבַי לֶחִי שִׁנֵּי רְשָׁעִים שִׁבַּרְתָּ:

[תהילים פרק-ז-ז]  קוּמָה יְהוָה בְּאַפֶּךָ הִנָּשֵׂא בְּעַבְרוֹת צוֹרְרָי וְעוּרָה אֵלַי מִשְׁפָּט צִוִּיתָ:

[תהילים פרק-מט-יב]  קִרְבָּם בָּתֵּימוֹ לְעוֹלָם מִשְׁכְּנֹתָם לְדֹר וָדֹר קָרְאוּ בִשְׁמוֹתָם עֲלֵי אֲדָמוֹת:

[תהילים פרק-סח-כו]  קִדְּמוּ שָׁרִים אַחַר נֹגְנִים בְּתוֹךְ עֲלָמוֹת תּוֹפֵפוֹת:

[תהילים פרק-קיט-קנא]  קָרוֹב אַתָּה יְהוָה וְכָל מִצְוֹתֶיךָ אֱמֶת:

[תהילים פרק-קמה-יח]  קָרוֹב יְהוָה לְכָל קֹרְאָיו לְכֹל אֲשֶׁר יִקְרָאֻהוּ בֶאֱמֶת:

[איוב פרק-כח-ג]  קֵץ שָׂם לַחֹשֶׁךְ וּלְכָל תַּכְלִית הוּא חוֹקֵר אֶבֶן אֹפֶל וְצַלְמָוֶת:

[שיר השירים פרק-ב-ח]  קוֹל דּוֹדִי הִנֵּה זֶה בָּא מְדַלֵּג עַל הֶהָרִים מְקַפֵּץ עַל הַגְּבָעוֹת:

[איכה פרק-ב-יט]  קוּמִי רֹנִּי (בליל) בַלַּיְלָה לְרֹאשׁ אַשְׁמֻרוֹת שִׁפְכִי כַמַּיִם לִבֵּךְ נֹכַח פְּנֵי אֲדֹנָי שְׂאִי אֵלָיו כַּפַּיִךְ עַל נֶפֶשׁ עוֹלָלַיִךְ הָעֲטוּפִים בְּרָעָב בְּרֹאשׁ כָּל חוּצוֹת:

[איכה פרק-ג-נה]  קָרָאתִי שִׁמְךָ יְהוָה מִבּוֹר תַּחְתִּיּוֹת:
