פרשת ויחי


{כח} וַיְחִ֤י יַֽעֲקֹב֙ בְּאֶ֣רֶץ מִצְרַ֔יִם שְׁבַ֥ע עֶשְׂרֵ֖ה שָׁנָ֑ה וַיְהִ֤י יְמֵי-יַֽעֲקֹב֙ שְׁנֵ֣י חַיָּ֔יו שֶׁ֣בַע שָׁנִ֔ים וְאַרְבָּעִ֥ים וּמְאַ֖ת שָׁנָֽה:  מ  וַיְחִ֤י יַֽעֲקֹב֙ בְּאֶ֣רֶץ מִצְרַ֔יִם שְׁבַ֥ע עֶשְׂרֵ֖ה שָׁנָ֑ה וַיְהִ֤י יְמֵי-יַֽעֲקֹב֙ שְׁנֵ֣י חַיָּ֔יו שֶׁ֣בַע שָׁנִ֔ים וְאַרְבָּעִ֥ים וּמְאַ֖ת שָׁנָֽה:  ת  וַחֲיָא יַעֲקֹב בְּאַרְעָא דְמִצְרַיִם שְׁבַע עֶשְׂרֵי שְׁנִין וַהֲווֹ יוֹמֵי יַעֲקֹב שְׁנֵי חַיוֹהִי מְאָה וְאַרְבְּעִין וּשְׁבַע שְׁנִין:

{כט} וַיִּקְרְב֣וּ יְמֵי-יִשְׂרָאֵל֘ לָמוּת֒ וַיִּקְרָ֣א | לִבְנ֣וֹ לְיוֹסֵ֗ף וַיֹּ֤אמֶר לוֹ֙ אִם-נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵינֶ֔יךָ שִׂים-נָ֥א יָֽדְךָ֖ תַּ֣חַת יְרֵכִ֑י וְעָשִׂ֤יתָ עִמָּדִי֙ חֶ֣סֶד וֶֽאֱמֶ֔ת אַל-נָ֥א תִקְבְּרֵ֖נִי בְּמִצְרָֽיִם:  מ  וַיִּקְרְב֣וּ יְמֵי-יִשְׂרָאֵל֘ לָמוּת֒ וַיִּקְרָ֣א | לִבְנ֣וֹ לְיוֹסֵ֗ף וַיֹּ֤אמֶר לוֹ֙ אִם-נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵינֶ֔יךָ שִׂים-נָ֥א יָֽדְךָ֖ תַּ֣חַת יְרֵכִ֑י וְעָשִׂ֤יתָ עִמָּדִי֙ חֶ֣סֶד וֶֽאֱמֶ֔ת אַל-נָ֥א תִקְבְּרֵ֖נִי בְּמִצְרָֽיִם:  ת  וּקְרִיבוּ יוֹמֵי יִשְׂרָאֵל לִמְמַת וּקְרָא לִבְרֵהּ לְיוֹסֵף וַאֲמַר לֵהּ אִם כְּעַן אַשְׁכָּחִית רַחֲמִין בְּעֵינָיךְ שַׁוִּי כְעַן יְדָךְ תְּחוֹת יַרְכִּי וְתַעְבֵּד עִמִּי טִיבוּ וּקְשׁוֹט לָא כְעַן תִּקְבְּרִנַּנִי בְּמִצְרַיִם:

{ל} וְשָֽׁכַבְתִּי֙ עִם-אֲבֹתַ֔י וּנְשָׂאתַ֨נִי֙ מִמִּצְרַ֔יִם וּקְבַרְתַּ֖נִי בִּקְבֻֽרָתָ֑ם וַיֹּאמַ֕ר אָֽנֹכִ֖י אֶֽעֱשֶׂ֥ה כִדְבָרֶֽךָ:  מ  וְשָֽׁכַבְתִּי֙ עִם-אֲבֹתַ֔י וּנְשָׂאתַ֨נִי֙ מִמִּצְרַ֔יִם וּקְבַרְתַּ֖נִי בִּקְבֻֽרָתָ֑ם וַיֹּאמַ֕ר אָֽנֹכִ֖י אֶֽעֱשֶׂ֥ה כִדְבָרֶֽךָ:  ת  וְאִשְׁכּוּב עִם אֲבָהָתַי וְתִטְּלִנַנִּי מִמִּצְרַיִם וְתִקְבְּרִנַנִּי בִּקְבֻרְתְּהוֹן וַאֲמַר אֲנָא אֶעְבַּד כְּפִתְגָּמָךְ:

{לא} וַיֹּ֗אמֶר הִשָּֽׁבְעָה֙ לִ֔י וַיִּשָּׁבַ֖ע ל֑וֹ וַיִּשְׁתַּ֥חוּ יִשְׂרָאֵ֖ל עַל-רֹ֥אשׁ הַמִּטָּֽה:  מ  וַיֹּ֗אמֶר הִשָּֽׁבְעָה֙ לִ֔י וַיִּשָּׁבַ֖ע ל֑וֹ וַיִּשְׁתַּ֥חוּ יִשְׂרָאֵ֖ל עַל-רֹ֥אשׁ הַמִּטָּֽה:  ת  וַאֲמַר קַיֵּם לִי וְקַיִים לֵהּ וּסְגִיד יִשְׂרָאֵל עַל רֵישׁ עַרְסָא: (פ)




בראשית פרק-מח

{א} וַיְהִ֗י אַֽחֲרֵי֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וַיֹּ֣אמֶר לְיוֹסֵ֔ף הִנֵּ֥ה אָבִ֖יךָ חֹלֶ֑ה וַיִּקַּ֞ח אֶת-שְׁנֵ֤י בָנָיו֙ עִמּ֔וֹ אֶת-מְנַשֶּׁ֖ה וְאֶת-אֶפְרָֽיִם:  מ  וַיְהִ֗י אַֽחֲרֵי֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וַיֹּ֣אמֶר לְיוֹסֵ֔ף הִנֵּ֥ה אָבִ֖יךָ חֹלֶ֑ה וַיִּקַּ֞ח אֶת-שְׁנֵ֤י בָנָיו֙ עִמּ֔וֹ אֶת-מְנַשֶּׁ֖ה וְאֶת-אֶפְרָֽיִם:  ת  וַהֲוָה בָּתַר פִּתְגָּמַיָּא הָאִלֵּין וַאֲמַר לְיוֹסֵף הָא אֲבוּךְ שְׁכִיב מְרָע וּדְבַר יָת תְּרֵין בְּנוֹהִי עִמֵּהּ יָת מְנַשֶּׁה וְיָת אֶפְרָיִם:

{ב} וַיַּגֵּ֣ד לְיַֽעֲקֹ֔ב וַיֹּ֕אמֶר הִנֵּ֛ה בִּנְךָ֥ יוֹסֵ֖ף בָּ֣א אֵלֶ֑יךָ וַיִּתְחַזֵּק֙ יִשְׂרָאֵ֔ל וַיֵּ֖שֶׁב עַל-הַמִּטָּֽה:  מ  וַיַּגֵּ֣ד לְיַֽעֲקֹ֔ב וַיֹּ֕אמֶר הִנֵּ֛ה בִּנְךָ֥ יוֹסֵ֖ף בָּ֣א אֵלֶ֑יךָ וַיִּתְחַזֵּק֙ יִשְׂרָאֵ֔ל וַיֵּ֖שֶׁב עַל-הַמִּטָּֽה:  ת  וְחַוִּי לְיַעֲקֹב וַאֲמַר הָא בְּרָךְ יוֹסֵף אֲתָא לְוָתָךְ וְאִתַּקַף יִשְׂרָאֵל וִיתֵיב עַל עַרְסָא:

{ג} וַיֹּ֤אמֶר יַֽעֲקֹב֙ אֶל-יוֹסֵ֔ף אֵ֥ל שַׁדַּ֛י נִרְאָֽה-אֵלַ֥י בְּל֖וּז בְּאֶ֣רֶץ כְּנָ֑עַן וַיְבָ֖רֶךְ אֹתִֽי:  מ  וַיֹּ֤אמֶר יַֽעֲקֹב֙ אֶל-יוֹסֵ֔ף אֵ֥ל שַׁדַּ֛י נִרְאָֽה-אֵלַ֥י בְּל֖וּז בְּאֶ֣רֶץ כְּנָ֑עַן וַיְבָ֖רֶךְ אֹתִֽי:  ת  וַאֲמַר יַעֲקֹב לְיוֹסֵף אֵל שַׁדַּי אִתְגְּלִי לִי בְלוּז בְּאַרְעָא דִכְנָעַן וּבָרִיךְ יָתִי:

{ד} וַיֹּ֣אמֶר אֵלַ֗י הִֽנְנִ֤י מַפְרְךָ֙ וְהִרְבִּיתִ֔ךָ וּנְתַתִּ֖יךָ לִקְהַ֣ל עַמִּ֑ים וְנָ֨תַתִּ֜י אֶת-הָאָ֧רֶץ הַזֹּ֛את לְזַרְעֲךָ֥ אַֽחֲרֶ֖יךָ אֲחֻזַּ֥ת עוֹלָֽם:  מ  וַיֹּ֣אמֶר אֵלַ֗י הִֽנְנִ֤י מַפְרְךָ֙ וְהִרְבִּיתִ֔ךָ וּנְתַתִּ֖יךָ לִקְהַ֣ל עַמִּ֑ים וְנָ֨תַתִּ֜י אֶת-הָאָ֧רֶץ הַזֹּ֛את לְזַרְעֲךָ֥ אַֽחֲרֶ֖יךָ אֲחֻזַּ֥ת עוֹלָֽם:  ת  וַאֲמַר לִי הָא אֲנָא מַפְשִׁינָךְ וְאַסְגִּנָּךְ וְאֶתְּנִנָּךְ לְכִנְשַׁת שִׁבְטִין וְאֶתֵּן יָת אַרְעָא הָדָא לִבְנָיךְ בַּתְרָךְ אַחֲסָנַת עֲלָם:

{ה} וְעַתָּ֡ה שְׁנֵֽי-בָנֶ֩יךָ֩ הַנּֽוֹלָדִ֨ים לְךָ֜ בְּאֶ֣רֶץ מִצְרַ֗יִם עַד-בֹּאִ֥י אֵלֶ֛יךָ מִצְרַ֖יְמָה לִי-הֵ֑ם אֶפְרַ֨יִם֙ וּמְנַשֶּׁ֔ה כִּרְאוּבֵ֥ן וְשִׁמְע֖וֹן יִֽהְיוּ-לִֽי:  מ  וְעַתָּ֡ה שְׁנֵֽי-בָנֶ֩יךָ֩ הַנּֽוֹלָדִ֨ים לְךָ֜ בְּאֶ֣רֶץ מִצְרַ֗יִם עַד-בֹּאִ֥י אֵלֶ֛יךָ מִצְרַ֖יְמָה לִי-הֵ֑ם אֶפְרַ֨יִם֙ וּמְנַשֶּׁ֔ה כִּרְאוּבֵ֥ן וְשִׁמְע֖וֹן יִֽהְיוּ-לִֽי:  ת  וּכְעַן תְּרֵין בְּנָיךְ דְּאִתְיְלִידוּ לָךְ בְּאַרְעָא דְמִצְרַיִם עַד מֵיתִי לְוָתָךְ לְמִצְרַיִם דִּילִי אִנּוּן אֶפְרַיִם וּמְנַשֶּׁה כִּרְאוּבֵן וְשִׁמְעוֹן יְהוֹן קֳדָמָי:

{ו} וּמֽוֹלַדְתְּךָ֛ אֲשֶׁר-הוֹלַ֥דְתָּ אַֽחֲרֵיהֶ֖ם לְךָ֣ יִֽהְי֑וּ עַ֣ל שֵׁ֧ם אֲחֵיהֶ֛ם יִקָּֽרְא֖וּ בְּנַֽחֲלָתָֽם:  מ  וּמֽוֹלַדְתְּךָ֛ אֲשֶׁר-הוֹלַ֥דְתָּ אַֽחֲרֵיהֶ֖ם לְךָ֣ יִֽהְי֑וּ עַ֣ל שֵׁ֧ם אֲחֵיהֶ֛ם יִקָּֽרְא֖וּ בְּנַֽחֲלָתָֽם:  ת  וּבְנָיךְ דִּי תוֹלִיד בַּתְרֵיהוֹן דִּילָךְ יְהוֹן עַל שׁוּם אֲחוּהוֹן יִתְקְרוּן בְּאַחֲסַנְתְּהוֹן:

{ז} וַֽאֲנִ֣י | בְּבֹאִ֣י מִפַּדָּ֗ן מֵ֩תָה֩ עָלַ֨י רָחֵ֜ל בְּאֶ֤רֶץ כְּנַ֨עַן֙ בַּדֶּ֔רֶךְ בְּע֥וֹד כִּבְרַת-אֶ֖רֶץ לָבֹ֣א אֶפְרָ֑תָה וָֽאֶקְבְּרֶ֤הָ שָּׁם֙ בְּדֶ֣רֶךְ אֶפְרָ֔ת הִ֖וא בֵּ֥ית לָֽחֶם:  מ  וַֽאֲנִ֣י | בְּבֹאִ֣י מִפַּדָּ֗ן מֵ֩תָה֩ עָלַ֨י רָחֵ֜ל בְּאֶ֤רֶץ כְּנַ֨עַן֙ בַּדֶּ֔רֶךְ בְּע֥וֹד כִּבְרַת-אֶ֖רֶץ לָבֹ֣א אֶפְרָ֑תָה וָֽאֶקְבְּרֶ֤הָ שָּׁם֙ בְּדֶ֣רֶךְ אֶפְרָ֔ת הִ֖וא בֵּ֥ית לָֽחֶם:  ת  וַאֲנָא בְּמֵיתִי מִפַּדָן מִיתַת עָלַי רָחֵל בְּאַרְעָא דִּכְנַעַן בְּאָרְחָא בְּעוֹד כְּרוּבָא דְאַרְעָא לְמֵיעַל לְאֶפְרָת וּקְבָרִתַּהּ תַּמָּן בְּאֹרַח אֶפְרָת הִיא בֵּית לָחֶם:

{ח} וַיַּ֥רְא יִשְׂרָאֵ֖ל אֶת-בְּנֵ֣י יוֹסֵ֑ף וַיֹּ֖אמֶר מִי-אֵֽלֶּה:  מ  וַיַּ֥רְא יִשְׂרָאֵ֖ל אֶת-בְּנֵ֣י יוֹסֵ֑ף וַיֹּ֖אמֶר מִי-אֵֽלֶּה:  ת  וַחֲזָא יִשְׂרָאֵל יָת בְּנֵי יוֹסֵף וַאֲמַר מַן אִלֵּין:

{ט} וַיֹּ֤אמֶר יוֹסֵף֙ אֶל-אָבִ֔יו בָּנַ֣י הֵ֔ם אֲשֶׁר-נָֽתַן-לִ֥י אֱלֹהִ֖ים בָּזֶ֑ה וַיֹּאמַ֕ר קָֽחֶם-נָ֥א אֵלַ֖י וַֽאֲבָֽרֲכֵֽם:  מ  וַיֹּ֤אמֶר יוֹסֵף֙ אֶל-אָבִ֔יו בָּנַ֣י הֵ֔ם אֲשֶׁר-נָֽתַן-לִ֥י אֱלֹהִ֖ים בָּזֶ֑ה וַיֹּאמַ֕ר קָֽחֶם-נָ֥א אֵלַ֖י וַֽאֲבָֽרֲכֵֽם:  ת  וַאֲמַר יוֹסֵף לַאֲבוּהִי בְּנַי אִנּוּן דִּי יְהַב לִי יְיָ הָכָא וַאֲמַר קָרֵבִנּוּן כְּעַן לְוָתִי וְאֵבָרֵכִנּוּן:

{י}  שני  וְעֵינֵ֤י יִשְׂרָאֵל֙ כָּֽבְד֣וּ מִזֹּ֔קֶן לֹ֥א יוּכַ֖ל לִרְא֑וֹת וַיַּגֵּ֤שׁ אֹתָם֙ אֵלָ֔יו וַיִּשַּׁ֥ק לָהֶ֖ם וַיְחַבֵּ֥ק לָהֶֽם:  מ  וְעֵינֵ֤י יִשְׂרָאֵל֙ כָּֽבְד֣וּ מִזֹּ֔קֶן לֹ֥א יוּכַ֖ל לִרְא֑וֹת וַיַּגֵּ֤שׁ אֹתָם֙ אֵלָ֔יו וַיִּשַּׁ֥ק לָהֶ֖ם וַיְחַבֵּ֥ק לָהֶֽם:  ת  וְעֵינֵי יִשְׂרָאֵל יְקָרָן מִסֵּיבוּ לָא יִכּוּל לְמֶחֱזֵי וְקָרִיב יָתְהוֹן לְוָתֵהּ וְנַשִּׁיק לְהוֹן וְגַפִּיף לְהוֹן:

{יא} וַיֹּ֤אמֶר יִשְׂרָאֵל֙ אֶל-יוֹסֵ֔ף רְאֹ֥ה פָנֶ֖יךָ לֹ֣א פִלָּ֑לְתִּי וְהִנֵּ֨ה הֶרְאָ֥ה אֹתִ֛י אֱלֹהִ֖ים גַּ֥ם אֶת-זַרְעֶֽךָ:  מ  וַיֹּ֤אמֶר יִשְׂרָאֵל֙ אֶל-יוֹסֵ֔ף רְאֹ֥ה פָנֶ֖יךָ לֹ֣א פִלָּ֑לְתִּי וְהִנֵּ֨ה הֶרְאָ֥ה אֹתִ֛י אֱלֹהִ֖ים גַּ֥ם אֶת-זַרְעֶֽךָ:  ת  וַאֲמַר יִשְׂרָאֵל לְיוֹסֵף לְמֶחֱזֵי אַפָיךְ לָא סְבָרִית וְהָא אַחֲזִי יָתִי יְיָ אַף יָת בְּנָיךְ:

{יב} וַיּוֹצֵ֥א יוֹסֵ֛ף אֹתָ֖ם מֵעִ֣ם בִּרְכָּ֑יו וַיִּשְׁתַּ֥חוּ לְאַפָּ֖יו אָֽרְצָה:  מ  וַיּוֹצֵ֥א יוֹסֵ֛ף אֹתָ֖ם מֵעִ֣ם בִּרְכָּ֑יו וַיִּשְׁתַּ֥חוּ לְאַפָּ֖יו אָֽרְצָה:  ת  וְאַפֵּיק יוֹסֵף יָתְהוֹן מִן קֳדָמוֹהִי וּסְגִיד עַל אַפּוֹהִי עַל אַרְעָא:

{יג} וַיִּקַּ֣ח יוֹסֵף֘ אֶת-שְׁנֵיהֶם֒ אֶת-אֶפְרַ֤יִם בִּֽימִינוֹ֙ מִשְּׂמֹ֣אל יִשְׂרָאֵ֔ל וְאֶת-מְנַשֶּׁ֥ה בִשְׂמֹאל֖וֹ מִימִ֣ין יִשְׂרָאֵ֑ל וַיַּגֵּ֖שׁ אֵלָֽיו:  מ  וַיִּקַּ֣ח יוֹסֵף֘ אֶת-שְׁנֵיהֶם֒ אֶת-אֶפְרַ֤יִם בִּֽימִינוֹ֙ מִשְּׂמֹ֣אל יִשְׂרָאֵ֔ל וְאֶת-מְנַשֶּׁ֥ה בִשְׂמֹאל֖וֹ מִימִ֣ין יִשְׂרָאֵ֑ל וַיַּגֵּ֖שׁ אֵלָֽיו:  ת  וּדְבַר יוֹסֵף יָת תַּרְוֵיהוֹן יָת אֶפְרַיִם בִּימִינֵהּ מִשְׂמָאלָא דְיִשְׂרָאֵל וְיָת מְנַשֶּׁה בִשְׂמָאלֵהּ מִימִינָא דְיִשְׂרָאֵל וְקָרֵיב לְוָתֵהּ:

{יד} וַיִּשְׁלַח֩ יִשְׂרָאֵ֨ל אֶת-יְמִינ֜וֹ וַיָּ֨שֶׁת עַל-רֹ֤אשׁ אֶפְרַ֨יִם֙ וְה֣וּא הַצָּעִ֔יר וְאֶת-שְׂמֹאל֖וֹ עַל-רֹ֣אשׁ מְנַשֶּׁ֑ה שִׂכֵּל֙ אֶת-יָדָ֔יו כִּ֥י מְנַשֶּׁ֖ה הַבְּכֽוֹר:  מ  וַיִּשְׁלַח֩ יִשְׂרָאֵ֨ל אֶת-יְמִינ֜וֹ וַיָּ֨שֶׁת עַל-רֹ֤אשׁ אֶפְרַ֨יִם֙ וְה֣וּא הַצָּעִ֔יר וְאֶת-שְׂמֹאל֖וֹ עַל-רֹ֣אשׁ מְנַשֶּׁ֑ה שִׂכֵּל֙ אֶת-יָדָ֔יו כִּ֥י מְנַשֶּׁ֖ה הַבְּכֽוֹר:  ת  וְאוֹשִׁיט יִשְׂרָאֵל יָת יְמִינֵהּ וְשַׁוִּי עַל רֵישָׁא דְאֶפְרַיִם וְהוּא זְעֵירָא וְיָת שְׂמָאלֵהּ עַל רֵישָׁא דִּמְנַשֶּׁה אַחְכִּמִינּוּן לִידוֹהִי אֲרֵי מְנַשֶּׁה בּוּכְרָא:

{טו} וַיְבָ֥רֶךְ אֶת-יוֹסֵ֖ף וַיֹּאמַ֑ר הָֽאֱלֹהִ֡ים אֲשֶׁר֩ הִתְהַלְּכ֨וּ אֲבֹתַ֤י לְפָנָיו֙ אַבְרָהָ֣ם וְיִצְחָ֔ק הָֽאֱלֹהִים֙ הָֽרֹעֶ֣ה אֹתִ֔י מֵֽעוֹדִ֖י עַד-הַיּ֥וֹם הַזֶּֽה:  מ  וַיְבָ֥רֶךְ אֶת-יוֹסֵ֖ף וַיֹּאמַ֑ר הָֽאֱלֹהִ֡ים אֲשֶׁר֩ הִתְהַלְּכ֨וּ אֲבֹתַ֤י לְפָנָיו֙ אַבְרָהָ֣ם וְיִצְחָ֔ק הָֽאֱלֹהִים֙ הָֽרֹעֶ֣ה אֹתִ֔י מֵֽעוֹדִ֖י עַד-הַיּ֥וֹם הַזֶּֽה:  ת  וּבְרִיךְ יָת יוֹסֵף וַאֲמָר יְיָ דִי פְלָחוּ אֲבָהָתַי קֳדָמוֹהִי אַבְרָהָם וְיִצְחָק יְיָ דְזַן יָתִי מִדְאִיתָנִי עַד יוֹמָא הָדֵין:

{טז} הַמַּלְאָךְ֩ הַגֹּאֵ֨ל אֹתִ֜י מִכָּל-רָ֗ע יְבָרֵךְ֘ אֶת-הַנְּעָרִים֒ וְיִקָּרֵ֤א בָהֶם֙ שְׁמִ֔י וְשֵׁ֥ם אֲבֹתַ֖י אַבְרָהָ֣ם וְיִצְחָ֑ק וְיִדְגּ֥וּ לָרֹ֖ב בְּקֶ֥רֶב הָאָֽרֶץ:  מ  הַמַּלְאָךְ֩ הַגֹּאֵ֨ל אֹתִ֜י מִכָּל-רָ֗ע יְבָרֵךְ֘ אֶת-הַנְּעָרִים֒ וְיִקָּרֵ֤א בָהֶם֙ שְׁמִ֔י וְשֵׁ֥ם אֲבֹתַ֖י אַבְרָהָ֣ם וְיִצְחָ֑ק וְיִדְגּ֥וּ לָרֹ֖ב בְּקֶ֥רֶב הָאָֽרֶץ:  ת  מַלְאָכָא דִי פְרַק יָתִי מִכָּל בִּישָׁא יְבָרֵךְ יָת עוּלֵמַיָּא וְיִתְקְרֵי בְהוֹן שְׁמִי וְשׁוּם אֲבָהָתַי אַבְרָהָם וְיִצְחָק וּכְנוּנֵי יַמָּא יִסְגוּן בְּגוֹ בְנֵי אֱנָשָׁא עַל אַרְעָא:

{יז}  שלישי  וַיַּ֣רְא יוֹסֵ֗ף כִּֽי-יָשִׁ֨ית אָבִ֧יו יַד-יְמִינ֛וֹ עַל-רֹ֥אשׁ אֶפְרַ֖יִם וַיֵּ֣רַע בְּעֵינָ֑יו וַיִּתְמֹ֣ךְ יַד-אָבִ֗יו לְהָסִ֥יר אֹתָ֛הּ מֵעַ֥ל רֹאשׁ-אֶפְרַ֖יִם עַל-רֹ֥אשׁ מְנַשֶּֽׁה:  מ  וַיַּ֣רְא יוֹסֵ֗ף כִּֽי-יָשִׁ֨ית אָבִ֧יו יַד-יְמִינ֛וֹ עַל-רֹ֥אשׁ אֶפְרַ֖יִם וַיֵּ֣רַע בְּעֵינָ֑יו וַיִּתְמֹ֣ךְ יַד-אָבִ֗יו לְהָסִ֥יר אֹתָ֛הּ מֵעַ֥ל רֹאשׁ-אֶפְרַ֖יִם עַל-רֹ֥אשׁ מְנַשֶּֽׁה:  ת  וַחֲזָא יוֹסֵף אֲרֵי שַׁוִּי אָבוּהִי יַד יְמִינֵהּ עַל רֵישָׁא דְאֶפְרַיִם וּבְאֵישׁ בְּעֵינוֹהִי וְסָעֵד יְדָא דְאָבוּהִי לְאַעֲדָאָה יָתַהּ מֵעַל רֵישָׁא דְאֶפְרַיִם לְאַנָחוּתַהּ עַל רֵישָׁא דִמְנַשֶּׁה:

{יח} וַיֹּ֧אמֶר יוֹסֵ֛ף אֶל-אָבִ֖יו לֹֽא-כֵ֣ן אָבִ֑י כִּי-זֶ֣ה הַבְּכֹ֔ר שִׂ֥ים יְמִֽינְךָ֖ עַל-רֹאשֽׁוֹ:  מ  וַיֹּ֧אמֶר יוֹסֵ֛ף אֶל-אָבִ֖יו לֹֽא-כֵ֣ן אָבִ֑י כִּי-זֶ֣ה הַבְּכֹ֔ר שִׂ֥ים יְמִֽינְךָ֖ עַל-רֹאשֽׁוֹ:  ת  וַאֲמַר יוֹסֵף לְאָבוּהִי לָא כֵן אַבָּא אֲרֵי דֵין בּוּכְרָא שַׁו יְמִינָךְ עַל רֵישֵׁהּ:

{יט} וַיְמָאֵ֣ן אָבִ֗יו וַיֹּ֨אמֶר֙ יָדַ֤עְתִּי בְנִי֙ יָדַ֔עְתִּי גַּם-ה֥וּא יִֽהְיֶה-לְּעָ֖ם וְגַם-הוּא יִגְדָּ֑ל וְאוּלָ֗ם אָחִ֤יו הַקָּטֹן֙ יִגְדַּ֣ל מִמֶּ֔נּוּ וְזַרְע֖וֹ יִֽהְיֶ֥ה מְלֹֽא-הַגּוֹיִֽם:  מ  וַיְמָאֵ֣ן אָבִ֗יו וַיֹּ֨אמֶר֙ יָדַ֤עְתִּי בְנִי֙ יָדַ֔עְתִּי גַּם-ה֥וּא יִֽהְיֶה-לְּעָ֖ם וְגַם-הוּא יִגְדָּ֑ל וְאוּלָ֗ם אָחִ֤יו הַקָּטֹן֙ יִגְדַּ֣ל מִמֶּ֔נּוּ וְזַרְע֖וֹ יִֽהְיֶ֥ה מְלֹֽא-הַגּוֹיִֽם:  ת  וְסָרֵיב אָבוּהִי וַאֲמַר יְדַעְנָא בְרִי יְדַעְנָא אַף הוּא יְהֵי לְעַמָּא וְאַף הוּא יִסְגֵּי וּבְרַם אָחוּהִי זְעִירָא יִסְגֵּי מִנֵּהּ וּבְנוֹהִי יְהוֹן שַׁלִּיטִין בְּעַמְמַיָּא:

{כ} וַיְבָ֨רֲכֵ֜ם בַּיּ֣וֹם הַהוּא֘ לֵאמוֹר֒ בְּךָ֗ יְבָרֵ֤ךְ יִשְׂרָאֵל֙ לֵאמֹ֔ר יְשִֽׂמְךָ֣ אֱלֹהִ֔ים כְּאֶפְרַ֖יִם וְכִמְנַשֶּׁ֑ה וַיָּ֥שֶׂם אֶת-אֶפְרַ֖יִם לִפְנֵ֥י מְנַשֶּֽׁה:  מ  וַיְבָ֨רֲכֵ֜ם בַּיּ֣וֹם הַהוּא֘ לֵאמוֹר֒ בְּךָ֗ יְבָרֵ֤ךְ יִשְׂרָאֵל֙ לֵאמֹ֔ר יְשִֽׂמְךָ֣ אֱלֹהִ֔ים כְּאֶפְרַ֖יִם וְכִמְנַשֶּׁ֑ה וַיָּ֥שֶׂם אֶת-אֶפְרַ֖יִם לִפְנֵ֥י מְנַשֶּֽׁה:  ת  וּבְרֵיכִנּוּן בְּיוֹמָא הַהוּא לְמֵימַר בָּךְ יְבָרֵךְ יִשְׂרָאֵל לְמֵימַר יְשַׁוִּינָךְ יְיָ כְּאֶפְרַיִם וְכִמְנַשֶּׁה וְשַׁוִּי יָת אֶפְרַיִם קֳדָם מְנַשֶּׁה:

{כא} וַיֹּ֤אמֶר יִשְׂרָאֵל֙ אֶל-יוֹסֵ֔ף הִנֵּ֥ה אָֽנֹכִ֖י מֵ֑ת וְהָיָ֤ה אֱלֹהִים֙ עִמָּכֶ֔ם וְהֵשִׁ֣יב אֶתְכֶ֔ם אֶל-אֶ֖רֶץ אֲבֹֽתֵיכֶֽם:  מ  וַיֹּ֤אמֶר יִשְׂרָאֵל֙ אֶל-יוֹסֵ֔ף הִנֵּ֥ה אָֽנֹכִ֖י מֵ֑ת וְהָיָ֤ה אֱלֹהִים֙ עִמָּכֶ֔ם וְהֵשִׁ֣יב אֶתְכֶ֔ם אֶל-אֶ֖רֶץ אֲבֹֽתֵיכֶֽם:  ת  וַאֲמַר יִשְׂרָאֵל לְיוֹסֵף הָא אֲנָא מָאִית וִיהֵי מֵימְרָא דַיְיָ בְּסַעְדְכוֹן וְיָתֵיב יָתְכוֹן לְאַרְעָא דַאֲבָהַתְכוֹן:

{כב} וַֽאֲנִ֗י נָתַ֧תִּי לְךָ֛ שְׁכֶ֥ם אַחַ֖ד עַל-אַחֶ֑יךָ אֲשֶׁ֤ר לָקַ֨חְתִּי֙ מִיַּ֣ד הָֽאֱמֹרִ֔י בְּחַרְבִּ֖י וּבְקַשְׁתִּֽי:  מ  וַֽאֲנִ֗י נָתַ֧תִּי לְךָ֛ שְׁכֶ֥ם אַחַ֖ד עַל-אַחֶ֑יךָ אֲשֶׁ֤ר לָקַ֨חְתִּי֙ מִיַּ֣ד הָֽאֱמֹרִ֔י בְּחַרְבִּ֖י וּבְקַשְׁתִּֽי:  ת  וַאֲנָא יְהָבִית לָךְ חוּלַק חַד יַתִּיר עַל אַחָיךְ דִּי נְסֵיבִית מִידָא דֶאֱמוֹרָאָה בִּצְלוֹתִי וּבְבָעוּתִי: (פ)




בראשית פרק-מט

{א}  רביעי  וַיִּקְרָ֥א יַֽעֲקֹ֖ב אֶל-בָּנָ֑יו וַיֹּ֗אמֶר הֵאָֽסְפוּ֙ וְאַגִּ֣ידָה לָכֶ֔ם אֵ֛ת אֲשֶׁר-יִקְרָ֥א אֶתְכֶ֖ם בְּאַֽחֲרִ֥ית הַיָּמִֽים:  מ  וַיִּקְרָ֥א יַֽעֲקֹ֖ב אֶל-בָּנָ֑יו וַיֹּ֗אמֶר הֵאָֽסְפוּ֙ וְאַגִּ֣ידָה לָכֶ֔ם אֵ֛ת אֲשֶׁר-יִקְרָ֥א אֶתְכֶ֖ם בְּאַֽחֲרִ֥ית הַיָּמִֽים:  ת  וּקְרָא יַעֲקֹב לִבְנוֹהִי וַאֲמַר אִתְכַּנָשׁוּ וַאֲחַוֵּי לְכוֹן יָת דִּי יְעָרַע יָתְכוֹן בְּסוֹף יוֹמַיָּא:

{ב} הִקָּֽבְצ֥וּ וְשִׁמְע֖וּ בְּנֵ֣י יַֽעֲקֹ֑ב וְשִׁמְע֖וּ אֶל-יִשְׂרָאֵ֥ל אֲבִיכֶֽם:  מ  הִקָּֽבְצ֥וּ וְשִׁמְע֖וּ בְּנֵ֣י יַֽעֲקֹ֑ב וְשִׁמְע֖וּ אֶל-יִשְׂרָאֵ֥ל אֲבִיכֶֽם:  ת  אִתְכַּנָּשׁוּ וּשְׁמָעוּ בְּנֵי יַעֲקֹב וְקַבִּילוּ אוּלְפַן מִן יִשְׂרָאֵל אֲבוּכוֹן:

{ג} רְאוּבֵן֙ בְּכֹ֣רִי אַ֔תָּה כֹּחִ֖י וְרֵאשִׁ֣ית אוֹנִ֑י יֶ֥תֶר שְׂאֵ֖ת וְיֶ֥תֶר עָֽז:  מ  רְאוּבֵן֙ בְּכֹ֣רִי אַ֔תָּה כֹּחִ֖י וְרֵאשִׁ֣ית אוֹנִ֑י יֶ֥תֶר שְׂאֵ֖ת וְיֶ֥תֶר עָֽז:  ת  רְאוּבֵן בּוּכְרִי אַתְּ חֵילִי וְרֵישׁ תָּקְפִּי לָךְ הֲוָה חָזֵי לְמִסַּב תְּלָתָא חוּלָקִין בְּכֵירוּתָא כְּהֻנְתָּא וּמַלְכוּתָא:

{ד} פַּ֤חַז כַּמַּ֨יִם֙ אַל-תּוֹתַ֔ר כִּ֥י עָלִ֖יתָ מִשְׁכְּבֵ֣י אָבִ֑יךָ אָ֥ז חִלַּ֖לְתָּ יְצוּעִ֥י עָלָֽה:  מ  פַּ֤חַז כַּמַּ֨יִם֙ אַל-תּוֹתַ֔ר כִּ֥י עָלִ֖יתָ מִשְׁכְּבֵ֣י אָבִ֑יךָ אָ֥ז חִלַּ֖לְתָּ יְצוּעִ֥י עָלָֽה:  ת  עַל דַאֲזַלְתָּא לָקֳבֵל אַפָּיךְ הָא כְמַיָּא בְּרַם לָא אַהֲנֵיתָא חוֹלָק יַתִּיר לָא תִסַּב אֲרֵי סְלֶקְתָּא בֵּית מִשְׁכָּבֵי אָבוּךְ בְּכֵן אֲחֶלְתָּא לְשִׁוּוּיִי בְּרִי סְלֶקְתָּא: (פ)

{ה} שִׁמְע֥וֹן וְלֵוִ֖י אַחִ֑ים כְּלֵ֥י חָמָ֖ס מְכֵרֹֽתֵיהֶֽם:  מ  שִׁמְע֥וֹן וְלֵוִ֖י אַחִ֑ים כְּלֵ֥י חָמָ֖ס מְכֵרֹֽתֵיהֶֽם:  ת  שִׁמְעוֹן וְלֵוִי אַחִין גַּבְרִין גִּבָּרִין בְּאַרַע תּוֹתָבוּתְהוֹן עֲבָדוּ גְּבוּרָא:

{ו} בְּסֹדָם֙ אַל-תָּבֹ֣א נַפְשִׁ֔י בִּקְהָלָ֖ם אַל-תֵּחַ֣ד כְּבֹדִ֑י כִּ֤י בְאַפָּם֙ הָ֣רְגוּ אִ֔ישׁ וּבִרְצֹנָ֖ם עִקְּרוּ-שֽׁוֹר:  מ  בְּסֹדָם֙ אַל-תָּבֹ֣א נַפְשִׁ֔י בִּקְהָלָ֖ם אַל-תֵּחַ֣ד כְּבֹדִ֑י כִּ֤י בְאַפָּם֙ הָ֣רְגוּ אִ֔ישׁ וּבִרְצֹנָ֖ם עִקְּרוּ-שֽׁוֹר:  ת  בְּרָזְהוֹן לָא הֲוַת נַפְשִׁי בְּאִתְכַּנְשֵׁהוֹן לִמְהַךְ לָא נְחָתִית מִן יְקָרִי אֲרֵי בְרָגְזְהוֹן קְטָלוּ קְטוֹל וּבִרְעוּתְהוֹן תַּרָעוּ שׁוּר סַנְּאָה:

{ז} אָר֤וּר אַפָּם֙ כִּ֣י עָ֔ז וְעֶבְרָתָ֖ם כִּ֣י קָשָׁ֑תָה אֲחַלְּקֵ֣ם בְּיַֽעֲקֹ֔ב וַֽאֲפִיצֵ֖ם בְּיִשְׂרָאֵֽל:  מ  אָר֤וּר אַפָּם֙ כִּ֣י עָ֔ז וְעֶבְרָתָ֖ם כִּ֣י קָשָׁ֑תָה אֲחַלְּקֵ֣ם בְּיַֽעֲקֹ֔ב וַֽאֲפִיצֵ֖ם בְּיִשְׂרָאֵֽל:  ת  לִיט רָגְזְהוֹן אֲרֵי תַקִיף וְחֵמַתְהוֹן אֲרֵי קַשְׁיָא אֲפַלְגִנוּן בְּיַעֲקֹב וְאֵבַדָרִנוּן בְּיִשְׂרָאֵל: (פ)

{ח} יְהוּדָ֗ה אַתָּה֙ יוֹדוּךָ אַחֶ֔יךָ יָֽדְךָ֖ בְּעֹ֣רֶף אֹֽיְבֶ֑יךָ יִשְׁתַּֽחֲו֥וּ לְךָ֖ בְּנֵ֥י אָבִֽיךָ:  מ  יְהוּדָ֗ה אַתָּה֙ יוֹדוּךָ אַחֶ֔יךָ יָֽדְךָ֖ בְּעֹ֣רֶף אֹֽיְבֶ֑יךָ יִשְׁתַּֽחֲו֥וּ לְךָ֖ בְּנֵ֥י אָבִֽיךָ:  ת  יְהוּדָה אַתְּ אוֹדֵיתָא וְלָא בְהֵיתְתָא בָּךְ יוֹדוּן אַחָיךְ יְדָךְ תִּתְקֵף עַל בַּעֲלֵי דְבָבָךְ יִתְבַּדְרוּן סַנְאָךְ יְהוֹן מְחַזְרִין קְדַּל קֳדָמָךְ וִיְהוֹן מְקַדְמִין לְמִשְׁאַל בִּשְׁלָמָךְ בְּנֵי אָבוּךְ:

{ט} גּ֤וּר אַרְיֵה֙ יְהוּדָ֔ה מִטֶּ֖רֶף בְּנִ֣י עָלִ֑יתָ כָּרַ֨ע רָבַ֧ץ כְּאַרְיֵ֛ה וּכְלָבִ֖יא מִ֥י יְקִימֶֽנּוּ:  מ  גּ֤וּר אַרְיֵה֙ יְהוּדָ֔ה מִטֶּ֖רֶף בְּנִ֣י עָלִ֑יתָ כָּרַ֨ע רָבַ֧ץ כְּאַרְיֵ֛ה וּכְלָבִ֖יא מִ֥י יְקִימֶֽנּוּ:  ת  שִׁלְטוֹן יְהֵי בְשֵׁרוּיָא וּבְסוֹפָא יִתְרַבָּא מַלְכָּא מִדְבֵית יְהוּדָה אֲרֵי מִדִין קַטְלָא בְּרִי נַפְשָׁךְ סְלֶקְתָּא יְנוּחַ יִשְׁרֵי בִתְקוֹף כְּאַרְיָא וּכְלֵיתָא וְלֵית מַלְכוּ דִתְזַעְזְעִנֵהּ:

{י} לֹֽא-יָס֥וּר שֵׁ֨בֶט֙ מִֽיהוּדָ֔ה וּמְחֹקֵ֖ק מִבֵּ֣ין רַגְלָ֑יו עַ֚ד כִּֽי-יָבֹ֣א (שילה) שִׁיל֔וֹ וְל֖וֹ יִקְּהַ֥ת עַמִּֽים:  מ  לֹֽא-יָס֥וּר שֵׁ֨בֶט֙ מִֽיהוּדָ֔ה וּמְחֹקֵ֖ק מִבֵּ֣ין רַגְלָ֑יו עַ֚ד כִּֽי-יָבֹ֣א (שילה) שִׁיל֔וֹ וְל֖וֹ יִקְּהַ֥ת עַמִּֽים:  ת  לָא יַעֲדֵי עָבֵד שָׁלְטַן מִדְבֵית יְהוּדָה וְסַפְרָא מִבְּנֵי בְנוֹהִי עַד עָלְמָא עַד דְיֵיתֵי מְשִׁיחָא דְדִילֵהּ הִיא מַלְכוּתָא וְלֵהּ יִשְׁתַּמְעוּן עַמְמַיָּא:

{כה} מֵאֵ֨ל אָבִ֜יךָ וְיַעְזְרֶ֗ךָּ וְאֵ֤ת שַׁדַּי֙ וִֽיבָ֣רֲכֶ֔ךָּ בִּרְכֹ֤ת שָׁמַ֨יִם֙ מֵעָ֔ל בִּרְכֹ֥ת תְּה֖וֹם רֹבֶ֣צֶת תָּ֑חַת בִּרְכֹ֥ת שָׁדַ֖יִם וָרָֽחַם:  מ  מֵאֵ֨ל אָבִ֜יךָ וְיַעְזְרֶ֗ךָּ וְאֵ֤ת שַׁדַּי֙ וִֽיבָ֣רֲכֶ֔ךָּ בִּרְכֹ֤ת שָׁמַ֨יִם֙ מֵעָ֔ל בִּרְכֹ֥ת תְּה֖וֹם רֹבֶ֣צֶת תָּ֑חַת בִּרְכֹ֥ת שָׁדַ֖יִם וָרָֽחַם:  ת  מֵימַר אֱלָהָא דְאָבוּךְ יְהֵי בְסַעְדָךְ וְיָת שַׁדַּי וִיבָרְכִנָּךְ בִּרְכַן דְּנַחֲתָן מִטַּלָּא דִשְׁמַיָּא מִלְעֵלָא בִּרְכַן דְנַגְדַן מִמַעֲמַקֵי אַרְעָא מִלְרָע בִּרְכְּתָא דְאָבוּךְ וּדְאִמָּךְ:

{כו} בִּרְכֹ֣ת אָבִ֗יךָ גָּֽבְרוּ֙ עַל-בִּרְכֹ֣ת הוֹרַ֔י עַֽד-תַּֽאֲוַ֖ת גִּבְעֹ֣ת עוֹלָ֑ם תִּֽהְיֶ֨יןָ֨ לְרֹ֣אשׁ יוֹסֵ֔ף וּלְקָדְקֹ֖ד נְזִ֥יר אֶחָֽיו:  מ  בִּרְכֹ֣ת אָבִ֗יךָ גָּֽבְרוּ֙ עַל-בִּרְכֹ֣ת הוֹרַ֔י עַֽד-תַּֽאֲוַ֖ת גִּבְעֹ֣ת עוֹלָ֑ם תִּֽהְיֶ֨יןָ֨ לְרֹ֣אשׁ יוֹסֵ֔ף וּלְקָדְקֹ֖ד נְזִ֥יר אֶחָֽיו:  ת  בִּרְכְּתָא דְאָבוּךְ יִתּוֹסְפָן עַל בִּרְכְּתָא דִילִי בָּרִיכוּ אֲבָהָתַי דַחֲמִידוּ לְהֶן רַבְרְבַיָּא דְמִן עָלְמָא יֶהֶוְיַן כָּל אִלֵּין לְרֵישָׁא דְיוֹסֵף וּלְגַבְרָא פָּרִישָׁא דַאֲחוֹהִי: (פ)

{כז}  שישי  בִּנְיָמִין֙ זְאֵ֣ב יִטְרָ֔ף בַּבֹּ֖קֶר יֹ֣אכַל עַ֑ד וְלָעֶ֖רֶב יְחַלֵּ֥ק שָׁלָֽל:  מ  בִּנְיָמִין֙ זְאֵ֣ב יִטְרָ֔ף בַּבֹּ֖קֶר יֹ֣אכַל עַ֑ד וְלָעֶ֖רֶב יְחַלֵּ֥ק שָׁלָֽל:  ת  בִּנְיָמִין בְּאַרְעֵהּ תִּשְׁרֵי שְׁכִנְתָּא וּבְאַחֲסַנְתֵּהּ יִתְבְּנֵי מַקְדְשָׁא בְּצַפְרָא וּבְפַנְיָא יְהוֹן מְקָרְבִין כַּהֲנַיָּא קָרְבָּנָא וּבְעִדַּן רַמְשָׁא יְהוֹן מְפַלְגִין מוֹתַר חוּלְקְהוֹן מִשְׁאַר קָדְשַׁיָּא:

{כח} כָּל-אֵ֛לֶּה שִׁבְטֵ֥י יִשְׂרָאֵ֖ל שְׁנֵ֣ים עָשָׂ֑ר וְ֠זֹ֠את אֲשֶׁר-דִּבֶּ֨ר לָהֶ֤ם אֲבִיהֶם֙ וַיְבָ֣רֶךְ אוֹתָ֔ם אִ֛ישׁ אֲשֶׁ֥ר כְּבִרְכָת֖וֹ בֵּרַ֥ךְ אֹתָֽם:  מ  כָּל-אֵ֛לֶּה שִׁבְטֵ֥י יִשְׂרָאֵ֖ל שְׁנֵ֣ים עָשָׂ֑ר וְ֠זֹ֠את אֲשֶׁר-דִּבֶּ֨ר לָהֶ֤ם אֲבִיהֶם֙ וַיְבָ֣רֶךְ אוֹתָ֔ם אִ֛ישׁ אֲשֶׁ֥ר כְּבִרְכָת֖וֹ בֵּרַ֥ךְ אֹתָֽם:  ת  כָּל אִלֵּין שִׁבְטַיָּא דְיִשְׂרָאֵל תְּרֵין עֲסַר וְדָא דְמַלֵּיל לְהוֹן אֲבוּהוֹן וּבָרִיךְ יָתְהוֹן גְבַר דִּי כְבִרְכְּתֵהּ בָּרֵךְ יָתְהוֹן:

{כט} וַיְצַ֣ו אוֹתָ֗ם וַיֹּ֤אמֶר אֲלֵהֶם֙ אֲנִי֙ נֶֽאֱסָ֣ף אֶל-עַמִּ֔י קִבְר֥וּ אֹתִ֖י אֶל-אֲבֹתָ֑י אֶ֨ל-הַמְּעָרָ֔ה אֲשֶׁ֥ר בִּשְׂדֵ֖ה עֶפְר֥וֹן הַֽחִתִּֽי:  מ  וַיְצַ֣ו אוֹתָ֗ם וַיֹּ֤אמֶר אֲלֵהֶם֙ אֲנִי֙ נֶֽאֱסָ֣ף אֶל-עַמִּ֔י קִבְר֥וּ אֹתִ֖י אֶל-אֲבֹתָ֑י אֶ֨ל-הַמְּעָרָ֔ה אֲשֶׁ֥ר בִּשְׂדֵ֖ה עֶפְר֥וֹן הַֽחִתִּֽי:  ת  וּפַקֵד יָתְהוֹן וַאֲמַר לְהוֹן אֲנָא מִתְכְּנֵשׁ לְעַמִּי קְבָרוּ יָתִי לְוַת אֲבָהָתִי בִּמְעַרְתָּא דִי בַחֲקַל עֶפְרוֹן חִתָּאָה:

{ל} בַּמְּעָרָ֞ה אֲשֶׁ֨ר בִּשְׂדֵ֧ה הַמַּכְפֵּלָ֛ה אֲשֶׁר עַל-פְּנֵ֥י-מַמְרֵ֖א בְּאֶ֣רֶץ כְּנָ֑עַן אֲשֶׁר֩ קָנָ֨ה אַבְרָהָ֜ם אֶת-הַשָּׂדֶ֗ה מֵאֵ֛ת עֶפְרֹ֥ן הַֽחִתִּ֖י לַֽאֲחֻזַּת-קָֽבֶר:  מ  בַּמְּעָרָ֞ה אֲשֶׁ֨ר בִּשְׂדֵ֧ה הַמַּכְפֵּלָ֛ה אֲשֶׁר עַל-פְּנֵ֥י-מַמְרֵ֖א בְּאֶ֣רֶץ כְּנָ֑עַן אֲשֶׁר֩ קָנָ֨ה אַבְרָהָ֜ם אֶת-הַשָּׂדֶ֗ה מֵאֵ֛ת עֶפְרֹ֥ן הַֽחִתִּ֖י לַֽאֲחֻזַּת-קָֽבֶר:  ת  בִּמְעַרְתָּא דִי בַחֲקַל כָּפֶלְתָּא דִי עַל אַפֵּי מַמְרֵא בְּאַרְעָא דִכְנָעַן דִי זְבַן אַבְרָהָם יָת חַקְלָא מִן עֶפְרֹן חִתָּאָה לְאַחֲסָנַת קְבוּרָא:

{לא} שָׁ֣מָּה קָּֽבְר֞וּ אֶת-אַבְרָהָ֗ם וְאֵת֙ שָׂרָ֣ה אִשְׁתּ֔וֹ שָׁ֚מָּה קָֽבְר֣וּ אֶת-יִצְחָ֔ק וְאֵ֖ת רִבְקָ֣ה אִשְׁתּ֑וֹ וְשָׁ֥מָּה קָבַ֖רְתִּי אֶת-לֵאָֽה:  מ  שָׁ֣מָּה קָּֽבְר֞וּ אֶת-אַבְרָהָ֗ם וְאֵת֙ שָׂרָ֣ה אִשְׁתּ֔וֹ שָׁ֚מָּה קָֽבְר֣וּ אֶת-יִצְחָ֔ק וְאֵ֖ת רִבְקָ֣ה אִשְׁתּ֑וֹ וְשָׁ֥מָּה קָבַ֖רְתִּי אֶת-לֵאָֽה:  ת  תַּמָּן קְבָרוּ יָת אַבְרָהָם וְיָת שָׂרָה אִתְּתֵהּ תַּמָּן קְבָרוּ יָת יִצְחָק וְיָת רִבְקָה אִתְּתֵהּ וְתַמָּן קְבָרִית יָת לֵאָה:

{לב} מִקְנֵ֧ה הַשָּׂדֶ֛ה וְהַמְּעָרָ֥ה אֲשֶׁר-בּ֖וֹ מֵאֵ֥ת בְּנֵי-חֵֽת:  מ  מִקְנֵ֧ה הַשָּׂדֶ֛ה וְהַמְּעָרָ֥ה אֲשֶׁר-בּ֖וֹ מֵאֵ֥ת בְּנֵי-חֵֽת:  ת  זְבִינֵי חַקְלָא וּמְעַרְתָּא דִי בֵהּ מִן בְּנֵי חִתָּאָהּ:

{לג} וַיְכַ֤ל יַֽעֲקֹב֙ לְצַוֹּ֣ת אֶת-בָּנָ֔יו וַיֶּֽאֱסֹ֥ף רַגְלָ֖יו אֶל-הַמִּטָּ֑ה וַיִּגְוַ֖ע וַיֵּאָ֥סֶף אֶל-עַמָּֽיו:  מ  וַיְכַ֤ל יַֽעֲקֹב֙ לְצַוֹּ֣ת אֶת-בָּנָ֔יו וַיֶּֽאֱסֹ֥ף רַגְלָ֖יו אֶל-הַמִּטָּ֑ה וַיִּגְוַ֖ע וַיֵּאָ֥סֶף אֶל-עַמָּֽיו:  ת  וְשֵׁיצֵי יַעֲקֹב לְפַקָּדָא יָת בְּנוֹהִי וּכְנַשׁ רַגְלוֹהִי לְעַרְסָא וְאִתְנְגִיד וְאִתְכְּנִישׁ לְעַמֵּהּ:

{א} וַיִּפֹּ֥ל יוֹסֵ֖ף עַל-פְּנֵ֣י אָבִ֑יו וַיֵּ֥בְךְּ עָלָ֖יו וַיִּֽשַּׁק-לֽוֹ:  מ  וַיִּפֹּ֥ל יוֹסֵ֖ף עַל-פְּנֵ֣י אָבִ֑יו וַיֵּ֥בְךְּ עָלָ֖יו וַיִּֽשַּׁק-לֽוֹ:  ת  וּנְפַל יוֹסֵף עַל אַפֵּי אָבוּהִי וּבְכָא עֲלוֹהִי וּנְשַׁק לֵהּ:

{ב} וַיְצַ֨ו יוֹסֵ֤ף אֶת-עֲבָדָיו֙ אֶת-הָרֹ֣פְאִ֔ים לַֽחֲנֹ֖ט אֶת-אָבִ֑יו וַיַּֽחַנְט֥וּ הָרֹֽפְאִ֖ים אֶת-יִשְׂרָאֵֽל:  מ  וַיְצַ֨ו יוֹסֵ֤ף אֶת-עֲבָדָיו֙ אֶת-הָרֹ֣פְאִ֔ים לַֽחֲנֹ֖ט אֶת-אָבִ֑יו וַיַּֽחַנְט֥וּ הָרֹֽפְאִ֖ים אֶת-יִשְׂרָאֵֽל:  ת  וּפַקֵּיד יוֹסֵף יָת עַבְדוֹהִי יָת אַסְוָתָא לְמֶחְנַט יָת אָבוּהִי וְחָנְטוּ אַסְוָתָא יָת יִשְׂרָאֵל:

{ג} וַיִּמְלְאוּ-לוֹ֙ אַרְבָּעִ֣ים י֔וֹם כִּ֛י כֵּ֥ן יִמְלְא֖וּ יְמֵ֣י הַֽחֲנֻטִ֑ים וַיִּבְכּ֥וּ אֹת֛וֹ מִצְרַ֖יִם שִׁבְעִ֥ים יֽוֹם:  מ  וַיִּמְלְאוּ-לוֹ֙ אַרְבָּעִ֣ים י֔וֹם כִּ֛י כֵּ֥ן יִמְלְא֖וּ יְמֵ֣י הַֽחֲנֻטִ֑ים וַיִּבְכּ֥וּ אֹת֛וֹ מִצְרַ֖יִם שִׁבְעִ֥ים יֽוֹם:  ת  וּשְׁלִימוּ לֵהּ אַרְבְּעִין יוֹמִין אֲרֵי כֵּן שְׁלִימוּ יוֹמֵי חֲנִיטַיָּא וּבְכוֹ יָתֵהּ מִצְרָאֵי שַׁבְעִין יוֹמִין:

{ד} וַיַּֽעַבְרוּ֙ יְמֵ֣י בְכִית֔וֹ וַיְדַבֵּ֣ר יוֹסֵ֔ף אֶל-בֵּ֥ית פַּרְעֹ֖ה לֵאמֹ֑ר אִם-נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵ֣ינֵיכֶ֔ם דַּבְּרוּ-נָ֕א בְּאָזְנֵ֥י פַרְעֹ֖ה לֵאמֹֽר:  מ  וַיַּֽעַבְרוּ֙ יְמֵ֣י בְכִית֔וֹ וַיְדַבֵּ֣ר יוֹסֵ֔ף אֶל-בֵּ֥ית פַּרְעֹ֖ה לֵאמֹ֑ר אִם-נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵ֣ינֵיכֶ֔ם דַּבְּרוּ-נָ֕א בְּאָזְנֵ֥י פַרְעֹ֖ה לֵאמֹֽר:  ת  וַעֲבָרוּ יוֹמֵי בְכִיתֵהּ וּמַלִיל יוֹסֵף עִם בֵּית פַּרְעֹה לְמֵימָר אִם כְּעַן אַשְׁכָּחִית רַחֲמִין בְּעֵינֵיכוֹן מַלִּילוּ כְעַן קֳדָם פַּרְעֹה לְמֵימָר:

{ה} אָבִ֞י הִשְׁבִּיעַ֣נִי לֵאמֹ֗ר הִנֵּ֣ה אָֽנֹכִי֘ מֵת֒ בְּקִבְרִ֗י אֲשֶׁ֨ר כָּרִ֤יתִי לִי֙ בְּאֶ֣רֶץ כְּנַ֔עַן שָׁ֖מָּה תִּקְבְּרֵ֑נִי וְעַתָּ֗ה אֶֽעֱלֶה-נָּ֛א וְאֶקְבְּרָ֥ה אֶת-אָבִ֖י וְאָשֽׁוּבָה:  מ  אָבִ֞י הִשְׁבִּיעַ֣נִי לֵאמֹ֗ר הִנֵּ֣ה אָֽנֹכִי֘ מֵת֒ בְּקִבְרִ֗י אֲשֶׁ֨ר כָּרִ֤יתִי לִי֙ בְּאֶ֣רֶץ כְּנַ֔עַן שָׁ֖מָּה תִּקְבְּרֵ֑נִי וְעַתָּ֗ה אֶֽעֱלֶה-נָּ֛א וְאֶקְבְּרָ֥ה אֶת-אָבִ֖י וְאָשֽׁוּבָה:  ת  אַבָּא קַיֵּים עָלַי לְמֵימָר הָא אֲנָא מָאִית בְּקִבְרִי דְאַתְקֵנִית לִי בְּאַרְעָא דִכְנָעַן תַּמָּן תִּקְבְּרִנָנִי וּכְעַן אֶסַּק כְּעַן וְאֶקְבַּר יָת אַבָּא וְאֵ יתוּב:

{ו} וַיֹּ֖אמֶר פַּרְעֹ֑ה עֲלֵ֛ה וּקְבֹ֥ר אֶת-אָבִ֖יךָ כַּֽאֲשֶׁ֥ר הִשְׁבִּיעֶֽךָ:  מ  וַיֹּ֖אמֶר פַּרְעֹ֑ה עֲלֵ֛ה וּקְבֹ֥ר אֶת-אָבִ֖יךָ כַּֽאֲשֶׁ֥ר הִשְׁבִּיעֶֽךָ:  ת  וַאֲמַר פַּרְעֹה סַק וּקְבוֹר יָת אֲבוּךְ כְּמָא דְקַיֵּים עֲלָךְ:

{ז} וַיַּ֥עַל יוֹסֵ֖ף לִקְבֹּ֣ר אֶת-אָבִ֑יו וַיַּֽעֲל֨וּ אִתּ֜וֹ כָּל-עַבְדֵ֤י פַרְעֹה֙ זִקְנֵ֣י בֵית֔וֹ וְכֹ֖ל זִקְנֵ֥י אֶֽרֶץ-מִצְרָֽיִם:  מ  וַיַּ֥עַל יוֹסֵ֖ף לִקְבֹּ֣ר אֶת-אָבִ֑יו וַיַּֽעֲל֨וּ אִתּ֜וֹ כָּל-עַבְדֵ֤י פַרְעֹה֙ זִקְנֵ֣י בֵית֔וֹ וְכֹ֖ל זִקְנֵ֥י אֶֽרֶץ-מִצְרָֽיִם:  ת  וּסְלֵיק יוֹסֵף לְמִקְבַּר יָת אָבוּהִי וּסְלִיקוּ עִמֵּהּ כָּל עַבְדֵי פַרְעֹה סָבֵי בֵיתֵהּ וְכֹל סָבֵי אַרְעָא דְמִצְרָיִם:

{ח} וְכֹל֙ בֵּ֣ית יוֹסֵ֔ף וְאֶחָ֖יו וּבֵ֣ית אָבִ֑יו רַ֗ק טַפָּם֙ וְצֹאנָ֣ם וּבְקָרָ֔ם עָֽזְב֖וּ בְּאֶ֥רֶץ גּֽשֶׁן:  מ  וְכֹל֙ בֵּ֣ית יוֹסֵ֔ף וְאֶחָ֖יו וּבֵ֣ית אָבִ֑יו רַ֗ק טַפָּם֙ וְצֹאנָ֣ם וּבְקָרָ֔ם עָֽזְב֖וּ בְּאֶ֥רֶץ גּֽשֶׁן:  ת  וְכֹל בֵּית יוֹסֵף וַאֲחוֹהִי וּבֵית אָבוּהִי לְחוֹד טַפְלְהוֹן וְעָנְהוֹן וְתוֹרֵיהוֹן שְׁבָקוּ בְּאַרְעָא דְגשֶׁן:

{ט} וַיַּ֣עַל עִמּ֔וֹ גַּם-רֶ֖כֶב גַּם-פָּֽרָשִׁ֑ים וַיְהִ֥י הַֽמַּֽחֲנֶ֖ה כָּבֵ֥ד מְאֹֽד:  מ  וַיַּ֣עַל עִמּ֔וֹ גַּם-רֶ֖כֶב גַּם-פָּֽרָשִׁ֑ים וַיְהִ֥י הַֽמַּֽחֲנֶ֖ה כָּבֵ֥ד מְאֹֽד:  ת  וּסְלִיקוּ עִמֵּהּ אַף רְתִכִּין אַף פָּרָשִׁין וַהֲוָה מַשְׁרִיתָא סַגִּיאָה לַחֲדָא:

{י} וַיָּבֹ֜אוּ עַד-גֹּ֣רֶן הָֽאָטָ֗ד אֲשֶׁר֙ בְּעֵ֣בֶר הַיַּרְדֵּ֔ן וַיִּ֨סְפְּדוּ-שָׁ֔ם מִסְפֵּ֛ד גָּד֥וֹל וְכָבֵ֖ד מְאֹ֑ד וַיַּ֧עַשׂ לְאָבִ֛יו אֵ֖בֶל שִׁבְעַ֥ת יָמִֽים:  מ  וַיָּבֹ֜אוּ עַד-גֹּ֣רֶן הָֽאָטָ֗ד אֲשֶׁר֙ בְּעֵ֣בֶר הַיַּרְדֵּ֔ן וַיִּ֨סְפְּדוּ-שָׁ֔ם מִסְפֵּ֛ד גָּד֥וֹל וְכָבֵ֖ד מְאֹ֑ד וַיַּ֧עַשׂ לְאָבִ֛יו אֵ֖בֶל שִׁבְעַ֥ת יָמִֽים:  ת  וַאֲתוֹ עַד בֵּית אִדְרֵי דְאָטָד דִּי בְּעִבְרָא דְיַרְדְנָא וּסְפָדוּ תַמָּן מִסְפֵּד רַב וְתַקִּיף לַחֲדָא וְעָבַד לַאֲבוּהִי אֶבְלָא שַׁבְעָא יוֹמִין:

{יא} וַיַּ֡רְא יוֹשֵׁב֩ הָאָ֨רֶץ הַֽכְּנַֽעֲנִ֜י אֶת-הָאֵ֗בֶל בְּגֹ֨רֶן֙ הָֽאָטָ֔ד וַיֹּ֣אמְר֔וּ אֵֽבֶל-כָּבֵ֥ד זֶ֖ה לְמִצְרָ֑יִם עַל-כֵּ֞ן קָרָ֤א שְׁמָהּ֙ אָבֵ֣ל מִצְרַ֔יִם אֲשֶׁ֖ר בְּעֵ֥בֶר הַיַּרְדֵּֽן:  מ  וַיַּ֡רְא יוֹשֵׁב֩ הָאָ֨רֶץ הַֽכְּנַֽעֲנִ֜י אֶת-הָאֵ֗בֶל בְּגֹ֨רֶן֙ הָֽאָטָ֔ד וַיֹּ֣אמְר֔וּ אֵֽבֶל-כָּבֵ֥ד זֶ֖ה לְמִצְרָ֑יִם עַל-כֵּ֞ן קָרָ֤א שְׁמָהּ֙ אָבֵ֣ל מִצְרַ֔יִם אֲשֶׁ֖ר בְּעֵ֥בֶר הַיַּרְדֵּֽן:  ת  וַחֲזָא יָתֵב אַרְעָא כְנַעֲנָאָה יָת אֶבְלָא בְּבֵית אִדְרֵי דְאָטָד וַאֲמָרוּ אָבֵל תַּקִּיף דֵּין לְמִצְרָאֵי עַל כֵּן קְרָא שְׁמַהּ אָבֵל מִצְרַיִם דִּי בְּעִבְרָא דְיַרְדְנָא:

{יב} וַיַּֽעֲשׂ֥וּ בָנָ֖יו ל֑וֹ כֵּ֖ן כַּֽאֲשֶׁ֥ר צִוָּֽם:  מ  וַיַּֽעֲשׂ֥וּ בָנָ֖יו ל֑וֹ כֵּ֖ן כַּֽאֲשֶׁ֥ר צִוָּֽם:  ת  וַעֲבָדוּ בְּנוֹהִי לֵהּ כֵּן כְּמָא דִי פַּקֵדִנּוּן:

{יג} וַיִּשְׂא֨וּ אֹת֤וֹ בָנָיו֙ אַ֣רְצָה כְּנַ֔עַן וַיִּקְבְּר֣וּ אֹת֔וֹ בִּמְעָרַ֖ת שְׂדֵ֣ה הַמַּכְפֵּלָ֑ה אֲשֶׁ֣ר קָנָה֩ אַבְרָהָ֨ם אֶת-הַשָּׂדֶ֜ה לַֽאֲחֻזַּת-קֶ֗בֶר מֵאֵ֛ת עֶפְרֹ֥ן הַֽחִתִּ֖י עַל-פְּנֵ֥י מַמְרֵֽא:  מ  וַיִּשְׂא֨וּ אֹת֤וֹ בָנָיו֙ אַ֣רְצָה כְּנַ֔עַן וַיִּקְבְּר֣וּ אֹת֔וֹ בִּמְעָרַ֖ת שְׂדֵ֣ה הַמַּכְפֵּלָ֑ה אֲשֶׁ֣ר קָנָה֩ אַבְרָהָ֨ם אֶת-הַשָּׂדֶ֜ה לַֽאֲחֻזַּת-קֶ֗בֶר מֵאֵ֛ת עֶפְרֹ֥ן הַֽחִתִּ֖י עַל-פְּנֵ֥י מַמְרֵֽא:  ת  וּנְטָלוּ יָתֵהּ בְּנוֹהִי לְאַרְעָא דִכְנַעַן וּקְבָרוּ יָתֵהּ בִּמְעָרַת חֲקַל כָּפֶלְתָּא דִי זְבַן אַבְרָהָם יָת חַקְלָא לְאַחֲסָנַת קְבוּרָא מִן עֶפְרֹן חִתָּאָה עַל אַפֵּי מַמְרֵא:

{יד} וַיָּ֨שָׁב יוֹסֵ֤ף מִצְרַ֨יְמָה֙ הוּא וְאֶחָ֔יו וְכָל-הָֽעֹלִ֥ים אִתּ֖וֹ לִקְבֹּ֣ר אֶת-אָבִ֑יו אַֽחֲרֵ֖י קָבְר֥וֹ אֶת-אָבִֽיו:  מ  וַיָּ֨שָׁב יוֹסֵ֤ף מִצְרַ֨יְמָה֙ הוּא וְאֶחָ֔יו וְכָל-הָֽעֹלִ֥ים אִתּ֖וֹ לִקְבֹּ֣ר אֶת-אָבִ֑יו אַֽחֲרֵ֖י קָבְר֥וֹ אֶת-אָבִֽיו:  ת  וְתַב יוֹסֵף לְמִצְרַיִם הוּא וַאֲחוֹהִי וְכָל דִּסְלִיקוּ עִמֵּהּ לְמִקְבַּר יָת אָבוּהִי בָּתַר דִּקְבַר יָת אָבוּהִי:

{טו} וַיִּרְא֤וּ אֲחֵֽי-יוֹסֵף֙ כִּי-מֵ֣ת אֲבִיהֶ֔ם וַיֹּ֣אמְר֔וּ ל֥וּ יִשְׂטְמֵ֖נוּ יוֹסֵ֑ף וְהָשֵׁ֤ב יָשִׁיב֙ לָ֔נוּ אֵ֚ת כָּל-הָ֣רָעָ֔ה אֲשֶׁ֥ר גָּמַ֖לְנוּ אֹתֽוֹ:  מ  וַיִּרְא֤וּ אֲחֵֽי-יוֹסֵף֙ כִּי-מֵ֣ת אֲבִיהֶ֔ם וַיֹּ֣אמְר֔וּ ל֥וּ יִשְׂטְמֵ֖נוּ יוֹסֵ֑ף וְהָשֵׁ֤ב יָשִׁיב֙ לָ֔נוּ אֵ֚ת כָּל-הָ֣רָעָ֔ה אֲשֶׁ֥ר גָּמַ֖לְנוּ אֹתֽוֹ:  ת  וַחֲזוֹ אֲחֵי יוֹסֵף אֲרֵי מִית אֲבוּהוֹן וַאֲמָרוּ דִּלְמָא יִטַּר לָנָא דְבָבוּ יוֹסֵף וְאֲתָבָא יָתֵיב לָנָא יָת כָּל בִּישְׁתָא דִי גְמַלְנָא יָתֵהּ:

{טז} וַיְצַוּ֕וּ אֶל-יוֹסֵ֖ף לֵאמֹ֑ר אָבִ֣יךָ צִוָּ֔ה לִפְנֵ֥י מוֹת֖וֹ לֵאמֹֽר:  מ  וַיְצַוּ֕וּ אֶל-יוֹסֵ֖ף לֵאמֹ֑ר אָבִ֣יךָ צִוָּ֔ה לִפְנֵ֥י מוֹת֖וֹ לֵאמֹֽר:  ת  וּפַקִידוּ לְוַת יוֹסֵף לְמֵימָר אָבוּךְ פַּקֵּד קֳדָם מוֹתֵהּ לְמֵימָר:

{יז} כֹּה-תֹֽאמְר֣וּ לְיוֹסֵ֗ף אָ֣נָּ֡א שָׂ֣א נָ֠א פֶּ֣שַׁע אַחֶ֤יךָ וְחַטָּאתָם֙ כִּֽי-רָעָ֣ה גְמָל֔וּךָ וְעַתָּה֙ שָׂ֣א נָ֔א לְפֶ֥שַׁע עַבְדֵ֖י אֱלֹהֵ֣י אָבִ֑יךָ וַיֵּ֥בְךְּ יוֹסֵ֖ף בְּדַבְּרָ֥ם אֵלָֽיו:  מ  כֹּה-תֹֽאמְר֣וּ לְיוֹסֵ֗ף אָ֣נָּ֡א שָׂ֣א נָ֠א פֶּ֣שַׁע אַחֶ֤יךָ וְחַטָּאתָם֙ כִּֽי-רָעָ֣ה גְמָל֔וּךָ וְעַתָּה֙ שָׂ֣א נָ֔א לְפֶ֥שַׁע עַבְדֵ֖י אֱלֹהֵ֣י אָבִ֑יךָ וַיֵּ֥בְךְּ יוֹסֵ֖ף בְּדַבְּרָ֥ם אֵלָֽיו:  ת  כְּדֵין תֵּימְרוּן לְיוֹסֵף בְּבָעוּ שְׁבוֹק כְּעַן לְחוֹבָא אַחָיךְ וְלַחֲטָאֵהוֹן אֲרֵי בִישְׁתָא גְמָלוּךְ וּכְעַן שְׁבוֹק כְּעַן לְחוֹבָא עַבְדֵי אֱלָהָא דְאָבוּךְ וּבְכָא יוֹסֵף בְּמַלָלוּתְהוֹן עִמֵּהּ:

{יח} וַיֵּֽלְכוּ֙ גַּם-אֶחָ֔יו וַֽיִּפְּל֖וּ לְפָנָ֑יו וַיֹּ֣אמְר֔וּ הִנֶּ֥נּוּ לְךָ֖ לַֽעֲבָדִֽים:  מ  וַיֵּֽלְכוּ֙ גַּם-אֶחָ֔יו וַֽיִּפְּל֖וּ לְפָנָ֑יו וַיֹּ֣אמְר֔וּ הִנֶּ֥נּוּ לְךָ֖ לַֽעֲבָדִֽים:  ת  וַאֲזָלוּ אַף אֲחוֹהִי וּנְפָלוּ קֳדָמוֹהִי וַאֲמָרוּ הָא אֲנַחְנָא לָךְ לְעַבְדִין:

{יט} וַיֹּ֧אמֶר אֲלֵהֶ֛ם יוֹסֵ֖ף אַל-תִּירָ֑אוּ כִּ֛י הֲתַ֥חַת אֱלֹהִ֖ים אָֽנִי:  מ  וַיֹּ֧אמֶר אֲלֵהֶ֛ם יוֹסֵ֖ף אַל-תִּירָ֑אוּ כִּ֛י הֲתַ֥חַת אֱלֹהִ֖ים אָֽנִי:  ת  וַאֲמַר לְהוֹן יוֹסֵף לָא תִדַחֲלוּן אֲרֵי דַחֲלָא דַּיְיָ אֲנָא:

{כ} וְאַתֶּ֕ם חֲשַׁבְתֶּ֥ם עָלַ֖י רָעָ֑ה אֱלֹהִים֙ חֲשָׁבָ֣הּ לְטֹבָ֔ה לְמַ֗עַן עֲשׂ֛ה כַּיּ֥וֹם הַזֶּ֖ה לְהַֽחֲיֹ֥ת עַם-רָֽב:  מ  וְאַתֶּ֕ם חֲשַׁבְתֶּ֥ם עָלַ֖י רָעָ֑ה אֱלֹהִים֙ חֲשָׁבָ֣הּ לְטֹבָ֔ה לְמַ֗עַן עֲשׂ֛ה כַּיּ֥וֹם הַזֶּ֖ה לְהַֽחֲיֹ֥ת עַם-רָֽב:  ת  וְאַתּוּן חֲשַׁבְתּוּן עֲלַי בִּישָׁא מִן קֳדָם יְיָ אִתְחַשְׁבַת לְטָבָא בְּדִיל לְמֶעְבַּד כְּיוֹמָא הָדֵין לְקַיָּמָא עַם סַגִּי:

{כא}  שביעי  וְעַתָּה֙ אַל-תִּירָ֔אוּ אָֽנֹכִ֛י אֲכַלְכֵּ֥ל אֶתְכֶ֖ם וְאֶֽת-טַפְּכֶ֑ם וַיְנַחֵ֣ם אוֹתָ֔ם וַיְדַבֵּ֖ר עַל-לִבָּֽם:  מ  וְעַתָּה֙ אַל-תִּירָ֔אוּ אָֽנֹכִ֛י אֲכַלְכֵּ֥ל אֶתְכֶ֖ם וְאֶֽת-טַפְּכֶ֑ם וַיְנַחֵ֣ם אוֹתָ֔ם וַיְדַבֵּ֖ר עַל-לִבָּֽם:  ת  וּכְעַן לָא תִדַחֲלוּן אֲנָא אֵזוּן יָתְכוֹן וְיָת טַפְלְכוֹן וְנַחֵם יָתְהוֹן וּמַלֵּיל תַּנְחוּמִין עַל לִבְּהוֹן:

{כב} וַיֵּ֤שֶׁב יוֹסֵף֙ בְּמִצְרַ֔יִם ה֖וּא וּבֵ֣ית אָבִ֑יו וַֽיְחִ֣י יוֹסֵ֔ף מֵאָ֥ה וָעֶ֖שֶׂר שָׁנִֽים:  מ  וַיֵּ֤שֶׁב יוֹסֵף֙ בְּמִצְרַ֔יִם ה֖וּא וּבֵ֣ית אָבִ֑יו וַֽיְחִ֣י יוֹסֵ֔ף מֵאָ֥ה וָעֶ֖שֶׂר שָׁנִֽים:  ת  וִיתֵיב יוֹסֵף בְּמִצְרַיִם הוּא וּבֵית אָבוּהִי וַחֲיָא יוֹסֵף מְאָה וַעֲסַר שְׁנִין:

{כג}  מפטיר  וַיַּ֤רְא יוֹסֵף֙ לְאֶפְרַ֔יִם בְּנֵ֖י שִׁלֵּשִׁ֑ים גַּ֗ם בְּנֵ֤י מָכִיר֙ בֶּן-מְנַשֶּׁ֔ה יֻלְּד֖וּ עַל-בִּרְכֵּ֥י יוֹסֵֽף:  מ  וַיַּ֤רְא יוֹסֵף֙ לְאֶפְרַ֔יִם בְּנֵ֖י שִׁלֵּשִׁ֑ים גַּ֗ם בְּנֵ֤י מָכִיר֙ בֶּן-מְנַשֶּׁ֔ה יֻלְּד֖וּ עַל-בִּרְכֵּ֥י יוֹסֵֽף:  ת  וַחֲזָא יוֹסֵף לְאֶפְרַיִם בְּנִין תְּלִיתָאִין אַף בְּנֵי מָכִיר בַּר מְנַשֶּׁה אִתְיְלִידוּ וְרַבִּי יוֹסֵף:

{כד} וַיֹּ֤אמֶר יוֹסֵף֙ אֶל-אֶחָ֔יו אָֽנֹכִ֖י מֵ֑ת וֵֽאלֹהִ֞ים פָּקֹ֧ד יִפְקֹ֣ד אֶתְכֶ֗ם וְהֶֽעֱלָ֤ה אֶתְכֶם֙ מִן-הָאָ֣רֶץ הַזֹּ֔את אֶל-הָאָ֕רֶץ אֲשֶׁ֥ר נִשְׁבַּ֛ע לְאַבְרָהָ֥ם לְיִצְחָ֖ק וּֽלְיַֽעֲקֹֽב:  מ  וַיֹּ֤אמֶר יוֹסֵף֙ אֶל-אֶחָ֔יו אָֽנֹכִ֖י מֵ֑ת וֵֽאלֹהִ֞ים פָּקֹ֧ד יִפְקֹ֣ד אֶתְכֶ֗ם וְהֶֽעֱלָ֤ה אֶתְכֶם֙ מִן-הָאָ֣רֶץ הַזֹּ֔את אֶל-הָאָ֕רֶץ אֲשֶׁ֥ר נִשְׁבַּ֛ע לְאַבְרָהָ֥ם לְיִצְחָ֖ק וּֽלְיַֽעֲקֹֽב:  ת  וַאֲמַר יוֹסֵף לַאֲחוֹהִי אֲנָא מָאִית וַיְיָ מִדְכַּר דְּכִיר יָתְכוֹן וְיִסַק יָתְכוֹן מִן אַרְעָא הָדָא לְאַרְעָא דִי קַיִים לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב:

{כה} וַיַּשְׁבַּ֣ע יוֹסֵ֔ף אֶת-בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר פָּקֹ֨ד יִפְקֹ֤ד אֱלֹהִים֙ אֶתְכֶ֔ם וְהַֽעֲלִתֶ֥ם אֶת-עַצְמֹתַ֖י מִזֶּֽה:  מ  וַיַּשְׁבַּ֣ע יוֹסֵ֔ף אֶת-בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר פָּקֹ֨ד יִפְקֹ֤ד אֱלֹהִים֙ אֶתְכֶ֔ם וְהַֽעֲלִתֶ֥ם אֶת-עַצְמֹתַ֖י מִזֶּֽה:  ת  וְאוֹמֵי יוֹסֵף יָת בְּנֵי יִשְׂרָאֵל לְמֵימָר מִדְכַּר דְּכִיר יְיָ יָתְכוֹן וְתַסְקוּן יָת גַּרְמַי מִכָּא:

{כו} וַיָּ֣מָת יוֹסֵ֔ף בֶּן-מֵאָ֥ה וָעֶ֖שֶׂר שָׁנִ֑ים וַיַּֽחַנְט֣וּ אֹת֔וֹ וַיִּ֥ישֶׂם בָּֽאָר֖וֹן בְּמִצְרָֽיִם:  מ  וַיָּ֣מָת יוֹסֵ֔ף בֶּן-מֵאָ֥ה וָעֶ֖שֶׂר שָׁנִ֑ים וַיַּֽחַנְט֣וּ אֹת֔וֹ וַיִּ֥ישֶׂם בָּֽאָר֖וֹן בְּמִצְרָֽיִם:  ת  וּמִית יוֹסֵף בַּר מְאָה וַעֲסַר שְׁנִין וַחֲנָטוּ יָתֵהּ וְשָׂמוּהִי בָאֳרוֹנָא בְּמִצְרָיִם: (חזק)
