פרק א - משנה א

הָרוֹאֶה רְאִיָּה אַחַת שֶׁל זוֹב, בֵּית שַׁמַּאי אוֹמְרִים, כְּשׁוֹמֶרֶת יוֹם כְּנֶגֶד יוֹם. וּבֵית הִלֵּל אוֹמְרִים, כְּבַעַל קֶרִי. רָאָה אַחַת, וּבַשֵּׁנִי הִפְסִיק, וּבַשְּׁלִישִׁי רָאָה שְׁתַּיִם {ב} אוֹ אַחַת מְרֻבָּה כִשְׁתַּיִם, בֵּית שַׁמַּאי אוֹמְרִים, זָב גָּמוּר. וּבֵית הִלֵּל אוֹמְרִים, מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב, וְצָרִיךְ בִּיאַת מַיִם חַיִּים, וּפָטוּר מִן הַקָּרְבָּן. אָמַר רַבִּי אֱלִיעֶזֶר בֶּן יְהוּדָה, מוֹדִים בֵּית שַׁמַּאי בָּזֶה שֶׁאֵינוֹ זָב גָּמוּר. וְעַל מַה נֶּחְלְקוּ, עַל הָרוֹאֶה שְׁתַּיִם אוֹ אַחַת מְרֻבָּה כִשְׁתַּיִם, וּבַשֵּׁנִי הִפְסִיק, וּבַשְּׁלִישִׁי רָאָה אַחַת, בֵּית שַׁמַּאי אוֹמְרִים, זָב גָּמוּר. וּבֵית הִלֵּל אוֹמְרִים, מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב, וְצָרִיךְ בִּיאַת מַיִם חַיִּים, וּפָטוּר מִן הַקָּרְבָּן:

 ר"ע מברטנורה  הָרוֹאֶה רְאִיָּה אַחַת שֶׁל זוֹב. זוֹב בָּא מִבָּשָׂר הַמֵּת וְדוֹמֶה לְלֹבֶן בֵּיצָה הַמּוּזֶרֶת. וְשִׁכְבַת זֶרַע בָּא מִבָּשָׂר שֶׁאֵינוֹ מֵת וּקְשׁוּרָה כְלֹבֶן בֵּיצָה שֶׁאֵינָהּ מוּזֶרֶת: בֵּית שַׁמַּאי אוֹמְרִים כְּשׁוֹמֶרֶת יוֹם. דְּאַף עַל פִּי שֶׁאֵינָהּ זָבָה גְמוּרָה עַד שָׁלֹשׁ רְאִיּוֹת, מִכָּל מָקוֹם בִּרְאִיָּה אַחַת עוֹשָׂה מִשְׁכָּב וּמוֹשָׁב. כְּמוֹ כֵן זָב, אַף עַל פִּי שֶׁאֵינוֹ נַעֲשֶׂה זָב גָּמוּר עַד שָׁלֹשׁ רְאִיּוֹת, מִכָּל מָקוֹם בִּרְאִיָּה אַחַת עוֹשֶׂה מִשְׁכָּב וּמוֹשָׁב {א}: וּבֵית הִלֵּל אוֹמְרִים כְּבַעַל קֶרִי. וְאֵינוֹ עוֹשֶׂה מִשְׁכָּב וּמוֹשָׁב, וְאֵין הָרְאִיָּה מְטַמְּאָה בְמַשָּׂא, אֶלָּא בְמַגָּע: רָאָה אַחַת וּבַשֵּׁנִי הִפְסִיק וּבַשְּׁלִישִׁי רָאָה שְׁתַּיִם. דְּכִי רָאָה שְׁתֵּי רְאִיּוֹת בְּיוֹם אֶחָד אוֹ בִשְׁנֵי יָמִים רְצוּפִים, צָרִיךְ לִמְנוֹת שִׁבְעָה נְקִיִּים, וּפָטוּר מִן הַקָּרְבָּן. וְאִם רָאָה שָׁלֹשׁ רְאִיּוֹת בְּיוֹם אֶחָד אוֹ בִשְׁלֹשָׁה יָמִים רְצוּפִים, צָרִיךְ לִמְנוֹת שִׁבְעָה נְקִיִּים וְחַיָּב בְּקָרְבָּן. דְּתַנְיָא, מָנָה הַכָּתוּב שְׁתַּיִם וּקְרָאוֹ טָמֵא, דִּכְתִיב (וַיִּקְרָא טו) זָב מִבְּשָׂרוֹ זוֹבוֹ טָמֵא הוּא. שָׁלֹשׁ וּקְרָאוֹ טָמֵא, (שָׁם) וְזֹאת תִּהְיֶה טֻמְאָתוֹ בְּזוֹבוֹ רָר בְּשָׂרוֹ אֶת זוֹבוֹ אוֹ הֶחְתִּים בְּשָׂרוֹ מִזּוֹבוֹ טֻמְאָתוֹ הִוא, הָא כֵיצַד, שְׁתַּיִם לְטֻמְאָה וְשָׁלֹשׁ לְקָרְבָּן. אֲבָל הַזָּבָה אִם רָאֲתָה רְאִיּוֹת רַבּוֹת בְּיוֹם אֶחָד אוֹ בִשְׁנֵי יָמִים בְּלֹא עֵת נִדָּתָהּ, שׁוֹמֶרֶת יוֹם אֶחָד בִּנְקִיּוּת וּטְהוֹרָה לָעֶרֶב. וְאִם רָאֲתָה שָׁלֹשׁ רְאִיּוֹת בִּשְׁלֹשָׁה יָמִים רְצוּפִים, צְרִיכָה לִמְנוֹת שִׁבְעָה נְקִיִּים וְחַיֶּבֶת בְּקָרְבָּן. דְּגַבֵּי זָבָה כְּתִיב (שָׁם) וְאִשָּׁה כִּי יָזוּב זוֹב דָּמָהּ יָמִים רַבִּים בְּלֹא עֵת נִדָּתָהּ, יָמִים שְׁנַיִם, רַבִּים שְׁלֹשָׁה, תָּלָה הַכָּתוּב הַזָּב בִּרְאִיּוֹת, וְהַזָּבָה בְּיָמִים. וְהַשְׁתָּא קָאָמַר שֶׁאִם רָאָה הַזָּב שָׁלֹשׁ רְאִיּוֹת אֶלָּא שֶׁהִפְסִיק יוֹם אֶחָד בֵּין רְאִיָּה רִאשׁוֹנָה לַשְּׁתַּיִם הָאַחֲרוֹנוֹת: בֵּית שַׁמַּאי אוֹמְרִים זָב גָּמוּר. וְחַיָּב בְּקָרְבָּן כְּבַעַל שָׁלֹשׁ רְאִיּוֹת: וּבֵית הִלֵּל אוֹמְרִים. בִּטֵּל הַיּוֹם הַטָּהוֹר אֶת הָרְאִיָּה הָרִאשׁוֹנָה, וְאֵין כָּאן אֶלָּא שְׁתֵּי רְאִיּוֹת הָאַחֲרוֹנוֹת, וּפָטוּר מִן הַקָּרְבָּן: מוֹדִים בֵּית שַׁמַּאי בָּזֶה שֶׁאֵינוֹ זָב גָּמוּר. דְּבִטֵּל הַיּוֹם הַטָּהוֹר שֶׁהִפְסִיק, אֶת הָרְאִיָּה הָרִאשׁוֹנָה, כְּדַאֲמַרַן: עַל מַה נֶּחְלְקוּ עַל הָרוֹאֶה שְׁתַּיִם וְכוּ' בֵּית שַׁמַּאי אוֹמְרִים זָב גָּמוּר. דְּהוֹאִיל וְנִזְקָק לִסְפִירַת שִׁבְעָה כְּשֶׁרָאָה שְׁתַּיִם בַּתְּחִלָּה, בִּטְּלָה רְאִיָּה אֶת יוֹם הַטָּהוֹר, וְיֵשׁ בְּיָדוֹ שָׁלֹשׁ רְאִיּוֹת: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} וְקָשֶׁה, דְּלֹא פֵרֵשׁ לְבֵית שַׁמַּאי דִּין רְאִיָּה גוּפַהּ כְּמוֹ שֶׁפֵּרֵשׁ לְבֵית הִלֵּל דְּאֵינָהּ מְטַמְּאָה בְמַשָּׂא. וְהָרַמְבַּ"ם לֹא פֵרֵשׁ בִּדְבֵית הִלֵּל כְּלוּם. וְהָרַ"שׁ פֵּרֵשׁ, דִּלְבֵית שַׁמַּאי מַסִּיט אֶת הָרְאִיָּה, תּוֹלִין. וּפֵרֵשׁ נַמִּי דְּמִשְׁכָּב וּמוֹשָׁב תּוֹלִין, כִּדְתָנֵי בַתּוֹסֶפְתָּא. וּמִנַּהּ אָמְרִינַן בַּגְּמָרָא, דִּכְשׁוֹמֶרֶת יוֹם דְּבֵית שַׁמַּאי, הַיְנוּ כְּבוֹעֵל שׁוֹמֶרֶת יוֹם. וְהָרַ"ב עֵרֵב שְׁנֵי הַפֵּרוּשִׁים: {ב} שְׁתַּיִם. דְּאִלּוּ אַחַת, לְבֵית הִלֵּל אֵינוֹ מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב: (עיקר תוי"ט).

פרק א - משנה ב

הָרוֹאֶה קֶרִי בַּיּוֹם הַשְּׁלִישִׁי לִסְפִירַת זוֹבוֹ, בֵּית שַׁמַּאי אוֹמְרִים, סוֹתֵר שְׁנֵי יָמִים שֶׁלְּפָנָיו. וּבֵית הִלֵּל אוֹמְרִים, לֹא סָתַר אֶלָּא יוֹמוֹ. רַבִּי יִשְׁמָעֵאל אוֹמֵר, הָרוֹאֶה בַשֵּׁנִי, סוֹתֵר אֶת שֶׁלְּפָנָיו. רַבִּי עֲקִיבָא אוֹמֵר {ג}, אֶחָד הָרוֹאֶה בַשֵּׁנִי וְאֶחָד הָרוֹאֶה בַשְּׁלִישִׁי, שֶׁבֵּית שַׁמַּאי אוֹמְרִים, סָתַר שְׁנֵי יָמִים שֶׁלְּפָנָיו. וּבֵית הִלֵּל אוֹמְרִים, לֹא סָתַר אֶלָּא יוֹמוֹ. וּמוֹדִים בְּרוֹאֶה בָרְבִיעִי שֶׁלֹּא סָתַר אֶלָּא יוֹמוֹ, בְּרָאָה קֶרִי. אֲבָל אִם רָאָה זוֹב, אֲפִלּוּ יוֹם שְׁבִיעִי, סָתַר אֶת שֶׁלְּפָנָיו:

 ר"ע מברטנורה  בַּיּוֹם הַשְּׁלִישִׁי לִסְפִירַת זוֹבוֹ. כְּגוֹן בַּעַל שְׁתֵּי רְאִיּוֹת שֶׁצָּרִיךְ לִסְפֹּר שִׁבְעָה נְקִיִּים, וּמָנָה רִאשׁוֹן וְשֵׁנִי, וּבַשְּׁלִישִׁי רָאָה קֶרִי, סָתַר שְׁנֵי יָמִים שֶׁלְּפָנָיו, וְחוֹזֵר וּמוֹנֶה שִׁבְעָה כְּבַתְּחִלָּה: לֹא סָתַר אֶלָּא יוֹמוֹ. וּמוֹנֶה חֲמִשָּׁה לְהַשְׁלִים שִׁבְעָה, דִּשְׁנַיִם הָרִאשׁוֹנִים עָלוּ לוֹ: הָרוֹאֶה בַשֵּׁנִי. שֶׁמָּנָה אֶחָד, וּבַשֵּׁנִי רָאָה קֶרִי, סָתַר אֶת שֶׁלְּפָנָיו אֲפִלּוּ לְבֵית הִלֵּל. וְרַבִּי עֲקִיבָא סָבַר דִּבְהָא נַמִּי פְלִיגֵי. וְאֵין הֲלָכָה כְּרַבִּי יִשְׁמָעֵאל: וּמוֹדִים בֵּית שַׁמַּאי בְּרוֹאֶה קֶרִי בָרְבִיעִי שֶׁלֹּא סָתַר אֶלָּא יוֹמוֹ. וְשֶׁלְּפָנָיו עָלוּ לוֹ. דְּדַוְקָא כְשֶׁלֹּא נִפְסַק הַזִּיבוּת בִּשְׁלֹשָׁה יָמִים טְהוֹרִים כְּמִסְפַּר הַיָּמִים שֶׁעוֹשֶׂה זָב גָּמוּר, הוּא דְאָמְרֵי בֵית שַׁמַּאי דְּשִׁכְבַת זֶרַע סוֹתֶרֶת, אֲבָל לְאַחַר שֶׁעָבְרוּ שְׁלֹשָׁה יָמִים טְהוֹרִים, מוֹדִים בֵּית שַׁמַּאי שֶׁאֵין קֶרִי סוֹתֵר אֶלָּא יוֹמוֹ. וְהַאי דְּקֶרִי סוֹתֵר בְּזָב, נַפְקָא לָן מִקְּרָא דִּכְתִיב (וַיִּקְרָא טו) זֹאת תּוֹרַת הַזָּב וַאֲשֶׁר תֵּצֵא מִמֶּנּוּ שִׁכְבַת זֶרַע, מַה זּוֹב סוֹתֵר, אַף שִׁכְבַת זֶרַע סוֹתֵר. אִי מַה זּוֹב סוֹתֵר אֶת הַכֹּל אַף שִׁכְבַת זֶרַע תִּסְתֹּר אֶת הַכֹּל, תַּלְמוּד לוֹמַר לְטָמְאָה בָהּ, אֵין לָהּ אֶלָּא יוֹם אֶחָד {ד}. אֲבָל אִם רָאָה זוֹב, אֲפִלּוּ יוֹם שְׁבִיעִי, סוֹתֵר אֶת שֶׁלְּפָנָיו, דִּכְתִיב שִׁבְעַת יָמִים לְטָהֳרָתוֹ {ה}, עַד שֶׁיִּהְיוּ כָל שִׁבְעַת הַיָּמִים טְהוֹרִים מִן הַזִּיבוּת: (ר"ע מברטנורה).

 עיקר תוי"ט  {ג} רַבִּי עֲקִיבָא אוֹמֵר. הַיְנוּ תַּנָּא קַמָּא דִּסְבִירָא לֵיהּ נַמִּי הַךְ וּמוֹדִים, דְּהָא לֹא אָמַר אֶלָּא בַּשְּׁלִישִׁי. וְצָרִיךְ לוֹמַר דְּקָא מַשְׁמַע לָן דְּמַאן תַּנָּא קַמָּא, רַבִּי עֲקִיבָא: {ד} בַּגְּמָרָא, אֵין לְךָ בָּהּ אֶלָּא מַה שֶּׁאָמוּר בָּהּ. וּפֵרֵשׁ רַשִׁ"י, כְּטֻמְאָתָהּ, מַה טֻּמְאָתָהּ יוֹם אֶחָד, אַף סְתִירָתָהּ כֵּן: {ה} מַשְׁמַע טָהֳרָה אַחַת בְּלִי שׁוּם הַפְסָקָה. רַשִׁ"י: (עיקר תוי"ט).

פרק א - משנה ג

רָאָה אַחַת הַיּוֹם וּשְׁתַּיִם לְמָחָר, שְׁתַּיִם הַיּוֹם וְאַחַת לְמָחָר, שָׁלשׁ לִשְׁלשָׁה יָמִים אוֹ לִשְׁלשָׁה לֵילוֹת, הֲרֵי זֶה זָב גָּמוּר:

 ר"ע מברטנורה  הֲרֵי זֶה זָב גָּמוּר. וְחַיָּב בְּקָרְבָּן. דְּבִרְאִיּוֹת תָּלָה רַחֲמָנָא, דִּכְתִיב טֻמְאָתוֹ בְּזוֹבוֹ, טֻמְאָתוֹ תְּלוּיָה בְּזוֹבוֹ וְאֵינָהּ תְּלוּיָה בְּיוֹמֵי. וּמִיהוּ בְּהִפְסִיק יוֹם בֵּינְתַיִם קָאָמְרֵי בֵית הִלֵּל לְעֵיל דְּלֹא הָוֵי זָב גָּמוּר וְאֵין הָרְאִיּוֹת מִצְטָרְפוֹת, לֹא שְׁנָא לְעִנְיַן טֻמְאָה כְּגוֹן רָאָה שְׁתֵּי רְאִיּוֹת בִּלְבַד בְּהַפְסָקַת יוֹם בֵּינְתַיִם, לֹא שְׁנָא לְעִנְיַן קָרְבָּן כְּגוֹן רָאָה שָׁלֹשׁ רְאִיּוֹת: (ר"ע מברטנורה).

פרק א - משנה ד

רָאָה אַחַת וְהִפְסִיק כְּדֵי טְבִילָה וְסִפּוּג, וְאַחַר כָּךְ רָאָה שְׁתַּיִם אוֹ אַחַת מְרֻבָּה כִשְׁתַּיִם, אוֹ רָאָה שְׁתַּיִם אוֹ אַחַת מְרֻבָּה כִשְׁתַּיִם, וְהִפְסִיק כְּדֵי טְבִילָה וְסִפּוּג, וְאַחַר כָּךְ רָאָה אַחַת, הֲרֵי זֶה זָב גָּמוּר:

 ר"ע מברטנורה  כְּדֵי טְבִילָה וְסִפּוּג. כְּדֵי שֶׁיִּטְבֹּל וִיקַנַּח עַצְמוֹ אַחַר טְבִילָה. וּפָחוֹת מִשִּׁעוּר זֶה, לֹא הָוֵי הֶפְסֵק וְלָא חֲשִׁיבָא אֶלָּא רְאִיָּה אַחַת: אוֹ אַחַת מְרֻבָּה כִשְׁתַּיִם. שֶׁהָיָה בֵּין הַתְחָלַת הָרְאִיָּה {ו} לְסוֹף הָרְאִיָּה כְּדֵי טְבִילָה וְסִפּוּג, וְהָיָה רוֹאֶה תָּמִיד בְּלֹא הֶפְסֵק: (ר"ע מברטנורה).

 עיקר תוי"ט  {ו} הָרַמְבַּ"ם. וּמִשּׁוּם דְּלִרְאִיָּה אֵין שִׁעוּר וַאֲפִלּוּ כָל שֶׁהוּא, הִלְכָּךְ אֵין פּוֹחֲתִין שִׁעוּר הַהֶפְסֵק, דְּכָל שֶׁהוּא אֵינוֹ מִן הַשִּׁעוּר. וְאֶפְשָׁר גַּם כֵּן דְּעִנְיָנוֹ בֵּין מִשֶּׁהִתְחִיל עַד לְמַה שֶּׁהוּא בַסּוֹף, וּלְהַאי כָּל שֶׁהוּא קְרָאֻהוּ הַתְחָלָה וְסוֹף כוּ': (עיקר תוי"ט).

פרק א - משנה ה

רָאָה אַחַת מְרֻבָּה כְשָׁלשׁ, שֶׁהִיא כְמִין גַּד יוֹן לַשִּׁילוֹחַ, שֶׁהֵן כְּדֵי שְׁתֵּי טְבִילוֹת וְכִשְׁנֵי סִפּוּגִין {ז}, הֲרֵי זֶה זָב גָּמוּר. רָאָה אַחַת מְרֻבָּה כִשְׁתַּיִם, מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב, וְצָרִיךְ בִּיאַת מַיִם חַיִּים, וּפָטוּר מִן הַקָּרְבָּן. אָמַר רַבִּי יוֹסֵי, לֹא אָמְרוּ אַחַת מְרֻבָּה אֶלָּא אִם כֵּן יֶשׁ בָּהּ כְּדֵי שָׁלשׁ {ח}:

 ר"ע מברטנורה  כְּמִין גַּד יוֹן לַשִּׁילוֹחַ. מָקוֹם בִּירוּשָׁלַיִם שֶׁהֶעֱמִידוּ בוֹ מַלְכֵי יְוָנִים עֲבוֹדָה זָרָה, וְהָיָה אוֹתוֹ מָקוֹם רָחוֹק מִן הַשִּׁילוֹחַ שִׁעוּר שְׁתֵּי טְבִילוֹת וּשְׁנֵי סִפּוּגִים. כְּמִגַּד יוֹן, לְשׁוֹן הָעוֹרְכִים לַגַּד שֻׁלְחָן (יְשַׁעְיָה סה). שִׁלּוֹחַ, שֵׁם הַנָּהָר, כְּדִכְתִיב (שָׁם ח) יַעַן כִּי מָאַס הָעָם הַזֶּה אֶת מֵי הַשִּׁלֹחַ: מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב. דְּחָשְׁבִינָא לַהּ כִּשְׁתֵּי רְאִיּוֹת: אֶלָּא אִם כֵּן יֶשׁ בָּהּ כְּדֵי שָׁלֹשׁ. כְּגוֹן כְּמִגַּד יוֹן לַשִּׁילוֹחַ, דְּאָז נֶחֱשָׁב כְּזָב גָּמוּר אֲפִלּוּ לְעִנְיַן קָרְבָּן. אֲבָל בְּאַחַת מְרֻבָּה כִשְׁתַּיִם, אֵין חוֹלְקִין אוֹתָהּ, וְכִרְאִיָּה אַחַת הִיא חֲשׁוּבָה. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: (ר"ע מברטנורה).

 עיקר תוי"ט  {ז} כְּדֵי שְׁתֵּי כוּ'. וְהוּא שֶׁיִּרְאֶה הָאַחַת בִּתְחִלַּת הַשִּׁעוּר וְאַחַת בְּאֶמְצָעָהּ וְהָאַחַת בְּסוֹפָהּ. הָרַמְבַּ"ם: {ח} אָמַר רַבִּי יוֹסֵי כוּ'. בַּסִּפְרָא פְלִיגֵי בִּדְרָשֵׁי דִקְרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק א - משנה ו

רָאָה אַחַת הַיּוֹם וְאַחַת בֵּין הַשְּׁמָשׁוֹת, אַחַת בֵּין הַשְּׁמָשׁוֹת וְאַחַת לְמָחָר, אִם יָדוּעַ שֶׁמִּקְצָת הָרְאִיָּה מֵהַיּוֹם וּמִקְצָתָהּ לְמָחָר, וַדַּאי לַקָּרְבָּן וְלַטֻּמְאָה. אִם סָפֵק שֶׁמִּקְצָת הָרְאִיָּה מֵהַיּוֹם וּמִקְצָתָהּ לְמָחָר, וַדַּאי לַטֻּמְאָה וְסָפֵק לַקָּרְבָּן. רָאָה שְׁנֵי יָמִים בֵּין הַשְּׁמָשׁוֹת, סָפֵק לַטֻּמְאָה וְלַקָּרְבָּן. אַחַת בֵּין הַשְּׁמָשׁוֹת, סָפֵק לַטֻּמְאָה:

 ר"ע מברטנורה  הַכֹּל מִטַּמְּאִים בְּזִיבָה. הַכֹּל, לַאֲתוֹיֵי תִּינוֹק בֶּן יוֹמוֹ {א} שֶׁהוּא מִטַּמֵּא בְּזִיבָה: אַף הַגֵּרִים. דִּכְתִיב בְּפָרָשַׁת זָב (וַיִּקְרָא טו) דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל, אֵין לִי אֶלָּא בְּנֵי יִשְׂרָאֵל, מִנַּיִן לְרַבּוֹת גֵּרִים וַעֲבָדִים, תַּלְמוּד לוֹמַר וְאָמַרְתָּ אֲלֵיהֶם אִישׁ, לְרַבּוֹת אֶת הַגֵּרִים וְאֶת הָעֲבָדִים. מִנַּיִן לְרַבּוֹת אֶת הַקְּטַנִּים, תַּלְמוּד לוֹמַר אִישׁ אִישׁ: וּמִטַּמְּאִין בְּדָם כָּאִשָּׁה. שֶׁאֵין הָאִשָּׁה מִטַּמְּאָה בְּלֹבֶן אֶלָּא בְאֹדֶם, דְּדָם כְּתִיב בָּהּ, שֶׁהוּא אָדֹם: וּבְלֹבֶן כָּאִישׁ. דְּזוֹבוֹ כְּתִיב גַּבֵּי אִישׁ {ב}. וְהוּא דוֹמֶה לְמֵי בָצֵק שֶׁל שְׂעֹרִים: וְטֻמְאָתָן בְּסָפֵק. אִם רָאוּ לֹבֶן, שֶׁמָּא הֵן אִשָּׁה, וְלֹבֶן טָהוֹר בָּהּ. וְאִם רָאוּ אֹדֶם, שֶׁמָּא הֵן אִישׁ, וְאֹדֶם טָהוֹר גַּבֵּיהּ. אֲבָל אִם רָאָה לֹבֶן וְאֹדֶם כְּאֶחָד, טֻמְאָתָן וַדַּאי מִמַּה נַּפְשָׁךְ, וְשׂוֹרְפִין עֲלֵיהֶם אֶת הַתְּרוּמָה, אֲבָל אֵין חַיָּבִים עֲלֵיהֶם עַל בִּיאַת מִקְדָּשׁ, דְּגַבֵּי שִׁלּוּחַ טְמֵאִים כְּתִיב (בְּמִדְבַּר ה) מִזָּכָר עַד נְקֵבָה תְּשַׁלֵּחוּ, זָכָר וַדַּאי, נְקֵבָה וַדָּאִית {ג}, וְלֹא טֻמְטוּם וְאַנְדְּרוֹגִינוֹס {ד}: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} מֵרִבּוּי דְאִישׁ, דִּמְרַבִּינַן קְטַנִּים, כִּדְאִיתָא בַגְּמָרָא. וְנִרְאֶה לִי דַּהֲוֵי כְעֵין מִעוּט אַחַר מִעוּט שֶׁהוּא לְרַבּוֹת. דְּהָא אִישׁ מַשְׁמָעוֹ בֶּן שְׁלֹשׁ עֶשְׂרֵה, וּמְמַעֵט הַפָּחוֹת מִכָּן: {ב} כְּלוֹמַר מַה שֶּׁהוּא רָגִיל שֶׁיָּזוּב מִמֶּנּוּ, וְהוּא לֹבֶן וְדוֹמֶה לְמֵי בָצֵק, וְגַם כְּלֹבֶן בֵּיצָה, דְּכֻלַּהּ חֲדָא מִלְּתָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} אִי הָכִי, כִּי אִטַּמּוּ בִּשְׁאָר טֻמְאוֹת [בְּמֵת אוֹ בְשֶׁרֶץ] לֹא לִשַּׁלְּחוּ. אָמַר קְרָא מִזָּכָר, מִטֻּמְאָה הַפּוֹרֶשֶׁת מִן הַזָּכָר. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} וְאַף עַל גַּב דְּאַנְדְּרוֹגִינוֹס קָמֵי שְׁמַיָּא גַּלְיָא אִם זָכָר הוּא אִם נְקֵבָה, יֵשׁ לוֹמַר דִּמְמַעֵט לֵיהּ מִטַּעַם שֶׁהוּא מְשֻׁנֶּה. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

 ר"ע מברטנורה  בְּשִׁבְעָה דְרָכִים בּוֹדְקִין אֶת הַזָּב. מִשּׁוּם דְּאֵינוֹ מְטַמֵּא מֵחֲמַת אֹנֶס, דִּכְתִיב זָב מִבְּשָׂרוֹ, וְלֹא מֵחֲמַת אָנְסוֹ: עַד שֶׁלֹּא נִזְקַק לְטֻמְאָה. דְּהַיְנוּ כְּשֶׁרוֹאֶה רְאִיָּה שְׁנִיָּה שֶׁבָּהּ הוּא נַעֲשֶׂה זָב לְטַמֵּא מִשְׁכָּב וּמוֹשָׁב וּלְהַצְרִיכוֹ לִסְפֹּר שִׁבְעָה נְקִיִּים. אֲבָל רְאִיָּה רִאשׁוֹנָה, מְטַמְּאָה בְאֹנֶס טֻמְאַת עֶרֶב, כְּדִין בַּעַל קֶרִי, וּמִצְטָרֶפֶת עִם הַשְּׁנִיָּה וַאֲפִלּוּ שֶׁהָיְתָה בְאֹנֶס: בְּמַאֲכָל. אִם אָכַל אֲכִילָה גַּסָּה אוֹ אָכַל דְּבָרִים הַמְבִיאִין לִידֵי זִיבָה, וַאֲפִלּוּ כָל שֶׁהֵן, כְּגוֹן בָּשָׂר שָׁמֵן, חָלָב, וּגְבִינָה, בֵּיצִים, וְיַיִן יָשָׁן, וּגְרִיסִין שֶׁל פּוֹל, וְשַׁחֲלִין: וּבְמִשְׁתֶּה. בְּרִבּוּי שְׁתִיָּה: בְּמַשָּׂא. כָּבֵד שֶׁנָּשָׂא: בְּמַרְאֶה. רָאָה אִשָּׁה, אֲפִלּוּ לֹא הִרְהֵר: וּבְהִרְהוּר. אַף עַל פִּי שֶׁלֹּא רָאָה. אִי אֵרַע לוֹ אֶחָד מֵאֵלּוּ הַשִּׁבְעָה קֹדֶם שֶׁרָאָה רְאִיָּה שְׁנִיָּה, אֵינוֹ נַעֲשֶׂה זָב מֵחֲמַת אוֹתָהּ רְאִיָּה, וְהַטִּפָּה אֵינָהּ מְטַמְּאָה בְמַשָּׂא כִּשְׁאָר טִפָּה שֶׁל זוֹב: אֵין אַחֲרָיוּת זָבִין עֲלֵיכֶם. וְאִם לֹא יִהְיוּ זָבִים לְעוֹלָם, מָה אַתֶּם חוֹשְׁשִׁים, הֲלֹא לֹא קִבַּלְתֶּם אַחֲרָיוּת שֶׁל זָבִים עֲלֵיכֶם. וְאֵין הֲלָכָה לֹא כְרַבִּי יְהוּדָה וְלֹא כְרַבִּי עֲקִיבָא: מִשֶּׁנִּזְקַק לְטֻמְאָה. לְאַחַר שֶׁרָאָה רְאִיָּה שְׁנִיָּה שֶׁלֹּא בְאֹנֶס, שֶׁהֲרֵי מִכָּאן וָאֵילָךְ הוּא עוֹשֶׂה מִשְׁכָּב וּמוֹשָׁב: אֵין בּוֹדְקִין אוֹתוֹ. וְאַף אִם רָאָה רְאִיָּה שְׁלִישִׁית בְּאֹנֶס, נַעֲשָׂה זָב לַקָּרְבָּן. וְכֵן אִם רָאָה רְאִיָּה שֶׁל זוֹב בְּשִׁבְעָה יָמִים נְקִיִּים שֶׁהוּא מוֹנֶה, סָתַר הַכֹּל וַאֲפִלּוּ הָיְתָה אוֹתָהּ רְאִיָּה בְּאֹנֶס: סְפֵקוֹ וְשִׁכְבַת זַרְעוֹ טְמֵאִין. סְפֵקוֹ מֵחֲמַת שִׁכְבַת זַרְעוֹ, כְּגוֹן אִם רָאָה שִׁכְבַת זֶרַע תְּחִלָּה, וְטִפָּה זִיבָה הַבָּאָה אַחַר כָּךְ אֵינָהּ מְטַמְּאָה, שֶׁהָרוֹאֶה קֶרִי אֵינוֹ מְטַמֵּא בְזִיבָה מֵעֵת לְעֵת. וְעַד שֶׁלֹּא נִזְקַק לְטֻמְאָה, הַקֶּרִי מְטַהֵר אֶת הַזִּיבָה, מִשּׁוּם דַּהֲוֵי אֹנֶס. וּלְאַחַר שֶׁנִּזְקַק לְטֻמְאָה, אֵין הַקֶּרִי מְטַהֵר אֶת הַזִּיבָה, שֶׁאֵין תּוֹלִין שֶׁמֵּחֲמַת הַקֶּרִי בָּאָה הַזִּיבָה: שֶׁרַגְלַיִם לַדָּבָר. שֶׁאֵין אוֹתָהּ רְאִיָּה מֵחֲמַת אֹנֶס, כֵּיוָן שֶׁכְּבָר נַעֲשָׂה זָב: רָאָה רְאִיָּה רִאשׁוֹנָה בּוֹדְקִין אוֹתוֹ. הַיְנוּ דַּוְקָא לַקָּרְבָּן. דְּאִי מֵחֲמַת אֹנֶס רָאָה, אֵינָהּ מִצְטָרֶפֶת עִם הַשְּׁלִישִׁית לְקָרְבָּן. אֲבָל מִצְטָרֶפֶת עִם הַשְּׁנִיָּה לַטֻּמְאָה. דְּאִם רְאִיָּה שְׁנִיָּה שֶׁלֹּא בְאֹנֶס, אַף עַל פִּי שֶׁהָיְתָה רִאשׁוֹנָה בְּאֹנֶס, עוֹשָׂה מִשְׁכָּב וּמוֹשָׁב וּבָעֵי שִׁבְעָה נְקִיִּים וּבִיאַת מַיִם חַיִּים, דְּכִי כְתִיב מִבְּשָׂרוֹ וְלֹא מֵחֲמַת אָנְסוֹ, בַּשְּׁנִיָּה כְּתִיב {ה}. אֲבָל רִאשׁוֹנָה אִתַּקַּשׁ לְשִׁכְבַת זֶרַע, דִּכְתִיב זֹאת תּוֹרַת הַזָּב וַאֲשֶׁר תֵּצֵא מִמֶּנּוּ שִׁכְבַת זֶרַע, מַה שִּׁכְבַת זֶרַע מְטַמְּאָה בְאֹנֶס {ו}, אַף רְאִיָּה רִאשׁוֹנָה שֶׁל זוֹב מְטַמְּאָה בְאֹנֶס: בַּשְּׁנִיָּה בּוֹדְקִין אוֹתוֹ. לַטֻּמְאָה וְלַקָּרְבָּן: בַּשְּׁלִישִׁית אֵין בּוֹדְקִין אוֹתוֹ. לֹא לַטֻּמְאָה וְלֹא לַקָּרְבָּן: רַבִּי אֱלִיעֶזֶר אוֹמֵר כוּ'. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: (ר"ע מברטנורה).

 עיקר תוי"ט  {ה} כְּלוֹמַר, עַל כָּרְחָךְ אַשְּׁנִיָּה דִּכְתִיב בָּתְרֵיהּ קָאֵי, וְלֹא אַרִאשׁוֹנָה, דְּהָא אִתַּקַּשׁ לְשִׁכְבַת זֶרַע כוּ': {ו} דְּכָל עַצְמוֹ שֶׁל שִׁכְבַת זֶרַע מֵחֲמַת אֹנֶס חִמּוּם בָּא. רַשִׁ"י. וְנִרְאֶה לִי, דְּמִשּׁוּם דְּמִשִּׁכְבַת זֶרַע יָלְפִינַן, דַּיּוֹ לְמֵילַף לְעִנְיַן טֻמְאָה, אֲבָל לְקָרְבָּן דְּלֹא שַׁיָּךְ בְּשִׁכְבַת זֶרַע, לֵית לָן לְמֵילַף מִנֵּיהּ: {ז} רַבִּי אֱלִיעֶזֶר אוֹמֵר. בַּגְּמָרָא פְלִיגֵי בִּדְרָשֵׁי דִקְרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ב - משנה ג

הָרוֹאֶה קֶרִי, אֵינוֹ מִטַּמֵּא בְזִיבָה מֵעֵת לְעֵת. רַבִּי יוֹסֵי אוֹמֵר {ט}, יוֹמוֹ. נָכְרִי שֶׁרָאָה קֶרִי וְנִתְגַּיֵּר, מִיָּד הוּא מִטַּמֵּא בְזִיבָה. הָרוֹאָה דָם וְהַמַּקְשָׁה, מֵעֵת לְעֵת. וְהַמַּכֶּה אֶת עַבְדּוֹ יוֹם יוֹמַיִם, מֵעֵת לְעֵת. כֶּלֶב שֶׁאָכַל בְּשַׂר הַמֵּת, שְׁלשָׁה יָמִים מֵעֵת לְעֵת, הֲרֵי הוּא כִבְרִיָּתוֹ:

 ר"ע מברטנורה  אֵינוֹ מִטַּמֵּא בְזִיבָה מֵעֵת לְעֵת. דְּהַאי זִיבוּת מֵחֲמַת חוּלְשָׁא דְקֶרִי הוּא. וּבִרְאִיָּה שְׁנִיָּה אַיְרֵי, דְּרִאשׁוֹנָה מְטַמְּאָה בְאֹנֶס. וּכְשֵׁם שֶׁתּוֹלִין לְקֶרִי מֵעֵת לְעֵת, כָּךְ לְמַרְאֶה וּלְהִרְהוּר תּוֹלִין מֵעֵת לְעֵת. אֲבָל לְמַאֲכָל וּלְמִשְׁתֶּה לְמַשָּׂא וְלִקְפִיצָה {ח}, תּוֹלִין לוֹ כָּל זְמַן שֶׁהוּא מִצְטַעֵר, וְאָמְרִינַן דְּמֵחֲמַת הַצַּעַר הָוֵי הַזִּיבָה וְלֹא מִבְּשָׂרוֹ: רַבִּי יוֹסֵי אוֹמֵר יוֹמוֹ. אֵין תּוֹלִין אֶלָּא אוֹתוֹ הַיּוֹם שֶׁרָאָה קֶרִי בִּלְבַד, וְלֹא מֵעֵת לְעֵת. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: מִיָּד מִטַּמֵּא בְזִיבָה. דְּגֵר שֶׁנִּתְגַּיֵּר כְּקָטָן שֶׁנּוֹלָד דָּמֵי. וְאַף עַל פִּי שֶׁרָאָה זִיבוּת כְּשֶׁהוּא נָכְרִי, מִטַּמֵּא בְזִיבָה בְהַךְ שְׁנִיָּה {י} שֶׁרוֹאֶה בְתוֹךְ מֵעֵת לְעֵת שֶׁל קֶרִי, דְּנַעֲשֵׂית לוֹ כִּרְאִיָּה רִאשׁוֹנָה דִּמְטַמֵּא בְאֹנֶס, וּמִצְטָרֶפֶת עִם הַשְּׁנִיָּה שֶׁיִּרְאֶה אַחַר מֵעֵת לְעֵת שֶׁל קֶרִי. דִּרְאִיַּת זִיבָה שֶׁל נָכְרִי כְּמַאן דְּלֵיתָא, דְּאֵין הַנָּכְרִים מִטַּמְּאִין בְּזִיבָה: הָרוֹאָה דָם. מְטַמְּאָה כָל טָהֳרוֹת שֶׁנִּתְעַסְּקָה בָהֶן לְמַפְרֵעַ מֵעֵת לְעֵת, כִּדְפָרִישְׁנָא בְרֵישׁ נִדָּה. וְאַיְדֵי דְאַיְרֵי בְמֵעֵת לְעֵת, קָחָשֵׁיב כָּל הָנֵי שֶׁשִּׁעוּרָן בְּמֵעֵת לְעֵת: וְכֵן הַמַּקְשָׁה. בִּימֵי זִיבָה. דְּאָמְרִינַן כִּי יָזוּב זוֹב דָּמָהּ, מֵחֲמַת עַצְמָהּ וְלֹא מֵחֲמַת וָלָד. וְאִם שָׁפְתָה מִן הַחֲבָלִין מֵעֵת לְעֵת וְיָלְדָה, לֹא תָלִינַן דַּם קֹשִׁי בְּדַם לֵידָה: הַמַּכֶּה אֶת עַבְדּוֹ. הָאָמוּר בַּתּוֹרָה (שְׁמוֹת כא) אַךְ אִם יוֹם אוֹ יוֹמַיִם יַעֲמֹד, וְדָרְשִׁינַן יוֹם שֶׁהוּא כְיוֹמַיִם, זֶהוּ מֵעֵת לְעֵת: כֶּלֶב שֶׁאָכַל בְּשַׂר הַמֵּת. דְּקַיְמָא לָן דְּאֵין מִתְעַכֵּל בְּמֵעָיו עַד לְאַחַר שְׁלֹשָׁה יָמִים מֵעֵת לְעֵת, דְּהָכִי תְנַן בְּמַסֶּכֶת אֳהָלוֹת פֶּרֶק י"א, וְכַמָּה תִשְׁהֶה בְּמֵעָיו, שְׁלֹשָׁה יָמִים מֵעֵת לְעֵת: וְהוּא שֶׁיְּהֵא כִבְרִיָּתוֹ. שֶׁלֹּא נִשְׁתַּנָּה {יא}. וְעוֹד אִיכָּא טוּבָא שֶׁשִּׁעוּרָן מֵעֵת לְעֵת וְלָא חֲשִׁיב, וְתָנָא וְשִׁיֵּר: (ר"ע מברטנורה).

מְגִיפִין אוֹ פוֹתְחִין. וַחֲכָמִים אוֹמְרִים, עַד שֶׁיְּהֵא זֶה מֵגִיף וְזֶה פוֹתֵחַ. מַעֲלִין זֶה אֶת זֶה מִן הַבּוֹר. רַבִּי יְהוּדָה אוֹמֵר, עַד שֶׁיְּהֵא הַטָּהוֹר מַעֲלֶה אֶת הַטָּמֵא. מַפְשִׁילִין בַּחֲבָלִין. וַחֲכָמִים אוֹמְרִים, עַד שֶׁיִּהְיֶה זֶה מוֹשֵׁךְ הֵילֵךְ וְזֶה מוֹשֵׁךְ הֵילֵךְ. אוֹרְגִים, בֵּין בְּעוֹמְדִין בֵּין בְּיוֹשְׁבִין אוֹ טוֹחֲנִין. רַבִּי שִׁמְעוֹן מְטַהֵר בְּכֻלָּן, חוּץ מִן הַטּוֹחֲנִין בְּרֵחַיִם שֶׁל יָד. פּוֹרְקִין מִן הַחֲמוֹר אוֹ טוֹעֲנִין, בִּזְמַן שֶׁמַּשָּׂאָם כָּבֵד, טְמֵאִין. בִּזְמַן שֶׁמַּשָׂאָם קַל, טְהוֹרִין. וְכֻלָּן טְהוֹרִין לִבְנֵי הַכְּנֶסֶת וּטְמֵאִין לַתְּרוּמָה:

 עיקר תוי"ט  {ט} וְטָמֵא, דְּאִי אֶפְשָׁר שֶׁלֹּא יִהְיוּ נִתְלִין זֶה עַל זֶה קְצָת. מַהֲרַ"ם: {י} וְהוֹאִיל וְאֵינוֹ אֶלָּא סָפֵק, אֵין שׂוֹרְפִין עָלָיו אֶת הַתְּרוּמָה. הָרַאֲבַ"ד: (עיקר תוי"ט).

הַזָּב וְהַטָּהוֹר שֶׁיָּשְׁבוּ בִסְפִינָה גְדוֹלָה, אֵיזוֹ הִיא סְפִינָה גְדוֹלָה. רַבִּי יְהוּדָה אוֹמֵר, כֹּל שֶׁאֵינָהּ יְכוֹלָה לְהָמִיט בָּאָדָם. יָשְׁבוּ עַל הַנֶּסֶר, עַל הַסַּפְסָל, עַל הַגָּשִׁישׁ שֶׁל מִטָּה וְעַל הָאַכְלוֹנָס בִּזְמַן שֶׁאֵינָן מַחְגִּירִין, עָלוּ בְאִילָן שֶׁכֹּחוֹ יָפֶה, בְּסוֹכָה שֶׁכֹּחָהּ יָפֶה, וּבְסֻלָּם צוֹרִי אוֹ מִצְרִי בִּזְמַן שֶׁהוּא קָבוּעַ בְּמַסְמֵר, עַל הַכֶּבֶשׁ, עַל הַקּוֹרָה וְעַל הַדֶּלֶת בִּזְמַן שֶׁהֵן עֲשׂוּיִין בְּטִיט אֲפִלּוּ מִצַּד אֶחָד {יא}, טְהוֹרִין. הַטָּהוֹר מַכֶּה אֶת הַטָּמֵא, טָהוֹר. הַטָּמֵא מַכֶּה אֶת הַטָּהוֹר, טָמֵא, שֶׁאִם יִמָּשֵׁךְ הַטָּהוֹר, הֲרֵי הַטָּמֵא נוֹפֵל:

 ר"ע מברטנורה  כֹּל שֶׁאֵינָהּ יְכוֹלָה לְהָמִיט בָּאָדָם. שֶׁהִיא גְדוֹלָה כָּל כָּךְ שֶׁכְּשֶׁעוֹלֶה בָהּ אָדָם אֵינָהּ מִתְמוֹטֶטֶת: הַטָּהוֹר מַכֶּה אֶת הַטָּמֵא. אֶת הַזָּב: הֲרֵי הַטָּמֵא נוֹפֵל. שֶׁכֵּן דֶּרֶךְ הַמַּכֶּה אֶת חֲבֵרוֹ, אִם הָיָה חֲבֵרוֹ נִמְשָׁךְ הָיָה הַמַּכֶּה נוֹפֵל, וּלְכָךְ נֶחְשָׁב זֶה הַמַּכֶּה כְּנִשְׁעָן עַל הַטָּהוֹר. וְדַוְקָא לְעִנְיַן טֻמְאַת מִדְרָס אַיְרֵי, דְּטָהוֹר הַמַּכֶּה אֶת הַטָּמֵא, טָהוֹר מִטֻּמְאַת מִדְרָס. אֲבָל טָמֵא הוּא, שֶׁהֲרֵי נִשְׁעָן עַל הַזָּב. וְטָמֵא הַמַּכֶּה אֶת הַטָּהוֹר, בְּגָדָיו טְמֵאִים מִדְרָס, דַּהֲוֵי כְאִלּוּ הַזָּב נִשְׁעָן עָלָיו: (ר"ע מברטנורה).

 עיקר תוי"ט  {יא} אֲפִלּוּ כוּ'. פֵּרוּשׁ, אֲפִלּוּ שֶׁעָלוּ מִצַּד אֶחָד. הָרַמְבַּ"ם: (עיקר תוי"ט).

פרק ד - משנה א

רַבִּי יְהוֹשֻׁעַ אוֹמֵר, נִדָּה שֶׁיָּשְׁבָה עִם הַטְּהוֹרָה בְּמִטָּה {א}, כִּפָּה שֶׁבְּרֹאשָׁהּ טָמֵא מִדְרָס. יָשְׁבָה בִסְפִינָה, כֵּלִים שֶׁבְּרֹאשׁ הַנֵּס שֶׁבַּסְּפִינָה טְמֵאִין מִדְרָס. נוֹטֶלֶת עֲרֵבָה מְלֵאָה בְגָדִים, בִּזְמַן שֶׁמַּשָּׂאָן כָּבֵד, טְמֵאִין {ב}. בִּזְמַן שֶׁמַּשָּׂאָן קַל, טְהוֹרִין {ג}. זָב שֶׁהִקִּישׁ עַל כְּצוֹצְרָא, וְנָפַל כִּכָּר שֶׁל תְּרוּמָה, טָהוֹר:

 ר"ע מברטנורה  רַבִּי יְהוֹשֻׁעַ אוֹמֵר כִּפָּה שֶׁבְּרֹאשָׁהּ. לִרְבוּתָא נְקָטָהּ, דַּאֲפִלּוּ צָעִיף שֶׁבְּרֹאשׁ הַטְּהוֹרָה נֶחְשָׁב כְּאִלּוּ יָשְׁבָה עָלָיו הַנִּדָּה, אַף עַל פִּי שֶׁאֵינָהּ יוֹשֶׁבֶת עָלָיו. וּפָלֵיג אַסְּתָם מַתְנִיתִין דְּפִרְקִין דִּלְעֵיל דְּלֹא מְטַמֵּא אֶלָּא בִגְדֵיהֶן וְלֹא כִפָּה שֶׁבְּרֹאשָׁן: כֵּלִים שֶׁבְּרֹאשׁ הַנֵּס. הָא נַמִּי לִרְבוּתָא נָקַט, דְּאַף עַל פִּי שֶׁהֵם בַּגֹּבַהּ בְּרֹאשׁ הַנֵּס שֶׁלֹּא בִמְקוֹם דְּרִיסָה, אֲפִלּוּ הָכִי טְמֵאִים מִדְרָס. וּבִסְפִינָה קְטַנָּה וּבְמִטָּה מַחְגֶּרֶת אַיְרֵי, כְּהַהִיא דְרֵישׁ פִּרְקִין דִּלְעֵיל, אֶלָּא דְּתַנָּא דִלְעֵיל לֹא מְטַמֵּא כֵלִים שֶׁבְּרֹאשׁ הַנֵּס. וְאֵין הֲלָכָה כְּרַבִּי יְהוֹשֻׁעַ: בִּזְמַן שֶׁמַּשָּׂאָן כָּבֵד טְמֵאִים מִדְרָס. וְאַף עַל פִּי שֶׁלֹּא נָגְעָה בַבְּגָדִים, דְּחַשְׁבִינָא לַהּ כְּאִלּוּ נִשְׁעֶנֶת עַל הַבְּגָדִים: כְּסוֹסְטְרָא. כְּמוֹ גְּזוּזְטְרָא, כְּמִין נֶסֶר הַבּוֹלֵט מִכֹּתֶל הָעֲלִיָּה לַחוּץ: וְנָפַל כִּכָּר שֶׁל תְּרוּמָה. שֶׁעַל הַכְּסוֹסְטְרָא: טָהוֹר. הַכִּכָּר. וְאַף עַל פִּי שֶׁנָּפַל מֵחֲמַת הֶקֵּשׁ שֶׁל זָב, לָא חֲשִׁיב לֵיהּ כְּהֶסֵּט, מִשּׁוּם דְּכֹחָהּ שֶׁל כְּסוֹסְטְרָא יָפֶה: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} נִדָּה כוּ'. וְהוּא הַדִּין זָב וְטָהוֹר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} טְמֵאִין. נִרְאֶה לִי דְּמַיְרֵי שֶׁנּוֹשֵׂאת הָעֲרֵבָה בְּחֵיקָהּ לְפָנֶיהָ, וּכְשֶׁמַּשָּׂאָן כָּבֵד מַכְרִיעוֹת אוֹתָהּ עַד שֶׁהִיא נִשְׁעֶנֶת וְנִתְלֵית עֲלֵיהֶם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} טְהוֹרִין. הַבְּגָדִים מִטֻּמְאַת מִדְרָס. וְאַף שֶׁהָעֲרֵבָה טְמֵאָה מַגַּע זָב, וְאָדָם הַנּוֹגֵעַ בְּזָב מְטַמֵּא בְגָדִים כָּל זְמַן שֶׁלֹּא פֵרַשׁ, כְּלִי הַנּוֹגֵעַ בְּזָב אֵינוֹ מְטַמֵּא בְגָדִים אֲפִלּוּ בְעוֹד שֶׁלֹּא פֵרַשׁ. הָרַ"שׁ: (עיקר תוי"ט).

פרק ד - משנה ב

הִקִּישׁ עַל הַמָּרִישׁ {ד}, עַל הַמַּלְבֵּן, עַל הַצִּנּוֹר, וְעַל הַדַּף, אַף עַל פִּי שֶׁהוּא עָשׂוּי בַּחֲבָלִים, עַל הַתַּנּוּר, וְעַל הַיָּם, וְעַל אִצְטְרֻבָּל, וְעַל חֲמוֹר שֶׁל רֵחַיִם שֶׁל יָד, וְעַל סְאָה שֶׁל רֵחַיִם שֶׁל זֵיתִים, רַבִּי יוֹסֵי אוֹמֵר, אַף עַל קוֹרַת הַבַּלָּנִין, טָהוֹר:

 ר"ע מברטנורה  מָרִישׁ. קוֹרָה גְדוֹלָה שֶׁבּוֹנִים בַּבִּירָה וְרָאשֵׁי קוֹרוֹת הַגַּג כֻּלָּן סְמוּכוֹת עָלֶיהָ, וְהִיא מֻנַּחַת עַל גַּבֵּי עַמּוּדִים: מַלְבֵּן. כְּמִין רִבּוּעַ שֶׁל קוֹרוֹת קְבוּעוֹת בָּאָרֶץ: צִנּוֹר. מַרְזֵב שֶׁל מַיִם: אַף עַל פִּי שֶׁהוּא עָשׂוּי בַּחֲבָלִים. שֶׁאֵינוֹ קָבוּעַ בָּאָרֶץ אֶלָּא קָשׁוּר בַּחֲבָלִים וְאֵין כֹּחוֹ יָפֶה כָּל כָּךְ: וְעַל הַיָּם. הֶקֵּף שֶׁל עֵץ שֶׁהַקֶּמַח מִתְקַבֵּץ בּוֹ {ה} בִּשְׁעַת הַטְּחִינָה: אִצְטְרֻבָּל. עִגּוּל שֶׁל עֵץ שֶׁמַּעֲמִידִין הָרֵחַיִם עָלָיו. וּשְׁנֵיהֶן קְבוּעִים הֵן: חֲמוֹר שֶׁל רֵחַיִם שֶׁל יָד. בִּנְיָן שֶׁל עֵצִים שֶׁהָרֵחַיִם שֶׁל יָד קְבוּעִים בָּהֶן: וּסְאָה שֶׁל רֵחַיִם. מִדָּה גְדוֹלָה וּקְבוּעָה בַקַּרְקַע: קוֹרַת הַבַּלָּנִין. קוֹרָה שֶׁבַּעַל הַמֶּרְחָץ יוֹשֵׁב עָלֶיהָ. וְכָל הָנֵי אִם הִקִּישׁ עֲלֵיהֶן הַזָּב וְנָפַל הַכִּכָּר שֶׁל תְּרוּמָה מִכֹּחַ הַהֶקֵּשׁ, טָהוֹר, דְּכֹחָן יָפֶה וְלָא חֲשִׁיב הֶסֵּט. וּבְקוֹרַת הַבַּלָּנִים נֶחְלְקוּ. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: (ר"ע מברטנורה).

 עיקר תוי"ט  {ד} הִקִּישׁ כוּ'. בַּתּוֹסֶפְתָּא, זֶה הַכְּלָל, כָּל שֶׁנָּפַל מִכֹּחַ הֶסֵּטוֹ [הַיְנוּ שֶׁכֹּחוֹ רַע], טָמֵא. מִכֹּחַ הָרְעָדָה [שֶׁכֹּחוֹ יָפֶה], טָהוֹר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} וְקֶלֶת דִּלְקַמָּן עָשׂוּי שֶׁלֹּא יִתְפַּזֵּר הַקֶּמַח וְשֶׁיִּפֹּל לְתוֹךְ הַיָּם. אֲבָל הָרַ"שׁ פֵּרֵשׁ, יָם שֶׁל בֵּית הַבַּד: (עיקר תוי"ט).

פרק ד - משנה ג

הִקִּישׁ עַל הַדֶּלֶת, עַל הַנֶּגֶר, עַל הַמַּנְעוּל, עַל הַמָּשׁוֹט, וְעַל הַקֶּלֶת, וְעַל אִילָן שֶׁכֹּחוֹ רַע, וְעַל סוֹכָה שֶׁכֹּחָהּ רַע, עַל אִילָן יָפֶה, עַל סֻלָּם מִצְרִי בִּזְמַן שֶׁאֵינוֹ קָבוּעַ בְּמַסְמֵר, עַל הַכֶּבֶשׁ, עַל הַקּוֹרָה וְעַל הַדֶּלֶת בִּזְמַן שֶׁאֵינָן עֲשׂוּיִין בְּטִיט, טְמֵאִין. עַל הַשִּׁדָּה, עַל הַתֵּבָה וְעַל הַמִּגְדָּל, טְמֵאִין. רַבִּי נְחֶמְיָה וְרַבִּי שִׁמְעוֹן מְטַהֲרִין בָּאֵלּוּ:

 ר"ע מברטנורה  הַנֶּגֶר. יָתֵד הַנָּעוּץ אֲחוֹרֵי הַדֶּלֶת בַּנֶּקֶב שֶׁבָּאַסְקֻפָּה: מַנְעוּל. שֶׁסּוֹגְרִים בּוֹ הַפֶּתַח: מָשׁוֹט. שֶׁמַּנְהִיגִין בּוֹ אֶת הַסְּפִינָה. וּלְשׁוֹן מִקְרָא הוּא, כָּל תּוֹפְשֵׂי מָשׁוֹט, בִּיחֶזְקֵאל [כז]: קֶלֶת. עִגּוּל שֶׁל עֵץ שֶׁסָּבִיב לָרֵחַיִם הַתַּחְתּוֹנָה לְקַבֵּל הַקֶּמַח הַנִּטְחָן שֶׁלֹּא יִפֹּל לָאָרֶץ, וּמִטַּלְטֵל הוּא: וְעַל סוֹכָה שֶׁכֹּחָהּ רַע. וְעוֹמֶדֶת בְּאִילָן שֶׁכֹּחוֹ יָפֶה: טְמֵאִין. דְּכָל הָנֵי אֵין כֹּחָן יָפֶה, וְכֹחַ הֶקֵּשׁ הַזָּב בְּהוּ חֲשִׁיב הֶסֵּט: מְטַהֲרִים בָּאֵלּוּ. אַשִּׁדָּה תֵּבָה וּמִגְדָּל קָאֵי. לְהָכִי תְנִינְהוּ תַנָּא קַמָּא לְחוֹדַיְהוּ. וְאֵין הֲלָכָה כְּרַבִּי נְחֶמְיָה וְרַבִּי שִׁמְעוֹן: (ר"ע מברטנורה).

פרק ד - משנה ד

זָב שֶׁהָיָה מֻטָּל עַל חֲמִשָּׁה סַפְסָלִים אוֹ עַל חָמֵשׁ פֻּנְדִּיּוֹת, לְאָרְכָּן, טְמֵאִין. לְרָחְבָּן, טְהוֹרִין. יָשֵׁן, סָפֵק שֶׁנִּתְהַפֵּךְ עֲלֵיהֶן, טְמֵאִין. הָיָה מֻטָּל עַל שִׁשָּׁה כִסְיוֹת, שְׁתֵּי יָדָיו עַל שְׁנַיִם, שְׁתֵּי רַגְלָיו עַל שְׁנַיִם, רֹאשׁוֹ עַל אֶחָד, גּוּפוֹ עַל אֶחָד, אֵין טָמֵא אֶלָּא זֶה שֶׁתַּחַת הַגּוּף. עוֹמֵד עַל שְׁנֵי כִסְאוֹת, רַבִּי שִׁמְעוֹן אוֹמֵר, אִם רְחוֹקִין זֶה מִזֶּה, טְהוֹרִין:

 ר"ע מברטנורה  פֻּנְדִּיּוֹת. כְּעֵין כִּיסִין אֲרֻכִּין {ו}: לְאָרְכָּן. שֶׁשּׁוֹכֵב לְאָרְכָּן שֶׁל סַפְסָלִים אוֹ שֶׁל פֻּנְדִּיּוֹת {ז}: טְמֵאִים. שֶׁכָּל אֶחָד וְאֶחָד נִשְׁעָן רֻבּוֹ שֶׁל זָב עֲלֵיהֶן, פְּעָמִים עַל זֶה פְּעָמִים עַל זֶה: לְרָחְבָּן. שֶׁשּׁוֹכֵב אָרְכּוֹ לְרָחְבָּן: טְהוֹרִים. דְּאֵין רֻבּוֹ נִשְׁעָן עַל אֶחָד מֵהֶן {ח}, וְאֵין [טָמֵא] מִשְׁכָּב וּמוֹשָׁב לְעוֹלָם, עַד שֶׁיִּשָּׁעֵן רֻבּוֹ עָלָיו, דִּכְתִיב אִם עַל הַמִּשְׁכָּב הוּא {ט}, עַד שֶׁיִּנָּשֵׂא רֻבּוֹ עָלָיו: יָשֵׁן. לְרָחְבָּן, וְסָפֵק אִם נִתְהַפֵּךְ לְאָרְכָּן, טְמֵאִין: כִּסְיוֹת. לְשׁוֹן רַבִּים שֶׁל כִּסֵּא {י}: שֶׁתַּחַת הַגּוּף. לְפִי שֶׁרֻבּוֹ נִשָּׂא עָלָיו. וְהָאֲחֵרִים טְמֵאִים מַגַּע הַזָּב. אֲבָל אֵין לָהֶם תּוֹרַת מִשְׁכָּב וּמוֹשָׁב לִהְיוֹת אַב הַטֻּמְאָה לְטַמֵּא אָדָם וְכֵלִים: אִם רְחוֹקִים זֶה מִזֶּה טְהוֹרִים. לְפִי שֶׁאֵין רֻבּוֹ נִשָּׂא {יא} לֹא עַל זֶה וְלֹא עַל זֶה. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: (ר"ע מברטנורה).

 עיקר תוי"ט  {ו} וְנוֹתְנִין בָּהֶם מָעוֹת וּרְאוּיִים לְשַׁמֵּשׁ שְׁכִיבָה עִם מְלַאכְתָּן, שֶׁיּוֹשֵׁב עֲלֵיהֶן וְהֵן מְלֵאוֹת מָעוֹת וּפוֹתֵחַ פִּיהֶם וְנוֹטֵל, וְאֵין צָרִיךְ לוֹמַר לוֹ עֲמֹד וְנַעֲשֶׂה מְלַאכְתֵּנוּ. רַשִׁ"י: {ז} כְּגוֹן שֶׁשּׁוֹכֵב עַל מֵעָיו אוֹ פְרַקְדָן וְאָרְכּוֹ מֻטָּל עַל אָרְכָּן. רַשִׁ"י: {ח} וּלְפִי זֶה אָתְיָא כְרַבִּי שִׁמְעוֹן דְּסוֹף פִּרְקִין. וְרַבִּי יוֹסֵי מְפָרֵשׁ דְּאָתְיָא אֲפִלּוּ כְרַבָּנָן. וְטַעֲמָא דִטְהוֹרִין, מִשּׁוּם דַּהֲווּ לְהוּ כָּל אֶחָד מְסַיֵּעַ וְאֵין בּוֹ מַמָּשׁ: {ט} וְלֹא דַק, דְּזֶה נֶאֱמַר עַל הַטָּהוֹר הַשּׁוֹכֵב עַל מִשְׁכַּב הַזָּב. אֲבָל בְּזָב הָעוֹשֶׂה מִשְׁכָּב נִדְרָשׁ מִדִּכְתִיב וְכָל הַמִּשְׁכָּב אֲשֶׁר יִשְׁכַּב וְגוֹ'. וְהָא דְדָרְשִׁינַן רֻבּוֹ, מִשּׁוּם דְּרֻבּוֹ כְּכֻלּוֹ: {י} וְקָשֶׁה קְצָת, דְּהָכָא תָנֵי בְּיוֹ"ד וּלְקַמָּן בְּאָלֶ"ף. וְנִרְאֶה לִי כְּפֵרוּשׁ הָרַמְבַּ"ם שֶׁכָּתַב כִּסְיוֹת כְּלֵי חֲגוֹרוֹת שֶׁל עוֹר נְאוֹתִים לְמִשְׁכָּב: {יא} אֶלָּא עַל שְׁתֵּיהֶן בְּשָׁוֶה, חֶצְיוֹ עַל כָּל אֶחָד. אֲבָל אִם הֵם קְרוֹבִים, זִמְנִין נִשְׁעָן רֻבּוֹ עַל אֶחָד. מַהֲרַ"ם: (עיקר תוי"ט).

פרק ד - משנה ה

עֶשֶׂר טַלִּיּוֹת זוֹ עַל גַּב זוֹ, יָשֵׁן עַל גַּבֵּי הָעֶלְיוֹנָה, כֻּלָּן טְמֵאוֹת. הַזָּב בְּכַף מֹאזְנַיִם וּמִשְׁכָּב וּמוֹשָׁב כְּנֶגְדּוֹ, כָּרַע הַזָּב, טְהוֹרִין. כָּרְעוּ הֵן, טְמֵאִין. רַבִּי שִׁמְעוֹן אוֹמֵר, בִּיחִידִי, טָמֵא. וּבִמְרֻבִּין, טָהוֹר, שֶׁאֵין אֶחָד נוֹשֵׂא אֶת רֻבּוֹ:

 ר"ע מברטנורה  עֶשֶׂר טַלִּיּוֹת. לְשׁוֹן רַבִּים שֶׁל טַלִּית. וְעֶשֶׂר לָאו דַּוְקָא, דַּאֲפִלּוּ אֶלֶף טַלִּיּוֹת וְאֶבֶן גְּדוֹלָה עַל גַּבָּן, וְיָשַׁב הַזָּב עַל הָאֶבֶן, כָּל הַטַּלִּיּוֹת שֶׁתַּחְתֶּיהָ מִטַּמְּאוֹת מוֹשָׁב, דְּמִשְׁכָּב וּמוֹשָׁב מִטַּמְּאִין תַּחַת אֶבֶן מְסָמָא: כָּרַע הַזָּב טְהוֹרִים. מִדִּין מִשְׁכָּב, אֲבָל טְמֵאִין בְּמַשָּׂא הַזָּב: בִּיחִידִי. מִשְׁכָּב אֶחָד: טָמֵא. שֶׁהֲרֵי רֹב הַזָּב נִשָּׂא עָלָיו: וּבִמְרֻבִּים. בְּמִשְׁכָּבוֹת רַבּוֹת בְּכַף מֹאזְנַיִם כְּנֶגֶד הַזָּב: שֶׁאֵין אֶחָד מֵהֶן נוֹשֵׂא אֶת רֻבּוֹ. דְּכִי כָּרְעוּ הֵן, אֵין רֻבּוֹ שֶׁל זָב נִשָּׂא עַל אֶחָד מֵהֶן, אֶלָּא כָּל אֶחָד מֵהַמִּשְׁכָּבוֹת נָשָׂא מִעוּטוֹ שֶׁל זָב: (ר"ע מברטנורה).

פרק ד - משנה ו

הַזָּב בְּכַף מֹאזְנַיִם וְאֳכָלִין וּמַשְׁקִין בְּכַף שְׁנִיָּה, טְמֵאִין. וּבַמֵּת, הַכֹּל טָהוֹר, חוּץ מִן הָאָדָם {יב}. זֶה חֹמֶר בַּזָּב מִבַּמֵּת. חֹמֶר בַּמֵּת מִבַּזָּב, שֶׁהַזָּב עוֹשֶׂה מִשְׁכָּב וּמוֹשָׁב מִתַּחְתָּיו לְטַמֵּא אָדָם וּלְטַמֵּא בְגָדִים, וְעַל גַּבָּיו מַדָּף {יג} לְטַמֵּא אֳכָלִין וּמַשְׁקִין, מַה שֶּׁאֵין הַמֵּת מְטַמֵּא. חֹמֶר בַּמֵּת, שֶׁהַמֵּת מְטַמֵּא בְאֹהֶל וּמְטַמֵּא טֻמְאַת שִׁבְעָה, מַה שֶּׁאֵין הַזָּב מְטַמֵּא:

 ר"ע מברטנורה  טְמֵאִין. בֵּין כָּרַע הַזָּב, בֵּין כָּרְעוּ אֳכָלִין וּמַשְׁקִין, נִטְמְאוּ טֻמְאַת הֶסֵּט: וּבַמֵּת הַכֹּל טָהוֹר. בֵּין מוֹשָׁב וּמִשְׁכָּב, בֵּין אֳכָלִים וּמַשְׁקִין, בֵּין כָּרַע הַמֵּת בֵּין כָּרְעוּ הֵם, דְּכָל טֻמְאוֹת הַמַּסִּיטוֹת טְהוֹרוֹת, חוּץ מֵהֶסֵּטוֹ שֶׁל זָב. וְאִם כָּרְעוּ הֵן, נַמִּי טְהוֹרִים, דְּכָל שֶׁהַמֵּת נִשָּׂא עָלָיו כְּגוֹן בְּכַף מֹאזְנַיִם לְבַד מֵאֹהֶל, טָהוֹר, חוּץ מִן הָאָדָם, דְּבִזְמַן שֶׁמַּכְרִיעַ אֶת הַמֵּת, טָמֵא מִפְּנֵי שֶׁהִסִּיטוֹ, כִּדְאָמְרִינַן בְּפֶרֶק בַּתְרָא דְּנִדָּה (דַּף סט), אַטּוּ מֵת מִי לֹא נִטְמָא בְמַשָּׂא: מַדָּף. לְשׁוֹן טֻמְאָה קַלָּה, כְּדִכְתִיב (וַיִּקְרָא כו) קוֹל עָלֶה נִדָּף. וְרַמְבַּ"ם פֵּרֵשׁ, מַדָּף מִלְּשׁוֹן רֵיחוֹ נוֹדֵף, שֶׁרֵיחַ טֻמְאַת הַזָּב הוֹלֶכֶת לְמֵרָחוֹק לְטַמֵּא כָל הַכֵּלִים הָיָה יוֹשֵׁב עַל גַּבֵּי הַמִּטָּה וְאַרְבַּע טַלִּיּוֹת תַּחַת אַרְבַּע רַגְלֵי הַמִּטָּה, טְמֵאוֹת, מִפְּנֵי שֶׁאֵינָהּ יְכוֹלָה לַעֲמֹד עַל שָׁלשׁ. רַבִּי שִׁמְעוֹן מְטַהֵר. הָיָה רוֹכֵב עַל גַּבֵּי בְהֵמָה וְאַרְבַּע טַלִּיּוֹת תַּחַת אַרְבַּע רַגְלֵי בְהֵמָה, טְהוֹרוֹת, מִפְּנֵי שֶׁהִיא יְכוֹלָה לַעֲמֹד עַל שְׁלשָׁה. הָיְתָה טַלִּית אַחַת תַּחַת שְׁנֵי יָדַיִם, תַּחַת שְׁנֵי רַגְלַיִם, תַּחַת יָד וָרֶגֶל, טְמֵאָה. רַבִּי יוֹסֵי אוֹמֵר, הַסּוּס מְטַמֵּא בְרַגְלָיו וְהַחֲמוֹר בְּיָדָיו, שֶׁמִּשְׁעֶנֶת הַסּוּס עַל רַגְלָיו וּמִשְׁעֶנֶת הַחֲמוֹר עַל יָדָיו. יָשַׁב עַל קוֹרַת בֵּית הַבַּד, כֵּלִים שֶׁבָּעֵקֶל טְמֵאִין. עַל הַמַּכְבֵּשׁ שֶׁל כּוֹבֵס, כֵּלִים שֶׁתַּחְתָּיו, טְהוֹרִין. רַבִּי נְחֶמְיָה מְטַמֵּא:

פרק ה - משנה א

הַנּוֹגֵעַ בַּזָּב אוֹ שֶׁהַזָּב נוֹגֵע בּוֹ, הַמַּסִּיט אֶת הַזָּב אוֹ שֶׁהַזָּב מַסִּיטוֹ, מְטַמֵּא אֳכָלִים וּמַשְׁקִים וּכְלֵי שֶׁטֶף בְּמַגָּע, אֲבָל לֹא בְמַשָּׂא. כְּלָל אָמַר רַבִּי יְהוֹשֻׁעַ, כָּל הַמְטַמֵּא בְגָדִים בִּשְׁעַת מַגָּעוֹ {ג}, מְטַמֵּא אֳכָלִים וּמַשְׁקִין לִהְיוֹת תְּחִלָּה, וְהַיָּדַיִם לִהְיוֹת שְׁנִיּוֹת, וְאֵינוֹ מְטַמֵּא לֹא אָדָם וְלֹא כְלֵי חֶרֶס. לְאַחַר פְּרִישָׁתוֹ מִמְּטַמְּאָיו, מְטַמֵּא מַשְׁקִין לִהְיוֹת תְּחִלָּה, וְהָאֳכָלִין וְהַיָּדַיִם לִהְיוֹת שְׁנִיּוֹת, וְאֵינוֹ מְטַמֵּא בְגָדִים:

 ר"ע מברטנורה  הַנּוֹגֵעַ בַּזָּב. וּכְלֵי שֶׁטֶף בְּמַגָּע. דְּכִי הֵיכִי דְּאָדָם הַנּוֹגֵעַ בַּזָּב מְטַמֵּא בְגָדִים בִּשְׁעַת מַגָּעוֹ, כָּךְ מְטַמֵּא כָל כֵּלִים שֶׁנּוֹגֵעַ בָּהֶן בִּשְׁעַת מַגָּעוֹ בַזָּב, חוּץ מֵאָדָם וּכְלֵי חֶרֶס, דְּתַנְיָא בְתוֹרַת כֹּהֲנִים, מִנַּיִן לַעֲשׂוֹת שְׁאָר כֵּלִים כִּבְגָדִים, תַּלְמוּד לוֹמַר (וַיִּקְרָא טו) וְטָמֵא {א}. יָכוֹל יְטַמֵּא אָדָם וּכְלֵי חֶרֶס, תַּלְמוּד לוֹמַר בֶּגֶד {ב}, בֶּגֶד הוּא מְטַמֵּא וְאֵינוֹ מְטַמֵּא אָדָם וּכְלֵי חֶרֶס. וְדַוְקָא בִשְׁעַת מַגָּעוֹ בַזָּב. אֲבָל פֵּרַשׁ, לֹא, דִּכְתִיב (וַיִּקְרָא טו) וְאִישׁ אֲשֶׁר יִגַּע בְּמִשְׁכָּבוֹ יְכַבֵּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא, בִּשְׁעַת מַגָּעוֹ מְטַמֵּא בְגָדִים. פֵּרַשׁ, אֵינוֹ מְטַמֵּא בְגָדִים: אֲבָל לֹא בְמַשָּׂא. אִם בִּשְׁעַת מַגָּעוֹ בַזָּב נָשָׂא בְגָדִים וְלֹא נָגַע בָּהֶן, טְהוֹרִים: כֹּל הַמְטַמֵּא בְגָדִים בִּשְׁעַת מַגָּעוֹ. כְּגוֹן הַנּוֹגֵעַ בְּזָב וְזָבָה וְיוֹלֶדֶת, וּבְמַעַיְנוֹתֵיהֶם, בְּמִשְׁכָּבָן, וּבְמוֹשָׁבָן: מְטַמֵּא אֳכָלִים. לִהְיוֹת תְּחִלָּה, מִפְּנֵי שֶׁהוּא נֶחְשָׁב כְּאַב הַטֻּמְאָה {ד} בִּשְׁעַת מַגָּעוֹ, כֵּיוָן דְּאָמְרָה תוֹרָה שֶׁהוּא מְטַמֵּא בְגָדִים: וְאֶת הַיָּדַיִם לִהְיוֹת שְׁנִיּוֹת. אֲבָל גּוּפוֹ שֶׁל אָדָם, אֵינוֹ מְטַמֵּא, כִּדְקָתָנֵי סֵיפָא שֶׁאֵינוֹ מְטַמֵּא אָדָם וּכְלֵי חֶרֶס: לְאַחַר פְּרִישָׁתוֹ. מִמְּטַמְּאָיו מְטַמֵּא אֶת הַמַּשְׁקִין לִהְיוֹת תְּחִלָּה. וְאַף עַל פִּי שֶׁאֵינוֹ אֶלָּא רִאשׁוֹן לַטֻּמְאָה, עוֹשֶׂה הַמַּשְׁקִין תְּחִלָּה, דַּאֲפִלּוּ שֵׁנִי לַטֻּמְאָה מְטַמֵּא מַשְׁקִין לִהְיוֹת תְּחִלָּה. לְפִי שֶׁהֵן עֲלוּלִין לְקַבֵּל טֻמְאָה וְאֵינָן צְרִיכִין הֶכְשֵׁר כְּמוֹ הָאֳכָלִים, גָּזְרוּ בָהֶן שֶׁיִּהְיוּ לְעוֹלָם תְּחִלָּה, גְּזֵרָה מִשּׁוּם מַשְׁקִין הַבָּאִין מֵחֲמַת שֶׁרֶץ שֶׁהֵם תְּחִלָּה מִן הַתּוֹרָה: וְהָאֳכָלִים וְהַיָּדַיִם לִהְיוֹת שְׁנִיּוֹת. דְּכֵיוָן דְּפֵרַשׁ מִמְּטַמְּאָיו, אֵינוֹ אֶלָּא רִאשׁוֹן בְּעָלְמָא: (ר"ע מברטנורה).

 עיקר תוי"ט  {א} דַּהֲוָה לֵיהּ לְמֵימַר וְרָחַץ בַּמַּיִם בָּעֶרֶב, אֶלָּא לוֹמַר דְּעוֹד טָמֵא גַּם כֵּן כָּל כְּלִי שֶׁיִּגַּע בּוֹ. וּכְלֵי שֶׁטֶף הַיְנוּ כָּל שְׁאָר כֵּלִים חוּץ מִכְּלֵי חֶרֶס. עַל שֵׁם שֶׁיֵּשׁ לָהֶם טָהֳרָה בְּמִקְוֶה. רַשִׁ"י: {ב} דִּכְלִי שֶׁטֶף דּוֹמֶה קְצָת לְבֶגֶד, דַּהֲוֵי נַמִּי כְּלִי, מַה שֶּׁאֵין כֵּן הָאָדָם. וּכְלִי חֶרֶס אֵין דּוֹמֶה לוֹ, שֶׁאֵין לוֹ טָהֳרָה בְּמִקְוֶה: {ג} כָּל הַמְטַמֵּא כוּ'. כָּל אָדָם הַמְטַמֵּא קָאָמַר, דְּהָא מַשְׁכַּחַת כֵּלִים שֶׁאֵין מְטַמְּאִין כֵּלִים, וְאִם נָגְעוּ מְטַמְּאִין אֳכָלִים לִהְיוֹת תְּחִלָּה כִּדְמוּכָח בְּמַתְנִיתִין דִּלְקַמָּן. דְּדַוְקָא בְּאֵין נוֹגְעִים מַיְרֵי סֵיפָא, הָא נוֹגְעִים מְטַמְּאִין שְׁנַיִם. וְכֵן הוּא בְּהֶדְיָא בַתּוֹסֶפְתָּא. וְשָׁם מְפָרֵשׁ הָרַ"ב דְּמַתְנִיתִין כָּל אָדָם קָאָמַר: {ד} כְּלוֹמַר, לְעִנְיַן אֳכָלִים וּמַשְׁקִים. וְאַף עַל פִּי שֶׁאֵינוֹ אַב הַטֻּמְאָה לְעִנְיַן אָדָם וּכְלֵי חֶרֶס: (עיקר תוי"ט).

פרק ה - משנה ב

וְעוֹד כְּלָל אַחֵר אָמְרוּ. כֹּל הַנִּשָּׂא עַל גַּבֵּי הַזָּב, טָמֵא {ו}. וְכֹל שֶׁהַזָּב נִשָּׂא עָלָיו, טָהוֹר {ז}, חוּץ מִן הָרָאוּי לְמִשְׁכָּב וּלְמוֹשָׁב וְהָאָדָם. כֵּיצַד {ט}. אֶצְבָּעוֹ {י} שֶׁל זָב תַּחַת הַנִּדְבָּךְ, הַטָּהוֹר מִלְמַעְלָן, מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד. פֵּרַשׁ, מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד. הַטָּמֵא מִלְמַעְלָן וְהַטָּהוֹר מִלְּמַטָּן, מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד. פֵּרַשׁ, מְטַמֵא אֶחָד וּפוֹסֵל אֶחָד. הָאֳכָלִין וְהַמַּשְׁקִין, הַמִּשְׁכָּב, וְהַמּוֹשָׁב וְהַמַּדָּף מִלְמַעְלָן, מְטַמְּאִין שְׁנַיִם וּפוֹסְלִין אֶחָד. פֵּרְשׁוּ, מְטַמְּאִין אֶחָד וּפוֹסְלִין אֶחָד. וְהַמִּשְׁכָּב וְהַמּוֹשָׁב מִלְּמַטָּן, מְטַמְּאִין שְׁנַיִם וּפוֹסְלִין אֶחָד. פֵּרְשׁוּ, מְטַמְּאִין שְׁנַיִם {יא} וּפוֹסְלִין אֶחָד. הָאֳכָלִין וְהַמַּשְׁקִין וְהַמַּדָּף מִלְּמַטָּן, טְהוֹרִין:

 ר"ע מברטנורה  כֹּל הַנִּשָּׂא עַל גַּבֵּי הַזָּב. עֶלְיוֹנוֹ שֶׁל זָב וַאֲפִלּוּ לֹא נָגַע בּוֹ הַזָּב {ה}, וַאֲפִלּוּ יֵשׁ דְּבָרִים הַרְבֵּה חוֹצְצִים בֵּינוֹ לְבֵין הַזָּב, הַכֹּל טָמֵא, דִּכְתִיב (שָׁם) וְהַנּוֹגֵעַ בְּכָל אֲשֶׁר יִהְיֶה תַחְתָּיו, מַאי תַחְתָּיו, אִילֵימָא תַּחְתָּיו דְּזָב, הַיְנוּ מִשְׁכָּב, אֶלָּא בְּכָל אֲשֶׁר יִהְיֶה הַזָּב תַּחְתָּיו, דְּהַיְנוּ עֶלְיוֹנוֹ שֶׁל זָב, וְנוֹהֵג בְּכָל דָּבָר, אֲפִלּוּ בְמִידִי דְּאֵין רְגִילוּת לִהְיוֹת עָלָיו, וְלֹא הָווּ כְעֵין תַּחְתָּיו, דְּאֵין נוֹהֵג אֶלָּא בְמִידִי דַּחֲזִי לֵיהּ: חוּץ מִן הָרָאוּי לְמִשְׁכָּב וּלְמוֹשָׁב. וְהוּא הַדִּין לְמֶרְכָּב {ח}. אֲבָל שֶׁאֵינוֹ רָאוּי, טָהוֹר, כְּגוֹן אִם כָּפָה סְאָה שֶׁאֵינָהּ עֲשׂוּיָה לִישִׁיבָה וְיָשַׁב עָלֶיהָ, יָכוֹל יְהֵא טָמֵא מוֹשָׁב, תַּלְמוּד לוֹמַר (שָׁם) וְהַיֹּשֵׁב עַל הַכְּלִי אֲשֶׁר יֵשֵׁב עָלָיו הַזָּב, מַה שֶּׁמְּיֻחָד לִישִׁיבָה, יָצָא זֶה שֶׁאוֹמְרִים לוֹ עֲמֹד וְנַעֲשֶׂה מְלַאכְתֵּנוּ: נִדְבָּךְ. שׁוּרַת בִּנְיַן הַחוֹמָה. כְּמוֹ נִדְבָּכִין דִּי אֶבֶן גְּלָל (בְּעֶזְרָא ו): הַטָּהוֹר מִלְּמַעְלָה. עַל הַנִּדְבָּךְ וּמַכְבִּיד עַל אֶצְבָּעוֹ שֶׁל זָב, דְּהַשְׁתָּא הָוֵי נִשָּׂא עַל גַּבֵּי הַזָּב וְנֶחְשָׁב אַב הַטֻּמְאָה: מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד. מְטַמֵּא רִאשׁוֹן וְשֵׁנִי, וּפוֹסֵל שְׁלִישִׁי בַּתְּרוּמָה: פֵּרַשׁ. שֶׁיָּרַד הַטָּהוֹר מִן הַנִּדְבָּךְ, אוֹ שֶׁסִּלֵּק הַזָּב אֶצְבָּעוֹ מִתַּחַת הַנִּדְבָּךְ. מְטַמֵּא הַטָּהוֹר אֶחָד וּפוֹסֵל אֶחָד, לְפִי שֶׁאֵין הַטָּהוֹר אֶלָּא רִאשׁוֹן מִכֵּיוָן שֶׁפֵּרַשׁ, דְּתוּ לֹא מְטַמֵּא בְגָדִים, וְעוֹשֶׂה שֵׁנִי וּשְׁלִישִׁי בַּתְּרוּמָה: הַטָּמֵא מִלְמַעְלָה. עַל הַנִּדְבָּךְ, וְאֶצְבַּע הַטָּהוֹר תַּחַת הַנִּדְבָּךְ, וּמַכְבִּיד הַטָּמֵא עָלָיו, הֲרֵי הַזָּב נִשָּׂא עָלָיו, לְפִיכָךְ מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד: וְהַמַּדָּף. כָּל כֵּלִים שֶׁאֵינָם מִשְׁכָּב וּמוֹשָׁב מַדָּף קָרֵי לְהוּ: מִלְמַעְלָן. שֶׁל נִדְבָּךְ, וְאֶצְבַּע הַזָּב מִלְּמַטָּה, אֵינָם אֶלָּא רִאשׁוֹן: מְטַמְּאִין אֶחָד וּפוֹסְלִים אֶחָד. וּלְאַחַר שֶׁפֵּרְשׁוּ נַמִּי כָךְ דִּינָא: פֵּרְשׁוּ מְטַמְּאִין שְׁנַיִם וּפוֹסְלִים אֶחָד. הַמִּשְׁכָּב וּמוֹשָׁב לְעוֹלָם אַב הַטֻּמְאָה אַף עַל פִּי שֶׁפֵּרְשׁוּ: טְהוֹרִים. כִּדְאָמְרִינַן שֶׁכָּל שֶׁהַזָּב נִשָּׂא עָלָיו, טָהוֹר, חוּץ מִמִּשְׁכָּב וּמוֹשָׁב וְהָאָדָם: (ר"ע מברטנורה).

 עיקר תוי"ט  {ה} אֲבָל סֵיפָא כָּל שֶׁהַזָּב נִשָּׂא עָלָיו טָהוֹר, דַּוְקָא בְלֹא נָגַע. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ו} טָמֵא. וּלְקַמָּן מְפָרֵשׁ שֶׁאֵין כֻּלָּם שָׁוִין: {ז} טָהוֹר. לְגַמְרֵי. וְאֵין בָּהֶן אֲפִלּוּ טֻמְאַת מַדָּף, כִּדְאִיתָא בְתוֹרַת כֹּהֲנִים. תּוֹסָפוֹת. וְצָרִיךְ לוֹמַר דְּמַיְרֵי שֶׁלֹּא הוּסַט. דְּעַל יְדֵי הֶסֵּט הָא אֲמַרַן לְעֵיל דְּמִטַּמְאִין. וְהָרַ"שׁ פֵּרֵשׁ, דַּוְקָא עַל יְדֵי הֶסֵּט, וְטָהוֹר מִשּׁוּם דְּהַנּוֹשֵׂא אֵין בּוֹ רוּחַ חַיִּים, וּלְכָךְ תָּנֵי חוּץ מִן הָאָדָם. וְקָשֶׁה, דְּהֵיכִי מְטַהֲרִין הָכָא לְגַמְרֵי: {ח} דְּאֵין לְמֵדִין מִן הַכְּלָלוֹת: {ט} כֵּיצַד. אַרֵישָׁא נַמִּי קָאֵי: {י} אֶצְבָּעוֹ. נִרְאֶה לִי שֶׁטֻּמְאָה זוֹ מִדִּבְרֵיהֶם. וּלְפִי זֶה צָרִיךְ לוֹמַר דְּמִשְׁכָּב וּמוֹשָׁב דְּסֵיפָא לֹא מַיְרֵי כְּשֶׁאֶצְבָּעוֹ בִּלְבַד מִלְמַעְלָן, דִּבְהוּ לְעוֹלָם רֻבּוֹ בָעִינַן. אֶלָּא מַתְנִיתִין דְּהָכָא וּדְהָתָם לָאו בַּחֲדָא שִׁיטָה, וְהָא כִדְאִיתָא וְהָא כִדְאִיתָא: {יא} שְׁנַיִם. לְפִי שֶׁהַמִּשְׁכָּב אַב הַטֻּמְאָה אֲפִלּוּ כְשֶׁנִּתְרוֹקֵן הַזָּב מִמֶּנּוּ. כִּדְיָלֵיף לֵיהּ בְּתוֹרַת כֹּהֲנִים: (עיקר תוי"ט).

פרק ה - משנה ג

מִפְּנֵי שֶׁאָמְרוּ, כֹּל הַנּוֹשֵׂא וְנִשָּׂא עַל גַּבֵּי מִשְׁכָּב, טָהוֹר, חוּץ מִן הָאָדָם. כֹּל הַנּוֹשֵׂא וְנִשָּׂא עַל גַּבֵּי הַנְּבֵלָה, טָהוֹר, חוּץ מִן הַמַּסִּיט. רַבִּי אֱלִיעֶזֶר אוֹמֵר, אַף הַנּוֹשֵׂא {יז}. כֹּל הַנּוֹשֵׂא וְנִשָּׂא עַל גַּבֵּי הַמֵּת, טָהוֹר, חוּץ מִן הַמַּאֲהִיל {יח}, וְאָדָם בִּזְמַן שֶׁהוּא מַסִּיט:

 ר"ע מברטנורה  מִפְּנֵי שֶׁאָמְרוּ. כְּלוֹמַר מִפְּנֵי מָה אָמְרוּ הָאֳכָלִים וְהַמַּשְׁקִים וְהַמַּדָּף מִלְּמַטָּה תַּחַת מִשְׁכַּב הַזָּב {יב} טְהוֹרִים, מִפְּנֵי שֶׁאָמְרוּ כָּל הַנּוֹשֵׂא וְנִשָּׂא כוּ': כֹּל הַנּוֹשֵׂא. כְּגוֹן אֳכָלִין וּמַשְׁקִין וּמַדָּף שֶׁהֵן לְמַטָּה מִמִּשְׁכַּב הַזָּב וְנוֹשְׂאִים אוֹתוֹ וְאֵינָן נוֹגְעִין בּוֹ {יג}, שֶׁיֵּשׁ דְּבָרִים חוֹצְצִים בֵּינְתַיִם: נִשָּׂא עַל גַּבֵּי הַמִּשְׁכָּב. שֶׁאֵלּוּ לְמַעְלָה וְהַמִּשְׁכָּב לְמַטָּה וְנִשָּׂא עָלָיו וְאֵין נוֹגְעִין בּוֹ, טְהוֹרִים: חוּץ מִן הָאָדָם. שֶׁהָאָדָם הַנּוֹשֵׂא הַמִּשְׁכָּב, נִטְמָא אַף עַל פִּי שֶׁלֹּא נָגַע בּוֹ וְלֹא הִסִּיטוֹ, מִפְּנֵי שֶׁמִּשְׁכָּב הַזָּב מְטַמֵּא בְמַשָּׂא. וְכֵן אִם נָתְנוּ אֶבֶן מְסָמָא לְמַעְלָה מִמִּשְׁכָּבוֹ שֶׁל זָב וְיָשַׁב אָדָם טָהוֹר עַל הָאֶבֶן, נִטְמָא, כֵּיוָן שֶׁנִּשָּׂא עַל גַּבֵּי מִשְׁכָּב. כְּשֵׁם שֶׁאִם יָשַׁב הַזָּב עַל הָאֶבֶן נִטְמָא הַמִּשְׁכָּב שֶׁלְּמַטָּה מִן הָאֶבֶן, כָּךְ כְּשֶׁיָּשַׁב הַטָּהוֹר עַל הָאֶבֶן, נִטְמָא וְנִשָּׂא מִן הַמִּשְׁכָּב שֶׁתַּחְתֶּיהָ: כֹּל הַנּוֹשֵׂא עַל גַּבֵּי נְבֵלָה. בְּאָדָם קָמַיְרֵי {יד}, כְּלוֹמַר וְכָל אָדָם הַנּוֹשֵׂא וְנִתָּן {טו} עַל גַּבֵּי נְבֵלָה: טָהוֹר חוּץ מִן הַמַּסִּיט. דְּנוֹשֵׂא אֶת הַנְּבֵלָה וְלֹא זָז מִמְּקוֹמוֹ גַּם לֹא הֱנִיעָהּ מִמְּקוֹמָהּ אֶלָּא שֶׁמֻּנַּחַת עַל גַּבָּיו, כְּגוֹן שֶׁבָּא חֲבֵרוֹ וְהִנִּיחָהּ עָלָיו, טָהוֹר, עַד שֶׁיַּסִּיטֶנָּה {טז}: רַבִּי אֱלִיעֶזֶר אוֹמֵר אַף הַנּוֹשֵׂא. בַּגְּמָרָא מְפָרֵשׁ דְּהָכִי קָאָמַר, וְהוּא שֶׁנּוֹשֵׂא, כְּלוֹמַר אֵין הַמַּסִּיט אֶת הַנְּבֵלָה טָמֵא אֶלָּא כְשֶׁנּוֹשֵׂא אוֹתָהּ, שֶׁנְּבֵלָה לְמַעְלָה וְהוּא לְמַטָּה. אֲבָל אִם הַנְּבֵלָה לְמַטָּה וְהוּא לְמַעְלָה, אַף עַל פִּי שֶׁמַּסִּיטָהּ, טָהוֹר, כָּל זְמַן שֶׁלֹּא נָגַע בָּהּ. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: חוּץ מִן הַמַּאֲהִיל. שֶׁאֶפְשָׁר לִהְיוֹת נוֹשֵׂא וְנִשָּׂא וְלֹא יַאֲהִיל, כְּגוֹן עֲלִיָּה מַפְסֶקֶת, שֶׁאִם הַמֵּת בָּעֲלִיָּה וָאֳכָלִין וּמַשְׁקִין וְכֵלִים בַּבַּיִת מַגִּיעִין עַד שְׁמֵי קוֹרָה וְהַנְּסָרִים נִכְפָּפִים מִכֹּבֶד הַמֵּת וּמַכְבִּידִין עַל הַכֵּלִים, אוֹ שֶׁהַמֵּת בַּבַּיִת סָמוּךְ לִשְׁמֵי קוֹרָה וְהַכֵּלִים בָּעֲלִיָּה וְנִכְפָּפִים הַנְּסָרִים שֶׁבָּעֲלִיָּה מִכֹּבֶד הַכֵּלִים וּמַכְבִּידִים עַל הַמֵּת {יט}, בִּכְהַאי גַּוְנָא הָוֵי נוֹשֵׂא וְנִשָּׂא עַל גַּבֵּי הַמֵּת וְאֵינוֹ מַאֲהִיל, וְטָהוֹר: וְאָדָם בִּזְמַן שֶׁהוּא מַסִּיט. בֵּין נוֹשֵׂא אֶת הַמֵּת, בֵּין נִשָּׂא עַל גַּבֵּי מֵת, טָמֵא אַף עַל פִּי שֶׁאֵינוֹ נוֹגֵעַ בּוֹ. אֲבָל אִם אֵין הַמֵּת זָז מִמְּקוֹמוֹ, שֶׁאֵין כָּאן הֶסֵּט, טָהוֹר {כ}: (ר"ע מברטנורה).

 עיקר תוי"ט  {יב} וְאַף שֶׁלֹּא אָמְרוּ לְעֵיל אֶלָּא תַחַת הַזָּב, וַדַּאי דְּלֹא חֲמִירָא מִשְׁכָּב מִזָּב עַצְמוֹ, וּמִמֵּילָא נַשְׁמְעִינֵיהּ: {יג} וַאֲפִלּוּ לֹא הִפְסִיק בֵּינֵיהֶם אֶלָּא נְיָר. הָרַמְבַּ"ם: {יד} דְּאֵין לְךָ מִטַּמֵּא בְמַשָּׂא בִּנְבֵלָה אֶלָּא הָאָדָם: {טו} צָרִיךְ לוֹמַר וְנִשָּׂא. וְכֵן הוּא בָּרַ"שׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טז} וּמִדְּלֹא תָנֵי גַּבֵּי מִשְׁכָּב חוּץ מִן הַמַּסִּיט, צָרִיךְ לוֹמַר דִּמְטַמֵּא בְלֹא מַסִּיט. וְטַעֲמָא נִרְאֶה לִי, מִשּׁוּם דְּאַחְמִיר רַחֲמָנָא בְּמִשְׁכָּב דַּאֲפִלּוּ הַנּוֹגֵעַ מְטַמֵּא בְגָדִים, וְגַבֵּי נְבֵלָה תְּנַן חֲשׂוּכֵי בְגָדִים בְּמַגָּע. הָרַ"שׁ: {יז} הַנּוֹשֵׂא. נַמִּי בְאָדָם. דּוּמְיָא דִנְבֵלָה: {יח} הַמַּאֲהִיל. בֵּין שֶׁהַמֵּת תַּחְתָּיו וְהָאָדָם נִשָּׂא בֵּין שֶׁהַמֵּת מַאֲהִיל עָלָיו וְהוּא נִשָּׂא, מְטַמֵּא טֻמְאַת אֹהֶל: {יט} לִרְבוּתָא כָּתַב כֵּן, דַּאֲפִלּוּ מִשּׁוּם מַשָּׂא שֶׁמַּכְבִּיד עָלָיו אֵינוֹ מְטַמֵּא, שֶׁאֵין בַּמֵּת מִשּׁוּם אֶבֶן מְסָמָא. וּמִכָּל מָקוֹם בְּאֵינָהּ רְצוּצָה מַיְרֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} וְטַעֲמָא נִרְאֶה לִי, דְּטֻמְאַת מַשָּׂא דְּמֵת יָלְפִינַן בְּקַל וָחֹמֶר מִנְּבֵלָה, וְאָמְרִינַן דַּיּוֹ שֶׁלֹּא יְטַמֵּא מַשָּׂאוֹ אֶלָּא אִם כֵּן הִסִּיט, כְּמוֹ בִּנְבֵלָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ה - משנה ד

מִקְצָת טָמֵא עַל הַטָּהוֹר וּמִקְצָת טָהוֹר עַל הַטָּמֵא, חִבּוּרֵי טָמֵא עַל הַטָּהוֹר וְחִבּוּרֵי טָהוֹר עַל הַטָּמֵא, טָמֵא. רַבִּי שִׁמְעוֹן אוֹמֵר, מִקְצָת טָמֵא עַל הַטָּהוֹר, טָמֵא. וּמִקְצָת טָהוֹר עַל הַטָּמֵא, טָהוֹר:

 ר"ע מברטנורה  מִקְצָת טָמֵא עַל הַטָּהוֹר. אֲפִלּוּ אֶצְבָּעוֹ שֶׁל זָב עַל הַטָּהוֹר, אוֹ אֶצְבָּעוֹ שֶׁל טָהוֹר עַל הַזָּב, אוֹ חִבּוּרֵי טָמֵא, כְּגוֹן שְׂעָרוֹ וְצִפָּרְנָיו וְשִׁנָּיו, כָּל אֵלּוּ נִקְרָאִין חִבּוּרֵי, אַף עַל גַּב דְּלֵיכָּא אֶלָּא מִעוּט הַטָּמֵא עַל הַטָּהוֹר אוֹ מִעוּט הַטָּהוֹר עַל הַטָּמֵא, נִטְמָא הַטָּהוֹר, דְּלֹא בָעִינַן שֶׁיִּנָּשֵׂא רֻבּוֹ שֶׁל זָב אֶלָּא עַל הַמִּשְׁכָּב וְלֹא עַל הָאָדָם: וּמִקְצָת טָהוֹר עַל הַטָּמֵא טָהוֹר. דְּבָעִינַן שֶׁיִּנָּשֵׂא רֻבּוֹ. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: (ר"ע מברטנורה).

פרק ה - משנה ה

הַטָּמֵא עַל מִקְצָת הַמִּשְׁכָּב וְהַטָּהוֹר עַל מִקְצָת הַמִּשְׁכָּב, טָמֵא. מִקְצָת טָמֵא עַל הַמִּשְׁכָּב וּמִקְצָת טָהוֹר עַל הַמִּשְׁכָּב, טָהוֹר. נִמְצֵאת טֻמְאָה נִכְנֶסֶת לוֹ וְיוֹצֵאת מִמֶּנּוּ בְּמִעוּטוֹ. וְכֵן כִּכָּר שֶׁל תְּרוּמָה שֶׁהוּא נָתוּן עַל גַּבֵּי מִשְׁכָּב וְהַנְּיָר בֵּינְתַיִם, בֵּין מִלְמַעְלָן בֵּין מִלְּמַטָּן, טָהוֹר. וְכֵן בְּאֶבֶן הַמְנֻגַּעַת, טָהוֹר. רַבִּי שִׁמְעוֹן מְטַמֵּא בָזוֹ:

 ר"ע מברטנורה  הַטָּמֵא עַל מִקְצָת הַמִּשְׁכָּב. רֻבּוֹ שֶׁל טָמֵא עַל מִעוּטוֹ שֶׁל מִשְׁכָּב, נִטְמָא הַמִּשְׁכָּב. וְכֵן רֻבּוֹ שֶׁל טָהוֹר עַל מִעוּטוֹ שֶׁל הַמִּשְׁכָּב הַטָּמֵא, נִטְמָא הַטָּהוֹר. וְלֹא חָיְשִׁינַן עַל מִעוּט מִשְׁכָּב, כֵּיוָן דְּרֻבָּא דְטָהוֹר וְרֻבָּא דְטָמֵא {כא} יוֹשֵׁב עָלָיו, דִּבְמִעוּטוֹ שֶׁל מִשְׁכָּב טֻמְאָה נִכְנֶסֶת, וּבְמִעוּטוֹ יוֹצֵאת, כִּדְקָתָנֵי סֵיפָא נִמְצֵאת טֻמְאָה נִכְנֶסֶת לוֹ וְיוֹצְאָה מִמֶּנּוּ בְּמִעוּטוֹ: מִקְצָת הַטָּמֵא עַל הַמִּשְׁכָּב. כֵּיוָן דְּמִעוּטוֹ שֶׁל טָמֵא הוּא, אַף עַל פִּי שֶׁנִּשְׁעָן עַל כָּל הַמִּשְׁכָּב, טָהוֹר הַמִּשְׁכָּב, עַד שֶׁיִּנָּשֵׂא עָלָיו רֻבּוֹ שֶׁל זָב: וּמִקְצָת טָהוֹר עַל הַמִּשְׁכָּב. כְּלוֹמַר אוֹ מִקְצָת טָהוֹר אֲפִלּוּ עַל כָּל הַמִּשְׁכָּב: טָהוֹר. עַד שֶׁיִּשָּׁעֵן עָלָיו רֻבּוֹ שֶׁל טָהוֹר: וְכֵן כִּכָּר שֶׁל תְּרוּמָה. אַמַּתְנִיתִין דִּלְעֵיל קָאֵי, דִּתְנַן כָּל הַנּוֹשֵׂא וְנִשָּׂא עַל גַּבֵּי הַמִּשְׁכָּב, טָהוֹר. וַאֲפִלּוּ אֳכָלִין שֶׁל תְּרוּמָה, אִם נִשְּׂאוּ עַל גַּבֵּי מִשְׁכָּב וְיֵשׁ בֵּינֵיהֶן דָּבָר חוֹצֵץ אֲפִלּוּ הוּא דַק בְּיוֹתֵר כְּגוֹן נְיָר וְכַיּוֹצֵא בוֹ, אֵינָן מְקַבְּלִין טֻמְאָה מִן הַמִּשְׁכָּב, דְּאֵין מְטַמֵּא אֲחֵרִים וְלֹא מְקַבֵּל טֻמְאָה בְּהֶסֵּט אֶלָּא דָבָר שֶׁיֶּשׁ בּוֹ רוּחַ חַיִּים {כב}: וְרַבִּי שִׁמְעוֹן מְטַמֵּא בָזוֹ. הוֹאִיל וּמְקוֹמָהּ טָמֵא, מְטַמֵּא בְהֶסֵּט אֲפִלּוּ דָבָר שֶׁאֵין בּוֹ רוּחַ חַיִּים. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: (ר"ע מברטנורה).

 עיקר תוי"ט  {כא} כִּדְיָלֵיף לְהוּ בְּסִפְרָא. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כב} הָרַ"שׁ. וּלְשִׁיטָתֵיהּ אָזֵיל כוּ', אֲבָל הָרַ"ב בְּחִנָּם נַקְטֵיהּ וְאֵין בּוֹ צֹרֶךְ לְפֵרוּשׁ הַמִּשְׁנָה. וְגַם בְּעִקַּר הַדִּין נִרְאֶה לִי דְּלָאו כְּהִלְכָתָא הוּא. וּרְאָיָה מִמִּשְׁנָה ה' ו' דְּפֶרֶק ד': (עיקר תוי"ט).

פרק ה - משנה ו

הַנּוֹגֵעַ בְּזָב, וּבְזָבָה, וּבְנִדָּה, וּבְיוֹלֶדֶת, וּבִמְצֹרָע, בְּמִשְׁכָּב, וּמוֹשָׁב, מְטַמֵּא שְׁנַיִם {כג} וּפוֹסֵל אֶחָד. פֵּרַשׁ, מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד, אֶחָד הַנּוֹגֵעַ וְאֶחָד הַמַּסִּיט {כד} וְאֶחָד הַנּוֹשֵׂא וְאֶחָד הַנִּשָּׂא {כו}:

 ר"ע מברטנורה  הַנּוֹגֵעַ בְּזָב וּבְזָבָה כוּ'. כָּל הָנֵי שָׁוֶה מַגָּעָן לְמַשָּׂאָן דִּמְטַמְּאוּ אָדָם לְטַמֵּא בְגָדִים, הִלְכָּךְ חֲשִׁיבֵי אַב הַטֻּמְאָה כָּל זְמַן שֶׁלֹּא פֵרְשׁוּ, מֵאַחַר דִּמְטַמְּאוּ בְגָדִים, וּלְכָךְ הַנּוֹגֵעַ עוֹשֶׂה רִאשׁוֹן וְשֵׁנִי וּפוֹסֵל שְׁלִישִׁי בַּתְּרוּמָה: פֵּרַשׁ. מִמְּטַמְּאָיו, אֵינוֹ אֶלָּא רִאשׁוֹן בְּעָלְמָא וּמְטַמֵּא אֶחָד לַעֲשׂוֹתוֹ שֵׁנִי, וּפוֹסֵל שְׁלִישִׁי בַּתְּרוּמָה: וְאֶחָד הַמַּסִּיט וְאֶחָד הַנּוֹשֵׂא. וְאַף עַל פִּי שֶׁאֵינָן נוֹגְעִים. זָב וְזָבָה וְנִדָּה בְּהֶדְיָא כְתִיבֵי, דִּכְתִיב זֹאת תּוֹרַת הַזָּב וְגוֹ' וְהַדָּוָה בְּנִדָּתָהּ. וּמְצֹרָע וְיוֹלֶדֶת {כה} נַפְקָא לָן מִלַּזָּכָר וְלַנְּקֵבָה, לַזָּכָר לְרַבּוֹת אֶת הַמְצֹרָע, וְלַנְּקֵבָה לְרַבּוֹת אֶת הַיּוֹלֶדֶת: (ר"ע מברטנורה).

 עיקר תוי"ט  {כג} שְׁנַיִם. בְּמַגָּע וְלֹא בְמַשָּׂא: {כד} הַמַּסִּיט. כְּלוֹמַר, וְאֵינוֹ נוֹשֵׂא וְנִשָּׂא. וּבֵין שֶׁהַטָּהוֹר הִסִּיט לַטָּמֵא אוֹ אִפְּכָא. וְקָאֵי אַזָּב וְכוּ'. אֲבָל בִּמְצֹרָע וּמִשְׁכָּב וּמוֹשָׁב לֵיכָּא לְפָרוֹשֵׁי הָכִי, דְּהָא כָּל טֻמְאוֹת הַמַּסִּיט טְהוֹרוֹת חוּץ מִשֶּׁל זָב: {כה} בְּלֵידָה יָבֵשְׁתָּא: {כו} הַנּוֹשֵׂא כוּ'. כְּלוֹמַר וַאֲפִלּוּ אֵינוֹ מַסִּיט. מַה שֶּׁאֵין כֵּן נְבֵלָה וּמֵת דְּמִשְׁנָה ג' דְּבָעִינַן דַּוְקָא שֶׁיְּהֵא נִיסָט, עַיֵּן שָׁם: (עיקר תוי"ט).

פרק ה - משנה ז

הַנּוֹגֵעַ בְּזוֹבוֹ שֶׁל זָב, וּבְרֻקּוֹ, בְּשִׁכְבַת זַרְעוֹ, בְּמֵימֵי רַגְלָיו, וּבְדַם הַנִּדָּה, מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד. פֵּרַשׁ, מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד, אֶחָד הַנּוֹגֵעַ וְאֶחָד הַמַּסִּיט {כז}. רַבִּי אֱלִיעֶזֶר אוֹמֵר, אַף הַנּוֹשֵׂא:

 ר"ע מברטנורה  וְאֶחָד הַמַּסִּיט. הַמֵּזִיז אוֹתָהּ מִמְּקוֹמָהּ אַף עַל פִּי שֶׁאֵינוֹ לֹא נוֹגֵעַ וְלֹא נוֹשֵׂא: רַבִּי אֱלִיעֶזֶר אוֹמֵר אַף הַנּוֹשֵׂא. וְהוּא שֶׁנּוֹשֵׂא קָאָמַר, כְּלוֹמַר אֵין הַמַּסִּיט טָמֵא אֶלָּא אִם כֵּן נוֹשֵׂא {כח}, שֶׁהֵן לְמַעְלָה וְהַמַּסִּיט לְמַטָּה. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: (ר"ע מברטנורה).

 עיקר תוי"ט  {כז} הַמַּסִּיט. אֲבָל בְּמַשָּׂא בְּלֹא הֶסֵּט לֹא מְטַמֵּא, וְדָמוּ לִנְבֵלָה וּמֵת, דְּכֻלְּהוּ אִמְּעִיטוּ מִקְּרָא מֵאֶבֶן מְסָמָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כח} הָרַמְבַּ"ם. וּלְשִׁיטָתֵיהּ דִּסְבִירָא לֵיהּ דִּלְתַנָּא קַמָּא נַמִּי נוֹשֵׂא מְטַמֵּא בְּלֹא הֶסֵּט. אֲבָל הָרַ"שׁ מְפָרֵשׁ כָּאן כִּפְשׁוּטוֹ, דְּרַבִּי אֱלִיעֶזֶר פָּלֵיג וּסְבִירָא לֵיהּ דְּנוֹשֵׂא בְלֹא מַסִּיט נַמִּי מְטַמֵּא, דְּלֹא מְמַעֵט לְהוּ מִקְּרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ה - משנה ח

הַנּוֹשֵׂא אֶת הַמֶּרְכָּב וְהַנִּשָּׂא עָלָיו וְהַמַּסִּיטוֹ {כט}, מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד. פֵּרַשׁ, מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד. הַנּוֹשֵׂא אֶת הַנְּבֵלָה, וְאֶת מֵי חַטָּאת שֶׁיֶּשׁ בָּהֶם כְּדֵי הַזָּיָה, מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד. פֵּרַשׁ, מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד:

 ר"ע מברטנורה  הַנּוֹשֵׂא אֶת הַמֶּרְכָּב. בְּנוֹשֵׂא אֶת הַמֶּרְכָּב כְּתִיב כִּבּוּס בְּגָדִים, וְלֹא בְנוֹגֵעַ, וְאֵין נוֹגֵעַ אֶלָּא רִאשׁוֹן בְּעָלְמָא: הַנּוֹשֵׂא אֶת הַנְּבֵלָה. הַיְנוּ מַסִּיט, דְּנוֹשֵׂא בְּלֹא מַסִּיט, טָהוֹר כְּדִפְרִישִׁית לְעֵיל. וְנוֹגֵעַ נַמִּי בִּנְבֵלָה, אַף עַל פִּי שֶׁהוּא טָמֵא, כֵּיוָן דְּלֹא מְטַמֵּא בְגָדִים אֵינוֹ מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד, דְּאֵינוֹ אֶלָּא רִאשׁוֹן בְּעָלְמָא וּמְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד: מֵי חַטָּאת שֶׁיֵּשׁ בָּהֶן כְּדֵי הַזָּיָה. מְטַמְּאִין בְּגָדִים בְּמַשָּׂא אֲבָל לֹא בְמַגָּע: (ר"ע מברטנורה).

 עיקר תוי"ט  {כט} וְהַמַּסִּיטוֹ. לַמֶּרְכָּב. אֲבָל מֶרְכָּב הַמַּסִּיט לַטָּהוֹר, לֹא, וּכְמוֹ בְמִשְׁכָּב: (עיקר תוי"ט).

פרק ה - משנה ט

הָאוֹכֵל מִנִּבְלַת עוֹף טָהוֹר וְהִיא בְּבֵית הַבְּלִיעָה, מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד. הִכְנִיס רֹאשׁוֹ לַאֲוִיר הַתַּנּוּר, טָהוֹר {ל}, וְטָהוֹר הַתַּנּוּר. הֱקִיאָהּ אוֹ בְלָעָהּ, מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד. וּכְשֶׁהִיא בְתוֹךְ פִּיו, עַד שֶׁיִּבְלָעֶנָּה, טָהוֹר:

 ר"ע מברטנורה  וְהִיא בְּבֵית הַבְּלִיעָה. כָּל זְמַן שֶׁהִיא בְּבֵית הַבְּלִיעָה מְטַמֵּא בְגָדִים, וְכֵיוָן דִּמְטַמֵּא בְגָדִים מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד: הִכְנִיס רֹאשׁוֹ בַּאֲוִיר הַתַּנּוּר. אַף עַל גַּב דְּקַיְמָא נִבְלַת הָעוֹף עִם בֵּית הַבְּלִיעָה שֶׁל הָאוֹכֵל הַכֹּל בַּאֲוִיר הַתַּנּוּר, אֲפִלּוּ הָכִי טָהוֹר, דְּמֵחֲמַת הָאָדָם לֹא מִטַּמֵּא הַתַּנּוּר, דְּכָל הַמְטַמְּאִים בְּגָדִים אֵין מְטַמְּאִין אָדָם וּכְלֵי חֶרֶס {לא}. וּמֵחֲמַת נִבְלַת הָעוֹף לֹא מִטַּמֵּא, דְּאֵין לָהּ טֻמְאָה אֶלָּא לָאָדָם בְּבֵית הַבְּלִיעָה: הֱקִיאָהּ אוֹ שֶׁבְּלָעָהּ. הַיְנוּ פֵּרַשׁ. דְּמִשֶּׁיָּרְדָה לַבֶּטֶן אֵין לָהּ טֻמְאָה, וּכְשֶׁהִיא לְתוֹךְ פִּיו אֵין לָהּ טֻמְאָה, דְּלֹא יֹאכַל כְּתִיב (וַיִּקְרָא כב), אֲכִילָה בְּבֵית הַבְּלִיעָה: (ר"ע מברטנורה).

 עיקר תוי"ט  {ל} טָהוֹר. הָאָדָם שֶׁהַנְּבֵלָה בְּפִיו וַעֲדַיִן לֹא בְלָעָהּ. וְקָא מַשְׁמַע לָן דְּלֹא תֵימָא דְּבִהְיוֹתָהּ נִסְתֶּרֶת וְנִכְנֶסֶת לַאֲוִיר הַתַּנּוּר תִּהְיֶה כְמוֹ בְלוּעָה, וְטָהוֹר הַתַּנּוּר, מַיְרֵי כְשֶׁהִיא בְלוּעָה. הָרַמְבַּ"ם: {לא} כְּלוֹמַר, דְּכָל אֵלּוּ דְַּאֲמַרַן דִּמְטַמְּאִין שְׁנַיִם מִשּׁוּם בְּגָדִים, כָּל אֵלּוּ אֵין כוּ': (עיקר תוי"ט).

פרק ה - משנה י

הַנּוֹגֵעַ בְּשֶׁרֶץ וּבְשִׁכְבַת זֶרַע, וּבִטְמֵא מֵת, וּבִמְצֹרָע בִּימֵי סָפְרוֹ, וּבְמֵי חַטָּאת שֶׁאֵין בָּהֶם כְּדֵי הַזָּיָה {לב}, וּבִנְבֵלָה, וּבְמֶרְכָּב, מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד. זֶה הַכְּלָל, כֹּל הַנּוֹגֵעַ בְּאַחַד מִכָּל אֲבוֹת הַטֻּמְאוֹת שֶׁבַּתּוֹרָה, מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד, חוּץ מִן הָאָדָם. פֵּרַשׁ, מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד:

 ר"ע מברטנורה  וּבִנְבֵלָה וּבְמֶרְכָּב. כָּל הָנֵי בְּמַשָּׂא בִּלְבַד מְטַמְּאִין בְּגָדִים וְלֹא בְמַגָּע, הִלְכָּךְ אָדָם הַנּוֹגֵעַ בָּהֶן אַף עַל פִּי שֶׁלֹּא פֵרַשׁ אֵינוֹ אֶלָּא רִאשׁוֹן, וּמְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד: כֹּל הַנּוֹגֵעַ. כְּגוֹן אֳכָלִים וְכֵלִים {לג}. דַּאֲפִלּוּ בְאוֹתָן שֶׁאָדָם הַנּוֹגֵעַ בָּהֶן מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד, כֵּלִים הַנּוֹגְעִים בָּהֶן מְטַמְּאִים אֶחָד וּפוֹסְלִין אֶחָד וַאֲפִלּוּ בִשְׁעַת מַגָּעוֹ: חוּץ מִן הָאָדָם. דְּיֵשׁ מֵהֶן דְּאִם נָגַע בָּהֶן אוֹ נָשָׂא דִּמְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד, כִּי הָנֵי דְַאֲמַרַן לְעֵיל: פֵּרַשׁ. אֲפִלּוּ אָדָם, בְּכֻלְּהוּ מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד, דְּכֵיוָן דְּפֵרַשׁ אֵינוֹ אֶלָּא רִאשׁוֹן: (ר"ע מברטנורה).

 עיקר תוי"ט  {לב} שֶׁאֵין כוּ'. דְּאִלּוּ יֵשׁ בָּהֶם, אַף הַנּוֹגֵעַ בָּהֶם מְטַמֵּא בְגָדִים, לְפִי שֶׁאִי אֶפְשָׁר שֶׁיִּגַּע בַּמַּיִם שֶׁלֹּא יַסִּיט אוֹתָם, וַהֲרֵי הוּא מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד. אֲבָל אִם קָרְשׁוּ וְנַעֲשׂוּ כִגְלִיד דְּאֶפְשָׁר לִגַּע בְּלֹא הֶסֵּט, תָּנֵי בְפֶרֶק קַמָּא דְכֵלִים מִשְׁנָה ב' חֲשׂוּכֵי בְגָדִים בְּמַגָּע. וּכְמוֹ שֶׁכָּתַב הָרַ"ב בְּמִשְׁנָה ח'. וְהַתַּנָּא הָתָם וְהָכָא מִלְּתָא דִפְסִיקָא נָקַט: {לג} וְאֵין הַלָּשׁוֹן מַשְׁמַע כֵּן, דְּהָכִי הֲוָה לֵיהּ לְמִתְנֵי, בַּמֶּה דְבָרִים אֲמוּרִים בְּאָדָם, אֲבָל אֳכָלִים וְכֵלִים הַנּוֹגְעִים בְּאֶחָד כוּ'. וְהָרַמְבַּ"ם פֵּרֵשׁ, כָּל הַנּוֹגֵעַ בִּשְׁאָר אֲבוֹת הַטֻּמְאוֹת חוּץ מֵאֵלּוּ שֶׁמָּנָה עַד הֵנָּה [שֶׁמְּטַמְּאִין שְׁנַיִם כוּ']. וְאָמְרוֹ חוּץ מִן הָאָדָם, רָצָה לוֹמַר הַמֵּת, שֶׁגַּם הוּא לֹא נִמְנָה כָאן, וְנוֹגֵעַ מְטַמֵּא שְׁנַיִם כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (עיקר תוי"ט).

פרק ה - משנה יא

בַּעַל קֶרִי, כְּמַגַּע שֶׁרֶץ. וּבוֹעֵל נִדָּה, כִּטְמֵא מֵת, אֶלָּא שֶׁחָמוּר מִמֶּנּוּ בּוֹעֵל נִדָּה {לו}, שֶׁהוּא מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב טֻמְאָה קַלָּה לְטַמֵּא אֳכָלִין וּמַשְׁקִין:

 ר"ע מברטנורה  בַּעַל קֶרִי כְּמַגַּע שֶׁרֶץ. כְּשֵׁם שֶׁנּוֹגֵעַ בַּשֶּׁרֶץ רִאשׁוֹן לַטֻּמְאָה, כָּךְ בַּעַל קֶרִי רִאשׁוֹן לַטֻּמְאָה, וְטֻמְאָתוֹ שָׁוָה לְטֻמְאַת הַנּוֹגֵעַ בְּשִׁכְבַת זֶרַע {לד}: וּבוֹעֵל נִדָּה כִּטְמֵא מֵת. שֶׁהוּא אַב הַטֻּמְאָה וְטָמֵא טֻמְאַת שִׁבְעָה {לה}: שֶׁהוּא מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב. וּמֵת עַצְמוֹ וְכָל שֶׁכֵּן טְמֵא מֵת אֵינוֹ מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב, לֹא טֻמְאָה קַלָּה וְלֹא טֻמְאָה חֲמוּרָה, דִּכְתִיב (וַיִּקְרָא טו) וְכָל הַכְּלִי אֲשֶׁר יֵשֵׁב עָלָיו, הַזָּב, וְלֹא טְמֵא מֵת: (ר"ע מברטנורה).

 עיקר תוי"ט  {לד} וּמִשּׁוּם דְּנוֹגֵעַ לֹא כְתִיב בְּהֶדְיָא תָּלֵי לֵיהּ בְּשֶׁרֶץ. אֲבָל בַּגְּמָרָא מַסִּיק דְּמַה שֶּׁרֶץ מְטַמֵּא בְאֹנֶס [אִם נָפַל עָלָיו שֶׁרֶץ מֵת], אַף בַּעַל קֶרִי: {לה} אֲבָל בַּגְּמָרָא, דְּזֶהוּ מִלְּתָא דִפְשִׁיטָא הוּא, דִּקְרָא כְתִיב. וּלְמַחֲנוֹתָם אִצְטְרִיךְ, סָלְקָא דַעְתָּךְ אֲמֵינָא וּתְהִי נִדָּתָהּ עָלָיו כְּתִיב, דְּיִשְׁתַּלַּח חוּץ לִשְׁתֵּי מַחֲנוֹת כְּנִדָּה, קָא מַשְׁמַע לָן. וּבְבַעַל קֶרִי כְּמַגַּע שֶׁרֶץ לֵיכָּא לְמֵימַר הָכִי, דְּבַעַל קֶרִי חוּץ לִשְׁתֵּי מַחֲנוֹת, וְשֶׁרֶץ שָׁרֵי בְּמַחֲנֵה לְוִיָּה. אִי נַמִּי, הָכָא סֵיפָא אִצְטְרִיךְ לֵיהּ אֶלָּא כוּ': {לו} שֶׁחָמוּר כוּ'. וְקָשֶׁה, הָא חָמוּר נַמִּי טְמֵא מֵת מִבּוֹעֵל נִדָּה, שֶׁטָּעוּן הַזָּאָה שְׁלִישִׁי וּשְׁבִיעִי וְחֶרֶב הֲרֵי הוּא כְּחָלָל. וְשֶׁמָּא בַסֵּיפָא דִבְרַיְתָא חֲשִׁיב לְהוּ. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב וְתוֹסָפוֹת חֲדָשִׁים: (עיקר תוי"ט).

פרק ה - משנה יב

אֵלּוּ פוֹסְלִים אֶת הַתְּרוּמָה. הָאוֹכֵל אֹכֶל רִאשׁוֹן, וְהָאוֹכֵל אֹכֶל שֵׁנִי, וְהַשּׁוֹתֶה מַשְׁקִין טְמֵאִין, וְהַבָּא רֹאשׁוֹ וְרֻבּוֹ בְּמַיִם שְׁאוּבִין, וְטָהוֹר שֶׁנָּפְלוּ עַל רֹאשׁוֹ וְעַל רֻבּוֹ שְׁלשָׁה לֻגִּין מַיִם שְׁאוּבִין, וְהַסֵּפֶר, וְהַיָּדַיִם, וּטְבוּל יוֹם, וְהָאֳכָלִים וְהַכֵּלִים שֶׁנִּטְמְאוּ בְמַשְׁקִים:

 ר"ע מברטנורה  אֵלּוּ פוֹסְלִים אֶת הַתְּרוּמָה. מִדְּרַבָּנָן. וְכֻלְּהוּ מִשְּׁמֹנָה עָשָׂר דָּבָר שֶׁגָּזְרוּ בּוֹ בַיּוֹם. חוּץ מִטְּבוּל יוֹם דְּמִדְּאוֹרַיְתָא הוּא: הָאוֹכֵל אֹכֶל. שֶׁהוּא רִאשׁוֹן לַטֻּמְאָה אוֹ שֵׁנִי לַטֻּמְאָה, גָּזְרוּ שֶׁיְּהֵא גוּפוֹ שֵׁנִי לַטֻּמְאָה, וּפוֹסֵל אֶת הַתְּרוּמָה בְּמַגָּעוֹ, שֶׁשֵּׁנִי פוֹסֵל בַּתְּרוּמָה: וְהַשּׁוֹתֶה מַשְׁקִים טְמֵאִים. נַעֲשָׂה גַם כֵּן שֵׁנִי לַטֻּמְאָה וּפוֹסֵל אֶת הַתְּרוּמָה. וְטַעֲמָא דְגַזּוּר רַבָּנָן בְּהָנֵי, דְּזִמְנִין דְּאָכִיל אֳכָלִים טְמֵאִים, וְשָׁדֵי מַשְׁקִים דִּתְרוּמָה בְּפֻמֵּיהּ בְּעוֹד שֶׁהָאֳכָלִים טְמֵאִים בְּפִיו, וּפָסֵיל לְהוּ. וְכֵן נַמִּי זִמְנִין דְּשָׁתֵי מַשְׁקִין טְמֵאִין, וְשָׁדֵי אֳכָלִים דִּתְרוּמָה בְּפֻמֵּיהּ בְּעוֹדָן בְּפִיו, וּפָסֵיל לְהוּ: וְהַבָּא רֹאשׁוֹ וְרֻבּוֹ. לְאַחַר שֶׁטָּבַל בּוֹ בַיּוֹם: בְּמַיִם שְׁאוּבִין. לְפִי שֶׁהָיוּ טוֹבְלִים בְּמֵי מְעָרוֹת סְרוּחִים וְהָיוּ נוֹתְנִין עֲלֵיהֶם אַחַר כָּךְ מַיִם שְׁאוּבִין לְהַעֲבִיר סִרְחוֹן הַמַּיִם, הִתְחִילוּ וַעֲשָׂאוּם קֶבַע לוֹמַר לֹא מֵי הַמְּעָרוֹת מְטַהֲרִים אֶלָּא מַיִם שְׁאוּבִים מְטַהֲרִים, עָמְדוּ וְגָזְרוּ עֲלֵיהֶם טֻמְאָה, דִּלְמָא אָתוּ לְבַטּוֹלֵי תּוֹרַת מִקְוֶה וְטָבְלֵי בִשְׁאוּבִים, וּכְדֵי לְקַיֵּם גְּזֵרָה זוֹ לְהַעֲמִידָהּ, הוֹסִיפוּ וְגָזְרוּ עַל טָהוֹר גָּמוּר שֶׁנָּפְלוּ עַל רֹאשׁוֹ [וְרֻבּוֹ] שְׁלֹשָׁה לֻגִּין {לז} מַיִם שְׁאוּבִין, שֶׁיִּהְיֶה פוֹסֵל אֶת הַתְּרוּמָה עַד שֶׁיִּטְבֹּל: וְהַסֵּפֶר. כָּל כִּתְבֵי הַקֹּדֶשׁ פּוֹסְלִים אֶת הַתְּרוּמָה בְּמַגָּע. לְפִי שֶׁבַּתְּחִלָּה הָיוּ מַצְנִיעִים אֳכָלִים דִּתְרוּמָה אֵצֶל סֵפֶר תּוֹרָה, אָמְרֵי הַאי קֹדֶשׁ וְהַאי קֹדֶשׁ, כֵּיוָן דְּחָזוּ דְּקָא אָתוּ סְפָרִים לִידֵי פְסֵידָא שֶׁהָעַכְבָּרִים הַמְּצוּיִין אֵצֶל הָאֳכָלִים הָיוּ מַפְסִידִים אֶת הַסְּפָרִים, גָּזְרוּ עַל תּוֹרָה נְבִיאִים וּכְתוּבִים שֶׁיִּהְיוּ פוֹסְלִים בְּמַגָּעָן אֶת הַתְּרוּמָה: וּטְבוּל יוֹם. פּוֹסֵל אֶת הַתְּרוּמָה מִן הַתּוֹרָה עַד שֶׁיַּעֲרִיב שִׁמְשׁוֹ {לח}. וְאֵינוֹ מִשְּׁמֹנָה עָשָׂר דָּבָר שֶׁגָּזְרוּ בּוֹ בַיּוֹם: וְהָאֳכָלִים. שֶׁנִּטְמְאוּ בְמַשְׁקִים. חֲכָמִים גָּזְרוּ עַל הַמַּשְׁקִים שֶׁנִּטְמְאוּ בְּשֵׁנִי לַטֻּמְאָה, לִהְיוֹת תְּחִלָּה, וִיטַמְּאוּ אֶת הָאֳכָלִים, גְּזֵרָה מִשּׁוּם מַשְׁקִין הַבָּאִין מֵחֲמַת שֶׁרֶץ שֶׁהֵם תְּחִלָּה מִדְּאוֹרַיְתָא. וְלָמָּה לֹא גָזְרוּ עַל הָאֳכָלִים נַמִּי לִהְיוֹת תְּחִלָּה גְּזֵרָה מִשּׁוּם אֳכָלִים הַבָּאִים מֵחֲמַת שֶׁרֶץ, לְפִי שֶׁהָאֳכָלִים הֵן צְרִיכִין הֶכְשֵׁר וְאֵין מְקַבְּלִין טֻמְאָה אֶלָּא עַל יְדֵי בִּיאַת מַיִם, אֲבָל הַמַּשְׁקִין הֵן עֲלוּלִים לְקַבֵּל טֻמְאָה וְאֵין צְרִיכִין הֶכְשֵׁר וְתִקּוּן, וְטֻמְאָתָן יוֹתֵר מְצוּיָה: וְכֵלִים שֶׁנִּטְמְאוּ בְמַשְׁקִין. דְּמִדְּאוֹרַיְתָא אֲפִלּוּ נִטְמְאוּ הַמַּשְׁקִים בְּשֶׁרֶץ אֵין מְטַמְּאִין כֵּלִים, דְּאֵין אָדָם וְכֵלִים מְקַבְּלִין טֻמְאָה אֶלָּא מֵאַב הַטֻּמְאָה, וְרַבָּנָן גָּזְרוּ אֲפִלּוּ עַל מַשְׁקִין שֶׁנִּטְמְאוּ מֵחֲמַת יָדַיִם שֶׁהֵן שְׁנִיּוֹת, שֶׁיְּטַמְּאוּ כֵלִים, גְּזֵרָה מִשּׁוּם מַשְׁקֶה דְזָב וְזָבָה, כְּגוֹן רֻקָּן וּמֵימֵי רַגְלֵיהֶן שֶׁהֵן אַב הַטֻּמְאָה וּמְטַמְּאִין כֵּלִים מִדְּאוֹרַיְתָא: (ר"ע מברטנורה).

 עיקר תוי"ט  {לז} דְּמִסְתַּבְּרָא דְּהַבָּא כוּ' נַמִּי בִשְׁלֹשָׁה לֻגִּין הוּא, כֵּיוָן דְּהָא בְהָא תַּלְיָא. וְכֵן כָּתַב הָרַמְבַּ"ם: {לח} דִּכְתִיב וּבָא הַשֶּׁמֶשׁ וְטָהֵר. גְּמָרָא: (עיקר תוי"ט).
