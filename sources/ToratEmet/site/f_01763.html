רות פרק-א

{א}  וַיְהִ֗י בִּימֵי֙ שְׁפֹ֣ט הַשֹּׁפְטִ֔ים וַיְהִ֥י רָעָ֖ב בָּאָ֑רֶץ וַיֵּ֨לֶךְ אִ֜ישׁ מִבֵּ֧ית לֶ֣חֶם יְהוּדָ֗ה לָגוּר֙ בִּשְׂדֵ֣י מוֹאָ֔ב ה֥וּא וְאִשְׁתּ֖וֹ וּשְׁנֵ֥י בָנָֽיו:

 רש"י  ויהי בימי שפוט השופטים. לפני מלוך מלך שאול שהיו הדורות מתפרנסים על ידי שופטים ובימי אבצן היה שאמרו רבותינו אבצן זה בועז : וילך איש. עשיר גדול היה ופרנס הדור ויצא מארץ ישראל לחוצה לארץ מפני צרות העין שהיתה עינו צרה בעניי' הבאים לדוחקו לכך נענש : (רש"י)
{ב}  וְשֵׁ֣ם הָאִ֣ישׁ אֱֽלִימֶ֡לֶךְ וְשֵׁם֩ אִשְׁתּ֨וֹ נָעֳמִ֜י וְשֵׁ֥ם שְׁנֵֽי-בָנָ֣יו | מַחְל֤וֹן וְכִלְיוֹן֙ אֶפְרָתִ֔ים מִבֵּ֥ית לֶ֖חֶם יְהוּדָ֑ה וַיָּבֹ֥אוּ שְׂדֵי-מוֹאָ֖ב וַיִּֽהְיוּ-שָֽׁם:

 רש"י  אפרתים. חשובים וכן (שמואל א א) בן תוחו בן צוף אפרתי אבגי''נוס, ראה חשיבותו שהרי השיא עגלון מלך מואב את בתו למחלון דאמר מר רות בתו של עגלון היתה, ד''א אפרתים בית לחם קרויה אפרת : (רש"י)
{ג}  וַיָּ֥מָת אֱלִימֶ֖לֶךְ אִ֣ישׁ נָעֳמִ֑י וַתִּשָּׁאֵ֥ר הִ֖יא וּשְׁנֵ֥י בָנֶֽיהָ:

 רש"י  איש נעמי. למה נאמר מכאן אמרו אין איש מת אלא לאשתו (ואמר איש נעמי כלומר לפי שהוא היה איש נעמי ושולט עליה והיא טפלה לו לכן פגעה בו מדת הדין ולא בה) : (רש"י)
{ד}  וַיִּשְׂא֣וּ לָהֶ֗ם נָשִׁים֙ מֹֽאֲבִיּ֔וֹת שֵׁ֤ם הָֽאַחַת֙ עָרְפָּ֔ה וְשֵׁ֥ם הַשֵּׁנִ֖ית ר֑וּת וַיֵּ֥שְׁבוּ שָׁ֖ם כְּעֶ֥שֶׂר שָׁנִֽים: {ה}  וַיָּמ֥וּתוּ גַם-שְׁנֵיהֶ֖ם מַחְל֣וֹן וְכִלְי֑וֹן וַתִּשָּׁאֵר֙ הָֽאִשָּׁ֔ה מִשְּׁנֵ֥י יְלָדֶ֖יהָ וּמֵאִישָֽׁהּ:

 רש"י  גם שניהם. מהו גם בתחלה לקו בממונם ומתו גמליהם ומקניהם אח''כ מתו גם הם : (רש"י)
{ו}  וַתָּ֤קָם הִיא֙ וְכַלֹּתֶ֔יהָ וַתָּ֖שָׁב מִשְּׂדֵ֣י מוֹאָ֑ב כִּ֤י שָֽׁמְעָה֙ בִּשְׂדֵ֣ה מוֹאָ֔ב כִּֽי-פָקַ֤ד יְהוָה֙ אֶת-עַמּ֔וֹ לָתֵ֥ת לָהֶ֖ם לָֽחֶם: {ז}  וַתֵּצֵ֗א מִן-הַמָּקוֹם֙ אֲשֶׁ֣ר הָיְתָה-שָׁ֔מָּה וּשְׁתֵּ֥י כַלֹּתֶ֖יהָ עִמָּ֑הּ וַתֵּלַ֣כְנָה בַדֶּ֔רֶךְ לָשׁ֖וּב אֶל-אֶ֥רֶץ יְהוּדָֽה:

 רש"י  ותצא מן המקום. למה נאמר הרי כבר נאמר ותשב משדה מואב ומהיכן תשוב אם לא תצא מן המקום שהיתה שם אלא מגיד שיציאת צדיק מן המקום ניכרת ועושה רושם פנה זיוה פנה הדרה פנה שבחה של עיר וכן (בראשית כה) ויצא יעקב מבאר שבע : (רש"י)
{ח}  וַתֹּ֤אמֶר נָעֳמִי֙ לִשְׁתֵּ֣י כַלֹּתֶ֔יהָ לֵ֣כְנָה שֹּׁ֔בְנָה אִשָּׁ֖ה לְבֵ֣ית אִמָּ֑הּ (יעשה) יַ֣עַשׂ יְהוָ֤ה עִמָּכֶם֙ חֶ֔סֶד כַּאֲשֶׁ֧ר עֲשִׂיתֶ֛ם עִם-הַמֵּתִ֖ים וְעִמָּדִֽי: {ט}  יִתֵּ֤ן יְהוָה֙ לָכֶ֔ם וּמְצֶ֣אןָ מְנוּחָ֔ה אִשָּׁ֖ה בֵּ֣ית אִישָׁ֑הּ וַתִּשַּׁ֣ק לָהֶ֔ן וַתִּשֶּׂ֥אנָה קוֹלָ֖ן וַתִּבְכֶּֽינָה: {י}  וַתֹּאמַ֖רְנָה-לָּ֑הּ כִּי-אִתָּ֥ךְ נָשׁ֖וּב לְעַמֵּֽךְ: {יא}  וַתֹּ֤אמֶר נָעֳמִי֙ שֹׁ֣בְנָה בְנֹתַ֔י לָ֥מָּה תֵלַ֖כְנָה עִמִּ֑י הַֽעֽוֹד-לִ֤י בָנִים֙ בְּֽמֵעַ֔י וְהָי֥וּ לָכֶ֖ם לַאֲנָשִֽׁים: {יב}  שֹׁ֤בְנָה בְנֹתַי֙ לֵ֔כְןָ כִּ֥י זָקַ֖נְתִּי מִהְי֣וֹת לְאִ֑ישׁ כִּ֤י אָמַ֙רְתִּי֙ יֶשׁ-לִ֣י תִקְוָ֔ה גַּ֣ם הָיִ֤יתִי הַלַּ֙יְלָה֙ לְאִ֔ישׁ וְגַ֖ם יָלַ֥דְתִּי בָנִֽים:

 רש"י  כי זקנתי מהיות לאיש. שאנשא לו ואוליד בנים ותנשאו להם שאינם אסורים לכם ואינכם אסורים להם משום אשת אחיו שלא היה בעולמו שאינה זקוקה ליבם לפי שלא היו למחלון וכליון קדושין בהן שנכריות היו ולא נתגיירו ועכשיו הן באות להתגייר כמו שנאמר כי אתך נשוב לעמך מעתה נהיה לעם אחד : כי אמרתי יש לו תקוה. כי אפילו אמר לי לבי יש לי תקוה לינשא עוד וללדת בנים : וגם הייתי הלילה לאיש. ויותר מכן אפילו הריתי הלילה זכרים : וגם ילדתי בנים. או אפילו כבר ילדתי בנים : (רש"י)
{יג}  הֲלָהֵ֣ן | תְּשַׂבֵּ֗רְנָה עַ֚ד אֲשֶׁ֣ר יִגְדָּ֔לוּ הֲלָהֵן֙ תֵּֽעָגֵ֔נָה לְבִלְתִּ֖י הֱי֣וֹת לְאִ֑ישׁ אַ֣ל בְּנֹתַ֗י כִּֽי-מַר-לִ֤י מְאֹד֙ מִכֶּ֔ם כִּֽי-יָצְאָ֥ה בִ֖י יַד-יְהוָֽה:

 רש"י  הלהן תשברנה. בתמיה שמא להם תצפנה עד אשר יגדלו לשון (תהלים קמו) שברו על ה' אלהיו : תעגנה. לשון אסור כלא כמו עג עוגה ועמד בתוכה ויש פותרין לשון עיגון ולא יתכן שאם כן היה לו לינקד הנו''ן דגש או לכתוב שני נוני''ן : כי יצאה בי יד ה'. אמר רבי לוי כל מקום שנאמר יד ה' מכת דבר היא ובנין אב לכולם (שמות ט) הנה יד ה' הויה : (רש"י)
{יד}  וַתִּשֶּׂ֣נָה קוֹלָ֔ן וַתִּבְכֶּ֖ינָה ע֑וֹד וַתִּשַּׁ֤ק עָרְפָּה֙ לַחֲמוֹתָ֔הּ וְר֖וּת דָּ֥בְקָה בָּֽהּ: {טו}  וַתֹּ֗אמֶר הִנֵּה֙ שָׁ֣בָה יְבִמְתֵּ֔ךְ אֶל-עַמָּ֖הּ וְאֶל-אֱלֹהֶ֑יהָ שׁ֖וּבִי אַחֲרֵ֥י יְבִמְתֵּֽךְ:

 רש"י  הנה שבה יבמתך. זה טעמו למעלה תחת השי''ן לפי שהוא לשון עבר (אסתר ב) ובבקר היא שבה טעמו למטה בבי''ת לפי שהוא לשון הווה וכן כל כיוצא בהם : (רש"י)
{טז}  וַתֹּ֤אמֶר רוּת֙ אַל-תִּפְגְּעִי-בִ֔י לְעָזְבֵ֖ךְ לָשׁ֣וּב מֵאַחֲרָ֑יִךְ כִּ֠י אֶל-אֲשֶׁ֨ר תֵּלְכִ֜י אֵלֵ֗ךְ וּבַאֲשֶׁ֤ר תָּלִ֙ינִי֙ אָלִ֔ין עַמֵּ֣ךְ עַמִּ֔י וֵאלֹהַ֖יִךְ אֱלֹהָֽי:

 רש"י  אל תפגע בי. אל תפצרי בי : כי אל אשר תלכי אלך. מכאן אמרו רבותינו ז''ל גר שבא להתגייר מודיעים לו מקצת עונשים שאם בא לחזור בו יחזור שמתוך דברים של רות אתה למד שאמרה לה נעמי אסור לנו לצאת חוץ לתחום בשבת א''ל באשר תלכי אלך, אסור לנו להתייחד נקבה עם זכר שאינה אישה אמרה לה באשר תליני אלין, עמנו מובדלים משאר עמים בשש מאות ושלש עשרה מצות עמך עמי, אסור לנו ע''א אלהיך אלהי, ארבע מיתות נמסרו לבית דין באשר תמותי אמות, שני קברים נמסרו לבית דין אחד לנסקלין ונשרפין ואחד לנהרגין ונחנקין אמרה לה ושם אקבר : (רש"י)
{יז}  בַּאֲשֶׁ֤ר תָּמ֙וּתִי֙ אָמ֔וּת וְשָׁ֖ם אֶקָּבֵ֑ר כֹּה֩ יַעֲשֶׂ֨ה יְהוָ֥ה לִי֙ וְכֹ֣ה יֹסִ֔יף כִּ֣י הַמָּ֔וֶת יַפְרִ֖יד בֵּינִ֥י וּבֵינֵֽךְ:

 רש"י  כה יעשה ה' לי. כאשר התחיל להרע שיצאה בי ידו להמית אישי ולירד מנכסי : וכה יוסיף. אם יפריד ביני וביניך כי אם המות : (רש"י)
{יח}  וַתֵּ֕רֶא כִּֽי-מִתְאַמֶּ֥צֶת הִ֖יא לָלֶ֣כֶת אִתָּ֑הּ וַתֶּחְדַּ֖ל לְדַבֵּ֥ר אֵלֶֽיהָ:

 רש"י  ותחדל לדבר אליה. מכאן אמרו אין מרבין עליו ואין מדקדקין עליו : (רש"י)
{יט}  וַתֵּלַ֣כְנָה שְׁתֵּיהֶ֔ם עַד-בֹּאָ֖נָה בֵּ֣ית לָ֑חֶם וַיְהִ֗י כְּבֹאָ֙נָה֙ בֵּ֣ית לֶ֔חֶם וַתֵּהֹ֤ם כָּל-הָעִיר֙ עֲלֵיהֶ֔ן וַתֹּאמַ֖רְנָה הֲזֹ֥את נָעֳמִֽי:

 רש"י  ותלכנה שתיהם. אמר רבי אבהו בא וראה כמה חביבים הגרים לפני הקב''ה כיון שנתנה דעתה להתגייר השוה אותה הכתוב לנעמי : ותהם כל העיר. נעשית הומיה כל העיר כולם נתקבצו לקבור אשתו של בועז שמתה בו ביום : הזאת נעמי. ה''א נקודה חטף מפני שהיא בתמיה הזאת נעמי שרגילה לצאת בצבים ובפרדים חזיתם מה עלתה לה על אשר יצאתה לחוצה לארץ : (רש"י)
{כ}  וַתֹּ֣אמֶר אֲלֵיהֶ֔ן אַל-תִּקְרֶ֥אנָה לִ֖י נָעֳמִ֑י קְרֶ֤אןָ לִי֙ מָרָ֔א כִּי-הֵמַ֥ר שַׁדַּ֛י לִ֖י מְאֹֽד: {כא}  אֲנִי֙ מְלֵאָ֣ה הָלַ֔כְתִּי וְרֵיקָ֖ם הֱשִׁיבַ֣נִי יְהוָ֑ה לָ֣מָּה תִקְרֶ֤אנָה לִי֙ נָעֳמִ֔י וַֽיהוָה֙ עָ֣נָה בִ֔י וְשַׁדַּ֖י הֵ֥רַֽע לִֽי:

 רש"י  מלאה הלכתי. בעושר ובנים, ד''א שהיתה מעוברת : ענה בי. העיד עלי שהרשעתי לפניו, ד''א ענה בי מדת הדין כמו וענה גאון ישראל : (רש"י)
{כב}  וַתָּ֣שָׁב נָעֳמִ֗י וְר֨וּת הַמּוֹאֲבִיָּ֤ה כַלָּתָהּ֙ עִמָּ֔הּ הַשָּׁ֖בָה מִשְּׂדֵ֣י מוֹאָ֑ב וְהֵ֗מָּה בָּ֚אוּ בֵּ֣ית לֶ֔חֶם בִּתְחִלַּ֖ת קְצִ֥יר שְׂעֹרִֽים:

 רש"י  בתחלת קציר שעורים. בקצירת העומר הכתוב מדבר : (רש"י)

רות פרק-ב

{א}  וּֽלְנָעֳמִ֞י (מידע) מוֹדַ֣ע לְאִישָׁ֗הּ אִ֚ישׁ גִּבּ֣וֹר חַ֔יִל מִמִּשְׁפַּ֖חַת אֱלִימֶ֑לֶךְ וּשְׁמ֖וֹ בֹּֽעַז:

 רש"י  מודע. קרוב בן אחיו של אלימלך היה אמרו רבותינו ז''ל אלימלך ושלמון אבי בועז ופלוני אלמוני הגואל ואבי נעמי כולם בני נחשון בן עמינדב היו ולא הועילו להם זכות אבותם בצאתם מהארץ לחוצה לארץ : (רש"י)
{ב}  וַתֹּאמֶר֩ ר֨וּת הַמּוֹאֲבִיָּ֜ה אֶֽל-נָעֳמִ֗י אֵֽלְכָה-נָּ֤א הַשָּׂדֶה֙ וַאֲלַקֳטָ֣ה בַשִּׁבֳּלִ֔ים אַחַ֕ר אֲשֶׁ֥ר אֶמְצָא-חֵ֖ן בְּעֵינָ֑יו וַתֹּ֥אמֶר לָ֖הּ לְכִ֥י בִתִּֽי:

 רש"י  אלכה נא השדה. לאחד משדות אנשי העיר אחרי אחד מהם אשר אמצא חן בעיניו שלא יגער בי : ואלקטה בשבלים אחר אשר אמצא חן בעיניו. אחר מי אשר אמצא חן בעיניו : (רש"י)

 רש"י  ותבא ותלקט בשדה. מצינו במדרש רות עד לא אזלת אתת שהוא אומר ותבא ואחר ותלקט אלא שהיתה מסמנת הדרכים קודם שנכנסה לשדה והלכה ובאה וחזרה לעיר כדי לעשות סימנים וציונים שלא תטעה בשבילין ותדע לשוב : ויקר מקרה לבא בחלקת השדה אשר לבועז : (רש"י)
{ד}  וְהִנֵּה-בֹ֗עַז בָּ֚א מִבֵּ֣ית לֶ֔חֶם וַיֹּ֥אמֶר לַקּוֹצְרִ֖ים יְהוָ֣ה עִמָּכֶ֑ם וַיֹּ֥אמְרוּ ל֖וֹ יְבָרֶכְךָ֥ יְהוָֽה: {ה}  וַיֹּ֤אמֶר בֹּ֙עַז֙ לְנַעֲר֔וֹ הַנִּצָּ֖ב עַל-הַקּֽוֹצְרִ֑ים לְמִ֖י הַנַּעֲרָ֥ה הַזֹּֽאת:

 רש"י  למי הנערה הזאת. וכי דרכו של בועז לשאול בנשים אלא דברי צניעות וחכמה ראה בה שתי שבלים לקטה שלשה אינה לקטה והיתה מלקטת עומדות מעומד ושוכבות מיושב כדי שלא תשחה : (רש"י)
{ו}  וַיַּ֗עַן הַנַּ֛עַר הַנִּצָּ֥ב עַל-הַקּוֹצְרִ֖ים וַיֹּאמַ֑ר נַעֲרָ֤ה מֽוֹאֲבִיָּה֙ הִ֔יא הַשָּׁ֥בָה עִֽם-נָעֳמִ֖י מִשְּׂדֵ֥ה מוֹאָֽב:

 רש"י  השבה עם נעמי. הטעם למעלה תחת השי''ן לפי שהוא לשון עבר ואינה לשון פועלת : (רש"י)
{ז}  וַתֹּ֗אמֶר אֲלַקֳטָה-נָּא֙ וְאָסַפְתִּ֣י בָֽעֳמָרִ֔ים אַחֲרֵ֖י הַקּוֹצְרִ֑ים וַתָּב֣וֹא וַֽתַּעֲמ֗וֹד מֵאָ֤ז הַבֹּ֙קֶר֙ וְעַד-עַ֔תָּה זֶ֛ה שִׁבְתָּ֥הּ הַבַּ֖יִת מְעָֽט:

{ח}  וַיֹּאמֶר֩ בֹּ֨עַז אֶל-ר֜וּת הֲל֧וֹא שָׁמַ֣עַתְּ בִּתִּ֗י אַל-תֵּלְכִי֙ לִלְקֹט֙ בְּשָׂדֶ֣ה אַחֵ֔ר וְגַ֛ם לֹ֥א תַעֲבוּרִ֖י מִזֶּ֑ה וְכֹ֥ה תִדְבָּקִ֖ין עִם-נַעֲרֹתָֽי: {ט}  עֵינַ֜יִךְ בַּשָּׂדֶ֤ה אֲשֶׁר-יִקְצֹרוּן֙ וְהָלַ֣כְתְּ אַחֲרֵיהֶ֔ן הֲל֥וֹא צִוִּ֛יתִי אֶת-הַנְּעָרִ֖ים לְבִלְתִּ֣י נָגְעֵ֑ךְ וְצָמִ֗ת וְהָלַכְתְּ֙ אֶל-הַכֵּלִ֔ים וְשָׁתִ֕ית מֵאֲשֶׁ֥ר יִשְׁאֲב֖וּן הַנְּעָרִֽים:

 רש"י  וצמית והלכת אל הכלים. ואם תצמאי אל תכלמי מלכת ולשתות מכלי המים אשר ישאבון הנערים : (רש"י)
{י}  וַתִּפֹּל֙ עַל-פָּנֶ֔יהָ וַתִּשְׁתַּ֖חוּ אָ֑רְצָה וַתֹּ֣אמֶר אֵלָ֗יו מַדּוּעַ֩ מָצָ֨אתִי חֵ֤ן בְּעֵינֶ֙יךָ֙ לְהַכִּירֵ֔נִי וְאָנֹכִ֖י נָכְרִיָּֽה: {יא}  וַיַּ֤עַן בֹּ֙עַז֙ וַיֹּ֣אמֶר לָ֔הּ הֻגֵּ֨ד הֻגַּ֜ד לִ֗י כֹּ֤ל אֲשֶׁר-עָשִׂית֙ אֶת-חֲמוֹתֵ֔ךְ אַחֲרֵ֖י מ֣וֹת אִישֵׁ֑ךְ וַתַּֽעַזְבִ֞י אָבִ֣יךְ וְאִמֵּ֗ךְ וְאֶ֙רֶץ֙ מֽוֹלַדְתֵּ֔ךְ וַתֵּ֣לְכִ֔י אֶל-עַ֕ם אֲשֶׁ֥ר לֹא-יָדַ֖עַתְּ תְּמ֥וֹל שִׁלְשֽׁוֹם: {יב}  יְשַׁלֵּ֥ם יְהוָ֖ה פָּעֳלֵ֑ךְ וּתְהִ֨י מַשְׂכֻּרְתֵּ֜ךְ שְׁלֵמָ֗ה מֵעִ֤ם יְהוָה֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁר-בָּ֖את לַחֲס֥וֹת תַּֽחַת-כְּנָפָֽיו: {יג}  וַ֠תֹּאמֶר אֶמְצָא-חֵ֨ן בְּעֵינֶ֤יךָ אֲדֹנִי֙ כִּ֣י נִֽחַמְתָּ֔נִי וְכִ֥י דִבַּ֖רְתָּ עַל-לֵ֣ב שִׁפְחָתֶ֑ךָ וְאָנֹכִי֙ לֹ֣א אֶֽהְיֶ֔ה כְּאַחַ֖ת שִׁפְחֹתֶֽיךָ:

 רש"י  לא אהיה כאחת שפחתך. איני חשובה כאחת מן השפחות שלך : (רש"י)
{יד}  וַיֹּאמֶר֩ לָ֨ה בֹ֜עַז לְעֵ֣ת הָאֹ֗כֶל גֹּ֤שִֽׁי הֲלֹם֙ וְאָכַ֣לְתְּ מִן-הַלֶּ֔חֶם וְטָבַ֥לְתְּ פִּתֵּ֖ךְ בַּחֹ֑מֶץ וַתֵּ֙שֶׁב֙ מִצַּ֣ד הַקּֽוֹצְרִ֔ים וַיִּצְבָּט-לָ֣הּ קָלִ֔י וַתֹּ֥אכַל וַתִּשְׂבַּ֖ע וַתֹּתַֽר:

 רש"י  וטבלת פתך בחומץ. מכאן שהחומץ יפה לשרב : ויצבט לה קלי. ויושט לה ואין לו דמיון במקרא אלא בלשון משנה אחוריים ותוך ובית הצביטה : (רש"י)
{טו}  וַתָּ֖קָם לְלַקֵּ֑ט וַיְצַו֩ בֹּ֨עַז אֶת-נְעָרָ֜יו לֵאמֹ֗ר גַּ֣ם בֵּ֧ין הָֽעֳמָרִ֛ים תְּלַקֵּ֖ט וְלֹ֥א תַכְלִימֽוּהָ: {טז}  וְגַ֛ם שֹׁל-תָּשֹׁ֥לּוּ לָ֖הּ מִן-הַצְּבָתִ֑ים וַעֲזַבְתֶּ֥ם וְלִקְּטָ֖ה וְלֹ֥א תִגְעֲרוּ-בָֽהּ:

 רש"י  וגם של תשלו. שכח תשכחו עשו עצמיכם כאילו אתם שוכחים תרגום כל שגגה שלותא וכן על השל, דבר אחר לשון כי ישל זיתך : צבתים. עמרים קטנים יש דוגמתו בלשון משנה מצאן צבתים או כריכות : (רש"י)
{יז}  וַתְּלַקֵּ֥ט בַּשָּׂדֶ֖ה עַד-הָעָ֑רֶב וַתַּחְבֹּט֙ אֵ֣ת אֲשֶׁר-לִקֵּ֔טָה וַיְהִ֖י כְּאֵיפָ֥ה שְׂעֹרִֽים: {יח}  וַתִּשָּׂא֙ וַתָּב֣וֹא הָעִ֔יר וַתֵּ֥רֶא חֲמוֹתָ֖הּ אֵ֣ת אֲשֶׁר-לִקֵּ֑טָה וַתּוֹצֵא֙ וַתִּתֶּן-לָ֔הּ אֵ֥ת אֲשֶׁר-הוֹתִ֖רָה מִשָּׂבְעָֽהּ: {יט}  וַתֹּאמֶר֩ לָ֨הּ חֲמוֹתָ֜הּ אֵיפֹ֨ה לִקַּ֤טְתְּ הַיּוֹם֙ וְאָ֣נָה עָשִׂ֔ית יְהִ֥י מַכִּירֵ֖ךְ בָּר֑וּךְ וַתַּגֵּ֣ד לַחֲמוֹתָ֗הּ אֵ֤ת אֲשֶׁר-עָשְׂתָה֙ עִמּ֔וֹ וַתֹּ֗אמֶר שֵׁ֤ם הָאִישׁ֙ אֲשֶׁ֨ר עָשִׂ֧יתִי עִמּ֛וֹ הַיּ֖וֹם בֹּֽעַז:

 רש"י  יהי מכירך ברוך. בעל השדה שנשא ונתן לך פנים ללקט בשדהו : (רש"י)
{כ}  וַתֹּ֨אמֶר נָעֳמִ֜י לְכַלָּתָ֗הּ בָּר֥וּךְ הוּא֙ לַיהוָ֔ה אֲשֶׁר֙ לֹא-עָזַ֣ב חַסְדּ֔וֹ אֶת-הַחַיִּ֖ים וְאֶת-הַמֵּתִ֑ים וַתֹּ֧אמֶר לָ֣הּ נָעֳמִ֗י קָר֥וֹב לָ֙נוּ֙ הָאִ֔ישׁ מִֽגֹּאֲלֵ֖נוּ הֽוּא:

 רש"י  את החיים ואת המתים. שזן ומפרנס את החיים ונטפל בצורכי המתים : (רש"י)
{כא}  וַתֹּ֖אמֶר ר֣וּת הַמּוֹאֲבִיָּ֑ה גַּ֣ם | כִּי-אָמַ֣ר אֵלַ֗י עִם-הַנְּעָרִ֤ים אֲשֶׁר-לִי֙ תִּדְבָּקִ֔ין עַ֣ד אִם-כִּלּ֔וּ אֵ֥ת כָּל-הַקָּצִ֖יר אֲשֶׁר-לִֽי: {כב}  וַתֹּ֥אמֶר נָעֳמִ֖י אֶל-ר֣וּת כַּלָּתָ֑הּ ט֣וֹב בִּתִּ֗י כִּ֤י תֵֽצְאִי֙ עִם-נַ֣עֲרוֹתָ֔יו וְלֹ֥א יִפְגְּעוּ-בָ֖ךְ בְּשָׂדֶ֥ה אַחֵֽר: {כג}  וַתִּדְבַּ֞ק בְּנַעֲר֥וֹת בֹּ֙עַז֙ לְלַקֵּ֔ט עַד-כְּל֥וֹת קְצִֽיר-הַשְּׂעֹרִ֖ים וּקְצִ֣יר הַֽחִטִּ֑ים וַתֵּ֖שֶׁב אֶת-חֲמוֹתָֽהּ:

{א}  וַתֹּ֥אמֶר לָ֖הּ נָעֳמִ֣י חֲמוֹתָ֑הּ בִּתִּ֞י הֲלֹ֧א אֲבַקֶּשׁ-לָ֛ךְ מָנ֖וֹחַ אֲשֶׁ֥ר יִֽיטַב-לָֽךְ: {ב}  וְעַתָּ֗ה הֲלֹ֥א בֹ֙עַז֙ מֹֽדַעְתָּ֔נוּ אֲשֶׁ֥ר הָיִ֖ית אֶת-נַעֲרוֹתָ֑יו הִנֵּה-ה֗וּא זֹרֶ֛ה אֶת-גֹּ֥רֶן הַשְּׂעֹרִ֖ים הַלָּֽיְלָה:

 רש"י  מדעתנו. קרובינו : הנה הוא זורה. המוץ ווונטו''ר בלע''ז : הלילה. שהיה הדור פרוץ בגנבה וגזל והיה ישן בגורנו לשמור גורנו : (רש"י)
{ג}  וְרָחַ֣צְתְּ | וָסַ֗כְתְּ וְשַׂ֧מְתְּ (שמלתך) שִׂמְלֹתַ֛יִךְ עָלַ֖יִךְ (וירדתי) וְיָרַ֣דְתְּ הַגֹּ֑רֶן אַל-תִּוָּדְעִ֣י לָאִ֔ישׁ עַ֥ד כַּלֹּת֖וֹ לֶאֱכֹ֥ל וְלִשְׁתּֽוֹת:

 רש"י  ורחצת. מטנוף ע''ז שלך : וסכת. אלו מצות : ושמת שמלותיך. בגדים של שבת : וירדת הגורן. וירדתי כתיב זכותי תרד עמך : אל תודעי לאיש. לבועז : (רש"י)
{ד}  וִיהִ֣י בְשָׁכְב֗וֹ וְיָדַ֙עַתְּ֙ אֶת-הַמָּקוֹם֙ אֲשֶׁ֣ר יִשְׁכַּב-שָׁ֔ם וּבָ֛את וְגִלִּ֥ית מַרְגְּלֹתָ֖יו (ושכבתי) וְשָׁכָ֑בְתְּ וְהוּא֙ יַגִּ֣יד לָ֔ךְ אֵ֖ת אֲשֶׁ֥ר תַּעֲשִֽׂין: {ה}  וַתֹּ֖אמֶר אֵלֶ֑יהָ כֹּ֛ל אֲשֶׁר-תֹּאמְרִ֥י אֵלַ֖י אֶֽעֱשֶֽׂה: {ו}  וַתֵּ֖רֶד הַגֹּ֑רֶן וַתַּ֕עַשׂ כְּכֹ֥ל אֲשֶׁר-צִוַּ֖תָּה חֲמוֹתָֽהּ:

 רש"י  ותרד הגורן ותעש. היא אמרה לה ורחצת וסכת ושמת שמלותיך עליך ואחר כך וירדת הגורן והיא לא עשתה כן אלא אמרה אם ארד כשאני מקושטת הפוגע בי והרואה אותי יאמר שאני זונה לפיכך ירדה בתחלה הגורן ואחר כך קשטה את עצמה כאשר צותה חמותה : (רש"י)

{יג}  לִ֣ינִי | הַלַּ֗יְלָה וְהָיָ֤ה בַבֹּ֙קֶר֙ אִם-יִגְאָלֵ֥ךְ טוֹב֙ יִגְאָ֔ל וְאִם-לֹ֨א יַחְפֹּ֧ץ לְגָֽאֳלֵ֛ךְ וּגְאַלְתִּ֥יךְ אָנֹ֖כִי חַי-יְהוָ֑ה שִׁכְבִ֖י עַד-הַבֹּֽקֶר:

{יד}  וַתִּשְׁכַּ֤ב (מרגלתו) מַרְגְּלוֹתָיו֙ עַד-הַבֹּ֔קֶר וַתָּ֕קָם (בטרום) בְּטֶ֛רֶם יַכִּ֥יר אִ֖ישׁ אֶת-רֵעֵ֑הוּ וַיֹּ֙אמֶר֙ אַל-יִוָּדַ֔ע כִּי-בָ֥אָה הָאִשָּׁ֖ה הַגֹּֽרֶן:

רות פרק-ד

{א}  וּבֹ֨עַז עָלָ֣ה הַשַּׁעַר֮ וַיֵּ֣שֶׁב שָׁם֒ וְהִנֵּ֨ה הַגֹּאֵ֤ל עֹבֵר֙ אֲשֶׁ֣ר דִּבֶּר-בֹּ֔עַז וַיֹּ֛אמֶר ס֥וּרָה שְׁבָה-פֹּ֖ה פְּלֹנִ֣י אַלְמֹנִ֑י וַיָּ֖סַר וַיֵּשֵֽׁב:

 רש"י  פלני אלמני. ולא נכתב שמו לפי שלא אבה לגאול : פלני אלמני. מתורגם בנביאים (שמואל א' כא) כסי וטמיר : פלני. מכוסה ונעלם לשון (דברים יז) כי יפלא (בראשית יח) היפלא מה' דבר : אלמני. אלמון מבלי שם, (ס''א אלמוני שהיה אלמן מד''ת שהיה לו לדרוש עמוני ולא עמונית מואבי ולא מואבית והוא אמר פן אשחית את נחלתי) : (רש"י)
{ב}  וַיִּקַּ֞ח עֲשָׂרָ֧ה אֲנָשִׁ֛ים מִזִּקְנֵ֥י הָעִ֖יר וַיֹּ֣אמֶר שְׁבוּ-פֹ֑ה וַיֵּשֵֽׁבוּ: {ג}  וַיֹּ֙אמֶר֙ לַגֹּאֵ֔ל חֶלְקַת֙ הַשָּׂדֶ֔ה אֲשֶׁ֥ר לְאָחִ֖ינוּ לֶאֱלִימֶ֑לֶךְ מָכְרָ֣ה נָעֳמִ֔י הַשָּׁ֖בָה מִשְּׂדֵ֥ה מוֹאָֽב: {ד}  וַאֲנִ֨י אָמַ֜רְתִּי אֶגְלֶ֧ה אָזְנְךָ֣ לֵאמֹ֗ר קְ֠נֵה נֶ֥גֶד הַֽיֹּשְׁבִים֮ וְנֶ֣גֶד זִקְנֵ֣י עַמִּי֒ אִם-תִּגְאַל֙ גְּאָ֔ל וְאִם-לֹ֨א יִגְאַ֜ל הַגִּ֣ידָה לִּ֗י (ואדע) וְאֵֽדְעָה֙ כִּ֣י אֵ֤ין זוּלָֽתְךָ֙ לִגְא֔וֹל וְאָנֹכִ֖י אַחֲרֶ֑יךָ וַיֹּ֖אמֶר אָנֹכִ֥י אֶגְאָֽל:

 רש"י  ואדעה כי אין זולתך. קרוב לגאול : (רש"י)
{ה}  וַיֹּ֣אמֶר בֹּ֔עַז בְּיוֹם-קְנוֹתְךָ֥ הַשָּׂדֶ֖ה מִיַּ֣ד נָעֳמִ֑י וּ֠מֵאֵת ר֣וּת הַמּוֹאֲבִיָּ֤ה אֵֽשֶׁת-הַמֵּת֙ (קניתי) קָנִ֔יתָה לְהָקִ֥ים שֵׁם-הַמֵּ֖ת עַל-נַחֲלָתֽוֹ:

 רש"י  ומאת רות המואביה. אתה צריך לקנות והיא אינה מתרצה אלא אם כן תשאנה : (רש"י)
{ו}  וַיֹּ֣אמֶר הַגֹּאֵ֗ל לֹ֤א אוּכַל֙ (לגאול)-לִגְאָל-לִ֔י פֶּן-אַשְׁחִ֖ית אֶת-נַחֲלָתִ֑י גְּאַל-לְךָ֤ אַתָּה֙ אֶת-גְּאֻלָּתִ֔י כִּ֥י לֹא-אוּכַ֖ל לִגְאֹֽל:

 רש"י  פן אשחית את נחלתי. זרעי כמו (תהלים קכז) נחלת ה' בנים לתת פגם בזרעי שנאמר (דברים כג) לא יבא עמוני ומואבי וטעה בעמוני ולא עמונית : (רש"י)
{ז}  וְזֹאת֩ לְפָנִ֨ים בְּיִשְׂרָאֵ֜ל עַל-הַגְּאוּלָּ֤ה וְעַל-הַתְּמוּרָה֙ לְקַיֵּ֣ם כָּל-דָּבָ֔ר שָׁלַ֥ף אִ֛ישׁ נַעֲל֖וֹ וְנָתַ֣ן לְרֵעֵ֑הוּ וְזֹ֥את הַתְּעוּדָ֖ה בְּיִשְׂרָאֵֽל:

 רש"י  על הגאלה. זו מכירה : תמורה. זו חליפין : שלף איש נעלו. זהו קניין כמו שאנו קונין בסודר במקום נעל, ורבותינו ז''ל נחלקו בדבר מי נתן למי יש אומרים קונין בכליו של קונה ובועז נתן לגואל ויש אומרים קונין בכליו של מקנה וגואל נתן לבועז : וזאת התעודה בישראל. משפט העדות : (רש"י)
{ח}  וַיֹּ֧אמֶר הַגֹּאֵ֛ל לְבֹ֖עַז קְנֵה-לָ֑ךְ וַיִּשְׁלֹ֖ף נַעֲלֽוֹ: {ט}  וַיֹּאמֶר֩ בֹּ֨עַז לַזְּקֵנִ֜ים וְכָל-הָעָ֗ם עֵדִ֤ים אַתֶּם֙ הַיּ֔וֹם כִּ֤י קָנִ֙יתִי֙ אֶת-כָּל-אֲשֶׁ֣ר לֶֽאֱלִימֶ֔לֶךְ וְאֵ֛ת כָּל-אֲשֶׁ֥ר לְכִלְי֖וֹן וּמַחְל֑וֹן מִיַּ֖ד נָעֳמִֽי: {י}  וְגַ֣ם אֶת-ר֣וּת הַמֹּאֲבִיָּה֩ אֵ֨שֶׁת מַחְל֜וֹן קָנִ֧יתִי לִ֣י לְאִשָּׁ֗ה לְהָקִ֤ים שֵׁם-הַמֵּת֙ עַל-נַ֣חֲלָת֔וֹ וְלֹא-יִכָּרֵ֧ת שֵׁם-הַמֵּ֛ת מֵעִ֥ם אֶחָ֖יו וּמִשַּׁ֣עַר מְקוֹמ֑וֹ עֵדִ֥ים אַתֶּ֖ם הַיּֽוֹם:

 רש"י  להקים שם המת על נחלתו. מתוך שאשתו יוצאה ובאה בנחלה ומכנסת ומוציאה אומרים זאת היתה אשת מחלון שמו נזכר עליה : (רש"י)
{יא}  וַיֹּ֨אמְר֜וּ כָּל-הָעָ֧ם אֲשֶׁר-בַּשַּׁ֛עַר וְהַזְּקֵנִ֖ים עֵדִ֑ים יִתֵּן֩ יְהוָ֨ה אֶֽת-הָאִשָּׁ֜ה הַבָּאָ֣ה אֶל-בֵּיתֶ֗ךָ כְּרָחֵ֤ל | וּכְלֵאָה֙ אֲשֶׁ֨ר בָּנ֤וּ שְׁתֵּיהֶם֙ אֶת-בֵּ֣ית יִשְׂרָאֵ֔ל וַעֲשֵׂה-חַ֣יִל בְּאֶפְרָ֔תָה וּקְרָא-שֵׁ֖ם בְּבֵ֥ית לָֽחֶם:

 רש"י  כרחל וכלאה. אף על פי שהיו משבט יהודה ומבני לאה מודים הם על רחל שהיתה עיקרה של בית והקדימו רחל ללאה : וקרא שם. כלומר יגדל שמך : (רש"י)
{יב}  וִיהִ֤י בֵֽיתְךָ֙ כְּבֵ֣ית פֶּ֔רֶץ אֲשֶׁר-יָלְדָ֥ה תָמָ֖ר לִֽיהוּדָ֑ה מִן-הַזֶּ֗רַע אֲשֶׁ֨ר יִתֵּ֤ן יְהוָה֙ לְךָ֔ מִן-הַֽנַּעֲרָ֖ה הַזֹּֽאת:

 רש"י  כבית פרץ. שיצאנו ממנו : (רש"י)
{יג}  וַיִּקַּ֨ח בֹּ֤עַז אֶת-רוּת֙ וַתְּהִי-ל֣וֹ לְאִשָּׁ֔ה וַיָּבֹ֖א אֵלֶ֑יהָ וַיִּתֵּ֨ן יְהוָ֥ה לָ֛הּ הֵרָי֖וֹן וַתֵּ֥לֶד בֵּֽן: {יד}  וַתֹּאמַ֤רְנָה הַנָּשִׁים֙ אֶֽל-נָעֳמִ֔י בָּר֣וּךְ יְהוָ֔ה אֲ֠שֶׁר לֹ֣א הִשְׁבִּ֥ית לָ֛ךְ גֹּאֵ֖ל הַיּ֑וֹם וְיִקָּרֵ֥א שְׁמ֖וֹ בְּיִשְׂרָאֵֽל: {טו}  וְהָ֤יָה לָךְ֙ לְמֵשִׁ֣יב נֶ֔פֶשׁ וּלְכַלְכֵּ֖ל אֶת-שֵׂיבָתֵ֑ךְ כִּ֣י כַלָּתֵ֤ךְ אֲֽשֶׁר-אֲהֵבַ֙תֶךְ֙ יְלָדַ֔תּוּ אֲשֶׁר-הִיא֙ ט֣וֹבָה לָ֔ךְ מִשִּׁבְעָ֖ה בָּנִֽים: {טז}  וַתִּקַּ֨ח נָעֳמִ֤י אֶת-הַיֶּ֙לֶד֙ וַתְּשִׁתֵ֣הוּ בְחֵיקָ֔הּ וַתְּהִי-ל֖וֹ לְאֹמֶֽנֶת: {יז}  וַתִּקְרֶאנָה֩ ל֨וֹ הַשְּׁכֵנ֥וֹת שֵׁם֙ לֵאמֹ֔ר יֻלַּד-בֵּ֖ן לְנָעֳמִ֑י וַתִּקְרֶ֤אנָֽה שְׁמוֹ֙ עוֹבֵ֔ד ה֥וּא אֲבִי-יִשַׁ֖י אֲבִ֥י דָוִֽד: (פ) {יח}  וְאֵ֙לֶּה֙ תּוֹלְד֣וֹת פָּ֔רֶץ פֶּ֖רֶץ הוֹלִ֥יד אֶת-חֶצְרֽוֹן: {יט}  וְחֶצְרוֹן֙ הוֹלִ֣יד אֶת-רָ֔ם וְרָ֖ם הוֹלִ֥יד אֶת-עַמִּֽינָדָֽב:

 רש"י  ואלה תולדות פרץ. לפי שייחס את דוד על שמה של רות המואביה חזר ויחסו על שם יהודה : חסלת מגלת רות : (רש"י)
{כ}  וְעַמִּֽינָדָב֙ הוֹלִ֣יד אֶת-נַחְשׁ֔וֹן וְנַחְשׁ֖וֹן הוֹלִ֥יד אֶת-שַׂלְמָֽה: {כא}  וְשַׂלְמוֹן֙ הוֹלִ֣יד אֶת-בֹּ֔עַז וּבֹ֖עַז הוֹלִ֥יד אֶת-עוֹבֵֽד: {כב}  וְעֹבֵד֙ הוֹלִ֣יד אֶת-יִשָׁ֔י וְיִשַׁ֖י הוֹלִ֥יד אֶת-דָּוִֽד: (פפפ)
