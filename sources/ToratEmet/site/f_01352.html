פרק א - משנה א

{א} וְקָשֶׁה, דְּלֹא פֵרֵשׁ לְבֵית שַׁמַּאי דִּין רְאִיָּה גוּפַהּ כְּמוֹ שֶׁפֵּרֵשׁ לְבֵית הִלֵּל דְּאֵינָהּ מְטַמְּאָה בְמַשָּׂא. וְהָרַמְבַּ"ם לֹא פֵרֵשׁ בִּדְבֵית הִלֵּל כְּלוּם. וְהָרַ"שׁ פֵּרֵשׁ, דִּלְבֵית שַׁמַּאי מַסִּיט אֶת הָרְאִיָּה, תּוֹלִין. וּפֵרֵשׁ נַמִּי דְּמִשְׁכָּב וּמוֹשָׁב תּוֹלִין, כִּדְתָנֵי בַתּוֹסֶפְתָּא. וּמִנַּהּ אָמְרִינַן בַּגְּמָרָא, דִּכְשׁוֹמֶרֶת יוֹם דְּבֵית שַׁמַּאי, הַיְנוּ כְּבוֹעֵל שׁוֹמֶרֶת יוֹם. וְהָרַ"ב עֵרֵב שְׁנֵי הַפֵּרוּשִׁים: {ב} שְׁתַּיִם. דְּאִלּוּ אַחַת, לְבֵית הִלֵּל אֵינוֹ מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב:


פרק א - משנה ב

{ג} רַבִּי עֲקִיבָא אוֹמֵר. הַיְנוּ תַּנָּא קַמָּא דִּסְבִירָא לֵיהּ נַמִּי הַךְ וּמוֹדִים, דְּהָא לֹא אָמַר אֶלָּא בַּשְּׁלִישִׁי. וְצָרִיךְ לוֹמַר דְּקָא מַשְׁמַע לָן דְּמַאן תַּנָּא קַמָּא, רַבִּי עֲקִיבָא: {ד} בַּגְּמָרָא, אֵין לְךָ בָּהּ אֶלָּא מַה שֶּׁאָמוּר בָּהּ. וּפֵרֵשׁ רַשִׁ"י, כְּטֻמְאָתָהּ, מַה טֻּמְאָתָהּ יוֹם אֶחָד, אַף סְתִירָתָהּ כֵּן: {ה} מַשְׁמַע טָהֳרָה אַחַת בְּלִי שׁוּם הַפְסָקָה. רַשִׁ"י:


פרק א - משנה ד

{ו} הָרַמְבַּ"ם. וּמִשּׁוּם דְּלִרְאִיָּה אֵין שִׁעוּר וַאֲפִלּוּ כָל שֶׁהוּא, הִלְכָּךְ אֵין פּוֹחֲתִין שִׁעוּר הַהֶפְסֵק, דְּכָל שֶׁהוּא אֵינוֹ מִן הַשִּׁעוּר. וְאֶפְשָׁר גַּם כֵּן דְּעִנְיָנוֹ בֵּין מִשֶּׁהִתְחִיל עַד לְמַה שֶּׁהוּא בַסּוֹף, וּלְהַאי כָּל שֶׁהוּא קְרָאֻהוּ הַתְחָלָה וְסוֹף כוּ':


פרק א - משנה ה

{ז} כְּדֵי שְׁתֵּי כוּ'. וְהוּא שֶׁיִּרְאֶה הָאַחַת בִּתְחִלַּת הַשִּׁעוּר וְאַחַת בְּאֶמְצָעָהּ וְהָאַחַת בְּסוֹפָהּ. הָרַמְבַּ"ם: {ח} אָמַר רַבִּי יוֹסֵי כוּ'. בַּסִּפְרָא פְלִיגֵי בִּדְרָשֵׁי דִקְרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:


פרק א - משנה ו

{ט} וּבְהָכִי אַף רַבִּי יוֹסֵי מוֹדֶה. כִּדְאִיתָא בַתּוֹסֶפְתָּא הוּבָא בָּרַמְבַּ"ם וּבָרַ"שׁ: {י} דְּאִי בְּרֶגַע אֶחָד מְכֻוָּן, הָיָה מִמַּה נַּפְשָׁךְ טָמֵא, אִם זֶה מִן הַיּוֹם אוֹ מִן הַלַּיְלָה אַף זֶה כֵּן, וְהָווּ לְהוּ שְׁתֵּי רְאִיּוֹת רְצוּפִים. הָרַ"שׁ:

פרק ב - משנה א

{א} מֵרִבּוּי דְאִישׁ, דִּמְרַבִּינַן קְטַנִּים, כִּדְאִיתָא בַגְּמָרָא. וְנִרְאֶה לִי דַּהֲוֵי כְעֵין מִעוּט אַחַר מִעוּט שֶׁהוּא לְרַבּוֹת. דְּהָא אִישׁ מַשְׁמָעוֹ בֶּן שְׁלֹשׁ עֶשְׂרֵה, וּמְמַעֵט הַפָּחוֹת מִכָּן: {ב} כְּלוֹמַר מַה שֶּׁהוּא רָגִיל שֶׁיָּזוּב מִמֶּנּוּ, וְהוּא לֹבֶן וְדוֹמֶה לְמֵי בָצֵק, וְגַם כְּלֹבֶן בֵּיצָה, דְּכֻלַּהּ חֲדָא מִלְּתָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} אִי הָכִי, כִּי אִטַּמּוּ בִּשְׁאָר טֻמְאוֹת [בְּמֵת אוֹ בְשֶׁרֶץ] לֹא לִשַּׁלְּחוּ. אָמַר קְרָא מִזָּכָר, מִטֻּמְאָה הַפּוֹרֶשֶׁת מִן הַזָּכָר. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} וְאַף עַל גַּב דְּאַנְדְּרוֹגִינוֹס קָמֵי שְׁמַיָּא גַּלְיָא אִם זָכָר הוּא אִם נְקֵבָה, יֵשׁ לוֹמַר דִּמְמַעֵט לֵיהּ מִטַּעַם שֶׁהוּא מְשֻׁנֶּה. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ה} כְּלוֹמַר, עַל כָּרְחָךְ אַשְּׁנִיָּה דִּכְתִיב בָּתְרֵיהּ קָאֵי, וְלֹא אַרִאשׁוֹנָה, דְּהָא אִתַּקַּשׁ לְשִׁכְבַת זֶרַע כוּ': {ו} דְּכָל עַצְמוֹ שֶׁל שִׁכְבַת זֶרַע מֵחֲמַת אֹנֶס חִמּוּם בָּא. רַשִׁ"י. וְנִרְאֶה לִי, דְּמִשּׁוּם דְּמִשִּׁכְבַת זֶרַע יָלְפִינַן, דַּיּוֹ לְמֵילַף לְעִנְיַן טֻמְאָה, אֲבָל לְקָרְבָּן דְּלֹא שַׁיָּךְ בְּשִׁכְבַת זֶרַע, לֵית לָן לְמֵילַף מִנֵּיהּ: {ז} רַבִּי אֱלִיעֶזֶר אוֹמֵר. בַּגְּמָרָא פְלִיגֵי בִּדְרָשֵׁי דִקְרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ח} תּוֹסֶפְתָּא. וְחֹלִי אֵין צָרִיךְ לְמִתְנֵי, דְּוַדַּאי דְּכָל זְמַן שֶׁמִּצְטַעֵר: {ט} רַבִּי יוֹסֵי אוֹמֵר. בְּסִפְרִי פְּלִיגֵי בִּדְרָשֵׁי דִקְרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} דְּאַרֵישָׁא מְהַדַּר, דְּרוֹאֶה קֶרִי אֵינוֹ מְטַמֵּא מֵעֵת לְעֵת דְּהַיְנוּ בְּזִיבָה שְׁנִיָּה, וְקָאָמַר דְּנָכְרִי שֶׁרָאָה קֶרִי מְטַמֵּא רְאִיָּה שֶׁהִיא שְׁנִיָּה לָרְאִיָּה שֶׁרָאָה כְּבָר קֹדֶם שֶׁנִּתְגַּיֵּר. וּבִתְנַאי שֶׁתִּהְיֶה רְאִיָּה הַשְּׁנִיָּה אַחַר שֶׁפָּסַק מִן הַדְּאָגָה שֶׁל קַבָּלַת מִצְוֹת: {יא} הָרַ"שׁ. וְקָשֶׁה, דְּאִם כֵּן תַּנָּאָה הוּא וַהֲוָה לֵיהּ לְמִתְנֵי בְּאֳהָלוֹת פֶּרֶק י"א מִשְׁנָה ז'. וְלָכֵן נִרְאֶה כְּגִרְסַת הַסֵּפֶר, וְהִיא גִרְסַת הָרַמְבַּ"ם, וְאֵינוֹ אֶלָּא תּוֹסֶפֶת בֵּאוּר שֶׁכָּךְ הוּא הַהֹוֶה:

{א} דְּלֵיכָּא לְמֵיחַשׁ שֶׁמָּא יָשַׁב עֲלֵיהֶם הַזָּב. מַהֲרַ"ם: {ב} וְנִרְאֶה שֶׁטֻּמְאָתוֹ בְסָפֵק, וְכֵן דַּעַת הָרַמְבַּ"ם. אֶלָּא שֶׁכָּתַב שֶׁמָּא דָרַס עֲלֵיהֶן הַזָּב, וּלְהָרַאֲבַ"ד אֵין כָּאן סָפֵק: {ג} שֶׁכָּל הַמִּתְבַּשֵּׁל אֵין כֻּלּוֹ נוֹגֵעַ בּוֹ, אֲבָל יֵשׁ מִקְצָתוֹ בַּאֲוִירוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} דִּפְשִׁיטָא, מִי גָרַע מִשְּׁאָר טֻמְאוֹת שֶׁמְּטַמְּאוֹת כְּלֵי חֶרֶס מֵאֲוִירָן, כֵּיוָן דִּכְתִיב גַּבֵּי שְׁרָצִים, לָמָּה לֵיהּ לְמִכְתְּבֵיהּ הָכָא. רַשִׁ"י: {ה} דְּגַבֵּי אָדָם נַמִּי כְתִיב [וַיִּקְרָא טו] וְכֹל אֲשֶׁר יִגַּע בּוֹ הַזָּב, וּמַגָּע נַמִּי כְּתִיב [וַיִּקְרָא טו] וְהַנֹּגֵעַ בִּבְשַׂר הַזָּב, אֶלָּא לְהָבִיא הֶסֵּטוֹ. רַשִׁ"י: {ו} וְהוּא שֶׁכָּרַע הַטָּהוֹר. אֲבָל אִם כָּרַע הַזָּב, לֹא נַעֲשָׂה הַטָּהוֹר אֶלָּא רִאשׁוֹן וְאֵינוֹ מְטַמֵּא בְגָדִים: {ז} וּמִכָּל מָקוֹם אֵין טֻמְאָתָן שָׁוָה, וּכְמוֹ שֶׁכָּתַבְתִּי בְּסָמוּךְ: {ח} וְקָשֶׁה, הֲלֹא אֲפִלּוּ אֵינָהּ נֶחְבֵּאת הָוֵי כֹחָהּ רַע לְפִי שֶׁרוֹתֵת תַּחְתָּיו. תּוֹסָפוֹת. וְהַכֶּסֶף מִשְׁנֶה פֵּרֵשׁ לְדִבְרֵי הָרַמְבַּ"ם, כְּשֶׁאוֹחֵז בָּהּ וְהִיא נֶחְבֵּאת תּוֹךְ יָדוֹ וְכוֹפֵף אוֹתָהּ הִיא מֵזִיזָה אוֹתָהּ:

{א} נִדָּה כוּ'. וְהוּא הַדִּין זָב וְטָהוֹר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} טְמֵאִין. נִרְאֶה לִי דְּמַיְרֵי שֶׁנּוֹשֵׂאת הָעֲרֵבָה בְּחֵיקָהּ לְפָנֶיהָ, וּכְשֶׁמַּשָּׂאָן כָּבֵד מַכְרִיעוֹת אוֹתָהּ עַד שֶׁהִיא נִשְׁעֶנֶת וְנִתְלֵית עֲלֵיהֶם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} טְהוֹרִין. הַבְּגָדִים מִטֻּמְאַת מִדְרָס. וְאַף שֶׁהָעֲרֵבָה טְמֵאָה מַגַּע זָב, וְאָדָם הַנּוֹגֵעַ בְּזָב מְטַמֵּא בְגָדִים כָּל זְמַן שֶׁלֹּא פֵרַשׁ, כְּלִי הַנּוֹגֵעַ בְּזָב אֵינוֹ מְטַמֵּא בְגָדִים אֲפִלּוּ בְעוֹד שֶׁלֹּא פֵרַשׁ. הָרַ"שׁ:

{ד} הִקִּישׁ כוּ'. בַּתּוֹסֶפְתָּא, זֶה הַכְּלָל, כָּל שֶׁנָּפַל מִכֹּחַ הֶסֵּטוֹ [הַיְנוּ שֶׁכֹּחוֹ רַע], טָמֵא. מִכֹּחַ הָרְעָדָה [שֶׁכֹּחוֹ יָפֶה], טָהוֹר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} וְקֶלֶת דִּלְקַמָּן עָשׂוּי שֶׁלֹּא יִתְפַּזֵּר הַקֶּמַח וְשֶׁיִּפֹּל לְתוֹךְ הַיָּם. אֲבָל הָרַ"שׁ פֵּרֵשׁ, יָם שֶׁל בֵּית הַבַּד:

{ו} וְנוֹתְנִין בָּהֶם מָעוֹת וּרְאוּיִים לְשַׁמֵּשׁ שְׁכִיבָה עִם מְלַאכְתָּן, שֶׁיּוֹשֵׁב עֲלֵיהֶן וְהֵן מְלֵאוֹת מָעוֹת וּפוֹתֵחַ פִּיהֶם וְנוֹטֵל, וְאֵין צָרִיךְ לוֹמַר לוֹ עֲמֹד וְנַעֲשֶׂה מְלַאכְתֵּנוּ. רַשִׁ"י: {ז} כְּגוֹן שֶׁשּׁוֹכֵב עַל מֵעָיו אוֹ פְרַקְדָן וְאָרְכּוֹ מֻטָּל עַל אָרְכָּן. רַשִׁ"י: {ח} וּלְפִי זֶה אָתְיָא כְרַבִּי שִׁמְעוֹן דְּסוֹף פִּרְקִין. וְרַבִּי יוֹסֵי מְפָרֵשׁ דְּאָתְיָא אֲפִלּוּ כְרַבָּנָן. וְטַעֲמָא דִטְהוֹרִין, מִשּׁוּם דַּהֲווּ לְהוּ כָּל אֶחָד מְסַיֵּעַ וְאֵין בּוֹ מַמָּשׁ: {ט} וְלֹא דַק, דְּזֶה נֶאֱמַר עַל הַטָּהוֹר הַשּׁוֹכֵב עַל מִשְׁכַּב הַזָּב. אֲבָל בְּזָב הָעוֹשֶׂה מִשְׁכָּב נִדְרָשׁ מִדִּכְתִיב וְכָל הַמִּשְׁכָּב אֲשֶׁר יִשְׁכַּב וְגוֹ'. וְהָא דְדָרְשִׁינַן רֻבּוֹ, מִשּׁוּם דְּרֻבּוֹ כְּכֻלּוֹ: {י} וְקָשֶׁה קְצָת, דְּהָכָא תָנֵי בְּיוֹ"ד וּלְקַמָּן בְּאָלֶ"ף. וְנִרְאֶה לִי כְּפֵרוּשׁ הָרַמְבַּ"ם שֶׁכָּתַב כִּסְיוֹת כְּלֵי חֲגוֹרוֹת שֶׁל עוֹר נְאוֹתִים לְמִשְׁכָּב: {יא} אֶלָּא עַל שְׁתֵּיהֶן בְּשָׁוֶה, חֶצְיוֹ עַל כָּל אֶחָד. אֲבָל אִם הֵם קְרוֹבִים, זִמְנִין נִשְׁעָן רֻבּוֹ עַל אֶחָד. מַהֲרַ"ם:

{יב} וּבַמֵּת כוּ'. וַהֲוָה מָצֵי נַמִּי לְמִתְנֵי וּבִנְבֵלָה הַכֹּל כוּ' חוּץ כוּ': {יג} מַדָּף כוּ'. בְּסִפְרִי יָלֵיף לְהוּ מִקְּרָאֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יד} פֵּרוּשׁ, הָרְאוּיִים לְמִשְׁכָּב וּמוֹשָׁב: {טו} כָּל זֶה הוּגָה מִלְּשׁוֹן הָרַ"שׁ. אֲבָל לְפִי שִׁגְרַת לְשׁוֹן הָרַ"ב אֵין צָרִיךְ לְהַכְנִיס כָּל זֶה בִּלְשׁוֹנוֹ. אַךְ לְפִי זֶה צָרִיךְ לְהַגִּיהַּ בְּסָמוּךְ וְשֶׁכֵּן צָרִיךְ לוֹמַר, כִּי הַאי גַּוְנָא בְּזָב טָמֵא לְטַמֵּא כוּ' וּבְמֵת טָהוֹר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{טז} וְהוֹאִיל וְכָל אַחַת רְאוּיָה לִהְיוֹת מְסַיֵּעַ, וְאֵין אָנוּ יוֹדְעִים אֵיזוֹ יָד וְאֵיזוֹ רֶגֶל הִיא שֶׁלֹּא הָיְתָה נִשְׁעֶנֶת עָלֶיהָ, הֲרֵי לֹא הֻחְזְקָה טֻמְאָה בְּאַחַת מֵהֶן, וּלְפִיכָךְ כֻּלָּן טְהוֹרוֹת. הָרַמְבַּ"ם: {יז} עַיֵּן בְּתוֹסְפוֹת יוֹם טוֹב שֶׁהֶאֱרִיךְ בְּטַעַם פְּלֻגְתָּתָן עַל פִּי צוּרַת כֵּלִים הַלָּלוּ שֶׁהָיוּ אֶצְלָם בְּאוֹתוֹ הַזְּמַן:

פרק ה - משנה א

{א} דַּהֲוָה לֵיהּ לְמֵימַר וְרָחַץ בַּמַּיִם בָּעֶרֶב, אֶלָּא לוֹמַר דְּעוֹד טָמֵא גַּם כֵּן כָּל כְּלִי שֶׁיִּגַּע בּוֹ. וּכְלֵי שֶׁטֶף הַיְנוּ כָּל שְׁאָר כֵּלִים חוּץ מִכְּלֵי חֶרֶס. עַל שֵׁם שֶׁיֵּשׁ לָהֶם טָהֳרָה בְּמִקְוֶה. רַשִׁ"י: {ב} דִּכְלִי שֶׁטֶף דּוֹמֶה קְצָת לְבֶגֶד, דַּהֲוֵי נַמִּי כְּלִי, מַה שֶּׁאֵין כֵּן הָאָדָם. וּכְלִי חֶרֶס אֵין דּוֹמֶה לוֹ, שֶׁאֵין לוֹ טָהֳרָה בְּמִקְוֶה: {ג} כָּל הַמְטַמֵּא כוּ'. כָּל אָדָם הַמְטַמֵּא קָאָמַר, דְּהָא מַשְׁכַּחַת כֵּלִים שֶׁאֵין מְטַמְּאִין כֵּלִים, וְאִם נָגְעוּ מְטַמְּאִין אֳכָלִים לִהְיוֹת תְּחִלָּה כִּדְמוּכָח בְּמַתְנִיתִין דִּלְקַמָּן. דְּדַוְקָא בְּאֵין נוֹגְעִים מַיְרֵי סֵיפָא, הָא נוֹגְעִים מְטַמְּאִין שְׁנַיִם. וְכֵן הוּא בְּהֶדְיָא בַתּוֹסֶפְתָּא. וְשָׁם מְפָרֵשׁ הָרַ"ב דְּמַתְנִיתִין כָּל אָדָם קָאָמַר: {ד} כְּלוֹמַר, לְעִנְיַן אֳכָלִים וּמַשְׁקִים. וְאַף עַל פִּי שֶׁאֵינוֹ אַב הַטֻּמְאָה לְעִנְיַן אָדָם וּכְלֵי חֶרֶס:


פרק ה - משנה ב

{ה} אֲבָל סֵיפָא כָּל שֶׁהַזָּב נִשָּׂא עָלָיו טָהוֹר, דַּוְקָא בְלֹא נָגַע. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ו} טָמֵא. וּלְקַמָּן מְפָרֵשׁ שֶׁאֵין כֻּלָּם שָׁוִין: {ז} טָהוֹר. לְגַמְרֵי. וְאֵין בָּהֶן אֲפִלּוּ טֻמְאַת מַדָּף, כִּדְאִיתָא בְתוֹרַת כֹּהֲנִים. תּוֹסָפוֹת. וְצָרִיךְ לוֹמַר דְּמַיְרֵי שֶׁלֹּא הוּסַט. דְּעַל יְדֵי הֶסֵּט הָא אֲמַרַן לְעֵיל דְּמִטַּמְאִין. וְהָרַ"שׁ פֵּרֵשׁ, דַּוְקָא עַל יְדֵי הֶסֵּט, וְטָהוֹר מִשּׁוּם דְּהַנּוֹשֵׂא אֵין בּוֹ רוּחַ חַיִּים, וּלְכָךְ תָּנֵי חוּץ מִן הָאָדָם. וְקָשֶׁה, דְּהֵיכִי מְטַהֲרִין הָכָא לְגַמְרֵי: {ח} דְּאֵין לְמֵדִין מִן הַכְּלָלוֹת: {ט} כֵּיצַד. אַרֵישָׁא נַמִּי קָאֵי: {י} אֶצְבָּעוֹ. נִרְאֶה לִי שֶׁטֻּמְאָה זוֹ מִדִּבְרֵיהֶם. וּלְפִי זֶה צָרִיךְ לוֹמַר דְּמִשְׁכָּב וּמוֹשָׁב דְּסֵיפָא לֹא מַיְרֵי כְּשֶׁאֶצְבָּעוֹ בִּלְבַד מִלְמַעְלָן, דִּבְהוּ לְעוֹלָם רֻבּוֹ בָעִינַן. אֶלָּא מַתְנִיתִין דְּהָכָא וּדְהָתָם לָאו בַּחֲדָא שִׁיטָה, וְהָא כִדְאִיתָא וְהָא כִדְאִיתָא: {יא} שְׁנַיִם. לְפִי שֶׁהַמִּשְׁכָּב אַב הַטֻּמְאָה אֲפִלּוּ כְשֶׁנִּתְרוֹקֵן הַזָּב מִמֶּנּוּ. כִּדְיָלֵיף לֵיהּ בְּתוֹרַת כֹּהֲנִים:

{יב} וְאַף שֶׁלֹּא אָמְרוּ לְעֵיל אֶלָּא תַחַת הַזָּב, וַדַּאי דְּלֹא חֲמִירָא מִשְׁכָּב מִזָּב עַצְמוֹ, וּמִמֵּילָא נַשְׁמְעִינֵיהּ: {יג} וַאֲפִלּוּ לֹא הִפְסִיק בֵּינֵיהֶם אֶלָּא נְיָר. הָרַמְבַּ"ם: {יד} דְּאֵין לְךָ מִטַּמֵּא בְמַשָּׂא בִּנְבֵלָה אֶלָּא הָאָדָם: {טו} צָרִיךְ לוֹמַר וְנִשָּׂא. וְכֵן הוּא בָּרַ"שׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טז} וּמִדְּלֹא תָנֵי גַּבֵּי מִשְׁכָּב חוּץ מִן הַמַּסִּיט, צָרִיךְ לוֹמַר דִּמְטַמֵּא בְלֹא מַסִּיט. וְטַעֲמָא נִרְאֶה לִי, מִשּׁוּם דְּאַחְמִיר רַחֲמָנָא בְּמִשְׁכָּב דַּאֲפִלּוּ הַנּוֹגֵעַ מְטַמֵּא בְגָדִים, וְגַבֵּי נְבֵלָה תְּנַן חֲשׂוּכֵי בְגָדִים בְּמַגָּע. הָרַ"שׁ: {יז} הַנּוֹשֵׂא. נַמִּי בְאָדָם. דּוּמְיָא דִנְבֵלָה: {יח} הַמַּאֲהִיל. בֵּין שֶׁהַמֵּת תַּחְתָּיו וְהָאָדָם נִשָּׂא בֵּין שֶׁהַמֵּת מַאֲהִיל עָלָיו וְהוּא נִשָּׂא, מְטַמֵּא טֻמְאַת אֹהֶל: {יט} לִרְבוּתָא כָּתַב כֵּן, דַּאֲפִלּוּ מִשּׁוּם מַשָּׂא שֶׁמַּכְבִּיד עָלָיו אֵינוֹ מְטַמֵּא, שֶׁאֵין בַּמֵּת מִשּׁוּם אֶבֶן מְסָמָא. וּמִכָּל מָקוֹם בְּאֵינָהּ רְצוּצָה מַיְרֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} וְטַעֲמָא נִרְאֶה לִי, דְּטֻמְאַת מַשָּׂא דְּמֵת יָלְפִינַן בְּקַל וָחֹמֶר מִנְּבֵלָה, וְאָמְרִינַן דַּיּוֹ שֶׁלֹּא יְטַמֵּא מַשָּׂאוֹ אֶלָּא אִם כֵּן הִסִּיט, כְּמוֹ בִּנְבֵלָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:


פרק ה - משנה ה

{כא} כִּדְיָלֵיף לְהוּ בְּסִפְרָא. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כב} הָרַ"שׁ. וּלְשִׁיטָתֵיהּ אָזֵיל כוּ', אֲבָל הָרַ"ב בְּחִנָּם נַקְטֵיהּ וְאֵין בּוֹ צֹרֶךְ לְפֵרוּשׁ הַמִּשְׁנָה. וְגַם בְּעִקַּר הַדִּין נִרְאֶה לִי דְּלָאו כְּהִלְכָתָא הוּא. וּרְאָיָה מִמִּשְׁנָה ה' ו' דְּפֶרֶק ד':


פרק ה - משנה ו

{כג} שְׁנַיִם. בְּמַגָּע וְלֹא בְמַשָּׂא: {כד} הַמַּסִּיט. כְּלוֹמַר, וְאֵינוֹ נוֹשֵׂא וְנִשָּׂא. וּבֵין שֶׁהַטָּהוֹר הִסִּיט לַטָּמֵא אוֹ אִפְּכָא. וְקָאֵי אַזָּב וְכוּ'. אֲבָל בִּמְצֹרָע וּמִשְׁכָּב וּמוֹשָׁב לֵיכָּא לְפָרוֹשֵׁי הָכִי, דְּהָא כָּל טֻמְאוֹת הַמַּסִּיט טְהוֹרוֹת חוּץ מִשֶּׁל זָב: {כה} בְּלֵידָה יָבֵשְׁתָּא: {כו} הַנּוֹשֵׂא כוּ'. כְּלוֹמַר וַאֲפִלּוּ אֵינוֹ מַסִּיט. מַה שֶּׁאֵין כֵּן נְבֵלָה וּמֵת דְּמִשְׁנָה ג' דְּבָעִינַן דַּוְקָא שֶׁיְּהֵא נִיסָט, עַיֵּן שָׁם:


פרק ה - משנה ז

{כז} הַמַּסִּיט. אֲבָל בְּמַשָּׂא בְּלֹא הֶסֵּט לֹא מְטַמֵּא, וְדָמוּ לִנְבֵלָה וּמֵת, דְּכֻלְּהוּ אִמְּעִיטוּ מִקְּרָא מֵאֶבֶן מְסָמָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כח} הָרַמְבַּ"ם. וּלְשִׁיטָתֵיהּ דִּסְבִירָא לֵיהּ דִּלְתַנָּא קַמָּא נַמִּי נוֹשֵׂא מְטַמֵּא בְּלֹא הֶסֵּט. אֲבָל הָרַ"שׁ מְפָרֵשׁ כָּאן כִּפְשׁוּטוֹ, דְּרַבִּי אֱלִיעֶזֶר פָּלֵיג וּסְבִירָא לֵיהּ דְּנוֹשֵׂא בְלֹא מַסִּיט נַמִּי מְטַמֵּא, דְּלֹא מְמַעֵט לְהוּ מִקְּרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:


פרק ה - משנה ח

{כט} וְהַמַּסִּיטוֹ. לַמֶּרְכָּב. אֲבָל מֶרְכָּב הַמַּסִּיט לַטָּהוֹר, לֹא, וּכְמוֹ בְמִשְׁכָּב:


פרק ה - משנה ט

{ל} טָהוֹר. הָאָדָם שֶׁהַנְּבֵלָה בְּפִיו וַעֲדַיִן לֹא בְלָעָהּ. וְקָא מַשְׁמַע לָן דְּלֹא תֵימָא דְּבִהְיוֹתָהּ נִסְתֶּרֶת וְנִכְנֶסֶת לַאֲוִיר הַתַּנּוּר תִּהְיֶה כְמוֹ בְלוּעָה, וְטָהוֹר הַתַּנּוּר, מַיְרֵי כְשֶׁהִיא בְלוּעָה. הָרַמְבַּ"ם: {לא} כְּלוֹמַר, דְּכָל אֵלּוּ דְַּאֲמַרַן דִּמְטַמְּאִין שְׁנַיִם מִשּׁוּם בְּגָדִים, כָּל אֵלּוּ אֵין כוּ':


פרק ה - משנה י

{לב} שֶׁאֵין כוּ'. דְּאִלּוּ יֵשׁ בָּהֶם, אַף הַנּוֹגֵעַ בָּהֶם מְטַמֵּא בְגָדִים, לְפִי שֶׁאִי אֶפְשָׁר שֶׁיִּגַּע בַּמַּיִם שֶׁלֹּא יַסִּיט אוֹתָם, וַהֲרֵי הוּא מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד. אֲבָל אִם קָרְשׁוּ וְנַעֲשׂוּ כִגְלִיד דְּאֶפְשָׁר לִגַּע בְּלֹא הֶסֵּט, תָּנֵי בְפֶרֶק קַמָּא דְכֵלִים מִשְׁנָה ב' חֲשׂוּכֵי בְגָדִים בְּמַגָּע. וּכְמוֹ שֶׁכָּתַב הָרַ"ב בְּמִשְׁנָה ח'. וְהַתַּנָּא הָתָם וְהָכָא מִלְּתָא דִפְסִיקָא נָקַט: {לג} וְאֵין הַלָּשׁוֹן מַשְׁמַע כֵּן, דְּהָכִי הֲוָה לֵיהּ לְמִתְנֵי, בַּמֶּה דְבָרִים אֲמוּרִים בְּאָדָם, אֲבָל אֳכָלִים וְכֵלִים הַנּוֹגְעִים בְּאֶחָד כוּ'. וְהָרַמְבַּ"ם פֵּרֵשׁ, כָּל הַנּוֹגֵעַ בִּשְׁאָר אֲבוֹת הַטֻּמְאוֹת חוּץ מֵאֵלּוּ שֶׁמָּנָה עַד הֵנָּה [שֶׁמְּטַמְּאִין שְׁנַיִם כוּ']. וְאָמְרוֹ חוּץ מִן הָאָדָם, רָצָה לוֹמַר הַמֵּת, שֶׁגַּם הוּא לֹא נִמְנָה כָאן, וְנוֹגֵעַ מְטַמֵּא שְׁנַיִם כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:


פרק ה - משנה יא

{לד} וּמִשּׁוּם דְּנוֹגֵעַ לֹא כְתִיב בְּהֶדְיָא תָּלֵי לֵיהּ בְּשֶׁרֶץ. אֲבָל בַּגְּמָרָא מַסִּיק דְּמַה שֶּׁרֶץ מְטַמֵּא בְאֹנֶס [אִם נָפַל עָלָיו שֶׁרֶץ מֵת], אַף בַּעַל קֶרִי: {לה} אֲבָל בַּגְּמָרָא, דְּזֶהוּ מִלְּתָא דִפְשִׁיטָא הוּא, דִּקְרָא כְתִיב. וּלְמַחֲנוֹתָם אִצְטְרִיךְ, סָלְקָא דַעְתָּךְ אֲמֵינָא וּתְהִי נִדָּתָהּ עָלָיו כְּתִיב, דְּיִשְׁתַּלַּח חוּץ לִשְׁתֵּי מַחֲנוֹת כְּנִדָּה, קָא מַשְׁמַע לָן. וּבְבַעַל קֶרִי כְּמַגַּע שֶׁרֶץ לֵיכָּא לְמֵימַר הָכִי, דְּבַעַל קֶרִי חוּץ לִשְׁתֵּי מַחֲנוֹת, וְשֶׁרֶץ שָׁרֵי בְּמַחֲנֵה לְוִיָּה. אִי נַמִּי, הָכָא סֵיפָא אִצְטְרִיךְ לֵיהּ אֶלָּא כוּ': {לו} שֶׁחָמוּר כוּ'. וְקָשֶׁה, הָא חָמוּר נַמִּי טְמֵא מֵת מִבּוֹעֵל נִדָּה, שֶׁטָּעוּן הַזָּאָה שְׁלִישִׁי וּשְׁבִיעִי וְחֶרֶב הֲרֵי הוּא כְּחָלָל. וְשֶׁמָּא בַסֵּיפָא דִבְרַיְתָא חֲשִׁיב לְהוּ. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב וְתוֹסָפוֹת חֲדָשִׁים:


פרק ה - משנה יב

{לז} דְּמִסְתַּבְּרָא דְּהַבָּא כוּ' נַמִּי בִשְׁלֹשָׁה לֻגִּין הוּא, כֵּיוָן דְּהָא בְהָא תַּלְיָא. וְכֵן כָּתַב הָרַמְבַּ"ם: {לח} דִּכְתִיב וּבָא הַשֶּׁמֶשׁ וְטָהֵר. גְּמָרָא:
