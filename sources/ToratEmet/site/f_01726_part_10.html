פרשת האזינו




דברים פרק-לב

{א} הַֽאֲזִ֥ינוּ הַשָּׁמַ֖יִם וַֽאֲדַבֵּ֑רָה וְתִשְׁמַ֥ע הָאָ֖רֶץ אִמְרֵי-פִֽי:  מ  הַֽאֲזִ֥ינוּ הַשָּׁמַ֖יִם וַֽאֲדַבֵּ֑רָה וְתִשְׁמַ֥ע הָאָ֖רֶץ אִמְרֵי-פִֽי:  ת  אַצִיתוּ שְׁמַיָא וֶאֱמַלֵל וְתִשְׁמַע אַרְעָא מֵימְרֵי פֻמִי:

{ב} יַֽעֲרֹ֤ף כַּמָּטָר֙ לִקְחִ֔י תִּזַּ֥ל כַּטַּ֖ל אִמְרָתִ֑י כִּשְׂעִירִ֣ם עֲלֵי-דֶ֔שֶׁא וְכִרְבִיבִ֖ים עֲלֵי-עֵֽשֶׂב:  מ  יַֽעֲרֹ֤ף כַּמָּטָר֙ לִקְחִ֔י תִּזַּ֥ל כַּטַּ֖ל אִמְרָתִ֑י כִּשְׂעִירִ֣ם עֲלֵי-דֶ֔שֶׁא וְכִרְבִיבִ֖ים עֲלֵי-עֵֽשֶׂב:  ת  יְבַסַם כְּמִטְרָא אוּלְפָּנִי יִתְקַבַּל כְּטַלָא מֵימְרִי כְּרוּחֵי מִטְרָא דְנָשְׁבִין עַל דִתְאָה וְכִרְסִיסֵי מַלְקוֹשָׁא דִי עַל עִשְׂבָּא:

{ג} כִּ֛י שֵׁ֥ם יְהֹוָ֖ה אֶקְרָ֑א הָב֥וּ גֹ֖דֶל לֵֽאלֹהֵֽינוּ:  מ  כִּ֛י שֵׁ֥ם יְהֹוָ֖ה אֶקְרָ֑א הָב֥וּ גֹ֖דֶל לֵֽאלֹהֵֽינוּ:  ת  אֲרֵי בִּשְׁמָא דַיְיָ אֲנָא מְצַלֵי הָבוּ רְבוּתָא קֳדָם אֱלָהָנָא:

{ד} הַצּוּר֙ תָּמִ֣ים פָּֽעֳל֔וֹ כִּ֥י כָל-דְּרָכָ֖יו מִשְׁפָּ֑ט אֵ֤ל אֱמוּנָה֙ וְאֵ֣ין עָ֔וֶל צַדִּ֥יק וְיָשָׁ֖ר הֽוּא:  מ  הַצּוּר֙ תָּמִ֣ים פָּֽעֳל֔וֹ כִּ֥י כָל-דְּרָכָ֖יו מִשְׁפָּ֑ט אֵ֤ל אֱמוּנָה֙ וְאֵ֣ין עָ֔וֶל צַדִּ֥יק וְיָשָׁ֖ר הֽוּא:  ת  תַּקִיפָא דְשָׁלְמִין עוֹבָדוֹהִי אֲרֵי כָּל אֹרְחָתֵהּ דִינָא אֱלָהָא מְהֵימְנָא דְמִן קֳדָמוֹהִי עַוְלָה לָא נָפֵק מִן קֳדָם דְזַכַּי וְקַשִׁיט הוּא:

{ה} שִׁחֵ֥ת ל֛וֹ לֹ֖א בָּנָ֣יו מוּמָ֑ם דּ֥וֹר עִקֵּ֖שׁ וּפְתַלְתֹּֽל:  מ  שִׁחֵ֥ת ל֛וֹ לֹ֖א בָּנָ֣יו מוּמָ֑ם דּ֥וֹר עִקֵּ֖שׁ וּפְתַלְתֹּֽל:  ת  חַבִּילוּ לְהוֹן לָא לֵהּ בְּנַיָא דִי פְלָחוּ לְטַעֲוָתָא דָרָא דְאַשְׁנִיוּ עוֹבָדוֹהִי וְאִשְׁתַּנִיוּ:

{ו} הַלְיהֹוָה֙ תִּגְמְלוּ-זֹ֔את עַ֥ם נָבָ֖ל וְלֹ֣א חָכָ֑ם הֲלוֹא-הוּא֙ אָבִ֣יךָ קָּנֶ֔ךָ ה֥וּא עָֽשְׂךָ֖ וַֽיְכֹֽנְנֶֽךָ:  מ  הַלְיהֹוָה֙ תִּגְמְלוּ-זֹ֔את עַ֥ם נָבָ֖ל וְלֹ֣א חָכָ֑ם הֲלוֹא-הוּא֙ אָבִ֣יךָ קָּנֶ֔ךָ ה֥וּא עָֽשְׂךָ֖ וַֽיְכֹֽנְנֶֽךָ:  ת  הָא קֳדָם יְיָ אַתּוּן גָמְלִין דָא עַמָא דְקַבִּילוּ אוֹרַיְתָא וְלָא חַכִּימוּ הֲלָא הוּא אֲבוּךְ וְאַתְּ דִי לֵהּ הוּא עָבְדָךְ וְאַתְקְנָךְ:

{ז}  שני  זְכֹר֙ יְמ֣וֹת עוֹלָ֔ם בִּ֖ינוּ שְׁנ֣וֹת דֹּר-וָדֹ֑ר שְׁאַ֤ל אָבִ֨יךָ֙ וְיַגֵּ֔דְךָ זְקֵנֶ֖יךָ וְיֹ֥אמְרוּ לָֽךְ:  מ  זְכֹר֙ יְמ֣וֹת עוֹלָ֔ם בִּ֖ינוּ שְׁנ֣וֹת דֹּר-וָדֹ֑ר שְׁאַ֤ל אָבִ֨יךָ֙ וְיַגֵּ֔דְךָ זְקֵנֶ֖יךָ וְיֹ֥אמְרוּ לָֽךְ:  ת  אִדְכַּר יוֹמִין דְמִן עָלְמָא אִסְתַּכַּל בִּשְׁנֵי דַר וְדַר שְׁאַל אֲבוּךְ וִיחַוִּי לָךְ סָבָיךְ וְיֵימְרוּן לָךְ:

{ח} בְּהַנְחֵ֤ל עֶלְיוֹן֙ גּוֹיִ֔ם בְּהַפְרִיד֖וֹ בְּנֵ֣י אָדָ֑ם יַצֵּב֙ גְּבֻלֹ֣ת עַמִּ֔ים לְמִסְפַּ֖ר בְּנֵ֥י יִשְׂרָאֵֽל:  מ  בְּהַנְחֵ֤ל עֶלְיוֹן֙ גּוֹיִ֔ם בְּהַפְרִיד֖וֹ בְּנֵ֣י אָדָ֑ם יַצֵּב֙ גְּבֻלֹ֣ת עַמִּ֔ים לְמִסְפַּ֖ר בְּנֵ֥י יִשְׂרָאֵֽל:  ת  בְּאַחֲסָנָא עִלָאָה עַמְמַיָא בְּפָרָשׁוּתֵהּ בְּנֵי אֲנָשָׁא קַיִים תְּחוּמֵי עַמְמַיָא לְמִנְיַן בְּנֵי יִשְׂרָאֵל:

{ט} כִּ֛י חֵ֥לֶק יְהֹוָ֖ה עַמּ֑וֹ יַֽעֲקֹ֖ב חֶ֥בֶל נַֽחֲלָתֽוֹ:  מ  כִּ֛י חֵ֥לֶק יְהֹוָ֖ה עַמּ֑וֹ יַֽעֲקֹ֖ב חֶ֥בֶל נַֽחֲלָתֽוֹ:  ת  אֲרֵי חֳלָקָא דַיְיָ עַמֵהּ יַעֲקֹב עֲדַב אַחֲסַנְתֵּהּ:

{י} יִמְצָאֵ֨הוּ֙ בְּאֶ֣רֶץ מִדְבָּ֔ר וּבְתֹ֖הוּ יְלֵ֣ל יְשִׁמֹ֑ן יְסֹֽבְבֶ֨נְהוּ֙ יְב֣וֹנְנֵ֔הוּ יִצְּרֶ֖נְהוּ כְּאִישׁ֥וֹן עֵינֽוֹ:  מ  יִמְצָאֵ֨הוּ֙ בְּאֶ֣רֶץ מִדְבָּ֔ר וּבְתֹ֖הוּ יְלֵ֣ל יְשִׁמֹ֑ן יְסֹֽבְבֶ֨נְהוּ֙ יְב֣וֹנְנֵ֔הוּ יִצְּרֶ֖נְהוּ כְּאִישׁ֥וֹן עֵינֽוֹ:  ת  סַפִּיק צָרְכֵּיהוֹן בְּאַרְעָא מַדְבְּרָא וּבֵית צַחֲוָנָא אֲתַר דִי לֵית מַיָא אַשְׁרִנוּן סְחוֹר סְחוֹר לִשְׁכִנְתֵּהּ אַלֵפִנוּן פִּתְגָמֵי אוֹרַיְתָא נְטָרִנוּן כְּבָבַת עֵינֵיהוֹן:

{יא} כְּנֶ֨שֶׁר֙ יָעִ֣יר קִנּ֔וֹ עַל-גּֽוֹזָלָ֖יו יְרַחֵ֑ף יִפְרֹ֤שׂ כְּנָפָיו֙ יִקָּחֵ֔הוּ יִשָּׂאֵ֖הוּ עַל-אֶבְרָתֽוֹ:  מ  כְּנֶ֨שֶׁר֙ יָעִ֣יר קִנּ֔וֹ עַל-גּֽוֹזָלָ֖יו יְרַחֵ֑ף יִפְרֹ֤שׂ כְּנָפָיו֙ יִקָּחֵ֔הוּ יִשָּׂאֵ֖הוּ עַל-אֶבְרָתֽוֹ:  ת  כְּנִשְׁרָא דִמְחִישׁ לְקִנֵהּ עַל בְּנוֹהִי מִתְחוֹפֵף פָּרִיס גַדְפּוֹהִי מְקַבֶּלְהוֹן מְנַטֶלְהוֹן עַל תְּקוֹף אֵבָרוֹהִי:

{יב} יְהֹוָ֖ה בָּדָ֣ד יַנְחֶ֑נּוּ וְאֵ֥ין עִמּ֖וֹ אֵ֥ל נֵכָֽר:  מ  יְהֹוָ֖ה בָּדָ֣ד יַנְחֶ֑נּוּ וְאֵ֥ין עִמּ֖וֹ אֵ֥ל נֵכָֽר:  ת  יְיָ בִּלְחוֹדוֹהִי עֲתִיד לְאַשְׁרָיוּתְהוֹן בְּעָלְמָא דְּהוּא עָתִיד לְאִתְחַדָתָא וְלָא יִתְקַיַם קֳדָמוֹהִי פָּלְחַן טַּעֲוָן:

{יג}  שלישי  יַרְכִּבֵ֨הוּ֙ עַל-בָּ֣מֳתֵי אָ֔רֶץ וַיֹּאכַ֖ל תְּנוּבֹ֣ת שָׂדָ֑י וַיֵּֽנִקֵ֤הוּ דְבַשׁ֙ מִסֶּ֔לַע וְשֶׁ֖מֶן מֵֽחַלְמִ֥ישׁ צֽוּר:  מ  יַרְכִּבֵ֨הוּ֙ עַל-בָּ֣מֳתֵי אָ֔רֶץ וַיֹּאכַ֖ל תְּנוּבֹ֣ת שָׂדָ֑י וַיֵּֽנִקֵ֤הוּ דְבַשׁ֙ מִסֶּ֔לַע וְשֶׁ֖מֶן מֵֽחַלְמִ֥ישׁ צֽוּר:  ת  אַשְׁרִנוּן עַל תָּקְפֵּי אַרְעָא וְאוֹכָלִנוּן בִּזַת סָנְאֵיהוֹן וִיהַב לְהוֹן בִּזַת שַׁלִיטֵי קִרְוִין וְנִכְסֵי יָתְבֵי כְּרַכִּין תַּקִיפִין:

{יד} חֶמְאַ֨ת בָּקָ֜ר וַֽחֲלֵ֣ב צֹ֗אן עִם-חֵ֨לֶב כָּרִ֜ים וְאֵילִ֤ים בְּנֵֽי-בָשָׁן֙ וְעַתּוּדִ֔ים עִם-חֵ֖לֶב כִּלְי֣וֹת חִטָּ֑ה וְדַם-עֵנָ֖ב תִּשְׁתֶּה-חָֽמֶר:  מ  חֶמְאַ֨ת בָּקָ֜ר וַֽחֲלֵ֣ב צֹ֗אן עִם-חֵ֨לֶב כָּרִ֜ים וְאֵילִ֤ים בְּנֵֽי-בָשָׁן֙ וְעַתּוּדִ֔ים עִם-חֵ֖לֶב כִּלְי֣וֹת חִטָּ֑ה וְדַם-עֵנָ֖ב תִּשְׁתֶּה-חָֽמֶר:  ת  יְהַב לְהוֹן בִּזַת מַלְכֵיהוֹן וְשַׁלִיטֵיהוֹן עִם עוֹתַר רַבְרְבֵיהוֹן וְתַקִפֵיהוֹן עַמָא דְאַרְעֲהוֹן וְאַחֲסַנְתְּהוֹן עִם בִּזַת חֵילֵיהוֹן וּמַשִׁרְיָתְהוֹן וְדַם גִבָּרֵיהוֹן יִתָּשַׁד כְּמַיָא:

{טו} וַיִּשְׁמַ֤ן יְשֻׁרוּן֙ וַיִּבְעָ֔ט שָׁמַ֖נְתָּ עָבִ֣יתָ כָּשִׂ֑יתָ וַיִּטּשׁ֙ אֱל֣וֹהַּ עָשָׂ֔הוּ וַיְנַבֵּ֖ל צ֥וּר יְשֻֽׁעָתֽוֹ:  מ  וַיִּשְׁמַ֤ן יְשֻׁרוּן֙ וַיִּבְעָ֔ט שָׁמַ֖נְתָּ עָבִ֣יתָ כָּשִׂ֑יתָ וַיִּטּשׁ֙ אֱל֣וֹהַּ עָשָׂ֔הוּ וַיְנַבֵּ֖ל צ֥וּר יְשֻֽׁעָתֽוֹ:  ת  וְעַתַּר יִשְׂרָאֵל וּבְעַט אַצְלַח תְּקוֹף קְנָא נִכְסִין וּשְׁבַק פָּלְחַן אֱלָהָא דְעָבְדֵהּ וְאַרְגֵז קֳדָם תַּקִיפָא דְפָרְקֵהּ:

{טז} יַקְנִאֻ֖הוּ בְּזָרִ֑ים בְּתֽוֹעֵבֹ֖ת יַכְעִיסֻֽהוּ:  מ  יַקְנִאֻ֖הוּ בְּזָרִ֑ים בְּתֽוֹעֵבֹ֖ת יַכְעִיסֻֽהוּ:  ת  אַקְנִיאוּ קֳדָמוֹהִי בְּפָלְחַן טַעֲוָן בְּתוֹעֶבְתָּא אַרְגִיזוּ קֳדָמוֹהי:

{יז} יִזְבְּח֗וּ לַשֵּׁדִים֙ לֹ֣א אֱלֹ֔הַּ אֱלֹהִ֖ים לֹ֣א יְדָע֑וּם חֲדָשִׁים֙ מִקָּרֹ֣ב בָּ֔אוּ לֹ֥א שְׂעָר֖וּם אֲבֹֽתֵיכֶֽם:  מ  יִזְבְּח֗וּ לַשֵּׁדִים֙ לֹ֣א אֱלֹ֔הַּ אֱלֹהִ֖ים לֹ֣א יְדָע֑וּם חֲדָשִׁים֙ מִקָּרֹ֣ב בָּ֔אוּ לֹ֥א שְׂעָר֖וּם אֲבֹֽתֵיכֶֽם:  ת  דבָּחוּ לשֵׁדין דִלֵית בְּהוֹן צְרוֹךְ דַחֲלָן דְלָא יְדָעֻנוּן חַדְתִּין דִי מִקָרִיב אִתְעֲבִידוּ לָא אִתְעַסְקוּ בְהוֹן אֲבָהָתְכוֹן:

{יח} צ֥וּר יְלָֽדְךָ֖ תֶּ֑שִׁי וַתִּשְׁכַּ֖ח אֵ֥ל מְחֹֽלְלֶֽךָ:  מ  צ֥וּר יְלָֽדְךָ֖ תֶּ֑שִׁי וַתִּשְׁכַּ֖ח אֵ֥ל מְחֹֽלְלֶֽךָ:  ת  דַחֲלַת תַקִיפָא דִבְרָאָךְ אִתְנְשֵׁתָא שְׁבַקְתָּא פָלְחַן אֱלָהָא דְעָבְדָךְ:

{יט}  רביעי  וַיַּ֥רְא יְהֹוָ֖ה וַיִּנְאָ֑ץ מִכַּ֥עַס בָּנָ֖יו וּבְנֹתָֽיו:  מ  וַיַּ֥רְא יְהֹוָ֖ה וַיִּנְאָ֑ץ מִכַּ֥עַס בָּנָ֖יו וּבְנֹתָֽיו:  ת  וּגְלֵי קֳדָם יְיָ וּתְקֵף רָגְזֵהּ מִדְאַרְגִיזוּ קֳדָמוֹהִי בְּנִין וּבְנָן:

{כ} וַיֹּ֗אמֶר אַסְתִּ֤ירָה פָנַי֙ מֵהֶ֔ם אֶרְאֶ֖ה מָ֣ה אַֽחֲרִיתָ֑ם כִּ֣י ד֤וֹר תַּהְפֻּכֹת֙ הֵ֔מָּה בָּנִ֖ים לֹֽא-אֵמֻ֥ן בָּֽם:  מ  וַיֹּ֗אמֶר אַסְתִּ֤ירָה פָנַי֙ מֵהֶ֔ם אֶרְאֶ֖ה מָ֣ה אַֽחֲרִיתָ֑ם כִּ֣י ד֤וֹר תַּהְפֻּכֹת֙ הֵ֔מָּה בָּנִ֖ים לֹֽא-אֵמֻ֥ן בָּֽם:  ת  וַאֲמַר אֶסַלֵק שְׁכִנְתִּי מִנְהוֹן גְלֵי קֳדָמַי מָה יְהֵי בְסוֹפֵיהוֹן אֲרֵי דָרָא דְאַשְׁנִיוּ אִנוּן בְנַיָא דִלֵית בְּהוֹן הֵימָנוּ:

{כא} הֵ֚ם קִנְא֣וּנִי בְלֹא-אֵ֔ל כִּֽעֲס֖וּנִי בְּהַבְלֵיהֶ֑ם וַֽאֲנִי֙ אַקְנִיאֵ֣ם בְּלֹא-עָ֔ם בְּג֥וֹי נָבָ֖ל אַכְעִיסֵֽם:  מ  הֵ֚ם קִנְא֣וּנִי בְלֹא-אֵ֔ל כִּֽעֲס֖וּנִי בְּהַבְלֵיהֶ֑ם וַֽאֲנִי֙ אַקְנִיאֵ֣ם בְּלֹא-עָ֔ם בְּג֥וֹי נָבָ֖ל אַכְעִיסֵֽם:  ת  אִנוּן אַקְנִיאוּ קֳדָמַי בְּלָא דַחֲלָא אַרְגִיזוּ קֳדָמַי בְּפָלְחַן טַעֲוָן וַאֲנָא אַקְנִנוּן בְּלָא עַם בְּעַמָא טַּפְּשָׁא אַרְגֵזִנוּן:

{כב} כִּי-אֵשׁ֙ קָֽדְחָ֣ה בְאַפִּ֔י וַתִּיקַ֖ד עַד-שְׁא֣וֹל תַּחְתִּ֑ית וַתֹּ֤אכַל אֶ֨רֶץ֙ וִֽיבֻלָ֔הּ וַתְּלַהֵ֖ט מֽוֹסְדֵ֥י הָרִֽים:  מ  כִּי-אֵשׁ֙ קָֽדְחָ֣ה בְאַפִּ֔י וַתִּיקַ֖ד עַד-שְׁא֣וֹל תַּחְתִּ֑ית וַתֹּ֤אכַל אֶ֨רֶץ֙ וִֽיבֻלָ֔הּ וַתְּלַהֵ֖ט מֽוֹסְדֵ֥י הָרִֽים:  ת  אֲרֵי קִדוּם תַּקִיף כְּאֶשָׁא נְפַק מִקָדָמַי בִּרְגַז וְשֵׁצֵי עַד שְׁאוֹל אֲרָעִית וְאָסֵף אַרְעָא וַעֲלַלְתַּהּ וְשֵׁצְיַת עַד סְיָפֵי טוּרַיָא:

{כג} אַסְפֶּ֥ה עָלֵ֖ימוֹ רָע֑וֹת חִצַּ֖י אֲכַלֶּה-בָּֽם:  מ  אַסְפֶּ֥ה עָלֵ֖ימוֹ רָע֑וֹת חִצַּ֖י אֲכַלֶּה-בָּֽם:  ת  אָסֵף עֲלֵיהוֹן בִּישִׁין מַכְתָּשַׁי אֱשֵׁצֵי בְהוֹן:

{כד} מְזֵ֥י רָעָ֛ב וּלְחֻ֥מֵי רֶ֖שֶׁף וְקֶ֣טֶב מְרִירִ֑י וְשֶׁן-בְּהֵמֹת֙ אֲשַׁלַּח-בָּ֔ם עִם-חֲמַ֖ת זֹֽחֲלֵ֥י עָפָֽר:  מ  מְזֵ֥י רָעָ֛ב וּלְחֻ֥מֵי רֶ֖שֶׁף וְקֶ֣טֶב מְרִירִ֑י וְשֶׁן-בְּהֵמֹת֙ אֲשַׁלַּח-בָּ֔ם עִם-חֲמַ֖ת זֹֽחֲלֵ֥י עָפָֽר:  ת  נְפִיחֵי כָפָן וַאֲכוּלֵי עוּף וּכְתִישֵׁי רוּחִין בִּישָׁן וְשֵׁן חֵיוַת בָּרָא אֱגָרֵי בְהוֹן עִם חֲמַת תַּנִינַיָא דְזָחֲלִין בְּעַפְרָא:

{כה} מִחוּץ֙ תְּשַׁכֶּל-חֶ֔רֶב וּמֵֽחֲדָרִ֖ים אֵימָ֑ה גַּם-בָּחוּר֙ גַּם-בְּתוּלָ֔ה יוֹנֵ֖ק עִם-אִ֥ישׁ שֵׂיבָֽה:  מ  מִחוּץ֙ תְּשַׁכֶּל-חֶ֔רֶב וּמֵֽחֲדָרִ֖ים אֵימָ֑ה גַּם-בָּחוּר֙ גַּם-בְּתוּלָ֔ה יוֹנֵ֖ק עִם-אִ֥ישׁ שֵׂיבָֽה:  ת  מִבָּרָא תְּתַכֵּל חַרְבָּא וּמִתְּוָנַיָא חַרְגַת מוֹתָא אַף עוּלֵמֵיהוֹן אַף עוּלֵמָתְהוֹן יָנְקֵיהוֹן עִם אֱנַשׁ סָבֵיהוֹן:

{כו} אָמַ֖רְתִּי אַפְאֵיהֶ֑ם אַשְׁבִּ֥יתָה מֵּֽאֱנ֖וֹשׁ זִכְרָֽם:  מ  אָמַ֖רְתִּי אַפְאֵיהֶ֑ם אַשְׁבִּ֥יתָה מֵּֽאֱנ֖וֹשׁ זִכְרָֽם:  ת  אֲמָרִית יֵחוֹל רוּגְזִי עֲלֵיהוֹן וֶאֱשֵׁצִנוּן אֱבַטַל מִבְּנֵי אֲנָשָׁא דוּכְרָנֵיהוֹן:

{כז} לוּלֵ֗י כַּ֤עַס אוֹיֵב֙ אָג֔וּר פֶּן-יְנַכְּר֖וּ צָרֵ֑ימוֹ פֶּן-יֹֽאמְרוּ֙ יָדֵ֣נוּ רָ֔מָה וְלֹ֥א יְהֹוָ֖ה פָּעַ֥ל כָּל-זֹֽאת:  מ  לוּלֵ֗י כַּ֤עַס אוֹיֵב֙ אָג֔וּר פֶּן-יְנַכְּר֖וּ צָרֵ֑ימוֹ פֶּן-יֹֽאמְרוּ֙ יָדֵ֣נוּ רָ֔מָה וְלֹ֥א יְהֹוָ֖ה פָּעַ֥ל כָּל-זֹֽאת:  ת  אִלוּלָפוֹן רָגְזָא דְסָנְאָה כְּנִישׁ דִילְמָא יִתְרַבְרַב בְּעֵל דְבָבָא דִילְמָא יֵימְרוּן יְדָנָא תְקִיפַת לָנָא וְלָא מִן קֳדָם יְיָ הֲוַת כָּל דָא:

{כח} כִּי-ג֛וֹי אֹבַ֥ד עֵצ֖וֹת הֵ֑מָּה וְאֵ֥ין בָּהֶ֖ם תְּבוּנָֽה:  מ  כִּי-ג֛וֹי אֹבַ֥ד עֵצ֖וֹת הֵ֑מָּה וְאֵ֥ין בָּהֶ֖ם תְּבוּנָֽה:  ת  אֲרֵי עַם מְאַבְּדֵי עֵצָה אִנוּן וְלֵית בְּהוֹן סוּכְלְתָנוּ:

{כט}  חמישי  ל֥וּ חָֽכְמ֖וּ יַשְׂכִּ֣ילוּ זֹ֑את יָבִ֖ינוּ לְאַֽחֲרִיתָֽם:  מ  ל֥וּ חָֽכְמ֖וּ יַשְׂכִּ֣ילוּ זֹ֑את יָבִ֖ינוּ לְאַֽחֲרִיתָֽם:  ת  אִלוּ חַכִּימוּ אִסְתַּכָּלוּ בְדָא סְבָרוּ מָה יְהֵי בְסוֹפֵיהוֹן:

{ל} אֵיכָ֞ה יִרְדֹּ֤ף אֶחָד֙ אֶ֔לֶף וּשְׁנַ֖יִם יָנִ֣יסוּ רְבָבָ֑ה אִם-לֹא֙ כִּֽי-צוּרָ֣ם מְכָרָ֔ם וַֽיהֹוָ֖ה הִסְגִּירָֽם:  מ  אֵיכָ֞ה יִרְדֹּ֤ף אֶחָד֙ אֶ֔לֶף וּשְׁנַ֖יִם יָנִ֣יסוּ רְבָבָ֑ה אִם-לֹא֙ כִּֽי-צוּרָ֣ם מְכָרָ֔ם וַֽיהֹוָ֖ה הִסְגִּירָֽם:  ת  אֶכְדֵין יִרְדוֹף חַד אַלְפָא וּתְרֵין יֶעֶרְקוּן לְרִבּוֹתָא אֶלָהֵן (אֲרֵי) תַּקִיפְהוֹן מְסָרִנוּן וַייָ אַשְׁלֵמִנוּן:

{לא} כִּ֛י לֹ֥א כְצוּרֵ֖נוּ צוּרָ֑ם וְאֹֽיְבֵ֖ינוּ פְּלִילִֽים:  מ  כִּ֛י לֹ֥א כְצוּרֵ֖נוּ צוּרָ֑ם וְאֹֽיְבֵ֖ינוּ פְּלִילִֽים:  ת  אֲרֵי לָא כְתָקְפָּנָא תָּקְפְהוֹן וּבַעֲלֵי דְבָבָנָא הֲווֹ דַיָנָנָא:

{לב} כִּֽי-מִגֶּ֤פֶן סְדֹם֙ גַּפְנָ֔ם וּמִשַּׁדְמֹ֖ת עֲמֹרָ֑ה עֲנָבֵ֨מוֹ֙ עִנְּבֵי-ר֔וֹשׁ אַשְׁכְּלֹ֥ת מְרֹרֹ֖ת לָֽמוֹ:  מ  כִּֽי-מִגֶּ֤פֶן סְדֹם֙ גַּפְנָ֔ם וּמִשַּׁדְמֹ֖ת עֲמֹרָ֑ה עֲנָבֵ֨מוֹ֙ עִנְּבֵי-ר֔וֹשׁ אַשְׁכְּלֹ֥ת מְרֹרֹ֖ת לָֽמוֹ:  ת  אֲרֵי כְפֻרְעֲנוּת עַמָא דִסְדוֹם פֻּרְעֲנוּתְהוֹן וְלָקוּתְהוֹן כְּעַם עֲמֹרָה מָחָתְהוֹן בִּישִׁין כְּרֵישֵׁי חֶוְיָן וְתֻשְׁלָמַת עוֹבָדֵיהוֹן כִּמְרָרוּתְהוֹן:

{לג} חֲמַ֥ת תַּנִּינִ֖ם יֵינָ֑ם וְרֹ֥אשׁ פְּתָנִ֖ים אַכְזָֽר:  מ  חֲמַ֥ת תַּנִּינִ֖ם יֵינָ֑ם וְרֹ֥אשׁ פְּתָנִ֖ים אַכְזָֽר:  ת  הָא כְמָרַת תַּנִינַיָא כַּס פֻּרְעֲנוּתְהוֹן וּכְרֵישׁ פְּתָנֵי חֶוְיָן אַכְזְרָאִין:

{לד} הֲלֹא-ה֖וּא כָּמֻ֣ס עִמָּדִ֑י חָת֖וּם בְּאֽוֹצְרֹתָֽי:  מ  הֲלֹא-ה֖וּא כָּמֻ֣ס עִמָּדִ֑י חָת֖וּם בְּאֽוֹצְרֹתָֽי:  ת  הֲלָא כָל עוֹבָדֵיהוֹן גְלָן קֳדָמַי גְנִיזִין לְיוֹם דִינָא בְּאוֹצְרָי:

{לה} לִ֤י נָקָם֙ וְשִׁלֵּ֔ם לְעֵ֖ת תָּמ֣וּט רַגְלָ֑ם כִּ֤י קָרוֹב֙ יוֹם אֵידָ֔ם וְחָ֖שׁ עֲתִדֹ֥ת לָֽמוֹ:  מ  לִ֤י נָקָם֙ וְשִׁלֵּ֔ם לְעֵ֖ת תָּמ֣וּט רַגְלָ֑ם כִּ֤י קָרוֹב֙ יוֹם אֵידָ֔ם וְחָ֖שׁ עֲתִדֹ֥ת לָֽמוֹ:  ת  קֳדָמַי פֻּרְעֲנוּתָא וַאֲנָא אֲשַׁלֵם לְעִדָן דְיִגְלוּן מֵאַרְעֲהוֹן אֲרֵי קָרִיב יוֹם תְּבָרְהוֹן וּמַבַּע דְעָתִיד לְהוֹן:

{לו} כִּֽי-יָדִ֤ין יְהֹוָה֙ עַמּ֔וֹ וְעַל-עֲבָדָ֖יו יִתְנֶחָ֑ם כִּ֤י יִרְאֶה֙ כִּי-אָ֣זְלַת יָ֔ד וְאֶ֖פֶס עָצ֥וּר וְעָזֽוּב:  מ  כִּֽי-יָדִ֤ין יְהֹוָה֙ עַמּ֔וֹ וְעַל-עֲבָדָ֖יו יִתְנֶחָ֑ם כִּ֤י יִרְאֶה֙ כִּי-אָ֣זְלַת יָ֔ד וְאֶ֖פֶס עָצ֥וּר וְעָזֽוּב:  ת  אֲרֵי יְדִין יְיָ דִינָא דְעַמֵהּ וּפֻרְעֲנוּת עֲבְדוֹהִי צַדִיקַיָא יִתְפְרַע אֲרֵי גְלֵי קֳדָמוֹהִי דִבְעִדָן דְתִתְקֵף עֲלֵיהוֹן מָחַת סָנְאָה יְהוֹן מִטַלְטְלִין וּשְׁבִיקִין:

{לז} וְאָמַ֖ר אֵ֣י אֱלֹהֵ֑ימוֹ צ֖וּר חָסָ֥יוּ בֽוֹ:  מ  וְאָמַ֖ר אֵ֣י אֱלֹהֵ֑ימוֹ צ֖וּר חָסָ֥יוּ בֽוֹ:  ת  וְיֵימַר אָן דַחֲלָתְהוֹן תַּקִיפָא דַהֲווֹ רְחִיצִין בֵּהּ:

{לח} אֲשֶׁ֨ר חֵ֤לֶב זְבָחֵ֨ימוֹ֙ יֹאכֵ֔לוּ יִשְׁתּ֖וּ יֵ֣ין נְסִיכָ֑ם יָק֨וּמוּ֙ וְיַעְזְרֻכֶ֔ם יְהִ֥י עֲלֵיכֶ֖ם סִתְרָֽה:  מ  אֲשֶׁ֨ר חֵ֤לֶב זְבָחֵ֨ימוֹ֙ יֹאכֵ֔לוּ יִשְׁתּ֖וּ יֵ֣ין נְסִיכָ֑ם יָק֨וּמוּ֙ וְיַעְזְרֻכֶ֔ם יְהִ֥י עֲלֵיכֶ֖ם סִתְרָֽה:  ת  דִי תְרַב נִכְסָתְהוֹן הֲווֹ אָכְלִין שָׁתָן חֲמַר נִסְכֵּיהוֹן יְקוּמוּן כְּעַן וִיסַעֲדוּנְכוֹן יְהוֹן עֲלֵיכוֹן מָגֵן:

{לט} רְא֣וּ | עַתָּ֗ה כִּ֣י אֲנִ֤י אֲנִי֙ ה֔וּא וְאֵ֥ין אֱלֹהִ֖ים עִמָּדִ֑י אֲנִ֧י אָמִ֣ית וַֽאֲחַיֶּ֗ה מָחַ֨צְתִּי֙ וַֽאֲנִ֣י אֶרְפָּ֔א וְאֵ֥ין מִיָּדִ֖י מַצִּֽיל:  מ  רְא֣וּ | עַתָּ֗ה כִּ֣י אֲנִ֤י אֲנִי֙ ה֔וּא וְאֵ֥ין אֱלֹהִ֖ים עִמָּדִ֑י אֲנִ֧י אָמִ֣ית וַֽאֲחַיֶּ֗ה מָחַ֨צְתִּי֙ וַֽאֲנִ֣י אֶרְפָּ֔א וְאֵ֥ין מִיָּדִ֖י מַצִּֽיל:  ת  חֲזוֹ כְּעַן אֲרֵי אֲנָא אֲנָא הוּא וְלֵית אֱלָהּ בַּר מִנִי אֲנָא מֵמִית וּמְחַיֵה מָחֵינָא וְאַף מַסֵינָא וְלֵית מִן יְדִי מְשֵׁזֵב:

{מ}  שישי  כִּֽי-אֶשָּׂ֥א אֶל-שָׁמַ֖יִם יָדִ֑י וְאָמַ֕רְתִּי חַ֥י אָֽנֹכִ֖י לְעֹלָֽם:  מ  כִּֽי-אֶשָּׂ֥א אֶל-שָׁמַ֖יִם יָדִ֑י וְאָמַ֕רְתִּי חַ֥י אָֽנֹכִ֖י לְעֹלָֽם:  ת  אֲרֵי אַתְקֵנִית בִּשְׁמַיָא בֵּית שְׁכִנְתִּי וַאֲמָרִית קַיָם אֲנָא לְעָלְמִין:

{מא} אִם-שַׁנּוֹתִי֙ בְּרַ֣ק חַרְבִּ֔י וְתֹאחֵ֥ז בְּמִשְׁפָּ֖ט יָדִ֑י אָשִׁ֤יב נָקָם֙ לְצָרָ֔י וְלִמְשַׂנְאַ֖י אֲשַׁלֵּֽם:  מ  אִם-שַׁנּוֹתִי֙ בְּרַ֣ק חַרְבִּ֔י וְתֹאחֵ֥ז בְּמִשְׁפָּ֖ט יָדִ֑י אָשִׁ֤יב נָקָם֙ לְצָרָ֔י וְלִמְשַׂנְאַ֖י אֲשַׁלֵּֽם:  ת  אִם עַל חַד תְּרֵין כְּחֵיזוּ בַרְקָא מִסוֹף שְׁמַיָא וְעַד סוֹף שְׁמַיָא תִּתְגְלִי חַרְבִּי וְתִתְקֵף בְּדִינָא יְדִי אָתֵב פֻּרְעֲנוּתָא לְסָנְאַי וּלְבַעֲלֵי דְבָבַי אֱשַׁלֵם:

{מב} אַשְׁכִּ֤יר חִצַּי֙ מִדָּ֔ם וְחַרְבִּ֖י תֹּאכַ֣ל בָּשָׂ֑ר מִדַּ֤ם חָלָל֙ וְשִׁבְיָ֔ה מֵרֹ֖אשׁ פַּרְע֥וֹת אוֹיֵֽב:  מ  אַשְׁכִּ֤יר חִצַּי֙ מִדָּ֔ם וְחַרְבִּ֖י תֹּאכַ֣ל בָּשָׂ֑ר מִדַּ֤ם חָלָל֙ וְשִׁבְיָ֔ה מֵרֹ֖אשׁ פַּרְע֥וֹת אוֹיֵֽב:  ת  אֱרַוֵי גִירַי מִדְמָא וְחַרְבִּי תִּקְטוֹל בְּעַמְמַיָא מִדַם קְטִילָן וְשִׁבְיָן לְאַעְדָאָה כִתְרִין מֵרֵישׁ סָנְאָה וּבְעַל דְבָבָא:

{מג} הַרְנִ֤ינוּ גוֹיִם֙ עַמּ֔וֹ כִּ֥י דַם-עֲבָדָ֖יו יִקּ֑וֹם וְנָקָם֙ יָשִׁ֣יב לְצָרָ֔יו וְכִפֶּ֥ר אַדְמָת֖וֹ עַמּֽוֹ:  מ  הַרְנִ֤ינוּ גוֹיִם֙ עַמּ֔וֹ כִּ֥י דַם-עֲבָדָ֖יו יִקּ֑וֹם וְנָקָם֙ יָשִׁ֣יב לְצָרָ֔יו וְכִפֶּ֥ר אַדְמָת֖וֹ עַמּֽוֹ:  ת  שַׁבָּחוּ עַמְמַיָא עַמֵהּ אֲרֵי פֻּרְעֲנוּת עַבְדוֹהִי צַדִיקַיָא יִתְפְּרַע וּפֻרְעֲנוּת יָתֵיב לְסָנְאוֹהִי וִיכַפֵּר עַל אַרְעֵהּ וְעַל עַמֵהּ: (פ)

{מד}  שביעי  וַיָּבֹ֣א מֹשֶׁ֗ה וַיְדַבֵּ֛ר אֶת-כָּל-דִּבְרֵ֥י הַשִּׁירָֽה-הַזֹּ֖את בְּאָזְנֵ֣י הָעָ֑ם ה֖וּא וְהוֹשֵׁ֥עַ בִּן-נֽוּן:  מ  וַיָּבֹ֣א מֹשֶׁ֗ה וַיְדַבֵּ֛ר אֶת-כָּל-דִּבְרֵ֥י הַשִּׁירָֽה-הַזֹּ֖את בְּאָזְנֵ֣י הָעָ֑ם ה֖וּא וְהוֹשֵׁ֥עַ בִּן-נֽוּן:  ת  וַאֲתָא מֹשֶׁה וּמַלִּיל יָת כָּל פִּתְגָמֵי תֻשְׁבַּחְתָּא הָדָא קֳדָם עַמָא הוּא וְהוֹשֵׁעַ בַּר נוּן:

{מה} וַיְכַ֣ל מֹשֶׁ֗ה לְדַבֵּ֛ר אֶת-כָּל-הַדְּבָרִ֥ים הָאֵ֖לֶּה אֶל-כָּל-יִשְׂרָאֵֽל:  מ  וַיְכַ֣ל מֹשֶׁ֗ה לְדַבֵּ֛ר אֶת-כָּל-הַדְּבָרִ֥ים הָאֵ֖לֶּה אֶל-כָּל-יִשְׂרָאֵֽל:  ת  וְשֵׁצִי מֹשֶׁה לְמַלָלָא יָת כָּל פִּתְגָמַיָא הָאִלֵין עִם כָּל יִשְׂרָאֵל:

{מו} וַיֹּ֤אמֶר אֲלֵהֶם֙ שִׂ֣ימוּ לְבַבְכֶ֔ם לְכָ֨ל-הַדְּבָרִ֔ים אֲשֶׁ֧ר אָֽנֹכִ֛י מֵעִ֥יד בָּכֶ֖ם הַיּ֑וֹם אֲשֶׁ֤ר תְּצַוֻּם֙ אֶת-בְּנֵיכֶ֔ם לִשְׁמֹ֣ר לַֽעֲשׂ֔וֹת אֶת-כָּל-דִּבְרֵ֖י הַתּוֹרָ֥ה הַזֹּֽאת:  מ  וַיֹּ֤אמֶר אֲלֵהֶם֙ שִׂ֣ימוּ לְבַבְכֶ֔ם לְכָ֨ל-הַדְּבָרִ֔ים אֲשֶׁ֧ר אָֽנֹכִ֛י מֵעִ֥יד בָּכֶ֖ם הַיּ֑וֹם אֲשֶׁ֤ר תְּצַוֻּם֙ אֶת-בְּנֵיכֶ֔ם לִשְׁמֹ֣ר לַֽעֲשׂ֔וֹת אֶת-כָּל-דִּבְרֵ֖י הַתּוֹרָ֥ה הַזֹּֽאת:  ת  וַאֲמַר לְהוֹן שַׁווּ לִבְּכוֹן לְכָל פִּתְגָמַיָא דִי אֲנָא מַסְהֵד בְּכוֹן יוֹמָא דֵין דִי תְפַקְדִנוּן יָת בְּנֵיכוֹן לְמִטַר לְמֶעְבַּד יָת כָּל פִּתְגָמֵי אוֹרַיְתָא הָדָא:

{מז} כִּ֠י לֹֽא-דָבָ֨ר רֵ֥ק הוּא֙ מִכֶּ֔ם כִּי-ה֖וּא חַיֵּיכֶ֑ם וּבַדָּבָ֣ר הַזֶּ֗ה תַּֽאֲרִ֤יכוּ יָמִים֙ עַל-הָ֣אֲדָמָ֔ה אֲשֶׁ֨ר אַתֶּ֜ם עֹֽבְרִ֧ים אֶת-הַיַּרְדֵּ֛ן שָׁ֖מָּה לְרִשְׁתָּֽהּ:  מ  כִּ֠י לֹֽא-דָבָ֨ר רֵ֥ק הוּא֙ מִכֶּ֔ם כִּי-ה֖וּא חַיֵּיכֶ֑ם וּבַדָּבָ֣ר הַזֶּ֗ה תַּֽאֲרִ֤יכוּ יָמִים֙ עַל-הָ֣אֲדָמָ֔ה אֲשֶׁ֨ר אַתֶּ֜ם עֹֽבְרִ֧ים אֶת-הַיַּרְדֵּ֛ן שָׁ֖מָּה לְרִשְׁתָּֽהּ:  ת  אֲרֵי לָא פִתְגָם רֵקָא הוּא מִנְכוֹן אֲרֵי הוּא חַיֵיכוֹן וּבְפִתְגָמָא הָדֵין תּוֹרְכוּן יוֹמִין עַל אַרְעָא דִי אַתּוּן עָבְרִין יָת יַרְדְנָא תַּמָן לְמֵירְתַהּ: (פ)

{מח}  מפטיר  וַיְדַבֵּ֤ר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה בְּעֶ֛צֶם הַיּ֥וֹם הַזֶּ֖ה לֵאמֹֽר:  מ  וַיְדַבֵּ֤ר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה בְּעֶ֛צֶם הַיּ֥וֹם הַזֶּ֖ה לֵאמֹֽר:  ת  וּמַלִיל יְיָ עִם מֹשֶׁה בִּכְרַן יוֹמָא הָדֵין לְמֵימָר:

{מט} עֲלֵ֡ה אֶל-הַר֩ הָֽעֲבָרִ֨ים הַזֶּ֜ה הַר-נְב֗וֹ אֲשֶׁר֙ בְּאֶ֣רֶץ מוֹאָ֔ב אֲשֶׁ֖ר עַל-פְּנֵ֣י יְרֵח֑וֹ וּרְאֵה֙ אֶת-אֶ֣רֶץ כְּנַ֔עַן אֲשֶׁ֨ר אֲנִ֥י נֹתֵ֛ן לִבְנֵ֥י יִשְׂרָאֵ֖ל לַֽאֲחֻזָּֽה:  מ  עֲלֵ֡ה אֶל-הַר֩ הָֽעֲבָרִ֨ים הַזֶּ֜ה הַר-נְב֗וֹ אֲשֶׁר֙ בְּאֶ֣רֶץ מוֹאָ֔ב אֲשֶׁ֖ר עַל-פְּנֵ֣י יְרֵח֑וֹ וּרְאֵה֙ אֶת-אֶ֣רֶץ כְּנַ֔עַן אֲשֶׁ֨ר אֲנִ֥י נֹתֵ֛ן לִבְנֵ֥י יִשְׂרָאֵ֖ל לַֽאֲחֻזָּֽה:  ת  סַק לְטוּרָא דַעֲבָרָאֵי הָדֵין טוּרָא דִנְבוֹ דִי בְּאַרְעָא דְמוֹאָב דִי עַל אַפֵּי יְרֵחוֹ וַחֲזֵי יָת אַרְעָא דִכְנַעַן דִי אֲנָא יָהֵב לִבְנֵי יִשְׂרָאֵל לְאַחֲסָנָא:

{נ} וּמֻ֗ת בָּהָר֙ אֲשֶׁ֤ר אַתָּה֙ עֹלֶ֣ה שָׁ֔מָּה וְהֵֽאָסֵ֖ף אֶל-עַמֶּ֑יךָ כַּֽאֲשֶׁר-מֵ֞ת אַֽהֲרֹ֤ן אָחִ֨יךָ֙ בְּהֹ֣ר הָהָ֔ר וַיֵּאָ֖סֶף אֶל-עַמָּֽיו:  מ  וּמֻ֗ת בָּהָר֙ אֲשֶׁ֤ר אַתָּה֙ עֹלֶ֣ה שָׁ֔מָּה וְהֵֽאָסֵ֖ף אֶל-עַמֶּ֑יךָ כַּֽאֲשֶׁר-מֵ֞ת אַֽהֲרֹ֤ן אָחִ֨יךָ֙ בְּהֹ֣ר הָהָ֔ר וַיֵּאָ֖סֶף אֶל-עַמָּֽיו:  ת  וּמוּת בְּטוּרָא דִי אַתְּ סָלֵק לְתַמָן וְתִתְכְּנֵשׁ לְעַמָךְ כְּמָא דְמִית אַהֲרֹן אֲחוּךְ בְּהוֹר טוּרָא וְאִתְכְּנִישׁ לְעַמֵהּ:

{נא} עַל֩ אֲשֶׁ֨ר מְעַלְתֶּ֜ם בִּ֗י בְּתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל בְּמֵֽי-מְרִיבַ֥ת קָדֵ֖שׁ מִדְבַּר-צִ֑ן עַ֣ל אֲשֶׁ֤ר לֹֽא-קִדַּשְׁתֶּם֙ אוֹתִ֔י בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל:  מ  עַל֩ אֲשֶׁ֨ר מְעַלְתֶּ֜ם בִּ֗י בְּתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל בְּמֵֽי-מְרִיבַ֥ת קָדֵ֖שׁ מִדְבַּר-צִ֑ן עַ֣ל אֲשֶׁ֤ר לֹֽא-קִדַּשְׁתֶּם֙ אוֹתִ֔י בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל:  ת  עַל דִי שַׁקַרְתּוּן בְּמֵימְרִי בְּגוֹ בְּנֵי יִשְׂרָאֵל בְּמֵי מַצוּת רְקָם מַדְבְּרָא דְצִין עַל דִי לָא קַדִשְׁתּוּן יָתִי בְּגוֹ בְּנֵי יִשְׂרָאֵל:

{נב} כִּ֥י מִנֶּ֖גֶד תִּרְאֶ֣ה אֶת-הָאָ֑רֶץ וְשָׁ֨מָּה֙ לֹ֣א תָב֔וֹא אֶל-הָאָ֕רֶץ אֲשֶׁר-אֲנִ֥י נֹתֵ֖ן לִבְנֵ֥י יִשְׂרָאֵֽל:  מ  כִּ֥י מִנֶּ֖גֶד תִּרְאֶ֣ה אֶת-הָאָ֑רֶץ וְשָׁ֨מָּה֙ לֹ֣א תָב֔וֹא אֶל-הָאָ֕רֶץ אֲשֶׁר-אֲנִ֥י נֹתֵ֖ן לִבְנֵ֥י יִשְׂרָאֵֽל:  ת  אֲרֵי מִקֳבֵל תֶּחֱזֵי יָת אַרְעָא וּלְתַמָן לָא תֵעוּל לְאַרְעָא דִי אֲנָא יָהֵב לִבְנֵי יִשְׂרָאֵל: (פפפ)
