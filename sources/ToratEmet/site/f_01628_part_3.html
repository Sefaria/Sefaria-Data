פרשת לך לך



בראשית פרק-יב

{א} וַיֹּאמֶר יְהֹוָה אֶל אַבְרָם לֶךְ לְךָ מֵֽאַרְצְךָ וּמִמּֽוֹלַדְתְּךָ וּמִבֵּית אָבִיךָ אֶל הָאָרֶץ אֲשֶׁר אַרְאֶֽךָּ:  מ  וַיֹּאמֶר יְהֹוָה אֶל אַבְרָם לֶךְ לְךָ מֵֽאַרְצְךָ וּמִמּֽוֹלַדְתְּךָ וּמִבֵּית אָבִיךָ אֶל הָאָרֶץ אֲשֶׁר אַרְאֶֽךָּ:  ת  וַאֲמַר יְיָ לְאַבְרָם אִיזֵל לָךְ מֵאַרְעָךְ וּמִיַּלָּדוּתָךְ וּמִבֵּית אֲבוּךְ לְאַרְעָא דִּי אַחֲזִנָּךְ:

{ב} וְאֶֽעֶשְׂךָ לְגוֹי גָּדוֹל וַֽאֲבָרֶכְךָ וַֽאֲגַדְּלָה שְׁמֶךָ וֶֽהְיֵה בְּרָכָֽה:  מ  וְאֶֽעֶשְׂךָ לְגוֹי גָּדוֹל וַֽאֲבָרֶכְךָ וַֽאֲגַדְּלָה שְׁמֶךָ וֶֽהְיֵה בְּרָכָֽה:  ת  וְאֶעְבְּדִנָּךְ לְעַם סַגִּי וֶאֱבָרֵכִנָּךְ וַאֲרַבֵּי שְׁמָךְ וּתְהֵא מְבָרֵךְ:

{ג} וַֽאֲבָֽרְכָה מְבָרֲכֶיךָ וּמְקַלֶּלְךָ אָאֹר וְנִבְרְכוּ בְךָ כֹּל מִשְׁפְּחֹת הָֽאֲדָמָֽה:  מ  וַֽאֲבָֽרְכָה מְבָרֲכֶיךָ וּמְקַלֶּלְךָ אָאֹר וְנִבְרְכוּ בְךָ כֹּל מִשְׁפְּחֹת הָֽאֲדָמָֽה:  ת  וֶאֱבָרֵךְ מְבָרְכָךְ וּמְלַטְטָךְ אֵילוּט וְיִתְבָּרֲכוּן בְּדִילָךְ כֹּל זַרְעֲיַת אַרְעָא:

{ד} וַיֵּלֶךְ אַבְרָם כַּֽאֲשֶׁר דִּבֶּר אֵלָיו יְהֹוָה וַיֵּלֶךְ אִתּוֹ לוֹט וְאַבְרָם בֶּן חָמֵשׁ שָׁנִים וְשִׁבְעִים שָׁנָה בְּצֵאתוֹ מֵֽחָרָֽן:  מ  וַיֵּלֶךְ אַבְרָם כַּֽאֲשֶׁר דִּבֶּר אֵלָיו יְהֹוָה וַיֵּלֶךְ אִתּוֹ לוֹט וְאַבְרָם בֶּן חָמֵשׁ שָׁנִים וְשִׁבְעִים שָׁנָה בְּצֵאתוֹ מֵֽחָרָֽן:  ת  וַאֲזַל אַבְרָם כְּמָא דְּמַלֵּיל עִמֵּהּ יְיָ וַאֲזַל עִמֵּהּ לוֹטּ וְאַבְרָם בַּר שִׁבְעִין וַחֲמֵשׁ שְׁנִין בְּמִפְקֵהּ מֵחָרָן:

{ה} וַיִּקַּח אַבְרָם אֶת שָׂרַי אִשְׁתּוֹ וְאֶת לוֹט בֶּן אָחִיו וְאֶת כָּל רְכוּשָׁם אֲשֶׁר רָכָשׁוּ וְאֶת הַנֶּפֶשׁ אֲשֶׁר עָשׂוּ בְחָרָן וַיֵּֽצְאוּ לָלֶכֶת אַרְצָה כְּנַעַן וַיָּבֹאוּ אַרְצָה כְּנָֽעַן:  מ  וַיִּקַּח אַבְרָם אֶת שָׂרַי אִשְׁתּוֹ וְאֶת לוֹט בֶּן אָחִיו וְאֶת כָּל רְכוּשָׁם אֲשֶׁר רָכָשׁוּ וְאֶת הַנֶּפֶשׁ אֲשֶׁר עָשׂוּ בְחָרָן וַיֵּֽצְאוּ לָלֶכֶת אַרְצָה כְּנַעַן וַיָּבֹאוּ אַרְצָה כְּנָֽעַן:  ת  וּדְבַר אַבְרָם יָת שָׂרַי אִתְּתֵהּ וְיָת לוֹט בַר אֲחוּהִי וְיָת כָּל קִנְיָנְהוֹן דִּי קְנוֹ וְיָת נַפְשָׁתָא דְּשַׁעֲבִּידוּ לְאוֹרַיְתָא בְחָרָן וּנְפָקוּ לְמֵיזַל לְאַרְעָא דִּכְנַעַן וַאֲתוֹ לְאַרְעָא דִּכְנָעַן:

{ו} וַיַּעֲבֹר אַבְרָם בָּאָרֶץ עַד מְקוֹם שְׁכֶם עַד אֵלוֹן מוֹרֶה וְהַֽכְּנַֽעֲנִי אָז בָּאָֽרֶץ:  מ  וַיַּעֲבֹר אַבְרָם בָּאָרֶץ עַד מְקוֹם שְׁכֶם עַד אֵלוֹן מוֹרֶה וְהַֽכְּנַֽעֲנִי אָז בָּאָֽרֶץ:  ת  וַעֲבַר אַבְרָם בְּאַרְעָא עַד אֲתַר שְׁכֶם עַד מֵישַׁר מוֹרֶה וּכְנַעֲנָאָה בְּכֵן בְּאַרְעָא:

{ז} וַיֵּרָא יְהֹוָה אֶל אַבְרָם וַיֹּאמֶר לְזַרְעֲךָ אֶתֵּן אֶת הָאָרֶץ הַזֹּאת וַיִּבֶן שָׁם מִזְבֵּחַ לַֽיהֹוָה הַנִּרְאֶה אֵלָֽיו:  מ  וַיֵּרָא יְהֹוָה אֶל אַבְרָם וַיֹּאמֶר לְזַרְעֲךָ אֶתֵּן אֶת הָאָרֶץ הַזֹּאת וַיִּבֶן שָׁם מִזְבֵּחַ לַֽיהֹוָה הַנִּרְאֶה אֵלָֽיו:  ת  וְאִתְגְּלִי יְיָ לְאַבְרָם וַאֲמַר לִבְנָךְ אֶתֵּן יָת אַרְעָא הָדָא וּבְנָא תַמָּן מַדְבְּחָא קֳדָם יְיָ דְּאִתְגְּלִי לֵהּ:

{ח} וַיַּעְתֵּק מִשָּׁם הָהָרָה מִקֶּדֶם לְבֵית אֵל וַיֵּט אָֽהֳלֹה בֵּית אֵל מִיָּם וְהָעַי מִקֶּדֶם וַיִּֽבֶן שָׁם מִזְבֵּחַ לַֽיהֹוָה וַיִּקְרָא בְּשֵׁם יְהֹוָֽה:  מ  וַיַּעְתֵּק מִשָּׁם הָהָרָה מִקֶּדֶם לְבֵית אֵל וַיֵּט אָֽהֳלֹה בֵּית אֵל מִיָּם וְהָעַי מִקֶּדֶם וַיִּֽבֶן שָׁם מִזְבֵּחַ לַֽיהֹוָה וַיִּקְרָא בְּשֵׁם יְהֹוָֽה:  ת  וְאִסְתַּלַּק מִתַּמָּן לְטוּרָא מִמַדְנַח לְבֵית אֵל וּפְרַס מַשְׁכְּנֵהּ בֵּית אֵל מִמַּעַרְבָא וְעַי מִמָּדִינְחָא וּבְנָא תַמָּן מִדְבְּחָא קֳדָם יְיָ וְצַלִּי בִּשְׁמָא דַּיְיָ:

{ט} וַיִּסַּע אַבְרָם הָלוֹךְ וְנָסוֹעַ הַנֶּֽגְבָּה:  מ  וַיִּסַּע אַבְרָם הָלוֹךְ וְנָסוֹעַ הַנֶּֽגְבָּה:  ת  וּנְטַּל אַבְרָם אָזֵל וְנָטֵל לְדָרוֹמָא: (פ)

{י} וַיְהִי רָעָב בָּאָרֶץ וַיֵּרֶד אַבְרָם מִצְרַיְמָה לָגוּר שָׁם כִּֽי כָבֵד הָֽרָעָב בָּאָֽרֶץ:  מ  וַיְהִי רָעָב בָּאָרֶץ וַיֵּרֶד אַבְרָם מִצְרַיְמָה לָגוּר שָׁם כִּֽי כָבֵד הָֽרָעָב בָּאָֽרֶץ:  ת  וַהֲוָה כַפְנָא בְּאַרְעָא וּנְחַת אַבְרָם לְמִצְרַיִם לְאִתּוֹתָבָא תַמָּן אֲרֵי תַקִּיף כַּפְנָא בְּאַרְעָא:

{יא} וַיְהִי כַּֽאֲשֶׁר הִקְרִיב לָבוֹא מִצְרָיְמָה וַיֹּאמֶר אֶל שָׂרַי אִשְׁתּוֹ הִנֵּה נָא יָדַעְתִּי כִּי אִשָּׁה יְפַת מַרְאֶה אָֽתְּ:  מ  וַיְהִי כַּֽאֲשֶׁר הִקְרִיב לָבוֹא מִצְרָיְמָה וַיֹּאמֶר אֶל שָׂרַי אִשְׁתּוֹ הִנֵּה נָא יָדַעְתִּי כִּי אִשָּׁה יְפַת מַרְאֶה אָֽתְּ:  ת  וַהֲוָה כַּד קְרֵיב לְמֵיעַל לְמִצְרָיִם וַאֲמַר לְשָׂרַי אִתְּתֵה הָא כְעַן יְדָעִית אֲרֵי אִתְּתָא שַׁפִּירַת חֵיזוּ אָתְּ:

{יב} וְהָיָה כִּֽי יִרְאוּ אֹתָךְ הַמִּצְרִים וְאָֽמְרוּ אִשְׁתּוֹ זֹאת וְהָֽרְגוּ אֹתִי וְאֹתָךְ יְחַיּֽוּ:  מ  וְהָיָה כִּֽי יִרְאוּ אֹתָךְ הַמִּצְרִים וְאָֽמְרוּ אִשְׁתּוֹ זֹאת וְהָֽרְגוּ אֹתִי וְאֹתָךְ יְחַיּֽוּ:  ת  וִיהֵי כַּד (נ''י אֲרֵי) יֶחֱזוּן יָתִיךְ מִצְרָאֵי וְיֵימְרוּן אִתְּתֵהּ דָּא וְיִקְטְלוּן יָתִי וְיָתִיךְ יְקַיְמוּן:

{יג} אִמְרִי נָא אֲחֹתִי אָתְּ לְמַעַן יִֽיטַב לִי בַֽעֲבוּרֵךְ וְחָֽיְתָה נַפְשִׁי בִּגְלָלֵֽךְ:  מ  אִמְרִי נָא אֲחֹתִי אָתְּ לְמַעַן יִֽיטַב לִי בַֽעֲבוּרֵךְ וְחָֽיְתָה נַפְשִׁי בִּגְלָלֵֽךְ:  ת  אִמְרִי כְעַן אֲחָתִי אָתְּ בְּדִיל דְּיֵיטַב לִי בְּדִילָךְ וְתִתְקַיֵּם נַפְשִׁי בְּפִתְגָּמָיְכִי:

{יד}  שני  וַיְהִי כְּבוֹא אַבְרָם מִצְרָיְמָה וַיִּרְאוּ הַמִּצְרִים אֶת הָאִשָּׁה כִּֽי יָפָה הִוא מְאֹֽד:  מ  וַיְהִי כְּבוֹא אַבְרָם מִצְרָיְמָה וַיִּרְאוּ הַמִּצְרִים אֶת הָאִשָּׁה כִּֽי יָפָה הִוא מְאֹֽד:  ת  וַהֲוָה כַּד עַל אַבְרָם לְמִצְרָיִם וַחֲזוֹ מִצְרָאֵי יָת אִתְּתָא אֲרֵי שַׁפִּירָא הִיא לַחֲדָא:

{טו} וַיִּרְאוּ אֹתָהּ שָׂרֵי פַרְעֹה וַיְהַֽלֲלוּ אֹתָהּ אֶל פַּרְעֹה וַתֻּקַּח הָֽאִשָּׁה בֵּית פַּרְעֹֽה:  מ  וַיִּרְאוּ אֹתָהּ שָׂרֵי פַרְעֹה וַיְהַֽלֲלוּ אֹתָהּ אֶל פַּרְעֹה וַתֻּקַּח הָֽאִשָּׁה בֵּית פַּרְעֹֽה:  ת  וַחֲזוֹ יָתַהּ רַבְרְבֵי פַרְעֹה וְשַׁבָּחוּ יָתַהּ לְפַרְעֹה וְאִדַּבָּרַת אִתְּתָא לְבֵית פַּרְעֹה:

{טז} וּלְאַבְרָם הֵיטִיב בַּֽעֲבוּרָהּ וַֽיְהִי לוֹ צֹאן וּבָקָר וַֽחֲמֹרִים וַֽעֲבָדִים וּשְׁפָחֹת וַֽאֲתֹנֹת וּגְמַלִּֽים:  מ  וּלְאַבְרָם הֵיטִיב בַּֽעֲבוּרָהּ וַֽיְהִי לוֹ צֹאן וּבָקָר וַֽחֲמֹרִים וַֽעֲבָדִים וּשְׁפָחֹת וַֽאֲתֹנֹת וּגְמַלִּֽים:  ת  וּלְאַבְרָם אוֹטִיב בְּדִילַהּ וַהֲווֹ לֵהּ עָאן וְתוֹרִין וַחֲמָרִין וְעַבְדִין וְאַמְהָן וְאַתְנָן וְגַמְלִין:

{יז} וַיְנַגַּע יְהֹוָה אֶת פַּרְעֹה נְגָעִים גְּדֹלִים וְאֶת בֵּיתוֹ עַל דְּבַר שָׂרַי אֵשֶׁת אַבְרָֽם:  מ  וַיְנַגַּע יְהֹוָה אֶת פַּרְעֹה נְגָעִים גְּדֹלִים וְאֶת בֵּיתוֹ עַל דְּבַר שָׂרַי אֵשֶׁת אַבְרָֽם:  ת  וְאַיְתִי יְיָ עַל פַּרְעֹה מַכְתָּשִׁין רַבְרְבִין וְעַל אֱנַשׁ בֵּיתֵהּ עַל עֵיסַק שָׂרַי אִתַּת אַבְרָם:

{יח} וַיִּקְרָא פַרְעֹה לְאַבְרָם וַיֹּאמֶר מַה זֹּאת עָשִׂיתָ לִּי לָמָּה לֹֽא הִגַּדְתָּ לִּי כִּי אִשְׁתְּךָ הִֽוא:  מ  וַיִּקְרָא פַרְעֹה לְאַבְרָם וַיֹּאמֶר מַה זֹּאת עָשִׂיתָ לִּי לָמָּה לֹֽא הִגַּדְתָּ לִּי כִּי אִשְׁתְּךָ הִֽוא:  ת  וּקְרָא פַרְעֹה לְאַבְרָם וַאֲמַר מָה דָּא עֲבַדְתְּ לִי לְמָא לָא חַוֵּיתָא לִי אֲרֵי אִתְּתָךְ הִיא:

{יט} לָמָה אָמַרְתָּ אֲחֹתִי הִוא וָֽאֶקַּח אֹתָהּ לִי לְאִשָּׁה וְעַתָּה הִנֵּה אִשְׁתְּךָ קַח וָלֵֽךְ:  מ  לָמָה אָמַרְתָּ אֲחֹתִי הִוא וָֽאֶקַּח אֹתָהּ לִי לְאִשָּׁה וְעַתָּה הִנֵּה אִשְׁתְּךָ קַח וָלֵֽךְ:  ת  לְמָא אֲמַרְתְּ אֲחָתִי הִיא וּדְבָרִית יָתַהּ לִי לְאִתּוּ וּכְעַן הָא אִתְּתָךְ דְּבַר וְאִיזֵל:

{כ} וַיְצַו עָלָיו פַּרְעֹה אֲנָשִׁים וַיְשַׁלְּחוּ אֹתוֹ וְאֶת אִשְׁתּוֹ וְאֶת כָּל אֲשֶׁר לֽוֹ:  מ  וַיְצַו עָלָיו פַּרְעֹה אֲנָשִׁים וַיְשַׁלְּחוּ אֹתוֹ וְאֶת אִשְׁתּוֹ וְאֶת כָּל אֲשֶׁר לֽוֹ:  ת  וּפַקִּיד עֲלוֹהִי פַּרְעֹה גּוּבְרִין וְאַלְוִיאוּ יָתֵהּ וְיָת אִתְּתֵהּ וְיָת כָּל דִילֵהּ:




בראשית פרק-יג

{א} וַיַּעַל אַבְרָם מִמִּצְרַיִם הוּא וְאִשְׁתּוֹ וְכָל אֲשֶׁר לוֹ וְלוֹט עִמּוֹ הַנֶּֽגְבָּה:  מ  וַיַּעַל אַבְרָם מִמִּצְרַיִם הוּא וְאִשְׁתּוֹ וְכָל אֲשֶׁר לוֹ וְלוֹט עִמּוֹ הַנֶּֽגְבָּה:  ת  וּסְלֵיק אַבְרָם מִמִּצְרַיִם הוּא וְאִתְּתֵהּ וְכָל דִּילֵהּ וְלוֹט עִמֵּהּ לְדָרוֹמָא:

{ב} וְאַבְרָם כָּבֵד מְאֹד בַּמִּקְנֶה בַּכֶּסֶף וּבַזָּהָֽב:  מ  וְאַבְרָם כָּבֵד מְאֹד בַּמִּקְנֶה בַּכֶּסֶף וּבַזָּהָֽב:  ת  וְאַבְרָם תַּקִּיף לַחֲדָא בִּבְעִירָא בְּכַסְפָּא וּבְדַהֲבָא:

{ג} וַיֵּלֶךְ לְמַסָּעָיו מִנֶּגֶב וְעַד בֵּית אֵל עַד הַמָּקוֹם אֲשֶׁר הָיָה שָׁם אָֽהֳלֹה בַּתְּחִלָּה בֵּין בֵּית אֵל וּבֵין הָעָֽי:  מ  וַיֵּלֶךְ לְמַסָּעָיו מִנֶּגֶב וְעַד בֵּית אֵל עַד הַמָּקוֹם אֲשֶׁר הָיָה שָׁם אָֽהֳלֹה בַּתְּחִלָּה בֵּין בֵּית אֵל וּבֵין הָעָֽי:  ת  וַאֲזַל לְמַטְלָנוֹהִי מִדָּרוֹמָא וְעַד בֵּית אֵל עַד אַתְרָא דִּי פְּרַס תַּמָּן מַשְׁכְּנֵהּ בְּקַדְמֵיתָא בֵּין בֵּית אֵל וּבֵין עָי:

{ד} אֶל מְקוֹם הַמִּזְבֵּחַ אֲשֶׁר עָשָׂה שָׁם בָּרִֽאשֹׁנָה וַיִּקְרָא שָׁם אַבְרָם בְּשֵׁם יְהֹוָֽה:  מ  אֶל מְקוֹם הַמִּזְבֵּחַ אֲשֶׁר עָשָׂה שָׁם בָּרִֽאשֹׁנָה וַיִּקְרָא שָׁם אַבְרָם בְּשֵׁם יְהֹוָֽה:  ת  לַאֲתַר מַדְבְּחָא דִּי עֲבַד תַּמָּן בְּקַדְמֵיתָא וְצַלֵּי תַמָּן אַבְרָם בִּשְׁמָא דַּיְיָ:

{ה}  שלישי  וְגַם לְלוֹט הַֽהֹלֵךְ אֶת אַבְרָם הָיָה צֹאן וּבָקָר וְאֹֽהָלִֽים:  מ  וְגַם לְלוֹט הַֽהֹלֵךְ אֶת אַבְרָם הָיָה צֹאן וּבָקָר וְאֹֽהָלִֽים:  ת  וְאַף לְלוֹט דְּאָזִיל עִם אַבְרָם הֲוָה עָאן וְתוֹרִין וּמַשְׁכְּנִין:

{ו} וְלֹֽא נָשָׂא אֹתָם הָאָרֶץ לָשֶׁבֶת יַחְדָּו כִּֽי הָיָה רְכוּשָׁם רָב וְלֹא יָֽכְלוּ לָשֶׁבֶת יַחְדָּֽו:  מ  וְלֹֽא נָשָׂא אֹתָם הָאָרֶץ לָשֶׁבֶת יַחְדָּו כִּֽי הָיָה רְכוּשָׁם רָב וְלֹא יָֽכְלוּ לָשֶׁבֶת יַחְדָּֽו:  ת  וְלָא סוֹבָרַת יָתְהוֹן אַרְעָא לְמִיתַב כַּחֲדָא אֲרֵי הֲוָה קִנְיָנְהוֹן סַגִּי וְלָא יְכִילוּ לְמִיתַּב כַּחֲדָא:

{ז} וַֽיְהִי רִיב בֵּין רֹעֵי מִקְנֵֽה אַבְרָם וּבֵין רֹעֵי מִקְנֵה לוֹט וְהַֽכְּנַֽעֲנִי וְהַפְּרִזִּי אָז ישֵׁב בָּאָֽרֶץ:  מ  וַֽיְהִי רִיב בֵּין רֹעֵי מִקְנֵֽה אַבְרָם וּבֵין רֹעֵי מִקְנֵה לוֹט וְהַֽכְּנַֽעֲנִי וְהַפְּרִזִּי אָז ישֵׁב בָּאָֽרֶץ:  ת  וַהַוַת מַצוּתָא בֵּין רָעֵי בְּעִירֵה דְאַבְרָם וּבֵין רָעֵי בְּעִירֵה דְלוֹט וּכְנַעֲנָאָה וּפְרִזָאָה בְּכֵן יָתִיב בְּאַרְעָא:

{ח} וַיֹּאמֶר אַבְרָם אֶל לוֹט אַל נָא תְהִי מְרִיבָה בֵּינִי וּבֵינֶךָ וּבֵין רֹעַי וּבֵין רֹעֶיךָ כִּֽי אֲנָשִׁים אַחִים אֲנָֽחְנוּ:  מ  וַיֹּאמֶר אַבְרָם אֶל לוֹט אַל נָא תְהִי מְרִיבָה בֵּינִי וּבֵינֶךָ וּבֵין רֹעַי וּבֵין רֹעֶיךָ כִּֽי אֲנָשִׁים אַחִים אֲנָֽחְנוּ:  ת  וַאֲמַר אַבְרָם לְלוֹט לָא כְעַן תְּהֵי מַצוּתָא בֵּינָא וּבֵינָךְ וּבֵין רַעֲוָתִי וּבֵין רַעֲוָתָךְ אֲרֵי גוּבְרִין אַחִין אֲנָחְנָא:

{ט} הֲלֹא כָל הָאָרֶץ לְפָנֶיךָ הִפָּרֶד נָא מֵֽעָלָי אִם הַשְּׂמֹאל וְאֵימִנָה וְאִם הַיָּמִין וְאַשְׂמְאִֽילָה:  מ  הֲלֹא כָל הָאָרֶץ לְפָנֶיךָ הִפָּרֶד נָא מֵֽעָלָי אִם הַשְּׂמֹאל וְאֵימִנָה וְאִם הַיָּמִין וְאַשְׂמְאִֽילָה:  ת  הֲלָא כָל אַרְעָא קֳדָמָךְ אִתְפָּרַשׁ כְּעַן מִלְוָתִי אִם אַתְּ לְצִפּוּנָא וַאֲנָא לְדָרוֹמָא וְאִם אַתְּ לְדָרוֹמָא וַאֲנָא לְצִפּוּנָא:

{י} וַיִּשָּׂא לוֹט אֶת עֵינָיו וַיַּרְא אֶת כָּל כִּכַּר הַיַּרְדֵּן כִּי כֻלָּהּ מַשְׁקֶה לִפְנֵי שַׁחֵת יְהֹוָה אֶת סְדֹם וְאֶת עֲמֹרָה כְּגַן יְהֹוָה כְּאֶרֶץ מִצְרַיִם בֹּֽאֲכָה צֹֽעַר:  מ  וַיִּשָּׂא לוֹט אֶת עֵינָיו וַיַּרְא אֶת כָּל כִּכַּר הַיַּרְדֵּן כִּי כֻלָּהּ מַשְׁקֶה לִפְנֵי שַׁחֵת יְהֹוָה אֶת סְדֹם וְאֶת עֲמֹרָה כְּגַן יְהֹוָה כְּאֶרֶץ מִצְרַיִם בֹּֽאֲכָה צֹֽעַר:  ת  וּזְקַף לוֹט יָת עֵינוֹהִי וַחֲזָא יָת כָּל מֵישַׁר יַרְדְנָא אֲרֵי כֻלַהּ בֵּית שַׁקְיָא קֳדָם חַבָּלוּת יְיָ יָת סְדוֹם וְיָת עֲמֹרָא כְּגִנְתָא דַיְיָ כְּאַרְעָא דְמִצְרַיִם מָטֵּי לְצוֹעַר:

{יא} וַיִּבְחַר לוֹ לוֹט אֵת כָּל כִּכַּר הַיַּרְדֵּן וַיִּסַּע לוֹט מִקֶּדֶם וַיִּפָּרְדוּ אִישׁ מֵעַל אָחִֽיו:  מ  וַיִּבְחַר לוֹ לוֹט אֵת כָּל כִּכַּר הַיַּרְדֵּן וַיִּסַּע לוֹט מִקֶּדֶם וַיִּפָּרְדוּ אִישׁ מֵעַל אָחִֽיו:  ת  וּבְחַר לֵיהּ לוֹט יַת כָּל מֵישַׁר יַרְדְנָא וּנְטַל לוֹט מִלְקַדְמִין וְאִתַּפְרָשׁוּ גְבַר מִלְוַת אָחוּהִי:

{יב} אַבְרָם יָשַׁב בְּאֶֽרֶץ כְּנָעַן וְלוֹט יָשַׁב בְּעָרֵי הַכִּכָּר וַיֶּֽאֱהַל עַד סְדֹֽם:  מ  אַבְרָם יָשַׁב בְּאֶֽרֶץ כְּנָעַן וְלוֹט יָשַׁב בְּעָרֵי הַכִּכָּר וַיֶּֽאֱהַל עַד סְדֹֽם:  ת  אַבְרָם יָתֵיב בְּאַרְעָא דִכְנָעַן וְלוֹט יָתֵיב בְּקִרְוֵי מֵישְׁרָא וּפְרַס עַד סְדוֹם:

{יג} וְאַנְשֵׁי סְדֹם רָעִים וְחַטָּאִים לַֽיהֹוָה מְאֹֽד:  מ  וְאַנְשֵׁי סְדֹם רָעִים וְחַטָּאִים לַֽיהֹוָה מְאֹֽד:  ת  וֶאֱנָשִׁין דִסְדוֹם בִּישִׁין בְּמָמוֹנְהוֹן וְחַיָבִין בִּגְוִיָתְהוֹן קֳדָם יְיָ לַחֲדָא:

{יד} וַֽיהֹוָה אָמַר אֶל אַבְרָם אַֽחֲרֵי הִפָּֽרֶד לוֹט מֵֽעִמּוֹ שָׂא נָא עֵינֶיךָ וּרְאֵה מִן הַמָּקוֹם אֲשֶׁר אַתָּה שָׁם צָפֹנָה וָנֶגְבָּה וָקֵדְמָה וָיָֽמָּה:  מ  וַֽיהֹוָה אָמַר אֶל אַבְרָם אַֽחֲרֵי הִפָּֽרֶד לוֹט מֵֽעִמּוֹ שָׂא נָא עֵינֶיךָ וּרְאֵה מִן הַמָּקוֹם אֲשֶׁר אַתָּה שָׁם צָפֹנָה וָנֶגְבָּה וָקֵדְמָה וָיָֽמָּה:  ת  וַיְיָ אֲמַר לְאַבְרָם בָּתַר דְאִתְפָּרֶשׁ לוֹט מֵעִמֵהּ זְקוֹף כְּעַן עֵינָךְ וַחֲזֵי מִן אַתְרָא דְאַתְּ תַּמָן לְצִפּוּנָא וּלְדָרוֹמָא וּלְמַדִינְחָא וּלְמַעֲרָבָא:

{טו} כִּי אֶת כָּל הָאָרֶץ אֲשֶׁר אַתָּה רֹאֶה לְךָ אֶתְּנֶנָּה וּלְזַרְעֲךָ עַד עוֹלָֽם:  מ  כִּי אֶת כָּל הָאָרֶץ אֲשֶׁר אַתָּה רֹאֶה לְךָ אֶתְּנֶנָּה וּלְזַרְעֲךָ עַד עוֹלָֽם:  ת  אֲרֵי יָת כָּל אַרְעָא דִי אַתְּ חָזֵי לָךְ אֶתְּנִנַהּ וְלִבְנָיךְ עַד עָלָם:

{טז} וְשַׂמְתִּי אֶת זַרְעֲךָ כַּֽעֲפַר הָאָרֶץ אֲשֶׁר אִם יוּכַל אִישׁ לִמְנוֹת אֶת עֲפַר הָאָרֶץ גַּם זַרְעֲךָ יִמָּנֶֽה:  מ  וְשַׂמְתִּי אֶת זַרְעֲךָ כַּֽעֲפַר הָאָרֶץ אֲשֶׁר אִם יוּכַל אִישׁ לִמְנוֹת אֶת עֲפַר הָאָרֶץ גַּם זַרְעֲךָ יִמָּנֶֽה:  ת  וֶאֱשַׁוֵּי יָת בְּנָיךְ סַגִיאִין כְּעַפְרָא דְאַרְעָא כְּמָא דִי לָא אֶפְשַׁר לִגְבַר לְּמִמְנֵי יָת עַפְרָא דְאַרְעָא אַף בְּנָיךְ לָא יִתִּמְנוּן:

{יז} קוּם הִתְהַלֵּךְ בָּאָרֶץ לְאָרְכָּהּ וּלְרָחְבָּהּ כִּי לְךָ אֶתְּנֶֽנָּה:  מ  קוּם הִתְהַלֵּךְ בָּאָרֶץ לְאָרְכָּהּ וּלְרָחְבָּהּ כִּי לְךָ אֶתְּנֶֽנָּה:  ת  קוּם הֲלִיךְ בְּאַרְעָא לְאָרְכַּהּ וְלִפְתָיַהּ אֲרֵי לָךְ אֶתְּנִינַהּ:

{יח} וַיֶּֽאֱהַל אַבְרָם וַיָּבֹא וַיֵּשֶׁב בְּאֵֽלֹנֵי מַמְרֵא אֲשֶׁר בְּחֶבְרוֹן וַיִּֽבֶן שָׁם מִזְבֵּחַ לַֽיהֹוָֽה:  מ  וַיֶּֽאֱהַל אַבְרָם וַיָּבֹא וַיֵּשֶׁב בְּאֵֽלֹנֵי מַמְרֵא אֲשֶׁר בְּחֶבְרוֹן וַיִּֽבֶן שָׁם מִזְבֵּחַ לַֽיהֹוָֽה:  ת  וּפְרַס אַבְרָם וַאֲתָא וִיתֵב בְּמֵישְׁרֵי מַמְרֵא דִי בְחֶבְרוֹן וּבְנָא תַמָן מַדְבְּחָא קֳדָם יְיָ: (פ)




בראשית פרק-יד

{א}  רביעי  וַיְהִי בִּימֵי אַמְרָפֶל מֶֽלֶךְ שִׁנְעָר אַרְיוֹךְ מֶלֶךְ אֶלָּסָר כְּדָרְלָעֹמֶר מֶלֶךְ עֵילָם וְתִדְעָל מֶלֶךְ גּוֹיִֽם:  מ  וַיְהִי בִּימֵי אַמְרָפֶל מֶֽלֶךְ שִׁנְעָר אַרְיוֹךְ מֶלֶךְ אֶלָּסָר כְּדָרְלָעֹמֶר מֶלֶךְ עֵילָם וְתִדְעָל מֶלֶךְ גּוֹיִֽם:  ת  וַהֲוָה בְיוֹמֵי אַמְרָפֶל מַלְכָּא דְבָבֶל אַרְיוֹךְ מַלְכָּא דְאֶלָּסָר כְּדָרְלָעוֹמֶר מַלְכָּא דְעֵילָם וְתִדְעָל מַלְכָּא דְעַמְמֵי:

{ב} עָשׂוּ מִלְחָמָה אֶת בֶּרַע מֶלֶךְ סְדֹם וְאֶת בִּרְשַׁע מֶלֶךְ עֲמֹרָה שִׁנְאָב מֶלֶךְ אַדְמָה וְשֶׁמְאֵבֶר מֶלֶךְ (צביים) צְבוֹיִים וּמֶלֶךְ בֶּלַע הִיא צֹֽעַר:  מ  עָשׂוּ מִלְחָמָה אֶת בֶּרַע מֶלֶךְ סְדֹם וְאֶת בִּרְשַׁע מֶלֶךְ עֲמֹרָה שִׁנְאָב מֶלֶךְ אַדְמָה וְשֶׁמְאֵבֶר מֶלֶךְ (צביים) צְבוֹיִים וּמֶלֶךְ בֶּלַע הִיא צֹֽעַר:  ת  סְדָרוּ (נ''א עֲבָדוּ) קְרָבָא עִם בֶּרַע מַלְכָּא דִסְדוֹם וְעִם בִּרְשַׁע מַלְכָּא דַעֲמֹרָה שִׁנְאָב מַלְכָּא דְאַדְמָה וְשֶׁמְאֵבֶר מַלְכָּא דִצְבוֹיִם וּמַלְכָּא דְבֶלַּע הִיא צוֹעַר:

{ג} כָּל אֵלֶּה חָֽבְרוּ אֶל עֵמֶק הַשִּׂדִּים הוּא יָם הַמֶּֽלַח:  מ  כָּל אֵלֶּה חָֽבְרוּ אֶל עֵמֶק הַשִּׂדִּים הוּא יָם הַמֶּֽלַח:  ת  כָּל אִלֵין אִתְכְּנָשׁוּ לְמֵישַׁר חַקְלַיָא הוּא אֲתַר יַמָא דְמִלְחָא:

{ד} שְׁתֵּים עֶשְׂרֵה שָׁנָה עָֽבְדוּ אֶת כְּדָרְלָעֹמֶר וּשְׁלשׁ עֶשְׂרֵה שָׁנָה מָרָֽדוּ:  מ  שְׁתֵּים עֶשְׂרֵה שָׁנָה עָֽבְדוּ אֶת כְּדָרְלָעֹמֶר וּשְׁלשׁ עֶשְׂרֵה שָׁנָה מָרָֽדוּ:  ת  תַּרְתֵּי עַשְׂרֵי שְׁנִין פְּלָחוּ יָת כְּדָרְלָעוֹמֶר וּתְלָת עֶשְׂרֵי שְׁנִין מְרָדוּ:

{ה} וּבְאַרְבַּע עֶשְׂרֵה שָׁנָה בָּא כְדָרְלָעֹמֶר וְהַמְּלָכִים אֲשֶׁר אִתּוֹ וַיַּכּוּ אֶת רְפָאִים בְּעַשְׁתְּרֹת קַרְנַיִם וְאֶת הַזּוּזִים בְּהָם וְאֵת הָֽאֵימִים בְּשָׁוֵה קִרְיָתָֽיִם:  מ  וּבְאַרְבַּע עֶשְׂרֵה שָׁנָה בָּא כְדָרְלָעֹמֶר וְהַמְּלָכִים אֲשֶׁר אִתּוֹ וַיַּכּוּ אֶת רְפָאִים בְּעַשְׁתְּרֹת קַרְנַיִם וְאֶת הַזּוּזִים בְּהָם וְאֵת הָֽאֵימִים בְּשָׁוֵה קִרְיָתָֽיִם:  ת  וּבְאַרְבַּע עֶשְׂרֵי שְׁנִין אֲתָא כְּדָרְלָעֹמֶר וּמַלְכַיָא דְעִמֵהּ וּמְחוֹ יָת גִבָּרַיָא דִי בְעַשְׁתְּרוֹת קַרְנַיִם וְיָת תַּקִיפַיָא דִבְהֶמְתָּא וְיָת אֵמְתָנֵי דִבְשָׁוֵה קִרְיָתָיִם:

{ו} וְאֶת הַֽחֹרִי בְּהַֽרְרָם שֵׂעִיר עַד אֵיל פָּארָן אֲשֶׁר עַל הַמִּדְבָּֽר:  מ  וְאֶת הַֽחֹרִי בְּהַֽרְרָם שֵׂעִיר עַד אֵיל פָּארָן אֲשֶׁר עַל הַמִּדְבָּֽר:  ת  וְיָת חוֹרָאֵי דִי בְּטוּרְהוֹן דְשֵׂעִיר עַד מֵישַׁר פָּארָן דִסְמִיךְ עַל מַדְבְּרָא:

{ז} וַיָּשֻׁבוּ וַיָּבֹאוּ אֶל עֵין מִשְׁפָּט הִוא קָדֵשׁ וַיַּכּוּ אֶת כָּל שְׂדֵה הָֽעֲמָֽלֵקִי וְגַם אֶת הָאֱמֹרִי הַיּשֵׁב בְּחַֽצֲצֹן תָּמָֽר:  מ  וַיָּשֻׁבוּ וַיָּבֹאוּ אֶל עֵין מִשְׁפָּט הִוא קָדֵשׁ וַיַּכּוּ אֶת כָּל שְׂדֵה הָֽעֲמָֽלֵקִי וְגַם אֶת הָאֱמֹרִי הַיּשֵׁב בְּחַֽצֲצֹן תָּמָֽר:  ת  וְתָבוּ וַאֲתוּ לְמֵישַׁר פִּלּוּג דִינָא הִיא רְקָם וּמְחוֹ יָת כָּל חֲקַל עֲמַלְקָאָה וְאַף יָת אֱמוֹרָאָה דְיָתֵיב בְּעֵין גֶּדִי:

{ח} וַיֵּצֵא מֶֽלֶךְ סְדֹם וּמֶלֶךְ עֲמֹרָה וּמֶלֶךְ אַדְמָה וּמֶלֶךְ (צביים) צְבוֹיִם וּמֶלֶךְ בֶּלַע הִוא צֹעַר וַיַּֽעַרְכוּ אִתָּם מִלְחָמָה בְּעֵמֶק הַשִּׂדִּֽים:  מ  וַיֵּצֵא מֶֽלֶךְ סְדֹם וּמֶלֶךְ עֲמֹרָה וּמֶלֶךְ אַדְמָה וּמֶלֶךְ (צביים) צְבוֹיִם וּמֶלֶךְ בֶּלַע הִוא צֹעַר וַיַּֽעַרְכוּ אִתָּם מִלְחָמָה בְּעֵמֶק הַשִּׂדִּֽים:  ת  וּנְפַק מַלְכָּא דִסְדוֹם וּמַלְכָּא דַעֲמוֹרָה וּמַלְכָּא דְאַדְמָה וּמַלְכָּא דִצְבוֹיִם וּמַלְכָּא דְבֶלַע הִיא צוֹעַר וְסַדָרוּ עִמְהוֹן קְרָבָא בְּמֵישַׁר חַקְלַיָא:

{ט} אֵת כְּדָרְלָעֹמֶר מֶלֶךְ עֵילָם וְתִדְעָל מֶלֶךְ גּוֹיִם וְאַמְרָפֶל מֶלֶךְ שִׁנְעָר וְאַרְיוֹךְ מֶלֶךְ אֶלָּסָר אַרְבָּעָה מְלָכִים אֶת הַֽחֲמִשָּֽׁה:  מ  אֵת כְּדָרְלָעֹמֶר מֶלֶךְ עֵילָם וְתִדְעָל מֶלֶךְ גּוֹיִם וְאַמְרָפֶל מֶלֶךְ שִׁנְעָר וְאַרְיוֹךְ מֶלֶךְ אֶלָּסָר אַרְבָּעָה מְלָכִים אֶת הַֽחֲמִשָּֽׁה:  ת  עִם כְּדָרְלָעֹמֶר מַלְכָּא דְעֵילָם וְתִדְעָל מַלְכָּא דְעַמְמִין וְאַמְרָפֶל מַלְכָּא דְבָבֶל וְאַרְיוֹךְ מַלְכָּא דְאֶלָּסָר אַרְבְּעָה מַלְכִין לָקֳבֵיל חַמְשָׁא:

{י} וְעֵמֶק הַשִּׂדִּים בֶּֽאֱרֹת בֶּֽאֱרֹת חֵמָר וַיָּנֻסוּ מֶֽלֶךְ סְדֹם וַֽעֲמֹרָה וַיִּפְּלוּ שָׁמָּה וְהַנִּשְׁאָרִים הֶרָה נָּֽסוּ:  מ  וְעֵמֶק הַשִּׂדִּים בֶּֽאֱרֹת בֶּֽאֱרֹת חֵמָר וַיָּנֻסוּ מֶֽלֶךְ סְדֹם וַֽעֲמֹרָה וַיִּפְּלוּ שָׁמָּה וְהַנִּשְׁאָרִים הֶרָה נָּֽסוּ:  ת  וּמֵישַׁר חַקְלַיָא בֵּירִין בֵּירִין מַסְקַן חֵימָרָא וַעֲרָקוּ מַלְכָּא דִסְדוֹם וַעֲמוֹרָה וּנְפָלוּ תַמָן וּדְאִשְׁתָּאֲרוּ לְטוּרָא עֲרָקוּ:

{יא} וַיִּקְחוּ אֶת כָּל רְכֻשׁ סְדֹם וַֽעֲמֹרָה וְאֶת כָּל אָכְלָם וַיֵּלֵֽכוּ:  מ  וַיִּקְחוּ אֶת כָּל רְכֻשׁ סְדֹם וַֽעֲמֹרָה וְאֶת כָּל אָכְלָם וַיֵּלֵֽכוּ:  ת  וּשְׁבוֹ יָת כָּל קִנְיָנָא דִסְדוֹם וַעֲמוֹרָה וְיָת כָּל מֵיכָלְהוֹן וַאֲזָלוּ:

{יב} וַיִּקְחוּ אֶת לוֹט וְאֶת רְכֻשׁוֹ בֶּן אֲחִי אַבְרָם וַיֵּלֵכוּ וְהוּא ישֵׁב בִּסְדֹֽם:  מ  וַיִּקְחוּ אֶת לוֹט וְאֶת רְכֻשׁוֹ בֶּן אֲחִי אַבְרָם וַיֵּלֵכוּ וְהוּא ישֵׁב בִּסְדֹֽם:  ת  וּשְׁבוֹ יָת לוֹט וְיָת קִנְיָנֵהּ בַּר אַחוּהִי דְאַבְרָם וַאֲזָלוּ וְהוּא יָתֵיב בִּסְדוֹם:

{יג} וַיָּבֹא הַפָּלִיט וַיַּגֵּד לְאַבְרָם הָֽעִבְרִי וְהוּא שֹׁכֵן בְּאֵֽלֹנֵי מַמְרֵא הָֽאֱמֹרִי אֲחִי אֶשְׁכֹּל וַֽאֲחִי עָנֵר וְהֵם בַּֽעֲלֵי בְרִית אַבְרָֽם:  מ  וַיָּבֹא הַפָּלִיט וַיַּגֵּד לְאַבְרָם הָֽעִבְרִי וְהוּא שֹׁכֵן בְּאֵֽלֹנֵי מַמְרֵא הָֽאֱמֹרִי אֲחִי אֶשְׁכֹּל וַֽאֲחִי עָנֵר וְהֵם בַּֽעֲלֵי בְרִית אַבְרָֽם:  ת  וַאֲתָא מְשֵׁיזָבָא וְחַוִּי לְאַבְרָם עִבְרָאָה וְהוּא שָׁרֵי בְּמֵישְׁרֵי מַמְרֵא אֱמוֹרָאָה אֲחוּהִי דְאֶשְׁכּוֹל וַאֲחוּהִי דְעָנֵר וְאִינוּן אֱנָשֵׁי קְיָמֵהּ דְאַבְרָם:

{יד} וַיִּשְׁמַע אַבְרָם כִּי נִשְׁבָּה אָחִיו וַיָּרֶק אֶת חֲנִיכָיו יְלִידֵי בֵיתוֹ שְׁמֹנָה עָשָׂר וּשְׁלֹשׁ מֵאוֹת וַיִּרְדֹּף עַד דָּֽן:  מ  וַיִּשְׁמַע אַבְרָם כִּי נִשְׁבָּה אָחִיו וַיָּרֶק אֶת חֲנִיכָיו יְלִידֵי בֵיתוֹ שְׁמֹנָה עָשָׂר וּשְׁלֹשׁ מֵאוֹת וַיִּרְדֹּף עַד דָּֽן:  ת  וּשְׁמַע אַבְרָם אֲרֵי אִשְׁתְּבִי אֲחוּהִי וְזָרֵיז יָת עוּלֵמוֹהִי יְלִידֵי בֵיתֵהּ תְּלַת מְאָה וְתַמְנֵי עֲסַר וּרְדַף עַד דָן:

{טו} וַיֵּֽחָלֵק עֲלֵיהֶם לַיְלָה הוּא וַֽעֲבָדָיו וַיַּכֵּם וַֽיִּרְדְּפֵם עַד חוֹבָה אֲשֶׁר מִשְּׂמֹאל לְדַמָּֽשֶׂק:  מ  וַיֵּֽחָלֵק עֲלֵיהֶם לַיְלָה הוּא וַֽעֲבָדָיו וַיַּכֵּם וַֽיִּרְדְּפֵם עַד חוֹבָה אֲשֶׁר מִשְּׂמֹאל לְדַמָּֽשֶׂק:  ת  וְאִתְפְּלֵג עֲלֵיהוֹן לֵילְיָא הוּא וְעַבְדוֹהִי וּמְחוֹנוּן וּרְדָפִנוּן עַד חוֹבָא דִי מִצִפּוּנָא לְדַמָשֶׂק:

{טז} וַיָּשֶׁב אֵת כָּל הָֽרְכֻשׁ וְגַם אֶת לוֹט אָחִיו וּרְכֻשׁוֹ הֵשִׁיב וְגַם אֶת הַנָּשִׁים וְאֶת הָעָֽם:  מ  וַיָּשֶׁב אֵת כָּל הָֽרְכֻשׁ וְגַם אֶת לוֹט אָחִיו וּרְכֻשׁוֹ הֵשִׁיב וְגַם אֶת הַנָּשִׁים וְאֶת הָעָֽם:  ת  וַאֲתֵיב יָת כָּל קִנְיָנָא וְאַף יָת לוֹט בַּר אֲחוּהִי וְקִנְיָנֵהּ אֲתֵיב וְאַף יָת נְשַׁיָא וְיָת עַמָא:

{יז} וַיֵּצֵא מֶֽלֶךְ סְדֹם לִקְרָאתוֹ אַֽחֲרֵי שׁוּבוֹ מֵֽהַכּוֹת אֶת כְּדָרְלָעֹמֶר וְאֶת הַמְּלָכִים אֲשֶׁר אִתּוֹ אֶל עֵמֶק שָׁוֵה הוּא עֵמֶק הַמֶּֽלֶךְ:  מ  וַיֵּצֵא מֶֽלֶךְ סְדֹם לִקְרָאתוֹ אַֽחֲרֵי שׁוּבוֹ מֵֽהַכּוֹת אֶת כְּדָרְלָעֹמֶר וְאֶת הַמְּלָכִים אֲשֶׁר אִתּוֹ אֶל עֵמֶק שָׁוֵה הוּא עֵמֶק הַמֶּֽלֶךְ:  ת  וּנְפַק מַלְכָּא דִסְדוֹם לְקַדָמוּתֵה בָּתַר דְתָב מִלְמִמְחֵי יָת כְּדָרְלָעוֹמֶר וְיָת מַלְכַיָא דִי עִמֵהּ לְמֵישַׁר מַפְנָא הוּא אֲתַר בֵּית רֵיסָא דְמַלְכָּא:

{יח} וּמַלְכִּי צֶדֶק מֶלֶךְ שָׁלֵם הוֹצִיא לֶחֶם וָיָיִן וְהוּא כֹהֵן לְאֵל עֶלְיֽוֹן:  מ  וּמַלְכִּי צֶדֶק מֶלֶךְ שָׁלֵם הוֹצִיא לֶחֶם וָיָיִן וְהוּא כֹהֵן לְאֵל עֶלְיֽוֹן:  ת  וּמַלְכִּי צֶדֶק מַלְכָּא דִירוּשְׁלֶם אַפֵּיק לְחֵם וַחֲמָר וְהוּא מְשַׁמֵשׁ קֳדָם אֵל עִלָאָה:

{יט} וַיְבָֽרְכֵהוּ וַיֹּאמַר בָּרוּךְ אַבְרָם לְאֵל עֶלְיוֹן קֹנֵה שָׁמַיִם וָאָֽרֶץ:  מ  וַיְבָֽרְכֵהוּ וַיֹּאמַר בָּרוּךְ אַבְרָם לְאֵל עֶלְיוֹן קֹנֵה שָׁמַיִם וָאָֽרֶץ:  ת  וּבָרְכֵיהּ וַאֲמָר בְּרִיךְ אַבְרָם לְאֵל עִלָאָה דְקִנְיָנֵהּ שְׁמַיָא וְאַרְעָא:

{כ} וּבָרוּךְ אֵל עֶלְיוֹן אֲשֶׁר מִגֵּן צָרֶיךָ בְּיָדֶךָ וַיִּתֶּן לוֹ מַֽעֲשֵׂר מִכֹּֽל:  מ  וּבָרוּךְ אֵל עֶלְיוֹן אֲשֶׁר מִגֵּן צָרֶיךָ בְּיָדֶךָ וַיִּתֶּן לוֹ מַֽעֲשֵׂר מִכֹּֽל:  ת  וּבְרִיךְ אֵל עִלָאָה דִמְסַר סָנְאָיךְ בִּידָךְ וִיהַב לֵיהּ חַד מִן עַשְׂרָא מִכֹּלָא:

{כא}  חמישי  וַיֹּאמֶר מֶֽלֶךְ סְדֹם אֶל אַבְרָם תֶּן לִי הַנֶּפֶשׁ וְהָֽרְכֻשׁ קַח לָֽךְ:  מ  וַיֹּאמֶר מֶֽלֶךְ סְדֹם אֶל אַבְרָם תֶּן לִי הַנֶּפֶשׁ וְהָֽרְכֻשׁ קַח לָֽךְ:  ת  וַאֲמַר מַלְכָּא דִסְדוֹם לְאַבְרָם הַב לִי נַפְשָׁתָא וְקִנְיָנָא (סַב) דְבַר לָךְ:

{כב} וַיֹּאמֶר אַבְרָם אֶל מֶלֶךְ סְדֹם הֲרִמֹתִי יָדִי אֶל יְהֹוָה אֵל עֶלְיוֹן קֹנֵה שָׁמַיִם וָאָֽרֶץ:  מ  וַיֹּאמֶר אַבְרָם אֶל מֶלֶךְ סְדֹם הֲרִמֹתִי יָדִי אֶל יְהֹוָה אֵל עֶלְיוֹן קֹנֵה שָׁמַיִם וָאָֽרֶץ:  ת  וַאֲמַר אַבְרָם לְמַלְכָּא דִסְדוֹם אֲרֵמִית יְדַי בִּצְלוֹ קֳדָם יְיָ אֵל עִלָאָה דְקִנְיָנֵהּ שְׁמַיָא וְאַרְעָא:

{כג} אִם מִחוּט וְעַד שְׂרוֹךְ נַעַל וְאִם אֶקַּח מִכָּל אֲשֶׁר לָךְ וְלֹא תֹאמַר אֲנִי הֶֽעֱשַׁרְתִּי אֶת אַבְרָֽם:  מ  אִם מִחוּט וְעַד שְׂרוֹךְ נַעַל וְאִם אֶקַּח מִכָּל אֲשֶׁר לָךְ וְלֹא תֹאמַר אֲנִי הֶֽעֱשַׁרְתִּי אֶת אַבְרָֽם:  ת  אִם מִחוּטָא וְעַד עַרְקַת מְסָנָא וְאִם אֶסַב מִכָּל דִי לָךְ וְלָא תֵימַר אֲנָא אַעְתָּרִית יָת אַבְרָם:

{כד} בִּלְעָדַי רַק אֲשֶׁר אָֽכְלוּ הַנְּעָרִים וְחֵלֶק הָֽאֲנָשִׁים אֲשֶׁר הָֽלְכוּ אִתִּי עָנֵר אֶשְׁכֹּל וּמַמְרֵא הֵם יִקְחוּ חֶלְקָֽם:  מ  בִּלְעָדַי רַק אֲשֶׁר אָֽכְלוּ הַנְּעָרִים וְחֵלֶק הָֽאֲנָשִׁים אֲשֶׁר הָֽלְכוּ אִתִּי עָנֵר אֶשְׁכֹּל וּמַמְרֵא הֵם יִקְחוּ חֶלְקָֽם:  ת  לְחוֹד (בַּר) מִדִאֲכָלוּ עוּלֵמַיָא וְחָלָק גֻבְרַיָא דִי אֲזָלוּ עִמִי עָנֵר אֶשְׁכּוֹל וּמַמְרֵא אִינוּן יְקַבְּלוּן חַלָקְהוֹן: (ס)




בראשית פרק-טו

{א} אַחַר הַדְּבָרִים הָאֵלֶּה הָיָה דְבַר יְהֹוָה אֶל אַבְרָם בַּֽמַּֽחֲזֶה לֵאמֹר אַל תִּירָא אַבְרָם אָֽנֹכִי מָגֵן לָךְ שְׂכָֽרְךָ הַרְבֵּה מְאֹֽד:  מ  אַחַר הַדְּבָרִים הָאֵלֶּה הָיָה דְבַר יְהֹוָה אֶל אַבְרָם בַּֽמַּֽחֲזֶה לֵאמֹר אַל תִּירָא אַבְרָם אָֽנֹכִי מָגֵן לָךְ שְׂכָֽרְךָ הַרְבֵּה מְאֹֽד:  ת  בָּתַר פִּתְגָמַיָא הָאִלֵין הֲוָה פִּתְגָמָא דַיְיָ עִם אַבְרָם בִּנְבוּאָה לְמֵימָר לָא תִדְחַל אַבְרָם מֵימְרִי תְּקוֹף לָךְ אַגְרָךְ סַגִי לַחֲדָא:

{ב} וַיֹּאמֶר אַבְרָם אֲדֹנָי יֱהוִֹה מַה תִּתֶּן לִי וְאָֽנֹכִי הוֹלֵךְ עֲרִירִי וּבֶן מֶשֶׁק בֵּיתִי הוּא דַּמֶּשֶׂק אֱלִיעֶֽזֶר:  מ  וַיֹּאמֶר אַבְרָם אֲדֹנָי יֱהוִֹה מַה תִּתֶּן לִי וְאָֽנֹכִי הוֹלֵךְ עֲרִירִי וּבֶן מֶשֶׁק בֵּיתִי הוּא דַּמֶּשֶׂק אֱלִיעֶֽזֶר:  ת  וַאֲמַר אַבְרָם יְיָ אֱלֹהִים מַה תִּתֶּן לִי וַאֲנָא אָזֵיל בְּלָא וְלָד וּבַר פַּרְנָסָא הָדֵין דִבְּבֵיתִי הוּא דַמַשְׂקָאָה אֱלִיעֶזֶר:

{ג} וַיֹּאמֶר אַבְרָם הֵן לִי לֹא נָתַתָּה זָרַע וְהִנֵּה בֶן בֵּיתִי יוֹרֵשׁ אֹתִֽי:  מ  וַיֹּאמֶר אַבְרָם הֵן לִי לֹא נָתַתָּה זָרַע וְהִנֵּה בֶן בֵּיתִי יוֹרֵשׁ אֹתִֽי:  ת  וַאֲמַר אַבְרָם הָא לִי לָא יְהַבְתְּ וְלָד וְהָא בַר בֵּיתִי יָרִית יָתִי:

{ד} וְהִנֵּה דְבַר יְהֹוָה אֵלָיו לֵאמֹר לֹא יִירָֽשְׁךָ זֶה כִּי אִם אֲשֶׁר יֵצֵא מִמֵּעֶיךָ הוּא יִֽירָשֶֽׁךָ:  מ  וְהִנֵּה דְבַר יְהֹוָה אֵלָיו לֵאמֹר לֹא יִירָֽשְׁךָ זֶה כִּי אִם אֲשֶׁר יֵצֵא מִמֵּעֶיךָ הוּא יִֽירָשֶֽׁךָ:  ת  וְהָא פִּתְגָמָא דַיְיָ עִמֵהּ לְמֵימַר לָא יִרְתִינָךְ דֵין אֱלָהֵין בַּר דְתוֹלִיד הוּא יִרְתִינָךְ:

{ה} וַיּוֹצֵא אֹתוֹ הַחוּצָה וַיֹּאמֶר הַבֶּט נָא הַשָּׁמַיְמָה וּסְפֹר הַכּוֹכָבִים אִם תּוּכַל לִסְפֹּר אֹתָם וַיֹּאמֶר לוֹ כֹּה יִֽהְיֶה זַרְעֶֽךָ:  מ  וַיּוֹצֵא אֹתוֹ הַחוּצָה וַיֹּאמֶר הַבֶּט נָא הַשָּׁמַיְמָה וּסְפֹר הַכּוֹכָבִים אִם תּוּכַל לִסְפֹּר אֹתָם וַיֹּאמֶר לוֹ כֹּה יִֽהְיֶה זַרְעֶֽךָ:  ת  וְאַפֵּיק יָתֵהּ לְבָרָא וַאֲמַר אִסְתְּכִי כְעַן לִשְׁמַיָא וּמְנֵי כּוֹכְבַיָא אִם תִּכּוֹל לְמִמְנֵי יָתְהוֹן וַאֲמַר לֵיהּ כְּדֵין יְהוֹן בְּנָךְ:

{ו} וְהֶֽאֱמִן בַּֽיהֹוָה וַיַּחְשְׁבֶהָ לּוֹ צְדָקָֽה:  מ  וְהֶֽאֱמִן בַּֽיהֹוָה וַיַּחְשְׁבֶהָ לּוֹ צְדָקָֽה:  ת  וְהֵימִין בְּמֵימְרָא דַיְיָ וְחָשְׁבַהּ לֵיהּ לְזָכוּ:

{ז}  שישי  וַיֹּאמֶר אֵלָיו אֲנִי יְהֹוָה אֲשֶׁר הֽוֹצֵאתִיךָ מֵאוּר כַּשְׂדִּים לָתֶת לְךָ אֶת הָאָרֶץ הַזֹּאת לְרִשְׁתָּֽהּ:  מ  וַיֹּאמֶר אֵלָיו אֲנִי יְהֹוָה אֲשֶׁר הֽוֹצֵאתִיךָ מֵאוּר כַּשְׂדִּים לָתֶת לְךָ אֶת הָאָרֶץ הַזֹּאת לְרִשְׁתָּֽהּ:  ת  וַאֲמַר לֵהּ אֲנָא יְיָ דְאַפֵּקְתָּךְ מֵאוּרָא דְכַשְׂדָאֵי לְמִתַּן לָךְ יָת אַרְעָא הָדָא לְמֵירְתַהּ:

{ח} וַיֹּאמַר אֲדֹנָי יֱהֹוִה בַּמָּה אֵדַע כִּי אִֽירָשֶֽׁנָּה:  מ  וַיֹּאמַר אֲדֹנָי יֱהֹוִה בַּמָּה אֵדַע כִּי אִֽירָשֶֽׁנָּה:  ת  וַאֲמָר יְיָ אֱלֹהִים בְּמָה אִידַע אֲרֵי אֵרְתִנַהּ:

{ט} וַיֹּאמֶר אֵלָיו קְחָה לִי עֶגְלָה מְשֻׁלֶּשֶׁת וְעֵז מְשֻׁלֶּשֶׁת וְאַיִל מְשֻׁלָּשׁ וְתֹר וְגוֹזָֽל:  מ  וַיֹּאמֶר אֵלָיו קְחָה לִי עֶגְלָה מְשֻׁלֶּשֶׁת וְעֵז מְשֻׁלֶּשֶׁת וְאַיִל מְשֻׁלָּשׁ וְתֹר וְגוֹזָֽל:  ת  וַאֲמַר לֵהּ קָרֵב קֳדָמַי עֶגְלִין תְּלָתָא וְעִזִין תְּלָתָא וְדִכְרִין תְּלָת וְשַׁפְנִינָא וּבַר יוֹנָה:

{י} וַיִּֽקַּח לוֹ אֶת כָּל אֵלֶּה וַיְבַתֵּר אֹתָם בַּתָּוֶךְ וַיִּתֵּן אִישׁ בִּתְרוֹ לִקְרַאת רֵעֵהוּ וְאֶת הַצִּפֹּר לֹא בָתָֽר:  מ  וַיִּֽקַּח לוֹ אֶת כָּל אֵלֶּה וַיְבַתֵּר אֹתָם בַּתָּוֶךְ וַיִּתֵּן אִישׁ בִּתְרוֹ לִקְרַאת רֵעֵהוּ וְאֶת הַצִּפֹּר לֹא בָתָֽר:  ת  וְקָרֵב קֳדָמוֹהִי יָת כָּל אִלֵין וּפַלִיג יָתְהוֹן בְּשָׁוֶה וִיהַב פַּלְגַיָא פִּלוּג לָקֳבֵיל חַבְרֵיהּ וְיָת עוֹפָא לָא פַּלִיג:

{יא} וַיֵּרֶד הָעַיִט עַל הַפְּגָרִים וַיַּשֵּׁב אֹתָם אַבְרָֽם:  מ  וַיֵּרֶד הָעַיִט עַל הַפְּגָרִים וַיַּשֵּׁב אֹתָם אַבְרָֽם:  ת  וּנְחַת עוֹפָא עַל פַּגְלַיָא וְאַפְרַח יָתְהוֹן אַבְרָם:

{יב} וַיְהִי הַשֶּׁמֶשׁ לָבוֹא וְתַרְדֵּמָה נָֽפְלָה עַל אַבְרָם וְהִנֵּה אֵימָה חֲשֵׁכָה גְדֹלָה נֹפֶלֶת עָלָֽיו:  מ  וַיְהִי הַשֶּׁמֶשׁ לָבוֹא וְתַרְדֵּמָה נָֽפְלָה עַל אַבְרָם וְהִנֵּה אֵימָה חֲשֵׁכָה גְדֹלָה נֹפֶלֶת עָלָֽיו:  ת  וַהֲוָה שִׁמְשָׁא לְמֵיעַל וְשִׁינְתָא נְפָלַת עַל אַבְרָם וְהָא אֵימָתָא קְבֵל סַגִי נְפָלַת עֲלוֹהִי:

{יג} וַיֹּאמֶר לְאַבְרָם יָדֹעַ תֵּדַע כִּי גֵר יִֽהְיֶה זַרְעֲךָ בְּאֶרֶץ לֹא לָהֶם וַֽעֲבָדוּם וְעִנּוּ אֹתָם אַרְבַּע מֵאוֹת שָׁנָֽה:  מ  וַיֹּאמֶר לְאַבְרָם יָדֹעַ תֵּדַע כִּי גֵר יִֽהְיֶה זַרְעֲךָ בְּאֶרֶץ לֹא לָהֶם וַֽעֲבָדוּם וְעִנּוּ אֹתָם אַרְבַּע מֵאוֹת שָׁנָֽה:  ת  וַאֲמַר לְאַבְרָם מִדַע תִּדַע אֲרֵי דַיָרִין יְהוֹן בְּנָיךְ בְּאַרְעָא דְלָא דִילְהוֹן וְיִפְלְחוּן בְּהוֹן וִיעַנוּן יָתְהוֹן אַרְבַּע מְאָה שְׁנִין:

{יד} וְגַם אֶת הַגּוֹי אֲשֶׁר יַֽעֲבֹדוּ דָּן אָנֹכִי וְאַֽחֲרֵי כֵן יֵֽצְאוּ בִּרְכֻשׁ גָּדֽוֹל:  מ  וְגַם אֶת הַגּוֹי אֲשֶׁר יַֽעֲבֹדוּ דָּן אָנֹכִי וְאַֽחֲרֵי כֵן יֵֽצְאוּ בִּרְכֻשׁ גָּדֽוֹל:  ת  וְאַף יָת עַמָא דְיִפְלְחוּן בְּהוֹן דָיִין אֲנָא וּבָתַר כֵּן יִפְקוּן בְּקִנְיָנָא סַגִי:

{טו} וְאַתָּה תָּבוֹא אֶל אֲבֹתֶיךָ בְּשָׁלוֹם תִּקָּבֵר בְּשֵׂיבָה טוֹבָֽה:  מ  וְאַתָּה תָּבוֹא אֶל אֲבֹתֶיךָ בְּשָׁלוֹם תִּקָּבֵר בְּשֵׂיבָה טוֹבָֽה:  ת  וְאַתְּ תֵעוֹל לְוָת אַבָהָתָךְ בִּשְׁלָם תִּתְקְבַר בְּסֵיבוּ טָבָא:

{טז} וְדוֹר רְבִיעִי יָשׁוּבוּ הֵנָּה כִּי לֹֽא שָׁלֵם עֲוֹן הָֽאֱמֹרִי עַד הֵֽנָּה:  מ  וְדוֹר רְבִיעִי יָשׁוּבוּ הֵנָּה כִּי לֹֽא שָׁלֵם עֲוֹן הָֽאֱמֹרִי עַד הֵֽנָּה:  ת  וְדָרָא רְבִיעָאָה יְתוּבוּן הָכָא אֲרֵי לָא שְׁלִים חוֹבָא דֶאֱמוֹרָאָה עַד כְּעָן:

{יז} וַיְהִי הַשֶּׁמֶשׁ בָּאָה וַֽעֲלָטָה הָיָה וְהִנֵּה תַנּוּר עָשָׁן וְלַפִּיד אֵשׁ אֲשֶׁר עָבַר בֵּין הַגְּזָרִים הָאֵֽלֶּה:  מ  וַיְהִי הַשֶּׁמֶשׁ בָּאָה וַֽעֲלָטָה הָיָה וְהִנֵּה תַנּוּר עָשָׁן וְלַפִּיד אֵשׁ אֲשֶׁר עָבַר בֵּין הַגְּזָרִים הָאֵֽלֶּה:  ת  וַהֲוָה שִׁמְשָׁא עַלַת וְקִבְלָא הֲוָה וְהָא תַּנוּר דִתְנַן וּבָעוּר דְאֶשָׁתָא דִעֲדָא בֵּין פַּלְגַיָא הָאִלֵין:

{יח} בַּיּוֹם הַהוּא כָּרַת יְהֹוָה אֶת אַבְרָם בְּרִית לֵאמֹר לְזַרְעֲךָ נָתַתִּי אֶת הָאָרֶץ הַזֹּאת מִנְּהַר מִצְרַיִם עַד הַנָּהָר הַגָּדֹל נְהַר פְּרָֽת:  מ  בַּיּוֹם הַהוּא כָּרַת יְהֹוָה אֶת אַבְרָם בְּרִית לֵאמֹר לְזַרְעֲךָ נָתַתִּי אֶת הָאָרֶץ הַזֹּאת מִנְּהַר מִצְרַיִם עַד הַנָּהָר הַגָּדֹל נְהַר פְּרָֽת:  ת  בְּיוֹמָא הַהוּא גְזַר יְיָ עִם אַבְרָם קְיָם לְמֵימָר לִבְנָךְ יְהָבִית יָת אַרְעָא הָדָא מִן נַהֲרָא דְמִצְרַיִם וְעַד נַהֲרָא רַבָּא נַהֲרָא פְרָת:

{יט} אֶת הַקֵּינִי וְאֶת הַקְּנִזִּי וְאֵת הַקַּדְמֹנִֽי:  מ  אֶת הַקֵּינִי וְאֶת הַקְּנִזִּי וְאֵת הַקַּדְמֹנִֽי:  ת  יָת שַׁלְמָאֵי וְיָת קְנִזָאֵי וְיָת קַדְמוֹנָאֵי:

{כ} וְאֶת הַֽחִתִּי וְאֶת הַפְּרִזִּי וְאֶת הָֽרְפָאִֽים:  מ  וְאֶת הַֽחִתִּי וְאֶת הַפְּרִזִּי וְאֶת הָֽרְפָאִֽים:  ת  וְיָת חִתָּאֵי וְיָת פְּרִזָאֵי וְיָת גִבָּרָאֵי:

{כא} וְאֶת הָֽאֱמֹרִי וְאֶת הַֽכְּנַֽעֲנִי וְאֶת הַגִּרְגָּשִׁי וְאֶת הַיְבוּסִֽי:  מ  וְאֶת הָֽאֱמֹרִי וְאֶת הַֽכְּנַֽעֲנִי וְאֶת הַגִּרְגָּשִׁי וְאֶת הַיְבוּסִֽי:  ת  וְיָת אֱמוֹרָאֵי וְיָת כְנַעֲנָאֵי וְיָת גִרְגָשָׁאֵי וְיָת יְבוּסָאֵי: (ס)




בראשית פרק-טז

{א} וְשָׂרַי אֵשֶׁת אַבְרָם לֹא יָֽלְדָה לוֹ וְלָהּ שִׁפְחָה מִצְרִית וּשְׁמָהּ הָגָֽר:  מ  וְשָׂרַי אֵשֶׁת אַבְרָם לֹא יָֽלְדָה לוֹ וְלָהּ שִׁפְחָה מִצְרִית וּשְׁמָהּ הָגָֽר:  ת  וְשָׂרַי אִתַּת אַבְרָם לָא יְלִידַת לֵהּ וְלַהּ אַמְתָא מִצְרֵיתָא וּשְׁמַהּ הָגָר:

{ב} וַתֹּאמֶר שָׂרַי אֶל אַבְרָם הִנֵּה נָא עֲצָרַנִי יְהֹוָה מִלֶּדֶת בֹּא נָא אֶל שִׁפְחָתִי אוּלַי אִבָּנֶה מִמֶּנָּה וַיִּשְׁמַע אַבְרָם לְקוֹל שָׂרָֽי:  מ  וַתֹּאמֶר שָׂרַי אֶל אַבְרָם הִנֵּה נָא עֲצָרַנִי יְהֹוָה מִלֶּדֶת בֹּא נָא אֶל שִׁפְחָתִי אוּלַי אִבָּנֶה מִמֶּנָּה וַיִּשְׁמַע אַבְרָם לְקוֹל שָׂרָֽי:  ת  וַאֲמַרַת שָׂרַי לְאַבְרָם הָא כְעַן מַנְעֲנִי יְיָ מִלְמֵילַד עוּל כְּעַן לְוַת אַמְתִי מָאִים אִתְבְּנֵי מִנַהּ וְקַבִּיל אַבְרָם לְמֵימַר שָׂרָי:

{ג} וַתִּקַּח שָׂרַי אֵשֶׁת אַבְרָם אֶת הָגָר הַמִּצְרִית שִׁפְחָתָהּ מִקֵּץ עֶשֶׂר שָׁנִים לְשֶׁבֶת אַבְרָם בְּאֶרֶץ כְּנָעַן וַתִּתֵּן אֹתָהּ לְאַבְרָם אִישָׁהּ לוֹ לְאִשָּֽׁה:  מ  וַתִּקַּח שָׂרַי אֵשֶׁת אַבְרָם אֶת הָגָר הַמִּצְרִית שִׁפְחָתָהּ מִקֵּץ עֶשֶׂר שָׁנִים לְשֶׁבֶת אַבְרָם בְּאֶרֶץ כְּנָעַן וַתִּתֵּן אֹתָהּ לְאַבְרָם אִישָׁהּ לוֹ לְאִשָּֽׁה:  ת  וּדְבָרַת שָׂרַי אִתַּת אַבְרָם יָת הָגָר מִצְרֵיתָא אַמְתַהּ מִסוֹף עֲשַׂר שְׁנִין לְמִיתַּב אַבְרָם בְּאַרְעָא דִכְנָעַן וִיהָבַת יָתַהּ לְאַבְרָם בַּעֲלַהּ לֵהּ לְאִנְתּוּ:

{ד} וַיָּבֹא אֶל הָגָר וַתַּהַר וַתֵּרֶא כִּי הָרָתָה וַתֵּקַל גְּבִרְתָּהּ בְּעֵינֶֽיהָ:  מ  וַיָּבֹא אֶל הָגָר וַתַּהַר וַתֵּרֶא כִּי הָרָתָה וַתֵּקַל גְּבִרְתָּהּ בְּעֵינֶֽיהָ:  ת  וְעַל לְוַת הָגָר וְעַדִיאַת וַחֲזַת אֲרֵי עֲדִיאַת וּקְלַת רִבָּנְתַּהּ בְּעֵינָהַהּ:

{ה} וַתֹּאמֶר שָׂרַי אֶל אַבְרָם חֲמָסִי עָלֶיךָ אָֽנֹכִי נָתַתִּי שִׁפְחָתִי בְּחֵיקֶךָ וַתֵּרֶא כִּי הָרָתָה וָֽאֵקַל בְּעֵינֶיהָ יִשְׁפֹּט יְהֹוָה בֵּינִי וּבֵינֶֽיךָ:  מ  וַתֹּאמֶר שָׂרַי אֶל אַבְרָם חֲמָסִי עָלֶיךָ אָֽנֹכִי נָתַתִּי שִׁפְחָתִי בְּחֵיקֶךָ וַתֵּרֶא כִּי הָרָתָה וָֽאֵקַל בְּעֵינֶיהָ יִשְׁפֹּט יְהֹוָה בֵּינִי וּבֵינֶֽיךָ:  ת  וַאֲמַרַת שָׂרַי לְאַבְרָם דִין לִי עֲלָךְ אֲנָא יְהָבִית אַמְתִי לָךְ וַחֲזַת אֲרֵי עֲדִיאַת וּקְלֵית בְּעֵינָהַהּ יְדִין יְיָ בֵּינִי וּבֵינָךְ:

{ו} וַיֹּאמֶר אַבְרָם אֶל שָׂרַי הִנֵּה שִׁפְחָתֵךְ בְּיָדֵךְ עֲשִׂי לָהּ הַטּוֹב בְּעֵינָיִךְ וַתְּעַנֶּהָ שָׂרַי וַתִּבְרַח מִפָּנֶֽיהָ:  מ  וַיֹּאמֶר אַבְרָם אֶל שָׂרַי הִנֵּה שִׁפְחָתֵךְ בְּיָדֵךְ עֲשִׂי לָהּ הַטּוֹב בְּעֵינָיִךְ וַתְּעַנֶּהָ שָׂרַי וַתִּבְרַח מִפָּנֶֽיהָ:  ת  וַאֲמַר אַבְרָם לְשָׂרַי הָא אַמְתִיךְ בִּידִךְ עִבִידִי לַהּ כִּדְתַקִין בְעֵינָיְכִי וְעֲנִיתַהּ שָׂרַי וְעָרְקַת מִקֳדָמָהַהּ:

{ז} וַֽיִּמְצָאָהּ מַלְאַךְ יְהֹוָה עַל עֵין הַמַּיִם בַּמִּדְבָּר עַל הָעַיִן בְּדֶרֶךְ שֽׁוּר:  מ  וַֽיִּמְצָאָהּ מַלְאַךְ יְהֹוָה עַל עֵין הַמַּיִם בַּמִּדְבָּר עַל הָעַיִן בְּדֶרֶךְ שֽׁוּר:  ת  וְאַשְׁכְּחַהּ מַלְאָכָא דַיְיָ עַל עֵינָא דְמַיָא בְּמַדְבְּרָא עַל עֵינָא בְּאָרְחָא דְחַגְרָא:

{ח} וַיֹּאמַר הָגָר שִׁפְחַת שָׂרַי אֵֽי מִזֶּה בָאת וְאָנָה תֵלֵכִי וַתֹּאמֶר מִפְּנֵי שָׂרַי גְּבִרְתִּי אָֽנֹכִי בֹּרַֽחַת:  מ  וַיֹּאמַר הָגָר שִׁפְחַת שָׂרַי אֵֽי מִזֶּה בָאת וְאָנָה תֵלֵכִי וַתֹּאמֶר מִפְּנֵי שָׂרַי גְּבִרְתִּי אָֽנֹכִי בֹּרַֽחַת:  ת  וַאֲמַר הָגָר אַמְתָא דְשָׂרַי מְנָן אַתְּ אָתְיָא וּלְאָן אַתְּ אָזְלָא וַאֲמַרַת מִן קֳדָם שָׂרַי רִבָּנְתִּי אֲנָא עֲרָקָא (עֲרָקַת):

{ט} וַיֹּאמֶר לָהּ מַלְאַךְ יְהֹוָה שׁוּבִי אֶל גְּבִרְתֵּךְ וְהִתְעַנִּי תַּחַת יָדֶֽיהָ:  מ  וַיֹּאמֶר לָהּ מַלְאַךְ יְהֹוָה שׁוּבִי אֶל גְּבִרְתֵּךְ וְהִתְעַנִּי תַּחַת יָדֶֽיהָ:  ת  וַאֲמַר לַהּ מַלְאֲכָא דַיְיָ תּוּבִי לְוַת רִבּוֹנְתִּיךְ וְאִשְׁתַּעְבְּדִי תְּחוֹת יְדָהָא:

{י} וַיֹּאמֶר לָהּ מַלְאַךְ יְהֹוָה הַרְבָּה אַרְבֶּה אֶת זַרְעֵךְ וְלֹא יִסָּפֵר מֵרֹֽב:  מ  וַיֹּאמֶר לָהּ מַלְאַךְ יְהֹוָה הַרְבָּה אַרְבֶּה אֶת זַרְעֵךְ וְלֹא יִסָּפֵר מֵרֹֽב:  ת  וַאֲמַר לַהּ מַלְאֲכָא דַיְיָ אַסְגָאָה אַסְגֵי יָת בְּנָיְכִי וְלָא יִתִּמְנוּן מִסְגֵי:

{יא} וַיֹּאמֶר לָהּ מַלְאַךְ יְהֹוָה הִנָּךְ הָרָה וְיֹלַדְתְּ בֵּן וְקָרָאת שְׁמוֹ יִשְׁמָעֵאל כִּֽי שָׁמַע יְהֹוָה אֶל עָנְיֵֽךְ:  מ  וַיֹּאמֶר לָהּ מַלְאַךְ יְהֹוָה הִנָּךְ הָרָה וְיֹלַדְתְּ בֵּן וְקָרָאת שְׁמוֹ יִשְׁמָעֵאל כִּֽי שָׁמַע יְהֹוָה אֶל עָנְיֵֽךְ:  ת  וַאֲמַר לַהּ מַלְאֲכָא דַיְיָ הָא אַתְּ מְעַדְיָא וְתֵלְדִין בַּר וְתִקְרִין שְׁמֵהּ יִשְׁמָעֵאל אֲרֵי קַבִּיל יְיָ צְלוֹתִיךְ:

{יב} וְהוּא יִֽהְיֶה פֶּרֶא אָדָם יָדוֹ בַכֹּל וְיַד כֹּל בּוֹ וְעַל פְּנֵי כָל אֶחָיו יִשְׁכֹּֽן:  מ  וְהוּא יִֽהְיֶה פֶּרֶא אָדָם יָדוֹ בַכֹּל וְיַד כֹּל בּוֹ וְעַל פְּנֵי כָל אֶחָיו יִשְׁכֹּֽן:  ת  וְהוּא יְהֵא מָרוֹד בֶּאֱנָשָׁא הוּא יְהֵא צְרִיךְ לְכֹלָא וְאַף כָּל בְנֵי אֲנָשָׁא יְהוֹן צְרִיכִין לֵהּ וְעַל אַפֵּי כָל אֲחוֹהִי יִשְׁרֵי:

{יג} וַתִּקְרָא שֵׁם יְהֹוָה הַדֹּבֵר אֵלֶיהָ אַתָּה אֵל רֳאִי כִּי אָֽמְרָה הֲגַם הֲלֹם רָאִיתִי אַֽחֲרֵי רֹאִֽי:  מ  וַתִּקְרָא שֵׁם יְהֹוָה הַדֹּבֵר אֵלֶיהָ אַתָּה אֵל רֳאִי כִּי אָֽמְרָה הֲגַם הֲלֹם רָאִיתִי אַֽחֲרֵי רֹאִֽי:  ת  וְצַלִיאַת בִּשְׁמָא דַיְיָ דְמִתְמַלֵל עִמַהּ אֲמַרַת אַתְּ הוּא אֱלָהָא דְחָזֵי כֹלָא אֲרֵי אֲמַרַת הַבְרַם הָכָא (נ''א הַאַף אֲנָא) שָּׁרֵיתִי חַזְיָא בָּתַר דְאִתְגְלִי לִי:

{יד} עַל כֵּן קָרָא לַבְּאֵר בְּאֵר לַחַי רֹאִי הִנֵּה בֵין קָדֵשׁ וּבֵין בָּֽרֶד:  מ  עַל כֵּן קָרָא לַבְּאֵר בְּאֵר לַחַי רֹאִי הִנֵּה בֵין קָדֵשׁ וּבֵין בָּֽרֶד:  ת  עַל כֵּן קְרָא לִבְאֵרָא בְּאֵרָא דְמַלְאַךְ קַיָימָא אִתַּחֲזִי עֲלַהּ הָא הִיא בֵּין רְקָם וּבֵין חַגְרָא:

{טו} וַתֵּלֶד הָגָר לְאַבְרָם בֵּן וַיִּקְרָא אַבְרָם שֶׁם בְּנוֹ אֲשֶׁר יָֽלְדָה הָגָר יִשְׁמָעֵֽאל:  מ  וַתֵּלֶד הָגָר לְאַבְרָם בֵּן וַיִּקְרָא אַבְרָם שֶׁם בְּנוֹ אֲשֶׁר יָֽלְדָה הָגָר יִשְׁמָעֵֽאל:  ת  וִילִידַת הָגָר לְאַבְרָם בָּר וקְרָא אַבְרָם שׁוּם בְּרֵיהּ דִילִידַת הָגָר יִשְׁמָעֵאל:

{טז} וְאַבְרָם בֶּן שְׁמֹנִים שָׁנָה וְשֵׁשׁ שָׁנִים בְּלֶֽדֶת הָגָר אֶת יִשְׁמָעֵאל לְאַבְרָֽם:  מ  וְאַבְרָם בֶּן שְׁמֹנִים שָׁנָה וְשֵׁשׁ שָׁנִים בְּלֶֽדֶת הָגָר אֶת יִשְׁמָעֵאל לְאַבְרָֽם:  ת  וְאַבְרָם בַּר תְּמָנָן וְשִׁית שְׁנִין כַּד יְלִידַת הָגָר יָת יִשְׁמָעֵאל לְאַבְרָם: (ס)




בראשית פרק-יז

{א} וַיְהִי אַבְרָם בֶּן תִּשְׁעִים שָׁנָה וְתֵשַׁע שָׁנִים וַיֵּרָא יְהֹוָה אֶל אַבְרָם וַיֹּאמֶר אֵלָיו אֲנִי אֵל שַׁדַּי הִתְהַלֵּךְ לְפָנַי וֶֽהְיֵה תָמִֽים:  מ  וַיְהִי אַבְרָם בֶּן תִּשְׁעִים שָׁנָה וְתֵשַׁע שָׁנִים וַיֵּרָא יְהֹוָה אֶל אַבְרָם וַיֹּאמֶר אֵלָיו אֲנִי אֵל שַׁדַּי הִתְהַלֵּךְ לְפָנַי וֶֽהְיֵה תָמִֽים:  ת  וַהֲוָה אַבְרָם בַּר תִּשְׁעִין וּתְשַׁע שְׁנִין וְאִתְגְלִי יְיָ לְאַבְרָם וַאֲמַר לֵיהּ אֲנָא אֵל שַׁדַי פְּלַח קֳדָמַי וֶהֱוֵי שְׁלִים:

{ב} וְאֶתְּנָה בְרִיתִי בֵּינִי וּבֵינֶךָ וְאַרְבֶּה אֽוֹתְךָ בִּמְאֹד מְאֹֽד:  מ  וְאֶתְּנָה בְרִיתִי בֵּינִי וּבֵינֶךָ וְאַרְבֶּה אֽוֹתְךָ בִּמְאֹד מְאֹֽד:  ת  וְאֶתֵּן קְיָמִי בֵּין מֵימְרִי וּבֵינָךְ וְאַסְגֵי יָתָךְ לַחֲדָא לַחֲדָא:

{ג} וַיִּפֹּל אַבְרָם עַל פָּנָיו וַיְדַבֵּר אִתּוֹ אֱלֹהִים לֵאמֹֽר:  מ  וַיִּפֹּל אַבְרָם עַל פָּנָיו וַיְדַבֵּר אִתּוֹ אֱלֹהִים לֵאמֹֽר:  ת  וּנְפַל אַבְרָם עַל אַפּוֹהִי וּמַלֵיל עִמֵהּ יְיָ לְמֵימָר:

{ד} אֲנִי הִנֵּה בְרִיתִי אִתָּךְ וְהָיִיתָ לְאַב הֲמוֹן גּוֹיִֽם:  מ  אֲנִי הִנֵּה בְרִיתִי אִתָּךְ וְהָיִיתָ לְאַב הֲמוֹן גּוֹיִֽם:  ת  אֲנָא הָא (גְזַר) קְיָמִי עִמָךְ וּתְהֵי לְאַב סַגִי עַמְמִין:

{ה} וְלֹֽא יִקָּרֵא עוֹד אֶת שִׁמְךָ אַבְרָם וְהָיָה שִׁמְךָ אַבְרָהָם כִּי אַב הֲמוֹן גּוֹיִם נְתַתִּֽיךָ:  מ  וְלֹֽא יִקָּרֵא עוֹד אֶת שִׁמְךָ אַבְרָם וְהָיָה שִׁמְךָ אַבְרָהָם כִּי אַב הֲמוֹן גּוֹיִם נְתַתִּֽיךָ:  ת  וְלָא יִתְקְרֵי עוֹד יָת שְׁמָךְ אַבְרָם וִיהֵי שְׁמָךְ אַבְרָהָם אֲרֵי אַב סַגִי עַמְמִין יְהָבִתָךְ:

{ו} וְהִפְרֵתִי אֹֽתְךָ בִּמְאֹד מְאֹד וּנְתַתִּיךָ לְגוֹיִם וּמְלָכִים מִמְּךָ יֵצֵֽאוּ:  מ  וְהִפְרֵתִי אֹֽתְךָ בִּמְאֹד מְאֹד וּנְתַתִּיךָ לְגוֹיִם וּמְלָכִים מִמְּךָ יֵצֵֽאוּ:  ת  וְאַפִּישׁ יָתָךְ לַחֲדָא לַחֲדָא וְאֶתְּנִנָךְ לְכִנְשִׁין וּמַלְכִין דְשַׁלִיטִין בְּעַמְמַיָא מִנָךְ יִפְקוּן:

{ז}  שביעי  וַֽהֲקִֽמֹתִי אֶת בְּרִיתִי בֵּינִי וּבֵינֶךָ וּבֵין זַרְעֲךָ אַֽחֲרֶיךָ לְדֹֽרֹתָם לִבְרִית עוֹלָם לִֽהְיוֹת לְךָ לֵֽאלֹהִים וּלְזַרְעֲךָ אַֽחֲרֶֽיךָ:  מ  וַֽהֲקִֽמֹתִי אֶת בְּרִיתִי בֵּינִי וּבֵינֶךָ וּבֵין זַרְעֲךָ אַֽחֲרֶיךָ לְדֹֽרֹתָם לִבְרִית עוֹלָם לִֽהְיוֹת לְךָ לֵֽאלֹהִים וּלְזַרְעֲךָ אַֽחֲרֶֽיךָ:  ת  וַאַקֵים יָת קְיָמִי בֵּין מֵימְרִי וּבֵינָךְ וּבֵין בְּנָךְ בַּתְרָךְ לְדָרֵיהוֹן לִקְיָם עָלָם לְמֵהֲוֵי לָךְ לֵאלָהָא וְלִבְנָיךְ בַּתְרָךְ:

{ח} וְנָֽתַתִּי לְךָ וּלְזַרְעֲךָ אַֽחֲרֶיךָ אֵת אֶרֶץ מְגֻרֶיךָ אֵת כָּל אֶרֶץ כְּנַעַן לַֽאֲחֻזַּת עוֹלָם וְהָיִיתִי לָהֶם לֵֽאלֹהִֽים:  מ  וְנָֽתַתִּי לְךָ וּלְזַרְעֲךָ אַֽחֲרֶיךָ אֵת אֶרֶץ מְגֻרֶיךָ אֵת כָּל אֶרֶץ כְּנַעַן לַֽאֲחֻזַּת עוֹלָם וְהָיִיתִי לָהֶם לֵֽאלֹהִֽים:  ת  וְאֶתֵּן לָךְ וְלִבְנָיךְ בַּתְרָךְ יָת אַרְעָא תּוֹתָבוּתָךְ יָת כָּל אַרְעָא דִכְנַעַן לְאַחֲסָנַת עָלָם וְאֱהֶוֵי לְהוֹן לֵאלָהָא:

{ט} וַיֹּאמֶר אֱלֹהִים אֶל אַבְרָהָם וְאַתָּה אֶת בְּרִיתִי תִשְׁמֹר אַתָּה וְזַרְעֲךָ אַֽחֲרֶיךָ לְדֹֽרֹתָֽם:  מ  וַיֹּאמֶר אֱלֹהִים אֶל אַבְרָהָם וְאַתָּה אֶת בְּרִיתִי תִשְׁמֹר אַתָּה וְזַרְעֲךָ אַֽחֲרֶיךָ לְדֹֽרֹתָֽם:  ת  וַאֲמַר יְיָ לְאַבְרָהָם וְאַתְּ יָת קְיָמִי תִטָר אַתְּ וּבְנָךְ בַּתְרָךְ לְדָרֵיהוֹן:

{י} זֹאת בְּרִיתִי אֲשֶׁר תִּשְׁמְרוּ בֵּינִי וּבֵינֵיכֶם וּבֵין זַרְעֲךָ אַֽחֲרֶיךָ הִמּוֹל לָכֶם כָּל זָכָֽר:  מ  זֹאת בְּרִיתִי אֲשֶׁר תִּשְׁמְרוּ בֵּינִי וּבֵינֵיכֶם וּבֵין זַרְעֲךָ אַֽחֲרֶיךָ הִמּוֹל לָכֶם כָּל זָכָֽר:  ת  דָא קְיָמִי דִי תִטְרוּן בֵּין מֵימְרִי וּבֵינֵיכוֹן וּבֵין בְּנָיךְ בַּתְרָךְ מִגְזַר לְכוֹן כָּל דְכוּרָא:

{יא} וּנְמַלְתֶּם אֵת בְּשַׂר עָרְלַתְכֶם וְהָיָה לְאוֹת בְּרִית בֵּינִי וּבֵֽינֵיכֶֽם:  מ  וּנְמַלְתֶּם אֵת בְּשַׂר עָרְלַתְכֶם וְהָיָה לְאוֹת בְּרִית בֵּינִי וּבֵֽינֵיכֶֽם:  ת  וְתִגְזְרוּן יָת בִּשְׂרָא דְעָרְלָתְכוֹן וּתְהֵי (נ''י וּתְהֵא) לְאָת קְיָם בֵּין מֵימְרִי וּבֵינֵיכוֹן:

{יב} וּבֶן שְׁמֹנַת יָמִים יִמּוֹל לָכֶם כָּל זָכָר לְדֹרֹֽתֵיכֶם יְלִיד בָּיִת וּמִקְנַת כֶּסֶף מִכֹּל בֶּן נֵכָר אֲשֶׁר לֹא מִזַּרְעֲךָ הֽוּא:  מ  וּבֶן שְׁמֹנַת יָמִים יִמּוֹל לָכֶם כָּל זָכָר לְדֹרֹֽתֵיכֶם יְלִיד בָּיִת וּמִקְנַת כֶּסֶף מִכֹּל בֶּן נֵכָר אֲשֶׁר לֹא מִזַּרְעֲךָ הֽוּא:  ת  וּבַר תְּמַנְיָא יוֹמִין יִתְגְזַר (יִגְזַר) לְכוֹן כָּל דְכוּרָא לְדָרָתֵיכוֹן יְלִידֵי בֵּיתָא וּזְבִינֵי כַּסְפָּא מִכֹּל בַּר עַמְמִין דְלָא מִבְּנָךְ הוּא:

{יג} הִמּוֹל יִמּוֹל יְלִיד בֵּֽיתְךָ וּמִקְנַת כַּסְפֶּךָ וְהָֽיְתָה בְרִיתִי בִּבְשַׂרְכֶם לִבְרִית עוֹלָֽם:  מ  הִמּוֹל יִמּוֹל יְלִיד בֵּֽיתְךָ וּמִקְנַת כַּסְפֶּךָ וְהָֽיְתָה בְרִיתִי בִּבְשַׂרְכֶם לִבְרִית עוֹלָֽם:  ת  אִתְגַזָרָא יִתְגַזְרוּן (נ''י מִגְזַר יִגְזַר) יְלִיד בֵּיתָךְ וּזְבִין כַּסְפָּךְ וּתְהֵי קְיָמִי בְּבִשְׂרֵיכוֹן לִקְיָם עָלָם:

{יד} וְעָרֵל זָכָר אֲשֶׁר לֹֽא יִמּוֹל אֶת בְּשַׂר עָרְלָתוֹ וְנִכְרְתָה הַנֶּפֶשׁ הַהִוא מֵֽעַמֶּיהָ אֶת בְּרִיתִי הֵפַֽר:  מ  וְעָרֵל זָכָר אֲשֶׁר לֹֽא יִמּוֹל אֶת בְּשַׂר עָרְלָתוֹ וְנִכְרְתָה הַנֶּפֶשׁ הַהִוא מֵֽעַמֶּיהָ אֶת בְּרִיתִי הֵפַֽר:  ת  וְעָרֵל דְכוּרָא דִי לָא יִגְזַר יָת בְּשַׂר עָרְלָתֵהּ וְיִשְׁתֵּיצֵי אֱנָשָׁא הַהוּא מֵעַמֵּיהּ יָת קְיָמִי אַשְׁנִי: (ס)

{טו} וַיֹּאמֶר אֱלֹהִים אֶל אַבְרָהָם שָׂרַי אִשְׁתְּךָ לֹֽא תִקְרָא אֶת שְׁמָהּ שָׂרָי כִּי שָׂרָה שְׁמָֽהּ:  מ  וַיֹּאמֶר אֱלֹהִים אֶל אַבְרָהָם שָׂרַי אִשְׁתְּךָ לֹֽא תִקְרָא אֶת שְׁמָהּ שָׂרָי כִּי שָׂרָה שְׁמָֽהּ:  ת  וַאֲמַר יְיָ לְאַבְרָהָם שָׂרַי אִתְּתָךְ לָא תִקְרֵי יָת שְׁמַהּ שָׂרָי אֲרֵי שָׂרָה שְׁמַהּ:

{טז} וּבֵֽרַכְתִּי אֹתָהּ וְגַם נָתַתִּי מִמֶּנָּה לְךָ בֵּן וּבֵֽרַכְתִּיהָ וְהָֽיְתָה לְגוֹיִם מַלְכֵי עַמִּים מִמֶּנָּה יִֽהְיֽוּ:  מ  וּבֵֽרַכְתִּי אֹתָהּ וְגַם נָתַתִּי מִמֶּנָּה לְךָ בֵּן וּבֵֽרַכְתִּיהָ וְהָֽיְתָה לְגוֹיִם מַלְכֵי עַמִּים מִמֶּנָּה יִֽהְיֽוּ:  ת  וֶאֱבָרֵךְ יָתַהּ וְאַף אֶתֵּן מִנַהּ לָךְ בָּר וֶאֱבָרֲכִנַהּ וּתְהֵי לְכִנְשִׁין מַלְכִין דְשַׁלִּיטִין בְעַמְמַיָּא מִנַהּ יְהוֹן:

{יז} וַיִּפֹּל אַבְרָהָם עַל פָּנָיו וַיִּצְחָק וַיֹּאמֶר בְּלִבּוֹ הַלְּבֶן מֵאָֽה שָׁנָה יִוָּלֵד וְאִם שָׂרָה הֲבַת תִּשְׁעִים שָׁנָה תֵּלֵֽד:  מ  וַיִּפֹּל אַבְרָהָם עַל פָּנָיו וַיִּצְחָק וַיֹּאמֶר בְּלִבּוֹ הַלְּבֶן מֵאָֽה שָׁנָה יִוָּלֵד וְאִם שָׂרָה הֲבַת תִּשְׁעִים שָׁנָה תֵּלֵֽד:  ת  וּנְפַל אַבְרָהָם עַל אַפּוֹהִי וַחֲדִי וַאֲמַר בְּלִבֵּהּ הַלְּבַר מְאָה שְׁנִין יְהֵי וְלַד וְאִם שָׂרָה הֲבַת תִּשְׁעִין שְׁנִין תְּלִיד:

{יח} וַיֹּאמֶר אַבְרָהָם אֶל הָֽאֱלֹהִים לוּ יִשְׁמָעֵאל יִחְיֶה לְפָנֶֽיךָ:  מ  וַיֹּאמֶר אַבְרָהָם אֶל הָֽאֱלֹהִים לוּ יִשְׁמָעֵאל יִחְיֶה לְפָנֶֽיךָ:  ת  וַאֲמַר אַבְרָהָם קֳדָם יְיָ לְוַי יִשְׁמָעֵאל יִתְקַיָּם קֳדָמָךְ:

{יט} וַיֹּאמֶר אֱלֹהִים אֲבָל שָׂרָה אִשְׁתְּךָ יֹלֶדֶת לְךָ בֵּן וְקָרָאתָ אֶת שְׁמוֹ יִצְחָק וַֽהֲקִֽמֹתִי אֶת בְּרִיתִי אִתּוֹ לִבְרִית עוֹלָם לְזַרְעוֹ אַֽחֲרָֽיו:  מ  וַיֹּאמֶר אֱלֹהִים אֲבָל שָׂרָה אִשְׁתְּךָ יֹלֶדֶת לְךָ בֵּן וְקָרָאתָ אֶת שְׁמוֹ יִצְחָק וַֽהֲקִֽמֹתִי אֶת בְּרִיתִי אִתּוֹ לִבְרִית עוֹלָם לְזַרְעוֹ אַֽחֲרָֽיו:  ת  וַאֲמַר יְיָ בְּקוּשְׁטָא שָׂרָה אִתְּתָךְ תְּלִיד לָךְ בָּר וְתִקְרֵי יָת שְׁמֵהּ יִצְחָק וַאֲקִים יָת קְיָמִי עִמֵּהּ לִקְיָם עָלָם לִבְנוֹהִי בַּתְרוֹהִי:

{כ} וּלְיִשְׁמָעֵאל שְׁמַעְתִּיךָ הִנֵּה בֵּרַכְתִּי אֹתוֹ וְהִפְרֵיתִי אֹתוֹ וְהִרְבֵּיתִי אֹתוֹ בִּמְאֹד מְאֹד שְׁנֵים עָשָׂר נְשִׂיאִם יוֹלִיד וּנְתַתִּיו לְגוֹי גָּדֽוֹל:  מ  וּלְיִשְׁמָעֵאל שְׁמַעְתִּיךָ הִנֵּה בֵּרַכְתִּי אֹתוֹ וְהִפְרֵיתִי אֹתוֹ וְהִרְבֵּיתִי אֹתוֹ בִּמְאֹד מְאֹד שְׁנֵים עָשָׂר נְשִׂיאִם יוֹלִיד וּנְתַתִּיו לְגוֹי גָּדֽוֹל:  ת  וְעַל יִשְׁמָעֵאל קַבֵּלִית צְלוֹתָךְ הָא בָּרֵכִית יָתֵהּ וְאַפֵּישּׁ יָתֵהּ וְאַסְגֵּי יָתֵהּ לַחֲדָא לַחֲדָא תְּרֵין עֲשַׂר רַבְרְבַיָּא יְלִיד וְאֶתְּנִנֵּיהּ לְעַם סַגִּי:

{כא} וְאֶת בְּרִיתִי אָקִים אֶת יִצְחָק אֲשֶׁר תֵּלֵד לְךָ שָׂרָה לַמּוֹעֵד הַזֶּה בַּשָּׁנָה הָֽאַחֶֽרֶת:  מ  וְאֶת בְּרִיתִי אָקִים אֶת יִצְחָק אֲשֶׁר תֵּלֵד לְךָ שָׂרָה לַמּוֹעֵד הַזֶּה בַּשָּׁנָה הָֽאַחֶֽרֶת:  ת  וְיָת קְיָמִי אָקִים עִם יִצְחָק דִי תְלִיד לָךְ שָׂרָה לְזִמְנָא הָדֵין בְּשַׁתָּא אָחֳרַנְתָּא:

{כב} וַיְכַל לְדַבֵּר אִתּוֹ וַיַּעַל אֱלֹהִים מֵעַל אַבְרָהָֽם:  מ  וַיְכַל לְדַבֵּר אִתּוֹ וַיַּעַל אֱלֹהִים מֵעַל אַבְרָהָֽם:  ת  וְשֵׁיצֵי לְמַלָּלָא עִמֵהּ וְאִסְתַּלַּק יְקָרָא דַיְיָ מֵעִלָּווֹהִי דְּאַבְרָהָם:

{כג} וַיִּקַּח אַבְרָהָם אֶת יִשְׁמָעֵאל בְּנוֹ וְאֵת כָּל יְלִידֵי בֵיתוֹ וְאֵת כָּל מִקְנַת כַּסְפּוֹ כָּל זָכָר בְּאַנְשֵׁי בֵּית אַבְרָהָם וַיָּמָל אֶת בְּשַׂר עָרְלָתָם בְּעֶצֶם הַיּוֹם הַזֶּה כַּֽאֲשֶׁר דִּבֶּר אִתּוֹ אֱלֹהִֽים:  מ  וַיִּקַּח אַבְרָהָם אֶת יִשְׁמָעֵאל בְּנוֹ וְאֵת כָּל יְלִידֵי בֵיתוֹ וְאֵת כָּל מִקְנַת כַּסְפּוֹ כָּל זָכָר בְּאַנְשֵׁי בֵּית אַבְרָהָם וַיָּמָל אֶת בְּשַׂר עָרְלָתָם בְּעֶצֶם הַיּוֹם הַזֶּה כַּֽאֲשֶׁר דִּבֶּר אִתּוֹ אֱלֹהִֽים:  ת  וּדְבַר אַבְרָהָם יָת יִשְׁמָעֵאל בְּרֵהּ וְיָת כָּל יְלִידֵי בֵיתֵהּ וְיָת כָּל זְבִינֵי כַסְפֵּה כָּל דְּכוּרָא בַּאֲנָשֵׁי בֵּית אַבְרָהָם וּגְזַר יָת בִּשְׂרָא דְעָרְלָתְהוֹן בִּכְרַן יוֹמָא הָדֵין כְּמָא דִי מַלֵיל עִמֵּהּ יְיָ:

{כד}  מפטיר  וְאַבְרָהָם בֶּן תִּשְׁעִים וָתֵשַׁע שָׁנָה בְּהִמֹּלוֹ בְּשַׂר עָרְלָתֽוֹ:  מ  וְאַבְרָהָם בֶּן תִּשְׁעִים וָתֵשַׁע שָׁנָה בְּהִמֹּלוֹ בְּשַׂר עָרְלָתֽוֹ:  ת  וְאַבְרָהָם בַּר תִּשְׁעִין וּתְשַׁע שְׁנִין כַּד גְזַר בִּשְׂרָא דְעָרְלְתֵהּ:

{כה} וְיִשְׁמָעֵאל בְּנוֹ בֶּן שְׁלֹשׁ עֶשְׂרֵה שָׁנָה בְּהִמֹּלוֹ אֵת בְּשַׂר עָרְלָתֽוֹ:  מ  וְיִשְׁמָעֵאל בְּנוֹ בֶּן שְׁלֹשׁ עֶשְׂרֵה שָׁנָה בְּהִמֹּלוֹ אֵת בְּשַׂר עָרְלָתֽוֹ:  ת  וְיִשְׁמָעֵאל בְּרֵהּ בַּר תְּלַת עַשְׂרֵי שְׁנִין כַּד גְּזַר יַת בִּשְׂרָא דְעָרְלְתֵהּ:

{כו} בְּעֶצֶם הַיּוֹם הַזֶּה נִמּוֹל אַבְרָהָם וְיִשְׁמָעֵאל בְּנֽוֹ:  מ  בְּעֶצֶם הַיּוֹם הַזֶּה נִמּוֹל אַבְרָהָם וְיִשְׁמָעֵאל בְּנֽוֹ:  ת  בִּכְרַן יוֹמָא הָדֵין (גְּזַר) אִתְגְּזַר אַבְרָהָם וְיִשְׁמָעֵאל בְּרֵהּ:

{כז} וְכָל אַנְשֵׁי בֵיתוֹ יְלִיד בָּיִת וּמִקְנַת כֶּסֶף מֵאֵת בֶּן נֵכָר נִמֹּלוּ אִתּֽוֹ:  מ  וְכָל אַנְשֵׁי בֵיתוֹ יְלִיד בָּיִת וּמִקְנַת כֶּסֶף מֵאֵת בֶּן נֵכָר נִמֹּלוּ אִתּֽוֹ:  ת  וְכָל אֱנָשֵׁי בֵיתֵהּ יְלִידֵי בֵיתָא וּזְבִינֵי כַסְפָּא מִן בַּר עַמְמִין אִתְגְּזָרוּ (גְזָרוּ) עִמֵּהּ: (פפפ)
