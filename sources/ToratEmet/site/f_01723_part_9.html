פרשת כי תשא


{יא} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{יב} כִּ֣י תִשָּׂ֞א אֶת-רֹ֥אשׁ בְּנֵֽי-יִשְׂרָאֵל֘ לִפְקֻֽדֵיהֶם֒ וְנָ֨תְנ֜וּ אִ֣ישׁ כֹּ֧פֶר נַפְשׁ֛וֹ לַֽיהוָֹ֖ה בִּפְקֹ֣ד אֹתָ֑ם וְלֹא-יִֽהְיֶ֥ה בָהֶ֛ם נֶ֖גֶף בִּפְקֹ֥ד אֹתָֽם:  מ  כִּ֣י תִשָּׂ֞א אֶת-רֹ֥אשׁ בְּנֵֽי-יִשְׂרָאֵל֘ לִפְקֻֽדֵיהֶם֒ וְנָ֨תְנ֜וּ אִ֣ישׁ כֹּ֧פֶר נַפְשׁ֛וֹ לַֽיהוָֹ֖ה בִּפְקֹ֣ד אֹתָ֑ם וְלֹא-יִֽהְיֶ֥ה בָהֶ֛ם נֶ֖גֶף בִּפְקֹ֥ד אֹתָֽם:  ת  אֲרֵי תְקַבֵּל יָת חֻשְׁבַּן בְּנֵי יִשְׂרָאֵל לְמִנְיָנֵיהוֹן וְיִתְּנוּן גְבַר פֻּרְקַן נַפְשֵׁהּ קֳדָם יְיָ כַּד תִּמְנֵי יָתְהוֹן וְלָא יְהֵי בְהוֹן מוֹתָא כַּד תִּמְנֵי יָתְהוֹן:

{יג} זֶ֣ה | יִתְּנ֗וּ כָּל-הָֽעֹבֵר֙ עַל-הַפְּקֻדִ֔ים מַֽחֲצִ֥ית הַשֶּׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ עֶשְׂרִ֤ים גֵּרָה֙ הַשֶּׁ֔קֶל מַֽחֲצִ֣ית הַשֶּׁ֔קֶל תְּרוּמָ֖ה לַֽיהֹוָֽה:  מ  זֶ֣ה | יִתְּנ֗וּ כָּל-הָֽעֹבֵר֙ עַל-הַפְּקֻדִ֔ים מַֽחֲצִ֥ית הַשֶּׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ עֶשְׂרִ֤ים גֵּרָה֙ הַשֶּׁ֔קֶל מַֽחֲצִ֣ית הַשֶּׁ֔קֶל תְּרוּמָ֖ה לַֽיהֹוָֽה:  ת  דֵין יִתְּנוּן כָּל דְעָבַר עַל מִנְיָנַיָּא פַּלְגוּת סִלְעָא בְּסִלְעֵי קוּדְשָׁא עֶסְרִין מָעִין סִלְעָא פַּלְגוּת סִלְעָא אַפְרָשׁוּתָא קֳדָם יְיָ:

{יד} כֹּ֗ל הָֽעֹבֵר֙ עַל-הַפְּקֻדִ֔ים מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה יִתֵּ֖ן תְּרוּמַ֥ת יְהוָֹֽה:  מ  כֹּ֗ל הָֽעֹבֵר֙ עַל-הַפְּקֻדִ֔ים מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה יִתֵּ֖ן תְּרוּמַ֥ת יְהוָֹֽה:  ת  כֹּל דְעָבַר עַל מִנְיָנַיָא מִבַּר עֶסְרִין שְׁנִין וּלְעֵלָא יִתֵּן אַפְרָשׁוּתָא קֳדָם יְיָ:

{טו} הֶֽעָשִׁ֣יר לֹֽא-יַרְבֶּ֗ה וְהַדַּל֙ לֹ֣א יַמְעִ֔יט מִֽמַּֽחֲצִ֖ית הַשָּׁ֑קֶל לָתֵת֙ אֶת-תְּרוּמַ֣ת יְהֹוָ֔ה לְכַפֵּ֖ר עַל-נַפְשֹֽׁתֵיכֶֽם:  מ  הֶֽעָשִׁ֣יר לֹֽא-יַרְבֶּ֗ה וְהַדַּל֙ לֹ֣א יַמְעִ֔יט מִֽמַּֽחֲצִ֖ית הַשָּׁ֑קֶל לָתֵת֙ אֶת-תְּרוּמַ֣ת יְהֹוָ֔ה לְכַפֵּ֖ר עַל-נַפְשֹֽׁתֵיכֶֽם:  ת  דְעַתִּיר לָא יַסְגֵי וּדְמִסְכֵּן לָא יַזְעֵר מִפַּלְגוּת סִלְעָא לְמִתַּן יָת אַפְרָשׁוּתָא קֳדָם יְיָ לְכַפָּרָא עַל נַפְשָׁתֵיכוֹן:

{טז} וְלָֽקַחְתָּ֞ אֶת-כֶּ֣סֶף הַכִּפֻּרִ֗ים מֵאֵת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וְנָֽתַתָּ֣ אֹת֔וֹ עַל-עֲבֹדַ֖ת אֹ֣הֶל מוֹעֵ֑ד וְהָיָה֩ לִבְנֵ֨י יִשְׂרָאֵ֤ל לְזִכָּרוֹן֙ לִפְנֵ֣י יְהֹוָ֔ה לְכַפֵּ֖ר עַל-נַפְשֹֽׁתֵיכֶֽם:  מ  וְלָֽקַחְתָּ֞ אֶת-כֶּ֣סֶף הַכִּפֻּרִ֗ים מֵאֵת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וְנָֽתַתָּ֣ אֹת֔וֹ עַל-עֲבֹדַ֖ת אֹ֣הֶל מוֹעֵ֑ד וְהָיָה֩ לִבְנֵ֨י יִשְׂרָאֵ֤ל לְזִכָּרוֹן֙ לִפְנֵ֣י יְהֹוָ֔ה לְכַפֵּ֖ר עַל-נַפְשֹֽׁתֵיכֶֽם:  ת  וְתִסַב יָת כְּסַף כִּפּוּרַיָא מִן בְּנֵי יִשְׂרָאֵל וְתִתֵּן יָתֵהּ עַל פָּלְחַן מַשְׁכַּן זִמְנָא וִיהֵי לִבְנֵי יִשְׂרָאֵל לְדָכְרָנָא קֳדָם יְיָ לְכַפָּרָא עַל נַפְשָׁתֵיכוֹן: (פ)

{יז} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{יח} וְעָשִׂ֜יתָ כִּיּ֥וֹר נְחֹ֛שֶׁת וְכַנּ֥וֹ נְחֹ֖שֶׁת לְרָחְצָ֑ה וְנָֽתַתָּ֣ אֹת֗וֹ בֵּֽין-אֹ֤הֶל מוֹעֵד֙ וּבֵ֣ין הַמִּזְבֵּ֔חַ וְנָֽתַתָּ֥ שָׁ֖מָּה מָֽיִם:  מ  וְעָשִׂ֜יתָ כִּיּ֥וֹר נְחֹ֛שֶׁת וְכַנּ֥וֹ נְחֹ֖שֶׁת לְרָחְצָ֑ה וְנָֽתַתָּ֣ אֹת֗וֹ בֵּֽין-אֹ֤הֶל מוֹעֵד֙ וּבֵ֣ין הַמִּזְבֵּ֔חַ וְנָֽתַתָּ֥ שָׁ֖מָּה מָֽיִם:  ת  וְתַעְבֵּד כִּיוֹרָא דִנְחָשָׁא וּבְסִיסֵהּ נְחָשָׁא לְקִדוּשׁ וְתִתֵּן יָתֵהּ בֵּין מַשְׁכַּן זִמְנָא וּבֵין מַדְבְּחָא וְתִתֵּן תַּמָן מַיָא:

{יט} וְרָֽחֲצ֛וּ אַֽהֲרֹ֥ן וּבָנָ֖יו מִמֶּ֑נּוּ אֶת-יְדֵיהֶ֖ם וְאֶת-רַגְלֵיהֶֽם:  מ  וְרָֽחֲצ֛וּ אַֽהֲרֹ֥ן וּבָנָ֖יו מִמֶּ֑נּוּ אֶת-יְדֵיהֶ֖ם וְאֶת-רַגְלֵיהֶֽם:  ת  וִיקַדְשׁוּן אַהֲרֹן וּבְנוֹהִי מִנֵהּ יָת יְדֵיהוֹן וְיָת רַגְלֵיהוֹן:

{כ} בְּבֹאָ֞ם אֶל-אֹ֧הֶל מוֹעֵ֛ד יִרְחֲצוּ-מַ֖יִם וְלֹ֣א יָמֻ֑תוּ א֣וֹ בְגִשְׁתָּ֤ם אֶל-הַמִּזְבֵּ֨חַ֙ לְשָׁרֵ֔ת לְהַקְטִ֥יר אִשֶּׁ֖ה לַֽיהוָֹֽה:  מ  בְּבֹאָ֞ם אֶל-אֹ֧הֶל מוֹעֵ֛ד יִרְחֲצוּ-מַ֖יִם וְלֹ֣א יָמֻ֑תוּ א֣וֹ בְגִשְׁתָּ֤ם אֶל-הַמִּזְבֵּ֨חַ֙ לְשָׁרֵ֔ת לְהַקְטִ֥יר אִשֶּׁ֖ה לַֽיהוָֹֽה:  ת  בְּמֵיעַלְהוֹן לְמַשְׁכַּן זִמְנָא יְקַדְשׁוּן מַיָא וְלָא יְמוּתוּן אוֹ בְמִקְרָבְהוֹן לְמַדְבְּחָא לְשַׁמָשָׁא לְאַסָקָא קֻרְבָּנָא קֳדָם יְיָ:

{כא} וְרָֽחֲצ֛וּ יְדֵיהֶ֥ם וְרַגְלֵיהֶ֖ם וְלֹ֣א יָמֻ֑תוּ וְהָֽיְתָ֨ה לָהֶ֧ם חָק-עוֹלָ֛ם ל֥וֹ וּלְזַרְע֖וֹ לְדֹֽרֹתָֽם:  מ  וְרָֽחֲצ֛וּ יְדֵיהֶ֥ם וְרַגְלֵיהֶ֖ם וְלֹ֣א יָמֻ֑תוּ וְהָֽיְתָ֨ה לָהֶ֧ם חָק-עוֹלָ֛ם ל֥וֹ וּלְזַרְע֖וֹ לְדֹֽרֹתָֽם:  ת  וִיקַדְשׁוּן יְדֵיהוֹן וְרַגְלֵיהוֹן וְלָא יְמוּתוּן וּתְהֵי לְהוֹן קְיָם עָלָם לֵהּ וְלִבְנוֹהִי לְדָרֵיהוֹן: (פ)

{כב} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{כג} וְאַתָּ֣ה קַח-לְךָ֘ בְּשָׂמִ֣ים רֹאשׁ֒ מָר-דְּרוֹר֙ חֲמֵ֣שׁ מֵא֔וֹת וְקִנְּמָן-בֶּ֥שֶׂם מַֽחֲצִית֖וֹ חֲמִשִּׁ֣ים וּמָאתָ֑יִם וּקְנֵה-בֹ֖שֶׂם חֲמִשִּׁ֥ים וּמָאתָֽיִם:  מ  וְאַתָּ֣ה קַח-לְךָ֘ בְּשָׂמִ֣ים רֹאשׁ֒ מָר-דְּרוֹר֙ חֲמֵ֣שׁ מֵא֔וֹת וְקִנְּמָן-בֶּ֥שֶׂם מַֽחֲצִית֖וֹ חֲמִשִּׁ֣ים וּמָאתָ֑יִם וּקְנֵה-בֹ֖שֶׂם חֲמִשִּׁ֥ים וּמָאתָֽיִם:  ת  וְאַתְּ סַב לָךְ בּוּסְמִין רֵישָׁא מֵירָא דַכְיָא מַתְקַל חֲמֵשׁ מְאָה וְקִנְמָן בֶּשֶׂם פַּלְגוּתֵהּ מַתְקַל מָאתָן וְחַמְשִׁין וּקְנֵה בוּסְמָא מַתְקַל מָאתָן וְחַמְשִׁין:

{כד} וְקִדָּ֕ה חֲמֵ֥שׁ מֵא֖וֹת בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ וְשֶׁ֥מֶן זַ֖יִת הִֽין:  מ  וְקִדָּ֕ה חֲמֵ֥שׁ מֵא֖וֹת בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ וְשֶׁ֥מֶן זַ֖יִת הִֽין:  ת  וּקְצִיעֵתָא מַתְקַל חֲמֵשׁ מְאָה בְּסִלְעֵי קוּדְשָׁא וּמְשַׁח זֵיתָא מְלֵא הִינָא:

{כה} וְעָשִׂ֣יתָ אֹת֗וֹ שֶׁ֚מֶן מִשְׁחַת-קֹ֔דֶשׁ רֹ֥קַח מִרְקַ֖חַת מַֽעֲשֵׂ֣ה רֹקֵ֑חַ שֶׁ֥מֶן מִשְׁחַת-קֹ֖דֶשׁ יִֽהְיֶֽה:  מ  וְעָשִׂ֣יתָ אֹת֗וֹ שֶׁ֚מֶן מִשְׁחַת-קֹ֔דֶשׁ רֹ֥קַח מִרְקַ֖חַת מַֽעֲשֵׂ֣ה רֹקֵ֑חַ שֶׁ֥מֶן מִשְׁחַת-קֹ֖דֶשׁ יִֽהְיֶֽה:  ת  וְתַעְבֵּד יָתֵהּ מְשַׁח רְבוּת קוּדְשָׁא בּוֹסֶם מְבַסָם עוֹבַד בּוּסְמָנוּ מְשַׁח רְבוּת קוּדְשָׁא יְהֵא:

{כו} וּמָֽשַׁחְתָּ֥ ב֖וֹ אֶת-אֹ֣הֶל מוֹעֵ֑ד וְאֵ֖ת אֲר֥וֹן הָֽעֵדֻֽת:  מ  וּמָֽשַׁחְתָּ֥ ב֖וֹ אֶת-אֹ֣הֶל מוֹעֵ֑ד וְאֵ֖ת אֲר֥וֹן הָֽעֵדֻֽת:  ת  וּתְרַבִּי בֵהּ יָת מַשְׁכַּן זִמְנָא וְיָת אֲרוֹנָא דְסַהֲדוּתָא:

{כז} וְאֶת-הַשֻּׁלְחָן֙ וְאֶת-כָּל-כֵּלָ֔יו וְאֶת-הַמְּנֹרָ֖ה וְאֶת-כֵּלֶ֑יהָ וְאֵ֖ת מִזְבַּ֥ח הַקְּטֹֽרֶת:  מ  וְאֶת-הַשֻּׁלְחָן֙ וְאֶת-כָּל-כֵּלָ֔יו וְאֶת-הַמְּנֹרָ֖ה וְאֶת-כֵּלֶ֑יהָ וְאֵ֖ת מִזְבַּ֥ח הַקְּטֹֽרֶת:  ת  וְיָת פָּתוֹרָא וְיָת כָּל מָנוֹהִי וְיָת מְנַרְתָּא וְיָת מָנָהָא וְיָת מַדְבְּחָא דִקְטֹרֶת בּוּסְמַיָא:

{כח} וְאֶת-מִזְבַּ֥ח הָֽעֹלָ֖ה וְאֶת-כָּל-כֵּלָ֑יו וְאֶת-הַכִּיֹּ֖ר וְאֶת-כַּנּֽוֹ:  מ  וְאֶת-מִזְבַּ֥ח הָֽעֹלָ֖ה וְאֶת-כָּל-כֵּלָ֑יו וְאֶת-הַכִּיֹּ֖ר וְאֶת-כַּנּֽוֹ:  ת  וְיָת מַדְבַּח דַעֲלָתָא וְיָת כָּל מָנוֹהִי וְיָת כִּיוֹרָא וְיָת בְּסִיסֵהּ:

{כט} וְקִדַּשְׁתָּ֣ אֹתָ֔ם וְהָי֖וּ קֹ֣דֶשׁ קָֽדָשִׁ֑ים כָּל-הַנֹּגֵ֥עַ בָּהֶ֖ם יִקְדָּֽשׁ:  מ  וְקִדַּשְׁתָּ֣ אֹתָ֔ם וְהָי֖וּ קֹ֣דֶשׁ קָֽדָשִׁ֑ים כָּל-הַנֹּגֵ֥עַ בָּהֶ֖ם יִקְדָּֽשׁ:  ת  וּתְקַדֵשׁ יָתְהוֹן וִיהוֹן קֹדֶשׁ קוּדְשַׁיָא כָּל דְיִקְרַב בְּהוֹן יִתְקַדָשׁ:

{ל} וְאֶת-אַֽהֲרֹ֥ן וְאֶת-בָּנָ֖יו תִּמְשָׁ֑ח וְקִדַּשְׁתָּ֥ אֹתָ֖ם לְכַהֵ֥ן לִֽי:  מ  וְאֶת-אַֽהֲרֹ֥ן וְאֶת-בָּנָ֖יו תִּמְשָׁ֑ח וְקִדַּשְׁתָּ֥ אֹתָ֖ם לְכַהֵ֥ן לִֽי:  ת  וְיָת אַהֲרֹן וְיָת בְּנוֹהִי תְּרַבִּי וּתְקַדֵשׁ יָתְהוֹן לְשַׁמָשָׁא קֳדָמַי:

{לא} וְאֶל-בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר שֶׁ֠מֶן מִשְׁחַת-קֹ֨דֶשׁ יִהְיֶ֥ה זֶ֛ה לִ֖י לְדֹרֹֽתֵיכֶֽם:  מ  וְאֶל-בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר שֶׁ֠מֶן מִשְׁחַת-קֹ֨דֶשׁ יִהְיֶ֥ה זֶ֛ה לִ֖י לְדֹרֹֽתֵיכֶֽם:  ת  וְעִם בְּנֵי יִשְׂרָאֵל תְּמַלֵל לְמֵימָר מְשַׁח רְבוּת קוּדְשָׁא יְהֵי דֵין קֳדָמַי לְדָרֵיכוֹן:

{לב} עַל-בְּשַׂ֤ר אָדָם֙ לֹ֣א יִיסָ֔ךְ וּ֨בְמַתְכֻּנְתּ֔וֹ לֹ֥א תַֽעֲשׂ֖וּ כָּמֹ֑הוּ קֹ֣דֶשׁ ה֔וּא קֹ֖דֶשׁ יִהְיֶ֥ה לָכֶֽם:  מ  עַל-בְּשַׂ֤ר אָדָם֙ לֹ֣א יִיסָ֔ךְ וּ֨בְמַתְכֻּנְתּ֔וֹ לֹ֥א תַֽעֲשׂ֖וּ כָּמֹ֑הוּ קֹ֣דֶשׁ ה֔וּא קֹ֖דֶשׁ יִהְיֶ֥ה לָכֶֽם:  ת  עַל בִּסְרָא דַאֲנָשָׁא לָא יִתְנַסַךְ וּבִדְמוּתֵהּ לָא תַעְבְּדוּן כְּוָתֵהּ קוּדְשָׁא הוּא קוּדְשָׁא יְהֵי לְכוֹן:

{לג} אִ֚ישׁ אֲשֶׁ֣ר יִרְקַ֣ח כָּמֹ֔הוּ וַֽאֲשֶׁ֥ר יִתֵּ֛ן מִמֶּ֖נּוּ עַל-זָ֑ר וְנִכְרַ֖ת מֵֽעַמָּֽיו:  מ  אִ֚ישׁ אֲשֶׁ֣ר יִרְקַ֣ח כָּמֹ֔הוּ וַֽאֲשֶׁ֥ר יִתֵּ֛ן מִמֶּ֖נּוּ עַל-זָ֑ר וְנִכְרַ֖ת מֵֽעַמָּֽיו:  ת  גְבַר דִי יְבַסֵם דִכְוָתֵהּ וְדִי יִתֵּן מִנֵהּ עַל חִלוֹנָי וְיִשְׁתֵּצֵי מֵעַמֵהּ: (ס)

{לד} וַיֹּ֩אמֶר֩ יְהֹוָ֨ה אֶל-מֹשֶׁ֜ה קַח-לְךָ֣ סַמִּ֗ים נָטָ֤ף | וּשְׁחֵ֨לֶת֙ וְחֶלְבְּנָ֔ה סַמִּ֖ים וּלְבֹנָ֣ה זַכָּ֑ה בַּ֥ד בְּבַ֖ד יִֽהְיֶֽה:  מ  וַיֹּ֩אמֶר֩ יְהֹוָ֨ה אֶל-מֹשֶׁ֜ה קַח-לְךָ֣ סַמִּ֗ים נָטָ֤ף | וּשְׁחֵ֨לֶת֙ וְחֶלְבְּנָ֔ה סַמִּ֖ים וּלְבֹנָ֣ה זַכָּ֑ה בַּ֥ד בְּבַ֖ד יִֽהְיֶֽה:  ת  וַאֲמַר יְיָ לְמשֶׁה סַב לָךְ בּוּסְמִין נְטוֹפָא וְטוּפְרָא וְחֶלְבְּנֵתָא בּוּסְמִין וּלְבוּנְתָא דָכִיתָא מַתְקַל בְּמַתְקַל יְהֵי:

{לה} וְעָשִׂ֤יתָ אֹתָהּ֙ קְטֹ֔רֶת רֹ֖קַח מַֽעֲשֵׂ֣ה רוֹקֵ֑חַ מְמֻלָּ֖ח טָה֥וֹר קֹֽדֶשׁ:  מ  וְעָשִׂ֤יתָ אֹתָהּ֙ קְטֹ֔רֶת רֹ֖קַח מַֽעֲשֵׂ֣ה רוֹקֵ֑חַ מְמֻלָּ֖ח טָה֥וֹר קֹֽדֶשׁ:  ת  וְתַעְבֵּד יָתַהּ קְטוֹרֶת בּוּסְמִין בּוֹסֶם עוֹבַד בּוּסְמָנוּ מְעָרַב דְכֵי קוּדְשָׁא:

{לו} וְשָֽׁחַקְתָּ֣ מִמֶּ֘נָּה֘ הָדֵק֒ וְנָֽתַתָּ֨ה מִמֶּ֜נָּה לִפְנֵ֤י הָֽעֵדֻת֙ בְּאֹ֣הֶל מוֹעֵ֔ד אֲשֶׁ֛ר אִוָּעֵ֥ד לְךָ֖ שָׁ֑מָּה קֹ֥דֶשׁ קָֽדָשִׁ֖ים תִּהְיֶ֥ה לָכֶֽם:  מ  וְשָֽׁחַקְתָּ֣ מִמֶּ֘נָּה֘ הָדֵק֒ וְנָֽתַתָּ֨ה מִמֶּ֜נָּה לִפְנֵ֤י הָֽעֵדֻת֙ בְּאֹ֣הֶל מוֹעֵ֔ד אֲשֶׁ֛ר אִוָּעֵ֥ד לְךָ֖ שָׁ֑מָּה קֹ֥דֶשׁ קָֽדָשִׁ֖ים תִּהְיֶ֥ה לָכֶֽם:  ת  וְתִשְׁחוֹק מִנַהּ וְתַדִיק וְתִתֵּן מִנַהּ קֳדָם סַהֲדוּתָא בְּמַשְׁכַּן זִמְנָא דִי אֱזַמֵן מֵימְרִי לָךְ תַּמָן קֹדֶשׁ קוּדְשִׁין תְּהֵי לְכוֹן:

{לז} וְהַקְּטֹ֨רֶת֙ אֲשֶׁ֣ר תַּֽעֲשֶׂ֔ה בְּמַ֨תְכֻּנְתָּ֔הּ לֹ֥א תַֽעֲשׂ֖וּ לָכֶ֑ם קֹ֛דֶשׁ תִּהְיֶ֥ה לְךָ֖ לַֽיהוָֹֽה:  מ  וְהַקְּטֹ֨רֶת֙ אֲשֶׁ֣ר תַּֽעֲשֶׂ֔ה בְּמַ֨תְכֻּנְתָּ֔הּ לֹ֥א תַֽעֲשׂ֖וּ לָכֶ֑ם קֹ֛דֶשׁ תִּהְיֶ֥ה לְךָ֖ לַֽיהוָֹֽה:  ת  וּקְטֹרֶת בּוּסְמִין דִי תַעְבֵּד בִּדְמוּתַהּ לָא תַעְבְּדוּן לְכוֹן קוּדְשָׁא תְּהֵי לָךְ קֳדָם יְיָ:

{לח} אִ֛ישׁ אֲשֶׁר-יַֽעֲשֶׂ֥ה כָמ֖וֹהָ לְהָרִ֣יחַ בָּ֑הּ וְנִכְרַ֖ת מֵֽעַמָּֽיו:  מ  אִ֛ישׁ אֲשֶׁר-יַֽעֲשֶׂ֥ה כָמ֖וֹהָ לְהָרִ֣יחַ בָּ֑הּ וְנִכְרַ֖ת מֵֽעַמָּֽיו:  ת  גְבַר דִיַעְבֵּד דִכְוָתַהּ לַאֲרָחָא בַהּ וְיִשְׁתֵּצֵי מֵעַמֵּהּ: (ס)




שמות פרק-לא

{א} וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהוָֹ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{ב} רְאֵ֖ה קָרָ֣אתִי בְשֵׁ֑ם בְּצַלְאֵ֛ל בֶּן-אוּרִ֥י בֶן-ח֖וּר לְמַטֵּ֥ה יְהוּדָֽה:  מ  רְאֵ֖ה קָרָ֣אתִי בְשֵׁ֑ם בְּצַלְאֵ֛ל בֶּן-אוּרִ֥י בֶן-ח֖וּר לְמַטֵּ֥ה יְהוּדָֽה:  ת  חֲזֵי דְרַבֵּיתִי בְשׁוּם בְּצַלְאֵל בַּר אוּרִי בַּר חוּר לְשִׁבְטָא דִיהוּדָה:

{ג} וָֽאֲמַלֵּ֥א אֹת֖וֹ ר֣וּחַ אֱלֹהִ֑ים בְּחָכְמָ֛ה וּבִתְבוּנָ֥ה וּבְדַ֖עַת וּבְכָל-מְלָאכָֽה:  מ  וָֽאֲמַלֵּ֥א אֹת֖וֹ ר֣וּחַ אֱלֹהִ֑ים בְּחָכְמָ֛ה וּבִתְבוּנָ֥ה וּבְדַ֖עַת וּבְכָל-מְלָאכָֽה:  ת  וְאַשְׁלֵמִית עִמֵהּ רוּחַ מִן קֳדָם יְיָ בְּחָכְמְתָא וּבְסוּכְלְתָנוּ וּבְמַדַע וּבְכָל עִבִידָא:

{ד} לַחְשֹׁ֖ב מַֽחֲשָׁבֹ֑ת לַֽעֲשׂ֛וֹת בַּזָּהָ֥ב וּבַכֶּ֖סֶף וּבַנְּחֹֽשֶׁת:  מ  לַחְשֹׁ֖ב מַֽחֲשָׁבֹ֑ת לַֽעֲשׂ֛וֹת בַּזָּהָ֥ב וּבַכֶּ֖סֶף וּבַנְּחֹֽשֶׁת:  ת  לְאַלָפָא אוּמַנְוָן לְמֶעְבַּד בְּדַהֲבָא וּבְכַסְפָּא וּבִנְחָשָׁא:

{ה} וּבַֽחֲרֹ֥שֶׁת אֶ֛בֶן לְמַלֹּ֖את וּבַֽחֲרֹ֣שֶׁת עֵ֑ץ לַֽעֲשׂ֖וֹת בְּכָל-מְלָאכָֽה:  מ  וּבַֽחֲרֹ֥שֶׁת אֶ֛בֶן לְמַלֹּ֖את וּבַֽחֲרֹ֣שֶׁת עֵ֑ץ לַֽעֲשׂ֖וֹת בְּכָל-מְלָאכָֽה:  ת  וּבְאָמָנוּת אֶבֶן טָבָא לְאַשְׁלָמָא וּבְנַגָרוּת אָעָא לְמֶעְבַּד בְּכָל עִבִידָא:

{ו} וַֽאֲנִ֞י הִנֵּ֧ה נָתַ֣תִּי אִתּ֗וֹ אֵ֣ת אָֽהֳלִיאָ֞ב בֶּן-אֲחִֽיסָמָךְ֙ לְמַטֵּה-דָ֔ן וּבְלֵ֥ב כָּל-חֲכַם-לֵ֖ב נָתַ֣תִּי חָכְמָ֑ה וְעָשׂ֕וּ אֵ֖ת כָּל-אֲשֶׁ֥ר צִוִּיתִֽךָ:  מ  וַֽאֲנִ֞י הִנֵּ֧ה נָתַ֣תִּי אִתּ֗וֹ אֵ֣ת אָֽהֳלִיאָ֞ב בֶּן-אֲחִֽיסָמָךְ֙ לְמַטֵּה-דָ֔ן וּבְלֵ֥ב כָּל-חֲכַם-לֵ֖ב נָתַ֣תִּי חָכְמָ֑ה וְעָשׂ֕וּ אֵ֖ת כָּל-אֲשֶׁ֥ר צִוִּיתִֽךָ:  ת  וַאֲנָא הָא יְהָבִית עִמֵהּ יָת אָהֳלִיאָב בַּר אֲחִיסָמָךְ לְשִׁבְטָא דְדָן וּבְלִבָּא כָל חַכִּימֵי לִבָּא יְהָבִית חָכִמְתָא וְיַעְבִּדוּן יָת כָּל דִי פַקֵדִתָּךְ:

{ז} אֵ֣ת | אֹ֣הֶל מוֹעֵ֗ד וְאֶת-הָֽאָרֹן֙ לָֽעֵדֻ֔ת וְאֶת-הַכַּפֹּ֖רֶת אֲשֶׁ֣ר עָלָ֑יו וְאֵ֖ת כָּל-כְּלֵ֥י הָאֹֽהֶל:  מ  אֵ֣ת | אֹ֣הֶל מוֹעֵ֗ד וְאֶת-הָֽאָרֹן֙ לָֽעֵדֻ֔ת וְאֶת-הַכַּפֹּ֖רֶת אֲשֶׁ֣ר עָלָ֑יו וְאֵ֖ת כָּל-כְּלֵ֥י הָאֹֽהֶל:  ת  יָת מַשְׁכַּן זִמְנָא וְיָת אֲרוֹנָא לְסַהֲדוּתָא וְיָת כַּפֻּרְתָּא דִי עֲלוֹהִי וְיָת כָּל מָנֵי מַשְׁכְּנָא:

{ח} וְאֶת-הַשֻּׁלְחָן֙ וְאֶת-כֵּלָ֔יו וְאֶת-הַמְּנֹרָ֥ה הַטְּהֹרָ֖ה וְאֶת-כָּל-כֵּלֶ֑יהָ וְאֵ֖ת מִזְבַּ֥ח הַקְּטֹֽרֶת:  מ  וְאֶת-הַשֻּׁלְחָן֙ וְאֶת-כֵּלָ֔יו וְאֶת-הַמְּנֹרָ֥ה הַטְּהֹרָ֖ה וְאֶת-כָּל-כֵּלֶ֑יהָ וְאֵ֖ת מִזְבַּ֥ח הַקְּטֹֽרֶת:  ת  וְיָת פָּתוֹרָא וְיָת מָנוֹהִי וְיָת מְנַרְתָּא דָכִיתָא וְיָת כָּל מָנָהָא וְיָת מַדְבְּחָא דִקְטּוֹרֶת בּוּסְמַיָא:

{ט} וְאֶת-מִזְבַּ֥ח הָֽעֹלָ֖ה וְאֶת-כָּל-כֵּלָ֑יו וְאֶת-הַכִּיּ֖וֹר וְאֶת-כַּנּֽוֹ:  מ  וְאֶת-מִזְבַּ֥ח הָֽעֹלָ֖ה וְאֶת-כָּל-כֵּלָ֑יו וְאֶת-הַכִּיּ֖וֹר וְאֶת-כַּנּֽוֹ:  ת  וְיָת מַדְבְּחָא דַעֲלָתָא וְיָת כָּל מָנוֹהִי וְיָת כִּיוֹרָא וְיָת בְּסִיסֵהּ:

{י} וְאֵ֖ת בִּגְדֵ֣י הַשְּׂרָ֑ד וְאֶת-בִּגְדֵ֤י הַקֹּ֨דֶשׁ֙ לְאַֽהֲרֹ֣ן הַכֹּהֵ֔ן וְאֶת-בִּגְדֵ֥י בָנָ֖יו לְכַהֵֽן:  מ  וְאֵ֖ת בִּגְדֵ֣י הַשְּׂרָ֑ד וְאֶת-בִּגְדֵ֤י הַקֹּ֨דֶשׁ֙ לְאַֽהֲרֹ֣ן הַכֹּהֵ֔ן וְאֶת-בִּגְדֵ֥י בָנָ֖יו לְכַהֵֽן:  ת  וְיָת לְבוּשֵׁי שִׁמוּשָׁא וְיָת לְבוּשֵׁי קוּדְשָׁא לְאַהֲרֹן כַּהֲנָא וְיָת לְבוּשֵׁי בְנוֹהִי לְשַׁמָשָׁא:

{יא} וְאֵ֨ת שֶׁ֧מֶן הַמִּשְׁחָ֛ה וְאֶת-קְטֹ֥רֶת הַסַּמִּ֖ים לַקֹּ֑דֶשׁ כְּכֹ֥ל אֲשֶׁר-צִוִּיתִ֖ךָ יַֽעֲשֽׂוּ:  מ  וְאֵ֨ת שֶׁ֧מֶן הַמִּשְׁחָ֛ה וְאֶת-קְטֹ֥רֶת הַסַּמִּ֖ים לַקֹּ֑דֶשׁ כְּכֹ֥ל אֲשֶׁר-צִוִּיתִ֖ךָ יַֽעֲשֽׂוּ:  ת  וְיָת מִשְׁחָא דִרְבוּתָא וְיָת קְטוֹרֶת בּוּסְמַיָא לְקוּדְשָׁא כְּכֹל דִי פַקֵדִתָּךְ יַעְבְּדוּן: (פ)

{יב} וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וַאֲמַר יְיָ לְמשֶׁה לְמֵימָר:

{יג} וְאַתָּ֞ה דַּבֵּ֨ר אֶל-בְּנֵ֤י יִשְׂרָאֵל֙ לֵאמֹ֔ר אַ֥ךְ אֶת-שַׁבְּתֹתַ֖י תִּשְׁמֹ֑רוּ כִּי֩ א֨וֹת הִ֜וא בֵּינִ֤י וּבֵֽינֵיכֶם֙ לְדֹרֹ֣תֵיכֶ֔ם לָדַ֕עַת כִּ֛י אֲנִ֥י יְהוָֹ֖ה מְקַדִּשְׁכֶֽם:  מ  וְאַתָּ֞ה דַּבֵּ֨ר אֶל-בְּנֵ֤י יִשְׂרָאֵל֙ לֵאמֹ֔ר אַ֥ךְ אֶת-שַׁבְּתֹתַ֖י תִּשְׁמֹ֑רוּ כִּי֩ א֨וֹת הִ֜וא בֵּינִ֤י וּבֵֽינֵיכֶם֙ לְדֹרֹ֣תֵיכֶ֔ם לָדַ֕עַת כִּ֛י אֲנִ֥י יְהוָֹ֖ה מְקַדִּשְׁכֶֽם:  ת  וְאַתְּ מַלֵל עִם בְּנֵי יִשְׂרָאֵל לְמֵימַר בְּרַם יָת יוֹמֵי שַׁבַּיָא דִי לִי תִּטְּרוּן אֲרֵי אָת הִיא בֵּין מֵימְרִי וּבֵינֵיכוֹן לְדָרֵיכוֹן לְמִדַע אֲרֵי אֲנָא יְיָ מְקַדִשְׁכוֹן:

{יד} וּשְׁמַרְתֶּם֙ אֶת-הַשַּׁבָּ֔ת כִּ֛י קֹ֥דֶשׁ הִ֖וא לָכֶ֑ם מְחַֽלְלֶ֨יהָ֙ מ֣וֹת יוּמָ֔ת כִּ֗י כָּל-הָֽעֹשֶׂ֥ה בָהּ֙ מְלָאכָ֔ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מִקֶּ֥רֶב עַמֶּֽיהָ:  מ  וּשְׁמַרְתֶּם֙ אֶת-הַשַּׁבָּ֔ת כִּ֛י קֹ֥דֶשׁ הִ֖וא לָכֶ֑ם מְחַֽלְלֶ֨יהָ֙ מ֣וֹת יוּמָ֔ת כִּ֗י כָּל-הָֽעֹשֶׂ֥ה בָהּ֙ מְלָאכָ֔ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מִקֶּ֥רֶב עַמֶּֽיהָ:  ת  וְתִטְּרוּן יָת שַׁבְּתָא אֲרֵי קוּדְשָׁא הִיא לְכוֹן מְחַלְלִנַהּ אִתְקְטָלָא יִתְקְטֵל אֲרֵי כָּל דְיַעְבֶּד בַּהּ עִבִידָא וְיִשְׁתֵּצֵי אֲנָשָׁא הַהוּא מִגוֹ עַמֵּהּ:

{טו} שֵׁ֣שֶׁת יָמִים֘ יֵֽעָשֶׂ֣ה מְלָאכָה֒ וּבַיּ֣וֹם הַשְּׁבִיעִ֗י שַׁבַּ֧ת שַׁבָּת֛וֹן קֹ֖דֶשׁ לַֽיהוָֹ֑ה כָּל-הָֽעֹשֶׂ֧ה מְלָאכָ֛ה בְּי֥וֹם הַשַּׁבָּ֖ת מ֥וֹת יוּמָֽת:  מ  שֵׁ֣שֶׁת יָמִים֘ יֵֽעָשֶׂ֣ה מְלָאכָה֒ וּבַיּ֣וֹם הַשְּׁבִיעִ֗י שַׁבַּ֧ת שַׁבָּת֛וֹן קֹ֖דֶשׁ לַֽיהוָֹ֑ה כָּל-הָֽעֹשֶׂ֧ה מְלָאכָ֛ה בְּי֥וֹם הַשַּׁבָּ֖ת מ֥וֹת יוּמָֽת:  ת  שִׁתָּא יוֹמִין יִתְעֲבֵד עִבִידָא וּבְיוֹמָא שְׁבִיעָאָה שְׁבַת שַׁבְּתָא קוּדְשָׁא קֳדָם יְיָ כָּל דְיַעְבֵּד עִבִדְתָּא בְּיוֹמָא דְשַׁבְּתָא אִתְקְטָלָא יִתְקְטֵל:

{טז} וְשָֽׁמְר֥וּ בְנֵֽי-יִשְׂרָאֵ֖ל אֶת-הַשַּׁבָּ֑ת לַֽעֲשׂ֧וֹת אֶת-הַשַּׁבָּ֛ת לְדֹֽרֹתָ֖ם בְּרִ֥ית עוֹלָֽם:  מ  וְשָֽׁמְר֥וּ בְנֵֽי-יִשְׂרָאֵ֖ל אֶת-הַשַּׁבָּ֑ת לַֽעֲשׂ֧וֹת אֶת-הַשַּׁבָּ֛ת לְדֹֽרֹתָ֖ם בְּרִ֥ית עוֹלָֽם:  ת  וְיִטְרוּן בְּנֵי יִשְׂרָאֵל יָת שַׁבְּתָא לְמֶעְבַּד יָת שַׁבְּתָא לְדָרֵיהוֹן קְיַם עָלָם:

{יז} בֵּינִ֗י וּבֵין֙ בְּנֵ֣י יִשְׂרָאֵ֔ל א֥וֹת הִ֖וא לְעֹלָ֑ם כִּי-שֵׁ֣שֶׁת יָמִ֗ים עָשָׂ֤ה יְהוָֹה֙ אֶת-הַשָּׁמַ֣יִם וְאֶת-הָאָ֔רֶץ וּבַיּוֹם֙ הַשְּׁבִיעִ֔י שָׁבַ֖ת וַיִּנָּפַֽשׁ:  מ  בֵּינִ֗י וּבֵין֙ בְּנֵ֣י יִשְׂרָאֵ֔ל א֥וֹת הִ֖וא לְעֹלָ֑ם כִּי-שֵׁ֣שֶׁת יָמִ֗ים עָשָׂ֤ה יְהוָֹה֙ אֶת-הַשָּׁמַ֣יִם וְאֶת-הָאָ֔רֶץ וּבַיּוֹם֙ הַשְּׁבִיעִ֔י שָׁבַ֖ת וַיִּנָּפַֽשׁ:  ת  בֵּין מֵימְרִי וּבֵין בְּנֵי יִשְׂרָאֵל אָת הִיא לְעָלָם אֲרֵי שִׁתָּא יוֹמִין עֲבַד יְיָ יָת שְׁמַיָא וְיָת אַרְעָא וּבְיוֹמָא שְׁבִיעָאָה שְׁבַת וְנַח: (ס)

{יח}  שני  וַיִּתֵּ֣ן אֶל-מֹשֶׁ֗ה כְּכַלֹּתוֹ֙ לְדַבֵּ֤ר אִתּוֹ֙ בְּהַ֣ר סִינַ֔י שְׁנֵ֖י לֻחֹ֣ת הָֽעֵדֻ֑ת לֻחֹ֣ת אֶ֔בֶן כְּתֻבִ֖ים בְּאֶצְבַּ֥ע אֱלֹהִֽים:  מ  וַיִּתֵּ֣ן אֶל-מֹשֶׁ֗ה כְּכַלֹּתוֹ֙ לְדַבֵּ֤ר אִתּוֹ֙ בְּהַ֣ר סִינַ֔י שְׁנֵ֖י לֻחֹ֣ת הָֽעֵדֻ֑ת לֻחֹ֣ת אֶ֔בֶן כְּתֻבִ֖ים בְּאֶצְבַּ֥ע אֱלֹהִֽים:  ת  וִיהַב לְמשֶׁה כַּד שֵׁיצֵי לְמַלָלָא עִמֵהּ בְּטוּרָא דְסִינָי תְּרֵין לוּחֵי סַהֲדוּתָא לוּחֵי אַבְנָא כְּתִיבִין בְּאֶצְבְּעָא דַיְיָ:




שמות פרק-לב

{א} וַיַּ֣רְא הָעָ֔ם כִּֽי-בֹשֵׁ֥שׁ מֹשֶׁ֖ה לָרֶ֣דֶת מִן-הָהָ֑ר וַיִּקָּהֵ֨ל הָעָ֜ם עַֽל-אַֽהֲרֹ֗ן וַיֹּֽאמְר֤וּ אֵלָיו֙ ק֣וּם | עֲשֵׂה-לָ֣נוּ אֱלֹהִ֗ים אֲשֶׁ֤ר יֵֽלְכוּ֙ לְפָנֵ֔ינוּ כִּי-זֶ֣ה | מֹשֶׁ֣ה הָאִ֗ישׁ אֲשֶׁ֤ר הֶֽעֱלָ֨נוּ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לֹ֥א יָדַ֖עְנוּ מֶה-הָ֥יָה לֽוֹ:  מ  וַיַּ֣רְא הָעָ֔ם כִּֽי-בֹשֵׁ֥שׁ מֹשֶׁ֖ה לָרֶ֣דֶת מִן-הָהָ֑ר וַיִּקָּהֵ֨ל הָעָ֜ם עַֽל-אַֽהֲרֹ֗ן וַיֹּֽאמְר֤וּ אֵלָיו֙ ק֣וּם | עֲשֵׂה-לָ֣נוּ אֱלֹהִ֗ים אֲשֶׁ֤ר יֵֽלְכוּ֙ לְפָנֵ֔ינוּ כִּי-זֶ֣ה | מֹשֶׁ֣ה הָאִ֗ישׁ אֲשֶׁ֤ר הֶֽעֱלָ֨נוּ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לֹ֥א יָדַ֖עְנוּ מֶה-הָ֥יָה לֽוֹ:  ת  וַחֲזָא עַמָא אֲרֵי אוֹחַר משֶׁה לְמֵחַת מִן טוּרָא וְאִתְכְּנֵשׁ עַמָא עַל אַהֲרֹן וַאֲמָרוּ לֵהּ קוּם עֲבֵד לָנָא דַחֲלָן דִיְהָכוּן קֳדָמָנָא אֲרֵי דֵין משֶׁה גַבְרָא דִי אַסְקָנָא מֵאַרְעָא דְמִצְרַיִם לָא יְדַעְנָא מָא הֲוָה לֵהּ:

{ב} וַיֹּ֤אמֶר אֲלֵהֶם֙ אַֽהֲרֹ֔ן פָּֽרְקוּ֙ נִזְמֵ֣י הַזָּהָ֔ב אֲשֶׁר֙ בְּאָזְנֵ֣י נְשֵׁיכֶ֔ם בְּנֵיכֶ֖ם וּבְנֹֽתֵיכֶ֑ם וְהָבִ֖יאוּ אֵלָֽי:  מ  וַיֹּ֤אמֶר אֲלֵהֶם֙ אַֽהֲרֹ֔ן פָּֽרְקוּ֙ נִזְמֵ֣י הַזָּהָ֔ב אֲשֶׁר֙ בְּאָזְנֵ֣י נְשֵׁיכֶ֔ם בְּנֵיכֶ֖ם וּבְנֹֽתֵיכֶ֑ם וְהָבִ֖יאוּ אֵלָֽי:  ת  וַאֲמַר לְהוֹן אַהֲרֹן פְּרִיקוּ קָדָשֵׁי דְדַהֲבָא דִי בְאוּדְנֵי נְשֵׁיכוֹן בְּנֵיכוֹן וּבְנָתֵיכוֹן וְאַיְתִיוּ לְוָתִי:

{ג} וַיִּֽתְפָּֽרְקוּ֙ כָּל-הָעָ֔ם אֶת-נִזְמֵ֥י הַזָּהָ֖ב אֲשֶׁ֣ר בְּאָזְנֵיהֶ֑ם וַיָּבִ֖יאוּ אֶֽל-אַֽהֲרֹֽן:  מ  וַיִּֽתְפָּֽרְקוּ֙ כָּל-הָעָ֔ם אֶת-נִזְמֵ֥י הַזָּהָ֖ב אֲשֶׁ֣ר בְּאָזְנֵיהֶ֑ם וַיָּבִ֖יאוּ אֶֽל-אַֽהֲרֹֽן:  ת  וְאִתְפְּרִיקוּ כָּל עַמָא יָת קָדָשֵׁי דְדַהֲבָא דִי בְאוּדְנֵיהוֹן וְאַיְתִיוּ לְאַהֲרֹן:

{ד} וַיִּקַּ֣ח מִיָּדָ֗ם וַיָּ֤צַר אֹתוֹ֙ בַּחֶ֔רֶט וַֽיַּֽעֲשֵׂ֖הוּ עֵ֣גֶל מַסֵּכָ֑ה וַיֹּ֣אמְר֔וּ אֵ֤לֶּה אֱלֹהֶ֨יךָ֙ יִשְׂרָאֵ֔ל אֲשֶׁ֥ר הֶֽעֱל֖וּךָ מֵאֶ֥רֶץ מִצְרָֽיִם:  מ  וַיִּקַּ֣ח מִיָּדָ֗ם וַיָּ֤צַר אֹתוֹ֙ בַּחֶ֔רֶט וַֽיַּֽעֲשֵׂ֖הוּ עֵ֣גֶל מַסֵּכָ֑ה וַיֹּ֣אמְר֔וּ אֵ֤לֶּה אֱלֹהֶ֨יךָ֙ יִשְׂרָאֵ֔ל אֲשֶׁ֥ר הֶֽעֱל֖וּךָ מֵאֶ֥רֶץ מִצְרָֽיִם:  ת  וּנְסִיב מִידֵיהוֹן וְצַר יָתֵהּ בְּזִיפָא וְעָבְדֵהּ עֵגֶל מַתְּכָא וַאֲמָרוּ אִלֵין דַחַלְתָּךְ יִשְׂרָאֵל דִי אַסְקוּךְ מֵאַרְעָא דְמִצְרָיִם:

{ה} וַיַּ֣רְא אַֽהֲרֹ֔ן וַיִּ֥בֶן מִזְבֵּ֖חַ לְפָנָ֑יו וַיִּקְרָ֤א אַֽהֲרֹן֙ וַיֹּאמַ֔ר חַ֥ג לַֽיהוָֹ֖ה מָחָֽר:  מ  וַיַּ֣רְא אַֽהֲרֹ֔ן וַיִּ֥בֶן מִזְבֵּ֖חַ לְפָנָ֑יו וַיִּקְרָ֤א אַֽהֲרֹן֙ וַיֹּאמַ֔ר חַ֥ג לַֽיהוָֹ֖ה מָחָֽר:  ת  וַחֲזָא אַהֲרֹן וּבְנָא מַדְבְּחָא קֳדָמוֹהִי וּקְרָא אַהֲרֹן וַאֲמַר חַגָא קֳדָם יְיָ מְחָר:

{ו} וַיַּשְׁכִּ֨ימוּ֙ מִֽמָּ֣חֳרָ֔ת וַיַּֽעֲל֣וּ עֹלֹ֔ת וַיַּגִּ֖שׁוּ שְׁלָמִ֑ים וַיֵּ֤שֶׁב הָעָם֙ לֶֽאֱכֹ֣ל וְשָׁת֔וֹ וַיָּקֻ֖מוּ לְצַחֵֽק:  מ  וַיַּשְׁכִּ֨ימוּ֙ מִֽמָּ֣חֳרָ֔ת וַיַּֽעֲל֣וּ עֹלֹ֔ת וַיַּגִּ֖שׁוּ שְׁלָמִ֑ים וַיֵּ֤שֶׁב הָעָם֙ לֶֽאֱכֹ֣ל וְשָׁת֔וֹ וַיָּקֻ֖מוּ לְצַחֵֽק:  ת  וְאַקְדִימוּ בְּיוֹמָא דְבַתְרוֹהִי וְאַסִיקוּ עֲלָוָן וְקָרִיבוּ נִכְסִין וְאַסְחַר עַמָא לְמֵיכַל וּלְמִשְׁתֵּי וְקָמוּ לְחַיָכָא: (פ)

{ז} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֑ה לֶךְ-רֵ֕ד כִּ֚י שִׁחֵ֣ת עַמְּךָ֔ אֲשֶׁ֥ר הֶֽעֱלֵ֖יתָ מֵאֶ֥רֶץ מִצְרָֽיִם:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֑ה לֶךְ-רֵ֕ד כִּ֚י שִׁחֵ֣ת עַמְּךָ֔ אֲשֶׁ֥ר הֶֽעֱלֵ֖יתָ מֵאֶ֥רֶץ מִצְרָֽיִם:  ת  וּמַלִיל יְיָ עִם משֶׁה אִזֵל חוּת אֲרֵי חַבִּיל עַמָּךְ דִי אַסֶקְתָּא מֵאַרְעָא דְמִצְרָיִם:

{ח} סָ֣רוּ מַהֵ֗ר מִן-הַדֶּ֨רֶךְ֙ אֲשֶׁ֣ר צִוִּיתִ֔ם עָשׂ֣וּ לָהֶ֔ם עֵ֖גֶל מַסֵּכָ֑ה וַיִּשְׁתַּֽחֲווּ-לוֹ֙ וַיִּזְבְּחוּ-ל֔וֹ וַיֹּ֣אמְר֔וּ אֵ֤לֶּה אֱלֹהֶ֨יךָ֙ יִשְׂרָאֵ֔ל אֲשֶׁ֥ר הֶֽעֱל֖וּךָ מֵאֶ֥רֶץ מִצְרָֽיִם:  מ  סָ֣רוּ מַהֵ֗ר מִן-הַדֶּ֨רֶךְ֙ אֲשֶׁ֣ר צִוִּיתִ֔ם עָשׂ֣וּ לָהֶ֔ם עֵ֖גֶל מַסֵּכָ֑ה וַיִּשְׁתַּֽחֲווּ-לוֹ֙ וַיִּזְבְּחוּ-ל֔וֹ וַיֹּ֣אמְר֔וּ אֵ֤לֶּה אֱלֹהֶ֨יךָ֙ יִשְׂרָאֵ֔ל אֲשֶׁ֥ר הֶֽעֱל֖וּךָ מֵאֶ֥רֶץ מִצְרָֽיִם:  ת  סָטוּ בִּפְרִיעַ מִן אוֹרְחָא דִי פַקֶדְתִּנוּן עֲבָדוּ לְהוֹן עֵגֶל מַתְּכָא וּסְגִידוּ לֵהּ וְדַבָּחוּ לֵהּ וַאֲמָרוּ אִלֵין דַחַלְתָּךְ יִשְׂרָאֵל דִי אַסְקוּךְ מֵאַרְעָא דְמִצְרָיִם:

{ט} וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֑ה רָאִ֨יתִי֙ אֶת-הָעָ֣ם הַזֶּ֔ה וְהִנֵּ֥ה עַם-קְשֵׁה-עֹ֖רֶף הֽוּא:  מ  וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֑ה רָאִ֨יתִי֙ אֶת-הָעָ֣ם הַזֶּ֔ה וְהִנֵּ֥ה עַם-קְשֵׁה-עֹ֖רֶף הֽוּא:  ת  וַאֲמַר יְיָ לְמשֶׁה גְלִי קֳדָמַי עַמָא הָדֵין וְהָא עַמָא קְשֵׁי קְדַל הוּא:

{י} וְעַתָּה֙ הַנִּ֣יחָה לִּ֔י וְיִֽחַר-אַפִּ֥י בָהֶ֖ם וַֽאֲכַלֵּ֑ם וְאֶֽעֱשֶׂ֥ה אֽוֹתְךָ֖ לְג֥וֹי גָּדֽוֹל:  מ  וְעַתָּה֙ הַנִּ֣יחָה לִּ֔י וְיִֽחַר-אַפִּ֥י בָהֶ֖ם וַֽאֲכַלֵּ֑ם וְאֶֽעֱשֶׂ֥ה אֽוֹתְךָ֖ לְג֥וֹי גָּדֽוֹל:  ת  וּכְעַן הַנַּח בָּעוּתָךְ מִן קֳדָמַי וְיִתְקֵף רוּגְזִי בְהוֹן וְאֵישֵׁיצִנוּן וְאֶעְבֵּד יָתָךְ לְעַם סַגִי:

{יא} וַיְחַ֣ל מֹשֶׁ֔ה אֶת-פְּנֵ֖י יְהוָֹ֣ה אֱלֹהָ֑יו וַיֹּ֗אמֶר לָמָ֤ה יְהוָֹה֙ יֶֽחֱרֶ֤ה אַפְּךָ֙ בְּעַמֶּ֔ךָ אֲשֶׁ֤ר הוֹצֵ֨אתָ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם בְּכֹ֥חַ גָּד֖וֹל וּבְיָ֥ד חֲזָקָֽה:  מ  וַיְחַ֣ל מֹשֶׁ֔ה אֶת-פְּנֵ֖י יְהוָֹ֣ה אֱלֹהָ֑יו וַיֹּ֗אמֶר לָמָ֤ה יְהוָֹה֙ יֶֽחֱרֶ֤ה אַפְּךָ֙ בְּעַמֶּ֔ךָ אֲשֶׁ֤ר הוֹצֵ֨אתָ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם בְּכֹ֥חַ גָּד֖וֹל וּבְיָ֥ד חֲזָקָֽה:  ת  וְצַלִי משֶׁה קֳדָם יְיָ אֱלָהֵהּ וַאֲמַר לְמָא יְיָ יִתְקֵף רוּגְזָךְ בְּעַמָךְ דִי אַפֶּקְתָּא מֵאַרְעָא דְמִצְרַיִם בְּחֵיל רַב וּבִידָא תַקִיפָא:

{יב} לָ֩מָּה֩ יֹֽאמְר֨וּ מִצְרַ֜יִם לֵאמֹ֗ר בְּרָעָ֤ה הֽוֹצִיאָם֙ לַֽהֲרֹ֤ג אֹתָם֙ בֶּֽהָרִ֔ים וּ֨לְכַלֹּתָ֔ם מֵעַ֖ל פְּנֵ֣י הָֽאֲדָמָ֑ה שׁ֚וּב מֵֽחֲר֣וֹן אַפֶּ֔ךָ וְהִנָּחֵ֥ם עַל-הָֽרָעָ֖ה לְעַמֶּֽךָ:  מ  לָ֩מָּה֩ יֹֽאמְר֨וּ מִצְרַ֜יִם לֵאמֹ֗ר בְּרָעָ֤ה הֽוֹצִיאָם֙ לַֽהֲרֹ֤ג אֹתָם֙ בֶּֽהָרִ֔ים וּ֨לְכַלֹּתָ֔ם מֵעַ֖ל פְּנֵ֣י הָֽאֲדָמָ֑ה שׁ֚וּב מֵֽחֲר֣וֹן אַפֶּ֔ךָ וְהִנָּחֵ֥ם עַל-הָֽרָעָ֖ה לְעַמֶּֽךָ:  ת  לְמָא יֵימְרוּן מִצְרָאֵי לְמֵימַר בְּבִישְׁתָּא אַפֶּקְתִּנוּן לְקַטָלָא יָתְהוֹן בֵינֵי טוּרַיָא וּלְשֵׁיצָיוּתְהוֹן מֵעַל אַפֵּי אַרְעָא תּוּב מִתְּקוֹף רוּגְזָךְ וְאָתֵיב מִן בִּשָׁתָא דְמַלֶלְתָּא לְמֶעְבַּד לְעַמָךְ:

{יג} זְכֹ֡ר לְאַבְרָהָם֩ לְיִצְחָ֨ק וּֽלְיִשְׂרָאֵ֜ל עֲבָדֶ֗יךָ אֲשֶׁ֨ר נִשְׁבַּ֣עְתָּ לָהֶם֘ בָּךְ֒ וַתְּדַבֵּ֣ר אֲלֵהֶ֔ם אַרְבֶּה֙ אֶֽת-זַרְעֲכֶ֔ם כְּכֽוֹכְבֵ֖י הַשָּׁמָ֑יִם וְכָל-הָאָ֨רֶץ הַזֹּ֜את אֲשֶׁ֣ר אָמַ֗רְתִּי אֶתֵּן֙ לְזַֽרְעֲכֶ֔ם וְנָֽחֲל֖וּ לְעֹלָֽם:  מ  זְכֹ֡ר לְאַבְרָהָם֩ לְיִצְחָ֨ק וּֽלְיִשְׂרָאֵ֜ל עֲבָדֶ֗יךָ אֲשֶׁ֨ר נִשְׁבַּ֣עְתָּ לָהֶם֘ בָּךְ֒ וַתְּדַבֵּ֣ר אֲלֵהֶ֔ם אַרְבֶּה֙ אֶֽת-זַרְעֲכֶ֔ם כְּכֽוֹכְבֵ֖י הַשָּׁמָ֑יִם וְכָל-הָאָ֨רֶץ הַזֹּ֜את אֲשֶׁ֣ר אָמַ֗רְתִּי אֶתֵּן֙ לְזַֽרְעֲכֶ֔ם וְנָֽחֲל֖וּ לְעֹלָֽם:  ת  אִדְכַּר לְאַבְרָהָם לְיִצְחָק וּלְיִשְׂרָאֵל עַבְדָיךְ דִי קַיֶמְתָּא לְהוֹן בְּמֵימְרָךְ וּמַלֶלְתָּא עִמְהוֹן אַסְגֵי יָת בְּנֵיכוֹן כְּכוֹכְבֵי שְׁמַיָא וְכָל אַרְעָא הָדָא דִי אֲמָרִית אֶתֵּן לִבְנֵיכוֹן וְיַחְסְנוּן לַעֲלָם:

{יד} וַיִּנָּ֖חֶם יְהוָֹ֑ה עַל-הָ֣רָעָ֔ה אֲשֶׁ֥ר דִּבֶּ֖ר לַֽעֲשׂ֥וֹת לְעַמּֽוֹ:  מ  וַיִּנָּ֖חֶם יְהוָֹ֑ה עַל-הָ֣רָעָ֔ה אֲשֶׁ֥ר דִּבֶּ֖ר לַֽעֲשׂ֥וֹת לְעַמּֽוֹ:  ת  וְתַב יְיָ מִן בִּישְׁתָא דִי מַלִיל לְמֶעְבַּד לְעַמֵהּ: (פ)

{טו} וַיִּ֜פֶן וַיֵּ֤רֶד מֹשֶׁה֙ מִן-הָהָ֔ר וּשְׁנֵ֛י לֻחֹ֥ת הָֽעֵדֻ֖ת בְּיָד֑וֹ לֻחֹ֗ת כְּתֻבִים֙ מִשְּׁנֵ֣י עֶבְרֵיהֶ֔ם מִזֶּ֥ה וּמִזֶּ֖ה הֵ֥ם כְּתֻבִֽים:  מ  וַיִּ֜פֶן וַיֵּ֤רֶד מֹשֶׁה֙ מִן-הָהָ֔ר וּשְׁנֵ֛י לֻחֹ֥ת הָֽעֵדֻ֖ת בְּיָד֑וֹ לֻחֹ֗ת כְּתֻבִים֙ מִשְּׁנֵ֣י עֶבְרֵיהֶ֔ם מִזֶּ֥ה וּמִזֶּ֖ה הֵ֥ם כְּתֻבִֽים:  ת  וְאִתְפְּנֵי וּנְחַת משֶׁה מִן טוּרָא וּתְרֵין לוּחֵי סַהֲדוּתָא בִּידֵהּ לוּחֵי כְּתִיבִין מִתְּרֵין עִבְרֵיהוֹן מִכָּא וּמִכָּא אִנוּן כְּתִיבִין:

{טז} וְהַ֨לֻּחֹ֔ת מַֽעֲשֵׂ֥ה אֱלֹהִ֖ים הֵ֑מָּה וְהַמִּכְתָּ֗ב מִכְתַּ֤ב אֱלֹהִים֙ ה֔וּא חָר֖וּת עַל-הַלֻּחֹֽת:  מ  וְהַ֨לֻּחֹ֔ת מַֽעֲשֵׂ֥ה אֱלֹהִ֖ים הֵ֑מָּה וְהַמִּכְתָּ֗ב מִכְתַּ֤ב אֱלֹהִים֙ ה֔וּא חָר֖וּת עַל-הַלֻּחֹֽת:  ת  וְלוּחַיָא עוֹבָדָא דַיְיָ אִנוּן וּכְתָבָא כְּתָבָא דַיְיָ הוּא מְפָרַשׁ עַל לוּחַיָא:

{יז} וַיִּשְׁמַ֧ע יְהוֹשֻׁ֛עַ אֶת-ק֥וֹל הָעָ֖ם בְּרֵעֹ֑ה וַיֹּ֨אמֶר֙ אֶל-מֹשֶׁ֔ה ק֥וֹל מִלְחָמָ֖ה בַּֽמַּֽחֲנֶֽה:  מ  וַיִּשְׁמַ֧ע יְהוֹשֻׁ֛עַ אֶת-ק֥וֹל הָעָ֖ם בְּרֵעֹ֑ה וַיֹּ֨אמֶר֙ אֶל-מֹשֶׁ֔ה ק֥וֹל מִלְחָמָ֖ה בַּֽמַּֽחֲנֶֽה:  ת  וּשְׁמַע יְהוֹשֻׁעַ יָת קַל עַמָא כַּד מְיַבְּבִן וַאֲמַר לְמשֶׁה קַל קְרָבָא בְּמַשְׁרִיתָא:

{יח} וַיֹּ֗אמֶר אֵ֥ין קוֹל֙ עֲנ֣וֹת גְּבוּרָ֔ה וְאֵ֥ין ק֖וֹל עֲנ֣וֹת חֲלוּשָׁ֑ה ק֣וֹל עַנּ֔וֹת אָֽנֹכִ֖י שֹׁמֵֽעַ:  מ  וַיֹּ֗אמֶר אֵ֥ין קוֹל֙ עֲנ֣וֹת גְּבוּרָ֔ה וְאֵ֥ין ק֖וֹל עֲנ֣וֹת חֲלוּשָׁ֑ה ק֣וֹל עַנּ֔וֹת אָֽנֹכִ֖י שֹׁמֵֽעַ:  ת  וַאֲמַר לָא קַל גִבָּרִין דְנַצְחִין בִּקְרָבָא וְאַף לָא קַל חַלָשִׁין דְמִתַּבְּרִין קַל דִמְחַיְכִין אֲנָא שְׁמָע:

{יט} וַיְהִ֗י כַּֽאֲשֶׁ֤ר קָרַב֙ אֶל-הַֽמַּֽחֲנֶ֔ה וַיַּ֥רְא אֶת-הָעֵ֖גֶל וּמְחֹלֹ֑ת וַיִּֽחַר-אַ֣ף מֹשֶׁ֗ה וַיַּשְׁלֵ֤ךְ (מידו) מִיָּדָיו֙ אֶת-הַלֻּחֹ֔ת וַיְשַׁבֵּ֥ר אֹתָ֖ם תַּ֥חַת הָהָֽר:  מ  וַיְהִ֗י כַּֽאֲשֶׁ֤ר קָרַב֙ אֶל-הַֽמַּֽחֲנֶ֔ה וַיַּ֥רְא אֶת-הָעֵ֖גֶל וּמְחֹלֹ֑ת וַיִּֽחַר-אַ֣ף מֹשֶׁ֗ה וַיַּשְׁלֵ֤ךְ (מידו) מִיָּדָיו֙ אֶת-הַלֻּחֹ֔ת וַיְשַׁבֵּ֥ר אֹתָ֖ם תַּ֥חַת הָהָֽר:  ת  וַהֲוָה כַּד קָרִיב לְמַשְׁרִיתָא וַחֲזָא יָת עֶגְלָא וְחִנְגִין וּתְקֵיף רוּגְזָא דְמשֶׁה וּרְמָא מִידוֹהִי יָת לוּחַיָא וְתַבַּר יָתְהוֹן בְּשִׁפּוֹלֵי טּוּרָא:

{כ} וַיִּקַּ֞ח אֶת-הָעֵ֨גֶל אֲשֶׁ֤ר עָשׂוּ֙ וַיִּשְׂרֹ֣ף בָּאֵ֔שׁ וַיִּטְחַ֖ן עַ֣ד אֲשֶׁר-דָּ֑ק וַיִּ֨זֶר֙ עַל-פְּנֵ֣י הַמַּ֔יִם וַיַּ֖שְׁקְ אֶת-בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וַיִּקַּ֞ח אֶת-הָעֵ֨גֶל אֲשֶׁ֤ר עָשׂוּ֙ וַיִּשְׂרֹ֣ף בָּאֵ֔שׁ וַיִּטְחַ֖ן עַ֣ד אֲשֶׁר-דָּ֑ק וַיִּ֨זֶר֙ עַל-פְּנֵ֣י הַמַּ֔יִם וַיַּ֖שְׁקְ אֶת-בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וּנְסֵיב יָת עֶגְלָא דִי עֲבָדוּ וְאוֹקֵד בְּנוּרָא וְשַׁף עַד דַהֲוָה דַקִיק וּזְרָא עַל אַפֵּי מַיָא וְאַשְׁקֵי יָת בְּנֵי יִשְׂרָאֵל:

{כא} וַיֹּ֤אמֶר מֹשֶׁה֙ אֶֽל-אַֽהֲרֹ֔ן מֶה-עָשָׂ֥ה לְךָ֖ הָעָ֣ם הַזֶּ֑ה כִּֽי-הֵבֵ֥אתָ עָלָ֖יו חֲטָאָ֥ה גְדֹלָֽה:  מ  וַיֹּ֤אמֶר מֹשֶׁה֙ אֶֽל-אַֽהֲרֹ֔ן מֶה-עָשָׂ֥ה לְךָ֖ הָעָ֣ם הַזֶּ֑ה כִּֽי-הֵבֵ֥אתָ עָלָ֖יו חֲטָאָ֥ה גְדֹלָֽה:  ת  וַאֲמַר משֶׁה לְאַהֲרֹן מָא עֲבַד לָךְ עַמָא הָדֵין אֲרֵי אַיְתֵיתָא עֲלוֹהִי חוֹבָא רַבָּא:

{כב} וַיֹּ֣אמֶר אַֽהֲרֹ֔ן אַל-יִ֥חַר אַ֖ף אֲדֹנִ֑י אַתָּה֙ יָדַ֣עְתָּ אֶת-הָעָ֔ם כִּ֥י בְרָ֖ע הֽוּא:  מ  וַיֹּ֣אמֶר אַֽהֲרֹ֔ן אַל-יִ֥חַר אַ֖ף אֲדֹנִ֑י אַתָּה֙ יָדַ֣עְתָּ אֶת-הָעָ֔ם כִּ֥י בְרָ֖ע הֽוּא:  ת  וַאֲמַר אַהֲרֹן לָא יִתְקֵף רוּגְזָא דְרִבּוֹנִי אַתְּ יְדַעְתְּ יָת עַמָא אֲרֵי בְּבִישׁ הוּא:

{כג} וַיֹּ֣אמְרוּ לִ֔י עֲשֵׂה-לָ֣נוּ אֱלֹהִ֔ים אֲשֶׁ֥ר יֵֽלְכ֖וּ לְפָנֵ֑ינוּ כִּי-זֶ֣ה | מֹשֶׁ֣ה הָאִ֗ישׁ אֲשֶׁ֤ר הֶֽעֱלָ֨נוּ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לֹ֥א יָדַ֖עְנוּ מֶה-הָ֥יָה לֽוֹ:  מ  וַיֹּ֣אמְרוּ לִ֔י עֲשֵׂה-לָ֣נוּ אֱלֹהִ֔ים אֲשֶׁ֥ר יֵֽלְכ֖וּ לְפָנֵ֑ינוּ כִּי-זֶ֣ה | מֹשֶׁ֣ה הָאִ֗ישׁ אֲשֶׁ֤ר הֶֽעֱלָ֨נוּ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לֹ֥א יָדַ֖עְנוּ מֶה-הָ֥יָה לֽוֹ:  ת  וַאֲמָרוּ לִי עִבֵיד לָנָא דַחֲלָן דִי יְהָכָן קֳדָמָנָא אֲרֵי דֵין משֶׁה גַבְרָא דִי אַסְקָנָא מֵאַרְעָא דְמִצְרַיִם לָא יְדַעְנָא מָא הֲוָה לֵהּ:

{כד} וָֽאֹמַ֤ר לָהֶם֙ לְמִ֣י זָהָ֔ב הִתְפָּרָ֖קוּ וַיִּתְּנוּ-לִ֑י וָֽאַשְׁלִכֵ֣הוּ בָאֵ֔שׁ וַיֵּצֵ֖א הָעֵ֥גֶל הַזֶּֽה:  מ  וָֽאֹמַ֤ר לָהֶם֙ לְמִ֣י זָהָ֔ב הִתְפָּרָ֖קוּ וַיִּתְּנוּ-לִ֑י וָֽאַשְׁלִכֵ֣הוּ בָאֵ֔שׁ וַיֵּצֵ֖א הָעֵ֥גֶל הַזֶּֽה:  ת  וַאֲמָרִית לְהוֹן לְמָן דַהֲבָא פְּרִיקוּ וִיהָבוּ לִי וּרְמִיתֵהּ בְּנוּרָא וּנְפַק עֶגְלָא הָדֵין:

{כה} וַיַּ֤רְא מֹשֶׁה֙ אֶת-הָעָ֔ם כִּ֥י פָרֻ֖עַ ה֑וּא כִּֽי-פְרָעֹ֣ה אַֽהֲרֹ֔ן לְשִׁמְצָ֖ה בְּקָֽמֵיהֶֽם:  מ  וַיַּ֤רְא מֹשֶׁה֙ אֶת-הָעָ֔ם כִּ֥י פָרֻ֖עַ ה֑וּא כִּֽי-פְרָעֹ֣ה אַֽהֲרֹ֔ן לְשִׁמְצָ֖ה בְּקָֽמֵיהֶֽם:  ת  וַחֲזָא משֶׁה יָת עַמָא אֲרֵי בְטִיל הוּא אֲרֵי בָטֵילִנוּן אַהֲרֹן לַאֲסָבוּתְהוֹן שׁוּם בִּישׁ לְדָרֵיהוֹן:

{כו} וַיַּֽעֲמֹ֤ד מֹשֶׁה֙ בְּשַׁ֣עַר הַֽמַּֽחֲנֶ֔ה וַיֹּ֕אמֶר מִ֥י לַֽיהוָֹ֖ה אֵלָ֑י וַיֵּאָֽסְפ֥וּ אֵלָ֖יו כָּל-בְּנֵ֥י לֵוִֽי:  מ  וַיַּֽעֲמֹ֤ד מֹשֶׁה֙ בְּשַׁ֣עַר הַֽמַּֽחֲנֶ֔ה וַיֹּ֕אמֶר מִ֥י לַֽיהוָֹ֖ה אֵלָ֑י וַיֵּאָֽסְפ֥וּ אֵלָ֖יו כָּל-בְּנֵ֥י לֵוִֽי:  ת  וְקָם משֶׁה בִּתְרַע מַשְׁרִיתָא וַאֲמַר מָן דַחֲלַיָא דַיְיָ יֵיתוּן לְוָתִי וְאִתְכַּנָשׁוּ לְוָתֵהּ כָּל בְּנֵי לֵוִי:

{כז} וַיֹּ֣אמֶר לָהֶ֗ם כֹּֽה-אָמַ֤ר יְהוָֹה֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל שִׂ֥ימוּ אִישׁ-חַרְבּ֖וֹ עַל-יְרֵכ֑וֹ עִבְר֨וּ וָשׁ֜וּבוּ מִשַּׁ֤עַר לָשַׁ֨עַר֙ בַּֽמַּֽחֲנֶ֔ה וְהִרְג֧וּ אִֽישׁ-אֶת-אָחִ֛יו וְאִ֥ישׁ אֶת-רֵעֵ֖הוּ וְאִ֥ישׁ אֶת-קְרֹבֽוֹ:  מ  וַיֹּ֣אמֶר לָהֶ֗ם כֹּֽה-אָמַ֤ר יְהוָֹה֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל שִׂ֥ימוּ אִישׁ-חַרְבּ֖וֹ עַל-יְרֵכ֑וֹ עִבְר֨וּ וָשׁ֜וּבוּ מִשַּׁ֤עַר לָשַׁ֨עַר֙ בַּֽמַּֽחֲנֶ֔ה וְהִרְג֧וּ אִֽישׁ-אֶת-אָחִ֛יו וְאִ֥ישׁ אֶת-רֵעֵ֖הוּ וְאִ֥ישׁ אֶת-קְרֹבֽוֹ:  ת  וַאֲמַר לְהוֹן כִּדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל שַׁווּ גְבַר חַרְבֵּהּ עַל יַרְכֵּהּ עִבַרוּ וְתוּבוּ מִתְּרַע לִתְרַע בְּמַשְׁרִיתָא וּקְטוּלוּ גְבַר יָת אָחוּהִי וּגְבַר יָת חַבְרֵהּ וֶאֱנַשׁ יָת קָרִיבֵהּ:

{כח} וַיַּֽעֲשׂ֥וּ בְנֵֽי-לֵוִ֖י כִּדְבַ֣ר מֹשֶׁ֑ה וַיִּפֹּ֤ל מִן-הָעָם֙ בַּיּ֣וֹם הַה֔וּא כִּשְׁל֥שֶׁת אַלְפֵ֖י אִֽישׁ:  מ  וַיַּֽעֲשׂ֥וּ בְנֵֽי-לֵוִ֖י כִּדְבַ֣ר מֹשֶׁ֑ה וַיִּפֹּ֤ל מִן-הָעָם֙ בַּיּ֣וֹם הַה֔וּא כִּשְׁל֥שֶׁת אַלְפֵ֖י אִֽישׁ:  ת  וַעֲבָדוּ בְנֵי לֵוִי כְּפִתְגָמָא דְמשֶׁה וּנְפַל מִן עַמָא בְּיוֹמָא הַהוּא כִּתְלָתָא אַלְפִין גַבְרָא:

{כט} וַיֹּ֣אמֶר מֹשֶׁ֗ה מִלְא֨וּ יֶדְכֶ֤ם הַיּוֹם֙ לַֽיהֹוָ֔ה כִּ֛י אִ֥ישׁ בִּבְנ֖וֹ וּבְאָחִ֑יו וְלָתֵ֧ת עֲלֵיכֶ֛ם הַיּ֖וֹם בְּרָכָֽה:  מ  וַיֹּ֣אמֶר מֹשֶׁ֗ה מִלְא֨וּ יֶדְכֶ֤ם הַיּוֹם֙ לַֽיהֹוָ֔ה כִּ֛י אִ֥ישׁ בִּבְנ֖וֹ וּבְאָחִ֑יו וְלָתֵ֧ת עֲלֵיכֶ֛ם הַיּ֖וֹם בְּרָכָֽה:  ת  וַאֲמַר משֶׁה קָרִיבוּ יֶדְכוֹן קֻרְבָּנָא יוֹמָא דֵין קֳדָם יְיָ אֲרֵי גְבַר בִּבְרֵהּ וּבְאָחוּהִי וּלְאַיְתָאָה עֲלֵיכוֹן יוֹמָא דֵין בִּרְכָן:

{ל} וַֽיְהִי֙ מִֽמָּ֣חֳרָ֔ת וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל-הָעָ֔ם אַתֶּ֥ם חֲטָאתֶ֖ם חֲטָאָ֣ה גְדֹלָ֑ה וְעַתָּה֙ אֶֽעֱלֶ֣ה אֶל-יְהֹוָ֔ה אוּלַ֥י אֲכַפְּרָ֖ה בְּעַ֥ד חַטַּאתְכֶֽם:  מ  וַֽיְהִי֙ מִֽמָּ֣חֳרָ֔ת וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל-הָעָ֔ם אַתֶּ֥ם חֲטָאתֶ֖ם חֲטָאָ֣ה גְדֹלָ֑ה וְעַתָּה֙ אֶֽעֱלֶ֣ה אֶל-יְהֹוָ֔ה אוּלַ֥י אֲכַפְּרָ֖ה בְּעַ֥ד חַטַּאתְכֶֽם:  ת  וַהֲוָה בְּיוֹמָא דְבַתְרוֹהִי וַאֲמַר משֶׁה לְעַמָא אַתּוּן חַבְתּוּן חוֹבָא רַבָּא וּכְעַן אֶסַק קֳדָם יְיָ מָאִים אֵכַפֵּר עַל חוֹבֵיכוֹן:

{לא} וַיָּ֧שָׁב מֹשֶׁ֛ה אֶל-יְהוָֹ֖ה וַיֹּאמַ֑ר אָ֣נָּ֗א חָטָ֞א הָעָ֤ם הַזֶּה֙ חֲטָאָ֣ה גְדֹלָ֔ה וַיַּֽעֲשׂ֥וּ לָהֶ֖ם אֱלֹהֵ֥י זָהָֽב:  מ  וַיָּ֧שָׁב מֹשֶׁ֛ה אֶל-יְהוָֹ֖ה וַיֹּאמַ֑ר אָ֣נָּ֗א חָטָ֞א הָעָ֤ם הַזֶּה֙ חֲטָאָ֣ה גְדֹלָ֔ה וַיַּֽעֲשׂ֥וּ לָהֶ֖ם אֱלֹהֵ֥י זָהָֽב:  ת  וְתַב משֶׁה לִקֳדָם יְיָ וַאֲמָר בְּבָעוּ חַב עַמָא הָדֵין חוֹבָא רַבָּא וַעֲבָדוּ לְהוֹן דַחֲלָן דִדְהָב:

{לב} וְעַתָּ֖ה אִם-תִּשָּׂ֣א חַטָּאתָ֑ם וְאִם-אַ֕יִן מְחֵ֣נִי נָ֔א מִֽסִּפְרְךָ֖ אֲשֶׁ֥ר כָּתָֽבְתָּ:  מ  וְעַתָּ֖ה אִם-תִּשָּׂ֣א חַטָּאתָ֑ם וְאִם-אַ֕יִן מְחֵ֣נִי נָ֔א מִֽסִּפְרְךָ֖ אֲשֶׁ֥ר כָּתָֽבְתָּ:  ת  וּכְעַן אִם שַׁבְקַת לְחוֹבֵיהוֹן וְאִם לָא מְחֵנִי כְעַן מִסִפְרָךְ דִי כְתַבְתָּא:

{לג} וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֑ה מִ֚י אֲשֶׁ֣ר חָֽטָא-לִ֔י אֶמְחֶ֖נּוּ מִסִּפְרִֽי:  מ  וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֑ה מִ֚י אֲשֶׁ֣ר חָֽטָא-לִ֔י אֶמְחֶ֖נּוּ מִסִּפְרִֽי:  ת  וַאֲמַר יְיָ לְמשֶׁה מָן דִי חַב קֳדָמַי אֶמְחִנֵהּ מִסִפְרִי:

{לד} וְעַתָּ֞ה לֵ֣ךְ | נְחֵ֣ה אֶת-הָעָ֗ם אֶ֤ל אֲשֶׁר-דִּבַּ֨רְתִּי֙ לָ֔ךְ הִנֵּ֥ה מַלְאָכִ֖י יֵלֵ֣ךְ לְפָנֶ֑יךָ וּבְי֣וֹם פָּקְדִ֔י וּפָֽקַדְתִּ֥י עֲלֵהֶ֖ם חַטָּאתָֽם:  מ  וְעַתָּ֞ה לֵ֣ךְ | נְחֵ֣ה אֶת-הָעָ֗ם אֶ֤ל אֲשֶׁר-דִּבַּ֨רְתִּי֙ לָ֔ךְ הִנֵּ֥ה מַלְאָכִ֖י יֵלֵ֣ךְ לְפָנֶ֑יךָ וּבְי֣וֹם פָּקְדִ֔י וּפָֽקַדְתִּ֥י עֲלֵהֶ֖ם חַטָּאתָֽם:  ת  וּכְעַן אִיזֵל דַבַּר יָת עַמָא לַאֲתַר דִי מַלֵלִית לָךְ הָא מַלְאָכִי יְהַךְ קֳדָמָךְ וּבְיוֹם אַסַעֲרוּתִי וְאַסְעַר עֲלֵיהוֹן חוֹבֵיהוֹן:

{לה} וַיִּגֹּ֥ף יְהוָֹ֖ה אֶת-הָעָ֑ם עַ֚ל אֲשֶׁ֣ר עָשׂ֣וּ אֶת-הָעֵ֔גֶל אֲשֶׁ֥ר עָשָׂ֖ה אַֽהֲרֹֽן:  מ  וַיִּגֹּ֥ף יְהוָֹ֖ה אֶת-הָעָ֑ם עַ֚ל אֲשֶׁ֣ר עָשׂ֣וּ אֶת-הָעֵ֔גֶל אֲשֶׁ֥ר עָשָׂ֖ה אַֽהֲרֹֽן:  ת  וּמְחָא יְיָ יָת עַמָא עַל דִי אִשְׁתַּעְבָּדוּ לְעֶגְלָא דִי עֲבַד אַהֲרֹן: (ס)




שמות פרק-לג

{א} וַיְדַבֵּ֨ר יְהֹוָ֤ה אֶל-מֹשֶׁה֙ לֵ֣ךְ עֲלֵ֣ה מִזֶּ֔ה אַתָּ֣ה וְהָעָ֔ם אֲשֶׁ֥ר הֶֽעֱלִ֖יתָ מֵאֶ֣רֶץ מִצְרָ֑יִם אֶל-הָאָ֗רֶץ אֲשֶׁ֣ר נִ֠שְׁבַּ֠עְתִּי לְאַבְרָהָ֨ם לְיִצְחָ֤ק וּֽלְיַֽעֲקֹב֙ לֵאמֹ֔ר לְזַֽרְעֲךָ֖ אֶתְּנֶֽנָּה:  מ  וַיְדַבֵּ֨ר יְהֹוָ֤ה אֶל-מֹשֶׁה֙ לֵ֣ךְ עֲלֵ֣ה מִזֶּ֔ה אַתָּ֣ה וְהָעָ֔ם אֲשֶׁ֥ר הֶֽעֱלִ֖יתָ מֵאֶ֣רֶץ מִצְרָ֑יִם אֶל-הָאָ֗רֶץ אֲשֶׁ֣ר נִ֠שְׁבַּ֠עְתִּי לְאַבְרָהָ֨ם לְיִצְחָ֤ק וּֽלְיַֽעֲקֹב֙ לֵאמֹ֔ר לְזַֽרְעֲךָ֖ אֶתְּנֶֽנָּה:  ת  וּמַלִיל יְיָ עִם משֶׁה אִיזֵל סַק מִכָּא אַתְּ וְעַמָא דְאַסֶקְתָּא מֵאַרְעָא דְמִצְרַיִם לְאַרְעָא דִי קַיֵמִית לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לְמֵימַר לִבְנָיךְ אֶתְּנִנַהּ:

{ב} וְשָֽׁלַחְתִּ֥י לְפָנֶ֖יךָ מַלְאָ֑ךְ וְגֵֽרַשְׁתִּ֗י אֶת-הַֽכְּנַֽעֲנִי֙ הָֽאֱמֹרִ֔י וְהַֽחִתִּי֙ וְהַפְּרִזִּ֔י הַֽחִוִּ֖י וְהַיְבוּסִֽי:  מ  וְשָֽׁלַחְתִּ֥י לְפָנֶ֖יךָ מַלְאָ֑ךְ וְגֵֽרַשְׁתִּ֗י אֶת-הַֽכְּנַֽעֲנִי֙ הָֽאֱמֹרִ֔י וְהַֽחִתִּי֙ וְהַפְּרִזִּ֔י הַֽחִוִּ֖י וְהַיְבוּסִֽי:  ת  וְאֶשְׁלַח קֳדָמָךְ מַלְאָכָא וְאֵתָרֵךְ יָת כְּנַעֲנָאֵי אֱמוֹרָאֵי וְחִתָּאֵי וּפְרִזָאֵי חִוָאֵי וִיבוּסָאֵי:

{ג} אֶל-אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ כִּי֩ לֹ֨א אֶֽעֱלֶ֜ה בְּקִרְבְּךָ֗ כִּ֤י עַם-קְשֵׁה-עֹ֨רֶף֙ אַ֔תָּה פֶּן-אֲכֶלְךָ֖ בַּדָּֽרֶךְ:  מ  אֶל-אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ כִּי֩ לֹ֨א אֶֽעֱלֶ֜ה בְּקִרְבְּךָ֗ כִּ֤י עַם-קְשֵׁה-עֹ֨רֶף֙ אַ֔תָּה פֶּן-אֲכֶלְךָ֖ בַּדָּֽרֶךְ:  ת  לְאַרְעָא עַבְדָא חֲלָב וּדְבָשׁ אֲרֵי לָא אֲסַלֵק שְׁכִנְתִּי מִבֵּינָךְ אֲרֵי עַם קְשֵׁי קְדַל אַתְּ דִילְמָא אֲשֵׁצִינָךְ בְּאָרְחָא:

{ד} וַיִּשְׁמַ֣ע הָעָ֗ם אֶת-הַדָּבָ֥ר הָרָ֛ע הַזֶּ֖ה וַיִּתְאַבָּ֑לוּ וְלֹא-שָׁ֛תוּ אִ֥ישׁ עֶדְי֖וֹ עָלָֽיו:  מ  וַיִּשְׁמַ֣ע הָעָ֗ם אֶת-הַדָּבָ֥ר הָרָ֛ע הַזֶּ֖ה וַיִּתְאַבָּ֑לוּ וְלֹא-שָׁ֛תוּ אִ֥ישׁ עֶדְי֖וֹ עָלָֽיו:  ת  וּשְׁמַע עַמָא יָת פִּתְגָמָא בִישָׁא הָדֵין וְאִתְאַבָּלוּ וְלָא שַׁוִיוּ גְבַר תִּקוּן זֵינֵהּ עֲלוֹהִי:

{ה} וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל-מֹשֶׁ֗ה אֱמֹ֤ר אֶל-בְּנֵֽי-יִשְׂרָאֵל֙ אַתֶּ֣ם עַם-קְשֵׁה-עֹ֔רֶף רֶ֧גַע אֶחָ֛ד אֶֽעֱלֶ֥ה בְקִרְבְּךָ֖ וְכִלִּיתִ֑יךָ וְעַתָּ֗ה הוֹרֵ֤ד עֶדְיְךָ֙ מֵֽעָלֶ֔יךָ וְאֵֽדְעָ֖ה מָ֥ה אֶֽעֱשֶׂה-לָּֽךְ:  מ  וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל-מֹשֶׁ֗ה אֱמֹ֤ר אֶל-בְּנֵֽי-יִשְׂרָאֵל֙ אַתֶּ֣ם עַם-קְשֵׁה-עֹ֔רֶף רֶ֧גַע אֶחָ֛ד אֶֽעֱלֶ֥ה בְקִרְבְּךָ֖ וְכִלִּיתִ֑יךָ וְעַתָּ֗ה הוֹרֵ֤ד עֶדְיְךָ֙ מֵֽעָלֶ֔יךָ וְאֵֽדְעָ֖ה מָ֥ה אֶֽעֱשֶׂה-לָּֽךְ:  ת  וַאֲמַר יְיָ לְמשֶׁה אֵימַר לִבְנֵי יִשְׂרָאֵל אַתּוּן עַם קְשֵׁי קְדַל שָׁעָה חֲדָא אֲסַלֵק שְׁכִנְתִּי מִבֵּינָךְ וְאֵשֵׁיצִינָךְ וּכְעַן אַעְדִי תִּקוּן זֵינָךְ מִנָךְ וּגְלֵי קֳדָמַי מָה אַעְבֵּד לָךְ:

{ו} וַיִּתְנַצְּל֧וּ בְנֵֽי-יִשְׂרָאֵ֛ל אֶת-עֶדְיָ֖ם מֵהַ֥ר חוֹרֵֽב:  מ  וַיִּתְנַצְּל֧וּ בְנֵֽי-יִשְׂרָאֵ֛ל אֶת-עֶדְיָ֖ם מֵהַ֥ר חוֹרֵֽב:  ת  וְאַעְדִיאוּ בְנֵי יִשְׂרָאֵל יָת תִּקוּן זֵינֵיהוֹן מִטוּרָא דְחוֹרֵב:

{ז} וּמֹשֶׁה֩ יִקַּ֨ח אֶת-הָאֹ֜הֶל וְנָֽטָה-ל֣וֹ | מִח֣וּץ לַֽמַּֽחֲנֶ֗ה הַרְחֵק֙ מִן-הַֽמַּֽחֲנֶ֔ה וְקָ֥רָא ל֖וֹ אֹ֣הֶל מוֹעֵ֑ד וְהָיָה֙ כָּל-מְבַקֵּ֣שׁ יְהֹוָ֔ה יֵצֵא֙ אֶל-אֹ֣הֶל מוֹעֵ֔ד אֲשֶׁ֖ר מִח֥וּץ לַֽמַּֽחֲנֶֽה:  מ  וּמֹשֶׁה֩ יִקַּ֨ח אֶת-הָאֹ֜הֶל וְנָֽטָה-ל֣וֹ | מִח֣וּץ לַֽמַּֽחֲנֶ֗ה הַרְחֵק֙ מִן-הַֽמַּֽחֲנֶ֔ה וְקָ֥רָא ל֖וֹ אֹ֣הֶל מוֹעֵ֑ד וְהָיָה֙ כָּל-מְבַקֵּ֣שׁ יְהֹוָ֔ה יֵצֵא֙ אֶל-אֹ֣הֶל מוֹעֵ֔ד אֲשֶׁ֖ר מִח֥וּץ לַֽמַּֽחֲנֶֽה:  ת  וּמשֶׁה נְסִיב יָת מַשְׁכְּנָא וּפַרְסֵהּ לֵהּ מִבָּרָא לְמַשְׁרִיתָא אַרְחֵיק מִן מַשְׁרִיתָא וְקָרֵי לֵהּ מַשְׁכַּן בֵּית אוּלְפָנָא וִיהֵי כָּל דְתָבַע אוּלְפַן מִן קֳדָם יְיָ נְפַק לְמַשְׁכַּן בֵּית אוּלְפָנָא דִי מִבָּרָא לְמַשְׁרִיתָא:

{ח} וְהָיָ֗ה כְּצֵ֤את מֹשֶׁה֙ אֶל-הָאֹ֔הֶל יָק֨וּמוּ֙ כָּל-הָעָ֔ם וְנִ֨צְּב֔וּ אִ֖ישׁ פֶּ֣תַח אָֽהֳל֑וֹ וְהִבִּ֨יטוּ֙ אַֽחֲרֵ֣י מֹשֶׁ֔ה עַד-בֹּא֖וֹ הָאֹֽהֱלָה:  מ  וְהָיָ֗ה כְּצֵ֤את מֹשֶׁה֙ אֶל-הָאֹ֔הֶל יָק֨וּמוּ֙ כָּל-הָעָ֔ם וְנִ֨צְּב֔וּ אִ֖ישׁ פֶּ֣תַח אָֽהֳל֑וֹ וְהִבִּ֨יטוּ֙ אַֽחֲרֵ֣י מֹשֶׁ֔ה עַד-בֹּא֖וֹ הָאֹֽהֱלָה:  ת  וַהֲוָה כַּד נְפַק משֶׁה לְמַשְׁכְּנָא קַיְמִין כָּל עַמָא וּמִתְעַתְּדִין גְבַר בִּתְרַע מַשְׁכְּנֵהּ וּמִסְתַּכְּלִין אֲחוֹרֵי משֶׁה עַד דְעָלֵל לְמַשְׁכְּנָא:

{ט} וְהָיָ֗ה כְּבֹ֤א מֹשֶׁה֙ הָאֹ֔הֱלָה יֵרֵד֙ עַמּ֣וּד הֶֽעָנָ֔ן וְעָמַ֖ד פֶּ֣תַֽח הָאֹ֑הֶל וְדִבֶּ֖ר עִם-מֹשֶֽׁה:  מ  וְהָיָ֗ה כְּבֹ֤א מֹשֶׁה֙ הָאֹ֔הֱלָה יֵרֵד֙ עַמּ֣וּד הֶֽעָנָ֔ן וְעָמַ֖ד פֶּ֣תַֽח הָאֹ֑הֶל וְדִבֶּ֖ר עִם-מֹשֶֽׁה:  ת  וַהֲוָה כַּד עָלֵל משֶׁה לְמַשְׁכְּנָא נָחִית עַמוּדָא דַעֲנָנָא וְקָאִים בִּתְרַע מַשְׁכְּנָא וּמִתְמַלֵל עִם משֶׁה:

{י} וְרָאָ֤ה כָל-הָעָם֙ אֶת-עַמּ֣וּד הֶֽעָנָ֔ן עֹמֵ֖ד פֶּ֣תַֽח הָאֹ֑הֶל וְקָ֤ם כָּל-הָעָם֙ וְהִֽשְׁתַּֽחֲו֔וּ אִ֖ישׁ פֶּ֥תַח אָֽהֳלֽוֹ:  מ  וְרָאָ֤ה כָל-הָעָם֙ אֶת-עַמּ֣וּד הֶֽעָנָ֔ן עֹמֵ֖ד פֶּ֣תַֽח הָאֹ֑הֶל וְקָ֤ם כָּל-הָעָם֙ וְהִֽשְׁתַּֽחֲו֔וּ אִ֖ישׁ פֶּ֥תַח אָֽהֳלֽוֹ:  ת  וַחֲזָן כָּל עַמָא יָת עַמוּדָא דַעֲנָנָא קָאֵם בִּתְרַע מַשְׁכְּנָא וְקַיְמִין כָּל עַמָא וְסָגְדִין גְבַר בִּתְרַע מַשְׁכְּנֵהּ:

{יא} וְדִבֶּ֨ר יְהֹוָ֤ה אֶל-מֹשֶׁה֙ פָּנִ֣ים אֶל-פָּנִ֔ים כַּֽאֲשֶׁ֛ר יְדַבֵּ֥ר אִ֖ישׁ אֶל-רֵעֵ֑הוּ וְשָׁב֙ אֶל-הַֽמַּֽחֲנֶ֔ה וּמְשָׁ֨רְת֜וֹ יְהוֹשֻׁ֤עַ בִּן-נוּן֙ נַ֔עַר לֹ֥א יָמִ֖ישׁ מִתּ֥וֹךְ הָאֹֽהֶל:  מ  וְדִבֶּ֨ר יְהֹוָ֤ה אֶל-מֹשֶׁה֙ פָּנִ֣ים אֶל-פָּנִ֔ים כַּֽאֲשֶׁ֛ר יְדַבֵּ֥ר אִ֖ישׁ אֶל-רֵעֵ֑הוּ וְשָׁב֙ אֶל-הַֽמַּֽחֲנֶ֔ה וּמְשָׁ֨רְת֜וֹ יְהוֹשֻׁ֤עַ בִּן-נוּן֙ נַ֔עַר לֹ֥א יָמִ֖ישׁ מִתּ֥וֹךְ הָאֹֽהֶל:  ת  וּמִתְמַלֵל יְיָ עִם משֶׁה מַמְלַל עִם מַמְלַל כְּמָא דִמְמַלֵל גְבַר עִם חַבְרֵהּ וְתָב לְמַשְׁרִיתָא וּמְשׁוּמְשָׁנֵהּ יְהוֹשֻׁעַ בַּר נוּן עוּלֵימָא לָא עֲדֵי מִגוֹ מַשְׁכְּנָא: (פ)

{יב}  שלישי  וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל-יְהֹוָ֗ה רְ֠אֵ֠ה אַתָּ֞ה אֹמֵ֤ר אֵלַי֙ הַ֚עַל אֶת-הָעָ֣ם הַזֶּ֔ה וְאַתָּה֙ לֹ֣א הֽוֹדַעְתַּ֔נִי אֵ֥ת אֲשֶׁר-תִּשְׁלַ֖ח עִמִּ֑י וְאַתָּ֤ה אָמַ֨רְתָּ֙ יְדַעְתִּ֣יךָ בְשֵׁ֔ם וְגַם-מָצָ֥אתָ חֵ֖ן בְּעֵינָֽי:  מ  וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל-יְהֹוָ֗ה רְ֠אֵ֠ה אַתָּ֞ה אֹמֵ֤ר אֵלַי֙ הַ֚עַל אֶת-הָעָ֣ם הַזֶּ֔ה וְאַתָּה֙ לֹ֣א הֽוֹדַעְתַּ֔נִי אֵ֥ת אֲשֶׁר-תִּשְׁלַ֖ח עִמִּ֑י וְאַתָּ֤ה אָמַ֨רְתָּ֙ יְדַעְתִּ֣יךָ בְשֵׁ֔ם וְגַם-מָצָ֥אתָ חֵ֖ן בְּעֵינָֽי:  ת  וַאֲמַר משֶׁה קֳדָם יְיָ חֲזֵי דְאַתְּ אָמַר לִי אַסֵיק יָת עַמָא הָדֵין וְאַתְּ לֹא הוֹדַעְתַּנִי יָת דִי תִשְׁלַח עִמִי וְאַתְּ אָמַרְתְּ רַבִּיתָךְ בְּשׁוּם וְאַף אַשְׁכַּחְתָּא רַחֲמִין קֳדָמָי:

{יג} וְעַתָּ֡ה אִם-נָא֩ מָצָ֨אתִי חֵ֜ן בְּעֵינֶ֗יךָ הֽוֹדִעֵ֤נִי נָא֙ אֶת-דְּרָכֶ֔ךָ וְאֵדָ֣עֲךָ֔ לְמַ֥עַן אֶמְצָא-חֵ֖ן בְּעֵינֶ֑יךָ וּרְאֵ֕ה כִּ֥י עַמְּךָ֖ הַגּ֥וֹי הַזֶּֽה:  מ  וְעַתָּ֡ה אִם-נָא֩ מָצָ֨אתִי חֵ֜ן בְּעֵינֶ֗יךָ הֽוֹדִעֵ֤נִי נָא֙ אֶת-דְּרָכֶ֔ךָ וְאֵדָ֣עֲךָ֔ לְמַ֥עַן אֶמְצָא-חֵ֖ן בְּעֵינֶ֑יךָ וּרְאֵ֕ה כִּ֥י עַמְּךָ֖ הַגּ֥וֹי הַזֶּֽה:  ת  וּכְעַן אִם כְּעַן אַשְׁכָּחִית רַחֲמִין קֳדָמָךְ הוֹדַעְנִי כְעַן יָת אוֹרַח טּוּבָךְ וְאִדַע רַחֲמָךְ בְּדִיל דְאַשְׁכַּח רַחֲמִין קֳדָמָךְ וּגְלֵי קֳדָמָךְ אֲרֵי עַמָךְ עַמָא הָדֵין:

{יד} וַיֹּאמַ֑ר פָּנַ֥י יֵלֵ֖כוּ וַֽהֲנִחֹ֥תִי לָֽךְ:  מ  וַיֹּאמַ֑ר פָּנַ֥י יֵלֵ֖כוּ וַֽהֲנִחֹ֥תִי לָֽךְ:  ת  וַאֲמַר שְׁכִנְתִּי תְהַךְ וְאָנִיחַ לָךְ:

{טו} וַיֹּ֖אמֶר אֵלָ֑יו אִם-אֵ֤ין פָּנֶ֨יךָ֙ הֹֽלְכִ֔ים אַֽל-תַּֽעֲלֵ֖נוּ מִזֶּֽה:  מ  וַיֹּ֖אמֶר אֵלָ֑יו אִם-אֵ֤ין פָּנֶ֨יךָ֙ הֹֽלְכִ֔ים אַֽל-תַּֽעֲלֵ֖נוּ מִזֶּֽה:  ת  וַאֲמַר קֳדָמוֹהִי אִם לֵית שְׁכִנְתָּךְ מֵהַלְכָא בֵינָנָא לָא תַסְקִינָנָא מִכָּא:

{טז} וּבַמֶּ֣ה | יִוָּדַ֣ע אֵפ֗וֹא כִּֽי-מָצָ֨אתִי חֵ֤ן בְּעֵינֶ֨יךָ֙ אֲנִ֣י וְעַמֶּ֔ךָ הֲל֖וֹא בְּלֶכְתְּךָ֣ עִמָּ֑נוּ וְנִפְלִ֨ינוּ֙ אֲנִ֣י וְעַמְּךָ֔ מִכָּ֨ל-הָעָ֔ם אֲשֶׁ֖ר עַל-פְּנֵ֥י הָֽאֲדָמָֽה:  מ  וּבַמֶּ֣ה | יִוָּדַ֣ע אֵפ֗וֹא כִּֽי-מָצָ֨אתִי חֵ֤ן בְּעֵינֶ֨יךָ֙ אֲנִ֣י וְעַמֶּ֔ךָ הֲל֖וֹא בְּלֶכְתְּךָ֣ עִמָּ֑נוּ וְנִפְלִ֨ינוּ֙ אֲנִ֣י וְעַמְּךָ֔ מִכָּ֨ל-הָעָ֔ם אֲשֶׁ֖ר עַל-פְּנֵ֥י הָֽאֲדָמָֽה:  ת  וּבַמֶה יִתְיַדַע הָכָא אֲרֵי אַשְׁכָּחִית רַחֲמִין קֳדָמָךְ אֲנָא וְעַמָךְ הֲלָא בְּמֵהַךְ שְׁכִנְתָּךְ עִמָנָא וְיִתְעַבְּדַן לָנָא פְרִישָׁן לִי וּלְעַמָךְ מְשַׁנֵי מִכָּל עַמָא דִי עַל אַפֵּי אַרְעָא: (פ)

{יז} וַיֹּ֤אמֶר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה גַּ֣ם אֶת-הַדָּבָ֥ר הַזֶּ֛ה אֲשֶׁ֥ר דִּבַּ֖רְתָּ אֶֽעֱשֶׂ֑ה כִּֽי-מָצָ֤אתָ חֵן֙ בְּעֵינַ֔י וָֽאֵדָֽעֲךָ֖ בְּשֵֽׁם:  מ  וַיֹּ֤אמֶר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה גַּ֣ם אֶת-הַדָּבָ֥ר הַזֶּ֛ה אֲשֶׁ֥ר דִּבַּ֖רְתָּ אֶֽעֱשֶׂ֑ה כִּֽי-מָצָ֤אתָ חֵן֙ בְּעֵינַ֔י וָֽאֵדָֽעֲךָ֖ בְּשֵֽׁם:  ת  וַאֲמַר יְיָ לְמשֶׁה אַף יָת פִּתְגָמָא הָדֵין דִי מַלֶלְתָּא אֶעְבֵּד אֲרֵי אַשְׁכַּחְתָּ רַחֲמִין קֳדָמַי וְרַבִּיתָךְ בְּשׁוּם:

{יח}  רביעי  וַיֹּאמַ֑ר הַרְאֵ֥נִי נָ֖א אֶת-כְּבֹדֶֽךָ:  מ  וַיֹּאמַ֑ר הַרְאֵ֥נִי נָ֖א אֶת-כְּבֹדֶֽךָ:  ת  וַאֲמַר אַחֲזֵינִי כְעַן יָת יְקָרָךְ:

{יט} וַיֹּ֗אמֶר אֲנִ֨י אַֽעֲבִ֤יר כָּל-טוּבִי֙ עַל-פָּנֶ֔יךָ וְקָרָ֧אתִי בְשֵׁ֛ם יְהוָֹ֖ה לְפָנֶ֑יךָ וְחַנֹּתִי֙ אֶת-אֲשֶׁ֣ר אָחֹ֔ן וְרִֽחַמְתִּ֖י אֶת-אֲשֶׁ֥ר אֲרַחֵֽם:  מ  וַיֹּ֗אמֶר אֲנִ֨י אַֽעֲבִ֤יר כָּל-טוּבִי֙ עַל-פָּנֶ֔יךָ וְקָרָ֧אתִי בְשֵׁ֛ם יְהוָֹ֖ה לְפָנֶ֑יךָ וְחַנֹּתִי֙ אֶת-אֲשֶׁ֣ר אָחֹ֔ן וְרִֽחַמְתִּ֖י אֶת-אֲשֶׁ֥ר אֲרַחֵֽם:  ת  וַאֲמַר אֲנָא אַעְבַּר כָּל טוּבִי עַל אַפָּיךְ וְאֶקְרֵי בִשְׁמָא דַיְיָ קֳדָמָךְ וְאֵחוֹן לְמָן דְאֵחוֹן וְאֵרַחֵם עַל מָן דְאֵרַחֵם:

{כ} וַיֹּ֕אמֶר לֹ֥א תוּכַ֖ל לִרְאֹ֣ת אֶת-פָּנָ֑י כִּ֛י לֹֽא-יִרְאַ֥נִי הָֽאָדָ֖ם וָחָֽי:  מ  וַיֹּ֕אמֶר לֹ֥א תוּכַ֖ל לִרְאֹ֣ת אֶת-פָּנָ֑י כִּ֛י לֹֽא-יִרְאַ֥נִי הָֽאָדָ֖ם וָחָֽי:  ת  וַאֲמַר לָא תִיכוּל לְמֶחֱזֵי יָת אַפֵּי שְׁכִנְתִּי אֲרֵי לָא יֶחֱזִינַנִי אֶנָשָׁא וְיִתְקַיָם:

{כא} וַיֹּ֣אמֶר יְהֹוָ֔ה הִנֵּ֥ה מָק֖וֹם אִתִּ֑י וְנִצַּבְתָּ֖ עַל-הַצּֽוּר:  מ  וַיֹּ֣אמֶר יְהֹוָ֔ה הִנֵּ֥ה מָק֖וֹם אִתִּ֑י וְנִצַּבְתָּ֖ עַל-הַצּֽוּר:  ת  וַאֲמַר יְיָ הָא אָתַר מַתְקַן קֳדָמָי וְתִתְעַתַּד עַל טִנָרָא:

{כב} וְהָיָה֙ בַּֽעֲבֹ֣ר כְּבֹדִ֔י וְשַׂמְתִּ֖יךָ בְּנִקְרַ֣ת הַצּ֑וּר וְשַׂכֹּתִ֥י כַפִּ֛י עָלֶ֖יךָ עַד-עָבְרִֽי:  מ  וְהָיָה֙ בַּֽעֲבֹ֣ר כְּבֹדִ֔י וְשַׂמְתִּ֖יךָ בְּנִקְרַ֣ת הַצּ֑וּר וְשַׂכֹּתִ֥י כַפִּ֛י עָלֶ֖יךָ עַד-עָבְרִֽי:  ת  וִיהֵי בְּמֵעֲבַר יְקָרִי וְאֵשַׁוִינָךְ בִּמְעָרַת טִנָרָא וְאָגֵין בְּמֵימְרִי עֲלָךְ עַד דְאֶעֱבַר:

{כג} וַֽהֲסִרֹתִי֙ אֶת-כַּפִּ֔י וְרָאִ֖יתָ אֶת-אֲחֹרָ֑י וּפָנַ֖י לֹ֥א יֵֽרָאֽוּ:  מ  וַֽהֲסִרֹתִי֙ אֶת-כַּפִּ֔י וְרָאִ֖יתָ אֶת-אֲחֹרָ֑י וּפָנַ֖י לֹ֥א יֵֽרָאֽוּ:  ת  וְאַעְדֵי יָת דִבְרַת יְקָרִי וְתֶחֱזֵי יָת דְבַתְרָי וְדִקֳדָמַי לָא יִתְחֲזוּן: (פ)

{א}  חמישי  וַיֹּ֤אמֶר יְהוָֹה֙ אֶל-מֹשֶׁ֔ה פְּסָל-לְךָ֛ שְׁנֵֽי-לֻחֹ֥ת אֲבָנִ֖ים כָּרִֽאשֹׁנִ֑ים וְכָֽתַבְתִּי֙ עַל-הַלֻּחֹ֔ת אֶ֨ת-הַדְּבָרִ֔ים אֲשֶׁ֥ר הָי֛וּ עַל-הַלֻּחֹ֥ת הָרִֽאשֹׁנִ֖ים אֲשֶׁ֥ר שִׁבַּֽרְתָּ:  מ  וַיֹּ֤אמֶר יְהוָֹה֙ אֶל-מֹשֶׁ֔ה פְּסָל-לְךָ֛ שְׁנֵֽי-לֻחֹ֥ת אֲבָנִ֖ים כָּרִֽאשֹׁנִ֑ים וְכָֽתַבְתִּי֙ עַל-הַלֻּחֹ֔ת אֶ֨ת-הַדְּבָרִ֔ים אֲשֶׁ֥ר הָי֛וּ עַל-הַלֻּחֹ֥ת הָרִֽאשֹׁנִ֖ים אֲשֶׁ֥ר שִׁבַּֽרְתָּ:  ת  וַאֲמַר יְיָ לְמשֶׁה פְּסָל לָךְ תְּרֵין לוּחֵי אַבְנַיָא כְּקַדְמָאֵי וְאֶכְתּוֹב עַל לוּחַיָא יָת פִּתְגָמַיָא דִי הֲווֹ עַל לוּחַיָא קַדְמָאֵי דִי תְבַרְתָּא:

{ב} וֶהְיֵ֥ה נָכ֖וֹן לַבֹּ֑קֶר וְעָלִ֤יתָ בַבֹּ֨קֶר֙ אֶל-הַ֣ר סִינַ֔י וְנִצַּבְתָּ֥ לִ֛י שָׁ֖ם עַל-רֹ֥אשׁ הָהָֽר:  מ  וֶהְיֵ֥ה נָכ֖וֹן לַבֹּ֑קֶר וְעָלִ֤יתָ בַבֹּ֨קֶר֙ אֶל-הַ֣ר סִינַ֔י וְנִצַּבְתָּ֥ לִ֛י שָׁ֖ם עַל-רֹ֥אשׁ הָהָֽר:  ת  וֶהֱוֵי זַמִין לְצַפְרָא וְתִסַק בְּצַפְרָא לְטוּרָא דְסִינַי וְתִתְעַתַּד קֳדָמַי תַּמָן עַל רֵישׁ טוּרָא:

{ג} וְאִישׁ֙ לֹא-יַֽעֲלֶ֣ה עִמָּ֔ךְ וְגַם-אִ֥ישׁ אַל-יֵרָ֖א בְּכָל-הָהָ֑ר גַּם-הַצֹּ֤אן וְהַבָּקָר֙ אַל-יִרְע֔וּ אֶל-מ֖וּל הָהָ֥ר הַהֽוּא:  מ  וְאִישׁ֙ לֹא-יַֽעֲלֶ֣ה עִמָּ֔ךְ וְגַם-אִ֥ישׁ אַל-יֵרָ֖א בְּכָל-הָהָ֑ר גַּם-הַצֹּ֤אן וְהַבָּקָר֙ אַל-יִרְע֔וּ אֶל-מ֖וּל הָהָ֥ר הַהֽוּא:  ת  וֶאֱנַשׁ לָא יִסַק עִמָךְ וְאַף אֱנַשׁ לָא יִתְחֲזֵי בְּכָל טוּרָא אַף עֲנָא וְתוֹרֵי לָא יִרְעוּן לָקֳבֵל טוּרָא הַהוּא:

{ד} וַיִּפְסֹ֡ל שְׁנֵֽי-לֻחֹ֨ת אֲבָנִ֜ים כָּרִֽאשֹׁנִ֗ים וַיַּשְׁכֵּ֨ם מֹשֶׁ֤ה בַבֹּ֨קֶר֙ וַיַּ֨עַל֙ אֶל-הַ֣ר סִינַ֔י כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֹת֑וֹ וַיִּקַּ֣ח בְּיָד֔וֹ שְׁנֵ֖י לֻחֹ֥ת אֲבָנִֽים:  מ  וַיִּפְסֹ֡ל שְׁנֵֽי-לֻחֹ֨ת אֲבָנִ֜ים כָּרִֽאשֹׁנִ֗ים וַיַּשְׁכֵּ֨ם מֹשֶׁ֤ה בַבֹּ֨קֶר֙ וַיַּ֨עַל֙ אֶל-הַ֣ר סִינַ֔י כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָֹ֖ה אֹת֑וֹ וַיִּקַּ֣ח בְּיָד֔וֹ שְׁנֵ֖י לֻחֹ֥ת אֲבָנִֽים:  ת  וּפְסַל תְּרֵין לוּחֵי אַבְנַיָא כְּקַדְמָאֵי וְאַקְדֵם משֶׁה בְּצַפְרָא וּסְלֵיק לְטוּרָא דְסִינַי כְּמָא דְפַקֵד יְיָ יָתֵהּ וּנְסֵב בִּידֵהּ תְּרֵין לוּחֵי אַבְנַיָא:

{ה} וַיֵּ֤רֶד יְהוָֹה֙ בֶּֽעָנָ֔ן וַיִּתְיַצֵּ֥ב עִמּ֖וֹ שָׁ֑ם וַיִּקְרָ֥א בְשֵׁ֖ם יְהוָֹֽה:  מ  וַיֵּ֤רֶד יְהוָֹה֙ בֶּֽעָנָ֔ן וַיִּתְיַצֵּ֥ב עִמּ֖וֹ שָׁ֑ם וַיִּקְרָ֥א בְשֵׁ֖ם יְהוָֹֽה:  ת  וְאִתְגְלִי יְיָ בַּעֲנָנָא וְאִתְעַתַּד עִמֵהּ תַּמָן וּקְרָא בִשְׁמָא דַיְיָ:

{ו} וַיַּֽעֲבֹ֨ר יְהוָֹ֥ה | עַל-פָּנָיו֘ וַיִּקְרָא֒ יְהוָֹ֣ה | יְהֹוָ֔ה אֵ֥ל רַח֖וּם וְחַנּ֑וּן אֶ֥רֶךְ אַפַּ֖יִם וְרַב-חֶ֥סֶד וֶֽאֱמֶֽת:  מ  וַיַּֽעֲבֹ֨ר יְהוָֹ֥ה | עַל-פָּנָיו֘ וַיִּקְרָא֒ יְהוָֹ֣ה | יְהֹוָ֔ה אֵ֥ל רַח֖וּם וְחַנּ֑וּן אֶ֥רֶךְ אַפַּ֖יִם וְרַב-חֶ֥סֶד וֶֽאֱמֶֽת:  ת  וְאַעְבַּר יְיָ שְׁכִנְתֵּהּ עַל אַפּוֹהִי וּקְרָא יְיָ יְיָ אֱלָהָא רַחֲמָנָא וְחַנָנָא מַרְחִיק רְגַז וּמַסְגֵי לְמֶעְבַּד טַבְוָן וּקְשׁוֹט:

{ז} נֹצֵ֥ר חֶ֨סֶד֙ לָֽאֲלָפִ֔ים נֹשֵׂ֥א עָוֹ֛ן וָפֶ֖שַׁע וְחַטָּאָ֑ה וְנַקֵּה֙ לֹ֣א יְנַקֶּ֔ה פֹּקֵ֣ד | עֲוֹ֣ן אָב֗וֹת עַל-בָּנִים֙ וְעַל-בְּנֵ֣י בָנִ֔ים עַל-שִׁלֵּשִׁ֖ים וְעַל-רִבֵּעִֽים:  מ  נֹצֵ֥ר חֶ֨סֶד֙ לָֽאֲלָפִ֔ים נֹשֵׂ֥א עָוֹ֛ן וָפֶ֖שַׁע וְחַטָּאָ֑ה וְנַקֵּה֙ לֹ֣א יְנַקֶּ֔ה פֹּקֵ֣ד | עֲוֹ֣ן אָב֗וֹת עַל-בָּנִים֙ וְעַל-בְּנֵ֣י בָנִ֔ים עַל-שִׁלֵּשִׁ֖ים וְעַל-רִבֵּעִֽים:  ת  נָטַר טִיבוּ לְאַלְפֵי דָרִין שָׁבֵק לַעֲוָיָן וְלִמְרוֹד וּלְחוֹבִין סָלַח לִדְתַיְבִין לְאוֹרַיְתֵהּ וְלִדְלָא תַיְבִין לָא מְזַכֵּי מַסְעַר חוֹבֵי אֲבָהָן עַל בְּנִין וְעַל בְּנֵי בְנִין מְרָדִין עַל דַר תְּלִיתָי וְעַל דַר רְבִיעָי:

{ח} וַיְמַהֵ֖ר מֹשֶׁ֑ה וַיִּקֹּ֥ד אַ֖רְצָה וַיִּשְׁתָּֽחוּ:  מ  וַיְמַהֵ֖ר מֹשֶׁ֑ה וַיִּקֹּ֥ד אַ֖רְצָה וַיִּשְׁתָּֽחוּ:  ת  וְאוֹחִי משֶׁה וּכְרַע עַל אַרְעָא וּסְגִיד:

{ט} וַיֹּ֡אמֶר אִם-נָא֩ מָצָ֨אתִי חֵ֤ן בְּעֵינֶ֨יךָ֙ אֲדֹנָ֔י יֵֽלֶךְ-נָ֥א אֲדֹנָ֖י בְּקִרְבֵּ֑נוּ כִּ֤י עַם-קְשֵׁה-עֹ֨רֶף֙ ה֔וּא וְסָֽלַחְתָּ֛ לַֽעֲוֹנֵ֥נוּ וּלְחַטָּאתֵ֖נוּ וּנְחַלְתָּֽנוּ:  מ  וַיֹּ֡אמֶר אִם-נָא֩ מָצָ֨אתִי חֵ֤ן בְּעֵינֶ֨יךָ֙ אֲדֹנָ֔י יֵֽלֶךְ-נָ֥א אֲדֹנָ֖י בְּקִרְבֵּ֑נוּ כִּ֤י עַם-קְשֵׁה-עֹ֨רֶף֙ ה֔וּא וְסָֽלַחְתָּ֛ לַֽעֲוֹנֵ֥נוּ וּלְחַטָּאתֵ֖נוּ וּנְחַלְתָּֽנוּ:  ת  וַאֲמַר אִם כְּעַן אַשְׁכָּחִית רַחֲמִין קֳדָמָךְ יְיָ תְּהַךְ כְּעַן שְׁכִנְתָּא דַיְיָ בֵּינָנָא אֲרֵי עַם קְשֵׁי קְדַל הוּא וְתִשְׁבּוֹק לְחוֹבָנָא וְלַחֲטָאָנָא וְתַחְסְנָנָא:

{י}  שישי  וַיֹּ֗אמֶר הִנֵּ֣ה אָנֹכִי֘ כֹּרֵ֣ת בְּרִית֒ נֶ֤גֶד כָּֽל-עַמְּךָ֙ אֶֽעֱשֶׂ֣ה נִפְלָאֹ֔ת אֲשֶׁ֛ר לֹֽא-נִבְרְא֥וּ בְכָל-הָאָ֖רֶץ וּבְכָל-הַגּוֹיִ֑ם וְרָאָ֣ה כָל-הָ֠עָ֠ם אֲשֶׁר-אַתָּ֨ה בְקִרְבּ֜וֹ אֶת-מַֽעֲשֵׂ֤ה יְהוָֹה֙ כִּֽי-נוֹרָ֣א ה֔וּא אֲשֶׁ֥ר אֲנִ֖י עֹשֶׂ֥ה עִמָּֽךְ:  מ  וַיֹּ֗אמֶר הִנֵּ֣ה אָנֹכִי֘ כֹּרֵ֣ת בְּרִית֒ נֶ֤גֶד כָּֽל-עַמְּךָ֙ אֶֽעֱשֶׂ֣ה נִפְלָאֹ֔ת אֲשֶׁ֛ר לֹֽא-נִבְרְא֥וּ בְכָל-הָאָ֖רֶץ וּבְכָל-הַגּוֹיִ֑ם וְרָאָ֣ה כָל-הָ֠עָ֠ם אֲשֶׁר-אַתָּ֨ה בְקִרְבּ֜וֹ אֶת-מַֽעֲשֵׂ֤ה יְהוָֹה֙ כִּֽי-נוֹרָ֣א ה֔וּא אֲשֶׁ֥ר אֲנִ֖י עֹשֶׂ֥ה עִמָּֽךְ:  ת  וַאֲמַר הָא אֲנָא גְזַר קְיַם קֳדָם כָּל עַמָךְ אֶעְבֵּד פְּרִישָׁן דִי לָא אִתְבְּרִיאוּ בְּכָל אַרְעָא וּבְכָל עַמְמַיָא וְיֶחֱזֵי כָל עַמָא דִי אַתְּ בֵּינֵיהוֹן יָת עוֹבָדָא דַיְיָ אֲרֵי דָחֵל הוּא דִי אֲנָא עָבֵד עִמָךְ:

{יא} שְׁמָ֨ר-לְךָ֔ אֵ֛ת אֲשֶׁ֥ר אָֽנֹכִ֖י מְצַוְּךָ֣ הַיּ֑וֹם הִנְנִ֧י גֹרֵ֣שׁ מִפָּנֶ֗יךָ אֶת-הָֽאֱמֹרִי֙ וְהַֽכְּנַֽעֲנִ֔י וְהַֽחִתִּי֙ וְהַפְּרִזִּ֔י וְהַֽחִוִּ֖י וְהַיְבוּסִֽי:  מ  שְׁמָ֨ר-לְךָ֔ אֵ֛ת אֲשֶׁ֥ר אָֽנֹכִ֖י מְצַוְּךָ֣ הַיּ֑וֹם הִנְנִ֧י גֹרֵ֣שׁ מִפָּנֶ֗יךָ אֶת-הָֽאֱמֹרִי֙ וְהַֽכְּנַֽעֲנִ֔י וְהַֽחִתִּי֙ וְהַפְּרִזִּ֔י וְהַֽחִוִּ֖י וְהַיְבוּסִֽי:  ת  טָר לָךְ יָת דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין הָא אֲנָא מְתָרֵךְ מִן קֳדָמָךְ יָת אֱמוֹרָאֵי וּכְנַעֲנָאֵי וְחִתָּאֵי וּפְרִזָאֵי וְחִוָאֵי וִיבוּסָאֵי:

{יב} הִשָּׁ֣מֶר לְךָ֗ פֶּן-תִּכְרֹ֤ת בְּרִית֙ לְיוֹשֵׁ֣ב הָאָ֔רֶץ אֲשֶׁ֥ר אַתָּ֖ה בָּ֣א עָלֶ֑יהָ פֶּן-יִֽהְיֶ֥ה לְמוֹקֵ֖שׁ בְּקִרְבֶּֽךָ:  מ  הִשָּׁ֣מֶר לְךָ֗ פֶּן-תִּכְרֹ֤ת בְּרִית֙ לְיוֹשֵׁ֣ב הָאָ֔רֶץ אֲשֶׁ֥ר אַתָּ֖ה בָּ֣א עָלֶ֑יהָ פֶּן-יִֽהְיֶ֥ה לְמוֹקֵ֖שׁ בְּקִרְבֶּֽךָ:  ת  אִסְתַּמַר לָךְ דִילְמָא תִגְזַר קְיַם לְיָתֵב אַרְעָא דִי אַתְּ עָלֵל עֲלַהּ דִילְמָא יְהֵי לְתַקְלָא בֵּינָךְ:

{יג} כִּ֤י אֶת-מִזְבְּחֹתָם֙ תִּתֹּצ֔וּן וְאֶת-מַצֵּֽבֹתָ֖ם תְּשַׁבֵּר֑וּן וְאֶת-אֲשֵׁרָ֖יו תִּכְרֹתֽוּן:  מ  כִּ֤י אֶת-מִזְבְּחֹתָם֙ תִּתֹּצ֔וּן וְאֶת-מַצֵּֽבֹתָ֖ם תְּשַׁבֵּר֑וּן וְאֶת-אֲשֵׁרָ֖יו תִּכְרֹתֽוּן:  ת  אֲרֵי יָת אֲגוֹרֵיהוֹן תְּתַרְעוּן וְיָת קָמַתְהוֹן תְּתַבְּרוּן וְיָת אַשְׁרֵיהוֹן תְּקַצְצוּן:

{יד} כִּ֛י לֹ֥א תִֽשְׁתַּֽחֲוֶ֖ה לְאֵ֣ל אַחֵ֑ר כִּ֤י יְהוָֹה֙ קַנָּ֣א שְׁמ֔וֹ אֵ֥ל קַנָּ֖א הֽוּא:  מ  כִּ֛י לֹ֥א תִֽשְׁתַּֽחֲוֶ֖ה לְאֵ֣ל אַחֵ֑ר כִּ֤י יְהוָֹה֙ קַנָּ֣א שְׁמ֔וֹ אֵ֥ל קַנָּ֖א הֽוּא:  ת  אֲרֵי לָא תִסְגוֹד לְטַעֲוַת עַמְמַיָא אֲרֵי יְיָ קַנָא שְׁמֵהּ אֵל קַנָא הוּא:

{טו} פֶּן-תִּכְרֹ֥ת בְּרִ֖ית לְיוֹשֵׁ֣ב הָאָ֑רֶץ וְזָנ֣וּ | אַֽחֲרֵ֣י אֱלֹֽהֵיהֶ֗ם וְזָבְחוּ֙ לֵאלֹ֣הֵיהֶ֔ם וְקָרָ֣א לְךָ֔ וְאָֽכַלְתָּ֖ מִזִּבְחֽוֹ:  מ  פֶּן-תִּכְרֹ֥ת בְּרִ֖ית לְיוֹשֵׁ֣ב הָאָ֑רֶץ וְזָנ֣וּ | אַֽחֲרֵ֣י אֱלֹֽהֵיהֶ֗ם וְזָבְחוּ֙ לֵאלֹ֣הֵיהֶ֔ם וְקָרָ֣א לְךָ֔ וְאָֽכַלְתָּ֖ מִזִּבְחֽוֹ:  ת  דִילְמָא תִגְזַר קְיַם לְיָתֵב אַרְעָא וְיִטְּעוּן בָּתַר טַּעֲוַתְהוֹן וְיִדְבְּחוּן לְטַּעֲוַתְהוֹן וְיִקְרוּן לָךְ וְתֵיכוּל מִדִבְחֵיהוֹן:

{טז} וְלָֽקַחְתָּ֥ מִבְּנֹתָ֖יו לְבָנֶ֑יךָ וְזָנ֣וּ בְנֹתָ֗יו אַֽחֲרֵי֙ אֱלֹ֣הֵיהֶ֔ן וְהִזְנוּ֙ אֶת-בָּנֶ֔יךָ אַֽחֲרֵ֖י אֱלֹֽהֵיהֶֽן:  מ  וְלָֽקַחְתָּ֥ מִבְּנֹתָ֖יו לְבָנֶ֑יךָ וְזָנ֣וּ בְנֹתָ֗יו אַֽחֲרֵי֙ אֱלֹ֣הֵיהֶ֔ן וְהִזְנוּ֙ אֶת-בָּנֶ֔יךָ אַֽחֲרֵ֖י אֱלֹֽהֵיהֶֽן:  ת  וְתִסַב מִבְּנַתְהוֹן לִבְנָיךְ וְיִטְעַיָן בְּנַתְהוֹן בָּתַר טַעֲוַתְהוֹן וְיִטְעַיַן יָת בְּנָיךְ בָּתַר טַעֲוַתְהוֹן:

{יז} אֱלֹהֵ֥י מַסֵּכָ֖ה לֹ֥א תַֽעֲשֶׂה-לָּֽךְ:  מ  אֱלֹהֵ֥י מַסֵּכָ֖ה לֹ֥א תַֽעֲשֶׂה-לָּֽךְ:  ת  דַחֲלַן דְמַתְּכָא לָא תַעְבֵּד לָךְ:

{יח} אֶת-חַ֣ג הַמַּצּוֹת֘ תִּשְׁמֹר֒ שִׁבְעַ֨ת יָמִ֜ים תֹּאכַ֤ל מַצּוֹת֙ אֲשֶׁ֣ר צִוִּיתִ֔ךָ לְמוֹעֵ֖ד חֹ֣דֶשׁ הָֽאָבִ֑יב כִּ֚י בְּחֹ֣דֶשׁ הָֽאָבִ֔יב יָצָ֖אתָ מִמִּצְרָֽיִם:  מ  אֶת-חַ֣ג הַמַּצּוֹת֘ תִּשְׁמֹר֒ שִׁבְעַ֨ת יָמִ֜ים תֹּאכַ֤ל מַצּוֹת֙ אֲשֶׁ֣ר צִוִּיתִ֔ךָ לְמוֹעֵ֖ד חֹ֣דֶשׁ הָֽאָבִ֑יב כִּ֚י בְּחֹ֣דֶשׁ הָֽאָבִ֔יב יָצָ֖אתָ מִמִּצְרָֽיִם:  ת  יָת חַגָא דְפַטִּירַיָא תִּטַר שַׁבְעָא יוֹמִין תֵּיכוֹל פַּטִירָא כְּמָא דִי פַקֵדִתָּךְ לִזְמַן יַרְחָא דְאַבִּיבָא אֲרֵי בְּיַרְחָא דְאַבִּיבָא נְפַקְתָּא מִמִצְרָיִם:

{יט} כָּל-פֶּ֥טֶר רֶ֖חֶם לִ֑י וְכָֽל-מִקְנְךָ֙ תִּזָּכָ֔ר פֶּ֖טֶר שׁ֥וֹר וָשֶֽׂה:  מ  כָּל-פֶּ֥טֶר רֶ֖חֶם לִ֑י וְכָֽל-מִקְנְךָ֙ תִּזָּכָ֔ר פֶּ֖טֶר שׁ֥וֹר וָשֶֽׂה:  ת  כָּל פָּתַח וַלְדָא דִילִי וְכָל בְּעִירָךְ דִכְרִין תַּקְדֵשׁ בְּכוֹר תּוֹר וְאִימָר:

{כ} וּפֶ֤טֶר חֲמוֹר֙ תִּפְדֶּ֣ה בְשֶׂ֔ה וְאִם-לֹ֥א תִפְדֶּ֖ה וַֽעֲרַפְתּ֑וֹ כֹּ֣ל בְּכ֤וֹר בָּנֶ֨יךָ֙ תִּפְדֶּה וְלֹא-יֵֽרָא֥וּ פָנַ֖י רֵיקָֽם:  מ  וּפֶ֤טֶר חֲמוֹר֙ תִּפְדֶּ֣ה בְשֶׂ֔ה וְאִם-לֹ֥א תִפְדֶּ֖ה וַֽעֲרַפְתּ֑וֹ כֹּ֣ל בְּכ֤וֹר בָּנֶ֨יךָ֙ תִּפְדֶּה וְלֹא-יֵֽרָא֥וּ פָנַ֖י רֵיקָֽם:  ת  וּבוּכְרָא דַחֲמָרָא תִּפְרוֹק בְּאִימְרָא וְאִם לָא תִפְרוֹק וְתַנְקְפֵהּ כָּל בּוּכְרָא דִבְנָיךְ תִּפְרוֹק וְלָא יִתְחֲזוּן קֳדָמַי רֵיקָנִין:

{כא} שֵׁ֤שֶׁת יָמִים֙ תַּֽעֲבֹ֔ד וּבַיּ֥וֹם הַשְּׁבִיעִ֖י תִּשְׁבֹּ֑ת בֶּֽחָרִ֥ישׁ וּבַקָּצִ֖יר תִּשְׁבֹּֽת:  מ  שֵׁ֤שֶׁת יָמִים֙ תַּֽעֲבֹ֔ד וּבַיּ֥וֹם הַשְּׁבִיעִ֖י תִּשְׁבֹּ֑ת בֶּֽחָרִ֥ישׁ וּבַקָּצִ֖יר תִּשְׁבֹּֽת:  ת  שִׁתָּא יוֹמִין תִּפְלָח וּבְיוֹמָא שְׁבִיעָאָה תְּנוּחַ בִּזְרוֹעָא וּבַחֲצָדָא תְּנוּחַ:

{כב} וְחַ֤ג שָֽׁבֻעֹת֙ תַּֽעֲשֶׂ֣ה לְךָ֔ בִּכּוּרֵ֖י קְצִ֣יר חִטִּ֑ים וְחַג֙ הָֽאָסִ֔יף תְּקוּפַ֖ת הַשָּׁנָֽה:  מ  וְחַ֤ג שָֽׁבֻעֹת֙ תַּֽעֲשֶׂ֣ה לְךָ֔ בִּכּוּרֵ֖י קְצִ֣יר חִטִּ֑ים וְחַג֙ הָֽאָסִ֔יף תְּקוּפַ֖ת הַשָּׁנָֽה:  ת  וְחַגָא דְשָׁבוּעַיָא תַּעְבֵּד לָךְ בִּכּוּרֵי חֲצַד חִטִין וְחַגָא דִכְנָשָׁא בְּמִפְקָא דְשַׁתָּא:

{כג} שָׁל֥שׁ פְּעָמִ֖ים בַּשָּׁנָ֑ה יֵֽרָאֶה֙ כָּל-זְכ֣וּרְךָ֔ אֶת-פְּנֵ֛י הָֽאָדֹ֥ן | יְהוָֹ֖ה אֱלֹהֵ֥י יִשְׂרָאֵֽל:  מ  שָׁל֥שׁ פְּעָמִ֖ים בַּשָּׁנָ֑ה יֵֽרָאֶה֙ כָּל-זְכ֣וּרְךָ֔ אֶת-פְּנֵ֛י הָֽאָדֹ֥ן | יְהוָֹ֖ה אֱלֹהֵ֥י יִשְׂרָאֵֽל:  ת  תְּלַת זִמְנִין בְּשַׁתָּא יִתְחֲזוּן כָּל דְכוּרָךְ קֳדָם רִבּוֹן עָלְמָא יְיָ אֱלָהָא דְיִשְׂרָאֵל:

{כד} כִּֽי-אוֹרִ֤ישׁ גּוֹיִם֙ מִפָּנֶ֔יךָ וְהִרְחַבְתִּ֖י אֶת-גְּבֻלֶ֑ךָ וְלֹֽא-יַחְמֹ֥ד אִישׁ֙ אֶֽת-אַרְצְךָ֔ בַּֽעֲלֹֽתְךָ֗ לֵֽרָאוֹת֙ אֶת-פְּנֵי֙ יְהוָֹ֣ה אֱלֹהֶ֔יךָ שָׁל֥שׁ פְּעָמִ֖ים בַּשָּׁנָֽה:  מ  כִּֽי-אוֹרִ֤ישׁ גּוֹיִם֙ מִפָּנֶ֔יךָ וְהִרְחַבְתִּ֖י אֶת-גְּבֻלֶ֑ךָ וְלֹֽא-יַחְמֹ֥ד אִישׁ֙ אֶֽת-אַרְצְךָ֔ בַּֽעֲלֹֽתְךָ֗ לֵֽרָאוֹת֙ אֶת-פְּנֵי֙ יְהוָֹ֣ה אֱלֹהֶ֔יךָ שָׁל֥שׁ פְּעָמִ֖ים בַּשָּׁנָֽה:  ת  אֲרֵי אֲתָרֵךְ עַמְמִין מִן קֳדָמָךְ וְאַפְתֵּי יָת תְּחוּמָךְ וְלָא יַחְמֵד אֱנַשׁ יָת אַרְעָךְ בְּמִסְקָךְ לְאִתְחַזָאָה קֳדָם יְיָ אֱלָהָךְ תְּלַת זִמְנִין בְּשַׁתָּא:

{כה} לֹֽא-תִשְׁחַ֥ט עַל-חָמֵ֖ץ דַּם-זִבְחִ֑י וְלֹֽא-יָלִ֣ין לַבֹּ֔קֶר זֶ֖בַח חַ֥ג הַפָּֽסַח:  מ  לֹֽא-תִשְׁחַ֥ט עַל-חָמֵ֖ץ דַּם-זִבְחִ֑י וְלֹֽא-יָלִ֣ין לַבֹּ֔קֶר זֶ֖בַח חַ֥ג הַפָּֽסַח:  ת  לָא תִכּוֹס עַל חֲמִיעַ דַם פִּסְחִי וְלָא יְבִיתוּן לְצַפְרָא בַּר מִמַדְבְּחָא תַּרְבֵּי נִכְסַת חַגָא דְפִסְחָא:

{כו} רֵאשִׁ֗ית בִּכּוּרֵי֙ אַדְמָ֣תְךָ֔ תָּבִ֕יא בֵּ֖ית יְהוָֹ֣ה אֱלֹהֶ֑יךָ לֹֽא-תְבַשֵּׁ֥ל גְּדִ֖י בַּֽחֲלֵ֥ב אִמּֽוֹ:  מ  רֵאשִׁ֗ית בִּכּוּרֵי֙ אַדְמָ֣תְךָ֔ תָּבִ֕יא בֵּ֖ית יְהוָֹ֣ה אֱלֹהֶ֑יךָ לֹֽא-תְבַשֵּׁ֥ל גְּדִ֖י בַּֽחֲלֵ֥ב אִמּֽוֹ:  ת  רֵישׁ בִּכּוּרֵי אַרְעָךְ תַּיְתֵי לְבֵית מַקְדְשָׁא דַיְיָ אֱלָהָךְ לָא תֵיכְלוּן בְּשַׂר בַּחֲלָב: (פ)

{כז}  שביעי  וַיֹּ֤אמֶר יְהוָֹה֙ אֶל-מֹשֶׁ֔ה כְּתָב-לְךָ֖ אֶת-הַדְּבָרִ֣ים הָאֵ֑לֶּה כִּ֞י עַל-פִּ֣י | הַדְּבָרִ֣ים הָאֵ֗לֶּה כָּרַ֧תִּי אִתְּךָ֛ בְּרִ֖ית וְאֶת-יִשְׂרָאֵֽל:  מ  וַיֹּ֤אמֶר יְהוָֹה֙ אֶל-מֹשֶׁ֔ה כְּתָב-לְךָ֖ אֶת-הַדְּבָרִ֣ים הָאֵ֑לֶּה כִּ֞י עַל-פִּ֣י | הַדְּבָרִ֣ים הָאֵ֗לֶּה כָּרַ֧תִּי אִתְּךָ֛ בְּרִ֖ית וְאֶת-יִשְׂרָאֵֽל:  ת  וַאֲמַר יְיָ לְמשֶׁה כְּתַב לָךְ יָת פִּתְגָמַיָא הָאִלֵין אֲרֵי עַל מֵימַר פִּתְגָמַיָא הָאִלֵין גְזָרִית עִמָךְ קְיַם וְעִם יִשְׂרָאֵל:

{כח} וַֽיְהִי-שָׁ֣ם עִם-יְהֹוָ֗ה אַרְבָּעִ֥ים יוֹם֙ וְאַרְבָּעִ֣ים לַ֔יְלָה לֶ֚חֶם לֹ֣א אָכַ֔ל וּמַ֖יִם לֹ֣א שָׁתָ֑ה וַיִּכְתֹּ֣ב עַל-הַלֻּחֹ֗ת אֵ֚ת דִּבְרֵ֣י הַבְּרִ֔ית עֲשֶׂ֖רֶת הַדְּבָרִֽים:  מ  וַֽיְהִי-שָׁ֣ם עִם-יְהֹוָ֗ה אַרְבָּעִ֥ים יוֹם֙ וְאַרְבָּעִ֣ים לַ֔יְלָה לֶ֚חֶם לֹ֣א אָכַ֔ל וּמַ֖יִם לֹ֣א שָׁתָ֑ה וַיִּכְתֹּ֣ב עַל-הַלֻּחֹ֗ת אֵ֚ת דִּבְרֵ֣י הַבְּרִ֔ית עֲשֶׂ֖רֶת הַדְּבָרִֽים:  ת  וַהֲוָה תַמָן קֳדָם יְיָ אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן לַחְמָא לָא אֲכַל וּמַיָא לָא שְׁתֵי וּכְתַב עַל לוּחַיָא יָת פִּתְגָמֵי קְיָמָא עַשְׂרָא פִּתְגָמִין:

{כט} וַיְהִ֗י בְּרֶ֤דֶת מֹשֶׁה֙ מֵהַ֣ר סִינַ֔י וּשְׁנֵ֨י לֻחֹ֤ת הָֽעֵדֻת֙ בְּיַד-מֹשֶׁ֔ה בְּרִדְתּ֖וֹ מִן-הָהָ֑ר וּמֹשֶׁ֣ה לֹֽא-יָדַ֗ע כִּ֥י קָרַ֛ן ע֥וֹר פָּנָ֖יו בְּדַבְּר֥וֹ אִתּֽוֹ:  מ  וַיְהִ֗י בְּרֶ֤דֶת מֹשֶׁה֙ מֵהַ֣ר סִינַ֔י וּשְׁנֵ֨י לֻחֹ֤ת הָֽעֵדֻת֙ בְּיַד-מֹשֶׁ֔ה בְּרִדְתּ֖וֹ מִן-הָהָ֑ר וּמֹשֶׁ֣ה לֹֽא-יָדַ֗ע כִּ֥י קָרַ֛ן ע֥וֹר פָּנָ֖יו בְּדַבְּר֥וֹ אִתּֽוֹ:  ת  וַהֲוָה כַּד נְחַת משֶׁה מִטוּרָא דְסִינַי וּתְרֵין לוּחֵי סַהֲדוּתָא בִּידָא דְמשֶׁה בְּמֵחֲתֵהּ מִן טוּרָא וּמשֶׁה לָא יְדַע אֲרֵי סְגֵי זִיו יְקָרָא דְאַפּוֹהִי בְּמַלָלוּתֵהּ עִמֵהּ:

{ל} וַיַּ֨רְא אַֽהֲרֹ֜ן וְכָל-בְּנֵ֤י יִשְׂרָאֵל֙ אֶת-מֹשֶׁ֔ה וְהִנֵּ֥ה קָרַ֖ן ע֣וֹר פָּנָ֑יו וַיִּֽירְא֖וּ מִגֶּ֥שֶׁת אֵלָֽיו:  מ  וַיַּ֨רְא אַֽהֲרֹ֜ן וְכָל-בְּנֵ֤י יִשְׂרָאֵל֙ אֶת-מֹשֶׁ֔ה וְהִנֵּ֥ה קָרַ֖ן ע֣וֹר פָּנָ֑יו וַיִּֽירְא֖וּ מִגֶּ֥שֶׁת אֵלָֽיו:  ת  וַחֲזָא אַהֲרֹן וְכָל בְּנֵי יִשְׂרָאֵל יָת משֶׁה וְהָא סְגֵי זִיו יְקָרָא דְאַפּוֹהִי וּדְחִילוּ מִלְאִתְקְרְבָא לְוָתֵהּ:

{לא} וַיִּקְרָ֤א אֲלֵהֶם֙ מֹשֶׁ֔ה וַיָּשֻׁ֧בוּ אֵלָ֛יו אַֽהֲרֹ֥ן וְכָל-הַנְּשִׂאִ֖ים בָּֽעֵדָ֑ה וַיְדַבֵּ֥ר מֹשֶׁ֖ה אֲלֵהֶֽם:  מ  וַיִּקְרָ֤א אֲלֵהֶם֙ מֹשֶׁ֔ה וַיָּשֻׁ֧בוּ אֵלָ֛יו אַֽהֲרֹ֥ן וְכָל-הַנְּשִׂאִ֖ים בָּֽעֵדָ֑ה וַיְדַבֵּ֥ר מֹשֶׁ֖ה אֲלֵהֶֽם:  ת  וּקְרָא לְהוֹן משֶׁה וְתָבוּ לְוָתֵהּ אַהֲרֹן וְכָל רַבְרְבַיָא בִּכְנִשְׁתָּא וּמַלֵל משֶׁה עִמְהוֹן:

{לב} וְאַֽחֲרֵי-כֵ֥ן נִגְּשׁ֖וּ כָּל-בְּנֵ֣י יִשְׂרָאֵ֑ל וַיְצַוֵּ֕ם אֵת֩ כָּל-אֲשֶׁ֨ר דִּבֶּ֧ר יְהוָֹ֛ה אִתּ֖וֹ בְּהַ֥ר סִינָֽי:  מ  וְאַֽחֲרֵי-כֵ֥ן נִגְּשׁ֖וּ כָּל-בְּנֵ֣י יִשְׂרָאֵ֑ל וַיְצַוֵּ֕ם אֵת֩ כָּל-אֲשֶׁ֨ר דִּבֶּ֧ר יְהוָֹ֛ה אִתּ֖וֹ בְּהַ֥ר סִינָֽי:  ת  וּבָתַר כֵּן אִתְקְרִיבוּ כָּל בְּנֵי יִשְׂרָאֵל וּפַקֵדִנוּן יָת כָּל דִי מַלֵל יְיָ עִמֵהּ בְּטוּרָא דְסִינָי:

{לג}  מפטיר  וַיְכַ֣ל מֹשֶׁ֔ה מִדַּבֵּ֖ר אִתָּ֑ם וַיִּתֵּ֥ן עַל-פָּנָ֖יו מַסְוֶֽה:  מ  וַיְכַ֣ל מֹשֶׁ֔ה מִדַּבֵּ֖ר אִתָּ֑ם וַיִּתֵּ֥ן עַל-פָּנָ֖יו מַסְוֶֽה:  ת  וְשֵׁיצֵי משֶׁה מִלְמַלָלָא עִמְהוֹן וִיהַב עַל אַפּוֹהִי בֵּית אַפֵּי:

{לד} וּבְבֹ֨א מֹשֶׁ֜ה לִפְנֵ֤י יְהוָֹה֙ לְדַבֵּ֣ר אִתּ֔וֹ יָסִ֥יר אֶת-הַמַּסְוֶ֖ה עַד-צֵאת֑וֹ וְיָצָ֗א וְדִבֶּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל אֵ֖ת אֲשֶׁ֥ר יְצֻוֶּֽה:  מ  וּבְבֹ֨א מֹשֶׁ֜ה לִפְנֵ֤י יְהוָֹה֙ לְדַבֵּ֣ר אִתּ֔וֹ יָסִ֥יר אֶת-הַמַּסְוֶ֖ה עַד-צֵאת֑וֹ וְיָצָ֗א וְדִבֶּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל אֵ֖ת אֲשֶׁ֥ר יְצֻוֶּֽה:  ת  וְכַד עָלֵל משֶׁה לִקֳדָם יְיָ לְמַלָלָא עִמֵה אַעְדִי יָת בֵּית אַפּוֹהִי עַד מִפְקֵהּ וְנָפֵק וּמְמַלֵל עִם בְּנֵי יִשְׂרָאֵל יָת דִי מִתְפַּקָד:

{לה} וְרָא֤וּ בְנֵֽי-יִשְׂרָאֵל֙ אֶת-פְּנֵ֣י מֹשֶׁ֔ה כִּ֣י קָרַ֔ן ע֖וֹר פְּנֵ֣י מֹשֶׁ֑ה וְהֵשִׁ֨יב מֹשֶׁ֤ה אֶת-הַמַּסְוֶה֙ עַל-פָּנָ֔יו עַד-בֹּא֖וֹ לְדַבֵּ֥ר אִתּֽוֹ:  מ  וְרָא֤וּ בְנֵֽי-יִשְׂרָאֵל֙ אֶת-פְּנֵ֣י מֹשֶׁ֔ה כִּ֣י קָרַ֔ן ע֖וֹר פְּנֵ֣י מֹשֶׁ֑ה וְהֵשִׁ֨יב מֹשֶׁ֤ה אֶת-הַמַּסְוֶה֙ עַל-פָּנָ֔יו עַד-בֹּא֖וֹ לְדַבֵּ֥ר אִתּֽוֹ:  ת  וַחֲזָן בְּנֵי יִשְׂרָאֵל יָת אַפֵּי משֶׁה אֲרֵי סְגֵי זִיו יְקָרָא דְאַפֵי משֶׁה וַאֲתֵב משֶׁה יָת בֵּית אַפֵּי עַל אַפּוֹהִי עַד דְעָלֵל לְמַלָלָא עִמֵהּ: ססס: (ססס)
