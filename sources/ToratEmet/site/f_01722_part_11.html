פרשת ויגש


{יח} וַיִּגַּ֨שׁ אֵלָ֜יו יְהוּדָ֗ה וַיֹּ֘אמֶר֘ בִּ֣י אֲדֹנִי֒ יְדַבֶּר-נָ֨א עַבְדְּךָ֤ דָבָר֙ בְּאָזְנֵ֣י אֲדֹנִ֔י וְאַל-יִ֥חַר אַפְּךָ֖ בְּעַבְדֶּ֑ךָ כִּ֥י כָמ֖וֹךָ כְּפַרְעֹֽה:  מ  וַיִּגַּ֨שׁ אֵלָ֜יו יְהוּדָ֗ה וַיֹּ֘אמֶר֘ בִּ֣י אֲדֹנִי֒ יְדַבֶּר-נָ֨א עַבְדְּךָ֤ דָבָר֙ בְּאָזְנֵ֣י אֲדֹנִ֔י וְאַל-יִ֥חַר אַפְּךָ֖ בְּעַבְדֶּ֑ךָ כִּ֥י כָמ֖וֹךָ כְּפַרְעֹֽה:  ת  וּקְרֵב לְוָתֵהּ יְהוּדָה וַאֲמַר בְּבָעוּ רִבּוֹנִי יְמַלֵּל כְּעַן עַבְדָּךְ פִּתְגָּמָא קֳדָם רִבּוֹנִי וְלָא יִתְקֵף רוּגְזָךְ בְּעַבְדָּךְ אֲרֵי כְפַרְעֹה כֵּן אָתְּ:

{יט} אֲדֹנִ֣י שָׁאַ֔ל אֶת-עֲבָדָ֖יו לֵאמֹ֑ר הֲיֵשׁ-לָכֶ֥ם אָ֖ב אוֹ-אָֽח:  מ  אֲדֹנִ֣י שָׁאַ֔ל אֶת-עֲבָדָ֖יו לֵאמֹ֑ר הֲיֵשׁ-לָכֶ֥ם אָ֖ב אוֹ-אָֽח:  ת  רִבּוֹנִי שְׁאִיל יָת עַבְדּוֹהִי לְמֵימָר הַאִית לְכוֹן אַבָּא אוֹ אָחָא:

{כ} וַנֹּ֨אמֶר֙ אֶל-אֲדֹנִ֔י יֶשׁ-לָ֨נוּ֙ אָ֣ב זָקֵ֔ן וְיֶ֥לֶד זְקֻנִ֖ים קָטָ֑ן וְאָחִ֣יו מֵ֔ת וַיִּוָּתֵ֨ר ה֧וּא לְבַדּ֛וֹ לְאִמּ֖וֹ וְאָבִ֥יו אֲהֵבֽוֹ:  מ  וַנֹּ֨אמֶר֙ אֶל-אֲדֹנִ֔י יֶשׁ-לָ֨נוּ֙ אָ֣ב זָקֵ֔ן וְיֶ֥לֶד זְקֻנִ֖ים קָטָ֑ן וְאָחִ֣יו מֵ֔ת וַיִּוָּתֵ֨ר ה֧וּא לְבַדּ֛וֹ לְאִמּ֖וֹ וְאָבִ֥יו אֲהֵבֽוֹ:  ת  וַאֲמַרְנָא לְרִבּוֹנִי אִית לָנָא אַבָּא סָבָא וּבַר סִיבְתִין זְעֵיר וְאָחוּהִי מִית וְאִשְׁתָּאַר הוּא בִלְחוֹדוֹהִי לְאִמֵּהּ וַאֲבוּהִי רְחִים לֵהּ:

{כא} וַתֹּ֨אמֶר֙ אֶל-עֲבָדֶ֔יךָ הֽוֹרִדֻ֖הוּ אֵלָ֑י וְאָשִׂ֥ימָה עֵינִ֖י עָלָֽיו:  מ  וַתֹּ֨אמֶר֙ אֶל-עֲבָדֶ֔יךָ הֽוֹרִדֻ֖הוּ אֵלָ֑י וְאָשִׂ֥ימָה עֵינִ֖י עָלָֽיו:  ת  וַאֲמַרְתָּ לְעַבְדָּיךְ אֲחִיתוֹהִי לְוָתִי וֶאֱשַׁוֵּי עֵינִי עֲלוֹהִי:

{כב} וַנֹּ֨אמֶר֙ אֶל-אֲדֹנִ֔י לֹֽא-יוּכַ֥ל הַנַּ֖עַר לַֽעֲזֹ֣ב אֶת-אָבִ֑יו וְעָזַ֥ב אֶת-אָבִ֖יו וָמֵֽת:  מ  וַנֹּ֨אמֶר֙ אֶל-אֲדֹנִ֔י לֹֽא-יוּכַ֥ל הַנַּ֖עַר לַֽעֲזֹ֣ב אֶת-אָבִ֑יו וְעָזַ֥ב אֶת-אָבִ֖יו וָמֵֽת:  ת  וַאֲמַרְנָא לְרִבּוֹנִי לָא יִכּוּל עוּלֵימָא לְמִשְׁבַּק יָת אֲבוּהִי וְאִם יִשְׁבּוֹק יָת אֲבוּהִי וּמִית:

{כג} וַתֹּ֨אמֶר֙ אֶל-עֲבָדֶ֔יךָ אִם-לֹ֥א יֵרֵ֛ד אֲחִיכֶ֥ם הַקָּטֹ֖ן אִתְּכֶ֑ם לֹ֥א תֹֽסִפ֖וּן לִרְא֥וֹת פָּנָֽי:  מ  וַתֹּ֨אמֶר֙ אֶל-עֲבָדֶ֔יךָ אִם-לֹ֥א יֵרֵ֛ד אֲחִיכֶ֥ם הַקָּטֹ֖ן אִתְּכֶ֑ם לֹ֥א תֹֽסִפ֖וּן לִרְא֥וֹת פָּנָֽי:  ת  וַאֲמַרְתָּ לְעַבְדָּיךְ אִם לָא יֵחוֹת אֲחוּכוֹן זְעֵירָא עִמְּכוֹן לָא תוֹסְפוּן לְמֶחֱזֵי אַפָּי:

{כד} וַֽיְהִי֙ כִּ֣י עָלִ֔ינוּ אֶֽל-עַבְדְּךָ֖ אָבִ֑י וַנַּ֨גֶּד-ל֔וֹ אֵ֖ת דִּבְרֵ֥י אֲדֹנִֽי:  מ  וַֽיְהִי֙ כִּ֣י עָלִ֔ינוּ אֶֽל-עַבְדְּךָ֖ אָבִ֑י וַנַּ֨גֶּד-ל֔וֹ אֵ֖ת דִּבְרֵ֥י אֲדֹנִֽי:  ת  וַהֲוָה כַּד סְלֵיקְנָא לְעַבְדָּךְ אַבָּא וְחַוֵּינָא לֵהּ יָת פִּתְגָּמֵי רִבּוֹנִי:

{כה} וַיֹּ֖אמֶר אָבִ֑ינוּ שֻׁ֖בוּ שִׁבְרוּ-לָ֥נוּ מְעַט-אֹֽכֶל:  מ  וַיֹּ֖אמֶר אָבִ֑ינוּ שֻׁ֖בוּ שִׁבְרוּ-לָ֥נוּ מְעַט-אֹֽכֶל:  ת  וַאֲמַר אָבוּנָא תּוּבוּ זְבוּנוּ לָנָא זְעֵיר עִיבוּרָא:

{כו} וַנֹּ֕אמֶר לֹ֥א נוּכַ֖ל לָרֶ֑דֶת אִם-יֵשׁ֩ אָחִ֨ינוּ הַקָּטֹ֤ן אִתָּ֨נוּ֙ וְיָרַ֔דְנוּ כִּי-לֹ֣א נוּכַ֗ל לִרְאוֹת֙ פְּנֵ֣י הָאִ֔ישׁ וְאָחִ֥ינוּ הַקָּטֹ֖ן אֵינֶ֥נּוּ אִתָּֽנוּ:  מ  וַנֹּ֕אמֶר לֹ֥א נוּכַ֖ל לָרֶ֑דֶת אִם-יֵשׁ֩ אָחִ֨ינוּ הַקָּטֹ֤ן אִתָּ֨נוּ֙ וְיָרַ֔דְנוּ כִּי-לֹ֣א נוּכַ֗ל לִרְאוֹת֙ פְּנֵ֣י הָאִ֔ישׁ וְאָחִ֥ינוּ הַקָּטֹ֖ן אֵינֶ֥נּוּ אִתָּֽנוּ:  ת  וַאֲמַרְנָא לָא נִכּוּל לְמֵיחָת אִם אִית אָחוּנָא זְעֵירָא עִמָּנָא וְנֵחוֹת אֲרֵי לָא נִכּוּל לְמֶחֱזֵי אַפֵּי גַּבְרָא וְאָחוּנָא זְעֵירָא לֵיתוֹהִי עִמָּנָא:

{כז} וַיֹּ֛אמֶר עַבְדְּךָ֥ אָבִ֖י אֵלֵ֑ינוּ אַתֶּ֣ם יְדַעְתֶּ֔ם כִּ֥י שְׁנַ֖יִם יָֽלְדָה-לִּ֥י אִשְׁתִּֽי:  מ  וַיֹּ֛אמֶר עַבְדְּךָ֥ אָבִ֖י אֵלֵ֑ינוּ אַתֶּ֣ם יְדַעְתֶּ֔ם כִּ֥י שְׁנַ֖יִם יָֽלְדָה-לִּ֥י אִשְׁתִּֽי:  ת  וַאֲמַר עַבְדָּךְ אַבָּא לָנָא אַתּוּן יְדַעְתּוּן אֲרֵי תְרֵין יְלִידַת לִי אִתְּתִי:

{כח} וַיֵּצֵ֤א הָֽאֶחָד֙ מֵֽאִתִּ֔י וָֽאֹמַ֕ר אַ֖ךְ טָרֹ֣ף טֹרָ֑ף וְלֹ֥א רְאִיתִ֖יו עַד-הֵֽנָּה:  מ  וַיֵּצֵ֤א הָֽאֶחָד֙ מֵֽאִתִּ֔י וָֽאֹמַ֕ר אַ֖ךְ טָרֹ֣ף טֹרָ֑ף וְלֹ֥א רְאִיתִ֖יו עַד-הֵֽנָּה:  ת  וּנְפַק חַד מִלְוָתִי וַאֲמָרִית בְּרַם מִקְטַל קְטִיל וְלָא חֲזִתֵּהּ עַד כְּעָן:

{כט} וּלְקַחְתֶּ֧ם גַּם-אֶת-זֶ֛ה מֵעִ֥ם פָּנַ֖י וְקָרָ֣הוּ אָס֑וֹן וְהֽוֹרַדְתֶּ֧ם אֶת-שֵֽׂיבָתִ֛י בְּרָעָ֖ה שְׁאֹֽלָה:  מ  וּלְקַחְתֶּ֧ם גַּם-אֶת-זֶ֛ה מֵעִ֥ם פָּנַ֖י וְקָרָ֣הוּ אָס֑וֹן וְהֽוֹרַדְתֶּ֧ם אֶת-שֵֽׂיבָתִ֛י בְּרָעָ֖ה שְׁאֹֽלָה:  ת  וְתַדְבְּרוּן אַף יָת דֵּין מִן קֳדָמַי וִיעַרְעִנֵּהּ מוֹתָא וְתַחֲתוּן יָת שִׂבְתִי בְּבִישְׁתָא לִשְׁאוֹל:

{ל} וְעַתָּ֗ה כְּבֹאִי֙ אֶל-עַבְדְּךָ֣ אָבִ֔י וְהַנַּ֖עַר אֵינֶ֣נּוּ אִתָּ֑נוּ וְנַפְשׁ֖וֹ קְשׁוּרָ֥ה בְנַפְשֽׁוֹ:  מ  וְעַתָּ֗ה כְּבֹאִי֙ אֶל-עַבְדְּךָ֣ אָבִ֔י וְהַנַּ֖עַר אֵינֶ֣נּוּ אִתָּ֑נוּ וְנַפְשׁ֖וֹ קְשׁוּרָ֥ה בְנַפְשֽׁוֹ:  ת  וּכְעַן כְּמֵיתִי לְוַת עַבְדָּךְ אַבָּא וְעוּלֵימָא לֵיתוֹהִי עִמָּנָא וְנַפְשֵׁהּ חֲבִיבָא לֵהּ כְּנַפְשֵׁהּ:

{לא}  שני  וְהָיָ֗ה כִּרְאוֹת֛וֹ כִּי-אֵ֥ין הַנַּ֖עַר וָמֵ֑ת וְהוֹרִ֨ידוּ עֲבָדֶ֜יךָ אֶת-שֵׂיבַ֨ת עַבְדְּךָ֥ אָבִ֛ינוּ בְּיָג֖וֹן שְׁאֹֽלָה:  מ  וְהָיָ֗ה כִּרְאוֹת֛וֹ כִּי-אֵ֥ין הַנַּ֖עַר וָמֵ֑ת וְהוֹרִ֨ידוּ עֲבָדֶ֜יךָ אֶת-שֵׂיבַ֨ת עַבְדְּךָ֥ אָבִ֛ינוּ בְּיָג֖וֹן שְׁאֹֽלָה:  ת  וִיהֵי כַּד יֶחֱזֵי אֲרֵי לֵית עוּלֵימָא וּמִית וְיַחֲתוּן עַבְדָּיךְ יָת שִׂיבַת עַבְדָּךְ אָבוּנָא בְּדָווֹנָא לִשְׁאוֹל:

{לב} כִּ֤י עַבְדְּךָ֙ עָרַ֣ב אֶת-הַנַּ֔עַר מֵעִ֥ם אָבִ֖י לֵאמֹ֑ר אִם-לֹ֤א אֲבִיאֶ֨נּוּ֙ אֵלֶ֔יךָ וְחָטָ֥אתִי לְאָבִ֖י כָּל-הַיָּמִֽים:  מ  כִּ֤י עַבְדְּךָ֙ עָרַ֣ב אֶת-הַנַּ֔עַר מֵעִ֥ם אָבִ֖י לֵאמֹ֑ר אִם-לֹ֤א אֲבִיאֶ֨נּוּ֙ אֵלֶ֔יךָ וְחָטָ֥אתִי לְאָבִ֖י כָּל-הַיָּמִֽים:  ת  אֲרֵי עַבְדָּךְ מְעָרַב בְּעוּלֵימָא מִן אַבָּא לְמֵימָר אִם לָא אַיְתִנֵּהּ לְוָתָךְ וָאֱהִי חָטֵי לְאַבָּא כָּל יוֹמַיָּא:

{לג} וְעַתָּ֗ה יֵֽשֶׁב-נָ֤א עַבְדְּךָ֙ תַּ֣חַת הַנַּ֔עַר עֶ֖בֶד לַֽאדֹנִ֑י וְהַנַּ֖עַר יַ֥עַל עִם-אֶחָֽיו:  מ  וְעַתָּ֗ה יֵֽשֶׁב-נָ֤א עַבְדְּךָ֙ תַּ֣חַת הַנַּ֔עַר עֶ֖בֶד לַֽאדֹנִ֑י וְהַנַּ֖עַר יַ֥עַל עִם-אֶחָֽיו:  ת  וּכְעַן יֵתֶב כְּעַן עַבְדָּךְ תְּחוֹת עוּלֵימָא עַבְדָּא לְרִבּוֹנִי וְעוּלֵימָא יִסַּק עִם אֲחוֹהִי:

{לד} כִּי-אֵיךְ֙ אֶֽעֱלֶ֣ה אֶל-אָבִ֔י וְהַנַּ֖עַר אֵינֶנּ֣וּ אִתִּ֑י פֶּ֚ן אֶרְאֶ֣ה בָרָ֔ע אֲשֶׁ֥ר יִמְצָ֖א אֶת-אָבִֽי:  מ  כִּי-אֵיךְ֙ אֶֽעֱלֶ֣ה אֶל-אָבִ֔י וְהַנַּ֖עַר אֵינֶנּ֣וּ אִתִּ֑י פֶּ֚ן אֶרְאֶ֣ה בָרָ֔ע אֲשֶׁ֥ר יִמְצָ֖א אֶת-אָבִֽי:  ת  אֲרֵי אֶכְדֵּין אֶסַּק לְוַת אַבָּא וְעוּלֵימָא לֵיתוֹהִי עִמִּי דִּלְמָא אֶחֱזֵי בְּבִישׁוּ דִּי יִשְׁכַּח יָת אַבָּא:




בראשית פרק-מה

{א} וְלֹֽא-יָכֹ֨ל יוֹסֵ֜ף לְהִתְאַפֵּ֗ק לְכֹ֤ל הַנִּצָּבִים֙ עָלָ֔יו וַיִּקְרָ֕א הוֹצִ֥יאוּ כָל-אִ֖ישׁ מֵֽעָלָ֑י וְלֹא-עָ֤מַד אִישׁ֙ אִתּ֔וֹ בְּהִתְוַדַּ֥ע יוֹסֵ֖ף אֶל-אֶחָֽיו:  מ  וְלֹֽא-יָכֹ֨ל יוֹסֵ֜ף לְהִתְאַפֵּ֗ק לְכֹ֤ל הַנִּצָּבִים֙ עָלָ֔יו וַיִּקְרָ֕א הוֹצִ֥יאוּ כָל-אִ֖ישׁ מֵֽעָלָ֑י וְלֹא-עָ֤מַד אִישׁ֙ אִתּ֔וֹ בְּהִתְוַדַּ֥ע יוֹסֵ֖ף אֶל-אֶחָֽיו:  ת  וְלָא יְכִיל יוֹסֵף לְאִתְחַסָּנָא לְכֹל דְּקָיְמִין עִלָּוֹהִי וּקְרָא אַפִּיקוּ כָל אֱנַשׁ מֵעִלָּוָי וְלָא קָם אֱנַשׁ עִמֵּהּ כַּד אִתְיְדַע יוֹסֵף לְוַת אֲחוֹהִי:

{ב} וַיִּתֵּ֥ן אֶת-קֹל֖וֹ בִּבְכִ֑י וַיִּשְׁמְע֣וּ מִצְרַ֔יִם וַיִּשְׁמַ֖ע בֵּ֥ית פַּרְעֹֽה:  מ  וַיִּתֵּ֥ן אֶת-קֹל֖וֹ בִּבְכִ֑י וַיִּשְׁמְע֣וּ מִצְרַ֔יִם וַיִּשְׁמַ֖ע בֵּ֥ית פַּרְעֹֽה:  ת  וִיהַב יָת קָלֵהּ בִּבְכִיתָא וּשְׁמָעוּ מִצְרַיִם וּשְׁמַע אֱנַשׁ בֵּית פַּרְעֹה:

{ג} וַיֹּ֨אמֶר יוֹסֵ֤ף אֶל-אֶחָיו֙ אֲנִ֣י יוֹסֵ֔ף הַע֥וֹד אָבִ֖י חָ֑י וְלֹא-יָֽכְל֤וּ אֶחָיו֙ לַֽעֲנ֣וֹת אֹת֔וֹ כִּ֥י נִבְהֲל֖וּ מִפָּנָֽיו:  מ  וַיֹּ֨אמֶר יוֹסֵ֤ף אֶל-אֶחָיו֙ אֲנִ֣י יוֹסֵ֔ף הַע֥וֹד אָבִ֖י חָ֑י וְלֹא-יָֽכְל֤וּ אֶחָיו֙ לַֽעֲנ֣וֹת אֹת֔וֹ כִּ֥י נִבְהֲל֖וּ מִפָּנָֽיו:  ת  וַאֲמַר יוֹסֵף לַאֲחוֹהִי אֲנָא יוֹסֵף הַעַד כְּעַן אַבָּא קַיָּם וְלָא יָכִילוּ אֲחוֹהִי לַאֲתָבָא יָתֵהּ פִּתְגָּם אֳרֵי אִתְבְּהִילוּ מִן קֳדָמוֹהִי:

{ד} וַיֹּ֨אמֶר יוֹסֵ֧ף אֶל-אֶחָ֛יו גְּשׁוּ-נָ֥א אֵלַ֖י וַיִּגָּ֑שׁוּ וַיֹּ֗אמֶר אֲנִי֙ יוֹסֵ֣ף אֲחִיכֶ֔ם אֲשֶׁר-מְכַרְתֶּ֥ם אֹתִ֖י מִצְרָֽיְמָה:  מ  וַיֹּ֨אמֶר יוֹסֵ֧ף אֶל-אֶחָ֛יו גְּשׁוּ-נָ֥א אֵלַ֖י וַיִּגָּ֑שׁוּ וַיֹּ֗אמֶר אֲנִי֙ יוֹסֵ֣ף אֲחִיכֶ֔ם אֲשֶׁר-מְכַרְתֶּ֥ם אֹתִ֖י מִצְרָֽיְמָה:  ת  וַאֲמַר יוֹסֵף לַאֲחוֹהִי קְרִיבוּ כְעַן לְוָתִי וּקְרִיבוּ וַאֲמַר אֲנָא יוֹסֵף אֲחוּכוֹן דִּי זַבִּנְתּוּן יָתִי לְמִצְרָיִם:

{ה} וְעַתָּ֣ה | אַל-תֵּעָ֣צְב֗וּ וְאַל-יִ֨חַר֙ בְּעֵ֣ינֵיכֶ֔ם כִּֽי-מְכַרְתֶּ֥ם אֹתִ֖י הֵ֑נָּה כִּ֣י לְמִֽחְיָ֔ה שְׁלָחַ֥נִי אֱלֹהִ֖ים לִפְנֵיכֶֽם:  מ  וְעַתָּ֣ה | אַל-תֵּעָ֣צְב֗וּ וְאַל-יִ֨חַר֙ בְּעֵ֣ינֵיכֶ֔ם כִּֽי-מְכַרְתֶּ֥ם אֹתִ֖י הֵ֑נָּה כִּ֣י לְמִֽחְיָ֔ה שְׁלָחַ֥נִי אֱלֹהִ֖ים לִפְנֵיכֶֽם:  ת  וּכְעַן לָא תִתְנַסְסוּן וְלָא יִתְקָף בְּעֵינֵיכוֹן אֲרֵי זַבִּנְתּוּן יָתִי הַלְכָא אֲרֵי לְקַיָּמָא שַׁלְחַנִּי יְיָ קֳדָמֵיכוֹן:

{ו} כִּי-זֶ֛ה שְׁנָתַ֥יִם הָֽרָעָ֖ב בְּקֶ֣רֶב הָאָ֑רֶץ וְעוֹד֙ חָמֵ֣שׁ שָׁנִ֔ים אֲשֶׁ֥ר אֵין-חָרִ֖ישׁ וְקָצִֽיר:  מ  כִּי-זֶ֛ה שְׁנָתַ֥יִם הָֽרָעָ֖ב בְּקֶ֣רֶב הָאָ֑רֶץ וְעוֹד֙ חָמֵ֣שׁ שָׁנִ֔ים אֲשֶׁ֥ר אֵין-חָרִ֖ישׁ וְקָצִֽיר:  ת  אֲרֵי דְנַן תַּרְתֵּין שְׁנִין כַּפְנָא בְּגוֹ אַרְעָא וְעוֹד חָמֵשׁ שְׁנִין דִּי לֵית זְרוֹעָא וַחֲצָדָא:

{ז} וַיִּשְׁלָחֵ֤נִי אֱלֹהִים֙ לִפְנֵיכֶ֔ם לָשׂ֥וּם לָכֶ֛ם שְׁאֵרִ֖ית בָּאָ֑רֶץ וּלְהַֽחֲי֣וֹת לָכֶ֔ם לִפְלֵיטָ֖ה גְּדֹלָֽה:  מ  וַיִּשְׁלָחֵ֤נִי אֱלֹהִים֙ לִפְנֵיכֶ֔ם לָשׂ֥וּם לָכֶ֛ם שְׁאֵרִ֖ית בָּאָ֑רֶץ וּלְהַֽחֲי֣וֹת לָכֶ֔ם לִפְלֵיטָ֖ה גְּדֹלָֽה:  ת  וְשַׁלְחַנִּי יְיָ קֳדָמֵיכוֹן לְשַׁוָּאָה לְכוֹן שְׁאָרָא בְּאַרְעָא וּלְקַיָּמָא לְכוֹן לְשֵׁיזָבָא רַבָּא:

{ח}  שלישי  וְעַתָּ֗ה לֹֽא-אַתֶּ֞ם שְׁלַחְתֶּ֤ם אֹתִי֙ הֵ֔נָּה כִּ֖י הָֽאֱלֹהִ֑ים וַיְשִׂימֵ֨נִי לְאָ֜ב לְפַרְעֹ֗ה וּלְאָדוֹן֙ לְכָל-בֵּית֔וֹ וּמשֵׁ֖ל בְּכָל-אֶ֥רֶץ מִצְרָֽיִם:  מ  וְעַתָּ֗ה לֹֽא-אַתֶּ֞ם שְׁלַחְתֶּ֤ם אֹתִי֙ הֵ֔נָּה כִּ֖י הָֽאֱלֹהִ֑ים וַיְשִׂימֵ֨נִי לְאָ֜ב לְפַרְעֹ֗ה וּלְאָדוֹן֙ לְכָל-בֵּית֔וֹ וּמשֵׁ֖ל בְּכָל-אֶ֥רֶץ מִצְרָֽיִם:  ת  וּכְעַן לָא אַתּוּן שְׁלַחְתּוּן יָתִי הַלְכָא אֶלָּהֵן מִן קֳדָם יְיָ וְשַׁוְּיַנִי לְאַבָּא לְפַרְעֹה וּלְרִבּוֹן לְכָל אֱנַשׁ בֵּיתֵהּ וְשַׁלִּיט בְּכָל אַרְעָא דְמִצְרָיִם:

{ט} מַֽהֲרוּ֘ וַֽעֲל֣וּ אֶל-אָבִי֒ וַֽאֲמַרְתֶּ֣ם אֵלָ֗יו כֹּ֤ה אָמַר֙ בִּנְךָ֣ יוֹסֵ֔ף שָׂמַ֧נִי אֱלֹהִ֛ים לְאָד֖וֹן לְכָל-מִצְרָ֑יִם רְדָ֥ה אֵלַ֖י אַל-תַּֽעֲמֹֽד:  מ  מַֽהֲרוּ֘ וַֽעֲל֣וּ אֶל-אָבִי֒ וַֽאֲמַרְתֶּ֣ם אֵלָ֗יו כֹּ֤ה אָמַר֙ בִּנְךָ֣ יוֹסֵ֔ף שָׂמַ֧נִי אֱלֹהִ֛ים לְאָד֖וֹן לְכָל-מִצְרָ֑יִם רְדָ֥ה אֵלַ֖י אַל-תַּֽעֲמֹֽד:  ת  אוֹחוּ וּסְקוֹ לְוַת אַבָּא וְתֵימְרוּן לֵהּ כִּדְנַן אֲמַר בְּרָךְ יוֹסֵף שַׁוְּיַנִי יְיָ לְרִבּוֹן לְכָל מִצְרָיִם חוּת לְוָתִי לָא תִתְעַכָּב:

{י} וְיָֽשַׁבְתָּ֣ בְאֶֽרֶץ-גּ֗שֶׁן וְהָיִ֤יתָ קָרוֹב֙ אֵלַ֔י אַתָּ֕ה וּבָנֶ֖יךָ וּבְנֵ֣י בָנֶ֑יךָ וְצֹֽאנְךָ֥ וּבְקָֽרְךָ֖ וְכָל-אֲשֶׁר-לָֽךְ:  מ  וְיָֽשַׁבְתָּ֣ בְאֶֽרֶץ-גּ֗שֶׁן וְהָיִ֤יתָ קָרוֹב֙ אֵלַ֔י אַתָּ֕ה וּבָנֶ֖יךָ וּבְנֵ֣י בָנֶ֑יךָ וְצֹֽאנְךָ֥ וּבְקָֽרְךָ֖ וְכָל-אֲשֶׁר-לָֽךְ:  ת  וְתֵתֵב בְּאַרְעָא דְגשֶׁן וּתְהֵי קָרִיב לִי אַתְּ וּבְנָיךְ וּבְנֵי בְנָיךְ וְעָנָךְ וְתוֹרָךְ וְכָל דִּי לָךְ:

{יא} וְכִלְכַּלְתִּ֤י אֹֽתְךָ֙ שָׁ֔ם כִּי-ע֛וֹד חָמֵ֥שׁ שָׁנִ֖ים רָעָ֑ב פֶּן-תִּוָּרֵ֛שׁ אַתָּ֥ה וּבֵֽיתְךָ֖ וְכָל-אֲשֶׁר-לָֽךְ:  מ  וְכִלְכַּלְתִּ֤י אֹֽתְךָ֙ שָׁ֔ם כִּי-ע֛וֹד חָמֵ֥שׁ שָׁנִ֖ים רָעָ֑ב פֶּן-תִּוָּרֵ֛שׁ אַתָּ֥ה וּבֵֽיתְךָ֖ וְכָל-אֲשֶׁר-לָֽךְ:  ת  וְאֵזוּן יָתָךְ תַּמָּן אֲרֵי עוֹד חֲמֵשׁ שְׁנִין כַּפְנָא דִּלְמָא תִתְמַסְכַּן אַתְּ וֶאֱנַשׁ בֵּיתָךְ וְכָל דִּי לָךְ:

{יב} וְהִנֵּ֤ה עֵֽינֵיכֶם֙ רֹא֔וֹת וְעֵינֵ֖י אָחִ֣י בִנְיָמִ֑ין כִּי-פִ֖י הַֽמְדַבֵּ֥ר אֲלֵיכֶֽם:  מ  וְהִנֵּ֤ה עֵֽינֵיכֶם֙ רֹא֔וֹת וְעֵינֵ֖י אָחִ֣י בִנְיָמִ֑ין כִּי-פִ֖י הַֽמְדַבֵּ֥ר אֲלֵיכֶֽם:  ת  וְהָא עֵינֵיכוֹן חֲזָן וְעֵינֵי אָחִי בִנְיָמִין אֲרֵי בְלִישַׁנְכוֹן אֲנָא מְמַלֵּל עִמְּכוֹן:

{יג} וְהִגַּדְתֶּ֣ם לְאָבִ֗י אֶת-כָּל-כְּבוֹדִי֙ בְּמִצְרַ֔יִם וְאֵ֖ת כָּל-אֲשֶׁ֣ר רְאִיתֶ֑ם וּמִֽהַרְתֶּ֛ם וְהֽוֹרַדְתֶּ֥ם אֶת-אָבִ֖י הֵֽנָּה:  מ  וְהִגַּדְתֶּ֣ם לְאָבִ֗י אֶת-כָּל-כְּבוֹדִי֙ בְּמִצְרַ֔יִם וְאֵ֖ת כָּל-אֲשֶׁ֣ר רְאִיתֶ֑ם וּמִֽהַרְתֶּ֛ם וְהֽוֹרַדְתֶּ֥ם אֶת-אָבִ֖י הֵֽנָּה:  ת  וּתְחַווּן לְאַבָּא יָת כָּל יְקָרִי בְּמִצְרַיִם וְיָת כָּל דִּי חֲזֵיתוּן וְתוֹחוּן וְתַחֲתוּן יָת אַבָּא הָכָא:

{יד} וַיִּפֹּ֛ל עַל-צַוְּארֵ֥י בִנְיָֽמִן-אָחִ֖יו וַיֵּ֑בְךְּ וּבִ֨נְיָמִ֔ן בָּכָ֖ה עַל-צַוָּארָֽיו:  מ  וַיִּפֹּ֛ל עַל-צַוְּארֵ֥י בִנְיָֽמִן-אָחִ֖יו וַיֵּ֑בְךְּ וּבִ֨נְיָמִ֔ן בָּכָ֖ה עַל-צַוָּארָֽיו:  ת  וּנְפַל עַל צַוְּארֵי בִנְיָמִן אֲחוּהִי וּבְכָא וּבִנְיָמִן בְּכָא עַל צַוְּארֵהּ:

{טו} וַיְנַשֵּׁ֥ק לְכָל-אֶחָ֖יו וַיֵּ֣בְךְּ עֲלֵהֶ֑ם וְאַ֣חֲרֵי כֵ֔ן דִּבְּר֥וּ אֶחָ֖יו אִתּֽוֹ:  מ  וַיְנַשֵּׁ֥ק לְכָל-אֶחָ֖יו וַיֵּ֣בְךְּ עֲלֵהֶ֑ם וְאַ֣חֲרֵי כֵ֔ן דִּבְּר֥וּ אֶחָ֖יו אִתּֽוֹ:  ת  וּנְשַׁק לְכָל אֲחוֹהִי וּבְכָא עֲלֵיהוֹן וּבָתַר כֵּן מַלִּילוּ אֲחוֹהִי עִמֵּהּ:

{טז} וְהַקֹּ֣ל נִשְׁמַ֗ע בֵּ֤ית פַּרְעֹה֙ לֵאמֹ֔ר בָּ֖אוּ אֲחֵ֣י יוֹסֵ֑ף וַיִּיטַב֙ בְּעֵינֵ֣י פַרְעֹ֔ה וּבְעֵינֵ֖י עֲבָדָֽיו:  מ  וְהַקֹּ֣ל נִשְׁמַ֗ע בֵּ֤ית פַּרְעֹה֙ לֵאמֹ֔ר בָּ֖אוּ אֲחֵ֣י יוֹסֵ֑ף וַיִּיטַב֙ בְּעֵינֵ֣י פַרְעֹ֔ה וּבְעֵינֵ֖י עֲבָדָֽיו:  ת  וְקָלָא אִשְׁתְּמַע בֵית פַּרְעֹה לְמֵימַר עַלּוּ אֲחֵי יוֹסֵף וּשְׁפַר בְּעֵינֵי פַרְעֹה וּבְעֵינֵי עַבְדּוֹהִי:

{יז} וַיֹּ֤אמֶר פַּרְעֹה֙ אֶל-יוֹסֵ֔ף אֱמֹ֥ר אֶל-אַחֶ֖יךָ זֹ֣את עֲשׂ֑וּ טַֽעֲנוּ֙ אֶת-בְּעִ֣ירְכֶ֔ם וּֽלְכוּ-בֹ֖אוּ אַ֥רְצָה כְּנָֽעַן:  מ  וַיֹּ֤אמֶר פַּרְעֹה֙ אֶל-יוֹסֵ֔ף אֱמֹ֥ר אֶל-אַחֶ֖יךָ זֹ֣את עֲשׂ֑וּ טַֽעֲנוּ֙ אֶת-בְּעִ֣ירְכֶ֔ם וּֽלְכוּ-בֹ֖אוּ אַ֥רְצָה כְּנָֽעַן:  ת  וַאֲמַר פַּרְעֹה לְיוֹסֵף אֱמַר לְאַחָיךְ דָּא עִיבִידוּ טְעוּנוּ יָת בְּעִירְכוֹן וֶאֱזִילוּ אוֹבִילוּ לְאַרְעָא דִּכְנָעַן:

{יח} וּקְח֧וּ אֶת-אֲבִיכֶ֛ם וְאֶת-בָּֽתֵּיכֶ֖ם וּבֹ֣אוּ אֵלָ֑י וְאֶתְּנָ֣ה לָכֶ֗ם אֶת-טוּב֙ אֶ֣רֶץ מִצְרַ֔יִם וְאִכְל֖וּ אֶת-חֵ֥לֶב הָאָֽרֶץ:  מ  וּקְח֧וּ אֶת-אֲבִיכֶ֛ם וְאֶת-בָּֽתֵּיכֶ֖ם וּבֹ֣אוּ אֵלָ֑י וְאֶתְּנָ֣ה לָכֶ֗ם אֶת-טוּב֙ אֶ֣רֶץ מִצְרַ֔יִם וְאִכְל֖וּ אֶת-חֵ֥לֶב הָאָֽרֶץ:  ת  וּדְבָרוּ יָת אֲבוּכוֹן וְיָת אֱנַשׁ בָּתֵּיכוֹן וְעוּלוּ לְוָתִי וְאֶתֵּן לְכוֹן יָת טוּב אַרְעָא דְמִצְרַיִם וְתֵיכְלוּן יָת טוּבָא דְאַרְעָא:

{יט}  רביעי  וְאַתָּ֥ה צֻוֵּ֖יתָה זֹ֣את עֲשׂ֑וּ קְחֽוּ-לָכֶם֩ מֵאֶ֨רֶץ מִצְרַ֜יִם עֲגָל֗וֹת לְטַפְּכֶם֙ וְלִנְשֵׁיכֶ֔ם וּנְשָׂאתֶ֥ם אֶת-אֲבִיכֶ֖ם וּבָאתֶֽם:  מ  וְאַתָּ֥ה צֻוֵּ֖יתָה זֹ֣את עֲשׂ֑וּ קְחֽוּ-לָכֶם֩ מֵאֶ֨רֶץ מִצְרַ֜יִם עֲגָל֗וֹת לְטַפְּכֶם֙ וְלִנְשֵׁיכֶ֔ם וּנְשָׂאתֶ֥ם אֶת-אֲבִיכֶ֖ם וּבָאתֶֽם:  ת  וְאַתְּ מְפַקָּדָא דָּא עִיבִידוּ סִיבוּ לְכוֹן מֵאַרְעָא דְמִצְרַיִם עֶגְלָן לְטַפְלֵּיכוֹן וְלִנְשֵׁיכוֹן וְתִטְּלוּן יָת אֲבוּכוֹן וְתֵיתוּן:

{כ} וְעֵ֣ינְכֶ֔ם אַל-תָּחֹ֖ס עַל-כְּלֵיכֶ֑ם כִּי-ט֛וּב כָּל-אֶ֥רֶץ מִצְרַ֖יִם לָכֶ֥ם הֽוּא:  מ  וְעֵ֣ינְכֶ֔ם אַל-תָּחֹ֖ס עַל-כְּלֵיכֶ֑ם כִּי-ט֛וּב כָּל-אֶ֥רֶץ מִצְרַ֖יִם לָכֶ֥ם הֽוּא:  ת  וְעֵינְכוֹן לָא תְחוּס עַל מָנֵיכוֹן אֲרֵי טַב כָּל אַרְעָא דְמִצְרַיִם דִּלְכוֹן הוּא:

{כא} וַיַּֽעֲשׂוּ-כֵן֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיִּתֵּ֨ן לָהֶ֥ם יוֹסֵ֛ף עֲגָל֖וֹת עַל-פִּ֣י פַרְעֹ֑ה וַיִּתֵּ֥ן לָהֶ֛ם צֵדָ֖ה לַדָּֽרֶךְ:  מ  וַיַּֽעֲשׂוּ-כֵן֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיִּתֵּ֨ן לָהֶ֥ם יוֹסֵ֛ף עֲגָל֖וֹת עַל-פִּ֣י פַרְעֹ֑ה וַיִּתֵּ֥ן לָהֶ֛ם צֵדָ֖ה לַדָּֽרֶךְ:  ת  וַעֲבָדוּ כֵן בְּנֵי יִשְׂרָאֵל וִיהַב לְהוֹן יוֹסֵף עֶגְלָן עַל מֵימַר פַּרְעֹה וִיהַב לְהוֹן זְוָדִין לְאָרְחָא:

{כב} לְכֻלָּ֥ם נָתַ֛ן לָאִ֖ישׁ חֲלִפ֣וֹת שְׂמָלֹ֑ת וּלְבִנְיָמִ֤ן נָתַן֙ שְׁלֹ֣שׁ מֵא֣וֹת כֶּ֔סֶף וְחָמֵ֖שׁ חֲלִפֹ֥ת שְׂמָלֹֽת:  מ  לְכֻלָּ֥ם נָתַ֛ן לָאִ֖ישׁ חֲלִפ֣וֹת שְׂמָלֹ֑ת וּלְבִנְיָמִ֤ן נָתַן֙ שְׁלֹ֣שׁ מֵא֣וֹת כֶּ֔סֶף וְחָמֵ֖שׁ חֲלִפֹ֥ת שְׂמָלֹֽת:  ת  לְכָלְהוֹן יְהַב לִגְבַר אִצְטְלַּוָּן דִּלְבוּשִׁין וּלְבִנְיָמִן יְהַב תְּלַת מְאָה סִלְעִין דִּכְסַף וַחֲמֵשׁ אִצְטְלַוָּן דִּלְבוּשִׁין:

{כג} וּלְאָבִ֞יו שָׁלַ֤ח כְּזֹאת֙ עֲשָׂרָ֣ה חֲמֹרִ֔ים נֹֽשְׂאִ֖ים מִטּ֣וּב מִצְרָ֑יִם וְעֶ֣שֶׂר אֲתֹנֹ֡ת נֹֽ֠שְׂאֹ֠ת בָּ֣ר וָלֶ֧חֶם וּמָז֛וֹן לְאָבִ֖יו לַדָּֽרֶךְ:  מ  וּלְאָבִ֞יו שָׁלַ֤ח כְּזֹאת֙ עֲשָׂרָ֣ה חֲמֹרִ֔ים נֹֽשְׂאִ֖ים מִטּ֣וּב מִצְרָ֑יִם וְעֶ֣שֶׂר אֲתֹנֹ֡ת נֹֽ֠שְׂאֹ֠ת בָּ֣ר וָלֶ֧חֶם וּמָז֛וֹן לְאָבִ֖יו לַדָּֽרֶךְ:  ת  וְלַאֲבוּהִי שְׁלַח כְּדָא עַסְרָא חֲמָרִין טְעִינִין מִטּוּבָא דְמִצְרָיִם וַעֲסַר אַתְנָן טְעִינָן עִיבוּר וּלְחֵם וּזְוָדִין לַאֲבוּהִי לְאָרְחָא:

{כד} וַיְשַׁלַּ֥ח אֶת-אֶחָ֖יו וַיֵּלֵ֑כוּ וַיֹּ֣אמֶר אֲלֵהֶ֔ם אַֽל-תִּרְגְּז֖וּ בַּדָּֽרֶךְ:  מ  וַיְשַׁלַּ֥ח אֶת-אֶחָ֖יו וַיֵּלֵ֑כוּ וַיֹּ֣אמֶר אֲלֵהֶ֔ם אַֽל-תִּרְגְּז֖וּ בַּדָּֽרֶךְ:  ת  וְשַׁלַּח יָת אֲחוֹהִי וַאֲזָלוּ וַאֲמַר לְהוֹן לָא תִתְנַצוּן בְּאָרְחָא:

{כה} וַֽיַּֽעֲל֖וּ מִמִּצְרָ֑יִם וַיָּבֹ֨אוּ֙ אֶ֣רֶץ כְּנַ֔עַן אֶֽל-יַֽעֲקֹ֖ב אֲבִיהֶֽם:  מ  וַֽיַּֽעֲל֖וּ מִמִּצְרָ֑יִם וַיָּבֹ֨אוּ֙ אֶ֣רֶץ כְּנַ֔עַן אֶֽל-יַֽעֲקֹ֖ב אֲבִיהֶֽם:  ת  וּסְלִיקוּ מִמִּצְרָיִם וַאֲתוֹ לְאַרְעָא דִּכְנַעַן לְוַת יַעֲקֹב אֲבוּהוֹן:

{כו} וַיַּגִּ֨דוּ ל֜וֹ לֵאמֹ֗ר ע֚וֹד יוֹסֵ֣ף חַ֔י וְכִי-ה֥וּא משֵׁ֖ל בְּכָל-אֶ֣רֶץ מִצְרָ֑יִם וַיָּ֣פָג לִבּ֔וֹ כִּ֥י לֹא-הֶֽאֱמִ֖ין לָהֶֽם:  מ  וַיַּגִּ֨דוּ ל֜וֹ לֵאמֹ֗ר ע֚וֹד יוֹסֵ֣ף חַ֔י וְכִי-ה֥וּא משֵׁ֖ל בְּכָל-אֶ֣רֶץ מִצְרָ֑יִם וַיָּ֣פָג לִבּ֔וֹ כִּ֥י לֹא-הֶֽאֱמִ֖ין לָהֶֽם:  ת  וְחַוִּיאוּ לֵהּ לְמֵימַר עַד כְּעַן יוֹסֵף קַיָּם וַאֲרֵי הוּא שַׁלִּיט בְּכָל אַרְעָא דְמִצְרָיִם וַהֲווֹ מִלַּיָּא פָּיְגָן עַל לִבֵּהּ אֲרֵי לָא הֵימִין לְהוֹן:

{כז} וַיְדַבְּר֣וּ אֵלָ֗יו אֵ֣ת כָּל-דִּבְרֵ֤י יוֹסֵף֙ אֲשֶׁ֣ר דִּבֶּ֣ר אֲלֵהֶ֔ם וַיַּרְא֙ אֶת-הָ֣עֲגָל֔וֹת אֲשֶׁר-שָׁלַ֥ח יוֹסֵ֖ף לָשֵׂ֣את אֹת֑וֹ וַתְּחִ֕י ר֖וּחַ יַֽעֲקֹ֥ב אֲבִיהֶֽם:  מ  וַיְדַבְּר֣וּ אֵלָ֗יו אֵ֣ת כָּל-דִּבְרֵ֤י יוֹסֵף֙ אֲשֶׁ֣ר דִּבֶּ֣ר אֲלֵהֶ֔ם וַיַּרְא֙ אֶת-הָ֣עֲגָל֔וֹת אֲשֶׁר-שָׁלַ֥ח יוֹסֵ֖ף לָשֵׂ֣את אֹת֑וֹ וַתְּחִ֕י ר֖וּחַ יַֽעֲקֹ֥ב אֲבִיהֶֽם:  ת  וּמַלִּילוּ עִמֵּהּ יָת כָּל פִּתְגָּמֵי יוֹסֵף דִּי מַלִּיל עִמְּהוֹן וַחֲזָא יָת עֲגַלָּתָא דִּשְׁלַח יוֹסֵף לְמִטַּל יָתֵהּ וּשְׁרַת רוּחַ נְבוּאָה עַל יַעֲקֹב אֲבוּהוֹן:

{כח}  חמישי  וַיֹּ֨אמֶר֙ יִשְׂרָאֵ֔ל רַ֛ב עוֹד-יוֹסֵ֥ף בְּנִ֖י חָ֑י אֵֽלְכָ֥ה וְאֶרְאֶ֖נּוּ בְּטֶ֥רֶם אָמֽוּת:  מ  וַיֹּ֨אמֶר֙ יִשְׂרָאֵ֔ל רַ֛ב עוֹד-יוֹסֵ֥ף בְּנִ֖י חָ֑י אֵֽלְכָ֥ה וְאֶרְאֶ֖נּוּ בְּטֶ֥רֶם אָמֽוּת:  ת  וַאֲמַר יִשְׂרָאֵל סַגִּי לִי חֶדְוָא עַד כְּעַן יוֹסֵף בְּרִי קַיָּם אֱזֵיל וְאֶחֱזִנֵּהּ עַד לָא אֵמוּת:




בראשית פרק-מו

{א} וַיִּסַּ֤ע יִשְׂרָאֵל֙ וְכָל-אֲשֶׁר-ל֔וֹ וַיָּבֹ֖א בְּאֵ֣רָה שָּׁ֑בַע וַיִּזְבַּ֣ח זְבָחִ֔ים לֵֽאלֹהֵ֖י אָבִ֥יו יִצְחָֽק:  מ  וַיִּסַּ֤ע יִשְׂרָאֵל֙ וְכָל-אֲשֶׁר-ל֔וֹ וַיָּבֹ֖א בְּאֵ֣רָה שָּׁ֑בַע וַיִּזְבַּ֣ח זְבָחִ֔ים לֵֽאלֹהֵ֖י אָבִ֥יו יִצְחָֽק:  ת  וּנְטַל יִשְׂרָאֵל וְכָל דִּי לֵהּ וַאֲתָא לִבְאֵר שָׁבַע וּדְבַח דִּבְחִין לֵאלָהָא דְאֲבוּהִי יִצְחָק:

{ב} וַיֹּ֨אמֶר אֱלֹהִ֤ים | לְיִשְׂרָאֵל֙ בְּמַרְאֹ֣ת הַלַּ֔יְלָה וַיֹּ֖אמֶר יַֽעֲקֹ֣ב | יַֽעֲקֹ֑ב וַיֹּ֖אמֶר הִנֵּֽנִי:  מ  וַיֹּ֨אמֶר אֱלֹהִ֤ים | לְיִשְׂרָאֵל֙ בְּמַרְאֹ֣ת הַלַּ֔יְלָה וַיֹּ֖אמֶר יַֽעֲקֹ֣ב | יַֽעֲקֹ֑ב וַיֹּ֖אמֶר הִנֵּֽנִי:  ת  וַאֲמַר יְיָ לְיִשְׂרָאֵל בְּחֶזְוֵי דְלֵילְיָא וַאֲמַר יַעֲקֹב יַעֲקֹב וַאֲמַר הָא אֲנָא:

{ג} וַיֹּ֕אמֶר אָֽנֹכִ֥י הָאֵ֖ל אֱלֹהֵ֣י אָבִ֑יךָ אַל-תִּירָא֙ מֵֽרְדָ֣ה מִצְרַ֔יְמָה כִּֽי-לְג֥וֹי גָּד֖וֹל אֲשִֽׂימְךָ֥ שָֽׁם:  מ  וַיֹּ֕אמֶר אָֽנֹכִ֥י הָאֵ֖ל אֱלֹהֵ֣י אָבִ֑יךָ אַל-תִּירָא֙ מֵֽרְדָ֣ה מִצְרַ֔יְמָה כִּֽי-לְג֥וֹי גָּד֖וֹל אֲשִֽׂימְךָ֥ שָֽׁם:  ת  וַאֲמַר אֲנָא אֵל אֱלָהָא דְאָבוּךְ לָא תִדְחַל מִלְּמֵיחַת לְמִצְרַיִם אֲרֵי לְעַם סַגִּי אֲשֵׁוִּנָּךְ תַּמָּן:

{ד} אָֽנֹכִ֗י אֵרֵ֤ד עִמְּךָ֙ מִצְרַ֔יְמָה וְאָֽנֹכִ֖י אַֽעַלְךָ֣ גַם-עָלֹ֑ה וְיוֹסֵ֕ף יָשִׁ֥ית יָד֖וֹ עַל-עֵינֶֽיךָ:  מ  אָֽנֹכִ֗י אֵרֵ֤ד עִמְּךָ֙ מִצְרַ֔יְמָה וְאָֽנֹכִ֖י אַֽעַלְךָ֣ גַם-עָלֹ֑ה וְיוֹסֵ֕ף יָשִׁ֥ית יָד֖וֹ עַל-עֵינֶֽיךָ:  ת  אֲנָא אֵחוֹת עִמָּךְ לְמִצְרַיִם וַאֲנָא אַסֵּקִנָּךְ אַף אַסָּקָא וְיוֹסֵף יְשַׁוֵּי יְדוֹהִי עַל עֵינָיךְ:

{ה} וַיָּ֥קָם יַֽעֲקֹ֖ב מִבְּאֵ֣ר שָׁ֑בַע וַיִּשְׂא֨וּ בְנֵֽי-יִשְׂרָאֵ֜ל אֶת-יַֽעֲקֹ֣ב אֲבִיהֶ֗ם וְאֶת-טַפָּם֙ וְאֶת-נְשֵׁיהֶ֔ם בָּֽעֲגָל֕וֹת אֲשֶׁר-שָׁלַ֥ח פַּרְעֹ֖ה לָשֵׂ֥את אֹתֽוֹ:  מ  וַיָּ֥קָם יַֽעֲקֹ֖ב מִבְּאֵ֣ר שָׁ֑בַע וַיִּשְׂא֨וּ בְנֵֽי-יִשְׂרָאֵ֜ל אֶת-יַֽעֲקֹ֣ב אֲבִיהֶ֗ם וְאֶת-טַפָּם֙ וְאֶת-נְשֵׁיהֶ֔ם בָּֽעֲגָל֕וֹת אֲשֶׁר-שָׁלַ֥ח פַּרְעֹ֖ה לָשֵׂ֥את אֹתֽוֹ:  ת  וְקָם יַעֲקֹב מִבְּאֵרָא דְשָׁבַע וּנְטָלוּ בְנֵי יִשְׂרָאֵל יָת יַעֲקֹב אֲבוּהוֹן וְיָת טַפְלְהוֹן וְיָת נְשֵׁיהוֹן בַּעֲגַלְתָּא דִּי שְׁלַח פַּרְעֹה לְמִטַּל יָתֵהּ:

{ו} וַיִּקְח֣וּ אֶת-מִקְנֵיהֶ֗ם וְאֶת-רְכוּשָׁם֙ אֲשֶׁ֤ר רָֽכְשׁוּ֙ בְּאֶ֣רֶץ כְּנַ֔עַן וַיָּבֹ֖אוּ מִצְרָ֑יְמָה יַֽעֲקֹ֖ב וְכָל-זַרְע֥וֹ אִתּֽוֹ:  מ  וַיִּקְח֣וּ אֶת-מִקְנֵיהֶ֗ם וְאֶת-רְכוּשָׁם֙ אֲשֶׁ֤ר רָֽכְשׁוּ֙ בְּאֶ֣רֶץ כְּנַ֔עַן וַיָּבֹ֖אוּ מִצְרָ֑יְמָה יַֽעֲקֹ֖ב וְכָל-זַרְע֥וֹ אִתּֽוֹ:  ת  וּדְבָרוּ יָת גֵּיתֵיהוֹן וְיָת קִנְיָנְהוֹן דִּי קְנוֹ בְּאַרְעָא דִּכְנַעַן וַאֲתוֹ לְמִצְרָיִם יַעֲקֹב וְכָל בְּנוֹהִי עִמֵּהּ:

{ז} בָּנָ֞יו וּבְנֵ֤י בָנָיו֙ אִתּ֔וֹ בְּנֹתָ֛יו וּבְנ֥וֹת בָּנָ֖יו וְכָל-זַרְע֑וֹ הֵבִ֥יא אִתּ֖וֹ מִצְרָֽיְמָה:  מ  בָּנָ֞יו וּבְנֵ֤י בָנָיו֙ אִתּ֔וֹ בְּנֹתָ֛יו וּבְנ֥וֹת בָּנָ֖יו וְכָל-זַרְע֑וֹ הֵבִ֥יא אִתּ֖וֹ מִצְרָֽיְמָה:  ת  בְּנוֹהִי וּבְנֵי בְנוֹהִי עִמֵּהּ בְּנָתֵהּ וּבְנַת בְּנוֹהִי וְכָל זַרְעֵהּ אַיְתִי עִמֵּהּ לְמִצְרָיִם: (ס)

{ח} וְאֵ֨לֶּה שְׁמ֧וֹת בְּנֵֽי-יִשְׂרָאֵ֛ל הַבָּאִ֥ים מִצְרַ֖יְמָה יַֽעֲקֹ֣ב וּבָנָ֑יו בְּכֹ֥ר יַֽעֲקֹ֖ב רְאוּבֵֽן:  מ  וְאֵ֨לֶּה שְׁמ֧וֹת בְּנֵֽי-יִשְׂרָאֵ֛ל הַבָּאִ֥ים מִצְרַ֖יְמָה יַֽעֲקֹ֣ב וּבָנָ֑יו בְּכֹ֥ר יַֽעֲקֹ֖ב רְאוּבֵֽן:  ת  וְאִלֵּין שְׁמָהַת בְּנֵי יִשְׂרָאֵל דְּעַלּוּ לְמִצְרַיִם יַעֲקֹב וּבְנוֹהִי בּוּכְרָא דְיַעֲקֹב רְאוּבֵן:

{ט} וּבְנֵ֖י רְאוּבֵ֑ן חֲנ֥וֹךְ וּפַלּ֖וּא וְחֶצְרֹ֥ן וְכַרְמִֽי:  מ  וּבְנֵ֖י רְאוּבֵ֑ן חֲנ֥וֹךְ וּפַלּ֖וּא וְחֶצְרֹ֥ן וְכַרְמִֽי:  ת  וּבְנֵי רְאוּבֵן חֲנוֹךְ וּפַלּוּא וְחֶצְרֹן וְכַרְמִי:

{י} וּבְנֵ֣י שִׁמְע֗וֹן יְמוּאֵ֧ל וְיָמִ֛ין וְאֹ֖הַד וְיָכִ֣ין וְצֹ֑חַר וְשָׁא֖וּל בֶּן-הַֽכְּנַֽעֲנִֽית:  מ  וּבְנֵ֣י שִׁמְע֗וֹן יְמוּאֵ֧ל וְיָמִ֛ין וְאֹ֖הַד וְיָכִ֣ין וְצֹ֑חַר וְשָׁא֖וּל בֶּן-הַֽכְּנַֽעֲנִֽית:  ת  וּבְנֵי שִׁמְעוֹן יְמוּאֵל וְיָמִין וְאֹהַד וְיָכִין וְצֹחַר וְשָׁאוּל בַּר כְּנַעֲנֵיתָא:

{יא} וּבְנֵ֖י לֵוִ֑י גֵּֽרְשׁ֕וֹן קְהָ֖ת וּמְרָרִֽי:  מ  וּבְנֵ֖י לֵוִ֑י גֵּֽרְשׁ֕וֹן קְהָ֖ת וּמְרָרִֽי:  ת  וּבְנֵי לֵוִי גֵּרְשׁוֹן קְהָת וּמְרָרִי:

{יב} וּבְנֵ֣י יְהוּדָ֗ה עֵ֧ר וְאוֹנָ֛ן וְשֵׁלָ֖ה וָפֶ֣רֶץ וָזָ֑רַח וַיָּ֨מָת עֵ֤ר וְאוֹנָן֙ בְּאֶ֣רֶץ כְּנַ֔עַן וַיִּֽהְי֥וּ בְנֵי-פֶ֖רֶץ חֶצְרֹ֥ן וְחָמֽוּל:  מ  וּבְנֵ֣י יְהוּדָ֗ה עֵ֧ר וְאוֹנָ֛ן וְשֵׁלָ֖ה וָפֶ֣רֶץ וָזָ֑רַח וַיָּ֨מָת עֵ֤ר וְאוֹנָן֙ בְּאֶ֣רֶץ כְּנַ֔עַן וַיִּֽהְי֥וּ בְנֵי-פֶ֖רֶץ חֶצְרֹ֥ן וְחָמֽוּל:  ת  וּבְנֵי יְהוּדָה עֵר וְאוֹנָן וְשֵׁלָה וָפֶרֶץ וָזָרַח וּמִית עֵר וְאוֹנָן בְּאַרְעָא דִּכְנַעַן וַהֲווֹ בְנֵי פֶרֶץ חֶצְרֹן וְחָמוּל:

{יג} וּבְנֵ֖י יִשָּׂשׂכָ֑ר תּוֹלָ֥ע וּפֻוָּ֖ה וְי֥וֹב וְשִׁמְרֹֽן:  מ  וּבְנֵ֖י יִשָּׂשׂכָ֑ר תּוֹלָ֥ע וּפֻוָּ֖ה וְי֥וֹב וְשִׁמְרֹֽן:  ת  וּבְנֵי יִשָּׂשכָר תּוֹלָע וּפֻוָּה וְיוֹב וְשִׁמְרוֹן:

{יד} וּבְנֵ֖י זְבֻל֑וּן סֶ֥רֶד וְאֵל֖וֹן וְיַחְלְאֵֽל:  מ  וּבְנֵ֖י זְבֻל֑וּן סֶ֥רֶד וְאֵל֖וֹן וְיַחְלְאֵֽל:  ת  וּבְנֵי זְבֻלוּן סֶרֶד וְאֵלוֹן וְיַחְלְאֵל:

{טו} אֵ֣לֶּה | בְּנֵ֣י לֵאָ֗ה אֲשֶׁ֨ר יָֽלְדָ֤ה לְיַֽעֲקֹב֙ בְּפַדַּ֣ן אֲרָ֔ם וְאֵ֖ת דִּינָ֣ה בִתּ֑וֹ כָּל-נֶ֧פֶשׁ בָּנָ֛יו וּבְנוֹתָ֖יו שְׁלֹשִׁ֥ים וְשָׁלֹֽשׁ:  מ  אֵ֣לֶּה | בְּנֵ֣י לֵאָ֗ה אֲשֶׁ֨ר יָֽלְדָ֤ה לְיַֽעֲקֹב֙ בְּפַדַּ֣ן אֲרָ֔ם וְאֵ֖ת דִּינָ֣ה בִתּ֑וֹ כָּל-נֶ֧פֶשׁ בָּנָ֛יו וּבְנוֹתָ֖יו שְׁלֹשִׁ֥ים וְשָׁלֹֽשׁ:  ת  אִלֵּין בְּנֵי לֵאָה דִּילִידַת לְיַעֲקֹב בְּפַדַּן אֲרָם וְיָת דִּינָה בְרַתֵּהּ כָּל נֶפֶשׁ בְּנוֹהִי וּבְנָתֵהּ תְּלָתִין וּתְלָת:

{טז} וּבְנֵ֣י גָ֔ד צִפְי֥וֹן וְחַגִּ֖י שׁוּנִ֣י וְאֶצְבֹּ֑ן עֵרִ֥י וַֽאֲרוֹדִ֖י וְאַרְאֵלִֽי:  מ  וּבְנֵ֣י גָ֔ד צִפְי֥וֹן וְחַגִּ֖י שׁוּנִ֣י וְאֶצְבֹּ֑ן עֵרִ֥י וַֽאֲרוֹדִ֖י וְאַרְאֵלִֽי:  ת  וּבְנֵי גָד צִפְיוֹן וְחַגִּי שׁוּנִי וְאֶצְבּוֹן עֵרִי וַאֲרוֹדִי וְאַרְאֵלִי:

{יז} וּבְנֵ֣י אָשֵׁ֗ר יִמְנָ֧ה וְיִשְׁוָ֛ה וְיִשְׁוִ֥י וּבְרִיעָ֖ה וְשֶׂ֣רַח אֲחֹתָ֑ם וּבְנֵ֣י בְרִיעָ֔ה חֶ֖בֶר וּמַלְכִּיאֵֽל:  מ  וּבְנֵ֣י אָשֵׁ֗ר יִמְנָ֧ה וְיִשְׁוָ֛ה וְיִשְׁוִ֥י וּבְרִיעָ֖ה וְשֶׂ֣רַח אֲחֹתָ֑ם וּבְנֵ֣י בְרִיעָ֔ה חֶ֖בֶר וּמַלְכִּיאֵֽל:  ת  וּבְנֵי אָשֵׁר יִמְנָה וְיִשְׁוָה וְיִשְׁוִי וּבְרִיעָה וְשֶׂרַח אֲחַתְהוֹן וּבְנֵי בְרִיעָה חֶבֶר וּמַלְכִּיאֵל:

{יח} אֵ֚לֶּה בְּנֵ֣י זִלְפָּ֔ה אֲשֶׁר-נָתַ֥ן לָבָ֖ן לְלֵאָ֣ה בִתּ֑וֹ וַתֵּ֤לֶד אֶת-אֵ֨לֶּה֙ לְיַֽעֲקֹ֔ב שֵׁ֥שׁ עֶשְׂרֵ֖ה נָֽפֶשׁ:  מ  אֵ֚לֶּה בְּנֵ֣י זִלְפָּ֔ה אֲשֶׁר-נָתַ֥ן לָבָ֖ן לְלֵאָ֣ה בִתּ֑וֹ וַתֵּ֤לֶד אֶת-אֵ֨לֶּה֙ לְיַֽעֲקֹ֔ב שֵׁ֥שׁ עֶשְׂרֵ֖ה נָֽפֶשׁ:  ת  אִלֵּין בְּנֵי זִלְפָּה דִי יְהַב לָבָן לְלֵאָה בְרַתֵּהּ וִילִידַת יָת אִלֵּין לְיַעֲקֹב שִׁית עֶשְׂרֵי נַפְשָׁא:

{יט} בְּנֵ֤י רָחֵל֙ אֵ֣שֶׁת יַֽעֲקֹ֔ב יוֹסֵ֖ף וּבִנְיָמִֽן:  מ  בְּנֵ֤י רָחֵל֙ אֵ֣שֶׁת יַֽעֲקֹ֔ב יוֹסֵ֖ף וּבִנְיָמִֽן:  ת  בְּנֵי רָחֵל אִתַּת יַעֲקֹב יוֹסֵף וּבִנְיָמִן:

{כ} וַיִּוָּלֵ֣ד לְיוֹסֵף֘ בְּאֶ֣רֶץ מִצְרַ֒יִם֒ אֲשֶׁ֤ר יָֽלְדָה-לּוֹ֙ אָֽסְנַ֔ת בַּת-פּ֥וֹטִי פֶ֖רַע כֹּהֵ֣ן אֹ֑ן אֶת-מְנַשֶּׁ֖ה וְאֶת-אֶפְרָֽיִם:  מ  וַיִּוָּלֵ֣ד לְיוֹסֵף֘ בְּאֶ֣רֶץ מִצְרַ֒יִם֒ אֲשֶׁ֤ר יָֽלְדָה-לּוֹ֙ אָֽסְנַ֔ת בַּת-פּ֥וֹטִי פֶ֖רַע כֹּהֵ֣ן אֹ֑ן אֶת-מְנַשֶּׁ֖ה וְאֶת-אֶפְרָֽיִם:  ת  וְאִתְיְלִידוּ לְיוֹסֵף בְּאַרְעָא דְמִצְרַיִם דִּילִידַת לֵהּ אָסְנַת בַּת פּוֹטִי פֶרַע רַבָּא דְאוֹן יָת מְנַשֶּׁה וְיָת אֶפְרָיִם:

{כא} וּבְנֵ֣י בִנְיָמִ֗ן בֶּ֤לַע וָבֶ֨כֶר֙ וְאַשְׁבֵּ֔ל גֵּרָ֥א וְנַֽעֲמָ֖ן אֵחִ֣י וָרֹ֑אשׁ מֻפִּ֥ים וְחֻפִּ֖ים וָאָֽרְדְּ:  מ  וּבְנֵ֣י בִנְיָמִ֗ן בֶּ֤לַע וָבֶ֨כֶר֙ וְאַשְׁבֵּ֔ל גֵּרָ֥א וְנַֽעֲמָ֖ן אֵחִ֣י וָרֹ֑אשׁ מֻפִּ֥ים וְחֻפִּ֖ים וָאָֽרְדְּ:  ת  וּבְנֵי בִנְיָמִן בֶּלַע וָבֶכֶר וְאַשְׁבֵּל גֵּרָא וְנַעֲמָן אֵחִי וָרֹאשׁ מֻפִּים וְחֻפִּים וָאָרְדְּ:

{כב} אֵ֚לֶּה בְּנֵ֣י רָחֵ֔ל אֲשֶׁ֥ר יֻלַּ֖ד לְיַֽעֲקֹ֑ב כָּל-נֶ֖פֶשׁ אַרְבָּעָ֥ה עָשָֽׂר:  מ  אֵ֚לֶּה בְּנֵ֣י רָחֵ֔ל אֲשֶׁ֥ר יֻלַּ֖ד לְיַֽעֲקֹ֑ב כָּל-נֶ֖פֶשׁ אַרְבָּעָ֥ה עָשָֽׂר:  ת  אִלֵּין בְּנֵי רָחֵל דְּאִתְיְלִידוּ לְיַעֲקֹב כָּל נַפְשָׁא אַרְבָּעָה עֲשָׂר:

{כג} וּבְנֵי-דָ֖ן חֻשִֽׁים:  מ  וּבְנֵי-דָ֖ן חֻשִֽׁים:  ת  וּבְנֵי דָן חֻשִׁים:

{כד} וּבְנֵ֖י נַפְתָּלִ֑י יַחְצְאֵ֥ל וְגוּנִ֖י וְיֵ֥צֶר וְשִׁלֵּֽם:  מ  וּבְנֵ֖י נַפְתָּלִ֑י יַחְצְאֵ֥ל וְגוּנִ֖י וְיֵ֥צֶר וְשִׁלֵּֽם:  ת  וּבְנֵי נַפְתָּלִי יַחְצְאֵל וְגוּנִי וְיֵצֶר וְשִׁלֵּם:

{כה} אֵ֚לֶּה בְּנֵ֣י בִלְהָ֔ה אֲשֶׁר-נָתַ֥ן לָבָ֖ן לְרָחֵ֣ל בִּתּ֑וֹ וַתֵּ֧לֶד אֶת-אֵ֛לֶּה לְיַֽעֲקֹ֖ב כָּל-נֶ֥פֶשׁ שִׁבְעָֽה:  מ  אֵ֚לֶּה בְּנֵ֣י בִלְהָ֔ה אֲשֶׁר-נָתַ֥ן לָבָ֖ן לְרָחֵ֣ל בִּתּ֑וֹ וַתֵּ֧לֶד אֶת-אֵ֛לֶּה לְיַֽעֲקֹ֖ב כָּל-נֶ֥פֶשׁ שִׁבְעָֽה:  ת  אִלֵּין בְּנֵי בִלְהָה דִּיהַב לָבָן לְרָחֵל בְּרַתֵּהּ וִילִידַת יָת אִלֵּין לְיַעֲקֹב כָּל נַפְשָׁא שִׁבְעָא:

{כו} כָּל-הַ֠נֶּ֠פֶשׁ הַבָּאָ֨ה לְיַֽעֲקֹ֤ב מִצְרַ֨יְמָה֙ יֹֽצְאֵ֣י יְרֵכ֔וֹ מִלְּבַ֖ד נְשֵׁ֣י בְנֵי-יַֽעֲקֹ֑ב כָּל-נֶ֖פֶשׁ שִׁשִּׁ֥ים וָשֵֽׁשׁ:  מ  כָּל-הַ֠נֶּ֠פֶשׁ הַבָּאָ֨ה לְיַֽעֲקֹ֤ב מִצְרַ֨יְמָה֙ יֹֽצְאֵ֣י יְרֵכ֔וֹ מִלְּבַ֖ד נְשֵׁ֣י בְנֵי-יַֽעֲקֹ֑ב כָּל-נֶ֖פֶשׁ שִׁשִּׁ֥ים וָשֵֽׁשׁ:  ת  כָּל נַפְשָׁא דְּעַלָּא לְיַעֲקֹב לְמִצְרַיִם נָפְקֵי יַרְכֵּהּ בַּר מִנְּשֵׁי בְנֵי יַעֲקֹב כָּל נַפְשָׁתָא שִׁתִּין וְשִׁית:

{כז} וּבְנֵ֥י יוֹסֵ֛ף אֲשֶׁר-יֻלַּד-ל֥וֹ בְמִצְרַ֖יִם נֶ֣פֶשׁ שְׁנָ֑יִם כָּל-הַנֶּ֧פֶשׁ לְבֵית-יַֽעֲקֹ֛ב הַבָּ֥אָה מִצְרַ֖יְמָה שִׁבְעִֽים:  מ  וּבְנֵ֥י יוֹסֵ֛ף אֲשֶׁר-יֻלַּד-ל֥וֹ בְמִצְרַ֖יִם נֶ֣פֶשׁ שְׁנָ֑יִם כָּל-הַנֶּ֧פֶשׁ לְבֵית-יַֽעֲקֹ֛ב הַבָּ֥אָה מִצְרַ֖יְמָה שִׁבְעִֽים:  ת  וּבְנֵי יוֹסֵף דִּי אִתְיְלִידוּ לֵהּ בְּמִצְרַיִם נַפְשָׁתָא תְרֵין כָּל נַפְשָׁתָא לְבֵית יַעֲקֹב דְּעַלּוּ לְמִצְרַיִם שִׁבְעִין: (ס)

{כח}  שישי  וְאֶת-יְהוּדָ֞ה שָׁלַ֤ח לְפָנָיו֙ אֶל-יוֹסֵ֔ף לְהוֹרֹ֥ת לְפָנָ֖יו גֹּ֑שְׁנָה וַיָּבֹ֖אוּ אַ֥רְצָה גֹּֽשֶׁן:  מ  וְאֶת-יְהוּדָ֞ה שָׁלַ֤ח לְפָנָיו֙ אֶל-יוֹסֵ֔ף לְהוֹרֹ֥ת לְפָנָ֖יו גֹּ֑שְׁנָה וַיָּבֹ֖אוּ אַ֥רְצָה גֹּֽשֶׁן:  ת  וְיָת יְהוּדָה שְׁלַח קֳדָמוֹהִי לְוַת יוֹסֵף לְפַנָּאָה קֳדָמוֹהִי לְגשֶׁן וַאֲתוֹ לְאַרְעָא דְגשֶׁן:

{כט} וַיֶּאְסֹ֤ר יוֹסֵף֙ מֶרְכַּבְתּ֔וֹ וַיַּ֛עַל לִקְרַֽאת-יִשְׂרָאֵ֥ל אָבִ֖יו גֹּ֑שְׁנָה וַיֵּרָ֣א אֵלָ֗יו וַיִּפֹּל֙ עַל-צַוָּארָ֔יו וַיֵּ֥בְךְּ עַל-צַוָּארָ֖יו עֽוֹד:  מ  וַיֶּאְסֹ֤ר יוֹסֵף֙ מֶרְכַּבְתּ֔וֹ וַיַּ֛עַל לִקְרַֽאת-יִשְׂרָאֵ֥ל אָבִ֖יו גֹּ֑שְׁנָה וַיֵּרָ֣א אֵלָ֗יו וַיִּפֹּל֙ עַל-צַוָּארָ֔יו וַיֵּ֥בְךְּ עַל-צַוָּארָ֖יו עֽוֹד:  ת  וְטַקֵּיס יוֹסֵף רְתִכּוֹהִי וּסְלֵיק לְקַדָּמוּת יִשְׂרָאֵל אֲבוּהִי לְגשֶׁן וְאִתְגְּלִי לֵהּ וּנְפַל עַל צַוְארֵהּ וּבְכָא עַל צַוְארֵהּ עוֹד:

{ל} וַיֹּ֧אמֶר יִשְׂרָאֵ֛ל אֶל-יוֹסֵ֖ף אָמ֣וּתָה הַפָּ֑עַם אַֽחֲרֵי֙ רְאוֹתִ֣י אֶת-פָּנֶ֔יךָ כִּ֥י עֽוֹדְךָ֖ חָֽי:  מ  וַיֹּ֧אמֶר יִשְׂרָאֵ֛ל אֶל-יוֹסֵ֖ף אָמ֣וּתָה הַפָּ֑עַם אַֽחֲרֵי֙ רְאוֹתִ֣י אֶת-פָּנֶ֔יךָ כִּ֥י עֽוֹדְךָ֖ חָֽי:  ת  וַאֲמַר יִשְׂרָאֵל לְיוֹסֵף אִלּוּ אֲנָא מָאִית זִמְנָא הָדָא מְנַחֵם אֲנָא בָּתַר דַּחֲזֵיתִי יָת אַפָיךְ אֲרֵי עַד כְּעַן אַתְּ קַיָּם:

{לא} וַיֹּ֨אמֶר יוֹסֵ֤ף אֶל-אֶחָיו֙ וְאֶל-בֵּ֣ית אָבִ֔יו אֶֽעֱלֶ֖ה וְאַגִּ֣ידָה לְפַרְעֹ֑ה וְאֹֽמְרָ֣ה אֵלָ֗יו אַחַ֧י וּבֵית-אָבִ֛י אֲשֶׁ֥ר בְּאֶֽרֶץ-כְּנַ֖עַן בָּ֥אוּ אֵלָֽי:  מ  וַיֹּ֨אמֶר יוֹסֵ֤ף אֶל-אֶחָיו֙ וְאֶל-בֵּ֣ית אָבִ֔יו אֶֽעֱלֶ֖ה וְאַגִּ֣ידָה לְפַרְעֹ֑ה וְאֹֽמְרָ֣ה אֵלָ֗יו אַחַ֧י וּבֵית-אָבִ֛י אֲשֶׁ֥ר בְּאֶֽרֶץ-כְּנַ֖עַן בָּ֥אוּ אֵלָֽי:  ת  וַאֲמַר יוֹסֵף לַאֲחוֹהִי וּלְבֵית אֲבוּהִי אֶסַּק וְאֵחַוֵּי לְפַרְעֹה וְאֵימַר לֵהּ אַחַי וּבֵית אַבָּא דִּי בְאַרְעָא דִּכְנַעַן עַלּוּ לְוָתִי:

{לב} וְהָֽאֲנָשִׁים֙ רֹ֣עֵי צֹ֔אן כִּֽי-אַנְשֵׁ֥י מִקְנֶ֖ה הָי֑וּ וְצֹאנָ֧ם וּבְקָרָ֛ם וְכָל-אֲשֶׁ֥ר לָהֶ֖ם הֵבִֽיאוּ:  מ  וְהָֽאֲנָשִׁים֙ רֹ֣עֵי צֹ֔אן כִּֽי-אַנְשֵׁ֥י מִקְנֶ֖ה הָי֑וּ וְצֹאנָ֧ם וּבְקָרָ֛ם וְכָל-אֲשֶׁ֥ר לָהֶ֖ם הֵבִֽיאוּ:  ת  וְגֻּבְרַיָּא רָעֵי עָנָא אֲרֵי גֻּבְרֵי מָרֵי גֵּיתֵי הֲווֹ וְעָנְהוֹן וְתוֹרְהוֹן וְכָל דִּי לְהוֹן אַיְתִיאוּ:

{לג} וְהָיָ֕ה כִּֽי-יִקְרָ֥א לָכֶ֖ם פַּרְעֹ֑ה וְאָמַ֖ר מַה-מַּֽעֲשֵׂיכֶֽם:  מ  וְהָיָ֕ה כִּֽי-יִקְרָ֥א לָכֶ֖ם פַּרְעֹ֑ה וְאָמַ֖ר מַה-מַּֽעֲשֵׂיכֶֽם:  ת  וִיהֵי אֲרֵי יִקְרֵי לְכוֹן פַּרְעֹה וְיֵימַר מָה עוֹבָדֵיכוֹן:

{לד} וַֽאֲמַרְתֶּ֗ם אַנְשֵׁ֨י מִקְנֶ֜ה הָי֤וּ עֲבָדֶ֨יךָ֙ מִנְּעוּרֵ֣ינוּ וְעַד-עַ֔תָּה גַּם-אֲנַ֖חְנוּ גַּם-אֲבֹתֵ֑ינוּ בַּֽעֲב֗וּר תֵּֽשְׁבוּ֙ בְּאֶ֣רֶץ גֹּ֔שֶׁן כִּי-תֽוֹעֲבַ֥ת מִצְרַ֖יִם כָּל-רֹ֥עֵה צֹֽאן:  מ  וַֽאֲמַרְתֶּ֗ם אַנְשֵׁ֨י מִקְנֶ֜ה הָי֤וּ עֲבָדֶ֨יךָ֙ מִנְּעוּרֵ֣ינוּ וְעַד-עַ֔תָּה גַּם-אֲנַ֖חְנוּ גַּם-אֲבֹתֵ֑ינוּ בַּֽעֲב֗וּר תֵּֽשְׁבוּ֙ בְּאֶ֣רֶץ גֹּ֔שֶׁן כִּי-תֽוֹעֲבַ֥ת מִצְרַ֖יִם כָּל-רֹ֥עֵה צֹֽאן:  ת  וְתֵימְרוּן גֻּבְרֵי מָרֵי גֵּיתֵי הֲווֹ עַבְדָּיךְ מֵעוּלֵמָנָא וְעַד כְּעַן אַף אֲנַחְנָא אַף אַבְהָתָנָא בְּדִיל דִּי תֵיתְבוּן בְּאַרְעָא דְגשֶׁן אֲרֵי מְרַחֲקִין מִצְרָאֵי כָּל רָעֵי עָנָא:




בראשית פרק-מז

{א} וַיָּבֹ֣א יוֹסֵף֘ וַיַּגֵּ֣ד לְפַרְעֹה֒ וַיֹּ֗אמֶר אָבִ֨י וְאַחַ֜י וְצֹאנָ֤ם וּבְקָרָם֙ וְכָל-אֲשֶׁ֣ר לָהֶ֔ם בָּ֖אוּ מֵאֶ֣רֶץ כְּנָ֑עַן וְהִנָּ֖ם בְּאֶ֥רֶץ גֹּֽשֶׁן:  מ  וַיָּבֹ֣א יוֹסֵף֘ וַיַּגֵּ֣ד לְפַרְעֹה֒ וַיֹּ֗אמֶר אָבִ֨י וְאַחַ֜י וְצֹאנָ֤ם וּבְקָרָם֙ וְכָל-אֲשֶׁ֣ר לָהֶ֔ם בָּ֖אוּ מֵאֶ֣רֶץ כְּנָ֑עַן וְהִנָּ֖ם בְּאֶ֥רֶץ גֹּֽשֶׁן:  ת  וַאֲתָא יוֹסֵף וְחַוִּי לְפַרְעֹה וַאֲמַר אַבָּא וְאַחַי וְעָנְהוֹן וְתוֹרְהוֹן וְכָל דִּי לְהוֹן אֲתוֹ מֵאַרְעָא דִּכְנָעַן וְהָא אִנּוּן בְּאַרְעָא דְגשֶׁן:

{ב} וּמִקְצֵ֣ה אֶחָ֔יו לָקַ֖ח חֲמִשָּׁ֣ה אֲנָשִׁ֑ים וַיַּצִּגֵ֖ם לִפְנֵ֥י פַרְעֹֽה:  מ  וּמִקְצֵ֣ה אֶחָ֔יו לָקַ֖ח חֲמִשָּׁ֣ה אֲנָשִׁ֑ים וַיַּצִּגֵ֖ם לִפְנֵ֥י פַרְעֹֽה:  ת  וּמִקְצַת מִן אֲחוֹהִי דְּבַר חַמְשָׁא גֻּבְרִין וַאֲקֵימִנּוּן קֳדָם פַּרְעֹה:

{ג} וַיֹּ֧אמֶר פַּרְעֹ֛ה אֶל-אֶחָ֖יו מַה-מַּֽעֲשֵׂיכֶ֑ם וַיֹּֽאמְר֣וּ אֶל-פַּרְעֹ֗ה רֹעֵ֥ה צֹאן֙ עֲבָדֶ֔יךָ גַּם-אֲנַ֖חְנוּ גַּם-אֲבוֹתֵֽינוּ:  מ  וַיֹּ֧אמֶר פַּרְעֹ֛ה אֶל-אֶחָ֖יו מַה-מַּֽעֲשֵׂיכֶ֑ם וַיֹּֽאמְר֣וּ אֶל-פַּרְעֹ֗ה רֹעֵ֥ה צֹאן֙ עֲבָדֶ֔יךָ גַּם-אֲנַ֖חְנוּ גַּם-אֲבוֹתֵֽינוּ:  ת  וַאֲמַר פַּרְעֹה לַאֲחוֹהִי מָה עוֹבָדֵיכוֹן וַאֲמָרוּ לְפַרְעֹה רָעֵי עָנָא עַבְדָּיךְ אַף אֲנַחְנָא אַף אֲבָהָתָנָא:

{ד} וַיֹּֽאמְר֣וּ אֶל-פַּרְעֹ֗ה לָג֣וּר בָּאָ֘רֶץ֘ בָּ֒אנוּ֒ כִּי-אֵ֣ין מִרְעֶ֗ה לַצֹּאן֙ אֲשֶׁ֣ר לַֽעֲבָדֶ֔יךָ כִּֽי-כָבֵ֥ד הָֽרָעָ֖ב בְּאֶ֣רֶץ כְּנָ֑עַן וְעַתָּ֛ה יֵֽשְׁבוּ-נָ֥א עֲבָדֶ֖יךָ בְּאֶ֥רֶץ גֹּֽשֶׁן:  מ  וַיֹּֽאמְר֣וּ אֶל-פַּרְעֹ֗ה לָג֣וּר בָּאָ֘רֶץ֘ בָּ֒אנוּ֒ כִּי-אֵ֣ין מִרְעֶ֗ה לַצֹּאן֙ אֲשֶׁ֣ר לַֽעֲבָדֶ֔יךָ כִּֽי-כָבֵ֥ד הָֽרָעָ֖ב בְּאֶ֣רֶץ כְּנָ֑עַן וְעַתָּ֛ה יֵֽשְׁבוּ-נָ֥א עֲבָדֶ֖יךָ בְּאֶ֥רֶץ גֹּֽשֶׁן:  ת  וַאֲמָרוּ לְוַת פַּרְעֹה לְאִתּוֹתָבָא בְאַרְעָא אֲתֵינָא אֲרֵי לֵית רַעֲיָא לְעָנָא דִּי לְעַבְדָּיךְ אֲרֵי תַקִּיף כַּפְנָא בְּאַרְעָא דִּכְנָעַן וּכְעַן יֵתְבוּן כְּעַן עַבְדָּיךְ בְּאַרְעָא דְגשֶׁן:

{ה} וַיֹּ֣אמֶר פַּרְעֹ֔ה אֶל-יוֹסֵ֖ף לֵאמֹ֑ר אָבִ֥יךָ וְאַחֶ֖יךָ בָּ֥אוּ אֵלֶֽיךָ:  מ  וַיֹּ֣אמֶר פַּרְעֹ֔ה אֶל-יוֹסֵ֖ף לֵאמֹ֑ר אָבִ֥יךָ וְאַחֶ֖יךָ בָּ֥אוּ אֵלֶֽיךָ:  ת  וַאֲמַר פַּרְעֹה לְיוֹסֵף לְמֵימָר אֲבוּךְ וְאַחָיךְ אֲתוֹ לְוָתָךְ:

{ו} אֶ֤רֶץ מִצְרַ֨יִם֙ לְפָנֶ֣יךָ הִ֔וא בְּמֵיטַ֣ב הָאָ֔רֶץ הוֹשֵׁ֥ב אֶת-אָבִ֖יךָ וְאֶת-אַחֶ֑יךָ יֵֽשְׁבוּ֙ בְּאֶ֣רֶץ גֹּ֔שֶׁן וְאִם-יָדַ֗עְתָּ וְיֶשׁ-בָּם֙ אַנְשֵׁי-חַ֔יִל וְשַׂמְתָּ֛ם שָׂרֵ֥י מִקְנֶ֖ה עַל-אֲשֶׁר-לִֽי:  מ  אֶ֤רֶץ מִצְרַ֨יִם֙ לְפָנֶ֣יךָ הִ֔וא בְּמֵיטַ֣ב הָאָ֔רֶץ הוֹשֵׁ֥ב אֶת-אָבִ֖יךָ וְאֶת-אַחֶ֑יךָ יֵֽשְׁבוּ֙ בְּאֶ֣רֶץ גֹּ֔שֶׁן וְאִם-יָדַ֗עְתָּ וְיֶשׁ-בָּם֙ אַנְשֵׁי-חַ֔יִל וְשַׂמְתָּ֛ם שָׂרֵ֥י מִקְנֶ֖ה עַל-אֲשֶׁר-לִֽי:  ת  אַרְעָא דְמִצְרַיִם קֳדָמָךְ הִיא בִּדְשַׁפִּיר בְּאַרְעָא אוֹתֵיב יָת אֲבוּךְ וְיָת אַחָיךְ יֵתְבוּן בְּאַרְעָא דְגשֶׁן וְאִם יָדַעְתְּ וְאִית בְּהוֹן גֻּבְרִין דְּחֵילָא וּתְמַנִּנוּן רַבָּנֵי גֵיתֵי עַל דִּי לִי:

{ז} וַיָּבֵ֤א יוֹסֵף֙ אֶת-יַֽעֲקֹ֣ב אָבִ֔יו וַיַּֽעֲמִדֵ֖הוּ לִפְנֵ֣י פַרְעֹ֑ה וַיְבָ֥רֶךְ יַֽעֲקֹ֖ב אֶת-פַּרְעֹֽה:  מ  וַיָּבֵ֤א יוֹסֵף֙ אֶת-יַֽעֲקֹ֣ב אָבִ֔יו וַיַּֽעֲמִדֵ֖הוּ לִפְנֵ֣י פַרְעֹ֑ה וַיְבָ֥רֶךְ יַֽעֲקֹ֖ב אֶת-פַּרְעֹֽה:  ת  וְאַיְתִי יוֹסֵף יָת יַעֲקֹב אֲבוּהִי וַאֲקִימִנֵּהּ קֳדָם פַּרְעֹה וּבְרִיךְ יַעֲקֹב יָת פַּרְעֹה:

{ח} וַיֹּ֥אמֶר פַּרְעֹ֖ה אֶל-יַֽעֲקֹ֑ב כַּמָּ֕ה יְמֵ֖י שְׁנֵ֥י חַיֶּיֽךָ:  מ  וַיֹּ֥אמֶר פַּרְעֹ֖ה אֶל-יַֽעֲקֹ֑ב כַּמָּ֕ה יְמֵ֖י שְׁנֵ֥י חַיֶּיֽךָ:  ת  וַאֲמַר פַּרְעֹה לְּיַעֲקֹב כַּמָּה יוֹמֵי שְׁנֵי חַיָּיךְ:

{ט} וַיֹּ֤אמֶר יַֽעֲקֹב֙ אֶל-פַּרְעֹ֔ה יְמֵי֙ שְׁנֵ֣י מְגוּרַ֔י שְׁלשִׁ֥ים וּמְאַ֖ת שָׁנָ֑ה מְעַ֣ט וְרָעִ֗ים הָיוּ֙ יְמֵי֙ שְׁנֵ֣י חַיַּ֔י וְלֹ֣א הִשִּׂ֗יגוּ אֶת-יְמֵי֙ שְׁנֵי֙ חַיֵּ֣י אֲבֹתַ֔י בִּימֵ֖י מְגֽוּרֵיהֶֽם:  מ  וַיֹּ֤אמֶר יַֽעֲקֹב֙ אֶל-פַּרְעֹ֔ה יְמֵי֙ שְׁנֵ֣י מְגוּרַ֔י שְׁלשִׁ֥ים וּמְאַ֖ת שָׁנָ֑ה מְעַ֣ט וְרָעִ֗ים הָיוּ֙ יְמֵי֙ שְׁנֵ֣י חַיַּ֔י וְלֹ֣א הִשִּׂ֗יגוּ אֶת-יְמֵי֙ שְׁנֵי֙ חַיֵּ֣י אֲבֹתַ֔י בִּימֵ֖י מְגֽוּרֵיהֶֽם:  ת  וַאֲמַר יַעֲקֹב לְפַרְעֹה יוֹמֵי שְׁנֵי תוֹתָבוּתַי מְאָה וּתְלָתִין שְׁנִין זְעֵירִין וּבִישִׁין הֲווֹ יוֹמֵי שְׁנֵי חַיַּי וְלָא אַדְבִּיקוּ יָת יוֹמֵי שְׁנֵי חַיֵּי אֲבָהָתַי בְּיוֹמֵי תּוֹתָבוּתְהוֹן:

{י} וַיְבָ֥רֶךְ יַֽעֲקֹ֖ב אֶת-פַּרְעֹ֑ה וַיֵּצֵ֖א מִלִּפְנֵ֥י פַרְעֹֽה:  מ  וַיְבָ֥רֶךְ יַֽעֲקֹ֖ב אֶת-פַּרְעֹ֑ה וַיֵּצֵ֖א מִלִּפְנֵ֥י פַרְעֹֽה:  ת  וּבָרִיךְ יַעֲקֹב יָת פַּרְעֹה וּנְפַק מִן קֳדָם פַּרְעֹה:

{יא}  שביעי  וַיּוֹשֵׁ֣ב יוֹסֵף֘ אֶת-אָבִ֣יו וְאֶת-אֶחָיו֒ וַיִּתֵּ֨ן לָהֶ֤ם אֲחֻזָּה֙ בְּאֶ֣רֶץ מִצְרַ֔יִם בְּמֵיטַ֥ב הָאָ֖רֶץ בְּאֶ֣רֶץ רַעְמְסֵ֑ס כַּֽאֲשֶׁ֖ר צִוָּ֥ה פַרְעֹֽה:  מ  וַיּוֹשֵׁ֣ב יוֹסֵף֘ אֶת-אָבִ֣יו וְאֶת-אֶחָיו֒ וַיִּתֵּ֨ן לָהֶ֤ם אֲחֻזָּה֙ בְּאֶ֣רֶץ מִצְרַ֔יִם בְּמֵיטַ֥ב הָאָ֖רֶץ בְּאֶ֣רֶץ רַעְמְסֵ֑ס כַּֽאֲשֶׁ֖ר צִוָּ֥ה פַרְעֹֽה:  ת  וְאוֹתֵיב יוֹסֵף יָת אֲבוּהִי וְיָת אֲחוֹהִי וִיהַב לְהוֹן אַחֲסָנָא בְּאַרְעָא דְמִצְרַיִם בִּדְשַׁפִּיר בְּאַרְעָא בְּאַרְעָא דְרַעְמְסֵס כְּמָא דִי פַקֵּיד פַרְעֹה:

{יב} וַיְכַלְכֵּ֤ל יוֹסֵף֙ אֶת-אָבִ֣יו וְאֶת-אֶחָ֔יו וְאֵ֖ת כָּל-בֵּ֣ית אָבִ֑יו לֶ֖חֶם לְפִ֥י הַטָּֽף:  מ  וַיְכַלְכֵּ֤ל יוֹסֵף֙ אֶת-אָבִ֣יו וְאֶת-אֶחָ֔יו וְאֵ֖ת כָּל-בֵּ֣ית אָבִ֑יו לֶ֖חֶם לְפִ֥י הַטָּֽף:  ת  וְזָן יוֹסֵף יָת אֲבוּהִי וְיָת אֲחוֹהִי וְיָת כָּל בֵּית אֲבוּהִי לַחְמָא לְפוּם טַפְלָא:

{יג} וְלֶ֤חֶם אֵין֙ בְּכָל-הָאָ֔רֶץ כִּֽי-כָבֵ֥ד הָֽרָעָ֖ב מְאֹ֑ד וַתֵּ֜לַהּ אֶ֤רֶץ מִצְרַ֨יִם֙ וְאֶ֣רֶץ כְּנַ֔עַן מִפְּנֵ֖י הָֽרָעָֽב:  מ  וְלֶ֤חֶם אֵין֙ בְּכָל-הָאָ֔רֶץ כִּֽי-כָבֵ֥ד הָֽרָעָ֖ב מְאֹ֑ד וַתֵּ֜לַהּ אֶ֤רֶץ מִצְרַ֨יִם֙ וְאֶ֣רֶץ כְּנַ֔עַן מִפְּנֵ֖י הָֽרָעָֽב:  ת  וְלַחֲמָא לֵית בְּכָל אַרְעָא אֲרֵי תַקִּיף כַּפְנָא לַחֲדָא וְאִשְׁתַּלְהֵי עַמָּא דְאַרְעָא דְמִצְרַיִם וְעַמָּא דְאַרְעָא דִּכְנַעַן מִן קֳדָם כַּפְנָא:

{יד} וַיְלַקֵּ֣ט יוֹסֵ֗ף אֶת-כָּל-הַכֶּ֨סֶף֙ הַנִּמְצָ֤א בְאֶֽרֶץ-מִצְרַ֨יִם֙ וּבְאֶ֣רֶץ כְּנַ֔עַן בַּשֶּׁ֖בֶר אֲשֶׁר-הֵ֣ם שֹֽׁבְרִ֑ים וַיָּבֵ֥א יוֹסֵ֛ף אֶת-הַכֶּ֖סֶף בֵּ֥יתָה פַרְעֹֽה:  מ  וַיְלַקֵּ֣ט יוֹסֵ֗ף אֶת-כָּל-הַכֶּ֨סֶף֙ הַנִּמְצָ֤א בְאֶֽרֶץ-מִצְרַ֨יִם֙ וּבְאֶ֣רֶץ כְּנַ֔עַן בַּשֶּׁ֖בֶר אֲשֶׁר-הֵ֣ם שֹֽׁבְרִ֑ים וַיָּבֵ֥א יוֹסֵ֛ף אֶת-הַכֶּ֖סֶף בֵּ֥יתָה פַרְעֹֽה:  ת  וְלַקִּיט יוֹסֵף יָת כָּל כַּסְפָּא דְּאִשְׁתְּכַח בְּאַרְעָא דְמִצְרַיִם וּבְאַרְעָא דִּכְנַעַן בְּעִיבוּרָא דִּי אִנּוּן זָבְנִין וְאַיְתִי יוֹסֵף יָת כַּסְפָּא לְבֵית פַּרְעֹה:

{טו} וַיִּתֹּ֣ם הַכֶּ֗סֶף מֵאֶ֣רֶץ מִצְרַ֘יִם֘ וּמֵאֶ֣רֶץ כְּנַ֒עַן֒ וַיָּבֹ֩אוּ֩ כָל-מִצְרַ֨יִם אֶל-יוֹסֵ֤ף לֵאמֹר֙ הָֽבָה-לָּ֣נוּ לֶ֔חֶם וְלָ֥מָּה נָמ֖וּת נֶגְדֶּ֑ךָ כִּ֥י אָפֵ֖ס כָּֽסֶף:  מ  וַיִּתֹּ֣ם הַכֶּ֗סֶף מֵאֶ֣רֶץ מִצְרַ֘יִם֘ וּמֵאֶ֣רֶץ כְּנַ֒עַן֒ וַיָּבֹ֩אוּ֩ כָל-מִצְרַ֨יִם אֶל-יוֹסֵ֤ף לֵאמֹר֙ הָֽבָה-לָּ֣נוּ לֶ֔חֶם וְלָ֥מָּה נָמ֖וּת נֶגְדֶּ֑ךָ כִּ֥י אָפֵ֖ס כָּֽסֶף:  ת  וּשְׁלִים כַּסְפָא מֵאַרְעָא דְמִצְרַיִם וּמֵאַרְעָא דִּכְנַעַן וַאֲתוֹ כָּל מִצְרָאֵי לְוַת יוֹסֵף לְמֵימַר הַב לָנָא לַחֲמָא וּלְמָה נְמוּת לְקִבְלָךְ אֲרֵי שְׁלִים כַּסְפָּא:

{טז} וַיֹּ֤אמֶר יוֹסֵף֙ הָב֣וּ מִקְנֵיכֶ֔ם וְאֶתְּנָ֥ה לָכֶ֖ם בְּמִקְנֵיכֶ֑ם אִם-אָפֵ֖ס כָּֽסֶף:  מ  וַיֹּ֤אמֶר יוֹסֵף֙ הָב֣וּ מִקְנֵיכֶ֔ם וְאֶתְּנָ֥ה לָכֶ֖ם בְּמִקְנֵיכֶ֑ם אִם-אָפֵ֖ס כָּֽסֶף:  ת  וַאֲמַר יוֹסֵף הָבוּ גֵּיתֵיכוֹן וְאֶתֵּן לְכוֹן בְּגֵּיתֵיכוֹן אִם שְׁלִים כַּסְפָּא:

{יז} וַיָּבִ֣יאוּ אֶת-מִקְנֵיהֶם֘ אֶל-יוֹסֵף֒ וַיִּתֵּ֣ן לָהֶם֩ יוֹסֵ֨ף לֶ֜חֶם בַּסּוּסִ֗ים וּבְמִקְנֵ֥ה הַצֹּ֛אן וּבְמִקְנֵ֥ה הַבָּקָ֖ר וּבַֽחֲמֹרִ֑ים וַיְנַֽהֲלֵ֤ם בַּלֶּ֨חֶם֙ בְּכָל-מִקְנֵהֶ֔ם בַּשָּׁנָ֖ה הַהִֽוא:  מ  וַיָּבִ֣יאוּ אֶת-מִקְנֵיהֶם֘ אֶל-יוֹסֵף֒ וַיִּתֵּ֣ן לָהֶם֩ יוֹסֵ֨ף לֶ֜חֶם בַּסּוּסִ֗ים וּבְמִקְנֵ֥ה הַצֹּ֛אן וּבְמִקְנֵ֥ה הַבָּקָ֖ר וּבַֽחֲמֹרִ֑ים וַיְנַֽהֲלֵ֤ם בַּלֶּ֨חֶם֙ בְּכָל-מִקְנֵהֶ֔ם בַּשָּׁנָ֖ה הַהִֽוא:  ת  וְאַיְתִיּוּ יָת גֵּיתֵיהוֹן לְוַת יוֹסֵף וִיהַב לְהוֹן יוֹסֵף לַחְמָא בְּסוּסְוָתָא וּבְגֵיתֵי עָנָא וּבְגֵיתֵי תוֹרִין וּבַחֲמָרִין וְזָנִנּוּן בְּלַחֲמָא בְּכָל גֵּיתֵיהוֹן בְּשַׁתָּא הַהִיא:

{יח} וַתִּתֹּם֘ הַשָּׁנָ֣ה הַהִוא֒ וַיָּבֹ֨אוּ אֵלָ֜יו בַּשָּׁנָ֣ה הַשֵּׁנִ֗ית וַיֹּ֤אמְרוּ לוֹ֙ לֹֽא-נְכַחֵ֣ד מֵֽאֲדֹנִ֔י כִּ֚י אִם-תַּ֣ם הַכֶּ֔סֶף וּמִקְנֵ֥ה הַבְּהֵמָ֖ה אֶל-אֲדֹנִ֑י לֹ֤א נִשְׁאַר֙ לִפְנֵ֣י אֲדֹנִ֔י בִּלְתִּ֥י אִם-גְּוִיָּתֵ֖נוּ וְאַדְמָתֵֽנוּ:  מ  וַתִּתֹּם֘ הַשָּׁנָ֣ה הַהִוא֒ וַיָּבֹ֨אוּ אֵלָ֜יו בַּשָּׁנָ֣ה הַשֵּׁנִ֗ית וַיֹּ֤אמְרוּ לוֹ֙ לֹֽא-נְכַחֵ֣ד מֵֽאֲדֹנִ֔י כִּ֚י אִם-תַּ֣ם הַכֶּ֔סֶף וּמִקְנֵ֥ה הַבְּהֵמָ֖ה אֶל-אֲדֹנִ֑י לֹ֤א נִשְׁאַר֙ לִפְנֵ֣י אֲדֹנִ֔י בִּלְתִּ֥י אִם-גְּוִיָּתֵ֖נוּ וְאַדְמָתֵֽנוּ:  ת  וּשְׁלִימַת שַׁתָּא הַהִיא וַאֲתוֹ לְוָתֵהּ בְּשַׁתָּא תִנְיֵתָא וַאֲמָרוּ לֵהּ לָא נְכַסֵּי מִן רִבּוֹנִי אֶלָּהֵין שְׁלִים כַּסְפָּא וְגֵּיתֵי בְעִירָא לְוַת רִבּוֹנִי לָא אִשְׁתָּאַר קֳדָם רִבּוֹנִי אֶלָּהֵין גְּוִיָתָנָא וְאַרְעָנָא:

{יט} לָ֧מָּה נָמ֣וּת לְעֵינֶ֗יךָ גַּם-אֲנַ֨חְנוּ֙ גַּ֣ם אַדְמָתֵ֔נוּ קְנֵֽה-אֹתָ֥נוּ וְאֶת-אַדְמָתֵ֖נוּ בַּלָּ֑חֶם וְנִֽהְיֶ֞ה אֲנַ֤חְנוּ וְאַדְמָתֵ֨נוּ֙ עֲבָדִ֣ים לְפַרְעֹ֔ה וְתֶן-זֶ֗רַע וְנִֽחְיֶה֙ וְלֹ֣א נָמ֔וּת וְהָֽאֲדָמָ֖ה לֹ֥א תֵשָֽׁם:  מ  לָ֧מָּה נָמ֣וּת לְעֵינֶ֗יךָ גַּם-אֲנַ֨חְנוּ֙ גַּ֣ם אַדְמָתֵ֔נוּ קְנֵֽה-אֹתָ֥נוּ וְאֶת-אַדְמָתֵ֖נוּ בַּלָּ֑חֶם וְנִֽהְיֶ֞ה אֲנַ֤חְנוּ וְאַדְמָתֵ֨נוּ֙ עֲבָדִ֣ים לְפַרְעֹ֔ה וְתֶן-זֶ֗רַע וְנִֽחְיֶה֙ וְלֹ֣א נָמ֔וּת וְהָֽאֲדָמָ֖ה לֹ֥א תֵשָֽׁם:  ת  לְמָה נְמוּת לְעֵינָיךְ אַף אֲנַחְנָא אַף אַרְעָתָנָא קְנֵי יָתָנָא וְיָת אַרְעָתָנָא בְּלַחְמָא וּנְהֵי אֲנַחְנָא וְאַרְעָתָנָא עַבְדִּין לְפַרְעֹה וְהַב בַּר זַרְעָא וְנֵיחֵי וְלָא נְמוּת וְאַרְעָא לָא תְבוּר:

{כ} וַיִּ֨קֶן יוֹסֵ֜ף אֶת-כָּל-אַדְמַ֤ת מִצְרַ֨יִם֙ לְפַרְעֹ֔ה כִּי-מָֽכְר֤וּ מִצְרַ֨יִם֙ אִ֣ישׁ שָׂדֵ֔הוּ כִּֽי-חָזַ֥ק עֲלֵהֶ֖ם הָרָעָ֑ב וַתְּהִ֥י הָאָ֖רֶץ לְפַרְעֹֽה:  מ  וַיִּ֨קֶן יוֹסֵ֜ף אֶת-כָּל-אַדְמַ֤ת מִצְרַ֨יִם֙ לְפַרְעֹ֔ה כִּי-מָֽכְר֤וּ מִצְרַ֨יִם֙ אִ֣ישׁ שָׂדֵ֔הוּ כִּֽי-חָזַ֥ק עֲלֵהֶ֖ם הָרָעָ֑ב וַתְּהִ֥י הָאָ֖רֶץ לְפַרְעֹֽה:  ת  וּקְנָא יוֹסֵף יָת כָּל אַרְעָא דְמִצְרַיִם לְפַרְעֹה אֲרֵי זַבִּינוּ מִצְרַיִם גְּבַר חַקְלֵהּ אֲרֵי תְקֵיף עֲלֵיהוֹן כַּפְנָא וַהֲוַת אַרְעָא לְפַרְעֹה:

{כא} וְאֶ֨ת-הָעָ֔ם הֶֽעֱבִ֥יר אֹת֖וֹ לֶֽעָרִ֑ים מִקְצֵ֥ה גְבוּל-מִצְרַ֖יִם וְעַד-קָצֵֽהוּ:  מ  וְאֶ֨ת-הָעָ֔ם הֶֽעֱבִ֥יר אֹת֖וֹ לֶֽעָרִ֑ים מִקְצֵ֥ה גְבוּל-מִצְרַ֖יִם וְעַד-קָצֵֽהוּ:  ת  וְיָת עַמָּא אַעְבֵּר יָתֵהּ מִקְּרֵי לִקְרֵי מִסְּיָפֵי תְּחוּם מִצְרַיִם וְעַד סוֹפֵהּ:

{כב} רַ֛ק אַדְמַ֥ת הַכֹּֽהֲנִ֖ים לֹ֣א קָנָ֑ה כִּי֩ חֹ֨ק לַכֹּֽהֲנִ֜ים מֵאֵ֣ת פַּרְעֹ֗ה וְאָֽכְל֤וּ אֶת-חֻקָּם֙ אֲשֶׁ֨ר נָתַ֤ן לָהֶם֙ פַּרְעֹ֔ה עַל-כֵּ֕ן לֹ֥א מָֽכְר֖וּ אֶת-אַדְמָתָֽם:  מ  רַ֛ק אַדְמַ֥ת הַכֹּֽהֲנִ֖ים לֹ֣א קָנָ֑ה כִּי֩ חֹ֨ק לַכֹּֽהֲנִ֜ים מֵאֵ֣ת פַּרְעֹ֗ה וְאָֽכְל֤וּ אֶת-חֻקָּם֙ אֲשֶׁ֨ר נָתַ֤ן לָהֶם֙ פַּרְעֹ֔ה עַל-כֵּ֕ן לֹ֥א מָֽכְר֖וּ אֶת-אַדְמָתָֽם:  ת  לְחוֹד אַרְעָא דְכֻמָּרַיָּא לָא זַבִּין אֲרֵי חֳלָקָא לְכֻמָּרַיָּא מִלְּוַת פַּרְעֹה וְאָכְלִין יָת חֳלָקְהוֹן דִּי יְהַב לְהוֹן פַּרְעֹה עַל כֵּן לָא זַבִּינוּ יָת אַרְעֲהוֹן:

{כג} וַיֹּ֤אמֶר יוֹסֵף֙ אֶל-הָעָ֔ם הֵן֩ קָנִ֨יתִי אֶתְכֶ֥ם הַיּ֛וֹם וְאֶת-אַדְמַתְכֶ֖ם לְפַרְעֹ֑ה הֵֽא-לָכֶ֣ם זֶ֔רַע וּזְרַעְתֶּ֖ם אֶת-הָֽאֲדָמָֽה:  מ  וַיֹּ֤אמֶר יוֹסֵף֙ אֶל-הָעָ֔ם הֵן֩ קָנִ֨יתִי אֶתְכֶ֥ם הַיּ֛וֹם וְאֶת-אַדְמַתְכֶ֖ם לְפַרְעֹ֑ה הֵֽא-לָכֶ֣ם זֶ֔רַע וּזְרַעְתֶּ֖ם אֶת-הָֽאֲדָמָֽה:  ת  וַאֲמַר יוֹסֵף לְעַמָּא הָא זְבֵנִית יָתְכוֹן יוֹמָא דֵּין וְיָת אַרְעֲכוֹן לְפַרְעֹה הֵא לְכוֹן בַּר זַרְעָא וְתִזְרְעוּן יָת אַרְעָא:

{כד} וְהָיָה֙ בַּתְּבוּאֹ֔ת וּנְתַתֶּ֥ם חֲמִישִׁ֖ית לְפַרְעֹ֑ה וְאַרְבַּ֣ע הַיָּדֹ֡ת יִֽהְיֶ֣ה לָכֶם֩ לְזֶ֨רַע הַשָּׂדֶ֧ה וּֽלְאָכְלְכֶ֛ם וְלַֽאֲשֶׁ֥ר בְּבָֽתֵּיכֶ֖ם וְלֶֽאֱכֹ֥ל לְטַפְּכֶֽם:  מ  וְהָיָה֙ בַּתְּבוּאֹ֔ת וּנְתַתֶּ֥ם חֲמִישִׁ֖ית לְפַרְעֹ֑ה וְאַרְבַּ֣ע הַיָּדֹ֡ת יִֽהְיֶ֣ה לָכֶם֩ לְזֶ֨רַע הַשָּׂדֶ֧ה וּֽלְאָכְלְכֶ֛ם וְלַֽאֲשֶׁ֥ר בְּבָֽתֵּיכֶ֖ם וְלֶֽאֱכֹ֥ל לְטַפְּכֶֽם:  ת  וִיהֵי בְּאָעוֹלֵי עֲלַלְתָּא וְתִתְּנוּן חַד מִן חַמְשָׁא לְפַרְעֹה וְאַרְבַּע חֳלָקִין יְהֵא לְכוֹן לְבַר זְרַע חַקְלָא וּלְמֵיכַלְכוֹן וְלֶאֱנַשׁ בָתֵּיכוֹן וּלְמֵיכַל לְטַפְלְכוֹן:

{כה}  מפטיר  וַיֹּֽאמְר֖וּ הֶֽחֱיִתָ֑נוּ נִמְצָא-חֵן֙ בְּעֵינֵ֣י אֲדֹנִ֔י וְהָיִ֥ינוּ עֲבָדִ֖ים לְפַרְעֹֽה:  מ  וַיֹּֽאמְר֖וּ הֶֽחֱיִתָ֑נוּ נִמְצָא-חֵן֙ בְּעֵינֵ֣י אֲדֹנִ֔י וְהָיִ֥ינוּ עֲבָדִ֖ים לְפַרְעֹֽה:  ת  וַאֲמָרוּ קַיִּמְתָּנָא נִשְׁכַּח רַחֲמִין בְּעֵינֵי רִבּוֹנִי וּנְהֵי עַבְדִּין לְפַרְעֹה:

{כו} וַיָּ֣שֶׂם אֹתָ֣הּ יוֹסֵ֡ף לְחֹק֩ עַד-הַיּ֨וֹם הַזֶּ֜ה עַל-אַדְמַ֥ת מִצְרַ֛יִם לְפַרְעֹ֖ה לַחֹ֑מֶשׁ רַ֞ק אַדְמַ֤ת הַכֹּֽהֲנִים֙ לְבַדָּ֔ם לֹ֥א הָֽיְתָ֖ה לְפַרְעֹֽה:  מ  וַיָּ֣שֶׂם אֹתָ֣הּ יוֹסֵ֡ף לְחֹק֩ עַד-הַיּ֨וֹם הַזֶּ֜ה עַל-אַדְמַ֥ת מִצְרַ֛יִם לְפַרְעֹ֖ה לַחֹ֑מֶשׁ רַ֞ק אַדְמַ֤ת הַכֹּֽהֲנִים֙ לְבַדָּ֔ם לֹ֥א הָֽיְתָ֖ה לְפַרְעֹֽה:  ת  וְשַׁוִּי יָתַהּ יוֹסֵף לִגְזֵרָא עַד יוֹמָא הָדֵין עַל אַרְעָא דְמִצְרַיִם דִּיהוֹן יָהֲבִין חַד מִן חַמְשָׁא לְפַרְעֹה לְחוֹד אַרְעָא דְכֻמָּרַיָּא בִּלְחוֹדֵיהוֹן לָא הֲוַת לְפַרְעֹה:
