פרשת נשא


{כא} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{כב} נָשֹׂ֗א אֶת-רֹ֛אשׁ בְּנֵ֥י גֵֽרְשׁ֖וֹן גַּם-הֵ֑ם לְבֵ֥ית אֲבֹתָ֖ם לְמִשְׁפְּחֹתָֽם:  מ  נָשֹׂ֗א אֶת-רֹ֛אשׁ בְּנֵ֥י גֵֽרְשׁ֖וֹן גַּם-הֵ֑ם לְבֵ֥ית אֲבֹתָ֖ם לְמִשְׁפְּחֹתָֽם:  ת  קַבֵּל יָת חֻשְׁבַּן בְּנֵי גֵרְשׁוֹן אַף אִנוּן לְבֵית אֲבָהַתְהוֹן לְזַרְעֲיַתְהוֹן:

{כג} מִבֶּן֩ שְׁלֹשִׁ֨ים שָׁנָ֜ה וָמַ֗עְלָה עַ֛ד בֶּן-חֲמִשִּׁ֥ים שָׁנָ֖ה תִּפְקֹ֣ד אוֹתָ֑ם כָּל-הַבָּא֨ לִצְבֹ֣א צָבָ֔א לַֽעֲבֹ֥ד עֲבֹדָ֖ה בְּאֹ֥הֶל מוֹעֵֽד:  מ  מִבֶּן֩ שְׁלֹשִׁ֨ים שָׁנָ֜ה וָמַ֗עְלָה עַ֛ד בֶּן-חֲמִשִּׁ֥ים שָׁנָ֖ה תִּפְקֹ֣ד אוֹתָ֑ם כָּל-הַבָּא֨ לִצְבֹ֣א צָבָ֔א לַֽעֲבֹ֥ד עֲבֹדָ֖ה בְּאֹ֥הֶל מוֹעֵֽד:  ת  מִבַּר תְּלָתִין שְׁנִין וּלְעֵלָא עַד בַּר חַמְשִׁין שְׁנִין תִּמְנֵי יַתְהוֹן כָּל דְאָתֵי לְחַיָלָא חֵילָא לְמִפְלַח פּוּלְחָנָא בְּמַשְׁכַּן זִמְנָא:

{כד} זֹ֣את עֲבֹדַ֔ת מִשְׁפְּחֹ֖ת הַגֵּֽרְשֻׁנִּ֑י לַֽעֲבֹ֖ד וּלְמַשָּֽׂא:  מ  זֹ֣את עֲבֹדַ֔ת מִשְׁפְּחֹ֖ת הַגֵּֽרְשֻׁנִּ֑י לַֽעֲבֹ֖ד וּלְמַשָּֽׂא:  ת  דֵין פּוּלְחַן זַרְעֲיַת גֵרְשׁוֹן לְמִפְלַח וּלְמַטוּל:

{כה} וְנָ֨שְׂא֜וּ אֶת-יְרִיעֹ֤ת הַמִּשְׁכָּן֙ וְאֶת-אֹ֣הֶל מוֹעֵ֔ד מִכְסֵ֕הוּ וּמִכְסֵ֛ה הַתַּ֥חַשׁ אֲשֶׁר-עָלָ֖יו מִלְמָ֑עְלָה וְאֶ֨ת-מָסַ֔ךְ פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  מ  וְנָ֨שְׂא֜וּ אֶת-יְרִיעֹ֤ת הַמִּשְׁכָּן֙ וְאֶת-אֹ֣הֶל מוֹעֵ֔ד מִכְסֵ֕הוּ וּמִכְסֵ֛ה הַתַּ֥חַשׁ אֲשֶׁר-עָלָ֖יו מִלְמָ֑עְלָה וְאֶ֨ת-מָסַ֔ךְ פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  ת  וְיִטְלוּן יָת יְרִיעַת מַשְׁכְּנָא וְיָת מַשְׁכַּן זִמְנָא חוֹפָאֵהּ וְחוֹפָאָה דְסַסְגוֹנָא דִי עֲלוֹהִי מִלְעֵלָא וְיָת פְּרָסָא דִתְרַע מַשְׁכַּן זִמְנָא:

{כו} וְאֵת֩ קַלְעֵ֨י הֶֽחָצֵ֜ר וְאֶת-מָסַ֣ךְ | פֶּ֣תַח | שַׁ֣עַר הֶֽחָצֵ֗ר אֲשֶׁ֨ר עַל-הַמִּשְׁכָּ֤ן וְעַל-הַמִּזְבֵּ֨חַ֙ סָבִ֔יב וְאֵת֙ מֵֽיתְרֵיהֶ֔ם וְאֶֽת-כָּל-כְּלֵ֖י עֲבֹֽדָתָ֑ם וְאֵ֨ת כָּל-אֲשֶׁ֧ר יֵֽעָשֶׂ֛ה לָהֶ֖ם וְעָבָֽדוּ:  מ  וְאֵת֩ קַלְעֵ֨י הֶֽחָצֵ֜ר וְאֶת-מָסַ֣ךְ | פֶּ֣תַח | שַׁ֣עַר הֶֽחָצֵ֗ר אֲשֶׁ֨ר עַל-הַמִּשְׁכָּ֤ן וְעַל-הַמִּזְבֵּ֨חַ֙ סָבִ֔יב וְאֵת֙ מֵֽיתְרֵיהֶ֔ם וְאֶֽת-כָּל-כְּלֵ֖י עֲבֹֽדָתָ֑ם וְאֵ֨ת כָּל-אֲשֶׁ֧ר יֵֽעָשֶׂ֛ה לָהֶ֖ם וְעָבָֽדוּ:  ת  וְיָת סְרָדֵי דְדַרְתָּא וְיָת פְּרָסָא דְמַעֲלָנָא דִתְרַע דַרְתָּא דִי עַל מַשְׁכְּנָא וְעַל מַדְבְּחָא סְחוֹר סְחוֹר וְיָת אַטוּנֵיהוֹן וְיָת כָּל מָנֵי פּוּלְחַנְהוֹן וְיַת כָּל דִי יִתְמְסַר לְהוֹן וְיִפְלְחוּן:

{כז} עַל-פִּי֩ אַֽהֲרֹ֨ן וּבָנָ֜יו תִּֽהְיֶ֗ה כָּל-עֲבֹדַת֙ בְּנֵ֣י הַגֵּֽרְשֻׁנִּ֔י לְכָ֨ל-מַשָּׂאָ֔ם וּלְכֹ֖ל עֲבֹֽדָתָ֑ם וּפְקַדְתֶּ֤ם עֲלֵהֶם֙ בְּמִשְׁמֶ֔רֶת אֵ֖ת כָּל-מַשָּׂאָֽם:  מ  עַל-פִּי֩ אַֽהֲרֹ֨ן וּבָנָ֜יו תִּֽהְיֶ֗ה כָּל-עֲבֹדַת֙ בְּנֵ֣י הַגֵּֽרְשֻׁנִּ֔י לְכָ֨ל-מַשָּׂאָ֔ם וּלְכֹ֖ל עֲבֹֽדָתָ֑ם וּפְקַדְתֶּ֤ם עֲלֵהֶם֙ בְּמִשְׁמֶ֔רֶת אֵ֖ת כָּל-מַשָּׂאָֽם:  ת  עַל מֵימַר אַהֲרֹן וּבְנוֹהִי יְהֵי כָּל פּוּלְחַן בְּנֵי גֵרְשׁוֹן לְכָל מַטוּלְהוֹן וּלְכֹל פּוּלְחַנְהוֹן וּתְמַנוּן עֲלֵיהוֹן בְּמַטָרָא יָת כָּל מַטוּלְהוֹן:

{כח} זֹ֣את עֲבֹדַ֗ת מִשְׁפְּחֹ֛ת בְּנֵ֥י הַגֵּֽרְשֻׁנִּ֖י בְּאֹ֣הֶל מוֹעֵ֑ד וּמִ֨שְׁמַרְתָּ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן-אַהֲרֹ֖ן הַכֹּהֵֽן:  מ  זֹ֣את עֲבֹדַ֗ת מִשְׁפְּחֹ֛ת בְּנֵ֥י הַגֵּֽרְשֻׁנִּ֖י בְּאֹ֣הֶל מוֹעֵ֑ד וּמִ֨שְׁמַרְתָּ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן-אַהֲרֹ֖ן הַכֹּהֵֽן:  ת  דֵין פּוּלְחַן זַרְעֲיַת בְּנֵי גֵרְשּׁוֹן בְּמַשְׁכַּן זִמְנָא וּמַטַרְתְּהוֹן בִּידָא דְאִיתָמָר בַּר אַהֲרֹן כַּהֲנָא: (ס)

{כט} בְּנֵ֖י מְרָרִ֑י לְמִשְׁפְּחֹתָ֥ם לְבֵית-אֲבֹתָ֖ם תִּפְקֹ֥ד אֹתָֽם:  מ  בְּנֵ֖י מְרָרִ֑י לְמִשְׁפְּחֹתָ֥ם לְבֵית-אֲבֹתָ֖ם תִּפְקֹ֥ד אֹתָֽם:  ת  בְּנֵי מְרָרִי לְזַרְעֲיַתְהוֹן לְבֵית אֲבָהַתְהוֹן תִּמְנֵי יַתְהוֹן:

{ל} מִבֶּן֩ שְׁלֹשִׁ֨ים שָׁנָ֜ה וָמַ֗עְלָה וְעַ֛ד בֶּן-חֲמִשִּׁ֥ים שָׁנָ֖ה תִּפְקְדֵ֑ם כָּל-הַבָּא֙ לַצָּבָ֔א לַֽעֲבֹ֕ד אֶת-עֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד:  מ  מִבֶּן֩ שְׁלֹשִׁ֨ים שָׁנָ֜ה וָמַ֗עְלָה וְעַ֛ד בֶּן-חֲמִשִּׁ֥ים שָׁנָ֖ה תִּפְקְדֵ֑ם כָּל-הַבָּא֙ לַצָּבָ֔א לַֽעֲבֹ֕ד אֶת-עֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד:  ת  מִבַּר תְּלָתִין שְׁנִין וּלְעֵלָא וְעַד בַּר חַמְשִׁין שְׁנִין תִּמְנִנוּן כָּל דְאָתֵי לְחֵילָא לְמִפְלָח יָת פּוּלְחַן מַשְׁכַּן זִמְנָא:

{לא} וְזֹאת֙ מִשְׁמֶ֣רֶת מַשָּׂאָ֔ם לְכָל-עֲבֹֽדָתָ֖ם בְּאֹ֣הֶל מוֹעֵ֑ד קַרְשֵׁי֙ הַמִּשְׁכָּ֔ן וּבְרִיחָ֖יו וְעַמּוּדָ֥יו וַֽאֲדָנָֽיו:  מ  וְזֹאת֙ מִשְׁמֶ֣רֶת מַשָּׂאָ֔ם לְכָל-עֲבֹֽדָתָ֖ם בְּאֹ֣הֶל מוֹעֵ֑ד קַרְשֵׁי֙ הַמִּשְׁכָּ֔ן וּבְרִיחָ֖יו וְעַמּוּדָ֥יו וַֽאֲדָנָֽיו:  ת  וְדָא מַטְרַת מַטוּלְהוֹן לְכָל פּוּלְחַנְהוֹן בְּמַשְׁכַּן זִמְנָא דַפֵּי מַשְׁכְּנָא וְעַבְרוֹהִי וְעַמוּדוֹהִי וְסַמְכוֹהִי:

{לב} וְעַמּוּדֵי֩ הֶֽחָצֵ֨ר סָבִ֜יב וְאַדְנֵיהֶ֗ם וִֽיתֵדֹתָם֙ וּמֵ֣יתְרֵיהֶ֔ם לְכָ֨ל-כְּלֵיהֶ֔ם וּלְכֹ֖ל עֲבֹֽדָתָ֑ם וּבְשֵׁמֹ֣ת תִּפְקְד֔וּ אֶת-כְּלֵ֖י מִשְׁמֶ֥רֶת מַשָּׂאָֽם:  מ  וְעַמּוּדֵי֩ הֶֽחָצֵ֨ר סָבִ֜יב וְאַדְנֵיהֶ֗ם וִֽיתֵדֹתָם֙ וּמֵ֣יתְרֵיהֶ֔ם לְכָ֨ל-כְּלֵיהֶ֔ם וּלְכֹ֖ל עֲבֹֽדָתָ֑ם וּבְשֵׁמֹ֣ת תִּפְקְד֔וּ אֶת-כְּלֵ֖י מִשְׁמֶ֥רֶת מַשָּׂאָֽם:  ת  וְעֲמוּדֵי דְדַרְתָּא סְחוֹר סְחוֹר וְסַמְכֵיהוֹן וְסִכֵּיהוֹן וְאַטוּנֵיהוֹן לְכָל מָנֵיהוֹן וּלְכֹל פּוּלְחַנְהוֹן וּבִשְׁמָהָן תְּמַנוּן יָת מָנֵי מַטְרַת מַטוּלְהוֹן:

{לג} זֹ֣את עֲבֹדַ֗ת מִשְׁפְּחֹת֙ בְּנֵ֣י מְרָרִ֔י לְכָל-עֲבֹֽדָתָ֖ם בְּאֹ֣הֶל מוֹעֵ֑ד בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן-אַהֲרֹ֖ן הַכֹּהֵֽן:  מ  זֹ֣את עֲבֹדַ֗ת מִשְׁפְּחֹת֙ בְּנֵ֣י מְרָרִ֔י לְכָל-עֲבֹֽדָתָ֖ם בְּאֹ֣הֶל מוֹעֵ֑ד בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן-אַהֲרֹ֖ן הַכֹּהֵֽן:  ת  דֵין פּוּלְחַן זַרְעֲיַת בְּנֵי מְרָרִי לְכָל פּוּלְחַנְהוֹן בְּמַשְׁכַּן זִמְנָא בִּידָא דְאִיתָמָר בַּר אַהֲרֹן כַּהֲנָא:

{לד} וַיִּפְקֹ֨ד מֹשֶׁ֧ה וְאַֽהֲרֹ֛ן וּנְשִׂיאֵ֥י הָֽעֵדָ֖ה אֶת-בְּנֵ֣י הַקְּהָתִ֑י לְמִשְׁפְּחֹתָ֖ם וּלְבֵ֥ית אֲבֹתָֽם:  מ  וַיִּפְקֹ֨ד מֹשֶׁ֧ה וְאַֽהֲרֹ֛ן וּנְשִׂיאֵ֥י הָֽעֵדָ֖ה אֶת-בְּנֵ֣י הַקְּהָתִ֑י לְמִשְׁפְּחֹתָ֖ם וּלְבֵ֥ית אֲבֹתָֽם:  ת  וּמְנָא משֶׁה וְאַהֲרֹן וְרַבְרְבֵי כְנִשְׁתָּא יָת בְּנֵי קְהָת לְזַרְעֲיַתְהוֹן וּלְבֵית אֲבָהַתְהוֹן:

{מב} וּפְקוּדֵ֕י מִשְׁפְּחֹ֖ת בְּנֵ֣י מְרָרִ֑י לְמִשְׁפְּחֹתָ֖ם לְבֵ֥ית אֲבֹתָֽם:  מ  וּפְקוּדֵ֕י מִשְׁפְּחֹ֖ת בְּנֵ֣י מְרָרִ֑י לְמִשְׁפְּחֹתָ֖ם לְבֵ֥ית אֲבֹתָֽם:  ת  וּמִנְיָנֵי זַרְעֲיַת בְּנֵי מְרָרִי לְזַרְעֲיַתְהוֹן לְבֵית אֲבָהַתְהוֹן:

{מז} מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן-חֲמִשִּׁ֣ים שָׁנָ֑ה כָּל-הַבָּ֗א לַֽעֲבֹ֨ד עֲבֹדַ֧ת עֲבֹדָ֛ה וַֽעֲבֹדַ֥ת מַשָּׂ֖א בְּאֹ֥הֶל מוֹעֵֽד:  מ  מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן-חֲמִשִּׁ֣ים שָׁנָ֑ה כָּל-הַבָּ֗א לַֽעֲבֹ֨ד עֲבֹדַ֧ת עֲבֹדָ֛ה וַֽעֲבֹדַ֥ת מַשָּׂ֖א בְּאֹ֥הֶל מוֹעֵֽד:  ת  מִבַּר תְּלָתִין שְׁנִין וּלְעֵלָא וְעַד בַּר חַמְשִׁין שְׁנִין כָּל דְאָתֵי לְמִפְלַח פּוּלְחַן פּוּלְחָנָא וּפוּלְחַן מַטוּל בְּמַשְׁכַּן זִמְנָא:

{מח} וַיִּֽהְי֖וּ פְּקֻֽדֵיהֶ֑ם שְׁמֹנַ֣ת אֲלָפִ֔ים וַֽחֲמֵ֥שׁ מֵא֖וֹת וּשְׁמֹנִֽים:  מ  וַיִּֽהְי֖וּ פְּקֻֽדֵיהֶ֑ם שְׁמֹנַ֣ת אֲלָפִ֔ים וַֽחֲמֵ֥שׁ מֵא֖וֹת וּשְׁמֹנִֽים:  ת  וַהֲווֹ מִנְיָנֵיהוֹן תְּמַנְיָא אַלְפִין וַחֲמֵשׁ מְאָה וְתַמְנָן:

{מט} עַל-פִּ֨י יְהֹוָ֜ה פָּקַ֤ד אוֹתָם֙ בְּיַד-מֹשֶׁ֔ה אִ֥ישׁ אִ֛ישׁ עַל-עֲבֹֽדָת֖וֹ וְעַל-מַשָּׂא֑וֹ וּפְקֻדָ֕יו אֲשֶׁר-צִוָּ֥ה יְהֹוָ֖ה אֶת-מֹשֶֽׁה:  מ  עַל-פִּ֨י יְהֹוָ֜ה פָּקַ֤ד אוֹתָם֙ בְּיַד-מֹשֶׁ֔ה אִ֥ישׁ אִ֛ישׁ עַל-עֲבֹֽדָת֖וֹ וְעַל-מַשָּׂא֑וֹ וּפְקֻדָ֕יו אֲשֶׁר-צִוָּ֥ה יְהֹוָ֖ה אֶת-מֹשֶֽׁה:  ת  עַל מֵימְרָא דַיְיָ מְנָא יָתְהוֹן בִּידָא דְמשֶׁה גְבַר גְבַר עַל פּוּלְחָנֵהּ וְעַל מַטוּלֵהּ וּמִנְיָנוֹהִי דִי פַקִיד יְיָ יָת משֶׁה: (פ)




במדבר פרק-ה

{א}  שלישי  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{ב} צַ֚ו אֶת-בְּנֵ֣י יִשְׂרָאֵ֔ל וִֽישַׁלְּחוּ֙ מִן-הַֽמַּחֲנֶ֔ה כָּל-צָר֖וּעַ וְכָל-זָ֑ב וְכֹ֖ל טָמֵ֥א לָנָֽפֶשׁ:  מ  צַ֚ו אֶת-בְּנֵ֣י יִשְׂרָאֵ֔ל וִֽישַׁלְּחוּ֙ מִן-הַֽמַּחֲנֶ֔ה כָּל-צָר֖וּעַ וְכָל-זָ֑ב וְכֹ֖ל טָמֵ֥א לָנָֽפֶשׁ:  ת  פַּקֵיד יָת בְּנֵי יִשְׂרָאֵל וִישַׁלְחוּן מִן מַשְׁרִיתָא כָּל דִסְגִיר וְכָל דְדָאֵב וְכֹל דִמְסָאָב לִטְמֵי נַפְשָׁא דֶאֱנָשָׁא:

{ג} מִזָּכָ֤ר עַד-נְקֵבָה֙ תְּשַׁלֵּ֔חוּ אֶל-מִח֥וּץ לַֽמַּחֲנֶ֖ה תְּשַׁלְּח֑וּם וְלֹ֤א יְטַמְּאוּ֙ אֶת-מַ֣חֲנֵיהֶ֔ם אֲשֶׁ֥ר אֲנִ֖י שֹׁכֵ֥ן בְּתוֹכָֽם:  מ  מִזָּכָ֤ר עַד-נְקֵבָה֙ תְּשַׁלֵּ֔חוּ אֶל-מִח֥וּץ לַֽמַּחֲנֶ֖ה תְּשַׁלְּח֑וּם וְלֹ֤א יְטַמְּאוּ֙ אֶת-מַ֣חֲנֵיהֶ֔ם אֲשֶׁ֥ר אֲנִ֖י שֹׁכֵ֥ן בְּתוֹכָֽם:  ת  מִדְכַר עַד נוּקְבָא תְּשַׁלְחוּן לְמִבָּרָא לְמַשְׁרִיתָא תְּשַׁלְחֻנוּן וְלָא יְסַאֲבוּן יָת מַשְׁרִיתְהוֹן דִי שְׁכִינְתִּי שַׁרְיָא בֵינֵיהוֹן:

{ד} וַיַּֽעֲשׂוּ-כֵן֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיְשַׁלְּח֣וּ אוֹתָ֔ם אֶל-מִח֖וּץ לַֽמַּחֲנֶ֑ה כַּֽאֲשֶׁ֨ר דִּבֶּ֤ר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה כֵּ֥ן עָשׂ֖וּ בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וַיַּֽעֲשׂוּ-כֵן֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיְשַׁלְּח֣וּ אוֹתָ֔ם אֶל-מִח֖וּץ לַֽמַּחֲנֶ֑ה כַּֽאֲשֶׁ֨ר דִּבֶּ֤ר יְהֹוָה֙ אֶל-מֹשֶׁ֔ה כֵּ֥ן עָשׂ֖וּ בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וַעֲבָדוּ כֵן בְּנֵי יִשְׂרָאֵל וְשַׁלְחוּ יָתְהוֹן לְמִבָּרָא לְמַשְׁרִיתָא כְּמָא דִי מַלִיל יְיָ עִם משֶׁה כֵּן עֲבָדוּ בְּנֵי יִשְׂרָאֵל: (פ)

{ה} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{ו} דַּבֵּר֘ אֶל-בְּנֵ֣י יִשְׂרָאֵל֒ אִ֣ישׁ אֽוֹ-אִשָּׁ֗ה כִּ֤י יַֽעֲשׂוּ֙ מִכָּל-חַטֹּ֣את הָֽאָדָ֔ם לִמְעֹ֥ל מַ֖עַל בַּֽיהֹוָ֑ה וְאָֽשְׁמָ֖ה הַנֶּ֥פֶשׁ הַהִֽוא:  מ  דַּבֵּר֘ אֶל-בְּנֵ֣י יִשְׂרָאֵל֒ אִ֣ישׁ אֽוֹ-אִשָּׁ֗ה כִּ֤י יַֽעֲשׂוּ֙ מִכָּל-חַטֹּ֣את הָֽאָדָ֔ם לִמְעֹ֥ל מַ֖עַל בַּֽיהֹוָ֑ה וְאָֽשְׁמָ֖ה הַנֶּ֥פֶשׁ הַהִֽוא:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל גְבַר אוֹ אִתְּתָא אֲרֵי יַעְבְּדוּן מִכָּל חוֹבֵי אֱנָשָׁא לְשַׁקָרָא שְׁקָר קֳדָם יְיָ וְיֵחוֹב אֱנָשָׁא הַהוּא:

{ז} וְהִתְוַדּ֗וּ אֶֽת-חַטָּאתָם֘ אֲשֶׁ֣ר עָשׂוּ֒ וְהֵשִׁ֤יב אֶת-אֲשָׁמוֹ֙ בְּרֹאשׁ֔וֹ וַחֲמִֽישִׁת֖וֹ יֹסֵ֣ף עָלָ֑יו וְנָתַ֕ן לַֽאֲשֶׁ֖ר אָשַׁ֥ם לֽוֹ:  מ  וְהִתְוַדּ֗וּ אֶֽת-חַטָּאתָם֘ אֲשֶׁ֣ר עָשׂוּ֒ וְהֵשִׁ֤יב אֶת-אֲשָׁמוֹ֙ בְּרֹאשׁ֔וֹ וַחֲמִֽישִׁת֖וֹ יֹסֵ֣ף עָלָ֑יו וְנָתַ֕ן לַֽאֲשֶׁ֖ר אָשַׁ֥ם לֽוֹ:  ת  וִיוַדוּן יָת חוֹבֵיהוֹן דִי עֲבָדוּ וְיָתֵב יָת חוֹבְתֵהּ בְּרֵישֵׁהּ וְחֻמְשֵׁהּ יוֹסֵף עֲלוֹהִי וְיִתֵּן לִדְחָב לֵהּ:

{ח} וְאִם-אֵ֨ין לָאִ֜ישׁ גֹּאֵ֗ל לְהָשִׁ֤יב הָֽאָשָׁם֙ אֵלָ֔יו הָֽאָשָׁ֛ם הַמּוּשָׁ֥ב לַֽיהֹוָ֖ה לַכֹּהֵ֑ן מִלְּבַ֗ד אֵ֚יל הַכִּפֻּרִ֔ים אֲשֶׁ֥ר יְכַפֶּר-בּ֖וֹ עָלָֽיו:  מ  וְאִם-אֵ֨ין לָאִ֜ישׁ גֹּאֵ֗ל לְהָשִׁ֤יב הָֽאָשָׁם֙ אֵלָ֔יו הָֽאָשָׁ֛ם הַמּוּשָׁ֥ב לַֽיהֹוָ֖ה לַכֹּהֵ֑ן מִלְּבַ֗ד אֵ֚יל הַכִּפֻּרִ֔ים אֲשֶׁ֥ר יְכַפֶּר-בּ֖וֹ עָלָֽיו:  ת  וְאִם לֵית לִגְבַר פָּרִיק לַאֲתָבָא חוֹבְתָא לֵהּ חוֹבְתָא דִמְתֵב קֳדָם יְיָ לְכַהֲנָא בַּר מִדְכַר כִּפּוּרַיָא דִי יְכַפֵּר בֵּהּ עֲלוֹהִי:

{ט} וְכָל-תְּרוּמָ֞ה לְכָל-קָדְשֵׁ֧י בְנֵֽי-יִשְׂרָאֵ֛ל אֲשֶׁר-יַקְרִ֥יבוּ לַכֹּהֵ֖ן ל֥וֹ יִֽהְיֶֽה:  מ  וְכָל-תְּרוּמָ֞ה לְכָל-קָדְשֵׁ֧י בְנֵֽי-יִשְׂרָאֵ֛ל אֲשֶׁר-יַקְרִ֥יבוּ לַכֹּהֵ֖ן ל֥וֹ יִֽהְיֶֽה:  ת  וְכָל אַפְרָשׁוּתָא לְכָל קוּדְשַׁיָא דִבְנֵי יִשְׂרָאֵל דִי יְקָרְבוּן לְכַהֲנָא דִלֵהּ יְהֵא:

{י} וְאִ֥ישׁ אֶת-קֳדָשָׁ֖יו לוֹ יִֽהְי֑וּ אִ֛ישׁ אֲשֶׁר-יִתֵּ֥ן לַכֹּהֵ֖ן ל֥וֹ יִֽהְיֶֽה:  מ  וְאִ֥ישׁ אֶת-קֳדָשָׁ֖יו לוֹ יִֽהְי֑וּ אִ֛ישׁ אֲשֶׁר-יִתֵּ֥ן לַכֹּהֵ֖ן ל֥וֹ יִֽהְיֶֽה:  ת  וּגְבַר יָת מַעֲשַׂר קוּדְשֵׁהּ דִלֵהּ יְהוֹן גְבַר דְיִתֵּן לְכַהֲנָא דִלֵהּ יְהֵא: (פ)

{יא}  רביעי  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{יב} דַּבֵּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם אִ֥ישׁ אִישׁ֙ כִּֽי-תִשְׂטֶ֣ה אִשְׁתּ֔וֹ וּמָֽעֲלָ֥ה ב֖וֹ מָֽעַל:  מ  דַּבֵּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם אִ֥ישׁ אִישׁ֙ כִּֽי-תִשְׂטֶ֣ה אִשְׁתּ֔וֹ וּמָֽעֲלָ֥ה ב֖וֹ מָֽעַל:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן גְבַר גְבַר אֲרֵי תִסְטֵי אִתְּתֵהּ וּתְשַׁקֵר בֵּהּ שְׁקָר:

{יג} וְשָׁכַ֨ב אִ֣ישׁ אֹתָהּ֘ שִׁכְבַת-זֶ֒רַע֒ וְנֶעְלַם֙ מֵֽעֵינֵ֣י אִישָׁ֔הּ וְנִסְתְּרָ֖ה וְהִ֣יא נִטְמָ֑אָה וְעֵד֙ אֵ֣ין בָּ֔הּ וְהִ֖וא לֹ֥א נִתְפָּֽשָׂה:  מ  וְשָׁכַ֨ב אִ֣ישׁ אֹתָהּ֘ שִׁכְבַת-זֶ֒רַע֒ וְנֶעְלַם֙ מֵֽעֵינֵ֣י אִישָׁ֔הּ וְנִסְתְּרָ֖ה וְהִ֣יא נִטְמָ֑אָה וְעֵד֙ אֵ֣ין בָּ֔הּ וְהִ֖וא לֹ֥א נִתְפָּֽשָׂה:  ת  וְיִשְׁכֻּב גְבַר יָתַהּ שָׁכְבַת זַרְעָא וִיהֵי מְכַסָא מֵעֵינֵי בַעְלַהּ וּמִטַמְרָא וְהִיא מְסָאֲבָא וְסָהִיד לֵית בַּהּ וְהִיא לָא אִתַּחָדַת:

{יד} וְעָבַ֨ר עָלָ֧יו רֽוּחַ-קִנְאָ֛ה וְקִנֵּ֥א אֶת-אִשְׁתּ֖וֹ וְהִ֣וא נִטְמָ֑אָה אֽוֹ-עָבַ֨ר עָלָ֤יו רֽוּחַ-קִנְאָה֙ וְקִנֵּ֣א אֶת-אִשְׁתּ֔וֹ וְהִ֖יא לֹ֥א נִטְמָֽאָה:  מ  וְעָבַ֨ר עָלָ֧יו רֽוּחַ-קִנְאָ֛ה וְקִנֵּ֥א אֶת-אִשְׁתּ֖וֹ וְהִ֣וא נִטְמָ֑אָה אֽוֹ-עָבַ֨ר עָלָ֤יו רֽוּחַ-קִנְאָה֙ וְקִנֵּ֣א אֶת-אִשְׁתּ֔וֹ וְהִ֖יא לֹ֥א נִטְמָֽאָה:  ת  וְיֶעְבַּר עֲלוֹהִי רוּחַ קִנְאָה וִיקַנֵי יָת אִתְּתֵהּ וְהִיא מְסָאֲבָא אוֹ עֲבַר עֲלוֹהִי רוּחַ קִנְאָה וִיקַנֵי יָת אִתְּתֵהּ וְהִיא לָא מְסָאֲבָא:

{טו} וְהֵבִ֨יא הָאִ֣ישׁ אֶת-אִשְׁתּוֹ֘ אֶל-הַכֹּהֵן֒ וְהֵבִ֤יא אֶת-קָרְבָּנָהּ֙ עָלֶ֔יהָ עֲשִׂירִ֥ת הָֽאֵיפָ֖ה קֶ֣מַח שְׂעֹרִ֑ים לֹֽא-יִצֹ֨ק עָלָ֜יו שֶׁ֗מֶן וְלֹֽא-יִתֵּ֤ן עָלָיו֙ לְבֹנָ֔ה כִּֽי-מִנְחַ֥ת קְנָאֹת֙ ה֔וּא מִנְחַ֥ת זִכָּר֖וֹן מַזְכֶּ֥רֶת עָוֹֽן:  מ  וְהֵבִ֨יא הָאִ֣ישׁ אֶת-אִשְׁתּוֹ֘ אֶל-הַכֹּהֵן֒ וְהֵבִ֤יא אֶת-קָרְבָּנָהּ֙ עָלֶ֔יהָ עֲשִׂירִ֥ת הָֽאֵיפָ֖ה קֶ֣מַח שְׂעֹרִ֑ים לֹֽא-יִצֹ֨ק עָלָ֜יו שֶׁ֗מֶן וְלֹֽא-יִתֵּ֤ן עָלָיו֙ לְבֹנָ֔ה כִּֽי-מִנְחַ֥ת קְנָאֹת֙ ה֔וּא מִנְחַ֥ת זִכָּר֖וֹן מַזְכֶּ֥רֶת עָוֹֽן:  ת  וְיַיְתֵי גַבְרָא יָת אִתְּתֵהּ לְוַת כַּהֲנָא וְיַיְתֵי יָת קֻרְבָּנַהּ עֲלַהּ חַד מִן עַסְרָא בִּתְלַת סְאִין קִמְחָא שְׂעָרִין לָא יְרִיק עֲלוֹהִי מִשְׁחָא וְלָא יִתֵּן עֲלוֹהִי לְבֻנְתָּא אֲרֵי מִנְחַת קִנְאָתָא הוּא מִנְחַת דוּכְרָנָא מַדְכָּרַת חוֹבָא:

{טז} וְהִקְרִ֥יב אֹתָ֖הּ הַכֹּהֵ֑ן וְהֶֽעֱמִדָ֖הּ לִפְנֵ֥י יְהֹוָֽה:  מ  וְהִקְרִ֥יב אֹתָ֖הּ הַכֹּהֵ֑ן וְהֶֽעֱמִדָ֖הּ לִפְנֵ֥י יְהֹוָֽה:  ת  וִיקָרֵב יָתַהּ כַּהֲנָא וִיקִימִנַהּ קֳדָם יְיָ:

{יז} וְלָקַ֧ח הַכֹּהֵ֛ן מַ֥יִם קְדֹשִׁ֖ים בִּכְלִי-חָ֑רֶשׂ וּמִן-הֶֽעָפָ֗ר אֲשֶׁ֤ר יִֽהְיֶה֙ בְּקַרְקַ֣ע הַמִּשְׁכָּ֔ן יִקַּ֥ח הַכֹּהֵ֖ן וְנָתַ֥ן אֶל-הַמָּֽיִם:  מ  וְלָקַ֧ח הַכֹּהֵ֛ן מַ֥יִם קְדֹשִׁ֖ים בִּכְלִי-חָ֑רֶשׂ וּמִן-הֶֽעָפָ֗ר אֲשֶׁ֤ר יִֽהְיֶה֙ בְּקַרְקַ֣ע הַמִּשְׁכָּ֔ן יִקַּ֥ח הַכֹּהֵ֖ן וְנָתַ֥ן אֶל-הַמָּֽיִם:  ת  וְיִסַב כַּהֲנָא מֵי כִיוֹר בְּמַן דַחֲסָף וּמִן עַפְרָא דִי יְהֵי בִּיסוֹדֵי מַשְׁכְּנָא יִסַב כַּהֲנָא וְיִתֵּן לְמַיָא:

{יח} וְהֶֽעֱמִ֨יד הַכֹּהֵ֥ן אֶֽת-הָאִשָּׁה֘ לִפְנֵ֣י יְהֹוָה֒ וּפָרַע֙ אֶת-רֹ֣אשׁ הָֽאִשָּׁ֔ה וְנָתַ֣ן עַל-כַּפֶּ֗יהָ אֵ֚ת מִנְחַ֣ת הַזִכָּר֔וֹן מִנְחַ֥ת קְנָאֹ֖ת הִ֑וא וּבְיַ֤ד הַכֹּהֵן֙ יִֽהְי֔וּ מֵ֥י הַמָּרִ֖ים הַמְאָֽרְרִֽים:  מ  וְהֶֽעֱמִ֨יד הַכֹּהֵ֥ן אֶֽת-הָאִשָּׁה֘ לִפְנֵ֣י יְהֹוָה֒ וּפָרַע֙ אֶת-רֹ֣אשׁ הָֽאִשָּׁ֔ה וְנָתַ֣ן עַל-כַּפֶּ֗יהָ אֵ֚ת מִנְחַ֣ת הַזִכָּר֔וֹן מִנְחַ֥ת קְנָאֹ֖ת הִ֑וא וּבְיַ֤ד הַכֹּהֵן֙ יִֽהְי֔וּ מֵ֥י הַמָּרִ֖ים הַמְאָֽרְרִֽים:  ת  וִיקֵים כַּהֲנָא יָת אִתְּתָא קֳדָם יְיָ וְיִפְרַע יָת רֵישָׁא דְאִתְּתָא וְיתֵּן עַל יְדָהָא יָת מִנְחַת דוּכְרָנָא מִנְחַת קִנְאָתָא הִיא וּבִידָא דְכַהֲנָא יְהוֹן מַיָא מְרִירַיָא מְלַטְטַיָא:

{יט} וְהִשְׁבִּ֨יעַ אֹתָ֜הּ הַכֹּהֵ֗ן וְאָמַ֤ר אֶל-הָֽאִשָּׁה֙ אִם-לֹ֨א שָׁכַ֥ב אִישׁ֙ אֹתָ֔ךְ וְאִם-לֹ֥א שָׂטִ֛ית טֻמְאָ֖ה תַּ֣חַת אִישֵׁ֑ךְ הִנָּקִ֕י מִמֵ֛י הַמָּרִ֥ים הַמְאָֽרְרִ֖ים הָאֵֽלֶּה:  מ  וְהִשְׁבִּ֨יעַ אֹתָ֜הּ הַכֹּהֵ֗ן וְאָמַ֤ר אֶל-הָֽאִשָּׁה֙ אִם-לֹ֨א שָׁכַ֥ב אִישׁ֙ אֹתָ֔ךְ וְאִם-לֹ֥א שָׂטִ֛ית טֻמְאָ֖ה תַּ֣חַת אִישֵׁ֑ךְ הִנָּקִ֕י מִמֵ֛י הַמָּרִ֥ים הַמְאָֽרְרִ֖ים הָאֵֽלֶּה:  ת  וְיוֹמֵי יָתַהּ כַּהֲנָא וְיֵימַר לְאִתְּתָא אִם לָא שְׁכִיב גְבַר יָתִיךְ וְאִם לָא סְטִית לְאִסְתְּאָבָא בַּר מִבַּעֲלִיךְ הֱוֵי זַכָּאָה מִמַיָא מְרִירַיָא מְלַטְטַיָא הָאִלֵין:

{כ} וְאַ֗תְּ כִּ֥י שָׂטִ֛ית תַּ֥חַת אִישֵׁ֖ךְ וְכִ֣י נִטְמֵ֑את וַיִּתֵּ֨ן אִ֥ישׁ בָּךְ֙ אֶת-שְׁכָבְתּ֔וֹ מִֽבַּלְעֲדֵ֖י אִישֵֽׁךְ:  מ  וְאַ֗תְּ כִּ֥י שָׂטִ֛ית תַּ֥חַת אִישֵׁ֖ךְ וְכִ֣י נִטְמֵ֑את וַיִּתֵּ֨ן אִ֥ישׁ בָּךְ֙ אֶת-שְׁכָבְתּ֔וֹ מִֽבַּלְעֲדֵ֖י אִישֵֽׁךְ:  ת  וְאַתְּ אֲרֵי סְטִית בַּר מִבַּעֲלִיךְ וַאֲרֵי אִסְתָּאָבְתְּ וִיהַב גְבַר בִּיךְ יָת שְׁכֻבְתֵּהּ בַּר מִבַּעֲלִיךְ:

{כא} וְהִשְׁבִּ֨יעַ הַכֹּהֵ֥ן אֶֽת-הָֽאִשָּׁה֘ בִּשְׁבֻעַ֣ת הָֽאָלָה֒ וְאָמַ֤ר הַכֹּהֵן֙ לָֽאִשָּׁ֔ה יִתֵּ֨ן יְהֹוָ֥ה אוֹתָ֛ךְ לְאָלָ֥ה וְלִשְׁבֻעָ֖ה בְּת֣וֹךְ עַמֵּ֑ךְ בְּתֵ֨ת יְהֹוָ֤ה אֶת-יְרֵכֵךְ֙ נֹפֶ֔לֶת וְאֶת-בִּטְנֵ֖ךְ צָבָֽה:  מ  וְהִשְׁבִּ֨יעַ הַכֹּהֵ֥ן אֶֽת-הָֽאִשָּׁה֘ בִּשְׁבֻעַ֣ת הָֽאָלָה֒ וְאָמַ֤ר הַכֹּהֵן֙ לָֽאִשָּׁ֔ה יִתֵּ֨ן יְהֹוָ֥ה אוֹתָ֛ךְ לְאָלָ֥ה וְלִשְׁבֻעָ֖ה בְּת֣וֹךְ עַמֵּ֑ךְ בְּתֵ֨ת יְהֹוָ֤ה אֶת-יְרֵכֵךְ֙ נֹפֶ֔לֶת וְאֶת-בִּטְנֵ֖ךְ צָבָֽה:  ת  וְיוֹמֵי כַהֲנָא יָת אִתְּתָא בְּמוֹמָתָא דִלְוָטָּא וְיֵימַר כַּהֲנָא לְאִתְּתָא יִתֵּן יְיָ יָתִיךְ לִלְוַט וּלְמוֹמֵי בְּגוֹ עַמִיךְ בִּדְיִתֵּן יְיָ יָת יַרְכִּיךְ מַסְיָא וְיָת מְעַיְכִי נְפִיחִין:

{כב} וּ֠בָ֠אוּ הַמַּ֨יִם הַמְאָֽרְרִ֤ים הָאֵ֨לֶּה֙ בְּֽמֵעַ֔יִךְ לַצְבּ֥וֹת בֶּ֖טֶן וְלַנְפִּ֣ל יָרֵ֑ךְ וְאָֽמְרָ֥ה הָֽאִשָּׁ֖ה אָמֵ֥ן | אָמֵֽן:  מ  וּ֠בָ֠אוּ הַמַּ֨יִם הַמְאָֽרְרִ֤ים הָאֵ֨לֶּה֙ בְּֽמֵעַ֔יִךְ לַצְבּ֥וֹת בֶּ֖טֶן וְלַנְפִּ֣ל יָרֵ֑ךְ וְאָֽמְרָ֥ה הָֽאִשָּׁ֖ה אָמֵ֥ן | אָמֵֽן:  ת  וְיֵעֲלוּן מַיָא מְלַטְטַיָא הָאִלֵין בִּמְעַיְכִי לַאֲפָחָא מְעִין וּלְאַמְסָאָה יַרְכָּא וְתֵימַר אִתְּתָא אָמֵן אָמֵן:

{כג} וְ֠כָתַ֠ב אֶת-הָֽאָלֹ֥ת הָאֵ֛לֶּה הַכֹּהֵ֖ן בַּסֵּ֑פֶר וּמָחָ֖ה אֶל-מֵ֥י הַמָּרִֽים:  מ  וְ֠כָתַ֠ב אֶת-הָֽאָלֹ֥ת הָאֵ֛לֶּה הַכֹּהֵ֖ן בַּסֵּ֑פֶר וּמָחָ֖ה אֶל-מֵ֥י הַמָּרִֽים:  ת  וְיִכְתּוֹב יָת לְוָטַיָא הָאִלֵין כַּהֲנָא בְּסִפְרָא וְיִמְחוֹק לְמַיָא מְרִירַיָא:

{כד} וְהִשְׁקָה֙ אֶת-הָ֣אִשָּׁ֔ה אֶת-מֵ֥י הַמָּרִ֖ים הַמְאָֽרְרִ֑ים וּבָ֥אוּ בָ֛הּ הַמַּ֥יִם הַֽמְאָרְרִ֖ים לְמָרִֽים:  מ  וְהִשְׁקָה֙ אֶת-הָ֣אִשָּׁ֔ה אֶת-מֵ֥י הַמָּרִ֖ים הַמְאָֽרְרִ֑ים וּבָ֥אוּ בָ֛הּ הַמַּ֥יִם הַֽמְאָרְרִ֖ים לְמָרִֽים:  ת  וְיַשְׁקֵי יָת אִתְּתָא יָת מַיָא מְרִירַיָא מְלַטְטַיָא וְיֵעֲלוּן בַּהּ מַיָא מְלַטְטַיָא לִמְרִירוּ:

{כה} וְלָקַ֤ח הַכֹּהֵן֙ מִיַּ֣ד הָֽאִשָּׁ֔ה אֵ֖ת מִנְחַ֣ת הַקְּנָאֹ֑ת וְהֵנִ֤יף אֶת-הַמִּנְחָה֙ לִפְנֵ֣י יְהֹוָ֔ה וְהִקְרִ֥יב אֹתָ֖הּ אֶל-הַמִּזְבֵּֽחַ:  מ  וְלָקַ֤ח הַכֹּהֵן֙ מִיַּ֣ד הָֽאִשָּׁ֔ה אֵ֖ת מִנְחַ֣ת הַקְּנָאֹ֑ת וְהֵנִ֤יף אֶת-הַמִּנְחָה֙ לִפְנֵ֣י יְהֹוָ֔ה וְהִקְרִ֥יב אֹתָ֖הּ אֶל-הַמִּזְבֵּֽחַ:  ת  וְיִסַב כַּהֲנָא מִידָא דְאִתְּתָא יָת מִנְחַת קִנְאָתָא וִירֵים יָת מִנְחָתָא קָדָם יְיָ וִיקָרֵב יָתַהּ לְמַדְבְּחָא:

{כו} וְקָמַ֨ץ הַכֹּהֵ֤ן מִן-הַמִּנְחָה֙ אֶת-אַזְכָּ֣רָתָ֔הּ וְהִקְטִ֖יר הַמִּזְבֵּ֑חָה וְאַחַ֛ר יַשְׁקֶ֥ה אֶת-הָֽאִשָּׁ֖ה אֶת-הַמָּֽיִם:  מ  וְקָמַ֨ץ הַכֹּהֵ֤ן מִן-הַמִּנְחָה֙ אֶת-אַזְכָּ֣רָתָ֔הּ וְהִקְטִ֖יר הַמִּזְבֵּ֑חָה וְאַחַ֛ר יַשְׁקֶ֥ה אֶת-הָֽאִשָּׁ֖ה אֶת-הַמָּֽיִם:  ת  וְיִקְמוֹץ כַּהֲנָא מִן מִנְחָתָא יָת אַדְכַּרְתָּהּ וְיַסֵק לְמַדְבְּחָא וּבָתַר כֵּן יַשְׁקֵי יָת אִתְּתָא יָת מַיָא:

{כז} וְהִשְׁקָ֣הּ אֶת-הַמַּ֗יִם וְהָֽיְתָ֣ה אִֽם-נִטְמְאָה֘ וַתִּמְעֹ֣ל מַ֣עַל בְּאִישָׁהּ֒ וּבָ֨אוּ בָ֜הּ הַמַּ֤יִם הַמְאָֽרְרִים֙ לְמָרִ֔ים וְצָֽבְתָ֣ה בִטְנָ֔הּ וְנָֽפְלָ֖ה יְרֵכָ֑הּ וְהָֽיְתָ֧ה הָֽאִשָּׁ֛ה לְאָלָ֖ה בְּקֶ֥רֶב עַמָּֽהּ:  מ  וְהִשְׁקָ֣הּ אֶת-הַמַּ֗יִם וְהָֽיְתָ֣ה אִֽם-נִטְמְאָה֘ וַתִּמְעֹ֣ל מַ֣עַל בְּאִישָׁהּ֒ וּבָ֨אוּ בָ֜הּ הַמַּ֤יִם הַמְאָֽרְרִים֙ לְמָרִ֔ים וְצָֽבְתָ֣ה בִטְנָ֔הּ וְנָֽפְלָ֖ה יְרֵכָ֑הּ וְהָֽיְתָ֧ה הָֽאִשָּׁ֛ה לְאָלָ֖ה בְּקֶ֥רֶב עַמָּֽהּ:  ת  וְיַשְׁקִנָהּ יָת מַיָא וּתְהֵי אִם אִסְתָּאָבַת וְשַׁקָרַת שְׁקָר בְּבַעֲלַהּ וְיֵעֲלוּן בַּהּ מַיָא מְלַטְטַיָא לִמְרִירוּ וְיִפְּחוּן מְעָהָא וְתִתְמְסֵי יַרְכַהּ וּתְהֵי אִתְּתָא לִלְוַט בְּגוֹ עַמָהּ:

{כח} וְאִם-לֹ֤א נִטְמְאָה֙ הָֽאִשָּׁ֔ה וּטְהֹרָ֖ה הִ֑וא וְנִקְּתָ֖ה וְנִזְרְעָ֥ה זָֽרַע:  מ  וְאִם-לֹ֤א נִטְמְאָה֙ הָֽאִשָּׁ֔ה וּטְהֹרָ֖ה הִ֑וא וְנִקְּתָ֖ה וְנִזְרְעָ֥ה זָֽרַע:  ת  וְאִם לָא אִסְתָּאָבַת אִתְּתָא וְדַכְוָאָה הִיא וְתִפּוֹק זַכָּאָה וְתַעְדִי עִדוּי:

{כט} זֹ֥את תּוֹרַ֖ת הַקְּנָאֹ֑ת אֲשֶׁ֨ר תִּשְׂטֶ֥ה אִשָּׁ֛ה תַּ֥חַת אִישָׁ֖הּ וְנִטְמָֽאָה:  מ  זֹ֥את תּוֹרַ֖ת הַקְּנָאֹ֑ת אֲשֶׁ֨ר תִּשְׂטֶ֥ה אִשָּׁ֛ה תַּ֥חַת אִישָׁ֖הּ וְנִטְמָֽאָה:  ת  דָא אוֹרַיְתָא דְקִנְאָתָא דִי תִסְטֵי אִתְּתָא בַּר מִבַּעֲלַהּ וְתִסְתָּאָב:

{ל} אוֹ אִ֗ישׁ אֲשֶׁ֨ר תַּֽעֲבֹ֥ר עָלָ֛יו ר֥וּחַ קִנְאָ֖ה וְקִנֵּ֣א אֶת-אִשְׁתּ֑וֹ וְהֶֽעֱמִ֤יד אֶת-הָֽאִשָּׁה֙ לִפְנֵ֣י יְהֹוָ֔ה וְעָ֤שָׂה לָהּ֙ הַכֹּהֵ֔ן אֵ֥ת כָּל-הַתּוֹרָ֖ה הַזֹּֽאת:  מ  אוֹ אִ֗ישׁ אֲשֶׁ֨ר תַּֽעֲבֹ֥ר עָלָ֛יו ר֥וּחַ קִנְאָ֖ה וְקִנֵּ֣א אֶת-אִשְׁתּ֑וֹ וְהֶֽעֱמִ֤יד אֶת-הָֽאִשָּׁה֙ לִפְנֵ֣י יְהֹוָ֔ה וְעָ֤שָׂה לָהּ֙ הַכֹּהֵ֔ן אֵ֥ת כָּל-הַתּוֹרָ֖ה הַזֹּֽאת:  ת  אוֹ גְבַר דִי תֶעְבַּר עֲלוֹהִי רוּחַ קִנְאָה וִיקַנֵי יָת אִתְּתֵהּ וִיקֵם יָת אִתְּתָא קֳדָם יְיָ וְיַעֲבֵד לַהּ כַּהֲנָא יָת כָּל אוֹרַיְתָא הָדָא:

{לא} וְנִקָּ֥ה הָאִ֖ישׁ מֵֽעָוֹ֑ן וְהָֽאִשָּׁ֣ה הַהִ֔וא תִּשָּׂ֖א אֶת-עֲוֹנָֽהּ:  מ  וְנִקָּ֥ה הָאִ֖ישׁ מֵֽעָוֹ֑ן וְהָֽאִשָּׁ֣ה הַהִ֔וא תִּשָּׂ֖א אֶת-עֲוֹנָֽהּ:  ת  וִיהֵי זַכָּאָה גַבְרָא מֵחוֹבָא וְאִתְּתָא הַהִיא תְּקַבֵּל יָת חוֹבָהּ: (פ)




במדבר פרק-ו

{א} וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:

{ב} דַּבֵּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם אִ֣ישׁ אֽוֹ-אִשָּׁ֗ה כִּ֤י יַפְלִא֙ לִנְדֹּר֙ נֶ֣דֶר נָזִ֔יר לְהַזִּ֖יר לַֽיהֹוָֽה:  מ  דַּבֵּר֙ אֶל-בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם אִ֣ישׁ אֽוֹ-אִשָּׁ֗ה כִּ֤י יַפְלִא֙ לִנְדֹּר֙ נֶ֣דֶר נָזִ֔יר לְהַזִּ֖יר לַֽיהֹוָֽה:  ת  מַלֵל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן גְבַר אוֹ אִתְּתָא אֲרֵי יְפָרֵשׁ לְמִדַר נְדַר נְזִירוּ לְמֵיזַר קֳדָם יְיָ:

{ג} מִיַּ֤יִן וְשֵׁכָר֙ יַזִּ֔יר חֹ֥מֶץ יַ֛יִן וְחֹ֥מֶץ שֵׁכָ֖ר לֹ֣א יִשְׁתֶּ֑ה וְכָל-מִשְׁרַ֤ת עֲנָבִים֙ לֹ֣א יִשְׁתֶּ֔ה וַֽעֲנָבִ֛ים לַחִ֥ים וִֽיבֵשִׁ֖ים לֹ֥א יֹאכֵֽל:  מ  מִיַּ֤יִן וְשֵׁכָר֙ יַזִּ֔יר חֹ֥מֶץ יַ֛יִן וְחֹ֥מֶץ שֵׁכָ֖ר לֹ֣א יִשְׁתֶּ֑ה וְכָל-מִשְׁרַ֤ת עֲנָבִים֙ לֹ֣א יִשְׁתֶּ֔ה וַֽעֲנָבִ֛ים לַחִ֥ים וִֽיבֵשִׁ֖ים לֹ֥א יֹאכֵֽל:  ת  מֵחֲמַר חֲדַת וְעַתִּיק יַנְזִיר חַל דַחֲמַר חֲדַת וְחַל דַחֲמַר עַתִּיק לָא יִשְׁתֵּי וְכָל מַתְרוּת עִנְבִין לָא יִשְׁתֵּי וְעִנְבִין רַטִיבִין וִיבֵשִׁין לָא יֵיכוּל:

{ד} כֹּ֖ל יְמֵ֣י נִזְר֑וֹ מִכֹּל֩ אֲשֶׁ֨ר יֵֽעָשֶׂ֜ה מִגֶּ֣פֶן הַיַּ֗יִן מֵֽחַרְצַנִּ֛ים וְעַד-זָ֖ג לֹ֥א יֹאכֵֽל:  מ  כֹּ֖ל יְמֵ֣י נִזְר֑וֹ מִכֹּל֩ אֲשֶׁ֨ר יֵֽעָשֶׂ֜ה מִגֶּ֣פֶן הַיַּ֗יִן מֵֽחַרְצַנִּ֛ים וְעַד-זָ֖ג לֹ֥א יֹאכֵֽל:  ת  כֹּל יוֹמֵי נִזְרֵהּ מִכֹּל דְיִתְעֲבֵד מִגֻפְנָא דְחַמְרָא מִפּוּרְצְנִין וְעַד עִצוּרִין לָא יֵיכוּל:

{ה} כָּל-יְמֵי֙ נֶ֣דֶר נִזְר֔וֹ תַּ֖עַר לֹֽא-יַעֲבֹ֣ר עַל-רֹאשׁ֑וֹ עַד-מְלֹ֨את הַיָּמִ֜ם אֲשֶׁר-יַזִּ֤יר לַֽיהֹוָה֙ קָדֹ֣שׁ יִֽהְיֶ֔ה גַּדֵּ֥ל פֶּ֖רַע שְׂעַ֥ר רֹאשֽׁוֹ:  מ  כָּל-יְמֵי֙ נֶ֣דֶר נִזְר֔וֹ תַּ֖עַר לֹֽא-יַעֲבֹ֣ר עַל-רֹאשׁ֑וֹ עַד-מְלֹ֨את הַיָּמִ֜ם אֲשֶׁר-יַזִּ֤יר לַֽיהֹוָה֙ קָדֹ֣שׁ יִֽהְיֶ֔ה גַּדֵּ֥ל פֶּ֖רַע שְׂעַ֥ר רֹאשֽׁוֹ:  ת  כָּל יוֹמֵי נְדַר נִזְרֵהּ מַסְפַּר לָא יֶעְבַּר עַל רֵישֵׁהּ עַד מִשְׁלַם יוֹמַיָא דְיַנְזֵר קֳדָם יְיָ קַדִישׁ יְהֵי מַרְבֵּי פֵרוּעַ שְׂעַר רֵישֵׁהּ:

{ו} כָּל-יְמֵ֥י הַזִּיר֖וֹ לַֽיהֹוָ֑ה עַל-נֶ֥פֶשׁ מֵ֖ת לֹ֥א יָבֹֽא:  מ  כָּל-יְמֵ֥י הַזִּיר֖וֹ לַֽיהֹוָ֑ה עַל-נֶ֥פֶשׁ מֵ֖ת לֹ֥א יָבֹֽא:  ת  כָּל יוֹמִין דְיַנְזֵר קֳדָם יְיָ עַל נַפְשָׁא דְמֵתָא לָא יֵיעוּל:

{ז} לְאָבִ֣יו וּלְאִמּ֗וֹ לְאָחִיו֙ וּלְאַ֣חֹת֔וֹ לֹֽא-יִטַּמָּ֥א לָהֶ֖ם בְּמֹתָ֑ם כִּ֛י נֵ֥זֶר אֱלֹהָ֖יו עַל-רֹאשֽׁוֹ:  מ  לְאָבִ֣יו וּלְאִמּ֗וֹ לְאָחִיו֙ וּלְאַ֣חֹת֔וֹ לֹֽא-יִטַּמָּ֥א לָהֶ֖ם בְּמֹתָ֑ם כִּ֛י נֵ֥זֶר אֱלֹהָ֖יו עַל-רֹאשֽׁוֹ:  ת  לְאָבוּהִי וּלְאִמֵהּ לְאַחוֹהִי וּלְאַחָתֵהּ לָא יִסְתָּאַב לְהוֹן בְּמוֹתְהוֹן אֲרֵי כְּלִילָא דֶאֱלָהֵהּ עַל רֵישֵׁהּ:

{ח} כֹּ֖ל יְמֵ֣י נִזְר֑וֹ קָדֹ֥שׁ ה֖וּא לַֽיהֹוָֽה:  מ  כֹּ֖ל יְמֵ֣י נִזְר֑וֹ קָדֹ֥שׁ ה֖וּא לַֽיהֹוָֽה:  ת  כֹּל יוֹמֵי נִזְרֵהּ קַדִישׁ הוּא קֳדָם יְיָ:

{ט} וְכִֽי-יָמ֨וּת מֵ֤ת עָלָיו֙ בְּפֶ֣תַע פִּתְאֹ֔ם וְטִמֵּ֖א רֹ֣אשׁ נִזְר֑וֹ וְגִלַּ֤ח רֹאשׁוֹ֙ בְּי֣וֹם טָֽהֳרָת֔וֹ בַּיּ֥וֹם הַשְּׁבִיעִ֖י יְגַלְּחֶֽנּוּ:  מ  וְכִֽי-יָמ֨וּת מֵ֤ת עָלָיו֙ בְּפֶ֣תַע פִּתְאֹ֔ם וְטִמֵּ֖א רֹ֣אשׁ נִזְר֑וֹ וְגִלַּ֤ח רֹאשׁוֹ֙ בְּי֣וֹם טָֽהֳרָת֔וֹ בַּיּ֥וֹם הַשְּׁבִיעִ֖י יְגַלְּחֶֽנּוּ:  ת  וַאֲרֵי יְמוּת מֵתָא עֲלוֹהִי בִּתְכֵּף שָׁלוּ וִיסָאֵב רֵישׁ נִזְרֵהּ וִיגַלַח רֵישֵׁהּ בְּיוֹמָא דִדְכוּתֵהּ בְּיוֹמָא שְׁבִיעָאָה יְגַלְחִנֵהּ:

{י} וּבַיּ֣וֹם הַשְּׁמִינִ֗י יָבִא֙ שְׁתֵּ֣י תֹרִ֔ים א֥וֹ שְׁנֵ֖י בְּנֵ֣י יוֹנָ֑ה אֶ֨ל-הַכֹּהֵ֔ן אֶל-פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  מ  וּבַיּ֣וֹם הַשְּׁמִינִ֗י יָבִא֙ שְׁתֵּ֣י תֹרִ֔ים א֥וֹ שְׁנֵ֖י בְּנֵ֣י יוֹנָ֑ה אֶ֨ל-הַכֹּהֵ֔ן אֶל-פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  ת  וּבְיוֹמָא תְמִינָאָה יַיְתֵי תַּרְתֵּין שַׁפְנִינִין אוֹ תְרֵין בְּנֵי יוֹנָה לְוָת כַּהֲנָא לִתְרַע מַשְׁכַּן זִמְנָא:

{יא} וְעָשָׂ֣ה הַכֹּהֵ֗ן אֶחָ֤ד לְחַטָּאת֙ וְאֶחָ֣ד לְעֹלָ֔ה וְכִפֶּ֣ר עָלָ֔יו מֵֽאֲשֶׁ֥ר חָטָ֖א עַל-הַנָּ֑פֶשׁ וְקִדַּ֥שׁ אֶת-רֹאשׁ֖וֹ בַּיּ֥וֹם הַהֽוּא:  מ  וְעָשָׂ֣ה הַכֹּהֵ֗ן אֶחָ֤ד לְחַטָּאת֙ וְאֶחָ֣ד לְעֹלָ֔ה וְכִפֶּ֣ר עָלָ֔יו מֵֽאֲשֶׁ֥ר חָטָ֖א עַל-הַנָּ֑פֶשׁ וְקִדַּ֥שׁ אֶת-רֹאשׁ֖וֹ בַּיּ֥וֹם הַהֽוּא:  ת  וְיַעְבֵּד כַּהֲנָא חַד לְחַטָאתָא וְחַד לַעֲלָתָא וִיכַפֵּר עֲלוֹהִי מִדְחָב עַל מִתָא וִיקַדַשׁ יָת רֵישֵׁהּ בְּיוֹמָא הַהוּא:

{יב} וְהִזִּ֤יר לַֽיהֹוָה֙ אֶת-יְמֵ֣י נִזְר֔וֹ וְהֵבִ֛יא כֶּ֥בֶשׂ בֶּן-שְׁנָת֖וֹ לְאָשָׁ֑ם וְהַיָּמִ֤ים הָרִֽאשֹׁנִים֙ יִפְּל֔וּ כִּ֥י טָמֵ֖א נִזְרֽוֹ:  מ  וְהִזִּ֤יר לַֽיהֹוָה֙ אֶת-יְמֵ֣י נִזְר֔וֹ וְהֵבִ֛יא כֶּ֥בֶשׂ בֶּן-שְׁנָת֖וֹ לְאָשָׁ֑ם וְהַיָּמִ֤ים הָרִֽאשֹׁנִים֙ יִפְּל֔וּ כִּ֥י טָמֵ֖א נִזְרֽוֹ:  ת  וְיַזֵר קֳדָם יְיָ יָת יוֹמֵי נִזְרֵהּ וְיַיְתִי אִמַר בַּר שַׁתֵּהּ לַאֲשָׁמָא וְיוֹמַיָא קַדְמָאֵי יִבְטְלוּן אֲרֵי אִסְתָּאַב נִזְרֵהּ:

{יג} וְזֹ֥את תּוֹרַ֖ת הַנָּזִ֑יר בְּי֗וֹם מְלֹאת֙ יְמֵ֣י נִזְר֔וֹ יָבִ֣יא אֹת֔וֹ אֶל-פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  מ  וְזֹ֥את תּוֹרַ֖ת הַנָּזִ֑יר בְּי֗וֹם מְלֹאת֙ יְמֵ֣י נִזְר֔וֹ יָבִ֣יא אֹת֔וֹ אֶל-פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד:  ת  וְדָא אוֹרַיְתָא דִנְזִירָא בְּיוֹם מִשְׁלַם יוֹמֵי נִזְרֵהּ יַיְתִי יָתֵהּ לִתְרַע מַשְׁכַּן זִמְנָא:

{יד} וְהִקְרִ֣יב אֶת-קָרְבָּנ֣וֹ לַֽיהֹוָ֡ה כֶּבֶשׂ֩ בֶּן-שְׁנָת֨וֹ תָמִ֤ים אֶחָד֙ לְעֹלָ֔ה וְכַבְשָׂ֨ה אַחַ֧ת בַּת-שְׁנָתָ֛הּ תְּמִימָ֖ה לְחַטָּ֑את וְאַֽיִל-אֶחָ֥ד תָּמִ֖ים לִשְׁלָמִֽים:  מ  וְהִקְרִ֣יב אֶת-קָרְבָּנ֣וֹ לַֽיהֹוָ֡ה כֶּבֶשׂ֩ בֶּן-שְׁנָת֨וֹ תָמִ֤ים אֶחָד֙ לְעֹלָ֔ה וְכַבְשָׂ֨ה אַחַ֧ת בַּת-שְׁנָתָ֛הּ תְּמִימָ֖ה לְחַטָּ֑את וְאַֽיִל-אֶחָ֥ד תָּמִ֖ים לִשְׁלָמִֽים:  ת  וִיקָרֵב יָת קֻרְבָּנֵהּ קֳדָם יְיָ אִמַר בַּר שַׁתֵּהּ שְׁלִים חַד לַעֲלָתָא וְאִמַרְתָּא חֲדָא בַּת שַׁתָּהּ שְׁלֶמְתָּא לְחַטָאתָא וּדְכַר חַד שְׁלִים לְנִכְסַת קוּדְשַׁיָא:

{טו} וְסַ֣ל מַצּ֗וֹת סֹ֤לֶת חַלֹּת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן וּמִנְחָתָ֖ם וְנִסְכֵּיהֶֽם:  מ  וְסַ֣ל מַצּ֗וֹת סֹ֤לֶת חַלֹּת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן וּמִנְחָתָ֖ם וְנִסְכֵּיהֶֽם:  ת  וְסַלָא דְפַטִּרַיָא סֻלְּתָּא גְרִיצַן דְפִילָן בִּמְשַׁח וְאִסְפּוֹגִין פַּטִירִין דִמְשִׁיחִין בִּמְשָׁח וּמִנְחַתְהוֹן וְנִסְכֵּיהוֹן:

{טז} וְהִקְרִ֥יב הַכֹּהֵ֖ן לִפְנֵ֣י יְהֹוָ֑ה וְעָשָׂ֥ה אֶת-חַטָּאת֖וֹ וְאֶת-עֹֽלָתֽוֹ:  מ  וְהִקְרִ֥יב הַכֹּהֵ֖ן לִפְנֵ֣י יְהֹוָ֑ה וְעָשָׂ֥ה אֶת-חַטָּאת֖וֹ וְאֶת-עֹֽלָתֽוֹ:  ת  וִיקָרֵב כַּהֲנָא קֳדָם יְיָ וְיַעְבֵּד יָת חַטָאתֵהּ וְיָת עֲלָתֵהּ:

{יז} וְאֶת-הָאַ֜יִל יַֽעֲשֶׂ֨ה זֶ֤בַח שְׁלָמִים֙ לַֽיהֹוָ֔ה עַ֖ל סַ֣ל הַמַּצּ֑וֹת וְעָשָׂה֙ הַכֹּהֵ֔ן אֶת-מִנְחָת֖וֹ וְאֶת-נִסְכּֽוֹ:  מ  וְאֶת-הָאַ֜יִל יַֽעֲשֶׂ֨ה זֶ֤בַח שְׁלָמִים֙ לַֽיהֹוָ֔ה עַ֖ל סַ֣ל הַמַּצּ֑וֹת וְעָשָׂה֙ הַכֹּהֵ֔ן אֶת-מִנְחָת֖וֹ וְאֶת-נִסְכּֽוֹ:  ת  וְיָת דִכְרָא יַעְבֵּד נִכְסַת קוּדְשַׁיָא קֳדָם יְיָ עַל סַלָא דְפַטִּירַיָא וְיַעְבֵּד כַּהֲנָא יָת מִנְחָתֵהּ וְיָת נִסְכֵּהּ:

{יח} וְגִלַּ֣ח הַנָּזִ֗יר פֶּ֛תַח אֹ֥הֶל מוֹעֵ֖ד אֶת-רֹ֣אשׁ נִזְר֑וֹ וְלָקַ֗ח אֶת-שְׂעַר֙ רֹ֣אשׁ נִזְר֔וֹ וְנָתַן֙ עַל-הָאֵ֔שׁ אֲשֶׁר-תַּ֖חַת זֶ֥בַח הַשְּׁלָמִֽים:  מ  וְגִלַּ֣ח הַנָּזִ֗יר פֶּ֛תַח אֹ֥הֶל מוֹעֵ֖ד אֶת-רֹ֣אשׁ נִזְר֑וֹ וְלָקַ֗ח אֶת-שְׂעַר֙ רֹ֣אשׁ נִזְר֔וֹ וְנָתַן֙ עַל-הָאֵ֔שׁ אֲשֶׁר-תַּ֖חַת זֶ֥בַח הַשְּׁלָמִֽים:  ת  וִיגַלַח נְזִירָא בִּתְרַע מַשְׁכַּן זִמְנָא יָת רֵישׁ נִזְרֵהּ וְיִסַב יָת שְׂעַר רֵישׁ נִזְרֵהּ וְיִתֵּן עַל אֶשָׁתָא דִי תְחוֹת דוּדָא דְנִכְסַת קוּדְשַׁיָא:

{כא} זֹ֣את תּוֹרַ֣ת הַנָּזִיר֘ אֲשֶׁ֣ר יִדֹּר֒ קָרְבָּנ֤וֹ לַֽיהֹוָה֙ עַל-נִזְר֔וֹ מִלְּבַ֖ד אֲשֶׁר-תַּשִּׂ֣יג יָד֑וֹ כְּפִ֤י נִדְרוֹ֙ אֲשֶׁ֣ר יִדֹּ֔ר כֵּ֣ן יַֽעֲשֶׂ֔ה עַ֖ל תּוֹרַ֥ת נִזְרֽוֹ:  מ  זֹ֣את תּוֹרַ֣ת הַנָּזִיר֘ אֲשֶׁ֣ר יִדֹּר֒ קָרְבָּנ֤וֹ לַֽיהֹוָה֙ עַל-נִזְר֔וֹ מִלְּבַ֖ד אֲשֶׁר-תַּשִּׂ֣יג יָד֑וֹ כְּפִ֤י נִדְרוֹ֙ אֲשֶׁ֣ר יִדֹּ֔ר כֵּ֣ן יַֽעֲשֶׂ֔ה עַ֖ל תּוֹרַ֥ת נִזְרֽוֹ:  ת  דָא אוֹרַיְתָא דִנְזִירָא דִי יִדַר קֻרְבָּנֵהּ קֳדָם יְיָ עַל נִזְרֵהּ בַּר מִדְתַדְבֵּק יְדֵהּ כְּפוּם נִדְרֵהּ דִי יִדַר כֵּן יַעְבֵּד עַל אוֹרַיְתָא דְנִזְרֵהּ: (פ)

במדבר פרק-ז

{א}  חמישי  וַיְהִ֡י בְּיוֹם֩ כַּלּ֨וֹת מֹשֶׁ֜ה לְהָקִ֣ים אֶת-הַמִּשְׁכָּ֗ן וַיִּמְשַׁ֨ח אֹת֜וֹ וַיְקַדֵּ֤שׁ אֹתוֹ֙ וְאֶת-כָּל-כֵּלָ֔יו וְאֶת-הַמִּזְבֵּ֖חַ וְאֶת-כָּל-כֵּלָ֑יו וַיִּמְשָׁחֵ֖ם וַיְקַדֵּ֥שׁ אֹתָֽם:  מ  וַיְהִ֡י בְּיוֹם֩ כַּלּ֨וֹת מֹשֶׁ֜ה לְהָקִ֣ים אֶת-הַמִּשְׁכָּ֗ן וַיִּמְשַׁ֨ח אֹת֜וֹ וַיְקַדֵּ֤שׁ אֹתוֹ֙ וְאֶת-כָּל-כֵּלָ֔יו וְאֶת-הַמִּזְבֵּ֖חַ וְאֶת-כָּל-כֵּלָ֑יו וַיִּמְשָׁחֵ֖ם וַיְקַדֵּ֥שׁ אֹתָֽם:  ת  וַהֲוָה בְּיוֹמָא דְשֵׁיצֵי משֶׁה לַאֲקָמָא יָת מַשְׁכְּנָא וְרַבִּי יָתֵהּ וְקַדִישׁ יָתֵהּ וְיָת כָּל מָנוֹהִי וְיָת מַדְבְּחָא וְיָת כָּל מָנוֹהִי וְרַבִּנוּן וְקַדֵישׁ יָתְהוֹן:

{ב} וַיַּקְרִ֨יבוּ֙ נְשִׂיאֵ֣י יִשְׂרָאֵ֔ל רָאשֵׁ֖י בֵּ֣ית אֲבֹתָ֑ם הֵ֚ם נְשִׂיאֵ֣י הַמַּטֹּ֔ת הֵ֥ם הָעֹֽמְדִ֖ים עַל-הַפְּקֻדִֽים:  מ  וַיַּקְרִ֨יבוּ֙ נְשִׂיאֵ֣י יִשְׂרָאֵ֔ל רָאשֵׁ֖י בֵּ֣ית אֲבֹתָ֑ם הֵ֚ם נְשִׂיאֵ֣י הַמַּטֹּ֔ת הֵ֥ם הָעֹֽמְדִ֖ים עַל-הַפְּקֻדִֽים:  ת  וּקְרִיבוּ רַבְרְבֵי יִשְׂרָאֵל רֵישֵׁי בֵּית אֲבָהַתְהוֹן אִנוּן רַבְרְבֵי שִׁבְטַיָא אִנוּן דְקַיְמִין עַל מִנְיָנַיָא:

{ג} וַיָּבִ֨יאוּ אֶת-קָרְבָּנָ֜ם לִפְנֵ֣י יְהֹוָ֗ה שֵׁשׁ-עֶגְלֹ֥ת צָב֙ וּשְׁנֵ֣י עָשָׂ֣ר בָּקָ֔ר עֲגָלָ֛ה עַל-שְׁנֵ֥י הַנְּשִׂאִ֖ים וְשׁ֣וֹר לְאֶחָ֑ד וַיַּקְרִ֥יבוּ אוֹתָ֖ם לִפְנֵ֥י הַמִּשְׁכָּֽן:  מ  וַיָּבִ֨יאוּ אֶת-קָרְבָּנָ֜ם לִפְנֵ֣י יְהֹוָ֗ה שֵׁשׁ-עֶגְלֹ֥ת צָב֙ וּשְׁנֵ֣י עָשָׂ֣ר בָּקָ֔ר עֲגָלָ֛ה עַל-שְׁנֵ֥י הַנְּשִׂאִ֖ים וְשׁ֣וֹר לְאֶחָ֑ד וַיַּקְרִ֥יבוּ אוֹתָ֖ם לִפְנֵ֥י הַמִּשְׁכָּֽן:  ת  וְאַיְתִיוּ יָת קֻרְבָּנֵיהוֹן קֳדָם יְיָ שִׁית עֶגְלָן כַּד מְחַפְיָן וּתְרֵי עֲשַׂר תּוֹרִין עֶגְלְתָא עַל תְּרֵין רַבְרְבַיָא וְתוֹר לְחַד וְקָרִיבוּ יָתְהוֹן לָקֳדָם מַשְׁכְּנָא:

{ד} וַ֥יֹּאמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  מ  וַ֥יֹּאמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֥ה לֵּאמֹֽר:  ת  וַאֲמַר יְיָ לְמשֶׁה לְמֵימָר:

{ה} קַ֚ח מֵֽאִתָּ֔ם וְהָי֕וּ לַֽעֲבֹ֕ד אֶת-עֲבֹדַ֖ת אֹ֣הֶל מוֹעֵ֑ד וְנָֽתַתָּ֤ה אוֹתָם֙ אֶל-הַלְוִיִּ֔ם אִ֖ישׁ כְּפִ֥י עֲבֹֽדָתֽוֹ:  מ  קַ֚ח מֵֽאִתָּ֔ם וְהָי֕וּ לַֽעֲבֹ֕ד אֶת-עֲבֹדַ֖ת אֹ֣הֶל מוֹעֵ֑ד וְנָֽתַתָּ֤ה אוֹתָם֙ אֶל-הַלְוִיִּ֔ם אִ֖ישׁ כְּפִ֥י עֲבֹֽדָתֽוֹ:  ת  קַבֵּל מִנְהוֹן וִיהוֹן לְמִפְלַח יָת פּוּלְחַן מַשְׁכַּן זִמְנָא וְתִתֵּן יָתְהוֹן לְלֵוָאֵי גְבַר כְּמִסַת פּוּלְחָנֵהּ:

{ו} וַיִּקַּ֣ח מֹשֶׁ֔ה אֶת-הָֽעֲגָלֹ֖ת וְאֶת-הַבָּקָ֑ר וַיִּתֵּ֥ן אוֹתָ֖ם אֶל-הַלְוִיִּֽם:  מ  וַיִּקַּ֣ח מֹשֶׁ֔ה אֶת-הָֽעֲגָלֹ֖ת וְאֶת-הַבָּקָ֑ר וַיִּתֵּ֥ן אוֹתָ֖ם אֶל-הַלְוִיִּֽם:  ת  וּנְסֵיב משֶׁה יָת עֶגְלָתָא וְיָת תּוֹרִין וִיהַב יָתְהוֹן לְלֵוָאֵי:

{ז} אֵ֣ת | שְׁתֵּ֣י הָֽעֲגָל֗וֹת וְאֵת֙ אַרְבַּ֣עַת הַבָּקָ֔ר נָתַ֖ן לִבְנֵ֣י גֵֽרְשׁ֑וֹן כְּפִ֖י עֲבֹֽדָתָֽם:  מ  אֵ֣ת | שְׁתֵּ֣י הָֽעֲגָל֗וֹת וְאֵת֙ אַרְבַּ֣עַת הַבָּקָ֔ר נָתַ֖ן לִבְנֵ֣י גֵֽרְשׁ֑וֹן כְּפִ֖י עֲבֹֽדָתָֽם:  ת  יָת תַּרְתֵּין עֶגְלָתָא וְיָת אַרְבַּעַת תוֹרֵי יְהַב לִבְנֵי גֵרְשׁוֹן כְּמִסַת פּוּלְחַנְהוֹן:

{ח} וְאֵ֣ת | אַרְבַּ֣ע הָֽעֲגָלֹ֗ת וְאֵת֙ שְׁמֹנַ֣ת הַבָּקָ֔ר נָתַ֖ן לִבְנֵ֣י מְרָרִ֑י כְּפִי֙ עֲבֹ֣דָתָ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן-אַהֲרֹ֖ן הַכֹּהֵֽן:  מ  וְאֵ֣ת | אַרְבַּ֣ע הָֽעֲגָלֹ֗ת וְאֵת֙ שְׁמֹנַ֣ת הַבָּקָ֔ר נָתַ֖ן לִבְנֵ֣י מְרָרִ֑י כְּפִי֙ עֲבֹ֣דָתָ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן-אַהֲרֹ֖ן הַכֹּהֵֽן:  ת  וְיָת אַרְבַּע עֶגְלָתָא וְיָת תְּמַנְיָא תוֹרֵי יְהַב לִבְנֵי מְרָרִי כְּמִסַת פּוּלְחַנְהוֹן בִּידָא דְאִיתָמָר בַּר אַהֲרֹן כַּהֲנָא:

{ט} וְלִבְנֵ֥י קְהָ֖ת לֹ֣א נָתָ֑ן כִּֽי-עֲבֹדַ֤ת הַקֹּ֨דֶשׁ֙ עֲלֵהֶ֔ם בַּכָּתֵ֖ף יִשָּֽׂאוּ:  מ  וְלִבְנֵ֥י קְהָ֖ת לֹ֣א נָתָ֑ן כִּֽי-עֲבֹדַ֤ת הַקֹּ֨דֶשׁ֙ עֲלֵהֶ֔ם בַּכָּתֵ֖ף יִשָּֽׂאוּ:  ת  וְלִבְנֵי קְהָת לָא יְהָב אֲרֵי פוּלְחַן קוּדְשָׁא עֲלֵיהוֹן בְּכַתְפָּא נָטְלִין:

{י} וַיַּקְרִ֣יבוּ הַנְּשִׂאִ֗ים אֵ֚ת חֲנֻכַּ֣ת הַמִּזְבֵּ֔חַ בְּי֖וֹם הִמָּשַׁ֣ח אֹת֑וֹ וַיַּקְרִ֧יבוּ הַנְּשִׂיאִ֛ם אֶת-קָרְבָּנָ֖ם לִפְנֵ֥י הַמִּזְבֵּֽחַ:  מ  וַיַּקְרִ֣יבוּ הַנְּשִׂאִ֗ים אֵ֚ת חֲנֻכַּ֣ת הַמִּזְבֵּ֔חַ בְּי֖וֹם הִמָּשַׁ֣ח אֹת֑וֹ וַיַּקְרִ֧יבוּ הַנְּשִׂיאִ֛ם אֶת-קָרְבָּנָ֖ם לִפְנֵ֥י הַמִּזְבֵּֽחַ:  ת  וְקָרִיבוּ רַבְרְבַיָא יָת חֲנֻכַּת מַדְבְּחָא בְּיוֹמָא דְרַבִּיוּ יָתֵהּ וְקָרִיבוּ רַבְרְבַיָא יָת קֻרְבַּנְהוֹן לְקֳדָם מַדְבְּחָא:

{יא} וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֑ה נָשִׂ֨יא אֶחָ֜ד לַיּ֗וֹם נָשִׂ֤יא אֶחָד֙ לַיּ֔וֹם יַקְרִ֨יבוּ֙ אֶת-קָרְבָּנָ֔ם לַֽחֲנֻכַּ֖ת הַמִּזְבֵּֽחַ:  מ  וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל-מֹשֶׁ֑ה נָשִׂ֨יא אֶחָ֜ד לַיּ֗וֹם נָשִׂ֤יא אֶחָד֙ לַיּ֔וֹם יַקְרִ֨יבוּ֙ אֶת-קָרְבָּנָ֔ם לַֽחֲנֻכַּ֖ת הַמִּזְבֵּֽחַ:  ת  וַאֲמַר יְיָ לְמשֶׁה רַבָּא חַד לְיוֹמָא רַבָּא חַד לְיוֹמָא יְקָרְבוּן יָת קֻרְבַּנְהוֹן לַחֲנֻכַּת מַדְבְּחָא: (ס)

{יב} וַיְהִ֗י הַמַּקְרִ֛יב בַּיּ֥וֹם הָֽרִאשׁ֖וֹן אֶת-קָרְבָּנ֑וֹ נַחְשׁ֥וֹן בֶּן-עַמִּֽינָדָ֖ב לְמַטֵּ֥ה יְהוּדָֽה:  מ  וַיְהִ֗י הַמַּקְרִ֛יב בַּיּ֥וֹם הָֽרִאשׁ֖וֹן אֶת-קָרְבָּנ֑וֹ נַחְשׁ֥וֹן בֶּן-עַמִּֽינָדָ֖ב לְמַטֵּ֥ה יְהוּדָֽה:  ת  וַהֲוָה דִמְקָרַב בְּיוֹמָא קַדְמָאָה יָת קֻרְבָּנֵהּ נַחְשׁוֹן בַּר עַמִינָדָב לְשִׁבְטָא דִיהוּדָה:

{יג} וְקָרְבָּנ֞וֹ קַֽעֲרַת-כֶּ֣סֶף אַחַ֗ת שְׁלֹשִׁ֣ים וּמֵאָה֘ מִשְׁקָלָהּ֒ מִזְרָ֤ק אֶחָד֙ כֶּ֔סֶף שִׁבְעִ֥ים שֶׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ שְׁנֵיהֶ֣ם | מְלֵאִ֗ים סֹ֛לֶת בְּלוּלָ֥ה בַשֶּׁ֖מֶן לְמִנְחָֽה:  מ  וְקָרְבָּנ֞וֹ קַֽעֲרַת-כֶּ֣סֶף אַחַ֗ת שְׁלֹשִׁ֣ים וּמֵאָה֘ מִשְׁקָלָהּ֒ מִזְרָ֤ק אֶחָד֙ כֶּ֔סֶף שִׁבְעִ֥ים שֶׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ שְׁנֵיהֶ֣ם | מְלֵאִ֗ים סֹ֛לֶת בְּלוּלָ֥ה בַשֶּׁ֖מֶן לְמִנְחָֽה:  ת  וְקֻרְבָּנֵהּ מְגִסְתָּא דִכְסַף חֲדָא מְאָה וּתְלָתִין סִלְעִין הֲוָה מַתְקְלַהּ מִזְרְקָא חַד דִכְסַף מַתְקְלֵהּ שַׁבְעִין סִלְעִין בְּסִלְעֵי קוּדְשָׁא תַּרְוֵיהוֹן מְלַן סֻלְתָּא דְּפִילָא בִמְשַׁח לְמִנְחָתָא:

{יד} כַּ֥ף אַחַ֛ת עֲשָׂרָ֥ה זָהָ֖ב מְלֵאָ֥ה קְטֹֽרֶת:  מ  כַּ֥ף אַחַ֛ת עֲשָׂרָ֥ה זָהָ֖ב מְלֵאָ֥ה קְטֹֽרֶת:  ת  בָּזִיכָא חֲדָא מַתְקַל עֲשַׂר סִלְעִין הִיא דִדְהַב מַלְיָא קְטוֹרֶת בּוּסְמַיָא:

{טו} פַּ֣ר אֶחָ֞ד בֶּן-בָּקָ֗ר אַ֧יִל אֶחָ֛ד כֶּֽבֶשׂ-אֶחָ֥ד בֶּן-שְׁנָת֖וֹ לְעֹלָֽה:  מ  פַּ֣ר אֶחָ֞ד בֶּן-בָּקָ֗ר אַ֧יִל אֶחָ֛ד כֶּֽבֶשׂ-אֶחָ֥ד בֶּן-שְׁנָת֖וֹ לְעֹלָֽה:  ת  תּוֹר חַד בַּר תּוֹרֵי דְּכַר חַד אִמַּר חַד בַּר שַׁתֵּהּ לַעֲלָתָא:

{טז} שְׂעִיר-עִזִּ֥ים אֶחָ֖ד לְחַטָּֽאת:  מ  שְׂעִיר-עִזִּ֥ים אֶחָ֖ד לְחַטָּֽאת:  ת  צְפִיר בַּר עִזִּין חַד לְחַטָּאתָא:

{יז} וּלְזֶ֣בַח הַשְּׁלָמִים֘ בָּקָ֣ר שְׁנַ֒יִם֒ אֵילִ֤ם חֲמִשָּׁה֙ עַתּוּדִ֣ים חֲמִשָּׁ֔ה כְּבָשִׂ֥ים בְּנֵֽי-שָׁנָ֖ה חֲמִשָּׁ֑ה זֶ֛ה קָרְבַּ֥ן נַחְשׁ֖וֹן בֶּן-עַמִּֽינָדָֽב:  מ  וּלְזֶ֣בַח הַשְּׁלָמִים֘ בָּקָ֣ר שְׁנַ֒יִם֒ אֵילִ֤ם חֲמִשָּׁה֙ עַתּוּדִ֣ים חֲמִשָּׁ֔ה כְּבָשִׂ֥ים בְּנֵֽי-שָׁנָ֖ה חֲמִשָּׁ֑ה זֶ֛ה קָרְבַּ֥ן נַחְשׁ֖וֹן בֶּן-עַמִּֽינָדָֽב:  ת  וּלְנִכְסַת קוּדְשַׁיָּא תּוֹרִין תְּרֵין דִּכְרִין חַמְשָׁא גְּדָיֵי חַמְשָׁא אִמְּרִין בְּנֵי שַׁתָּא חַמְשָׁא דֵּין קֻרְבָּנָא דְנַחְשׁוֹן בַּר עַמִּינָדָב: (פ)

{יח} בַּיּוֹם֙ הַשֵּׁנִ֔י הִקְרִ֖יב נְתַנְאֵ֣ל בֶּן-צוּעָ֑ר נְשִׂ֖יא יִשָּׂשכָֽר:  מ  בַּיּוֹם֙ הַשֵּׁנִ֔י הִקְרִ֖יב נְתַנְאֵ֣ל בֶּן-צוּעָ֑ר נְשִׂ֖יא יִשָּׂשכָֽר:  ת  בְּיוֹמָא תִּנְיָנָא קָרִיב נְתַנְאֵל בַּר צוּעָר רַבָּה דְיִשָׂשׂכָר:

{פה} שְׁלֹשִׁ֣ים וּמֵאָ֗ה הַקְּעָרָ֤ה הָֽאַחַת֙ כֶּ֔סֶף וְשִׁבְעִ֖ים הַמִּזְרָ֣ק הָֽאֶחָ֑ד כֹּ֚ל כֶּ֣סֶף הַכֵּלִ֔ים אַלְפַּ֥יִם וְאַרְבַּע-מֵא֖וֹת בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ:  מ  שְׁלֹשִׁ֣ים וּמֵאָ֗ה הַקְּעָרָ֤ה הָֽאַחַת֙ כֶּ֔סֶף וְשִׁבְעִ֖ים הַמִּזְרָ֣ק הָֽאֶחָ֑ד כֹּ֚ל כֶּ֣סֶף הַכֵּלִ֔ים אַלְפַּ֥יִם וְאַרְבַּע-מֵא֖וֹת בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ:  ת  מְאָה וּתְלָתִין סִלְעִין הֲוָה מַתְקְלָא דִמְגִיסְתָּא חֲדָא דְכַסְפָּא וְשַׁבְעִין דְמִזְרְקָא חָד כֹּל כְּסַף מָנַיָא תְּרֵין אַלְפִין וְאַרְבַּע מְאָה בְּסִלְעֵי קוּדְשָׁא:

{פו} כַּפּ֨וֹת זָהָ֤ב שְׁתֵּים-עֶשְׂרֵה֙ מְלֵאֹ֣ת קְטֹ֔רֶת עֲשָׂרָ֧ה עֲשָׂרָ֛ה הַכַּ֖ף בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ כָּל-זְהַ֥ב הַכַּפּ֖וֹת עֶשְׂרִ֥ים וּמֵאָֽה:  מ  כַּפּ֨וֹת זָהָ֤ב שְׁתֵּים-עֶשְׂרֵה֙ מְלֵאֹ֣ת קְטֹ֔רֶת עֲשָׂרָ֧ה עֲשָׂרָ֛ה הַכַּ֖ף בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ כָּל-זְהַ֥ב הַכַּפּ֖וֹת עֶשְׂרִ֥ים וּמֵאָֽה:  ת  בָּזִיכֵי דְדַהֲבָא תַּרְתֵּי עַשְׂרֵי מַלְיָן קְטֹרֶת בּוּסְמַיָא מַתְקַל עֲשַׂר סִלְעִין הֲוָה מַתְקְלָא דְבָזִיכָא בְּסִלְעֵי קוּדְשָׁא כָּל דְהַב בָּזִיכַיָא מְאָה וְעַשְׂרִין:

{פז}  מפטיר  כָּל-הַבָּקָ֨ר לָֽעֹלָ֜ה שְׁנֵ֧ים עָשָׂ֣ר פָּרִ֗ים אֵילִ֤ם שְׁנֵים-עָשָׂר֙ כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה שְׁנֵ֥ים עָשָׂ֖ר וּמִנְחָתָ֑ם וּשְׂעִירֵ֥י עִזִּ֛ים שְׁנֵ֥ים עָשָׂ֖ר לְחַטָּֽאת:  מ  כָּל-הַבָּקָ֨ר לָֽעֹלָ֜ה שְׁנֵ֧ים עָשָׂ֣ר פָּרִ֗ים אֵילִ֤ם שְׁנֵים-עָשָׂר֙ כְּבָשִׂ֧ים בְּנֵֽי-שָׁנָ֛ה שְׁנֵ֥ים עָשָׂ֖ר וּמִנְחָתָ֑ם וּשְׂעִירֵ֥י עִזִּ֛ים שְׁנֵ֥ים עָשָׂ֖ר לְחַטָּֽאת:  ת  כָּל תּוֹרֵי לַעֲלָתָא תְּרֵי עֲשַׂר תּוֹרִין דִכְרִין תְּרֵי עֲשַׂר אִמְרִין בְּנֵי שַׁתָּא תְּרֵי עֲשַׂר וּמִנְחַתְהוֹן וּצְפִירֵי בַּר עִזִין תְּרֵי עֲשַׂר לְחַטָאתָא:
