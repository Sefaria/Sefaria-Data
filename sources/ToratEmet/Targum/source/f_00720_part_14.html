נ-א


[ישעיה פרק-כד-י]  נִשְׁבְּרָה קִרְיַת תֹּהוּ סֻגַּר כָּל בַּיִת מִבּוֹא:

[תהילים פרק-לג-כ]  נַפְשֵׁנוּ חִכְּתָה לַיהוָה עֶזְרֵנוּ וּמָגִנֵּנוּ הוּא:

[תהילים פרק-עו-יב]  נִדֲרוּ וְשַׁלְּמוּ לַיהוָה אֱלֹהֵיכֶם כָּל סְבִיבָיו יוֹבִילוּ שַׁי לַמּוֹרָא:

[משלי פרק-יא-כה]  נֶפֶשׁ בְּרָכָה תְדֻשָּׁן וּמַרְוֶה גַּם הוּא יוֹרֶא:

[משלי פרק-כז-ו]  נֶאֱמָנִים פִּצְעֵי אוֹהֵב וְנַעְתָּרוֹת נְשִׁיקוֹת שׂוֹנֵא:

[איוב פרק-מא-יג]  נַפְשׁוֹ גֶּחָלִים תְּלַהֵט וְלַהַב מִפִּיו יֵצֵא:

[דניאל פרק-ג-לא]  נְבוּכַדְנֶצַּר מַלְכָּא לְכָל עַמְמַיָּא אֻמַיָּא וְלִשָּׁנַיָּא דִּי (דארין) דָיְרִין בְּכָל אַרְעָא שְׁלָמְכוֹן יִשְׂגֵּא:


נ-ב


[תהילים פרק-קיט-קיב]  נָטִיתִי לִבִּי לַעֲשׂוֹת חֻקֶּיךָ לְעוֹלָם עֵקֶב:

[משלי פרק-כב-א]  נִבְחָר שֵׁם מֵעֹשֶׁר רָב מִכֶּסֶף וּמִזָּהָב חֵן טוֹב:

[עזרא פרק-ו-ד]  נִדְבָּכִין דִּי אֶבֶן גְּלָל תְּלָתָא וְנִדְבָּךְ דִּי אָע חֲדַת וְנִפְקְתָא מִן בֵּית מַלְכָּא תִּתְיְהִב:


נ-ד


[ירמיה פרק-יג-כז]  נִאֻפַיִךְ וּמִצְהֲלוֹתַיִךְ זִמַּת זְנוּתֵךְ עַל גְּבָעוֹת בַּשָּׂדֶה רָאִיתִי שִׁקּוּצָיִךְ אוֹי לָךְ יְרוּשָׁלִַם לֹא תִטְהֲרִי אַחֲרֵי מָתַי עֹד:

[ירמיה פרק-נא-יד]  נִשְׁבַּע יְהוָה צְבָאוֹת בְּנַפְשׁוֹ כִּי אִם מִלֵּאתִיךְ אָדָם כַּיֶּלֶק וְעָנוּ עָלַיִךְ הֵידָד:

[תהילים פרק-לא-יג]  נִשְׁכַּחְתִּי כְּמֵת מִלֵּב הָיִיתִי כִּכְלִי אֹבֵד:

[משלי פרק-כז-יח]  נֹצֵר תְּאֵנָה יֹאכַל פִּרְיָהּ וְשֹׁמֵר אֲדֹנָיו יְכֻבָּד:


נ-ה


[ויקרא פרק-כב-ח]  נְבֵלָה וּטְרֵפָה לֹא יֹאכַל לְטָמְאָה בָהּ אֲנִי יְהוָה:

[שמואל א פרק-טו-יא]  נִחַמְתִּי כִּי הִמְלַכְתִּי אֶת שָׁאוּל לְמֶלֶךְ כִּי שָׁב מֵאַחֲרַי וְאֶת דְּבָרַי לֹא הֵקִים וַיִּחַר לִשְׁמוּאֵל וַיִּזְעַק אֶל יְהוָה כָּל הַלָּיְלָה:

[מלכים ב פרק-ד-י]  נַעֲשֶׂה נָּא עֲלִיַּת קִיר קְטַנָּה וְנָשִׂים לוֹ שָׁם מִטָּה וְשֻׁלְחָן וְכִסֵּא וּמְנוֹרָה וְהָיָה בְּבֹאוֹ אֵלֵינוּ יָסוּר שָׁמָּה:

[ישעיה פרק-יד-ז]  נָחָה שָׁקְטָה כָּל הָאָרֶץ פָּצְחוּ רִנָּה:

[ישעיה פרק-יט-יג]  נוֹאֲלוּ שָׂרֵי צֹעַן נִשְּׁאוּ שָׂרֵי נֹף הִתְעוּ אֶת מִצְרַיִם פִּנַּת שְׁבָטֶיהָ:

[ישעיה פרק-לג-ה]  נִשְׂגָּב יְהוָה כִּי שֹׁכֵן מָרוֹם מִלֵּא צִיּוֹן מִשְׁפָּט וּצְדָקָה:

[ישעיה פרק-מ-כט]  נֹתֵן לַיָּעֵף כֹּחַ וּלְאֵין אוֹנִים עָצְמָה יַרְבֶּה:

[ירמיה פרק-מח-ד]  נִשְׁבְּרָה מוֹאָב הִשְׁמִיעוּ זְּעָקָה (צעוריה) צְעִירֶיהָ:

[ירמיה פרק-מח-יז]  נֻדוּ לוֹ כָּל סְבִיבָיו וְכֹל יֹדְעֵי שְׁמוֹ אִמְרוּ אֵיכָה נִשְׁבַּר מַטֵּה עֹז מַקֵּל תִּפְאָרָה:

[ירמיה פרק-מח-כה]  נִגְדְּעָה קֶרֶן מוֹאָב וּזְרֹעוֹ נִשְׁבָּרָה נְאֻם יְהוָה:

[ירמיה פרק-מח-מא]  נִלְכְּדָה הַקְּרִיּוֹת וְהַמְּצָדוֹת נִתְפָּשָׂה וְהָיָה לֵב גִּבּוֹרֵי מוֹאָב בַּיּוֹם הַהוּא כְּלֵב אִשָּׁה מְצֵרָה:

[ירמיה פרק-מט-ל]  נֻסוּ נֻּדוּ מְאֹד הֶעְמִיקוּ לָשֶׁבֶת יֹשְׁבֵי חָצוֹר נְאֻם יְהוָה כִּי יָעַץ עֲלֵיכֶם נְבוּכַדְרֶאצַּר מֶלֶךְ בָּבֶל עֵצָה וְחָשַׁב (עליהם) עֲלֵיכֶם מַחֲשָׁבָה:

[ירמיה פרק-נא-ו]  נֻסוּ מִתּוֹךְ בָּבֶל וּמַלְּטוּ אִישׁ נַפְשׁוֹ אַל תִּדַּמּוּ בַּעֲוֹנָהּ כִּי עֵת נְקָמָה הִיא לַיהוָה גְּמוּל הוּא מְשַׁלֵּם לָהּ:

[יחזקאל פרק-יג-טז]  נְבִיאֵי יִשְׂרָאֵל הַנִבְּאִים אֶל יְרוּשָׁלִַם וְהַחֹזִים לָהּ חֲזוֹן שָׁלֹם וְאֵין שָׁלֹם נְאֻם אֲדֹנָי יְהוִֹה:

[עמוס פרק-ד-ב]  נִשְׁבַּע אֲדֹנָי יְהוִה בְּקָדְשׁוֹ כִּי הִנֵּה יָמִים בָּאִים עֲלֵיכֶם וְנִשָּׂא אֶתְכֶם בְּצִנּוֹת וְאַחֲרִיתְכֶן בְּסִירוֹת דּוּגָה:

[עמוס פרק-ה-ב]  נָפְלָה לֹא תוֹסִיף קוּם בְּתוּלַת יִשְׂרָאֵל נִטְּשָׁה עַל אַדְמָתָהּ אֵין מְקִימָהּ:

[עמוס פרק-ו-ח]  נִשְׁבַּע אֲדֹנָי יְהוִה בְּנַפְשׁוֹ נְאֻם יְהוָה אֱלֹהֵי צְבָאוֹת מְתָאֵב אָנֹכִי אֶת גְּאוֹן יַעֲקֹב וְאַרְמְנֹתָיו שָׂנֵאתִי וְהִסְגַּרְתִּי עִיר וּמְלֹאָהּ:

[עמוס פרק-ז-ג]  נִחַם יְהוָה עַל זֹאת לֹא תִהְיֶה אָמַר יְהוָה:

[עמוס פרק-ז-ו]  נִחַם יְהוָה עַל זֹאת גַּם הִיא לֹא תִהְיֶה אָמַר אֲדֹנָי יְהוִה:

[צפניה פרק-ג-ד]  נְבִיאֶיהָ פֹּחֲזִים אַנְשֵׁי בֹּגְדוֹת כֹּהֲנֶיהָ חִלְּלוּ קֹדֶשׁ חָמְסוּ תּוֹרָה:

[צפניה פרק-ג-יח]  נוּגֵי מִמּוֹעֵד אָסַפְתִּי מִמֵּךְ הָיוּ מַשְׂאֵת עָלֶיהָ חֶרְפָּה:

[תהילים פרק-ט-יז]  נוֹדַע יְהוָה מִשְׁפָּט עָשָׂה בְּפֹעַל כַּפָּיו נוֹקֵשׁ רָשָׁע הִגָּיוֹן סֶלָה:

[תהילים פרק-כד-ד]  נְקִי כַפַּיִם וּבַר לֵבָב אֲשֶׁר לֹא נָשָׂא לַשָּׁוְא נַפְשִׁי וְלֹא נִשְׁבַּע לְמִרְמָה:

[תהילים פרק-לד-יד]  נְצֹר לְשׁוֹנְךָ מֵרָע וּשְׂפָתֶיךָ מִדַּבֵּר מִרְמָה:

[תהילים פרק-מז-י]  נְדִיבֵי עַמִּים נֶאֱסָפוּ עַם אֱלֹהֵי אַבְרָהָם כִּי לֵאלֹהִים מָגִנֵּי אֶרֶץ מְאֹד נַעֲלָה:

[תהילים פרק-נז-ה]  נַפְשִׁי בְּתוֹךְ לְבָאִם אֶשְׁכְּבָה לֹהֲטִים בְּנֵי אָדָם שִׁנֵּיהֶם חֲנִית וְחִצִּים וּלְשׁוֹנָם חֶרֶב חַדָּה:

[תהילים פרק-נז-ח]  נָכוֹן לִבִּי אֱלֹהִים נָכוֹן לִבִּי אָשִׁירָה וַאֲזַמֵּרָה:

[תהילים פרק-ס-ו]  נָתַתָּה לִּירֵאֶיךָ נֵּס לְהִתְנוֹסֵס מִפְּנֵי קֹשֶׁט סֶלָה:

[תהילים פרק-עה-ד]  נְמֹגִים אֶרֶץ וְכָל יֹשְׁבֶיהָ אָנֹכִי תִכַּנְתִּי עַמּוּדֶיהָ סֶּלָה:

[תהילים פרק-פג-יא]  נִשְׁמְדוּ בְעֵין דֹּאר הָיוּ דֹּמֶן לָאֲדָמָה:

[תהילים פרק-פה-ג]  נָשָׂאתָ עֲוֹן עַמֶּךָ כִּסִּיתָ כָל חַטָּאתָם סֶלָה:

[תהילים פרק-פז-ג]  נִכְבָּדוֹת מְדֻבָּר בָּךְ עִיר הָאֱלֹהִים סֶלָה:

[תהילים פרק-צג-ב]  נָכוֹן כִּסְאֲךָ מֵאָז מֵעוֹלָם אָתָּה:

[תהילים פרק-קיט-קיא]  נָחַלְתִּי עֵדְוֹתֶיךָ לְעוֹלָם כִּי שְׂשׂוֹן לִבִּי הֵמָּה:

[משלי פרק-ו-לב]  נֹאֵף אִשָּׁה חֲסַר לֵב מַשְׁחִית נַפְשׁוֹ הוּא יַעֲשֶׂנָּה:

[משלי פרק-ו-לג]  נֶגַע וְקָלוֹן יִמְצָא וְחֶרְפָּתוֹ לֹא תִמָּחֶה:

[איוב פרק-כו-ז]  נֹטֶה צָפוֹן עַל תֹּהוּ תֹּלֶה אֶרֶץ עַל בְּלִי מָה:

[איוב פרק-כח-ז]  נָתִיב לֹא יְדָעוֹ עָיִט וְלֹא שְׁזָפַתּוּ עֵין אַיָּה:

[איכה פרק-ג-מ]  נַחְפְּשָׂה דְרָכֵינוּ וְנַחְקֹרָה וְנָשׁוּבָה עַד יְהוָה:

[איכה פרק-ה-יא]  נָשִׁים בְּצִיּוֹן עִנּוּ בְּתֻלֹת בְּעָרֵי יְהוּדָה:

[דניאל פרק-א-יב]  נַס נָא אֶת עֲבָדֶיךָ יָמִים עֲשָׂרָה וְיִתְּנוּ לָנוּ מִן הַזֵּרֹעִים וְנֹאכְלָה וּמַיִם וְנִשְׁתֶּה:


נ-ו


[שמות פרק-כז-ח]  נְבוּב לֻחֹת תַּעֲשֶׂה אֹתוֹ כַּאֲשֶׁר הֶרְאָה אֹתְךָ בָּהָר כֵּן יַעֲשׂוּ:

[ויקרא פרק-ה-כא]  נֶפֶשׁ כִּי תֶחֱטָא וּמָעֲלָה מַעַל בַּיהוָה וְכִחֵשׁ בַּעֲמִיתוֹ בְּפִקָּדוֹן אוֹ בִתְשׂוּמֶת יָד אוֹ בְגָזֵל אוֹ עָשַׁק אֶת עֲמִיתוֹ:

[דברים פרק-יח-יח]  נָבִיא אָקִים לָהֶם מִקֶּרֶב אֲחֵיהֶם כָּמוֹךָ וְנָתַתִּי דְבָרַי בְּפִיו וְדִבֶּר אֲלֵיהֶם אֵת כָּל אֲשֶׁר אֲצַוֶּנּוּ:

[מלכים ב פרק-ו-ב]  נֵלְכָה נָּא עַד הַיַּרְדֵּן וְנִקְחָה מִשָּׁם אִישׁ קוֹרָה אֶחָת וְנַעֲשֶׂה לָּנוּ שָׁם מָקוֹם לָשֶׁבֶת שָׁם וַיֹּאמֶר לֵכוּ:

[ישעיה פרק-י-לא]  נָדְדָה מַדְמֵנָה יֹשְׁבֵי הַגֵּבִים הֵעִיזוּ:

[ישעיה פרק-מב-יז]  נָסֹגוּ אָחוֹר יֵבֹשׁוּ בֹשֶׁת הַבֹּטְחִים בַּפָּסֶל הָאֹמְרִים לְמַסֵּכָה אַתֶּם אֱלֹהֵינוּ:

[ישעיה פרק-נג-ג]  נִבְזֶה וַחֲדַל אִישִׁים אִישׁ מַכְאֹבוֹת וִידוּעַ חֹלִי וּכְמַסְתֵּר פָּנִים מִמֶּנּוּ נִבְזֶה וְלֹא חֲשַׁבְנֻהוּ:

[ישעיה פרק-נג-ז]  נִגַּשׂ וְהוּא נַעֲנֶה וְלֹא יִפְתַּח פִּיו כַּשֶּׂה לַטֶּבַח יוּבָל וּכְרָחֵל לִפְנֵי גֹזְזֶיהָ נֶאֱלָמָה וְלֹא יִפְתַּח פִּיו:

[ישעיה פרק-נו-ח]  נְאֻם אֲדֹנָי יְהוִה מְקַבֵּץ נִדְחֵי יִשְׂרָאֵל עוֹד אֲקַבֵּץ עָלָיו לְנִקְבָּצָיו:

[ישעיה פרק-נט-יא]  נֶהֱמֶה כַדֻּבִּים כֻּלָּנוּ וְכַיּוֹנִים הָגֹה נֶהְגֶּה נְקַוֶּה לַמִּשְׁפָּט וָאַיִן לִישׁוּעָה רָחֲקָה מִמֶּנּוּ:

[ישעיה פרק-סב-ח]  נִשְׁבַּע יְהוָה בִּימִינוֹ וּבִזְרוֹעַ עֻזּוֹ אִם אֶתֵּן אֶת דְּגָנֵךְ עוֹד מַאֲכָל לְאֹיְבַיִךְ וְאִם יִשְׁתּוּ בְנֵי נֵכָר תִּירוֹשֵׁךְ אֲשֶׁר יָגַעַתְּ בּוֹ:

[ירמיה פרק-ג-כה]  נִשְׁכְּבָה בְּבָשְׁתֵּנוּ וּתְכַסֵּנוּ כְּלִמָּתֵנוּ כִּי לַיהוָה אֱלֹהֵינוּ חָטָאנוּ אֲנַחְנוּ וַאֲבוֹתֵינוּ מִנְּעוּרֵינוּ וְעַד הַיּוֹם הַזֶּה וְלֹא שָׁמַעְנוּ בְּקוֹל יְהוָֹה אֱלֹהֵינוּ:

[ירמיה פרק-ו-כט]  נָחַר מַפֻּחַ (מאשתם) מֵאֵשׁ תַּם עֹפָרֶת לַשָּׁוְא צָרַף צָרוֹף וְרָעִים לֹא נִתָּקוּ:

[ירמיה פרק-מט-ח]  נֻסוּ הָפְנוּ הֶעְמִיקוּ לָשֶׁבֶת יֹשְׁבֵי דְּדָן כִּי אֵיד עֵשָׂו הֵבֵאתִי עָלָיו עֵת פְּקַדְתִּיו:

[הושע פרק-ח-ח]  נִבְלַע יִשְׂרָאֵל עַתָּה הָיוּ בַגּוֹיִם כִּכְלִי אֵין חֵפֶץ בּוֹ:

[תהילים פרק-ב-ג]  נְנַתְּקָה אֶת מוֹסְרוֹתֵימוֹ וְנַשְׁלִיכָה מִמֶּנּוּ עֲבֹתֵימוֹ:

[תהילים פרק-ב-יב]  נַשְּׁקוּ בַר פֶּן יֶאֱנַף וְתֹאבְדוּ דֶרֶךְ כִּי יִבְעַר כִּמְעַט אַפּוֹ אַשְׁרֵי כָּל חוֹסֵי בוֹ:

[תהילים פרק-ד-ח]  נָתַתָּה שִׂמְחָה בְלִבִּי מֵעֵת דְּגָנָם וְתִירוֹשָׁם רָבּוּ:

[תהילים פרק-כג-ג]  נַפְשִׁי יְשׁוֹבֵב יַנְחֵנִי בְמַעְגְּלֵי צֶדֶק לְמַעַן שְׁמוֹ:

[תהילים פרק-לו-ב]  נְאֻם פֶּשַׁע לָרָשָׁע בְּקֶרֶב לִבִּי אֵין פַּחַד אֱלֹהִים לְנֶגֶד עֵינָיו:

[תהילים פרק-עו-ב]  נוֹדָע בִּיהוּדָה אֱלֹהִים בְּיִשְׂרָאֵל גָּדוֹל שְׁמוֹ:

[תהילים פרק-פט-מ]  נֵאַרְתָּה בְּרִית עַבְדֶּךָ חִלַּלְתָּ לָאָרֶץ נִזְרוֹ:

[תהילים פרק-צה-ב]  נְקַדְּמָה פָנָיו בְּתוֹדָה בִּזְמִרוֹת נָרִיעַ לוֹ:

[תהילים פרק-צח-ח]  נְהָרוֹת יִמְחֲאוּ כָף יַחַד הָרִים יְרַנֵּנוּ:

[תהילים פרק-קטז-יד]  נְדָרַי לַיהוָה אֲשַׁלֵּם נֶגְדָה נָּא לְכָל עַמּוֹ:

[תהילים פרק-קטז-יח]  נְדָרַי לַיהוָה אֲשַׁלֵּם נֶגְדָה נָּא לְכָל עַמּוֹ:

[תהילים פרק-קכד-ז]  נַפְשֵׁנוּ כְּצִפּוֹר נִמְלְטָה מִפַּח יוֹקְשִׁים הַפַּח נִשְׁבָּר וַאֲנַחְנוּ נִמְלָטְנוּ:

[תהילים פרק-קלב-ז]  נָבוֹאָה לְמִשְׁכְּנוֹתָיו נִשְׁתַּחֲוֶה לַהֲדֹם רַגְלָיו:

[תהילים פרק-קלו-כב]  נַחֲלָה לְיִשְׂרָאֵל עַבְדּוֹ כִּי לְעוֹלָם חַסְדּוֹ:

[תהילים פרק-קלו-כה]  נֹתֵן לֶחֶם לְכָל בָּשָׂר כִּי לְעוֹלָם חַסְדּוֹ:

[תהילים פרק-קמא-ו]  נִשְׁמְטוּ בִידֵי סֶלַע שֹׁפְטֵיהֶם וְשָׁמְעוּ אֲמָרַי כִּי נָעֵמוּ:

[תהילים פרק-קמז-ט]  נוֹתֵן לִבְהֵמָה לַחְמָהּ לִבְנֵי עֹרֵב אֲשֶׁר יִקְרָאוּ:

[משלי פרק-יג-ג]  נֹצֵר פִּיו שֹׁמֵר נַפְשׁוֹ פֹּשֵׂק שְׂפָתָיו מְחִתָּה לוֹ:

[משלי פרק-טז-כו]  נֶפֶשׁ עָמֵל עָמְלָה לּוֹ כִּי אָכַף עָלָיו פִּיהוּ:

[משלי פרק-יט-יב]  נַהַם כַּכְּפִיר זַעַף מֶלֶךְ וּכְטַל עַל עֵשֶׂב רְצוֹנוֹ:

[משלי פרק-כ-ב]  נַהַם כַּכְּפִיר אֵימַת מֶלֶךְ מִתְעַבְּרוֹ חוֹטֵא נַפְשׁוֹ:

[משלי פרק-כא-י]  נֶפֶשׁ רָשָׁע אִוְּתָה רָע לֹא יֻחַן בְּעֵינָיו רֵעֵהוּ:

[משלי פרק-כח-ז]  נוֹצֵר תּוֹרָה בֵּן מֵבִין וְרֹעֶה זוֹלְלִים יַכְלִים אָבִיו:

[משלי פרק-כח-כב]  נִבֳהָל לַהוֹן אִישׁ רַע עָיִן וְלֹא יֵדַע כִּי חֶסֶר יְבֹאֶנּוּ:

[איוב פרק-ל-יג]  נָתְסוּ נְתִיבָתִי לְהַוָּתִי יֹעִילוּ לֹא עֹזֵר לָמוֹ:

[איכה פרק-ה-טז]  נָפְלָה עֲטֶרֶת רֹאשֵׁנוּ אוֹי נָא לָנוּ כִּי חָטָאנוּ:

[קהלת פרק-י-ו]  נִתַּן הַסֶּכֶל בַּמְּרוֹמִים רַבִּים וַעֲשִׁירִים בַּשֵּׁפֶל יֵשֵׁבוּ:

[דניאל פרק-ז-י]  נְהַר דִּי נוּר נָגֵד וְנָפֵק מִן קֳדָמוֹהִי אֶלֶף (אלפים) אַלְפִין יְשַׁמְּשׁוּנֵּהּ וְרִבּוֹ (רבון) רִבְבָן קָדָמוֹהִי יְקוּמוּן דִּינָא יְתִב וְסִפְרִין פְּתִיחוּ:

[דברי הימים א פרק-ז-כז]  נוֹן בְּנוֹ יְהוֹשֻׁעַ בְּנוֹ:


נ-ז


[שופטים פרק-ה-כא]  נַחַל קִישׁוֹן גְּרָפָם נַחַל קְדוּמִים נַחַל קִישׁוֹן תִּדְרְכִי נַפְשִׁי עֹז:

[ישעיה פרק-לג-כג]  נִטְּשׁוּ חֲבָלָיִךְ בַּל יְחַזְּקוּ כֵן תָּרְנָם בַּל פָּרְשׂוּ נֵס אָז חֻלַּק עַד שָׁלָל מַרְבֶּה פִּסְחִים בָּזְזוּ בַז:


נ-ח


[יהושע פרק-יז-י]  נֶגְבָּה לְאֶפְרַיִם וְצָפוֹנָה לִמְנַשֶּׁה וַיְהִי הַיָּם גְּבוּלוֹ וּבְאָשֵׁר יִפְגְּעוּן מִצָּפוֹן וּבְיִשָּׂשכָר מִמִּזְרָח:

[משלי פרק-כח-א]  נָסוּ וְאֵין רֹדֵף רָשָׁע וְצַדִּיקִים כִּכְפִיר יִבְטָח:


נ-י


[ישעיה פרק-לב-ט]  נָשִׁים שַׁאֲנַנּוֹת קֹמְנָה שְׁמַעְנָה קוֹלִי בָּנוֹת בֹּטחוֹת הַאְזֵנָּה אִמְרָתִי:

[ישעיה פרק-סה-א]  נִדְרַשְׁתִּי לְלוֹא שָׁאָלוּ נִמְצֵאתִי לְלֹא בִקְשֻׁנִי אָמַרְתִּי הִנֵּנִי הִנֵּנִי אֶל גּוֹי לֹא קֹרָא בִשְׁמִי:

[הושע פרק-ד-ו]  נִדְמוּ עַמִּי מִבְּלִי הַדָּעַת כִּי אַתָּה הַדַּעַת מָאַסְתָּ וְאֶמְאָסְאךָ מִכַּהֵן לִי וַתִּשְׁכַּח תּוֹרַת אֱלֹהֶיךָ אֶשְׁכַּח בָּנֶיךָ גַּם אָנִי:

[תהילים פרק-לח-ז]  נַעֲוֵיתִי שַׁחֹתִי עַד מְאֹד כָּל הַיּוֹם קֹדֵר הִלָּכְתִּי:

[תהילים פרק-לח-ט]  נְפוּגוֹתִי וְנִדְכֵּיתִי עַד מְאֹד שָׁאַגְתִּי מִנַּהֲמַת לִבִּי:

[תהילים פרק-פד-ג]  נִכְסְפָה וְגַם כָּלְתָה נַפְשִׁי לְחַצְרוֹת יְהוָה לִבִּי וּבְשָׂרִי יְרַנְּנוּ אֶל אֵל חָי:

[תהילים פרק-קח-ב]  נָכוֹן לִבִּי אֱלֹהִים אָשִׁירָה וַאֲזַמְּרָה אַף כְּבוֹדִי:

[תהילים פרק-קיט-קה]  נֵר לְרַגְלִי דְבָרֶךָ וְאוֹר לִנְתִיבָתִי:

[תהילים פרק-קיט-קח]  נִדְבוֹת פִּי רְצֵה נָא יְהוָה וּמִשְׁפָּטֶיךָ לַמְּדֵנִי:

[תהילים פרק-קיט-קט]  נַפְשִׁי בְכַפִּי תָמִיד וְתוֹרָתְךָ לֹא שָׁכָחְתִּי:

[תהילים פרק-קיט-קי]  נָתְנוּ רְשָׁעִים פַּח לִי וּמִפִּקּוּדֶיךָ לֹא תָעִיתִי:

[איוב פרק-י-א]  נָקְטָה נַפְשִׁי בְּחַיָּי אֶעֶזְבָה עָלַי שִׂיחִי אֲדַבְּרָה בְּמַר נַפְשִׁי:

[עזרא פרק-ד-יח]  נִשְׁתְּוָנָא דִּי שְׁלַחְתּוּן עֲלֶינָא מְפָרַשׁ קֱרִי קָדָמָי:

[דברי הימים א פרק-ב-יד]  נְתַנְאֵל הָרְבִיעִי רַדַּי הַחֲמִישִׁי:

[דברי הימים א פרק-ח-ב]  נוֹחָה הָרְבִיעִי וְרָפָא הַחֲמִישִׁי:


נ-כ


[שמות פרק-טו-יג]  נָחִיתָ בְחַסְדְּךָ עַם זוּ גָּאָלְתָּ נֵהַלְתָּ בְעָזְּךָ אֶל נְוֵה קָדְשֶׁךָ:

[שמות פרק-יח-יח]  נָבֹל תִּבֹּל גַּם אַתָּה גַּם הָעָם הַזֶּה אֲשֶׁר עִמָּךְ כִּי כָבֵד מִמְּךָ הַדָּבָר לֹא תוּכַל עֲשֹׂהוּ לְבַדֶּךָ:

[במדבר פרק-כ-יז]  נַעְבְּרָה נָּא בְאַרְצֶךָ לֹא נַעֲבֹר בְּשָׂדֶה וּבְכֶרֶם וְלֹא נִשְׁתֶּה מֵי בְאֵר דֶּרֶךְ הַמֶּלֶךְ נֵלֵךְ לֹא נִטֶּה יָמִין וּשְׂמֹאול עַד אֲשֶׁר נַעֲבֹר גְּבוּלֶךָ:

[במדבר פרק-לא-ב]  נְקֹם נִקְמַת בְּנֵי יִשְׂרָאֵל מֵאֵת הַמִּדְיָנִים אַחַר תֵּאָסֵף אֶל עַמֶּיךָ:

[דברים פרק-טו-י]  נָתוֹן תִּתֵּן לוֹ וְלֹא יֵרַע לְבָבְךָ בְּתִתְּךָ לוֹ כִּי בִּגְלַל הַדָּבָר הַזֶּה יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכָל מַעֲשֶׂךָ וּבְכֹל מִשְׁלַח יָדֶךָ:

[ישעיה פרק-מז-יג]  נִלְאֵית בְּרֹב עֲצָתָיִךְ יַעַמְדוּ נָא וְיוֹשִׁיעֻךְ (הברו) הֹבְרֵי שָׁמַיִם הַחֹזִים בַּכּוֹכָבִים מוֹדִיעִם לֶחֳדָשִׁים מֵאֲשֶׁר יָבֹאוּ עָלָיִךְ:

[תהילים פרק-כ-ו]  נְרַנְּנָה בִּישׁוּעָתֶךָ וּבְשֵׁם אֱלֹהֵינוּ נִדְגֹּל יְמַלֵּא יְהוָה כָּל מִשְׁאֲלוֹתֶיךָ:

[תהילים פרק-נו-ט]  נֹדִי סָפַרְתָּה אָתָּה שִׂימָה דִמְעָתִי בְנֹאדֶךָ הֲלֹא בְּסִפְרָתֶךָ:

[תהילים פרק-קיט-קו]  נִשְׁבַּעְתִּי וָאֲקַיֵּמָה לִשְׁמֹר מִשְׁפְּטֵי צִדְקֶךָ:

[תהילים פרק-קיט-קז]  נַעֲנֵיתִי עַד מְאֹד יְהוָה חַיֵּנִי כִדְבָרֶךָ:

[תהילים פרק-קלב-יא]  נִשְׁבַּע יְהוָה לְדָוִד אֱמֶת לֹא יָשׁוּב מִמֶּנָּה מִפְּרִי בִטְנְךָ אָשִׁית לְכִסֵּא לָךְ:

[משלי פרק-ו-ב]  נוֹקַשְׁתָּ בְאִמְרֵי פִיךָ נִלְכַּדְתָּ בְּאִמְרֵי פִיךָ:

[משלי פרק-ו-כ]  נְצֹר בְּנִי מִצְוַת אָבִיךָ וְאַל תִּטֹּשׁ תּוֹרַת אִמֶּךָ:

[משלי פרק-כ-כא]  נַחֲלָה (מבחלת) מְבֹהֶלֶת בָּרִאשֹׁנָה וְאַחֲרִיתָהּ לֹא תְבֹרָךְ:

[איוב פרק-טו-כג]  נֹדֵד הוּא לַלֶּחֶם אַיֵּה יָדַע כִּי נָכוֹן בְּיָדוֹ יוֹם חֹשֶׁךְ:

[שיר השירים פרק-ז-יג]  נַשְׁכִּימָה לַכְּרָמִים נִרְאֶה אִם פָּרְחָה הַגֶּפֶן פִּתַּח הַסְּמָדַר הֵנֵצוּ הָרִמּוֹנִים שָׁם אֶתֵּן אֶת דֹּדַי לָךְ:


נ-ל


[ישעיה פרק-ז-ו]  נַעֲלֶה בִיהוּדָה וּנְקִיצֶנָּה וְנַבְקִעֶנָּה אֵלֵינוּ וְנַמְלִיךְ מֶלֶךְ בְּתוֹכָהּ אֵת בֶּן טָבְאַל:

[ישעיה פרק-כו-ט]  נַפְשִׁי אִוִּיתִיךָ בַּלַּיְלָה אַף רוּחִי בְקִרְבִּי אֲשַׁחֲרֶךָּ כִּי כַּאֲשֶׁר מִשְׁפָּטֶיךָ לָאָרֶץ צֶדֶק לָמְדוּ יֹשְׁבֵי תֵבֵל:

[תהילים פרק-פח-ה]  נֶחְשַׁבְתִּי עִם יוֹרְדֵי בוֹר הָיִיתִי כְּגֶבֶר אֵין אֱיָל:

[דניאל פרק-ג-א]  נְבוּכַדְנֶצַּר מַלְכָּא עֲבַד צְלֵם דִּי דְהַב רוּמֵהּ אַמִּין שִׁתִּין פְּתָיֵהּ אַמִּין שִׁת אֲקִימֵהּ בְּבִקְעַת דּוּרָא בִּמְדִינַת בָּבֶל:


נ-מ


[שמות פרק-טו-י]  נָשַׁפְתָּ בְרוּחֲךָ כִּסָּמוֹ יָם צָלֲלוּ כַּעוֹפֶרֶת בְּמַיִם אַדִּירִים:

[שמות פרק-לד-ז]  נֹצֵר חֶסֶד לָאֲלָפִים נֹשֵׂא עָוֹן וָפֶשַׁע וְחַטָּאָה וְנַקֵּה לֹא יְנַקֶּה פֹּקֵד עֲוֹן אָבוֹת עַל בָּנִים וְעַל בְּנֵי בָנִים עַל שִׁלֵּשִׁים וְעַל רִבֵּעִים:

[ויקרא פרק-ה-טו]  נֶפֶשׁ כִּי תִמְעֹל מַעַל וְחָטְאָה בִּשְׁגָגָה מִקָּדְשֵׁי יְהוָה וְהֵבִיא אֶת אֲשָׁמוֹ לַיהוָה אַיִל תָּמִים מִן הַצֹּאן בְּעֶרְכְּךָ כֶּסֶף שְׁקָלִים בְּשֶׁקֶל הַקֹּדֶשׁ לְאָשָׁם:

[ויקרא פרק-כב-ו]  נֶפֶשׁ אֲשֶׁר תִּגַּע בּוֹ וְטָמְאָה עַד הָעָרֶב וְלֹא יֹאכַל מִן הַקֳּדָשִׁים כִּי אִם רָחַץ בְּשָׂרוֹ בַּמָּיִם:

[במדבר פרק-ד-ב]  נָשֹׂא אֶת רֹאשׁ בְּנֵי קְהָת מִתּוֹךְ בְּנֵי לֵוִי לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם:

[במדבר פרק-ד-כב]  נָשֹׂא אֶת רֹאשׁ בְּנֵי גֵרְשׁוֹן גַּם הֵם לְבֵית אֲבֹתָם לְמִשְׁפְּחֹתָם:

[במדבר פרק-כד-ד]  נְאֻם שֹׁמֵעַ אִמְרֵי אֵל אֲשֶׁר מַחֲזֵה שַׁדַּי יֶחֱזֶה נֹפֵל וּגְלוּי עֵינָיִם:

[במדבר פרק-כד-טז]  נְאֻם שֹׁמֵעַ אִמְרֵי אֵל וְיֹדֵעַ דַּעַת עֶלְיוֹן מַחֲזֵה שַׁדַּי יֶחֱזֶה נֹפֵל וּגְלוּי עֵינָיִם:

[יהושע פרק-א-יד]  נְשֵׁיכֶם טַפְּכֶם וּמִקְנֵיכֶם יֵשְׁבוּ בָּאָרֶץ אֲשֶׁר נָתַן לָכֶם מֹשֶׁה בְּעֵבֶר הַיַּרְדֵּן וְאַתֶּם תַּעַבְרוּ חֲמֻשִׁים לִפְנֵי אֲחֵיכֶם כֹּל גִּבּוֹרֵי הַחַיִל וַעֲזַרְתֶּם אוֹתָם:

[ישעיה פרק-ג-יג]  נִצָּב לָרִיב יְהוָה וְעֹמֵד לָדִין עַמִּים:

[ישעיה פרק-יד-כד]  נִשְׁבַּע יְהוָה צְבָאוֹת לֵאמֹר אִם לֹא כַּאֲשֶׁר דִּמִּיתִי כֵּן הָיָתָה וְכַאֲשֶׁר יָעַצְתִּי הִיא תָקוּם:

[ישעיה פרק-כד-כ]  נוֹעַ תָּנוּעַ אֶרֶץ כַּשִּׁכּוֹר וְהִתְנוֹדְדָה כַּמְּלוּנָה וְכָבַד עָלֶיהָ פִּשְׁעָהּ וְנָפְלָה וְלֹא תֹסִיף קוּם:

[ישעיה פרק-מ-א]  נַחֲמוּ נַחֲמוּ עַמִּי יֹאמַר אֱלֹהֵיכֶם:

[ישעיה פרק-נט-י]  נְגַשְׁשָׁה כַעִוְרִים קִיר וּכְאֵין עֵינַיִם נְגַשֵּׁשָׁה כָּשַׁלְנוּ בַצָּהֳרַיִם כַּנֶּשֶׁף בָּאַשְׁמַנִּים כַּמֵּתִים:

[ירמיה פרק-י-יד]  נִבְעַר כָּל אָדָם מִדַּעַת הֹבִישׁ כָּל צוֹרֵף מִפָּסֶל כִּי שֶׁקֶר נִסְכּוֹ וְלֹא רוּחַ בָּם:

[ירמיה פרק-יב-ב]  נְטַעְתָּם גַּם שֹׁרָשׁוּ יֵלְכוּ גַּם עָשׂוּ פֶרִי קָרוֹב אַתָּה בְּפִיהֶם וְרָחוֹק מִכִּלְיוֹתֵיהֶם:

[ירמיה פרק-נא-יז]  נִבְעַר כָּל אָדָם מִדַּעַת הֹבִישׁ כָּל צֹרֵף מִפָּסֶל כִּי שֶׁקֶר נִסְכּוֹ וְלֹא רוּחַ בָּם:

[יחזקאל פרק-מב-ג]  נֶגֶד הָעֶשְׂרִים אֲשֶׁר לֶחָצֵר הַפְּנִימִי וְנֶגֶד רִצְפָה אֲשֶׁר לֶחָצֵר הַחִיצוֹנָה אַתִּיק אֶל פְּנֵי אַתִּיק בַּשְּׁלִשִׁים:

[הושע פרק-י-ז]  נִדְמֶה שֹׁמְרוֹן מַלְכָּהּ כְּקֶצֶף עַל פְּנֵי מָיִם:

[עמוס פרק-ח-ז]  נִשְׁבַּע יְהוָה בִּגְאוֹן יַעֲקֹב אִם אֶשְׁכַּח לָנֶצַח כָּל מַעֲשֵׂיהֶם:

[מיכה פרק-ב-ט]  נְשֵׁי עַמִּי תְּגָרְשׁוּן מִבֵּית תַּעֲנֻגֶיהָ מֵעַל עֹלָלֶיהָ תִּקְחוּ הֲדָרִי לְעוֹלָם:

[צפניה פרק-ב-יא]  נוֹרָא יְהוָה עֲלֵיהֶם כִּי רָזָה אֵת כָּל אֱלֹהֵי הָאָרֶץ וְיִשְׁתַּחֲווּ לוֹ אִישׁ מִמְּקוֹמוֹ כֹּל אִיֵּי הַגּוֹיִם:

[תהילים פרק-לז-כה]  נַעַר הָיִיתִי גַּם זָקַנְתִּי וְלֹא רָאִיתִי צַדִּיק נֶעֱזָב וְזַרְעוֹ מְבַקֶּשׁ לָחֶם:

[תהילים פרק-סה-ו]  נוֹרָאוֹת בְּצֶדֶק תַּעֲנֵנוּ אֱלֹהֵי יִשְׁעֵנוּ מִבְטָח כָּל קַצְוֵי אֶרֶץ וְיָם רְחֹקִים:

[תהילים פרק-סח-לו]  נוֹרָא אֱלֹהִים מִמִּקְדָּשֶׁיךָ אֵל יִשְׂרָאֵל הוּא נֹתֵן עֹז וְתַעֲצֻמוֹת לָעָם בָּרוּךְ אֱלֹהִים:

[תהילים פרק-צג-ג]  נָשְׂאוּ נְהָרוֹת יְהוָה נָשְׂאוּ נְהָרוֹת קוֹלָם יִשְׂאוּ נְהָרוֹת דָּכְיָם:

[תהילים פרק-קה-לב]  נָתַן גִּשְׁמֵיהֶם בָּרָד אֵשׁ לֶהָבוֹת בְּאַרְצָם:

[משלי פרק-יא-כב]  נֶזֶם זָהָב בְּאַף חֲזִיר אִשָּׁה יָפָה וְסָרַת טָעַם:

[משלי פרק-יט-כט]  נָכוֹנוּ לַלֵּצִים שְׁפָטִים וּמַהֲלֻמוֹת לְגֵו כְּסִילִים:

[משלי פרק-כח-טז]  נָגִיד חֲסַר תְּבוּנוֹת וְרַב מַעֲשַׁקּוֹת (שנאי) שֹׂנֵא בֶצַע יַאֲרִיךְ יָמִים:

[איוב פרק-ט-ח]  נֹטֶה שָׁמַיִם לְבַדּוֹ וְדוֹרֵךְ עַל בָּמֳתֵי יָם:

[שיר השירים פרק-א-י]  נָאווּ לְחָיַיִךְ בַּתֹּרִים צַוָּארֵךְ בַּחֲרוּזִים:

[שיר השירים פרק-ד-יד]  נֵרְדְּ וְכַרְכֹּם קָנֶה וְקִנָּמוֹן עִם כָּל עֲצֵי לְבוֹנָה מֹר וַאֲהָלוֹת עִם כָּל רָאשֵׁי בְשָׂמִים:

[איכה פרק-א-יד]  נִשְׂקַד עֹל פְּשָׁעַי בְּיָדוֹ יִשְׂתָּרְגוּ עָלוּ עַל צַוָּארִי הִכְשִׁיל כֹּחִי נְתָנַנִי אֲדֹנָי בִּידֵי לֹא אוּכַל קוּם:

[איכה פרק-ב-יד]  נְבִיאַיִךְ חָזוּ לָךְ שָׁוְא וְתָפֵל וְלֹא גִלּוּ עַל עֲוֹנֵךְ לְהָשִׁיב (שביתך) שְׁבוּתֵךְ וַיֶּחֱזוּ לָךְ מַשְׂאוֹת שָׁוְא וּמַדּוּחִים:

[איכה פרק-ג-מא]  נִשָּׂא לְבָבֵנוּ אֶל כַּפָּיִם אֶל אֵל בַּשָּׁמָיִם:

[איכה פרק-ד-יד]  נָעוּ עִוְרִים בַּחוּצוֹת נְגֹאֲלוּ בַּדָּם בְּלֹא יוּכְלוּ יִגְּעוּ בִּלְבֻשֵׁיהֶם:

[איכה פרק-ה-ב]  נַחֲלָתֵנוּ נֶהֶפְכָה לְזָרִים בָּתֵּינוּ לְנָכְרִים:

[אסתר פרק-ט-ב]  נִקְהֲלוּ הַיְּהוּדִים בְּעָרֵיהֶם בְּכָל מְדִינוֹת הַמֶּלֶךְ אֳחַשְׁוֵרוֹשׁ לִשְׁלֹחַ יָד בִּמְבַקְשֵׁי רָעָתָם וְאִישׁ לֹא עָמַד לִפְנֵיהֶם כִּי נָפַל פַּחְדָּם עַל כָּל הָעַמִּים:


נ-נ


[ויקרא פרק-יג-ט]  נֶגַע צָרַעַת כִּי תִהְיֶה בְּאָדָם וְהוּבָא אֶל הַכֹּהֵן:

[במדבר פרק-לב-לב]  נַחְנוּ נַעֲבֹר חֲלוּצִים לִפְנֵי יְהוָה אֶרֶץ כְּנָעַן וְאִתָּנוּ אֲחֻזַּת נַחֲלָתֵנוּ מֵעֵבֶר לַיַּרְדֵּן:

[דברים פרק-יח-טו]  נָבִיא מִקִּרְבְּךָ מֵאַחֶיךָ כָּמֹנִי יָקִים לְךָ יְהוָה אֱלֹהֶיךָ אֵלָיו תִּשְׁמָעוּן:

[ירמיה פרק-נ-ח]  נֻדוּ מִתּוֹךְ בָּבֶל וּמֵאֶרֶץ כַּשְׂדִּים (יצאו) צֵאוּ וִהְיוּ כְּעַתּוּדִים לִפְנֵי צֹאן:

[תהילים פרק-מו-ה]  נָהָר פְּלָגָיו יְשַׂמְּחוּ עִיר אֱלֹהִים קְדֹשׁ מִשְׁכְּנֵי עֶלְיוֹן:

[תהילים פרק-עז-כא]  נָחִיתָ כַצֹּאן עַמֶּךָ בְּיַד מֹשֶׁה וְאַהֲרֹן:

[תהילים פרק-עח-יב]  נֶגֶד אֲבוֹתָם עָשָׂה פֶלֶא בְּאֶרֶץ מִצְרַיִם שְׂדֵה צֹעַן:

[משלי פרק-ז-יז]  נַפְתִּי מִשְׁכָּבִי מֹר אֲהָלִים וְקִנָּמוֹן:

[משלי פרק-כ-כז]  נֵר יְהוָה נִשְׁמַת אָדָם חֹפֵשׂ כָּל חַדְרֵי בָטֶן:

[שיר השירים פרק-ד-יא]  נֹפֶת תִּטֹּפְנָה שִׂפְתוֹתַיִךְ כַּלָּה דְּבַשׁ וְחָלָב תַּחַת לְשׁוֹנֵךְ וְרֵיחַ שַׂלְמֹתַיִךְ כְּרֵיחַ לְבָנוֹן:

[דברי הימים א פרק-יב-ב]  נֹשְׁקֵי קֶשֶׁת מַיְמִינִים וּמַשְׂמִאלִים בָּאֲבָנִים וּבַחִצִּים בַּקָּשֶׁת מֵאֲחֵי שָׁאוּל מִבִּנְיָמִן:


נ-ס


[שופטים פרק-א-לג]  נַפְתָּלִי לֹא הוֹרִישׁ אֶת יֹשְׁבֵי בֵית שֶׁמֶשׁ וְאֶת יֹשְׁבֵי בֵית עֲנָת וַיֵּשֶׁב בְּקֶרֶב הַכְּנַעֲנִי יֹשְׁבֵי הָאָרֶץ וְיֹשְׁבֵי בֵית שֶׁמֶשׁ וּבֵית עֲנָת הָיוּ לָהֶם לָמַס:


נ-פ


[תהילים פרק-עו-ה]  נָאוֹר אַתָּה אַדִּיר מֵהַרְרֵי טָרֶף:

[תהילים פרק-קו-כב]  נִפְלָאוֹת בְּאֶרֶץ חָם נוֹרָאוֹת עַל יַם סוּף:


נ-צ


[שמות פרק-טו-יב]  נָטִיתָ יְמִינְךָ תִּבְלָעֵמוֹ אָרֶץ:

[נחום פרק-ג-יח]  נָמוּ רֹעֶיךָ מֶלֶךְ אַשּׁוּר יִשְׁכְּנוּ אַדִּירֶיךָ נָפֹשׁוּ עַמְּךָ עַל הֶהָרִים וְאֵין מְקַבֵּץ:

[תהילים פרק-כה-יג]  נַפְשׁוֹ בְּטוֹב תָּלִין וְזַרְעוֹ יִירַשׁ אָרֶץ:

[תהילים פרק-עט-ב]  נָתְנוּ אֶת נִבְלַת עֲבָדֶיךָ מַאֲכָל לְעוֹף הַשָּׁמָיִם בְּשַׂר חֲסִידֶיךָ לְחַיְתוֹ אָרֶץ:

[משלי פרק-לא-כג]  נוֹדָע בַּשְּׁעָרִים בַּעְלָהּ בְּשִׁבְתּוֹ עִם זִקְנֵי אָרֶץ:


נ-ק


[תהילים פרק-קי-ד]  נִשְׁבַּע יְהוָה וְלֹא יִנָּחֵם אַתָּה כֹהֵן לְעוֹלָם עַל דִּבְרָתִי מַלְכִּי צֶדֶק:

[משלי פרק-כז-ז]  נֶפֶשׁ שְׂבֵעָה תָּבוּס נֹפֶת וְנֶפֶשׁ רְעֵבָה כָּל מַר מָתוֹק:


נ-ר


[בראשית פרק-מט-כא]  נַפְתָּלִי אַיָּלָה שְׁלֻחָה הַנֹּתֵן אִמְרֵי שָׁפֶר:

[ישעיה פרק-כד-יב]  נִשְׁאַר בָּעִיר שַׁמָּה וּשְׁאִיָּה יֻכַּת שָׁעַר:

[ירמיה פרק-מח-ו]  נֻסוּ מַלְּטוּ נַפְשְׁכֶם וְתִהְיֶינָה כַּעֲרוֹעֵר בַּמִּדְבָּר:

[חבקוק פרק-ג-יד]  נָקַבְתָּ בְמַטָּיו רֹאשׁ (פרזו) פְּרָזָיו יִסְעֲרוּ לַהֲפִיצֵנִי עֲלִיצֻתָם כְּמוֹ לֶאֱכֹל עָנִי בַּמִּסְתָּר:

[תהילים פרק-טו-ד]  נִבְזֶה בְּעֵינָיו נִמְאָס וְאֶת יִרְאֵי יְהוָה יְכַבֵּד נִשְׁבַּע לְהָרַע וְלֹא יָמִר:

[תהילים פרק-לט-ג]  נֶאֱלַמְתִּי דוּמִיָּה הֶחֱשֵׁיתִי מִטּוֹב וּכְאֵבִי נֶעְכָּר:

[תהילים פרק-קל-ו]  נַפְשִׁי לַאדֹנָי מִשֹּׁמְרִים לַבֹּקֶר שֹׁמְרִים לַבֹּקֶר:

[משלי פרק-א-יב]  נִבְלָעֵם כִּשְׁאוֹל חַיִּים וּתְמִימִים כְּיוֹרְדֵי בוֹר:

[משלי פרק-כה-יד]  נְשִׂיאִים וְרוּחַ וְגֶשֶׁם אָיִן אִישׁ מִתְהַלֵּל בְּמַתַּת שָׁקֶר:


נ-ש


[ישעיה פרק-לג-ח]  נָשַׁמּוּ מְסִלּוֹת שָׁבַת עֹבֵר אֹרַח הֵפֵר בְּרִית מָאַס עָרִים לֹא חָשַׁב אֱנוֹשׁ:


נ-ת


[ירמיה פרק-טו-טז]  נִמְצְאוּ דְבָרֶיךָ וָאֹכְלֵם וַיְהִי (דבריך) דְבָרְךָ לִי לְשָׂשׂוֹן וּלְשִׂמְחַת לְבָבִי כִּי נִקְרָא שִׁמְךָ עָלַי יְהוָה אֱלֹהֵי צְבָאוֹת:

[תהילים פרק-לט-י]  נֶאֱלַמְתִּי לֹא אֶפְתַּח פִּי כִּי אַתָּה עָשִׂיתָ:

[משלי פרק-יד-יח]  נָחֲלוּ פְתָאיִם אִוֶּלֶת וַעֲרוּמִים יַכְתִּרוּ דָעַת:

[משלי פרק-כה-יב]  נֶזֶם זָהָב וַחֲלִי כָתֶם מוֹכִיחַ חָכָם עַל אֹזֶן שֹׁמָעַת:

[משלי פרק-כח-כז]  נוֹתֵן לָרָשׁ אֵין מַחְסוֹר וּמַעְלִים עֵינָיו רַב מְאֵרוֹת:

[איכה פרק-ג-מב]  נַחְנוּ פָשַׁעְנוּ וּמָרִינוּ אַתָּה לֹא סָלָחְתָּ:

[דברי הימים א פרק-א-ד]  נֹחַ שֵׁם חָם וָיָפֶת:
