פרשת וישלח


{ד} וַיִּשְׁלַ֨ח יַעֲקֹ֤ב מַלְאָכִים֙ לְפָנָ֔יו אֶל-עֵשָׂ֖ו אָחִ֑יו אַ֥רְצָה שֵׂעִ֖יר שְׂדֵ֥ה אֱדֽוֹם:  מ  וַיִּשְׁלַ֨ח יַעֲקֹ֤ב מַלְאָכִים֙ לְפָנָ֔יו אֶל-עֵשָׂ֖ו אָחִ֑יו אַ֥רְצָה שֵׂעִ֖יר שְׂדֵ֥ה אֱדֽוֹם:  ת  וּשְׁלַח יַעֲקֹב אִזְגַּדִּין קֳדָמוֹהִי לְוַת עֵשָׂו אֲחוּהִי לְאַרְעָא דְשֵׂעִיר לַחֲקַל דְאֱדוֹם:

{ה} וַיְצַ֤ו אֹתָם֙ לֵאמֹ֔ר כֹּ֣ה תֹאמְר֔וּן לַֽאדֹנִ֖י לְעֵשָׂ֑ו כֹּ֤ה אָמַר֙ עַבְדְּךָ֣ יַעֲקֹ֔ב עִם-לָבָ֣ן גַּ֔רְתִּי וָאֵחַ֖ר עַד-עָֽתָּה:  מ  וַיְצַ֤ו אֹתָם֙ לֵאמֹ֔ר כֹּ֣ה תֹאמְר֔וּן לַֽאדֹנִ֖י לְעֵשָׂ֑ו כֹּ֤ה אָמַר֙ עַבְדְּךָ֣ יַעֲקֹ֔ב עִם-לָבָ֣ן גַּ֔רְתִּי וָאֵחַ֖ר עַד-עָֽתָּה:  ת  וּפַקִּיד יָתְהוֹן לְמֵימָר כְּדֵין תֵּימְרוּן לְרִבּוֹנִי לְעֵשָׂו כִּדְנַן אֲמַר עַבְדָּךְ יַעֲקֹב עִם לָבָן דָּרִית וְאוֹחָרִית עַד כְּעָן:

{ו} וַֽיְהִי-לִי֙ שׁ֣וֹר וַחֲמ֔וֹר צֹ֖אן וְעֶ֣בֶד וְשִׁפְחָ֑ה וָֽאֶשְׁלְחָה֙ לְהַגִּ֣יד לַֽאדֹנִ֔י לִמְצֹא-חֵ֖ן בְּעֵינֶֽיךָ:  מ  וַֽיְהִי-לִי֙ שׁ֣וֹר וַחֲמ֔וֹר צֹ֖אן וְעֶ֣בֶד וְשִׁפְחָ֑ה וָֽאֶשְׁלְחָה֙ לְהַגִּ֣יד לַֽאדֹנִ֔י לִמְצֹא-חֵ֖ן בְּעֵינֶֽיךָ:  ת  וַהֲוָה לִי תּוֹרִין וַחֲמָרִין עָאן וְעַבְדִּין וְאַמְהָן וּשְׁלָחִית לְחַוָּאָה לְרִבּוֹנִי לְאַשְׁכָּחָא רַחֲמִין בְּעֵינָךְ:

{ז} וַיָּשֻׁ֙בוּ֙ הַמַּלְאָכִ֔ים אֶֽל-יַעֲקֹ֖ב לֵאמֹ֑ר בָּ֤אנוּ אֶל-אָחִ֙יךָ֙ אֶל-עֵשָׂ֔ו וְגַם֙ הֹלֵ֣ךְ לִקְרָֽאתְךָ֔ וְאַרְבַּע-מֵא֥וֹת אִ֖ישׁ עִמּֽוֹ:  מ  וַיָּשֻׁ֙בוּ֙ הַמַּלְאָכִ֔ים אֶֽל-יַעֲקֹ֖ב לֵאמֹ֑ר בָּ֤אנוּ אֶל-אָחִ֙יךָ֙ אֶל-עֵשָׂ֔ו וְגַם֙ הֹלֵ֣ךְ לִקְרָֽאתְךָ֔ וְאַרְבַּע-מֵא֥וֹת אִ֖ישׁ עִמּֽוֹ:  ת  וְתָבוּ אִזְגַּדַּיָּא לְוַת יַעֲקֹב לְמֵימָר אֲתֵינָא לְוַת אָחוּךְ לְוַת עֵשָׂו וְאַף אָתֵי לְקַדָּמוּתָךְ וְאַרְבַּע מְאָה גַּבְרָא עִמֵּהּ:

{ח} וַיִּירָ֧א יַעֲקֹ֛ב מְאֹ֖ד וַיֵּ֣צֶר ל֑וֹ וַיַּ֜חַץ אֶת-הָעָ֣ם אֲשֶׁר-אִתּ֗וֹ וְאֶת-הַצֹּ֧אן וְאֶת-הַבָּקָ֛ר וְהַגְּמַלִּ֖ים לִשְׁנֵ֥י מַחֲנֽוֹת:  מ  וַיִּירָ֧א יַעֲקֹ֛ב מְאֹ֖ד וַיֵּ֣צֶר ל֑וֹ וַיַּ֜חַץ אֶת-הָעָ֣ם אֲשֶׁר-אִתּ֗וֹ וְאֶת-הַצֹּ֧אן וְאֶת-הַבָּקָ֛ר וְהַגְּמַלִּ֖ים לִשְׁנֵ֥י מַחֲנֽוֹת:  ת  וּדְחִיל יַעֲקֹב לַחֲדָא וַעֲקַת לֵהּ וּפַלִּיג יָת עַמָּא דִּי עִמֵּהּ וְיָת עָנָא וְיָת תּוֹרֵי וְגַמְלַיָּא לִ:תרֵין מַשִּׁירְיָן:

{ט} וַיֹּ֕אמֶר אִם-יָב֥וֹא עֵשָׂ֛ו אֶל-הַמַּחֲנֶ֥ה הָאַחַ֖ת וְהִכָּ֑הוּ וְהָיָ֛ה הַמַּחֲנֶ֥ה הַנִּשְׁאָ֖ר לִפְלֵיטָֽה:  מ  וַיֹּ֕אמֶר אִם-יָב֥וֹא עֵשָׂ֛ו אֶל-הַמַּחֲנֶ֥ה הָאַחַ֖ת וְהִכָּ֑הוּ וְהָיָ֛ה הַמַּחֲנֶ֥ה הַנִּשְׁאָ֖ר לִפְלֵיטָֽה:  ת  וַאֲמַר אִם יֵיתֵי עֵשָׂו לְמַשְׁרִיתָא חֲדָא וְיִמְחִנֵּהּ וִיהֵי (נ''י וְיִמְחֵנַּהּ וּתְהִי) מַשְׁרִיתָא דְאִשְׁתָּאַר לְשֵׁיזָבָא:

{י} וַיֹּאמֶר֮ יַעֲקֹב֒ אֱלֹהֵי֙ אָבִ֣י אַבְרָהָ֔ם וֵאלֹהֵ֖י אָבִ֣י יִצְחָ֑ק יְהוָ֞ה הָאֹמֵ֣ר אֵלַ֗י שׁ֧וּב לְאַרְצְךָ֛ וּלְמוֹלַדְתְּךָ֖ וְאֵיטִ֥יבָה עִמָּֽךְ:  מ  וַיֹּאמֶר֮ יַעֲקֹב֒ אֱלֹהֵי֙ אָבִ֣י אַבְרָהָ֔ם וֵאלֹהֵ֖י אָבִ֣י יִצְחָ֑ק יְהוָ֞ה הָאֹמֵ֣ר אֵלַ֗י שׁ֧וּב לְאַרְצְךָ֛ וּלְמוֹלַדְתְּךָ֖ וְאֵיטִ֥יבָה עִמָּֽךְ:  ת  וַאֲמַר יַעֲקֹב אֱלָהֵהּ דְּאַבָּא אַבְרָהָם וֵאֱלָהֵהּ דְּאַבָּא יִצְחָק יְיָ דִּי אֲמַר לִי תּוּב לְאַרְעָךְ וּלְיַלָּדוּתָךְ וְאוֹטֵיב עִמָּךְ:

{יא} קָטֹ֜נְתִּי מִכֹּ֤ל הַחֲסָדִים֙ וּמִכָּל-הָ֣אֱמֶ֔ת אֲשֶׁ֥ר עָשִׂ֖יתָ אֶת-עַבְדֶּ֑ךָ כִּ֣י בְמַקְלִ֗י עָבַ֙רְתִּי֙ אֶת-הַיַּרְדֵּ֣ן הַזֶּ֔ה וְעַתָּ֥ה הָיִ֖יתִי לִשְׁנֵ֥י מַחֲנֽוֹת:  מ  קָטֹ֜נְתִּי מִכֹּ֤ל הַחֲסָדִים֙ וּמִכָּל-הָ֣אֱמֶ֔ת אֲשֶׁ֥ר עָשִׂ֖יתָ אֶת-עַבְדֶּ֑ךָ כִּ֣י בְמַקְלִ֗י עָבַ֙רְתִּי֙ אֶת-הַיַּרְדֵּ֣ן הַזֶּ֔ה וְעַתָּ֥ה הָיִ֖יתִי לִשְׁנֵ֥י מַחֲנֽוֹת:  ת  זְעִירָן זָכוּתַי מִכֹּל חִסְדִּן וּמִכָּל טַבְוָן דִּי עֲבַדְתָּ עִם עַבְדָּךְ אֲרֵי יְחִידִי עֲבָרִית יָת יַרְדְּנָא הָדֵין וּכְעַן הֲוֵיתִי לִתְרֵין (נ''י לְתַרְתֵּין) מַשִּׁרְיָן:

{יב} הַצִּילֵ֥נִי נָ֛א מִיַּ֥ד אָחִ֖י מִיַּ֣ד עֵשָׂ֑ו כִּֽי-יָרֵ֤א אָנֹכִי֙ אֹת֔וֹ פֶּן-יָב֣וֹא וְהִכַּ֔נִי אֵ֖ם עַל-בָּנִֽים:  מ  הַצִּילֵ֥נִי נָ֛א מִיַּ֥ד אָחִ֖י מִיַּ֣ד עֵשָׂ֑ו כִּֽי-יָרֵ֤א אָנֹכִי֙ אֹת֔וֹ פֶּן-יָב֣וֹא וְהִכַּ֔נִי אֵ֖ם עַל-בָּנִֽים:  ת  שֵׁזַבְנִי כְעַן מִידָא דְאָחִי מִידָא דְעֵשָׁו אֲרֵי דַחֵל אֲנָא מִנֵּהּ דִּילְמָא יֵיתֵי וְיִמְחִנַּנִי אִמָּא עַל בְּנַיָּא:

{יג} וְאַתָּ֣ה אָמַ֔רְתָּ הֵיטֵ֥ב אֵיטִ֖יב עִמָּ֑ךְ וְשַׂמְתִּ֤י אֶֽת-זַרְעֲךָ֙ כְּח֣וֹל הַיָּ֔ם אֲשֶׁ֥ר לֹא-יִסָּפֵ֖ר מֵרֹֽב:  מ  וְאַתָּ֣ה אָמַ֔רְתָּ הֵיטֵ֥ב אֵיטִ֖יב עִמָּ֑ךְ וְשַׂמְתִּ֤י אֶֽת-זַרְעֲךָ֙ כְּח֣וֹל הַיָּ֔ם אֲשֶׁ֥ר לֹא-יִסָּפֵ֖ר מֵרֹֽב:  ת  וְאַתְּ אֲמַרְתָּ אוֹטָבָא אוֹטִיב עִמָּךְ וֶאֱשַׁוֵּי יָת בְּנָיךְ סַגִּיאִין כְּחַלָּא דְיַמָּא דִּי לָא יִתִּמְנוּן מִסַּגִּי:

{יד}  שני  וַיָּ֥לֶן שָׁ֖ם בַּלַּ֣יְלָה הַה֑וּא וַיִּקַּ֞ח מִן-הַבָּ֧א בְיָד֛וֹ מִנְחָ֖ה לְעֵשָׂ֥ו אָחִֽיו:  מ  וַיָּ֥לֶן שָׁ֖ם בַּלַּ֣יְלָה הַה֑וּא וַיִּקַּ֞ח מִן-הַבָּ֧א בְיָד֛וֹ מִנְחָ֖ה לְעֵשָׂ֥ו אָחִֽיו:  ת  וּבָת תַּמָּן בְּלֵילְיָא הַהוּא וּנְסִיב מִן דְּאַיְתִי בִּידֵהּ תִּקְרֻבְתָּא לְעֵשָׂו אֲחוּהִי:

{טו} עִזִּ֣ים מָאתַ֔יִם וּתְיָשִׁ֖ים עֶשְׂרִ֑ים רְחֵלִ֥ים מָאתַ֖יִם וְאֵילִ֥ים עֶשְׂרִֽים:  מ  עִזִּ֣ים מָאתַ֔יִם וּתְיָשִׁ֖ים עֶשְׂרִ֑ים רְחֵלִ֥ים מָאתַ֖יִם וְאֵילִ֥ים עֶשְׂרִֽים:  ת  עִזֵּי מָאתָן וּתְיָשַׁיָא עֶסְרִין רְחֵלִין מָאתָן וְדִכְרִין עֶסְרִין:

{טז} גְּמַלִּ֧ים מֵינִיק֛וֹת וּבְנֵיהֶ֖ם שְׁלֹשִׁ֑ים פָּר֤וֹת אַרְבָּעִים֙ וּפָרִ֣ים עֲשָׂרָ֔ה אֲתֹנֹ֣ת עֶשְׂרִ֔ים וַעְיָרִ֖ם עֲשָׂרָֽה:  מ  גְּמַלִּ֧ים מֵינִיק֛וֹת וּבְנֵיהֶ֖ם שְׁלֹשִׁ֑ים פָּר֤וֹת אַרְבָּעִים֙ וּפָרִ֣ים עֲשָׂרָ֔ה אֲתֹנֹ֣ת עֶשְׂרִ֔ים וַעְיָרִ֖ם עֲשָׂרָֽה:  ת  גַּמְלֵי מֶנְקָתָא וּבְנֵיהוֹן תְּלָתִין תּוֹרְתָא אַרְבְּעִין וְתוֹרֵי עַסְרָא אַתְנָן עֶסְרִין וְעִירֵי עַסְרָא:

{יז} וַיִּתֵּן֙ בְּיַד-עֲבָדָ֔יו עֵ֥דֶר עֵ֖דֶר לְבַדּ֑וֹ וַ֤יֹּאמֶר אֶל-עֲבָדָיו֙ עִבְר֣וּ לְפָנַ֔י וְרֶ֣וַח תָּשִׂ֔ימוּ בֵּ֥ין עֵ֖דֶר וּבֵ֥ין עֵֽדֶר:  מ  וַיִּתֵּן֙ בְּיַד-עֲבָדָ֔יו עֵ֥דֶר עֵ֖דֶר לְבַדּ֑וֹ וַ֤יֹּאמֶר אֶל-עֲבָדָיו֙ עִבְר֣וּ לְפָנַ֔י וְרֶ֣וַח תָּשִׂ֔ימוּ בֵּ֥ין עֵ֖דֶר וּבֵ֥ין עֵֽדֶר:  ת  וִיהַב בְּיַד עַבְדּוֹהִי עֶדְרָא עֶדְרָא בִּלְחוֹדוֹהִי וַאֲמַר לְעַבְדּוֹהִי עִבָּרוּ קֳדָמַי וּרְוָחָא תְּשַׁווּן בֵּין עֶדְרָא וּבֵין עֶדְרָא:

{יח} וַיְצַ֥ו אֶת-הָרִאשׁ֖וֹן לֵאמֹ֑ר כִּ֣י יִֽפְגָּשְׁךָ֞ עֵשָׂ֣ו אָחִ֗י וּשְׁאֵֽלְךָ֙ לֵאמֹ֔ר לְמִי-אַ֙תָּה֙ וְאָ֣נָה תֵלֵ֔ךְ וּלְמִ֖י אֵ֥לֶּה לְפָנֶֽיךָ:  מ  וַיְצַ֥ו אֶת-הָרִאשׁ֖וֹן לֵאמֹ֑ר כִּ֣י יִֽפְגָּשְׁךָ֞ עֵשָׂ֣ו אָחִ֗י וּשְׁאֵֽלְךָ֙ לֵאמֹ֔ר לְמִי-אַ֙תָּה֙ וְאָ֣נָה תֵלֵ֔ךְ וּלְמִ֖י אֵ֥לֶּה לְפָנֶֽיךָ:  ת  וּפַקִד יָת קַדְמָאָה לְמֵימָר אֲרֵי יְעַרְעִנָּךְ עֵשָׂו אָחִי וּשְׁאֵלִנָּךְ לְמֵימָר לְמָן אַתְּ וּלְאָן אַתְּ אָזֵל וּלְמָן אִלֵּין דִּקֳדָמָךְ:

{יט} וְאָֽמַרְתָּ֙ לְעַבְדְּךָ֣ לְיַעֲקֹ֔ב מִנְחָ֥ה הִוא֙ שְׁלוּחָ֔ה לַֽאדֹנִ֖י לְעֵשָׂ֑ו וְהִנֵּ֥ה גַם-ה֖וּא אַחֲרֵֽינוּ:  מ  וְאָֽמַרְתָּ֙ לְעַבְדְּךָ֣ לְיַעֲקֹ֔ב מִנְחָ֥ה הִוא֙ שְׁלוּחָ֔ה לַֽאדֹנִ֖י לְעֵשָׂ֑ו וְהִנֵּ֥ה גַם-ה֖וּא אַחֲרֵֽינוּ:  ת  וְתֵימַר לְעַבְדָךְ לְיַעֲקֹב תִּקְרֻבְתָּא הִיא דִמְשַׁלְּחָא לְרִבּוֹנִי לְעֵשָׂו וְהָא אַף הוּא אָתֵי בַתְרָנָא:

{כ} וַיְצַ֞ו גַּ֣ם אֶת-הַשֵּׁנִ֗י גַּ֚ם אֶת-הַשְּׁלִישִׁ֔י גַּ֚ם אֶת-כָּל-הַהֹ֣לְכִ֔ים אַחֲרֵ֥י הָעֲדָרִ֖ים לֵאמֹ֑ר כַּדָּבָ֤ר הַזֶּה֙ תְּדַבְּר֣וּן אֶל-עֵשָׂ֔ו בְּמֹצַאֲכֶ֖ם אֹתֽוֹ:  מ  וַיְצַ֞ו גַּ֣ם אֶת-הַשֵּׁנִ֗י גַּ֚ם אֶת-הַשְּׁלִישִׁ֔י גַּ֚ם אֶת-כָּל-הַהֹ֣לְכִ֔ים אַחֲרֵ֥י הָעֲדָרִ֖ים לֵאמֹ֑ר כַּדָּבָ֤ר הַזֶּה֙ תְּדַבְּר֣וּן אֶל-עֵשָׂ֔ו בְּמֹצַאֲכֶ֖ם אֹתֽוֹ:  ת  וּפַקִּיד אַף יָת תִּנְיָנָא אַף יָת תְּלִיתָאָה אַף יָת כָּל דְּאָזְלִין בָּתַר עֲדָרַיָּא לְמֵימָר כְּפִתְגָּמָא הָדֵין תְּמַלְּלוּן עִם עֵשָׂו כַּד תַּשְׁכְּחוּן יָתֵהּ:

{כא} וַאֲמַרְתֶּ֕ם גַּ֗ם הִנֵּ֛ה עַבְדְּךָ֥ יַעֲקֹ֖ב אַחֲרֵ֑ינוּ כִּֽי-אָמַ֞ר אֲכַפְּרָ֣ה פָנָ֗יו בַּמִּנְחָה֙ הַהֹלֶ֣כֶת לְפָנָ֔י וְאַחֲרֵי-כֵן֙ אֶרְאֶ֣ה פָנָ֔יו אוּלַ֖י יִשָּׂ֥א פָנָֽי:  מ  וַאֲמַרְתֶּ֕ם גַּ֗ם הִנֵּ֛ה עַבְדְּךָ֥ יַעֲקֹ֖ב אַחֲרֵ֑ינוּ כִּֽי-אָמַ֞ר אֲכַפְּרָ֣ה פָנָ֗יו בַּמִּנְחָה֙ הַהֹלֶ֣כֶת לְפָנָ֔י וְאַחֲרֵי-כֵן֙ אֶרְאֶ֣ה פָנָ֔יו אוּלַ֖י יִשָּׂ֥א פָנָֽי:  ת  וְתֵימְרוּן אַף הָא עַבְדָּךְ יַעֲקֹב אָתֵי בַתְרָנָא אֲרֵי אֲמַר אַנְחִנֵּהּ לְרוּגְזֵהּ בְּתִקְרֻבְתָּא דְּאָזְלַת קֳדָמַי וּבָתַר כֵּן אֶחֱזֵי אַפּוֹהִי מָאִים יִסַּב אַפָּי:

{כב} וַתַּעֲבֹ֥ר הַמִּנְחָ֖ה עַל-פָּנָ֑יו וְה֛וּא לָ֥ן בַּלַּֽיְלָה-הַה֖וּא בַּֽמַּחֲנֶֽה:  מ  וַתַּעֲבֹ֥ר הַמִּנְחָ֖ה עַל-פָּנָ֑יו וְה֛וּא לָ֥ן בַּלַּֽיְלָה-הַה֖וּא בַּֽמַּחֲנֶֽה:  ת  וַעֲבָרַת תִּקְרֻבְתָּא עַל אַפּוֹהִי וְהוּא בָת בְּלֵילְיָא הַהוּא בְּמַשְׁרִיתָא:

{כג} וַיָּ֣קָם | בַּלַּ֣יְלָה ה֗וּא וַיִּקַּ֞ח אֶת-שְׁתֵּ֤י נָשָׁיו֙ וְאֶת-שְׁתֵּ֣י שִׁפְחֹתָ֔יו וְאֶת-אַחַ֥ד עָשָׂ֖ר יְלָדָ֑יו וַֽיַּעֲבֹ֔ר אֵ֖ת מַעֲבַ֥ר יַבֹּֽק:  מ  וַיָּ֣קָם | בַּלַּ֣יְלָה ה֗וּא וַיִּקַּ֞ח אֶת-שְׁתֵּ֤י נָשָׁיו֙ וְאֶת-שְׁתֵּ֣י שִׁפְחֹתָ֔יו וְאֶת-אַחַ֥ד עָשָׂ֖ר יְלָדָ֑יו וַֽיַּעֲבֹ֔ר אֵ֖ת מַעֲבַ֥ר יַבֹּֽק:  ת  וְקָם בְּלֵילְיָא הוּא וּדְבַר יָת תַּרְתֵּין נְשׁוֹהִי וְיָת תַּרְתֵּין לְחֵינָתֵהּ וְיָת חַד עֲסַר בְּנוֹהִי וַעֲבַר יָת מַעֲבַר יוּבְקָא:

{כד} וַיִּקָּחֵ֔ם וַיַּֽעֲבִרֵ֖ם אֶת-הַנָּ֑חַל וַֽיַּעֲבֵ֖ר אֶת-אֲשֶׁר-לוֹ:  מ  וַיִּקָּחֵ֔ם וַיַּֽעֲבִרֵ֖ם אֶת-הַנָּ֑חַל וַֽיַּעֲבֵ֖ר אֶת-אֲשֶׁר-לוֹ:  ת  וּדְבָרִנּוּן וְעַבָּרִנּוּן יָת נַחְלָא וְאַעְבַּר יָת דִּילֵהּ:

{כה} וַיִּוָּתֵ֥ר יַעֲקֹ֖ב לְבַדּ֑וֹ וַיֵּאָבֵ֥ק אִישׁ֙ עִמּ֔וֹ עַ֖ד עֲל֥וֹת הַשָּֽׁחַר:  מ  וַיִּוָּתֵ֥ר יַעֲקֹ֖ב לְבַדּ֑וֹ וַיֵּאָבֵ֥ק אִישׁ֙ עִמּ֔וֹ עַ֖ד עֲל֥וֹת הַשָּֽׁחַר:  ת  וְאִשְׁתָּאַר יַעֲקֹב בִּלְחוֹדוֹהִי וְאִשְׁתַּדֵּל גַּבְרָא עִמֵּהּ עַד דִּסְלֵק צַפְרָא:

{כו} וַיַּ֗רְא כִּ֣י לֹ֤א יָכֹל֙ ל֔וֹ וַיִּגַּ֖ע בְּכַף-יְרֵכ֑וֹ וַתֵּ֙קַע֙ כַּף-יֶ֣רֶךְ יַעֲקֹ֔ב בְּהֵֽאָבְק֖וֹ עִמּֽוֹ:  מ  וַיַּ֗רְא כִּ֣י לֹ֤א יָכֹל֙ ל֔וֹ וַיִּגַּ֖ע בְּכַף-יְרֵכ֑וֹ וַתֵּ֙קַע֙ כַּף-יֶ֣רֶךְ יַעֲקֹ֔ב בְּהֵֽאָבְק֖וֹ עִמּֽוֹ:  ת  וַחֲזָא אֲרֵי לָא יָכִיל לֵהּ וּקְרֵב בִּפְתֵי יַרְכֵּהּ וְזָע פְּתֵי יַרְכָּא דְיַעֲקֹב בְּאִשְׁתַּדָּלוּתֵהּ עִמֵּהּ:

{כז} וַיֹּ֣אמֶר שַׁלְּחֵ֔נִי כִּ֥י עָלָ֖ה הַשָּׁ֑חַר וַיֹּ֙אמֶר֙ לֹ֣א אֲשַֽׁלֵּחֲךָ֔ כִּ֖י אִם-בֵּרַכְתָּֽנִי:  מ  וַיֹּ֣אמֶר שַׁלְּחֵ֔נִי כִּ֥י עָלָ֖ה הַשָּׁ֑חַר וַיֹּ֙אמֶר֙ לֹ֣א אֲשַֽׁלֵּחֲךָ֔ כִּ֖י אִם-בֵּרַכְתָּֽנִי:  ת  וַאֲמַר שַׁלְּחַנִּי אֲרֵי סְלִיק צַפְרָא וַאֲמַר לָא אֲשַׁלְחִנָךְ אֶלָּהֵין בֵּרַכְתָּנִי:

{כח} וַיֹּ֥אמֶר אֵלָ֖יו מַה-שְּׁמֶ֑ךָ וַיֹּ֖אמֶר יַעֲקֹֽב:  מ  וַיֹּ֥אמֶר אֵלָ֖יו מַה-שְּׁמֶ֑ךָ וַיֹּ֖אמֶר יַעֲקֹֽב:  ת  וַאֲמַר לֵהּ מָה שְׁמָךְ וַאֲמַר יַעֲקֹב:

{כט} וַיֹּ֗אמֶר לֹ֤א יַעֲקֹב֙ יֵאָמֵ֥ר עוֹד֙ שִׁמְךָ֔ כִּ֖י אִם-יִשְׂרָאֵ֑ל כִּֽי-שָׂרִ֧יתָ עִם-אֱלֹהִ֛ים וְעִם-אֲנָשִׁ֖ים וַתּוּכָֽל:  מ  וַיֹּ֗אמֶר לֹ֤א יַעֲקֹב֙ יֵאָמֵ֥ר עוֹד֙ שִׁמְךָ֔ כִּ֖י אִם-יִשְׂרָאֵ֑ל כִּֽי-שָׂרִ֧יתָ עִם-אֱלֹהִ֛ים וְעִם-אֲנָשִׁ֖ים וַתּוּכָֽל:  ת  וַאֲמַר לָא יַעֲקֹב יִתְאֲמַר עוֹד שְׁמָךְ אֶלָּהֵין יִשְׂרָאֵל אֲרֵי רַבְרְבַת קֳדָם יְיָ וְעִם גּוּבְרַיָּא וִיכָלְתָּא:

{ל} וַיִּשְׁאַ֣ל יַעֲקֹ֗ב וַיֹּ֙אמֶר֙ הַגִּֽידָה-נָּ֣א שְׁמֶ֔ךָ וַיֹּ֕אמֶר לָ֥מָּה זֶּ֖ה תִּשְׁאַ֣ל לִשְׁמִ֑י וַיְבָ֥רֶךְ אֹת֖וֹ שָֽׁם:  מ  וַיִּשְׁאַ֣ל יַעֲקֹ֗ב וַיֹּ֙אמֶר֙ הַגִּֽידָה-נָּ֣א שְׁמֶ֔ךָ וַיֹּ֕אמֶר לָ֥מָּה זֶּ֖ה תִּשְׁאַ֣ל לִשְׁמִ֑י וַיְבָ֥רֶךְ אֹת֖וֹ שָֽׁם:  ת  וּשְׁאֵל יַעֲקֹב וַאֲמַר חַוִּי כְעַן שְׁמָךְ וַאֲמַר לְמָה דְנַן אַתְּ שָׁאֵל לִשְׁמִי וּבָרִיךְ יָתֵהּ תַּמָּן:

{לא}  שלישי  וַיִּקְרָ֧א יַעֲקֹ֛ב שֵׁ֥ם הַמָּק֖וֹם פְּנִיאֵ֑ל כִּֽי-רָאִ֤יתִי אֱלֹהִים֙ פָּנִ֣ים אֶל-פָּנִ֔ים וַתִּנָּצֵ֖ל נַפְשִֽׁי:  מ  וַיִּקְרָ֧א יַעֲקֹ֛ב שֵׁ֥ם הַמָּק֖וֹם פְּנִיאֵ֑ל כִּֽי-רָאִ֤יתִי אֱלֹהִים֙ פָּנִ֣ים אֶל-פָּנִ֔ים וַתִּנָּצֵ֖ל נַפְשִֽׁי:  ת  וּקְרָא יַעֲקֹב שְׁמָא דְאַתְרָא פְּנִיאֵל אֲרֵי חֲזֵיתִי מַלְאָכַיָּא דַּיְיָ אַפִּין בְּאַפִּין וְאִשְׁתְּזָבַת נַפְשִׁי:

{לב} וַיִּֽזְרַֽח-ל֣וֹ הַשֶּׁ֔מֶשׁ כַּאֲשֶׁ֥ר עָבַ֖ר אֶת-פְּנוּאֵ֑ל וְה֥וּא צֹלֵ֖עַ עַל-יְרֵכֽוֹ:  מ  וַיִּֽזְרַֽח-ל֣וֹ הַשֶּׁ֔מֶשׁ כַּאֲשֶׁ֥ר עָבַ֖ר אֶת-פְּנוּאֵ֑ל וְה֥וּא צֹלֵ֖עַ עַל-יְרֵכֽוֹ:  ת  וּדְנַח לֵהּ שִׁמְשָׁא כְּמָא דִּי עֲבַר יָת פְּנוּאֵל וְהוּא מַטְלַע עַל יַרְכֵּהּ:

{לג} עַל-כֵּ֡ן לֹֽא-יֹאכְל֨וּ בְנֵֽי-יִשְׂרָאֵ֜ל אֶת-גִּ֣יד הַנָּשֶׁ֗ה אֲשֶׁר֙ עַל-כַּ֣ף הַיָּרֵ֔ךְ עַ֖ד הַיּ֣וֹם הַזֶּ֑ה כִּ֤י נָגַע֙ בְּכַף-יֶ֣רֶךְ יַעֲקֹ֔ב בְּגִ֖יד הַנָּשֶֽׁה:  מ  עַל-כֵּ֡ן לֹֽא-יֹאכְל֨וּ בְנֵֽי-יִשְׂרָאֵ֜ל אֶת-גִּ֣יד הַנָּשֶׁ֗ה אֲשֶׁר֙ עַל-כַּ֣ף הַיָּרֵ֔ךְ עַ֖ד הַיּ֣וֹם הַזֶּ֑ה כִּ֤י נָגַע֙ בְּכַף-יֶ֣רֶךְ יַעֲקֹ֔ב בְּגִ֖יד הַנָּשֶֽׁה:  ת  עַל כֵן לָא יֵיכְלוּן בְּנֵי יִשְׂרָאֵל יָת גִּידָא דְנַשְׁיָא דִּי עַל פְּתֵי יַרְכָּא עַד יוֹמָא הָדֵין אֲרֵי קְרֵיב בִּפְתֵי יַרְכָּא דְיַעֲקֹב בְּגִידָא דְנַשְׁיָא:




בראשית פרק-לג

{א} וַיִּשָּׂ֨א יַעֲקֹ֜ב עֵינָ֗יו וַיַּרְא֙ וְהִנֵּ֣ה עֵשָׂ֣ו בָּ֔א וְעִמּ֕וֹ אַרְבַּ֥ע מֵא֖וֹת אִ֑ישׁ וַיַּ֣חַץ אֶת-הַיְלָדִ֗ים עַל-לֵאָה֙ וְעַל-רָחֵ֔ל וְעַ֖ל שְׁתֵּ֥י הַשְּׁפָחֽוֹת:  מ  וַיִּשָּׂ֨א יַעֲקֹ֜ב עֵינָ֗יו וַיַּרְא֙ וְהִנֵּ֣ה עֵשָׂ֣ו בָּ֔א וְעִמּ֕וֹ אַרְבַּ֥ע מֵא֖וֹת אִ֑ישׁ וַיַּ֣חַץ אֶת-הַיְלָדִ֗ים עַל-לֵאָה֙ וְעַל-רָחֵ֔ל וְעַ֖ל שְׁתֵּ֥י הַשְּׁפָחֽוֹת:  ת  וּזְקַף יַעֲקֹב עֵינוֹהִי וַחֲזָא וְהָא עֵשָׂו אֲתָא וְעִמֵּהּ אַרְבַּע מְאָה גֻּבְרִין וּפַלִּיג יָת בְּנַיָּא עַל לֵאָה וְעַל רָחֵל וְעַל תַּרְתֵּין לְחֵינָתָא:

{ב} וַיָּ֧שֶׂם אֶת-הַשְּׁפָח֛וֹת וְאֶת-יַלְדֵיהֶ֖ן רִֽאשֹׁנָ֑ה וְאֶת-לֵאָ֤ה וִֽילָדֶ֙יהָ֙ אַחֲרֹנִ֔ים וְאֶת-רָחֵ֥ל וְאֶת-יוֹסֵ֖ף אַחֲרֹנִֽים:  מ  וַיָּ֧שֶׂם אֶת-הַשְּׁפָח֛וֹת וְאֶת-יַלְדֵיהֶ֖ן רִֽאשֹׁנָ֑ה וְאֶת-לֵאָ֤ה וִֽילָדֶ֙יהָ֙ אַחֲרֹנִ֔ים וְאֶת-רָחֵ֥ל וְאֶת-יוֹסֵ֖ף אַחֲרֹנִֽים:  ת  וְשַׁוִּי יָת לְחֵינָתָא וְיָת בְּנֵיהֶן קַדְמָאִין וְיָת לֵאָה וּבְנָהָא בַּתְרָאִין וְיָת רָחֵל וְיָת יוֹסֵף בַּתְרָאִין:

{ג} וְה֖וּא עָבַ֣ר לִפְנֵיהֶ֑ם וַיִּשְׁתַּ֤חוּ אַ֙רְצָה֙ שֶׁ֣בַע פְּעָמִ֔ים עַד-גִּשְׁתּ֖וֹ עַד-אָחִֽיו:  מ  וְה֖וּא עָבַ֣ר לִפְנֵיהֶ֑ם וַיִּשְׁתַּ֤חוּ אַ֙רְצָה֙ שֶׁ֣בַע פְּעָמִ֔ים עַד-גִּשְׁתּ֖וֹ עַד-אָחִֽיו:  ת  וְהוּא עֲבַר קֳדָמֵיהוֹן וּסְגִיד עַל אַרְעָא שְׁבַע זִמְנִין עַד מִקְרְבֵהּ עַד (נ''י לְוַת) אֲחוּהִי:

{ד} וַיָּ֨רָץ עֵשָׂ֤ו לִקְרָאתוֹ֙ וַֽיְחַבְּקֵ֔הוּ וַיִּפֹּ֥ל עַל-צַוָּארָ֖ו וַׄיִּׄשָּׁׄקֵ֑ׄהׄוּׄ וַיִּבְכּֽוּ:  מ  וַיָּ֨רָץ עֵשָׂ֤ו לִקְרָאתוֹ֙ וַֽיְחַבְּקֵ֔הוּ וַיִּפֹּ֥ל עַל-צַוָּארָ֖ו וַׄיִּׄשָּׁׄקֵ֑ׄהׄוּׄ וַיִּבְכּֽוּ:  ת  וּרְהַט עֵשָׂו לְקַדָּמוּתֵהּ וְגַפְּפֵהּ וּנְפַל עַל צַוְּארֵהּ וְנַשְׁקֵהּ וּבְכוֹ:

{ה} וַיִּשָּׂ֣א אֶת-עֵינָ֗יו וַיַּ֤רְא אֶת-הַנָּשִׁים֙ וְאֶת-הַיְלָדִ֔ים וַיֹּ֖אמֶר מִי-אֵ֣לֶּה לָּ֑ךְ וַיֹּאמַ֕ר הַיְלָדִ֕ים אֲשֶׁר-חָנַ֥ן אֱלֹהִ֖ים אֶת-עַבְדֶּֽךָ:  מ  וַיִּשָּׂ֣א אֶת-עֵינָ֗יו וַיַּ֤רְא אֶת-הַנָּשִׁים֙ וְאֶת-הַיְלָדִ֔ים וַיֹּ֖אמֶר מִי-אֵ֣לֶּה לָּ֑ךְ וַיֹּאמַ֕ר הַיְלָדִ֕ים אֲשֶׁר-חָנַ֥ן אֱלֹהִ֖ים אֶת-עַבְדֶּֽךָ:  ת  וּזְקַף יָת עֵינוֹהִי וַחֲזָא יָת נְשַׁיָּא וְיָת בְּנַיָּא וַאֲמַר מָן אִלֵּין לָךְ וַאֲמַר בְּנַיָּא דִּי חַן יְיָ יָת עַבְדָּךְ:

{ו}  רביעי  וַתִּגַּ֧שְׁןָ הַשְּׁפָח֛וֹת הֵ֥נָּה וְיַלְדֵיהֶ֖ן וַתִּֽשְׁתַּחֲוֶֽיןָ:  מ  וַתִּגַּ֧שְׁןָ הַשְּׁפָח֛וֹת הֵ֥נָּה וְיַלְדֵיהֶ֖ן וַתִּֽשְׁתַּחֲוֶֽיןָ:  ת  וּקְרִיבָן לְחֵינָתָא אִנוּן וּבְנֵיהֶן וּסְגִידוּ:

{ז} וַתִּגַּ֧שׁ גַּם-לֵאָ֛ה וִילָדֶ֖יהָ וַיִּֽשְׁתַּחֲו֑וּ וְאַחַ֗ר נִגַּ֥שׁ יוֹסֵ֛ף וְרָחֵ֖ל וַיִּֽשְׁתַּחֲוֽוּ:  מ  וַתִּגַּ֧שׁ גַּם-לֵאָ֛ה וִילָדֶ֖יהָ וַיִּֽשְׁתַּחֲו֑וּ וְאַחַ֗ר נִגַּ֥שׁ יוֹסֵ֛ף וְרָחֵ֖ל וַיִּֽשְׁתַּחֲוֽוּ:  ת  וּקְרִיבָא אַף לֵאָה וּבְנָהָא וּסְגִידוּ וּבָתַר כֵּן קְרֵיב יוֹסֵף וְרָחֵל וּסְגִידוּ:

{ח} וַיֹּ֕אמֶר מִ֥י לְךָ֛ כָּל-הַמַּחֲנֶ֥ה הַזֶּ֖ה אֲשֶׁ֣ר פָּגָ֑שְׁתִּי וַיֹּ֕אמֶר לִמְצֹא-חֵ֖ן בְּעֵינֵ֥י אֲדֹנִֽי:  מ  וַיֹּ֕אמֶר מִ֥י לְךָ֛ כָּל-הַמַּחֲנֶ֥ה הַזֶּ֖ה אֲשֶׁ֣ר פָּגָ֑שְׁתִּי וַיֹּ֕אמֶר לִמְצֹא-חֵ֖ן בְּעֵינֵ֥י אֲדֹנִֽי:  ת  וַאֲמַר מָן לָךְ כָּל מַשְׁרִיתָא הָדֵין דִּי עֲרָעִית וַאֲמַר לְאַשְׁכָּחָא רַחֲמִין בְּעֵינֵי רִבּוֹנִי:

{ט} וַיֹּ֥אמֶר עֵשָׂ֖ו יֶשׁ-לִ֣י רָ֑ב אָחִ֕י יְהִ֥י לְךָ֖ אֲשֶׁר-לָֽךְ:  מ  וַיֹּ֥אמֶר עֵשָׂ֖ו יֶשׁ-לִ֣י רָ֑ב אָחִ֕י יְהִ֥י לְךָ֖ אֲשֶׁר-לָֽךְ:  ת  וַאֲמַר עֵשָׂו אִית לִי סַגִּי אָחִי אַצְלַח בְּדִילָךְ:

{י} וַיֹּ֣אמֶר יַעֲקֹ֗ב אַל-נָא֙ אִם-נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵינֶ֔יךָ וְלָקַחְתָּ֥ מִנְחָתִ֖י מִיָּדִ֑י כִּ֣י עַל-כֵּ֞ן רָאִ֣יתִי פָנֶ֗יךָ כִּרְאֹ֛ת פְּנֵ֥י אֱלֹהִ֖ים וַתִּרְצֵֽנִי:  מ  וַיֹּ֣אמֶר יַעֲקֹ֗ב אַל-נָא֙ אִם-נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵינֶ֔יךָ וְלָקַחְתָּ֥ מִנְחָתִ֖י מִיָּדִ֑י כִּ֣י עַל-כֵּ֞ן רָאִ֣יתִי פָנֶ֗יךָ כִּרְאֹ֛ת פְּנֵ֥י אֱלֹהִ֖ים וַתִּרְצֵֽנִי:  ת  וַאֲמַר יַעֲקֹב בְּבָעוּ אִם כְּעַן אַשְׁכָּחִית רַחֲמִין בְּעֵינָיךְ וּתְקַבֵּל תִּקְרֻבְתִּי מִן יְדִי אֲרֵי עַל כֵּן חֲזִתִּנּוּן לְאַפָּיךְ כְּחֵיזוּ אַפֵּי רַבְרְבַיָּא וְאִתְרְעֵית לִי:

{יא} קַח-נָ֤א אֶת-בִּרְכָתִי֙ אֲשֶׁ֣ר הֻבָ֣את לָ֔ךְ כִּֽי-חַנַּ֥נִי אֱלֹהִ֖ים וְכִ֣י יֶשׁ-לִי-כֹ֑ל וַיִּפְצַר-בּ֖וֹ וַיִּקָּֽח:  מ  קַח-נָ֤א אֶת-בִּרְכָתִי֙ אֲשֶׁ֣ר הֻבָ֣את לָ֔ךְ כִּֽי-חַנַּ֥נִי אֱלֹהִ֖ים וְכִ֣י יֶשׁ-לִי-כֹ֑ל וַיִּפְצַר-בּ֖וֹ וַיִּקָּֽח:  ת  קַבֵּל כְּעַן יָת תִּקְרֻבְתִּי דְּאִתּוֹתִיאַת לָךְ אֲרֵי רַחִים עֲלַי (קֳדָם) יְיָ וַאֲרֵי אִית לִי כֹלָּא וְאַתְקֵף בֵּהּ וְקַבִּיל:

{יב} וַיֹּ֖אמֶר נִסְעָ֣ה וְנֵלֵ֑כָה וְאֵלְכָ֖ה לְנֶגְדֶּֽךָ:  מ  וַיֹּ֖אמֶר נִסְעָ֣ה וְנֵלֵ֑כָה וְאֵלְכָ֖ה לְנֶגְדֶּֽךָ:  ת  וַאֲמַר נִטַל וְנֵהָךְ וְאֵיהַךְ לְקָבְלָךְ:

{יג} וַיֹּ֣אמֶר אֵלָ֗יו אֲדֹנִ֤י יֹדֵ֙עַ֙ כִּֽי-הַיְלָדִ֣ים רַכִּ֔ים וְהַצֹּ֥אן וְהַבָּקָ֖ר עָל֣וֹת עָלָ֑י וּדְפָקוּם֙ י֣וֹם אֶחָ֔ד וָמֵ֖תוּ כָּל-הַצֹּֽאן:  מ  וַיֹּ֣אמֶר אֵלָ֗יו אֲדֹנִ֤י יֹדֵ֙עַ֙ כִּֽי-הַיְלָדִ֣ים רַכִּ֔ים וְהַצֹּ֥אן וְהַבָּקָ֖ר עָל֣וֹת עָלָ֑י וּדְפָקוּם֙ י֣וֹם אֶחָ֔ד וָמֵ֖תוּ כָּל-הַצֹּֽאן:  ת  וַאֲמַר לֵהּ רִבּוֹנִי יְדַע אֲרֵי יָנְקַיָּא רַכִּיכִין וְעָנָא וְתוֹרֵי מֵינִקְתָּא עָלָי וּדְחוֹקִנּוּן יוֹמָא חַד וִימוּתוּן כָּל עָנָא:

{יד} יַעֲבָר-נָ֥א אֲדֹנִ֖י לִפְנֵ֣י עַבְדּ֑וֹ וַאֲנִ֞י אֶֽתְנָהֲלָ֣ה לְאִטִּ֗י לְרֶ֨גֶל הַמְּלָאכָ֤ה אֲשֶׁר-לְפָנַי֙ וּלְרֶ֣גֶל הַיְלָדִ֔ים עַ֛ד אֲשֶׁר-אָבֹ֥א אֶל-אֲדֹנִ֖י שֵׂעִֽירָה:  מ  יַעֲבָר-נָ֥א אֲדֹנִ֖י לִפְנֵ֣י עַבְדּ֑וֹ וַאֲנִ֞י אֶֽתְנָהֲלָ֣ה לְאִטִּ֗י לְרֶ֨גֶל הַמְּלָאכָ֤ה אֲשֶׁר-לְפָנַי֙ וּלְרֶ֣גֶל הַיְלָדִ֔ים עַ֛ד אֲשֶׁר-אָבֹ֥א אֶל-אֲדֹנִ֖י שֵׂעִֽירָה:  ת  יְעִבַּר כְּעַן רִבּוֹנִי קֳדָם עַבְדֵּהּ וַאֲנָא אֱדַבֵּר בְּנִיחַ לְרֶגֶל עוֹבַדְתָּא דִּי קֳדָמַי וּלְרֶגֶל יָנְקַיָּא עַד דִּי אֵעוֹל לְוַת רִבּוֹנִי לְשֵׂעִיר:

{טו} וַיֹּ֣אמֶר עֵשָׂ֔ו אַצִּֽיגָה-נָּ֣א עִמְּךָ֔ מִן-הָעָ֖ם אֲשֶׁ֣ר אִתִּ֑י וַיֹּ֙אמֶר֙ לָ֣מָּה זֶּ֔ה אֶמְצָא-חֵ֖ן בְּעֵינֵ֥י אֲדֹנִֽי:  מ  וַיֹּ֣אמֶר עֵשָׂ֔ו אַצִּֽיגָה-נָּ֣א עִמְּךָ֔ מִן-הָעָ֖ם אֲשֶׁ֣ר אִתִּ֑י וַיֹּ֙אמֶר֙ לָ֣מָּה זֶּ֔ה אֶמְצָא-חֵ֖ן בְּעֵינֵ֥י אֲדֹנִֽי:  ת  וַאֲמַר עֵשָׂו אֶשְׁבּוֹק כְּעַן עִמָּךְ מִן עַמָּא דְּעִמִּי וַאֲמַר לְמָא דְנַן אַשְׁכַּח רַחֲמִין בְּעֵינֵי רִבּוֹנִי:

{טז} וַיָּשָׁב֩ בַּיּ֨וֹם הַה֥וּא עֵשָׂ֛ו לְדַרְכּ֖וֹ שֵׂעִֽירָה:  מ  וַיָּשָׁב֩ בַּיּ֨וֹם הַה֥וּא עֵשָׂ֛ו לְדַרְכּ֖וֹ שֵׂעִֽירָה:  ת  וְתַב בְּיוֹמָא הַהוּא עֵשָׂו לְאָרְחֵהּ לְשֵׂעִיר:

{יז} וְיַעֲקֹב֙ נָסַ֣ע סֻכֹּ֔תָה וַיִּ֥בֶן ל֖וֹ בָּ֑יִת וּלְמִקְנֵ֙הוּ֙ עָשָׂ֣ה סֻכֹּ֔ת עַל-כֵּ֛ן קָרָ֥א שֵׁם-הַמָּק֖וֹם סֻכּֽוֹת:  מ  וְיַעֲקֹב֙ נָסַ֣ע סֻכֹּ֔תָה וַיִּ֥בֶן ל֖וֹ בָּ֑יִת וּלְמִקְנֵ֙הוּ֙ עָשָׂ֣ה סֻכֹּ֔ת עַל-כֵּ֛ן קָרָ֥א שֵׁם-הַמָּק֖וֹם סֻכּֽוֹת:  ת  וְיַעֲקֹב נְטַל לְסֻכּוֹת וּבְנָא לֵהּ בֵּיתָא וְלִבְעִירֵהּ עֲבַד מְטַלָּן עַל כֵּן קְרָא שְׁמָא דְאַתְרָא סֻכּוֹת: (ס)

{יח} וַיָּבֹא֩ יַעֲקֹ֨ב שָׁלֵ֜ם עִ֣יר שְׁכֶ֗ם אֲשֶׁר֙ בְּאֶ֣רֶץ כְּנַ֔עַן בְּבֹא֖וֹ מִפַּדַּ֣ן אֲרָ֑ם וַיִּ֖חַן אֶת-פְּנֵ֥י הָעִֽיר:  מ  וַיָּבֹא֩ יַעֲקֹ֨ב שָׁלֵ֜ם עִ֣יר שְׁכֶ֗ם אֲשֶׁר֙ בְּאֶ֣רֶץ כְּנַ֔עַן בְּבֹא֖וֹ מִפַּדַּ֣ן אֲרָ֑ם וַיִּ֖חַן אֶת-פְּנֵ֥י הָעִֽיר:  ת  וַאֲתָא יַעֲקֹב שְׁלִים קַרְתָּא דִּשְׁכֶם דִּי בְּאַרְעָא דִּכְנָעַן בְּמֵיתֵהּ מִפַּדַּן דַּאֲרָם וּשְׁרָא לָקֳבֵל (אַפֵּי) קַרְתָּא:

{יט} וַיִּ֜קֶן אֶת-חֶלְקַ֣ת הַשָּׂדֶ֗ה אֲשֶׁ֤ר נָֽטָה-שָׁם֙ אָהֳל֔וֹ מִיַּ֥ד בְּנֵֽי-חֲמ֖וֹר אֲבִ֣י שְׁכֶ֑ם בְּמֵאָ֖ה קְשִׂיטָֽה:  מ  וַיִּ֜קֶן אֶת-חֶלְקַ֣ת הַשָּׂדֶ֗ה אֲשֶׁ֤ר נָֽטָה-שָׁם֙ אָהֳל֔וֹ מִיַּ֥ד בְּנֵֽי-חֲמ֖וֹר אֲבִ֣י שְׁכֶ֑ם בְּמֵאָ֖ה קְשִׂיטָֽה:  ת  וּזְבַן יָת אַחֲסָנַת חַקְלָא דִּי פְרַס תַּמָּן מַשְׁכְּנֵהּ מִידָא דִּבְנֵי חֲמוֹר אֲבוּהִי דִּשְׁכֶם בְּמֵאָה חוּרְפָן:

{כ} וַיַּצֶּב-שָׁ֖ם מִזְבֵּ֑חַ וַיִּ֨קְרָא-ל֔וֹ אֵ֖ל אֱלֹהֵ֥י יִשְׂרָאֵֽל:  מ  וַיַּצֶּב-שָׁ֖ם מִזְבֵּ֑חַ וַיִּ֨קְרָא-ל֔וֹ אֵ֖ל אֱלֹהֵ֥י יִשְׂרָאֵֽל:  ת  וַאֲקֵם תַּמָּן מַדְבְּחָא וּפְלַח עֲלוֹהִי קֳדָם אֵל אֱלָהָא דְיִשְׂרָאֵל:




בראשית פרק-לד

{א}  חמישי  וַתֵּצֵ֤א דִינָה֙ בַּת-לֵאָ֔ה אֲשֶׁ֥ר יָלְדָ֖ה לְיַעֲקֹ֑ב לִרְא֖וֹת בִּבְנ֥וֹת הָאָֽרֶץ:  מ  וַתֵּצֵ֤א דִינָה֙ בַּת-לֵאָ֔ה אֲשֶׁ֥ר יָלְדָ֖ה לְיַעֲקֹ֑ב לִרְא֖וֹת בִּבְנ֥וֹת הָאָֽרֶץ:  ת  וּנְפָקַת דִּינָה בַּת לֵאָה דִּי יְלִידַת לְיַעֲקֹב לְמֶחֱזֵי בִּבְנַת אַרְעָא:

{ב} וַיַּ֨רְא אֹתָ֜הּ שְׁכֶ֧ם בֶּן-חֲמ֛וֹר הַֽחִוִּ֖י נְשִׂ֣יא הָאָ֑רֶץ וַיִּקַּ֥ח אֹתָ֛הּ וַיִּשְׁכַּ֥ב אֹתָ֖הּ וַיְעַנֶּֽהָ:  מ  וַיַּ֨רְא אֹתָ֜הּ שְׁכֶ֧ם בֶּן-חֲמ֛וֹר הַֽחִוִּ֖י נְשִׂ֣יא הָאָ֑רֶץ וַיִּקַּ֥ח אֹתָ֛הּ וַיִּשְׁכַּ֥ב אֹתָ֖הּ וַיְעַנֶּֽהָ:  ת  וַחֲזָא יָתַהּ שְׁכֶם בַּר חֲמוֹר חִוָאָה רַבָּא דְאַרְעָא וּדְבַר יָתַהּ וּשְׁכִיב יָתַהּ וְעַנְיַהּ:

{ג} וַתִּדְבַּ֣ק נַפְשׁ֔וֹ בְּדִינָ֖ה בַּֽת-יַעֲקֹ֑ב וַיֶּֽאֱהַב֙ אֶת-הַֽנַּעֲרָ֔ וַיְדַבֵּ֖ר עַל-לֵ֥ב הַֽנַּעֲרָֽ:  מ  וַתִּדְבַּ֣ק נַפְשׁ֔וֹ בְּדִינָ֖ה בַּֽת-יַעֲקֹ֑ב וַיֶּֽאֱהַב֙ אֶת-הַֽנַּעֲרָ֔ וַיְדַבֵּ֖ר עַל-לֵ֥ב הַֽנַּעֲרָֽ:  ת  וְאִתְרְעִיאַת נַפְשֵׁהּ בְּדִינָה בַּת יַעֲקֹב וּרְחֵים יָת עוּלֶמְתָּא וּמַלִּיל עַל לִבָּא דְעוּלֶמְתָּא:

{ד} וַיֹּ֣אמֶר שְׁכֶ֔ם אֶל-חֲמ֥וֹר אָבִ֖יו לֵאמֹ֑ר קַֽח-לִ֛י אֶת-הַיַּלְדָּ֥ה הַזֹּ֖את לְאִשָּֽׁה:  מ  וַיֹּ֣אמֶר שְׁכֶ֔ם אֶל-חֲמ֥וֹר אָבִ֖יו לֵאמֹ֑ר קַֽח-לִ֛י אֶת-הַיַּלְדָּ֥ה הַזֹּ֖את לְאִשָּֽׁה:  ת  וַאֲמַר שְׁכֶם לַחֲמוֹר אֲבוּהִי לְמֵימָר סַב לִי יָת עוּלֶמְתָּא הָדָא לְאִנְתּוּ:

{ה} וְיַעֲקֹ֣ב שָׁמַ֗ע כִּ֤י טִמֵּא֙ אֶת-דִּינָ֣ה בִתּ֔וֹ וּבָנָ֛יו הָי֥וּ אֶת-מִקְנֵ֖הוּ בַּשָּׂדֶ֑ה וְהֶחֱרִ֥שׁ יַעֲקֹ֖ב עַד-בֹּאָֽם:  מ  וְיַעֲקֹ֣ב שָׁמַ֗ע כִּ֤י טִמֵּא֙ אֶת-דִּינָ֣ה בִתּ֔וֹ וּבָנָ֛יו הָי֥וּ אֶת-מִקְנֵ֖הוּ בַּשָּׂדֶ֑ה וְהֶחֱרִ֥שׁ יַעֲקֹ֖ב עַד-בֹּאָֽם:  ת  וְיַעֲקֹב שְׁמַע אֲרֵי סָאֵיב יָת דִּינָה בְרַתֵּהּ וּבְנוֹהִי הֲווֹ עִם גֵּיתוֹהִי בְּחַקְלָא וְשָׁתִיק יַעֲקֹב עַד מֵיתֵיהוֹן:

{ו} וַיֵּצֵ֛א חֲמ֥וֹר אֲבִֽי-שְׁכֶ֖ם אֶֽל-יַעֲקֹ֑ב לְדַבֵּ֖ר אִתּֽוֹ:  מ  וַיֵּצֵ֛א חֲמ֥וֹר אֲבִֽי-שְׁכֶ֖ם אֶֽל-יַעֲקֹ֑ב לְדַבֵּ֖ר אִתּֽוֹ:  ת  וּנְפַק חֲמוֹר אֲבוּהִי דִּשְׁכֶם לְוַת יַעֲקֹב לְמַלָּלָא עִמֵּהּ:

{ז} וּבְנֵ֨י יַעֲקֹ֜ב בָּ֤אוּ מִן-הַשָּׂדֶה֙ כְּשָׁמְעָ֔ם וַיִּֽתְעַצְּבוּ֙ הָֽאֲנָשִׁ֔ים וַיִּ֥חַר לָהֶ֖ם מְאֹ֑ד כִּֽי-נְבָלָ֞ה עָשָׂ֣ה בְיִשְׂרָאֵ֗ל לִשְׁכַּב֙ אֶת-בַּֽת-יַעֲקֹ֔ב וְכֵ֖ן לֹ֥א יֵעָשֶֽׂה:  מ  וּבְנֵ֨י יַעֲקֹ֜ב בָּ֤אוּ מִן-הַשָּׂדֶה֙ כְּשָׁמְעָ֔ם וַיִּֽתְעַצְּבוּ֙ הָֽאֲנָשִׁ֔ים וַיִּ֥חַר לָהֶ֖ם מְאֹ֑ד כִּֽי-נְבָלָ֞ה עָשָׂ֣ה בְיִשְׂרָאֵ֗ל לִשְׁכַּב֙ אֶת-בַּֽת-יַעֲקֹ֔ב וְכֵ֖ן לֹ֥א יֵעָשֶֽׂה:  ת  וּבְנֵי יַעֲקֹב עַלּוּ מִן חַקְלָא כַּד שְׁמָעוּ וְאִתְנְסִיסוּ גֻּבְרַיָּא וּתְקֵיף לְהוֹן לַחֲדָא אֲרֵי קְלָנָא עֲבַד בְּיִשְׂרָאֵל לְמִשְׁכַּב עִם בַּת יַעֲקֹב וְכֵן לָא כָשַׁר לְאִתְעֲבָדָא:

{ח} וַיְדַבֵּ֥ר חֲמ֖וֹר אִתָּ֣ם לֵאמֹ֑ר שְׁכֶ֣ם בְּנִ֗י חָֽשְׁקָ֤ה נַפְשׁוֹ֙ בְּבִתְּכֶ֔ם תְּנ֨וּ נָ֥א אֹתָ֛הּ ל֖וֹ לְאִשָּֽׁה:  מ  וַיְדַבֵּ֥ר חֲמ֖וֹר אִתָּ֣ם לֵאמֹ֑ר שְׁכֶ֣ם בְּנִ֗י חָֽשְׁקָ֤ה נַפְשׁוֹ֙ בְּבִתְּכֶ֔ם תְּנ֨וּ נָ֥א אֹתָ֛הּ ל֖וֹ לְאִשָּֽׁה:  ת  וּמַלִּיל חֲמוֹר עִמְּהוֹן לְמֵימָר שְׁכֶם בְּרִי אִתְרְעִיאַת נַפְשֵׁהּ בִּבְרַתְּכוֹן הָבוּ כְעַן יָתַהּ לֵהּ לְאִנְתּוּ:

{ט} וְהִֽתְחַתְּנ֖וּ אֹתָ֑נוּ בְּנֹֽתֵיכֶם֙ תִּתְּנוּ-לָ֔נוּ וְאֶת-בְּנֹתֵ֖ינוּ תִּקְח֥וּ לָכֶֽם:  מ  וְהִֽתְחַתְּנ֖וּ אֹתָ֑נוּ בְּנֹֽתֵיכֶם֙ תִּתְּנוּ-לָ֔נוּ וְאֶת-בְּנֹתֵ֖ינוּ תִּקְח֥וּ לָכֶֽם:  ת  וְתִתְחַתְּנוּן בָּנָא בְּנָתֵיכוֹן תִּתְּנוּן לָנָא וְיָת בְּנָתָנָא תִּסְבוּן לְכוֹן:

{י} וְאִתָּ֖נוּ תֵּשֵׁ֑בוּ וְהָאָ֙רֶץ֙ תִּהְיֶ֣ה לִפְנֵיכֶ֔ם שְׁבוּ֙ וּסְחָר֔וּהָ וְהֵֽאָחֲז֖וּ בָּֽהּ:  מ  וְאִתָּ֖נוּ תֵּשֵׁ֑בוּ וְהָאָ֙רֶץ֙ תִּהְיֶ֣ה לִפְנֵיכֶ֔ם שְׁבוּ֙ וּסְחָר֔וּהָ וְהֵֽאָחֲז֖וּ בָּֽהּ:  ת  וְעִמָּנָא תֵּתְבוּן וְאַרְעָא תְּהֵי קֳדָמֵיכוֹן תִּיבוּ וְעֲבִידוּ בַהּ סְחוֹרְתָּא וְאַחֲסִינוּ בַּהּ:

{יא} וַיֹּ֤אמֶר שְׁכֶם֙ אֶל-אָבִ֣יה וְאֶל-אַחֶ֔יהָ אֶמְצָא-חֵ֖ן בְּעֵינֵיכֶ֑ם וַאֲשֶׁ֥ר תֹּאמְר֛וּ אֵלַ֖י אֶתֵּֽן:  מ  וַיֹּ֤אמֶר שְׁכֶם֙ אֶל-אָבִ֣יה וְאֶל-אַחֶ֔יהָ אֶמְצָא-חֵ֖ן בְּעֵינֵיכֶ֑ם וַאֲשֶׁ֥ר תֹּאמְר֛וּ אֵלַ֖י אֶתֵּֽן:  ת  וַאֲמַר שְׁכֶם לַאֲבוּהָא וּלְאַחָהָא אַשְׁכַּח רַחֲמִין בְּעֵינֵיכוֹן וְדִי תֵימְרוּן לִי אֶתֵּן:

{יב} הַרְבּ֨וּ עָלַ֤י מְאֹד֙ מֹ֣הַר וּמַתָּ֔ן וְאֶ֨תְּנָ֔ה כַּאֲשֶׁ֥ר תֹּאמְר֖וּ אֵלָ֑י וּתְנוּ-לִ֥י אֶת-הַֽנַּעֲרָ֖ לְאִשָּֽׁה:  מ  הַרְבּ֨וּ עָלַ֤י מְאֹד֙ מֹ֣הַר וּמַתָּ֔ן וְאֶ֨תְּנָ֔ה כַּאֲשֶׁ֥ר תֹּאמְר֖וּ אֵלָ֑י וּתְנוּ-לִ֥י אֶת-הַֽנַּעֲרָ֖ לְאִשָּֽׁה:  ת  אַסְגוּ עָלַי לַחֲדָא מוֹהֲרִין וּמַתְּנָן וְאֶתֵּן כְּמָא דִּי תֵימְרוּן לִי וְהָבוּ לִי יָת עוּלֶמְתָּא:לאִנְתּוּ:

{יג} וַיַּעֲנ֨וּ בְנֵֽי-יַעֲקֹ֜ב אֶת-שְׁכֶ֨ם וְאֶת-חֲמ֥וֹר אָבִ֛יו בְּמִרְמָ֖ה וַיְדַבֵּ֑רוּ אֲשֶׁ֣ר טִמֵּ֔א אֵ֖ת דִּינָ֥ה אֲחֹתָֽם:  מ  וַיַּעֲנ֨וּ בְנֵֽי-יַעֲקֹ֜ב אֶת-שְׁכֶ֨ם וְאֶת-חֲמ֥וֹר אָבִ֛יו בְּמִרְמָ֖ה וַיְדַבֵּ֑רוּ אֲשֶׁ֣ר טִמֵּ֔א אֵ֖ת דִּינָ֥ה אֲחֹתָֽם:  ת  וַאֲתִיבוּ בְנֵי יַעֲקֹב יָת שְׁכֶם וְיָת חֲמוֹר אֲבוּהִי בְּחָכְמְתָא וּמַלִּילוּ דִּי סָאֵב יָת דִּינָה אֲחָתְהוֹן:

{יד} וַיֹּאמְר֣וּ אֲלֵיהֶ֗ם לֹ֤א נוּכַל֙ לַעֲשׂוֹת֙ הַדָּבָ֣ר הַזֶּ֔ה לָתֵת֙ אֶת-אֲחֹתֵ֔נוּ לְאִ֖ישׁ אֲשֶׁר-ל֣וֹ עָרְלָ֑ה כִּֽי-חֶרְפָּ֥ה הִ֖וא לָֽנוּ:  מ  וַיֹּאמְר֣וּ אֲלֵיהֶ֗ם לֹ֤א נוּכַל֙ לַעֲשׂוֹת֙ הַדָּבָ֣ר הַזֶּ֔ה לָתֵת֙ אֶת-אֲחֹתֵ֔נוּ לְאִ֖ישׁ אֲשֶׁר-ל֣וֹ עָרְלָ֑ה כִּֽי-חֶרְפָּ֥ה הִ֖וא לָֽנוּ:  ת  וַאֲמָרוּ לְהוֹן לָא נִכּוּל לְמֶעְבַּד פִּתְגָּמָא הָדֵין לְמִתַּן יָת אֲחָתָנָא לִגְבַר דִּי לֵיהּ עָרְלְתָא אֲרֵי חִסּוּדָא הִיא לָנָא:

{טו} אַךְ-בְּזֹ֖את נֵא֣וֹת לָכֶ֑ם אִ֚ם תִּהְי֣וּ כָמֹ֔נוּ לְהִמֹּ֥ל לָכֶ֖ם כָּל-זָכָֽר:  מ  אַךְ-בְּזֹ֖את נֵא֣וֹת לָכֶ֑ם אִ֚ם תִּהְי֣וּ כָמֹ֔נוּ לְהִמֹּ֥ל לָכֶ֖ם כָּל-זָכָֽר:  ת  בְּרַם בְּדָא נִתְפֵּס לְכוֹן אִם תֶּהֱווֹן כְּוָתָנָא לְמִגְזַר לְכוֹן כָּל דְּכוּרָא:

{טז} וְנָתַ֤נּוּ אֶת-בְּנֹתֵ֙ינוּ֙ לָכֶ֔ם וְאֶת-בְּנֹתֵיכֶ֖ם נִֽקַּֽח-לָ֑נוּ וְיָשַׁ֣בְנוּ אִתְּכֶ֔ם וְהָיִ֖ינוּ לְעַ֥ם אֶחָֽד:  מ  וְנָתַ֤נּוּ אֶת-בְּנֹתֵ֙ינוּ֙ לָכֶ֔ם וְאֶת-בְּנֹתֵיכֶ֖ם נִֽקַּֽח-לָ֑נוּ וְיָשַׁ֣בְנוּ אִתְּכֶ֔ם וְהָיִ֖ינוּ לְעַ֥ם אֶחָֽד:  ת  וְנִתַּן יָת בְּנָתָנָא לְכוֹן וְיָת בְּנָתֵיכוֹן נִסַּב לָנָא וְנִתּוּב עִמְּכוֹן וְנֵיהֵי לְעַמָּא חָד:

{יז} וְאִם-לֹ֧א תִשְׁמְע֛וּ אֵלֵ֖ינוּ לְהִמּ֑וֹל וְלָקַ֥חְנוּ אֶת-בִּתֵּ֖נוּ וְהָלָֽכְנוּ:  מ  וְאִם-לֹ֧א תִשְׁמְע֛וּ אֵלֵ֖ינוּ לְהִמּ֑וֹל וְלָקַ֥חְנוּ אֶת-בִּתֵּ֖נוּ וְהָלָֽכְנוּ:  ת  וְאִם לָא תְקַבְּלוּן מִנָּנָא לְמִגְזָר וּנְדַבֵּר יָת בְּרַתָּנָא וְנֵזֵיל:

{יח} וַיִּֽיטְב֥וּ דִבְרֵיהֶ֖ם בְּעֵינֵ֣י חֲמ֑וֹר וּבְעֵינֵ֖י שְׁכֶ֥ם בֶּן-חֲמֽוֹר:  מ  וַיִּֽיטְב֥וּ דִבְרֵיהֶ֖ם בְּעֵינֵ֣י חֲמ֑וֹר וּבְעֵינֵ֖י שְׁכֶ֥ם בֶּן-חֲמֽוֹר:  ת  וּשְׁפָרוּ פִּתְגָּמֵיהוֹן בְּעֵינֵי חֲמוֹר וּבְעֵינֵי שְׁכֶם בַּר חֲמוֹר:

{יט} וְלֹֽא-אֵחַ֤ר הַנַּ֙עַר֙ לַעֲשׂ֣וֹת הַדָּבָ֔ר כִּ֥י חָפֵ֖ץ בְּבַֽת-יַעֲקֹ֑ב וְה֣וּא נִכְבָּ֔ד מִכֹּ֖ל בֵּ֥ית אָבִֽיו:  מ  וְלֹֽא-אֵחַ֤ר הַנַּ֙עַר֙ לַעֲשׂ֣וֹת הַדָּבָ֔ר כִּ֥י חָפֵ֖ץ בְּבַֽת-יַעֲקֹ֑ב וְה֣וּא נִכְבָּ֔ד מִכֹּ֖ל בֵּ֥ית אָבִֽיו:  ת  וְלָא אוֹחַר עוּלֵימָא לְמֶעְבַּד פִּתְגָּמָא אֲרֵי אִתְרְעִי בְּבַת יַעֲקֹב וְהוּא יַקִּיר מִכֹּל בֵּית אֲבוּהִי:

{כ} וַיָּבֹ֥א חֲמ֛וֹר וּשְׁכֶ֥ם בְּנ֖וֹ אֶל-שַׁ֣עַר עִירָ֑ם וַֽיְדַבְּר֛וּ אֶל-אַנְשֵׁ֥י עִירָ֖ם לֵאמֹֽר:  מ  וַיָּבֹ֥א חֲמ֛וֹר וּשְׁכֶ֥ם בְּנ֖וֹ אֶל-שַׁ֣עַר עִירָ֑ם וַֽיְדַבְּר֛וּ אֶל-אַנְשֵׁ֥י עִירָ֖ם לֵאמֹֽר:  ת  וַאֲתָא חֲמוֹר וּשְׁכֶם בְּרֵהּ לִתְרַע קַרְתְּהוֹן וּמַלִּילוּ עִם אֲנָשֵׁי קַרְתְּהוֹן לְמֵימָר:

{כא} הָאֲנָשִׁ֨ים הָאֵ֜לֶּה שְֽׁלֵמִ֧ים הֵ֣ם אִתָּ֗נוּ וְיֵשְׁב֤וּ בָאָ֙רֶץ֙ וְיִסְחֲר֣וּ אֹתָ֔הּ וְהָאָ֛רֶץ הִנֵּ֥ה רַֽחֲבַת-יָדַ֖יִם לִפְנֵיהֶ֑ם אֶת-בְּנֹתָם֙ נִקַּֽח-לָ֣נוּ לְנָשִׁ֔ים וְאֶת-בְּנֹתֵ֖ינוּ נִתֵּ֥ן לָהֶֽם:  מ  הָאֲנָשִׁ֨ים הָאֵ֜לֶּה שְֽׁלֵמִ֧ים הֵ֣ם אִתָּ֗נוּ וְיֵשְׁב֤וּ בָאָ֙רֶץ֙ וְיִסְחֲר֣וּ אֹתָ֔הּ וְהָאָ֛רֶץ הִנֵּ֥ה רַֽחֲבַת-יָדַ֖יִם לִפְנֵיהֶ֑ם אֶת-בְּנֹתָם֙ נִקַּֽח-לָ֣נוּ לְנָשִׁ֔ים וְאֶת-בְּנֹתֵ֖ינוּ נִתֵּ֥ן לָהֶֽם:  ת  גֻּבְרַיָּא הָאִלֵּין שָׁלְמִין אִנּוּן עִמָּנָא וִיתִיבוּן בְּאַרְעָא וְיַעְבְּדוּן בַּהּ סְחוֹרְתָא וְאַרְעָא הָא פְתָיוּת יְדִין קֳדָמֵיהוֹן יָת בְּנָתֵיהוֹן נִסַּב לָנָא לִנְשִׁין וְיָת בְּנָתָנָא נִתֵּן לְהוֹן:

{כב} אַךְ-בְּ֠זֹאת יֵאֹ֨תוּ לָ֤נוּ הָאֲנָשִׁים֙ לָשֶׁ֣בֶת אִתָּ֔נוּ לִהְי֖וֹת לְעַ֣ם אֶחָ֑ד בְּהִמּ֥וֹל לָ֙נוּ֙ כָּל-זָכָ֔ר כַּאֲשֶׁ֖ר הֵ֥ם נִמֹּלִֽים:  מ  אַךְ-בְּ֠זֹאת יֵאֹ֨תוּ לָ֤נוּ הָאֲנָשִׁים֙ לָשֶׁ֣בֶת אִתָּ֔נוּ לִהְי֖וֹת לְעַ֣ם אֶחָ֑ד בְּהִמּ֥וֹל לָ֙נוּ֙ כָּל-זָכָ֔ר כַּאֲשֶׁ֖ר הֵ֥ם נִמֹּלִֽים:  ת  בְּרַם בְּדָא יִתָּפְסוּן לָנָא גּוּבְרַיָּא לְמִתַּב עִמָּנָא לְמֶהֱוֵי לְעַמָּא חָד בְּמִגְזַר לָנָא כָּל דְּכוּרָא כְּמָא דִּי אִנּוּן גְּזִירִין:

{כג} מִקְנֵהֶ֤ם וְקִנְיָנָם֙ וְכָל-בְּהֶמְתָּ֔ם הֲל֥וֹא לָ֖נוּ הֵ֑ם אַ֚ךְ נֵא֣וֹתָה לָהֶ֔ם וְיֵשְׁב֖וּ אִתָּֽנוּ:  מ  מִקְנֵהֶ֤ם וְקִנְיָנָם֙ וְכָל-בְּהֶמְתָּ֔ם הֲל֥וֹא לָ֖נוּ הֵ֑ם אַ֚ךְ נֵא֣וֹתָה לָהֶ֔ם וְיֵשְׁב֖וּ אִתָּֽנוּ:  ת  גֵּיתֵיהוֹן וְקִנְיָנְהוֹן וְכָל בְּעִירְהוֹן הֲלָא דִּילָנָא אִנּוּן בְּרַם נִתְפֵּס לְהוֹן וִיתִיבוּן עִמָּנָא:

{כד} וַיִּשְׁמְע֤וּ אֶל-חֲמוֹר֙ וְאֶל-שְׁכֶ֣ם בְּנ֔וֹ כָּל-יֹצְאֵ֖י שַׁ֣עַר עִיר֑וֹ וַיִּמֹּ֙לוּ֙ כָּל-זָכָ֔ר כָּל-יֹצְאֵ֖י שַׁ֥עַר עִירֽוֹ:  מ  וַיִּשְׁמְע֤וּ אֶל-חֲמוֹר֙ וְאֶל-שְׁכֶ֣ם בְּנ֔וֹ כָּל-יֹצְאֵ֖י שַׁ֣עַר עִיר֑וֹ וַיִּמֹּ֙לוּ֙ כָּל-זָכָ֔ר כָּל-יֹצְאֵ֖י שַׁ֥עַר עִירֽוֹ:  ת  וְקַבִּילוּ מִן חֲמוֹר וּמִן שְׁכֶם בְּרֵהּ כָּל נָפְקֵי תְּרַע קַרְתֵּהּ וּגְזָרוּ כָּל דְּכוּרָא כָּל נָפְקֵי תְּרַע קַרְתֵּהּ:

{כה} וַיְהִי֩ בַיּ֨וֹם הַשְּׁלִישִׁ֜י בִּֽהְיוֹתָ֣ם כֹּֽאֲבִ֗ים וַיִּקְח֣וּ שְׁנֵֽי-בְנֵי-יַ֠עֲקֹב שִׁמְע֨וֹן וְלֵוִ֜י אֲחֵ֤י דִינָה֙ אִ֣ישׁ חַרְבּ֔וֹ וַיָּבֹ֥אוּ עַל-הָעִ֖יר בֶּ֑טַח וַיַּֽהַרְג֖וּ כָּל-זָכָֽר:  מ  וַיְהִי֩ בַיּ֨וֹם הַשְּׁלִישִׁ֜י בִּֽהְיוֹתָ֣ם כֹּֽאֲבִ֗ים וַיִּקְח֣וּ שְׁנֵֽי-בְנֵי-יַ֠עֲקֹב שִׁמְע֨וֹן וְלֵוִ֜י אֲחֵ֤י דִינָה֙ אִ֣ישׁ חַרְבּ֔וֹ וַיָּבֹ֥אוּ עַל-הָעִ֖יר בֶּ֑טַח וַיַּֽהַרְג֖וּ כָּל-זָכָֽר:  ת  וַהֲוָה בְיוֹמָא תְלִיתָאָה כַּד תְּקִיפוּ עֲלֵיהוֹן כֵּאבֵיהוֹן וּנְסִיבוּ תְרֵין בְּנֵי יַעֲקֹב שִׁמְעוֹן וְלֵוִי אֲחֵי דִּינָה גְּבַר חַרְבֵּהּ וְעַלּוּ עַל קַרְתָּא דְּיָתְבוּן לְרָחֲצָן וּקְטָלוּ כָּל דְּכוּרָא:

{כו} וְאֶת-חֲמוֹר֙ וְאֶת-שְׁכֶ֣ם בְּנ֔וֹ הָרְג֖וּ לְפִי-חָ֑רֶב וַיִּקְח֧וּ אֶת-דִּינָ֛ה מִבֵּ֥ית שְׁכֶ֖ם וַיֵּצֵֽאוּ:  מ  וְאֶת-חֲמוֹר֙ וְאֶת-שְׁכֶ֣ם בְּנ֔וֹ הָרְג֖וּ לְפִי-חָ֑רֶב וַיִּקְח֧וּ אֶת-דִּינָ֛ה מִבֵּ֥ית שְׁכֶ֖ם וַיֵּצֵֽאוּ:  ת  וְיָת חֲמוֹר וְיָת שְׁכֶם בְּרֵהּ קְטָלוּ לְפִתְגַּם דְּחָרֶב וּדְבָרוּ יָת דִּינָה מִבֵּית שְׁכֶם וּנְפָקוּ:

{כז} בְּנֵ֣י יַעֲקֹ֗ב בָּ֚אוּ עַל-הַ֣חֲלָלִ֔ים וַיָּבֹ֖זּוּ הָעִ֑יר אֲשֶׁ֥ר טִמְּא֖וּ אֲחוֹתָֽם:  מ  בְּנֵ֣י יַעֲקֹ֗ב בָּ֚אוּ עַל-הַ֣חֲלָלִ֔ים וַיָּבֹ֖זּוּ הָעִ֑יר אֲשֶׁ֥ר טִמְּא֖וּ אֲחוֹתָֽם:  ת  בְּנֵי יַעֲקֹב עַלּוּ לְחַלָּצָא קָטִילַיָּא וּבְזוֹ קַרְתָּא דִּי סָאִיבוּ אֲחָתְהוֹן:

{כח} אֶת-צֹאנָ֥ם וְאֶת-בְּקָרָ֖ם וְאֶת-חֲמֹרֵיהֶּ֑ם וְאֵ֧ת אֲשֶׁר-בָּעִ֛יר וְאֶת-אֲשֶׁ֥ר בַּשָּׂדֶ֖ה לָקָֽחוּ:  מ  אֶת-צֹאנָ֥ם וְאֶת-בְּקָרָ֖ם וְאֶת-חֲמֹרֵיהֶּ֑ם וְאֵ֧ת אֲשֶׁר-בָּעִ֛יר וְאֶת-אֲשֶׁ֥ר בַּשָּׂדֶ֖ה לָקָֽחוּ:  ת  יָת עָנְהוֹן וְיָת תּוֹרְהוֹן וְיָת חֲמָרֵיהוֹן וְיָת דִּי בְקַרְתָּא וְיָת דִּי בְחַקְלָא בְּזוֹ:

{כט} וְאֶת-כָּל-חֵילָ֤ם וְאֶת-כָּל-טַפָּם֙ וְאֶת-נְשֵׁיהֶ֔ם שָׁב֖וּ וַיָּבֹ֑זּוּ וְאֵ֖ת כָּל-אֲשֶׁ֥ר בַּבָּֽיִת:  מ  וְאֶת-כָּל-חֵילָ֤ם וְאֶת-כָּל-טַפָּם֙ וְאֶת-נְשֵׁיהֶ֔ם שָׁב֖וּ וַיָּבֹ֑זּוּ וְאֵ֖ת כָּל-אֲשֶׁ֥ר בַּבָּֽיִת:  ת  וְיָת כָּל נִכְסֵיהוֹן וְיָת כָּל טַפְלְהוֹן וְיָת נְשֵׁיהוֹן שְׁבוֹ וּבְזוֹ וְיָת כָּל דִּי בְבֵיתָא:

{ל} וַיֹּ֨אמֶר יַעֲקֹ֜ב אֶל-שִׁמְע֣וֹן וְאֶל-לֵוִי֮ עֲכַרְתֶּ֣ם אֹתִי֒ לְהַבְאִישֵׁ֙נִי֙ בְּיֹשֵׁ֣ב הָאָ֔רֶץ בַּֽכְּנַעֲנִ֖י וּבַפְּרִזִּ֑י וַאֲנִי֙ מְתֵ֣י מִסְפָּ֔ר וְנֶאֶסְפ֤וּ עָלַי֙ וְהִכּ֔וּנִי וְנִשְׁמַדְתִּ֖י אֲנִ֥י וּבֵיתִֽי:  מ  וַיֹּ֨אמֶר יַעֲקֹ֜ב אֶל-שִׁמְע֣וֹן וְאֶל-לֵוִי֮ עֲכַרְתֶּ֣ם אֹתִי֒ לְהַבְאִישֵׁ֙נִי֙ בְּיֹשֵׁ֣ב הָאָ֔רֶץ בַּֽכְּנַעֲנִ֖י וּבַפְּרִזִּ֑י וַאֲנִי֙ מְתֵ֣י מִסְפָּ֔ר וְנֶאֶסְפ֤וּ עָלַי֙ וְהִכּ֔וּנִי וְנִשְׁמַדְתִּ֖י אֲנִ֥י וּבֵיתִֽי:  ת  וַאֲמַר יַעֲקֹב לְשִׁמְעוֹן וּלְלֵוִי עֲכַרְתּוּן יָתִי לְמִתַּן דְּבָבוּ בֵּינָנָא וּבֵין יָתְבֵי אַרְעָא בִּכְנַעֲנָאָה וּבַפְּרִזָּאָה וַאֲנָא עַם דְּמִנְיַן וְיִתְכַּנְשׁוּן עֲלַי וְיִמְחִנּוּנִי וְאֶשְׁתֵּצֵי אֲנָא וְאֱנַשׁ בֵּיתִי:

{לא} וַיֹּאמְר֑וּ הַכְזוֹנָ֕ה יַעֲשֶׂ֖ה אֶת-אֲחוֹתֵֽנוּ:  מ  וַיֹּאמְר֑וּ הַכְזוֹנָ֕ה יַעֲשֶׂ֖ה אֶת-אֲחוֹתֵֽנוּ:  ת  וַאֲמָרוּ הַכְנָפְקַת בָּרָא יִתְעֲבֵד (נ''א יַעְבֵּד יָת אַחָתָנָא) לַאֲחָתָנָא: (פ)




בראשית פרק-לה

{א} וַיֹּ֤אמֶר אֱלֹהִים֙ אֶֽל-יַעֲקֹ֔ב ק֛וּם עֲלֵ֥ה בֵֽית-אֵ֖ל וְשֶׁב-שָׁ֑ם וַעֲשֵׂה-שָׁ֣ם מִזְבֵּ֔חַ לָאֵל֙ הַנִּרְאֶ֣ה אֵלֶ֔יךָ בְּבָרְחֲךָ֔ מִפְּנֵ֖י עֵשָׂ֥ו אָחִֽיךָ:  מ  וַיֹּ֤אמֶר אֱלֹהִים֙ אֶֽל-יַעֲקֹ֔ב ק֛וּם עֲלֵ֥ה בֵֽית-אֵ֖ל וְשֶׁב-שָׁ֑ם וַעֲשֵׂה-שָׁ֣ם מִזְבֵּ֔חַ לָאֵל֙ הַנִּרְאֶ֣ה אֵלֶ֔יךָ בְּבָרְחֲךָ֔ מִפְּנֵ֖י עֵשָׂ֥ו אָחִֽיךָ:  ת  וַאֲמַר יְיָ לְיַעֲקֹב קוּם סַק לְבֵית אֵל וְתִיב תַּמָן וְעִבֵּד תַּמָן מַדְבְּחָא לֵאלָהָא דְאִתְגְלִי לָךְ בְּמֵעִרְקָךְ מִן קֳדָם עֵשָׂו אָחוּךְ:

{ב} וַיֹּ֤אמֶר יַעֲקֹב֙ אֶל-בֵּית֔וֹ וְאֶ֖ל כָּל-אֲשֶׁ֣ר עִמּ֑וֹ הָסִ֜רוּ אֶת-אֱלֹהֵ֤י הַנֵּכָר֙ אֲשֶׁ֣ר בְּתֹכְכֶ֔ם וְהִֽטַּהֲר֔וּ וְהַחֲלִ֖יפוּ שִׂמְלֹתֵיכֶֽם:  מ  וַיֹּ֤אמֶר יַעֲקֹב֙ אֶל-בֵּית֔וֹ וְאֶ֖ל כָּל-אֲשֶׁ֣ר עִמּ֑וֹ הָסִ֜רוּ אֶת-אֱלֹהֵ֤י הַנֵּכָר֙ אֲשֶׁ֣ר בְּתֹכְכֶ֔ם וְהִֽטַּהֲר֔וּ וְהַחֲלִ֖יפוּ שִׂמְלֹתֵיכֶֽם:  ת  וַאֲמַר יַעֲקֹב לֶאֱנַשׁ בֵּיתֵהּ וּלְכֹל דִּי עִמֵּהּ אַעְדוּ יָת טַעֲוַת עַמְמַיָּא דִי בֵינֵיכוֹן וְאִדַכּוּ וְשַׁנּוּ כְּסוּתְכוֹן:

{ג} וְנָק֥וּמָה וְנַעֲלֶ֖ה בֵּֽית-אֵ֑ל וְאֶֽעֱשֶׂה-שָּׁ֣ם מִזְבֵּ֗חַ לָאֵ֞ל הָעֹנֶ֤ה אֹתִי֙ בְּי֣וֹם צָֽרָתִ֔י וַיְהִי֙ עִמָּדִ֔י בַּדֶּ֖רֶךְ אֲשֶׁ֥ר הָלָֽכְתִּי:  מ  וְנָק֥וּמָה וְנַעֲלֶ֖ה בֵּֽית-אֵ֑ל וְאֶֽעֱשֶׂה-שָּׁ֣ם מִזְבֵּ֗חַ לָאֵ֞ל הָעֹנֶ֤ה אֹתִי֙ בְּי֣וֹם צָֽרָתִ֔י וַיְהִי֙ עִמָּדִ֔י בַּדֶּ֖רֶךְ אֲשֶׁ֥ר הָלָֽכְתִּי:  ת  וְנֵקוֹם וְנִסַּק לְבֵית אֵל וְאֶעְבֵּד תַּמָּן מַדְבְּחָא לֵאלָהָא דְקַבִּיל צְלוֹתִי בְּיוֹמָא דְעַקְתִי וַהֲוָה מֵימְרֵהּ בְּסַעְדִי בְּאָרְחָא דִי אֲזָלִית:

{ד} וַיִּתְּנ֣וּ אֶֽל-יַעֲקֹ֗ב אֵ֣ת כָּל-אֱלֹהֵ֤י הַנֵּכָר֙ אֲשֶׁ֣ר בְּיָדָ֔ם וְאֶת-הַנְּזָמִ֖ים אֲשֶׁ֣ר בְּאָזְנֵיהֶ֑ם וַיִּטְמֹ֤ן אֹתָם֙ יַעֲקֹ֔ב תַּ֥חַת הָאֵלָ֖ה אֲשֶׁ֥ר עִם-שְׁכֶֽם:  מ  וַיִּתְּנ֣וּ אֶֽל-יַעֲקֹ֗ב אֵ֣ת כָּל-אֱלֹהֵ֤י הַנֵּכָר֙ אֲשֶׁ֣ר בְּיָדָ֔ם וְאֶת-הַנְּזָמִ֖ים אֲשֶׁ֣ר בְּאָזְנֵיהֶ֑ם וַיִּטְמֹ֤ן אֹתָם֙ יַעֲקֹ֔ב תַּ֥חַת הָאֵלָ֖ה אֲשֶׁ֥ר עִם-שְׁכֶֽם:  ת  וִיהָבוּ לְיַעֲקֹב יָת כָּל טַעֲוַת עַמְמַיָּא דִי בִידֵיהוֹן וְיָת קָדָשַׁיָּא דִי בְאֻדְנֵיהוֹן וְטַּמַּר יָתְהוֹן יַעֲקֹב תְּחוֹת בָּטְמָא דִי עִם שְׁכֶם:

{ה} וַיִּסָּ֑עוּ וַיְהִ֣י | חִתַּ֣ת אֱלֹהִ֗ים עַל-הֶֽעָרִים֙ אֲשֶׁר֙ סְבִיבֹ֣תֵיהֶ֔ם וְלֹ֣א רָֽדְפ֔וּ אַחֲרֵ֖י בְּנֵ֥י יַעֲקֹֽב:  מ  וַיִּסָּ֑עוּ וַיְהִ֣י | חִתַּ֣ת אֱלֹהִ֗ים עַל-הֶֽעָרִים֙ אֲשֶׁר֙ סְבִיבֹ֣תֵיהֶ֔ם וְלֹ֣א רָֽדְפ֔וּ אַחֲרֵ֖י בְּנֵ֥י יַעֲקֹֽב:  ת  וּנְטָלוּ וַהֲוַת דַּחֲלָא דַיְיָ עַל עַמְמַיָא דִי בְּקִרְוֵי סַחֲרָנֵיהוֹן וְלָא רְדָפוּ בָּתַר בְּנֵי יַעֲקֹב:

{ו} וַיָּבֹ֨א יַעֲקֹ֜ב ל֗וּזָה אֲשֶׁר֙ בְּאֶ֣רֶץ כְּנַ֔עַן הִ֖וא בֵּֽית-אֵ֑ל ה֖וּא וְכָל-הָעָ֥ם אֲשֶׁר-עִמּֽוֹ:  מ  וַיָּבֹ֨א יַעֲקֹ֜ב ל֗וּזָה אֲשֶׁר֙ בְּאֶ֣רֶץ כְּנַ֔עַן הִ֖וא בֵּֽית-אֵ֑ל ה֖וּא וְכָל-הָעָ֥ם אֲשֶׁר-עִמּֽוֹ:  ת  וַאֲתָא יַעֲקֹב לְלוּז דִּי בְאַרְעָא דִכְנַעַן הִיא בֵּית אֵל הוּא וְכָל עַמָּא דִי עִמֵּהּ:

{ז} וַיִּ֤בֶן שָׁם֙ מִזְבֵּ֔חַ וַיִּקְרָא֙ לַמָּק֔וֹם אֵ֖ל בֵּֽית-אֵ֑ל כִּ֣י שָׁ֗ם נִגְל֤וּ אֵלָיו֙ הָֽאֱלֹהִ֔ים בְּבָרְח֖וֹ מִפְּנֵ֥י אָחִֽיו:  מ  וַיִּ֤בֶן שָׁם֙ מִזְבֵּ֔חַ וַיִּקְרָא֙ לַמָּק֔וֹם אֵ֖ל בֵּֽית-אֵ֑ל כִּ֣י שָׁ֗ם נִגְל֤וּ אֵלָיו֙ הָֽאֱלֹהִ֔ים בְּבָרְח֖וֹ מִפְּנֵ֥י אָחִֽיו:  ת  וּבְנָא תַמָּן מַדְבְּחָא וּקְרָא לְאַתְרָא אֵל בֵּית אֵל אֲרֵי תַמָּן אִתְגְלֵי לֵהּ יְיָ בְּמֵעִרְקֵהּ מִן קֳדָם אָחוּהִי:

{ח} וַתָּ֤מָת דְּבֹרָה֙ מֵינֶ֣קֶת רִבְקָ֔ה וַתִּקָּבֵ֛ר מִתַּ֥חַת לְבֵֽית-אֵ֖ל תַּ֣חַת הָֽאַלּ֑וֹן וַיִּקְרָ֥א שְׁמ֖וֹ אַלּ֥וֹן בָּכֽוּת:  מ  וַתָּ֤מָת דְּבֹרָה֙ מֵינֶ֣קֶת רִבְקָ֔ה וַתִּקָּבֵ֛ר מִתַּ֥חַת לְבֵֽית-אֵ֖ל תַּ֣חַת הָֽאַלּ֑וֹן וַיִּקְרָ֥א שְׁמ֖וֹ אַלּ֥וֹן בָּכֽוּת:  ת  וּמִיתַת דְּבוֹרָה מֵנִקְתָּא דְרִבְקָה וְאִתְקְבָרַת מִלְרַע לְבֵית אֵל בְּשִׁפּוֹלֵי מֵישְׁרָא וּקְרָא שְׁמֵּהּ מֵישַׁר בָּכִיתָא: (פ)

{ט} וַיֵּרָ֨א אֱלֹהִ֤ים אֶֽל-יַעֲקֹב֙ ע֔וֹד בְּבֹא֖וֹ מִפַּדַּ֣ן אֲרָ֑ם וַיְבָ֖רֶךְ אֹתֽוֹ:  מ  וַיֵּרָ֨א אֱלֹהִ֤ים אֶֽל-יַעֲקֹב֙ ע֔וֹד בְּבֹא֖וֹ מִפַּדַּ֣ן אֲרָ֑ם וַיְבָ֖רֶךְ אֹתֽוֹ:  ת  וְאִתְגְלִי יְיָ לְיַעֲקֹב עוֹד בְּמֵיתוֹהִי מִפַּדַן אֲרָם וּבָרִיךְ יָתֵהּ:

{י} וַיֹּֽאמֶר-ל֥וֹ אֱלֹהִ֖ים שִׁמְךָ֣ יַעֲקֹ֑ב לֹֽא-יִקָּרֵא֩ שִׁמְךָ֨ ע֜וֹד יַעֲקֹ֗ב כִּ֤י אִם-יִשְׂרָאֵל֙ יִהְיֶ֣ה שְׁמֶ֔ךָ וַיִּקְרָ֥א אֶת-שְׁמ֖וֹ יִשְׂרָאֵֽל:  מ  וַיֹּֽאמֶר-ל֥וֹ אֱלֹהִ֖ים שִׁמְךָ֣ יַעֲקֹ֑ב לֹֽא-יִקָּרֵא֩ שִׁמְךָ֨ ע֜וֹד יַעֲקֹ֗ב כִּ֤י אִם-יִשְׂרָאֵל֙ יִהְיֶ֣ה שְׁמֶ֔ךָ וַיִּקְרָ֥א אֶת-שְׁמ֖וֹ יִשְׂרָאֵֽל:  ת  וַאֲמַר לֵהּ יְיָ שְׁמָךְ יַעֲקֹב לָא יִתְקְרֵי שְׁמָךְ עוֹד יַעֲקֹב אֶלָהֵין יִשְׂרָאֵל יְהֵא שְׁמָךְ וּקְרָא יָת שְׁמֵהּ יִשְׂרָאֵל:

{יא} וַיֹּאמֶר֩ ל֨וֹ אֱלֹהִ֜ים אֲנִ֨י אֵ֤ל שַׁדַּי֙ פְּרֵ֣ה וּרְבֵ֔ה גּ֛וֹי וּקְהַ֥ל גּוֹיִ֖ם יִהְיֶ֣ה מִמֶּ֑ךָּ וּמְלָכִ֖ים מֵחֲלָצֶ֥יךָ יֵצֵֽאוּ:  מ  וַיֹּאמֶר֩ ל֨וֹ אֱלֹהִ֜ים אֲנִ֨י אֵ֤ל שַׁדַּי֙ פְּרֵ֣ה וּרְבֵ֔ה גּ֛וֹי וּקְהַ֥ל גּוֹיִ֖ם יִהְיֶ֣ה מִמֶּ֑ךָּ וּמְלָכִ֖ים מֵחֲלָצֶ֥יךָ יֵצֵֽאוּ:  ת  וַאֲמַר לֵהּ יְיָ אֲנָא אֵל שַׁדַּי פּוּשׁ וּסְגֵי עַם וְכִנְשַׁת שִׁבְטִּין יְהֵא מִנָּךְ וּמַלְכִין דְשַׁלִּיטִין בְּעַמְמַיָּא מִנָּךְ יִפְקוּן:

{יב}  שישי  וְאֶת-הָאָ֗רֶץ אֲשֶׁ֥ר נָתַ֛תִּי לְאַבְרָהָ֥ם וּלְיִצְחָ֖ק לְךָ֣ אֶתְּנֶ֑נָּה וּֽלְזַרְעֲךָ֥ אַחֲרֶ֖יךָ אֶתֵּ֥ן אֶת-הָאָֽרֶץ:  מ  וְאֶת-הָאָ֗רֶץ אֲשֶׁ֥ר נָתַ֛תִּי לְאַבְרָהָ֥ם וּלְיִצְחָ֖ק לְךָ֣ אֶתְּנֶ֑נָּה וּֽלְזַרְעֲךָ֥ אַחֲרֶ֖יךָ אֶתֵּ֥ן אֶת-הָאָֽרֶץ:  ת  וְיָת אַרְעָא דִי יְהָבִית לְאַבְרָהָם וּלְיִצְחָק לָךְ אֶתְּנִנַּהּ וְלִבְנָיךְ בַּתְרָךְ אֶתֵּן יָת אַרְעָא:

{יג} וַיַּ֥עַל מֵעָלָ֖יו אֱלֹהִ֑ים בַּמָּק֖וֹם אֲשֶׁר-דִּבֶּ֥ר אִתּֽוֹ:  מ  וַיַּ֥עַל מֵעָלָ֖יו אֱלֹהִ֑ים בַּמָּק֖וֹם אֲשֶׁר-דִּבֶּ֥ר אִתּֽוֹ:  ת  וְאִסְתַּלַּק מֵעִלָּוֹהִי יְקָרָא דַיְיָ בְּאַתְרָא דִי מַלֵּיל עִמֵּהּ:

{יד} וַיַּצֵּ֨ב יַעֲקֹ֜ב מַצֵּבָ֗ה בַּמָּק֛וֹם אֲשֶׁר-דִּבֶּ֥ר אִתּ֖וֹ מַצֶּ֣בֶת אָ֑בֶן וַיַּסֵּ֤ךְ עָלֶ֙יהָ֙ נֶ֔סֶךְ וַיִּצֹ֥ק עָלֶ֖יהָ שָֽׁמֶן:  מ  וַיַּצֵּ֨ב יַעֲקֹ֜ב מַצֵּבָ֗ה בַּמָּק֛וֹם אֲשֶׁר-דִּבֶּ֥ר אִתּ֖וֹ מַצֶּ֣בֶת אָ֑בֶן וַיַּסֵּ֤ךְ עָלֶ֙יהָ֙ נֶ֔סֶךְ וַיִּצֹ֥ק עָלֶ֖יהָ שָֽׁמֶן:  ת  וַאֲקֵם יַעֲקֹב קַמְתָא בְּאַתְרָא דִי מַלִּיל עִמֵּהּ קָמַת אַבְנָא וְאַסִּיךְ עֲלַהּ נִסּוּכִין וַאֲרִיק עֲלַהּ מִשְׁחָא:

{טו} וַיִּקְרָ֨א יַעֲקֹ֜ב אֶת-שֵׁ֣ם הַמָּק֗וֹם אֲשֶׁר֩ דִּבֶּ֨ר אִתּ֥וֹ שָׁ֛ם אֱלֹהִ֖ים בֵּֽית-אֵֽל:  מ  וַיִּקְרָ֨א יַעֲקֹ֜ב אֶת-שֵׁ֣ם הַמָּק֗וֹם אֲשֶׁר֩ דִּבֶּ֨ר אִתּ֥וֹ שָׁ֛ם אֱלֹהִ֖ים בֵּֽית-אֵֽל:  ת  וּקְרָא יַעֲקֹב יָת שְׁמָא דְאַתְרָא דִי מַלִּיל עִמֵּהּ תַּמָּן יְיָ בֵּית אֵל:

{טז} וַיִּסְעוּ֙ מִבֵּ֣ית אֵ֔ל וַֽיְהִי-ע֥וֹד כִּבְרַת-הָאָ֖רֶץ לָב֣וֹא אֶפְרָ֑תָה וַתֵּ֥לֶד רָחֵ֖ל וַתְּקַ֥שׁ בְּלִדְתָּֽהּ:  מ  וַיִּסְעוּ֙ מִבֵּ֣ית אֵ֔ל וַֽיְהִי-ע֥וֹד כִּבְרַת-הָאָ֖רֶץ לָב֣וֹא אֶפְרָ֑תָה וַתֵּ֥לֶד רָחֵ֖ל וַתְּקַ֥שׁ בְּלִדְתָּֽהּ:  ת  וּנְטָלוּ מִבֵּית אֵל וַהֲוָה עוֹד כְּרוּבַת אַרְעָא לְמֵיעַל לְאֶפְרָת וִילִידַת רָחֵל וְקַשִׁיאַת בְּמֵילְדַהּ:

{יז} וַיְהִ֥י בְהַקְשֹׁתָ֖הּ בְּלִדְתָּ֑הּ וַתֹּ֨אמֶר לָ֤הּ הַמְיַלֶּ֙דֶת֙ אַל-תִּ֣ירְאִ֔י כִּֽי-גַם-זֶ֥ה לָ֖ךְ בֵּֽן:  מ  וַיְהִ֥י בְהַקְשֹׁתָ֖הּ בְּלִדְתָּ֑הּ וַתֹּ֨אמֶר לָ֤הּ הַמְיַלֶּ֙דֶת֙ אַל-תִּ֣ירְאִ֔י כִּֽי-גַם-זֶ֥ה לָ֖ךְ בֵּֽן:  ת  וַהֲוָה בְקַשְׁיוּתַהּ בְּמֵילְדַהּ וַאֲמֶרֶת לַהּ חָיְתָא לָא תִדְחֲלִי אֲרֵי אַף דֵּין לִיךְ בָּר:

{יח} וַיְהִ֞י בְּצֵ֤את נַפְשָׁהּ֙ כִּ֣י מֵ֔תָה וַתִּקְרָ֥א שְׁמ֖וֹ בֶּן-אוֹנִ֑י וְאָבִ֖יו קָֽרָא-ל֥וֹ בִנְיָמִֽין:  מ  וַיְהִ֞י בְּצֵ֤את נַפְשָׁהּ֙ כִּ֣י מֵ֔תָה וַתִּקְרָ֥א שְׁמ֖וֹ בֶּן-אוֹנִ֑י וְאָבִ֖יו קָֽרָא-ל֥וֹ בִנְיָמִֽין:  ת  וַהֲוָה בְּמִפַּק נַפְשַׁהּ אֲרֵי מִיתַת (נ''י מַיְתָא) וּקְרַת שְׁמֵהּ בַּר דְוָי וַאֲבוּהִי קְרָא לֵהּ בִּנְיָמִין:

{יט} וַתָּ֖מָת רָחֵ֑ל וַתִּקָּבֵר֙ בְּדֶ֣רֶךְ אֶפְרָ֔תָה הִ֖וא בֵּ֥ית לָֽחֶם:  מ  וַתָּ֖מָת רָחֵ֑ל וַתִּקָּבֵר֙ בְּדֶ֣רֶךְ אֶפְרָ֔תָה הִ֖וא בֵּ֥ית לָֽחֶם:  ת  וּמִיתַת רָחֵל וְאִתְקְבָרַת בְּאֹרַח אֶפְרָת הִיא בֵּית לָחֶם:

{כ} וַיַּצֵּ֧ב יַעֲקֹ֛ב מַצֵּבָ֖ה עַל-קְבֻרָתָ֑הּ הִ֛וא מַצֶּ֥בֶת קְבֻרַֽת-רָחֵ֖ל עַד-הַיּֽוֹם:  מ  וַיַּצֵּ֧ב יַעֲקֹ֛ב מַצֵּבָ֖ה עַל-קְבֻרָתָ֑הּ הִ֛וא מַצֶּ֥בֶת קְבֻרַֽת-רָחֵ֖ל עַד-הַיּֽוֹם:  ת  וַאֲקֵים יַעֲקֹב קַמְתָא עַל קְבֻרְתַּהּ הִיא קָמַת קְבֻרְתָּא דְרָחֵל עַד יוֹמָא דֵין:

{כא} וַיִּסַּ֖ע יִשְׂרָאֵ֑ל וַיֵּ֣ט אָֽהֳלֹ֔ה מֵהָ֖לְאָה לְמִגְדַּל-עֵֽדֶר:  מ  וַיִּסַּ֖ע יִשְׂרָאֵ֑ל וַיֵּ֣ט אָֽהֳלֹ֔ה מֵהָ֖לְאָה לְמִגְדַּל-עֵֽדֶר:  ת  וּנְטַל יִשְׂרָאֵל וּפְרַס מַשְׁכְּנֵהּ מִלְהַלָא לְמִגְדְלָא דְעֵדֶר:

{כב}  וַיְהִ֗י בִּשְׁכֹּ֤ן יִשְׂרָאֵל֙ בָּאָ֣רֶץ הַהִ֔וא וַיֵּ֣לֶךְ רְאוּבֵ֔ן וַיִּשְׁכַּ֕ב֙ אֶת בִּלְהָ֖ה֙ פִּילֶ֣גֶשׁ אָבִ֑֔יו וַיִּשְׁמַ֖ע יִשְׂרָאֵֽ֑ל וַיִּֽהְי֥וּ בְנֵֽי יַעֲקֹ֖ב שְׁנֵ֥ים עָשָֽׂר:  מ   וַיְהִ֗י בִּשְׁכֹּ֤ן יִשְׂרָאֵל֙ בָּאָ֣רֶץ הַהִ֔וא וַיֵּ֣לֶךְ רְאוּבֵ֔ן וַיִּשְׁכַּ֕ב֙ אֶת בִּלְהָ֖ה֙ פִּילֶ֣גֶשׁ אָבִ֑֔יו וַיִּשְׁמַ֖ע יִשְׂרָאֵֽ֑ל וַיִּֽהְי֥וּ בְנֵֽי יַעֲקֹ֖ב שְׁנֵ֥ים עָשָֽׂר:  ת   וַהֲוָה כַּד שְׁרָא יִשְׂרָאֵל בְּאַרְעָא הַהִיא וַאֲזַל רְאוּבֵן וּשְׁכִיב עִם בִּלְהָה לְחֵינָתָא דַאֲבוּהִי וּשְׁמַע יִשְׂרָאֵל וַהֲווֹ בְנֵי יַעֲקֹב תְּרֵי עֲסַר:

{כג} בְּנֵ֣י לֵאָ֔ה בְּכ֥וֹר יַעֲקֹ֖ב רְאוּבֵ֑ן וְשִׁמְעוֹן֙ וְלֵוִ֣י וִֽיהוּדָ֔ה וְיִשָּׂשכָ֖ר וּזְבוּלֻֽן:  מ  בְּנֵ֣י לֵאָ֔ה בְּכ֥וֹר יַעֲקֹ֖ב רְאוּבֵ֑ן וְשִׁמְעוֹן֙ וְלֵוִ֣י וִֽיהוּדָ֔ה וְיִשָּׂשכָ֖ר וּזְבוּלֻֽן:  ת  בְּנֵי לֵאָה בּוּכְרָא דְיַעֲקֹב רְאוּבֵן וְשִׁמְעוֹן וְלֵוִי וִיהוּדָה וְיִשָׂשׂכָר וּזְבֻלוּן:

{כד} בְּנֵ֣י רָחֵ֔ל יוֹסֵ֖ף וּבִנְיָמִֽן:  מ  בְּנֵ֣י רָחֵ֔ל יוֹסֵ֖ף וּבִנְיָמִֽן:  ת  בְּנֵי רָחֵל יוֹסֵף וּבִנְיָמִן:

{כה} וּבְנֵ֤י בִלְהָה֙ שִׁפְחַ֣ת רָחֵ֔ל דָּ֖ן וְנַפְתָּלִֽי:  מ  וּבְנֵ֤י בִלְהָה֙ שִׁפְחַ֣ת רָחֵ֔ל דָּ֖ן וְנַפְתָּלִֽי:  ת  וּבְנֵי בִלְהָה אַמְתָא דְרָחֵל דָּן וְנַפְתָּלִי:

{כו} וּבְנֵ֥י זִלְפָּ֛ה שִׁפְחַ֥ת לֵאָ֖ה גָּ֣ד וְאָשֵׁ֑ר אֵ֚לֶּה בְּנֵ֣י יַעֲקֹ֔ב אֲשֶׁ֥ר יֻלַּד-ל֖וֹ בְּפַדַּ֥ן אֲרָֽם:  מ  וּבְנֵ֥י זִלְפָּ֛ה שִׁפְחַ֥ת לֵאָ֖ה גָּ֣ד וְאָשֵׁ֑ר אֵ֚לֶּה בְּנֵ֣י יַעֲקֹ֔ב אֲשֶׁ֥ר יֻלַּד-ל֖וֹ בְּפַדַּ֥ן אֲרָֽם:  ת  וּבְנֵי זִלְפָּה אַמְתָא דְלֵאָה גָּד וְאָשֵׁר אִלֵּין בְּנֵי יַעֲקֹב דִּי אִתְיְלִידוּ לֵהּ בְּפַדַן אֲרָם:

{כז} וַיָּבֹ֤א יַעֲקֹב֙ אֶל-יִצְחָ֣ק אָבִ֔יו מַמְרֵ֖א קִרְיַ֣ת הָֽאַרְבַּ֑ע הִ֣וא חֶבְר֔וֹן אֲשֶׁר-גָּֽר-שָׁ֥ם אַבְרָהָ֖ם וְיִצְחָֽק:  מ  וַיָּבֹ֤א יַעֲקֹב֙ אֶל-יִצְחָ֣ק אָבִ֔יו מַמְרֵ֖א קִרְיַ֣ת הָֽאַרְבַּ֑ע הִ֣וא חֶבְר֔וֹן אֲשֶׁר-גָּֽר-שָׁ֥ם אַבְרָהָ֖ם וְיִצְחָֽק:  ת  וַאֲתָא יַעֲקֹב לְוַת יִצְחָק אֲבוּהִי מַמְרֵא קִרְיַת אַרְבַּע הִיא חֶבְרוֹן דִּי דָר תַּמָּן אַבְרָהָם וְיִצְחָק:

{כח} וַיִּֽהְי֖וּ יְמֵ֣י יִצְחָ֑ק מְאַ֥ת שָׁנָ֖ה וּשְׁמֹנִ֥ים שָׁנָֽה:  מ  וַיִּֽהְי֖וּ יְמֵ֣י יִצְחָ֑ק מְאַ֥ת שָׁנָ֖ה וּשְׁמֹנִ֥ים שָׁנָֽה:  ת  וַהֲווֹ יוֹמֵי יִצְחָק מְאָה וּתְמָנָן שְׁנִין:

{כט} וַיִּגְוַ֨ע יִצְחָ֤ק וַיָּ֙מָת֙ וַיֵּאָ֣סֶף אֶל-עַמָּ֔יו זָקֵ֖ן וּשְׂבַ֣ע יָמִ֑ים וַיִּקְבְּר֣וּ אֹת֔וֹ עֵשָׂ֥ו וְיַעֲקֹ֖ב בָּנָֽיו:  מ  וַיִּגְוַ֨ע יִצְחָ֤ק וַיָּ֙מָת֙ וַיֵּאָ֣סֶף אֶל-עַמָּ֔יו זָקֵ֖ן וּשְׂבַ֣ע יָמִ֑ים וַיִּקְבְּר֣וּ אֹת֔וֹ עֵשָׂ֥ו וְיַעֲקֹ֖ב בָּנָֽיו:  ת  וְאִתְנְגִיד יִצְחָק וּמִית וְאִתְכְּנִישׁ לְעַמֵהּ סִיב וּשְׂבַע יוֹמִין וּקְבָרוּ יָתֵהּ עֵשָׂו וְיַעֲקֹב בְּנוֹהִי: (פ)




בראשית פרק-לו

{א} וְאֵ֛לֶּה תֹּלְד֥וֹת עֵשָׂ֖ו ה֥וּא אֱדֽוֹם:  מ  וְאֵ֛לֶּה תֹּלְד֥וֹת עֵשָׂ֖ו ה֥וּא אֱדֽוֹם:  ת  וְאִלֵּין תּוּלְדַת עֵשָׂו הוּא אֱדוֹם:

{ב} עֵשָׂ֛ו לָקַ֥ח אֶת-נָשָׁ֖יו מִבְּנ֣וֹת כְּנָ֑עַן אֶת-עָדָ֗ה בַּת-אֵילוֹן֙ הַֽחִתִּ֔י וְאֶת-אָהֳלִֽיבָמָה֙ בַּת-עֲנָ֔ה בַּת-צִבְע֖וֹן הַֽחִוִּֽי:  מ  עֵשָׂ֛ו לָקַ֥ח אֶת-נָשָׁ֖יו מִבְּנ֣וֹת כְּנָ֑עַן אֶת-עָדָ֗ה בַּת-אֵילוֹן֙ הַֽחִתִּ֔י וְאֶת-אָהֳלִֽיבָמָה֙ בַּת-עֲנָ֔ה בַּת-צִבְע֖וֹן הַֽחִוִּֽי:  ת  עֵשָׂו נְסִיב יָת נְשׁוֹהִי מִבְּנַת כְּנָעַן יָת עָדָה בַּת אֵילוֹן חִתָּאָה וְיָת אָהֳלִיבָמָה בַּת עֲנָה בַּת צִבְעוֹן חִוָאָה:

{ג} וְאֶת-בָּשְׂמַ֥ת בַּת-יִשְׁמָעֵ֖אל אֲח֥וֹת נְבָיֽוֹת:  מ  וְאֶת-בָּשְׂמַ֥ת בַּת-יִשְׁמָעֵ֖אל אֲח֥וֹת נְבָיֽוֹת:  ת  וְיָת בָּשְׂמַת בַּת יִשְׁמָעֵאל אֲחָתֵהּ דִנְבָיוֹת:

{ד} וַתֵּ֧לֶד עָדָ֛ה לְעֵשָׂ֖ו אֶת-אֱלִיפָ֑ז וּבָ֣שְׂמַ֔ת יָלְדָ֖ה אֶת-רְעוּאֵֽל:  מ  וַתֵּ֧לֶד עָדָ֛ה לְעֵשָׂ֖ו אֶת-אֱלִיפָ֑ז וּבָ֣שְׂמַ֔ת יָלְדָ֖ה אֶת-רְעוּאֵֽל:  ת  וִילִידַת עָדָה לְעֵשָׂו יָת אֱלִיפָז וּבָשְׂמַת יְלִידַת יָת רְעוּאֵל:

{ה} וְאָהֳלִֽיבָמָה֙ יָֽלְדָ֔ה אֶת-(יעיש) יְע֥וּשׁ וְאֶת-יַעְלָ֖ם וְאֶת-קֹ֑רַח אֵ֚לֶּה בְּנֵ֣י עֵשָׂ֔ו אֲשֶׁ֥ר יֻלְּדוּ-ל֖וֹ בְּאֶ֥רֶץ כְּנָֽעַן:  מ  וְאָהֳלִֽיבָמָה֙ יָֽלְדָ֔ה אֶת-(יעיש) יְע֥וּשׁ וְאֶת-יַעְלָ֖ם וְאֶת-קֹ֑רַח אֵ֚לֶּה בְּנֵ֣י עֵשָׂ֔ו אֲשֶׁ֥ר יֻלְּדוּ-ל֖וֹ בְּאֶ֥רֶץ כְּנָֽעַן:  ת  וְאָהֳלִיבָמָה יְלִידַת יָת יְעוּשׁ וְיָת יַעְלָם וְיָת קֹרַח אִלֵּין בְּנֵי עֵשָׂו דִי אִתְיְלִידוּ לֵהּ בְּאַרְעָא דִכְנָעַן:

{ו} וַיִּקַּ֣ח עֵשָׂ֡ו אֶת-נָ֠שָׁיו וְאֶת-בָּנָ֣יו וְאֶת-בְּנֹתָיו֮ וְאֶת-כָּל-נַפְשׁ֣וֹת בֵּיתוֹ֒ וְאֶת-מִקְנֵ֣הוּ וְאֶת-כָּל-בְּהֶמְתּ֗וֹ וְאֵת֙ כָּל-קִנְיָנ֔וֹ אֲשֶׁ֥ר רָכַ֖שׁ בְּאֶ֣רֶץ כְּנָ֑עַן וַיֵּ֣לֶךְ אֶל-אֶ֔רֶץ מִפְּנֵ֖י יַעֲקֹ֥ב אָחִֽיו:  מ  וַיִּקַּ֣ח עֵשָׂ֡ו אֶת-נָ֠שָׁיו וְאֶת-בָּנָ֣יו וְאֶת-בְּנֹתָיו֮ וְאֶת-כָּל-נַפְשׁ֣וֹת בֵּיתוֹ֒ וְאֶת-מִקְנֵ֣הוּ וְאֶת-כָּל-בְּהֶמְתּ֗וֹ וְאֵת֙ כָּל-קִנְיָנ֔וֹ אֲשֶׁ֥ר רָכַ֖שׁ בְּאֶ֣רֶץ כְּנָ֑עַן וַיֵּ֣לֶךְ אֶל-אֶ֔רֶץ מִפְּנֵ֖י יַעֲקֹ֥ב אָחִֽיו:  ת  וּדְבַר עֵשָׂו יָת נְשׁוֹהִי וְיָת בְּנוֹהִי וְיָת בְּנָתֵהּ וְיָת כָּל נַפְשַׁת בֵּיתֵהּ וְיָת גֵּיתוֹהִי וְיָת כָּל בְּעִירֵהּ וְיָת כָּל קִנְיָנֵהּ דִי קְנָא בְּאַרְעָא דִכְנָעַן וַאֲזַל לְאַרְעָא אוֹחֲרִי מִן קֳדָם יַעֲקֹב אָחוּהִי:

{ז} כִּֽי-הָיָ֧ה רְכוּשָׁ֛ם רָ֖ב מִשֶּׁ֣בֶת יַחְדָּ֑ו וְלֹ֨א יָֽכְלָ֜ה אֶ֤רֶץ מְגֽוּרֵיהֶם֙ לָשֵׂ֣את אֹתָ֔ם מִפְּנֵ֖י מִקְנֵיהֶֽם:  מ  כִּֽי-הָיָ֧ה רְכוּשָׁ֛ם רָ֖ב מִשֶּׁ֣בֶת יַחְדָּ֑ו וְלֹ֨א יָֽכְלָ֜ה אֶ֤רֶץ מְגֽוּרֵיהֶם֙ לָשֵׂ֣את אֹתָ֔ם מִפְּנֵ֖י מִקְנֵיהֶֽם:  ת  אֲרֵי הֲוָה קִנְיָנְהוֹן סַגִּי מִלְמִתַּב כַּחֲדָא וְלָא יְכִילַת אֲרַע תּוֹתָבוּתְהוֹן לְסוֹבָרָא יָתְהוֹן מִן קֳדָם גֵּיתֵיהוֹן:

{ח} וַיֵּ֤שֶׁב עֵשָׂו֙ בְּהַ֣ר שֵׂעִ֔יר עֵשָׂ֖ו ה֥וּא אֱדֽוֹם:  מ  וַיֵּ֤שֶׁב עֵשָׂו֙ בְּהַ֣ר שֵׂעִ֔יר עֵשָׂ֖ו ה֥וּא אֱדֽוֹם:  ת  וִיתֵיב עֵשָׂו בְּטוּרָא דְשֵׂעִיר עֵשָׂו הוּא אֱדוֹם:

{ט} וְאֵ֛לֶּה תֹּלְד֥וֹת עֵשָׂ֖ו אֲבִ֣י אֱד֑וֹם בְּהַ֖ר שֵׂעִֽיר:  מ  וְאֵ֛לֶּה תֹּלְד֥וֹת עֵשָׂ֖ו אֲבִ֣י אֱד֑וֹם בְּהַ֖ר שֵׂעִֽיר:  ת  וְאִלֵּין תּוּלְדַת עֵשָׂו אֲבוּהוֹן דֶאֱדוֹמָאֵי בְּטוּרָא דְשֵׂעִיר:

{י} אֵ֖לֶּה שְׁמ֣וֹת בְּנֵֽי-עֵשָׂ֑ו אֱלִיפַ֗ז בֶּן-עָדָה֙ אֵ֣שֶׁת עֵשָׂ֔ו רְעוּאֵ֕ל בֶּן-בָּשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו:  מ  אֵ֖לֶּה שְׁמ֣וֹת בְּנֵֽי-עֵשָׂ֑ו אֱלִיפַ֗ז בֶּן-עָדָה֙ אֵ֣שֶׁת עֵשָׂ֔ו רְעוּאֵ֕ל בֶּן-בָּשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו:  ת  אִלֵּין שְׁמָהַת בְּנֵי עֵשָׂו אֱלִיפַז בַּר עָדָה אִתַּת עֵשָׂו רְעוּאֵל בַּר בָּשְׂמַת אִתַּת עֵשָׂו:

{יא} וַיִּהְי֖וּ בְּנֵ֣י אֱלִיפָ֑ז תֵּימָ֣ן אוֹמָ֔ר צְפ֥וֹ וְגַעְתָּ֖ם וּקְנַֽז:  מ  וַיִּהְי֖וּ בְּנֵ֣י אֱלִיפָ֑ז תֵּימָ֣ן אוֹמָ֔ר צְפ֥וֹ וְגַעְתָּ֖ם וּקְנַֽז:  ת  וַהֲווֹ בְּנֵי אֱלִיפָז תֵּימָן אוֹמָר צְפוֹ וְגַעְתָּם וּקְנַז:

{יב} וְתִמְנַ֣ע | הָיְתָ֣ה פִילֶ֗גֶשׁ לֶֽאֱלִיפַז֙ בֶּן-עֵשָׂ֔ו וַתֵּ֥לֶד לֶאֱלִיפַ֖ז אֶת-עֲמָלֵ֑ק אֵ֕לֶּה בְּנֵ֥י עָדָ֖ה אֵ֥שֶׁת עֵשָֽׂו:  מ  וְתִמְנַ֣ע | הָיְתָ֣ה פִילֶ֗גֶשׁ לֶֽאֱלִיפַז֙ בֶּן-עֵשָׂ֔ו וַתֵּ֥לֶד לֶאֱלִיפַ֖ז אֶת-עֲמָלֵ֑ק אֵ֕לֶּה בְּנֵ֥י עָדָ֖ה אֵ֥שֶׁת עֵשָֽׂו:  ת  וְתִמְנַע הֲוַת לְחֵינָתָא לֶאֱלִיפַז בַּר עֵשָׂו וִילִידַת לֶאֱלִיפַז יָת עֲמָלֵק אִלֵּין בְּנֵי עָדָה אִתַּת עֵשָׂו:

{יג} וְאֵ֙לֶּה֙ בְּנֵ֣י רְעוּאֵ֔ל נַ֥חַת וָזֶ֖רַח שַׁמָּ֣ה וּמִזָּ֑ה אֵ֣לֶּ֣ה הָי֔וּ בְּנֵ֥י בָשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו:  מ  וְאֵ֙לֶּה֙ בְּנֵ֣י רְעוּאֵ֔ל נַ֥חַת וָזֶ֖רַח שַׁמָּ֣ה וּמִזָּ֑ה אֵ֣לֶּ֣ה הָי֔וּ בְּנֵ֥י בָשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו:  ת  וְאִלֵּין בְּנֵי רְעוּאֵל נַחַת וָזֶרַח שַׁמָּה וּמִזָה אִלֵּין הֲווֹ בְּנֵי בָשְׂמַת אִתַּת עֵשָׂו:

{יד} וְאֵ֣לֶּה הָי֗וּ בְּנֵ֨י אָהֳלִיבָמָ֧ה בַת-עֲנָ֛ה בַּת-צִבְע֖וֹן אֵ֣שֶׁת עֵשָׂ֑ו וַתֵּ֣לֶד לְעֵשָׂ֔ו אֶת-(יעיש) יְע֥וּשׁ וְאֶת-יַעְלָ֖ם וְאֶת-קֹֽרַח:  מ  וְאֵ֣לֶּה הָי֗וּ בְּנֵ֨י אָהֳלִיבָמָ֧ה בַת-עֲנָ֛ה בַּת-צִבְע֖וֹן אֵ֣שֶׁת עֵשָׂ֑ו וַתֵּ֣לֶד לְעֵשָׂ֔ו אֶת-(יעיש) יְע֥וּשׁ וְאֶת-יַעְלָ֖ם וְאֶת-קֹֽרַח:  ת  וְאִלֵּין הֲווֹ בְּנֵי אָהֳלִיבָמָה בַת עֲנָה בַּת צִבְעוֹן אִתַּת עֵשָׂו וִילִידַת לְעֵשָׂו יָת יְעוּשׁ וְיָת יַעְלָם וְיָת קֹרַח:

{טו} אֵ֖לֶּה אַלּוּפֵ֣י בְנֵֽי-עֵשָׂ֑ו בְּנֵ֤י אֱלִיפַז֙ בְּכ֣וֹר עֵשָׂ֔ו אַלּ֤וּף תֵּימָן֙ אַלּ֣וּף אוֹמָ֔ר אַלּ֥וּף צְפ֖וֹ אַלּ֥וּף קְנַֽז:  מ  אֵ֖לֶּה אַלּוּפֵ֣י בְנֵֽי-עֵשָׂ֑ו בְּנֵ֤י אֱלִיפַז֙ בְּכ֣וֹר עֵשָׂ֔ו אַלּ֤וּף תֵּימָן֙ אַלּ֣וּף אוֹמָ֔ר אַלּ֥וּף צְפ֖וֹ אַלּ֥וּף קְנַֽז:  ת  אִלֵּין רַבְרְבֵי בְנֵי עֵשָׂו בְּנֵי אֱלִיפַז בּוּכְרָא דְעֵשָׂו רַבָּא תֵּימָן רַבָּא אוֹמָר רַבָּא צְפוֹ רַבָּא קְנַז:

{טז} אַלּֽוּף-קֹ֛רַח אַלּ֥וּף גַּעְתָּ֖ם אַלּ֣וּף עֲמָלֵ֑ק אֵ֣לֶּה אַלּוּפֵ֤י אֱלִיפַז֙ בְּאֶ֣רֶץ אֱד֔וֹם אֵ֖לֶּה בְּנֵ֥י עָדָֽה:  מ  אַלּֽוּף-קֹ֛רַח אַלּ֥וּף גַּעְתָּ֖ם אַלּ֣וּף עֲמָלֵ֑ק אֵ֣לֶּה אַלּוּפֵ֤י אֱלִיפַז֙ בְּאֶ֣רֶץ אֱד֔וֹם אֵ֖לֶּה בְּנֵ֥י עָדָֽה:  ת  רַבָּא קֹרַח רַבָּא גַעְתָּם רַבָּא עֲמָלֵק אִלֵּין רַבְרְבֵי אֱלִיפַז בְּאַרְעָא דֶאֱדוֹם אִלֵּין בְּנֵי עָדָה:

{יז} וְאֵ֗לֶּה בְּנֵ֤י רְעוּאֵל֙ בֶּן-עֵשָׂ֔ו אַלּ֥וּף נַ֙חַת֙ אַלּ֣וּף זֶ֔רַח אַלּ֥וּף שַׁמָּ֖ה אַלּ֣וּף מִזָּ֑ה אֵ֣לֶּה אַלּוּפֵ֤י רְעוּאֵל֙ בְּאֶ֣רֶץ אֱד֔וֹם אֵ֕לֶּה בְּנֵ֥י בָשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו:  מ  וְאֵ֗לֶּה בְּנֵ֤י רְעוּאֵל֙ בֶּן-עֵשָׂ֔ו אַלּ֥וּף נַ֙חַת֙ אַלּ֣וּף זֶ֔רַח אַלּ֥וּף שַׁמָּ֖ה אַלּ֣וּף מִזָּ֑ה אֵ֣לֶּה אַלּוּפֵ֤י רְעוּאֵל֙ בְּאֶ֣רֶץ אֱד֔וֹם אֵ֕לֶּה בְּנֵ֥י בָשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו:  ת  וְאִלֵּין בְּנֵי רְעוּאֵל בַּר עֵשָׂו רַבָּא נַחַת רַבָּא זֶרַח רַבָּא שַׁמָה רַבָּא מִזָּה אִלֵּין רַבְרְבֵי רְעוּאֵל בְּאַרְעָא דֶאֱדוֹם אִלֵין בְּנֵי בָשְׂמַת אִתַּת עֵשָׂו:

{יח} וְאֵ֗לֶּה בְּנֵ֤י אָהֳלִֽיבָמָה֙ אֵ֣שֶׁת עֵשָׂ֔ו אַלּ֥וּף יְע֛וּשׁ אַלּ֥וּף יַעְלָ֖ם אַלּ֣וּף קֹ֑רַח אֵ֣לֶּה אַלּוּפֵ֞י אָֽהֳלִיבָמָ֛ה בַּת-עֲנָ֖ה אֵ֥שֶׁת עֵשָֽׂו:  מ  וְאֵ֗לֶּה בְּנֵ֤י אָהֳלִֽיבָמָה֙ אֵ֣שֶׁת עֵשָׂ֔ו אַלּ֥וּף יְע֛וּשׁ אַלּ֥וּף יַעְלָ֖ם אַלּ֣וּף קֹ֑רַח אֵ֣לֶּה אַלּוּפֵ֞י אָֽהֳלִיבָמָ֛ה בַּת-עֲנָ֖ה אֵ֥שֶׁת עֵשָֽׂו:  ת  וְאִלֵּין בְּנֵי אָהֳלִיבָמָה אִתַּת עֵשָׂו רַבָּא יְעוּשׁ רַבָּא יַעְלָם רַבָּא קֹרַח אִלֵּין רַבְרְבֵי אָהֳלִיבָמָה בַּת עֲנָה אִתַּת עֵשָׂו:

{יט} אֵ֧לֶּה בְנֵי-עֵשָׂ֛ו וְאֵ֥לֶּה אַלּוּפֵיהֶ֖ם ה֥וּא אֱדֽוֹם:  מ  אֵ֧לֶּה בְנֵי-עֵשָׂ֛ו וְאֵ֥לֶּה אַלּוּפֵיהֶ֖ם ה֥וּא אֱדֽוֹם:  ת  אִלֵּין בְּנֵי עֵשָׂו וְאִלֵּין רַבְרְבָנֵיהוֹן הוּא אֱדוֹם: (ס)

{כ}  שביעי  אֵ֤לֶּה בְנֵֽי-שֵׂעִיר֙ הַחֹרִ֔י יֹשְׁבֵ֖י הָאָ֑רֶץ לוֹטָ֥ן וְשׁוֹבָ֖ל וְצִבְע֥וֹן וַעֲנָֽה:  מ  אֵ֤לֶּה בְנֵֽי-שֵׂעִיר֙ הַחֹרִ֔י יֹשְׁבֵ֖י הָאָ֑רֶץ לוֹטָ֥ן וְשׁוֹבָ֖ל וְצִבְע֥וֹן וַעֲנָֽה:  ת  אִלֵּין בְּנֵי שֵׂעִיר חוֹרָאָה יָתְבֵי דְּאַרְעָא לוֹטָן וְשׁוֹבָל וְצִבְעוֹן וַעֲנָה:

{כא} וְדִשׁ֥וֹן וְאֵ֖צֶר וְדִישָׁ֑ן אֵ֣לֶּה אַלּוּפֵ֧י הַחֹרִ֛י בְּנֵ֥י שֵׂעִ֖יר בְּאֶ֥רֶץ אֱדֽוֹם:  מ  וְדִשׁ֥וֹן וְאֵ֖צֶר וְדִישָׁ֑ן אֵ֣לֶּה אַלּוּפֵ֧י הַחֹרִ֛י בְּנֵ֥י שֵׂעִ֖יר בְּאֶ֥רֶץ אֱדֽוֹם:  ת  וְדִישׁוֹן וְאֵצֶר וְדִישָׁן אִלֵּין רַבְרְבֵי חוֹרָאֵי בְּנֵי שֵׂעִיר בְּאַרְעָא דֶאֱדוֹם:

{כב} וַיִּהְי֥וּ בְנֵי-לוֹטָ֖ן חֹרִ֣י וְהֵימָ֑ם וַאֲח֥וֹת לוֹטָ֖ן תִּמְנָֽע:  מ  וַיִּהְי֥וּ בְנֵי-לוֹטָ֖ן חֹרִ֣י וְהֵימָ֑ם וַאֲח֥וֹת לוֹטָ֖ן תִּמְנָֽע:  ת  וַהֲווֹ בְנֵי לוֹטָן חוֹרִי וְהֵימָם וַאֲחָתֵהּ דְלוֹטָן תִּמְנָע:

{כג} וְאֵ֙לֶּה֙ בְּנֵ֣י שׁוֹבָ֔ל עַלְוָ֥ן וּמָנַ֖חַת וְעֵיבָ֑ל שְׁפ֖וֹ וְאוֹנָֽם:  מ  וְאֵ֙לֶּה֙ בְּנֵ֣י שׁוֹבָ֔ל עַלְוָ֥ן וּמָנַ֖חַת וְעֵיבָ֑ל שְׁפ֖וֹ וְאוֹנָֽם:  ת  וְאִלֵּין בְּנֵי שׁוֹבָל עַלְוָן וּמָנַחַת וְעֵיבָל שְׁפוֹ וְאוֹנָם:

{כד} וְאֵ֥לֶּה בְנֵֽי-צִבְע֖וֹן וְאַיָּ֣ה וַעֲנָ֑ה ה֣וּא עֲנָ֗ה אֲשֶׁ֨ר מָצָ֤א אֶת-הַיֵּמִם֙ בַּמִּדְבָּ֔ר בִּרְעֹת֥וֹ אֶת-הַחֲמֹרִ֖ים לְצִבְע֥וֹן אָבִֽיו:  מ  וְאֵ֥לֶּה בְנֵֽי-צִבְע֖וֹן וְאַיָּ֣ה וַעֲנָ֑ה ה֣וּא עֲנָ֗ה אֲשֶׁ֨ר מָצָ֤א אֶת-הַיֵּמִם֙ בַּמִּדְבָּ֔ר בִּרְעֹת֥וֹ אֶת-הַחֲמֹרִ֖ים לְצִבְע֥וֹן אָבִֽיו:  ת  וְאִלֵּין בְּנֵי צִבְעוֹן וְאַיָּה וַעֲנָה הוּא עֲנָה דִּי אַשְׁכַּח יָתּ גִּבָּרַיָּא בְּמַדְבְּרָא כַּד הֲוָה רָעֵי יָת חֲמָרַיָּא לְצִבְעוֹן אֲבוּהִי:

{כה} וְאֵ֥לֶּה בְנֵֽי-עֲנָ֖ה דִּשֹׁ֑ן וְאָהֳלִיבָמָ֖ה בַּת-עֲנָֽה:  מ  וְאֵ֥לֶּה בְנֵֽי-עֲנָ֖ה דִּשֹׁ֑ן וְאָהֳלִיבָמָ֖ה בַּת-עֲנָֽה:  ת  וְאִלֵּין בְּנֵי עֲנָה דִּשֹׁן וְאָהֳלִיבָמָה בַּת עֲנָה:

{כו} וְאֵ֖לֶּה בְּנֵ֣י דִישָׁ֑ן חֶמְדָּ֥ן וְאֶשְׁבָּ֖ן וְיִתְרָ֥ן וּכְרָֽן:  מ  וְאֵ֖לֶּה בְּנֵ֣י דִישָׁ֑ן חֶמְדָּ֥ן וְאֶשְׁבָּ֖ן וְיִתְרָ֥ן וּכְרָֽן:  ת  וְאִלֵּין בְּנֵי דִישָׁן חֶמְדָן וְאֶשְׁבָּן וְיִתְּרָן וּכְרָן:

{כז} אֵ֖לֶּה בְּנֵי-אֵ֑צֶר בִּלְהָ֥ן וְזַעֲוָ֖ן וַעֲקָֽן:  מ  אֵ֖לֶּה בְּנֵי-אֵ֑צֶר בִּלְהָ֥ן וְזַעֲוָ֖ן וַעֲקָֽן:  ת  אִלֵּין בְּנֵי אֵצֶר בִּלְהָן וְזַעֲוָן וַעֲקָן:

{כח} אֵ֥לֶּה בְנֵֽי-דִישָׁ֖ן ע֥וּץ וַאֲרָֽן:  מ  אֵ֥לֶּה בְנֵֽי-דִישָׁ֖ן ע֥וּץ וַאֲרָֽן:  ת  אִלֵּין בְּנֵי דִישָׁן עוּץ וַאֲרָן:

{כט} אֵ֖לֶּה אַלּוּפֵ֣י הַחֹרִ֑י אַלּ֤וּף לוֹטָן֙ אַלּ֣וּף שׁוֹבָ֔ל אַלּ֥וּף צִבְע֖וֹן אַלּ֥וּף עֲנָֽה:  מ  אֵ֖לֶּה אַלּוּפֵ֣י הַחֹרִ֑י אַלּ֤וּף לוֹטָן֙ אַלּ֣וּף שׁוֹבָ֔ל אַלּ֥וּף צִבְע֖וֹן אַלּ֥וּף עֲנָֽה:  ת  אִלֵּין רַבְרְבֵי חוֹרָאָה רַבָּא לוֹטָן רַבָּא שׁוֹבָל רַבָּא צִבְעוֹן רַבָּא עֲנָה:

{ל} אַלּ֥וּף דִּשֹׁ֛ן אַלּ֥וּף אֵ֖צֶר אַלּ֣וּף דִּישָׁ֑ן אֵ֣לֶּה אַלּוּפֵ֧י הַחֹרִ֛י לְאַלֻּפֵיהֶ֖ם בְּאֶ֥רֶץ שֵׂעִֽיר:  מ  אַלּ֥וּף דִּשֹׁ֛ן אַלּ֥וּף אֵ֖צֶר אַלּ֣וּף דִּישָׁ֑ן אֵ֣לֶּה אַלּוּפֵ֧י הַחֹרִ֛י לְאַלֻּפֵיהֶ֖ם בְּאֶ֥רֶץ שֵׂעִֽיר:  ת  רַבָּא דִּשֹׁן רַבָּא אֵצֶר רַבָּא דִישָׁן אִלֵּין רַבְרְבֵי חוֹרָאָה לְרַבְרְבָנֵיהוֹן בְּאַרְעָא דְשֵׂעִיר: (פ)

{לא} וְאֵ֙לֶּה֙ הַמְּלָכִ֔ים אֲשֶׁ֥ר מָלְכ֖וּ בְּאֶ֣רֶץ אֱד֑וֹם לִפְנֵ֥י מְלָךְ-מֶ֖לֶךְ לִבְנֵ֥י יִשְׂרָאֵֽל:  מ  וְאֵ֙לֶּה֙ הַמְּלָכִ֔ים אֲשֶׁ֥ר מָלְכ֖וּ בְּאֶ֣רֶץ אֱד֑וֹם לִפְנֵ֥י מְלָךְ-מֶ֖לֶךְ לִבְנֵ֥י יִשְׂרָאֵֽל:  ת  וְאִלֵּין מַלְכַיָּא דִי מְלִיכוּ בְּאַרְעָא דֶאֱדוֹם קֳדָם דִּי מְלַךְ מַלְכָּא לִבְנֵי יִשְׂרָאֵל:

{לב} וַיִּמְלֹ֣ךְ בֶּאֱד֔וֹם בֶּ֖לַע בֶּן-בְּע֑וֹר וְשֵׁ֥ם עִיר֖וֹ דִּנְהָֽבָה:  מ  וַיִּמְלֹ֣ךְ בֶּאֱד֔וֹם בֶּ֖לַע בֶּן-בְּע֑וֹר וְשֵׁ֥ם עִיר֖וֹ דִּנְהָֽבָה:  ת  וּמְלַךְ בֶּאֱדוֹם בֶּלַע בַּר בְּעוֹר וְשׁוּם קַרְתֵּהּ דִּנְהָבָה:

{לג} וַיָּ֖מָת בָּ֑לַע וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו יוֹבָ֥ב בֶּן-זֶ֖רַח מִבָּצְרָֽה:  מ  וַיָּ֖מָת בָּ֑לַע וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו יוֹבָ֥ב בֶּן-זֶ֖רַח מִבָּצְרָֽה:  ת  וּמִית בָּלַע וּמְלַךְ תְּחוֹתוֹהִי יוֹבָב בַּר זֶרַח מִבָּצְרָה:

{לד} וַיָּ֖מָת יוֹבָ֑ב וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו חֻשָׁ֖ם מֵאֶ֥רֶץ הַתֵּימָנִֽי:  מ  וַיָּ֖מָת יוֹבָ֑ב וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו חֻשָׁ֖ם מֵאֶ֥רֶץ הַתֵּימָנִֽי:  ת  וּמִית יוֹבָב וּמְלַךְ תְּחוֹתוֹהִי חֻשָׁם מֵאַרְעָא דָרוֹמָא:

{לה} וַיָּ֖מָת חֻשָׁ֑ם וַיִּמְלֹ֨ךְ תַּחְתָּ֜יו הֲדַ֣ד בֶּן-בְּדַ֗ד הַמַּכֶּ֤ה אֶת-מִדְיָן֙ בִּשְׂדֵ֣ה מוֹאָ֔ב וְשֵׁ֥ם עִיר֖וֹ עֲוִֽית:  מ  וַיָּ֖מָת חֻשָׁ֑ם וַיִּמְלֹ֨ךְ תַּחְתָּ֜יו הֲדַ֣ד בֶּן-בְּדַ֗ד הַמַּכֶּ֤ה אֶת-מִדְיָן֙ בִּשְׂדֵ֣ה מוֹאָ֔ב וְשֵׁ֥ם עִיר֖וֹ עֲוִֽית:  ת  וּמִית חֻשָׁם וּמְלַךְ תְּחוֹתוֹהִי הֲדַד בַּר בְּדַד דִקְטִיל יָת מִדְיָנָאֵי בְּחַקְלָא דְמוֹאָב וְשׁוּם קַרְתֵּהּ עֲוִית:

{לו} וַיָּ֖מָת הֲדָ֑ד וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו שַׂמְלָ֖ה מִמַּשְׂרֵקָֽה:  מ  וַיָּ֖מָת הֲדָ֑ד וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו שַׂמְלָ֖ה מִמַּשְׂרֵקָֽה:  ת  וּמִית הֲדָד וּמְלַךְ תְּחוֹתוֹהִי שַׂמְלָה מִמַשְׂרֵקָה:

{לז} וַיָּ֖מָת שַׂמְלָ֑ה וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו שָׁא֖וּל מֵרְחֹב֥וֹת הַנָּהָֽר:  מ  וַיָּ֖מָת שַׂמְלָ֑ה וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו שָׁא֖וּל מֵרְחֹב֥וֹת הַנָּהָֽר:  ת  וּמִית שַׂמְלָה וּמְלַךְ תְּחוֹתוֹהִי שָׁאוּל מֵרְחוֹבוֹת דְעַל פְּרָת:

{לח} וַיָּ֖מָת שָׁא֑וּל וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו בַּ֥עַל חָנָ֖ן בֶּן-עַכְבּֽוֹר:  מ  וַיָּ֖מָת שָׁא֑וּל וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו בַּ֥עַל חָנָ֖ן בֶּן-עַכְבּֽוֹר:  ת  וּמִית שָׁאוּל וּמְלַךְ תְּחוֹתוֹהִי בַּעַל חָנָן בַּר עַכְבּוֹר:

{לט} וַיָּמָת֮ בַּ֣עַל חָנָ֣ן בֶּן-עַכְבּוֹר֒ וַיִּמְלֹ֤ךְ תַּחְתָּיו֙ הֲדַ֔ר וְשֵׁ֥ם עִיר֖וֹ פָּ֑עוּ וְשֵׁ֨ם אִשְׁתּ֤וֹ מְהֵֽיטַבְאֵל֙ בַּת-מַטְרֵ֔ד בַּ֖ת מֵ֥י זָהָֽב:  מ  וַיָּמָת֮ בַּ֣עַל חָנָ֣ן בֶּן-עַכְבּוֹר֒ וַיִּמְלֹ֤ךְ תַּחְתָּיו֙ הֲדַ֔ר וְשֵׁ֥ם עִיר֖וֹ פָּ֑עוּ וְשֵׁ֨ם אִשְׁתּ֤וֹ מְהֵֽיטַבְאֵל֙ בַּת-מַטְרֵ֔ד בַּ֖ת מֵ֥י זָהָֽב:  ת  וּמִית בַּעַל חָנָן בַּר עַכְבּוֹר וּמְלַךְ תְּחוֹתוֹהִי הֲדַר וְשׁוּם קַרְתֵּהּ פָּעוּ וְשׁוּם אִתְּתֵהּ מְהֵיטַבְאֵל בַּת מַטְּרֵד בַת מְצָרֵף דַּהֲבָא:

{מ}  מפטיר  וְ֠אֵלֶּה שְׁמ֞וֹת אַלּוּפֵ֤י עֵשָׂו֙ לְמִשְׁפְּחֹתָ֔ם לִמְקֹמֹתָ֖ם בִּשְׁמֹתָ֑ם אַלּ֥וּף תִּמְנָ֛ע אַלּ֥וּף עַֽלְוָ֖ה אַלּ֥וּף יְתֵֽת:  מ  וְ֠אֵלֶּה שְׁמ֞וֹת אַלּוּפֵ֤י עֵשָׂו֙ לְמִשְׁפְּחֹתָ֔ם לִמְקֹמֹתָ֖ם בִּשְׁמֹתָ֑ם אַלּ֥וּף תִּמְנָ֛ע אַלּ֥וּף עַֽלְוָ֖ה אַלּ֥וּף יְתֵֽת:  ת  וְאִלֵּין שְׁמָהַת רַבְרְבֵי עֵשָׂו לְזַרְעֲיָתְהוֹן לְאַתְרֵיהוֹן בִּשְׁמָהַתְהוֹן רַבָּא תִמְנָע רַבָּא עַלְוָה רַבָּא יְתֵת:

{מא} אַלּ֧וּף אָהֳלִיבָמָ֛ה אַלּ֥וּף אֵלָ֖ה אַלּ֥וּף פִּינֹֽן:  מ  אַלּ֧וּף אָהֳלִיבָמָ֛ה אַלּ֥וּף אֵלָ֖ה אַלּ֥וּף פִּינֹֽן:  ת  רַבָּא אָהֳלִיבָמָה רַבָּא אֵלָה רַבָּא פִּינֹן:

{מב} אַלּ֥וּף קְנַ֛ז אַלּ֥וּף תֵּימָ֖ן אַלּ֥וּף מִבְצָֽר:  מ  אַלּ֥וּף קְנַ֛ז אַלּ֥וּף תֵּימָ֖ן אַלּ֥וּף מִבְצָֽר:  ת  רַבָּא קְנַז רַבָּא תֵימָן רַבָּא מִבְצָר:

{מג} אַלּ֥וּף מַגְדִּיאֵ֖ל אַלּ֣וּף עִירָ֑ם אֵ֣לֶּה | אַלּוּפֵ֣י אֱד֗וֹם לְמֹֽשְׁבֹתָם֙ בְּאֶ֣רֶץ אֲחֻזָּתָ֔ם ה֥וּא עֵשָׂ֖ו אֲבִ֥י אֱדֽוֹם:  מ  אַלּ֥וּף מַגְדִּיאֵ֖ל אַלּ֣וּף עִירָ֑ם אֵ֣לֶּה | אַלּוּפֵ֣י אֱד֗וֹם לְמֹֽשְׁבֹתָם֙ בְּאֶ֣רֶץ אֲחֻזָּתָ֔ם ה֥וּא עֵשָׂ֖ו אֲבִ֥י אֱדֽוֹם:  ת  רַבָּא מַגְדִיאֵל רַבָּא עִירָם אִלֵּין רַבְרְבֵי אֱדוֹם לְמוֹתְבָנְהוֹן בְּאַרְעָא אַחֲסַנְתְּהוֹן הוּא עֵשָׂו אֲבוּהוֹן דֶאֱדוֹמָאֵי: (פפפ)
