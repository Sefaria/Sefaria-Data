הלכות שילוח הקן



סימן רצב - כמה דינים משלוח הקן, ובו י''ג סעיפים


א שִׁלּוּחַ הַקֵּן אֵינוֹ נוֹהֵג אֶלָּא בְּעוֹף טָהוֹר, אֲפִלּוּ אִם הָאֵם טְרֵפָה. וַאֲפִלּוּ אֵין שָׁם אֶלָּא אֶפְרוֹחַ אֶחָד אוֹ בֵּיצָה אַחַת, חַיָּב לְשַׁלֵּחַ.


ב קֵן (א) הַמְזֻמָּן אֶצְלוֹ, כְּגוֹן יוֹנִים שֶׁדַּרְכָּן לִגָּדֵל עִם בְּנֵי אָדָם בַּבָּתִּים, וְאַוָּזִים וְתַרְנְגוֹלִים שֶׁקִּנְּנוּ בְּבַיִת, פָּטוּר. אֲבָל יוֹנֵי שׁוֹבָךְ וַעֲלִיָּה, וְצִפֳּרִים שֶׁקִּנְּנוּ בַּטְּפִיחִים (פֵּרוּשׁ כְּלִי חֶרֶס הַבָּנוּי בַּכְּתָלִים אֲשֶׁר שָׁם צִפֳּרִים יְקַנֵּנוּ) וּבַבּוֹרוֹת, וְאַוָּזִים וְתַרְנְגוֹלִים שֶׁקִּנְּנוּ (ב) בְּפַרְדֵּס, חַיָּב. וְהוּא שֶׁלֹּא (ג) הֻגְבְּהָה הָאֵם מֵעַל הַבֵּיצִים כְּלָל, מִשֶּׁהֵטִילָם. אֲבָל אִם הֻגְבְּהָה (ד) הָאֵם מֵעֲלֵיהֶם, אִם הַמְּקוֹמוֹת הַלָּלוּ שֶׁלּוֹ, זָכְתָה לוֹ חֲצֵרוֹ וַהֲוָה לֵיהּ מְזֻמָּן, וּפָטוּר.


ג וּבְכָל מָקוֹם שֶׁיִּמְצָאֶנּוּ, בֵּין בְּאִילָן בֵּין בָּאָרֶץ, אוֹ בְּבוֹרוֹת שִׁיחִין וּמְעָרוֹת, אוֹ בַּיָּם אוֹ בְּרֹאשׁוֹ שֶׁל אָדָם, אוֹ בְּרֹאשׁ שְׁאָר בַּעֲלֵי חַיִּים, בֵּין בִּרְשׁוּת הָרַבִּים בֵּין בִּרְשׁוּת הַיָּחִיד, חַיָּב לְשַׁלֵּחַ.


ד צָרִיךְ לְשַׁלֵּחַ הָאֵם עַד שֶׁתֵּצֵא מִתַּחַת יָדוֹ, וְאַחַר כָּךְ יִקַּח אֶת הַבָּנִים. וְאִם רוֹצֶה לַחְתֹּךְ כְּנָפֶיהָ קֹדֶם שֶׁיְּשַׁלְּחֶנָּה, כְּדֵי שֶׁלֹּא תּוּכַל לִפְרֹחַ וְיִקָּחֶנָּה, אֵינוֹ רַשַּׁאי, אֶלָּא יְשַׁלְּחֶנָּה מִיָּד עִם כְּנָפֶיהָ כְּדֵי שֶׁתּוּכַל לִפְרֹחַ, וּלְאַחַר שֶׁתֵּצֵא מִיָּדוֹ יִקָּחֶנָּה, אִם יִרְצֶה. וְאִם חֲתָכָן, יַשְׁהֶנָּה עַד שֶׁיִּגְדְּלוּ כְּנָפֶיהָ, וִישַׁלְּחֶנָּה.


ה אִם רוֹצֶה לִקַּח הָאֵם וּלְשַׁלֵּחַ הַבָּנִים, אֵינוֹ רַשַּׁאי. שִׁלְּחָהּ (ה) וְחָזְרָה, שִׁלְּחָהּ וְחָזְרָה, אֲפִלּוּ כַּמָּה פְּעָמִים, חַיָּב לַחֲזֹר וּלְשַׁלְּחָהּ. אֲבָל אִם שִׁלְּחָהּ וְנָטַל הַבָּנִים וְהֶחֱזִירָן לַקֵּן וְחָזְרָה הָאֵם עֲלֵיהֶם, פָּטוּר מִלְּשַׁלֵּחַ.


ו אִם נָטַל הָאֵם מֵעַל הַבָּנִים וְלֹא שִׁלְּחָהּ, יְשַׁלְּחֶנָּה (ו) וּפָטוּר.


ז הָיוּ הַבֵּיצִים מוּזָרוֹת, אוֹ הָאֶפְרוֹחִים מַפְרִיחִים, אוֹ שֶׁהֵם טְרֵפָה, אוֹ שֶׁזָּכָר רוֹבֵץ עַל הַקֵּן, אוֹ שֶׁעוֹף טָמֵא רוֹבֵץ עַל בֵּיצֵי עוֹף טָהוֹר, אוֹ עוֹף טָהוֹר רוֹבֵץ עַל בֵּיצֵי עוֹף טָמֵא, פָּטוּר מִלְּשַׁלֵּחַ (ל' רַמְבַּ''ם פ' י''ג מהל' שְׁחִיטָה דִּין י') .


ח הָיְתָה רוֹבֶצֶת עַל בֵּיצִים שֶׁאֵינָם מִינָהּ, וְהֵם (ז) טְהוֹרִים, הֲרֵי זֶה מְשַׁלֵּחַ. וְאִם לֹא שִׁלַּח, אֵינוֹ לוֹקֶה (שָׁם דִּין י''א) .


ט הָיְתָה מַטְלִית אוֹ כַּנְפֵי נוֹצָה מֵעוֹף אַחֵר אוֹ בֵּיצִים מוּזָרוֹת מַפְסִיקִין בֵּינָהּ לַקֵּן, אוֹ שֶׁהָיוּ שְׁנֵי (ח) סִדְרֵי בֵּיצִים זֶה עַל זֶה וּכְנָפֶיהָ נוֹגְעוֹת בַּסֵדֶר הָעֶלְיוֹן, אוֹ שֶׁהָיְתָה אֵם עַל גַּבֵּי אֵם, אוֹ שֶׁהָיָה הַזָּכָר עַל הַקֵּן וְהָאֵם עַל הַזָּכָר, הֲרֵי זֶה מְשַׁלֵּחַ וְאִם לֹא שִׁלַּח אֵינוֹ לוֹקֶה (שָׁם דִּין י''ד) .


י (ט) שָׁחַט מִקְצָת סִימָנֵי הָאֶפְרוֹחִים בְּתוֹךְ הַקֵּן קֹדֶם שֶׁיִּקָּחֶנָּה, הֲרֵי זֶה מְשַׁלֵּחַ, וְאִם לֹא שִׁלַּח אֵינוֹ לוֹקֶה (שָׁם דִּין י''ב) .


יא הָיְתָה מְעוֹפֶפֶת עַל הַקֵּן, אִם כְּנָפֶיהָ נוֹגְעוֹת בַּקֵּן חַיָּב לְשַׁלֵּחַ, וְאִם לָאו, פָּטוּר אַף עַל פִּי שֶׁנּוֹגְעוֹת מִן הַצַּד.


יב הָיְתָה יוֹשֶׁבֶת בֵּין הַבֵּיצִים אוֹ בֵּין הָאֶפְרוֹחִים, אֲפִלּוּ נוֹגַעַת בָּהֶן, (י) פָּטוּר.


יג הָיְתָה יוֹשֶׁבֶת עַל שְׁנֵי עַנְפֵי הָאִילָן וְהַקֵּן תַּחְתֶּיהָ, רוֹאִים כָּל שֶׁאִלּוּ תִּשָּׁמֵט וְנוֹפֶלֶת עֲלֵיהֶם, חַיָּב, וְאִם לָאו, פָּטוּר.
