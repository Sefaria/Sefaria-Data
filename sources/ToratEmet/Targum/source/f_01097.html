פרק א

פרק א - משנה א

מַשְׁקִין בֵּית הַשְּׁלָחִין בַּמּוֹעֵד {ב} וּבַשְּׁבִיעִית, בֵּין מִמַּעְיָן שֶׁיָּצָא בַּתְּחִלָּה, בֵּין מִמַּעְיָן שֶׁלֹּא {ו} יָצָא בַּתְּחִלָּה. אֲבָל אֵין מַשְׁקִין {ז} לֹא מִמֵּי הַגְּשָׁמִים וְלֹא מִמֵּי הַקִּילוֹן {ח}. וְאֵין עוֹשִׂין {י} עוּגִיּוֹת לַגְּפָנִים:


פרק א - משנה ב

רַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה אוֹמֵר, אֵין עוֹשִׂין אֶת הָאַמָּה בַּתְּחִלָּה בַּמּוֹעֵד וּבַשְּׁבִיעִית, וַחֲכָמִים אוֹמְרִים {יג}, עוֹשִׂין אֶת הָאַמָּה בַּתְּחִלָּה בַּשְּׁבִיעִית, וּמְתַקְּנִין אֶת הַמְּקֻלְקָלוֹת {יד} בַּמּוֹעֵד. וּמְתַקְּנִין אֶת קִלְקוּלֵי הַמַּיִם שֶׁבִּרְשׁוּת הָרַבִּים {טז} וְחוֹטְטִין אוֹתָן. וּמְתַקְּנִין אֶת הַדְּרָכִים וְאֶת הָרְחוֹבוֹת {יז} וְאֶת מִקְוָאוֹת הַמַּיִם, וְעוֹשִׂין {יח} כָּל צָרְכֵי הָרַבִּים, וּמְצַיְּנִין אֶת הַקְּבָרוֹת, וְיוֹצְאִין {יט} (אַף) עַל הַכִּלְאַיִם:


פרק א - משנה ג

רַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר, מוֹשְׁכִים אֶת הַמַּיִם מֵאִילָן לְאִילָן, וּבִלְבַד שֶׁלֹּא יַשְׁקֶה אֶת כָּל הַשָּׂדֶה. זְרָעִים שֶׁלֹּא שָׁתוּ לִפְנֵי הַמּוֹעֵד, לֹא יַשְׁקֵם בַּמּוֹעֵד. וַחֲכָמִים מַתִּירִין בָּזֶה וּבָזֶה:


פרק א - משנה ד

צָדִין אֶת הָאֵישׁוּת וְאֶת הָעַכְבָּרִים מִשְּׂדֵה הָאִילָן וּמִשְּׂדֵה הַלָּבָן, (שֶׁלֹּא) כְּדַרְכּוֹ, בַּמּוֹעֵד וּבַשְּׁבִיעִית {כד}. וַחֲכָמִים אוֹמְרִים, מִשְּׂדֵה הָאִילָן כְּדַרְכּוֹ, וּמִשְּׂדֵה הַלָּבָן שֶׁלֹּא כְּדַרְכּוֹ. וּמְקָרִין {כה} אֶת הַפִּרְצָה בַּמּוֹעֵד, וּבַשְּׁבִיעִית בּוֹנֶה כְּדַרְכּוֹ:


פרק א - משנה ה

רַבִּי מֵאִיר אוֹמֵר, רוֹאִין אֶת הַנְּגָעִים בַּתְּחִלָּה {כו} לְהָקֵל, אֲבָל לֹא לְהַחְמִיר. וַחֲכָמִים אוֹמְרִים, לֹא לְהָקֵל {כח} וְלֹא לְהַחְמִיר. וְעוֹד {כט} אָמַר רַבִּי מֵאִיר, מְלַקֵט אָדָם עַצְמוֹת אָבִיו וְאִמּוֹ, מִפְּנֵי שֶׁשִּׂמְחָה הִיא לוֹ. רַבִּי יוֹסֵי אוֹמֵר, אֵבֶל הִיא לוֹ. לֹא יְעוֹרֵר אָדָם עַל מֵתוֹ וְלֹא יַסְפִּידֶנּוּ קֹדֶם לָרֶגֶל שְׁלשִׁים יוֹם:


פרק א - משנה ו

אֵין חוֹפְרִין כּוּכִין וּקְבָרוֹת בַּמּוֹעֵד, אֲבָל מְחַנְּכִים אֶת הַכּוּכִין {לא} בַּמּוֹעֵד. וְעוֹשִׂין נִבְרֶכֶת בַּמּוֹעֵד, וְאָרוֹן עִם הַמֵּת בֶּחָצֵר. רַבִּי יְהוּדָה אוֹסֵר, אֶלָּא אִם כֵּן יֵשׁ עִמּוֹ נְסָרִים {לד}:


פרק א - משנה ז

אֵין נוֹשְׂאִין נָשִׁים בַּמּוֹעֵד, לֹא בְּתוּלוֹת, וְלֹא אַלְמָנוֹת, וְלֹא מְיַבְּמִין, מִפְּנֵי שֶׁשִּׂמְחָה הִיא לוֹ. אֲבָל מַחֲזִיר הוּא אֶת גְּרוּשָׁתוֹ. וְעוֹשָׂה אִשָּׁה תַּכְשִׁיטֶיהָ בַּמּוֹעֵד. רַבִּי יְהוּדָה אוֹמֵר, לֹא תָּסוּד מִפְּנֵי שֶׁנִּוּוּל הוּא לָהּ:


פרק א - משנה ח

הַהֶדְיוֹט תּוֹפֵר כְּדַרְכּוֹ {לז}, וְהָאֻמָּן מַכְלִיב. וּמְסָרְגִין אֶת הַמִּטּוֹת. רַבִּי יוֹסֵי אוֹמֵר, (אַף) מְמַתְּחִין:


פרק א - משנה ט

מַעֲמִידִין תַּנּוּר וְכִירַיִם (וְרֵחַיִם) בַּמּוֹעֵד. רַבִּי יְהוּדָה אוֹמֵר, אֵין מְכַבְּשִׁין אֶת הָרֵחַיִם בַּתְּחִלָּה {מ}:


פרק א - משנה י

עוֹשִׂין מַעֲקֶה לַגַּג וְלַמִּרְפֶּסֶת מַעֲשֵׂה הֶדְיוֹט {מא}, אֲבָל לֹא מַעֲשֵׂה אֻמָּן. שָׁפִין אֶת הַסְּדָקִין וּמַעְגִּילִין אוֹתָן בַּמַּעְגִּילָה בַּיָּד וּבָרֶגֶל, אֲבָל לֹא בַּמַּחְלְצַיִם. הַצִּיר וְהַצִּנּוֹר וְהַקוֹרָה וְהַמַּנְעוּל וְהַמַּפְתֵּחַ שֶׁנִּשְׁבְּרוּ, מְתַקְנָן בַּמּוֹעֵד {מג}, וּבִלְבַד שֶׁלֹּא יְכַוֵּן מְלַאכְתּוֹ בַּמּוֹעֵד, וְכָל כְּבָשִׁין שֶׁהוּא יָכוֹל לֶאֱכוֹל מֵהֶן בַּמּוֹעֵד {מד}, כּוֹבְשָׁן:


פרק ב

פרק ב - משנה א

מִי שֶׁהָפַךְ אֶת זֵיתָיו וְאֵרְעוֹ אֵבֶל אוֹ אֹנֶס, אוֹ שֶׁהִטְעוּהוּ (פוֹעֲלִים), טוֹעֵן קוֹרָה רִאשׁוֹנָה וּמַנִּיחָהּ לְאַחַר הַמּוֹעֵד. דִּבְרֵי רַבִּי יְהוּדָה. רַבִּי יוֹסֵי אוֹמֵר, זוֹלֵף וְגוֹמֵר וְגָף כְּדַרְכּוֹ:


פרק ב - משנה ב

וְכֵן מִי שֶׁהָיָה יֵינוֹ {ג} בְּתוֹךְ הַבּוֹר וְאֵרְעוֹ אֵבֶל אוֹ אֹנֶס, אוֹ שֶׁהִטְעוּהוּ (פוֹעֲלִים), זוֹלֵף {ד} וְגוֹמֵר וְגָף כְּדַרְכּוֹ. דִּבְרֵי רַבִּי יוֹסֵי. רַבִּי יְהוּדָה אוֹמֵר עוֹשֶׂה לוֹ לִמּוּדִים, בִּשְׁבִיל שֶׁלֹּא יַחְמִיץ:


פרק ב - משנה ג

מַכְנִיס אָדָם פֵּרוֹתָיו מִפְּנֵי הַגַּנָּבִים וְשׁוֹלֶה פִשְׁתָּנוֹ מִן הַמִּשְׁרָה בִּשְׁבִיל שֶׁלֹּא תֹאבַד, וּבִלְבַד {ה} שֶׁלֹּא יְכַוֵּן אֶת מְלַאכְתּוֹ בַמּוֹעֵד. וְכֻלָּן אִם כִּוְּנוּ מְלַאכְתָּן בַּמּוֹעֵד, יֹאבֵדוּ:


פרק ב - משנה ד

אֵין לוֹקְחִין בָּתִּים, עֲבָדִים וּבְהֵמָה, אֶלָּא לְצֹרֶךְ הַמּוֹעֵד {ז}, אוֹ לְצֹרֶךְ הַמּוֹכֵר, שֶׁאֵין לוֹ מַה יֹּאכַל {ח}. אֵין מְפַנִּין מִבַּיִת לְבַיִת, אֲבָל מְפַנֶּה הוּא לַחֲצֵרוֹ. אֵין מְבִיאִין כֵּלִים מִבֵּית הָאֻמָּן {יא}. וְאִם חוֹשֵׁשׁ לָהֶם, מְפַנָּן לְחָצֵר אַחֶרֶת:


פרק ב - משנה ה

מְחַפִּין אֶת הַקְּצִיעוֹת בַּקַשׁ. רַבִּי יְהוּדָה אוֹמֵר, אַף מְעַבִּין. מוֹכְרֵי פֵרוֹת, כְּסוּת וְכֵלִים, מוֹכְרִים בְּצִנְעָה {יב} לְצֹרֶךְ הַמּוֹעֵד. הַצַּיָּדִין {יג} וְהַדָּשׁוֹשׁוֹת וְהַגָּרוֹסוֹת, עוֹשִׂין בְּצִנְעָה {יד} לְצֹרֶךְ הַמּוֹעֵד. רַבִּי יוֹסֵי אוֹמֵר, הֵם הֶחְמִירוּ עַל עַצְמָן:


פרק ג

פרק ג - משנה א

וְאֵלּוּ מְגַלְּחִין בַּמּוֹעֵד, הַבָּא מִמְּדִינַת הַיָּם, וּמִבֵּית הַשִּׁבְיָה, וְהַיּוֹצֵא מִבֵּית הָאֲסוּרִין, וְהַמְנֻדֶּה שֶׁהִתִּירוּ לוֹ חֲכָמִים, וְכֵן מִי שֶׁנִּשְׁאַל לֶחָכָם וְהֻתַּר, וְהַנָּזִיר, וְהַמְּצֹרָע הָעוֹלֶה מִטֻּמְאָתוֹ לְטַהֲרָתוֹ:


פרק ג - משנה ב

וְאֵלּוּ מְכַבְּסִין בַּמּוֹעֵד, הַבָּא מִמְּדִינַת הַיָּם, וּמִבֵּית הַשִּׁבְיָה, וְהַיּוֹצֵא מִבֵּית הָאֲסוּרִים, וְהַמְנֻדֶּה {ה} שֶׁהִתִּירוּ לוֹ חֲכָמִים, וְכֵן מִי שֶׁנִּשְׁאַל לֶחָכָם וְהֻתַּר, מִטְפְּחוֹת הַיָּדַיִם וּמִטְפְּחוֹת הַסַּפָּרִים וּמִטְפְּחוֹת הַסְּפָג, (הַזָּבִין וְהַזָּבוֹת וְהַנִּדּוֹת וְהַיּוֹלְדוֹת), וְכָל {ו} הָעוֹלִין מִטֻּמְאָה לְטָהֳרָה, הֲרֵי אֵלּוּ מֻתָּרִין. וּשְׁאָר כָּל אָדָם אֲסוּרִין:


פרק ג - משנה ג

וְאֵלּוּ כּוֹתְבִין בַּמּוֹעֵד, קִדּוּשֵׁי נָשִׁים, גִּטִּין וְשׁוֹבָרִין, דְּיַתִּיקִי, מַתָּנָה וּפְרוֹזְבּוּלִין, אִגְּרוֹת שׁוּם וְאִגְּרוֹת מָזוֹן, שְׁטָרֵי חֲלִיצָה וּמֵאוּנִים וּשְׁטָרֵי בֵרוּרִין וּגְזֵרוֹת בֵּית דִּין וְאִגְּרוֹת שֶׁל רְשׁוּת:


פרק ג - משנה ד

אֵין כּוֹתְבִין שְׁטָרֵי חוֹב בַּמּוֹעֵד. וְאִם אֵינוֹ מַאֲמִינוֹ אוֹ שֶׁאֵין לוֹ מַה יֹּאכַל, הֲרֵי זֶה יִכְתּוֹב. אֵין כּוֹתְבִין סְפָרִים, תְּפִלִּין {יז} וּמְזוּזוֹת, בַּמּוֹעֵד, וְאֵין מַגִּיהִין אוֹת אַחַת, אֲפִלּוּ בְּסֵפֶר עֶזְרָא. רַבִּי יְהוּדָה אוֹמֵר, כּוֹתֵב אָדָם תְּפִלִּין וּמְזוּזוֹת לְעַצְמוֹ, וְטווֶֹה עַל יְרֵכוֹ {יט} תְּכֵלֶת לְצִיצִית:


פרק ג - משנה ה

הַקּוֹבֵר {כא} אֶת מֵתוֹ שְׁלשָׁה יָמִים {כב} קֹדֶם לָרֶגֶל, בָּטְלָה הֵימֶנּוּ גְּזֵרַת שִׁבְעָה {כג}. שְׁמוֹנָה, בָּטְלָה הֵימֶנּוּ גְּזֵרַת שְׁלשִׁים {כה}, מִפְּנֵי שֶׁאָמְרוּ, שַׁבָּת עוֹלָה {כו} וְאֵינָהּ מַפְסֶקֶת, רְגָלִים מַפְסִיקִין וְאֵינָן עוֹלִין:


פרק ג - משנה ו

רַבִּי אֱלִיעֶזֶר אוֹמֵר, מִשֶּׁחָרַב בֵּית הַמִּקְדָּשׁ, עֲצֶרֶת כַּשַּׁבָּת {כח}. רַבָּן גַּמְלִיאֵל אוֹמֵר, רֹאשׁ הַשָּׁנָה וְיוֹם הַכִּפּוּרִים, כָּרְגָלִים {כט}. וַחֲכָמִים אוֹמְרִים, לֹא כְדִבְרֵי זֶה וְלֹא כְדִבְרֵי זֶה, אֶלָּא עֲצֶרֶת כָּרְגָלִים, רֹאשׁ הַשָּׁנָה וְיוֹם הַכִּפּוּרִים כַּשַּׁבָּת:


פרק ג - משנה ז

אֵין קוֹרְעִין {לד}, וְלֹא חוֹלְצִין, וְאֵין מַבְרִין, אֶלָּא קְרוֹבָיו שֶׁל מֵת. אֵין מַבְרִין אֶלָּא עַל מִטָּה זְקוּפָה, וְאֵין מוֹלִיכִין לְבֵית הָאֵבֶל לֹא בַּטַּבְלָא וְלֹא בָאִסְקוּטְלָא וְלֹא בַקָּנוֹן, אֶלָּא בַסַּלִּים. וְאֵין אוֹמְרִים בִּרְכַּת אֲבֵלִים בַּמּוֹעֵד, אֲבָל עוֹמְדִין בַּשּׁוּרָה וּמְנַחֲמִין וּפוֹטְרִין אֶת הָרַבִּים:


פרק ג - משנה ח

אֵין מַנִּיחִין אֶת הַמִּטָּה בָּרְחוֹב, שֶׁלֹּא לְהַרְגִּיל אֶת הַהֶסְפֵּד {לז}, וְלֹא שֶׁל נָשִׁים לְעוֹלָם, מִפְּנֵי הַכָּבוֹד {לח}. נָשִׁים בַּמּוֹעֵד מְעַנּוֹת, אֲבָל לֹא מְטַפְּחוֹת. רַבִּי יִשְׁמָעֵאל אוֹמֵר, הַסְּמוּכוֹת לַמִּטָּה, מְטַפְּחוֹת:


פרק ג - משנה ט

בְּרָאשֵׁי חֳדָשִׁים, בַּחֲנֻכָּה וּבַפּוּרִים, מְעַנּוֹת וּמְטַפְּחוֹת, בָּזֶה וּבָזֶה (אֲבָל) לֹא מְקוֹנְנוֹת. נִקְבַּר הַמֵּת {לט}, לֹא מְעַנּוֹת וְלֹא מְטַפְּחוֹת. אֵיזֶהוּ עִנּוּי, שֶׁכֻּלָּן עוֹנוֹת כְּאֶחָת. קִינָה, שֶׁאַחַת מְדַבֶּרֶת וְכֻלָּן עוֹנוֹת אַחֲרֶיהָ, שֶׁנֶּאֱמַר, (ירמיה ט) וְלַמֵּדְנָה בְנֹתֵיכֶם נֶהִי, וְאִשָּׁה רְעוּתָהּ קִינָה. אֲבָל לֶעָתִיד לָבוֹא הוּא אוֹמֵר, (ישעיה כה) בִּלַּע הַמָּוֶת לְנֶצַח, וּמָחָה ה' אֱלֹהִים דִּמְעָה מֵעַל כָּל פָּנִים וגו':
