סימן סט - דיני מליחה והדחה, ובו כ''א סעיפים


א צָרִיךְ (א) לְהָדִיחַ הַבָּשָׂר קֹדֶם מְלִיחָה. וְאִם הֱדִיחוֹ (ב) הַטַּבָּח אֵין צָרִיךְ לַהֲדִיחוֹ בַבַּיִת (טוּר) . וְאִם אַחַר שֶׁהֵדִיחַ חָתַךְ כָּל נֶתֶךְ לִשְׁנַיִם אוֹ לִשְׁלֹשָׁה אוֹ שֶׁהֵסִיר טַלְפֵי הָרַגְלַיִם, לְאַחַר הַהֲדָחָה (אָרֹךְ כְּלָל י''ד וי''ז) צָרִיךְ (ג) לַחֲזֹר לַהֲדִיחָם הגה: וְאִם לֹא עָשָׂה כֵן, הָוֵי כְּלֹא הוּדַח כְּלָל (שָׁם) . הֲדָחַת הַבָּשָׂר לְכַתְּחִלָּה, יִזָּהֵר לִשְׁרוֹתוֹ נֶגֶד חֲצִי שָׁעָה וְלַהֲדִיחוֹ (ד) הֵיטֵב בְּמֵי הַשְּׁרִיָּה, אֲבָל אִם לֹא שְׁרָאוֹ רַק הֱדִיחוֹ הֵיטֵב סָגֵי לֵיהּ, וְאַחַר כָּךְ יַמְתִּין מְעַט שֶׁיְּטַפְטְפוּ הַמַּיִם קֹדֶם שֶׁיִּמְלְחֶנּוּ, שֶׁלֹּא יִמַּס הַמֶּלַח מִן הַמַּיִם וְלֹא יוֹצִיא דָּם (הַגָּהוֹת ש''ד וְהַגָּהוֹת אֲשֵׁרִי פכ''ה וּבְאָרֹךְ כְּלָל א') . וְנָהֲגוּ שֶׁלֹּא (ה) לְהִשְׁתַּמֵּשׁ בִּדְבָרִים אֲחֵרִים בַּכְּלִי שֶׁשּׁוֹרִין בּוֹ בָּשָׂר. וְאִם נִשְׁתַּהָה הַבָּשָׂר בִּשְׁרִיָּתוֹ מֵעֵת לְעֵת, הַבָּשָׂר וְגַם הַכְּלִי (ו) אֲסוּרִין. וְעַיֵּן לְקַמָּן סִימָן זֶה. אֲבָל פָּחוֹת מֵעֵת לְעֵת אֵין לְהַקְפִּיד. וְאַף בְּמָקוֹם שֶׁנֶּאֱסַר הַכְּלִי, מֻתָּר לַחֲזֹר (ז) וְלִשְׁרוֹת בּוֹ (בְּאָרֹךְ) . דִּין בָּשָׂר שֶׁמָּלֵא (ח) קֶרַח, כֵּיצַד נוֹהֲגִין עִם שְׁרִיָּתוֹ עַיֵּן לְעֵיל סִימָן ס''ח.


ב אִם מָלַח וְלֹא הֵדִיחַ תְּחִלָּה, יְדִיחֶנּוּ וְיִמְלְחֶנּוּ שֵׁנִית. וְיֵשׁ אוֹסְרִין. הגה: וְכֵן נוֹהֲגִין, אֲפִלּוּ לֹא נִמְלַח רַק מְעַט כְּדֶרֶךְ שֶׁמּוֹלְחִין (ט) לִצְלִי, וַאֲפִלּוּ לֹא (י) שָׁהָה בְּמִלְחוֹ שִׁעוּר מְלִיחָה (אָרֹךְ וש''ד וּסְמַ''ג) . מִיהוּ בִּמְקוֹם הֶפְסֵד מְרֻבֶּה, יֵשׁ לְהַתִּיר (בֵּית יוֹסֵף) . וְאִם לֹא הוּדַח רַק מְעַט, קֹדֶם שֶׁמְּלָחוֹ, מֻתָּר (יא) בְּדִיעֲבַד. וְהוּא הַדִּין אִם הָיָה (יב) ס' בַּחֲתִיכָה נֶגֶד דָּם שֶׁעָלָיו (הַגָּהַת ש''ד בְּשֵׁם מהרא''י) . וְאִם נִמְלַח חֲתִיכָה בְּלֹא הֲדָחָה עִם שְׁאָר חֲתִיכוֹת, שְׁאָר חֲתִיכוֹת (יג) מֻתָּרוֹת וְהִיא אֲסוּרָה (בְּאָרֹךְ כְּלָל ד' דִּין ה') .


ג לֹא יִמְלַח (יד) בְּמֶלַח דַּקָּה כְּקֶמַח, וְלֹא בְּמֶלַח גַּסָה בְּיוֹתֵר שֶׁנּוֹפֶלֶת מֵעַל הַבָּשָׂר אֵילָךְ וָאֵילָךְ. וְאִם אֵין לוֹ מֶלַח אַחֵר רַק מֶלַח דַּק כְּקֶמַח, מֻתָּר לִמְלֹחַ בּוֹ (ד''ע) .


ד יְפַזֵּר עָלָיו מֶלַח שֶׁלֹּא יִשָּׁאֵר בּוֹ מָקוֹם מִבְּלִי מֶלַח, וְיִמְלַח כְּדֵי שֶׁלֹּא יְהֵא רָאוּי לֶאֱכֹל עִם אוֹתוֹ מֶלַח, וְאֵינוֹ צָרִיךְ (טו) לְהַרְבּוֹת עָלָיו מֶלַח יוֹתֵר מִזֶּה. וּמוֹלְחוֹ מִשְּׁנֵי צְדָדִים. וְעוֹפוֹת צָרִיךְ לְמָלְחָם גַּם מִבִּפְנִים. וְאִם לֹא מְלָחָם אֶלָּא מִבִּפְנִים אוֹ מִבַּחוּץ, וְכֵן חֲתִיכָה שֶׁלֹּא נִמְלְחָה אֶלָּא מִצַּד אֶחָד, מֻתָּר. הגה: וְיֵשׁ אוֹסְרִים (טז) אֲפִלוּ בְּדִיעֲבַד (מהרא''י בְּהַגָּהוֹת ש''ד וְאָרֹךְ כְּלָל ו'), וְהָכֵי נָהוּג, אִם לֹא לְצֹרֶךְ. וְדַוְקָא אִם כְּבַר (יז) נִתְבַּשֵּׁל כָּךְ; אֲבָל אִם לֹא נִתְבַּשֵּׁל עֲדַיִן, לֹא יְבַשְּׁלֶנּוּ כָּךְ, אֶלָּא אִם הוּא תּוֹךְ י''ב שָׁעוֹת (יח) שֶׁנִּמְלַח, יַחֲזֹר וְיִמְלַח צַד הַשֵּׁנִי שֶׁלֹּא נִמְלַח עֲדַיִן וִיבַשְּׁלֶנּוּ (יט) אַחַר כָּךְ, וְאִם הוּא אַחַר י''ב שָׁעוֹת, אֲזַי יִצְּלֶנּוּ דְּנוּרָא מִשְׁאָב שָׁאִיב וְאֵין הַצַּד שֶׁנִּמְלַח כְּבָר בּוֹלֵעַ מִצַּד שֶׁלֹּא נִמְלַח (שָׁם) .


ה אַחַר שֶׁנִּמְלְחָה הַחֲתִיכָה, אִם (כ) חָתַךְ מִמֶּנָּהּ אֵינוֹ צָרִיךְ לַחֲזֹר וְלִמְלֹחַ מְקוֹם הַחֲתָךְ.


ו שִׁעוּר שְׁהִיָּה בְּמֶלַח אֵינוֹ פָּחוֹת מִכְּדֵי הִלּוּךְ מִיל, שֶׁהוּא כְּדֵי שְׁלִישִׁית שָׁעָה (כא) בְּקֵרוּב. הגה: וְעַל זֶה יֵשׁ לִסְמֹךְ (כב) בְּדִיעֲבַד, אוֹ אֲפִלּוּ לְכַתְּחִלָּה לִכְבוֹד אוֹרְחִים אוֹ לְצֹרֶךְ שַׁבָּת, אֲבָל בְּלָאו הָכֵי הַמִּנְהָג לְהַשְׁהוֹת בִּמְלִיחָה שִׁעוּר שָׁעָה, וְאֵין לְשַׁנּוֹת (מהרא''י בת''ה סִימָן קס''ז וּבְהַגָּהַת ש''ד ואו''ה כְּלָל א') .


ז קֹדֶם שֶׁיִּתֵּן הַבָּשָׂר בַּכְּלִי שֶׁמְּדִיחוֹ בּוֹ, (כג) יְנַפֵּץ מֵעָלָיו הַמֶּלַח שֶׁעָלָיו, אוֹ יִשְׁטְפֶנּוּ בְּמַיִם, וְאַחַר כָּךְ יִתֵּן הַבָּשָׂר בַּכְּלִי שֶׁמְּדִיחִין בּוֹ, וִידִיחֶנּוּ פַּעֲמַיִם, וְיִשְׁטֹף הַכְּלִי בֵּין רְחִיצָה לִרְחִיצָה. הגה: וְיֵשׁ אוֹמְרִים שֶׁצְּרִיכִין לְהָדִיחַ ג' פְּעָמִים (חִדּוּשֵׁי אֲגֻדָּה בְּשֵׁם אוֹר זָרוּעַ ומהרא''י ואו''ה כ''א דט''ו והג''א), וְהָכֵי נוֹהֲגִין (כד) לְכַתְּחִלָּה, עַל כֵּן (כה) יִשְׁטְפֶנּוּ אוֹ יְנַפֵּץ הַמֶּלַח מֵעָלָיו וִידִיחֶנּוּ (כו) ב' פְּעָמִים דְּזֶה הָוֵי כַּהֲדָחָה ג' פְּעָמִים, אוֹ יָשִׂים הַמַּיִם תּוֹךְ הַכְּלִי וְאַחַר כָּךְ יַנִּיחַ בּוֹ הַבָּשָׂר וִידִיחֶנּוּ ג' פְּעָמִים, וְהָכֵי נָהוּג. וּלְכַתְּחִלָּה יִתֵּן מַיִם הַרְבֵּה בַּהֲדָחָה רִאשׁוֹנָה, כְּדֵי שֶׁיְּבַטֵּל כֹּחַ הַמֶּלַח שֶׁבַּצִּיר (ד''ע) . מֻתָּר לְהָדִיחַ הַבָּשָׂר בְּמֵי פֵּרוֹת וְאֵין צָרִיךְ מַיִם (הר''ן פ' הַשּׁוֹחֵט ותוספ' שָׁם) .


ח אִם לֹא נִפֵּץ הַמֶּלַח שֶׁעָלָיו וְלֹא שְׁטָפוֹ, אֵין לֶאֱסֹר, כִּי (כז) הַמַּיִם שֶׁבַּכְּלִי מְבַטְּלִין כֹּחַ הַמֶּלַח. הגה: וַאֲפִלּוּ הַמַּיִם מוּעָטִים מִכָּל מָקוֹם מְבַטְּלִים כֹּחַ הַצִּיר (ד''ע וארוךְ כלל ו' ומהרא''י) . וְיֵשׁ מַתִּירִין אֲפִלּוּ לֹא הָיָה מַיִם כְּלָל בַּכְּלִי וְהִנִּיחוּ בּוֹ הַבָּשָׂר, דְּאֵין לָחוּשׁ (כח) בְּשָׁעָה מוּעֶטֶת כָּזוֹ שֶׁעוֹסְקִין בַּהֲדָחָת הַבָּשָׂר (ש''ד ואו''ה כְּלָל א' וי' וְעַיֵּן לְעֵיל ס''כ), וְיֵשׁ לִסְמֹךְ עֲלַיְהוּ. וְאֵין לָחוּשׁ בְּכָל זֶה רַק בַּהֲדָחָה רִאשׁוֹנָה, אֲבָל לְאַחַר הֲדָחָה רִאשׁוֹנָה אֵין לָחוּשׁ (כָּךְ מַשְׁמָע מִמָּרְדְּכַי וְר''ן) . וְאֵין חִלּוּק בֵּין אִם הֱדִיחוֹ בִּכְלִי כָּשֵׁר אוֹ (כט) טְרֵפָה אוֹ חוֹלֶבֶת. וְאִם הָיָה מְעַט מַיִם בַּכְּלִי, שֶׁמְּבַטֵּל כֹּחַ הַמֶּלַח שֶׁאֵינוֹ חָשׁוּב עוֹד כְּרוֹתֵחַ, אֲפִלּוּ הַכְּלִי חוֹלֶבֶת אוֹ הַטְּרֵפָה (ל) מְלֻכְלָךְ עֲדַיִן בְּאִסּוּר, מֻתָּר (בְּאָרֹךְ) .


ט בָּשָׂר (לא) שֶׁנִּמְלַח וְנִתְבַּשֵּׁל בְּלֹא הֲדָחָה אַחֲרוֹנָה, צָרִיךְ שֶׁיְּהֵא בּוֹ שִׁשִּׁים כְּדֵי הַמֶּלַח שֶׁבּוֹ. הגה: וְכָל הַקְּדֵרָה (לב) מִצְטָרֵף לְשִׁשִּׁים (ש''ד וְעוֹד פּוֹסְקִים עַיֵּן ס''ק ל''ד) . וְאִם יֵשׁ בַּקְּדֵרָה כ''כ כְּמוֹ הַחֲתִיכָה (לג) שֶׁנִּמְלַח וְלֹא הוּדַח, הַכֹּל שָׁרֵי, דְּוַדַּאי אִכָּא שִׁשִּׁים נֶגֶד הַמֶּלַח שֶׁעַל הַחֲתִיכָה דְּהַחֲתִיכָה עַצְמָהּ בְּוַדַּאי הִיא שְׁלֹשִׁים נֶגֶד הַמֶּלַח שֶׁעָלָיו, (בְּאָרֹךְ כְּלָל י''א ד''ד) . וְאִי לֵיכָּא ס' בַּקְּדֵרָה נֶגֶד הַמֶּלַח, אֲפִלּוּ לֹא הוּשַׂם רַק בִּכְלִי שֵׁנִי, הַכֹּל אָסוּר, דְּמֵאַחַר שֶׁיֵּשׁ שָׁם מֶלַח וְצִיר אֲפִלּוּ בִּכְלִי שֵׁנִי מְבַשֵּׁל (שָׁם דִּין א' וּבַמָּרְדְּכַי) . וּבָשָׂר יָבֵשׁ יֵשׁ לְהַתִּיר אֲפִלּוּ בִּכְלִי רִאשׁוֹן, דְּוַדַּאי יֵשׁ ס' נֶגֶד הַמֶּלַח שֶׁעָלָיו מֵאַחַר שֶׁכְּבָר נִתְיַבֵּשׁ (מָרְדְּכַי פכ''ה וראב''ן וּבֵית יוֹסֵף בְּשֵׁם הַגָּהַת אֲשֵׁרִי וש''ד) . אֲבָל לְכַתְּחִלָּה יִזָּהֵר אֲפִלּוּ בְּבָשָׂר יָבֵשׁ שֶׁלֹּא לְבַשְּׁלוֹ אוֹ לַהֲדִיחוֹ בְּמַיִם שֶׁהַיָּד סוֹלֶדֶת בָּהֶם בְּלֹא הֲדָחָה אַחֲרוֹנָה (ש''ד שָׁם) . וְכָל זֶה מַיְרֵי שֶׁלֹּא הוּדַח בָּאַחֲרוֹנָה כְּלָל, אֲבָל אִם הוּדַח רַק פַּעַם אַחַת וְנִתְבַּשֵּׁל כָּךְ, מֻתָּר, דִּבְדִיעֲבַד סָגֵי לֵהּ בַּהֲדָחָה אַחַת בָּאַחֲרוֹנָה (מהרא''י בְּהַגָּהַת ש''ד) . מֶלַח שֶׁמָּלְחוּ בּוֹ פַּעַם אַחַת, אָסוּר לִמְלוֹחַ בּוֹ פַעַם (לד) שֵׁנִית (כָּל בּוֹ וד''ע וְדַעַת רֹב הַפּוֹסְקִים), וְכָל שֶׁכֵּן שֶׁאָסוּר לֶאֱכֹל (לה) הַמֶּלַח אַחַר שֶׁמָּלְחוּ בּוֹ.


י עוֹבֵד כּוֹכָבִים מְשַׁמֵּשׁ בְּבֵית יִשְׂרָאֵל וְנָתַן הַבָּשָׂר בַּקְּדֵרָה, וְאֵין (לו) יָדוּעַ אִם הֱדִיחוֹ, אִם יוֹדֵעַ הָעוֹבֵד כּוֹכָבִים מִנְהַג יִשְׂרָאֵל, סוֹמְכִין עַל דְּבָרָיו, אִם הָיָה שָׁם יִשְׂרָאֵל יוֹצֵא וְנִכְנָס אוֹ שׁוּם קָטָן בֶּן דַּעַת. הגה: וּבְחַד מִינַיְיהוּ סָגֵי (טוּר), אוֹ בְּמֵסִיחַ לְפִי תֻּמּוֹ שֶׁהֱדִיחוֹ יָפֶה אוֹ שֶׁיִּשְׂרָאֵל, אֲפִלּוּ קָטָן, יוֹצֵא וְנִכְנָס, דְּמִרְתָּת הוֹאִיל וְיוֹדֵעַ מִנְהֲגֵי יִשְׂרָאֵל. מִיהוּ אִם מִיחָה לָעוֹבֵד כּוֹכָבִים שֶׁלֹּא יְדִיחוֹ בְּלֹא רְשׁוּתוֹ, וְהוּא עָבָר עַל דְּבָרָיו, אָסוּר, דְּהָא חֲזֵינָן דְּאֵינוֹ מִרְתָּת וְאֵין לִסְמֹךְ גַּם כֵּן אַדְּבָרָיו (הַגָּהַת ש''ד ואו''ה) .


יא בָּשָׂר (לז) שֶׁנִּתְבַּשֵּׁל בְּלֹא מְלִיחָה, צָרִיךְ שֶׁיִּהְיֶה בַּתַּבְשִׁיל שִׁשִּׁים כְּנֶגֶד אוֹתוֹ (לח) בָּשָׂר וְאָז מֻתָּר הַכֹּל. הגה: וְיֵשׁ אוֹסְרִים אוֹתָהּ חֲתִיכָה אֲפִלּוּ בִּדְאִכָּא ס' נֶגֶד הַחֲתִיכָה (טוּר בְּשֵׁם י''א וְהַגָּהוֹת מַיְמוֹנִי בְּשֵׁם סמ''ק וְהַגָּהוֹת ש''ד וְאָרֹךְ ומהרא''י וּמָרְדְּכַי בְּשֵׁם ראב''ן), וְהָכֵי נָהוּג אִם לֹא לְצֹרֶךְ, כְּגוֹן לִכְבוֹד שַׁבָּת אוֹ (לט) לִכְבוֹד אוֹרְחִים, דְּאָז יֵשׁ לִסְמֹךְ אַדִּבְרֵי הַמְקִלִּין (בֵּית יוֹסֵף בְּשֵׁם א''ח וד''ע עד''מ) . וְאִם נִמְלְחָה חֲתִיכָה וְלֹא נִמְלְחָה (מ) כָּרָאוּי, דִּינָהּ כְּאִלּוּ לֹא נִמְלְחָה כְּלָל. וְכֵן בָּשָׂר שֶׁשָּׁהָה ג' יָמִים בְּלֹא מְלִיחָה, אַף אִם נִמְלַח דִּינוֹ כְּאִלּוּ לֹא נִמְלַח, וְאִם נִתְבַּשֵּׁל צָרִיךְ ס' (מא) כְּנֶגְדּוֹ (או''ה) .


יב בָּשָׂר שֶׁשָּׁהָה ג' יָמִים מֵעֵת לְעֵת בְּלֹא מְלִיחָה, נִתְיַבֵּשׁ דָּמוֹ בְּתוֹכוֹ וְלֹא יֵצֵא עוֹד עַל יְדֵי מְלִיחָה, וְאֵין לְאָכְלוֹ (מב) מְבֻשָּׁל, אֶלָּא צְלִי. וְאַחַר שֶׁצְּלָאוֹ (מג) לֹא יְבַשְּׁלֶנּוּ, וְאִם בִּשְּׁלוֹ מֻתָּר. וְאֵין לְהַשְׁהוֹת בָּשָׂר ג' יָמִים בְּלֹא מְלִיחָה, (מד) דְּחַיְישִׁינָן שֶׁמָּא יְבַשְּׁלוֹ (בֵּית יוֹסֵף בשם פסקי מהרא''י סימן קצ''א) .


יג וְאִם שָׁרוּ אוֹתוֹ בְּמַיִם תּוֹךְ הַג' יָמִים, יָכוֹל לְהַשְׁהוֹתוֹ עוֹד שְׁלֹשָׁה יָמִים אֲחֵרִים פָּחוֹת (מה) חֲצִי שָׁעָה (אָגוּר) . בָּשָׂר שֶׁנִּמְלַח וְסָפֵק אִם נִמְלַח תּוֹךְ ג', מֻתָּר (בְּאָרֹךְ) .


יד בָּשָׂר שֶׁשָּׁהָה שְׁלֹשָׁה יָמִים בְּלֹא מְלִיחָה וְנִתְעָרְבָה אוֹתָהּ חֲתִיכָה בַּחֲתִיכוֹת אֲחֵרוֹת, (מו) בְּטֵלָה בְּרֹב וּמֻתָּר לְבַשֵּׁל כֻּלָּן, וְאֲפִלּוּ הָיְתָה רְאוּיָה (מז) לְהִתְכַּבֵּד. וְכֵן הַדִּין בְּנִתְבַּשֵּׁל בְּלֹא (מח) מְלִיחָה וְנִתְעָרֵב אַחַר כָּךְ בַּאֲחֵרוֹת (או''ה) .


טו בָּשָׂר (מט) הַמְלֻכְלָךְ בְּדָמִים שֶׁנִּשְׁרָה בְּמַיִם מֵעֵת לְעֵת, יֵשׁ אוֹסְרִים לְאָכְלוֹ כִּי אִם צְלִי, אֶלָּא אִם כֵּן יֵשׁ בַּמַּיִם ס' (נ) כְּנֶגְדּוֹ. (עי' ס''ק מ''ו) וְיֵשׁ (נא) אוֹסְרִים אֲפִלּוּ לִצְלִי, וְהָכֵי נָהוּג (או''ה) .


טז אֵין מוֹלְחִין אֶלָּא בִּכְלִי מְנֻקָּב אוֹ עַל גַּבֵּי קַשִּׁין וְקִסְמִין אוֹ בִּמְקוֹם מִדְרוֹן, בְּעִנְיָן שֶׁאִם יִשָּׁפֵךְ שָׁם מַיִם יֵצְאוּ מִיָּד. הגה: וְדַף חָלָק בְּרָהִיטָנִי, שֶׁמַּיִם זָבִין מִמֶּנּוּ, אֵין צָרִיךְ לְהַנִּיחוֹ בְּמִדְרוֹן. אֲבָל אִם אֵינוֹ חָלָק בְּרָהִיטָנִי, צָרִיךְ לְהַנִּיחוֹ בְּמִדְרוֹן שֶׁיֵּצְאוּ הַמַּיִם מִמֶּנּוּ (שְׁעָרִים ש''ו ובבֵּית יוֹסֵף בְּשֵׁם ת''ה סִימָן קע''ג) . וַאֲפִלּוּ בִּכְלִי מְנֻקָּב יִזָּהֵר לְכַתְּחִלָּה שֶׁיִּהְיוּ הַנְּקָבִים פְּתוּחִים. (בְּאָרֹךְ כְּלָל ג') . וְלָכֵן לֹא יַעֲמִידוֹ עַל גַּבֵּי קַרְקַע, כִּי הוּא כִּכְלִי שֶׁאֵינוֹ מְנֻקָּב (עַיִּןִ לְעֵיל סוֹף סִימָן ע') . וְכֵן מַחְמִירִין קְצָת לָשׂוּם תּוֹךְ הַכְּלִי מְנֻקָּב קַשׁ אוֹ קִסְמִין, כִּי הַבָּשָׂר יִסְתֹּם הַנְּקָבִים (הַגָּהוֹת אֲשֵׁרִ''י פכ''ה וּבְאָרֹךְ כְּלָל א'), (נב) וּבְדִיעֲבַד אֵין לָחוּשׁ לְכָל זֶה. וְאִם מָלַח בִּכְלִי שֶׁאֵינוֹ מְנֻקָּב, אָסוּר לְהִשְׁתַּמֵּשׁ בְּאוֹתוֹ כְּלִי בְּדָבָר (נג) רוֹתֵחַ. הגה: וְאִם נִשְׁתַּמֵּשׁ בּוֹ, בָּעֵי קְלִיפָה אִם הוּא דָּבָר יָבֵשׁ (רַשְׁבָּ''א), וְאִם הוּא דָּבָר לַח, בָּעֵינָן שִׁשִּׁים נֶגֶד קְלִפָּה מִן (נד) הַקְּעָרָה. (בֵּית יוֹסֵף בשם י''א) . וְיֵשׁ אוֹמְרִים שֶׁאֲפִלּוּ בְּצוֹנֵן אָסוּר לְהִשְׁתַּמֵּשׁ בּוֹ בְּלֹא (נה) הֲדָחָה, וְאִם נִשְׁתַּמֵּשׁ בּוֹ בְּלֹא הֲדָחָה, יָדִיחַ מַה שֶּׁנִּשְׁתַּמֵּשׁ בּוֹ. הגה: אֲבָל מֻתָּר לַחֲזֹר וְלִמְלֹחַ בָּהּ בָּשָׂר לְאַחַר (נו) שֶׁנִּקְּבוּהָ, אוֹ אֲפִלּוּ בְּלֹא נְקִיבָה אִם הַבָּשָׂר שָׁהָה כְּבַר בִּמְלִיחָתוֹ וְהוּדַח (בֵּית יוֹסֵף בְּשֵׁם הַגָּהַת ש''ד וְאָרֹךְ) .


יז אֲבָל אִם הוּא מְנֻקָּב, מֻתָּר לֶאֱכֹל בּוֹ אֲפִלּוּ (נז) רוֹתֵחַ. וְיֵשׁ אוֹסְרִין בְּרוֹתֵחַ. וְיֵשׁ לִזָּהֵר לְכַתְּחִלָּה, וּבְדִיעֲבַד (נח) מֻתָּר. (אָרֹךְ כְּלָל ג') .


יח בָּשָׂר שֶׁנִּמְלַח בִּכְלִי שֶׁאֵינוֹ מְנֻקָּב וְשָׁהָה בּוֹ כְּשִׁעוּר שֶׁיִּתְּנוּ מַיִם עַל הָאֵשׁ וְיַתְחִילוּ לְהַרְתִּיחַ, כָּל מַה שֶּׁמִּמֶּנּוּ בַּצִּיר אָסוּר לְאָכְלוֹ אֲפִלּוּ (נט) צְלִי, וְחֵלֶק הַחֲתִיכָה שֶׁחוּץ לַצִּיר אֵין אָסוּר מִמֶּנּוּ אֶלָּא כְּדֵי (ס) קְלִיפָּה, וַאֲפִלּוּ אִם יֵשׁ בָּהּ שֻׁמָּן. הגה: וְיֵשׁ אוֹסְרִין כָּל הַחֲתִיכָה, אֲפִלּוּ מַה (סא) (סב) שֶּׁחוּץ לְצִיר. (טוּר וּמָרְדְּכַי בְּשֵׁם מהר''ף) . וַאֲפִלּוּ לֹא נִמְלַח רַק מְעַט, כְּדֶרֶךְ שֶׁמּוֹלְחִין לִצְלִי, וַאֲפִלּוּ לֹא שָׁהָה שִׁעוּר מְלִיחָה רַק מְעַט, עַד שֶׁנִּרְאֶה צִיר בִּכְלִי (מָרְדְּכַי ספכ''ה בְּשֵׁם ראבי''ה וְהַגָּהַת ש''ד וְאָרֹךְ כְּלָל י'), וְכֵן נוֹהֲגִין, וְאֵין לְשַׁנּוֹת. וּמִכָּל מָקוֹם אֵינוֹ אָסוּר רַק אוֹתָהּ חֲתִיכָה הַמֻּנַּחַת לְמַטָּה בַּכְּלִי וְנוֹגַעַת בַּצִּיר, אֲבָל שְׁאָר חֲתִיכוֹת שֶׁעָלֶיהָ הַנִּמְלָחִים עִמָּהּ, (סג) מֻתָּרוֹת. (שָׁם בְּאָרֹךְ וּבְכָל בּוֹ) וְהָכֵי נָהוּג.


יט אַחַר שֶׁנִּמְלַח הַבָּשָׂר וְהוּדַח, מֻתָּר לִתְּנוֹ אֲפִלּוּ בְּמַיִם שֶׁאֵינָם (סד) רוֹתְחִין. וְיֵשׁ מִי שֶׁמַּצְרִיךְ לִתְּנוֹ בְּמַיִם רוֹתְחִין. וְהַמִּנְהָג כִּסְבָרָא רִאשׁוֹנָה, וְכֵן (סה) עִקָּר (בֵּית יוֹסֵף וְדַעַת עַצְמוֹ) .


כ בָּשָׂר שֶׁנִּמְלַח, וְשָׁהָה כְּדֵי מְלִיחָה, וּנְתָנוּהוּ אַחַר כָּךְ בִּכְלִי בְּלֹא הֲדָחָה וְנִתְמַלֵּא מִצִּיר, (סו) מֻתָּר. וּלְפִי זֶה, בָּשָׂר שֶׁנָּפַל לְתוֹךְ צִיר הַיּוֹצֵא מֵהַבָּשָׂר אַחַר שֶׁשָּׁהָה כְּדֵי מְלִיחָה, מֻתָּר. וְיֵשׁ מִי שֶׁאוֹסֵר בָּזֶה וּבָזֶה, וְיֵשׁ לָחוּשׁ לוֹ לְכַתְּחִלָּה. הגה: וְיֵשׁ אוֹסְרִין (תּוֹסְפוֹת וְרֹא''שׁ וְכַמָּה מִן הָאַחֲרוֹנִים), אֲפִלּוּ בְּדִיעֲבַד, כְּדֵי (סז) קְלִפָּה. (הֲגָהַת ש''ד וּבַשְּׁעָרִים, ומהרי''ב ומהר''ש וּבֵית יוֹסֵף וְאָרֹךְ), וְהָכִי נָהוּג. וְדַוְקָא מַה שֶּׁמֻּנָּח בְּתוֹךְ הַצִּיר, אֲבָל מַה שֶּׁחוּץ לַצִּיר, שָׁרֵי. (סח) וְהַכְּלִי שֶׁנָּפַל בּוֹ אוֹתוֹ הַצִּיר, אָסוּר. וְכֵן אִם הָיָה הַכְּלִי חוֹלֶבֶת וְיֵשׁ בּוֹ לִכְלוּךְ, אָסוּר הַבָּשָׂר, דְּאַף לְאַחַר שִׁעוּר מְלִיחָתוֹ נֶחְשָׁב (סט) כְּרוֹתֵחַ (הֲגָהַת ש''ד ות''ה סִימָן קנ''ט) . אֲבָל אִם הוּא הֶפְסֵד מְרֻבֶּה וְאִית בֵּיהּ גַּם כֵּן צֹרֶךְ סְעֻדַּת מִצְוָה, יֵשׁ לְהָקֵל וְלוֹמַר דְּאַחַר שֶׁשָּׁהָה שִׁעוּר מְלִיחָה לֹא מִחְשַׁב צִירוֹ כְּרוֹתֵחַ (ד''ע) וְעַיִּןִ לְקַמָּן סִימָן צ''א. וּלְפִי דְּבָרָיו, אַף עַל פִּי שֶׁשָּׁהָה הַבָּשָׂר בְּמֶלַח כְּשִׁעוּר, אָסוּר לֶאֱכֹל מִמֶּנּוּ עַד (ע) שֶׁיְּדִיחֶנּוּ יָפֶה יָפֶה. וְאָסוּר לַחְתֹּךְ מִמֶּנּוּ בְּסַכִּין קֹדֶם שֶׁיְּדִיחֶנּוּ, וְאִם חָתַךְ צָרִיךְ לְהַגְעִילוֹ. הגה: וְיֵשׁ מַתִּירִין לַחְתֹּךְ בְּסַכִּין לְאַחַר שֶׁשָׁהָא שִׁעוּר מְלִיחָה, דְּאֵין מְלִיחָה (עא) לַכֵּלִים, וְכֵן עִקָּר דְּמֻתָּר (ת''ה סִימָן קס''ב ואו''ה כְּלָל ל''ז) . אֲבָל הַסַּכִּין צָרִיךְ (עב) הֲדָחָה אַחַר כָּךְ, אוֹ נְעִיצָה בַּקַּרְקַע אִם נִתְיַבְּשָׁה עָלָיו הַצִּיר (אָרֹךְ) . וַאֲפִלּוּ קֹדֶם שִׁעוּר מְלִיחָה אֵין הָאִסּוּר מִשּׁוּם הַסַּכִּין, אֶלָּא מִשּׁוּם שֶׁיָּצָא דָּם בְּעַיִן עַל הַבָּשָׂר, וְאִם רוֹצֶה לַהֲדִיחוֹ שָׁם הֵיטֵב וְלַחֲזֹר וּלְמָלְחוֹ שָׁם, מֻתָּר. וּכְשֶׁרוֹצִים לַעֲשׂוֹת מָלִיחַ לְהִתְקַיֵּם, לְאַחַר שֶׁשָּׁהָה בְּמִלְחוֹ בִּכְלִי מְנֻקָּב כַּשִּׁעוּר הָרָאוּי, (עג) מְדִיחִים אוֹתוֹ יָפֶה יָפֶה, וְאַחַר כָּךְ חוֹזְרִים וּמוֹלְחִים אוֹתוֹ כְּדֵי שֶׁיִּתְקַיֵּם, וַאֲפִלּוּ בִּכְלִי שֶׁאֵינוֹ מְנֻקָּב. וַאֲפִלּוּ לְפִי סְבָרָא זוֹ, אִם רוֹצֶה לִמְלֹחַ וְלֶאֱכֹל צְלִי בְּלֹא הֲדָחָה, עוֹשֶׂה וְאֵינוֹ חוֹשֵׁשׁ לְדָם שֶׁעַל הַמֶּלַח, שֶׁהָאֵשׁ שׁוֹאֲבוֹ וּמוֹנֵעַ הַמֶּלַח מִלִּבְלֹעַ דָּם. וְהָנֵי מִלֵּי בְּמוֹלְחוֹ וּמַעֲלֵהוּ לִצְלִי, אֲבָל אִם שָׁהָה בְּמִלְחוֹ, הַמֶּלַח בּוֹלֵעַ הַדָּם וְנֶאֱסַר, וּלְפִיכָךְ מְדִיחוֹ יָפֶה יָפֶה וְצוֹלֶה וְאוֹכֵל. וְע''ל סִימָן ע''ו מִדִּין בָּשָׂר שֶׁנִּמְלַח וְלֹא הוּדַח וְנִצְלָה כָּךְ.


כא בְּמָקוֹם שֶׁאֵין מֶלַח מָצוּי, יִצְלוּ הַבָּשָׂר עַד (עד) שֶׁיָּזוּב כָּל דָּמוֹ, וְאַחַר כָּךְ יְבַשְּׁלוּהוּ.

א מוֹלְחִין הַרְבֵּה חֲתִיכוֹת זוֹ עַל גַּב זוֹ. אַף עַל פִּי שֶׁהַתַּחְתּוֹנָה גּוֹמֶרֶת פְּלִיטָתָהּ קֹדֶם לָעֶלְיוֹנָה, לֹא אָמְרִינָן שֶׁחוֹזֶרֶת וּבוֹלַעַת מִדַּם הָעֶלְיוֹנָה, לְפִי שֶׁהוּא שׁוֹהֶה הַרְבֵּה לִפְלֹט צִיר וְכָל זְמַן (א) (ב) שֶׁפּוֹלֶטֶת צִירָהּ אֵינָהּ בּוֹלַעַת. וַאֲפִלּוּ מִתְקַבֵּץ הַרְבֵּה צִיר וְעוֹמֵד בַּגּוּמָא שֶׁבֵּין הַחֲתִיכוֹת, מֻתָּר. בַּמֶּה דְּבָרִים אֲמוּרִים, בְּמוֹלֵחַ בָּשָׂר עִם בָּשָׂר, וַאֲפִלּוּ בְּשַׂר שׁוֹר עִם בְּשַׂר גְּדָיִים וּטְלָאִים וַאֲפִלּוּ עִם עוֹפוֹת, שֶׁאִי אֶפְשָׁר לָהֶם לִגְמֹר כָּל פְּלִיטַת צִירָן עַד שֶׁיִּגְמֹר בְּשַׂר שׁוֹר לִפְלֹט אֶת דָּמוֹ. הגה: וּמִכָּל מָקוֹם נוֹהֲגִין לְהַחֲמִיר לְכַתְּחִלָּה שֶׁכָּל חֲתִיכָה שֶׁיֵּשׁ לָהּ בֵּית קִבּוּל, כְּגוֹן דֹּפֶן שְׁלֵמָה, מְהַפְּכִין אוֹתָהּ שֶׁיָּזוּב הַדָּם, אֲבָל בְּדִיעֲבַד אֵין לָחוּשׁ (א''ו הָאָרֹךְ) . חֲתִיכָה שֶׁמָּלַח אוֹתָהּ ב' פְּעָמִים, מֻתֶּרֶת, וְלֹא חַיְשִׁינָן שֶׁמֶּלַח הַשֵּׁנִי מַבְלִיעַ הַדָּם הַנִּשְׁאָר מִמְּלִיחָה הָרִאשׁוֹנָה (פִּסְקֵי מהרא''י סִימָן ס''ו) . אֲבָל בָּשָׂר עִם דָּגִים, אֲפִלּוּ בְּשַׂר עוֹפוֹת עִם דָּגִים, אָסוּר לִמְלֹחַ לְפִי שֶׁהַדָּגִים פּוֹלְטִים כָּל צִירָן קֹדֶם שֶׁיִּפְלֹט הַעוֹף אֶת דָּמוֹ. וְאִם עָבָר וּמְלָחָן (ג) יַחַד, הָעוֹפוֹת מֻתָּרִין, אֲבָל הַדָּגִים צָרִיךְ לִטֹּל מֵהֶם כְּדֵי קְלִפָּה. וְאִם לֹא (ד) נִטְּלוּ קַשְׂקַשֵּׂיהֶם כְּשֶׁנִּמְלְחוּ, מֻתָּרִים. הגה: וְיֵשׁ אוֹסְרִין כָּל הַדָּגִים (אָרֹךְ כְּלָל י''ג) אִם אֵינָן ס' נֶגֶד הָעוֹפוֹת, דְּאָנוּ מְשַׁעֲרִין בִּמְלִיחָה בְּס', וְהָכֵי נָהוּג. וְדַוְקָא דְּלֵית בְּהוּ קַשְׂקַשִּׂים, דְּרָפוּ קְרָמַיְהוּ וּפָלְטֵי מִיָּד, אֲבָל אִי אִית בְּהוּ קַשְׂקַשִּׂים, מֻתָּרִים, דְּאֵינָן פּוֹלְטִים מִיָּד וְלֹא בָּלְעֵי מִן הָעוֹפוֹת (הָרֹא''שׁ וְשָׁם), דְּאַיְדֵי דְּטָרִידֵי לִפְלֹט לָא בָּלְעֵי, מִידִי דְּהָוֵי אַשְּׁנֵי חֲתִיכוֹת שֶׁנִּמְלְחוּ יַחַד.


ב בַּמֶּה דְּבָרִים אֲמוּרִים, כְּשֶׁמָּלַח שְׁנֵיהֶם יַחַד, אוֹ שֶׁהָעוֹף מָלִיחַ וְדָג תָּפֵל, אֲבָל אִם דָּג מָלִיחַ וְעוֹף תָּפֵל (ה) וּנְתָנָם זֶה אֵצֶל זֶה, אוֹ זֶה (ו) עַל גַּב זֶה, אַף דָּגִים מֻתָּרִים בְּלֹא קְלִיפָה. הגה: וְכֵן אִם הַדָּגִים מֻנָּחִים עַל הָעוֹפוֹת, מֻתָּרִים אֲפִלּוּ נִמְלְחוּ יַחַד דְּדָם אֵינוֹ מְפַעְפֵּעַ מִלְּמַטָּה לְמַעְלָה, וְלֹא גָּרַע מֵחֲתִיכָה הַמֻּנַּחַת בַּצִּיר דְּמַה שֶּׁלְּמַעְלָה מִן הַצִּיר שָׁרֵי (אָרֹךְ כְּלָל י''ג) . וְכֵן אִם לֹא הִנִּיחַ הַדָּגִים אֵצֶל הָעוֹפוֹת אֶלָּא לְאַחַר שֶׁשָּׁהוּ הָעוֹפוֹת בְּמִלְחָן שִׁעוּר מְלִיחָה, אַף הַדָּגִים מֻתָּרִים, אַף עַל פִּי (ז) שֶׁשְּׁנֵיהֶם מְלוּחִים (ד''ע וְכֵן מַשְׁמָע בְּאָרֹךְ) . (ח) וְאִם נִמְלְחוּ דָּגִים בִּכְלִי שֶׁמָּלְחוּ בּוֹ בָּשָׂר, הַדָּגִים מֻתָּרִים דְּאֵין הַדָּגִים בּוֹלְעִים הַדָּם שֶׁבַּכְּלִי, דְּאֵין מְלִיחָה לְכֵלִים (בֵּית יוֹסֵף בְּשֵׁם רַשְׁבָּ''א סִימָן קכ''ט) . וְהָא דְּאָמְרִינָן דְּאִם הַדָּג תָּפֵל וְהָעוֹפוֹת מְלוּחִים אֲסוּרִים, הַיְנוּ דַּוְקָא שֶׁהַדָּג פָּלַט כְּבָר (ט) דָּמוֹ וְהוּדַח, אֲבָל אִם עֲדַיִן לֹא פָּלַט דָּמוֹ וְלֹא נִמְלַח מֵעוֹלָם, מֻתָּר, דְּאַגַּב דְּיִפְלֹט דָּם דִּידֵיהּ יִפְלֹט גַּם כֵּן הַדָּם שֶׁבּוֹלֵעַ מִן הָעוֹפוֹת (סְבָרַת עַצְמוֹ וְכֵן מַשְׁמָע בְּאָרֹךְ כְּלָל י''ג בְּשֵׁם הסמ''ק), כְּמוֹ חֲתִיכָה שֶׁנָּפְלָה לְצִיר קֹדֶם מְלִיחָה (י) דְּמֻתָּר מֵהַאי טַעֲמָא, כְּמוֹ שֶׁיִּתְבָּאֵר לְקַמָּן סִימָן זֶה. וְעַיֵּן לְקַמָּן סִימָן צ''א בְּאֵיזֶה מְלִיחָה אָמְרִינַן דְּהָוֵי כְּרוֹתֵחַ. (יא)


ג בְּשַׂר שְׁחוּטָה שֶׁמְּלָחוֹ עִם בְּשַׂר טְרֵפָה, אוֹ שֶׁהַטְּרֵפָה מְלוּחָה וְהַכְּשֵׁרָה תְּפֵלָה, וְהֵם נוֹגְעִים זֶה בָּזֶה, אָסוּר כְּדֵי קְלִפָּה. שֶׁאַף עַל פִּי שֶׁאֵינוֹ בּוֹלֵעַ מִדַּם הַטְּרֵפָה, בּוֹלַעַת מִצִּירָהּ. אֲבָל אִם הַכְּשֵׁרָה מְלוּחָה וְהַטְּרֵפָה תְּפֵלָה, מֻתֶּרֶת בַּהֲדָחָה בְּלֹא קְלִיפָה, בֵּין נָתַן כְּשֵׁרָה לְמַעְלָה, בֵּין נְתָנָהּ לְמַטָּה. וְיֵשׁ מִי שֶׁאוֹסֵר בְּנוֹגְעִים זֶה בָּזֶה, וְסוֹבֵר שֶׁלֹּא הֻתְּרוּ אֶלָּא (יב) בְּעוֹמְדִים בְּסָמוּךְ בִּכְדֵי שֶׁפְּלִיטָה שֶׁל זֶה נוֹגַעַת בָּזֶה. הגה: וּכְשֶׁהַטְּרֵפָה מָלוּחַ וְכָשֵׁר תָּפֵל, אֲפִלּוּ בִּכְהַאי גַּוְנָא אָסוּר. וְע''ל סִימָן ק''ה מֵאֵלּוּ דִּינִים.


ד הָא דְּאָמְרִינָן דְּאֵינוֹ אוֹסֵר אֶלָּא כְּדֵי קְלִפָּה, הַיְנוּ כְּשֶׁשְּׁתֵּיהֶן כְּחוּשׁוֹת, אֲבָל אִם אַחַת מֵהֶן שְׁמֵנָה, אֲפִלּוּ אִם חֲתִיכַת הַטְּרֵפָה כְּחוּשָׁה וַחֲתִיכַת הַכְּשֵׁרָה שְׁמֵנָה, מְפַעְפֵּעַ הָאִסוּר בְּכֻלָּהּ אִם הָיְתָה הַטְּרֵפָה הַמְּלוּחָה (יג) לְמַטָּה, מִשּׁוּם דְּתַתָּאָה גָּבַר. וְעַיֵּן לְקַמָּן סִימָן ק''ה כֵּיצַד (יד) נוֹהֲגִין.


ה יֵשׁ אוֹמְרִים שֶׁבָּשָׂר שֶׁנִּמְלַח אָסוּר לְהַשְׁהוֹתוֹ בְּמִלְחוֹ לְאַחַר פְּלִיטַת כָּל צִירוֹ דְּהַיְנוּ י''ב שָׁעוֹת, לְפִי שֶׁחוֹזֵר וּבוֹלֵעַ מִלַּחְלוּחִית דָּם שֶׁעָלָיו וְשֶׁעַל הַמֶּלַח. וְיֵשׁ (טו) מַתִּירִים לְהַשְׁהוֹתוֹ בְּמִלְחוֹ אֲפִלּוּ כַּמָּה יָמִים. וּלְכַתְּחִלָּה יֵשׁ לָחוּשׁ לְדִבְרֵי הָאוֹסְרִים, וּבְדִיעֲבַד מֻתָּר.


ו יֵשׁ אוֹסְרִים לִתֵּן בָּשָׂר שֶׁלֹּא נִמְלַח כְּלָל, אוֹ שֶׁנִּמְלַח (טז) (יז) וּפָלַט כָּל דָּמוֹ, עִם בָּשָׂר שֶׁנִּמְלַח קֹדֶם פְּלִיטָתוֹ דָּם, לְפִי שֶׁהַבָּשָׂר שֶׁלֹּא נִמְלַח, אוֹ שֶׁנִּמְלַח וּפָלַט כָּל דָּמוֹ, חוֹזֵר וּבוֹלֵעַ מִמַּה שֶּׁחֲבֵרוֹ פּוֹלֵט. וְיֵשׁ מַתִּירִים עַל יְדֵי מְלִיחָה שֶׁיִּמְלְחֶנּוּ אַחַר כָּךְ, כִּי אָז יַפְלִיט כָּל דָּם שֶׁבָּלַע, וְיֵשׁ מַתִּירִים בְּכָל זֶה. וּלְכַתְּחִלָּה יֵשׁ לָחוּשׁ לְדִבְרֵי הָאוֹסְרִים. הגה: וַאֲפִלּוּ בְּדִיעֲבַד נוֹהֲגִין (יח) לְאָסְרוֹ אִם נָפַל בָּשָׂר שֶׁכְּבָר פָּלַט כָּל דָּמוֹ וְצִירוֹ אֵצֶל בָּשָׂר שֶׁלֹּא שָׁהָה עֲדַיִן שִׁעוּר מְלִיחָה (מהרא''י בְּהַגָּהַת ש''ד) . וְיֵשׁ אוֹמְרִים שֶׁכָּל מֵעֵת לְעֵת לְאַחַר שֶׁנִּמְלַח פּוֹלֵט צִיר, וְאִם נָגַע תּוֹךְ זְמַן זֶה לַבָּשָׂר שֶׁנִּמְלַח וְלֹא שָׁהָה עֲדַיִן שִׁעוּר מְלִיחָה, (יט) אֵינוֹ נֶאֱסָר, וְכֵן נוֹהֲגִין (ש''ד בְּשֵׁם הַגְּאוֹנִים ומהרא''י וסה''ת וּמָרְדְּכַי בְּשֵׁם ר''י), אַךְ בְּמָקוֹם שֶׁאֵין הֶפְסֵד מְרֻבֶּה יֵשׁ לְאָסְרוֹ לְאַחַר י''ב שָׁעוֹת. וְקֹדֶם לָזֶה אֵין לְהַחֲמִיר (כ) כְּלָל. וְכָל שֶׁכֵּן אִם לֹא נִמְלַח כְּלָל, דַּאֲפִלּוּ אִם נָפַל לַצִּיר מַמָּשׁ אֵין לְאָסְרוֹ, דְּאָמְרִינָן עַל יְדֵי שֶׁיִּפְלֹט (כא) דָּם דִּידֵהּ יִפְלֹט גַּם כֵּן מַה שֶּׁבָּלַע מִמָּקוֹם אַחֵר. וַאֲפִלּוּ נִמְלַח הַבָּשָׂר וְלֹא שָׁהָה עֲדַיִן שִׁעוּר מְלִיחָה, וְנָפַל לַצִּיר, יֵשׁ (כב) לְהַתִּיר (אָרֹךְ כְּלָל ה') אִם לֹא נִכְבַּשׁ בְּתוֹכוֹ יוֹם (כג) שָׁלֵם. אַךְ אִם שָׁהָה שִׁעוּר מְלִיחָה וְנָפַל לַצִּיר, יֵשׁ אוֹסְרִין אוֹתוֹ אַף עַל פִּי שֶׁלֹּא כָּלָה זְמַן פְּלִיטַת צִירוֹ (כד) עֲדַיִן (רש''ל בא''ו שֶׁלּוֹ) . וּמִכָּל מָקוֹם לְצֹרֶךְ הֶפְסֵד גָּדוֹל יֵשׁ לְהַתִּיר גַּם בָּזֶה כָּל זְמַן פְּלִיטַת צִירוֹ, דְּהַיְנוּ תּוֹךְ י''ב (כה) (כו) שָׁעוֹת, עַל יְדֵי (כז) שֶׁיַּחֲזֹר וִידִיחֶנּוּ (הַגָּהַת ש''ד) וְיַחֲזֹר וְיִמְלְחֶנּוּ. וְכֵן אִם נָפַל לַצִּיר קֹדֶם שֶׁשָּׁהָה שִׁעוּר מְלִיחָה, יְדִיחֶנּוּ וְיַחֲזֹר וְיִמְלְחֶנּוּ. מִיהוּ אִם מְלָחוֹ בְּלֹא הֲדָחָה לְאַחַר שֶׁנָּפַל לַצִּיר, (כח) שָׁרֵי בְּדִיעֲבַד (אָרֹךְ כ''ה) . וְכָל זֶה לֹא מַיְרֵי אֶלָּא בְּצִיר שֶׁהוּא כְּרוֹתֵחַ, כַּמְּבֹאָר לְעֵיל סִימָן ס''ט, אֲבָל דָּם בְּעֵין שֶׁנָּפַל עַל בָּשָׂר שֶׁהוּא תּוֹךְ שִׁעוּר (כט) מְלִיחָתוֹ וְהוּא חָשׁוּב כְּרוֹתֵחַ כַּמְּבֹאָר לְקַמָּן סִימָן צ''א, נֶאֱסָר הַבָּשָׂר, דִּלְגַבֵּי דָּם בְּעֵין לֹא אָמְרִינָן אַיְדֵי דְּטָרִיד לִפְלֹט לֹא בָּלַע, וְלֹא כְּבוֹלְעוֹ כָּךְ פּוֹלְטוֹ (שם) . וְהָא דְּאָסְרִינָן אוֹתוֹ כְּשֶׁנָּפַל לַצִּיר, הַיְנוּ דַּוְקָא מַה שֶּׁבְּתוֹךְ הַצִּיר, אֲבָל מַה (ל) שֶּׁלְּמַעְלָה מִן הַצִּיר, שָׁרֵי. מִיהוּ, מַה שֶּׁבְּתוֹךְ הַצִּיר נֶאֱסָר מִיָּד וְאֵין (לא) (לב) שִׁעוּר לַדָּבָר. בָּשָׂר שֶׁנָּפַל לְתוֹךְ צִיר שֶׁעַל הַקַּרְקַע, דִּינָהּ כְּאִלּוּ הָיְתָה הַצִּיר בַּכְּלִי (בֵּית יוֹסֵף בְּשֵׁם ש''ד ואו''ה שָׁם) . בָּשָׂר שֶׁנָּגַע בַּחֲתִיכָה שֶׁנִּמְלְחָה בִּכְלִי שֶׁאֵינוֹ מְנֻקָּב וְנֶאֶסְרָה, דִּינָהּ כְּאִלּוּ (לג) נָגְעָה בַּצִּיר (ד''ע) . סָפֵק צִיר סָפֵק מַיִם, מֻתָּר, דְּצִיר דְּרַבָּנָן (לד) וּסְפֵיקָא לְהָקֵל (מָרְדְּכַי פכ''ה) . צִיר מְעֹרָב עִם מַיִם, אֲפִלּוּ הָיוּ (לה) הַמַּיִם מוּעָטִים, לֹא חָשִׁיב עוֹד רוֹתֵחַ וְאֵינָהּ אוֹסֶרֶת (בֵּית יוֹסֵף בְּשֵׁם אָגוּר) דִּין כָּבוּשׁ עַיֵּן לְקַמָּן סִימָן ק''ה.



סימן עא - דין מליחת הראש והטלפים והמח, ובו ג' סעיפים


א הָרֹאשׁ, חוֹתְכוֹ לִשְׁנַיִם וּמוֹלְחוֹ יָפֶה (א) לְצַד פְּנִים, וְחוֹזֵר וּמוֹלֵחַ עַל הַשֵּׂעָר, שֶׁאֵין הַשֵּׂעָר מְעַכֵּב עַל יְדֵי הַמֶּלַח לְהַפְלִיט דָּם.


ב הַטְּלָפַיִם, מְחַתֵּךְ אוֹתָם (ב) מְעַט לְמַטָּה, וּמוֹלֵחַ, וְיַנִּיחַ מְקוֹם הַחֲתָךְ לְמַטָּה וְיִמְלַח גַּם עַל הַשֵּׂעָר. וְעַיֵּן לְעֵיל סִימָן (ג) ס''ח.


ג הַקְּרוּם שֶׁעַל הַמֹּחַ, יֵשׁ בּוֹ חוּטִין, וְהַמֹּחַ עַצְמוֹ יֵשׁ בּוֹ דָּם, וְאֵינוֹ יוֹצֵא מִידֵי דָּמוֹ בִּמְלִיחַת הָרֹאשׁ, לְפִי שֶׁעֶצֶם הָרֹאשׁ מַקִּיפוֹ וְעוֹמֵד לְפָנָיו, וְאֵין מָקוֹם לַדָּם לָזוּב. לְפִיכָךְ הַבָּא לְמָלְחוֹ קוֹרְעוֹ וּמוֹצִיא הַמֹּחַ וְקוֹרֵעַ הַקְּרוּם וּמוֹלְחוֹ. וְאִם רָצָה לִמְלֹחַ הָרֹאשׁ וְהַמֹּחַ בְּתוֹכוֹ, יִנְקֹב הָעֶצֶם כְּנֶגֶד הַקְּרוּם, וְיִנְקֹב גַּם הַקְּרוּם, וְיִמְלַח, וְיַנִּיחַ הַנֶּקֶב (ד) לְמַטָּה, וּמֻתָּר אֲפִלּוּ לִקְדֵרָה. הגה: וְעַיֵּן לְעֵיל סִימָן ס''ח כֵּיצַד נוֹהֲגִין (ה) לְכַתְּחִלָּה (ש''ד בְּשֵׁם מ''ה) . וְאִם נִמְלַח הָרֹאשׁ שָׁלֵם בְּלֹא נְקִיבַת הָעֶצֶם, הַקְּרוּם (ו) וְהַמֹּחַ אֲסוּרִים, וּשְׁאָר הָרֹאשׁ מֻתָּר, וְכָל שֶׁכֵּן שְׁאָר (ז) בָּשָׂר שֶׁעִמּוֹ (אָרֹךְ כְּלָל י''ז וש''ד) . הַקּוּלִית וּשְׁאָר עֲצָמוֹת שֶׁיֵּשׁ בָּהֶן מֹחַ צְרִיכִין (ח) מְלִיחָה. וּמְלִיחַת הָעֶצֶם מוֹעִיל לַמֹּחַ שֶׁבְּקִרְבּוֹ, וְאֵין צָרִיךְ לִנְקֹב הָעֶצֶם (טוּר בְּשֵׁם הָרַשְׁבָּ''א) . מִיהוּ לְכַתְּחִלָּה לֹא יִמְלְחֶנּוּ (ט) עִם שְׁאָר בָּשָׂר, רַק לְבַד, וּבְדִיעֲבַד שָׁרֵי. (אָרֹךְ כְּלָל ל''א) .



סימן עב - דין מליחת הלב והראה, ובו ד' סעיפים


א הַלֵּב, מִתְקַבֵּץ הַדָּם בְּתוֹכוֹ בִּשְׁעַת שְׁחִיטָה. לְפִיכָךְ צָרִיךְ לְקָרְעוֹ קֹדֶם מְלִיחָה וּלְהוֹצִיא דָּמוֹ, וּלְמָלְחוֹ אַחַר כָּךְ. וְאָז מֻתָּר אֲפִלּוּ (א) לְבַשְּׁלוֹ (אָרֹךְ כְּלָל ט''ו) . וְיֵשׁ מַחְמִירִים לְבַשְּׁלוֹ רַק צוֹלִין אוֹתוֹ (וְאַחַר כָּךְ מְבַשְׁלוֹ) (ש''ד) .


ב מְלָחוֹ וְלֹא קְרָעוֹ, קוֹרְעוֹ אַחַר מְלִיחָתוֹ וּמֻתָּר אַף עַל פִּי שֶׁנִּמְלַח עִם הַדָּם שֶׁבְּתוֹכוֹ, דִּכְבוֹלְעוֹ (ב) כָּךְ פּוֹלְטוֹ. וְהוּא הַדִּין אִם צְלָאוֹ וְלֹא קְרָעוֹ, שֶׁקּוֹרְעוֹ לְאַחַר צְלִיָּתוֹ וּמֻתָּר. אֲבָל אִם בִּשְּׁלוֹ בְּלִי קְרִיעָה, אָסוּר עַד שֶׁיְּהֵא ס' כְּנֶגֶד הַלֵּב, דְּלָא יָדְעִינָן כַּמָּה (ג) נְפַק מִנֵהּ. הגה: וַאֲפִלּוּ בִּדְאִכָּא ס', הַלֵּב (ד) עַצְמוֹ אָסוּר (אָרֹךְ וש''ד מהרא''י ורי''ו), וְיִקְלֹף מְעַט סְבִיב (ה) הַלֵּב. וְיֵשׁ מַחְמִירִין בּוֹ, אֲפִלּוּ בִּמְלִיחָה, לֶאֱסֹר שְׁאָר בָּשָׂר שֶׁנִּמְלַח עִמּוֹ, וְאוֹמְרִין דְּלֹא אָמְרִינָן כְּבוֹלְעוֹ כָּךְ פּוֹלְטוֹ גַּבֵּי דַּם הַלֵּב הַכָּנוּס בְּתוֹכוֹ, דְּהָוֵי דָּם מַמָּשׁ וְלֹא מִקְרֵי דַּם פְּלִיטָה (הַגָּהַת ש''ד בְּשֵׁם ר' מַתִּתְיָה וכ''מ בְּמָרְדְּכַי בְּשֵׁם ר''י) . וְהַמִּנְהָג (ו) לְהָקֵל, וְכֵן עִקָּר דְּגַם זֶה מִקְרֵי דַּם פְּלִיטָה וְשַׁיָּךְ לְמֵימַר בֵּיהּ כְּבוֹלְעוֹ כָּךְ פּוֹלְטוֹ, מִדֵּי דְּהָוֵי אַדָּם שֶׁבַּחוּטִין, כְּמוֹ שֶׁנִּתְבָּאֵר לְעֵיל סִימָן כ''ב (וכ''ד האו''ה ומהר''א כ''ץ) . וְיֵשׁ (ז) מַחְמִירִין לִקְלֹף קְצָת בַּמָּקוֹם שֶׁהָיָה הַלֵּב דָּבוּק (בְּאָרֹךְ וּמָרְדְּכַי שָׁם), וְטוֹב לָחוּשׁ לְדִבְרֵיהֶם וְלִקְלֹף קְצָת סְבִיב הַלֵּב, וְאָז הַכֹּל מֻתָּר. וְאֵין חִלּוּק בֵּין בָּשָׂר שֶׁעִם הַלֵּב אוֹ הַלֵּב (ח) עַצְמוֹ, וְאֵין חִלּוּק בֵּין אִם הַלֵּב סָגוּר אוֹ פָּתוּחַ (ט) לְמַעְלָה (ד''ע וְלֹא כאו''ה) . וְנוֹהֲגִין לְכַתְּחִלָּה לַחְתֹּךְ עָרְלַת הַלֵּב וְלַחְתֹּךְ גִּידִין שֶׁבִּפְנִים (אָרֹךְ), וְאֵינוֹ אֶלָּא חֻמְרָא (י) וּזְהִירוּת בְּעָלְמָא.


ג אֵין עוֹף שֶׁלֹּא יְהֵא בּוֹ שִׁשִּׁים כְּנֶגֶד לִבּוֹ, וּמֻתָּר אֲפִלּוּ הוּא דָּבוּק בָּעוֹף. הגה: וְכָל עוֹף יֵשׁ בּוֹ ס', אֲפִלּוּ אֵין בּוֹ (יא) הָרֹאשׁ וְהָרַגְלַיִם הַתַּחְתּוֹנִים (פִּסְקֵי מהרא''י סִימָן צ''א וְר''י מִינְץ וּבְאָרֹךְ), כְּדֶרֶךְ שֶׁנּוֹהֲגִים לַהֲסִירָן דְּהַיְנוּ עַד הָאַרְכֻּבָּה הַתַּחְתּוֹנָה, וְלָכֵן אִם הָעוֹף שָׁלֵם, הַכֹּל מֻתָּר. וְאִם אֵין שָׁלֵם, וְלֵיכָּא בּוֹ ס' נֶגֶד הַלֵּב הַדָּבוּק בָּעוֹף, יֵשׁ אוֹמְרִים דְּהַחֲתִיכָה נַעֲשֵׂית (יב) (יג) נְבֵלָה וּבָעֵינָן ס' בַּקְּדֵרָה נֶגֶד כָּל הָעוֹף (טוּר בְּשֵׁם י''א וכ''ה בַּפּוֹסְקִים), וְכֵן נוֹהֲגִין. וַאֲפִלּוּ אִיכָּא ס' בַּקְּדֵרָה, הָעוֹף אָסוּר הוֹאִיל וְאֵין בּוֹ ס' נֶגֶד (יד) הַלֵּב הַדָּבוּק בּוֹ. וְאִם אֵין הַלֵּב דָּבוּק בָּעוֹף, מִצְטָרֵף כָּל הַקְּדֵרָה לְבַטֵּל הַלֵּב בְּס'. וְעַיֵּן לְקַמָּן סִימָן צ''ב בְּדִין חֲתִיכָה נַעֲשֵׂית נְבֵלָה. וְאֵין לְךָ בְּהֵמָה שֶׁהִיא ס' נֶגֶד לִבָּהּ.


ד הָרֵאָה אֵינָהּ צְרִיכָה חִתּוּךְ, אֲבָל נָהֲגוּ (טו) לְקָרְעָהּ וְלִפְתֹּחַ הַקְּנוֹקָנוֹת הַגְּדוֹלִים שֶׁלָּהּ, וּמִנְהָג יָפֶה הוּא.



סימן עג - דין צלית הכבד, ובו ו' סעיפים


א הַכָּבֵד, יֵשׁ בּוֹ רִבּוּי דָּם. לְפִיכָךְ לְכַתְּחִלָּה אֵין לוֹ (א) תַּקָּנָה לְבַשְּׁלוֹ עַל יְדֵי מְלִיחָה, אֶלָּא קוֹרְעוֹ שְׁתִי וָעֵרֶב וּמַנִּיחַ חִתּוּכוֹ לְמַטָּה, וְצוֹלֶהוּ שֶׁיְּהֵא רָאוּי (ב) לַאֲכִילָה (או''ה נתיב ט''ו), וְאַחַר כָּךְ יָכוֹל לְבַשְּׁלוֹ. הגה: וְאִם (ג)(ד)(ה) מְנַקְּבָהּ הַרְבֵּה פְּעָמִים בְּסַכִּין הָוֵי כִּקְרִיעָה שְׁתִי וָעֵרֶב (אָרֹךְ) . וְכֵן אִם נָטַל מִשָּׁם הַמָּרָה וַחֲתִיכַת בָּשָׂר מִן הַכָּבֵד, דְּאֶפְשָׁר לַדָּם לָזוּב מִשָּׁם (הַגָּהַת ש''ד) . וּמִכָּל מָקוֹם אִם לֹא עָשָׂה כֵן נוֹטֵל הַסִּמְפּוֹנוֹת לְאַחַר צְלִיָּה, וּמְבַשְּׁלָהּ (אָרֹךְ) . וְכָל זֶה בְּכָבֵד שְׁלֵמָה, אֲבָל כְּשֶׁהִיא חֲתוּכָה אֵין צָרִיךְ כְּלוּם (ר' יְרוּחָם בתא''ו) . וּכְשֶׁבָּא לְבַשְּׁלָהּ אַחַר הַצְּלִיָּה, יְדִיחֶנָּה תְּחִלָּה אַחַר הַצְּלִיָּה, קֹדֶם הַבִּשּׁוּל (דִּקְדּוּק בֵּית יוֹסֵף מִמָּרְדְּכַי פכ''ה וּבְאָרֹךְ) . מִיהוּ אִם לֹא הֱדִיחָהּ וּבִשְּׁלָהּ כָּךְ, מֻתָּר (ראבי''ה ואו''ה שָׁם) . וּבְדִיעֲבַד מֻתָּר אִם נִתְבַּשֵּׁל לְבַדּוֹ בַּקְּדֵרָה (בְּלֹא צְלִיָּה), אֲבָל הַקְּדֵרָה אֲסוּרָה, שֶׁפּוֹלֶטֶת וְאֵינָהּ בּוֹלַעַת. וְיֵשׁ מִי שֶׁאוֹסֵר. הגה: וְכֵן נוֹהֲגִין לֶאֱסֹר הַכֹּל (הַגָּהַת סמ''ק וּמהרא''י בְּהַגָּהַת ש''ד), אֲפִלּוּ (ו) נִמְלְחָה הַכָּבֵד קֹדֶם בִּשּׁוּלָהּ (דַּעַת רַשִׁ''י לַאֲפוּקֵי רַבֵּנוּ תָּם) .


ב אִם חֲלָטוֹ בְּחֹמֶץ אוֹ בְּרוֹתְחִין, וְנִקֵּב וְהוֹצִיא מִזְרְקֵי הַדָּם שֶׁבְּתוֹכוֹ, מִן הַדִּין (ז) מֻתָּר לְבַשְּׁלוֹ, אֶלָּא שֶׁהַגְּאוֹנִים אָסְרוּ לַעֲשׂוֹת כֵּן, וּבְדִיעֲבַד מֻתָּר.


ג לִצְלִי צָרִיךְ חֲתִיכָה מִשּׁוּם דָּם שֶׁבַּסִמְפּוֹנוֹת, וְאִם לֹא קְרָעוֹ קֹדֶם צְלִיָּה (ח) יִקְרְעֶנּוּ אַחַר כָּךְ. וְיֵשׁ אוֹמְרִים דְּאֵין צָרִיךְ לַצְּלִי שׁוּם חֲתִיכָה כְּלָל (טוּר בְּשֵׁם ר''י סמ''ק ואו''ה וּשְׁאָר פּוֹסְקִים), וְכֵן (ט) נוֹהֲגִין אֲפִלּוּ לְכַתְּחִלָּה.


ד אִם צְלָאוֹ עִם בָּשָׂר בַּתַּנּוּרִים שֶׁבִּימֵי חַכְמֵי הַגְּמָרָא, שֶׁפִּיהֶם לְמַעְלָה, יִהְיֶה הַכָּבֵד לְמַטָּה וְלֹא לְמַעְלָה. וּבְדִיעֲבַד, (י) מֻתָּר. וּבַשִּׁפּוּדִים שֶׁצּוֹלִים אֵצֶל הָאֵשׁ, אָסוּר לִצְלוֹתָן עִם הַבָּשָׂר, לְכַתְּחִלָּה, אֲפִלּוּ כָּבֵד (יא) לְמַטָּה. הגה: מִיהוּ אִם נִמְלְחָה הַכָּבֵד כְּבָר, מֻתָּר לִצְלוֹתָהּ עִם בָּשָׂר, (יב) אֲפִלּוּ עַל גַּבֵּי בִּשְׂרָא, דִּכְבָר נִתְמַעֵט דָּמוֹ וְהָוֵי כְּבָשָׂר עַל גַּבֵּי בָּשָׂר. (בֵּית יוֹסֵף בְּשֵׁם תּוֹסָפוֹת ר''פ כ''צ וְהַגָּהוֹת ש''ד וּמָרְדְּכַי פ' כ''ה) .


ה לא יִמְלְחֶנּוּ לְכַתְּחִלָּה (יג) עַל גַּבֵּי הַבָּשָׂר, אֶלָּא מִתַּחְתָּיו. הגה: וְנָהֲגוּ שֶׁלֹּא לִמְלֹחַ כָּבֵד כְּלָל, אֲפִלּוּ (יד) (טו) לְבַדָּהּ, וְאֵין לְשַׁנּוֹת רַק יֵשׁ לְמָלְחָהּ קְצָת כְּשֶׁהִיא תְּחוּבָה בְּשִׁפּוּד, אוֹ מֻנַּחַת עַל הָאֵשׁ לִצְלוֹתָהּ (הגהת ש''ד ואגור בשם אגודה) . וּמִיהוּ, אִם אִיתָא שֶׁמָּלַח כָּבֵד, בֵּין שֶׁמְּלָחָהּ לְבַדָּהּ אוֹ עִם בָּשָׂר, אֲפִלּוּ עַל גַּבֵּי בִּשְׂרָא, הַכֹּל מֻתָּר (טוּר וש''ד וְאָרֹךְ) . וְיֵשׁ אוֹמְרִים שֶׁיֵּשׁ לִקְלֹף מְעַט סְבִיב הַכָּבֵד, אִם הִיא דְּבֻקָּה בָּעוֹף, וְאֵינוֹ אֶלָּא חֻמְרָא בְּעָלְמָא (מָרְדְּכַי פכ''ה וּבְאָרֹךְ) . נָהֲגוּ לְהָדִיחַ כָּל כָּבֵד (טז) אַחַר צְלִיָּתָהּ, מִשּׁוּם דָּם הַדָּבוּק בָּהּ (אָגוּר בְּשֵׁם ר''י מוּלִין) . מִיהוּ אִם לֹא (יז) הֱדִיחָהּ, מֻתֶּרֶת.


ו נִמְצָא כָּבֵד בְּעוֹף צָלִי, מֻתָּר. הגה: וְיֵשׁ אוֹמְרִים (יח) (יט) לִקְלֹף מְעַט סְבִיב הַכָּבֵד, וְאֵינוֹ אֶלָּא חֻמְרָא בְּעָלְמָא (מָרְדְּכַי בְּשֵׁם ריב''ן וְש''ד וְאו''ה) . וְאִם הוּא מְבֻשָּׁל, צָרִיךְ ס' כְּנֶגֶד הַכָּבֵד. הגה: וְאֵין לְךָ עוֹף שֶׁהוּא ס' נֶגֶד הַכָּבֵד, כְּשֶׁהִיא שְׁלֵמָה (ש''ד וּמָרְדְּכַי בְּשֵׁם ראבי''ה וּמהרא''י וּבְאָרֹךְ כְּלָל י''ו), וְלָכֵן אִם הַכָּבֵד שְׁלֵמָה וּדְבוּקָה בָּעוֹף, נַעֲשֶׂה הָעוֹף חֲתִיכַת נְבֵלָה וּבָעֵינָן ס' מִשְּׁאָר דְּבָרִים שֶׁבַּקְּדֵרָה נֶגֶד כָּל הָעוֹף. וְהוּא הַדִּין אִם דָּבוּק חֲתִיכַת כָּבֵד בַּחֲתִיכַת עוֹף, וְאֵין ס' כְּנֶגְדּוֹ, (שָׁם) דְּאַנָּן קַיְמָא לָן בְּכָל הָאִסּוּרִים חֲתִיכָה נַעֲשֵׂית נְבֵלָה, כְּדִלְקַמָּן סִימָן צ''ב. וְאִם אֵין הַכָּבֵד דְּבוּקָה, מִצְטָרֵף כָּל מַה שֶּׁבַּקְּדֵרָה לְבַטֵּל הַכָּבֵד, וְאִם יֵשׁ שִׁשִּׁים בַּכֹּל, הַכֹּל מֻתָּר. מִיהוּ הַכָּבֵד (כ) עַצְמָהּ אֲסוּרָה כְּמוֹ בַּלֵּב (תא''ו נט''ו), כְּמוֹ שֶׁנִּתְבָּאֵר לְעֵיל סִימָן ע''ב. עוֹף (כא) שֶׁמִּלְּאוּהוּ (כב) בְּבֵיצִים וְנִמְצָא בּוֹ לֵב אוֹ כָּבֵד, דִּינוֹ כִּמְבֻשָּׁל וּבָעֵינָן ס' מִן הָעוֹף בְּלֹא הַמִּלּוּי, וְאִם לָאו, הַכֹּל אָסוּר. וְאִם מִלְּאוּהוּ בְּבָשָׂר וְאֵין שָׁם בֵּיצִים הַנִּקְרָשִׁים וּמְעַכְּבִים הַדָּם, דִּינוֹ כְּצָלִי. (אָרֹךְ בְּשֵׁם אֲגֻדָּה וּבִנְיָמִין זְאֵב סִימָן שי''ז) .



סימן עד - הטחול דינו כשאר בשר, ובו סעיף אחד


א הַטְּחוֹל, אַף עַל פִּי שֶׁיֵּשׁ בּוֹ מַרְאֵה אַדְמוּמִית וְנִרְאֶה כְּרִבּוּי דָּם, דִּינוֹ כִּשְׁאָר (א) בָּשָׂר. הגה: יֵשׁ אוֹמְרִים שֶׁנָּהֲגוּ שֶׁלֹּא לְבַשֵּׁל הַכְּלָיוֹת אוֹ בֵּיצֵי זָכָר, אֲפִלּוּ לְאַחַר שֶׁנִּקֵּר אוֹתָן, מִשּׁוּם דְּאִית בְּהוּ רֹב דָּם. וּבְדִיעֲבַד אֵין לָחוּשׁ (אָגוּר וְאָרֹךְ כְּלָל א' בְּשֵׁם רוֹקֵחַ), וּמֻתָּר לִמְלֹחַ כָּל אֵלּוּ עִם שְׁאָר בָּשָׂר אַף עַל פִּי שֶׁיֵּשׁ בָּהֶם רֹב דָּם, וּבִלְבַד שֶׁיָּסִירוּ הַקְּרוּמִים וְהַחֵלֶב מֵהֶם.



סימן עה - דין מליחת המעים, ובו ג' סעיפים


א אֵין מַחְזִיקִין דָּם בִּבְנֵי מֵעַיִם, כְּגוֹן בַּכֶּרֶס וּבַקֵּבָה וּבַדַּקִּין. וּבַחַלְחֹלֶת (פֵּרוּשׁ, הַכַּרְכָּשָׁא, וְהוּא הַמְּעִי הַדָּבוּק בְּפִי הַטַּבַּעַת) בְּלֹא שֻׁמָּן שֶׁעֲלֵיהֶם. לְפִיכָךְ אִם בִּשְּׁלָם בִּקְדֵרָה בְּלֹא מְלִיחָה, מֻתָּר, אֶלָּא אִם כֵּן (א) אַסְמִיק. הגה: וְכָל שֶׁכֵּן אִם נִמְלְחוּ בִּכְלִי שֶׁאֵינוֹ מְנֻקָּב, (ב) דְּמֻתָּרִים (מָרְדְּכַי פכ''ה) . מִיהוּ לְכַתְּחִלָּה צְרִיכִים מְלִיחָה בִּכְלִי מְנֻקָּב, וַהֲדָחָה תְּחִלָּה, כִּשְׁאָר בָּשָׂר (מָרְדְּכַי וְאָרֹךְ) . וְיֵשׁ חוֹלְקִים בַּכָּרֵס, וְאוֹמְרִים דְּמַחֲזִיקִין בָּהּ דָּם, וַאֲפִלּוּ בְּדִיעֲבַד אָסוּר בַּכָּרֵס וּבֵית הַכּוֹסוֹת כִּשְׁאָר בָּשָׂר (מָרְדְּכַי ור''ן ואו''ה) . אֲבָל (ג) שֻׁמָּן שֶׁעֲלֵיהֶם, דִּינוֹ כִּשְׁאָר בָּשָׂר. לְפִיכָךְ כְּשֶׁמּוֹלְחִים הַחַלְחֹלֶת וּשְׁאָר מֵעַיִם, אֵין מוֹלְחִין אוֹתָם בְּצַד פְּנִימִי מַעֲבַר הַמַּאֲכָל, אֶלָּא בַּצַּד הַחִיצוֹן שֶׁשָּׁם הַשֻּׁמָן דָּבוּק. הגה: וְאִם מָלַח הַחַלְחֹלֶת מִצַּד פְּנִים וְלֹא מִצַּד חוּץ, הֲוֵי כְּאִלּוּ לֹא נִמְלְחָה כְּלָל, וְאִם נִתְבַּשְּׁלָה כָּךְ, (ד) (ה) טְרֵפָה, אִם יֵשׁ עָלֶיהָ שֻׁמָּן מִבַּחוּץ. וְאִם לֹא נִתְבַּשְּׁלָה עֲדַיִן, יַחֲזֹר (ו) (ז) וְיִמְלְחֶנָּה מִבַּחוּץ. וּמֻתֶּרֶת (אָרֹךְ בְּשֵׁם הָרַשְׁבָּ''א) . בֵּיצִים הַנִּמְצָאִים בָּעוֹפוֹת לְאַחַר שְׁחִיטָה, אִם לֹא נִגְמַר הַחֶלְבּוֹן שֶׁלָּהֶם, רַק הַחֶלְמוֹן לְבַד, צְרִיכִים מְלִיחָה כִּשְׁאָר בָּשָׂר, וּמֻתָּר לְמָלְחָן עִם בָּשָׂר (בֵּית יוֹסֵף סִימָן פ''ז) . וְאִם נִגְמַר אַף הַחֶלְבּוֹן, אֲפִלּוּ הַקְּלִפָּה קָּשָׁה עָלָיו כְּמוֹ שֶׁהִיא נִמְכֶּרֶת בַּשּׁוּק, נוֹהֲגִין לְמָלְחָהּ, אַךְ יִזָּהֵר שֶׁלֹּא לְמָלְחָהּ עִם בָּשָׂר (מהרא''י בְּהַגָּהַת ש''ד) . מִיהוּ בְּדִיעֲבַד, אֵין לָחוּשׁ. וְכָל שֻׁמָּן, אֲפִלּוּ שֶׁל עוֹפוֹת, דִּינוֹ כִּשְׁאָר בָּשָׂר לְעִנְיָן מְלִיחָה וַהֲדָחָה (מָרְדְּכַי) .


ב בְּנֵי מֵעַיִם בְּלֹא שֻׁמָּן שֶׁעֲלֵיהֶם, אַף עַל פִּי שֶׁאֵין מַחֲזִיקִים בָּהֶם דָּם, מֻתָּר (ח) לְמָלְחָם עִם בָּשָׂר. הגה: וְיֵשׁ אוֹסְרִים לְמָלְחָם עִם בָּשָׂר (טוּר בְּשי''א), וְהָכֵי נוֹהֲגִין (ט) לְכַתְּחִלָּה, וּבְדִיעֲבַד מֻתָּר (מָרְדְּכַי בְּשֵׁם ר' יוֹאֵל וְהַגָּהוֹת אֲשֵׁרִ''י וְאו''ה בְּשֵׁם אוֹר זָרוּעַ) .


ג שֻׁמַּן הַכַּנְתָּא (פֵּרוּשׁ שֻׁמַּן הַדַּקִים אינטרי''לייא בְּלַעַ''ז) שֶׁעַל הַדַּקִּים, מְלֵאָה חוּטִים דַּקִּים מְלֵאִים דָּם וּלְפִיכָךְ יֵשׁ אוֹסְרִים אוֹתָהּ לִקְדֵרָה, אֲפִלּוּ עַל יְדֵי חֲתִיכָה וּמְלִיחָה. וְעַכְשָׁו נוֹהֲגִין הֶתֵּר, וְאוֹתָם חוּטִין שֶׁל דָּם כְּשֶׁאָדָם בָּקִי בָּהֶם מוֹתְחָן, וְנִפְשָׁלִים הֵיטֵב עַד שֶׁלֹּא יִשָּׁאֵר אֶחָד מֵהֶם.



סימן עו - דין בשר לצלי, ובו ו' סעיפים


א הַצְּלִי אֵין צָרִיךְ מְלִיחָה, (א) לְפִי שֶׁהָאֵשׁ שׁוֹאֵב הַדָּם שֶׁבּוֹ מֵעַצְמוֹ. אֲבָל אִם דָּם אַחֵר נָטַף עַל הַצְּלִי, אֲפִלּוּ אוֹתוֹ דָּם הוּא צוֹנֵן, לֹא אָמְרִינָן שֶׁהָאֵשׁ שׁוֹאֲבוֹ, וְאוֹסֵר מִמֶּנּוּ כְּדֵי (ב) נְטִילָה.


ב רָצָה לִמְלֹחַ צְלִי וּלְאָכְלוֹ (ג) בְּלֹא הֲדָחָה, עוֹשֶׂה, וְאֵין לָחוּשׁ לְדָם שֶׁעַל הַמֶּלַח. וְיֵשׁ מִי שֶׁאוֹמֵר דְּהָנֵי מִלֵּי כְּשֶׁמּוֹלְחוֹ (ד) וְצוֹלֶהוּ מִיָּד, אֲבָל אִם שָׁהָה בְּמִלְחוֹ, הַמֶּלַח בּוֹלֵעַ וְנֶאֱסָר, לְפִיכָךְ (ה) מְדִיחוֹ יָפֶה קֹדֶם צְלִיָּה. הגה: (ו) וְיֵשׁ אוֹמְרִים דְּצָלִּי בָּעֵי הֲדָחָה תְּחִלָּה (רַמְבַּ''ן) וְיֵשׁ אוֹמְרִים דְּצָרִיךְ גַּם כֵּן קְצָת מְלִיחָה תְּחִלָּה (רַשִׁ''י) . וְהַמִּנְהָג לְהָדִיחַ תְּחִלָּה וְגַם לְמָלְחוֹ קְצָת כַּאֲשֶׁר נִתְחָב בְּשִׁפּוּד, וְצוֹלֶהוּ מִיָּד קֹדֶם שֶׁיִּתְמַלֵּא הַמֶּלַח דָּם (הַגָּהוֹת ש''ד) . מִיהוּ אִם לֹא הֱדִיחוֹ וְלֹא מְלָחוֹ כְּלָל, אוֹ מְלָחוֹ בְּלֹא הֲדָחָה תְּחִלָּה, וְנִצְלָה כָּךְ, מֻתָּר, וּבִלְבַד שֶׁלֹּא שָׁהָה כָּךְ בִּמְלִיחָתוֹ בְּלֹא הֲדָחָה שִׁעוּר מְלִיחָתוֹ, אֲבָל אִם שָׁהָה כָּל כָּךְ קֹדֶם שֶׁצְּלָאוֹ, (ז) אָסוּר (כָּךְ מַשְׁמָע בְּאָרֹךְ כְּלָל ה') . וְאֵין חִלּוּק בְּכָל זֶה בֵּין אַוָּזוֹת וּשְׁאָר (ח) עוֹפוֹת הַחֲלוּלִים (מָרְדְּכַי וְש''ד וּבֵית יוֹסֵף וְעת''ח כ''ט דִּין ו'), וּבִלְבַד שֶׁלֹּא יִהְיוּ מְלוּאִים בְּבֵיצִים אוֹ בִּשְׁאָר (ט) בָּשָׂר. אֲבָל אִם הֵם מְלוּאִים, דִּינָם כְּבִשּׁוּל, וּצְרִיכִים (י) מְלִיחָה תְּחִלָּה כְּמוֹ לִקְדֵרָה (ש''ד) . וְנָהֲגוּ לְהַחְמִיר כְּשֶׁצּוֹלִין בָּשָׂר בְּלֹא מְלִיחָה, (שֶׁלֹּא) לְהַפֵּךְ הַשִּׁפּוּד תָּמִיד כְּדֵי שֶׁיָּזוּב הַדָּם (אָרֹךְ כְּלָל ט'), וּבְדִיעֲבַד אֵין לָחוּשׁ. וְאֵין חִלּוּק בְּכָל זֶה בֵּין אִם רוֹצֶה (יא) לְאָכְלוֹ כָּךְ צָלִי, אוֹ רוֹצֶה לְבַשְּׁלוֹ אַחַר כָּךְ, רַק שֶׁיִּצְלֶנּוּ תְּחִלָּה כְּדֵי שֶׁיְּהֵא רָאוּי לַאֲכִילָה, דְּהַיְנוּ כַּחֲצִי (יב) צְלִיָּתוֹ (שָׁם) . יֵשׁ אוֹמְרִים דְּכָל צְלִי צָרִיךְ הֲדָחָה אַחַר צְלִיָּתוֹ, מִשּׁוּם דָּם הַדָּבוּק בּוֹ (אֲגֻדָּה וְאָגוּר), וְכֵן נוֹהֲגִין לְכַתְּחִלָּה. מִיהוּ אִם לֹא הֱדִיחוֹ, וַאֲפִלּוּ בִּשְּׁלוֹ כָּךְ, מֻתָּר (עַיֵּן לְעֵיל סִימָן ע''ג) . וַאֲפִלּוּ נִמְלַח תְּחִלָּה קֹדֶם צְלִיָּתוֹ וְלֹא הוּדַח אַחַר הַמְּלִיחָה, אֲפִלּוּ הָכֵי מֻתָּר וְלֹא חַיְישִׁינָן לְדָם וּמֶלַח שֶׁעָלָיו, דְּנוּרָא מִשְׁאָב שָׁאִיב (או''ה) .


ג לֹא נָקַב הַוְּרִידִים (בָּעוֹף) בִּשְׁעַת שְׁחִיטָה, אָסוּר לְאָכְלוֹ, וַאֲפִלּוּ צָלִי, עַד שֶׁיַּחְתֹּךְ אֵבֶר אֵבֶר וְיִצְלֶה. רָצָה לֶאֱכֹל מִמֶּנּוּ בָּשָׂר חַי, אָסוּר עַד שֶׁיַּחְתֹּךְ (יג) וְיִמְלַח. וְאִם נִקֵּר הַבָּשָׂר מֵחוּטֵי דָּם, אוֹכֵל אֲפִלּוּ בְּלֹא מְלִיחָה, חַי אוֹ (יד) (טו) צָלִי, וַאֲפִלּוּ כֻּלּוֹ כְּאֶחָד. וְיֵשׁ שֶׁהֶחֱמִירוּ שֶׁלֹּא לְאָכְלוֹ כֻּלּוֹ כְּאֶחָד, אֲפִלּוּ בְּצָלִי אֶלָּא לְאַחַר חֲתִיכַת הַוְּרִידִין (וְרֹב שְׁנֵי הַסִימָנִים) .


ד יֵשׁ מִי שֶׁאוֹסֵר לַחְתֹּךְ בְּסַכִּין צָלִי שֶׁאֵצֶל הָאֵשׁ שֶׁלֹּא נִמְלַח, כָּל זְמַן שֶׁאֵינוֹ נִצְלֶה כָּל צָרְכּוֹ, מִפְּנֵי דָּם שֶׁנִּבְלַע בַּסַכִּין. וְיֵשׁ מִי שֶׁאוֹסֵר (טז) (יז) הַשִּׁפּוּד שֶׁצָּלוּ בּוֹ בָּשָׂר בְּלֹא מְלִיחָה. וְיֵשׁ מִי שֶׁהוֹרָה שֶׁאָסוּר לְהַשְׁהוֹת הַצָלִי עַל הַשִּׁפּוּד לְאַחַר שֶׁהוּסַר מִן הָאֵשׁ, לְאַחַר שֶׁפָּסַק הַבָּשָׂר מִלָּזוּב, שֶׁמָּא יַחֲזֹר הַבָּשָׂר הַחַם וְיִבְלַע מִמֶּנּוּ. וְיֵשׁ מַתִּירִין בְּכָל זֶה, וְכֵן הַמִּנְהָג לְהַתִּיר. וְאָנוּ נִזְהָרִין לְכַתְּחִלָּה, וּמַתִּירִין (יח) בְּדִיעֲבַד (מהרא''י וְהַגָּהָת ש''ד וְאו''ה כְּלָל ל''ז) .


ה צָלִי שֶׁלֹּא נִמְלַח וַחֲתָכוֹ עַל גַּבֵּי כִּכָּר, אַף עַל פִּי שֶׁיֵּשׁ בַּכִּכָּר מַרְאֵה אֹדֶם, מֻתָּר אִם נִצְלֶה עַד כְּדֵי שֶׁהוּא רָאוּי לַאֲכִילָה לְרֹב בְּנֵי אָדָם, דְּהַיְנוּ חֲצִי צְלִיָּתוֹ. וְהוּא הַדִּין (יט) שֶׁהַמּוֹהַל בְּלֹא כִּכָּר נָמֵי שָׁרֵי (בֵּית יוֹסֵף בְּשֵׁם הָרַשְׁבָּ''א) .


ו בָּשָׂר שֶׁצּוֹלִין בְּלֹא מְלִיחָה, אֵין נוֹתְנִין כְּלִי תַּחְתָּיו לְקַבֵּל שַׁמְנוּנִית הַנּוֹטֵף מִמֶּנּוּ, עַד שֶׁיִּצָּלֶה עַד שֶׁיְּהֵא רָאוּי (כ) לַאֲכִילָה עַל יְדֵי צְלִיָּה זוֹ.



סימן עז - עופות שנמלאו בשר שלא נמלח, ובו סעיף אחד


א עוֹפוֹת אוֹ גְּדָיִים שֶׁמְּמַלְּאִים אוֹתָם בָּשָׂר שֶׁלֹּא (א) נִמְלַח, אִם לְצָלִי, מֻתָּר אֲפִלּוּ פִּיהֶם לְמַעְלָה, וַאֲפִלּוּ אִם נִמְלַח הַחִיצוֹן, דִּכְמוֹ שֶׁבּוֹלֵעַ דַּם הַמִּלוּאִים כָּךְ פּוֹלְטוֹ. הגה: וכ''ש אִם נִמְלַח (ב) הַפְּנִימִי וְלֹא הַחִיצוֹן, דְּשָׁרֵי (ד''ע לְדַעַת אָרֹךְ כְּלָל ט'), דְּנוּרָא מִשְׁאָב שָׁאִיב הַדָּם מִן הַחִיצוֹן וְאֵינוֹ נִבְלָע בִּפְנִימִי (ד''ע) . וְלִקְדֵרָה אָסוּר עַד שֶׁיִּמְלַח (ג) חִיצוֹן לְבַדּוֹ וּפְנִימִי לְבַדּוֹ, אֲבָל אִם לְאַחַר שֶׁמִּלְּאוֹ מָלַח הַחִיצוֹן, אֵינוֹ (ד) מַפְלִיט דָּם (ה) שֶׁבַּפְּנִימִי. וְכָל זֶה בְּדִיעֲבַד, אֲבָל לְכַתְּחִלָּה אֵין לַעֲשׂוֹת שׁוּם מֻלְיָתָא, רַק אִם נִמְלְחוּ שְׁנֵיהֶם (אָגוּר ומָרְדְּכַי פכ''ה וְרַשִׁ''י) . וְכָל זֶה מַיְרֵי שֶׁאֵין בְּמוּלְיָתָא זוֹ רַק (ו) בָּשָׂר אוֹ עֲשָׂבִים, אֲבָל אִם יֵשׁ שָׁם בֵּיצִים הַנִּקְרָשִׁים, דִּינוֹ כְּאִלּוּ נִתְבַּשֵּׁל בִּקְדֵרָה (הַגָּהַת ש''ד בְּשֵׁם מהרי''ט וְאָרֹךְ וב''ז), וְאֲפִלּוּ בְּדִיעֲבַד יֵשׁ לְאֶסוֹר אִם לֹא נִמְלְחוּ (ז) שְׁנֵיהֶם. בָּשָׂר שֶׁלֹּא נִמְלַח וְלֹא הוּדַח שֶׁנִּצְלָה עִם בָּשָׂר שֶׁנִּמְלַח וְהוּדַח, מֻתָּר בְּדִיעֲבַד (שָׁם) . אֲבָל לְכַתְּחִלָּה אָסוּר לִצְלוֹת בָּשָׂר שֶׁנִּמְלַח עִם הַבָּשָׂר שֶׁלֹּא נִמְלַח, וְאֲפִלּוּ הוּדַח.



סימן עח - דין שלא לדבק בצק בבשר שלא נמלח, ובו סעיף אחד


א הַטּוֹפֵל בָּצֵק בְּעוֹף שֶׁלֹּא נִמְלַח, אַף עַל גַּב דְּבַגְּמָרָא מַפְלִיג בֵּין סְמִידָא לִשְׁאָר קִמְחֵי וּבֵין אַסְמִיק לְלֹא אַסְמִיק, אֲנַן הַשְׁתָּא לֹא בְּקִיאִין בְּמִלְּתָא, וּבְכָל (א) גַּוְנָא יֵשׁ לֶאֱסֹר. אֲבָל אִם נִמְלַח, וְשָׁהָה כְּדֵי מְלִיחָה וְאַחַר כָּךְ הוּדַח, מֻתָּר בְּכָל גַּוְנָא. הגה: וְדַוְקָא לְטַפֵּל בְּבָצֵק, אָסוּר, אֲבָל מֻתָּר לִמְשֹׁחַ בְּשׁוּמָן אוֹ (ב) בְּצִיר בָּשָׂר שֶׁלֹּא נִמְלַח, דְּאֵין זֶה מְעַכֵּב הַפְּלִיטָה (אָרֹךְ כְּלָל ט') . (ג) וּפַשְׁטִידָ''א יֵשׁ לוֹ כָּל דִּין בִּשּׁוּל בִּקְדֵרָה, בֵּין (ד) לְקֻלָּא בֵּין לְחֻמְרָא, (מָרְדְּכַי פכ''ה בְּשֵׁם רַשִׁ''י וּמהרא''י וְאָרֹךְ כְּלָל ט''ו) .
