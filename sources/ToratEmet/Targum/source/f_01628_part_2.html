פרשת נח


{ט} אֵלֶּה תּוֹלְדֹת נֹחַ נֹחַ אִישׁ צַדִּיק תָּמִים הָיָה בְּדֹרֹתָיו אֶת הָאֱלֹהִים הִתְהַלֶּךְ נֹחַ:  מ  אֵלֶּה תּוֹלְדֹת נֹחַ נֹחַ אִישׁ צַדִּיק תָּמִים הָיָה בְּדֹרֹתָיו אֶת הָאֱלֹהִים הִתְהַלֶּךְ נֹחַ:  ת  אִלֵן תּוּלְדַת נֹחַ נֹחַ גְּבַר זַכַּאי שְׁלִים הֲוָה בְּדָרוֹהִי בְּדַחַלְתָּא דַּיְיָ הַלִּיךְ נֹחַ:

{י} וַיּוֹלֶד נֹחַ שְׁלֹשָׁה בָנִים אֶת שֵׁם אֶת חָם וְאֶת יָפֶת:  מ  וַיּוֹלֶד נֹחַ שְׁלֹשָׁה בָנִים אֶת שֵׁם אֶת חָם וְאֶת יָפֶת:  ת  וְאוֹלִיד נֹחַ תְּלָתָא בְנִין יָת שֵׁם יָת חָם וְיָת יָפֶת:

{יא} וַתִּשָּׁחֵת הָאָרֶץ לִפְנֵי הָאֱלֹהִים וַתִּמָּלֵא הָאָרֶץ חָמָס:  מ  וַתִּשָּׁחֵת הָאָרֶץ לִפְנֵי הָאֱלֹהִים וַתִּמָּלֵא הָאָרֶץ חָמָס:  ת  וְאִתְחַבָּלַת אַרְעָא קֳדָם יְיָ וְאִתְמְלִיאַת אַרְעָא חֲטּוֹפִין:

{יב} וַיַּרְא אֱלֹהִים אֶת הָאָרֶץ וְהִנֵּה נִשְׁחָתָה כִּי הִשְׁחִית כָּל בָּשָׂר אֶת דַּרְכּוֹ עַל הָאָרֶץ:  מ  וַיַּרְא אֱלֹהִים אֶת הָאָרֶץ וְהִנֵּה נִשְׁחָתָה כִּי הִשְׁחִית כָּל בָּשָׂר אֶת דַּרְכּוֹ עַל הָאָרֶץ:  ת  וַחֲזָא יְיָ יָת אַרְעָא וְהָא אִתְחַבָּלַת אֲרֵי חַבִּילוּ כָּל בִּסְרָא אֱנַשׁ יָת אָרְחֵהּ עַל אַרְעָא: (ס)

{יג} וַיֹּאמֶר אֱלֹהִים לְנֹחַ קֵץ כָּל בָּשָׂר בָּא לְפָנַי כִּי מָלְאָה הָאָרֶץ חָמָס מִפְּנֵיהֶם וְהִנְנִי מַשְׁחִיתָם אֶת הָאָרֶץ:  מ  וַיֹּאמֶר אֱלֹהִים לְנֹחַ קֵץ כָּל בָּשָׂר בָּא לְפָנַי כִּי מָלְאָה הָאָרֶץ חָמָס מִפְּנֵיהֶם וְהִנְנִי מַשְׁחִיתָם אֶת הָאָרֶץ:  ת  וַאֲמַר יְיָ לְנֹחַ קִצָא דְכָל בִּסְרָא עַלַּת לִקֳדָמַי אֲרֵי אִתְמְלִיאַת אַרְעָא חֲטוֹפִין מִן קֳדָם עוֹבָדֵיהוֹן בִּישַׁיָּא וְהָא אֲנָא מְחַבֶּלְהוֹן עִם אַרְעָא:

{יד} עֲשֵׂה לְךָ תֵּבַת עֲצֵי גֹפֶר קִנִּים תַּעֲשֶׂה אֶת הַתֵּבָה וְכָפַרְתָּ אֹתָהּ מִבַּיִת וּמִחוּץ בַּכֹּפֶר:  מ  עֲשֵׂה לְךָ תֵּבַת עֲצֵי גֹפֶר קִנִּים תַּעֲשֶׂה אֶת הַתֵּבָה וְכָפַרְתָּ אֹתָהּ מִבַּיִת וּמִחוּץ בַּכֹּפֶר:  ת  עִבֵד לָךְ תֵּבוֹתָא דְאָעִין דְּקַדְרוֹם מְדוֹרִין תַּעְבֵּד יָת תֵּבוֹתָא וְתֶחְפֵּי יָתַהּ מִגָּו וּמִבָּרָא בְּכּוּפְרָא:

{טו} וְזֶה אֲשֶׁר תַּעֲשֶׂה אֹתָהּ שְׁלֹשׁ מֵאוֹת אַמָּה אֹרֶךְ הַתֵּבָה חֲמִשִּׁים אַמָּה רָחְבָּהּ וּשְׁלֹשִׁים אַמָּה קוֹמָתָהּ:  מ  וְזֶה אֲשֶׁר תַּעֲשֶׂה אֹתָהּ שְׁלֹשׁ מֵאוֹת אַמָּה אֹרֶךְ הַתֵּבָה חֲמִשִּׁים אַמָּה רָחְבָּהּ וּשְׁלֹשִׁים אַמָּה קוֹמָתָהּ:  ת  וְדֵין דְּתַעְבֵּד יָתַהּ תְּלַת מְאָה אַמִּין אֻרְכָּא דְתֵבוֹתָא חַמְשִׁין אַמִּין פְּתָיַהּ וּתְלָתִין אַמִּין רוּמַהּ:

{טז} צֹהַר תַּעֲשֶׂה לַתֵּבָה וְאֶל אַמָּה תְּכַלֶנָּה מִלְמַעְלָה וּפֶתַח הַתֵּבָה בְּצִדָּהּ תָּשִׂים תַּחְתִּיִּם שְׁנִיִּם וּשְׁלִשִׁים תַּעֲשֶׂהָ:  מ  צֹהַר תַּעֲשֶׂה לַתֵּבָה וְאֶל אַמָּה תְּכַלֶנָּה מִלְמַעְלָה וּפֶתַח הַתֵּבָה בְּצִדָּהּ תָּשִׂים תַּחְתִּיִּם שְׁנִיִּם וּשְׁלִשִׁים תַּעֲשֶׂהָ:  ת  נֵהוֹר תַּעְבֵּיד לְתֵבוֹתָא וּלְאַמְתָא תְּשַׁכְלְלִינַהּ מִלְּעֵלָּא וְתַרְעָא דְתֵבוֹתָא בְּסִטְרַהּ תְּשַׁוִּי מְדוֹרִין אֲרָעִין תִּנְיָנִין וּתְלִיתָאִין תַּעְבְּדִנַּהּ:

{יז} וַאֲנִי הִנְנִי מֵבִיא אֶת הַמַּבּוּל מַיִם עַל הָאָרֶץ לְשַׁחֵת כָּל בָּשָׂר אֲשֶׁר בּוֹ רוּחַ חַיִּים מִתַּחַת הַשָּׁמָיִם כֹּל אֲשֶׁר בָּאָרֶץ יִגְוָע:  מ  וַאֲנִי הִנְנִי מֵבִיא אֶת הַמַּבּוּל מַיִם עַל הָאָרֶץ לְשַׁחֵת כָּל בָּשָׂר אֲשֶׁר בּוֹ רוּחַ חַיִּים מִתַּחַת הַשָּׁמָיִם כֹּל אֲשֶׁר בָּאָרֶץ יִגְוָע:  ת  וַאֲנָא הָא אֲנָא מַיְתֵי יָת טוֹפָנָא מַיָּא עַל אַרְעָא לְחַבָּלָא כָּל בִּסְרָא דִּי בֵהּ רוּחָא דְחַיֵּי מִתְּחוֹת שְׁמַיָּא כֹּל דִּי בְאַרְעָא יְמוּת:

{יח} וַהֲקִמֹתִי אֶת בְּרִיתִי אִתָּךְ וּבָאתָ אֶל הַתֵּבָה אַתָּה וּבָנֶיךָ וְאִשְׁתְּךָ וּנְשֵׁי בָנֶיךָ אִתָּךְ:  מ  וַהֲקִמֹתִי אֶת בְּרִיתִי אִתָּךְ וּבָאתָ אֶל הַתֵּבָה אַתָּה וּבָנֶיךָ וְאִשְׁתְּךָ וּנְשֵׁי בָנֶיךָ אִתָּךְ:  ת  וְאָקֵם יָת קְיָמִי עִמָּךְ וְתֵעוֹל לְתֵבוֹתָא אַתְּ וּבְנָךְ וְאִתְּתָךְ וּנְשֵׁי בְנָיךְ עִמָּךְ:

{יט} וּמִכָּל הָחַי מִכָּל בָּשָׂר שְׁנַיִם מִכֹּל תָּבִיא אֶל הַתֵּבָה לְהַחֲיֹת אִתָּךְ זָכָר וּנְקֵבָה יִהְיוּ:  מ  וּמִכָּל הָחַי מִכָּל בָּשָׂר שְׁנַיִם מִכֹּל תָּבִיא אֶל הַתֵּבָה לְהַחֲיֹת אִתָּךְ זָכָר וּנְקֵבָה יִהְיוּ:  ת  וּמִכָּל דְּחַי מִכָּל בִּסְרָא תְּרֵין מִכֹּלָּא תָּעֵיל לְתֵבוֹתָא לְקַיָּמָא עִמָּךְ דְּכַר וְנוּקְבָא יְהוֹן:

{כ} מֵהָעוֹף לְמִינֵהוּ וּמִן הַבְּהֵמָה לְמִינָהּ מִכֹּל רֶמֶשׂ הָאֲדָמָה לְמִינֵהוּ שְׁנַיִם מִכֹּל יָבֹאוּ אֵלֶיךָ לְהַחֲיוֹת:  מ  מֵהָעוֹף לְמִינֵהוּ וּמִן הַבְּהֵמָה לְמִינָהּ מִכֹּל רֶמֶשׂ הָאֲדָמָה לְמִינֵהוּ שְׁנַיִם מִכֹּל יָבֹאוּ אֵלֶיךָ לְהַחֲיוֹת:  ת  מֵעוֹפָא לִזְנוֹהִי וּמִן בְּעִירָא לִזְנַהּ וּמִכֹּל רִחֲשָׁא דְאַרְעָא לִזְנוֹהִי תְּרֵין מִכֹּלָּא יַעֲלוּן לְוָתָךְ לְקַיָּמָא:

{כא} וְאַתָּה קַח לְךָ מִכָּל מַאֲכָל אֲשֶׁר יֵאָכֵל וְאָסַפְתָּ אֵלֶיךָ וְהָיָה לְךָ וְלָהֶם לְאָכְלָה:  מ  וְאַתָּה קַח לְךָ מִכָּל מַאֲכָל אֲשֶׁר יֵאָכֵל וְאָסַפְתָּ אֵלֶיךָ וְהָיָה לְךָ וְלָהֶם לְאָכְלָה:  ת  וְאַתְּ סַב לָךְ מִכָּל מֵיכַל דְּמִתְאֲכֵל וְתִכְנוֹשׁ לְוָתָךְ וִיהֵי לָךְ וּלְהוֹן לְמֵיכָל:

{כב} וַיַּעַשׂ נֹחַ כְּכֹל אֲשֶׁר צִוָּה אֹתוֹ אֱלֹהִים כֵּן עָשָׂה:  מ  וַיַּעַשׂ נֹחַ כְּכֹל אֲשֶׁר צִוָּה אֹתוֹ אֱלֹהִים כֵּן עָשָׂה:  ת  וַעֲבַד נֹחַ כְּכֹל דִּי פַקֵּיד יָתֵהּ יְיָ כֵּן עֲבָד:




בראשית פרק-ז

{א}  שני  וַיֹּאמֶר יְהוָה לְנֹחַ בֹּא אַתָּה וְכָל בֵּיתְךָ אֶל הַתֵּבָה כִּי אֹתְךָ רָאִיתִי צַדִּיק לְפָנַי בַּדּוֹר הַזֶּה:  מ  וַיֹּאמֶר יְהוָה לְנֹחַ בֹּא אַתָּה וְכָל בֵּיתְךָ אֶל הַתֵּבָה כִּי אֹתְךָ רָאִיתִי צַדִּיק לְפָנַי בַּדּוֹר הַזֶּה:  ת  וַאֲמַר יְיָ לְנֹחַ עוּל אַתְּ וְכָל אֱנַשׁ בֵּיתָךְ לְתֵבוֹתָא אֲרֵי יָתָךְ חֲזֵיתִי זַכַּאי קֳדָמַי בְּדָרָא הָדֵין:

{ב} מִכֹּל הַבְּהֵמָה הַטְּהוֹרָה תִּקַּח לְךָ שִׁבְעָה שִׁבְעָה אִישׁ וְאִשְׁתּוֹ וּמִן הַבְּהֵמָה אֲשֶׁר לֹא טְהֹרָה הִוא שְׁנַיִם אִישׁ וְאִשְׁתּוֹ:  מ  מִכֹּל הַבְּהֵמָה הַטְּהוֹרָה תִּקַּח לְךָ שִׁבְעָה שִׁבְעָה אִישׁ וְאִשְׁתּוֹ וּמִן הַבְּהֵמָה אֲשֶׁר לֹא טְהֹרָה הִוא שְׁנַיִם אִישׁ וְאִשְׁתּוֹ:  ת  מִכֹּל בְּעִירָא דַכְיָא תִּסַּב לָךְ שַׁבְעָא שַׁבְעָא דְּכַר וְנוּקְבָא וּמִן בְּעִירָא דְּלָא אִיתָהָא דַכְיָא הִיא תְּרֵין דְּכַר וְנוּקְבָא:

{ג} גַּם מֵעוֹף הַשָּׁמַיִם שִׁבְעָה שִׁבְעָה זָכָר וּנְקֵבָה לְחַיּוֹת זֶרַע עַל פְּנֵי כָל הָאָרֶץ:  מ  גַּם מֵעוֹף הַשָּׁמַיִם שִׁבְעָה שִׁבְעָה זָכָר וּנְקֵבָה לְחַיּוֹת זֶרַע עַל פְּנֵי כָל הָאָרֶץ:  ת  אַף מֵעוֹפָא דִשְׁמַיָּא שַׁבְעָא שַׁבְעָא דְּכַר וְנוּקְבָא לְקַיָּמָא זַרְעָא עַל אַפֵּי כָל אַרְעָא:

{ד} כִּי לְיָמִים עוֹד שִׁבְעָה אָנֹכִי מַמְטִיר עַל הָאָרֶץ אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה וּמָחִיתִי אֶת כָּל הַיְקוּם אֲשֶׁר עָשִׂיתִי מֵעַל פְּנֵי הָאֲדָמָה:  מ  כִּי לְיָמִים עוֹד שִׁבְעָה אָנֹכִי מַמְטִיר עַל הָאָרֶץ אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה וּמָחִיתִי אֶת כָּל הַיְקוּם אֲשֶׁר עָשִׂיתִי מֵעַל פְּנֵי הָאֲדָמָה:  ת  אֲרֵי לִזְמַן יוֹמִין עוֹד שַׁבְעָא אֲנָא מָחֵת מִטְרָא עַל אַרְעָא אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן וְאֶמְחֵי יָת כָּל יְקוּמָא דִּי עַבְדֵת מֵעַל אַפֵּי אַרְעָא:

{ה} וַיַּעַשׂ נֹחַ כְּכֹל אֲשֶׁר צִוָּהוּ יְהוָה:  מ  וַיַּעַשׂ נֹחַ כְּכֹל אֲשֶׁר צִוָּהוּ יְהוָה:  ת  וַעֲבַד נֹחַ כְּכֹל דִּי פַקְּדֵהּ יְיָ:

{ו} וְנֹחַ בֶּן שֵׁשׁ מֵאוֹת שָׁנָה וְהַמַּבּוּל הָיָה מַיִם עַל הָאָרֶץ:  מ  וְנֹחַ בֶּן שֵׁשׁ מֵאוֹת שָׁנָה וְהַמַּבּוּל הָיָה מַיִם עַל הָאָרֶץ:  ת  וְנֹחַ בַּר שִׁתְּ מְאָה שְׁנִין וְטּוֹפָנָא הֲוָה מַיָּא עַל אַרְעָא:

{ז} וַיָּבֹא נֹחַ וּבָנָיו וְאִשְׁתּוֹ וּנְשֵׁי בָנָיו אִתּוֹ אֶל הַתֵּבָה מִפְּנֵי מֵי הַמַּבּוּל:  מ  וַיָּבֹא נֹחַ וּבָנָיו וְאִשְׁתּוֹ וּנְשֵׁי בָנָיו אִתּוֹ אֶל הַתֵּבָה מִפְּנֵי מֵי הַמַּבּוּל:  ת  וְעָאל נֹחַ וּבְנוֹהִי וְאִתְּתֵהּ וּנְשֵׁי בְנוֹהִי עִמֵּהּ לְתֵבוֹתָא מִן קֳדָם מֵי טוֹפָנָא:

{ח} מִן הַבְּהֵמָה הַטְּהוֹרָה וּמִן הַבְּהֵמָה אֲשֶׁר אֵינֶנָּה טְהֹרָה וּמִן הָעוֹף וְכֹל אֲשֶׁר רֹמֵשׂ עַל הָאֲדָמָה:  מ  מִן הַבְּהֵמָה הַטְּהוֹרָה וּמִן הַבְּהֵמָה אֲשֶׁר אֵינֶנָּה טְהֹרָה וּמִן הָעוֹף וְכֹל אֲשֶׁר רֹמֵשׂ עַל הָאֲדָמָה:  ת  מִן בְּעִירָא דַּכְיָא וּמִן בְּעִירָא דִּי לֵיתָהָא דַּכְיָא וּמִן עוֹפָא וְכֹל דִּי רָחֵשׁ עַל אַרְעָא:

{ט} שְׁנַיִם שְׁנַיִם בָּאוּ אֶל נֹחַ אֶל הַתֵּבָה זָכָר וּנְקֵבָה כַּאֲשֶׁר צִוָּה אֱלֹהִים אֶת נֹחַ:  מ  שְׁנַיִם שְׁנַיִם בָּאוּ אֶל נֹחַ אֶל הַתֵּבָה זָכָר וּנְקֵבָה כַּאֲשֶׁר צִוָּה אֱלֹהִים אֶת נֹחַ:  ת  תְּרֵין תְּרֵין עַלּוּ לְוַת נֹחַ לְתֵבוֹתָא דְּכַר וְנוּקְבָא כְּמָא דִּי פַּקֵּיד יְיָ יָת נֹחַ:

{י} וַיְהִי לְשִׁבְעַת הַיָּמִים וּמֵי הַמַּבּוּל הָיוּ עַל הָאָרֶץ:  מ  וַיְהִי לְשִׁבְעַת הַיָּמִים וּמֵי הַמַּבּוּל הָיוּ עַל הָאָרֶץ:  ת  וַהֲוָה לִזְמַן שִׁבְעַת יוֹמִין וּמֵי טוֹפָנָא הֲווֹ עַל אַרְעָא:

{יא} בִּשְׁנַת שֵׁשׁ מֵאוֹת שָׁנָה לְחַיֵּי נֹחַ בַּחֹדֶשׁ הַשֵּׁנִי בְּשִׁבְעָה עָשָׂר יוֹם לַחֹדֶשׁ בַּיּוֹם הַזֶּה נִבְקְעוּ כָּל מַעְיְנֹת תְּהוֹם רַבָּה וַאֲרֻבֹּת הַשָּׁמַיִם נִפְתָּחוּ:  מ  בִּשְׁנַת שֵׁשׁ מֵאוֹת שָׁנָה לְחַיֵּי נֹחַ בַּחֹדֶשׁ הַשֵּׁנִי בְּשִׁבְעָה עָשָׂר יוֹם לַחֹדֶשׁ בַּיּוֹם הַזֶּה נִבְקְעוּ כָּל מַעְיְנֹת תְּהוֹם רַבָּה וַאֲרֻבֹּת הַשָּׁמַיִם נִפְתָּחוּ:  ת  בִּשְׁנַת שִׁתְּ מְאָה שְׁנִין לְחַיֵּי נֹחַ בְּיַרְחָא תִּנְיָנָא בְּשִׁבְעַת עַשְׂרָא יוֹמָא לְיַרְחָא בְּיוֹמָא הָדֵין אִתְבְּזָעוּ כָּל מַבּוּעֵי תְּהוֹמָא רַבָּא וְכַוֵּי שְׁמַיָּא אִתְפַּתָּחוּ:

{יב} וַיְהִי הַגֶּשֶׁם עַל הָאָרֶץ אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה:  מ  וַיְהִי הַגֶּשֶׁם עַל הָאָרֶץ אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה:  ת  וַהֲוָה מִטְרָא נָחֵת עַל אַרְעָא אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן:

{יג} בְּעֶצֶם הַיּוֹם הַזֶּה בָּא נֹחַ וְשֵׁם וְחָם וָיֶפֶת בְּנֵי נֹחַ וְאֵשֶׁת נֹחַ וּשְׁלֹשֶׁת נְשֵׁי בָנָיו אִתָּם אֶל הַתֵּבָה:  מ  בְּעֶצֶם הַיּוֹם הַזֶּה בָּא נֹחַ וְשֵׁם וְחָם וָיֶפֶת בְּנֵי נֹחַ וְאֵשֶׁת נֹחַ וּשְׁלֹשֶׁת נְשֵׁי בָנָיו אִתָּם אֶל הַתֵּבָה:  ת  בִּכְרַן יוֹמָא הָדֵין עָאל נֹחַ וְשֵׁם וְחָם וָיֶפֶת בְּנֵי נֹחַ וְאִתַּת נֹחַ וּתְלָתָא נְשֵׁי בְנוֹהִי עִמְהוֹן לְתֵבוֹתָא:

{יד} הֵמָּה וְכָל הַחַיָּה לְמִינָהּ וְכָל הַבְּהֵמָה לְמִינָהּ וְכָל הָרֶמֶשׂ הָרֹמֵשׂ עַל הָאָרֶץ לְמִינֵהוּ וְכָל הָעוֹף לְמִינֵהוּ כֹּל צִפּוֹר כָּל כָּנָף:  מ  הֵמָּה וְכָל הַחַיָּה לְמִינָהּ וְכָל הַבְּהֵמָה לְמִינָהּ וְכָל הָרֶמֶשׂ הָרֹמֵשׂ עַל הָאָרֶץ לְמִינֵהוּ וְכָל הָעוֹף לְמִינֵהוּ כֹּל צִפּוֹר כָּל כָּנָף:  ת  אִנּוּן וְכָל חַיְתָא לִזְנַהּ וְכָל בְּעִירָא לְזְנַהּ וְכָל רִחֲשָׁא דְּרָחֵשׁ עַל אַרְעָא לִזְנוֹהִי וְכָל עוֹפָא לִזְנוֹהִי כֹּל צִפַּר כָּל דְּפָרָח:

{טו} וַיָּבֹאוּ אֶל נֹחַ אֶל הַתֵּבָה שְׁנַיִם שְׁנַיִם מִכָּל הַבָּשָׂר אֲשֶׁר בּוֹ רוּחַ חַיִּים:  מ  וַיָּבֹאוּ אֶל נֹחַ אֶל הַתֵּבָה שְׁנַיִם שְׁנַיִם מִכָּל הַבָּשָׂר אֲשֶׁר בּוֹ רוּחַ חַיִּים:  ת  וְעַלּוּ עִם נֹחַ לְתֵבוֹתָא תְּרֵין תְּרֵין מִכָּל בִּשְׂרָא דִּי בֵהּ רוּחָא דְּחַיֵּי:

{טז} וְהַבָּאִים זָכָר וּנְקֵבָה מִכָּל בָּשָׂר בָּאוּ כַּאֲשֶׁר צִוָּה אֹתוֹ אֱלֹהִים וַיִּסְגֹּר יְהוָה בַּעֲדוֹ:  מ  וְהַבָּאִים זָכָר וּנְקֵבָה מִכָּל בָּשָׂר בָּאוּ כַּאֲשֶׁר צִוָּה אֹתוֹ אֱלֹהִים וַיִּסְגֹּר יְהוָה בַּעֲדוֹ:  ת  וְעָלַיָּא דְּכַר וְנוּקְבָא מִכָּל בִּשְׂרָא עַלּוּ כְּמָא דִּי פַּקֵּיד יָתֵהּ יְיָ וַאֲגֵן יְיָ (בְּמֵימְרֵהּ) עֲלוֹהִי:

{יז}  שלישי  וַיְהִי הַמַּבּוּל אַרְבָּעִים יוֹם עַל הָאָרֶץ וַיִּרְבּוּ הַמַּיִם וַיִּשְׂאוּ אֶת הַתֵּבָה וַתָּרָם מֵעַל הָאָרֶץ:  מ  וַיְהִי הַמַּבּוּל אַרְבָּעִים יוֹם עַל הָאָרֶץ וַיִּרְבּוּ הַמַּיִם וַיִּשְׂאוּ אֶת הַתֵּבָה וַתָּרָם מֵעַל הָאָרֶץ:  ת  וַהֲוָה טוֹפָנָא אַרְבְּעִין יְמָמִין עַל אַרְעָא וּסְגִיאוּ מַיָּא וּנְטָלוּ יָת תֵּבוֹתָא וְאִתָּרָמַת מֵעַל אַרְעָא:

{יח} וַיִּגְבְּרוּ הַמַּיִם וַיִּרְבּוּ מְאֹד עַל הָאָרֶץ וַתֵּלֶךְ הַתֵּבָה עַל פְּנֵי הַמָּיִם:  מ  וַיִּגְבְּרוּ הַמַּיִם וַיִּרְבּוּ מְאֹד עַל הָאָרֶץ וַתֵּלֶךְ הַתֵּבָה עַל פְּנֵי הַמָּיִם:  ת  וּתְקִיפוּ מַיָּא וּסְגִיאוּ לַחֲדָא עַל אַרְעָא וּמְהַלְּכָא תֵבוֹתָא עַל אַפֵּי מַיָּא:

{יט} וְהַמַּיִם גָּבְרוּ מְאֹד מְאֹד עַל הָאָרֶץ וַיְכֻסּוּ כָּל הֶהָרִים הַגְּבֹהִים אֲשֶׁר תַּחַת כָּל הַשָּׁמָיִם:  מ  וְהַמַּיִם גָּבְרוּ מְאֹד מְאֹד עַל הָאָרֶץ וַיְכֻסּוּ כָּל הֶהָרִים הַגְּבֹהִים אֲשֶׁר תַּחַת כָּל הַשָּׁמָיִם:  ת  וּמַיָּא תְּקִיפוּ לַחֲדָא לַחֲדָא עַל אַרְעָא וְאִתְחֲפִיאוּ כָּל טוּרַיָּא רָמַיָּא דִּי תְחוֹת כָּל שְׁמַיָּא:

{כ} חֲמֵשׁ עֶשְׂרֵה אַמָּה מִלְמַעְלָה גָּבְרוּ הַמָּיִם וַיְכֻסּוּ הֶהָרִים:  מ  חֲמֵשׁ עֶשְׂרֵה אַמָּה מִלְמַעְלָה גָּבְרוּ הַמָּיִם וַיְכֻסּוּ הֶהָרִים:  ת  חֲמֵשׁ עַשְׂרֵי אַמִּין מִלְעֵלָּא תְּקִיפוּ מַיָּא וְאִתְחֲפִיאוּ טוּרַיָּא:

{כא} וַיִּגְוַע כָּל בָּשָׂר הָרֹמֵשׂ עַל הָאָרֶץ בָּעוֹף וּבַבְּהֵמָה וּבַחַיָּה וּבְכָל הַשֶּׁרֶץ הַשֹּׁרֵץ עַל הָאָרֶץ וְכֹל הָאָדָם:  מ  וַיִּגְוַע כָּל בָּשָׂר הָרֹמֵשׂ עַל הָאָרֶץ בָּעוֹף וּבַבְּהֵמָה וּבַחַיָּה וּבְכָל הַשֶּׁרֶץ הַשֹּׁרֵץ עַל הָאָרֶץ וְכֹל הָאָדָם:  ת  וּמִית כָּל בִּשְׂרָא דְּרָחֵשׁ עַל אַרְעָא בְּעוֹפָא וּבִבְעִירָא וּבְחַיְתָא וּבְכָל רִחֲשָׁא דְּרָחֵשׁ עַל אַרְעָא וְכֹל אֱנָשָׁא:

{כב} כֹּל אֲשֶׁר נִשְׁמַת רוּחַ חַיִּים בְּאַפָּיו מִכֹּל אֲשֶׁר בֶּחָרָבָה מֵתוּ:  מ  כֹּל אֲשֶׁר נִשְׁמַת רוּחַ חַיִּים בְּאַפָּיו מִכֹּל אֲשֶׁר בֶּחָרָבָה מֵתוּ:  ת  כֹּל דִּי נִשְׁמְתָא רוּחָא דְּחַיִּין בְּאַפּוֹהִי מִכֹּל דִּי בְּיַבֶּשְׁתָּא מִיתוּ:

{כג} וַיִּמַח אֶת כָּל הַיְקוּם אֲשֶׁר עַל פְּנֵי הָאֲדָמָה מֵאָדָם עַד בְּהֵמָה עַד רֶמֶשׂ וְעַד עוֹף הַשָּׁמַיִם וַיִּמָּחוּ מִן הָאָרֶץ וַיִשָּׁאֶר אַךְ נֹחַ וַאֲשֶׁר אִתּוֹ בַּתֵּבָה:  מ  וַיִּמַח אֶת כָּל הַיְקוּם אֲשֶׁר עַל פְּנֵי הָאֲדָמָה מֵאָדָם עַד בְּהֵמָה עַד רֶמֶשׂ וְעַד עוֹף הַשָּׁמַיִם וַיִּמָּחוּ מִן הָאָרֶץ וַיִשָּׁאֶר אַךְ נֹחַ וַאֲשֶׁר אִתּוֹ בַּתֵּבָה:  ת  וּמְחָא יָת כָּל יְקוּמָא דִּי עַל אַפֵּי אַרְעָא מֵאֱנָשָׁא עַד בְּעִירָא עַד רִחֲשָׁא וְעַד עוֹפָא דִּשְׁמַיָּא וְאִתְמְחִיאוּ מִן אַרְעָא וְאִשְׁתָּאַר בְּרַם נֹחַ וְדִי עִמֵּהּ בְּתֵבוֹתָא:

{כד} וַיִּגְבְּרוּ הַמַּיִם עַל הָאָרֶץ חֲמִשִּׁים וּמְאַת יוֹם:  מ  וַיִּגְבְּרוּ הַמַּיִם עַל הָאָרֶץ חֲמִשִּׁים וּמְאַת יוֹם:  ת  וּתְקִיפוּ מַיָּא עַל אַרְעָא מְאָה וְחַמְשִׁין יוֹמִין:




בראשית פרק-ח

{א} וַיִּזְכֹּר אֱלֹהִים אֶת נֹחַ וְאֵת כָּל הַחַיָּה וְאֶת כָּל הַבְּהֵמָה אֲשֶׁר אִתּוֹ בַּתֵּבָה וַיַּעֲבֵר אֱלֹהִים רוּחַ עַל הָאָרֶץ וַיָּשֹׁכּוּ הַמָּיִם:  מ  וַיִּזְכֹּר אֱלֹהִים אֶת נֹחַ וְאֵת כָּל הַחַיָּה וְאֶת כָּל הַבְּהֵמָה אֲשֶׁר אִתּוֹ בַּתֵּבָה וַיַּעֲבֵר אֱלֹהִים רוּחַ עַל הָאָרֶץ וַיָּשֹׁכּוּ הַמָּיִם:  ת  וּדְכִיר יְיָ יָת נֹחַ וְיָת כָּל חַיְתָא וְיָת כָּל בְּעִירָא דִּי עִמֵּהּ בְּתֵבוֹתָא וְאַעְבַּר יְיָ רוּחָא עַל אַרְעָא וְנָחוּ מַיָּא:

{ב} וַיִּסָּכְרוּ מַעְיְנֹת תְּהוֹם וַאֲרֻבֹּת הַשָּׁמָיִם וַיִּכָּלֵא הַגֶּשֶׁם מִן הַשָּׁמָיִם:  מ  וַיִּסָּכְרוּ מַעְיְנֹת תְּהוֹם וַאֲרֻבֹּת הַשָּׁמָיִם וַיִּכָּלֵא הַגֶּשֶׁם מִן הַשָּׁמָיִם:  ת  וְאִסְתְּכָרוּ מַבּוּעֵי תְהוֹמָא וְכַוֵּי שְׁמַיָּא וְאִתְכְּלִי מִטְרָא מִן שְׁמַיָּא:

{ג} וַיָּשֻׁבוּ הַמַּיִם מֵעַל הָאָרֶץ הָלוֹךְ וָשׁוֹב וַיַּחְסְרוּ הַמַּיִם מִקְצֵה חֲמִשִּׁים וּמְאַת יוֹם:  מ  וַיָּשֻׁבוּ הַמַּיִם מֵעַל הָאָרֶץ הָלוֹךְ וָשׁוֹב וַיַּחְסְרוּ הַמַּיִם מִקְצֵה חֲמִשִּׁים וּמְאַת יוֹם:  ת  וְתָבוּ מַיָּא מֵעַל אַרְעָא אָזְלִין וְתָיְבִין וַחֲסָרוּ מַיָּא מִסּוֹף מְאָה וְחַמְשִׁין יוֹמִין:

{ד} וַתָּנַח הַתֵּבָה בַּחֹדֶשׁ הַשְּׁבִיעִי בְּשִׁבְעָה עָשָׂר יוֹם לַחֹדֶשׁ עַל הָרֵי אֲרָרָט:  מ  וַתָּנַח הַתֵּבָה בַּחֹדֶשׁ הַשְּׁבִיעִי בְּשִׁבְעָה עָשָׂר יוֹם לַחֹדֶשׁ עַל הָרֵי אֲרָרָט:  ת  וְנָחַת תֵּבוֹתָא בְּיַרְחָא שְׁבִיעָאָה בְּשִׁבְעַת עֲשַׂר יוֹמָא לְיַרְחָא עַל טוּרֵי קַרְדּוּ:

{ה} וְהַמַּיִם הָיוּ הָלוֹךְ וְחָסוֹר עַד הַחֹדֶשׁ הָעֲשִׂירִי בָּעֲשִׂירִי בְּאֶחָד לַחֹדֶשׁ נִרְאוּ רָאשֵׁי הֶהָרִים:  מ  וְהַמַּיִם הָיוּ הָלוֹךְ וְחָסוֹר עַד הַחֹדֶשׁ הָעֲשִׂירִי בָּעֲשִׂירִי בְּאֶחָד לַחֹדֶשׁ נִרְאוּ רָאשֵׁי הֶהָרִים:  ת  וּמַיָּא הֲווֹ אָזְלִין וְחָסְרִין עַד יַרְחָא עֲשִׂירָאָה בַּעֲשִׂירָאָה בְּחַד לְיַרְחָא אִתְחֲזִיאוּ רֵישֵׁי טוּרַיָּא:

{ו} וַיְהִי מִקֵּץ אַרְבָּעִים יוֹם וַיִּפְתַּח נֹחַ אֶת חַלּוֹן הַתֵּבָה אֲשֶׁר עָשָׂה:  מ  וַיְהִי מִקֵּץ אַרְבָּעִים יוֹם וַיִּפְתַּח נֹחַ אֶת חַלּוֹן הַתֵּבָה אֲשֶׁר עָשָׂה:  ת  וַהֲוָה מִסּוֹף אַרְבְּעִין יוֹמִין וּפְתַח נֹחַ יָת כַּוַּת תֵּבוֹתָא דִּי עֲבָד:

{ז} וַיְשַׁלַּח אֶת הָעֹרֵב וַיֵּצֵא יָצוֹא וָשׁוֹב עַד יְבֹשֶׁת הַמַּיִם מֵעַל הָאָרֶץ:  מ  וַיְשַׁלַּח אֶת הָעֹרֵב וַיֵּצֵא יָצוֹא וָשׁוֹב עַד יְבֹשֶׁת הַמַּיִם מֵעַל הָאָרֶץ:  ת  וְשַׁלַּח יָת עוֹרְבָא וּנְפַק מִפַּק וְתַיִּב עַד דְּיַבִּישׁוּ מַיָּא מֵעַל אַרְעָא:

{ח} וַיְשַׁלַּח אֶת הַיּוֹנָה מֵאִתּוֹ לִרְאוֹת הֲקַלּוּ הַמַּיִם מֵעַל פְּנֵי הָאֲדָמָה:  מ  וַיְשַׁלַּח אֶת הַיּוֹנָה מֵאִתּוֹ לִרְאוֹת הֲקַלּוּ הַמַּיִם מֵעַל פְּנֵי הָאֲדָמָה:  ת  וְשַׁלַּח יָת יוֹנָה מִלְּוָתֵהּ לְמֵחֲזֵי הֲקַלִּיאוּ מַיָּא מֵעַל אַפֵּי אַרְעָא:

{ט} וְלֹא מָצְאָה הַיּוֹנָה מָנוֹחַ לְכַף רַגְלָהּ וַתָּשָׁב אֵלָיו אֶל הַתֵּבָה כִּי מַיִם עַל פְּנֵי כָל הָאָרֶץ וַיִּשְׁלַח יָדוֹ וַיִּקָּחֶהָ וַיָּבֵא אֹתָהּ אֵלָיו אֶל הַתֵּבָה:  מ  וְלֹא מָצְאָה הַיּוֹנָה מָנוֹחַ לְכַף רַגְלָהּ וַתָּשָׁב אֵלָיו אֶל הַתֵּבָה כִּי מַיִם עַל פְּנֵי כָל הָאָרֶץ וַיִּשְׁלַח יָדוֹ וַיִּקָּחֶהָ וַיָּבֵא אֹתָהּ אֵלָיו אֶל הַתֵּבָה:  ת  וְלָא אַשְׁכַּחַת יוֹנָה מְנַח לְפַרְסַת רַגְלַהּ וְתָבַת לְוָתֵהּ לְתֵבוֹתָא אֲרֵי מַיָּא עַל אַפֵּי כָל אַרְעָא וְאוֹשִׁיט יְדֵהּ וְנָסְבַהּ וְאָעֵיל יָתַהּ לְוָתֵהּ לְתֵבוֹתָא:

{י} וַיָּחֶל עוֹד שִׁבְעַת יָמִים אֲחֵרִים וַיֹּסֶף שַׁלַּח אֶת הַיּוֹנָה מִן הַתֵּבָה:  מ  וַיָּחֶל עוֹד שִׁבְעַת יָמִים אֲחֵרִים וַיֹּסֶף שַׁלַּח אֶת הַיּוֹנָה מִן הַתֵּבָה:  ת  וְאוֹרִיךְ עוֹד שַׁבְעָא יוֹמִין אָחֳרָנִין וְאוֹסֵיף שַׁלַּח יָת יוֹנָה מִן תֵּבוֹתָא:

{יא} וַתָּבֹא אֵלָיו הַיּוֹנָה לְעֵת עֶרֶב וְהִנֵּה עֲלֵה זַיִת טָרָף בְּפִיהָ וַיֵּדַע נֹחַ כִּי קַלּוּ הַמַּיִם מֵעַל הָאָרֶץ:  מ  וַתָּבֹא אֵלָיו הַיּוֹנָה לְעֵת עֶרֶב וְהִנֵּה עֲלֵה זַיִת טָרָף בְּפִיהָ וַיֵּדַע נֹחַ כִּי קַלּוּ הַמַּיִם מֵעַל הָאָרֶץ:  ת  וַאֲתַת לְוָתֵהּ יוֹנָה לְעִדַּן רַמְשָׁא וְהָא טְּרַף זֵיתָא תְּבִיר נָחִית בְּפוּמַהּ וִידַע נֹחַ אֲרֵי קַלִּיאוּ מַיָּא מֵעַל אַרְעָא:

{יב} וַיִּיָּחֶל עוֹד שִׁבְעַת יָמִים אֲחֵרִים וַיְשַׁלַּח אֶת הַיּוֹנָה וְלֹא יָסְפָה שׁוּב אֵלָיו עוֹד:  מ  וַיִּיָּחֶל עוֹד שִׁבְעַת יָמִים אֲחֵרִים וַיְשַׁלַּח אֶת הַיּוֹנָה וְלֹא יָסְפָה שׁוּב אֵלָיו עוֹד:  ת  וְאוֹרִיךְ עוֹד שַׁבְעָא יוֹמִין אָחֳרָנִין וְשַׁלַּח יָת יוֹנָה וְלָא אוֹסֵיפַת לְמִתּוּב לְוָתֵהּ עוֹד:

{יג} וַיְהִי בְּאַחַת וְשֵׁשׁ מֵאוֹת שָׁנָה בָּרִאשׁוֹן בְּאֶחָד לַחֹדֶשׁ חָרְבוּ הַמַּיִם מֵעַל הָאָרֶץ וַיָּסַר נֹחַ אֶת מִכְסֵה הַתֵּבָה וַיַּרְא וְהִנֵּה חָרְבוּ פְּנֵי הָאֲדָמָה:  מ  וַיְהִי בְּאַחַת וְשֵׁשׁ מֵאוֹת שָׁנָה בָּרִאשׁוֹן בְּאֶחָד לַחֹדֶשׁ חָרְבוּ הַמַּיִם מֵעַל הָאָרֶץ וַיָּסַר נֹחַ אֶת מִכְסֵה הַתֵּבָה וַיַּרְא וְהִנֵּה חָרְבוּ פְּנֵי הָאֲדָמָה:  ת  וַהֲוָה בְּשִׁתְּ מְאָה וְחַד שְׁנִין בְּקַדְמָאָה בְּחַד לְיַרְחָא נְגוּבוּ מַיָּא מֵעַל אַרְעָא וְאַעְדִּי נֹחַ יָת חוֹפָאָה דְּתֵבוֹתָא וַחֲזָא וְהָא נְגוּבוּ אַפֵּי אַרְעָא:

{יד} וּבַחֹדֶשׁ הַשֵּׁנִי בְּשִׁבְעָה וְעֶשְׂרִים יוֹם לַחֹדֶשׁ יָבְשָׁה הָאָרֶץ:  מ  וּבַחֹדֶשׁ הַשֵּׁנִי בְּשִׁבְעָה וְעֶשְׂרִים יוֹם לַחֹדֶשׁ יָבְשָׁה הָאָרֶץ:  ת  וּבְיַרְחָא תִּנְיָנָא בְּעַשְׂרִין וְשַׁבְעָא יוֹמָא לְיַרְחָא יַבִּישַׁת אַרְעָא: (ס)

{טו}  רביעי  וַיְדַבֵּר אֱלֹהִים אֶל נֹחַ לֵאמֹר:  מ  וַיְדַבֵּר אֱלֹהִים אֶל נֹחַ לֵאמֹר:  ת  וּמַלִּיל יְיָ עִם נֹחַ לְמֵימָר:

{טז} צֵא מִן הַתֵּבָה אַתָּה וְאִשְׁתְּךָ וּבָנֶיךָ וּנְשֵׁי בָנֶיךָ אִתָּךְ:  מ  צֵא מִן הַתֵּבָה אַתָּה וְאִשְׁתְּךָ וּבָנֶיךָ וּנְשֵׁי בָנֶיךָ אִתָּךְ:  ת  פּוּק מִן תֵּבוֹתָא אַתְּ וְאִתְּתָךְ וּבְנָךְ וּנְשֵׁי בְנָיךְ עִמָךְ:

{יז} כָּל הַחַיָּה אֲשֶׁר אִתְּךָ מִכָּל בָּשָׂר בָּעוֹף וּבַבְּהֵמָה וּבְכָל הָרֶמֶשׂ הָרֹמֵשׂ עַל הָאָרֶץ (הוצא) הַיְצֵא אִתָּךְ וְשָׁרְצוּ בָאָרֶץ וּפָרוּ וְרָבוּ עַל הָאָרֶץ:  מ  כָּל הַחַיָּה אֲשֶׁר אִתְּךָ מִכָּל בָּשָׂר בָּעוֹף וּבַבְּהֵמָה וּבְכָל הָרֶמֶשׂ הָרֹמֵשׂ עַל הָאָרֶץ (הוצא) הַיְצֵא אִתָּךְ וְשָׁרְצוּ בָאָרֶץ וּפָרוּ וְרָבוּ עַל הָאָרֶץ:  ת  כָּל חַיְתָא דְּעִמָּךְ מִכָּל בִּשְׂרָא בְּעוֹפָא וּבִבְעִירָא וּבְכָל רִחֲשָׁא דְּרָחֵשׁ עַל אַרְעָא אַפֵּיק עִמָּךְ וְיִתְיַלְּדוּן בְּאַרְעָא וְיִפְּשׁוּן וְיִסְגּוּן עַל אַרְעָא:

{יח} וַיֵּצֵא נֹחַ וּבָנָיו וְאִשְׁתּוֹ וּנְשֵׁי בָנָיו אִתּוֹ:  מ  וַיֵּצֵא נֹחַ וּבָנָיו וְאִשְׁתּוֹ וּנְשֵׁי בָנָיו אִתּוֹ:  ת  וּנְפַק נֹחַ וּבְנוֹהִי וְאִתְּתֵהּ וּנְשֵׁי בְנוֹהִי עִמֵּהּ:

{יט} כָּל הַחַיָּה כָּל הָרֶמֶשׂ וְכָל הָעוֹף כֹּל רוֹמֵשׂ עַל הָאָרֶץ לְמִשְׁפְּחֹתֵיהֶם יָצְאוּ מִן הַתֵּבָה:  מ  כָּל הַחַיָּה כָּל הָרֶמֶשׂ וְכָל הָעוֹף כֹּל רוֹמֵשׂ עַל הָאָרֶץ לְמִשְׁפְּחֹתֵיהֶם יָצְאוּ מִן הַתֵּבָה:  ת  כָּל חַיְתָא כָּל רִחֲשָׁא וְכָל עוֹפָא כֹּל דְּרָחֵשׁ עַל אַרְעָא לְזַרְעֲיָתְהוֹן נְפָקוּ מִן תֵּבוֹתָא:

{כ} וַיִּבֶן נֹחַ מִזְבֵּחַ לַיהוָה וַיִּקַּח מִכֹּל הַבְּהֵמָה הַטְּהוֹרָה וּמִכֹּל הָעוֹף הַטָּהֹר וַיַּעַל עֹלֹת בַּמִּזְבֵּחַ:  מ  וַיִּבֶן נֹחַ מִזְבֵּחַ לַיהוָה וַיִּקַּח מִכֹּל הַבְּהֵמָה הַטְּהוֹרָה וּמִכֹּל הָעוֹף הַטָּהֹר וַיַּעַל עֹלֹת בַּמִּזְבֵּחַ:  ת  וּבְנָא נֹחַ מַדְבְּחָא קֳדָם יְיָ וּנְסִיב מִכֹּל בְּעִירָא דַּכְיָא וּמִכֹּל עוֹפָא דְכֵי וְאַסֵּיק עֲלָוָן בְּמִדְבְּחָא:

{כא} וַיָּרַח יְהוָה אֶת רֵיחַ הַנִּיחֹחַ וַיֹּאמֶר יְהוָה אֶל לִבּוֹ לֹא אֹסִף לְקַלֵּל עוֹד אֶת הָאֲדָמָה בַּעֲבוּר הָאָדָם כִּי יֵצֶר לֵב הָאָדָם רַע מִנְּעֻרָיו וְלֹא אֹסִף עוֹד לְהַכּוֹת אֶת כָּל חַי כַּאֲשֶׁר עָשִׂיתִי:  מ  וַיָּרַח יְהוָה אֶת רֵיחַ הַנִּיחֹחַ וַיֹּאמֶר יְהוָה אֶל לִבּוֹ לֹא אֹסִף לְקַלֵּל עוֹד אֶת הָאֲדָמָה בַּעֲבוּר הָאָדָם כִּי יֵצֶר לֵב הָאָדָם רַע מִנְּעֻרָיו וְלֹא אֹסִף עוֹד לְהַכּוֹת אֶת כָּל חַי כַּאֲשֶׁר עָשִׂיתִי:  ת  וְקַבִּיל יְיָ בְּרַעֲוָא יָת קוּרְבָּנֵהּ וַאֲמַר יְיָ בְּמֵימְרֵהּ לָא אוֹסֵף לְמֵילַט עוֹד יָת אַרְעָא בְּדִיל חוֹבֵי אֱנָשָׁא אֲרֵי יִצְרָא לִבָּא דֶּאֱנָשָׁא בִּישׁ מִזְעִירֵהּ וְלָא אוֹסֵף עוֹד לְמִמְחֵי יָת כָּל דְּחַי כְּמָא דִעַבָדִית:

{כב} עֹד כָּל יְמֵי הָאָרֶץ זֶרַע וְקָצִיר וְקֹר וָחֹם וְקַיִץ וָחֹרֶף וְיוֹם וָלַיְלָה לֹא יִשְׁבֹּתוּ:  מ  עֹד כָּל יְמֵי הָאָרֶץ זֶרַע וְקָצִיר וְקֹר וָחֹם וְקַיִץ וָחֹרֶף וְיוֹם וָלַיְלָה לֹא יִשְׁבֹּתוּ:  ת  עוֹד כָּל יוֹמֵי אַרְעָא זְרוֹעָא וַחֲצָדָא וְקוֹרָא וְחוֹמָא וְקַיְטָא וְסִתְוָא וִימָם וְלֵילְיָא לָא יִבְטְּלוּן:




בראשית פרק-ט

{א} וַיְבָרֶךְ אֱלֹהִים אֶת נֹחַ וְאֶת בָּנָיו וַיֹּאמֶר לָהֶם פְּרוּ וּרְבוּ וּמִלְאוּ אֶת הָאָרֶץ:  מ  וַיְבָרֶךְ אֱלֹהִים אֶת נֹחַ וְאֶת בָּנָיו וַיֹּאמֶר לָהֶם פְּרוּ וּרְבוּ וּמִלְאוּ אֶת הָאָרֶץ:  ת  וּבָרִיךְ יְיָ יָת נֹחַ וְיָת בְּנוֹהִי וַאֲמַר לְהוֹן פּוּשׁוּ וּסְגוּ וּמְלוּ יָת אַרְעָא:

{ב} וּמוֹרַאֲכֶם וְחִתְּכֶם יִהְיֶה עַל כָּל חַיַּת הָאָרֶץ וְעַל כָּל עוֹף הַשָּׁמָיִם בְּכֹל אֲשֶׁר תִּרְמֹשׂ הָאֲדָמָה וּבְכָל דְּגֵי הַיָּם בְּיֶדְכֶם נִתָּנוּ:  מ  וּמוֹרַאֲכֶם וְחִתְּכֶם יִהְיֶה עַל כָּל חַיַּת הָאָרֶץ וְעַל כָּל עוֹף הַשָּׁמָיִם בְּכֹל אֲשֶׁר תִּרְמֹשׂ הָאֲדָמָה וּבְכָל דְּגֵי הַיָּם בְּיֶדְכֶם נִתָּנוּ:  ת  וְדַחֲלַתְכוֹן וְאֵימַתְכוֹן תְּהֵי עַל כָּל חֵוַת אַרְעָא וְעַל כָּל עוֹפָא דִּשְׁמַיָּא בְּכֹל דִּי תַרְחֵישׁ אַרְעָא וּבְכָל נוּנֵי יַמָּא בְּיֶדְכוֹן יְהוֹן מְסִירִין:

{ג} כָּל רֶמֶשׂ אֲשֶׁר הוּא חַי לָכֶם יִהְיֶה לְאָכְלָה כְּיֶרֶק עֵשֶׂב נָתַתִּי לָכֶם אֶת כֹּל:  מ  כָּל רֶמֶשׂ אֲשֶׁר הוּא חַי לָכֶם יִהְיֶה לְאָכְלָה כְּיֶרֶק עֵשֶׂב נָתַתִּי לָכֶם אֶת כֹּל:  ת  כָּל רִחֲשָׁא דְּהוּא חַי לְכוֹן יְהֵי לְמֵיכָל כִּירוֹק עִשְׂבָּא יְהָבִית לְכוֹן יָת כֹּלָּא:

{ד} אַךְ בָּשָׂר בְּנַפְשׁוֹ דָמוֹ לֹא תֹאכֵלוּ:  מ  אַךְ בָּשָׂר בְּנַפְשׁוֹ דָמוֹ לֹא תֹאכֵלוּ:  ת  בְּרַם בִּשְׂרָא בְּנַפְשֵׁהּ דְּמֵהּ לָא תֵכְלוּן:

{ה} וְאַךְ אֶת דִּמְכֶם לְנַפְשֹׁתֵיכֶם אֶדְרֹשׁ מִיַּד כָּל חַיָּה אֶדְרְשֶׁנּוּ וּמִיַּד הָאָדָם מִיַּד אִישׁ אָחִיו אֶדְרֹשׁ אֶת נֶפֶשׁ הָאָדָם:  מ  וְאַךְ אֶת דִּמְכֶם לְנַפְשֹׁתֵיכֶם אֶדְרֹשׁ מִיַּד כָּל חַיָּה אֶדְרְשֶׁנּוּ וּמִיַּד הָאָדָם מִיַּד אִישׁ אָחִיו אֶדְרֹשׁ אֶת נֶפֶשׁ הָאָדָם:  ת  וּבְרַם יָת דִּמְכוֹן לְנַפְשָׁתֵיכוֹן אֶתְבַּע מִיַּד כָּל חַיְתָא אֶתְבְּעִנֵהּ וּמִיַד אֱנָשָׁא מִיַד גְבַר דְיֵשׁוֹד יָת דְמָא דַאֲחוֹהִי אֶתְבַּע יָת נַפְשָׁא דֶאֱנָשָׁא:

{ו} שֹׁפֵךְ דַּם הָאָדָם בָּאָדָם דָּמוֹ יִשָּׁפֵךְ כִּי בְּצֶלֶם אֱלֹהִים עָשָׂה אֶת הָאָדָם:  מ  שֹׁפֵךְ דַּם הָאָדָם בָּאָדָם דָּמוֹ יִשָּׁפֵךְ כִּי בְּצֶלֶם אֱלֹהִים עָשָׂה אֶת הָאָדָם:  ת  דְיֵשׁוֹד דְמָא דֶאֱנָשָׁא בְּסַעֲדִין עַל מֵימַר דַיָנַיָא דְמֵהּ יִתְּשָׁד אֲרֵי בְּצַלְמָא דַיְיָ עֲבַד יָת אֱנָשָׁא:

{ז} וְאַתֶּם פְּרוּ וּרְבוּ שִׁרְצוּ בָאָרֶץ וּרְבוּ בָהּ:  מ  וְאַתֶּם פְּרוּ וּרְבוּ שִׁרְצוּ בָאָרֶץ וּרְבוּ בָהּ:  ת  וְאַתּוּן פּוּשׁוּ וּסְגוּ אִתְיְלִידוּ בְאַרְעָא וּסְגוּ בַהּ: (ס)

{ח}  חמישי  וַיֹּאמֶר אֱלֹהִים אֶל נֹחַ וְאֶל בָּנָיו אִתּוֹ לֵאמֹר:  מ  וַיֹּאמֶר אֱלֹהִים אֶל נֹחַ וְאֶל בָּנָיו אִתּוֹ לֵאמֹר:  ת  וַאֲמַר יְיָ לְנֹחַ וְלִבְנוֹהִי עִמֵּהּ לְמֵימָר:

{ט} וַאֲנִי הִנְנִי מֵקִים אֶת בְּרִיתִי אִתְּכֶם וְאֶת זַרְעֲכֶם אַחֲרֵיכֶם:  מ  וַאֲנִי הִנְנִי מֵקִים אֶת בְּרִיתִי אִתְּכֶם וְאֶת זַרְעֲכֶם אַחֲרֵיכֶם:  ת  וַאֲנָא הָא אֲנָא מֵקִים יָת קְיָמִי עִמְּכוֹן וְעִם בְּנֵיכוֹן בַּתְרֵיכוֹן:

{י} וְאֵת כָּל נֶפֶשׁ הַחַיָּה אֲשֶׁר אִתְּכֶם בָּעוֹף בַּבְּהֵמָה וּבְכָל חַיַּת הָאָרֶץ אִתְּכֶם מִכֹּל יֹצְאֵי הַתֵּבָה לְכֹל חַיַּת הָאָרֶץ:  מ  וְאֵת כָּל נֶפֶשׁ הַחַיָּה אֲשֶׁר אִתְּכֶם בָּעוֹף בַּבְּהֵמָה וּבְכָל חַיַּת הָאָרֶץ אִתְּכֶם מִכֹּל יֹצְאֵי הַתֵּבָה לְכֹל חַיַּת הָאָרֶץ:  ת  וְעִם כָּל נַפְשָׁא חַיְתָא דְּעִמְּכוֹן בְּעוֹפָא בִבְעִירָא וּבְכָל חַיַת אַרְעָא דְּעִמְּכוֹן מִכֹּל נַפְקֵי תֵבוֹתָא לְכֹל חַיַת אַרְעָא:

{יא} וַהֲקִמֹתִי אֶת בְּרִיתִי אִתְּכֶם וְלֹא יִכָּרֵת כָּל בָּשָׂר עוֹד מִמֵּי הַמַּבּוּל וְלֹא יִהְיֶה עוֹד מַבּוּל לְשַׁחֵת הָאָרֶץ:  מ  וַהֲקִמֹתִי אֶת בְּרִיתִי אִתְּכֶם וְלֹא יִכָּרֵת כָּל בָּשָׂר עוֹד מִמֵּי הַמַּבּוּל וְלֹא יִהְיֶה עוֹד מַבּוּל לְשַׁחֵת הָאָרֶץ:  ת  וְאָקֵם יָת קְיָמִי עִמְּכוֹן וְלָא יִשְׁתֵּיצֵי כָל בִּשְׂרָא עוֹד מִמֵּי טוֹפָנָא וְלָא יְהֵי עוֹד טוֹפָנָא לְחַבָּלָא אַרְעָא:

{יב} וַיֹּאמֶר אֱלֹהִים זֹאת אוֹת הַבְּרִית אֲשֶׁר אֲנִי נֹתֵן בֵּינִי וּבֵינֵיכֶם וּבֵין כָּל נֶפֶשׁ חַיָּה אֲשֶׁר אִתְּכֶם לְדֹרֹת עוֹלָם:  מ  וַיֹּאמֶר אֱלֹהִים זֹאת אוֹת הַבְּרִית אֲשֶׁר אֲנִי נֹתֵן בֵּינִי וּבֵינֵיכֶם וּבֵין כָּל נֶפֶשׁ חַיָּה אֲשֶׁר אִתְּכֶם לְדֹרֹת עוֹלָם:  ת  וַאֲמַר יְיָ דָּא אָת קְיָם דִּי אֲנָא יָהֵב בֵּין מֵימְרִי וּבֵינֵכוֹן וּבֵין כָּל נַפְשָׁא חַיְתָא דִּי עִמְּכוֹן לְדָרֵי עַלְמָא:

{יג} אֶת קַשְׁתִּי נָתַתִּי בֶּעָנָן וְהָיְתָה לְאוֹת בְּרִית בֵּינִי וּבֵין הָאָרֶץ:  מ  אֶת קַשְׁתִּי נָתַתִּי בֶּעָנָן וְהָיְתָה לְאוֹת בְּרִית בֵּינִי וּבֵין הָאָרֶץ:  ת  יָת קַשְׁתִּי יְהָבִית בַּעֲנָנָא וּתְהִי לְאָת קְיָם בֵּין מֵימְרִי וּבֵין אַרְעָא:

{יד} וְהָיָה בְּעַנְנִי עָנָן עַל הָאָרֶץ וְנִרְאֲתָה הַקֶּשֶׁת בֶּעָנָן:  מ  וְהָיָה בְּעַנְנִי עָנָן עַל הָאָרֶץ וְנִרְאֲתָה הַקֶּשֶׁת בֶּעָנָן:  ת  וִיהִי בַּעֲנָנוּתִי עֲנָנָא עַל אַרְעָא וְתִתְחֲזֵי קַשְׁתָּא בַּעֲנָנָא:

{טו} וְזָכַרְתִּי אֶת בְּרִיתִי אֲשֶׁר בֵּינִי וּבֵינֵיכֶם וּבֵין כָּל נֶפֶשׁ חַיָּה בְּכָל בָּשָׂר וְלֹא יִהְיֶה עוֹד הַמַּיִם לְמַבּוּל לְשַׁחֵת כָּל בָּשָׂר:  מ  וְזָכַרְתִּי אֶת בְּרִיתִי אֲשֶׁר בֵּינִי וּבֵינֵיכֶם וּבֵין כָּל נֶפֶשׁ חַיָּה בְּכָל בָּשָׂר וְלֹא יִהְיֶה עוֹד הַמַּיִם לְמַבּוּל לְשַׁחֵת כָּל בָּשָׂר:  ת  וְדָכִירְנָא יָת קְיָמִי דִּי בֵין מֵימְרִי וּבֵינֵיכוֹן וּבֵין כָּל נַפְשָׁא חַיְתָא בְּכָל בִּשְׂרָא וְלָא יְהֵי עוֹד מַיָּא לְטוֹפָנָא לְחַבָּלָא כָל בִּשְׂרָא:

{טז} וְהָיְתָה הַקֶּשֶׁת בֶּעָנָן וּרְאִיתִיהָ לִזְכֹּר בְּרִית עוֹלָם בֵּין אֱלֹהִים וּבֵין כָּל נֶפֶשׁ חַיָּה בְּכָל בָּשָׂר אֲשֶׁר עַל הָאָרֶץ:  מ  וְהָיְתָה הַקֶּשֶׁת בֶּעָנָן וּרְאִיתִיהָ לִזְכֹּר בְּרִית עוֹלָם בֵּין אֱלֹהִים וּבֵין כָּל נֶפֶשׁ חַיָּה בְּכָל בָּשָׂר אֲשֶׁר עַל הָאָרֶץ:  ת  וּתְהֵי קַשְׁתָּא בַּעֲנָנָא וְאֵחֲזִנַּהּ לְמִדְכַּר קְיַם עָלָם בֵּין מֵימְרָא דַּיְיָ וּבֵין כָּל נַפְשָׁא חַיְתָא בְּכָל בִּשְׂרָא דִּי עַל אַרְעָא:

{יז} וַיֹּאמֶר אֱלֹהִים אֶל נֹחַ זֹאת אוֹת הַבְּרִית אֲשֶׁר הֲקִמֹתִי בֵּינִי וּבֵין כָּל בָּשָׂר אֲשֶׁר עַל הָאָרֶץ:  מ  וַיֹּאמֶר אֱלֹהִים אֶל נֹחַ זֹאת אוֹת הַבְּרִית אֲשֶׁר הֲקִמֹתִי בֵּינִי וּבֵין כָּל בָּשָׂר אֲשֶׁר עַל הָאָרֶץ:  ת  וַאֲמַר יְיָ לְנֹחַ דָּא אָת קְיָם דִּי אֲקֵמִית בֵּין מֵימְרִי וּבֵין כָּל בִּשְׂרָא דִּי עַל אַרְעָא: (פ)

{יח}  שישי  וַיִּהְיוּ בְנֵי נֹחַ הַיֹּצְאִים מִן הַתֵּבָה שֵׁם וְחָם וָיָפֶת וְחָם הוּא אֲבִי כְנָעַן:  מ  וַיִּהְיוּ בְנֵי נֹחַ הַיֹּצְאִים מִן הַתֵּבָה שֵׁם וְחָם וָיָפֶת וְחָם הוּא אֲבִי כְנָעַן:  ת  וַהֲווֹ בְנֵי נֹחַ דִּי נְפָקוּ מִן תֵּבוֹתָא שֵׁם וְחָם וָיָפֶת וְחָם הוּא אֲבוּהִי דִכְנָעַן:

{יט} שְׁלֹשָׁה אֵלֶּה בְּנֵי נֹחַ וּמֵאֵלֶּה נָפְצָה כָל הָאָרֶץ:  מ  שְׁלֹשָׁה אֵלֶּה בְּנֵי נֹחַ וּמֵאֵלֶּה נָפְצָה כָל הָאָרֶץ:  ת  תְּלָתָא אִלֵּין בְּנֵי נֹחַ וּמֵאִלֵּין אִתְבַּדָּרוּ כָּל אַרְעָא:

{כ} וַיָּחֶל נֹחַ אִישׁ הָאֲדָמָה וַיִּטַּע כָּרֶם:  מ  וַיָּחֶל נֹחַ אִישׁ הָאֲדָמָה וַיִּטַּע כָּרֶם:  ת  וְשָׁרִי נֹחַ גְּבַר פָּלַח בְּאַרְעָא וּנְצִיב כַּרְמָא:

{כא} וַיֵּשְׁתְּ מִן הַיַּיִן וַיִּשְׁכָּר וַיִּתְגַּל בְּתוֹךְ אָהֳלֹה:  מ  וַיֵּשְׁתְּ מִן הַיַּיִן וַיִּשְׁכָּר וַיִּתְגַּל בְּתוֹךְ אָהֳלֹה:  ת  וּשְׁתִי מִן חַמְרָא וּרְוִי וְאִתְגְּלִי בְּגוֹ מַשְׁכְּנֵהּ:

{כב} וַיַּרְא חָם אֲבִי כְנַעַן אֵת עֶרְוַת אָבִיו וַיַּגֵּד לִשְׁנֵי אֶחָיו בַּחוּץ:  מ  וַיַּרְא חָם אֲבִי כְנַעַן אֵת עֶרְוַת אָבִיו וַיַּגֵּד לִשְׁנֵי אֶחָיו בַּחוּץ:  ת  וַחֲזָא חָם אֲבוּהִי דִּכְנַעַן יָת עֶרְיְתָא דַּאֲבוּהִי וְחַוֵּי לִתְרֵין אֲחוֹהִי בְּשׁוּקָא:

{כג} וַיִּקַּח שֵׁם וָיֶפֶת אֶת הַשִּׂמְלָה וַיָּשִׂימוּ עַל שְׁכֶם שְׁנֵיהֶם וַיֵּלְכוּ אֲחֹרַנִּית וַיְכַסּוּ אֵת עֶרְוַת אֲבִיהֶם וּפְנֵיהֶם אֲחֹרַנִּית וְעֶרְוַת אֲבִיהֶם לֹא רָאוּ:  מ  וַיִּקַּח שֵׁם וָיֶפֶת אֶת הַשִּׂמְלָה וַיָּשִׂימוּ עַל שְׁכֶם שְׁנֵיהֶם וַיֵּלְכוּ אֲחֹרַנִּית וַיְכַסּוּ אֵת עֶרְוַת אֲבִיהֶם וּפְנֵיהֶם אֲחֹרַנִּית וְעֶרְוַת אֲבִיהֶם לֹא רָאוּ:  ת  וּנְסֵיב שֵׁם וָיֶפֶת יָת כְּסוּתָא וְשַׁוִּיוּ עַל כְּתַף תַּרְוֵיהוֹן וַאֲזָלוּ מְחַזְרִין וַחֲפִיאוּ יָת עֶרְיְתָא דַּאֲבוּהוֹן וְאַפֵּיהוֹן מְחַזְרִין וְעֶרְיְתָא דַּאֲבוּהוֹן לָא חֲזוֹ:

{כד} וַיִּיקֶץ נֹחַ מִיֵּינוֹ וַיֵּדַע אֵת אֲשֶׁר עָשָׂה לוֹ בְּנוֹ הַקָּטָן:  מ  וַיִּיקֶץ נֹחַ מִיֵּינוֹ וַיֵּדַע אֵת אֲשֶׁר עָשָׂה לוֹ בְּנוֹ הַקָּטָן:  ת  וְאִתְּעַר נֹחַ מֵחַמְרֵהּ וִידַע יָת דִּי עֲבַד לֵהּ בְּרֵהּ זְעֵירָא:

{כה} וַיֹּאמֶר אָרוּר כְּנָעַן עֶבֶד עֲבָדִים יִהְיֶה לְאֶחָיו:  מ  וַיֹּאמֶר אָרוּר כְּנָעַן עֶבֶד עֲבָדִים יִהְיֶה לְאֶחָיו:  ת  וַאֲמַר לִיט כְּנַעַן עֲבַד פְּלַח עַבְדִין יְהֵי לַאֲחוֹהִי:

{כו} וַיֹּאמֶר בָּרוּךְ יְהוָֹה אֱלֹהֵי שֵׁם וִיהִי כְנַעַן עֶבֶד לָמוֹ:  מ  וַיֹּאמֶר בָּרוּךְ יְהוָֹה אֱלֹהֵי שֵׁם וִיהִי כְנַעַן עֶבֶד לָמוֹ:  ת  וַאֲמַר בְּרִיךְ יְיָ אֱלָהֵהּ דְּשֵׁם וִיהֵי כְנַעַן עַבְדָא לְהוֹן:

{כז} יַפְתְּ אֱלֹהִים לְיֶפֶת וְיִשְׁכֹּן בְּאָהֳלֵי שֵׁם וִיהִי כְנַעַן עֶבֶד לָמוֹ:  מ  יַפְתְּ אֱלֹהִים לְיֶפֶת וְיִשְׁכֹּן בְּאָהֳלֵי שֵׁם וִיהִי כְנַעַן עֶבֶד לָמוֹ:  ת  יַפְתֵּי יְיָ לְיֶפֶת וְיִשְׁרֵי שְׁכִינְתֵּהּ בְּמַשְׁכְּנֵהּ דְּשֵׁם וִיהֵי כְנַעַן עַבְדָא לְהוֹן:

{כח} וַיְחִי נֹחַ אַחַר הַמַּבּוּל שְׁלֹשׁ מֵאוֹת שָׁנָה וַחֲמִשִּׁים שָׁנָה:  מ  וַיְחִי נֹחַ אַחַר הַמַּבּוּל שְׁלֹשׁ מֵאוֹת שָׁנָה וַחֲמִשִּׁים שָׁנָה:  ת  וַחֲיָא נֹחַ בָּתַר טוֹפָנָא תְּלַת מְאָה וְחַמְשִׁין שְׁנִין:

{כט} וַיְהִי כָּל יְמֵי נֹחַ תְּשַׁע מֵאוֹת שָׁנָה וַחֲמִשִּׁים שָׁנָה וַיָּמֹת:  מ  וַיְהִי כָּל יְמֵי נֹחַ תְּשַׁע מֵאוֹת שָׁנָה וַחֲמִשִּׁים שָׁנָה וַיָּמֹת:  ת  וַהֲווֹ כָל יוֹמֵי נֹחַ תְּשַׁע מְאָה וְחַמְשִׁין שְׁנִין וּמִית: (פ)




בראשית פרק-י

{א} וְאֵלֶּה תּוֹלְדֹת בְּנֵי נֹחַ שֵׁם חָם וָיָפֶת וַיִּוָּלְדוּ לָהֶם בָּנִים אַחַר הַמַּבּוּל:  מ  וְאֵלֶּה תּוֹלְדֹת בְּנֵי נֹחַ שֵׁם חָם וָיָפֶת וַיִּוָּלְדוּ לָהֶם בָּנִים אַחַר הַמַּבּוּל:  ת  וְאִלֵּין תּוּלְדַת בְּנֵי נֹחַ שֵׁם חָם וָיָפֶת וְאִתְיְלִידוּ לְהוֹן בְּנִין בָּתַר טוֹפָנָא:

{ב} בְּנֵי יֶפֶת גֹּמֶר וּמָגוֹג וּמָדַי וְיָוָן וְתֻבָל וּמֶשֶׁךְ וְתִירָס:  מ  בְּנֵי יֶפֶת גֹּמֶר וּמָגוֹג וּמָדַי וְיָוָן וְתֻבָל וּמֶשֶׁךְ וְתִירָס:  ת  בְּנֵי יֶפֶת גּוֹמֶר וּמָגוֹג וּמָדַי וְיָוָן וְתוּבָל וּמֶשֶׁךְ וְתִירָס:

{ג} וּבְנֵי גֹּמֶר אַשְׁכֲּנַז וְרִיפַת וְתֹגַרְמָה:  מ  וּבְנֵי גֹּמֶר אַשְׁכֲּנַז וְרִיפַת וְתֹגַרְמָה:  ת  וּבְנֵי גּוֹמֶר אַשְׁכְּנַז וְרִיפַת וְתוֹגַרְמָה:

{ד} וּבְנֵי יָוָן אֱלִישָׁה וְתַרְשִׁישׁ כִּתִּים וְדֹדָנִים:  מ  וּבְנֵי יָוָן אֱלִישָׁה וְתַרְשִׁישׁ כִּתִּים וְדֹדָנִים:  ת  וּבְנֵי יָוָן אֱלִישָׁה וְתַרְשִׁישׁ כִּתִּים וְדֹדָנִים:

{ה} מֵאֵלֶּה נִפְרְדוּ אִיֵּי הַגּוֹיִם בְּאַרְצֹתָם אִישׁ לִלְשֹׁנוֹ לְמִשְׁפְּחֹתָם בְּגוֹיֵהֶם:  מ  מֵאֵלֶּה נִפְרְדוּ אִיֵּי הַגּוֹיִם בְּאַרְצֹתָם אִישׁ לִלְשֹׁנוֹ לְמִשְׁפְּחֹתָם בְּגוֹיֵהֶם:  ת  מֵאִלֵּין אִתַּפְרָשׁוּ נַגְוַת עַמְמַיָּא בְּאַרְעֲהוֹן גְּבַר לְלִישָׁנֵהּ לְזַרְעֲיָתְהוֹן בְּעַמְמֵיהוֹן:

{ו} וּבְנֵי חָם כּוּשׁ וּמִצְרַיִם וּפוּט וּכְנָעַן:  מ  וּבְנֵי חָם כּוּשׁ וּמִצְרַיִם וּפוּט וּכְנָעַן:  ת  וּבְנֵי חָם כּוּשׁ וּמִצְרַיִם וּפוּט וּכְנָעַן:

{ז} וּבְנֵי כוּשׁ סְבָא וַחֲוִילָה וְסַבְתָּה וְרַעְמָה וְסַבְתְּכָא וּבְנֵי רַעְמָה שְׁבָא וּדְדָן:  מ  וּבְנֵי כוּשׁ סְבָא וַחֲוִילָה וְסַבְתָּה וְרַעְמָה וְסַבְתְּכָא וּבְנֵי רַעְמָה שְׁבָא וּדְדָן:  ת  וּבְנֵי כוּשׁ סְבָא וַחֲוִילָה וְסַבְתָּה וְרַעְמָה וְסַבְתְּכָא וּבְנֵי רַעְמָה שְׁבָא וּדְדָן:

{ח} וְכוּשׁ יָלַד אֶת נִמְרֹד הוּא הֵחֵל לִהְיוֹת גִּבֹּר בָּאָרֶץ:  מ  וְכוּשׁ יָלַד אֶת נִמְרֹד הוּא הֵחֵל לִהְיוֹת גִּבֹּר בָּאָרֶץ:  ת  וְכוּשׁ אוֹלִיד יָת נִמְרֹד הוּא שָׁרֵי לְמֵהֲוֵי גִבַּר תַּקִּיף בְּאַרְעָא:

{ט} הוּא הָיָה גִבֹּר צַיִד לִפְנֵי יְהוָה עַל כֵּן יֵאָמַר כְּנִמְרֹד גִּבּוֹר צַיִד לִפְנֵי יְהוָה:  מ  הוּא הָיָה גִבֹּר צַיִד לִפְנֵי יְהוָה עַל כֵּן יֵאָמַר כְּנִמְרֹד גִּבּוֹר צַיִד לִפְנֵי יְהוָה:  ת  הוּא הֲוָה גִבַּר תַּקִּיף קֳדָם יְיָ עַל כֵּן יִתְאֲמַר כְּנִמְרֹד גִּבַּר תַּקִּיף קֳדָם יְיָ:

{י} וַתְּהִי רֵאשִׁית מַמְלַכְתּוֹ בָּבֶל וְאֶרֶךְ וְאַכַּד וְכַלְנֵה בְּאֶרֶץ שִׁנְעָר:  מ  וַתְּהִי רֵאשִׁית מַמְלַכְתּוֹ בָּבֶל וְאֶרֶךְ וְאַכַּד וְכַלְנֵה בְּאֶרֶץ שִׁנְעָר:  ת  וַהֲוָה רֵישׁ מַלְכוּתֵהּ בָּבֶל וְאֶרֶךְ וְאַכַּד וְכַלְנֵה בְּאַרְעָא דְּבָבֶל:

{יא} מִן הָאָרֶץ הַהִוא יָצָא אַשּׁוּר וַיִּבֶן אֶת נִינְוֵה וְאֶת רְחֹבֹת עִיר וְאֶת כָּלַח:  מ  מִן הָאָרֶץ הַהִוא יָצָא אַשּׁוּר וַיִּבֶן אֶת נִינְוֵה וְאֶת רְחֹבֹת עִיר וְאֶת כָּלַח:  ת  מִן אַרְעָא (ס''א עֵיצָה) הַהִיא נְפַק אֲתוּרָאָה וּבְנָא יָת נִינְוֵה וְיָת רְחֹבֹת (נ''א רְחוֹבֵי) קַרְתָּא וְיָת כָּלַח:

{יב} וְאֶת רֶסֶן בֵּין נִינְוֵה וּבֵין כָּלַח הִוא הָעִיר הַגְּדֹלָה:  מ  וְאֶת רֶסֶן בֵּין נִינְוֵה וּבֵין כָּלַח הִוא הָעִיר הַגְּדֹלָה:  ת  וְיָת רֶסֶן בֵּין נִינְוֵה וּבֵין כָּלַח הִיא קַרְתָּא רַבְּתָא:

{יג} וּמִצְרַיִם יָלַד אֶת לוּדִים וְאֶת עֲנָמִים וְאֶת לְהָבִים וְאֶת נַפְתֻּחִים:  מ  וּמִצְרַיִם יָלַד אֶת לוּדִים וְאֶת עֲנָמִים וְאֶת לְהָבִים וְאֶת נַפְתֻּחִים:  ת  וּמִצרָאֵי (בכל הס':וּמִצְרַיִם) אוֹלִיד יָת לוּדָאֵי וְיָת עֲנָמָאֵי וְיָת לְהָבָאֵי וְיָת נַפְתּוּחָאֵי:

{יד} וְאֶת פַּתְרֻסִים וְאֶת כַּסְלֻחִים אֲשֶׁר יָצְאוּ מִשָּׁם פְּלִשְׁתִּים וְאֶת כַּפְתֹּרִים:  מ  וְאֶת פַּתְרֻסִים וְאֶת כַּסְלֻחִים אֲשֶׁר יָצְאוּ מִשָּׁם פְּלִשְׁתִּים וְאֶת כַּפְתֹּרִים:  ת  וְיָת פַּתְרוּסָאֵי וְיָת כַּסְלוּחָאֵי דִּי נְפָקוּ מִתַּמָּן פְּלִשְׁתָּאֵי וְיָת קַפּוּטְקָאֵי: (ס)

{טו} וּכְנַעַן יָלַד אֶת צִידֹן בְּכֹרוֹ וְאֶת חֵת:  מ  וּכְנַעַן יָלַד אֶת צִידֹן בְּכֹרוֹ וְאֶת חֵת:  ת  וּכְנַעַן אוֹלִיד יָת צִידוֹן בָּכְרֵהּ וְיָת חֵת:

{טז} וְאֶת הַיְבוּסִי וְאֶת הָאֱמֹרִי וְאֵת הַגִּרְגָּשִׁי:  מ  וְאֶת הַיְבוּסִי וְאֶת הָאֱמֹרִי וְאֵת הַגִּרְגָּשִׁי:  ת  וְיָת יְבוּסָאֵי וְיָת אֱמוֹרָאֵי וְיָת גִּרְגָּשָׁאֵי:

{יז} וְאֶת הַחִוִּי וְאֶת הַעַרְקִי וְאֶת הַסִּינִי:  מ  וְאֶת הַחִוִּי וְאֶת הַעַרְקִי וְאֶת הַסִּינִי:  ת  וְיָת חִוָּאֵי וְיָת עַרְקָאֵי וְיָת אַנְתּוֹסָאֵי:

{יח} וְאֶת הָאַרְוָדִי וְאֶת הַצְּמָרִי וְאֶת הַחֲמָתִי וְאַחַר נָפֹצוּ מִשְׁפְּחוֹת הַכְּנַעֲנִי:  מ  וְאֶת הָאַרְוָדִי וְאֶת הַצְּמָרִי וְאֶת הַחֲמָתִי וְאַחַר נָפֹצוּ מִשְׁפְּחוֹת הַכְּנַעֲנִי:  ת  וְיָת אַרְוָדָאֵי וְיָת צְמָרָאֵי וְיָת חֲמָתָאֵי וּבָתַר כֵּן אִתְבַּדָּרוּ זַרְעֲיַת כְּנַעֲנָאֵי:

{יט} וַיְהִי גְּבוּל הַכְּנַעֲנִי מִצִּידֹן בֹּאֲכָה גְרָרָה עַד עַזָּה בֹּאֲכָה סְדֹמָה וַעֲמֹרָה וְאַדְמָה וּצְבֹיִם עַד לָשַׁע:  מ  וַיְהִי גְּבוּל הַכְּנַעֲנִי מִצִּידֹן בֹּאֲכָה גְרָרָה עַד עַזָּה בֹּאֲכָה סְדֹמָה וַעֲמֹרָה וְאַדְמָה וּצְבֹיִם עַד לָשַׁע:  ת  וַהֲוָה תְּחוּם כְּנַעֲנָאֵי מִצִידוֹן מָטֵי לִגְרָר עַד עַזָּה מָטֵי לִסְדוֹם וַעֲמוֹרָה וְאַדְמָה וּצְבוֹיִם עַד לָשַׁע:

{כ} אֵלֶּה בְנֵי חָם לְמִשְׁפְּחֹתָם לִלְשֹׁנֹתָם בְּאַרְצֹתָם בְּגוֹיֵהֶם:  מ  אֵלֶּה בְנֵי חָם לְמִשְׁפְּחֹתָם לִלְשֹׁנֹתָם בְּאַרְצֹתָם בְּגוֹיֵהֶם:  ת  אִלֵּין בְּנֵי חָם לְזַרְעֲיָתְהוֹן לְלִישָׁנְהוֹן בְּאַרְעָתְהוֹן בְּעַמְמֵיהוֹן: (ס)

{כא} וּלְשֵׁם יֻלַּד גַּם הוּא אֲבִי כָּל בְּנֵי עֵבֶר אֲחִי יֶפֶת הַגָּדוֹל:  מ  וּלְשֵׁם יֻלַּד גַּם הוּא אֲבִי כָּל בְּנֵי עֵבֶר אֲחִי יֶפֶת הַגָּדוֹל:  ת  וּלְשֵׁם אִתְיְלִיד אַף הוּא אֲבוּהוֹן דְּכָל בְּנֵי עֵבֶר אָחוּהִי דְּיֶפֶת רַבָּא:

{כב} בְּנֵי שֵׁם עֵילָם וְאַשּׁוּר וְאַרְפַּכְשַׁד וְלוּד וַאֲרָם:  מ  בְּנֵי שֵׁם עֵילָם וְאַשּׁוּר וְאַרְפַּכְשַׁד וְלוּד וַאֲרָם:  ת  בְּנֵי שֵׁם עֵילָם וְאַשּׁוּר וְאַרְפַּכְשַׁד וְלוּד וַאֲרָם:

{כג} וּבְנֵי אֲרָם עוּץ וְחוּל וְגֶתֶר וָמַשׁ:  מ  וּבְנֵי אֲרָם עוּץ וְחוּל וְגֶתֶר וָמַשׁ:  ת  וּבְנֵי אֲרָם עוּץ וְחוּל וְגֶתֶר וָמַשׁ:

{כד} וְאַרְפַּכְשַׁד יָלַד אֶת שָׁלַח וְשֶׁלַח יָלַד אֶת עֵבֶר:  מ  וְאַרְפַּכְשַׁד יָלַד אֶת שָׁלַח וְשֶׁלַח יָלַד אֶת עֵבֶר:  ת  וְאַרְפַּכְשַׁד אוֹלִיד יָת שָׁלַח וְשֶׁלַח אוֹלִיד יָת עֵבֶר:

{כה} וּלְעֵבֶר יֻלַּד שְׁנֵי בָנִים שֵׁם הָאֶחָד פֶּלֶג כִּי בְיָמָיו נִפְלְגָה הָאָרֶץ וְשֵׁם אָחִיו יָקְטָן:  מ  וּלְעֵבֶר יֻלַּד שְׁנֵי בָנִים שֵׁם הָאֶחָד פֶּלֶג כִּי בְיָמָיו נִפְלְגָה הָאָרֶץ וְשֵׁם אָחִיו יָקְטָן:  ת  וּלְעֵבֶר אִתְיְלִידוּ תְּרֵין בְּנִין שׁוּם חַד פֶּלֶג אֲרֵי בְּיוֹמוֹהִי אִתְפְּלִיגַת אַרְעָא וְשׁוּם אֲחוּהִי יָקְטָן:

{כו} וְיָקְטָן יָלַד אֶת אַלְמוֹדָד וְאֶת שָׁלֶף וְאֶת חֲצַרְמָוֶת וְאֶת יָרַח:  מ  וְיָקְטָן יָלַד אֶת אַלְמוֹדָד וְאֶת שָׁלֶף וְאֶת חֲצַרְמָוֶת וְאֶת יָרַח:  ת  וְיָקְטָן אוֹלִיד יָת אַלְמוֹדָד וְיָת שָׁלֶף וְיָת חֲצַרְמָוֶת וְיָת יָרַח:

{כז} וְאֶת הֲדוֹרָם וְאֶת אוּזָל וְאֶת דִּקְלָה:  מ  וְאֶת הֲדוֹרָם וְאֶת אוּזָל וְאֶת דִּקְלָה:  ת  וְיָת הֲדוֹרָם וְיָת אוּזָל וְיָת דִּקְלָה:

{כח} וְאֶת עוֹבָל וְאֶת אֲבִימָאֵל וְאֶת שְׁבָא:  מ  וְאֶת עוֹבָל וְאֶת אֲבִימָאֵל וְאֶת שְׁבָא:  ת  וְיָת עוֹבָל וְיָת אֲבִימָאֵל וְיָת שְׁבָא:

{כט} וְאֶת אוֹפִר וְאֶת חֲוִילָה וְאֶת יוֹבָב כָּל אֵלֶּה בְּנֵי יָקְטָן:  מ  וְאֶת אוֹפִר וְאֶת חֲוִילָה וְאֶת יוֹבָב כָּל אֵלֶּה בְּנֵי יָקְטָן:  ת  וְיָת אוֹפִר וְיָת חֲוִילָה וְיָת יוֹבָב כָּל אִלֵּין בְּנֵי יָקְטָן:

{ל} וַיְהִי מוֹשָׁבָם מִמֵּשָׁא בֹּאֲכָה סְפָרָה הַר הַקֶּדֶם:  מ  וַיְהִי מוֹשָׁבָם מִמֵּשָׁא בֹּאֲכָה סְפָרָה הַר הַקֶּדֶם:  ת  וַהֲוָה מוֹתְבָנֵהוֹן מִמֵּשָׁא מָטֵי לִסְפַר טּוּר מָדִינְחָא:

{לא} אֵלֶּה בְנֵי שֵׁם לְמִשְׁפְּחֹתָם לִלְשֹׁנֹתָם בְּאַרְצֹתָם לְגוֹיֵהֶם:  מ  אֵלֶּה בְנֵי שֵׁם לְמִשְׁפְּחֹתָם לִלְשֹׁנֹתָם בְּאַרְצֹתָם לְגוֹיֵהֶם:  ת  אִלֵּין בְּנֵי שֵׁם לְזַרְעֲיָתְהוֹן לְלִישָׁנְהוֹן לְאַרְעָתְהוֹן בְּעַמְמֵיהוֹן:

{לב} אֵלֶּה מִשְׁפְּחֹת בְּנֵי נֹחַ לְתוֹלְדֹתָם בְּגוֹיֵהֶם וּמֵאֵלֶּה נִפְרְדוּ הַגּוֹיִם בָּאָרֶץ אַחַר הַמַּבּוּל:  מ  אֵלֶּה מִשְׁפְּחֹת בְּנֵי נֹחַ לְתוֹלְדֹתָם בְּגוֹיֵהֶם וּמֵאֵלֶּה נִפְרְדוּ הַגּוֹיִם בָּאָרֶץ אַחַר הַמַּבּוּל:  ת  אִלֵּין זַרְעֲיַת בְּנֵי נֹחַ לְתוּלְדָתְהוֹן בְּעַמְמֵיהוֹן וּמֵאִלֵּין אִתַּפְרָשׁוּ עַמְמַיָּא בְּאַרְעָא בָּתַר טוֹפָנָא: (פ)




בראשית פרק-יא

{א}  שביעי  וַיְהִי כָל הָאָרֶץ שָׂפָה אֶחָת וּדְבָרִים אֲחָדִים:  מ  וַיְהִי כָל הָאָרֶץ שָׂפָה אֶחָת וּדְבָרִים אֲחָדִים:  ת  וַהֲוָה כָל אַרְעָא לִישָׁן חָד וּמַמְלָל חָד:

{ב} וַיְהִי בְּנָסְעָם מִקֶּדֶם וַיִּמְצְאוּ בִקְעָה בְּאֶרֶץ שִׁנְעָר וַיֵּשְׁבוּ שָׁם:  מ  וַיְהִי בְּנָסְעָם מִקֶּדֶם וַיִּמְצְאוּ בִקְעָה בְּאֶרֶץ שִׁנְעָר וַיֵּשְׁבוּ שָׁם:  ת  וַהֲוָה בְּמִטַּלְהוֹן בְּקַדְמֵיתָא וְאַשְׁכָּחוּ בִקְעֲתָא בְּאַרְעָא דְבָבֶל וִיתִיבוּ תַמָּן:

{ג} וַיֹּאמְרוּ אִישׁ אֶל רֵעֵהוּ הָבָה נִלְבְּנָה לְבֵנִים וְנִשְׂרְפָה לִשְׂרֵפָה וַתְּהִי לָהֶם הַלְּבֵנָה לְאָבֶן וְהַחֵמָר הָיָה לָהֶם לַחֹמֶר:  מ  וַיֹּאמְרוּ אִישׁ אֶל רֵעֵהוּ הָבָה נִלְבְּנָה לְבֵנִים וְנִשְׂרְפָה לִשְׂרֵפָה וַתְּהִי לָהֶם הַלְּבֵנָה לְאָבֶן וְהַחֵמָר הָיָה לָהֶם לַחֹמֶר:  ת  וַאֲמָרוּ גְּבַר לְחַבְרֵהּ הָבוּ נִרְמֵי לִבְנִין וְנוֹקְדִינוּן בְּנוּרָא יוֹקַדְתָּא (נ''י בְּנוּרָא) וַהֲוַת לְהוֹן לְבִנְתָּא לְאַבְנָא וְחֵימָרָא הֲוָה לְהוֹן לְשִׁיעַ:

{ד} וַיֹּאמְרוּ הָבָה נִבְנֶה לָּנוּ עִיר וּמִגְדָּל וְרֹאשׁוֹ בַשָּׁמַיִם וְנַעֲשֶׂה לָּנוּ שֵׁם פֶּן נָפוּץ עַל פְּנֵי כָל הָאָרֶץ:  מ  וַיֹּאמְרוּ הָבָה נִבְנֶה לָּנוּ עִיר וּמִגְדָּל וְרֹאשׁוֹ בַשָּׁמַיִם וְנַעֲשֶׂה לָּנוּ שֵׁם פֶּן נָפוּץ עַל פְּנֵי כָל הָאָרֶץ:  ת  וַאֲמָרוּ הָבוּ נִבְנֵי לָנָא קַרְתָּא וּמַגְדְּלָא וְרֵישֵׁהּ מָטֵי עַד צֵית שְׁמַיָּא וְנַעְבֵּיד לָנָא שׁוּם דִּילְמָא נִתְבַּדַּר עַל אַפֵּי כָל אַרְעָא:

{ה} וַיֵּרֶד יְהוָה לִרְאֹת אֶת הָעִיר וְאֶת הַמִּגְדָּל אֲשֶׁר בָּנוּ בְּנֵי הָאָדָם:  מ  וַיֵּרֶד יְהוָה לִרְאֹת אֶת הָעִיר וְאֶת הַמִּגְדָּל אֲשֶׁר בָּנוּ בְּנֵי הָאָדָם:  ת  וְאִתְגְּלֵי יְיָ לְאִתְפְּרָעָא עַל עוֹבָדֵי קַרְתָּא וּמַגְדְּלָא דִּי בְנוֹ בְּנֵי אֱנָשָׁא:

{ו} וַיֹּאמֶר יְהוָה הֵן עַם אֶחָד וְשָׂפָה אַחַת לְכֻלָּם וְזֶה הַחִלָּם לַעֲשׂוֹת וְעַתָּה לֹא יִבָּצֵר מֵהֶם כֹּל אֲשֶׁר יָזְמוּ לַעֲשׂוֹת:  מ  וַיֹּאמֶר יְהוָה הֵן עַם אֶחָד וְשָׂפָה אַחַת לְכֻלָּם וְזֶה הַחִלָּם לַעֲשׂוֹת וְעַתָּה לֹא יִבָּצֵר מֵהֶם כֹּל אֲשֶׁר יָזְמוּ לַעֲשׂוֹת:  ת  וַאֲמַר יְיָ הָא עַמָּא חַד וְלִישָׁן חַד לְכֻלְּהוֹן וְדֵין דְּשָׁרִיוּ לְמֶעְבָּד וּכְעַן לָא יִתִּמְנַע מִנְּהוֹן כֹּל דִחַשִּׁיבוּ לְמֶעְבָּד:

{ז} הָבָה נֵרְדָה וְנָבְלָה שָׁם שְׂפָתָם אֲשֶׁר לֹא יִשְׁמְעוּ אִישׁ שְׂפַת רֵעֵהוּ:  מ  הָבָה נֵרְדָה וְנָבְלָה שָׁם שְׂפָתָם אֲשֶׁר לֹא יִשְׁמְעוּ אִישׁ שְׂפַת רֵעֵהוּ:  ת  הָבוּ נִתְגְּלִי וּנְבַלְבֵּל תַּמָּן לִישָׁנְהוֹן דְּלָא יִשְׁמְעוּן (גְּבַר) אֱנַשׁ לִישָׁן חַבְרֵהּ:

{ח} וַיָּפֶץ יְהוָה אֹתָם מִשָּׁם עַל פְּנֵי כָל הָאָרֶץ וַיַּחְדְּלוּ לִבְנֹת הָעִיר:  מ  וַיָּפֶץ יְהוָה אֹתָם מִשָּׁם עַל פְּנֵי כָל הָאָרֶץ וַיַּחְדְּלוּ לִבְנֹת הָעִיר:  ת  וּבַדַּר יְיָ יָתְהוֹן מִתַּמָּן עַל אַפֵּי כָל אַרְעָא וּמְנָעוּ לְמִבְנֵי (נ''י וְאִתְמְנָע מִלְּמִבְנֵי) קַרְתָּא:

{ט} עַל כֵּן קָרָא שְׁמָהּ בָּבֶל כִּי שָׁם בָּלַל יְהוָה שְׂפַת כָּל הָאָרֶץ וּמִשָּׁם הֱפִיצָם יְהוָה עַל פְּנֵי כָּל הָאָרֶץ:  מ  עַל כֵּן קָרָא שְׁמָהּ בָּבֶל כִּי שָׁם בָּלַל יְהוָה שְׂפַת כָּל הָאָרֶץ וּמִשָּׁם הֱפִיצָם יְהוָה עַל פְּנֵי כָּל הָאָרֶץ:  ת  עַל כֵּן קְרָא שְׁמַהּ בָּבֶל אֲרֵי תַמָּן בַּלְבֵּל יְיָ לִישַׁן כָּל אַרְעָא וּמִתַּמָּן בַּדָּרִנּוּן יְיָ עַל אַפֵּי כָּל אַרְעָא: (פ)

{י} אֵלֶּה תּוֹלְדֹת שֵׁם שֵׁם בֶּן מְאַת שָׁנָה וַיּוֹלֶד אֶת אַרְפַּכְשָׁד שְׁנָתַיִם אַחַר הַמַּבּוּל:  מ  אֵלֶּה תּוֹלְדֹת שֵׁם שֵׁם בֶּן מְאַת שָׁנָה וַיּוֹלֶד אֶת אַרְפַּכְשָׁד שְׁנָתַיִם אַחַר הַמַּבּוּל:  ת  אִלֵּין תּוּלְדַת שֵׁם שֵׁם בַּר מְאָה שְׁנִין וְאוֹלִיד יָת אַרְפַּכְשָׁד תַּרְתֵּין שְׁנִין בָּתַר טּוֹפָנָא:

{יא} וַיְחִי שֵׁם אַחֲרֵי הוֹלִידוֹ אֶת אַרְפַּכְשָׁד חֲמֵשׁ מֵאוֹת שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  מ  וַיְחִי שֵׁם אַחֲרֵי הוֹלִידוֹ אֶת אַרְפַּכְשָׁד חֲמֵשׁ מֵאוֹת שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  ת  וַחֲיָא שֵׁם בָּתַר דְּאוֹלִיד יָת אַרְפַּכְשָׁד חֲמֵשׁ מְאָה שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן: (ס)

{יב} וְאַרְפַּכְשַׁד חַי חָמֵשׁ וּשְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת שָׁלַח:  מ  וְאַרְפַּכְשַׁד חַי חָמֵשׁ וּשְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת שָׁלַח:  ת  וְאַרְפַּכְשַׁד חֲיָא תְּלָתִין וַחֲמֵשׁ שְׁנִין וְאוֹלִיד יָת שָׁלַח:

{יג} וַיְחִי אַרְפַּכְשַׁד אַחֲרֵי הוֹלִידוֹ אֶת שֶׁלַח שָׁלֹשׁ שָׁנִים וְאַרְבַּע מֵאוֹת שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  מ  וַיְחִי אַרְפַּכְשַׁד אַחֲרֵי הוֹלִידוֹ אֶת שֶׁלַח שָׁלֹשׁ שָׁנִים וְאַרְבַּע מֵאוֹת שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  ת  וַחֲיָא אַרְפַּכְשַׁד בָּתַר דְּאוֹלִיד יָת שֶׁלַח אַרְבַּע מְאָה וּתְלָת שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן: (ס)

{יד} וְשֶׁלַח חַי שְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת עֵבֶר:  מ  וְשֶׁלַח חַי שְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת עֵבֶר:  ת  וְשֶׁלַח חֲיָא תְּלָתִין שְׁנִין וְאוֹלִיד יָת עֵבֶר:

{טו} וַיְחִי שֶׁלַח אַחֲרֵי הוֹלִידוֹ אֶת עֵבֶר שָׁלֹשׁ שָׁנִים וְאַרְבַּע מֵאוֹת שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  מ  וַיְחִי שֶׁלַח אַחֲרֵי הוֹלִידוֹ אֶת עֵבֶר שָׁלֹשׁ שָׁנִים וְאַרְבַּע מֵאוֹת שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  ת  וַחֲיָא שֶׁלַח בָּתַר דְּאוֹלִיד יָת עֵבֶר אַרְבַּע מְאָה וּתְלָת שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן: (ס)

{טז} וַיְחִי עֵבֶר אַרְבַּע וּשְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת פָּלֶג:  מ  וַיְחִי עֵבֶר אַרְבַּע וּשְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת פָּלֶג:  ת  וַחֲיָא עֵבֶר תְּלָתִין וְאַרְבַּע שְׁנִין וְאוֹלִיד יָת פָּלֶג:

{יז} וַיְחִי עֵבֶר אַחֲרֵי הוֹלִידוֹ אֶת פֶּלֶג שְׁלֹשִׁים שָׁנָה וְאַרְבַּע מֵאוֹת שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  מ  וַיְחִי עֵבֶר אַחֲרֵי הוֹלִידוֹ אֶת פֶּלֶג שְׁלֹשִׁים שָׁנָה וְאַרְבַּע מֵאוֹת שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  ת  וַחֲיָא עֵבֶר בָּתַר דְּאוֹלִיד יָת פֶּלֶג אַרְבַּע מְאָה וּתְּלָתִין שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן: (ס)

{יח} וַיְחִי פֶלֶג שְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת רְעוּ:  מ  וַיְחִי פֶלֶג שְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת רְעוּ:  ת  וַחֲיָא פֶלֶג תְּלָתִין שְׁנִין וְאוֹלִיד יָת רְעוּ:

{יט} וַיְחִי פֶלֶג אַחֲרֵי הוֹלִידוֹ אֶת רְעוּ תֵּשַׁע שָׁנִים וּמָאתַיִם שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  מ  וַיְחִי פֶלֶג אַחֲרֵי הוֹלִידוֹ אֶת רְעוּ תֵּשַׁע שָׁנִים וּמָאתַיִם שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  ת  וַחֲיָא פֶלֶג בָּתַר דְּאוֹלִיד יָת רְעוּ מָאתָן וּתְשַׁע שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן: (ס)

{כ} וַיְחִי רְעוּ שְׁתַּיִם וּשְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת שְׂרוּג:  מ  וַיְחִי רְעוּ שְׁתַּיִם וּשְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת שְׂרוּג:  ת  וַחֲיָא רְעוּ תְּלָתִין וְתַרְתֵּין שְׁנִין וְאוֹלִיד יָת שְׂרוּג:

{כא} וַיְחִי רְעוּ אַחֲרֵי הוֹלִידוֹ אֶת שְׂרוּג שֶׁבַע שָׁנִים וּמָאתַיִם שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  מ  וַיְחִי רְעוּ אַחֲרֵי הוֹלִידוֹ אֶת שְׂרוּג שֶׁבַע שָׁנִים וּמָאתַיִם שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  ת  וַחֲיָא רְעוּ בָּתַר דְּאוֹלִיד יָת שְׂרוּג מָאתָן וּשְׁבַע שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן: (ס)

{כב} וַיְחִי שְׂרוּג שְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת נָחוֹר:  מ  וַיְחִי שְׂרוּג שְׁלֹשִׁים שָׁנָה וַיּוֹלֶד אֶת נָחוֹר:  ת  וַחֲיָא שְׂרוּג תְּלָתִין שְׁנִין וְאוֹלִיד יָת נָחוֹר:

{כג} וַיְחִי שְׂרוּג אַחֲרֵי הוֹלִידוֹ אֶת נָחוֹר מָאתַיִם שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  מ  וַיְחִי שְׂרוּג אַחֲרֵי הוֹלִידוֹ אֶת נָחוֹר מָאתַיִם שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  ת  וַחֲיָא שְׂרוּג בָּתַר דְּאוֹלִיד יָת נָחוֹר מָאתָן שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן: (ס)

{כד} וַיְחִי נָחוֹר תֵּשַׁע וְעֶשְׂרִים שָׁנָה וַיּוֹלֶד אֶת תָּרַח:  מ  וַיְחִי נָחוֹר תֵּשַׁע וְעֶשְׂרִים שָׁנָה וַיּוֹלֶד אֶת תָּרַח:  ת  וַחֲיָא נָחוֹר עֶשְׂרִין וּתְשַׁע שְׁנִין וְאוֹלִיד יָת תָּרַח:

{כה} וַיְחִי נָחוֹר אַחֲרֵי הוֹלִידוֹ אֶת תֶּרַח תְּשַׁע עֶשְׂרֵה שָׁנָה וּמְאַת שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  מ  וַיְחִי נָחוֹר אַחֲרֵי הוֹלִידוֹ אֶת תֶּרַח תְּשַׁע עֶשְׂרֵה שָׁנָה וּמְאַת שָׁנָה וַיּוֹלֶד בָּנִים וּבָנוֹת:  ת  וַחֲיָא נָחוֹר בָּתַר דְּאוֹלִיד יָת תֶּרַח מְאָה וּתְשַׁע עֲשַׂר שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן: (ס)

{כו} וַיְחִי תֶרַח שִׁבְעִים שָׁנָה וַיּוֹלֶד אֶת אַבְרָם אֶת נָחוֹר וְאֶת הָרָן:  מ  וַיְחִי תֶרַח שִׁבְעִים שָׁנָה וַיּוֹלֶד אֶת אַבְרָם אֶת נָחוֹר וְאֶת הָרָן:  ת  וַחֲיָא תֶרַח שִׁבְעִין שְׁנִין וְאוֹלִיד יָת אַבְרָם יָת נָחוֹר וְיָת הָרָן:

{כז} וְאֵלֶּה תּוֹלְדֹת תֶּרַח תֶּרַח הוֹלִיד אֶת אַבְרָם אֶת נָחוֹר וְאֶת הָרָן וְהָרָן הוֹלִיד אֶת לוֹט:  מ  וְאֵלֶּה תּוֹלְדֹת תֶּרַח תֶּרַח הוֹלִיד אֶת אַבְרָם אֶת נָחוֹר וְאֶת הָרָן וְהָרָן הוֹלִיד אֶת לוֹט:  ת  וְאִלֵּין תּוּלְדַת תֶּרַח תֶּרַח אוֹלִיד יָת אַבְרָם יָת נָחוֹר וְיָת הָרָן וְהָרָן אוֹלִיד יָת לוֹטּ:

{כח} וַיָּמָת הָרָן עַל פְּנֵי תֶּרַח אָבִיו בְּאֶרֶץ מוֹלַדְתּוֹ בְּאוּר כַּשְׂדִּים:  מ  וַיָּמָת הָרָן עַל פְּנֵי תֶּרַח אָבִיו בְּאֶרֶץ מוֹלַדְתּוֹ בְּאוּר כַּשְׂדִּים:  ת  וּמִית הָרָן עַל אַפֵּי תֶּרַח אֲבוּהִי בְּאַרְעָא דְּיַלָּדוּתֵהּ בְּאוּרָא דְכַסְדָּאֵי:

{כט}  מפטיר  וַיִּקַּח אַבְרָם וְנָחוֹר לָהֶם נָשִׁים שֵׁם אֵשֶׁת אַבְרָם שָׂרָי וְשֵׁם אֵשֶׁת נָחוֹר מִלְכָּה בַּת הָרָן אֲבִי מִלְכָּה וַאֲבִי יִסְכָּה:  מ  וַיִּקַּח אַבְרָם וְנָחוֹר לָהֶם נָשִׁים שֵׁם אֵשֶׁת אַבְרָם שָׂרָי וְשֵׁם אֵשֶׁת נָחוֹר מִלְכָּה בַּת הָרָן אֲבִי מִלְכָּה וַאֲבִי יִסְכָּה:  ת  וּנְסִיב אַבְרָם וְנָחוֹר לְהוֹן נְשִׁין שׁוּם אִתַּת אַבְרָם שָׂרָי וְשׁוּם אִתַּת נָחוֹר מִלְכָּה בַּת הָרָן אֲבוּהָא דְּמִלְכָּה וַאֲבוּהָא דְּיִסְכָּה:

{ל} וַתְּהִי שָׂרַי עֲקָרָה אֵין לָהּ וָלָד:  מ  וַתְּהִי שָׂרַי עֲקָרָה אֵין לָהּ וָלָד:  ת  וַהֲוַת שָׂרַי עֲקָרָה לֵית לָהּ וְלָד:

{לא} וַיִּקַּח תֶּרַח אֶת אַבְרָם בְּנוֹ וְאֶת לוֹט בֶּן הָרָן בֶּן בְּנוֹ וְאֵת שָׂרַי כַּלָּתוֹ אֵשֶׁת אַבְרָם בְּנוֹ וַיֵּצְאוּ אִתָּם מֵאוּר כַּשְׂדִּים לָלֶכֶת אַרְצָה כְּנַעַן וַיָּבֹאוּ עַד חָרָן וַיֵּשְׁבוּ שָׁם:  מ  וַיִּקַּח תֶּרַח אֶת אַבְרָם בְּנוֹ וְאֶת לוֹט בֶּן הָרָן בֶּן בְּנוֹ וְאֵת שָׂרַי כַּלָּתוֹ אֵשֶׁת אַבְרָם בְּנוֹ וַיֵּצְאוּ אִתָּם מֵאוּר כַּשְׂדִּים לָלֶכֶת אַרְצָה כְּנַעַן וַיָּבֹאוּ עַד חָרָן וַיֵּשְׁבוּ שָׁם:  ת  וּדְבַר תֶּרַח יָת אַבְרָם בְּרֵהּ וְיָת לוֹט בַּר הָרָן בַּר בְּרֵהּ וְיָת שָׂרַי כַּלָּתֵהּ אִתַּת אַבְרָם בְּרֵהּ וּנְפָקוּ עִמְּהוֹן מֵאוּרָא דְּכַסְדָּאֵי לְמֵיזַל לְאַרְעָא דִּכְנָעַן וַאֲתוֹ עַד חָרָן וִיתִיבוּ תַמָּן:

{לב} וַיִּהְיוּ יְמֵי תֶרַח חָמֵשׁ שָׁנִים וּמָאתַיִם שָׁנָה וַיָּמָת תֶּרַח בְּחָרָן:  מ  וַיִּהְיוּ יְמֵי תֶרַח חָמֵשׁ שָׁנִים וּמָאתַיִם שָׁנָה וַיָּמָת תֶּרַח בְּחָרָן:  ת  וַהֲווֹ יוֹמֵי תֶרַח מָאָתן וַחֲמֵשׁ שְׁנִין וּמִית תֶּרַח בְּחָרָן: (פפפ)
