שמות פרק-כא

{א} וְאֵ֙לֶּה֙ הַמִּשְׁפָּטִ֔ים אֲשֶׁ֥ר תָּשִׂ֖ים לִפְנֵיהֶֽם:  מ  וְאֵ֙לֶּה֙ הַמִּשְׁפָּטִ֔ים אֲשֶׁ֥ר תָּשִׂ֖ים לִפְנֵיהֶֽם:  ת  וְאִלֵין דִינַיָא דִי תַסְדַר קֳדָמֵיהוֹן:

{ב} כִּ֤י תִקְנֶה֙ עֶ֣בֶד עִבְרִ֔י שֵׁ֥שׁ שָׁנִ֖ים יַעֲבֹ֑ד וּבַ֨שְּׁבִעִ֔ת יֵצֵ֥א לַֽחָפְשִׁ֖י חִנָּֽם:  מ  כִּ֤י תִקְנֶה֙ עֶ֣בֶד עִבְרִ֔י שֵׁ֥שׁ שָׁנִ֖ים יַעֲבֹ֑ד וּבַ֨שְּׁבִעִ֔ת יֵצֵ֥א לַֽחָפְשִׁ֖י חִנָּֽם:  ת  אֲרֵי תִזְבַּן עַבְדָא בַר יִשְׂרָאֵל שִׁתְּ שְׁנִין יִפְלָח וּבַשְׁבִיעֵתָא יִפּוֹק לְבַר חוֹרִין מַגָן:

{ג} אִם-בְּגַפּ֥וֹ יָבֹ֖א בְּגַפּ֣וֹ יֵצֵ֑א אִם-בַּ֤עַל אִשָּׁה֙ ה֔וּא וְיָצְאָ֥ה אִשְׁתּ֖וֹ עִמּֽוֹ:  מ  אִם-בְּגַפּ֥וֹ יָבֹ֖א בְּגַפּ֣וֹ יֵצֵ֑א אִם-בַּ֤עַל אִשָּׁה֙ ה֔וּא וְיָצְאָ֥ה אִשְׁתּ֖וֹ עִמּֽוֹ:  ת  אִם בִּלְחוֹדוֹהִי יֵיעוֹל בִּלְחוֹדוֹהִי יִפּוֹק אִם בְּעֵל אִתְּתָא הוּא וְתִפּוֹק אִתְּתֵהּ עִמֵהּ:

{ד} אִם-אֲדֹנָיו֙ יִתֶּן-ל֣וֹ אִשָּׁ֔ה וְיָלְדָה-ל֥וֹ בָנִ֖ים א֣וֹ בָנ֑וֹת הָאִשָּׁ֣ה וִילָדֶ֗יהָ תִּהְיֶה֙ לַֽאדֹנֶ֔יהָ וְה֖וּא יֵצֵ֥א בְגַפּֽוֹ:  מ  אִם-אֲדֹנָיו֙ יִתֶּן-ל֣וֹ אִשָּׁ֔ה וְיָלְדָה-ל֥וֹ בָנִ֖ים א֣וֹ בָנ֑וֹת הָאִשָּׁ֣ה וִילָדֶ֗יהָ תִּהְיֶה֙ לַֽאדֹנֶ֔יהָ וְה֖וּא יֵצֵ֥א בְגַפּֽוֹ:  ת  אִם רִבּוֹנֵהּ יִתֶּן לֵהּ אִתְּתָא וּתְלִיד לֵהּ בְּנִין אוֹ בְנָן אִתְּתָא וּבְנָהָא תְּהֵי לְרִבּוֹנָהָא וְהוּא יִפּוֹק בִּלְחוֹדוֹהִי:

{ה} וְאִם-אָמֹ֤ר יֹאמַר֙ הָעֶ֔בֶד אָהַ֙בְתִּי֙ אֶת-אֲדֹנִ֔י אֶת-אִשְׁתִּ֖י וְאֶת-בָּנָ֑י לֹ֥א אֵצֵ֖א חָפְשִֽׁי:  מ  וְאִם-אָמֹ֤ר יֹאמַר֙ הָעֶ֔בֶד אָהַ֙בְתִּי֙ אֶת-אֲדֹנִ֔י אֶת-אִשְׁתִּ֖י וְאֶת-בָּנָ֑י לֹ֥א אֵצֵ֖א חָפְשִֽׁי:  ת  וְאִם מֵימַר יֵימַר עַבְדָא רָחֵימְנָא יָת רִבּוֹנִי יָת אִתְּתִי וְיָת בְּנָי לָא אֶפּוֹק לְבַר חוֹרִין:

{ו} וְהִגִּישׁ֤וֹ אֲדֹנָיו֙ אֶל-הָ֣אֱלֹהִ֔ים וְהִגִּישׁוֹ֙ אֶל-הַדֶּ֔לֶת א֖וֹ אֶל-הַמְּזוּזָ֑ה וְרָצַ֨ע אֲדֹנָ֤יו אֶת-אָזְנוֹ֙ בַּמַּרְצֵ֔עַ וַעֲבָד֖וֹ לְעֹלָֽם:  מ  וְהִגִּישׁ֤וֹ אֲדֹנָיו֙ אֶל-הָ֣אֱלֹהִ֔ים וְהִגִּישׁוֹ֙ אֶל-הַדֶּ֔לֶת א֖וֹ אֶל-הַמְּזוּזָ֑ה וְרָצַ֨ע אֲדֹנָ֤יו אֶת-אָזְנוֹ֙ בַּמַּרְצֵ֔עַ וַעֲבָד֖וֹ לְעֹלָֽם:  ת  וִיקָרְבִנֵהּ רִבּוֹנֵהּ לִקֳדָם דַיָנַיָא וִיקָרְבִנֵהּ לְוָת דָשָׁא אוֹ לְוָת (נ''א דִילְוָת) מְזְזֶתָּא וְיַרְצַע רִבּוֹנֵהּ יָת אוּדְנֵהּ בְּמַרְצְעָא וִיהֵי לֵהּ עֲבַד פָּלַח לְעָלָם: (ס)

{ז} וְכִֽי-יִמְכֹּ֥ר אִ֛ישׁ אֶת-בִּתּ֖וֹ לְאָמָ֑ה לֹ֥א תֵצֵ֖א כְּצֵ֥את הָעֲבָדִֽים:  מ  וְכִֽי-יִמְכֹּ֥ר אִ֛ישׁ אֶת-בִּתּ֖וֹ לְאָמָ֑ה לֹ֥א תֵצֵ֖א כְּצֵ֥את הָעֲבָדִֽים:  ת  וַאֲרֵי יְזַבֵּן גְבַר יָת בְּרַתֵּהּ לְאַמְהוּ לָא תִפּוֹק כְּמַפְקָנוּת עַבְדַיָא:

{ח} אִם-רָעָ֞ה בְּעֵינֵ֧י אֲדֹנֶ֛יהָ אֲשֶׁר-(לא) ל֥וֹ יְעָדָ֖הּ וְהֶפְדָּ֑הּ לְעַ֥ם נָכְרִ֛י לֹא-יִמְשֹׁ֥ל לְמָכְרָ֖הּ בְּבִגְדוֹ-בָֽהּ:  מ  אִם-רָעָ֞ה בְּעֵינֵ֧י אֲדֹנֶ֛יהָ אֲשֶׁר-(לא) ל֥וֹ יְעָדָ֖הּ וְהֶפְדָּ֑הּ לְעַ֥ם נָכְרִ֛י לֹא-יִמְשֹׁ֥ל לְמָכְרָ֖הּ בְּבִגְדוֹ-בָֽהּ:  ת  אִם בִּישָׁא בְּעֵינֵי רִבּוֹנַהּ דִיקַיְמַהּ לֵהּ וְיִפְרְקִנַהּ לִגְבַר אָחֳרָן לֵית לֵהּ רְשׁוּ לְזַבּוֹנַהּ בְּמִשְׁלְטֵהּ בַהּ:

{ט} וְאִם-לִבְנ֖וֹ יִֽיעָדֶ֑נָּה כְּמִשְׁפַּ֥ט הַבָּנ֖וֹת יַעֲשֶׂה-לָּֽהּ:  מ  וְאִם-לִבְנ֖וֹ יִֽיעָדֶ֑נָּה כְּמִשְׁפַּ֥ט הַבָּנ֖וֹת יַעֲשֶׂה-לָּֽהּ:  ת  וְאִם לִבְרֵהּ יְקַיְמִנַהּ כְּהִלְכַת בְּנַת יִשְׂרָאֵל יַעְבֵּד לַהּ:

{י} אִם-אַחֶ֖רֶת יִֽקַּֽח-ל֑וֹ שְׁאֵרָ֛הּ כְּסוּתָ֥הּ וְעֹנָתָ֖הּ לֹ֥א יִגְרָֽע:  מ  אִם-אַחֶ֖רֶת יִֽקַּֽח-ל֑וֹ שְׁאֵרָ֛הּ כְּסוּתָ֥הּ וְעֹנָתָ֖הּ לֹ֥א יִגְרָֽע:  ת  אִם אָחֳרַנְתָּא יִסַב לֵהּ זִיוּנַהּ כְּסוּתַהּ וְעוֹנָתַהּ לָא יִתְמְנָע:

{יא} וְאִם-שְׁלָ֨שׁ-אֵ֔לֶּה לֹ֥א יַעֲשֶׂ֖ה לָ֑הּ וְיָצְאָ֥ה חִנָּ֖ם אֵ֥ין כָּֽסֶף:  מ  וְאִם-שְׁלָ֨שׁ-אֵ֔לֶּה לֹ֥א יַעֲשֶׂ֖ה לָ֑הּ וְיָצְאָ֥ה חִנָּ֖ם אֵ֥ין כָּֽסֶף:  ת  וְאִם תְּלָת אִלֵין לָא יַעְבֵּד לַהּ וְתִפּוֹק מַגָן בְּלָא כְסַף: (ס)

{יב} מַכֵּ֥ה אִ֛ישׁ וָמֵ֖ת מ֥וֹת יוּמָֽת:  מ  מַכֵּ֥ה אִ֛ישׁ וָמֵ֖ת מ֥וֹת יוּמָֽת:  ת  דְיִמְחֵי לֶאֱנַשׁ וְיִקְטְלִנֵהּ אִתְקְטָלָא יִתְקְטֵל:

{יג} וַאֲשֶׁר֙ לֹ֣א צָדָ֔ה וְהָאֱלֹהִ֖ים אִנָּ֣ה לְיָד֑וֹ וְשַׂמְתִּ֤י לְךָ֙ מָק֔וֹם אֲשֶׁ֥ר יָנ֖וּס שָֽׁמָּה:  מ  וַאֲשֶׁר֙ לֹ֣א צָדָ֔ה וְהָאֱלֹהִ֖ים אִנָּ֣ה לְיָד֑וֹ וְשַׂמְתִּ֤י לְךָ֙ מָק֔וֹם אֲשֶׁ֥ר יָנ֖וּס שָֽׁמָּה:  ת  וּדְלָא כְמַן לֵהּ וּמִן קֳדָם יְיָ אִתִּמְסַר לִידֵהּ וֶאֱשַׁוִי לָךְ אֲתַר דִי יֵעִרוֹק לְתַמָן: (ס)

{יד} וְכִי-יָזִ֥ד אִ֛ישׁ עַל-רֵעֵ֖הוּ לְהָרְג֣וֹ בְעָרְמָ֑ה מֵעִ֣ם מִזְבְּחִ֔י תִּקָּחֶ֖נּוּ לָמֽוּת:  מ  וְכִי-יָזִ֥ד אִ֛ישׁ עַל-רֵעֵ֖הוּ לְהָרְג֣וֹ בְעָרְמָ֑ה מֵעִ֣ם מִזְבְּחִ֔י תִּקָּחֶ֖נּוּ לָמֽוּת:  ת  וַאֲרֵי יַרְשַׁע גְבַר עַל חַבְרֵהּ לְמִקְטְלֵהּ בִנְכִילוּ מִן מַדְבְּחִי תִּדְבְּרִנֵהּ לִמְמָת: (ס)

{טו} וּמַכֵּ֥ה אָבִ֛יו וְאִמּ֖וֹ מ֥וֹת יוּמָֽת:  מ  וּמַכֵּ֥ה אָבִ֛יו וְאִמּ֖וֹ מ֥וֹת יוּמָֽת:  ת  וּדְיִמְחֵי אֲבוּהִי וְאִמֵהּ אִתְקְטָלָא יִתְקְטֵל: (ס)

{טז} וְגֹנֵ֨ב אִ֧ישׁ וּמְכָר֛וֹ וְנִמְצָ֥א בְיָד֖וֹ מ֥וֹת יוּמָֽת:  מ  וְגֹנֵ֨ב אִ֧ישׁ וּמְכָר֛וֹ וְנִמְצָ֥א בְיָד֖וֹ מ֥וֹת יוּמָֽת:  ת  וְדִיִגְנוּב נַפְשָׁא מִן בְּנֵי יִשְׂרָאֵל וִיזַבְּנִנֵהּ וְיִשְׁתְּכַח בִּידֵהּ אִתְקְטָלָא יִתְקְטֵל: (ס)

{יז} וּמְקַלֵּ֥ל אָבִ֛יו וְאִמּ֖וֹ מ֥וֹת יוּמָֽת:  מ  וּמְקַלֵּ֥ל אָבִ֛יו וְאִמּ֖וֹ מ֥וֹת יוּמָֽת:  ת  וְדִיְלוּט אֲבוּהִי וְאִמֵהּ אִתְקְטָלָא יִתְקְטֵל: (ס)

{יח} וְכִֽי-יְרִיבֻ֣ן אֲנָשִׁ֔ים וְהִכָּה-אִישׁ֙ אֶת-רֵעֵ֔הוּ בְּאֶ֖בֶן א֣וֹ בְאֶגְרֹ֑ף וְלֹ֥א יָמ֖וּת וְנָפַ֥ל לְמִשְׁכָּֽב:  מ  וְכִֽי-יְרִיבֻ֣ן אֲנָשִׁ֔ים וְהִכָּה-אִישׁ֙ אֶת-רֵעֵ֔הוּ בְּאֶ֖בֶן א֣וֹ בְאֶגְרֹ֑ף וְלֹ֥א יָמ֖וּת וְנָפַ֥ל לְמִשְׁכָּֽב:  ת  וַאֲרֵי יִנְצוּן גֻבְרִין וְיִמְחֵי גְבַר יָת חַבְרֵהּ בְּאַבְנָא אוֹ בְכוּרְמֵיזָא וְלָא יְמוּת וְיִפּוֹל לְבוּטְלָן:

{יט} אִם-יָק֞וּם וְהִתְהַלֵּ֥ךְ בַּח֛וּץ עַל-מִשְׁעַנְתּ֖וֹ וְנִקָּ֣ה הַמַּכֶּ֑ה רַ֥ק שִׁבְתּ֛וֹ יִתֵּ֖ן וְרַפֹּ֥א יְרַפֵּֽא:  מ  אִם-יָק֞וּם וְהִתְהַלֵּ֥ךְ בַּח֛וּץ עַל-מִשְׁעַנְתּ֖וֹ וְנִקָּ֣ה הַמַּכֶּ֑ה רַ֥ק שִׁבְתּ֛וֹ יִתֵּ֖ן וְרַפֹּ֥א יְרַפֵּֽא:  ת  אִם יְקוּם וִיהַלִיךְ בְּבָרָא עַל בּוּרְיֵהּ וִיהֵי זַכָּאָה מָחַיָא לְחוֹד בּוּטְּלָנֵהּ יִתֵּן וַאֲגַר אַסְיָא יְשַׁלֵם: (ס)

{כ}  שני  וְכִֽי-יַכֶּה֩ אִ֨ישׁ אֶת-עַבְדּ֜וֹ א֤וֹ אֶת-אֲמָתוֹ֙ בַּשֵּׁ֔בֶט וּמֵ֖ת תַּ֣חַת יָד֑וֹ נָקֹ֖ם יִנָּקֵֽם:  מ  וְכִֽי-יַכֶּה֩ אִ֨ישׁ אֶת-עַבְדּ֜וֹ א֤וֹ אֶת-אֲמָתוֹ֙ בַּשֵּׁ֔בֶט וּמֵ֖ת תַּ֣חַת יָד֑וֹ נָקֹ֖ם יִנָּקֵֽם:  ת  וַאֲרֵי יִמְחֵי גְבַר יָת עַבְדֵהּ אוֹ יָת אַמְתֵהּ בְּשָׁלְטָּן וִימוּת תְּחוֹת יְדֵהּ אִתְדָנָא יִתְּדָן:

{כא} אַ֥ךְ אִם-י֛וֹם א֥וֹ יוֹמַ֖יִם יַעֲמֹ֑ד לֹ֣א יֻקַּ֔ם כִּ֥י כַסְפּ֖וֹ הֽוּא:  מ  אַ֥ךְ אִם-י֛וֹם א֥וֹ יוֹמַ֖יִם יַעֲמֹ֑ד לֹ֣א יֻקַּ֔ם כִּ֥י כַסְפּ֖וֹ הֽוּא:  ת  בְּרַם אִם יוֹמָא אוֹ תְרֵין יוֹמִין יִתְקַיָם לָא יִתְּדָן אֲרֵי כַסְפֵּהּ הוּא: (ס)

{כב} וְכִֽי-יִנָּצ֣וּ אֲנָשִׁ֗ים וְנָ֨גְפ֜וּ אִשָּׁ֤ה הָרָה֙ וְיָצְא֣וּ יְלָדֶ֔יהָ וְלֹ֥א יִהְיֶ֖ה אָס֑וֹן עָנ֣וֹשׁ יֵעָנֵ֗שׁ כַּֽאֲשֶׁ֨ר יָשִׁ֤ית עָלָיו֙ בַּ֣עַל הָֽאִשָּׁ֔ה וְנָתַ֖ן בִּפְלִלִֽים:  מ  וְכִֽי-יִנָּצ֣וּ אֲנָשִׁ֗ים וְנָ֨גְפ֜וּ אִשָּׁ֤ה הָרָה֙ וְיָצְא֣וּ יְלָדֶ֔יהָ וְלֹ֥א יִהְיֶ֖ה אָס֑וֹן עָנ֣וֹשׁ יֵעָנֵ֗שׁ כַּֽאֲשֶׁ֨ר יָשִׁ֤ית עָלָיו֙ בַּ֣עַל הָֽאִשָּׁ֔ה וְנָתַ֖ן בִּפְלִלִֽים:  ת  וַאֲרֵי יִנְצוּן גֻבְרִין וְיִמְחוּן אִתְּתָא מַעַדְיָא וְיִפְּקוּן וַלְדָהָא וְלָא יְהֵי מוֹתָא אִתְגְבָאָה יִתְגְבֵי כְּמָא דִי יְשַׁוִי עֲלוֹהִי בַּעֲלַהּ דְאִתְּתָא וְיִתֵּן עַל מֵימַר דַיָנַיָא:

{כג} וְאִם-אָס֖וֹן יִהְיֶ֑ה וְנָתַתָּ֥ה נֶ֖פֶשׁ תַּ֥חַת נָֽפֶשׁ:  מ  וְאִם-אָס֖וֹן יִהְיֶ֑ה וְנָתַתָּ֥ה נֶ֖פֶשׁ תַּ֥חַת נָֽפֶשׁ:  ת  וְאִם מוֹתָא יְהֵא וְתִתֵּן נַפְשָׁא חֳלָף נַפְשָׁא:

{כד} עַ֚יִן תַּ֣חַת עַ֔יִן שֵׁ֖ן תַּ֣חַת שֵׁ֑ן יָ֚ד תַּ֣חַת יָ֔ד רֶ֖גֶל תַּ֥חַת רָֽגֶל:  מ  עַ֚יִן תַּ֣חַת עַ֔יִן שֵׁ֖ן תַּ֣חַת שֵׁ֑ן יָ֚ד תַּ֣חַת יָ֔ד רֶ֖גֶל תַּ֥חַת רָֽגֶל:  ת  עֵינָא חֳלָף עֵינָא שִׁנָא חֳלָף שִׁנָא יְדָא חֳלָף יְדָא רַגְלָא חֳלָף רַגְלָא:

{כה} כְּוִיָּה֙ תַּ֣חַת כְּוִיָּ֔ה פֶּ֖צַע תַּ֣חַת פָּ֑צַע חַבּוּרָ֕ה תַּ֖חַת חַבּוּרָֽה:  מ  כְּוִיָּה֙ תַּ֣חַת כְּוִיָּ֔ה פֶּ֖צַע תַּ֣חַת פָּ֑צַע חַבּוּרָ֕ה תַּ֖חַת חַבּוּרָֽה:  ת  כְּוָאָה חֳלָף כְּוָאָה פִּדְעָא חֳלָף פִּדְעָא מַשְׁקוֹפֵי חֳלָף מַשְׁקוֹפֵי: (ס)

{כו} וְכִֽי-יַכֶּ֨ה אִ֜ישׁ אֶת-עֵ֥ין עַבְדּ֛וֹ אֽוֹ-אֶת-עֵ֥ין אֲמָת֖וֹ וְשִֽׁחֲתָ֑הּ לַֽחָפְשִׁ֥י יְשַׁלְּחֶ֖נּוּ תַּ֥חַת עֵינֽוֹ:  מ  וְכִֽי-יַכֶּ֨ה אִ֜ישׁ אֶת-עֵ֥ין עַבְדּ֛וֹ אֽוֹ-אֶת-עֵ֥ין אֲמָת֖וֹ וְשִֽׁחֲתָ֑הּ לַֽחָפְשִׁ֥י יְשַׁלְּחֶ֖נּוּ תַּ֥חַת עֵינֽוֹ:  ת  וַאֲרֵי יִמְחֵי גְבַר יָת עֵינָא דְעַבְדֵהּ אוֹ יָת עֵינָא דְאַמְתֵהּ וִיחַבְּלִנַהּ לְבַר חוֹרִין יִפְטְּרִנֵהּ חֳלָף עֵינֵהּ:

{כז} וְאִם-שֵׁ֥ן עַבְדּ֛וֹ אֽוֹ-שֵׁ֥ן אֲמָת֖וֹ יַפִּ֑יל לַֽחָפְשִׁ֥י יְשַׁלְּחֶ֖נּוּ תַּ֥חַת שִׁנּֽוֹ:  מ  וְאִם-שֵׁ֥ן עַבְדּ֛וֹ אֽוֹ-שֵׁ֥ן אֲמָת֖וֹ יַפִּ֑יל לַֽחָפְשִׁ֥י יְשַׁלְּחֶ֖נּוּ תַּ֥חַת שִׁנּֽוֹ:  ת  וְאִם שִׁנָא דְעַבְדֵהּ אוֹ שִּׁנָא דְאַמְתֵהּ יַפֵּל לְבַר חוֹרִין יִפְטְרִנֵהּ חֳלָף שִׁנֵהּ: (ס)

{כח} וְכִֽי-יִגַּ֨ח שׁ֥וֹר אֶת-אִ֛ישׁ א֥וֹ אֶת-אִשָּׁ֖ה וָמֵ֑ת סָק֨וֹל יִסָּקֵ֜ל הַשּׁ֗וֹר וְלֹ֤א יֵאָכֵל֙ אֶת-בְּשָׂר֔וֹ וּבַ֥עַל הַשּׁ֖וֹר נָקִֽי:  מ  וְכִֽי-יִגַּ֨ח שׁ֥וֹר אֶת-אִ֛ישׁ א֥וֹ אֶת-אִשָּׁ֖ה וָמֵ֑ת סָק֨וֹל יִסָּקֵ֜ל הַשּׁ֗וֹר וְלֹ֤א יֵאָכֵל֙ אֶת-בְּשָׂר֔וֹ וּבַ֥עַל הַשּׁ֖וֹר נָקִֽי:  ת  וַאֲרֵי יַגַח תּוֹרָא יָת גַבְרָא אוֹ יָת אִתְּתָא וִימוּת אִתְרְגְמָא יִתְרְגֵם תּוֹרָא וְלָא יִתְאָכֵל יָת בִּסְרֵהּ וּמָרֵהּ דְתוֹרָא יְהֵי זַכָּאָה:

{כט} וְאִ֡ם שׁוֹר֩ נַגָּ֨ח ה֜וּא מִתְּמֹ֣ל שִׁלְשֹׁ֗ם וְהוּעַ֤ד בִּבְעָלָיו֙ וְלֹ֣א יִשְׁמְרֶ֔נּוּ וְהֵמִ֥ית אִ֖ישׁ א֣וֹ אִשָּׁ֑ה הַשּׁוֹר֙ יִסָּקֵ֔ל וְגַם-בְּעָלָ֖יו יוּמָֽת:  מ  וְאִ֡ם שׁוֹר֩ נַגָּ֨ח ה֜וּא מִתְּמֹ֣ל שִׁלְשֹׁ֗ם וְהוּעַ֤ד בִּבְעָלָיו֙ וְלֹ֣א יִשְׁמְרֶ֔נּוּ וְהֵמִ֥ית אִ֖ישׁ א֣וֹ אִשָּׁ֑ה הַשּׁוֹר֙ יִסָּקֵ֔ל וְגַם-בְּעָלָ֖יו יוּמָֽת:  ת  וְאִם תּוֹר נַגַח הוּא מֵאִתְּמַלֵי וּמִדְקַמוֹהִי וְיִתַּסְהֵד בְּמָרֵהּ וְלָא נַטְרֵהּ וְיִקְטוֹל גַבְרָא אוֹ אִתְּתָא תּוֹרָא יִתְרְגֵם וְאַף מָרֵהּ יִתְקְטֵל:

{ל} אִם-כֹּ֖פֶר יוּשַׁ֣ת עָלָ֑יו וְנָתַן֙ פִּדְיֹ֣ן נַפְשׁ֔וֹ כְּכֹ֥ל אֲשֶׁר-יוּשַׁ֖ת עָלָֽיו:  מ  אִם-כֹּ֖פֶר יוּשַׁ֣ת עָלָ֑יו וְנָתַן֙ פִּדְיֹ֣ן נַפְשׁ֔וֹ כְּכֹ֥ל אֲשֶׁר-יוּשַׁ֖ת עָלָֽיו:  ת  אִם מָמוֹן יְשַׁווּן עֲלוֹהִי וְיִתֵּן פּוּרְקַן נַפְשֵׁהּ כְּכֹל דִי יְשַׁווּן עֲלוֹהִי:

{לא} אוֹ-בֵ֥ן יִגָּ֖ח אוֹ-בַ֣ת יִגָּ֑ח כַּמִּשְׁפָּ֥ט הַזֶּ֖ה יֵעָ֥שֶׂה לּֽוֹ:  מ  אוֹ-בֵ֥ן יִגָּ֖ח אוֹ-בַ֣ת יִגָּ֑ח כַּמִּשְׁפָּ֥ט הַזֶּ֖ה יֵעָ֥שֶׂה לּֽוֹ:  ת  אוֹ לְבַר יִשְׂרָאֵל יַגַח תּוֹרָא אוֹ לְבַת יִשְׂרָאֵל יַגָח כְּדִינָא הָדֵין יִתְעֲבֵד לֵהּ:

{לב} אִם-עֶ֛בֶד יִגַּ֥ח הַשּׁ֖וֹר א֣וֹ אָמָ֑ה כֶּ֣סֶף | שְׁלֹשִׁ֣ים שְׁקָלִ֗ים יִתֵּן֙ לַֽאדֹנָ֔יו וְהַשּׁ֖וֹר יִסָּקֵֽל:  מ  אִם-עֶ֛בֶד יִגַּ֥ח הַשּׁ֖וֹר א֣וֹ אָמָ֑ה כֶּ֣סֶף | שְׁלֹשִׁ֣ים שְׁקָלִ֗ים יִתֵּן֙ לַֽאדֹנָ֔יו וְהַשּׁ֖וֹר יִסָּקֵֽל:  ת  אִם לְעַבְדָא יַגַח תּוֹרָא אוֹ לְאַמְתָא כְּסַף תְּלָתִין סִלְעִין יִתֵּן לְרִבּוֹנֵהּ וְתוֹרָא יִתְרְגֵם: (ס)

{לג} וְכִֽי-יִפְתַּ֨ח אִ֜ישׁ בּ֗וֹר א֠וֹ כִּֽי-יִכְרֶ֥ה אִ֛ישׁ בֹּ֖ר וְלֹ֣א יְכַסֶּ֑נּוּ וְנָֽפַל-שָׁ֥מָּה שּׁ֖וֹר א֥וֹ חֲמֽוֹר:  מ  וְכִֽי-יִפְתַּ֨ח אִ֜ישׁ בּ֗וֹר א֠וֹ כִּֽי-יִכְרֶ֥ה אִ֛ישׁ בֹּ֖ר וְלֹ֣א יְכַסֶּ֑נּוּ וְנָֽפַל-שָׁ֥מָּה שּׁ֖וֹר א֥וֹ חֲמֽוֹר:  ת  וַאֲרֵי יִפְתַּח גְבַר גוֹב אוֹ אֲרֵי יִכְרֶה גְבַר גוֹב וְלָא יְכַסִנֵהּ וְיִפֵּל תַּמָן תּוֹרָא אוֹ חֲמָרָא:

{לד} בַּ֤עַל הַבּוֹר֙ יְשַׁלֵּ֔ם כֶּ֖סֶף יָשִׁ֣יב לִבְעָלָ֑יו וְהַמֵּ֖ת יִֽהְיֶה-לּֽוֹ:  מ  בַּ֤עַל הַבּוֹר֙ יְשַׁלֵּ֔ם כֶּ֖סֶף יָשִׁ֣יב לִבְעָלָ֑יו וְהַמֵּ֖ת יִֽהְיֶה-לּֽוֹ:  ת  מָרֵי דְגֻבָּא יְשַׁלֵם כַּסְפָּא יָתִיב לְמָרוֹהִי וּמֵיתָא יְהֵי דִילֵהּ: (ס)

{לה} וְכִֽי-יִגֹּ֧ף שֽׁוֹר-אִ֛ישׁ אֶת-שׁ֥וֹר רֵעֵ֖הוּ וָמֵ֑ת וּמָ֨כְר֜וּ אֶת-הַשּׁ֤וֹר הַחַי֙ וְחָצ֣וּ אֶת-כַּסְפּ֔וֹ וְגַ֥ם אֶת-הַמֵּ֖ת יֶֽחֱצֽוּן:  מ  וְכִֽי-יִגֹּ֧ף שֽׁוֹר-אִ֛ישׁ אֶת-שׁ֥וֹר רֵעֵ֖הוּ וָמֵ֑ת וּמָ֨כְר֜וּ אֶת-הַשּׁ֤וֹר הַחַי֙ וְחָצ֣וּ אֶת-כַּסְפּ֔וֹ וְגַ֥ם אֶת-הַמֵּ֖ת יֶֽחֱצֽוּן:  ת  וַאֲרֵי יִגוֹף תּוֹר דִגְבַר יָת תּוֹרָא דְחַבְרֵהּ וִימוּת וִזַבְּנוּן יָת תּוֹרָא חַיָא וִיפַלְגוּן יָת כַּסְפֵּהּ וְאַף יָת דְמֵי מֵיתָא יְפַלְגוּן:

{לו} א֣וֹ נוֹדַ֗ע כִּ֠י שׁ֣וֹר נַגָּ֥ח הוּא֙ מִתְּמ֣וֹל שִׁלְשֹׁ֔ם וְלֹ֥א יִשְׁמְרֶ֖נּוּ בְּעָלָ֑יו שַׁלֵּ֨ם יְשַׁלֵּ֥ם שׁוֹר֙ תַּ֣חַת הַשּׁ֔וֹר וְהַמֵּ֖ת יִֽהְיֶה-לּֽוֹ:  מ  א֣וֹ נוֹדַ֗ע כִּ֠י שׁ֣וֹר נַגָּ֥ח הוּא֙ מִתְּמ֣וֹל שִׁלְשֹׁ֔ם וְלֹ֥א יִשְׁמְרֶ֖נּוּ בְּעָלָ֑יו שַׁלֵּ֨ם יְשַׁלֵּ֥ם שׁוֹר֙ תַּ֣חַת הַשּׁ֔וֹר וְהַמֵּ֖ת יִֽהְיֶה-לּֽוֹ:  ת  אוֹ אִתְיְדַע אֲרֵי תוֹר נַגַח הוּא מֵאִתְּמַלֵי וּמִדְקַמוֹהִי וְלָא נָטְרֵהּ מָרֵהּ שַׁלָמָא יְשַׁלֵם תּוֹרָא חֲלַף תּוֹרָא וּמֵיתָא יְהֵא דִילֵהּ: (ס)

{לז} כִּ֤י יִגְנֹֽב-אִישׁ֙ שׁ֣וֹר אוֹ-שֶׂ֔ה וּטְבָח֖וֹ א֣וֹ מְכָר֑וֹ חֲמִשָּׁ֣ה בָקָ֗ר יְשַׁלֵּם֙ תַּ֣חַת הַשּׁ֔וֹר וְאַרְבַּע-צֹ֖אן תַּ֥חַת הַשֶּֽׂה:  מ  כִּ֤י יִגְנֹֽב-אִישׁ֙ שׁ֣וֹר אוֹ-שֶׂ֔ה וּטְבָח֖וֹ א֣וֹ מְכָר֑וֹ חֲמִשָּׁ֣ה בָקָ֗ר יְשַׁלֵּם֙ תַּ֣חַת הַשּׁ֔וֹר וְאַרְבַּע-צֹ֖אן תַּ֥חַת הַשֶּֽׂה:  ת  אֲרֵי יִגְנוּב גְבַר תּוֹר אוֹ אִמָר וְיִכְּסִנֵהּ אוֹ זַבְּנִנֵהּ חַמְשָׁא תוֹרִין יְשַׁלֵם חָלָף תּוֹרָא וְאַרְבַּע עָנָא חֳלָף אִמְרָא:




שמות פרק-כב

{א} אִם-בַּמַּחְתֶּ֛רֶת יִמָּצֵ֥א הַגַּנָּ֖ב וְהֻכָּ֣ה וָמֵ֑ת אֵ֥ין ל֖וֹ דָּמִֽים:  מ  אִם-בַּמַּחְתֶּ֛רֶת יִמָּצֵ֥א הַגַּנָּ֖ב וְהֻכָּ֣ה וָמֵ֑ת אֵ֥ין ל֖וֹ דָּמִֽים:  ת  אִם בְּמַחְתַּרְתָּא יִשְׁתְּכַח גַנָבָא וְיִתְמְחֵי וִימוּת לֵית לֵהּ דָם:

{ב} אִם-זָרְחָ֥ה הַשֶּׁ֛מֶשׁ עָלָ֖יו דָּמִ֣ים ל֑וֹ שַׁלֵּ֣ם יְשַׁלֵּ֔ם אִם-אֵ֣ין ל֔וֹ וְנִמְכַּ֖ר בִּגְנֵבָתֽוֹ:  מ  אִם-זָרְחָ֥ה הַשֶּׁ֛מֶשׁ עָלָ֖יו דָּמִ֣ים ל֑וֹ שַׁלֵּ֣ם יְשַׁלֵּ֔ם אִם-אֵ֣ין ל֔וֹ וְנִמְכַּ֖ר בִּגְנֵבָתֽוֹ:  ת  אִם עֵינָא דְסַהֲדַיָא נְפָלַת עֲלוֹהִי דְמָא לֵהּ שַׁלָמָא יְשַׁלֵם אִם לֵית לֵהּ וְיִזְדַבַּן בִּגְנֻבְתֵּהּ:

{ג} אִֽם-הִמָּצֵא֩ תִמָּצֵ֨א בְיָד֜וֹ הַגְּנֵבָ֗ה מִשּׁ֧וֹר עַד-חֲמ֛וֹר עַד-שֶׂ֖ה חַיִּ֑ים שְׁנַ֖יִם יְשַׁלֵּֽם:  מ  אִֽם-הִמָּצֵא֩ תִמָּצֵ֨א בְיָד֜וֹ הַגְּנֵבָ֗ה מִשּׁ֧וֹר עַד-חֲמ֛וֹר עַד-שֶׂ֖ה חַיִּ֑ים שְׁנַ֖יִם יְשַׁלֵּֽם:  ת  אִם אִשְׁתְּכָחָא יִשְׁתְּכַח בִּידֵהּ גְנֻבְתָּא מִתּוֹר עַד חֲמַר עַד אִמַר דְאִנוּן חַיִין עַל חַד תְּרֵין יְשַׁלֵם: (ס)

{ד}  שלישי  כִּ֤י יַבְעֶר-אִישׁ֙ שָׂדֶ֣ה אוֹ-כֶ֔רֶם וְשִׁלַּח֙ אֶת-(בעירה) בְּעִיר֔וֹ וּבִעֵ֖ר בִּשְׂדֵ֣ה אַחֵ֑ר מֵיטַ֥ב שָׂדֵ֛הוּ וּמֵיטַ֥ב כַּרְמ֖וֹ יְשַׁלֵּֽם:  מ  כִּ֤י יַבְעֶר-אִישׁ֙ שָׂדֶ֣ה אוֹ-כֶ֔רֶם וְשִׁלַּח֙ אֶת-(בעירה) בְּעִיר֔וֹ וּבִעֵ֖ר בִּשְׂדֵ֣ה אַחֵ֑ר מֵיטַ֥ב שָׂדֵ֛הוּ וּמֵיטַ֥ב כַּרְמ֖וֹ יְשַׁלֵּֽם:  ת  אֲרֵי יאכִיל גְבַר חֲקַל אוֹ כְרָם וִישַׁלַח יָת בְּעִירֵהּ וְיֵיכוּל בַּחֲקַל אָחֳרָן שְׁפַר חַקְלֵהּ וּשְׁפַר כַּרְמֵהּ יְשַׁלֵם: (ס)

{ה} כִּֽי-תֵצֵ֨א אֵ֜שׁ וּמָצְאָ֤ה קֹצִים֙ וְנֶאֱכַ֣ל גָּדִ֔ישׁ א֥וֹ הַקָּמָ֖ה א֣וֹ הַשָּׂדֶ֑ה שַׁלֵּ֣ם יְשַׁלֵּ֔ם הַמַּבְעִ֖ר אֶת-הַבְּעֵרָֽה:  מ  כִּֽי-תֵצֵ֨א אֵ֜שׁ וּמָצְאָ֤ה קֹצִים֙ וְנֶאֱכַ֣ל גָּדִ֔ישׁ א֥וֹ הַקָּמָ֖ה א֣וֹ הַשָּׂדֶ֑ה שַׁלֵּ֣ם יְשַׁלֵּ֔ם הַמַּבְעִ֖ר אֶת-הַבְּעֵרָֽה:  ת  אֲרֵי תִתַּפַּק נוּר וְיַשְׁכַּח כּוּבִין וְיֵיכוּל גְדִישִׁין אוֹ קֳמָא אוֹ חַקְלָא שַׁלָמָא יְשַׁלֵם דְאַדְלֵק יָת דְלֶקְתָּא: (ס)

{ו} כִּֽי-יִתֵּן֩ אִ֨ישׁ אֶל-רֵעֵ֜הוּ כֶּ֤סֶף אֽוֹ-כֵלִים֙ לִשְׁמֹ֔ר וְגֻנַּ֖ב מִבֵּ֣ית הָאִ֑ישׁ אִם-יִמָּצֵ֥א הַגַּנָּ֖ב יְשַׁלֵּ֥ם שְׁנָֽיִם:  מ  כִּֽי-יִתֵּן֩ אִ֨ישׁ אֶל-רֵעֵ֜הוּ כֶּ֤סֶף אֽוֹ-כֵלִים֙ לִשְׁמֹ֔ר וְגֻנַּ֖ב מִבֵּ֣ית הָאִ֑ישׁ אִם-יִמָּצֵ֥א הַגַּנָּ֖ב יְשַׁלֵּ֥ם שְׁנָֽיִם:  ת  אֲרֵי יִתֵּן גְבַר לְחַבְרֵהּ כְּסַף אוֹ מָנִין לְמִטַר וְיִתְגַנְבוּן מִבֵּית גַבְרָא אִם יִשְׁתְּכַח גַנָבָא יְשַׁלֵם עַל חַד תְּרֵין:

{ז} אִם-לֹ֤א יִמָּצֵא֙ הַגַּנָּ֔ב וְנִקְרַ֥ב בַּֽעַל-הַבַּ֖יִת אֶל-הָֽאֱלֹהִ֑ים אִם-לֹ֥א שָׁלַ֛ח יָד֖וֹ בִּמְלֶ֥אכֶת רֵעֵֽהוּ:  מ  אִם-לֹ֤א יִמָּצֵא֙ הַגַּנָּ֔ב וְנִקְרַ֥ב בַּֽעַל-הַבַּ֖יִת אֶל-הָֽאֱלֹהִ֑ים אִם-לֹ֥א שָׁלַ֛ח יָד֖וֹ בִּמְלֶ֥אכֶת רֵעֵֽהוּ:  ת  אִם לָא יִשְׁתְּכַח גַנָבָא וְיִתְקָרֵב מָרֵי דְבֵיתָא לִקֳדָם דַיָנַיָא אִם לָא אוֹשִׁיט יְדֵהּ בְּמָא דִמְסַר לֵהּ חַבְרֵהּ:

{ח} עַֽל-כָּל-דְּבַר-פֶּ֡שַׁע עַל-שׁ֡וֹר עַל-חֲ֠מוֹר עַל-שֶׂ֨ה עַל-שַׂלְמָ֜ה עַל-כָּל-אֲבֵדָ֗ה אֲשֶׁ֤ר יֹאמַר֙ כִּי-ה֣וּא זֶ֔ה עַ֚ד הָֽאֱלֹהִ֔ים יָבֹ֖א דְּבַר-שְׁנֵיהֶ֑ם אֲשֶׁ֤ר יַרְשִׁיעֻן֙ אֱלֹהִ֔ים יְשַׁלֵּ֥ם שְׁנַ֖יִם לְרֵעֵֽהוּ:  מ  עַֽל-כָּל-דְּבַר-פֶּ֡שַׁע עַל-שׁ֡וֹר עַל-חֲ֠מוֹר עַל-שֶׂ֨ה עַל-שַׂלְמָ֜ה עַל-כָּל-אֲבֵדָ֗ה אֲשֶׁ֤ר יֹאמַר֙ כִּי-ה֣וּא זֶ֔ה עַ֚ד הָֽאֱלֹהִ֔ים יָבֹ֖א דְּבַר-שְׁנֵיהֶ֑ם אֲשֶׁ֤ר יַרְשִׁיעֻן֙ אֱלֹהִ֔ים יְשַׁלֵּ֥ם שְׁנַ֖יִם לְרֵעֵֽהוּ:  ת  עַל כָּל פִּתְגַם דְחוֹב עַל תּוֹר עַל חֲמַר עַל אִמָר עַל כְּסוּ עַל כָּל אֲבֶדְתָּא דִי יֵימַר אֲרֵי הוּא דֵין לִקֳדָם דַיָנַיָא יֵיעוֹל דִין תַּרְוֵיהוֹן דִי יְחַיְבוּן דַיָנַיָא יְשַׁלֵם עַל חַד תְּרֵין לְחַבְרֵהּ: (ס)

{ט} כִּֽי-יִתֵּן֩ אִ֨ישׁ אֶל-רֵעֵ֜הוּ חֲמ֨וֹר אוֹ-שׁ֥וֹר אוֹ-שֶׂ֛ה וְכָל-בְּהֵמָ֖ה לִשְׁמֹ֑ר וּמֵ֛ת אוֹ-נִשְׁבַּ֥ר אוֹ-נִשְׁבָּ֖ה אֵ֥ין רֹאֶֽה:  מ  כִּֽי-יִתֵּן֩ אִ֨ישׁ אֶל-רֵעֵ֜הוּ חֲמ֨וֹר אוֹ-שׁ֥וֹר אוֹ-שֶׂ֛ה וְכָל-בְּהֵמָ֖ה לִשְׁמֹ֑ר וּמֵ֛ת אוֹ-נִשְׁבַּ֥ר אוֹ-נִשְׁבָּ֖ה אֵ֥ין רֹאֶֽה:  ת  אֲרֵי יִתֵּן גְבַר לְחַבְרֵהּ חֲמַר אוֹ תוֹר אוֹ אִמָר וְכָל בְּעִירָא לְמִטָּר וּמִית אוֹ אִתְּבַר אוֹ אִשְׁתְּבִי לֵית דְחָזֵי:

{י} שְׁבֻעַ֣ת יְהוָ֗ה תִּהְיֶה֙ בֵּ֣ין שְׁנֵיהֶ֔ם אִם-לֹ֥א שָׁלַ֛ח יָד֖וֹ בִּמְלֶ֣אכֶת רֵעֵ֑הוּ וְלָקַ֥ח בְּעָלָ֖יו וְלֹ֥א יְשַׁלֵּֽם:  מ  שְׁבֻעַ֣ת יְהוָ֗ה תִּהְיֶה֙ בֵּ֣ין שְׁנֵיהֶ֔ם אִם-לֹ֥א שָׁלַ֛ח יָד֖וֹ בִּמְלֶ֣אכֶת רֵעֵ֑הוּ וְלָקַ֥ח בְּעָלָ֖יו וְלֹ֥א יְשַׁלֵּֽם:  ת  מוֹמָתָא דַיְיָ תְּהֵי בֵּין תַּרְוֵיהוֹן אִם לָא אוֹשִׁיט יְדֵהּ בְּמָא דִמְסַר לֵהּ חַבְרֵהּ וִיקַבֵּל מָרֵהּ מִנֵהּ מוֹמָתָא וְלָא יְשַׁלֵם:

{יא} וְאִם-גָּנֹ֥ב יִגָּנֵ֖ב מֵעִמּ֑וֹ יְשַׁלֵּ֖ם לִבְעָלָֽיו:  מ  וְאִם-גָּנֹ֥ב יִגָּנֵ֖ב מֵעִמּ֑וֹ יְשַׁלֵּ֖ם לִבְעָלָֽיו:  ת  וְאִם אִתְגְנָבָא יִתְגְנֵב מֵעִמֵהּ יְשַׁלֵם לְמָרוֹהִי:

{יב} אִם-טָרֹ֥ף יִטָּרֵ֖ף יְבִאֵ֣הוּ עֵ֑ד הַטְּרֵפָ֖ה לֹ֥א יְשַׁלֵּֽם:  מ  אִם-טָרֹ֥ף יִטָּרֵ֖ף יְבִאֵ֣הוּ עֵ֑ד הַטְּרֵפָ֖ה לֹ֥א יְשַׁלֵּֽם:  ת  אִם אִתְּבָרָא יִתְּבַר יַיְתִי סַהֲדִין דִתְבִירָא לָא יְשַׁלֵם: (ס)

{יג} וְכִֽי-יִשְׁאַ֥ל אִ֛ישׁ מֵעִ֥ם רֵעֵ֖הוּ וְנִשְׁבַּ֣ר אוֹ-מֵ֑ת בְּעָלָ֥יו אֵין-עִמּ֖וֹ שַׁלֵּ֥ם יְשַׁלֵּֽם:  מ  וְכִֽי-יִשְׁאַ֥ל אִ֛ישׁ מֵעִ֥ם רֵעֵ֖הוּ וְנִשְׁבַּ֣ר אוֹ-מֵ֑ת בְּעָלָ֥יו אֵין-עִמּ֖וֹ שַׁלֵּ֥ם יְשַׁלֵּֽם:  ת  וַאֲרֵי יִשְׁאַל גְבַר מִן חַבְרֵהּ וְיִתְּבַר אוֹ מִית מָרֵהּ לֵית עִמֵהּ שַׁלָמָא יְשַׁלֵם:

{יד} אִם-בְּעָלָ֥יו עִמּ֖וֹ לֹ֣א יְשַׁלֵּ֑ם אִם-שָׂכִ֣יר ה֔וּא בָּ֖א בִּשְׂכָרֽוֹ:  מ  אִם-בְּעָלָ֥יו עִמּ֖וֹ לֹ֣א יְשַׁלֵּ֑ם אִם-שָׂכִ֣יר ה֔וּא בָּ֖א בִּשְׂכָרֽוֹ:  ת  אִם מָרֵהּ עִמֵהּ לָא יְשַׁלֵם אִם אֲגִירָא הוּא עַל בְּאַגְרֵהּ: (ס)

{טו} וְכִֽי-יְפַתֶּ֣ה אִ֗ישׁ בְּתוּלָ֛ה אֲשֶׁ֥ר לֹא-אֹרָ֖שָׂה וְשָׁכַ֣ב עִמָּ֑הּ מָהֹ֛ר יִמְהָרֶ֥נָּה לּ֖וֹ לְאִשָּֽׁה:  מ  וְכִֽי-יְפַתֶּ֣ה אִ֗ישׁ בְּתוּלָ֛ה אֲשֶׁ֥ר לֹא-אֹרָ֖שָׂה וְשָׁכַ֣ב עִמָּ֑הּ מָהֹ֛ר יִמְהָרֶ֥נָּה לּ֖וֹ לְאִשָּֽׁה:  ת  וַאֲרֵי יְשַׁדֵל גְבַר בְּתֻלְתָּא דִּי לָא מְאַרְסָא וְיִשְׁכּוּב עִמַהּ קַיָמָא יְקַיְמִנַהּ לֵהּ לְאִנְתּוּ:

{טז} אִם-מָאֵ֧ן יְמָאֵ֛ן אָבִ֖יהָ לְתִתָּ֣הּ ל֑וֹ כֶּ֣סֶף יִשְׁקֹ֔ל כְּמֹ֖הַר הַבְּתוּלֹֽת:  מ  אִם-מָאֵ֧ן יְמָאֵ֛ן אָבִ֖יהָ לְתִתָּ֣הּ ל֑וֹ כֶּ֣סֶף יִשְׁקֹ֔ל כְּמֹ֖הַר הַבְּתוּלֹֽת:  ת  אִם מִצְבָּא לָא יִצְבֵּי אֲבוּהָא לְמִתְּנַהּ לֵהּ כַּסְפָּא יִתְקוֹל כְּמוֹהֲרֵי בְּתֻלְתָּא: (ס)

{יז} מְכַשֵּׁפָ֖ה לֹ֥א תְחַיֶּֽה:  מ  מְכַשֵּׁפָ֖ה לֹ֥א תְחַיֶּֽה:  ת  חָרָשָׁא לָא תָחֵי: (ס)

{יח} כָּל-שֹׁכֵ֥ב עִם-בְּהֵמָ֖ה מ֥וֹת יוּמָֽת:  מ  כָּל-שֹׁכֵ֥ב עִם-בְּהֵמָ֖ה מ֥וֹת יוּמָֽת:  ת  כָּל דְיִשְׁכּוּב עִם בְּעִירָא אִתְקְטָלָא יִתְקְטֵל: (ס)

{יט} זֹבֵ֥חַ לָאֱלֹהִ֖ים יָֽחֳרָ֑ם בִּלְתִּ֥י לַיהוָ֖ה לְבַדּֽוֹ:  מ  זֹבֵ֥חַ לָאֱלֹהִ֖ים יָֽחֳרָ֑ם בִּלְתִּ֥י לַיהוָ֖ה לְבַדּֽוֹ:  ת  דְדַבַּח לְטַעְוַת עַמְמַיָא יִתְקְטֵל אֶלָהֵן לִשְׁמָא דַיְיָ בִּלְחוֹדוֹהִי:

{כ} וְגֵ֥ר לֹא-תוֹנֶ֖ה וְלֹ֣א תִלְחָצֶ֑נּוּ כִּֽי-גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֥רֶץ מִצְרָֽיִם:  מ  וְגֵ֥ר לֹא-תוֹנֶ֖ה וְלֹ֣א תִלְחָצֶ֑נּוּ כִּֽי-גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֥רֶץ מִצְרָֽיִם:  ת  וְגִיוֹרָא לָא תוֹנוּן וְלָא תְעִיקוּן לֵהּ אֲרֵי דַיָרִין הֱוֵיתוּן בְּאַרְעָא דְמִצְרָיִם:

{כא} כָּל-אַלְמָנָ֥ה וְיָת֖וֹם לֹ֥א תְעַנּֽוּן:  מ  כָּל-אַלְמָנָ֥ה וְיָת֖וֹם לֹ֥א תְעַנּֽוּן:  ת  כָּל אַרְמְלָא וְיִתָּם לָא תְעַנוּן:

{כב} אִם-עַנֵּ֥ה תְעַנֶּ֖ה אֹת֑וֹ כִּ֣י אִם-צָעֹ֤ק יִצְעַק֙ אֵלַ֔י שָׁמֹ֥עַ אֶשְׁמַ֖ע צַעֲקָתֽוֹ:  מ  אִם-עַנֵּ֥ה תְעַנֶּ֖ה אֹת֑וֹ כִּ֣י אִם-צָעֹ֤ק יִצְעַק֙ אֵלַ֔י שָׁמֹ֥עַ אֶשְׁמַ֖ע צַעֲקָתֽוֹ:  ת  אִם עַנָאָה תְעַנֵי יָתֵהּ אֲרֵי אִם מִקְבַל יִקְבֵל קֳדָמַי קַבָּלָא אֱקַבֵּל קְבִלְתֵּהּ:

{כג} וְחָרָ֣ה אַפִּ֔י וְהָרַגְתִּ֥י אֶתְכֶ֖ם בֶּחָ֑רֶב וְהָי֤וּ נְשֵׁיכֶם֙ אַלְמָנ֔וֹת וּבְנֵיכֶ֖ם יְתֹמִֽים:  מ  וְחָרָ֣ה אַפִּ֔י וְהָרַגְתִּ֥י אֶתְכֶ֖ם בֶּחָ֑רֶב וְהָי֤וּ נְשֵׁיכֶם֙ אַלְמָנ֔וֹת וּבְנֵיכֶ֖ם יְתֹמִֽים:  ת  וְיִתְקֵף רוּגְזִי וְאֶקְטוֹל יָתְכוֹן בְּחַרְבָּא וִיהֶוְיָן נְשֵׁיכוֹן אַרְמְלִין וּבְנֵיכוֹן יַתְמִין: (ס)

{כד} אִם-כֶּ֣סֶף | תַּלְוֶ֣ה אֶת-עַמִּ֗י אֶת-הֶֽעָנִי֙ עִמָּ֔ךְ לֹא-תִהְיֶ֥ה ל֖וֹ כְּנֹשֶׁ֑ה לֹֽא-תְשִׂימ֥וּן עָלָ֖יו נֶֽשֶׁךְ:  מ  אִם-כֶּ֣סֶף | תַּלְוֶ֣ה אֶת-עַמִּ֗י אֶת-הֶֽעָנִי֙ עִמָּ֔ךְ לֹא-תִהְיֶ֥ה ל֖וֹ כְּנֹשֶׁ֑ה לֹֽא-תְשִׂימ֥וּן עָלָ֖יו נֶֽשֶׁךְ:  ת  אִם כַּסְפָּא תּוֹזֵף בְּעַמִי לְּעַנְיָא דִי עִמָךְ לָא תְהֵי לֵהּ כְּרַשְׁיָא לָא תְשַׁוּוֹן עֲלוֹהִי חִבּוּלְיָא:

{כה} אִם-חָבֹ֥ל תַּחְבֹּ֖ל שַׂלְמַ֣ת רֵעֶ֑ךָ עַד-בֹּ֥א הַשֶּׁ֖מֶשׁ תְּשִׁיבֶ֥נּוּ לֽוֹ:  מ  אִם-חָבֹ֥ל תַּחְבֹּ֖ל שַׂלְמַ֣ת רֵעֶ֑ךָ עַד-בֹּ֥א הַשֶּׁ֖מֶשׁ תְּשִׁיבֶ֥נּוּ לֽוֹ:  ת  אִם מַשְׁכּוֹנָא תִסַב כְּסוּתָא דְחַבְרָךְ עַד מֵעַל שִׁמְשָׁא תְּתִיבִנֵהּ לֵהּ:

{כו} כִּ֣י הִ֤וא (כסותה) כְסוּתוֹ֙ לְבַדָּ֔הּ הִ֥וא שִׂמְלָת֖וֹ לְעֹר֑וֹ בַּמֶּ֣ה יִשְׁכָּ֔ב וְהָיָה֙ כִּֽי-יִצְעַ֣ק אֵלַ֔י וְשָׁמַעְתִּ֖י כִּֽי-חַנּ֥וּן אָֽנִי:  מ  כִּ֣י הִ֤וא (כסותה) כְסוּתוֹ֙ לְבַדָּ֔הּ הִ֥וא שִׂמְלָת֖וֹ לְעֹר֑וֹ בַּמֶּ֣ה יִשְׁכָּ֔ב וְהָיָה֙ כִּֽי-יִצְעַ֣ק אֵלַ֔י וְשָׁמַעְתִּ֖י כִּֽי-חַנּ֥וּן אָֽנִי:  ת  אֲרֵי הִיא כְסוּתֵהּ בִּלְחוֹדַהּ הִיא תוֹתְבֵהּ לְמַשְׁכֵּיהּ בְּמָא יִשְׁכּוּב וִיהֵי אֲרֵי יִקְבֵל קֳדָמַי וֶאֱקַבֵּל קְבִלְתֵּהּ אֲרֵי חַנָנָא אֲנָא: (ס)

{כז}  רביעי  אֱלֹהִ֖ים לֹ֣א תְקַלֵּ֑ל וְנָשִׂ֥יא בְעַמְּךָ֖ לֹ֥א תָאֹֽר:  מ  אֱלֹהִ֖ים לֹ֣א תְקַלֵּ֑ל וְנָשִׂ֥יא בְעַמְּךָ֖ לֹ֥א תָאֹֽר:  ת  דַיָנָא לָא תְקִיל וְרַבָּא בְעַמָךְ לָא תְלוּט:

{כח} מְלֵאָתְךָ֥ וְדִמְעֲךָ֖ לֹ֣א תְאַחֵ֑ר בְּכ֥וֹר בָּנֶ֖יךָ תִּתֶּן-לִּֽי:  מ  מְלֵאָתְךָ֥ וְדִמְעֲךָ֖ לֹ֣א תְאַחֵ֑ר בְּכ֥וֹר בָּנֶ֖יךָ תִּתֶּן-לִּֽי:  ת  בִּכּוּרָךְ וְדִמְעָךְ לָא תְאַחֵר בּוּכְרָא דִבְנָיךְ תַּפְרֵשׁ קֳדָמָי:

{כט} כֵּֽן-תַּעֲשֶׂ֥ה לְשֹׁרְךָ֖ לְצֹאנֶ֑ךָ שִׁבְעַ֤ת יָמִים֙ יִהְיֶ֣ה עִם-אִמּ֔וֹ בַּיּ֥וֹם הַשְּׁמִינִ֖י תִּתְּנוֹ-לִֽי:  מ  כֵּֽן-תַּעֲשֶׂ֥ה לְשֹׁרְךָ֖ לְצֹאנֶ֑ךָ שִׁבְעַ֤ת יָמִים֙ יִהְיֶ֣ה עִם-אִמּ֔וֹ בַּיּ֥וֹם הַשְּׁמִינִ֖י תִּתְּנוֹ-לִֽי:  ת  כֵּן תַּעְבֵּד לְתוֹרָךְ לְעָנָךְ שִׁבְעָא יוֹמִין יְהֵי עִם אִמֵהּ בְּיוֹמָא תְמִינָאָה תִּפְרְשִׁנֵהּ קֳדָמָי:

{ל} וְאַנְשֵׁי-קֹ֖דֶשׁ תִּהְי֣וּן לִ֑י וּבָשָׂ֨ר בַּשָּׂדֶ֤ה טְרֵפָה֙ לֹ֣א תֹאכֵ֔לוּ לַכֶּ֖לֶב תַּשְׁלִכ֥וּן אֹתֽוֹ:  מ  וְאַנְשֵׁי-קֹ֖דֶשׁ תִּהְי֣וּן לִ֑י וּבָשָׂ֨ר בַּשָּׂדֶ֤ה טְרֵפָה֙ לֹ֣א תֹאכֵ֔לוּ לַכֶּ֖לֶב תַּשְׁלִכ֥וּן אֹתֽוֹ:  ת  וֶאֱנָשִׁין קַדִישִׁין תְּהוֹן קֳדָמָי וּבְשַׂר תְּלִישׁ מִן חֵיוָא חַיְתָא לָא תֵיכְלוּן לְכַלְבָּא תִּרְמוּן יָתֵהּ: (ס)




שמות פרק-כג

{א} לֹ֥א תִשָּׂ֖א שֵׁ֣מַע שָׁ֑וְא אַל-תָּ֤שֶׁת יָֽדְךָ֙ עִם-רָשָׁ֔ע לִהְיֹ֖ת עֵ֥ד חָמָֽס:  מ  לֹ֥א תִשָּׂ֖א שֵׁ֣מַע שָׁ֑וְא אַל-תָּ֤שֶׁת יָֽדְךָ֙ עִם-רָשָׁ֔ע לִהְיֹ֖ת עֵ֥ד חָמָֽס:  ת  לָא תְקַבֵּל שְׁמַע דִשְׁקָר לָא תְשַׁוִי יְדָךְ עִם חַיָבָא לְמֶהֱוֵי לֵהּ סָהִיד שְׁקָר:

{ב} לֹֽא-תִהְיֶ֥ה אַחֲרֵֽי-רַבִּ֖ים לְרָעֹ֑ת וְלֹא-תַעֲנֶ֣ה עַל-רִ֗ב לִנְטֹ֛ת אַחֲרֵ֥י רַבִּ֖ים לְהַטֹּֽת:  מ  לֹֽא-תִהְיֶ֥ה אַחֲרֵֽי-רַבִּ֖ים לְרָעֹ֑ת וְלֹא-תַעֲנֶ֣ה עַל-רִ֗ב לִנְטֹ֛ת אַחֲרֵ֥י רַבִּ֖ים לְהַטֹּֽת:  ת  לָא תְהֵי בָּתַר סַגִיאִין לְאַבְאָשָׁא וְלָא תִתִּמְנַע מִלְאַלָפָא מָא דִבְעֵינָךְ עַל דִינָא בָּתַר סַגִיאֵי שְׁלַם דִינָא:

{ג} וְדָ֕ל לֹ֥א תֶהְדַּ֖ר בְּרִיבֽוֹ:  מ  וְדָ֕ל לֹ֥א תֶהְדַּ֖ר בְּרִיבֽוֹ:  ת  וְעַל מִסְכֵּינָא לָא תְרַחֵם בְּדִינֵהּ: (ס)

{ד} כִּ֣י תִפְגַּ֞ע שׁ֧וֹר אֹֽיִבְךָ֛ א֥וֹ חֲמֹר֖וֹ תֹּעֶ֑ה הָשֵׁ֥ב תְּשִׁיבֶ֖נּוּ לֽוֹ:  מ  כִּ֣י תִפְגַּ֞ע שׁ֧וֹר אֹֽיִבְךָ֛ א֥וֹ חֲמֹר֖וֹ תֹּעֶ֑ה הָשֵׁ֥ב תְּשִׁיבֶ֖נּוּ לֽוֹ:  ת  אֲרֵי תִפְגַע תּוֹרָא דְסָנְאָךְ אוֹ חֲמָרֵהּ דְטָּעֵי אָתָבָא תְּתִיבִנֵהּ לֵהּ: (ס)

{ה} כִּֽי-תִרְאֶ֞ה חֲמ֣וֹר שֹׂנַאֲךָ֗ רֹבֵץ֙ תַּ֣חַת מַשָּׂא֔וֹ וְחָדַלְתָּ֖ מֵעֲזֹ֣ב ל֑וֹ עָזֹ֥ב תַּעֲזֹ֖ב עִמּֽוֹ:  מ  כִּֽי-תִרְאֶ֞ה חֲמ֣וֹר שֹׂנַאֲךָ֗ רֹבֵץ֙ תַּ֣חַת מַשָּׂא֔וֹ וְחָדַלְתָּ֖ מֵעֲזֹ֣ב ל֑וֹ עָזֹ֥ב תַּעֲזֹ֖ב עִמּֽוֹ:  ת  אֲרֵי תֶחֱזֵי חֲמָרָא דְסָנְאָךְ רְבִיעַ תְּחוֹת טוֹעֲנֵיהּ וְתִתִּמְנַע מִלְמִשְׁבַּק לֵהּ מִשְׁבַּק תִּשְׁבּוֹק מָא דִבְלִבָּךְ עֲלוֹהִי וּתְפָרֵק עִמֵהּ: (ס)

{ו}  חמישי  לֹ֥א תַטֶּ֛ה מִשְׁפַּ֥ט אֶבְיֹנְךָ֖ בְּרִיבֽוֹ:  מ  לֹ֥א תַטֶּ֛ה מִשְׁפַּ֥ט אֶבְיֹנְךָ֖ בְּרִיבֽוֹ:  ת  לָא תַצְלֵי דִין מִסְכֵּינָךְ בְּדִינֵהּ:

{ז} מִדְּבַר-שֶׁ֖קֶר תִּרְחָ֑ק וְנָקִ֤י וְצַדִּיק֙ אַֽל-תַּהֲרֹ֔ג כִּ֥י לֹא-אַצְדִּ֖יק רָשָֽׁע:  מ  מִדְּבַר-שֶׁ֖קֶר תִּרְחָ֑ק וְנָקִ֤י וְצַדִּיק֙ אַֽל-תַּהֲרֹ֔ג כִּ֥י לֹא-אַצְדִּ֖יק רָשָֽׁע:  ת  מִפִּתְגָמָא דְשִׁקְרָא הֱוֵי רָחִיק וְדִזְכֵי וְדִי נְפַק (דְכֵי) מִן דִינָא לָא תִקְטוֹל אֲרֵי לָא אֱזַכֵּי חַיָבָא:

{ח} וְשֹׁ֖חַד לֹ֣א תִקָּ֑ח כִּ֤י הַשֹּׁ֙חַד֙ יְעַוֵּ֣ר פִּקְחִ֔ים וִֽיסַלֵּ֖ף דִּבְרֵ֥י צַדִּיקִֽים:  מ  וְשֹׁ֖חַד לֹ֣א תִקָּ֑ח כִּ֤י הַשֹּׁ֙חַד֙ יְעַוֵּ֣ר פִּקְחִ֔ים וִֽיסַלֵּ֖ף דִּבְרֵ֥י צַדִּיקִֽים:  ת  וְשׁוֹחֲדָא לָא תְקַבֵּל אֲרֵי שׁוֹחֲדָא מְעַוֵר עֵינֵי חַכִּימִין וּמְקַלְקֵל פִּתְגָמִין תְּרִיצִין:

{ט} וְגֵ֖ר לֹ֣א תִלְחָ֑ץ וְאַתֶּ֗ם יְדַעְתֶּם֙ אֶת-נֶ֣פֶשׁ הַגֵּ֔ר כִּֽי-גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֥רֶץ מִצְרָֽיִם:  מ  וְגֵ֖ר לֹ֣א תִלְחָ֑ץ וְאַתֶּ֗ם יְדַעְתֶּם֙ אֶת-נֶ֣פֶשׁ הַגֵּ֔ר כִּֽי-גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֥רֶץ מִצְרָֽיִם:  ת  וְגִיוֹרָא לָא תְעִיקוּן וְאַתּוּן יְדַעְתּוּן יָת נַפְשָׁא דְגִיוֹרָא אֲרֵי דַיָרִין הֲוֵיתוּן בְּאַרְעָא דְמִצְרָיִם:

{י} וְשֵׁ֥שׁ שָׁנִ֖ים תִּזְרַ֣ע אֶת-אַרְצֶ֑ךָ וְאָסַפְתָּ֖ אֶת-תְּבוּאָתָֽהּ:  מ  וְשֵׁ֥שׁ שָׁנִ֖ים תִּזְרַ֣ע אֶת-אַרְצֶ֑ךָ וְאָסַפְתָּ֖ אֶת-תְּבוּאָתָֽהּ:  ת  וְשִׁית שְׁנִין תִּזְרַע יָת אַרְעָךְ וְתִכְנוֹשׁ יָת עֲלַלְתַּהּ:

{יא} וְהַשְּׁבִיעִ֞ת תִּשְׁמְטֶ֣נָּה וּנְטַשְׁתָּ֗הּ וְאָֽכְלוּ֙ אֶבְיֹנֵ֣י עַמֶּ֔ךָ וְיִתְרָ֕ם תֹּאכַ֖ל חַיַּ֣ת הַשָּׂדֶ֑ה כֵּֽן-תַּעֲשֶׂ֥ה לְכַרְמְךָ֖ לְזֵיתֶֽךָ:  מ  וְהַשְּׁבִיעִ֞ת תִּשְׁמְטֶ֣נָּה וּנְטַשְׁתָּ֗הּ וְאָֽכְלוּ֙ אֶבְיֹנֵ֣י עַמֶּ֔ךָ וְיִתְרָ֕ם תֹּאכַ֖ל חַיַּ֣ת הַשָּׂדֶ֑ה כֵּֽן-תַּעֲשֶׂ֥ה לְכַרְמְךָ֖ לְזֵיתֶֽךָ:  ת  וּשְׁבִעֵתָא תַּשְׁמְטִּנַהּ וְתַרְטְשִׁנַהּ וְיֵיכְלוּן מִסְכְּנֵי עַמָךְ וּשְׁאָרְהוֹן תֵּיכוּל חֵיוַת בָּרָא כֵּן תַּעְבֵּד לְכַרְמָךְ לְזֵיתָךְ:

{יב} שֵׁ֤שֶׁת יָמִים֙ תַּעֲשֶׂ֣ה מַעֲשֶׂ֔יךָ וּבַיּ֥וֹם הַשְּׁבִיעִ֖י תִּשְׁבֹּ֑ת לְמַ֣עַן יָנ֗וּחַ שֽׁוֹרְךָ֙ וַחֲמֹרֶ֔ךָ וְיִנָּפֵ֥שׁ בֶּן-אֲמָתְךָ֖ וְהַגֵּֽר:  מ  שֵׁ֤שֶׁת יָמִים֙ תַּעֲשֶׂ֣ה מַעֲשֶׂ֔יךָ וּבַיּ֥וֹם הַשְּׁבִיעִ֖י תִּשְׁבֹּ֑ת לְמַ֣עַן יָנ֗וּחַ שֽׁוֹרְךָ֙ וַחֲמֹרֶ֔ךָ וְיִנָּפֵ֥שׁ בֶּן-אֲמָתְךָ֖ וְהַגֵּֽר:  ת  שִׁתָּא יוֹמִין תַּעְבֵּד עוֹבָדָיךְ וּבְיוֹמָא שְׁבִיעָאָה תָּנוּחַ בְּדִיל דִינוּחַ תּוֹרָךְ וְחַמָרָךְ וְיִשְּׁקוֹט בַּר אַמְתָךְ וְגִיוֹרָא:

{יג} וּבְכֹ֛ל אֲשֶׁר-אָמַ֥רְתִּי אֲלֵיכֶ֖ם תִּשָּׁמֵ֑רוּ וְשֵׁ֨ם אֱלֹהִ֤ים אֲחֵרִים֙ לֹ֣א תַזְכִּ֔ירוּ לֹ֥א יִשָּׁמַ֖ע עַל-פִּֽיךָ:  מ  וּבְכֹ֛ל אֲשֶׁר-אָמַ֥רְתִּי אֲלֵיכֶ֖ם תִּשָּׁמֵ֑רוּ וְשֵׁ֨ם אֱלֹהִ֤ים אֲחֵרִים֙ לֹ֣א תַזְכִּ֔ירוּ לֹ֥א יִשָּׁמַ֖ע עַל-פִּֽיךָ:  ת  וּבְכֹל דִי אֲמָרִית לְכוֹן תִּסְתַּמְרוּן וְשׁוּם טַעֲוַת עַמְמַיָא לָא תִדְכְּרוּן לָא יִשְׁתְּמַע עַל פּוּמְכוֹן:

{יד} שָׁלֹ֣שׁ רְגָלִ֔ים תָּחֹ֥ג לִ֖י בַּשָּׁנָֽה:  מ  שָׁלֹ֣שׁ רְגָלִ֔ים תָּחֹ֥ג לִ֖י בַּשָּׁנָֽה:  ת  תְּלַת זִמְנִין תֵּחוֹג קֳדָמַי בְּשַׁתָּא:

{טו} אֶת-חַ֣ג הַמַּצּוֹת֮ תִּשְׁמֹר֒ שִׁבְעַ֣ת יָמִים֩ תֹּאכַ֨ל מַצּ֜וֹת כַּֽאֲשֶׁ֣ר צִוִּיתִ֗ךָ לְמוֹעֵד֙ חֹ֣דֶשׁ הָֽאָבִ֔יב כִּי-ב֖וֹ יָצָ֣אתָ מִמִּצְרָ֑יִם וְלֹא-יֵרָא֥וּ פָנַ֖י רֵיקָֽם:  מ  אֶת-חַ֣ג הַמַּצּוֹת֮ תִּשְׁמֹר֒ שִׁבְעַ֣ת יָמִים֩ תֹּאכַ֨ל מַצּ֜וֹת כַּֽאֲשֶׁ֣ר צִוִּיתִ֗ךָ לְמוֹעֵד֙ חֹ֣דֶשׁ הָֽאָבִ֔יב כִּי-ב֖וֹ יָצָ֣אתָ מִמִּצְרָ֑יִם וְלֹא-יֵרָא֥וּ פָנַ֖י רֵיקָֽם:  ת  יָת חַגָא דְפַטִירַיָא תִּטַר שִׁבְעָא יוֹמִין תֵּיכוּל פַּטִירָא כְּמָא דִפַקֵידִתָּךְ לִזְמַן יַרְחָא דַאֲבִיבָא אֲרֵי בֵהּ נְפַקְתָּ מִמִצְרָיִם וְלָא יִתְחֲזוּן קֳדָמַי רֵיקָנוּן:

{טז} וְחַ֤ג הַקָּצִיר֙ בִּכּוּרֵ֣י מַעֲשֶׂ֔יךָ אֲשֶׁ֥ר תִּזְרַ֖ע בַּשָּׂדֶ֑ה וְחַ֤ג הָֽאָסִף֙ בְּצֵ֣את הַשָּׁנָ֔ה בְּאָסְפְּךָ֥ אֶֽת-מַעֲשֶׂ֖יךָ מִן-הַשָּׂדֶֽה:  מ  וְחַ֤ג הַקָּצִיר֙ בִּכּוּרֵ֣י מַעֲשֶׂ֔יךָ אֲשֶׁ֥ר תִּזְרַ֖ע בַּשָּׂדֶ֑ה וְחַ֤ג הָֽאָסִף֙ בְּצֵ֣את הַשָּׁנָ֔ה בְּאָסְפְּךָ֥ אֶֽת-מַעֲשֶׂ֖יךָ מִן-הַשָּׂדֶֽה:  ת  וְחַגָא דַחֲצָדָא בִּכּוּרֵי עוֹבָדָיךְ דִי תִזְרַע בְּחַקְלָא וְחַגָא דִכְנָשָׁא בְּמִפְּקָא דְשַׁתָּא בְּמִכְנְשָׁךְ יָת עוֹבָדָיךְ מִן חַקְלָא:

{יז} שָׁלֹ֥שׁ פְּעָמִ֖ים בַּשָּׁנָ֑ה יֵרָאֶה֙ כָּל-זְכ֣וּרְךָ֔ אֶל-פְּנֵ֖י הָאָדֹ֥ן | יְהוָֽה:  מ  שָׁלֹ֥שׁ פְּעָמִ֖ים בַּשָּׁנָ֑ה יֵרָאֶה֙ כָּל-זְכ֣וּרְךָ֔ אֶל-פְּנֵ֖י הָאָדֹ֥ן | יְהוָֽה:  ת  תְּלַת זִמְנִין בְּשַׁתָּא יִתְחֲזוּן כָּל דְכוּרָךְ קֳדָם רִבּוֹן עָלְמָא יְיָ:

{יח} לֹֽא-תִזְבַּ֥ח עַל-חָמֵ֖ץ דַּם-זִבְחִ֑י וְלֹֽא-יָלִ֥ין חֵֽלֶב-חַגִּ֖י עַד-בֹּֽקֶר:  מ  לֹֽא-תִזְבַּ֥ח עַל-חָמֵ֖ץ דַּם-זִבְחִ֑י וְלֹֽא-יָלִ֥ין חֵֽלֶב-חַגִּ֖י עַד-בֹּֽקֶר:  ת  לָא תִכּוֹס עַל חֲמִיעַ דַם פִּסְחִי וְלָא יְבִיתוּן בַּר מִן מַדְבְּחָא תַּרְבֵּי נִכְסַת חַגָא עַד צַפְרָא:

{יט} רֵאשִׁ֗ית בִּכּוּרֵי֙ אַדְמָ֣תְךָ֔ תָּבִ֕יא בֵּ֖ית יְהוָ֣ה אֱלֹהֶ֑יךָ לֹֽא-תְבַשֵּׁ֥ל גְּדִ֖י בַּחֲלֵ֥ב אִמּֽוֹ:  מ  רֵאשִׁ֗ית בִּכּוּרֵי֙ אַדְמָ֣תְךָ֔ תָּבִ֕יא בֵּ֖ית יְהוָ֣ה אֱלֹהֶ֑יךָ לֹֽא-תְבַשֵּׁ֥ל גְּדִ֖י בַּחֲלֵ֥ב אִמּֽוֹ:  ת  רֵישׁ בִּכּוּרֵי אַרְעָךְ תַּיְתִי לְבֵית מִקְדְשָׁא דַיְיָ אֱלָהָךְ לָא תֵיכְלוּן בְּשַׂר בַּחֲלָב: (ס)

{כ}  שישי  הִנֵּ֨ה אָנֹכִ֜י שֹׁלֵ֤חַ מַלְאָךְ֙ לְפָנֶ֔יךָ לִשְׁמָרְךָ֖ בַּדָּ֑רֶךְ וְלַהֲבִ֣יאֲךָ֔ אֶל-הַמָּק֖וֹם אֲשֶׁ֥ר הֲכִנֹֽתִי:  מ  הִנֵּ֨ה אָנֹכִ֜י שֹׁלֵ֤חַ מַלְאָךְ֙ לְפָנֶ֔יךָ לִשְׁמָרְךָ֖ בַּדָּ֑רֶךְ וְלַהֲבִ֣יאֲךָ֔ אֶל-הַמָּק֖וֹם אֲשֶׁ֥ר הֲכִנֹֽתִי:  ת  הָא אֲנָא שָׁלַח מַלְאָכָא קֳדָמָךְ לְמִטְרָךְ בְּאָרְחָא וּלְאָעֳלוּתָךְ לְאַתְרָא דִאַתְקֵנִית:

{כא} הִשָּׁ֧מֶר מִפָּנָ֛יו וּשְׁמַ֥ע בְּקֹל֖וֹ אַל-תַּמֵּ֣ר בּ֑וֹ כִּ֣י לֹ֤א יִשָּׂא֙ לְפִשְׁעֲכֶ֔ם כִּ֥י שְׁמִ֖י בְּקִרְבּֽוֹ:  מ  הִשָּׁ֧מֶר מִפָּנָ֛יו וּשְׁמַ֥ע בְּקֹל֖וֹ אַל-תַּמֵּ֣ר בּ֑וֹ כִּ֣י לֹ֤א יִשָּׂא֙ לְפִשְׁעֲכֶ֔ם כִּ֥י שְׁמִ֖י בְּקִרְבּֽוֹ:  ת  אִסְתַּמַּר מִן קֳדָמוֹהִי וְקַבֵּל לְמֵימְרֵהּ לָא תְסָרֵב לְקִבְלֵהּ אֲרֵי לָא יִשְׁבּוֹק לְחוֹבֵיכוֹן אֲרֵי בִשְׁמִי מֵימְרֵהּ:

{כב} כִּ֣י אִם-שָׁמֹ֤עַ תִּשְׁמַע֙ בְּקֹל֔וֹ וְעָשִׂ֕יתָ כֹּ֖ל אֲשֶׁ֣ר אֲדַבֵּ֑ר וְאָֽיַבְתִּי֙ אֶת-אֹ֣יְבֶ֔יךָ וְצַרְתִּ֖י אֶת-צֹרְרֶֽיךָ:  מ  כִּ֣י אִם-שָׁמֹ֤עַ תִּשְׁמַע֙ בְּקֹל֔וֹ וְעָשִׂ֕יתָ כֹּ֖ל אֲשֶׁ֣ר אֲדַבֵּ֑ר וְאָֽיַבְתִּי֙ אֶת-אֹ֣יְבֶ֔יךָ וְצַרְתִּ֖י אֶת-צֹרְרֶֽיךָ:  ת  אֲרֵי אִם קַבָּלָא תְקַבֵּל לְמֵימְרֵהּ וְתַעְבֵּד כֹּל דִאֱמַלֵל וְאֶסְנֵי יָת סָנְאָךְ וְאָעִיק לְדִמְעִיקִין לָךְ:

{כג} כִּֽי-יֵלֵ֣ךְ מַלְאָכִי֮ לְפָנֶיךָ֒ וֶהֱבִֽיאֲךָ֗ אֶל-הָֽאֱמֹרִי֙ וְהַ֣חִתִּ֔י וְהַפְּרִזִּי֙ וְהַֽכְּנַעֲנִ֔י הַחִוִּ֖י וְהַיְבוּסִ֑י וְהִכְחַדְתִּֽיו:  מ  כִּֽי-יֵלֵ֣ךְ מַלְאָכִי֮ לְפָנֶיךָ֒ וֶהֱבִֽיאֲךָ֗ אֶל-הָֽאֱמֹרִי֙ וְהַ֣חִתִּ֔י וְהַפְּרִזִּי֙ וְהַֽכְּנַעֲנִ֔י הַחִוִּ֖י וְהַיְבוּסִ֑י וְהִכְחַדְתִּֽיו:  ת  אֲרֵי יְהַךְ מַלְאָכִי קֳדָמָךְ וְיָעֵלִנָךְ לְוָת אֱמוֹרָאֵי וְחִתָּאֵי וּפְרִיזָאֵי וּכְנַעֲנָאֵי חִוָאֵי וִיבוּסָאֵי וֶאֱשֵׁצִנוּן:

{כד} לֹֽא-תִשְׁתַּחֲוֶ֤ה לֵאלֹֽהֵיהֶם֙ וְלֹ֣א תָֽעָבְדֵ֔ם וְלֹ֥א תַעֲשֶׂ֖ה כְּמַֽעֲשֵׂיהֶ֑ם כִּ֤י הָרֵס֙ תְּהָ֣רְסֵ֔ם וְשַׁבֵּ֥ר תְּשַׁבֵּ֖ר מַצֵּבֹתֵיהֶֽם:  מ  לֹֽא-תִשְׁתַּחֲוֶ֤ה לֵאלֹֽהֵיהֶם֙ וְלֹ֣א תָֽעָבְדֵ֔ם וְלֹ֥א תַעֲשֶׂ֖ה כְּמַֽעֲשֵׂיהֶ֑ם כִּ֤י הָרֵס֙ תְּהָ֣רְסֵ֔ם וְשַׁבֵּ֥ר תְּשַׁבֵּ֖ר מַצֵּבֹתֵיהֶֽם:  ת  לָא תִסְגוּד לְטַעֲוָתְהוֹן וְלָא תִפְלְחִנוּן וְלָא תַעְבֵּד כְּעוֹבָדֵיהוֹן אֲרֵי פַגָרָא תְּפַגְרִנוּן וְתַבָּרָא תְתַבַּר קָמָתְהוֹן:

{כה} וַעֲבַדְתֶּ֗ם אֵ֚ת יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם וּבֵרַ֥ךְ אֶֽת-לַחְמְךָ֖ וְאֶת-מֵימֶ֑יךָ וַהֲסִרֹתִ֥י מַחֲלָ֖ה מִקִּרְבֶּֽךָ:  מ  וַעֲבַדְתֶּ֗ם אֵ֚ת יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם וּבֵרַ֥ךְ אֶֽת-לַחְמְךָ֖ וְאֶת-מֵימֶ֑יךָ וַהֲסִרֹתִ֥י מַחֲלָ֖ה מִקִּרְבֶּֽךָ:  ת  וְתִפְלְחוּן קֳדָם יְיָ אֱלָהֲכוֹן וִיבָרֵךְ יָת מֵיכְלָךְ וְיָת מִשְׁתְּיָךְ וְאַעְדִי מַרְעִין בִּישִׁין מִבֵּינָךְ: (ס)

{כו}  שביעי  לֹ֥א תִהְיֶ֛ה מְשַׁכֵּלָ֥ה וַעֲקָרָ֖ה בְּאַרְצֶ֑ךָ אֶת-מִסְפַּ֥ר יָמֶ֖יךָ אֲמַלֵּֽא:  מ  לֹ֥א תִהְיֶ֛ה מְשַׁכֵּלָ֥ה וַעֲקָרָ֖ה בְּאַרְצֶ֑ךָ אֶת-מִסְפַּ֥ר יָמֶ֖יךָ אֲמַלֵּֽא:  ת  לָא תְהֵי מְתַכְּלָא וְעַקְרָא בְּאַרְעָךְ יָת מִנְיַן יוֹמָיךְ אַשְׁלִים:

{כז} אֶת-אֵֽימָתִי֙ אֲשַׁלַּ֣ח לְפָנֶ֔יךָ וְהַמֹּתִי֙ אֶת-כָּל-הָעָ֔ם אֲשֶׁ֥ר תָּבֹ֖א בָּהֶ֑ם וְנָתַתִּ֧י אֶת-כָּל-אֹיְבֶ֛יךָ אֵלֶ֖יךָ עֹֽרֶף:  מ  אֶת-אֵֽימָתִי֙ אֲשַׁלַּ֣ח לְפָנֶ֔יךָ וְהַמֹּתִי֙ אֶת-כָּל-הָעָ֔ם אֲשֶׁ֥ר תָּבֹ֖א בָּהֶ֑ם וְנָתַתִּ֧י אֶת-כָּל-אֹיְבֶ֛יךָ אֵלֶ֖יךָ עֹֽרֶף:  ת  יָת אֵימְתִי אֲשַׁלַח קֳדָמָךְ ואֶתְּפַר (נ''א וֶאֱשַׁגֵשׁ) יָת כָּל עַמָא דְאַתְּ אָתֵי לַאֲגָחָא בְּהוֹן וְאֶמְסַר יָת כָּל בַּעֲלֵי דְבָבָךְ קֳדָמָךְ מְחַזְרֵי קְדָל:

{כח} וְשָׁלַחְתִּ֥י אֶת-הַצִּרְעָ֖ה לְפָנֶ֑יךָ וְגֵרְשָׁ֗ה אֶת-הַחִוִּ֧י אֶת-הַֽכְּנַעֲנִ֛י וְאֶת-הַחִתִּ֖י מִלְּפָנֶֽיךָ:  מ  וְשָׁלַחְתִּ֥י אֶת-הַצִּרְעָ֖ה לְפָנֶ֑יךָ וְגֵרְשָׁ֗ה אֶת-הַחִוִּ֧י אֶת-הַֽכְּנַעֲנִ֛י וְאֶת-הַחִתִּ֖י מִלְּפָנֶֽיךָ:  ת  וְאֶשְׁלַח יָת עֲרָעִיתָא קֳדָמָךְ וּתְתָרֵךְ יָת חִוָאֵי יָת כְּנַעֲנָאֵי וְיָת חִתָּאֵי מִן קֳדָמָךְ:

{כט} לֹ֧א אֲגָרְשֶׁ֛נּוּ מִפָּנֶ֖יךָ בְּשָׁנָ֣ה אֶחָ֑ת פֶּן-תִּהְיֶ֤ה הָאָ֙רֶץ֙ שְׁמָמָ֔ה וְרַבָּ֥ה עָלֶ֖יךָ חַיַּ֥ת הַשָּׂדֶֽה:  מ  לֹ֧א אֲגָרְשֶׁ֛נּוּ מִפָּנֶ֖יךָ בְּשָׁנָ֣ה אֶחָ֑ת פֶּן-תִּהְיֶ֤ה הָאָ֙רֶץ֙ שְׁמָמָ֔ה וְרַבָּ֥ה עָלֶ֖יךָ חַיַּ֥ת הַשָּׂדֶֽה:  ת  לָא אֱתָרֵכִנוּן מִן קֳדָמָךְ בְּשַׁתָּא חֲדָא דִילְמָא תְהֵי אַרְעָא צַדְיָא וְתִסְגֵי עֲלָךְ חֵיוַת בָּרָא:

{ל} מְעַ֥ט מְעַ֛ט אֲגָרְשֶׁ֖נּוּ מִפָּנֶ֑יךָ עַ֚ד אֲשֶׁ֣ר תִּפְרֶ֔ה וְנָחַלְתָּ֖ אֶת-הָאָֽרֶץ:  מ  מְעַ֥ט מְעַ֛ט אֲגָרְשֶׁ֖נּוּ מִפָּנֶ֑יךָ עַ֚ד אֲשֶׁ֣ר תִּפְרֶ֔ה וְנָחַלְתָּ֖ אֶת-הָאָֽרֶץ:  ת  זְעֵיר זְעֵיר אֱתָרֵכִנוּן מִן קֳדָמָךְ עַד דְתִסְגֵי וְתַחְסֵן יָת אַרְעָא:

{לא} וְשַׁתִּ֣י אֶת-גְּבֻלְךָ֗ מִיַּם-סוּף֙ וְעַד-יָ֣ם פְּלִשְׁתִּ֔ים וּמִמִּדְבָּ֖ר עַד-הַנָּהָ֑ר כִּ֣י | אֶתֵּ֣ן בְּיֶדְכֶ֗ם אֵ֚ת יֹשְׁבֵ֣י הָאָ֔רֶץ וְגֵרַשְׁתָּ֖מוֹ מִפָּנֶֽיךָ:  מ  וְשַׁתִּ֣י אֶת-גְּבֻלְךָ֗ מִיַּם-סוּף֙ וְעַד-יָ֣ם פְּלִשְׁתִּ֔ים וּמִמִּדְבָּ֖ר עַד-הַנָּהָ֑ר כִּ֣י | אֶתֵּ֣ן בְּיֶדְכֶ֗ם אֵ֚ת יֹשְׁבֵ֣י הָאָ֔רֶץ וְגֵרַשְׁתָּ֖מוֹ מִפָּנֶֽיךָ:  ת  וֶאֱשַׁוֵי יָת תְּחוּמָךְ מִיַמָא דְסוּף וְעַד יַמָא דִפְלִשְׁתָּאֵי וּמִמַדְבְּרָא עַד פְּרָת אֲרֵי אֶמְסַר בִּידֵיכוֹן יָת יָתְבֵי אַרְעָא וּתְתָרֵכִנוּן מִן קֳדָמָךְ:

{לב} לֹא-תִכְרֹ֥ת לָהֶ֛ם וְלֵאלֹֽהֵיהֶ֖ם בְּרִֽית:  מ  לֹא-תִכְרֹ֥ת לָהֶ֛ם וְלֵאלֹֽהֵיהֶ֖ם בְּרִֽית:  ת  לָא תִגְזַר לְהוֹן וּלְּטַעֲוָתְהוֹן קְיָם:

{לג} לֹ֤א יֵשְׁבוּ֙ בְּאַרְצְךָ֔ פֶּן-יַחֲטִ֥יאוּ אֹתְךָ֖ לִ֑י כִּ֤י תַעֲבֹד֙ אֶת-אֱלֹ֣הֵיהֶ֔ם כִּֽי-יִהְיֶ֥ה לְךָ֖ לְמוֹקֵֽשׁ:  מ  לֹ֤א יֵשְׁבוּ֙ בְּאַרְצְךָ֔ פֶּן-יַחֲטִ֥יאוּ אֹתְךָ֖ לִ֑י כִּ֤י תַעֲבֹד֙ אֶת-אֱלֹ֣הֵיהֶ֔ם כִּֽי-יִהְיֶ֥ה לְךָ֖ לְמוֹקֵֽשׁ:  ת  לָא יֵתְבוּן בְּאַרְעָךְ דִילְמָא יְחַיְבוּן יָתָךְ קֳדָמָי אֲרֵי תִפְלַח יָת טַעֲוָתְהוֹן אֲרֵי יְהוֹן לָךְ לְתַקְלָא: (ס)




שמות פרק-כד

{א} וְאֶל-מֹשֶׁ֨ה אָמַ֜ר עֲלֵ֣ה אֶל-יְהוָ֗ה אַתָּה֙ וְאַהֲרֹן֙ נָדָ֣ב וַאֲבִיה֔וּא וְשִׁבְעִ֖ים מִזִּקְנֵ֣י יִשְׂרָאֵ֑ל וְהִשְׁתַּחֲוִיתֶ֖ם מֵרָחֹֽק:  מ  וְאֶל-מֹשֶׁ֨ה אָמַ֜ר עֲלֵ֣ה אֶל-יְהוָ֗ה אַתָּה֙ וְאַהֲרֹן֙ נָדָ֣ב וַאֲבִיה֔וּא וְשִׁבְעִ֖ים מִזִּקְנֵ֣י יִשְׂרָאֵ֑ל וְהִשְׁתַּחֲוִיתֶ֖ם מֵרָחֹֽק:  ת  וּלְמשֶׁה אֲמַר סַק לִקֳדָם יְיָ אַתְּ וְאַהֲרֹן נָדָב וַאֲבִיהוּא וְשִׁבְעִין מִסָבֵי יִשְׂרָאֵל וְתִסְגְדוּן מֵרָחִיק:

{ב} וְנִגַּ֨שׁ מֹשֶׁ֤ה לְבַדּוֹ֙ אֶל-יְהוָ֔ה וְהֵ֖ם לֹ֣א יִגָּ֑שׁוּ וְהָעָ֕ם לֹ֥א יַעֲל֖וּ עִמּֽוֹ:  מ  וְנִגַּ֨שׁ מֹשֶׁ֤ה לְבַדּוֹ֙ אֶל-יְהוָ֔ה וְהֵ֖ם לֹ֣א יִגָּ֑שׁוּ וְהָעָ֕ם לֹ֥א יַעֲל֖וּ עִמּֽוֹ:  ת  וְיִתְקְרֵב משֶׁה בִלְחוֹדוֹהִי לִקֳדָם יְיָ וְאִנוּן לָא יִתְקָרְבוּן וְעַמָא לָא יִסְקוּן עִמֵהּ:

{ג} וַיָּבֹ֣א מֹשֶׁ֗ה וַיְסַפֵּ֤ר לָעָם֙ אֵ֚ת כָּל-דִּבְרֵ֣י יְהוָ֔ה וְאֵ֖ת כָּל-הַמִּשְׁפָּטִ֑ים וַיַּ֨עַן כָּל-הָעָ֜ם ק֤וֹל אֶחָד֙ וַיֹּ֣אמְר֔וּ כָּל-הַדְּבָרִ֛ים אֲשֶׁר-דִּבֶּ֥ר יְהוָ֖ה נַעֲשֶֽׂה:  מ  וַיָּבֹ֣א מֹשֶׁ֗ה וַיְסַפֵּ֤ר לָעָם֙ אֵ֚ת כָּל-דִּבְרֵ֣י יְהוָ֔ה וְאֵ֖ת כָּל-הַמִּשְׁפָּטִ֑ים וַיַּ֨עַן כָּל-הָעָ֜ם ק֤וֹל אֶחָד֙ וַיֹּ֣אמְר֔וּ כָּל-הַדְּבָרִ֛ים אֲשֶׁר-דִּבֶּ֥ר יְהוָ֖ה נַעֲשֶֽׂה:  ת  וַאֲתָא משֶׁה וְאִשְׁתָּעִי לְעַמָא יָת כָּל פִּתְגָמַיָא דַיְיָ וְיָת כָּל דִינַיָא וַאֲתֵיב כָּל עַמָא קָלָא חַד וַאֲמָרוּ כָּל פִּתְגָמַיָא דִי מַלִיל יְיָ נַעְבֵּד:

{ד} וַיִּכְתֹּ֣ב מֹשֶׁ֗ה אֵ֚ת כָּל-דִּבְרֵ֣י יְהוָ֔ה וַיַּשְׁכֵּ֣ם בַּבֹּ֔קֶר וַיִּ֥בֶן מִזְבֵּ֖חַ תַּ֣חַת הָהָ֑ר וּשְׁתֵּ֤ים עֶשְׂרֵה֙ מַצֵּבָ֔ה לִשְׁנֵ֥ים עָשָׂ֖ר שִׁבְטֵ֥י יִשְׂרָאֵֽל:  מ  וַיִּכְתֹּ֣ב מֹשֶׁ֗ה אֵ֚ת כָּל-דִּבְרֵ֣י יְהוָ֔ה וַיַּשְׁכֵּ֣ם בַּבֹּ֔קֶר וַיִּ֥בֶן מִזְבֵּ֖חַ תַּ֣חַת הָהָ֑ר וּשְׁתֵּ֤ים עֶשְׂרֵה֙ מַצֵּבָ֔ה לִשְׁנֵ֥ים עָשָׂ֖ר שִׁבְטֵ֥י יִשְׂרָאֵֽל:  ת  וּכְתַב משֶׁה יָת כָּל פִּתְגָמַיָא דַיְיָ וְאַקְדֵם בְּצַפְרָא וּבְנָא מַדְבְּחָא בְּשִׁפּוֹלֵי טוּרָא וְתַרְתָּא עֶשְׂרֵי קָמָן לִתְרֵי עֲסַר שִׁבְטַיָא דְיִשְׂרָאֵל:

{ה} וַיִּשְׁלַ֗ח אֶֽת-נַעֲרֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיַּֽעֲל֖וּ עֹלֹ֑ת וַֽיִּזְבְּח֞וּ זְבָחִ֧ים שְׁלָמִ֛ים לַיהוָ֖ה פָּרִֽים:  מ  וַיִּשְׁלַ֗ח אֶֽת-נַעֲרֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיַּֽעֲל֖וּ עֹלֹ֑ת וַֽיִּזְבְּח֞וּ זְבָחִ֧ים שְׁלָמִ֛ים לַיהוָ֖ה פָּרִֽים:  ת  וּשְׁלַח יָת בּוּכְרֵי בְּנֵי יִשְׂרָאֵל וְאַסִיקוּ עֲלָוָן וְקָרִיבוּ נִכְסַת קוּדְשִׁין קֳדָם יְיָ תּוֹרִין:

{ו} וַיִּקַּ֤ח מֹשֶׁה֙ חֲצִ֣י הַדָּ֔ם וַיָּ֖שֶׂם בָּאַגָּנֹ֑ת וַחֲצִ֣י הַדָּ֔ם זָרַ֖ק עַל-הַמִּזְבֵּֽחַ:  מ  וַיִּקַּ֤ח מֹשֶׁה֙ חֲצִ֣י הַדָּ֔ם וַיָּ֖שֶׂם בָּאַגָּנֹ֑ת וַחֲצִ֣י הַדָּ֔ם זָרַ֖ק עַל-הַמִּזְבֵּֽחַ:  ת  וּנְסִיב משֶׁה פַּלְגוּת דְמָא וְשַׁוִי בְּמִזְרְקַיָא וּפַלְגוּת דְמָא זְרַק עַל מַדְבְּחָא:

{ז} וַיִּקַּח֙ סֵ֣פֶר הַבְּרִ֔ית וַיִּקְרָ֖א בְּאָזְנֵ֣י הָעָ֑ם וַיֹּ֣אמְר֔וּ כֹּ֛ל אֲשֶׁר-דִּבֶּ֥ר יְהוָ֖ה נַעֲשֶׂ֥ה וְנִשְׁמָֽע:  מ  וַיִּקַּח֙ סֵ֣פֶר הַבְּרִ֔ית וַיִּקְרָ֖א בְּאָזְנֵ֣י הָעָ֑ם וַיֹּ֣אמְר֔וּ כֹּ֛ל אֲשֶׁר-דִּבֶּ֥ר יְהוָ֖ה נַעֲשֶׂ֥ה וְנִשְׁמָֽע:  ת  וּנְסִיב סִפְרָא דִקְיָמָא וּקְרָא קֳדָם עַמָא וַאֲמָרוּ כֹּל דִי מַלִיל יְיָ נַעְבֵּד וּנְקַבֵּל:

{ח} וַיִּקַּ֤ח מֹשֶׁה֙ אֶת-הַדָּ֔ם וַיִּזְרֹ֖ק עַל-הָעָ֑ם וַיֹּ֗אמֶר הִנֵּ֤ה דַֽם-הַבְּרִית֙ אֲשֶׁ֨ר כָּרַ֤ת יְהוָה֙ עִמָּכֶ֔ם עַ֥ל כָּל-הַדְּבָרִ֖ים הָאֵֽלֶּה:  מ  וַיִּקַּ֤ח מֹשֶׁה֙ אֶת-הַדָּ֔ם וַיִּזְרֹ֖ק עַל-הָעָ֑ם וַיֹּ֗אמֶר הִנֵּ֤ה דַֽם-הַבְּרִית֙ אֲשֶׁ֨ר כָּרַ֤ת יְהוָה֙ עִמָּכֶ֔ם עַ֥ל כָּל-הַדְּבָרִ֖ים הָאֵֽלֶּה:  ת  וּנְסִיב משֶׁה יָת דְמָא וּזְרַק עַל מַדְבְּחָא לְכַפָּרָא עַל עַמָא וַאֲמַר הָא דֵין דָם קְיָמָא דִגְזַר יְיָ עִמְכוֹן עַל כָּל פִּתְגָמַיָא הָאִלֵין:

{ט} וַיַּ֥עַל מֹשֶׁ֖ה וְאַהֲרֹ֑ן נָדָב֙ וַאֲבִיה֔וּא וְשִׁבְעִ֖ים מִזִּקְנֵ֥י יִשְׂרָאֵֽל:  מ  וַיַּ֥עַל מֹשֶׁ֖ה וְאַהֲרֹ֑ן נָדָב֙ וַאֲבִיה֔וּא וְשִׁבְעִ֖ים מִזִּקְנֵ֥י יִשְׂרָאֵֽל:  ת  וּסְלֵיק משֶׁה וְאַהֲרֹן נָדָב וַאֲבִיהוּא וְשִׁבְעִין מִסָבֵי יִשְׂרָאֵל:

{י} וַיִּרְא֕וּ אֵ֖ת אֱלֹהֵ֣י יִשְׂרָאֵ֑ל וְתַ֣חַת רַגְלָ֗יו כְּמַעֲשֵׂה֙ לִבְנַ֣ת הַסַּפִּ֔יר וּכְעֶ֥צֶם הַשָּׁמַ֖יִם לָטֹֽהַר:  מ  וַיִּרְא֕וּ אֵ֖ת אֱלֹהֵ֣י יִשְׂרָאֵ֑ל וְתַ֣חַת רַגְלָ֗יו כְּמַעֲשֵׂה֙ לִבְנַ֣ת הַסַּפִּ֔יר וּכְעֶ֥צֶם הַשָּׁמַ֖יִם לָטֹֽהַר:  ת  וַחֲזוֹ יָת יְקַר אֱלָהָא דְיִשְׂרָאֵל וּתְחוֹת כָּרְסָא יְקָרֵהּ כְּעוֹבַד אֶבֶן טָבָא וּכְמֶחֱזֵי שְׁמַיָא לְבָרִירוּ:

{יא} וְאֶל-אֲצִילֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לֹ֥א שָׁלַ֖ח יָד֑וֹ וַֽיֶּחֱזוּ֙ אֶת-הָ֣אֱלֹהִ֔ים וַיֹּאכְל֖וּ וַיִּשְׁתּֽוּ:  מ  וְאֶל-אֲצִילֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לֹ֥א שָׁלַ֖ח יָד֑וֹ וַֽיֶּחֱזוּ֙ אֶת-הָ֣אֱלֹהִ֔ים וַיֹּאכְל֖וּ וַיִּשְׁתּֽוּ:  ת  וּלְרַבְרְבֵי בְּנֵי יִשְׂרָאֵל לָא הֲוָה נִזְקָא וַחֲזוֹ יָת יְקָרָא דַיְיָ וַהֲווֹ חָדָן בְּקָרְבָּנֵיהוֹן דְאִתְקַבָּלוּ בְּרַעֲוָא כְּאִלוּ אָכְלִין וְשָׁתָן: (ס)

{יב} וַיֹּ֨אמֶר יְהוָ֜ה אֶל-מֹשֶׁ֗ה עֲלֵ֥ה אֵלַ֛י הָהָ֖רָה וֶהְיֵה-שָׁ֑ם וְאֶתְּנָ֨ה לְךָ֜ אֶת-לֻחֹ֣ת הָאֶ֗בֶן וְהַתּוֹרָה֙ וְהַמִּצְוָ֔ה אֲשֶׁ֥ר כָּתַ֖בְתִּי לְהוֹרֹתָֽם:  מ  וַיֹּ֨אמֶר יְהוָ֜ה אֶל-מֹשֶׁ֗ה עֲלֵ֥ה אֵלַ֛י הָהָ֖רָה וֶהְיֵה-שָׁ֑ם וְאֶתְּנָ֨ה לְךָ֜ אֶת-לֻחֹ֣ת הָאֶ֗בֶן וְהַתּוֹרָה֙ וְהַמִּצְוָ֔ה אֲשֶׁ֥ר כָּתַ֖בְתִּי לְהוֹרֹתָֽם:  ת  וַאֲמַר יְיָ לְמשֶׁה סַק לִקֳדָמַי לְטוּרָא וֶהֱוֵי תַמָן וְאֶתֵּן לָךְ יָת לוּחֵי אַבְנָא וְאוֹרַיְתָא וְתַפְקֶדְתָּא דִכְתָבִית לְאַלוּפֵיהוֹן:

{יג} וַיָּ֣קָם מֹשֶׁ֔ה וִיהוֹשֻׁ֖עַ מְשָׁרְת֑וֹ וַיַּ֥עַל מֹשֶׁ֖ה אֶל-הַ֥ר הָאֱלֹהִֽים:  מ  וַיָּ֣קָם מֹשֶׁ֔ה וִיהוֹשֻׁ֖עַ מְשָׁרְת֑וֹ וַיַּ֥עַל מֹשֶׁ֖ה אֶל-הַ֥ר הָאֱלֹהִֽים:  ת  וְקָם משֶׁה וִיהוֹשֻׁעַ מְשֻׁמְשָׁנֵהּ וּסְלֵיק משֶׁה לְטוּרָא דְאִתְגְלִי עֲלוֹהִי יְקָרָא דַיְיָ:

{יד} וְאֶל-הַזְּקֵנִ֤ים אָמַר֙ שְׁבוּ-לָ֣נוּ בָזֶ֔ה עַ֥ד אֲשֶׁר-נָשׁ֖וּב אֲלֵיכֶ֑ם וְהִנֵּ֨ה אַהֲרֹ֤ן וְחוּר֙ עִמָּכֶ֔ם מִי-בַ֥עַל דְּבָרִ֖ים יִגַּ֥שׁ אֲלֵהֶֽם:  מ  וְאֶל-הַזְּקֵנִ֤ים אָמַר֙ שְׁבוּ-לָ֣נוּ בָזֶ֔ה עַ֥ד אֲשֶׁר-נָשׁ֖וּב אֲלֵיכֶ֑ם וְהִנֵּ֨ה אַהֲרֹ֤ן וְחוּר֙ עִמָּכֶ֔ם מִי-בַ֥עַל דְּבָרִ֖ים יִגַּ֥שׁ אֲלֵהֶֽם:  ת  וּלְסָבַיָא אֲמַר אוֹרִיכוּ לָנָא הָכָא עַד דִנְתוּב לְוָתְכוֹן וְהָא אַהֲרֹן וְחוּר עִמְכוֹן מָן דְאִית לֵהּ דִינָא יִתְקְרֵב לִקֳדָמֵיהוֹן:

{טו}  מפטיר  וַיַּ֥עַל מֹשֶׁ֖ה אֶל-הָהָ֑ר וַיְכַ֥ס הֶעָנָ֖ן אֶת-הָהָֽר:  מ  וַיַּ֥עַל מֹשֶׁ֖ה אֶל-הָהָ֑ר וַיְכַ֥ס הֶעָנָ֖ן אֶת-הָהָֽר:  ת  וּסְלֵק משֶׁה לְטוּרָא וַחֲפָא עֲנָנָא יָת טוּרָא:

{טז} וַיִּשְׁכֹּ֤ן כְּבוֹד-יְהוָה֙ עַל-הַ֣ר סִינַ֔י וַיְכַסֵּ֥הוּ הֶעָנָ֖ן שֵׁ֣שֶׁת יָמִ֑ים וַיִּקְרָ֧א אֶל-מֹשֶׁ֛ה בַּיּ֥וֹם הַשְּׁבִיעִ֖י מִתּ֥וֹךְ הֶעָנָֽן:  מ  וַיִּשְׁכֹּ֤ן כְּבוֹד-יְהוָה֙ עַל-הַ֣ר סִינַ֔י וַיְכַסֵּ֥הוּ הֶעָנָ֖ן שֵׁ֣שֶׁת יָמִ֑ים וַיִּקְרָ֧א אֶל-מֹשֶׁ֛ה בַּיּ֥וֹם הַשְּׁבִיעִ֖י מִתּ֥וֹךְ הֶעָנָֽן:  ת  וּשְׁרָא יְקָרָא דַיְיָ עַל טוּרָא דְסִינַי וַחֲפָהִי עֲנָנָא שִׁתָּא יוֹמִין וּקְרָא לְמשֶׁה בְּיוֹמָא שְׁבִיעָאָה מִגוֹ עֲנָנָא:

{יז} וּמַרְאֵה֙ כְּב֣וֹד יְהוָ֔ה כְּאֵ֥שׁ אֹכֶ֖לֶת בְּרֹ֣אשׁ הָהָ֑ר לְעֵינֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל:  מ  וּמַרְאֵה֙ כְּב֣וֹד יְהוָ֔ה כְּאֵ֥שׁ אֹכֶ֖לֶת בְּרֹ֣אשׁ הָהָ֑ר לְעֵינֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל:  ת  וְחֵיזוּ יְקָרָא דַיְיָ כְּחֵיזוּ אֶשָׁא אָכְלָא בְּרֵישׁ טוּרָא לְעֵינֵי בְּנֵי יִשְׂרָאֵל:

{יח} וַיָּבֹ֥א מֹשֶׁ֛ה בְּת֥וֹךְ הֶעָנָ֖ן וַיַּ֣עַל אֶל-הָהָ֑ר וַיְהִ֤י מֹשֶׁה֙ בָּהָ֔ר אַרְבָּעִ֣ים י֔וֹם וְאַרְבָּעִ֖ים לָֽיְלָה:  מ  וַיָּבֹ֥א מֹשֶׁ֛ה בְּת֥וֹךְ הֶעָנָ֖ן וַיַּ֣עַל אֶל-הָהָ֑ר וַיְהִ֤י מֹשֶׁה֙ בָּהָ֔ר אַרְבָּעִ֣ים י֔וֹם וְאַרְבָּעִ֖ים לָֽיְלָה:  ת  וְעַל משֶׁה בְּגוֹ עֲנָנָא וּסְלֵיק לְטוּרָא וַהֲוָה משֶׁה בְּטּוּרָא אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן: (פפפ)
