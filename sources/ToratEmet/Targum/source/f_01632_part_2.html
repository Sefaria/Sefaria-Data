פרשת ואתחנן


{כג} וָאֶתְחַנַּן אֶל יְהוָה בָּעֵת הַהִוא לֵאמֹר:  מ  וָאֶתְחַנַּן אֶל יְהוָה בָּעֵת הַהִוא לֵאמֹר:  ת  וְצַלֵיתִי קֳדָם יְיָ בְּעִדָנָא הַהִיא לְמֵימָר:

{כד} אֲדֹנָי יֱהוִה אַתָּה הַחִלּוֹתָ לְהַרְאוֹת אֶת עַבְדְּךָ אֶת גָּדְלְךָ וְאֶת יָדְךָ הַחֲזָקָה אֲשֶׁר מִי אֵל בַּשָּׁמַיִם וּבָאָרֶץ אֲשֶׁר יַעֲשֶׂה כְמַעֲשֶׂיךָ וְכִגְבוּרֹתֶךָ:  מ  אֲדֹנָי יֱהוִה אַתָּה הַחִלּוֹתָ לְהַרְאוֹת אֶת עַבְדְּךָ אֶת גָּדְלְךָ וְאֶת יָדְךָ הַחֲזָקָה אֲשֶׁר מִי אֵל בַּשָּׁמַיִם וּבָאָרֶץ אֲשֶׁר יַעֲשֶׂה כְמַעֲשֶׂיךָ וְכִגְבוּרֹתֶךָ:  ת  יְיָ אֱלֹהִים אַתְּ שָׁרֵיתָא לְאַחֲזָאָה יָת עַבְדָךְ יָת רְבוּתָךְ וְיָת יְדָךְ תַּקֶפְתָּא דִי אַתְּ הוּא אֱלָהָא דִשְׁכִנְתָּךְ בִּשְׁמַיָא מִלְעֵלָא וְשַׁלִיט בְּאַרְעָא וְלֵית דְיַעְבֵּד כְּעוֹבְדָיךְ וּכְגִבָּרְתָּךְ:

{כה} אֶעְבְּרָה נָּא וְאֶרְאֶה אֶת הָאָרֶץ הַטּוֹבָה אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן הָהָר הַטּוֹב הַזֶּה וְהַלְּבָנוֹן:  מ  אֶעְבְּרָה נָּא וְאֶרְאֶה אֶת הָאָרֶץ הַטּוֹבָה אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן הָהָר הַטּוֹב הַזֶּה וְהַלְּבָנוֹן:  ת  אֵעִבַר כְּעַן וְאֶחֱזֵי יָת אַרְעָא טַבְתָא דִּי בְּעִבְרָא דְיַרְדְנָא טוּרָא טָּבָא הָדֵין וּבֵית מַקְדָשָׁא:

{כו} וַיִּתְעַבֵּר יְהוָה בִּי לְמַעַנְכֶם וְלֹא שָׁמַע אֵלָי וַיֹּאמֶר יְהוָה אֵלַי רַב לָךְ אַל תּוֹסֶף דַּבֵּר אֵלַי עוֹד בַּדָּבָר הַזֶּה:  מ  וַיִּתְעַבֵּר יְהוָה בִּי לְמַעַנְכֶם וְלֹא שָׁמַע אֵלָי וַיֹּאמֶר יְהוָה אֵלַי רַב לָךְ אַל תּוֹסֶף דַּבֵּר אֵלַי עוֹד בַּדָּבָר הַזֶּה:  ת  וַהֲוָה רְגַז מִן קֳדָם יְיָ עֲלַי בְּדִילְכוֹן וְלָא קַבִּיל מִנִי וַאֲמַר יְיָ לִי סַגִי לָךְ לָא תוֹסֵף לְמַלָלָא קֳדָמַי עוֹד בְּפִתְגָמָא הָדֵין:

{כז} עֲלֵה רֹאשׁ הַפִּסְגָּה וְשָׂא עֵינֶיךָ יָמָּה וְצָפֹנָה וְתֵימָנָה וּמִזְרָחָה וּרְאֵה בְעֵינֶיךָ כִּי לֹא תַעֲבֹר אֶת הַיַּרְדֵּן הַזֶּה:  מ  עֲלֵה רֹאשׁ הַפִּסְגָּה וְשָׂא עֵינֶיךָ יָמָּה וְצָפֹנָה וְתֵימָנָה וּמִזְרָחָה וּרְאֵה בְעֵינֶיךָ כִּי לֹא תַעֲבֹר אֶת הַיַּרְדֵּן הַזֶּה:  ת  סַק לְרֵישׁ רָמָתָא וּזְקוֹף עֵינָיךְ לְמַעֲרָבָא וּלְצִפּוּנָא וְלִדְרוֹמָא וּלְמָדִינְחָא וַחֲזֵי בְעֵינָיךְ אֲרֵי לָא תְעִבַר יָת יַרְדְנָא הָדֵין:

{כח} וְצַו אֶת יְהוֹשֻׁעַ וְחַזְּקֵהוּ וְאַמְּצֵהוּ כִּי הוּא יַעֲבֹר לִפְנֵי הָעָם הַזֶּה וְהוּא יַנְחִיל אוֹתָם אֶת הָאָרֶץ אֲשֶׁר תִּרְאֶה:  מ  וְצַו אֶת יְהוֹשֻׁעַ וְחַזְּקֵהוּ וְאַמְּצֵהוּ כִּי הוּא יַעֲבֹר לִפְנֵי הָעָם הַזֶּה וְהוּא יַנְחִיל אוֹתָם אֶת הָאָרֶץ אֲשֶׁר תִּרְאֶה:  ת  וּפַקֵד יָת יְהוֹשֻׁעַ וְתַקֶפְהִי וְאַלֶמְהִי אֲרֵי הוּא יֵעִבַר קֳדָם עַמָא הָדֵין וְהוּא יַחְסֵן יָתְהוֹן יָת אַרְעָא דִי תֶחֱזֵי:

{כט} וַנֵּשֶׁב בַּגָּיְא מוּל בֵּית פְּעוֹר:  מ  וַנֵּשֶׁב בַּגָּיְא מוּל בֵּית פְּעוֹר:  ת  וִיתֵבְנָא בְחֵילָתָא לָקֳבֵל בֵּית פְּעוֹר: (פ)




דברים פרק-ד

{א} וְעַתָּה יִשְׂרָאֵל שְׁמַע אֶל הַחֻקִּים וְאֶל הַמִּשְׁפָּטִים אֲשֶׁר אָנֹכִי מְלַמֵּד אֶתְכֶם לַעֲשׂוֹת לְמַעַן תִּחְיוּ וּבָאתֶם וִירִשְׁתֶּם אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֵי אֲבֹתֵיכֶם נֹתֵן לָכֶם:  מ  וְעַתָּה יִשְׂרָאֵל שְׁמַע אֶל הַחֻקִּים וְאֶל הַמִּשְׁפָּטִים אֲשֶׁר אָנֹכִי מְלַמֵּד אֶתְכֶם לַעֲשׂוֹת לְמַעַן תִּחְיוּ וּבָאתֶם וִירִשְׁתֶּם אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֵי אֲבֹתֵיכֶם נֹתֵן לָכֶם:  ת  וּכְעַן יִשְׂרָאֵל שְׁמַע לִקְיָמַיָא וּלְדִינַיָא דִי אֲנָא מְאַלֵף יָתְכוֹן לְמֶעְבַּד בְּדִיל דְתֵיחוּן וְתֵיתוּן וְתֵירְתוּן יָת אַרְעָא דִי יְיָ אֱלָהָא דַאֲבָהָתְכוֹן יָהֵב לְכוֹן:

{ב} לֹא תֹסִפוּ עַל הַדָּבָר אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם וְלֹא תִגְרְעוּ מִמֶּנּוּ לִשְׁמֹר אֶת מִצְוֹת יְהוָה אֱלֹהֵיכֶם אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם:  מ  לֹא תֹסִפוּ עַל הַדָּבָר אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם וְלֹא תִגְרְעוּ מִמֶּנּוּ לִשְׁמֹר אֶת מִצְוֹת יְהוָה אֱלֹהֵיכֶם אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם:  ת  לָא תוֹסְפוּן עַל פִּתְגָמָא דִי אֲנָא מְפַקֵד יָתְכוֹן וְלָא תִמְנְעוּן מִנֵהּ לְמִטַר יָת פִּקוּדַיָא דַיְיָ אֱלָהֲכוֹן דִי אֲנָא מְפַקֵד יָתְכוֹן:

{ג} עֵינֵיכֶם הָרֹאֹת אֵת אֲשֶׁר עָשָׂה יְהוָה בְּבַעַל פְּעוֹר כִּי כָל הָאִישׁ אֲשֶׁר הָלַךְ אַחֲרֵי בַעַל פְּעוֹר הִשְׁמִידוֹ יְהוָה אֱלֹהֶיךָ מִקִּרְבֶּךָ:  מ  עֵינֵיכֶם הָרֹאֹת אֵת אֲשֶׁר עָשָׂה יְהוָה בְּבַעַל פְּעוֹר כִּי כָל הָאִישׁ אֲשֶׁר הָלַךְ אַחֲרֵי בַעַל פְּעוֹר הִשְׁמִידוֹ יְהוָה אֱלֹהֶיךָ מִקִּרְבֶּךָ:  ת  עֵינֵיכוֹן חָזָן יָת דִי עֲבַד יְיָ בְּבַעַל פְּעוֹר אֲרֵי כָּל גַבְרָא דִי אֲזַל בָּתַר בַּעַל פְּעוֹר שֵׁצְיֵהּ יְיָ אֱלָהָךְ מִבֵּינָךְ:

{ד} וְאַתֶּם הַדְּבֵקִים בַּיהוָה אֱלֹהֵיכֶם חַיִּים כֻּלְּכֶם הַיּוֹם:  מ  וְאַתֶּם הַדְּבֵקִים בַּיהוָה אֱלֹהֵיכֶם חַיִּים כֻּלְּכֶם הַיּוֹם:  ת  וְאַתּוּן דְאַדְבֶּקְתּוּן בְּדַחַלְתָּא דַיְיָ אֱלָהֲכוֹן קַיָמִין כֻּלְכוֹן יוֹמָא דֵין:

{ה}  שני  רְאֵה לִמַּדְתִּי אֶתְכֶם חֻקִּים וּמִשְׁפָּטִים כַּאֲשֶׁר צִוַּנִי יְהוָה אֱלֹהָי לַעֲשׂוֹת כֵּן בְּקֶרֶב הָאָרֶץ אֲשֶׁר אַתֶּם בָּאִים שָׁמָּה לְרִשְׁתָּהּ:  מ  רְאֵה לִמַּדְתִּי אֶתְכֶם חֻקִּים וּמִשְׁפָּטִים כַּאֲשֶׁר צִוַּנִי יְהוָה אֱלֹהָי לַעֲשׂוֹת כֵּן בְּקֶרֶב הָאָרֶץ אֲשֶׁר אַתֶּם בָּאִים שָׁמָּה לְרִשְׁתָּהּ:  ת  חֲזוֹ דְאַלֵפִית יָתְכוֹן קְיָמַיָא וְדִינַיָא כְּמָא דִי פַקְדַנִי יְיָ אֱלָהָי לְמֶעְבַּד כֵּן בְּגוֹ אַרְעָא דִי אַתּוּן עַלִין תַּמָן לְמֵירְתַהּ:

{ו} וּשְׁמַרְתֶּם וַעֲשִׂיתֶם כִּי הִוא חָכְמַתְכֶם וּבִינַתְכֶם לְעֵינֵי הָעַמִּים אֲשֶׁר יִשְׁמְעוּן אֵת כָּל הַחֻקִּים הָאֵלֶּה וְאָמְרוּ רַק עַם חָכָם וְנָבוֹן הַגּוֹי הַגָּדוֹל הַזֶּה:  מ  וּשְׁמַרְתֶּם וַעֲשִׂיתֶם כִּי הִוא חָכְמַתְכֶם וּבִינַתְכֶם לְעֵינֵי הָעַמִּים אֲשֶׁר יִשְׁמְעוּן אֵת כָּל הַחֻקִּים הָאֵלֶּה וְאָמְרוּ רַק עַם חָכָם וְנָבוֹן הַגּוֹי הַגָּדוֹל הַזֶּה:  ת  וְתִטְרוּן וְתַעְבְּדוּן אֲרֵי הִיא חָכְמַתְכוֹן וְסוּכְלְתָנוּתְכוֹן לְעֵינֵי עַמְמַיָא דִי יִשְׁמְעוּן יָת כָּל קְיָמַיָא הָאִלֵין וְיֵימְרוּן לְחוֹד עַם חַכִּים וְסוּכְלְתָן עַמָא רַבָּא הָדֵין:

{ז} כִּי מִי גוֹי גָּדוֹל אֲשֶׁר לוֹ אֱלֹהִים קְרֹבִים אֵלָיו כַּיהוָה אֱלֹהֵינוּ בְּכָל קָרְאֵנוּ אֵלָיו:  מ  כִּי מִי גוֹי גָּדוֹל אֲשֶׁר לוֹ אֱלֹהִים קְרֹבִים אֵלָיו כַּיהוָה אֱלֹהֵינוּ בְּכָל קָרְאֵנוּ אֵלָיו:  ת  אֲרֵי מָן עַם רַב דִי לֵהּ אֱלָהָא קָרִיב לֵהּ לְקַבָּלָא צְלוֹתֵהּ בְּעִדַן עַקְתֵהּ כַּייָ אֱלָהָנָא בְּכָל עִדָן דַאֲנַחְנָא מְצַלִין קֳדָמוֹהִי:

{ח} וּמִי גּוֹי גָּדוֹל אֲשֶׁר לוֹ חֻקִּים וּמִשְׁפָּטִים צַדִּיקִם כְּכֹל הַתּוֹרָה הַזֹּאת אֲשֶׁר אָנֹכִי נֹתֵן לִפְנֵיכֶם הַיּוֹם:  מ  וּמִי גּוֹי גָּדוֹל אֲשֶׁר לוֹ חֻקִּים וּמִשְׁפָּטִים צַדִּיקִם כְּכֹל הַתּוֹרָה הַזֹּאת אֲשֶׁר אָנֹכִי נֹתֵן לִפְנֵיכֶם הַיּוֹם:  ת  וּמָן עַם רַב דִי לֵהּ קְיָמִין וְדִינִין קַשִׁיטִין כְּכֹל אוֹרַיְתָא הָדָא דִי אֲנָא יָהֵב קֳדָמֵכוֹן יוֹמָא דֵין:

{ט} רַק הִשָּׁמֶר לְךָ וּשְׁמֹר נַפְשְׁךָ מְאֹד פֶּן תִּשְׁכַּח אֶת הַדְּבָרִים אֲשֶׁר רָאוּ עֵינֶיךָ וּפֶן יָסוּרוּ מִלְּבָבְךָ כֹּל יְמֵי חַיֶּיךָ וְהוֹדַעְתָּם לְבָנֶיךָ וְלִבְנֵי בָנֶיךָ:  מ  רַק הִשָּׁמֶר לְךָ וּשְׁמֹר נַפְשְׁךָ מְאֹד פֶּן תִּשְׁכַּח אֶת הַדְּבָרִים אֲשֶׁר רָאוּ עֵינֶיךָ וּפֶן יָסוּרוּ מִלְּבָבְךָ כֹּל יְמֵי חַיֶּיךָ וְהוֹדַעְתָּם לְבָנֶיךָ וְלִבְנֵי בָנֶיךָ:  ת  לְחוֹד אִסְתַּמַר לָךְ וְטַר נַפְשָׁךְ לַחֲדָא דִילְמָא תִנְשֵׁי יָת פִּתְגָמַיָא דִי חֲזוֹ עֵינָיִךְ וְדִילְמָא יְעִדוּן מִלִבָּךְ כֹּל יוֹמֵי חַיָיךְ וּתְהוֹדְעִנוּן לִבְנָיךְ וְלִבְנֵי בְנָיִךְ:

{י} יוֹם אֲשֶׁר עָמַדְתָּ לִפְנֵי יְהוָה אֱלֹהֶיךָ בְּחֹרֵב בֶּאֱמֹר יְהוָה אֵלַי הַקְהֶל לִי אֶת הָעָם וְאַשְׁמִעֵם אֶת דְּבָרָי אֲשֶׁר יִלְמְדוּן לְיִרְאָה אֹתִי כָּל הַיָּמִים אֲשֶׁר הֵם חַיִּים עַל הָאֲדָמָה וְאֶת בְּנֵיהֶם יְלַמֵּדוּן:  מ  יוֹם אֲשֶׁר עָמַדְתָּ לִפְנֵי יְהוָה אֱלֹהֶיךָ בְּחֹרֵב בֶּאֱמֹר יְהוָה אֵלַי הַקְהֶל לִי אֶת הָעָם וְאַשְׁמִעֵם אֶת דְּבָרָי אֲשֶׁר יִלְמְדוּן לְיִרְאָה אֹתִי כָּל הַיָּמִים אֲשֶׁר הֵם חַיִּים עַל הָאֲדָמָה וְאֶת בְּנֵיהֶם יְלַמֵּדוּן:  ת  יוֹמָא דִי קַמְתָּא קֳדָם יְיָ אֱלָהָךְ בְּחֹרֵב כַּד אֲמַר יְיָ לִי כְּנוֹשׁ קֳדָמַי יָת עַמָא וְאַשְׁמְעִנוּן יָת פִּתְגָמָי דִי יֵלְפוּן לְמִדְחַל קֳדָמַי כָּל יוֹמַיָא דִי אִנוּן קַיָמִין עַל אַרְעָא וְיָת בְּנֵיהוֹן יְאַלְפוּן:

{יא} וַתִּקְרְבוּן וַתַּעַמְדוּן תַּחַת הָהָר וְהָהָר בֹּעֵר בָּאֵשׁ עַד לֵב הַשָּׁמַיִם חֹשֶׁךְ עָנָן וַעֲרָפֶל:  מ  וַתִּקְרְבוּן וַתַּעַמְדוּן תַּחַת הָהָר וְהָהָר בֹּעֵר בָּאֵשׁ עַד לֵב הַשָּׁמַיִם חֹשֶׁךְ עָנָן וַעֲרָפֶל:  ת  וּקְרֶבְתּוּן וְקַמְתּוּן בְּשִׁפּוֹלֵי טוּרָא וְטוּרָא בָּעֵר בְּאֶשָׁתָא עַד צֵית שְׁמַיָא חֲשׁוֹכָא עֲנָנָא וַאֲמִיטְתָא:

{יב} וַיְדַבֵּר יְהוָה אֲלֵיכֶם מִתּוֹךְ הָאֵשׁ קוֹל דְּבָרִים אַתֶּם שֹׁמְעִים וּתְמוּנָה אֵינְכֶם רֹאִים זוּלָתִי קוֹל:  מ  וַיְדַבֵּר יְהוָה אֲלֵיכֶם מִתּוֹךְ הָאֵשׁ קוֹל דְּבָרִים אַתֶּם שֹׁמְעִים וּתְמוּנָה אֵינְכֶם רֹאִים זוּלָתִי קוֹל:  ת  וּמַלִיל יְיָ עִמְכוֹן מִגוֹ אֶשְׁתָא קַל פִּתְגָמִין אַתּוּן שָׁמְעִין וּדְמוּת לֵיתֵיכוֹן חָזָן אֱלָהֵן קָלָא:

{יג} וַיַּגֵּד לָכֶם אֶת בְּרִיתוֹ אֲשֶׁר צִוָּה אֶתְכֶם לַעֲשׂוֹת עֲשֶׂרֶת הַדְּבָרִים וַיִּכְתְּבֵם עַל שְׁנֵי לֻחוֹת אֲבָנִים:  מ  וַיַּגֵּד לָכֶם אֶת בְּרִיתוֹ אֲשֶׁר צִוָּה אֶתְכֶם לַעֲשׂוֹת עֲשֶׂרֶת הַדְּבָרִים וַיִּכְתְּבֵם עַל שְׁנֵי לֻחוֹת אֲבָנִים:  ת  וְחַוִי לְכוֹן יָת קְיָמֵהּ דִי פַקִיד יָתְכוֹן לְמֶעְבַּד עַשְׂרָא פִּתְגָמִין וּכְתָבִנוּן עַל תְּרֵין לוּחֵי אַבְנַיָא:

{יד} וְאֹתִי צִוָּה יְהוָה בָּעֵת הַהִוא לְלַמֵּד אֶתְכֶם חֻקִּים וּמִשְׁפָּטִים לַעֲשֹׂתְכֶם אֹתָם בָּאָרֶץ אֲשֶׁר אַתֶּם עֹבְרִים שָׁמָּה לְרִשְׁתָּהּ:  מ  וְאֹתִי צִוָּה יְהוָה בָּעֵת הַהִוא לְלַמֵּד אֶתְכֶם חֻקִּים וּמִשְׁפָּטִים לַעֲשֹׂתְכֶם אֹתָם בָּאָרֶץ אֲשֶׁר אַתֶּם עֹבְרִים שָׁמָּה לְרִשְׁתָּהּ:  ת  וְיָתִי פַּקִיד יְיָ בְּעִדְנָא הַהִיא לְאַלָפָא יַתְכוֹן קְיָמִין וְדִינִין לְמֶעְבָּדְכוֹן יָתְהוֹן בְּאַרְעָא דִי אַתּוּן עָבְרִין תַּמָן לְמֵירְתַהּ:

{טו} וְנִשְׁמַרְתֶּם מְאֹד לְנַפְשֹׁתֵיכֶם כִּי לֹא רְאִיתֶם כָּל תְּמוּנָה בְּיוֹם דִּבֶּר יְהוָה אֲלֵיכֶם בְּחֹרֵב מִתּוֹךְ הָאֵשׁ:  מ  וְנִשְׁמַרְתֶּם מְאֹד לְנַפְשֹׁתֵיכֶם כִּי לֹא רְאִיתֶם כָּל תְּמוּנָה בְּיוֹם דִּבֶּר יְהוָה אֲלֵיכֶם בְּחֹרֵב מִתּוֹךְ הָאֵשׁ:  ת  וְתִסְתַּמְרוּן לַחֲדָא לְנַפְשָׁתֵיכוֹן אֲרֵי לָא חֲזֵיתוּן כָּל דְמוּת בְּיוֹמָא דִי מַלִיל יְיָ עִמְכוֹן בְּחֹרֵב מִגוֹ אֶשָׁתָא:

{טז} פֶּן תַּשְׁחִתוּן וַעֲשִׂיתֶם לָכֶם פֶּסֶל תְּמוּנַת כָּל סָמֶל תַּבְנִית זָכָר אוֹ נְקֵבָה:  מ  פֶּן תַּשְׁחִתוּן וַעֲשִׂיתֶם לָכֶם פֶּסֶל תְּמוּנַת כָּל סָמֶל תַּבְנִית זָכָר אוֹ נְקֵבָה:  ת  דִילְמָא תְּחַבְּלוּן וְתַעְבְּדוּן לְכוֹן צֶלֶם דְמוּת כָּל צוּרָא דְמוּת דְכַר אוֹ נֻקְבָא:

{יז} תַּבְנִית כָּל בְּהֵמָה אֲשֶׁר בָּאָרֶץ תַּבְנִית כָּל צִפּוֹר כָּנָף אֲשֶׁר תָּעוּף בַּשָּׁמָיִם:  מ  תַּבְנִית כָּל בְּהֵמָה אֲשֶׁר בָּאָרֶץ תַּבְנִית כָּל צִפּוֹר כָּנָף אֲשֶׁר תָּעוּף בַּשָּׁמָיִם:  ת  דְמוּת כָּל בְּעִירָא דִי בְאַרְעָא דְמוּת כָּל צִפַּר גַדְפָא דִי פָּרַח בַּאֲוֵר רְקִיע שְׁמַיָא:

{יח} תַּבְנִית כָּל רֹמֵשׂ בָּאֲדָמָה תַּבְנִית כָּל דָּגָה אֲשֶׁר בַּמַּיִם מִתַּחַת לָאָרֶץ:  מ  תַּבְנִית כָּל רֹמֵשׂ בָּאֲדָמָה תַּבְנִית כָּל דָּגָה אֲשֶׁר בַּמַּיִם מִתַּחַת לָאָרֶץ:  ת  דְמוּת כָּל רִחֲשָׁא דִי בְאַרְעָא דְמוּת כָּל נוּנֵי דִי בְמַיָא מִלְרַע לְאַרְעָא:

{יט} וּפֶן תִּשָּׂא עֵינֶיךָ הַשָּׁמַיְמָה וְרָאִיתָ אֶת הַשֶּׁמֶשׁ וְאֶת הַיָּרֵחַ וְאֶת הַכּוֹכָבִים כֹּל צְבָא הַשָּׁמַיִם וְנִדַּחְתָּ וְהִשְׁתַּחֲוִיתָ לָהֶם וַעֲבַדְתָּם אֲשֶׁר חָלַק יְהוָה אֱלֹהֶיךָ אֹתָם לְכֹל הָעַמִּים תַּחַת כָּל הַשָּׁמָיִם:  מ  וּפֶן תִּשָּׂא עֵינֶיךָ הַשָּׁמַיְמָה וְרָאִיתָ אֶת הַשֶּׁמֶשׁ וְאֶת הַיָּרֵחַ וְאֶת הַכּוֹכָבִים כֹּל צְבָא הַשָּׁמַיִם וְנִדַּחְתָּ וְהִשְׁתַּחֲוִיתָ לָהֶם וַעֲבַדְתָּם אֲשֶׁר חָלַק יְהוָה אֱלֹהֶיךָ אֹתָם לְכֹל הָעַמִּים תַּחַת כָּל הַשָּׁמָיִם:  ת  וְדִילְמָא תִזְקוֹף עֵינָיךְ לִשְׁמַיָא וְתֶחֱזֵי יָת שִׁמְשָׁא וְיָת סִיהֲרָא וְיָת כּוֹכְבַיָא כֹּל חֵילֵי שְׁמַיָא וְתִטְעֵי וְתִסְגוּד לְהוֹן וְתִפְלְחִנוּן דִי זַמִין יְיָ אֱלָהָךְ יָתְהוֹן לְכֹל עַמְמַיָא תְּחוֹת כָּל שְׁמַיָא:

{כ} וְאֶתְכֶם לָקַח יְהוָה וַיּוֹצִא אֶתְכֶם מִכּוּר הַבַּרְזֶל מִמִּצְרָיִם לִהְיוֹת לוֹ לְעַם נַחֲלָה כַּיּוֹם הַזֶּה:  מ  וְאֶתְכֶם לָקַח יְהוָה וַיּוֹצִא אֶתְכֶם מִכּוּר הַבַּרְזֶל מִמִּצְרָיִם לִהְיוֹת לוֹ לְעַם נַחֲלָה כַּיּוֹם הַזֶּה:  ת  וְיָתְכוֹן קָרִיב יְיָ לְדַחַלְתֵּהּ וְאַפֵּק יָתְכוֹן מִכּוּרָא דְפַרְזְלָא מִמִצְרָיִם לְמֶהֱוֵי לֵהּ לְעַמָא אַחֲסָנָא כְּיוֹמָא הָדֵין:

{כא} וַיהוָה הִתְאַנֶּף בִּי עַל דִּבְרֵיכֶם וַיִּשָּׁבַע לְבִלְתִּי עָבְרִי אֶת הַיַּרְדֵּן וּלְבִלְתִּי בֹא אֶל הָאָרֶץ הַטּוֹבָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה:  מ  וַיהוָה הִתְאַנֶּף בִּי עַל דִּבְרֵיכֶם וַיִּשָּׁבַע לְבִלְתִּי עָבְרִי אֶת הַיַּרְדֵּן וּלְבִלְתִּי בֹא אֶל הָאָרֶץ הַטּוֹבָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה:  ת  וּמִן קֳדָם יְיָ הֲוָה רְגַז עֲלַי עַל פִּתְגָמֵיכוֹן וְקַיִים בְּדִיל דְלָא לְמֵעֲבַר יָת יַרְדְנָא וּבְדִיל דְלָא לְמֵיעַל לְאַרְעָא טָבָא דִי יְיָ אֱלָהָךְ יָהֵבּ לָךְ אַחֲסָנָא:

{כב} כִּי אָנֹכִי מֵת בָּאָרֶץ הַזֹּאת אֵינֶנִּי עֹבֵר אֶת הַיַּרְדֵּן וְאַתֶּם עֹבְרִים וִירִשְׁתֶּם אֶת הָאָרֶץ הַטּוֹבָה הַזֹּאת:  מ  כִּי אָנֹכִי מֵת בָּאָרֶץ הַזֹּאת אֵינֶנִּי עֹבֵר אֶת הַיַּרְדֵּן וְאַתֶּם עֹבְרִים וִירִשְׁתֶּם אֶת הָאָרֶץ הַטּוֹבָה הַזֹּאת:  ת  אֲרֵי אֲנָא מָאִית בְּאַרְעָא הָדָא לֵית אֲנָא עָבֵר יָת יַרְדְנָא וְאַתּוּן עָבְרִין וְתֵירְתוּן יָת אַרְעָא טָבָא הָדָא:

{כג} הִשָּׁמְרוּ לָכֶם פֶּן תִּשְׁכְּחוּ אֶת בְּרִית יְהוָה אֱלֹהֵיכֶם אֲשֶׁר כָּרַת עִמָּכֶם וַעֲשִׂיתֶם לָכֶם פֶּסֶל תְּמוּנַת כֹּל אֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ:  מ  הִשָּׁמְרוּ לָכֶם פֶּן תִּשְׁכְּחוּ אֶת בְּרִית יְהוָה אֱלֹהֵיכֶם אֲשֶׁר כָּרַת עִמָּכֶם וַעֲשִׂיתֶם לָכֶם פֶּסֶל תְּמוּנַת כֹּל אֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ:  ת  אִסְתַּמָרוּ לְכוֹן דִילְמָא תִתְנְשׁוּן יָת קְיָמָא דַיְיָ אֱלָהֲכוֹן דִי גְזַר עִמְכוֹן וְתַעְבְּדוּן לְכוֹן צֶלֶם דְמוּת כֹּלָא דִי פַקְדָךְ יְיָ אֱלָהָךְ:

{כד} כִּי יְהוָה אֱלֹהֶיךָ אֵשׁ אֹכְלָה הוּא אֵל קַנָּא:  מ  כִּי יְהוָה אֱלֹהֶיךָ אֵשׁ אֹכְלָה הוּא אֵל קַנָּא:  ת  אֲרֵי יְיָ אֱלָהָךְ מֵימְרֵהּ אֶשָׁא אָכְלָא הוּא אֵל קַנָא: (פ)

{כה} כִּי תוֹלִיד בָּנִים וּבְנֵי בָנִים וְנוֹשַׁנְתֶּם בָּאָרֶץ וְהִשְׁחַתֶּם וַעֲשִׂיתֶם פֶּסֶל תְּמוּנַת כֹּל וַעֲשִׂיתֶם הָרַע בְּעֵינֵי יְהוָה אֱלֹהֶיךָ לְהַכְעִיסוֹ:  מ  כִּי תוֹלִיד בָּנִים וּבְנֵי בָנִים וְנוֹשַׁנְתֶּם בָּאָרֶץ וְהִשְׁחַתֶּם וַעֲשִׂיתֶם פֶּסֶל תְּמוּנַת כֹּל וַעֲשִׂיתֶם הָרַע בְּעֵינֵי יְהוָה אֱלֹהֶיךָ לְהַכְעִיסוֹ:  ת  אֲרֵי תוֹלְדוּן בְּנִין וּבְנֵי בְנִין וְתִתְעַתְּקוּן בְּאַרְעָא וּתְחַבְּלוּן וְתַעְבְּדוּן צֶלֶם דְמוּת כֹּלָא וְתַעְבְּדוּן דְבִישׁ קֳדָם יְיָ אֱלָהָךְ לְאַרְגָזָא קֳדָמוֹהִי:

{כו} הַעִידֹתִי בָכֶם הַיּוֹם אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ כִּי אָבֹד תֹּאבֵדוּן מַהֵר מֵעַל הָאָרֶץ אֲשֶׁר אַתֶּם עֹבְרִים אֶת הַיַּרְדֵּן שָׁמָּה לְרִשְׁתָּהּ לֹא תַאֲרִיכֻן יָמִים עָלֶיהָ כִּי הִשָּׁמֵד תִּשָּׁמֵדוּן:  מ  הַעִידֹתִי בָכֶם הַיּוֹם אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ כִּי אָבֹד תֹּאבֵדוּן מַהֵר מֵעַל הָאָרֶץ אֲשֶׁר אַתֶּם עֹבְרִים אֶת הַיַּרְדֵּן שָׁמָּה לְרִשְׁתָּהּ לֹא תַאֲרִיכֻן יָמִים עָלֶיהָ כִּי הִשָּׁמֵד תִּשָּׁמֵדוּן:  ת  אַסְהֵדִית בְּכוֹן יוֹמָא דֵין יָת שְׁמַיָא וְיָת אַרְעָא אֲרֵי מֵיבַד תֵּיבְדוּן בִּפְרִיעַ מֵעַל אַרְעָא דִי אַתּוּן עָבְרִין יָת יַרְדְנָא לְתַמָן לְמֵירְתַהּ לָא תוֹרְכוּן יוֹמִין עֲלַהּ אֲרֵי אִשְׁתְּצָאָה תִּשְׁתֵּצוּן:

{כז} וְהֵפִיץ יְהוָה אֶתְכֶם בָּעַמִּים וְנִשְׁאַרְתֶּם מְתֵי מִסְפָּר בַּגּוֹיִם אֲשֶׁר יְנַהֵג יְהוָה אֶתְכֶם שָׁמָּה:  מ  וְהֵפִיץ יְהוָה אֶתְכֶם בָּעַמִּים וְנִשְׁאַרְתֶּם מְתֵי מִסְפָּר בַּגּוֹיִם אֲשֶׁר יְנַהֵג יְהוָה אֶתְכֶם שָׁמָּה:  ת  וִיבַדַר יְיָ יָתְכוֹן בְּעַמְמַיָא וְתִשְׁתָּאֲרוּן עַם דְמִנְיָן בְּעַמְמַיָא דִי יְדַבַּר יְיָ יָתְכוֹן לְתַמָן:

{כח} וַעֲבַדְתֶּם שָׁם אֱלֹהִים מַעֲשֵׂה יְדֵי אָדָם עֵץ וָאֶבֶן אֲשֶׁר לֹא יִרְאוּן וְלֹא יִשְׁמְעוּן וְלֹא יֹאכְלוּן וְלֹא יְרִיחֻן:  מ  וַעֲבַדְתֶּם שָׁם אֱלֹהִים מַעֲשֵׂה יְדֵי אָדָם עֵץ וָאֶבֶן אֲשֶׁר לֹא יִרְאוּן וְלֹא יִשְׁמְעוּן וְלֹא יֹאכְלוּן וְלֹא יְרִיחֻן:  ת  וְתִפְלְחוּן תַּמָן לְעַמְמַיָא פָלְּחֵי טַעֲוָתָא עֹבַד יְדֵי אֱנָשָׁא אָעָא וְאַבְנָא דִי לָא יֶחֱזוּן וְלָא שָׁמְעִין וְלָא אָכְלִין וְלָא מְרִיחִין:

{כט} וּבִקַּשְׁתֶּם מִשָּׁם אֶת יְהוָה אֱלֹהֶיךָ וּמָצָאתָ כִּי תִדְרְשֶׁנּוּ בְּכָל לְבָבְךָ וּבְכָל נַפְשֶׁךָ:  מ  וּבִקַּשְׁתֶּם מִשָּׁם אֶת יְהוָה אֱלֹהֶיךָ וּמָצָאתָ כִּי תִדְרְשֶׁנּוּ בְּכָל לְבָבְךָ וּבְכָל נַפְשֶׁךָ:  ת  וְתִתְבְּעוּן מִתַּמָן יָת דַחֲלְתָּא דַייָ אֱלָהָךְ וְתִשְׁכָּח אֲרֵי תִבְעֵי מִן קֳדָמוֹהִי בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ:

{ל} בַּצַּר לְךָ וּמְצָאוּךָ כֹּל הַדְּבָרִים הָאֵלֶּה בְּאַחֲרִית הַיָּמִים וְשַׁבְתָּ עַד יְהוָה אֱלֹהֶיךָ וְשָׁמַעְתָּ בְּקֹלוֹ:  מ  בַּצַּר לְךָ וּמְצָאוּךָ כֹּל הַדְּבָרִים הָאֵלֶּה בְּאַחֲרִית הַיָּמִים וְשַׁבְתָּ עַד יְהוָה אֱלֹהֶיךָ וְשָׁמַעְתָּ בְּקֹלוֹ:  ת  כַּד יֵיעוֹק לָךְ וְיַשְׁכְּחֻנָךְ כֹּל פִּתְגָמַיָא הָאִלֵין בְּסוֹף יוֹמַיָא וּתְתוּב לְדַחַלְתָּא דַיְיָ אֱלָהָךְ וּתְקַבֵּל לְמֵימְרֵהּ:

{לא} כִּי אֵל רַחוּם יְהוָה אֱלֹהֶיךָ לֹא יַרְפְּךָ וְלֹא יַשְׁחִיתֶךָ וְלֹא יִשְׁכַּח אֶת בְּרִית אֲבֹתֶיךָ אֲשֶׁר נִשְׁבַּע לָהֶם:  מ  כִּי אֵל רַחוּם יְהוָה אֱלֹהֶיךָ לֹא יַרְפְּךָ וְלֹא יַשְׁחִיתֶךָ וְלֹא יִשְׁכַּח אֶת בְּרִית אֲבֹתֶיךָ אֲשֶׁר נִשְׁבַּע לָהֶם:  ת  אֲרֵי אֱלָהָא רַחֲמָנָא יְיָ אֱלָהָךְ לָא יִשְׁבְּקִנָךְ וְלָא יְחַבְּלִנָךְ וְלָא יִתִּנְשֵׁי יָת קְיָמָא דַאֲבָהָתָךְ דִי קַיִים לְהוֹן:

{לב} כִּי שְׁאַל נָא לְיָמִים רִאשֹׁנִים אֲשֶׁר הָיוּ לְפָנֶיךָ לְמִן הַיּוֹם אֲשֶׁר בָּרָא אֱלֹהִים אָדָם עַל הָאָרֶץ וּלְמִקְצֵה הַשָּׁמַיִם וְעַד קְצֵה הַשָּׁמָיִם הֲנִהְיָה כַּדָּבָר הַגָּדוֹל הַזֶּה אוֹ הֲנִשְׁמַע כָּמֹהוּ:  מ  כִּי שְׁאַל נָא לְיָמִים רִאשֹׁנִים אֲשֶׁר הָיוּ לְפָנֶיךָ לְמִן הַיּוֹם אֲשֶׁר בָּרָא אֱלֹהִים אָדָם עַל הָאָרֶץ וּלְמִקְצֵה הַשָּׁמַיִם וְעַד קְצֵה הַשָּׁמָיִם הֲנִהְיָה כַּדָּבָר הַגָּדוֹל הַזֶּה אוֹ הֲנִשְׁמַע כָּמֹהוּ:  ת  אֲרֵי שְׁאֵל כְּעַן לְיוֹמַיָא קַדְמָאֵי דַהֲווֹ קֳדָמָךְ לְמִן יוֹמָא דִי בְרָא יְיָ אָדָם עַל אַרְעָא וּלְמִסְיָפֵי שְׁמַיָא וְעַד סְיָפֵי שְׁמַיָא הַהֲוָה כְּפִתְגָמָא רַבָּא הָדֵין אוֹ הַאִשְׁתְּמַע כְּוָתֵהּ:

{לג} הֲשָׁמַע עָם קוֹל אֱלֹהִים מְדַבֵּר מִתּוֹךְ הָאֵשׁ כַּאֲשֶׁר שָׁמַעְתָּ אַתָּה וַיֶּחִי:  מ  הֲשָׁמַע עָם קוֹל אֱלֹהִים מְדַבֵּר מִתּוֹךְ הָאֵשׁ כַּאֲשֶׁר שָׁמַעְתָּ אַתָּה וַיֶּחִי:  ת  הַשְׁמַע עַמָא קַל מֵימְרָא דַיְיָ מְמַלֵל מִגוֹ אֶשָׁתָא כְּמָא דִי שְׁמַעַתְּ אַתְּ וְיִתְקַיָם:

{לד} אוֹ הֲנִסָּה אֱלֹהִים לָבוֹא לָקַחַת לוֹ גוֹי מִקֶּרֶב גּוֹי בְּמַסֹּת בְּאֹתֹת וּבְמוֹפְתִים וּבְמִלְחָמָה וּבְיָד חֲזָקָה וּבִזְרוֹעַ נְטוּיָה וּבְמוֹרָאִים גְּדֹלִים כְּכֹל אֲשֶׁר עָשָׂה לָכֶם יְהוָה אֱלֹהֵיכֶם בְּמִצְרַיִם לְעֵינֶיךָ:  מ  אוֹ הֲנִסָּה אֱלֹהִים לָבוֹא לָקַחַת לוֹ גוֹי מִקֶּרֶב גּוֹי בְּמַסֹּת בְּאֹתֹת וּבְמוֹפְתִים וּבְמִלְחָמָה וּבְיָד חֲזָקָה וּבִזְרוֹעַ נְטוּיָה וּבְמוֹרָאִים גְּדֹלִים כְּכֹל אֲשֶׁר עָשָׂה לָכֶם יְהוָה אֱלֹהֵיכֶם בְּמִצְרַיִם לְעֵינֶיךָ:  ת  אוֹ נִסִין דִי עֲבַד יְיָ לְאִתְגַלָאָה לְמִפְרַק לֵהּ עַם מִגוֹ עַם בְּנִסִין בְּאָתִין וּבְמוֹפְתִין וּבִקְרָבָא וּבִידָא תַקִיפָא וּבִדְרָעָא מְרָמָא וּבְחֶזְוָנִין רַבְרְבִין כְּכֹל דִי עֲבַד לְכוֹן יְיָ אֱלָהֲכוֹן בְּמִצְרַיִם לְעֵינָיךְ:

{לה} אַתָּה הָרְאֵתָ לָדַעַת כִּי יְהוָה הוּא הָאֱלֹהִים אֵין עוֹד מִלְבַדּוֹ:  מ  אַתָּה הָרְאֵתָ לָדַעַת כִּי יְהוָה הוּא הָאֱלֹהִים אֵין עוֹד מִלְבַדּוֹ:  ת  אַתְּ אִתְחֲזֵיתָא לְמִדַע אֲרֵי יְיָ הוּא אֱלֹהִים לֵית עוֹד בַּר מִנֵהּ:

{לו} מִן הַשָּׁמַיִם הִשְׁמִיעֲךָ אֶת קֹלוֹ לְיַסְּרֶךָּ וְעַל הָאָרֶץ הֶרְאֲךָ אֶת אִשּׁוֹ הַגְּדוֹלָה וּדְבָרָיו שָׁמַעְתָּ מִתּוֹךְ הָאֵשׁ:  מ  מִן הַשָּׁמַיִם הִשְׁמִיעֲךָ אֶת קֹלוֹ לְיַסְּרֶךָּ וְעַל הָאָרֶץ הֶרְאֲךָ אֶת אִשּׁוֹ הַגְּדוֹלָה וּדְבָרָיו שָׁמַעְתָּ מִתּוֹךְ הָאֵשׁ:  ת  מִן שְׁמַיָא אַשְׁמְעָךְ יָת קָל מֵימְרֵהּ לְאַלָפוּתָךְ וְעַל אַרְעָא אַחְזְיָךְ יָת אֶשָׁתֵהּ רַבְּתָא וּפִתְגָמוֹהִי שְׁמַעַתְּ מִגוֹ אֶשָׁתָא:

{לז} וְתַחַת כִּי אָהַב אֶת אֲבֹתֶיךָ וַיִּבְחַר בְּזַרְעוֹ אַחֲרָיו וַיּוֹצִאֲךָ בְּפָנָיו בְּכֹחוֹ הַגָּדֹל מִמִּצְרָיִם:  מ  וְתַחַת כִּי אָהַב אֶת אֲבֹתֶיךָ וַיִּבְחַר בְּזַרְעוֹ אַחֲרָיו וַיּוֹצִאֲךָ בְּפָנָיו בְּכֹחוֹ הַגָּדֹל מִמִּצְרָיִם:  ת  וָחֳלַף אֲרֵי רְחִים יָת אֲבָהָתָךְ וְאִתִּרְעֵי בִּבְנוֹהִי בַּתְרוֹהִי וְאַפְּקָךְ בְּמֵימְרֵהּ בְּחֵילֵהּ רַבָּא מִמִצְרָיִם:

{לח} לְהוֹרִישׁ גּוֹיִם גְּדֹלִים וַעֲצֻמִים מִמְּךָ מִפָּנֶיךָ לַהֲבִיאֲךָ לָתֶת לְךָ אֶת אַרְצָם נַחֲלָה כַּיּוֹם הַזֶּה:  מ  לְהוֹרִישׁ גּוֹיִם גְּדֹלִים וַעֲצֻמִים מִמְּךָ מִפָּנֶיךָ לַהֲבִיאֲךָ לָתֶת לְךָ אֶת אַרְצָם נַחֲלָה כַּיּוֹם הַזֶּה:  ת  לְתָרָכָא עַמְמִין רַבְרְבִין וְתַקִיפִין מִנָךְ מִן קֳדָמָךְ לְאָעָלוּתָךְ לְמִתַּן לָךְ יָת אַרְעֲהוֹן אַחֲסָנָא כְּיוֹמָא הָדֵין:

{לט} וְיָדַעְתָּ הַיּוֹם וַהֲשֵׁבֹתָ אֶל לְבָבֶךָ כִּי יְהוָה הוּא הָאֱלֹהִים בַּשָּׁמַיִם מִמַּעַל וְעַל הָאָרֶץ מִתָּחַת אֵין עוֹד:  מ  וְיָדַעְתָּ הַיּוֹם וַהֲשֵׁבֹתָ אֶל לְבָבֶךָ כִּי יְהוָה הוּא הָאֱלֹהִים בַּשָּׁמַיִם מִמַּעַל וְעַל הָאָרֶץ מִתָּחַת אֵין עוֹד:  ת  וְתִדַע יוֹמָא דֵין וּתְתוּב לְלִבָּךְ אֲרֵי יְיָ הוּא אֱלָהָא דִשְׁכִנְתֵּהּ בִּשְׁמַיָא מִ לְעֵלָא וְשַׁלִיט עַל אַרְעָא מִלְרָע לֵית עוֹד:

{מ} וְשָׁמַרְתָּ אֶת חֻקָּיו וְאֶת מִצְוֹתָיו אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם אֲשֶׁר יִיטַב לְךָ וּלְבָנֶיךָ אַחֲרֶיךָ וּלְמַעַן תַּאֲרִיךְ יָמִים עַל הָאֲדָמָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ כָּל הַיָּמִים:  מ  וְשָׁמַרְתָּ אֶת חֻקָּיו וְאֶת מִצְוֹתָיו אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם אֲשֶׁר יִיטַב לְךָ וּלְבָנֶיךָ אַחֲרֶיךָ וּלְמַעַן תַּאֲרִיךְ יָמִים עַל הָאֲדָמָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ כָּל הַיָּמִים:  ת  וְתִטַר יָת קְיָמוֹהִי וְיָת פִּקוּדוֹהִי דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין דִי יֵיטַב לָךְ וְלִבְנָיךְ בַּתְרָךְ וּבְדִיל דְתוֹרִיךְ יוֹמִין עַל אַרְעָא דִי יְיָ אֱלָהָךְ יָהֵב לָךְ כָּל יוֹמַיָא: (פ)

{מא}  שלישי  אָז יַבְדִּיל מֹשֶׁה שָׁלֹשׁ עָרִים בְּעֵבֶר הַיַּרְדֵּן מִזְרְחָה שָׁמֶשׁ:  מ  אָז יַבְדִּיל מֹשֶׁה שָׁלֹשׁ עָרִים בְּעֵבֶר הַיַּרְדֵּן מִזְרְחָה שָׁמֶשׁ:  ת  בְּכֵן יַפְרֵשׁ מֹשֶׁה תְּלַת קִרְוִין בְּעִבְרָא דְיַרְדְנָא מִדְנַח שִׁמְשָׁא:

{מב} לָנֻס שָׁמָּה רוֹצֵחַ אֲשֶׁר יִרְצַח אֶת רֵעֵהוּ בִּבְלִי דַעַת וְהוּא לֹא שֹׂנֵא לוֹ מִתְּמוֹל שִׁלְשׁוֹם וְנָס אֶל אַחַת מִן הֶעָרִים הָאֵל וָחָי:  מ  לָנֻס שָׁמָּה רוֹצֵחַ אֲשֶׁר יִרְצַח אֶת רֵעֵהוּ בִּבְלִי דַעַת וְהוּא לֹא שֹׂנֵא לוֹ מִתְּמוֹל שִׁלְשׁוֹם וְנָס אֶל אַחַת מִן הֶעָרִים הָאֵל וָחָי:  ת  לְמֵעִירוֹק תַּמָן קָטוֹלָא דִי יִקְטוֹל יָת חַבְרֵהּ בְּלָא מַנְדְעֵי וְהוּא לָא סָנֵי לֵהּ מֵאִתְמַלֵי וּמִדְקַמוֹהִי וְיֵעִרוֹק לְחַד מִן קִרְוַיָא הָאִלֵין וְיִתְקַיָם:

{מג} אֶת בֶּצֶר בַּמִּדְבָּר בְּאֶרֶץ הַמִּישֹׁר לָראוּבֵנִי וְאֶת רָאמֹת בַּגִּלְעָד לַגָּדִי וְאֶת גּוֹלָן בַּבָּשָׁן לַמְנַשִּׁי:  מ  אֶת בֶּצֶר בַּמִּדְבָּר בְּאֶרֶץ הַמִּישֹׁר לָראוּבֵנִי וְאֶת רָאמֹת בַּגִּלְעָד לַגָּדִי וְאֶת גּוֹלָן בַּבָּשָׁן לַמְנַשִּׁי:  ת  יָת בֶּצֶר בְּמַדְבְּרָא בְּאַרְעָא מֵישְׁרָא לְשִׁבְטָּא דִרְאוּבֵן וְיָת רָאמוֹת בַּגִלְעָד לְשִׁבְטָא דְגָד וְיָת גוֹלָן בְּמַתְנָן לְשִׁבְטָא דִמְנַשֶׁה:

{מד} וְזֹאת הַתּוֹרָה אֲשֶׁר שָׂם מֹשֶׁה לִפְנֵי בְּנֵי יִשְׂרָאֵל:  מ  וְזֹאת הַתּוֹרָה אֲשֶׁר שָׂם מֹשֶׁה לִפְנֵי בְּנֵי יִשְׂרָאֵל:  ת  וְדָא אוֹרַיְתָא דִי סַדַר מֹשֶׁה קֳדָם בְּנֵי יִשְׂרָאֵל:

{מה} אֵלֶּה הָעֵדֹת וְהַחֻקִּים וְהַמִּשְׁפָּטִים אֲשֶׁר דִּבֶּר מֹשֶׁה אֶל בְּנֵי יִשְׂרָאֵל בְּצֵאתָם מִמִּצְרָיִם:  מ  אֵלֶּה הָעֵדֹת וְהַחֻקִּים וְהַמִּשְׁפָּטִים אֲשֶׁר דִּבֶּר מֹשֶׁה אֶל בְּנֵי יִשְׂרָאֵל בְּצֵאתָם מִמִּצְרָיִם:  ת  אִלֵין סָהֶדְוָתָא וּקְיָמַיָא וְדִינַיָא דִי מַלִיל מֹשֶׁה עִם בְּנֵי יִשְׂרָאֵל בְּמִפָּקְהוֹן מִמִצְרָיִם:

{מו} בְּעֵבֶר הַיַּרְדֵּן בַּגַּיְא מוּל בֵּית פְּעוֹר בְּאֶרֶץ סִיחֹן מֶלֶךְ הָאֱמֹרִי אֲשֶׁר יוֹשֵׁב בְּחֶשְׁבּוֹן אֲשֶׁר הִכָּה מֹשֶׁה וּבְנֵי יִשְׂרָאֵל בְּצֵאתָם מִמִּצְרָיִם:  מ  בְּעֵבֶר הַיַּרְדֵּן בַּגַּיְא מוּל בֵּית פְּעוֹר בְּאֶרֶץ סִיחֹן מֶלֶךְ הָאֱמֹרִי אֲשֶׁר יוֹשֵׁב בְּחֶשְׁבּוֹן אֲשֶׁר הִכָּה מֹשֶׁה וּבְנֵי יִשְׂרָאֵל בְּצֵאתָם מִמִּצְרָיִם:  ת  בְּעִבְרָא דְיַרְדְנָא בְּחֵילָתָא לָקֳבֵל בֵּית פְּעוֹר בְּאַרְעָא דְסִיחֹן מַלְכָּא דֶאֱמֹרָאָה דִי יָתֵב בְּחֶשְׁבּוֹן דִי מְחָא מֹשֶׁה וּבְנֵי יִשְׂרָאֵל בְּמִפָּקְהוֹן מִמִצְרָיִם:

{מז} וַיִּירְשׁוּ אֶת אַרְצוֹ וְאֶת אֶרֶץ עוֹג מֶלֶךְ הַבָּשָׁן שְׁנֵי מַלְכֵי הָאֱמֹרִי אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן מִזְרַח שָׁמֶשׁ:  מ  וַיִּירְשׁוּ אֶת אַרְצוֹ וְאֶת אֶרֶץ עוֹג מֶלֶךְ הַבָּשָׁן שְׁנֵי מַלְכֵי הָאֱמֹרִי אֲשֶׁר בְּעֵבֶר הַיַּרְדֵּן מִזְרַח שָׁמֶשׁ:  ת  וִירִיתוּ יָת אַרְעֵהּ וְיָת אַרְעָא דְעוֹג מַלְכָּא דְמַתְנָן תְּרֵין מַלְכֵי אֱמֹרָאָה דִי בְּעִבְרָא דְיַרְדְנָא מִדְנַח שִׁמְשָׁא:

{מח} מֵעֲרֹעֵר אֲשֶׁר עַל שְׂפַת נַחַל אַרְנֹן וְעַד הַר שִׂיאֹן הוּא חֶרְמוֹן:  מ  מֵעֲרֹעֵר אֲשֶׁר עַל שְׂפַת נַחַל אַרְנֹן וְעַד הַר שִׂיאֹן הוּא חֶרְמוֹן:  ת  מֵעֲרֹעֵר דִי עַל כֵּיף נַחֲלָא דְאַרְנֹן וְעַד טוּרָא דְשִׂיאֹן הוּא חֶרְמוֹן:

{מט} וְכָל הָעֲרָבָה עֵבֶר הַיַּרְדֵּן מִזְרָחָה וְעַד יָם הָעֲרָבָה תַּחַת אַשְׁדֹּת הַפִּסְגָּה:  מ  וְכָל הָעֲרָבָה עֵבֶר הַיַּרְדֵּן מִזְרָחָה וְעַד יָם הָעֲרָבָה תַּחַת אַשְׁדֹּת הַפִּסְגָּה:  ת  וְכָל מֵישְׁרָא עִבְרָא דְיַרְדְנָא לְמָדִינְחָא וְעַד יַמָא דְמֵישְׁרָא תְּחוֹת מַשְׁפַּךְ מְרָמָתָא:




דברים פרק-ה

{א}  רביעי  וַיִּקְרָא מֹשֶׁה אֶל כָּל יִשְׂרָאֵל וַיֹּאמֶר אֲלֵהֶם שְׁמַע יִשְׂרָאֵל אֶת הַחֻקִּים וְאֶת הַמִּשְׁפָּטִים אֲשֶׁר אָנֹכִי דֹּבֵר בְּאָזְנֵיכֶם הַיּוֹם וּלְמַדְתֶּם אֹתָם וּשְׁמַרְתֶּם לַעֲשֹׂתָם:  מ  וַיִּקְרָא מֹשֶׁה אֶל כָּל יִשְׂרָאֵל וַיֹּאמֶר אֲלֵהֶם שְׁמַע יִשְׂרָאֵל אֶת הַחֻקִּים וְאֶת הַמִּשְׁפָּטִים אֲשֶׁר אָנֹכִי דֹּבֵר בְּאָזְנֵיכֶם הַיּוֹם וּלְמַדְתֶּם אֹתָם וּשְׁמַרְתֶּם לַעֲשֹׂתָם:  ת  וּקְרָא מֹשֶׁה לְכָל יִשְׂרָאֵל וַאֲמַר לְהוֹן שְׁמַע יִשְׂרָאֵל יָת קְיָמַיָא וְיָת דִינַיָא דִי אֲנָא מְמַלֵל קֳדָמֵיכוֹן יוֹמָא דֵין וְתַלְפוּן יָתְהוֹן וְתִטְרוּן לְמֶעְבָּדְהוֹן:

{ב} יְהוָה אֱלֹהֵינוּ כָּרַת עִמָּנוּ בְּרִית בְּחֹרֵב:  מ  יְהוָה אֱלֹהֵינוּ כָּרַת עִמָּנוּ בְּרִית בְּחֹרֵב:  ת  יְיָ אֱלָהָנָא גְזַר עִמָנָא קְיָם בְּחֹרֵב:

{ג} לֹא אֶת אֲבֹתֵינוּ כָּרַת יְהוָה אֶת הַבְּרִית הַזֹּאת כִּי אִתָּנוּ אֲנַחְנוּ אֵלֶּה פֹה הַיּוֹם כֻּלָּנוּ חַיִּים:  מ  לֹא אֶת אֲבֹתֵינוּ כָּרַת יְהוָה אֶת הַבְּרִית הַזֹּאת כִּי אִתָּנוּ אֲנַחְנוּ אֵלֶּה פֹה הַיּוֹם כֻּלָּנוּ חַיִּים:  ת  לָא עִם אֲבְהָתָנָא גְזַר יְיָ יָת קְיָמָא הָדָא אֶלָהֵן עִמָנָא אֲנַחְנָא אִלֵין הָכָא יוֹמָא דֵין כֻּלָנָא קַיָמִין:

{ד} פָּנִים בְּפָנִים דִּבֶּר יְהוָה עִמָּכֶם בָּהָר מִתּוֹךְ הָאֵשׁ:  מ  פָּנִים בְּפָנִים דִּבֶּר יְהוָה עִמָּכֶם בָּהָר מִתּוֹךְ הָאֵשׁ:  ת  מַמְלָל עִם מַמְלָל מַלִיל יְיָ עִמְכוֹן בְּטוּרָא מִגוֹ אֶשָׁתָא:

{ה} אָנֹכִי עֹמֵד בֵּין יְהוָה וּבֵינֵיכֶם בָּעֵת הַהִוא לְהַגִּיד לָכֶם אֶת דְּבַר יְהוָה כִּי יְרֵאתֶם מִפְּנֵי הָאֵשׁ וְלֹא עֲלִיתֶם בָּהָר לֵאמֹר:  מ  אָנֹכִי עֹמֵד בֵּין יְהוָה וּבֵינֵיכֶם בָּעֵת הַהִוא לְהַגִּיד לָכֶם אֶת דְּבַר יְהוָה כִּי יְרֵאתֶם מִפְּנֵי הָאֵשׁ וְלֹא עֲלִיתֶם בָּהָר לֵאמֹר:  ת  אֲנָא הֲוֵיתִי קָאֵם בֵּין מֵימְרָא דַיְיָ וּבֵינֵיכוֹן בְּעִדָנָא הַהִיא לְחַוָאָה לְכוֹן יָת פִּתְגָמָא דַיְיָ אֲרֵי דְחֶלְתּוּן מִקֳדָם אֵשָׁתָא וְלָא סְלֶקְתּוּן בְּטוּרָא לְמֵימָר: (ס)

{ו} אָנֹכִי יְהוָה אֱלֹהֶיךָ אֲשֶׁר הוֹצֵאתִיךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים:  מ  אָנֹכִי יְהוָה אֱלֹהֶיךָ אֲשֶׁר הוֹצֵאתִיךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים:  ת  אֲנָא יְיָ אֱלָהָךְ דִי אַפֵּקְתָּךְ מֵאַרְעָא דְמִצְרַיִם מִבֵּית עֲבְדוּתָא:

{ז} לֹא יִהְיֶה לְךָ אֱלֹהִים אֲחֵרִים עַל פָּנָיַ:  מ  לֹא יִהְיֶה לְךָ אֱלֹהִים אֲחֵרִים עַל פָּנָיַ:  ת  לָא יְהֵא לָךְ אֱלָהּ אָחָרָן בַּר מִנִי:

{ח} לֹא תַעֲשֶׂה לְךָ פֶסֶל כָּל תְּמוּנָה אֲשֶׁר בַּשָּׁמַיִם מִמַּעַל וַאֲשֶׁר בָּאָרֶץ מִתָּחַת וַאֲשֶׁר בַּמַּיִם מִתַּחַת לָאָרֶץ:  מ  לֹא תַעֲשֶׂה לְךָ פֶסֶל כָּל תְּמוּנָה אֲשֶׁר בַּשָּׁמַיִם מִמַּעַל וַאֲשֶׁר בָּאָרֶץ מִתָּחַת וַאֲשֶׁר בַּמַּיִם מִתַּחַת לָאָרֶץ:  ת  לָא תַעְבֵּד לָךְ צֶלֶם כָּל דְמוּת דִי בִשְׁמַיָא מִלְעֵלָא וְדִי בְאַרְעָא מִלְרָע וְדִי בְמַיָא מִלְרַע לְאַרְעָא:

{ט} לֹא תִשְׁתַּחֲוֶה לָהֶם וְלֹא תָעָבְדֵם כִּי אָנֹכִי יְהוָה אֱלֹהֶיךָ אֵל קַנָּא פֹּקֵד עֲוֹן אָבוֹת עַל בָּנִים וְעַל שִׁלֵּשִׁים וְעַל רִבֵּעִים לְשֹׂנְאָי:  מ  לֹא תִשְׁתַּחֲוֶה לָהֶם וְלֹא תָעָבְדֵם כִּי אָנֹכִי יְהוָה אֱלֹהֶיךָ אֵל קַנָּא פֹּקֵד עֲוֹן אָבוֹת עַל בָּנִים וְעַל שִׁלֵּשִׁים וְעַל רִבֵּעִים לְשֹׂנְאָי:  ת  לָא תִסְגוּד לְהוֹן וְלָא תִפְלְחִנוּן אֲרֵי אֲנָא יְיָ אֱלָהָךְ אֵל קַנָא מַסְעַר חוֹבֵי אֲבָהָן עַל בְּנִין מְרָדִין עַל דַר תְּלִיתַי וְעַל דַר רְבִיעַי לְשַׂנְאָי כַּד מְשַׁלְמִין בְּנַיָא לְמֶחֱטֵּי בָּתַר אֲבָהָתְהוֹן:

{י} וְעֹשֶׂה חֶסֶד לַאֲלָפִים לְאֹהֲבַי וּלְשֹׁמְרֵי (מצותו) מִצְוֹתָי:  מ  וְעֹשֶׂה חֶסֶד לַאֲלָפִים לְאֹהֲבַי וּלְשֹׁמְרֵי (מצותו) מִצְוֹתָי:  ת  וְעָבֵד טֵּיבוּ לְאַלְפֵי דָרִין לְרַחֲמַי וּלְנָטְּרֵי פִקוֹדָי: (ס)

{יא} לֹא תִשָּׂא אֶת שֵׁם יְהוָה אֱלֹהֶיךָ לַשָּׁוְא כִּי לֹא יְנַקֶּה יְהוָה אֵת אֲשֶׁר יִשָּׂא אֶת שְׁמוֹ לַשָּׁוְא:  מ  לֹא תִשָּׂא אֶת שֵׁם יְהוָה אֱלֹהֶיךָ לַשָּׁוְא כִּי לֹא יְנַקֶּה יְהוָה אֵת אֲשֶׁר יִשָּׂא אֶת שְׁמוֹ לַשָּׁוְא:  ת  לָא תֵימֵי בִּשְׁמָא דַיְיָ אֱלָהָךְ לְמַגָנָא אֲרֵי לָא יְזַכֵּי יְיָ יָת דִי יֵמֵי בִשְׁמֵהּ לְשִׁקְרָא: (ס)

{יב} שָׁמוֹר אֶת יוֹם הַשַּׁבָּת לְקַדְּשׁוֹ כַּאֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ  מ  שָׁמוֹר אֶת יוֹם הַשַּׁבָּת לְקַדְּשׁוֹ כַּאֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ  ת  טַר יָת יוֹמָא דְשַׁבְּתָא לְקַדָשׁוּתֵהּ כְּמָא דִי פַקְדָךְ יְיָ אֱלָהָךְ:

{יג} שֵׁשֶׁת יָמִים תַּעֲבֹד וְעָשִׂיתָ כָּֿל מְלַאכְתֶּךָ:  מ  שֵׁשֶׁת יָמִים תַּעֲבֹד וְעָשִׂיתָ כָּֿל מְלַאכְתֶּךָ:  ת  שִׁתָּא יוֹמִין תִּפְלָח וְתַעְבֵּד כָּל עִבִדְתָּךְ:

{יד} וְיוֹם הַשְּׁבִיעִי שַׁבָּת לַיהוָה אֱלֹהֶיךָ לֹא תַעֲשֶׂה כָל מְלָאכָה אַתָּה וּבִנְךָ וּבִתֶּךָ וְעַבְדְּךָ וַאֲמָתֶךָ וְשׁוֹרְךָ וַחֲמֹרְךָ וְכָל בְּהֶמְתֶּךָ וְגֵרְךָ אֲשֶׁר בִּשְׁעָרֶיךָ לְמַעַן יָנוּחַ עַבְדְּךָ וַאֲמָתְךָ כָּמוֹךָ:  מ  וְיוֹם הַשְּׁבִיעִי שַׁבָּת לַיהוָה אֱלֹהֶיךָ לֹא תַעֲשֶׂה כָל מְלָאכָה אַתָּה וּבִנְךָ וּבִתֶּךָ וְעַבְדְּךָ וַאֲמָתֶךָ וְשׁוֹרְךָ וַחֲמֹרְךָ וְכָל בְּהֶמְתֶּךָ וְגֵרְךָ אֲשֶׁר בִּשְׁעָרֶיךָ לְמַעַן יָנוּחַ עַבְדְּךָ וַאֲמָתְךָ כָּמוֹךָ:  ת  וְיוֹמָא שְׁבִיעָאָה שַׁבְּתָא קֳדָם יְיָ אֱלָהָךְ לָא תַעְבֵּד כָּל עֲבִידָא אַתְּ וּבְרָךְ וּבְרַתָּךְ וְעַבְדָךְ וְאַמְתָךְ וְתוֹרָךְ וַחֲמָרָךְ וְכָל בְּעִירָךְ וְגִיוֹרָךְ דִי בְקִרְוָךְ בְּדִיל דִי יְנוּחַ עַבְדָךְ וְאַמְתָךְ כְּוָתָךְ:

{טו} וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרַיִם וַיֹּצִאֲךָ יְהוָה אֱלֹהֶיךָ מִשָּׁם בְּיָד חֲזָקָה וּבִזְרֹעַ נְטוּיָה עַל כֵּן צִוְּךָ יְהוָה אֱלֹהֶיךָ לַעֲשׂוֹת אֶת יוֹם הַשַּׁבָּת:  מ  וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרַיִם וַיֹּצִאֲךָ יְהוָה אֱלֹהֶיךָ מִשָּׁם בְּיָד חֲזָקָה וּבִזְרֹעַ נְטוּיָה עַל כֵּן צִוְּךָ יְהוָה אֱלֹהֶיךָ לַעֲשׂוֹת אֶת יוֹם הַשַּׁבָּת:  ת  וְתִדְכַּר אֲרֵי עַבְדָא הֲוֵיתָא בְּאַרְעָא דְמִצְרַיִם וְאַפְּקָךְ יְיָ אֱלָהָךְ מִתַּמָן בִּידָא תַקִיפָא וּבִדְרָעָא מְרָמְמָא עַל כֵּן פַּקְדָךְ יְיָ אֱלָהָךְ לְמֶעְבַּד יָת יוֹמָא דְשַׁבְּתָא: (ס)

{טז}  כַּבֵּד אֶת אָבִיךָ וְאֶת אִמֶּךָ כַּאֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ לְמַעַן יַאֲרִיכֻן יָמֶיךָ וּלְמַעַן יִיטַב לָךְ עַל הָאֲדָמָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:  מ   כַּבֵּד אֶת אָבִיךָ וְאֶת אִמֶּךָ כַּאֲשֶׁר צִוְּךָ יְהוָה אֱלֹהֶיךָ לְמַעַן יַאֲרִיכֻן יָמֶיךָ וּלְמַעַן יִיטַב לָךְ עַל הָאֲדָמָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ:  ת   יַקַר יָת אֲבוּךְ וְיָת אִמָךְ כְּמָא דִי פַקְדָךְ יְיָ אֱלָהָךְ בְּדִיל דְיוֹרְכוּן יוֹמָיךְ וּבְדִיל דְיֵיטַב לָךְ עַל אַרְעָא דַיְיָ אֱלָהָךְ יָהֵב לָךְ: (ס)

{יז}  לֹא תִּֿרְצָח: וְלֹא תִּֿנְאָף: וְלֹא תִּֿגְנֹב: וְלֹא תַעֲנֶה בְרֵעֲךָ עֵד שָׁוְא:  מ   לֹא תִּֿרְצָח: וְלֹא תִּֿנְאָף: וְלֹא תִּֿגְנֹב: וְלֹא תַעֲנֶה בְרֵעֲךָ עֵד שָׁוְא:  ת   לָא תִקְטוֹל נְפָשׁ וְלָא תְגוּף וְלָא תִגְנוֹב (נַפְשָׁא) וְלָא תַסְהֵד בְּחַבְרָךְ סַהֲדוּתָא דְשִׁקְרָא: (ס)

{יח}  וְלֹא תַחְמֹד אֵשֶׁת רֵעֶךָ וְלֹא תִתְאַוֶּה בֵּית רֵעֶךָ שָׂדֵהוּ וְעַבְדּוֹ וַאֲמָתוֹ שׁוֹרוֹ וַחֲמֹרוֹ וְכֹל אֲשֶׁר לְרֵעֶךָ:  מ   וְלֹא תַחְמֹד אֵשֶׁת רֵעֶךָ וְלֹא תִתְאַוֶּה בֵּית רֵעֶךָ שָׂדֵהוּ וְעַבְדּוֹ וַאֲמָתוֹ שׁוֹרוֹ וַחֲמֹרוֹ וְכֹל אֲשֶׁר לְרֵעֶךָ:  ת   וְלָא תַחְמֵד אִתַּת חַבְרָךְ וְלָא תֵרוֹג בֵּית חַבְרָךְ חַקְלֵהּ וְעַבְדֵהּ וְאַמְתֵהּ תּוֹרֵה וַחֲמָרֵהּ וְכֹל דִי לְחַבְרָךְ:

{יט}  חמישי  אֶת הַדְּבָרִים הָאֵלֶּה דִּבֶּר יְהוָה אֶל כָּל קְהַלְכֶם בָּהָר מִתּוֹךְ הָאֵשׁ הֶעָנָן וְהָעֲרָפֶל קוֹל גָּדוֹל וְלֹא יָסָף וַיִּכְתְּבֵם עַל שְׁנֵי לֻחֹת אֲבָנִים וַיִּתְּנֵם אֵלָי:  מ  אֶת הַדְּבָרִים הָאֵלֶּה דִּבֶּר יְהוָה אֶל כָּל קְהַלְכֶם בָּהָר מִתּוֹךְ הָאֵשׁ הֶעָנָן וְהָעֲרָפֶל קוֹל גָּדוֹל וְלֹא יָסָף וַיִּכְתְּבֵם עַל שְׁנֵי לֻחֹת אֲבָנִים וַיִּתְּנֵם אֵלָי:  ת  יָת פִּתְגָמַיָא הָאִלֵין מַלִיל יְיָ עִם כָּל קְהָלְכוֹן בְּטוּרָא מִגוֹ אֶשָׁתָא עֲנָנָא וַאֲמִיטְתָא קָל רַב וְלָא פְסַק וּכְתָבִנוּן עַל תְּרֵין לוּחֵי אַבְנַיָא וִיהָבִנָן לִי:

{כ} וַיְהִי כְּשָׁמְעֲכֶם אֶת הַקּוֹל מִתּוֹךְ הַחֹשֶׁךְ וְהָהָר בֹּעֵר בָּאֵשׁ וַתִּקְרְבוּן אֵלַי כָּל רָאשֵׁי שִׁבְטֵיכֶם וְזִקְנֵיכֶם:  מ  וַיְהִי כְּשָׁמְעֲכֶם אֶת הַקּוֹל מִתּוֹךְ הַחֹשֶׁךְ וְהָהָר בֹּעֵר בָּאֵשׁ וַתִּקְרְבוּן אֵלַי כָּל רָאשֵׁי שִׁבְטֵיכֶם וְזִקְנֵיכֶם:  ת  וַהֲוָה כַּד שְׁמַעֲכוֹן יָת קָלָא מִגוֹ חֲשׁוֹכָא וְטוּרָא בָּעֵר בְּאֶשָׁתָא וְקָרֶבְתּוּן לְוָתִי כָּל רֵישֵׁי שִׁבְטֵיכוֹן וְסָבֵיכוֹן:

{כא} וַתֹּאמְרוּ הֵן הֶרְאָנוּ יְהוָה אֱלֹהֵינוּ אֶת כְּבֹדוֹ וְאֶת גָּדְלוֹ וְאֶת קֹלוֹ שָׁמַעְנוּ מִתּוֹךְ הָאֵשׁ הַיּוֹם הַזֶּה רָאִינוּ כִּי יְדַבֵּר אֱלֹהִים אֶת הָאָדָם וָחָי:  מ  וַתֹּאמְרוּ הֵן הֶרְאָנוּ יְהוָה אֱלֹהֵינוּ אֶת כְּבֹדוֹ וְאֶת גָּדְלוֹ וְאֶת קֹלוֹ שָׁמַעְנוּ מִתּוֹךְ הָאֵשׁ הַיּוֹם הַזֶּה רָאִינוּ כִּי יְדַבֵּר אֱלֹהִים אֶת הָאָדָם וָחָי:  ת  וַאֲמַרְתּוּן הָא אַחְזְיָנָא יְיָ אֱלָהָנָא יָת יְקָרֵהּ וְיָת רְבוּתֵהּ וְיָת קָל מֵימְרֵהּ שְׁמַעְנָא מִגוֹ אֶשָׁתָא יוֹמָא הָדֵין חֲזֵינָא אֲרֵי יְמַלֵל יְיָ עִם אֲנָשָׁא וְיִתְקַיָם:

{כב} וְעַתָּה לָמָּה נָמוּת כִּי תֹאכְלֵנוּ הָאֵשׁ הַגְּדֹלָה הַזֹּאת אִם יֹסְפִים אֲנַחְנוּ לִשְׁמֹעַ אֶת קוֹל יְהוָה אֱלֹהֵינוּ עוֹד וָמָתְנוּ:  מ  וְעַתָּה לָמָּה נָמוּת כִּי תֹאכְלֵנוּ הָאֵשׁ הַגְּדֹלָה הַזֹּאת אִם יֹסְפִים אֲנַחְנוּ לִשְׁמֹעַ אֶת קוֹל יְהוָה אֱלֹהֵינוּ עוֹד וָמָתְנוּ:  ת  וּכְעַן לְמָא נְמוּת אֲרֵי תֵיכְלִנָנָא אֶשָׁתָא רַבְּתָא הָדָא אִם מוֹסְפִין אֲנַחְנָא לְמִשְׁמַע יָת קָל מֵימְרָא דַייָ אֱלָהָנָא עוֹד וּמָיְתִין אֲנַחְנָא:

{כג} כִּי מִי כָל בָּשָׂר אֲשֶׁר שָׁמַע קוֹל אֱלֹהִים חַיִּים מְדַבֵּר מִתּוֹךְ הָאֵשׁ כָּמֹנוּ וַיֶּחִי:  מ  כִּי מִי כָל בָּשָׂר אֲשֶׁר שָׁמַע קוֹל אֱלֹהִים חַיִּים מְדַבֵּר מִתּוֹךְ הָאֵשׁ כָּמֹנוּ וַיֶּחִי:  ת  אֲרֵי מָן כָּל בִּשְׂרָא דִי שְׁמַע קָל מֵימְרָא דַיְיָ קַיָמָא מְמַלֵל מִגוֹ אֶשָׁתָא כְּוָתָנָא וְאִתְקַיָם:

{כד} קְרַב אַתָּה וּשֲׁמָע אֵת כָּל אֲשֶׁר יֹאמַר יְהוָה אֱלֹהֵינוּ וְאַתְּ תְּדַבֵּר אֵלֵינוּ אֵת כָּל אֲשֶׁר יְדַבֵּר יְהוָה אֱלֹהֵינוּ אֵלֶיךָ וְשָׁמַעְנוּ וְעָשִׂינוּ:  מ  קְרַב אַתָּה וּשֲׁמָע אֵת כָּל אֲשֶׁר יֹאמַר יְהוָה אֱלֹהֵינוּ וְאַתְּ תְּדַבֵּר אֵלֵינוּ אֵת כָּל אֲשֶׁר יְדַבֵּר יְהוָה אֱלֹהֵינוּ אֵלֶיךָ וְשָׁמַעְנוּ וְעָשִׂינוּ:  ת  קְרַב אַתְּ וּשְׁמַע יָת כָּל דִי יֵימַר יְיָ אֱלָהָנָא וְאַתְּ תְּמַלֵל עִמָנָא יָת כָּל דִי מַלֵל יְיָ אֱלָהָנָא עִמָךְ וּנְקַבֵּל וְנַעְבֵּד:

{כה} וַיִּשְׁמַע יְהוָה אֶת קוֹל דִּבְרֵיכֶם בְּדַבֶּרְכֶם אֵלָי וַיֹּאמֶר יְהוָה אֵלַי שָׁמַעְתִּי אֶת קוֹל דִּבְרֵי הָעָם הַזֶּה אֲשֶׁר דִּבְּרוּ אֵלֶיךָ הֵיטִיבוּ כָּל אֲשֶׁר דִּבֵּרוּ:  מ  וַיִּשְׁמַע יְהוָה אֶת קוֹל דִּבְרֵיכֶם בְּדַבֶּרְכֶם אֵלָי וַיֹּאמֶר יְהוָה אֵלַי שָׁמַעְתִּי אֶת קוֹל דִּבְרֵי הָעָם הַזֶּה אֲשֶׁר דִּבְּרוּ אֵלֶיךָ הֵיטִיבוּ כָּל אֲשֶׁר דִּבֵּרוּ:  ת  וּשְׁמִיעַ קֳדָם יְיָ יָת קָל פִּתְגָמֵיכוֹן בְּמַלָלוּתְכוֹן עִמִי וַאֲמַר יְיָ לִי שְׁמִיעַ קֳדָמַי יָת קָל פִּתְגָמֵי עַמָא הָדֵין דִי מַלִילוּ עִמָךְ אַתְקִינוּ כָּל דִי מַלִילוּ:

{כו} מִי יִתֵּן וְהָיָה לְבָבָם זֶה לָהֶם לְיִרְאָה אֹתִי וְלִשְׁמֹר אֶת כָּל מִצְוֹתַי כָּל הַיָּמִים לְמַעַן יִיטַב לָהֶם וְלִבְנֵיהֶם לְעֹלָם:  מ  מִי יִתֵּן וְהָיָה לְבָבָם זֶה לָהֶם לְיִרְאָה אֹתִי וְלִשְׁמֹר אֶת כָּל מִצְוֹתַי כָּל הַיָּמִים לְמַעַן יִיטַב לָהֶם וְלִבְנֵיהֶם לְעֹלָם:  ת  לְוַי דִי יְהֵי לִבָּא הָדֵין לְהוֹן לְמִדְחַל קֳדָמַי וּלְמִטַר יָת כָּל פִּקוּדַי כָּל יוֹמַיָא בְּדִיל דִיֵיטַב לְהוֹן וְלִבְנֵיהוֹן לְעָלָם:

{כז} לֵךְ אֱמֹר לָהֶם שׁוּבוּ לָכֶם לְאָהֳלֵיכֶם:  מ  לֵךְ אֱמֹר לָהֶם שׁוּבוּ לָכֶם לְאָהֳלֵיכֶם:  ת  אִזֵל אֱמַר לְהוֹן תּוּבוּ לְכוֹן לְמַשְׁכְּנֵיכוֹן:

{כח} וְאַתָּה פֹּה עֲמֹד עִמָּדִי וַאֲדַבְּרָה אֵלֶיךָ אֵת כָּל הַמִּצְוָה וְהַחֻקִּים וְהַמִּשְׁפָּטִים אֲשֶׁר תְּלַמְּדֵם וְעָשׂוּ בָאָרֶץ אֲשֶׁר אָנֹכִי נֹתֵן לָהֶם לְרִשְׁתָּהּ:  מ  וְאַתָּה פֹּה עֲמֹד עִמָּדִי וַאֲדַבְּרָה אֵלֶיךָ אֵת כָּל הַמִּצְוָה וְהַחֻקִּים וְהַמִּשְׁפָּטִים אֲשֶׁר תְּלַמְּדֵם וְעָשׂוּ בָאָרֶץ אֲשֶׁר אָנֹכִי נֹתֵן לָהֶם לְרִשְׁתָּהּ:  ת  וְאַתְּ הָכָא קִים קֳדָמַי וֶאֱמַלֵל עִמָךְ יָת כָּלּ תַּפְקֶדְתָּא וּקְיָמַיָא וְדִינַיָא דִי תַלְפִנוּן וְיַעְבְּדוּן בְּאַרְעָא דִי אֲנָא יָהֵב לְהוֹן לְמֵירְתַהּ:

{כט} וּשְׁמַרְתֶּם לַעֲשׂוֹת כַּאֲשֶׁר צִוָּה יְהוָה אֱלֹהֵיכֶם אֶתְכֶם לֹא תָסֻרוּ יָמִין וּשְׂמֹאל:  מ  וּשְׁמַרְתֶּם לַעֲשׂוֹת כַּאֲשֶׁר צִוָּה יְהוָה אֱלֹהֵיכֶם אֶתְכֶם לֹא תָסֻרוּ יָמִין וּשְׂמֹאל:  ת  וְתִטְרוּן לְמֶעְבַּד כְּמָא דִי פַקִיד יְיָ אֱלָהֲכוֹן יָתְכוֹן לָא תִסְטוּן יַמִינָא וּשְׂמָאלָא:

{ל} בְּכָל הַדֶּרֶךְ אֲשֶׁר צִוָּה יְהוָה אֱלֹהֵיכֶם אֶתְכֶם תֵּלֵכוּ לְמַעַן תִּחְיוּן וְטוֹב לָכֶם וְהַאֲרַכְתֶּם יָמִים בָּאָרֶץ אֲשֶׁר תִּירָשׁוּן:  מ  בְּכָל הַדֶּרֶךְ אֲשֶׁר צִוָּה יְהוָה אֱלֹהֵיכֶם אֶתְכֶם תֵּלֵכוּ לְמַעַן תִּחְיוּן וְטוֹב לָכֶם וְהַאֲרַכְתֶּם יָמִים בָּאָרֶץ אֲשֶׁר תִּירָשׁוּן:  ת  בְּכָל אָרְחָא דִי פַקִיד יְיָ אֱלָהֲכוֹן יָתְכוֹן תְּהָכוּן בְּדִיל דְתֵיחוּן וְיֵיטַב לְכוֹן וְתוֹרְכוּן יוֹמִין בְּאַרְעָא דִי תֵירְתוּן:




דברים פרק-ו

{א} וְזֹאת הַמִּצְוָה הַחֻקִּים וְהַמִּשְׁפָּטִים אֲשֶׁר צִוָּה יְהוָה אֱלֹהֵיכֶם לְלַמֵּד אֶתְכֶם לַעֲשׂוֹת בָּאָרֶץ אֲשֶׁר אַתֶּם עֹבְרִים שָׁמָּה לְרִשְׁתָּהּ:  מ  וְזֹאת הַמִּצְוָה הַחֻקִּים וְהַמִּשְׁפָּטִים אֲשֶׁר צִוָּה יְהוָה אֱלֹהֵיכֶם לְלַמֵּד אֶתְכֶם לַעֲשׂוֹת בָּאָרֶץ אֲשֶׁר אַתֶּם עֹבְרִים שָׁמָּה לְרִשְׁתָּהּ:  ת  וְדָא תַפְקֶדְתָּא קְיָמַיָא וְדִינַיָא דִי פַקִיד יְיָ אֱלָהֲכוֹן לְאַלָפָא יָתְכוֹן לְמֶעְבַּד בְּאַרְעָא דִי אַתּוּן עָבְרִין תַּמָן לְמֵירְתַהּ:

{ב} לְמַעַן תִּירָא אֶת יְהוָה אֱלֹהֶיךָ לִשְׁמֹר אֶת כָּל חֻקֹּתָיו וּמִצְוֹתָיו אֲשֶׁר אָנֹכִי מְצַוֶּךָ אַתָּה וּבִנְךָ וּבֶן בִּנְךָ כֹּל יְמֵי חַיֶּיךָ וּלְמַעַן יַאֲרִכֻן יָמֶיךָ:  מ  לְמַעַן תִּירָא אֶת יְהוָה אֱלֹהֶיךָ לִשְׁמֹר אֶת כָּל חֻקֹּתָיו וּמִצְוֹתָיו אֲשֶׁר אָנֹכִי מְצַוֶּךָ אַתָּה וּבִנְךָ וּבֶן בִּנְךָ כֹּל יְמֵי חַיֶּיךָ וּלְמַעַן יַאֲרִכֻן יָמֶיךָ:  ת  בְּדִיל דְתִדְחַל קֳדָם יְיָ אֱלָהָךְ לְמִטַר יָת כָּל קְיָמוֹהִי וּפִקוּדוֹהִי דִי אֲנָא מְפַקְדָךְ אַתְּ וּבְרָךְ וּבַר בְּרָךְ כָּל יוֹמֵי חַיָיךְ וּבְדִיל דְיוֹרְכוּן יוֹמָיךְ:

{ג} וְשָׁמַעְתָּ יִשְׂרָאֵל וְשָׁמַרְתָּ לַעֲשׂוֹת אֲשֶׁר יִיטַב לְךָ וַאֲשֶׁר תִּרְבּוּן מְאֹד כַּאֲשֶׁר דִּבֶּר יְהוָה אֱלֹהֵי אֲבֹתֶיךָ לָךְ אֶרֶץ זָבַת חָלָב וּדְבָשׁ:  מ  וְשָׁמַעְתָּ יִשְׂרָאֵל וְשָׁמַרְתָּ לַעֲשׂוֹת אֲשֶׁר יִיטַב לְךָ וַאֲשֶׁר תִּרְבּוּן מְאֹד כַּאֲשֶׁר דִּבֶּר יְהוָה אֱלֹהֵי אֲבֹתֶיךָ לָךְ אֶרֶץ זָבַת חָלָב וּדְבָשׁ:  ת  וּתְקַבֵּל יִשְׂרָאֵל וְתִטַר לְמֶעְבַּד דִי יֵיטַב לָךְ וְדִי תִסְגוּן לַחֲדָא כְּמָא דִי מַלִיל יְיָ אֱלָהָא דַאֲבָהָתָךְ לָךְ אַרְעָא עָבְדָא חֲלַב וּדְבָשׁ: (פ)

{ד}  שישי  שְׁמַע יִשְׂרָאֵל יְהוָה אֱלֹהֵינוּ יְהוָה אֶחָד:  מ  שְׁמַע יִשְׂרָאֵל יְהוָה אֱלֹהֵינוּ יְהוָה אֶחָד:  ת  שְׁמַע יִשְׂרָאֵל יְיָ אֱלָהָנָא יְיָ חָד:

{ה} וְאָהַבְתָּ אֵת יְהוָה אֱלֹהֶיךָ בְּכָל לְבָבְךָ וּבְכָל נַפְשְׁךָ וּבְכָל מְאֹדֶךָ:  מ  וְאָהַבְתָּ אֵת יְהוָה אֱלֹהֶיךָ בְּכָל לְבָבְךָ וּבְכָל נַפְשְׁךָ וּבְכָל מְאֹדֶךָ:  ת  וְתִרְחַם יָת יְיָ אֱלָהָךְ בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ וּבְכָל נִכְסָךְ:

{ו} וְהָיוּ הַדְּבָרִים הָאֵלֶּה אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם עַל לְבָבֶךָ:  מ  וְהָיוּ הַדְּבָרִים הָאֵלֶּה אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם עַל לְבָבֶךָ:  ת  וִיהוֹן פִּתְגָמַיָא הָאִלֵין דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין עַל לִבָּךְ:

{ז} וְשִׁנַּנְתָּם לְבָנֶיךָ וְדִבַּרְתָּ בָּם בְּשִׁבְתְּךָ בְּבֵיתֶךָ וּבְלֶכְתְּךָ בַדֶּרֶךְ וּבְשָׁכְבְּךָ וּבְקוּמֶךָ:  מ  וְשִׁנַּנְתָּם לְבָנֶיךָ וְדִבַּרְתָּ בָּם בְּשִׁבְתְּךָ בְּבֵיתֶךָ וּבְלֶכְתְּךָ בַדֶּרֶךְ וּבְשָׁכְבְּךָ וּבְקוּמֶךָ:  ת  וּתְתָנִנוּן לִבְנָיךְ וּתְמַלֵל בְּהוֹן בְּמִתְּבָךְ בְּבֵיתָךְ וּבִמְהָכָךְ בְּאָרְחָא וּבְמִשְׁכְּבָךְ וּבִמְקִימָךְ:

{ח} וּקְשַׁרְתָּם לְאוֹת עַל יָדֶךָ וְהָיוּ לְטֹטָפֹת בֵּין עֵינֶיךָ:  מ  וּקְשַׁרְתָּם לְאוֹת עַל יָדֶךָ וְהָיוּ לְטֹטָפֹת בֵּין עֵינֶיךָ:  ת  וְתִקְטְרִנוּן לְאָת עַל יְדָךְ וִיהוֹן לִתְפִלִין בֵּין עֵינָיךְ:

{ט} וּכְתַבְתָּם עַל מְזוּזֹת בֵּיתֶךָ וּבִשְׁעָרֶיךָ:  מ  וּכְתַבְתָּם עַל מְזוּזֹת בֵּיתֶךָ וּבִשְׁעָרֶיךָ:  ת  וְתִכְתְּבִנוּן עַל מְזוּזְיָן וְתִקְבְּעִנוּן בְּסִפֵּי בֵיתָךְ וּבְתַרְעָיךְ: (ס)

{י} וְהָיָה כִּי יְבִיאֲךָ יְהוָה אֱלֹהֶיךָ אֶל הָאָרֶץ אֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לָתֶת לָךְ עָרִים גְּדֹלֹת וְטֹבֹת אֲשֶׁר לֹא בָנִיתָ:  מ  וְהָיָה כִּי יְבִיאֲךָ יְהוָה אֱלֹהֶיךָ אֶל הָאָרֶץ אֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לָתֶת לָךְ עָרִים גְּדֹלֹת וְטֹבֹת אֲשֶׁר לֹא בָנִיתָ:  ת  וִיהֵי אֲרֵי יָעֵלִנָךְ יְיָ אֱלָהָךְ לְאַרְעָא דִי קַיִים לַאֲבָהָתָךְ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לְמִתַּן לָךְ קִרְוִין רַבְרְבָן וְטָבָן דִי לָא בְנֵיתָא:

{יא} וּבָתִּים מְלֵאִים כָּל טוּב אֲשֶׁר לֹא מִלֵּאתָ וּבֹרֹת חֲצוּבִים אֲשֶׁר לֹא חָצַבְתָּ כְּרָמִים וְזֵיתִים אֲשֶׁר לֹא נָטָעְתָּ וְאָכַלְתָּ וְשָׂבָעְתָּ:  מ  וּבָתִּים מְלֵאִים כָּל טוּב אֲשֶׁר לֹא מִלֵּאתָ וּבֹרֹת חֲצוּבִים אֲשֶׁר לֹא חָצַבְתָּ כְּרָמִים וְזֵיתִים אֲשֶׁר לֹא נָטָעְתָּ וְאָכַלְתָּ וְשָׂבָעְתָּ:  ת  וּבָתִּין מְלָן כָּל טוּב דִי לָא מְלֵיתָא וְגֻבִּין פְּסִילָן דִי לָא פְּסַלְתָּא כַּרְמִין וְזֵיתִין דִי לָא נְצֶבְתָּא וְתֵיכוּל וְתִשְׂבָּע:

{יב} הִשָּׁמֶר לְךָ פֶּן תִּשְׁכַּח אֶת יְהוָה אֲשֶׁר הוֹצִיאֲךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים:  מ  הִשָּׁמֶר לְךָ פֶּן תִּשְׁכַּח אֶת יְהוָה אֲשֶׁר הוֹצִיאֲךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים:  ת  אִסְתַּמַר לָךְ דִילְמָא תִנְשֵׁי יָת (דַחַלְתָּא דַ) יְיָ דִי אַפְּקָךְ מֵאַרְעָא דְמִצְרַיִם מִבֵּית עַבְדוּתָא:

{יג} אֶת יְהוָה אֱלֹהֶיךָ תִּירָא וְאֹתוֹ תַעֲבֹד וּבִשְׁמוֹ תִּשָּׁבֵעַ:  מ  אֶת יְהוָה אֱלֹהֶיךָ תִּירָא וְאֹתוֹ תַעֲבֹד וּבִשְׁמוֹ תִּשָּׁבֵעַ:  ת  יָת יְיָ אֱלָהָךְ תִּדְחַל וְקָדָמוֹהִי תִּפְלָח וּבִשְׁמֵהּ תְּקַיַם:

{יד} לֹא תֵלְכוּן אַחֲרֵי אֱלֹהִים אֲחֵרִים מֵאֱלֹהֵי הָעַמִּים אֲשֶׁר סְבִיבוֹתֵיכֶם:  מ  לֹא תֵלְכוּן אַחֲרֵי אֱלֹהִים אֲחֵרִים מֵאֱלֹהֵי הָעַמִּים אֲשֶׁר סְבִיבוֹתֵיכֶם:  ת  לָא תְהָכוּן בָּתַר טַעֲוַת עַמְמַיָא מִטַעֲוַת עַמְמַיָא דִי בְסַחֲרָנֵיכוֹן:

{טו} כִּי אֵל קַנָּא יְהוָה אֱלֹהֶיךָ בְּקִרְבֶּךָ פֶּן יֶחֱרֶה אַף יְהוָה אֱלֹהֶיךָ בָּךְ וְהִשְׁמִידְךָ מֵעַל פְּנֵי הָאֲדָמָה:  מ  כִּי אֵל קַנָּא יְהוָה אֱלֹהֶיךָ בְּקִרְבֶּךָ פֶּן יֶחֱרֶה אַף יְהוָה אֱלֹהֶיךָ בָּךְ וְהִשְׁמִידְךָ מֵעַל פְּנֵי הָאֲדָמָה:  ת  אֲרֵי אֵל קַנָא יְיָ אֱלָהָךְ שְׁכִנְתֵּהּ בֵּינָךְ דִילְמָא יִתְקֵף רָגְזָא דַיְיָ אֱלָהָךְ בָּךְ וִישֵׁצָךְ מֵעַל אַפֵּי אַרְעָא: (ס)

{טז} לֹא תְנַסּוּ אֶת יְהוָה אֱלֹהֵיכֶם כַּאֲשֶׁר נִסִּיתֶם בַּמַּסָּה:  מ  לֹא תְנַסּוּ אֶת יְהוָה אֱלֹהֵיכֶם כַּאֲשֶׁר נִסִּיתֶם בַּמַּסָּה:  ת  לָא תְנַסוּן קֳדָם יְיָ אֱלָהֲכוֹן כְּמָא דִי נַסֵתוּן בְנִסֵיתָא:

{יז} שָׁמוֹר תִּשְׁמְרוּן אֶת מִצְוֹת יְהוָה אֱלֹהֵיכֶם וְעֵדֹתָיו וְחֻקָּיו אֲשֶׁר צִוָּךְ:  מ  שָׁמוֹר תִּשְׁמְרוּן אֶת מִצְוֹת יְהוָה אֱלֹהֵיכֶם וְעֵדֹתָיו וְחֻקָּיו אֲשֶׁר צִוָּךְ:  ת  מִטַּר תִּטְרוּן יָת פִּקוּדַיָא דַיְיָ אֱלָהֲכוֹן וּסְהֶדְוָתֵהּ וּקְיָמוֹהִי דִי פַקְדָךְ:

{יח} וְעָשִׂיתָ הַיָּשָׁר וְהַטּוֹב בְּעֵינֵי יְהוָה לְמַעַן יִיטַב לָךְ וּבָאתָ וְיָרַשְׁתָּ אֶת הָאָרֶץ הַטֹּבָה אֲשֶׁר נִשְׁבַּע יְהוָה לַאֲבֹתֶיךָ:  מ  וְעָשִׂיתָ הַיָּשָׁר וְהַטּוֹב בְּעֵינֵי יְהוָה לְמַעַן יִיטַב לָךְ וּבָאתָ וְיָרַשְׁתָּ אֶת הָאָרֶץ הַטֹּבָה אֲשֶׁר נִשְׁבַּע יְהוָה לַאֲבֹתֶיךָ:  ת  וְתַעְבֵּד דְכָשַׁר וּדְתַקֵן קֳדָם יְיָ בְּדִיל דְיֵיטַב לָךְ וְתֵעוּל וְתֵירַת יָת אַרְעָא טַבְתָא דִי קַיִים יְיָ לַאֲבָהָתָךְ:

{יט} לַהֲדֹף אֶת כָּל אֹיְבֶיךָ מִפָּנֶיךָ כַּאֲשֶׁר דִּבֶּר יְהוָה:  מ  לַהֲדֹף אֶת כָּל אֹיְבֶיךָ מִפָּנֶיךָ כַּאֲשֶׁר דִּבֶּר יְהוָה:  ת  לְמִתְּבַר יָת כָּל בַּעֲלֵי דְבָבָךְ מֵקָדָמָךְ כְּמָא דִי מַלִיל יְיָ: (ס)

{כ} כִּי יִשְׁאָלְךָ בִנְךָ מָחָר לֵאמֹר מָה הָעֵדֹת וְהַחֻקִּים וְהַמִּשְׁפָּטִים אֲשֶׁר צִוָּה יְהוָה אֱלֹהֵינוּ אֶתְכֶם:  מ  כִּי יִשְׁאָלְךָ בִנְךָ מָחָר לֵאמֹר מָה הָעֵדֹת וְהַחֻקִּים וְהַמִּשְׁפָּטִים אֲשֶׁר צִוָּה יְהוָה אֱלֹהֵינוּ אֶתְכֶם:  ת  אֲרֵי יִשְׁאֲלִנָךְ בְּרָךְ מְחָר לְמֵימָר מָא סָהֶדְוָתָא וּקְיָמַיָא וְדִינַיָא דִי פַקִיד יְיָ אֱלָהָנָא יָתְכוֹן:

{כא} וְאָמַרְתָּ לְבִנְךָ עֲבָדִים הָיִינוּ לְפַרְעֹה בְּמִצְרָיִם וַיּוֹצִיאֵנוּ יְהוָה מִמִּצְרַיִם בְּיָד חֲזָקָה:  מ  וְאָמַרְתָּ לְבִנְךָ עֲבָדִים הָיִינוּ לְפַרְעֹה בְּמִצְרָיִם וַיּוֹצִיאֵנוּ יְהוָה מִמִּצְרַיִם בְּיָד חֲזָקָה:  ת  וְתֵימַר לִבְרָךְ עַבְדִין הֲוֵינָא לְפַרְעֹה בְּמִצְרָיִם וְאַפְּקָנָא יְיָ מִמִצְרַיִם בִּידָא תַקִיפָא:

{כב} וַיִּתֵּן יְהוָה אוֹתֹת וּמֹפְתִים גְּדֹלִים וְרָעִים בְּמִצְרַיִם בְּפַרְעֹה וּבְכָל בֵּיתוֹ לְעֵינֵינוּ:  מ  וַיִּתֵּן יְהוָה אוֹתֹת וּמֹפְתִים גְּדֹלִים וְרָעִים בְּמִצְרַיִם בְּפַרְעֹה וּבְכָל בֵּיתוֹ לְעֵינֵינוּ:  ת  וִיהַב יְיָ אָתִין וּמוֹפְתִין רַבְרְבִין וּבִישִׁין בְּמִצְרַיִם בְּפַרְעֹה וּבְכָל אֱנַשׁ בֵּיתֵהּ לְעֵינָנָא:

{כג} וְאוֹתָנוּ הוֹצִיא מִשָּׁם לְמַעַן הָבִיא אֹתָנוּ לָתֶת לָנוּ אֶת הָאָרֶץ אֲשֶׁר נִשְׁבַּע לַאֲבֹתֵינוּ:  מ  וְאוֹתָנוּ הוֹצִיא מִשָּׁם לְמַעַן הָבִיא אֹתָנוּ לָתֶת לָנוּ אֶת הָאָרֶץ אֲשֶׁר נִשְׁבַּע לַאֲבֹתֵינוּ:  ת  וְיָתָנָא אַפֵּק מִתַּמָן בְּדִיל לְאַעֲלָא יָתָנָא לְמִתַּן לָנָא יָת אַרְעָא דִי קַיִים לַאֲבָהָתָנָא:

{כד} וַיְצַוֵּנוּ יְהוָה לַעֲשׂוֹת אֶת כָּל הַחֻקִּים הָאֵלֶּה לְיִרְאָה אֶת יְהוָה אֱלֹהֵינוּ לְטוֹב לָנוּ כָּל הַיָּמִים לְחַיֹּתֵנוּ כְּהַיּוֹם הַזֶּה:  מ  וַיְצַוֵּנוּ יְהוָה לַעֲשׂוֹת אֶת כָּל הַחֻקִּים הָאֵלֶּה לְיִרְאָה אֶת יְהוָה אֱלֹהֵינוּ לְטוֹב לָנוּ כָּל הַיָּמִים לְחַיֹּתֵנוּ כְּהַיּוֹם הַזֶּה:  ת  וּפַּקְדָנָא יְיָ לְמֶעְבַּד יָת כָּל קְיָמַיָא הָאִלֵין לְמִדְחַל יָת יְיָ אֱלָהָנָא לְטַב לָנָא כָּל יוֹמַיָא לְקַיָמוּתָנָא כְּיוֹמָא הָדֵין:

{כה} וּצְדָקָה תִּהְיֶה לָּנוּ כִּי נִשְׁמֹר לַעֲשׂוֹת אֶת כָּל הַמִּצְוָה הַזֹּאת לִפְנֵי יְהוָה אֱלֹהֵינוּ כַּאֲשֶׁר צִוָּנוּ:  מ  וּצְדָקָה תִּהְיֶה לָּנוּ כִּי נִשְׁמֹר לַעֲשׂוֹת אֶת כָּל הַמִּצְוָה הַזֹּאת לִפְנֵי יְהוָה אֱלֹהֵינוּ כַּאֲשֶׁר צִוָּנוּ:  ת  וְזָכוּתָא תְּהֵא לָנָא אֲרֵי נִטַר לְמֶעְבַּד יָת כָּל תַּפְקֶדְתָּא הָדָא קֳדָם יְיָ אֱלָהָנָא כְּמָא דִי פַקְדָנָא: (ס)




דברים פרק-ז

{א}  שביעי  כִּי יְבִיאֲךָ יְהוָה אֱלֹהֶיךָ אֶל הָאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ וְנָשַׁל גּוֹיִם רַבִּים מִפָּנֶיךָ הַחִתִּי וְהַגִּרְגָּשִׁי וְהָאֱמֹרִי וְהַכְּנַעֲנִי וְהַפְּרִזִּי וְהַחִוִּי וְהַיְבוּסִי שִׁבְעָה גוֹיִם רַבִּים וַעֲצוּמִים מִמֶּךָּ:  מ  כִּי יְבִיאֲךָ יְהוָה אֱלֹהֶיךָ אֶל הָאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ וְנָשַׁל גּוֹיִם רַבִּים מִפָּנֶיךָ הַחִתִּי וְהַגִּרְגָּשִׁי וְהָאֱמֹרִי וְהַכְּנַעֲנִי וְהַפְּרִזִּי וְהַחִוִּי וְהַיְבוּסִי שִׁבְעָה גוֹיִם רַבִּים וַעֲצוּמִים מִמֶּךָּ:  ת  אֲרֵי יְעֵלִנָךְ יְיָ אֱלָהָךְ לְאַרְעָא דִי אַתְּ עָלֵל לְתַמָן לְמֵירְתַהּ וִיתָרֵךְ עַמְמִין סַגִיאִין מִן קֳדָמָךְ חִתָּאֵי וְגִרְגָשָׁאֵי וֶאֱמֹרָאֵי וּכְנַעֲנָאֵי וּפְרִזָאֵי וְחִוָאֵי וִיבוּסָאֵי שִׁבְעָא עַמְמִין סַגִיאִין וְתַקִיפִין מִנָךְ:

{ב} וּנְתָנָם יְהוָה אֱלֹהֶיךָ לְפָנֶיךָ וְהִכִּיתָם הַחֲרֵם תַּחֲרִים אֹתָם לֹא תִכְרֹת לָהֶם בְּרִית וְלֹא תְחָנֵּם:  מ  וּנְתָנָם יְהוָה אֱלֹהֶיךָ לְפָנֶיךָ וְהִכִּיתָם הַחֲרֵם תַּחֲרִים אֹתָם לֹא תִכְרֹת לָהֶם בְּרִית וְלֹא תְחָנֵּם:  ת  וְיִמְסְרִנוּן יְיָ אֱלָהָךְ קֳדָמָךְ וְתִמְחִנוּן גַמָרָא תְגַמַר יָתְהוֹן לָא תִגְזַר לְהוֹן קְיָם וְלָא תְרַחַם עֲלֵיהוֹן:

{ג} וְלֹא תִתְחַתֵּן בָּם בִּתְּךָ לֹא תִתֵּן לִבְנוֹ וּבִתּוֹ לֹא תִקַּח לִבְנֶךָ:  מ  וְלֹא תִתְחַתֵּן בָּם בִּתְּךָ לֹא תִתֵּן לִבְנוֹ וּבִתּוֹ לֹא תִקַּח לִבְנֶךָ:  ת  וְלָא תִתְחַתַּן בְּהוֹן בְּרַתָּךְ לָא תִתֵּן לִבְרֵהּ וּבְרַתֵּהּ לָא תִסַב לִבְרָךְ:

{ד} כִּי יָסִיר אֶת בִּנְךָ מֵאַחֲרַי וְעָבְדוּ אֱלֹהִים אֲחֵרִים וְחָרָה אַף יְהוָה בָּכֶם וְהִשְׁמִידְךָ מַהֵר:  מ  כִּי יָסִיר אֶת בִּנְךָ מֵאַחֲרַי וְעָבְדוּ אֱלֹהִים אֲחֵרִים וְחָרָה אַף יְהוָה בָּכֶם וְהִשְׁמִידְךָ מַהֵר:  ת  אֲרֵי יַטְעֲיַן יָת בְּרָךְ מִבָּתַר פָּלְחָנִי וְיִפְלְחוּן לְטַעֲוַת עַמְמַיָא וְיִתְקֵף רָגְזָא דַיְיָ בְּכוֹן וִישֵׁצָךְ בִּפְרִיעַ:

{ה} כִּי אִם כֹּה תַעֲשׂוּ לָהֶם מִזְבְּחֹתֵיהֶם תִּתֹּצוּ וּמַצֵּבֹתָם תְּשַׁבֵּרוּ וַאֲשֵׁירֵהֶם תְּגַדֵּעוּן וּפְסִילֵיהֶם תִּשְׂרְפוּן בָּאֵשׁ:  מ  כִּי אִם כֹּה תַעֲשׂוּ לָהֶם מִזְבְּחֹתֵיהֶם תִּתֹּצוּ וּמַצֵּבֹתָם תְּשַׁבֵּרוּ וַאֲשֵׁירֵהֶם תְּגַדֵּעוּן וּפְסִילֵיהֶם תִּשְׂרְפוּן בָּאֵשׁ:  ת  אֲרֵי אִם כְּדֵין תַּעַבְּדוּן לְהוֹן אֱגוֹרֵיהוֹן תְּתָרְעוּן וְקָמָתְהוֹן תְּתַבְּרוּן וַאֲשֵׁרֵיהוֹן תְּקוֹצְצוּן וְצַלְמֵי טַּעֲוָתְהוֹן תּוֹקְדוּן בְּנוּרָא:

{ו} כִּי עַם קָדוֹשׁ אַתָּה לַיהוָה אֱלֹהֶיךָ בְּךָ בָּחַר יְהוָה אֱלֹהֶיךָ לִהְיוֹת לוֹ לְעַם סְגֻלָּה מִכֹּל הָעַמִּים אֲשֶׁר עַל פְּנֵי הָאֲדָמָה:  מ  כִּי עַם קָדוֹשׁ אַתָּה לַיהוָה אֱלֹהֶיךָ בְּךָ בָּחַר יְהוָה אֱלֹהֶיךָ לִהְיוֹת לוֹ לְעַם סְגֻלָּה מִכֹּל הָעַמִּים אֲשֶׁר עַל פְּנֵי הָאֲדָמָה:  ת  אֲרֵי עַם קַדִישׁ אַתְּ קֳדָם יְיָ אֱלָהָךְ בָּךְ אִתִּרְעֵי יְיָ אֱלָהָךְ לְמֶהֱוֵי לֵהּ לְעַם חַבִּיב מִכֹּל עַמְמַיָא דִי עַל אַפֵּי אַרְעָא:

{ז} לֹא מֵרֻבְּכֶם מִכָּל הָעַמִּים חָשַׁק יְהוָה בָּכֶם וַיִּבְחַר בָּכֶם כִּי אַתֶּם הַמְעַט מִכָּל הָעַמִּים:  מ  לֹא מֵרֻבְּכֶם מִכָּל הָעַמִּים חָשַׁק יְהוָה בָּכֶם וַיִּבְחַר בָּכֶם כִּי אַתֶּם הַמְעַט מִכָּל הָעַמִּים:  ת  לָא מִדְסַגִיאִין אַתּוּן מִכָּל עַמְמַיָא צָבֵי יְיָ בְּכוֹן וְאִתִּרְעֵי בְכוֹן אֲרֵי אַתּוּן זְעֵרִין מִכָּל עַמְמַיָא:

{ח} כִּי מֵאַהֲבַת יְהוָה אֶתְכֶם וּמִשָּׁמְרוֹ אֶת הַשְּׁבֻעָה אֲשֶׁר נִשְׁבַּע לַאֲבֹתֵיכֶם הוֹצִיא יְהוָה אֶתְכֶם בְּיָד חֲזָקָה וַיִּפְדְּךָ מִבֵּית עֲבָדִים מִיַּד פַּרְעֹה מֶלֶךְ מִצְרָיִם:  מ  כִּי מֵאַהֲבַת יְהוָה אֶתְכֶם וּמִשָּׁמְרוֹ אֶת הַשְּׁבֻעָה אֲשֶׁר נִשְׁבַּע לַאֲבֹתֵיכֶם הוֹצִיא יְהוָה אֶתְכֶם בְּיָד חֲזָקָה וַיִּפְדְּךָ מִבֵּית עֲבָדִים מִיַּד פַּרְעֹה מֶלֶךְ מִצְרָיִם:  ת  אֲרֵי מִדְרַחַם יְיָ יָתְכוֹן וּמִדִנְטַר יָת קְיָמָא דִי קַיִים לַאֲבָהָתְכוֹן אַפֵּיק יְיָ יָתְכוֹן בִּידָא תַקִיפָא וּפָרְקָךְ מִבֵּית עַבְדוּתָא מִידָא דְפַרְעֹה מַלְכָּא דְמִצְרָיִם:

{ט}  מפטיר  וְיָדַעְתָּ כִּי יְהוָה אֱלֹהֶיךָ הוּא הָאֱלֹהִים הָאֵל הַנֶּאֱמָן שֹׁמֵר הַבְּרִית וְהַחֶסֶד לְאֹהֲבָיו וּלְשֹׁמְרֵי (מצותו) מִצְוֹתָיו לְאֶלֶף דּוֹר:  מ  וְיָדַעְתָּ כִּי יְהוָה אֱלֹהֶיךָ הוּא הָאֱלֹהִים הָאֵל הַנֶּאֱמָן שֹׁמֵר הַבְּרִית וְהַחֶסֶד לְאֹהֲבָיו וּלְשֹׁמְרֵי (מצותו) מִצְוֹתָיו לְאֶלֶף דּוֹר:  ת  וְתִדַע אֲרֵי יְיָ אֱלָהָךְ הוּא אֱלֹהִים אֱלָהָא מְהֵימְנָא נָטֵר קְיָמָא וְחִסְדָא לְרַחֲמוֹהִי וּלְנָטְרֵי פִקוֹדוֹהִי לְאַלְפֵי דָרִין:

{י} וּמְשַׁלֵּם לְשֹׂנְאָיו אֶל פָּנָיו לְהַאֲבִידוֹ לֹא יְאַחֵר לְשֹׂנְאוֹ אֶל פָּנָיו יְשַׁלֶּם לוֹ:  מ  וּמְשַׁלֵּם לְשֹׂנְאָיו אֶל פָּנָיו לְהַאֲבִידוֹ לֹא יְאַחֵר לְשֹׂנְאוֹ אֶל פָּנָיו יְשַׁלֶּם לוֹ:  ת  וּמְשַׁלֵם לְסָנְאוֹהִי טָבָן דִי אִנוּן עָבְדִין קֳדָמוֹהִי בְּחַיֵיהוֹן לְאוֹבָדֵיהוֹן לָא מְאַחַר עֹבַד טָב לְסָנְאוֹהִי טַבְוָן דִי אִנוּן עָבְדִין קָדָמוֹהִי בְּחַיֵיהוֹן מְשַׁלֵם לְהוֹן:

{יא} וְשָׁמַרְתָּ אֶת הַמִּצְוָה וְאֶת הַחֻקִּים וְאֶת הַמִּשְׁפָּטִים אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לַעֲשׂוֹתָם:  מ  וְשָׁמַרְתָּ אֶת הַמִּצְוָה וְאֶת הַחֻקִּים וְאֶת הַמִּשְׁפָּטִים אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לַעֲשׂוֹתָם:  ת  וְתִטַר יָת תַּפְקֶדְתָּא וְיָת קְיָמַיָא וְיָת דִינַיָא דִי אֲנָא מְפַקְדָךְ יוֹמָא דֵין לְמֶעְבָּדְהוֹן: (פפפ)
