~ פרק א - משנה א

{א} דְּמִדְּקָתָנֵי מֵאֵימָתַי מִכְּלַל דִּפְשִׁיטָא לֵיהּ דְּחַיָּבִין לְהַזְכִּיר, וְהֵיכָא חָזִינַן, אֶלָּא אַיְדֵי כוּ'. וּמִבְּרָכוֹת פֶּרֶק ה' מִשְׁנָה ב' דְּתָנֵי מַזְכִּירִין כוּ' אֵין רְאָיָה, דַּהֲוָה אֲמִינָא דִּבְכָל הַשָּׁנָה קָאָמַר. אֲבָל אַדְּתָנֵי וּבֶחָג כוּ' אוֹרְחָא דְמִלְּתָא לְרַצּוֹת לִפְנֵי שְׁאֵלָה. תּוֹסָפוֹת: {ב} <B> מַזְכִּירִין.</B> עַיֵּן בְּרָכוֹת (שָׁם. וְעַיֵּן שָׁם אוֹת ה): {ג} <B> הָרִאשׁוֹן.</B> מְפָרֵשׁ בַּגְּמָרָא דְּמִלּוּלָב קָא גָּמַר לַהּ, שֶׁבָּא לְרַצּוֹת עַל הַמַּיִם וּמַתְחִילִין לִטֹּל מִיּוֹם טוֹב הָרִאשׁוֹן, הִלְכָּךְ בִּתְפִלַת יוֹצֵר נַמִּי שֶׁל יוֹם טוֹב הָרִאשׁוֹן מַזְכִּירִין גְּבוּרוֹת גְּשָׁמִים. וּמִיהוּ מֵאוּרְתָא לֹא מַתְחִילִין, כְּשֵׁם שֶׁאֵין מַתְחִילִין לִטֹּל אֶת הַלּוּלָב בַּלַּיְלָה: {ד} <B> לְהַזְכִּיר.</B> פֵּרֵשׁ רַשִׁ"י לֹא אָמַרְתִּי לִשְׁאוֹל שֶׁיִּתְפַּלְּלוּ עַל הַגְּשָׁמִים בֶּחָג כְּגוֹן וְתֵן טַל וּמָטָר, אֶלָּא לְהַזְכִּיר בֶּחָג גְּבוּרוֹת שֶׁל מָקוֹם שֶׁמּוֹרִיד גְּשָׁמִים בְּעוֹנָתָן כְּלוֹמַר בִּזְמַנָּן: {ה} <B> בְּעוֹנָתוֹ.</B> כְּשֵׁם שֶׁתְּחִיַּת הַמֵּתִים מַזְכִּיר כָּל הַשָּׁנָה כֻלָּהּ וְאֵינָהּ אֶלָּא בִּזְמַנָּהּ. וְרַבִּי יְהוֹשֻׁעַ אָמַר לְךָ בִּשְׁלָמָא תְּחִיַּת הַמֵּתִים מַזְכִּיר, דְּכֻלָּהּ יוֹמָא זִמְנֵיהּ הוּא, אֶלָּא גְּשָׁמִים הָתְנַן סוֹף פִּרְקִין יָצָא נִיסָן וְיָרְדוּ גְשָׁמִים סִימַן קְלָלָה. גְּמָרָא: {ו} <B> לְעוֹלָם כוּ'.</B> וְרַבִּי אֱלִיעֶזֶר סְבִירָא לֵיהּ בַּבְּרַיְתָא בַּגְּמָרָא דְּאִם בָּא לְהַזְכִּיר כָּל הַשָּׁנָה כֻלָּהּ מַזְכִּיר, דִּלְעוֹלָם בְּעוֹנָתוֹ מַשְׁמַע, מִיהוּ עַד הַשְׁתָּא לֹא רָמֵי עֲלֵיהּ חוֹבָה, אֲבָל בְּיוֹם טוֹב הָרִאשׁוֹן חוֹבָה לְרַצּוֹת לִפְנֵי שְׁאֵלָה, שֶׁכָּל הַבָּא לְבַקֵּשׁ מַקְדִּים וּמְרַצֶּה. תּוֹסָפוֹת:

{ז} וְאִם כֵּן רַבִּי יְהוּדָה פָּלֵיג.</B> וְכֵן הוּא בְּהֶדְיָא בַּגְּמָרָא: {ח} מִנְהָגָם הָיָה שֶׁהָעוֹבֵר לִפְנֵי הַתֵּבָה בְּמוּסָף אֵינוֹ אוֹתוֹ שֶׁעוֹבֵר בְּשַׁחֲרִית.</B> וּבַיְרוּשַׁלְמִי קָאָמַר אַמַּאי אֵינוֹ מַזְכִּיר מֵאוּרְתָא, פֵּרוּשׁ בִּתְפִלַּת עַרְבִית, וּמְשָׁנֵי דְּלֵית תַּמָּן כָּל עַמָּא, נִמְצָא זֶה מַזְכִּיר וְזֶה אֵינוֹ מַזְכִּיר, וְיִהְיֶה אֲגֻדּוֹת אֲגֻדּוֹת. וּפָרֵיךְ וְלִדְכְּרוּ בְּצַפְרָא, פֵּרוּשׁ בִּתְפִלַּת שַׁחֲרִית, וּמְשָׁנֵי דַּהֲווֹ סָבְרֵי מַדְכְּרֵי לֵיהּ מֵאוּרְתָא וּלְשָׁנָה הַבָּאָה יַזְכִּירוּ גַּם מִבָּעֶרֶב: {ט} כֵּן כָּתַב הָרַמְבַּ"ם אֲבָל אוֹמֵר וּמוֹרִיד הַטַּל.</B> וְזֶה כְּמִנְהַג בְּנֵי סְפָרַד שֶׁנּוֹהֲגִים לְהַזְכִּיר בִּימוֹת הַחַמָּה מַשִּׁיב הָרוּחַ וּמוֹרִיד הַטַּל: {י} <B> אֵינוֹ כוּ'.</B> יְרוּשַׁלְמִי, מַאי טַעֲמָא דְרַבִּי יְהוּדָה, כְּדֵי שֶׁיֵּצְאוּ הַמּוֹעֲדוֹת בַּטַּל, מִפְּנֵי שֶׁהַטַּל סִימָן יָפֶה לָעוֹלָם, וְיַזְכִּיר מִבָּעֶרֶב כוּ' כְּדִלְעֵיל: {יא} <B> עַד שֶׁיַּעֲבֹר.</B> אֶפְשָׁר לִי לוֹמַר שֶׁסּוֹבֵר דְּכֵיוָן דְּאֵין מַתְחִילִין לְהַזְכִּיר אֶלָּא בְּיוֹם טוֹב אַחֲרוֹן שֶׁל חָג, הָכִי נַמִּי אֵין פּוֹסְקִין אֶלָּא עַד שֶׁיַּעֲבֹר הַפֶּסַח, וְעוֹד שֶׁרְגִילִין לְהִתְאַחֵר בִּירוּשָׁלַיִם עַד שֶׁיַּעֲבֹר הַפֶּסַח וְאֵין כָּל כָּךְ עוֹבְרֵי דְרָכִים בְּחַג הַפֶּסַח שֶׁהַגְּשָׁמִים יַפְסִיקוּם, אֲבָל בִּימֵי הַסֻּכּוֹת סִימַן קְלָלָה הֵם: {יב} <B> שֶׁנֶּאֱמַר כוּ'.</B> וְאִינָךְ סְבִירָא לֵיהּ דְּאֵין לְמֵדִין מֵהָא דְּנִבֵּא לִשְׁעָתוֹ שֶׁהָיָה שְׁנַת רְעָבוֹן וַעֲצִירַת גְּשָׁמִים שֶׁבִּשֵּׂר לָהֶם שֶׁאַף עַל פִּי שֶׁיָּצָא אֲדָר וְלֹא יָרְדוּ גְשָׁמִים שֶׁיֵּצְאוּ וְיִזְרְעוּ: 

~ פרק א - משנה ג

{יג} <B> בִּשְׁלֹשָׁה כוּ'.</B> דְּאָז זְמַן רְבִיעָה רִאשׁוֹנָה: {יד} לְפִי שֶׁהוּא מָקוֹם נָמוּךְ וְאֵינוֹ צָרִיךְ מָטָר כָּל כָּךְ.</B> רַשִׁ"י: {טו} <B> בְּשִׁבְעָה כוּ'.</B> לְשׁוֹן רַשִׁ"י כְּלוֹמַר בְּז' בְּמַרְחֶשְׁוָן הוּא חֲמִשָּׁה עָשָׂר אַחַר הֶחָג. וְזֶה יִצְדַּק לְעוֹלָם כְּפִי סֵדֶר הָעִבּוּר הַמָּסוּר בְּיָדֵינוּ שֶׁאָנוּ חוֹשְׁבִים בּוֹ עַכְשָׁיו שֶׁלְּעוֹלָם תִּשְׁרֵי מָלֵא. אֲבָל כְּשֶׁקִּדְּשׁוּ עַל פִּי הָרְאִיָּה וְזֶה הָיָה בִּימֵי רַבָּן גַּמְלִיאֵל אֶפְשָׁר שֶׁהָיָה חָסֵר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טז} <B> אַחֲרוֹן.</B> שֶׁלִּפְעָמִים מִתְעַכְּבִין קְצָת בִּירוּשָׁלַיִם עַד שֶׁיַּעֲבֹר הֶחָג כֻּלּוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב. וּמְסַיֵּם הֲרֵי לְךָ שֶׁמִּן יְרוּשָׁלַיִם מִדְּרוֹמָהּ עַד קְצֵה הַצָּפוֹן שֶׁל אֶרֶץ יִשְׂרָאֵל שֶׁהוּא נְהַר פְּרָת, מַהֲלַךְ חֲמִשָּׁה עָשָׂר יוֹם. וְאַל תִּתְמַהּ עַל הַשַּׁבָּתוֹת שֶׁבְּתוֹךְ אֵלּוּ הַחֲמִשָּׁה עָשָׂר יוֹם, שֶׁהֲרֵי אֶפְשָׁר שֶׁיֵּלֵךְ עַל יְדֵי בּוּרְגָּנִין וְגַם עַל יְדֵי עִבּוּרֵי עִיר. וְהוֹאִיל וְאֶפְשָׁר לִהְיוֹת כֵּן שִׁעֲרוּהוּ חֲכָמִים בְּכָךְ יָמִים, כֵּיוָן שֶׁאֶפְשָׁר שֶׁאֵין הַשַּׁבָּתוֹת מְעַכְּבוֹת: 

~ פרק א - משנה ד

{יז} גְּמָרָא, דִּבְרֵי רַבִּי מֵאִיר: {יח} שֶׁמְּיֻחָדִים בְּמַעֲשֵׂיהֶם.</B> הָרֹא"שׁ. וּתְמֵהַנִי, דְּבַגְּמָרָא מַאן יְחִידִים אָמַר רַב הוּנָא רַבָּנָן, וּבָתַר הָכִי תָּנֵי אֵיזֶהוּ יָחִיד כוּ' כָּל שֶׁרָאוּי לְמַנּוֹתוֹ פַּרְנָס עַל הַצִּבּוּר, וְאֵיזֶהוּ תַּלְמִיד כָּל שֶׁשּׁוֹאֲלִין אוֹתוֹ דְּבַר הֲלָכָה וְאוֹמֵר אֲפִלּוּ בְּמַסֶּכֶת כַּלָּה. עַד כָּאן. שְׁמַע מִנָּהּ דְּיָחִיד עָדִיף מִתַּלְמִיד כוּ', וְאִם כֵּן קָשֶׁה עַל הָרֹא"שׁ וְהָרַ"ב שֶׁכָּתְבוּ תַּלְמִידֵי חֲכָמִים. אֶלָּא הֲוָה לְהוּ לְמִכְתָּב רַבָּנָן, אִי נַמִּי חֲכָמִים: {יט} וּבַגְּמָרָא, כְּמוֹ שֶׁאֵין גּוֹזְרִין תַּעֲנִית עַל הַצִּבּוּר לְכַתְּחִלָּה בַּחֲמִישִׁי, הוּא הַדִּין נַמִּי לִיחִידִים: {כ} כְּלוֹמַר שֶׁאֵין לָאָדָם קְבִיעַת זְמַן לְשֶׁיִּשְׁתֶּה בוֹ. וְאֵין לָשׁוֹן דְּהָכָא כְּלִישְׁנָא דִגְמָרָא אֵין קֶבַע לְשֵׁינָה [בְּפֶרֶק קַמָּא דְּסֻכָּה]: 

~ פרק א - משנה ה

שֶׁז' בְּתִשְׁרֵי שֶׁהֵם שְׁנֵי יָמִים קֹדֶם יוֹם הַכִּפּוּרִים הִתְחִילוּ יְמֵי הַחִנּוּךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
