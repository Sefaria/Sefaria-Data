~ פרק א - משנה א

{א} <b>יְצִיאוֹת הַשַּׁבָּת.</b> וְאַף עַל גַּב דִּמְלָאכָה זוֹ מְנוּיָה בְּסוֹף אֲבוֹת מְלָאכוֹת בְּפֶרֶק ז' הִקְדִּימוּהָ, לְפִי שֶׁזֹּאת הַמְּלָאכָה הִיא נִמְצֵאת תָּדִיר, וּבָהּ יִכָּשְׁלוּ בְּנֵי אָדָם עַל הָרֹב, לְפִי שֶׁאֵינָם צְרִיכִים בָּהּ לִכְלוּם. וְטַעַם שֵׁנִי לְלַמְּדֵנוּ שֶׁהִיא מְלָאכָה, אַף עַל פִּי שֶׁהַנִּרְאֶה מִמֶּנָּה שֶׁאֵינָהּ מְלָאכָה. הָרַמְבַּ"ם: {ב} וּמַה לִּי אַפּוֹקֵי מַה לִּי עַיּוֹלֵי.</B> גְּמָרָא רֵישׁ פֶּרֶק הַזּוֹרֵק. וּבְמַסֶּכֶת שְׁבוּעוֹת דַּף ה' (וּבִמְכִילְתִּין) אָמְרִינַן וְטַעְמָא מַאי, תַּנָּא כָּל עֲקִירַת חֵפֶץ מִמְּקוֹמוֹ הוֹצָאָה קָרֵי לַהּ: {ג} <B> בַּחוּץ.</B> וְכָתְבוּ הַתּוֹסָפוֹת תֵּימַהּ לְרִ"י אַמַּאי צָרִיךְ לְמִתְנֵי תַּרְתֵּי דְּעָנִי וּדְבַעַל הַבַּיִת, מַה לִּי עָנִי מַה לִּי עָשִׁיר. וְנִרְאֶה דְּאִיצְטְרִיך לְאַשְׁמוֹעִינַן מִשּׁוּם דְּהוֹצָאָה מְלָאכָה גְּרוּעָה הִיא, דְּמַה לִּי מוֹצִיא מֵרְשׁוּת הָרַבִּים לִרְשׁוּת הַיָּחִיד מַה לִּי מוֹצִיא מֵרְשׁוּת הַיָּחִיד לִרְשׁוּת הַיָּחִיד, וְלֹא הֲוָה גָּמְרִינַן דְּבַעַל הַבַּיִת מֵעָנִי וְלֹא דְּעָנִי מִבַּעַל הַבַּיִת, וּלְהָכִי אִיצְטְרִיכֵי תְּרֵי קְרָאֵי, חַד אַל יֵצֵא, וְחַד וַיִּכָּלֵא הָעָם מֵהָבִיא, דִּבְרֵישׁ פֶּרֶק הַזּוֹרֵק: {ד} אוֹרְחָא דְּמִלְּתָא נָקֵיט דְּבָעֵי לְפָרוֹשֵׁי דְּרוֹצֶה לְקַבֵּל צְדָקָה מִטַּעֲמָא דִּמְפָרֵשׁ: {ה} מִצַּד אִסּוּר שַׁבָּת הוּא דְּקָאָמְרִינַן פָּטוּר וּמֻתָּר, וְלֹא דָּיְקִינַן הַשְׁתָּא בְּאִסּוּר דְּלִפְנֵי עִוֵּר, דִּבְלָאו דִּידֵיהּ לֹא מַיְרֵי. הָרַ"ן: {ו} כְּלוֹמַר וְיָבֹא לִידֵי חִיּוּב חַטָּאת.</B> וְהַתּוֹסָפוֹת הִקְשׁוּ עַל זֶה דַּעֲקִירָה בְּלֹא הוֹצָאָה אֵינוֹ אֶלָּא טִלְטוּל בְּעַלְמָא, וְאֵין בָּהּ דְּרָרָא דְּחִיּוּב חַטָּאת כְּלָל וְעוֹד הִקְּשׁוּ כוּ', וּמְפָרֵשׁ רִיבָ"א דְּאוֹתוֹ שֶׁפּוֹשֵׁט אֶת יָדוֹ וּמַכְנִיסָהּ אוֹ מוֹצִיאָהּ, הַהִיא קָחָשֵׁיב דְּמָצֵי אָתְיָא לִידֵי חִיּוּב חַטָּאת: 

~ פרק א - משנה ב

{ז} פֵּרֵשׁ רַשִׁ"י הַתְחָלַת זְמַן תָּמִיד שֶׁל בֵּין הָעַרְבַּיִם, כִּדְאִיתָא בְּרֵישׁ פֶּרֶק ה' דִּפְסָחִים: {ח} <B>לַמֶּרְחָץ.</B> אֲפִלּוּ לְהַזִּיעַ בְּעָלְמָא. וְלֹא לַבֻּרְסְקִי, לְעַיּוֹנֵי בְּעָלְמָא. גְּמָרָא: {ט} וְסִיֵּם הָרַמְבַּ"ם וְנָשְׂאוּ וְנָתְנוּ הַדַּיָּנִין בַּדָּבָר וְלֹא נִשְׁאַר וְכוּ': {י} לְשׁוֹן הָרַמְבַּ"ם וְיִצְטָרֵךְ לְהַתְחִיל בַּדִּין מִתְּחִלָּתוֹ וְיִשְׁאַל טְעָנוֹת מִבַּעֲלֵי הַדִּינִין שֵׁנִית וְיִמְשֹׁךְ הַדָּבָר כוּ': {יא} מִשּׁוּם סֵיפָא דְּקָתָנֵי דְאֵין מַפְסִיקִין מוֹקִים לַהּ בַּגְּמָרָא הָכִי, דְּאִי לָאו הָכִי הָא כְּבָר תָּנָא לֵיהּ רֵישָׁא: {יב} גְּמָרָא. וְאִם תֹּאמַר וְכִי גָּרַע תַּלְמוּד תּוֹרָה מֵאֻמָּנוּת דִּלְעֵיל שֶׁאִם הִתְחִילוּ אֵין מַפְסִיקִין וַאֲפִלּוּ בְּהִתְחִילוּ בְּאִסּוּר, כִּדְאִיתָא בְּטוּר אֹרַח חַיִּים סִימָן רל"ב, וְהַתֵּרוּץ דְּהָכָא בִּדְלֵיכָּא שְׁהוּת, וַאֲפִלּוּ הָכִי לְרַבִּי שִׁמְעוֹן בֶּן יוֹחַאי וַחֲבֵרָיו מֻתָּר לַעֲקֹר הַתְּפִלָּה לְגַמְרֵי: 

~ פרק א - משנה ג

{יג} <B> לְאוֹר הַנֵּר.</B> קָאֵי נַמִּי עַל לֹא יְפַלֶּה, כִּדְמַסִּיק בַּגְּמָרָא: {יד} וְתַרְגּוּם רוֹאֶה חוֹזֶה, שֶׁהוּא צָרִיךְ לִרְאוֹת הֵיאַךְ תִּינוֹקוֹת קוֹרְאִין, וְעַל שֵׁם זֶה נִקְרָא שְׁלִיחַ צִבּוּר חַזָּן: {טו} וּבְשֻׁלְחָן עָרוּךְ סִימָן ער"ה סָעִיף ג' פָּסַק דְּמֻתָּר בְּאִשְׁתּוֹ שׁוֹמֶרֶת אוֹתוֹ. תּוֹסָפוֹת חֲדָשִׁים: {טז} רַשִׁ"י.</B> וּלְשִׁטָּתוֹ אָזַל שֶׁלֹּא מָנָה לְהַךְ דְּזָב וְזָבָה בִּכְלַל הַשְּׁמֹנָה עָשָׂר דָּבָר [שֶׁבַּמִשְׁנָה דְלְקַמָּן], וּמִפְּנֵי כָּךְ אִיצְטְרִיךְ לְתַנָּא לְמִתְנֵי דְּכַיּוֹצֵא בּוֹ הוּא וּדְמִשּׁוּם הָכִי תָּנֵי לֵיהּ הָכָא. תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ד

{יז} אִי תְּנַן אֹכֶל רִאשׁוֹן לְחוּד לֹא הֲוָה יָדְעִינַן אֹכֶל שֵׁנִי, וְאִי תָּנֵי אֹכֶל שֵׁנִי הֲוָה אֲמֵינָא דְּאֹכֶל רִאשׁוֹן נַמִּי טַמּוֹיֵי מְטַמֵּא, כְּלוֹמַר שֶׁהַתְּרוּמָה תִּהְיֶה טְמֵאָה לִפְסֹל אֲחֵרִים בְּמַגָּעָהּ. תּוֹסָפוֹת: {יח} וּמַקְשָׁה הַגְּמָרָא, הַיְנוּ הַךְ דִּגְזֵרַת אֳכָלִין.</B> וּמְשָׁנֵי מַהוּ דְּתֵימָא הָא שְׁכִיחַ שֶׁאָדָם שׁוֹתֶה עִם אֲכִילָתוֹ וְהָא לֹא שְׁכִיחַ שֶׁיְּהֵא אוֹכֵל עִם שְׁתִיָּתוֹ, קָא מַשְׁמַע לָן: {יט} וְלֹא חִלְּקָהּ נַמִּי לִשְׁתַּיִם כְּמוֹ בְאֹכֶל, מִשּׁוּם דְּלֹא שַׁיָּךְ בְּהוּ שֵׁנִי, דִּלְעוֹלָם מַשְׁקִין תְּחִלָּה הֲוָה, וְאַף עַל גַּב דְּאַכַּתִּי לֹא נִגְזָר [גְּזֵרָה דִּלְעוֹלָם מַשְׁקִין תְּחִלָּה], בְּדַעְתָּם הָיָה לִגְזֹר מִיָּד: {כ} [וְכֵן אִם נָפְלוּ עָלָיו שְׁלֹשָׁה לֻגִּין מַיִם שְׁאוּבִין]. וְגָזְרוּ דְפוֹסֵל אֶת הַתְּרוּמָה, וַאֲפִלּוּ כִּי הֶעֱרִיב שִׁמְשׁוֹ פּוֹסֵל עַד שֶׁיַּחְזֹר וְיִטְבֹּל: {כא} דִּבְטָהוֹר [שֶׁאֵינוֹ טְבוּל יוֹם] לֹא גָּזְרוּ אֶלָּא דַוְקָא בִּנְפִילָה כְּמוֹ שֶׁרְגִילִין לְהַפִּיל עֲלֵיהֶם מַיִם שְׁאוּבִים אַחַר שֶׁטָּבְלוּ בְּמַיִם סְרוּחִים. תּוֹסָפוֹת: {כב} וְלָאו גְּזֵרָה בִּפְנֵי עַצְמָהּ הִיא, דְּהָאֳכָלִין שֶׁנִּטְמְאוּ בְּמַשְׁקִין הַיְנוּ גְּזֵרָה דְּכָל הַפּוֹסֵל כוּ' וְעַד הַשְׁתָּא לֹא נִגְזְרָה. תּוֹסָפוֹת: {כג} הוּא הַדִּין דַּהֲוָה מָצֵי לְמִנְקָט מַשְׁקִים הַבָּאִים מֵחֲמַת יָדַיִם, שֶׁגַּם הֵם מְטַמְּאִים כֵּלִים לִפְסֹל אֶת הַתְּרוּמָה, כִּדְאִיתָא בִּבְרָכוֹת פֶּרֶק אֵלּוּ דְּבָרִים, אֶלָּא נָקַט מַשְׁקִים הַבָּאִים מֵחֲמַת שֶׁרֶץ לְאַשְׁמְעִינַן דַּאֲפִלּוּ הָנָךְ לֹא הָווּ אֶלָּא מִדְּרַבָּנָן. תּוֹסָפוֹת:

{א} לְפִי שֶׁהוּא מְפָרֵק אוֹתוֹ דָּבָר וּמַבְדִּילוֹ מֵחֲבֵרוֹ: {ב} כֵּן כָּתַב רַשִׁ"י.</B> וְהַתּוֹסָפוֹת כָּתְבוּ שֶׁאֵינוֹ נִרְאֶה, דְּאִם כֵּן אֲפִלּוּ לְצֹרֶךְ נַמִּי לִפְטֹר, אֶלָּא יֵשׁ לוֹמַר דַּאֲפִלּוּ בְּמִינוֹ נִצּוֹד פָּטוּר מִשּׁוּם דַּהֲוֵי מְלָאכָה שֶׁאֵין צְרִיכָה לְגוּפָהּ. וְכָתַב הָרַ"ן דְּהוּא הַדִּין נַמִּי בָּרֵישָׁא אֵינוֹ חַיָּב אֶלָּא לְצֹרֶךְ, אֶלָּא שֶׁאוֹתָן שְׁמֹנָה שְׁרָצִים אֵין דַּרְכָּן לְהַזִּיק וּלְפִיכָךְ כָּל צִידָתָן לְצֹרֶךְ, אֲבָל אִינָךְ דֶּרֶךְ לְצוּדָן שֶׁלֹּא יַזִּיקוּ. וּמַתְנִיתִין רַבִּי שִׁמְעוֹן הִיא דִּמְלָאכָה שֶׁאֵינָה צְרִיכָה לְגוּפָהּ פָּטוּר, וְתֵימַהּ עַל הָרַ"ב וְהָרַמְבַּ"ם שֶׁלֹּא כָּתְבוּ שֶׁאֵינָהּ הֲלָכָה: {ג} גְּמָרָא.</B> וְלַפֵּרוּשׁ שֶׁכָּתַב מִשּׁוּם מְפָרֵק נִיחָא, שֶׁצָּרִיךְ לַדָּם כְּדֵי לְהַאֲכִיל לִכְלָבָיו. אֲבָל לַפֵּרוּשׁ שֶׁמְּפָרֵשׁ מִשּׁוּם צוֹבֵעַ, וְאֵינוֹ יוֹצֵא, צָרִיךְ לוֹמַר כְּגוֹן שֶׁצָּרִיךְ וְנִיחָא לֵיהּ שֶׁיִּהְיֶה צָבוּעַ בְּעוֹרוֹ, וּדְנָקַט לְהַאֲכִיל לִכְלָבָיו חֲדָא מִתַּרְתֵּי נָקַט. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב. וְלַפֵּרוּשׁ דִּמְפָרֵק הָא דְּחַיָּב בְּנִצְרַר הַדָּם אַף עַל פִּי שֶׁלֹּא יָצָא וְלֹא נִתְקַיְּמָה מַחְשַׁבְתּוֹ, עַיֵּן פֶּרֶק י"ב מִשְׁנָה ג': 

~ פרק יד - משנה ב

{ד} הָכִי קָאָמַר רַבִּי יוֹסֵי בַּבְּרַיְתָא.</B> וְרַבָּנָן סְבִירָא לְהוּ שֶׁאֵין לֶאֱסֹר מִשּׁוּם מְעַבֵּד אֶלָּא כְּשֶׁעוֹשֶׂה כֵן בְּדָבָר הָעוֹמֵד לְהִתְקַיֵּם כְּמוֹ הַכְּבָשִׁים דְּהַשְׁתָּא דָּמִי לְעִבּוּד. אֲבָל לְצֹרֶךְ אֲכִילָה לִשְׁעָתוֹ לֹא דָמִי לְעִבּוּד. טוּר: 

~ פרק יד - משנה ג

{ה} וּמִשּׁוּם דְּאָיְרֵי בְּמֵי מֶלַח שֶׁהֵן עֲשׂוּיִין לְמַתֵּק הָאֹכֶל וְאֵינָן דְּבַר מַאֲכָל בְּעַצְמוֹ נָקַט נַמִּי לְהָנֵי שֶׁאֵינָן מַאֲכָל מַמָּשׁ וְאַגַּבַן נָקַט נַמִּי אִינָךְ שֶׁיֵּשׁ בָּהֵן רְפוּאָה. וְכָתַב הַמַּגִּיד בְּפֶרֶק כ"א דְּכָל אֵלּוּ הַדִּינִין הֵן בְּבָרִיא שֶׁאֵינוֹ נוֹפֵל לְמִשְׁכָּב וְלֹא חוֹלֶה, דְּהָתְנַן בְּמִשְׁנָה ו' פֶּרֶק ח' דְּיוֹמָא דְּחוֹשֵׁשׁ בִּגְרוֹנוֹ מְטִילִין לוֹ סַם בְּתוֹךְ פִּיו מִשּׁוּם סְפֵק נְפָשׁוֹת: {ו} <B> כָּל הָאֳכָלִין.</B> לַאֲתוֹיֵי טְחוֹל לַשִּׁנַּיִם וְכַרְשִׁינִין לִבְנֵי מֵעַיִן, כְּלוֹמַר שֶׁקָּשִׁים לָהֶם, וְסַלְקָא דַּעְתָּךְ אֲמֵינָא דְּלָאו אוֹרְחָא לְמֵיכְלִינְהוּ אִי לָאו לִרְפוּאָה שֶׁיֵּשׁ בָּהֶם וּמוּכְחָא מִלְּתָא וְלִתְּסַר, קָא מַשְׁמַע לָן. וְכָל הַמַּשְׁקִין לַאֲתוֹיֵי מֵי צְלָפִין בְּחֹמֶץ. גְּמָרָא: {ז} <B> וְכוֹס עִקָּרִים.</B> אַף עַל גַּב דִּמְרַפֵּא לַזָּבָה וְלֹא מֵעַקְרָא כִּדְאִיתָא בַּגְּמָרָא. קָרֵי לְהוּ אֱינָשֵׁי בְּשֵׁם עִקָּרִים כְּדֵי שֶׁיִּהְיֶה מְפֻרְסָם בְּשֵׁם זֶה וְיִזָּהֲרוּ מֵהֶם בְּנֵי אָדָם לִשְׁתּוֹתוֹ וְיִשְׁתַּדְּלוּ אַחַר רְפוּאָה אַחֶרֶת לְאֵלּוּ הַחֳלָיִים: {ח} <B> שֶׁמֶן עִקָּרִין.</B> נִרְאָה שֶׁרוֹצֶה לוֹמַר שֶׁנִּתְבַּשֵּׂם בְּעִקָּרֵי בְּשָׂמִים וִירָקוֹת. רַשִׁ"י פֵּרֵשׁ כֵּן בְּכוֹס עִקָּרִים: 

~ פרק יד - משנה ד

{ט} דְּחֹמֶץ מַעֲלֵי לְהוּ הֵיכָא דְּאִיכָּא מַכָּה דִּצְמִיתֵי, וּקְרָא דְּחֹמֶץ לַשִּׁנַּיִם כֵּן הֶעָצֵל לְשׁוֹלְחָיו (מִשְׁלֵי י) דְּמַשְׁמַע דְּקָשֶׁה לְהוּ, מַיְרֵי דְּלֵיכָּא מַכָּה, דִּמְרָפֵי. גְּמָרָא: {י} גְּמָרָא.</B> וְסֵיפָא דְּקָתָנֵי אֲבָל מְטַבֵּל כְּדַרְכּוֹ, הוּא הַדִּין דַּהֲוָה מָצֵי לְמִתְנֵי בִּמְגַמֵּעַ וּבוֹלֵעַ, אֶלָּא אוֹרְחָא דְּמִלְּתָא נָקַט. תּוֹסָפוֹת: {יא} וְהָרַמְבַּ"ם כָּתַב וַהֲלָכָה כְּרַבִּי שִׁמְעוֹן, וְצָרִיךְ שֶׁתִּהְיֶה אוֹתָהּ הַמְּשִׁיחָה מִשְׁתַּמְשִׁין בָּהּ בְּנֵי אָדָם הַרְבֵּה. וְהָכִי אָמַר רַב בַּגְּמָרָא דַּהֲלָכָה כְּרַבִּי שִׁמְעוֹן, וְלֹא מִטַּעֲמֵיהּ, דְּאִלּוּ רַבִּי שִׁמְעוֹן אַף עַל גַּב דְּלֹא שְׁכִיחַ שָׁרֵי, וְרַב סָבַר אִי שְׁכִיחַ אִין אִי לֹא לֹא, וּבְאַתְרָא דְּרַב שְׁכִיחַ מַשְׁחָא דְּוַרְדָא:

{א} <B> מִפְתַּח חֲלוּקָהּ.</B> בַּגְּמָרָא, פְּשִׁיטָא. לֹא צְרִיכָא דְּאִית לַהּ שְׁנֵי פְּתָחִים, דְּהַיְנוּ אוֹתָן לְשׁוֹנוֹת שֶׁזָּכַר הָרַ"ב, וְאִם אֵינָהּ מַתֶּרֶת אֶלָּא הָאֶחָד יְכוֹלָה לְפוֹשְׁטָהּ וּלְלוֹבְשָׁהּ בְּדֹחַק, מַהוּ דְּתֵימָא חַד מִנַּיְהוּ בִּטּוּלֵי מְבַטְּלָהּ לַהּ וְתֶיהֱוֵי קֶשֶׁר שֶׁל קְיָמָא וְתַרְוַיְהוּ לִיתַּסְרֵי, קָא מַשְׁמַע לָן: {ב} וַחֲסוּרֵי מִחַסְּרָא וְהָכִי קָתָנֵי, וְאֵלּוּ הֵן הַמֻתָּרִין לְכַתְּחִלָּה.</B> הָרַ"ן: {ג} <B> וְחוּטֵי סְבָכָה.</B> בַּגְּמָרָא, פְּשִׁיטָא. לֹא צְרִיכָא דִּרְוִיחָא לַהּ וְאֵינוֹ מְהֻדָּק בְּרֹאשָׁהּ, מַהוּ דְּתֵימָא מִישְׁלַף שָׁלְפָה לַהּ מֵרֹאשָׁהּ כְּשֶׁהִיא קְשׁוּרָה וַהֲוָה לֵיהּ קֶשֶׁר שֶׁל קְיָמָא, קָא מַשְׁמַע לָן דְּאִשָּׁה חָסָה עַל שַׂעֲרָה פֶּן תְּנַתֵּק הַשֵּׂעָר, וּמִישְׁרָא שָׁרֵי לֵיהּ: {ד} <B> וְשֶׁל פְּסִיקְיָא.</B> אִיכָּא נַמִּי לִשְׁנוּיֵי כְהַאי גַּוְנָא, מַהוּ דְּתֵימָא מְחַתָּה לֵיהּ דֶּרֶךְ רַגְלָהּ וְיוֹצְאָה, קָא מַשְׁמַע לָן דְּלֹא עָבְדֵי הָכִי מִשּׁוּם צְנִיעוּתָא. רַשִׁ"י: {ה} <B> וּרְצוּעוֹת וְכוּ'.</B> יֵשׁ בַּגְּמָרָא הַרְבֵּה צְדָדִין, מֵהֶן לְחִיּוּב מֵהֶן לִפְטוּר מֵהֶן לְהֶתֵּר, וְהַמֻּתָּרִים הֵם שֶׁקּוֹשְׁרִים אוֹתָם עַל הָרֶגֶל בִּשְׁעַת מַלְבּוּשׁ, שֶׁאֵלּוּ וַדַּאי אֵינָן מַעֲשֵׂה אֻמָּן וְלֹא שֶׁל קְיָמָא. הָרַמְבַּ"ם: {ו} וּבְאַחַת מֵהֶן יָכֹל לְהוֹצִיא הַיַּיִן.</B> רַשִׁ"י. וְנִרְאֶה לִי אֲבָל בְּדֹחַק, וְכֵן בִּקְדֵרָה שֶׁל בָּשָׂר, וּכְמוֹ שֶׁפֵּרְשׁוּ בְּכָל הָאֲחֵרִים. תּוֹסְפוֹת יוֹם טוֹב: {ז} <B> לִפְנֵי הַבְּהֵמָה.</B> בַּגְּמָרָא, פְּשִׁיטָא. לֹא צְרִיכָא דְּאִית לַהּ תַּרְתֵּי אִיסָּרֵי, פֵּרֵשׁ רַשִׁ"י שְׁנֵי חֲבָלִים קוֹשֵׁר בְּרֹחַב הַפֶּתַח זֶה לְמַעְלָה מִזֶּה. לִישָׁנָא אַחֲרִינָא, תְּרֵי אִיסָּרֵי שֶׁקּוֹשְׁרָן הַיּוֹם בִּשְׁנֵי צִדֵּי הַפֶּתַח וְלֹא הָיְתָה תְּלוּיָה בָּהּ מֵאֶתְמוֹל, מַהוּ דְּתֵימָא חַד מִנַּיְהוּ בַּטּוֹלֵי מְבַטֵּל וְיוֹצִיאֶנָּה בְּדֹחַק, קָא מַשְׁמַע לָן: {ח} מַסִּיק בַּגְּמָרָא דְּלֹא פָּלֵיג, דְּהָא דָּמִי לְהָנָךְ דִּלְעֵיל: {ט} שֶׁאֵין כְּלֵי הַגַּרְדִי מֻקְצִין. הָרַמְבַּ"ם. וְהָכִי {א} וְטַעְמָא שֶׁאֵלּוּ מְלַאכְתָּן לְהֶתֵּר הֵן וְאִלּוּ בִּשְׁאָר כֵּלִים פְּלִיגֵי לְקַמָּן.</B> הָרַ"ן: {ב} <B> בְּשַׁבָּת.</B> הָכִי קָאָמַר, אַף עַל פִּי שֶׁנִּתְפָּרְקוּ בְּחֹל נִטָּלִין בְּשַׁבָּת. גְּמָרָא:

{ה} <B> קְשׁוּרָה.</B> דְּשׁוּב לֹא יָבֹא לְתַקְּנָהּ. אֲבָל אֵינָהּ קְשׁוּרָה, אַף עַל פִּי שֶׁתִּקְנָהּ מִבְּעוֹד יוֹם וְיֵשׁ עָלֶיהָ תּוֹרַת כְּלִי, לֹא, דְּשֶׁמָּא לְמָחָר תְּהֵא לוֹ אֲרֻכָּה יוֹתֵר מִדַּאי וְיִקְטְמֶנָּה וְנִמְצָא מְתַקֵּן כְּלִי. גְּמָרָא:

{א} וּבַגְּמָרָא, כִּדְאָמְרֵי אֱינָשֵׁי, פֵּרוּשׁ חֶשְׁבּוֹן קָטָן תְּחִלָּה, וְאִם יִצְטָרֵךְ יוֹתֵר יְפַנֶּה יוֹתֵר (עַיֵּן בְּמַגִּיד מִשְׁנֶה פֶּרֶק כו הֲלָכָה טז וְשֻׁלְחָן עָרוּךְ סִימָן של"ג) (וּמִשּׁוּם הָכִי נָקַט אַרְבַּע בְּרֵישָׁא וַהֲדַר חָמֵשׁ). רַשִׁ"י. וְנִרְאֶה לִי דְּמִסְפָּר אַרְבַּע וְחָמֵשׁ מַסְרִיך סְרִיךְ אַלִּישָׁנָא דְּתַשְׁלוּמֵי אַרְבָּעָה וַחֲמִשָּׁה דְּפֶרֶק ז' דְּבָבָא קַמָּא, וְשָׁם בְּדַוְקָא: {ב} <B> קֻפּוֹת.</B> בַּיְרוּשַׁלְמִי, כְּהָא דִּתְנִינַן בִּשְׁקָלִים פֶּרֶק ג', קֻפּוֹת שֶׁל שָׁלֹשׁ סְאִים. תּוֹסָפוֹת: {ג} בַּגְּמָרָא. וְכָתְבוּ הַתּוֹסָפוֹת דַּהֲוֵי מָצֵי לְמֵימַר כְּדִלְעֵיל בִּתְרוּמָה כֵּיוָן דְּחָזֵי לְכֹהֵן, אֶלָּא כֵּיוָן דְּאַשְׁכַּח טַעְמָא דַּאֲפִלּוּ לְדִידֵיהּ מָצֵי חָזֵי אָמַר: {ד} <B> הַטֶּבֶל.</B> דִּלְכֹהֵן נַמִּי לֹא חָזֵי, דְּאָסוּר לְכֹהֵן לֶאֱכֹל מִפֵּרוֹתָיו בְּטִבְלָן, כְּמוֹ שֶׁכָּתַב הָרַמְבַּ"ם בַּפֶּרֶק א' מִמַּעַשְׂרוֹת: {ה} וּבְרֵישׁ פֶּרֶק ז' דִּבְרָכוֹת כָּתַב דִּכְתִיב וְנָתַן הַכֶּסֶף וְגוֹ', וְהָכִי אִיתָא בַּסּוּגְיָא דְּהָכָא וּדְהָתָם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ו} <B> עוֹרְבִין.</B> וְתַנָּא קַמָּא דְּפָלֵיג מִשּׁוּם דְּסָבַר דְּלֹא מְטַלְטְלִין דָּבָר שֶׁהוּא מַאֲכָל לַחַיָּה וְלָעוֹף כָּל הֵיכָא דְּלֵיתְנַיְהוּ בִּרְשׁוּתֵיהּ, אֶלָּא אִם כֵּן דֶּרֶךְ בְּנֵי אָדָם לְגַדֵּל אוֹתָן. תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק יח - משנה ב

{ז} <B> רוֹצֶה לוֹמַר מִשּׁוּם דִּסְתָמַיְהוּ לְהַסָּקָה קַיְמָא וּבָעֵי בְּהוּ הַזְמָנָה.</B> הָרַ"ן: {ח} <B> מְדַדִּין.</B> הִיא מִלָּה עִבְרִית כְּמוֹ אֶדַּדֵּם עַד בֵּית אֱלֹהִים [תְּהִלִּים מה]. הָרַמְבַּ"ם: {ט} קָשֶׁה דִּלְחַלֵּק בָּעוֹפוֹת בֵּין תַּרְנְגֹלֶת לְאַוָּזִין וּשְׁאָר עוֹפוֹת.</B> וּלְשִׁיטַת הָרַ"ן דִּמְחַלֵּק בֵּין גְּדוֹלִים לִקְטַנִּים דִּקְטַנִּים לֹא שָׁרֵי אֶלָּא בֶּחָצֵר אֲבָל לֹא בִּרְשׁוּת הָרַבִּים וּגְדוֹלִים אֲפִלּוּ בִּרְשׁוּת הָרַבִּים שָׁרֵי נִיחָא, דְּמַתְנִיתִין בִּרְשׁוּת הָרַבִּים מַיְרֵי מִכֹּחַ סֵיפָא דִּמְדַדָּה בְּנָהּ אוֹקְמִינַן לַהּ בַּגְּמָרָא בִּרְשׁוּת הָרַבִּים, וּלְהָכִי תָּנֵי עֲגָלִים וּסְיָחִים לְאַשְׁמְעִינַן דְּדַוְקָא בִּגְדוֹלִים שָׁרִינַן בִּרְשׁוּת הָרַבִּים: {י} <B> אָסוּר.</B> וְלֹא אָמְרִינַן חַי נוֹשֵׂא אֶת עַצְמוֹ כִּדִלְעֵיל מִשְׁנָה ה' פֶּרֶק י', כֵּיוָן שֶׁהוּא קָטָן כָּל כָּךְ שֶׁאֵינוֹ אֶלָּא גּוֹרֵר. הָרַ"ן. וְהַאי אָסוּר נִרְאֶה לִי דְּאָסוּר וְחַיָּב קָאָמַר, דְּהָא מְחַיֵּב בְּחוֹלֶה כְּשֶׁנְּשָׂאוֹ בְּרֵישׁ פִּרְקִין דִּלְקַמָּן. וְעַיֵּן שָׁם דְּלַתּוֹסָפוֹת נִרְאֶה דְּדַוְקָא בְּחוֹלֶה:

{יא} <B> אֶת הַשַּׁבָּת.</B> בַּגְּמָרָא לַאֲתוֹיֵי הָא דְּתָנוּ רַבָּנָן אִם הָיְתָה צְרִיכָה לְנֵר חֲבֵרְתָּהּ מַדְלֶקֶת לָהּ אֶת הַנֵּר. וּמֵהָכָא שָׁמְעִינַן דִּגְמָרָא סְבִירָא לַהּ דְּאִיכָּא אִסּוּר תְּחוּם מִן הַתּוֹרָה, וּכְמוֹ שֶׁכָּתַב הָרִי"ף דִּלְכֻלֵּי עָלְמָא מִשְּׁתֵּים עֶשְׂרֵה מִיל וּלְמַעְלָה מִדְּאוֹרַיְתָא, דְּאִם לֹא כֵן עַד הַשְׁתָּא לֹא שָׁמְעִינַן אֶלָּא לְאִסּוּר תְּחוּם מִדִּבְרֵי סוֹפְרִים. תּוֹסְפוֹת יוֹם טוֹב. וְעַיֵּן עוֹד: {יב} <B>צָרְכֵי מִילָה.</B> בַּשַּׁ"ס דַּף קל"ב נָפְקָא לֵיהּ מִקְּרָא:

~ פרק יט - משנה א

{א} וְקָאֵי אַמַּתְנִיתִין דִּלְעֵיל דִּסְלִיק מִנֵּיהּ וְכָל צָרְכֵי מִילָה עוֹשִׂין בְּשַׁבָּת: {ב} <B>מְבִיאוֹ בְשַׁבָּת.</B> וְאִם תֹּאמַר וְיָבִיאוּ הַתִּינוֹק אֵצֶל הַכְּלִי דְּלֵיכָּא אִסּוּר דְּהַחַי נוֹשֵׂא אֶת עַצְמוֹ. וְיֵשׁ לוֹמַר דִּלְאַחַר הַמִּילָה הָיָה צָרִיךְ לְהַחְזִירוֹ לְאִמּוֹ, וְאָז הוּא חוֹלֶה וַהֲוָה כָּפוּת דְּאֵינוֹ נוֹשֵׂא עַצְמוֹ, וְגַם אִמּוֹ מְסֻכֶּנֶת וְלֹא תּוּכַל לָבֹא אֶצְלוֹ. וְעוֹד יֵשׁ לוֹמַר כֵּיוָן דִּבְקַל יָבִיא הַכְּלִי מִשֶּׁיָּבִיא הַתִּינוֹק שָׁרִי רַבִּי אֱלִיעֶזֶר לְמַהֵר הַמִּצְוָה. תּוֹסָפוֹת: {ג} <B> עֵדִים.</B> בַּגְּמָרָא אִיבָּעֵי אִי עֵדִים שְׁנַיִם אוֹ אִיהוּ וְחַד שֶׁרָאוּי לְהָעִיד בְּמָקוֹם אַחֵר. וְאֵין לְהַקְשׁוֹת דְּאִם כֵּן לִתְנֵי עַל פִּי אֶחָד, דְּאִיצְטְרִיךְ לֵיהּ שֶׁהוּא וְחַד יִהְיוּ עֵדִים כְּשֵׁרִים יַחַד בְּמָקוֹם אַחֵר:

{ד} <B> עוֹשִׂין כָּל צָרְכֵי.</B> לַאֲתוֹיֵי הָא דְּתָנוּ רַבָּנָן הַמָּל כָּל זְמַן שֶׁהוּא עוֹסֵק בַּמִּילָה חוֹזֵר בֵּין עַל צִיצִין הַמְעַכְּבִין כוּ'. גְּמָרָא. וְאַף עַל גַּב דִּבְסוֹף פִּרְקִין דִּלְעֵיל קָתָנֵי לַהּ, הָכָא עִקָּר וְהָתָם אַגַּב. תּוֹסָפוֹת:

{ה} <B> בָּרָעִים.</B> כְּלִי מָלֵא מַיִם יָפִים לְתוֹךְ מִקְוֵה מַיִם רָעִים. רַשִׁ"י: {ו} כֵּן הוּא בְּפֵרוּשׁ רַשִׁ"י.</B> וְהָרַמְבַּ"ם בְּפֵרוּשׁוֹ, אֲבָל לֹא כְּנֶגֶד הָעָם, הִלְכָּךְ הָכָא נַמִּי טַעְמָא מְפָרֵשׁ דְּקָא מַשְׁמַע לָן דְּלֹא חָיֵשׁ לְהַאי חֲשָׁדָא. וּמִיהוּ אַכַּתִּי קָשֶׁה, מַאי שְׁנָא מִמִּשְׁנָה דִּלְקַמָּן דְּלֹא יָבִיא בְּיָדוֹ הָאֲלוּנְטִית שֶׁמָּא יִסְחֹט. וְהָרַמְבַּ"ם בְּחִבּוּרוֹ כָּתַב וְאֵין חוֹשֵׁשׁ שֶׁמָּא יִסְחֹט. וְיֵשׁ לְחַלֵּק דְּלֹא רָצוּ חֲזַ"ל לְהַצְרִיכוֹ לְהָסִיר בְּגָדָיו מֵעָלָיו מִשּׁוּם חֲשָׁשָׁא דְּשֶׁמָּא יִסְחֹט, כִּי גָּדוֹל כְּבוֹד הַבְּרִיּוֹת, וַאֲפִלּוּ בֶּגֶד אֶחָד מִבְּגָדָיו אַל יְשַׁנֶּה מִפְּנֵי כֵן, מַה שֶּׁאֵין כֵּן בַּאֲלוּנְטִיּוֹת שֶׁאֵינָן לְשׁוּם מַלְבּוּשׁ אֶלָּא מְבִיאָם בְּיָדוֹ גָּזְרוּ בָּהֶן שֶׁלֹּא יְבִיאֵם שֶׁמָּא יִסְחֹט:

{ז} <B> בְּמֵי מְעָרָה כוּ'.</B> מוֹקְמִינַן בַּגְּמָרָא דְּמֵי מְעָרָה חַמִּין דּוּמְיָא דְּמֵי טְבֶרְיָא, וּלְהָכִי קָתָנֵי הָרוֹחֵץ דִּיעֲבַד, דִּלְכַתְּחִלָּה לֹא יִרְחֹץ בְּחַמִּין, אֲבָל בְּמֵי טְבֶרְיָא רוֹחֵץ לְכַתְּחִלָּה, וְלֹא תָּנֵי לַהּ הָכָא אֶלָּא לְגִלּוּיֵי אַמֵּי מְעָרָה דְּחַמִּין הֵן, וְרוֹצֶה לוֹמַר חַמֵּי הָאוּר שֶׁהוּחֲמוּ מֵעֶרֶב שַׁבָּת. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} וְהוּא מִלְּשׁוֹן מִטְפַּחַת הַסְּפוֹג דְּמִשְׁנָה ג' פֶּרֶק ט' דְּכִלְאַיִם.</B> וְכָתַב הָרַ"ן דִּוְנִסְתַּפֵּג דְּמַשְׁמַע דִּיעֲבַד לָאו דַּוְקָא, דַּאֲפִלּוּ לְכַתְּחִלָּה מִסְתַּפֵּג כִּדְקָתָנֵי סֵיפָא כוּ' מִסְתַּפְּגִים כוּ', וְלֹא מַשְׁמַע לְחַלֵּק בֵּין אָדָם אֶחָד לַעֲשָׂרָה בְּנֵי אָדָם אֶלָּא לְעִנְיַן הֲבָאָה. וְכָתַב עוֹד וְלֹא גָּזְרִינַן שֶׁמָּא יָבֹא לִידֵי סְחִיטָה, שֶׁכֵּיוָן שֶׁכָּל הָרוֹחֲצִים מִסְתַּפְּגִים אִם אַתָּה אוֹסֵר עָלָיו לְהִסְתַּפֵּג אַף אַתָּה אוֹסְרוֹ בִּרְחִיצָה, וְאֵין הַדָּבָר עוֹמֵד כִּדְאִיתָא בְּפֶרֶק כִּירָה. עַד כָּאן. וְאִם תֹּאמַר וְהָאֵיךְ שָׁרִינַן לְהִסְתַּפֵּג בְּבֶגֶד, הָא תְּנַן בְּפֶרֶק כ"א מִשְׁנָה ב' הָיְתָה עָלָיו לַשְׁלֶשֶׁת מְקַנְּחָהּ, וּפֵרֵשׁ הָרַ"ב דְּלֹא יִתֵּן עָלֶיהָ מַיִם מִשּׁוּם דִּשְׁרִיָּתוֹ זֶה כִּבּוּסוֹ, וּמִיהוּ הָתָם כָּתַב הָרַ"ן כוּ' וּמַסִּיק דְּדַוְקָא בְּבֶגֶד מְטֻנָּף אָמְרִינַן שְׁרִיָּתוֹ הוּא כִּבּוּסוֹ. סֵפֶר הַתְּרוּמָה: {ט} <B> לֹא יְבִיאֵם בְּיָדוֹ.</B> שֶׁאִם נַתִּיר לוֹ לְהָבִיאָם יֵשׁ לָחוּשׁ שֶׁיִּסְחֹט כְּדֵי לַהֲבִיאוֹ וְהוּא יָבֵשׁ, אֲבָל כְּשֶׁנֹּאמַר לוֹ כִּי אֵין דֶּרֶךְ לְהוֹצִיא זֶה הַבֶּגֶד שֶׁנִּסְתַּפֵּג מִמְּקוֹם רְחִיצָתוֹ, יַנִּיחוֹ שָׁם וְלֹא יָחוּשׁ לוֹ, בֵּין שֶׁיִּהְיֶה יָבֵשׁ אוֹ שָׁרוּי. הָרַמְבַּ"ם:

{א} <B>כִּכָּרוֹת.</B> בַּגְּמָרָא אָמְרִינַן אֲפִלּוּ תֵּימָא מַתְנִיתִין כְּהִלֵּל דְּמִשְׁנָה ט' פֶּרֶק ה' דְּבָבָא מְצִיעָא, דִּסְבִירָא לֵיהּ דְּאַף בְּחֹל אָסוּר לְהַלְווֹת כִּכָּר בְּכִכָּר שֶׁמָּא יֻקְּרוּ חִטִּין וּבָאוֹת לִידֵי רִבִּית, דְּהָכָא בְּאַתְרָא דְּקִיץ דְּמֵיהּ. וּלְפִיכָךְ תָּנֵי וְכֵן הָאִשָּׁה, לְאַשְׁמְעִינַן דַּאֲפִלּוּ לְהִלֵּל כִּי קִיץ דְּמֵיהּ שָׁרֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} לַאֲפוֹקֵי חוֹבוֹת שֶׁאֵין קָבוּעַ לָהֶן זְמָן, דְּאֵין מַקְדִּישִׁין בְּיוֹם טוֹב, קַל וָחֹמֶר בְּשַׁבָּת. גְּמָרָא: {ג} <B> בְּשַׁבָּת.</B> וְלֹא נִזְכַּר מֵעֶרֶב שַׁבָּת לִקְנוֹת. רַשִׁ"י וְרַ"ן:

{ד} מִן הָאוֹרְחִים, שֶׁיִּרְאֶה שֶׁלֹּא הֵכִין כָּל צָרְכָּן וְיִתְחָרֵט וִיצַוֶּה לַשַּׁמָּשׁ שֶׁלֹּא לִקְרוֹתָם. רַשִׁ"י: {ה} שֶׁכֵּן הוּא בַּתַּרְגּוּם, גּוֹרָלְךָ תַּפִּיל בְּתוֹכֵנוּ [מִשְׁלֵי א], פַּיְסָךְ אַרְמֵי בֵּינָתָנָא. וְנִרְאֶה לִי שֶׁעִנְיָנוֹ כִּלְשׁוֹן לְהִתְחַנֵּן לוֹ [אֶסְתֵּר ד] שֶׁתִּרְגֵּם לְפַיְסָא לֵיהּ, שֶׁעַל יְדֵי גּוֹרָל מִתְפַּיְּסִין וּמִתְרַצִּין זֶה לָזֶה, לְפִי שֶׁהֻשְׁווּ הַמִּדּוֹת וְאֵין קִנְאָה וּמַחֲלֹקֶת: {ו} כְּלוֹמַר קְרוֹבִים לָבֹא לִידֵי כָּךְ שֶׁיַּעַבְרוּ מִשּׁוּם מִדָּה, שֶׁעַל יְדֵי שֶׁמַּקְפִּידִים זֶה לָזֶה וּרְגִילִים לִקַּח בְּמִדָּה זֶה מִזֶּה גַּם בְּיוֹם טוֹב יַעֲשׂוּ כָּךְ כוּ' וְכֵן בְּמִשְׁקָל וּמִנְיָן יָבֹאוּ לִידֵי מִשְׁקָל וּמִנְיָן מִפְּנֵי קְפִידָתָן, וּמִשּׁוּם לֹוִין וּפוֹרְעִין, דִּלְשׁוֹן הַלְוָאָה אָסוּר כְּדִלְעֵיל, וְהֵם שֶׁמַּקְפִּידִים יַזְכִּירוּ לְשׁוֹן הַלְוָאָה לְפִי שֶׁיֹּאמַר הַשּׁוֹאֵל אִם אַזְכִּיר לְשׁוֹן שְׁאֵלָה מַשְׁמַע דַּהֲדַרָה בְּעֵינֵיהּ, לְכָךְ אֹמַר לְשׁוֹן הַלְוָאָה. תּוֹסָפוֹת: {ז} מִשּׁוּם דְּכֵיוָן דְּהַכֹּל שֶׁל בַּעַל הַבַּיִת לֵיכָּא קְפֵדָא כְּלָל: {ח} סִירְכָא דְּלִישָׁנָא דִּגְמָרָא לְעֵיל בִּבְנֵי חֲבוּרָה כוּ' וּמִשּׁוּם לֹוִין וּפוֹרְעִין בְּיוֹם טוֹב (דִּבְבֵיצָה הֵם שְׁנוּיִין), אֲבָל הָכָא בְּשַׁבָּת הִיא: {ט} <B>קֻבְיָא.</B> שֵׁם נוֹפֵל עַל כָּל מִינֵי שְׂחוֹק שֶׁיִּקַּח אָדָם מְמוֹן חֲבֵרוֹ בְּהַסְכָּמַת אוֹתוֹ הַשְּׂחוֹק: {י} <B> חֲלָשִׁים.</B> מִין מִן הַגּוֹרָלוֹת: {יא} <B> בְּיוֹם טוֹב.</B> הָרַ"ן בְּשֵׁם הָרַמְבַּ"ם כָּתַב שֶׁהַטַּעַם כְּדֵי לְחַבֵּב אֶת הַמִּצְוָה. וּבַסֵּיפָא כָּתַב אֲבָל לֹא עַל הַמָּנוֹת, שֶׁל חֹל, קָדָשִׁים שֶׁל אֶתְמוֹל, כֵּיוָן דְּהָיָה אֶפְשָׁר לְמֶעְבַּד מֵאֶתְמוֹל. אֲבָל הָרַאֲבַ"ד וְהַמַּגִּיד מִשְׁנֶה פֵּרְשׁוּ דִּבְכָל קָדָשִׁים אֲפִלּוּ דְּאֶתְמוֹל מַתִּיר בְּיוֹם טוֹב מִשּׁוּם חִבּוּב מִצְוָה, דְּהוֹאִיל וּמִצְוָה יֵשׁ מַה לִּי דְּאֶתְמוֹל מַה לִּי דְּהַיּוֹם. וּמָנוֹת רוֹצֶה לוֹמַר שֶׁל חֻלִּין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{יב} וּכְתִיב לְעֵיל מִנֵּיהּ אִם תָּשִׁיב מִשַּׁבָּת רַגְלֶךָ וְגוֹ' מֵעֲשׂוֹת דְּרָכֶיךָ מִמְּצֹא וְגוֹ': {יג} דְּאִי לְגוּפֵיהּ פְּשִׁיטָא, מַאי שְׁנָא הוּא וּמַאי שְׁנָא חֲבֵרוֹ. גְּמָרָא. וּפֵרֵשׁ רַשִׁ"י דְּהָא יִשְׂרָאֵל כְּמוֹתוֹ הוּא וְלֹא מָצֵי לְאוֹגְרִינְהוּ, וְהַשּׁוֹלְחוֹ עוֹבֵר מִשּׁוּם וְלִפְנֵי עִוֵּר כוּ'. וְהִקְשׁוּ בַּתּוֹסָפוֹת וְנֵימָא אַלְּמָחָר קָאֵי, וְתֵרְצוּ דְּאִם כֵּן מַאי אַרְיָא חֲבֵרוֹ, אֲפִלּוּ הוּא נַמִּי לֹא מָצֵי לְמֵימַר אֶשְׂכֹּר לְמָחָר: {יד} <B> אֵין מַחְשִׁיכִין.</B> שֶׁהֲרֵי נִמְצָא הִלּוּכוֹ בְּשַׁבָּת לַעֲשׂוֹת חֲפָצָיו. הָרַמְבַּ"ם: {טו} כִּדְאִיתָא בַּגְּמָרָא מְשַׁדְּכִין עַל הַתִּינוֹקוֹת כוּ', אָמַר קְרָא מִמְּצֹא חֶפְצְךָ וְדַבֵּר דָּבָר, חֲפָצֶיךָ אָסוּר, חֶפְצֵי שָׁמַיִם מֻתָּר:

{טז} דְּאֵין דֶּרֶךְ לְהָבִיא חֲלִילִין אֶלָּא בִּשְׁבִיל מֵת. רַשִׁ"י. וְהָרַ"ב לֹא פֵּרֵשׁ עַד אֵימָתַי לֹא יִסְפֹּד בָּהֶן, נִרְאָה שֶׁדַּעְתּוֹ דְּבִכְדֵי שֶׁיָּבֹאוּ, וְהַיְנוּ דִּתְנַן אֶלָּא אִם כֵּן בָּאוּ מִמָּקוֹם קָרוֹב, וּמִסְּתָמָא כִּי בָּאוּ מִמָּקוֹם קָרוֹב יֵשׁ בּוֹ שִׁעוּר זֶה קֹדֶם שֶׁיִּסְפֹּד, שֶׁאֵין מַסְפִּידִין מִיָּד מִשֶּׁתֶּחְשַׁךְ. וְכֵן כָּתַב הָרַ"ן דְּהַחִלּוּק בֵּין חֲלִילִין לְאָרוֹן וְקֶבֶר, דְּאָרוֹן וְקֶבֶר בִּמְקוֹם פַרְהֶסְיָא הֵן מִן הַסְּתָם וְאָוְשָׁא מִלְּתָא טוּבָא וּגְנַאי הוּא לַמֵּת, אֲבָל חֲלִילִין מִסְּתָמָא בְּצִנְעָא מוּבָאִים, וְאֵין הָכִי נַמִּי דְּאָרוֹן וְקֶבֶר שֶׁנַּעֲשׂוּ בְּצִנְעָא דְּשָׁרֵי בִּכְדֵי שֶׁיַּעֲשׂוּ, וַחֲלִילִין שֶׁהוּבְאוּ בְּפַרְהֶסְיָא אֲסוּרִין עוֹלָמִית, אֶלָּא דְּמִשְׁנָתֵנוּ דִּבְּרָה בַּהֹוֶה לְהָא כִּדְאִיתָא וּלְהָא כִּדְאִיתָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יז} רַשִׁ"י.</B> וְהִקְשׁוּ בַּתּוֹסְפוֹת דַּהֲלֹא עִם כָּל זֶה הֱבִיאָם דֶּרֶךְ רְשׁוּת הָרַבִּים וּמַה לִּי אָרוֹן וְקֶבֶר שֶׁנַּעֲשָׂה בְּשַׁבָּת וּמַה לִּי חֲלִילִין שֶׁהוּבְאוּ דֶּרֶךְ רְשׁוּת הָרַבִּים, וְיֵשׁ לוֹמַר דְּהֵבִיא מִמָּקוֹם קָרוֹב לֹא מְהַנֵּי כָּל כָּךְ הֲבָאָתוֹ. וּלְשִׁיטַת הָרַ"ן צָרִיךְ לְהַמְתִּין עַד כְּדֵי שֶׁיְּבִיאוֹ מֵרְשׁוּת הָרַבִּים, מִמָּקוֹם שֶׁבָּאוּ: {יח} <B> יִקָּבֵר כוּ'.</B> מִיָּד, וְאֵין צָרִיךְ לְהַמְתִּין בִּכְדֵי שֶׁיַּעֲשֶׂה. וּלְרַשִׁ"י טַעְמָא כִּדְאִיתָא בַּגְּמָרָא דְּמוֹקִים לַהּ בְּעוֹמֵד בְּאַסְרַטְיָא, פֵּרֵשׁ רַשִׁ"י שֶׁאֵין דֶּרֶךְ יִשְׂרָאֵל לִקָּבֵר שָׁם, דְּוַדַּאי לֹא נַעֲשָׂה בִּשְׁבִיל יִשְׂרָאֵל. וְאָרוֹן מוֹקֵי בְּמֻטָּל עַל קִבְרוֹ שֶׁל נָכְרִי בְּאַסְרַטְיָא, וּלְהָרַ"ן מַיְרֵי דְּהַדָּבָר יָדוּעַ שֶׁעָשָׂה לוֹ לְעַצְמוֹ, וְאֵין צָרִיךְ הוֹכָחָה מִתּוֹךְ הַמַּעֲשֶׂה, אֶלָּא שֶׁיְּהֵא נוֹדָע שֶׁעֲשָׂאוֹ לְעַצְמוֹ:

{יט} לִישְׁנָא דִּבְרַיְתָא בַּגְּמָרָא הוּא: {כ} וְאִם תֹּאמַר דִּבְפֶרֶק י"ח מִשְׁנָה ב' דְּדוֹחִין לְתַרְנְגֹלֶת אֲבָל לֹא מְטַלְטְלִין, וְאַמַּאי דּוֹחִין שֶׁנִּמְצָא שֶׁמְּטַלְטֵל מִקְצָתוֹ, וְהָכָא אָסְרִינַן אֲפִלּוּ לָזוּז מִקְצָתוֹ. וְנִרְאָה דִּלְעֵיל מִשּׁוּם צַעַר בַּעֲלֵי חַיִּים נָגְעוּ, וְדַוְקָא בְּשֶׁצְּרִיכִין לְכָךְ, אֲבָל בְּשֶׁאֵין צְרִיכִין לְכָךְ לֹא. הָרַ"ן: {כא} וְעִנְיָנוֹ כָּאן שֶׁיִּתְלַחְלֵחַ.</B> כְּמוֹ בְּרֵישׁ פֶּרֶק לֹא יַחְפֹּר [בָּבָא בַּתְרָא דַּף יח], מְתוּנָא קָשָׁה לַכֹּתֶל: {כב} <B> בַּאֲרֻכּוֹת.</B> בְּכָ"ף גָּרְסִינַן בְּמִשְׁנָה ח' פֶּרֶק קַמָּא דְּסֻכָּה. וְהֵם הַלּוּחוֹת שֶׁבִּצְדָדֵי הַמִּטָּה שֶׁהֵם אַרֻכִּין: {כג} <B> אֵין מְעַמְּצִין.</B> כְּמוֹ עוֹצֵם עֵינָיו [יְשַעְיָה לג]. וְהַלָּשׁוֹן מְהֻפָּךְ כְּמוֹ כֶּבֶשׂ כֶּשֶׂב:

{א} וְכֵן פֵּרֵשׁ רַשִׁ"י.</B> וְיֵשׁ לְהַקְשׁוֹת דִּלְקַמָּן גַּבֵּי מֵנִיחַ כִּיסוֹ עַל הַחֲמוֹר מְפָרֵשׁ הָרַ"ב מִשֶּׁתֶּחְשַׁךְ, וְהָכָא בָּעִינַן מִבְּעוֹד יוֹם. וְהָרֹא"שׁ הִקְשָׁה דְּמַה צָּרִיךְ לַטַּעַם דְּנָכְרִי אִי אַתָּה מְצֻוֶּה כוּ', הָא בֵּית הִלֵּל מַתִּירִין מִבְּעוֹד יוֹם עַל יְדֵי קְצִיצַת שָׂכָר, וּמַסִּיק דְּאֵין הָכִי נַמִּי דְּהָכָא מִשֶּׁתֶּחְשַׁךְ מַיְרֵי. וְכָתַב בֵּית יוֹסֵף סִימָן רס"ו דְּאַף עַל פִּי שֶׁהַמָּעוֹת מֻקְצִין הֵם בִּמְקוֹם פְּסֵידָא לֹא גָּזְרוּ: {ב} וּלְהָכִי אִי לֵית לֵיהּ לֹא נָכְרִי וְלֹא חֲמוֹר עָשׂוּ חֲזַ"ל עוֹד תַּקָּנָה אַחֶרֶת, שֶׁמּוֹלִיכוֹ בְּעַצְמוֹ בְּפָחוֹת פָּחוֹת מֵאַרְבַּע אַמּוֹת, אֶלָּא שֶׁחַכְמֵי הַמִּשְׁנָה לֹא רָצוּ לְגַלּוֹת תַּקָּנָה זוֹ מִפְּנֵי שֶׁנֶּאֱמַר [מִשְׁלֵי כה] כְּבוֹד אֱלֹהִים הַסְתֵּר דָּבָר, דִּלְמָא אָתֵי לַאֲתוֹיֵי אַרְבַּע אַמּוֹת בִּרְשׁוּת הָרַבִּים: {ג} פֵּרוּשׁ קֹדֶם שֶׁתַּעֲמֹד, דְּלֹא תַּעֲשֶׂה הַנָּחָה, כְּמוֹ שֶׁכָּתַב הָרַמְבַּ"ם. וְכָתַב עוֹד דִּכְשֶׁיַּגִּיעַ לְפֶתַח בֵּיתוֹ יִטְּלֶנּוּ מֵעָלָיו כְּשֶׁהוּא מְהַלֵּךְ וְיִזְרְקֶנּוּ לְתוֹךְ בֵּיתוֹ כִּלְאַחַר יָד, כְּדֵי שֶׁלֹּא יְהֵא מַכְנִיס מֵרְשׁוּת הָרַבִּים לִרְשׁוּת הַיָּחִיד: 

~ פרק כד - משנה ב

{ד} וְאֶגֶד אֵין בּוֹ קֶשֶׁר הָאָסוּר לְהַתִּיר בְּשַׁבָּת: {ה} <B> הֶחָרוּבִין.</B> דּוּמְיָא דְּשַׁחַת דִּרְכִּיכָא. וּמִשּׁוּם הָכִי אֵין מְחַתְּכִין, דְּמִטְרָח בְּאֹכֶל לֹא מִטְרְחִינַן: 

~ פרק כד - משנה ג

{ו} <B> אֵין אוֹבְסִין.</B> לְשׁוֹן הָרַמְבַּ"ם, אֵין מַאֲכִילִין בְּהֵמָה חַיָּה וָעוֹף כְּדֶרֶךְ שֶׁהוּא מַאֲכִיל בְּחֹל, שֶׁמָּא יָבֹא לִידֵי כְּתִישָׁה אוֹ לִישָׁה: {ז} <B> מַלְעִיטִין.</B> לְשׁוֹן הַלְעִיטֵנִי נָא [בְּרֵאשִׁית כח]: 

~ פרק כד - משנה ד

{ח} הַתָּלוּשׁ מֵאֶתְמוֹל, דִּמְחֻבָּר כֵּיוָן דְּלֹא לִקְּטוֹ מִבְּעוֹד יוֹם אַקְצִינְהוּ מִדַּעְתֵּיהּ. בֵּית יוֹסֵף: {ט} וּתְמִיהָה דְּהָא אַף לְאָדָם לֹא קַיְּמֵי בֵּין הַשְּׁמָשׁוֹת [לִשְׁחֹט], מִסְּפֵק שַׁבָּת.</B> וּבַגְּמָרָא פְּסָחִים דַּף נ"ו קָאָמַר מֵהַךְ מִשְׁנָה דִּלְרַבִּי יְהוּדָה מוּכָן לָאָדָם לֹא הָוֵי מוּכָן לַבְּהֵמָה. וּפֵרְשׁוּ הַתּוֹסָפוֹת דְּמוּכָן לָאָדָם הַיְנוּ שֶׁעוֹמֶדֶת לַאֲכִילַת אָדָם וְלֹא הִקְצוּהָּ שֶׁלֹּא יֵהָנֶה בָּהּ אָדָם, אֶלָּא שֶׁאֲסוּרָה מֵחֲמַת אִסּוּר שַׁבָּת, וְהִלְכָּךְ אִי לָאו דְּאָמְרִינַן מוּכָן לָאָדָם לֹא הָוֵי מוּכָן לַכְּלָבִים הָוֵי שָׁרֵי לִתֵּן לַכְּלָבִים, דְּלִכְלָבִים רָאוּי אֲפִלּוּ חַיִּים כְּגוֹן עוֹפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק כד - משנה ה

{י} לְפִי שֶׁהַנֶּדֶר הַמֻּתָּר עַל יְדֵי חָכָם צָרִיךְ חֲקִירָה עַל הַחֲרָטָה אִם הִיא מֵעִקָּרָא וְעַל הַפֶּתַח אִם פּוֹתְחִין בָּהּ, לְכָךְ תָּנֵי נִשְׁאָלִין, שֶׁהֶחָכָם שׁוֹאֵל בָּהֶן. אֲבָל הֲפָרָה דִּלְאִשָּׁה הֵנִיחוּ בִּלְשׁוֹן הַכָּתוּב, הָפֵר יָפֵר [בְּמִדְבַּר ל], שֶׁאֵינוֹ צָרִיךְ שְׁאֵלָה: {יא} פֵּרוּשׁ אִם הָיְתָה מִתְּחִלָּה שָׁלֹשׁ עַל שָׁלֹשׁ וְנִטְמֵאת, וְנִפְחֲתָה, שׁוּב אֵינָהּ מְטַמְּאָה: {יב} פֵּרוּשׁ הַפָּתוּחַ כְּנֶגֶד הַבַּיִת שֶׁהַטֻּמְאָה בְּתוֹכוֹ. וּלְכָךְ נָתְנוּ הַטָּפִיחַ גַּבּוֹ מוּל הַשְּׁבִיל, כְּדֵי שֶׁלֹּא יְקַבֵּל טֻמְאָה מֵהַמֵּת שֶׁבַּבַּיִת, שֶׁהַטֻּמְאָה יוֹצְאָה מִמֶּנָּה וּמִתְפַּשֶּׁטֶת תַּחַת הַגִּיגִית עַד לַטָּפִיחַ. וּמִשּׁוּם הָכִי לֹא פֵּרֵשׁ שֶׁפָּקְקוּ הַחַלּוֹן שֶׁל הַבַּיִת שֶׁהַמֵּת בְּתוֹכוֹ, דְּמִסְּתָמָא אוֹתוֹ שֶׁחוֹשֵׁשׁ פֶּן יִתְטַמֵּא הוּא שֶׁפּוֹקֵק וְלֹא אוֹתוֹ שֶׁכְּבָר טָמֵא, דְּמַאי אִכְפַּת לֵיהּ. תּוֹסְפוֹת יוֹם טוֹב: {יג} וְצָרִיךְ לוֹמַר שֶׁבִּשְׁנֵי צִדֵּי הַסֶּדֶק לֹא הָיָה בְּרֹחַב שׁוּם אֶחָד מֵהֶן טֶפַח.</B> דְּאִי לָאו הָכִי אַכַּתִּי יֵשׁ אֹהֶל טֶפַח הַמֵּבִיא הַטֻּמְאָה לְבַיִת שֵׁנִי. תּוֹסְפוֹת יוֹם טוֹב: {יד} וְכָתְבוּ הַתּוֹסְפוֹת דִּפְקִיקָה לֹא בָעִינַן שֶׁל מִצְוָה, כְּלוֹמַר דְּשֶׁל מִצְוָה לֹא מַעֲלֶה וְלֹא מוֹרִיד בִּפְקִיקָה, דְּדַוְקָא מְדִידָה שֶׁאֵינָהּ אִסּוּר כָּל כָּךְ אֶלָּא מִשּׁוּם דַּהֲוֵי כְּעֻבְדָּא דְּחֹל מְהַנֵּי שֶׁל מִצְוָה, אֲבָל פְּקִיקָה דְּמִשּׁוּם תּוֹסֶפֶת אֹהֶל דְּדָמִי לְבִנְיָן לֵיכָּא לְמִשְׁרֵיהּ מִשּׁוּם מִצְוָה. וְהֶאֱרִיךְ הַתּוֹסְפוֹת יוֹם טוֹב וּמַסִּיק, כְּלָלוֹ שֶׁל דָּבָר, מְדִידָה שֶׁאֵינָהּ אֶלָּא כְּעֻבְדָּא דְּחֹל שַׁיָּךְ גַּבָּהּ לוֹמַר הוֹאִיל וְשֶׁל מִצְוָה הִיא שָׁרֵי. וּפְקִיקָה שֶׁיֵּשׁ בָּהּ מִשּׁוּם בִּנְיָן לֹא שַׁיָּךְ לוֹמַר גַּבָּהּ דְּמִשּׁוּם מִצְוָה נַתִּיר, אֶלָּא שֶׁאִם נַתִּיר יֵשׁ לְהַתִּיר אַף בְּלֹא מִצְוָה. וּקְשִׁירָה דְּלֹא נִכָּר בּוֹ בֵּין שֶׁל קְיָמָא לְאֵינָהּ שֶׁל קְיָמָא וְאִיכָּא לְמַרְאִית עַיִן מְלָאכָה דְּאוֹרַיְתָא, לֹא הִתִּירוּהָ אֶלָּא דַּוְקָא בְּמִצְוָה:
