{א} <b>נִקְנֵית.</b> לְבַעְלָהּ לְהַצְרִיכָהּ מִמֶּנּוּ גֵט. הָרַ"ן: {ב} וְאַכַּתִּי קָשְׁיָא דְלִתְנֵי מִתְקַדֶּשֶׁת. וּמְתָרֵץ בַּגְּמָרָא מִשּׁוּם דְּקָא בָּעֵי לְמִתְנֵי כֶסֶף, דְּיָלְפִינַן קִיחָה קִיחָה, וְקִיחָה אִקְּרֵי קִנְיָן דִּכְתִיב הַשָּׂדֶה אֲשֶׁר קָנָה אַבְרָהָם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} <B> בְּשָׁלֹשׁ כוּ'.</B> מִנְיָנָא לְמַעוֹטֵי חֲלִיפִין. דְּלֹא תֵימָא הוֹאִיל וְגָמְרִינַן מִשְּׂדֵה עֶפְרוֹן, מָה הָתָם בַּחֲלִיפִין אַף כָּאן בַּחֲלִיפִין, קָא מַשְׁמַע לָן דְּלֹא, דְּאִשָּׁה בְּפָחוֹת מִשָּׁוֶה פְרוּטָה לֹא מַקְנְיָא נַפְשָׁהּ, וַחֲלִיפִין אֲפִלּוּ בְּפָחוֹת מִשָּׁוֶה פְרוּטָה. וְלֹא דִּבֵּר הַכָּתוּב בְּקִיחָה דְאִשָּׁה אֶלָּא בְקִנְיַן דָּמִים. גְּמָרָא: {ד} <B> וְקוֹנָה.</B> לִהְיוֹת בִּרְשׁוּתָהּ לְהִנָּשֵׂא לְאַחֵר. רַשִׁ"י: {ה} <B> בִּשְׁתֵּי.</B> מִנְיָנָא לְמַעוֹטֵי חֲלִיצָה דְּלָא תֵּיתֵי מִקַּל וָחֹמֶר מִיבָמָה שֶׁאֵינָהּ יוֹצְאָה בְגֵט כוּ', תַּלְמוּד לוֹמַר כוּ'. גְּמָרָא: {ו} <B> בְּכֶסֶף.</B> וְצָרִיךְ שֶׁיִּהְיֶה לְשֵׁם קִדּוּשִׁין וְלִפְנֵי עֵדִים. וְתֵימַהּ עַל הָרַ"ב וְהָרַמְבַּ"ם שֶׁלֹּא פֵרְשׁוּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} אִיבְרָא דְבַגְּמָרָא הָכִי אִיתָא, אֲבָל הָרַ"ב מְסַיֵּם וּנְתָנוֹ לָהּ, מִשּׁוּם דְּהוּא הַדִּין בְּמִתְקַדֶּשֶׁת שֶׁלֹּא עַל יְדֵי אָבִיהָ אָיְרֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} וְלָאו דַוְקָא, אֶלָּא נִתְיַחֵד עִמָּהּ: {ט} <B> בְּדִינָר.</B> טַעֲמַיְיהוּ דְבֵית שַׁמַּאי שֶׁלֹּא יְהוּ בְנוֹת יִשְׂרָאֵל כְּהֶפְקֵר. גְּמָרָא. וּמָצָאנוּ שִׁעוּר דִּינָר שֶׁהוּא שִׁעוּר חָשׁוּב: {י} <B> וּבְשָׁוֶה.</B> וְהַטַּעַם דְּכֵיוָן שֶׁהִיא מִתְרַצֵּית בַּדָּבָר וְנִיחָא לָהּ בְּשָׁוֶה כֶסֶף כְּכֶסֶף דָּמֵי, וְאֵין צָרִיךְ קְרָא לְרַבּוֹיֵי. הָרַ"ן: {יא} <B> וְכַמָּה כוּ'.</B> הָא דְלָא מְפָרֵשׁ כַּמָּה הוּא דִינָר מִשּׁוּם דְּדִינָר הָיוּ יוֹדְעִים כַּמָּה הָיָה כוּ'. וְעוֹד דְּדִבְרֵי בֵית הִלֵּל דְּקָיְמָא לָן כְּוָתֵיהּ אִיבָּעֵי לֵיהּ לְפָרוֹשֵׁי טְפֵי. תּוֹסָפוֹת: {יב} <B> הָאִיטַלְקִי.</B> הִיא אִיטַלְיָא שֶׁל יָוָן. הָרַ"ב: {יג} <B> בְּגֵט כוּ'.</B> בַּגְּמָרָא יָלְפִינַן לְהוּ מִקְּרָאֵי: {יד} כִּדְיָלֵיף לַהּ בַּגְּמָרָא מִקְּרָאֵי: {טו} <B> בַּחֲלִיצָה.</B> יָלֵיף לֵיהּ בַּגְּמָרָא מִקְּרָא: {טז} <B> וּבְמִיתַת כוּ'.</B> קַל וָחֹמֶר, וּמָה אֵשֶׁת אִישׁ שֶׁהִיא בְחֶנֶק מִיתַת הַבַּעַל מַתִּירָתָהּ, יְבָמָה שֶׁהִיא בְלָאו לֹא כָל שֶׁכֵּן כוּ'. גְּמָרָא:

{יז} <B> בְּכֶסֶף.</B> וְלֹא פֵרְשָׁה מִשְׁנָתֵנוּ זוֹ בְּכַמָּה. וְהָרַמְבַּ"ם כָּתַב בְּאָמָה, וְאֵינָהּ נִקְנֵית בִּפְרוּטָה מִפְּנֵי שֶׁצָּרִיךְ לִקְנוֹתָהּ בְּדָמִים שֶׁרְאוּיִים לְגֵרָעוֹן כְּדֵי שֶׁתִּגְרַע פִּדְיוֹנָהּ וְתֵצֵא. וּמָצָאתִי לוֹ רְאָיָה כוּ', וּבְעֶבֶד עִבְרִי לֹא כָתַב כֵּן. וְכָתַב הַכֶּסֶף מִשְׁנֶה מַשְׁמַע דְּדִינוֹ בִּפְרוּטָה, וְקָשֶׁה דְהָא אִתַּקַּשׁ לְעִבְרִיָּה: {יח} וְכֵן כָּתַב הָרַמְבַּ"ם. וּבַגְּמָרָא, אַשְׁכְּחַן עֶבֶד עִבְרִי הַנִּמְכָּר לְנָכְרִי כוּ', נִמְכָּר לְיִשְׂרָאֵל מְנָלָן. אָמַר קְרָא וְהֶפְדָּהּ, מְלַמֵּד שֶׁמְּגָּרֵעַ[ת] פִּדְיוֹנָהּ וְיוֹצְאָה. וּפֵרֵשׁ רַשִׁ"י אַלְמָא בְּכֶסֶף נִקְנָה, דְּאִי לֹא קְנָאָהּ בְּדָמִים מַאי מְגָרְעָא. וְהֻקַּשׁ עִבְרִי לְעִבְרִיָּה. אַשְׁכְּחַן מְכָרוּהוּ בֵּית דִּין הוֹאִיל וְנִמְכָּר בְּעַל כָּרְחוֹ, מוֹכֵר עַצְמוֹ מְנָלָן, יָלֵיף שָׂכִיר שָׂכִיר. וּפֵרֵשׁ רַשִׁ"י נֶאֱמַר בְּמוֹכֵר עַצְמוֹ כְּשָׂכִיר כְּתוֹשָׁב יִהְיֶה עִמָּךְ, וְנֶאֱמַר בִּמְכָרוּהוּ בֵּית דִּין כִּי מִשְׁנֶה שְׂכַר שָׂכִיר: {יט} כְּלוֹמַר קִיחַת קִדּוּשִׁין. וּלְפִי זֶה שְׁטָר דְּאָמָה נַמִּי הַקּוֹנֶה כוֹתֵב בִּתְּךָ קְנוּיָה לִי. וּבַגְּמָרָא פְלִיגֵי בָהּ. וְהָרַמְבַּ"ם פּוֹסֵק כְּמַאן דְּאָמַר דְּכוֹתֵב בִּתִּי קְנוּיָה לְךָ: {כ} וְהַךְ קְרָא בִּמְכָרוּהוּ בֵּית דִּין כְּתִיב, אֲבָל בְּמוֹכֵר עַצְמוֹ דִּינוֹ שֶׁמּוֹכֵר עַצְמוֹ אֲפִלּוּ לְעֶשְׂרִים שָׁנָה, וְאִם פָּגַע בּוֹ יוֹבֵל אֲפִלּוּ אַחַר שָׁנָה אַחַת יוֹצֵא, שֶׁנֶּאֱמַר עַד שְׁנַת הַיּוֹבֵל יַעֲבֹד עִמָּךְ. וְנִרְאֶה לִי דְּהוֹאִיל וּמִכָּל מָקוֹם יוֹצֵא בַּשָּׁנִים שֶׁקָּצַב בִּמְכִירָתוֹ, שַׁפִּיר תָּנֵי סְתָם דְּעֶבֶד עִבְרִי יוֹצֵא בַּשָּׁנִים, וְאֶלָּא מִיהָא הָא כִדְאִיתָא וְהָא כִדְאִיתָא: {כא} <B> וּבַיּוֹבֵל.</B> בַּגְּמָרָא יָלֵיף לְהוּ מִקְּרָא: {כב} <B> וּבְגִרְעוֹן כֶּסֶף.</B> תָּנָא וְקוֹנֶה אֶת עַצְמוֹ בְכֶסֶף וּבְשָׁוֶה כֶסֶף. וְיָלֵיף לֵיהּ מִקְּרָא כוּ'. וְדַע דְּבַבְּרַיְתָא תָנֵי וּבִשְׁטָר, כְּלוֹמַר שְׁטַר שִׁחְרוּר. וְכָתְבוּ הַתּוֹסָפוֹת דְּתַנָּא דְמַתְנִיתִין לֹא חָשֵׁיב אֶלָּא יְצִיאוֹת שֶׁהֵם בְּעַל כָּרְחוֹ שֶׁל הָאָדוֹן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} מֵהֶקֵּישָׁא דְעִבְרִי וְעִבְרִיָּה. הָרַמְבַּ"ם: {כד} וְכֵיוָן דְּאִיתָא נַמִּי בָּאִישׁ לֹא קָתָנֵי. גְּמָרָא:

{כה} <B> בְּכֶסֶף כוּ'.</B> בַּגְּמָרָא דְּהוּא הַדִּין בַּחֲלִיפִין וּבִמְשִׁיכָה, וּמִלְּתָא דְאִיתָא בְּמִטַּלְטְלֵי לֹא קָתָנֵי: {כו} וְהַאי קְרָא בְפָרָשַׁת בְּהַר כְּתִיב: {כז} <B>בְּכֶסֶף.</B> וּבַיּוֹבֵל לֹא. וְאַף עַל גַּב דְּאִתַּקִּישׁ לְקַרְקָעוֹת לִשְׂדֵה אֲחֻזָּה, וּשְׂדֵה אֲחֻזָּה יוֹצֵאת בַּיּוֹבֵל כַּמְּפֹרָשׁ בַּכָּתוּב, בְּעֶבֶד כְּתִיב לְעוֹלָם בָּהֶם תַּעֲבֹדוּ, כִּדְאִיתָא בַגְּמָרָא: 

{כח} <B> וּבִשְׁטָר.</B> דְּגָמַר לָהּ לָהּ מֵאִשָּׁה: {כט} <B> וּלְפִי זֶה כָּל שֶׁכֵּן עַל יְדֵי עַצְמוֹ.</B> וְאִם תֹּאמַר לִיעָרְבִינְהוּ וְלִתְנֵי בְּכֶסֶף וּבִשְׁטָר בֵּין עַל יְדֵי עַצְמוֹ בֵּין עַל יְדֵי אֲחֵרִים. בַּגְּמָרָא מְשַׁנִּינַן דְּהַךְ בָּבָא וּבִשְׁטָר כוּ' לָאו חֲכָמִים אָמְרֵי לַהּ אֶלָּא רַבִּי שִׁמְעוֹן בֶּן אֶלְעָזָר הִיא, וְשָׁלֹשׁ מַחֲלֹקוֹת בַּדָּבָר. וּפֵרֵשׁ רַשִׁ"י הָכִי קָאָמְרֵי לֵיהּ רַבָּנָן לְרַבִּי מֵאִיר, אַף בְּכֶסֶף עַל יְדֵי עַצְמוֹ, דְּיֵשׁ קִנְיָן, וְכָל שֶׁכֵּן עַל יְדֵי אֲחֵרִים דִּזְכוּת הוּא לוֹ, וְכָל שֶׁכֵּן שְׁטָר דִּזְכוּת. וְרַבִּי שִׁמְעוֹן בֶּן אֶלְעָזָר אוֹמֵר בִּשְׁטָר עַל יְדֵי אֲחֵרִים, וְהָכִי קָאָמַר, אַף בִּשְׁטָר, וְכָל שֶׁכֵּן בְּכֶסֶף, דְּעַל יְדֵי אֲחֵרִים דַּוְקָא: {ל} יָלֵיף לֵיהּ בַּגְּמָרָא מִכְּלָל וּפְרָט וּכְלָל: {לא} יָלֵיף לַהּ שִׁלּוּחַ שִׁלּוּחַ מֵאִשָּׁה: {לב} שָׁמְעִינַן מִנָּהּ דִּסְבָרַת הָרַ"ב דַּהֲלָכָה כִּי אָמַר עַל מְנָת שֶׁאֵין לְרַבּוֹ רְשׁוּת בּוֹ מַהֲנֵי וְלֹא קָנָה רַבּוֹ, וְאִם כֵּן כָּל שֶׁכֵּן אִשָּׁה שֶׁאִם אָמְרוּ לָהּ כֵּן שֶׁלֹּא קָנָה בַּעְלָהּ. שֶׁהֲרֵי הָעֶבֶד גּוּפוֹ קָנוּי לְרַבּוֹ וְלֹא כֵן הָאִשָּׁה לְבַעְלָהּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{לג} לָאו דַוְקָא, דְּאֵין צָרִיךְ שֶׁיִּמְסְרֶנָּה מִיָּד לְיָד. אֶלָּא כָּל שֶׁאֲחָזָהּ בְּמִצְוָתוֹ סַגִּי. הָרַ"ן: {לד} <B> בְּהַגְבָּהָה.</B> שְׁלֹשָׁה טְפָחִים, דְּנָפְקָא לֵיהּ מִתּוֹרַת לָבוּד. רַשִׁ"י. וְלֹא דָּמֵי לְעֵרוּב דְּמִשְׁנָה ו' פֶּרֶק ז' דְּעֵרוּבִין דְּהַגְבָּהָה טֶפַח סַגִּי לִזְכּוֹת בּוֹ. דְּשָׁאנֵי שִׁתּוּפֵי מְבוֹאוֹת דְּרַבָּנָן לֹא הִצְרִיכוּ קְנִיָּה כָּל כָּךְ. וְרַבֵּנוּ תָּם מְפָרֵשׁ דְּכָל הַגְבָּהָה בְּטֶפַח קָנֵי: {לה} מַשְׁמַע דְּבִמְסִירָה לֹא. שְׁמַע מִנָּהּ דִּמְשִׁיכָה עֲדִיפָא מִמְּסִירָה: {לו} טַעֲמָא דְמִלְּתָא, מִשּׁוּם דִּמְשִׁיכָה הַיְנוּ שֶׁמּוֹשֵׁךְ הַדָּבָר לִרְשׁוּתוֹ, וְאִלּוּ רְשׁוּת הָרַבִּים לָאו רְשׁוּתוֹ הוּא, שֶׁהֲרֵי אֵין לוֹ רְשׁוּת לְהָבִיא שָׁם כֵּלָיו. וְאַף דְּבִכְלַל מְשִׁיכָה מְסִירָה, שֶׁהֲרֵי אֵין צָרִיךְ מִיָּד לְיָד (וּמְשִׁיכָה צְרִיכָה נַמִּי לִהְיוֹת בְּמִצְוַת הַמּוֹכֵר, אִי נַמִּי בְפָנָיו, וְכֵן דִּין הַמְּסִירָה). וְכֵיוָן שֶׁהַמְּסִירָה קוֹנָה בִרְשׁוּת הָרַבִּים אֵיךְ אֶפְשָׁר שֶׁהַמְּשִׁיכָה לֹא תִקְנֶה. לָאו קֻשְׁיָא הִיא, דְּכֵיוָן שֶׁמָּשַׁךְ גַּלֵּי דַעְתֵּיהּ דְּלֹא נִיחָא לֵיהּ דְּלִקְנֵי בִמְסִירָה אֶלָּא בִמְשִׁיכָה, וּמְשִׁיכָה בִּרְשׁוּת הָרַבִּים לָאו מִידֵי הִיא. וְהָא דִמְסִירָה לֹא קָנֵי בְּסִמְטָא וּבְחָצֵר שֶׁל שְׁנֵיהֶם, מִשּׁוּם דְּאֶפְשָׁר בִּמְשִׁיכָה, אִי נַמִּי כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לז} הַיְנוּ כְּשֶׁהִכְנִיסָהּ שֶׁלֹּא בִּרְשׁוּת בַּעַל הֶחָצֵר. דְּאִם לֹא כֵן הֲוֵי חָצֵר שֶׁלּוֹ אוֹ שֶׁל שְׁנֵיהֶם. הַמַּגִּיד: {לח} מְסַיֵּם הָרַמְבַּ"ם וַאֲפִלּוּ שֶׁאָנוּ אוֹמְרִים בּוֹ דְּנִקְנֶה בִמְשִׁיכָה. וּמִכָּל מָקוֹם יָצָא מִן הַכְּלָל בְּהֵמָה דַקָּה, מִשּׁוּם דְּמִסַּרְכָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לט} לֹא יָדַעְתִּי טַעַם הַדָּבָר, דְּוַדַּאי דִּמְשִׁיכָה דַעֲדִיפָא מִמְּסִירָה כְּמוֹ שֶׁכָּתַבְתִּי מַהֲנֵי בְּדָבָר הַנִּקְנֶה בִמְסִירָה דִּגְרִיעָה מִנָּהּ:

{מ} וּכְלוֹמַר שֶׁחוֹזֵר לְאָחוֹר. וּלְשׁוֹן הֶעָרוּךְ וַאַחֲרִיתֵךְ תֵּאָכֵל בָּאֵשׁ (יְחֶזְקֵאל כג), תַּרְגּוּם וּשְׁפַר אַרְעֵךְ תִּתּוֹקַד בְּנוּרָא: {מא} <B> בְּכֶסֶף.</B> בְּשָׁוֶה פְרוּטָה. דִּבְפָחוֹת מִשָּׁוֶה פְרוּטָה לֹא מִקְּרֵי כֶסֶף. בֵּית יוֹסֵף: {מב} וְאֵימָא עַד דְּאִיכָּא שְׁטָר, דִּכְתִיב וְכָתוֹב בַּסֵּפֶר וְחָתוֹם. אִי כְתִיב יִקְנוּ לְבַסּוֹף, כִּדְקָאָמַרְתְּ, הַשְׁתָּא לִקְנוֹת עַל הַשְּׁטָר הוּא. רַשִׁ"י. וּבְכָל שְׁטָר עָסְקִינַן, וַאֲפִלּוּ בִשְׁטַר הוֹדָאָה (כְּלוֹמַר שְׁטַר רְאָיָה. בֵּית דָּוִד). וּמִיהוּ כְּשֶׁרְגִילִין בִּשְׁטַר קִנְיָן תּוֹלֶה קְנִיָּתוֹ בַּשְּׁטָר וְאֵינוֹ קוֹנֶה מִשְּׁעַת נְתִינַת הַכֶּסֶף. וּכְשֶׁרְגִילִין בִּשְׁטַר רְאָיָה, כְּשֶׁנִּכְתַּב הַשְּׁטָר קָנָה לְמַפְרֵעַ מִשְּׁעַת נְתִינַת הַכֶּסֶף. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מד} אוֹ אוֹ קָאָמַר. וּמַשְׁכַּחַת לֵיהּ דְּקַנְיֵהּ, בְּמוֹכֵר שָׂדֵהוּ מִפְּנֵי רָעָתָהּ, כְּדִמְפָרֵשׁ וְאָזֵיל: {מה} דְּלֹא גָמַר מוֹכֵר וּמַקְנֶה עַד שֶׁמְּקַבֵּל דָּמִים, וְאַף עַל גַּב דְּקַבֵּיל דָּמִים בָּעֵי שְׁטָרָא כְדִלְעֵיל. רַשִׁ"י: {מו} דְּנִיחָא לֵיהּ דְּלִקְּנֵי לֵיהּ שְׁטָרָא לַלּוֹקֵחַ כִּי הֵיכִי דְלָא מָצֵי לְמִהֲדַר בֵּיהּ. רַשִׁ"י: {מז} וְכֵן כָּתַב הָרַמְבַּ"ם. דְנִקְנִין בִּמְסִירָה וְהַגְבָּהָה. וְנָקַט מְשִׁיכָה מִשּׁוּם דְּרֹב קְנִיַּת מִטַּלְטְלִין בִּמְשִׁיכָה כוּ'. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {נ} וְכֵן כָּתַב הָרַמְבַּ"ם. וְתֵימַהּ אָמְרִינַן וּלְרַבִּי יוֹחָנָן דְּאָמַר דְּבַר תּוֹרָה מָעוֹת קוֹנוֹת מַאי אִיכָּא לְמֵימַר, תַּנָּא תַּקַּנְתָּא דְרַבָּנָן קָתָנֵי: {נא} <B> עִם.</B> מַסְּקִינַן בַּגְּמָרָא דְלָא בָעִינַן צְבוּרִין {נג} וְאַף עַל גַּב דְּסוֹטָה אִסּוּרָא, יָלְפִינַן מִנָּהּ לְמָמוֹנָא, וּבְקַל וָחֹמֶר, וּמַה סּוֹטָה שֶׁלֹּא נִתְּנָה לְהִתָּבַע בְּעֵד אֶחָד מְגַלְגְּלִין, מָמוֹן שֶׁנִּתָּן לְהִתָּבַע בְּעֵד אֶחָד (לְהַשְׁבִּיעוֹ) אֵינוֹ דִּין שֶׁמְּגַלְגְּלִין. גְּמָרָא: {נד} כְּדִכְתִיב שָׁלַף אִישׁ נַעֲלוֹ [רוּת ד]: {נה} שֶׁהַמֶּלֶךְ פּוֹסְלָהּ לִפְעָמִים וְגוֹזֵר לָצוּר צוּרָה אַחֶרֶת, הִלְכָּךְ הֲוֵי לֵיהּ כְּדָבָר שֶׁאֵינוֹ מְסֻיָּם וְשָׁלֵם, דִּמְמַעֲטִינַן דָּבָר שֶׁאֵינוֹ מְסֻיָּם מִדִּכְתִיב נַעַל. רַשִׁ"י. וְטַעַם דִּבְפֵרוֹת לֹא, מִדִּכְתִיב נַעֲלוֹ, נַעַל אִין מִידֵי אַחֲרִינָא לֹא. גְּמָרָא. וְקָשֶׁה דִּלְמַאן דְּאָמַר אֵין קוֹנִין בְּפֵרוֹת לֹא קָיְמָא הַךְ אוֹקִימְתָא דְכָל הַנִּשּׁוֹם דְּהַיְנוּ כָּל מִטַּלְטְלִין שֶׁכָּתַב הָרַ"ב בָּרֵישָׁא. וּמוֹקְמִינַן שֶׁנִּתְחַיֵּב לוֹ דְּמֵי שׁוֹר וְאַחַר כָּךְ קוֹנֶה אֶצְלוֹ פָרָה בְאוֹתָן הַדָּמִים, וּמִשּׁוּם דְּלֹא שְׁכִיחַ לֹא תִקְּנוּ בוֹ רַבָּנָן שֶׁלֹּא יִהְיוּ מָעוֹת קוֹנוֹת כוּ'. וְהָרַמְבַּ"ם מְפָרֵשׁ כָּל הַנִּשּׁוֹם כוּ' כְּגוֹן שֶׁיֹּאמַר אֵלּוּ בְהֵמוֹת אוֹ זוֹ הַשָּׂדֶה שָׁוִים כָּךְ וְכָךְ לִטְרִין פִּלְפְּלִין, לְפִיכָךְ כְּשֶׁהֶחֱלִיף הַפִּלְפְּלִים עִם הַבְּהֵמָה כוּ' (כֵּיוָן שֶׁמָּשַׁךְ זֶה נִתְחַיֵּב כוּ') וְלֹא יִתְקַיֵּם הַקִּנְיָן, לֹא בְּפֵרוֹת וְלֹא בְמַטְבֵּעַ. וְלֹא קָשֶׁה מִידֵי, לְפִי דְלִדְבָרָיו הָרִאשׁוֹנִים אֵין הַקִּנְיָן כְּמוֹ קִנְיַן סוּדָר שֶׁבּוֹ אָמְרוּ אֵין קוֹנִין בְּפֵרוֹת דְּאִמַּעֲטוּ מִנַּעַל, אֲבָל דְּבָרָיו הָרִאשׁוֹנִים בְּתוֹרַת שִׁוּוּי דָּמִים הֵן וּסְבִירָא לֵיהּ דְּבִכְהַאי גַוְנָא כֻּלֵּי עָלְמָא סְבִירָא לְהוּ דְּקוֹנִין זֶה אֶת זֶה. וּמַה שֶּׁכָּתַב הָרַ"ב הוֹאִיל וְלֹא שָׁקַל כוּ', כֵּן כָּתַב הָרַמְבַּ"ם. וְקָשֶׁה, דְּהָא אֶפְשָׁר דְּאַף עַל פִּי כֵן דַּעְתּוֹ אַצּוּרָה. וּבַגְּמָרָא אָמְרוּ מִשּׁוּם דְּלֹא שְׁכִיחַ כוּ'. וְלִי נִרְאֶה שֶׁלָּשׁוֹן הָרַ"ב מְסֹרָס, שֶׁמַּה שֶּׁכָּתַב וְכָל וְכוּ' עַד בַּחֲלִיפִין, צָרִיךְ לוֹמַר אַחַר וְלֹא מָנָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {נו} <B> בְּכֶסֶף.</B> בַּגְּמָרָא יָלֵיף לֵיהּ: {נז} וַחֲזָקָה לָאו דַּוְקָא. וְהוּא הַדִּין מְשִׁיכָה לָאו דַּוְקָא: {נח} מַשְׁמַע דִּבְמִקָּח וּמִמְכָּר לֹא מַשְׁכַּחַת לַהּ. וּמִשְׁנָה ב' פֶּרֶק ח' דַּעֲרָכִין, מִדִּבְרֵיהֶם הִיא. הַמַּגִּיד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{נט} וְעִקֵּם הַתַּנָּא לְשׁוֹנוֹ לוֹמַר דְּמִצְוֹת הַבֵּן הֵם, שֶׁאִם לֹא עֲשָׂאָן לוֹ אָבִיו חַיָּב הוּא בְעַצְמוֹ לַעֲשׂוֹתָן כִּדְאִיתָא בַגְּמָרָא. וּבַסֵּיפָא אַגַּב רֵישָׁא נָקַט הַאי לִישָׁנָא: {ס} כֻּלְּהוּ יָלְפִינַן לְהוּ בַגְּמָרָא מִקְּרָאֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {סא} חַיּוּתֵיהּ הוּא, שֶׁמָּא יִפְרֹשׁ בַּסְּפִינָה וְתִטְבַּע וְיִסְתַּכֵּן אִם אֵינוֹ יוֹדֵעַ לָשׁוּט: {סב} כְּלָלִים הֵם וּלְדֻגְמָא נַקְטִינְהוּ, כְּמוֹ שֶׁכָּתַב הָרַמְבַּ"ם: {סג} אִם הָיָה חוֹלֵק עִם אַחֵר בִּדְבַר הֲלָכָה לֹא יֹאמַר נִרְאִים לִי דִבְרֵי פְלוֹנִי. רַשִׁ"י. וְהָרָמָ"ה כָּתַב, זֶה אֵין צָרִיךְ, דְּהַיְנוּ סוֹתֵר אֶת דְּבָרָיו, אֶלָּא אֲפִלּוּ נִרְאִין לוֹ דִּבְרֵי אָבִיו אֵין לוֹ לוֹמַר נִרְאִין דִּבְרֵי אַבָּא, שֶׁנִּרְאֶה כְמַכְרִיעַ דִּבְרֵי אָבִיו, אֶלָּא אִם יֵשׁ לוֹ תְשׁוּבָה לְהָשִׁיב עַל הַחוֹלְקִים יָשִׁיב. טוּר: {סד} <B> חַיָּבִין.</B> דִּכְתִיב תִּירָאוּ. וּמַה תַּלְמוּד לוֹמַר אִישׁ, שֶׁהוּא סִפֵּק בְּיָדוֹ לַעֲשׂוֹת (שֶׁאֵין מוֹחֶה בְיָדוֹ). אִשָּׁה רְשׁוּת אֲחֵרִים (בַּעְלָהּ. רַשִׁ"י) עָלֶיהָ. גְּמָרָא: {סה} וְאַף דְּבַיְרוּשַׁלְמִי אָמְרִינַן כְּסוּת יוֹם שֶׁלְּבָשָׁהּ בַּלַּיְלָה חַיֶּבֶת בְּצִיצִית, אֲפִלּוּ הָכִי כֵּיוָן שֶׁחִיּוּבוֹ תָלוּי בְּמַה שֶּׁהַזְּמַן גָּרַם לְבִישָׁתוֹ בַיּוֹם, יֵשׁ לָהּ לֵיחָשֵׁב זְמַן גְּרָמָהּ, שֶׁזְּמַן לְבִישָׁתָהּ גּוֹרֵם הַחִיּוּב. תּוֹסָפוֹת: {סו} <B> פְּטוּרוֹת.</B> מַה תְּפִלִּין נָשִׁים פְּטוּרוֹת, אַף כָּל מִצְוַת עֲשֵׂה שֶׁהַזְּמַן גְּרָמָהּ נָשִׁים פְּטוּרוֹת. גְּמָרָא: {סז} אַף עַל גַּב דְּאִית בֵּיהּ לָאו, וְלֹא תָשִׂים דָּמִים, וְאִם כֵּן תֵּפּוּק לֵיהּ דְּנָשִׁים חַיָּבוֹת מִשּׁוּם הַלָּאו, מַסִּיק הָרַ"ן, דְּלָאו וַעֲשֵׂה אֵינָן בָּאִין כְּאֶחָד, דְּלֹא תָשִׂים דָּמִים, הַהוּא כְּשֶׁבָּנָה עַל מְנָת שֶׁלֹּא לַעֲשׂוֹת, וְאַף עַל פִּי כֵן מְחֻיָּב אַחַר כָּךְ לַעֲשׂוֹת מַעֲקֶה מִשּׁוּם הָעֲשֵׂה: {סח} נַמִּי לָאו דְּלֹא תוּכַל לְהִתְעַלֵּם כְּשֶׁנְּטָלָהּ עַל מְנָת שֶׁלֹּא לְהַחְזִירָהּ עָבַר הַלָּאו, וּמִכָּל מָקוֹם מְחֻיָּב לְקַיֵּם הָעֲשֵׂה דְהָשֵׁב תְּשִׁיבֵם: {סט} אִם נְטָלָהּ עַל מְנָת שֶׁלֹּא לְשַׁלְּחָהּ עוֹבֵר מִשּׁוּם לֹא תִקַּח הָאֵם, וְאַחַר כָּךְ חַיָּב בַּעֲשֵׂה. דְּשַׁלֵּחַ לְרַבָּנָן לָאו מֵעִקָּרָא מַשְׁמַע. הָרַ"ן: {ע} וּמַה שֶּׁאָמַר כָּל, אָמְנָם רְצוֹנוֹ לוֹמַר הָרֹב. הָרַמְבַּ"ם: {עא} דִּזְמַן גָּרַם וְנָשִׁים חַיָּבוֹת, כִּדְאִיתָא בִּפְסָחִים דַּף מ"ג. וְשִׂמְחָה בַמּוֹעֲדִים דִּכְתִיב וְשָׂמַחְתָּ בְּחַגֶּךָ אַתָּה וּבִנְךָ וּבִתֶּךָ [דְּבָרִים טז]. רַשִׁ"י: {עב} כְּדִכְתִיב הַקְהֵל אֶת הָעָם הָאֲנָשִׁים וְהַנָּשִׁים [שָׁם לא]: {עג} כְּתַנָּא קַמָּא דְּסוֹף פֶּרֶק ו' דִּיבָמוֹת, דְּהִלְכְתָא כְּוָתֵיהּ: {עד} בַּגְּמָרָא יָלְפִינַן לַהּ: {עה} וְלָאוִין עָנְשֵׁי מַלְקוֹת הֵן. וּבַגְּמָרָא, אִי מֵהָכָא אֵימָא דַוְקָא מִשּׁוּם כַּפָּרָה חָס רַחֲמָנָא עֲלָהּ. פֵּרְשׁוּ הַתּוֹסָפוֹת כְּגוֹן עֲרָיוֹת דִּכְתִיב בְּנָשִׁים עֳנָשִׁין כָּתַב רַחֲמָנָא אֲשֶׁר תָּשִׂים לִפְנֵיהֶם, הִשְׁוָה כוּ' לְכָל דִּינִים. וְאִי מֵהָא הֲוָה אֲמִינָא מִשּׁוּם דְּחַיּוּתָהּ הִיא, דְּאִי אֵין עָלֶיהָ דִין הַכֹּל גּוֹזְלִים אוֹתָהּ וְהִיא גוֹזֶלֶת אֲחֵרִים וְהַכֹּל בְּדֵלִין הֵימֶנָּה, כָּתַב רַחֲמָנָא וְהֵמִית אִישׁ אוֹ אִשָּׁה בְּשׁוֹר הַמּוּעָד שֶׁהָרַג, הִשְׁוָה כוּ'. וְאִי מֵהָא מִשּׁוּם דְּאִיכָּא אִבּוּד נְשָׁמָה שֶׁהַלָּה עֲשָׂאוֹ שָׁלִיחַ לְקַדְּשָׁהּ לֹא הָיְתָה מִתְקַדֶּשֶׁת לוֹ, קָא מַשְׁמַע לָן. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} <B> לַשֵּׁנִי.</B> בַּגְּמָרָא, אָמַר רַב לְעוֹלָם. וְזֶה לְשׁוֹן הָרַמְבַּ"ם, לְפִי שֶׁבְּשָׁעָה שֶׁקִּדְּשָׁהּ הַשֵּׁנִי לֹא הָיְתָה מְקֻדֶּשֶׁת, וְתָפְסוּ בָהּ קִדּוּשֵׁי שֵׁנִי וְנַעֲשֵׂית אֵשֶׁת אִישׁ, וּלְאַחַר הַשְּׁלֹשִׁים יוֹם כְּשֶׁיָּבֹאוּ קִדּוּשֵׁי {כב} וְאַף עַל פִּי שֶׁאֵין דָּבָר שֶׁבָּעֶרְוָה פָּחוֹת מִשְּׁנַיִם, וְעֵד אֶחָד דִּמְהֵימָן לוֹמַר מֵת בַּעְלִיךְ הַיְנוּ טַעֲמָא מִשּׁוּם דְּאִיהִי דָיְקָא וּמִנָּסְבָא וּמִתּוֹךְ חֹמֶר שֶׁהֶחְמַרְתָּ עָלֶיהָ בְּסוֹפָהּ, וְהָכָא לֵיתֵיהּ לְהַאי טַעֲמָא, הָתָם הוּא לְהוֹצִיאָהּ מֵחֶזְקָתָהּ, אֲבָל זֶה שֶׁמַּעֲמִידָהּ בְּחֶזְקָתָהּ אֶלָּא שֶׁאוֹמֵר שֶׁנִּתְקַדְּשָׁה לוֹ, נֶאֱמָן. הָרַ"ן. וְהַגָּאוֹן פְּנֵי יְהוֹשֻׁעַ בַּסּוּגְיָא, שְׁדָא נַרְגָּא בְּדִבְרֵי הָרַ"ן. וּבְסִפְרִי פִּתְחֵי נִדָּה פֶּתַח מ"א מ"ב הֶאֱרַכְתִּי בָּעִנְיָן. וּמָקוֹם הִנִּיחוּ לִי לְהִתְגַּדֵּר בְּעֶזְרַת ה' יִתְבָּרַךְ:

{כג} <B> נֶאֱמָן.</B> דַּוְקָא בְתוֹךְ כְּדֵי דִבּוּר, אֲבָל הֵיכָא דְהֻחְזְקָה אֵשֶׁת אִישׁ עַל פִּיו אֵינוֹ נֶאֱמָן לוֹמַר לְאַחַר זְמַן גֵּרַשְׁתִּיהָ, דְּלֹא הֵימְנֵיהּ רַחֲמָנָא בְּגֵרוּשִׁין. הָרַ"ן. דְּהָא קְרָא אֶת בִּתִּי נָתַתִּי הוּא דִכְתִיב: {כד} וּקְרָא בֵּין בִּקְטַנָּה בֵּין בְּנַעֲרָה אֶלָּא שֶׁאֵינָהּ בּוֹגֶרֶת מַיְרֵי, וּלְפִיכָךְ צָרִיךְ לְפָרֵשׁ דִּקְטַנָּה דִתְנַן הַיְנוּ לַאֲפוֹקֵי בוֹגֶרֶת: {כה} וְהוּא הַדִּין בְּקִדּוּשֵׁי בִיאַת אִסּוּר לֹא הֵימְנֵיהּ רַחֲמָנָא, וּבְהָכִי נִיחָא, דְּלֵיכָּא לְמֵימַר מִגּוֹ דְקִדַּשְׁתִּיהָ לְפָסוּל לָהּ. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כו} <B> מִיתָתוֹ.</B> אוֹרְחָא דְמִלְּתָא נָקַט. רַשְׁבָּ"ם: {כז} וַאֲפִלּוּ כִּי מֻחְזָק בְּאַחֵי נֶאֱמָן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כח} <B> אֵין הַבּוֹגְרוֹת כוּ'.</B> הָא קְטַנּוֹת בִּכְלָל. דְּקִדּוּשִׁין שֶׁאֵין מְסוּרִין לְבִיאָה, דְּכָל אַחַת סָפֵק אֲחוֹת אִשָּׁה, אֲפִלּוּ הָכִי הֲווּ בִכְלַל סָפֵק קִדּוּשִׁין וּצְרִיכוֹת גֵּט: {כט} דְּאִי לָאו הָכִי לְמַאי תְּנַן אֵין הַבּוֹגְרוֹת בִּכְלָל, מִלְּתָא דִפְשִׁיטָא הִיא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ל} דְּמַתְנִיתִין סְתָמָא קָתָנֵי, וַאֲפִלּוּ אָמְרָה לוֹ קִדּוּשַׁי לְךָ, דְּאִית לֵיהּ נַמִּי הֲנָאָה מִנָּהּ. גְּמָרָא. וְהָא דְקָרֵי לֵיהּ מִצְוָה דְרַמְיָא עֲלֵיהּ, דִּכְתִיב [יִרְמְיָה כט] וְאֶת בְּנוֹתֵיכֶם תְּנוּ לַאֲנָשִׁים. רַשִׁ"י:

{לא} <B> מִשְּׁתֵּי נָשִׁים.</B> שֶׁנָּשָׂא הַשְּׁנִיָּה לְאַחַר מִיתַת הָרִאשׁוֹנָה. גְּדוֹלוֹת כַּת הָרִאשׁוֹנָה. קְטַנּוֹת כַּת שְׁנִיָּה. רַשִׁ"י. וְכֻלָּן בִּרְשׁוּתוֹ. הָרַמְבַּ"ם. וּכְלוֹמַר שֶׁכֻּלָּן קְטַנּוֹת אוֹ נַעֲרוֹת, אוֹ שֶׁכֻּלָּן בּוֹגְרוֹת וְנָתְנוּ לוֹ רְשׁוּת לְקַדְּשָׁן, דְּאִלּוּ בּוֹגְרוֹת בִּמְקוֹם קְטַנּוֹת אוֹ נַעֲרוֹת, מְבֹאָר לְעֵיל דַּאֲפִלּוּ שַׁוִּיתֵיהּ שָׁלִיחַ אֵינָם בִּכְלָל. בֵּית יוֹסֵף: {לב} <B> קִדַּשְׁתִּי.</B> נִרְאֶה בְעֵינַי דְּדוֹמֶה לְאוֹתָהּ שֶׁשָּׁנִינוּ בְּמִשְׁנָה ז' קִדַּשְׁתִּי וְאֵינִי יוֹדֵעַ לְמִי כוּ'. וְהָכָא נַמִּי שֶׁאוֹמֵר קִדַּשְׁתִּי כוּ' וְאֵינִי יוֹדֵעַ אֵיזוֹהִי. וְאַף עַל גַּב שֶׁהַמְקַדֵּשׁ יֹאמַר שֶׁזּוֹ הִיא שֶׁקִּדֵּשׁ, אֵינוֹ נֶאֱמָן, דְּהָתָם הַיְנוּ טַעֲמָא דְנֶאֱמָן כְּמוֹ שֶׁכָּתַבְתִּי בְּשֵׁם הָרַ"ן, דְּחֶזְקָתָהּ מְקֻדֶּשֶׁת וְאֵינוֹ מוֹצִיאָהּ מֵחֶזְקָתָהּ. אֲבָל הָכָא דִּלְמָא לֹא זוֹ הִיא שֶׁקִּדֵּשׁ וְנִמְצָא שֶׁאוֹתָהּ שֶׁנִּתְקַדְּשָׁה בֶאֱמֶת מוֹצִיאָהּ מִן הַסָּפֵק שֶׁיֵּשׁ לָנוּ בָּהּ. וְאֵין דָּבָר שֶׁבָּעֶרְוָה פָּחוֹת מִשְּׁנַיִם, מִלְּבַד הָאָב שֶׁהֶאֱמִינָתוֹ הַתּוֹרָה שֶׁנֶּאֱמַר אֶת בִּתִּי נָתַתִּי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לג} <B> כֻּלָּן.</B> מְפָרֵשׁ בַּגְּמָרָא דִּבְשֶׁאֵין בְּכַת שְׁנִיָּה אֶלָּא גְדוֹלָה וּקְטַנָּה עָסְקִינַן. שֶׁאִלּוּ הָיָה אֶמְצָעִית בְּכַת שְׁנִיָּה, אֲפִלּוּ לְרַבִּי מֵאִיר הָיְתָה מֻתֶּרֶת, מִשּׁוּם דְּלֹא מִקַּרְיָא גְדוֹלָה אֶלָּא אִם כֵּן הִיא גְדוֹלָה לְגַבֵּי כָּל כַּת אַחַת. וְהַיְנוּ דְּלֹא קָתָנֵי וְאֵינוֹ יוֹדֵעַ אִם אֶמְצָעִית שֶׁבַּקְּטַנּוֹת. אֲבָל אֶמְצָעִית שֶׁבַּגְּדוֹלוֹת נִכְלֶלֶת בַּקְּטַנָּה שֶׁבַּגְּדוֹלוֹת. הָרַ"ן: {לד} <B> דִּבְרֵי רַבִּי מֵאִיר, רַבִּי יוֹסֵי אוֹמֵר.</B> בְּהָא קָמִפַּלְּגֵי, רַבִּי מֵאִיר סָבַר מַחֵית אֱנָשׁ נַפְשֵׁיהּ לִסְפֵקָא, וְרַבִּי יוֹסֵי סָבַר לֹא מַחֵית אֱנָשׁ נַפְשֵׁיהּ לְמֵימַר מִלְּתָא דְּלֵיתֵי בָּהּ לִידֵי שְׁאֵלָה לַחֲכָמִים. וְכִי אָמַר גְּדוֹלָה, גְּדוֹלָה מַמָּשׁ שֶׁבְּכֻלָּן קָאָמַר. גְּמָרָא:

{לז} וְהַתּוֹסָפוֹת הִקְשׁוּ פְּשִׁיטָא כֵּיוָן דְּנִבְדֶּקֶת כְּבָר שׁוּב אֵין צָרִיךְ רְאָיָה. וְעוֹד דְּאֵינוֹ מֵבִיא רְאָיָה דְּסֵיפָא מְיֻתָּר כוּ'. וְלִי נִרְאֶה דְיֵשׁ לוֹמַר דְּאִין הָכִי נַמִּי דְּכֻלָּהּ מַתְנִיתִין לֹא אִיצְטְרִיךְ אֶלָּא מִשּׁוּם רְאָיָה דְעַל הַבָּנִים, דְּבָרֵישָׁא אֵין צָרִיךְ כֵּיוָן דִּכְרוּכִים, וּבַסֵּיפָא כֵּיוָן דְּמֵתָה צָרִיךְ, וְאַגַּב נָסֵיב נַמִּי בֵּין בָּרֵישָׁא בֵּין בַּסֵּיפָא דְאֵין צָרִיךְ רְאָיָה עַל הָאִשָּׁה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לח} נִדְבָּקִים אֶצְלָהּ.רַשִׁ"י: {לט} <B> עַל הַבָּנִים.</B> צָרִיךְ לוֹמַר דְּמַתְנִיתִין אֲפִלּוּ לְרַבִּי יְהוּדָה דְמִשְׁנָה ח' שֶׁסּוֹבֵר דְּדִין תּוֹרָה הָאָב נֶאֱמָן לְפָסְלוֹ וְכָל שֶׁכֵּן לְהַכְשִׁירוֹ וּכְמוֹ שֶׁכָּתוּב לְעֵיל, אֲפִלּוּ הָכִי מִשּׁוּם מַעֲלָה דְיוֹחֲסִין אָמְרוּ חֲכָמִים שֶׁלֹּא לְהַאֲמִינוֹ. וְכָל זֶה הוּא רַק נֶגֶד מִשְׁפָּחוֹת הַבְּדוּקוֹת שֶׁרוֹצִים שֶׁתִּהְיֶה מִשְׁפַּחְתָּן בְּדוּקָה, אֲבָל בְּלָאו הָכִי אֵין צָרִיךְ לְהָבִיא רְאָיָה, דְּכָל אָדָם בְּחֶזְקַת כַּשְׁרוּת עַד שֶׁיִּפָּסֵל (עַל יְדֵי עִרְעוּר). הַמַּגִּיד:

{מ} <B> הַבָּנִים.</B> הַקְּטַנִּים הַכְּרוּכִים אַחֲרֶיהָ. גְּמָרָא:

{מא} <B> לֹא יִתְיַחֵד.</B> רֶמֶז לְיִחוּד מִן הַתּוֹרָה מִנַּיִן, שֶׁנֶּאֱמַר כִּי יְסִיתְךָ אָחִיךָ בֶן אִמֶּךָ [דְּבָרִים יג], וְכִי בֶן אֵם מֵסִית בֶּן אָב אֵינוֹ מֵסִית, אֶלָּא לוֹמַר לְךָ, בֵּן מִתְיַחֵד עִם אִמּוֹ (וּלְהָכִי נָקַט בֶּן אִמֶּךָ, שֶׁהַבֵּן מָצוּי אֵצֶל הָאֵם), וְאָסוּר לְהִתְיַחֵד עִם כָּל עֲרָיוֹת שֶׁבַּתּוֹרָה. גְּמָרָא. וּלְפִי שֶׁלֹּא אָמְרוּ אֶלָּא רֶמֶז, כָּתַב הָרַמְבַּ"ם, וְאִסּוּר יִחוּד הָעֲרָיוֹת מִפִּי הַקַּבָּלָה. וְכָתַב עוֹד, דָּוִד וּבֵית דִּינוֹ גָּזְרוּ עַל יִחוּד הַפְּנוּיָה, שַׁמַּאי וְהִלֵּל גָּזְרוּ עַל יִחוּד כּוּתִית. עַד כָּאן. וּמִיהוּ כָּל זֶה בְּיוֹתֵר מִשָּׁלֹשׁ שָׁנִים וְיוֹם אֶחָד. יְרוּשַׁלְמִי. וְכֵן פָּסַק הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מב} וְרַשִׁ"י וְהָרַמְבַּ"ם כָּתְבוּ נַמִּי בַּת חֲמוֹתָהּ: {מג} כְּלוֹמַר, מַה הִיא בִּיאָה, שֶׁתֵּדַע לְסַפֵּר דְּבָרִים בַּשּׁוּק. וְאֵינָהּ מוֹסֶרֶת כוּ' מִתּוֹךְ קָטְנָהּ עֲדַיִן לֹא לְבָשָׁהּ יֵצֶר, הִלְכָּךְ אֵין כָּאן מִשּׁוּם דַּעְתָּהּ קַלָּה, שֶׁלֹּא תְהֵא נוֹחָה לְהִתְפַּתּוֹת. רַשִׁ"י: {מד} שְׁנֵיהֶם בִּכְלָל. הָרַמְבַּ"ם: {מה} <B> שֶׁאִשְׁתּוֹ.</B> הִקְשׁוּ בַּתּוֹסָפוֹת דִּבְרֵישׁ פֶּרֶק ב' דַּעֲבוֹדָה זָרָה מַשְׁמַע שֶׁאִשָּׁה אַחַת מִתְיַחֶדֶת עִם אִישׁ שֶׁאִשְׁתּוֹ עִמּוֹ, וְהָכָא מַשְׁמַע דְּדַוְקָא שְׁתֵּי נָשִׁים, וּמִכָּל שֶׁכֵּן לְרַבָּנָן, אִם כֵּן מַתְנִיתִין דְּהָתָם דְּלֹא כְמַאן כוּ'. לְכָךְ כָּתְבוּ דְּהָכִי גָרְסִינַן, רַבִּי שִׁמְעוֹן אוֹמֵר אַף אִישׁ אֶחָד מִתְיַחֵד עִם שְׁתֵּי נָשִׁים. וּבִזְמַן שֶׁאִשְׁתּוֹ עִמּוֹ יָשֵׁן עִמָּהֶן בְּפוּנְדְּקִי, מִפְּנֵי שֶׁאִשְׁתּוֹ מְשַׁמַּרְתּוֹ. וַאֲתָאן אֲפִלּוּ לְרַבָּנָן. וְכֵן גִּרְסַת הָרִי"ף:

{מו} בְּיָדוּעַ שֶׁיֵּשׁ בָּהּ טַעַם בִּיאָה. רַשִׁ"י. כְּלוֹמַר, וְיֵשׁ לָחוּשׁ שֶׁמּוֹסֶרֶת עַצְמָהּ לְבִיאָה. וְאֵין זֶה כְּטַעַם בִּיאָה דִלְעֵיל, דְּהָתָם אֵינָה מוֹסֶרֶת. וּמִשׁוּם הָכִי לְעֵיל אָמְרִינַן שֶׁיּוֹדַעַת, וְהָכָא שֶׁיֵּשׁ בָּהּ: {מז} <B> רַוָּק.</B> תַּרְגּוּם שֶׁל וּבַחוּרִים כָּשׁוֹל יִכָּשֵׁלוּ [יְשְׁעְיָה מ] הוּא רַוָּקִים. תּוֹסָפוֹת וְהָרַמְבַּ"ם: {מח} דְּאִלּוּ תִּינוֹקוֹת עַצְמָן לֹא נֶחְשְׁדוּ עַל הַזָּכוּר. גְּמָרָא: {מט} <B> אִשָּׁה.</B> מִפְּנֵי אֲבוֹתֵיהֶם שֶׁבָּאִים לְהָבִיא אֶת בְּנֵיהֶם. גְּמָרָא: {נ} אֶצְלוֹ בְּבֵית הַסֵּפֶר. וְטַעְמָא דְתַנָּא קַמָּא, דְּאַף עַל פִּי שֶׁהִיא בְּבֵיתָהּ מִקְּרֵי פַת בְּסַלּוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{נא} וְעַל הַבְּהֵמָה. רַשִׁ"י וְרַ"ן: {נב} כְּלוֹמַר דַּהֲוָה אֲמִינָא דְלִשְׁרֵי לְהִתְיַחֵד עִם שְׁתֵּי נָשִׁים מִשּׁוּם כְּדֵי חַיָּיו. וְכָתַב הָרַמְבַּ"ם וְכֵיצַד יַעֲשֶׂה, יִתְעַסֵּק עִמָּהֶן וְאִשְׁתּוֹ עִמּוֹ, אוֹ יִפְנֶה לִמְלָאכָה אַחֶרֶת: {נג} <B> נְקִיָּה כוּ'.</B> בַּגְּמָרָא, מַחֲטָא דְתַלְמִיּוּתָא, כְּלוֹמַר תּוֹפֵר בְּגָדִים קְרוּעִים. וְהַיְנוּ תַלְמִיּוּתָא שֶׁהַתְּפִירוֹת עֲשׂוּיוֹת שׁוּרוֹת שׁוּרוֹת כִּתְלָמִים שֶׁל מַחֲרֵשָׁה. רַשִׁ"י. וְהֶעָרוּךְ מְסַיֵּם נְקִיָּה מִגָּזֵל וְקַלָּה דְּאֵין בָּהּ חֶסְרוֹן כִּיס: {נד} <B> לְמִי כוּ'.</B> לְשׁוֹן רַשִׁ"י, וְאַל יֹאמַר בְּלִבּוֹ אֻמָּנוּת זוֹ אֵינָהּ מְעַשֶּׁרֶת, אֶלָּא יְבַקֵּשׁ רַחֲמִים לְמִי שֶׁהָעֹשֶׁר שֶׁלּוֹ: {נה} <B> שֶׁלּוֹ.</B> שֶׁנֶּאֱמַר לִי הַכֶּסֶף וְלִי הַזָּהָב נְאוּם ה' צְבָאוֹת [חַגַּי ב]. גְּמָרָא: {נו} <B> לְפִי זְכוּתוֹ.</B> וְקַשְׁיָא לִי דְּאִם הַכֹּל לְפִי זְכוּתוֹ, אִם כֵּן תְּפִלָּה מַאי מַהֲנְיָא לֵיהּ. שֶׁאִם זְכוּתוֹ כְּדַאי אֵין הַקָּדוֹשׁ בָּרוּךְ הוּא מְקַפֵּחַ שְׂכָרוֹ, וְאִם שֶׁנֶּאֱמַר עֲוֹנוֹתֵיכֶם הִטּוּ [יִרְמְיָה ה]. גְּמָרָא: {נט} <B> סַפָּר.</B> בַּגְּמָרָא וּבַיְרוּשַׁלְמִי גָּרַס קַדָּר. וּפֵרֵשׁ רַשִׁ"י בַּעֲלֵי קְרָנוֹת: <B>רְשָׁעִים.</B> מִפְּנֵי לִסְטִיּוּת: וְתָמִיד הֵם בִּרְעָדָה יוֹתֵר מִן הַגַּמָּלִים: {ס} <B> אֵינָהּ כֵּן.</B> שֶׁאַף הַבָּא לִידֵי זִקְנָה אוֹ יִסּוּרִין וְאֵינוֹ יָכוֹל לַעֲסֹק בָּהּ הוּא אוֹכֵל מִמַּתַּן שְׂכָרָהּ. דִּכְתִיב עוֹד
