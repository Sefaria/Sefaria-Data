~ פרק א - משנה א

{א} לָאו מֵהַמְּקַנֵּא דִּיֵּק, שֶׁיָּכוֹל לְפָרֵשׁ הַבָּא לְקַנְאוֹת, אֶלָּא כִּדְפֵרֵשׁ רַשִׁ"י, מִדְּלֹא תָּנֵי רַבִּי אֱלִיעֶזֶר אוֹמֵר מְקַנֵּא אָדָם לְאִשְׁתּוֹ. תּוֹסָפוֹת. וְלִישְׁנָא דִמְקַנֵּא לְשׁוֹן כַּעַס הוּא, שֶׁמַּכְעִיסָהּ וּמַקְנִיטָהּ. וּפֵרֵשׁ רַשִׁ"י כְּמוֹ הֵם קִנְאוּנִי [דְּבָרִים לב]. כִּי קִנְאָה חֲמַת גָּבֶר [מִשְׁלֵי ו]. וְלַהֲלָכָה שֶׁכָּתַב הָרַ"ב דְּחַיָּב לְקַנְאוֹת לִישְׁנָא דְּהַתְרָאָה כְּמוֹ וַיְקַנֵּא ה' לְאַרְצוֹ [יוֹאֵל ב]. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} <B> עַל פִּי כוּ'. </B> דִּסְתִירָה אִתַּקַּשׁ לְטֻמְאָה, דִּכְתִיב וְנִסְתְּרָה וְהִיא נִטְמָאָה. וּבְטֻמְאָה כֻּלֵּי עָלְמָא מוֹדוּ דְּעֵד אֶחָד נֶאֱמָן לְאַחַר קִנּוּי כוּ'. גְּמָרָא: {ג} הָכִי יָלֵיף לָהּ בַּגְּמָרָא דַּף כ"ח: {ד} פֵּרוּשׁ, אֲפִלּוּ עַל פִּי שְׁנַיִם כוּ'. אֲבָל בִּזְמַן שֶׁיֵּשׁ מֵי הַמָּרִים, אַף עַל פִּי שֶׁאִם תִּסָּתֵר עַל פִּי עַצְמוֹ אוֹ עַל פִּי עֵד אֶחָד תֵּאָסֵר עָלָיו עוֹלָמִית, הוֹאִיל וְאֶפְשָׁר שֶׁיָּכוֹל לִהְיוֹת שֶׁתִּסָּתֵר עַל פִּי שְׁנַיִם וְאָז יַשְׁקֶנָּה, לְפִיכָךְ רְשׁוּת אוֹ חוֹבָה לְקַנְאוֹת. מַה שֶּׁאֵין כֵּן בַּזְּמַן הַזֶּה שֶׁאִי אֶפְשָׁר שֶׁתָּשׁוּב לְהֵתֵּרָהּ, אָסְרוּ לְקַנְאוֹת כְּלָל:

{ה} וְאָמַר לָהּ לָאו פֵּרוּשָׁא דְכֵיצַד מְקַנֵּא לָהּ הוּא, אֶלָּא תַּרְתֵּי קָתָנֵי, וְהָכִי קָאָמַר, לֹא תְדַבְּרִי כוּ'. רַשִׁ"י: {ו} <B> לְבֵיתָהּ.</B> כְּלוֹמַר לְבַעְלָהּ, שֶׁלֹּא אֲסָרָתָהּ תּוֹרָה מִסָּפֵק אֶלָּא עַל יְדֵי קִנּוּי: {ז} בְּאֵשׁ בֵּינוֹנִית. הָרַ"מ: {ח} <B> בַּתְּרוּמָה.</B> כִּדְיָלְפִינַן לְקַמָּן שְׁלֹשָׁה וְנִטְמְאָה אֲמוּרִים בַּפָּרָשָׁה, אֶחָד לַבַּעַל וְאֶחָד לַבּוֹעֵל וְאֶחָד לַתְּרוּמָה. רַשִׁ"י: 

~ פרק א - משנה ג

{ט} <B> עֵדִים.</B> הוּא הַדִּין לְעֵד אֶחָד, שֶׁכָּל מָקוֹם שֶׁהֶאֱמִינָה תּוֹרָה לְעֵד אֶחָד הֲרֵי כָּאן שְׁנַיִם. תּוֹסָפוֹת. וְעַיֵּן מַה שֶּׁכָּתַבְנוּ מִשְׁנָה ב' פֶּרֶק ט"ז דִּיבָמוֹת: {י} אַף עַל גַּב דְּדָרְשִׁינַן לֵיהּ לְעֵד אֶחָד נֶאֱמָן בְּסוֹטָה, אֵין מִקְרָא יוֹצֵא מִיְדֵי פְשׁוּטוֹ וְדָרְשִׁינַן לֵיהּ נַמִּי הָכִי. רַשִׁ"י: {יא} וַהֲדַר אָמַר לְהוּ רַבִּי יְהוּדָה מִקְּרָא דִּכְתִיב וְהֵבִיא הָאִישׁ אֶת אִשְׁתּוֹ אֶל הַכֹּהֵן, מִן הַתּוֹרָה הָאִישׁ מֵבִיא אֶת אִשְׁתּוֹ. גְּמָרָא: 

~ פרק א - משנה ד

{יב} <B> כְּדֶרֶךְ.</B> כְּלוֹמַר כְּמוֹ שֶׁמְּאַיְּמִין יוֹתֵר עַל עֵדֵי נְפָשׁוֹת מֵעַל עֵדֵי מָמוֹנוֹת, כִּדְאִיתָא בְּסַנְהֶדְרִין, כְּמוֹ כֵן מְאַיְּמִין יוֹתֵר עַל הָאִשָּׁה. אֲבָל לֹא שֶׁאוֹתוֹ אִיּוּם עַצְמוֹ אוֹמְרִים בָּאִשָּׁה, שֶׁאֵין עִנְיָנוֹ לְעִנְיָנָהּ כְּלָל: {יג} כְּשֶׁאָמַר יְהוּדָה צָדְקָה מִמֶּנִּי, עָמַד רְאוּבֵן וְאָמַר בִּלְבַּלְתִּי יְצוּעֵי אָבִי. תַּנְחוּמָא: {יד} <B> וְכָל מִשְׁפַּחַת כוּ'.</B> שֶׁאִי אֶפְשָׁר שֶׁלֹּא יִהְיוּ נִמְצָאִין אַנְשֵׁי בֵּיתָהּ וּמִשְׁפַּחְתָּהּ בְּזִכְרוֹן אֵלּוּ הַדְּבָרִים. הָרַמְבַּ"ם: 

~ פרק א - משנה ה

{טו} <B> וְיוֹצֵאת.</B> וְאֵינָהּ נֶהֱרֶגֶת, שֶׁלֹּא הִתְרוּ בָּהּ עֵדִים בִּשְׁעַת מַעֲשֶׂה: {טז} <B> הַמִּזְרָח.</B> שַׁעַר הַחִיצוֹן שֶׁבּוֹ נִכְנָסִין לְהַר הַבַּיִת. לְשַׁעַר נִקָּנוֹר, וּמִשָּׁם לְשַׁעַר נִקָּנוֹר הוּא שַׁעַר הָעֶלְיוֹן שֶׁבֵּין עֶזְרַת יִשְׂרָאֵל לְעֶזְרַת נָשִׁים. רַשִׁ"י: {יז} דְּאִי לָאו הָכִי מַאי מַעֲלִין, הָא הָתָם קָיְמָא לִפְנֵי הַסַּנְהֶדְרִין שֶׁבְּלִשְׁכַּת הַגָּזִית שֶׁבָּעֲזָרָה. גְּמָרָא. וּרְצוֹנוֹ לוֹמַר מוֹרִידִין וּמַעֲלִין וּמוֹרִידִין: {יח} <B> שֶׁשָּׁם כוּ'.</B> וּבְמִשְׁנָה ג' פֶּרֶק ו' דִשְׁקָלִים פֵּרֵשׁ הָרַ"ב שַׁעַר הַנָּשִׁים לַעֲמֹד עַל קָרְבָּנָן, הַיְנוּ בְּנִדָּר וְנִדָּב אוֹ שְׁאָר חוֹבוֹת שֶׁאֵינָן מְחֻסְּרֵי כַפָּרָה: {יט} מִפְּנֵי שֶׁלֹּא נִתְקַדֵּשׁ בִּקְדֻשַּׁת עֲזָרָה עֹבִי חָלָל אוֹתוֹ שַׁעַר, מִפְּנֵי הַמְּצֹרָעִין שֶׁמַּכְנִיסִין יְדֵיהֶן לַבְּהוֹנוֹת לִתֵּן מִדַּם הָאָשָׁם עַל בֹּהֶן יָדָן, וּמְחֻסָּר כִּפּוּרִים שֶׁנִּכְנַס לָעֲזָרָה בְּכָרֵת, לְפִיכָךְ לֹא קִדְּשׁוּם, שֶׁיּוּכַל לַעֲמֹד בַּחֲלַל הַשַּׁעַר כוּ'. שֶׁאִם הָיָה כֹּהֵן מוֹצִיא אֶת דַּם הָאָשָׁם חוּץ לָעֲזָרָה הֲרֵי הוּא נִפְסַל בְּיוֹצֵא., לְפִיכָךְ צָרִיךְ לְהַכְנִיס יָדוֹ לְתוֹךְ חֲלַל הָעֲזָרָה, וּבִיאַת מִקְצָת שָׁרֵי רַחֲמָנָא. רַשִׁ"י: {כ} <B> לֹא הָיָה כוּ'.</B> אַף עַל גַּב דִּגְזֵרַת הַכָּתוּב הוּא, שֵׁב וְאַל תַּעֲשֶׂה לָאו מִעֲקַר הוּא. תּוֹסָפוֹת: {כא} עַל שֵׁם שֶׁהָיוּ מְצוּיִין שָׁם בָּעֲזָרָה יוֹתֵר מִשְּׁאָר הָעָם. רַשִׁ"י. וְאִם הָיוּ פּוֹרְסִין סָדִין בֵּינָהּ לְבֵין הָאֲנָשִׁים לֹא הָיָה נִוּוּל כָּל כָּךְ בִּפְנֵי הַנָּשִׁים כְּמוֹ בָּאֲנָשִׁים, שֶׁאֵין אִשָּׁה בּוֹשָׁה מִפְּנֵי חַבְרוֹתֶיהָ, וְלֹא יִוָּסְרוּ. תּוֹסָפוֹת: 

~ פרק א - משנה ו

{כב} <B> נְזָמִים.</B> הֵם תַּכְשִׁיטֵי הָאַף. הָרַמְבַּ"ם. וּמִלְּשׁוֹן הָרַ"ב פֶּרֶק ו' דְּשַׁבָּת גַּם דְּאֹזֶן בִּכְלַל: {כג} לְפִי שֶׁנֶּאֱמַר בִּתְחִלַּת פָּרָשַׁת עֲרָיוֹת כְּמַעֲשֵׂה אֶרֶץ מִצְרַיִם וְגוֹ'. וְאַף עַל פִּי שֶׁנֶּאֱמַר כְּמוֹ כֵן וּכְמַעֲשֵׂה אֶרֶץ כְּנַעַן וְגוֹ', וְיִקְּחוּ חֶבֶל שֶׁבַּמְּדִינָה, כָּל הַשִּׁנּוּיִים שֶׁנּוּכַל לַעֲשׂוֹת עוֹשִׂין כְּדֵי לְנַוְּלָהּ, וּלְפִיכָךְ לוֹקְחִין חֶבֶל מִצְרִי שֶׁלֹּא מֵהַמְּדִינָה. תּוֹסָפוֹת. וְכֵן יֵשׁ לְפָרֵשׁ כְּפִיפָה מִצְרִית דְּרֵישׁ פֶּרֶק ב'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כד} <B> גַּס.</B> כְּשֶׁהָאָדָם רוֹאֶה בְּנֵי בֵיתוֹ דַּעְתּוֹ מִתְגַּבֶּרֶת עָלָיו וְלֹא תִירָא וְלֹא תוֹדֶה. וְאָנוּ מְבַקְּשִׁין שֶׁתּוֹדֶה וְלֹא יִמָּחֶה שֵׁם הַקָּדוֹשׁ עַל הַמַּיִם. רַשִׁ"י:

{כה} <B> תְּחִלָּה.</B> לְקַלֵּל אוֹתָהּ, דִּכְתִיב בְּתֵת ה' אֶת יְרֵכֵךְ נוֹפֶלֶת וְאֶת בִּטְנֵךְ צָבָה. רַשִׁ"י. וּבַגְּמָרָא הָא כְּתִיב וְצָבְתָה בִּטְנָהּ וְנָפְלָה יְרֵכָהּ. אָמַר אַבַּיֵי כִּי לָיֵיט לָיֵיט תְּחִלָּה יָרֵךְ וַהֲדַר בֶּטֶן, וּמַיָּא כִּי אוֹרְחַיְהוּ בַּדְקֵי בֶּטֶן בָּרֵישָׁא וַהֲדַר יָרֵךְ. בִּקְלָלָה נַמִּי הָכְתִּיב לַצְבּוֹת בֶּטֶן וְלַנְפִּל יָרֵךְ, הַהִיא דְמוֹדַע לָהּ כֹּהֵן דְּבֶטֶן בָּרֵישָׁא וַהֲדַר יָרֵךְ, שֶׁלֹּא לְהוֹצִיא לַעַז עַל מַיִם הַמָּרִים:

{א} דְּאִי לָאו קְרָא הֲוָה אֲמִינָא נֵילַף בְּקַל וָחֹמֶר מִיְבָמָה דְּבִלְשׁוֹן הַקֹּדֶשׁ. רַשִׁ"י: {ב} וְרַשִׁ"י כָּתַב דְּהָכִי גָּרְסִינַן, וְאָמַר אֶל הָאִשָּׁה בְּכָל לָשׁוֹן שֶׁהִיא שׁוֹמַעַת, אֶל הָאִשָּׁה מַשְׁמַע דְּבָרִים הַנִּכְנָסִין בְּלִבָּהּ, שֶׁתְּהֵא מַכֶּרֶת בַּלָּשׁוֹן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} וְיָלֵיף אֲמִירָה מִסּוֹטָה. גְּמָרָא. וְאָתֵי שַׁפִּיר לְגִרְסַת רַשִׁ"י, דְּאִם לֹא כֵן מַאי אוּלְמֵיהּ דְּהַאי מֵהַאי: {ד}  וְאִצְטְרִיךְ לַאֲפוֹקֵי מֵרַבִּי דְּאָמַר בַּגְּמָרָא וְהָיוּ כִּכְתָבָם כוּ'. גְּמָרָא: {ה} <B> וּתְפִלָּה.</B> גְּמָרָא, רַחֲמֵי הוּא, כָּל דְּבָעֵי מְצַלֵּי. פֵּרֵשׁ רַשִׁ"י דְּהָא רַחֲמֵי נִינְהוּ, וְהַהוּא לִישָׁנָא דְּיָדַע לְכַוֵּן לְצַלֵּי: {ו} דְּאָמַר רַבִּי יוֹחָנָן כָּל הַשּׁוֹאֵל צְרָכָיו בִּלְשׁוֹן אֲרַמִּי אֵין מַלְאֲכֵי הַשָּׁרֵת נִזְקָקִין לוֹ. פֵּרֵשׁ רַשִׁ"י יָחִיד צָרִיךְ שֶׁיְּסַיְּעוּהוּ מַלְאֲכֵי הַשָּׁרֵת, צִבּוּר לֹא צְרִיכִי לְהוּ, דִּכְתִיב הֵן אֵל כַּבִּיר לֹא יִמְאַס [אִיּוֹב לו], אֵינוֹ מוֹאֵס בִּתְפִלָּתָן שֶׁל רַבִּים. וּמַשְׁמַע לְהָרַ"ב דְּכָל שֶׁכֵּן שְׁאָר לְשׁוֹנוֹת דְּלֹא, דְּהָא אֲרַמִּי קָרוֹב לִלְשׁוֹן הַקֹּדֶשׁ הוּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב דִּלְגִרְסָא שֶׁלָּנוּ צָרִיךְ לוֹמַר דְּתָנֵי וְשִׁיֵּר (אַף עַל פִּי שֶׁתָּנֵי אֵלּוּ) הַלֵּל וְקִדּוּשׁ וּבִרְכַּת הַפֵּרוֹת וְהַמִּצְוֹת: {ז} וְלֹא יָלְפִינַן מִבִּרְכוֹת הַר גְּרִזִּים כִּדְיָלְפִינַן מִנֵּיהּ בִּרְכַּת כֹּהֲנִים, לְהָכִי כְּתִיב וּבֵרַכְתָּ אֶת ה', כְּלוֹמַר מֵאַחַר שֶׁהַבְּרָכָה וְהַשֶּׁבַח כְּלַפֵּי הַשְּׁכִינָה תּוּכַל לְבָרֵךְ בְּכָל לָשׁוֹן שֶׁתִּרְצֶה, וּמִסְתָמָא בְּלָשׁוֹן שֶׁאַתָּה מַכִּיר שֶׁתִּתֵּן שֶׁבַח לְהַקָּדוֹשׁ בָּרוּךְ הוּא בְּלֵב שָׁלֵם לַהֲנָאָתְךָ: {ח} מִדְּלֹא כְּתִיב וְהִשְׁבִּיעוֹ בְּאָלָה. וְאִי לָאו קְרָא הֲוָה אֲמִינָא נֵילַף קוֹל קוֹל מִמֹּשֶׁה כְּדִלְקַמָּן מִשְׁנָה ג'. תּוֹסָפוֹת: {ט} אַגַּב דְּיָלְפִינַן בִּשְׁבוּעוֹת גְּזֵרָה שָׁוָה זוֹ לְעִנְיָן אַחֵר נָקַט לָהּ נַמִּי הָכָא. דִּמְהֵיכִי תֵּיתֵי דְּנִבְעֵי בִּלְשׁוֹן הַקֹּדֶשׁ. תּוֹסָפוֹת:

{י} וְאַיְדֵי דְּמִשְׁנָה ה' וּלְהַלָּן, תָּנֵי נַמִּי הָכָא כֵּיצַד: {יא} בְּמַתַּן תּוֹרָה. וּבִלְשׁוֹן הַקֹּדֶשׁ נִתְּנָה. רַשִׁ"י. וְלֹא יָלֵיף קוֹל קוֹל מִשְּׁבוּעַת הָעֵדוּת, דַּעֲנִיָּה וְקוֹל יָלֵיף. תּוֹסָפוֹת: 

~ פרק ז - משנה ד

{יב} <B> עַד כוּ'.</B> וְהָא דִּבְמִשְׁנָה ג' פֶּרֶק י"ב דִּיבָמוֹת קְרִיאָה אֵינָהּ מְעַכֶּבֶת בֵּין לְרַבִּי אֱלִיעֶזֶר בֵּין לְרַבִּי עֲקִיבָא, וְאִם קָרָא בִּלְשׁוֹן חֹל מְעַכֵּב, כָּתְבוּ הַתּוֹסָפוֹת דְּיֵשׁ לוֹמַר דִּסְבִירָא לְהוּ דִקְרִיאָה מְעַכֶּבֶת, וְקָסָבַר לֹא נֶחְלְקוּ רַבִּי עֲקִיבָא וְרַבִּי אֱלִיעֶזֶר בְּדָבָר זֶה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ז - משנה ה

{יג} <B> וְגוֹ'.</B> וְסֵיפָא דִּקְרָא, נֶגֶד הַכֹּהֲנִים וְהַלְוִיִּם: {יד} <B> בַּבְּרָכָה.</B> כָּל הָאֲרוּרִים הָיוּ אוֹמְרִים תְּחִלָּה בִּלְשׁוֹן בְּרָכָה. רַשִׁ"י. וּבַגְּמָרָא מַפִּיק לָהּ מִקְּרָא: {טו} <B> אֵלּוּ וָאֵלּוּ.</B> שֶׁבְּהַר גְּרִזִּים וּבְהַר עֵיבָל. וְהָא דִּכְתִיב וְאֵלֶּה יַעַמְדוּ לְבָרֵךְ, לֹא שֶׁהָיוּ הַשְּׁבָטִים מְבָרְכִין וּמְקַלְּלִין, אֶלָּא הַלְוִיִּם הָעוֹמְדִים בֵּין שְׁנֵי הֶהָרִים הוֹפְכִין פְּנֵיהֶם לְהַר גְּרִזִּים בַּבְּרָכָה וּלְהַר עֵיבָל בַּקְּלָלָה. רַשִׁ"י. שֶׁנֶּאֱמַר וְעָנוּ הַלְוִיִּם וְאָמְרוּ: {טז} <B> וּפָתְחוּ כוּ'.</B> בַּגְּמָרָא שָׁם, יָכוֹל יִהְיוּ כָּל הַבְּרָכוֹת קוֹדְמוֹת לַקְּלָלוֹת. תַּלְמוּד לוֹמַר בְּרָכָה וּקְלָלָה, בְּרָכָה אַחַת קוֹדֶמֶת לַקְּלָלָה וְאֵין כָּל הַבְּרָכוֹת קוֹדְמוֹת לַקְּלָלוֹת. וּלְהָקִישׁ בְּרָכָה לַקְּלָלָה, שֶׁשְּׁנֵיהֶם בַּלְוִיִּם, וּבְקוֹל רָם, וּבִלְשׁוֹן הַקֹּדֶשׁ, וּבִכְלָל וּפְרָט, וְאֵלּוּ וָאֵלּוּ עוֹנִין אָמֵן: {יז} <B> בְּשִׁבְעִים לָשׁוֹן.</B> כָּתַב רַבֵּנוּ בְּחַיֵּי וְכָתַבְתָּ עֲלֵיהֶן אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת, זֶה יוֹרֶה שֶׁכָּל הַתּוֹרָה כֻּלָּהּ כְּתוּבָה עֲלֵיהֶן. וְכֵן כָּתַב הָרַמְבַּ"ם שֶׁכָּל הַתּוֹרָה הָיְתָה כְּתוּבָה מִבְּרֵאשִׁית עַד לְעֵינֵי כָּל יִשְׂרָאֵל בְּתָגֶיהָ וְזִיּוּנֶיהָ, וּמִשָּׁם נֶעְתְּקוּ. וְדֶרֶךְ נֵס הָיָה בְּלֹא סָפֵק לְמַלְאוֹת שִׁבְעִים לָשׁוֹן בְּכָל הַתּוֹרָה עַל שְׁתֵּים עֶשְׂרֵה אֲבָנִים וּכְתִיבָתָם בְּיוֹם אֶחָד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יח} כְּלוֹמַר בִּכְתָב וְלָשׁוֹן: {יט} <B> הֵיטֵב.</B> דְּאִם לֹא כֵן בַּאֵר הֵיטֵב לָמָּה לִי. וְיֵשׁ מְפָרְשִׁים תֵּבַת הֵיטֵב עוֹלָה שִׁבְעִים כָּזֶה ה' ה"י הי"ט הֵיטֵ"ב (וְכֵן הִיא גִּרְסָא הַנְּכוֹנָה): {כ} <B> וְנָטְלוּ כוּ'.</B> וּבַיְרוּשַׁלְמִי, עוֹד הוּא מַעֲשֵׂה נִסִּים, נָתַן הַקָּדוֹשׁ בָּרוּךְ הוּא בִּינָה בְּלֵב כָּל אֻמָּה וְאֻמָּה וְהִשִּׂיאוּ אֶת הַתּוֹרָה שֶׁהָיְתָה כְּתוּבָה בְּשִׁבְעִים לָשׁוֹן: {כא} כְּדִכְתִיב וְהֶעֱלֵיתָ עָלָיו עוֹלוֹת. רַשִׁ"י. וּכְתִיב וְזָבַחְתָּ שְׁלָמִים. וְקָשֶׁה לִי, דַּעֲדִיפָא הֲוָה לֵיהּ לְפָרֵשׁ דְּאַף הַכְּתִיבָה הָיְתָה אַחַר הַקָּרְבָּנוֹת, שֶׁכֵּן סִדְרָן הַכָּתוּב:

{כב} <B> בִּרְכַּת כֹּהֲנִים.</B> וְתֵימַהּ דְּלֹא פֵּרֵשׁ מְנָלָן בִּלְשׁוֹן הַקֹּדֶשׁ. וּבַבְּרַיְתָא, כֹּה תְבָרְכוּ, בִּלְשׁוֹן הַקֹּדֶשׁ, נֶאֱמַר כָּאן תְּבָרְכוּ וְנֶאֱמַר לְהַלָּן אֵלֶּה יַעַמְדוּ לְבָרֵךְ אֶת הָעָם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} תֵּימַהּ, וַהֲלֹא עוֹנִין בָּרוּךְ שֵׁם כְּבוֹד מַלְכוּתוֹ לְעוֹלָם וָעֶד כְּשֶׁהָיוּ שׁוֹמְעִין אֶת הַשֵּׁם, כְּדִתְנַן מִשְׁנָה ח' פֶּרֶק ג' דְּיוֹמָא, וְזֶה הָיָה הֶפְסֵק גָּדוֹל מֵעֲנִיַּת אָמֵן. וְשֶׁמָּא הָיוּ עוֹנִין בָּרוּךְ שֵׁם כְּבוֹד מַלְכוּתוֹ לְעוֹלָם וָעֶד לְאַלְתַּר כְּשֶׁהָיוּ שׁוֹמְעִין הַשֵּׁם הַנִּכְבָּד וְהַנּוֹרָא וְלֹא הָיוּ מַמְתִּינִין עַד סוֹף הַפָּסוּק. וּלְפִי שֶׁלֹּא הָיוּ מַפְסִיקִין בֵּין פָּסוּק לְפָסוּק חָשֵׁיב לְהוּ בְּרָכָה אַחַת. אֲבָל בֵּין בְּרָכָה לִבְרָכָה לֹא הָיוּ עוֹנִין בָּרוּךְ שֵׁם כְּבוֹד מַלְכוּתוֹ לְעוֹלָם וָעֶד, דְּלֹא הָיוּ עוֹנִין רַק כְּשֶׁהַמְּבָרֵךְ הָיָה אוֹמֵר בָּרוּךְ ה' אֱלֹהֵי כוּ': {כד} וּמִיהוּ נְשִׂיאַת כַּפַּיִם הַגְבָּהָה בְּעָלְמָא בְּמַשְׁמָע: {כה} וּבָא וְשֵׁרֵת (דְּבָרִים יח) אִתַּקַּשׁ בְּרָכָה לְשֵׁירוּת, דִּכְתִיב (שָׁם י) לְשָׁרְתוֹ וּלְבָרֵךְ בִּשְׁמוֹ: {כו} <B>וַיִּשָּׂא.</B> נִרְאֶה לִי דְּלָאו רַבִּי יְהוּדָה אֲמָרָהּ, דְּאִי הָכִי לְדִידֵיהּ בִּגְבוּלִין נַמִּי לִבָּעֵי נְשִׂיאַת כַּפַּיִם לְמַעְלָה מֵרָאשֵׁיהֶן, אֶלָּא אַרֵישָׁא מְהַדַּר גַּם כֵּן מְנָלָן שֶׁנֶּאֱמַר וַיִּשָּׂא:

הָעֹשֶׁר.</B> כִּדְאָמְרִינַן תְּרֵיסָר אַלְפֵי עֶגְלֵי הֲוָה מְעַשֵּׂר מֵעֶדְרֵיהּ כָּל שַׁתָּא. רַשִׁ"י: {לט} וְלִבִּי מְהַסֵּס, שֶׁאִם כֵּן כִּנּוּי הַשֵּׁם שֶׁלֹּא לִכְבוֹדוֹ. וְהָרַמְבַּ"ם כָּתַב, וּפֵרוּשׁ קַטְנוּתָם שֶׁל חֲסִידִים בִּכְלָלָם וְשָׁרְשָׁם, לְפִי שֶׁשֹּׁרֶשׁ הַדָּבָר וּתְחִלָּתוֹ קָטָן. עַד כָּאן. וּבַיְרוּשַׁלְמִי גָּרַס שֶׁהָיָה תַּמְצִיתָן שֶׁל חֲסִידִים: {מ} <B> זִיו הַחָכְמָה.</B> כָּתַב רַשִׁ"י זֶה לֹא יָדַעְנָא מָה הִיא. עַד כָּאן. וַאַנָּא לֹא יָדַעְנָא אַמַּאי לֹא מְפָרֵשׁ כְּעֵין שֶׁמְּפָרֵשׁ לְזִיו הַכְּהֻנָּה: {מא} <B> וּמֵתָה.</B> לֹא יָדַעְנָא אַמַּאי נָקַט לָשׁוֹן מְשֻׁנֶּה וְלֹא קָאָמַר וּבָטְלָה: {מב} <B> זִיו הַכְּהֻנָּה.</B> שֶׁהָיָה חָכָם וְעָשִׁיר וְכֹהֲנִים רַבִּים אוֹכְלִים עַל שֻׁלְחָנוֹ. רַשִׁ"י. וְזֶה שֶׁכָּתַב חָכָם צָרִיךְ לוֹמַר כֹּהֵן: {מג} <B> חֲבֵרִים.</B> רְצוֹנוֹ לוֹמַר תַּלְמִידֵי חֲכָמִים, שֶׁנִּקְרְאוּ כֵן עַל שֵׁם חֶבְרָתָם חֶבְרָה נֶאֱמֶנֶת: <B>וּבְנֵי חוֹרִין.</B> מְיֻחָסִים. רַשִׁ"י: <B>חָפוּ רֹאשָׁם.</B> שֶׁגָּבְרָה יְדֵי עַזֵּי פָנִים שֶׁהֵם עֲשִׁירִים וְאֵלּוּ עֲנִיִּים: <B>וְנִדַּלְדְּלוּ כוּ'.</B> אֵין חָשׁ לָהֶם: <B>וְאֵין דּוֹרֵשׁ.</B> לְיִשְׂרָאֵל: <B>וְאֵין מְבַקֵּשׁ.</B> עֲלֵיהֶם רַחֲמִים. עַד כָּאן לְשׁוֹן רַשִׁ"י: <B>וְאֵין שׁוֹאֵל.</B> בִּשְׁלוֹם חֲבֵרוֹ. נִרְאֶה לִי: <B>הַגָּדוֹל.</B> הוּא רַבִּי אֱלִיעֶזֶר בֶּן הוּרְקְנוֹס תַּלְמִיד רַבָּן יוֹחָנָן בֶּן זַכַּאי: <B>שָׁרוּ חַכִּימַיָּא.</B> הִתְחִילוּ הַחֲכָמִים: <B>סָפְרַיָּא.</B> מְלַמְּדֵי תִינוֹקוֹת, שֶׁהֵם קְטַנִּים מִן הַחֲכָמִים: <B>כְּחַזָּנָא.</B> עַיֵּן רֵישׁ פֶּרֶק ז' דְּיוֹמָא: <B>כְּעַמָּא דְאַרְעָא.</B> כְּעַמֵּי הָאָרֶץ, דְּהַיְנוּ כָּאֲנָשִׁים הַנִּבְעָרִים מִדַּעַת וּמִיִּרְאַת ה': {מד} <B> מְבַקֵּשׁ.</B> וְלֹא קָשְׁיָא דְּלֹא אָמַר וְאֵין דּוֹרֵשׁ, כֵּיוָן דִּתְרֵי תַּנָּאֵי נִינְהוּ, וְלִישְׁנָא דְּמַר הָכֵין וְלִישָׁנָא דְּמַר הָכֵין: {מה} <B> בְּעִקְּבוֹת כוּ'.</B> בְּסוֹף הַגָּלוּת לִפְנֵי בִּיאַת הַמָּשִׁיחַ. רַשִׁ"י: {מו} <B> יַאֲמִיר.</B> בְּפֶרֶק חֵלֶק גָּרַס יְעַוֵּת. וּפֵרֵשׁ רַשִׁ"י שֶׁלֹּא יְכַבְּדוּ זֶה אֶת זֶה (מִלְשׁוֹן יְקַר, כָּבוֹד). לִישָׁנָא אַחֲרִינָא מְכֻבָּד שֶׁבָּהֶן יְהֵא עַוְּתָן וְרַמַּאי, עַד כָּאן. אוֹ יַאֲמִיר מֵעִנְיָן גַּבְהוּת כוּ', וְיִהְיֶה עִנְיָנוֹ הַיֹּקֶר יִגְדַּל עַד לְמַעְלָה, כְּאָדָם הָאוֹמֵר יֵשׁ יַקְּרוּת שֶׁאֵין לְמַעְלָה הֵימֶנּוּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מז} <B> בְּיֹקֶר.</B> שֶׁהַכֹּל עוֹסְקִין בְּמִשְׁתָּאוֹת: <B>וְהַמַּלְכוּת.</B> שֶׁל גּוֹיִם הַשּׁוֹלֶטֶת בָּעוֹלָם: <B>וְאֵין תּוֹכֵחָה.</B> אֵין לְךָ אָדָם שֶׁיּוּכַל לְהוֹכִיחַ, שֶׁכֻּלָּם נִכְשָׁלִים בְּחַטָּאוֹת, וּכְשֶׁמּוֹכִיחָן אוֹמְרִים לוֹ אַתָּה כָּמוֹנִי: <B>בֵּית וַעַד.</B> שֶׁל חֲכָמִים יִהְיֶה לְבֵית זְנוּת, שֶׁכָּלוּ הַחֲכָמִים וְאֵין לוֹמֵד תּוֹרָה, וִיהֵא חָרֵב מֵאֵין אִישׁ, וּבַעֲלֵי זִימָה מִתְגוֹדְדִים שָׁם, מִפְּנֵי שֶׁחוּץ לָעִיר הָיוּ בָתֵּי מִדְרָשׁוֹת שֶׁלָּהֶם. רַשִׁ"י: {מח} <B> זְרִיזוּת כוּ'.</B> פֵּרוּשׁ זְרִיזוּת שֶׁהוּא זָהִיר וּמָהִיר בְּמִצְוֹת, כִּדְאָמְרִינַן זְרִיזִין מַקְדִּימִין לְמִצְוֹת. הָרַ"ן. וְכֵן בִּכְלָלוֹ שֶׁהוּא זָרִיז לְהִשָּׁמֵר עַצְמוֹ שֶׁלֹּא תָּבֹא עֲבֵרָה לְיָדוֹ. רַשִׁ"י. וּפֵרוּשׁ נְקִיּוּת, נָקִי בְּאֵין חֵטְא: {מט} <B> טָהֳרָה.</B> פֵּרֵשׁ רַשִׁ"י צַח וּמְלֻבָּן, וְעָדִיף מִנָּקִי. וְהַנִמּוּקֵי יוֹסֵף מְפָרֵשׁ טָהֳרַת הַלֵּב, כְּשֶׁיִּמְנַע עַצְמוֹ מֵעֲבֵרוֹת יִגְרֹם לוֹ שֶׁאֲפִלּוּ בְּמַחְשַׁבְתּוֹ לֹא יֶחְטָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {נ} <B> פְּרִישׁוּת.</B> אַף מִדָּבָר הַמֻּתָּר פּוֹרֵשׁ, לְהַחְמִיר עַל עַצְמוֹ. רַשִׁ"י: {נא} <B> קְדֻשָּׁה.</B> שֶׁהוּא פְּרִישׁוּת טְפֵי לְהִתְקַדֵּשׁ אֲפִלּוּ מֵהַצָּרִיךְ לְגוּפוֹ, כְּגוֹן נְזִירוּת יַיִן וְתַעֲנִיּוֹת. יְפֵה מַרְאֶה. וְיִהְיֶה פֵּרוּשׁ קַדֵּשׁ עַצְמְךָ בַּמֻּתָּר לְךָ שֶׁאָמְרוּ חֲכָמֵינוּ זִכְרוֹנָם לִבְרָכָה, כְּלוֹמַר בַּמֻּתָּר וְצָרִיךְ לְךָ: {נב} <B> עֲנָוָה.</B> מִתּוֹךְ שֶׁהוּא מְבַזֶּה עִנְיְנֵי הָעוֹלָם בָּא לִידֵי עֲנָוָה. הָרַ"ן: {נג} <B> יִרְאַת חֵטְא.</B> הַיְנוּ שֶׁמִּצַּד הַחֵטְא בְּעַצְמוֹ מִתְיָרֵא. וְלֹא מִיִּרְאַת הַפֻּרְעָנוּת, אֶלָּא מֵאַהֲבַת הַמָּקוֹם אוֹ מִפְּנֵי מוֹרָאוֹ, וְאֵין זֶה מוֹרָא פֻרְעָנוּת אֶלָּא שֶׁמּוֹרָאוֹ וְגָדְלוֹ וְאֵימָתוֹ מֻטֶּלֶת עָלָיו. אִי נַמִּי רְצוֹנוֹ לוֹמַר שֶׁמְּדַקְדֵּק מְאֹד פֶּן יִכָּשֵׁל בַּחֵטְא אֲפִלּוּ בְּשׁוֹגֵג. יְפֵה מַרְאֶה: {נד} <B> חֲסִידוּת.</B> פֵּרוּשׁ שֶׁכָּל מַעֲשָׂיו לְשֵׁם שָׁמָיִם. הָרַ"ן. וְלִי נִרְאֶה לְפָרֵשׁ דְּהַיְנוּ וִתּוּרוֹ עִם חֲבֵרָיו לִפְנִים מִשּׁוּרַת הַדִּין בְּמַשָּׂאוֹ וּמַתָּנוֹ. יְפֵה מַרְאֶה: {נה} <B> רוּחַ הַקֹּדֶשׁ.</B> לְהַשְׁרוֹת עָלָיו שְׁכִינָה. רַשִׁ"י: {נו} <B> תְּחִיַּת הַמֵּתִים.</B> בַּיְרוּשַׁלְמִי, דִּכְתִיב וְנָתַתִּי רוּחִי בָּכֶם וִחְיִיתֶם [יְחֶזְקֵאל לז], שֶׁבִּתְחִלָּה יִתֵּן בָּהֶם רוּחַ הַקֹּדֶשׁ, וְעַל יְדֵי זֶה יִחְיוּ. וְכֵן פֵּרֵשׁ נִמּוּקֵי יוֹסֵף, וְנָתַתִּי רוּחִי בָּכֶם, שֶׁרוּחַ הַקֹּדֶשׁ נִדְבֶּקֶת בָּהֶם. וְאַף עַל גַּב דִּפְשׁוּטוֹ נְתִינַת הָרוּחַ בְּגוּפָם כְּדֵי שֶׁיִּחְיוּ, מִדִּכְתִיב רוּחִי וְלֹא רוּחַ מַשְׁמַע הָרוּחַ הַמְּיֻחָד לַה' דְּהַיְנוּ רוּחַ הַקֹּדֶשׁ. יְפֵה מַרְאֶה: {נז} <B> עַל יְדֵי אֵלִיָּהוּ זָכוּר לַטּוֹב.</B> וּבַיְרוּשַׁלְמִי, תְּחִיַּת הַמֵּתִים מְבִיאָה לִידֵי אֵלִיָּהוּ, וְמַיְתֵי מִדִּכְתִיב הִנֵּה אָנֹכִי שׁוֹלֵחַ לָכֶם אֶת אֵלִיָּהוּ הַנָּבִיא לִפְנֵי בֹּא יוֹם ה' הַגָּדוֹל וְהַנּוֹרָא [מַלְאָכִי ג']. וּפֵרֵשׁ הָרַ"ן וְכֵיוָן דְּבִשְׁבִיל אוֹתוֹ יוֹם אָתֵי אֵלִיָּהוּ, נִמְצָא שֶׁתְּחִיַּת הַמֵּתִים גּוֹרֶמֶת לְאֵלִיָּהוּ שֶׁיָּבֹא:
