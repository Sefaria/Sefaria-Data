~ פרק א - משנה א
<b>חַטַּאת הָעוֹף נַעֲשֵׂית לְמַטָּה.</b> מִחוּט הַסִּקְרָא {א}. דִּכְתִיב (וַיִּקְרָא ה) וְהִזָּה מִדַּם הַחַטָּאת עַל קִיר הַמִּזְבֵּחַ וְהַנִּשְׁאָר בַּדָּם יִמָּצֶה אֶל יְסוֹד הַמִּזְבֵּחַ, קִיר שֶׁהַשְּׁיָרִים שֶׁלּוֹ מִתְמַצִּין לַיְסוֹד, וְזֶהוּ קִיר הַתַּחְתּוֹן לְמַטָּה מִחוּט הַסִּקְרָא. דְּאִי קִיר הָעֶלְיוֹן דְּהַיְנוּ מִן הַחוּט וּלְמַעְלָה, פְּעָמִים שֶׁהוּא מִתְמַצֶּה לַסּוֹבֵב, כְּגוֹן שֶׁעוֹשֵׂהוּ לְמַעְלָה מִן הַסּוֹבֵב, שֶׁהַחוּט לְמַטָּה מִן הַסּוֹבֵב אַמָּה: <B>וְחַטַּאת בְּהֵמָה לְמַעְלָה.</B> דִּכְתִיב קַרְנוֹת, בְּגוּפָהּ שֶׁל קֶרֶן: <B>עוֹלַת הָעוֹף לְמַעְלָה.</B> דִּכְתִיב (שָׁם א) וּמָלַק וְהִקְטִיר וְנִמְצָה דָמוֹ, מַה הַקְטָרָה בְּרֹאשׁוֹ שֶׁל מִזְבֵּחַ, אַף מְלִיקָה וּמִצּוּי בְּרֹאשׁוֹ שֶׁל מִזְבֵּחַ: <B>וְשֶׁל בְּהֵמָה לְמַטָּה.</B> דִּכְתִיב בָּהּ יְסוֹד: <B>וְאִם שִׁנָּה בָּזֶה וּבָזֶה פָּסוּל.</B> הַאי דְּפָסוּל אִם עָשָׂה מַעֲשֵׂה חַטַּאת הָעוֹף לְמַעְלָה, דַּוְקָא בַהַזָּאָה {ב}. אֲבָל הַמְּלִיקָה שֶׁל חַטָּאת, אַף לְמַעְלָה כְּשֵׁרָה, דִּמְלִיקָה בְּכָל מָקוֹם בַּמִּזְבֵּחַ כְּשֵׁרָה. וְעוֹלַת הָעוֹף לֵית בָּהּ הַזָּאָה כִּי אִם מִצּוּי, וְאִם עֲשָׂאָהּ לְמַטָּה, פְּסוּלָה: <B>סֵדֶר קִנִּין שֶׁל חוֹבָה אֶחָד חַטָּאת וְאֶחָד עוֹלָה.</B> כְּגוֹן זָב וְזָבָה וְיוֹלֶדֶת, וְטֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו, וּשְׁמִיעַת קוֹל אָלָה, וּבִטּוּי שְׂפָתַיִם, דְּכֻלָּן מְבִיאִין קֵן בְּדַלּוּת, אֶלָּא שֶׁזָּב וְזָבָה מְבִיאִין קֵן בֵּין בְּדַלּוּת בֵּין בַּעֲשִׁירוּת. וְאַף עַל גַּב דְּאִיכָּא קִנִּין שֶׁל גֵּר דְּהֵן חוֹבָה וּשְׁנֵיהֶן עוֹלוֹת, לֹא חַיֵשׁ תַּנָּא עֲלַיְהוּ, מִשּׁוּם דְּלֹא שְׁכִיחֵי כֻּלֵּי הַאי {ה}: <B>בִּנְדָרִים וּנְדָבוֹת כֻּלָּן עוֹלוֹת.</B> לְפִי שֶׁאֵין בָּאִין בְּנֶדֶר וּנְדָבָה אֶלָּא עוֹלוֹת וּשְׁלָמִים בִּלְבַד, וְאֵין עוֹפוֹת בָּאִים שְׁלָמִים, הִלְכָּךְ נְדָרִים וּנְדָבוֹת שֶׁבְּעוֹפוֹת כֻּלָּן עוֹלוֹת: 

~ פרק א - משנה ב
<B>אֲפִלּוּ אֶחָד בְּרִבּוֹא יָמוּתוּ כֻלָּן.</B> וְלֹא בָטְלֵי בְרֻבָּא, דְּבַעֲלֵי חַיִּים חֲשִׁיבֵי וְלֹא בָטְלֵי: <B>חַטָּאת שֶׁנִּתְעָרְבָה בְחוֹבָה.</B> בֶּן יוֹנָה אוֹ תוֹר שֶׁל חַטָּאת שֶׁנִּתְעָרְבָה בִּשְׁנֵי קִנִּין שֶׁל חוֹבָה, דְּהַיְנוּ קִנִּין סְתָמָא שֶׁל יוֹלֶדֶת אוֹ שֶׁל זָבָה, שֶׁכָּל קֵן מֵהֶן הוּא אֶחָד עוֹלָה וְאֶחָד חַטָּאת, וְנִמְצָא עַתָּה חָמֵשׁ פְּרִידוֹת מְעֹרָבוֹת יַחַד: <B>אֵין כָּשֵׁר אֶלָּא מִנְיַן חַטָּאוֹת שֶׁבַּחוֹבָה.</B> וְאֵינוֹ יָכוֹל לְהָבִיא מֵאֵלּוּ הֶחָמֵשׁ פְּרִידוֹת רַק שְׁתֵּי חַטָּאוֹת בִּלְבַד, שֶׁזֶּהוּ מִנְיָן הַכָּשֵׁר בִּשְׁנֵי קִנִּין שֶׁל חוֹבָה. דְּאִי עֲבִיד שָׁלֹשׁ חַטָּאוֹת, שֶׁמָּא מִן הַשְּׁנֵי קִנִּין עֲבִיד לְהוּ וְלֹא מִן הַמְּעֹרָבוֹת בָּהֶן, וּמִשְּׁנֵי קִנִּין אֵין יְכוֹלִים לַעֲשׂוֹת רַק שְׁתֵּי חַטָּאוֹת. וְלֹא מָצֵי עֲבִיד אֲפִלּוּ עוֹלָה אַחַת, דְּשֶׁמָּא יִקַּח הַחַטָּאת הַמְּעֹרֶבֶת: <B>וְכֵן עוֹלָה.</B> בֶּן יוֹנָה אוֹ תוֹר שֶׁהֻפְרַשׁ לְשֵׁם עוֹלָה: <B>שֶׁנִּתְעָרְבָה בְחוֹבָה.</B> בִּשְׁנֵי קִנִּין סְתָמָא: <B>אֵין כָּשֵׁר אֶלָּא מִנְיַן עוֹלוֹת שֶׁבַּחוֹבָה.</B> כִּדְפָרֵישְׁנָא לְעֵיל גַּבֵּי חַטָּאת: <B>בֵּין שֶׁהַחוֹבָה מְרֻבָּה וְהַנְּדָבָה מֻעֶטֶת.</B> כְּגוֹן שֶׁקֵּן אֶחָד מְפֹרָשׁ שֶׁהֵן עוֹלוֹת, נִתְעָרֵב עִם כַּמָּה קִנִּין סְתָמָא שֶׁל יוֹלֶדֶת אוֹ שֶׁל זָבָה שֶׁכָּל קֵן מֵהֶן אֶחָד חַטָּאת וְאֶחָד עוֹלָה: <B>בֵּין וְכוּ'.</B> כְּגוֹן קֵן אֶחָד סְתוּמָה, שֶׁנִּתְעָרֵב בְּכַמָּה קִנִּין שֶׁל עוֹלוֹת. וְקָרִינַן לְעוֹלוֹת מְפֹרָשׁוֹת נְדָבָה, מִשּׁוּם דִּנְדָרִים וּנְדָבוֹת כֻּלָּן עוֹלוֹת כְּדִתְנַן בְּרֵישׁ פִּרְקִין {ז}: 

~ פרק א - משנה ג
<B>בְּחוֹבָה וּבִנְדָבָה.</B> שֶׁנִּתְעָרְבוּ קִנֵּי חוֹבָה שֶׁמִּקְצָתָן חַטָּאוֹת וּמִקְצָתָן עוֹלוֹת עִם קִנֵּי נְדָבָה שֶׁכֻּלָּן עוֹלוֹת: <B>אֲבָל בְּחוֹבָה שֶׁנִּתְעָרְבָה זוֹ בָזוֹ.</B> קֵן סְתוּמָה שֶׁל יוֹלֶדֶת אוֹ שֶׁל זָבָה שֶׁנִּתְעָרֵב בְּקֵן שֶׁל יוֹלֶדֶת אוֹ שֶׁל זָבָה אַחֶרֶת: <B>מֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל.</B> וּמַקְרִיב מֵהָאַרְבָּעָה הַמְעֹרָבִים שְׁנַיִם בִּלְבַד, אֶחָד לְעוֹלָה וְאֶחָד לְחַטָּאת וְלֹא יוֹתֵר, דְּאִי עֲבִיד שְׁתֵּי עוֹלוֹת שֶׁמָּא יַעֲשֶׂה אוֹתָן מִקֵּן אֶחָד, וְכָל קֵן סְתוּמָה הוּא אֶחָד עוֹלָה וְאֶחָד חַטָּאת: <B>הַמֻּעָט כָּשֵׁר.</B> אִם נִתְעָרֵב קֵן אֶחָד סְתוּמָה בִּשְׁנֵי קִנִּים אֲחֵרִים אוֹ בִּשְׁלֹשָׁה, אֵינוֹ יָכוֹל לְהַקְרִיב אֶלָּא קֵן אֶחָד, פְּרִידָה אַחַת לְחַטָּאת וּפְרִידָה אַחַת לְעוֹלָה. אֲבָל שְׁתֵּי עוֹלוֹת לֹא מָצֵי עֲבִיד, שֶׁמָּא יִקַּח הַקֵּן שֶׁהוּא מְבֹרָר תְּחִלָּה לְיוֹלֶדֶת אַחַת שֶׁאֵינוֹ יָכוֹל לַעֲשׂוֹת מִמֶּנּוּ אֶלָּא אֶחָד חַטָּאת וְאֶחָד עוֹלָה. וְכֵן אִם נִתְעָרְבוּ עֲשָׂרָה קִנִּין בְּמֵאָה, אֵינוֹ מַקְרִיב מִכֻּלָּן אֶלָּא עֲשָׂרָה קִנִּין, מִקְצָתָן חַטָּאוֹת וּמִקְצָתָן עוֹלוֹת, וְהַשְּׁאָר פְּסוּלִין, וְזֶהוּ הַמֻּעָט כָּשֵׁר: <B>בֵּין מִשֵּׁם אֶחָד.</B> לֵידָה וְלֵידָה, זִיבָה וְזִיבָה, מִשְּׁתֵּי נָשִׁים אוֹ מֵאִשָּׁה אֶחָת: <B>בֵּין מִשְּׁנֵי שֵׁמוֹת.</B> לֵידָה וְזִיבָה, בֵּין מִשְּׁתֵּי נָשִׁים בֵּין מֵאִשָּׁה אַחַת. כְּדִמְפָרֵשׁ וְאָזֵיל: 

~ פרק א - משנה ד
<B>רַבִּי יוֹסֵי אוֹמֵר כוּ'.</B> לָאו מִשּׁוּם דְּסָבַר רַבִּי יוֹסֵי יֵשׁ בְּרֵרָה וְנֵימָא דִּכְשֶׁהִקְרִיב הַכֹּהֵן קֵן אֶחָד לְשֵׁם אַחַת מֵהֶן הֻבְרַר הַדָּבָר לְמַפְרֵעַ דְּמִתְּחִלָּה בִּשְׁעַת לְקִיחַת הַקֵּן הָיָה קֵן זֶה שֶׁלָּהּ {י}, אֶלָּא מְפָרֵשׁ בַּגְּמָרָא בְּפֶרֶק בַּכֹּל מְעָרְבִין [עֵרוּבִין דַּף לו] דְּרַבִּי יוֹסֵי מַיְרֵי בְּשֶׁהִתְנוּ בִשְׁעַת לְקִיחָה שֶׁיּוּכַל הַכֹּהֵן לַעֲשׂוֹת קֵן שֶׁיִּרְצֶה לְשֵׁם אֵיזוֹ אִשָּׁה שֶׁיִּרְצֶה. וַהֲלָכָה כְּרַבִּי יוֹסֵי:

~ פרק ב - משנה א
<B>קֵן סְתוּמָה.</B> שֶׁלֹּא פֵרֵשׁ זוֹ חַטָּאת וְזוֹ עוֹלָה. וְלֹא מִבַּעְיָא מְפֹרֶשֶׁת, דִּפְשִׁיטָא שֶׁאִם פָּרַח הָאֶחָד יִקַּח זוּג לַשֵּׁנִי כֵּיוָן שֶׁהֵן נִפְרָשִׁים זֶה מִזֶּה וְכָל אֶחָד שְׁמוֹ עָלָיו {א}. אֶלָּא אֲפִלּוּ סְתוּמָה שֶׁעֲדַיִן הֵם יַחַד וְעוֹמְדִים לְהִתְפָּרֵשׁ וְסָלְקָא דַּעְתָּךְ אֲמֵינָא שֶׁאִם פָּרַח הָאֶחָד יָמוּת הַשֵּׁנִי, קָמַשְׁמַע לָן: <B>לְבֵין הַמֵּתוֹת.</B> לְבֵין קִנִּין שֶׁמִּשְׁפָּטָן שֶׁיָּמוּתוּ כֻלָּם, כְּגוֹן חַטָּאת שֶׁנִּתְעָרְבָה בְעוֹלָה דִּתְנַן בְּהוּ בְּפֶרֶק קַמָּא יָמוּתוּ כֻלָּם: <B>לְבֵין הַקְּרֵבוֹת.</B> לְבֵין קִנִּין סְתוּמוֹת הָעוֹמְדוֹת לִקָּרֵב: <B>פָּסוּל וּפוֹסֵל אֶחָד כְּנֶגְדּוֹ.</B> לְפִי שֶׁהוּא סְתָם וְלֹא נִתְפָּרֵשׁ אִם חַטָּאת אִם עוֹלָה, לְפִיכָךְ אֵינוֹ פוֹסֵל אֶלָּא אֶחָד כְּנֶגְדּוֹ, כְּדִמְפָרֵשׁ וְאָזֵיל:

פְרִידוֹת שֶׁפָּרְחוּ מִמֶּנָּה יִהְיוּ קְרֵבִים עוֹלוֹת שְׁתֵּיהֶן וְאִם כֵּן יַעֲשׂוּ בְשִׁשָּׁה קִנִּין שֶׁבַע עוֹלוֹת, אוֹ יִהְיוּ עֲשׂוּיִין שְׁתֵּיהֶן חַטָּאוֹת, וְאֵין יְכוֹלִים לַעֲשׂוֹת בְּשִׁשָּׁה קִנִּין אֶלָּא שֵׁשׁ עוֹלוֹת וְשֵׁשׁ חַטָּאוֹת. וְכֵן מֵחֲמִישִׁית לָרְבִיעִית. וּמֵרְבִיעִית לַשְּׁלִישִׁית. וּמִשְּׁלִישִׁית לַשְּׁנִיָּה, אֵין בַּשְּׁלִישִׁית כִּי אִם שְׁתֵּי פְרִידוֹת אַחַת חַטָּאת וְאַחַת עוֹלָה. וְאִם תֹּאמַר, וְלָמָה הִפְסִידָה הַשְּׁלִישִׁית שְׁתַּיִם, שֶׁהֲרֵי מֵאוֹתָן שֶׁפָּרְחוּ בַשְּׁנִיָּה בַחֲזָרָה לֹא יַקְרִיבוּ, דְּהָא קָתָנֵי הַשְּׁנִיָּה אֵין לָהּ כְּלוּם, וַהֲוָה לָן לְמֵימַר בַּשְּׁלִישִׁית שְׁנֵי קִנִּין וּבַשְּׁנִיָּה אֵין בָּהּ כְּלוּם, אוֹ בַּשְּׁנִיָּה קֵן אֶחָד וּבַשְּׁלִישִׁית קֵן אֶחָד. וְיֵשׁ לוֹמַר, כֵּיוָן דְּבִשְׁאָר קִנִּין מֵרְבִיעִי וָאֵילָךְ נִפְסְלוּ שְׁנֵי קִנִּין אֶחָד בַּהֲלִיכָה וְאַחַר בַּחֲזָרָה, גָּזְרִינַן כְּמוֹ כֵן בַּשְּׁלִישִׁית. אֲבָל בַּשְּׁבִיעִית לֹא נִפְסַל כִּי אִם קֵן אֶחָד, דְּלֹא הָלַךְ מִמֶּנָּה רַק פְּרִידָה אַחַת בַּחֲזָרָה, וְאִלּוּ בַהֲלִיכָה לֹא נִפְסַל בָּהּ כְּלוּם וְלֹא נֶחְסַר מִמֶּנָּה דָּבָר: <B>פָּרַח וְחָזַר.</B> לָאו אַרִאשׁוֹנָה וּשְׁנִיָּה קָאֵי, דְּכֵיוָן דְּאֵין לָהֶם כְּלוּם אֶלָּא מֵתִים, אִם פָּרַח אֶחָד מֵהֶם בֵּין שְׁאָר קִנִּין יָמוּתוּ כֻלָּם כְּדִתְנַן בְּסָמוּךְ וְאִם פָּרַח מִבֵּין הַמֵּתוֹת לְכֻלָּם כֻּלָּם יָמוּתוּ. אֶלָּא קָאֵי אַשְּׁלִישִׁית שֶׁפָּרַח מִמֶּנָּה לָרְבִיעִית, וּמֵרְבִיעִית לַחֲמִישִׁית, וּמֵחֲמִישִׁית לַשִּׁשִּׁית, וּמִשִּׁשִּׁית לַשְּׁבִיעִית, וְחָזַר מִשְּׁבִיעִית לַשִּׁשִּׁית עַד הַשְּׁלִישִׁית, הֲרֵי מִן הַשְּׁבִיעִית חָסֵר שְׁתֵּי פְרִידוֹת, אַחַת בַּחֲזָרָה רִאשׁוֹנָה וְאַחַת בַּחֲזָרָה שְׁנִיָּה. וְאֵין כָּשֵׁר בַּשְּׁבִיעִית רַק חָמֵשׁ פְּרִידוֹת לְעוֹלוֹת וְחָמֵשׁ פְּרִידוֹת לְחַטָּאוֹת, דְּאִי עֲבִיד שֵׁשׁ עוֹלוֹת, שֶׁמָּא הַשְּׁתַּיִם שֶׁהָלְכוּ מִמֶּנָּה יִהְיוּ קְרֵבִין עוֹלוֹת, הֲרֵי שְׁמֹנֶה עוֹלוֹת, וְאֵין בַּשֶּׁבַע קִנִּין רַק שֶׁבַע עוֹלוֹת. וּמִן הַשִּׁשִּׁית חָסֵר אַרְבַּע פְּרִידוֹת, שְׁתַּיִם בִּפְרִיחָה וַחֲזָרָה רִאשׁוֹנָה, וּשְׁתַּיִם בַּשְּׁנִיָּה, וְנִשְׁאֲרוּ שְׁנֵי קִנִּין כְּשֵׁרִים, דְּאִי עֲבִיד שָׁלֹשׁ עוֹלוֹת, שֶׁמָּא אַרְבַּע שֶׁהָלְכוּ מִמֶּנָּה יִהְיוּ עוֹלוֹת, הֲרֵי שֶׁבַע עוֹלוֹת, וְאֵין בַּשִּׁשָּׁה קִנִּים רַק שֵׁשׁ עוֹלוֹת. וְכֵן הַחֲמִישִׁית אֵין כָּשֵׁר בָּהּ כִּי אִם אֶחָד לְעוֹלָה וְאֶחָד לְחַטָּאת. וּכְמוֹ כֵן קַשְׁיָא בְכָאן כֵּיוָן דְּהָרְבִיעִית אֵין לָהּ כְּלוּם הָיָה יָכוֹל לְהַקְרִיב בַּחֲמִישִׁית שָׁלֹשׁ עוֹלוֹת וְשָׁלֹשׁ חַטָּאוֹת, דַּהֲרֵי אוֹתָן שֶׁהָלְכוּ בָרְבִיעִית אֵינָן קְרֵבוֹת, וּלְפִי זֶה שֶׁהַחֲמִישִׁית אֵין לָהּ אֶלָּא אַחַת הָיָה יָכוֹל לְהַקְרִיב בָּרְבִיעִית שְׁתֵּי עוֹלוֹת וּשְׁתֵּי חַטָּאוֹת. אֶלָּא גָּזְרִינַן בְּכָל פְּרִיחָה וַחֲזָרָה שְׁנֵי קִנִּין [גְּזֵרָה בָזוֹ אַטּוּ בָזוֹ]. וְכֵן כִּי פָרַח וְחָזַר פַּעַם שְׁלִישִׁית אֵין לָהֶם כְּלוּם לְבַד הַשְּׁבִיעִית שֶׁיֵּשׁ לָהּ אַרְבַּע, כְּמוֹ כֵן גָּזְרִינַן בְּכָל פְּרִיחָה וַחֲזָרָה שְׁנֵי קִנִּין: <B>וְיֵשׁ אוֹמְרִים הַשְּׁבִיעִית לֹא הִפְסִידָה כְלוּם.</B> פֵּרוּשׁ, בִּפְרִיחָה וַחֲזָרָה שְׁלִישִׁית. אֶלָּא לְעוֹלָם יַקְרִיבוּ מִמֶּנָּה חֲמִשָּׁה קִנִּין {ה}, דְּשָׁלֹשׁ פְּרִידוֹת פָּרְחוּ מִמֶּנָּה לְכָל הַיּוֹתֵר וְנִשְׁאֲרוּ בָהּ רַק אַחַת עֶשְׂרֵה פְרִידוֹת, וְיַעֲשֶׂה מֵהֶם חָמֵשׁ חַטָּאוֹת וְחָמֵשׁ עוֹלוֹת, כֵּיוָן דְּאַף הַשִּׁשִּׁית אֵין לָהּ כְּלוּם. וְהָכָא לֹא גָזְרִינַן בִּפְרִיחָה זוֹ כְּמוֹ בִּשְׁאָר קִנִּין, דְּכֵיוָן דְּאֵינָהּ מַפְסֶדֶת בַּפְּרִיחָה בְּשׁוּם פַּעַם, לֹא גָזְרוּ עַל הַחֲזָרָה. וְאֵין הֲלָכָה כְּיֵשׁ אוֹמְרִים:

<B>קֵן סְתוּמָה.</B> שֶׁלֹּא פֵרֵשׁ אֵיזוֹ חַטָּאת וְאֵיזוֹ עוֹלָה: <B>וְקֵן מְפֹרֶשֶׁת.</B> שֶׁיָּדוּעַ שֶׁזּוֹ חַטָּאת וְזוֹ עוֹלָה. וְהָכָא מַיְרֵי שֶׁנִּתְעָרְבוּ שְׁתֵּי הַפְּרִידוֹת יַחַד אַחַר שֶׁהָיוּ מְפֹרָשׁוֹת וְהַשְׁתָּא אֵין יָדוּעַ אֵיזוֹ חַטָּאת וְאֵיזוֹ עוֹלָה, וְהָכִי מוּכָח בַּסֵּיפָא דְּקָתָנֵי אוֹ שֶׁפָּרַח מִן הַמְפֹרֶשֶׁת רִאשׁוֹן לַסְּתוּמָה יָמוּתוּ כֻלָּם, וְאִם לֹא נִתְעָרְבוּ פְרִידוֹת הַמְפֹרָשׁוֹת זוֹ בָזוֹ אַמַּאי יָמוּתוּ, וְאִם הָעוֹלָה פָרְחָה, הַחַטָּאת הַנִּשְׁאֶרֶת תִּקָּרֵב, וְקֵן הַסְּתוּמָה יַקְרִיבוּ בָהּ עוֹלָה אַחַת מִפְּנֵי הָעוֹלָה שֶׁנִּתְעָרְבָה בָהּ. אֶלָּא מַיְרֵי שֶׁנִּתְעָרְבוּ פְרִידוֹת הַמְפֹרָשׁוֹת זוֹ בָזוֹ, וְאִם נִתְעָרְבָה פְרִידָה אַחַת מִן הַקֵּן הַסְּתוּמָה לְתוֹךְ הַמְפֹרֶשֶׁת, יִקַּח זוּג לַשֵּׁנִי וְיַפְרִישׁ אֶחָד חַטָּאת וְאֶחָד עוֹלָה, וְהַשְּׁלֹשָׁה שֶׁנִּתְעָרְבוּ יַחַד יָמוּתוּ כֻלָּן. חָזַר אֶחָד מִן הַשְּׁלֹשָׁה שֶׁנִּתְעָרְבוּ וְנִתְעָרֵב עִם הַיְחִידִי הַנִּשְׁאָר בַּסְּתוּמָה, אוֹ בִּתְחִלַּת הַפְּרִיחָה נִתְעָרֵב אֶחָד מִן הַמְפֹרָשׁוֹת לַסְּתוּמָה, וְהַיְנוּ רִאשׁוֹן דְּקָאָמַר, כְּלוֹמַר שֶׁבִּתְחִלַּת פְּרִיחָה הָלַךְ אֶחָד וְנִתְעָרֵב בַּסְּתוּמָה, יָמוּתוּ כֻלָּן, דְּכֵיוָן שֶׁפְּרִידוֹת הַמְפֹרָשׁוֹת נִתְעָרְבוּ יַחַד אֵינָן קְרֵבִין עוֹד, וְכִי פָרַח אֶחָד מֵהֶן בְּמָקוֹם אַחֵר הַיְנוּ אֶחָד מִן הַמֵּתוֹת שֶׁפָּרַח לְבֵין הַקְּרֵבוֹת, יָמוּתוּ כֻלָּן:

<B>חַטָּאת מִכָּאן.</B> פְּרִידָה הַמְפֹרֶשֶׁת לְחַטָּאת לְצַד אֶחָד, וּפְרִידָה הַמְפֹרֶשֶׁת לְעוֹלָה לַצַּד הַשֵּׁנִי, וּשְׁתֵּי פְרִידוֹת שֶׁל קֵן סְתוּמָה בָּאֶמְצַע: <B>פָּרַח מִן הָאֶמְצַע.</B> פְּרִידָה אַחַת לְיָמִין וּפְרִידָה אַחַת לִשְׂמֹאל: <B>לֹא הִפְסִיד כְּלוּם.</B> דְּאוֹתוֹ שֶׁמְּעֹרָב עִם הַחַטָּאת יַעֲשֶׂה חַטָּאת, וְאוֹתוֹ שֶׁעִם הָעוֹלָה יַעֲשֶׂה עוֹלָה, אֲבָל לֹא חַטָּאת, דְּשֶׁמָּא יַעֲשׂוּהוּ מִן הָעוֹלָה הַמְפֹרֶשֶׁת: <B>חָזַר לָאֶמְצַע.</B> חָזְרוּ מִן הַצְּדָדִים אֶחָד מִכָּאן וְאֶחָד מִכָּאן וְנִתְעָרְבוּ יַחַד, אֵלּוּ יָמוּתוּ, דְּחַטָּאת וְעוֹלָה נִתְעָרְבוּ יַחַד. אֲבָל אוֹתָן שֶׁנִּשְׁאֲרוּ כָּל אֶחָד לְבַדּוֹ, קְרֵבִים כָּל אֶחָד כְּדִינוֹ: <B>חָזַר.</B> מִן הָאֶמְצָעִיִּים לַצְּדָדִים, הֲרֵי חַטָּאוֹת וְעוֹלוֹת מְעֹרָבִים, וְיָמוּתוּ כֻלָּן {ו}: <B>אֵין מְבִיאִים תּוֹרִים כְּנֶגֶד בְּנֵי יוֹנָה.</B> אֶלָּא אוֹ שְׁנֵיהֶם תּוֹרִים אוֹ שְׁנֵיהֶן בְּנֵי יוֹנָה: <B>תִּכְפֹּל וְתָבִיא עוֹלָתָהּ בֶּן יוֹנָה.</B> כְּדִין הַחַטָּאת. שֶׁהַחַטָּאת הִיא עִקָּר {ח}, בֵּין שֶׁהִפְרִישַׁתּוּ תְחִלָּה בֵּין שֶׁהִפְרִישַׁתּוּ בַסּוֹף: <B>הוֹלְכִין אַחַר הָרִאשׁוֹן.</B> וְאֵינָהּ מְבִיאָה הַשֵּׁנִי אֶלָּא מִמִּין שֶׁהִפְרִישָׁה בָרִאשׁוֹן, בֵּין שֶׁהָיָה הָרִאשׁוֹן חַטָּאת בֵּין שֶׁהָיָה עוֹלָה. וַהֲלָכָה כְּתַנָּא קַמָּא: <B>לֹא יָבִיאוּ הַיּוֹרְשִׁים חַטָּאתָהּ.</B> דְּהַוְיָא חַטָּאת שֶׁמֵּתוּ בְעָלֶיהָ. וְאִי לֹא מֵהַאי טַעֲמָא, מְבִיאִים חַטָּאת אַף עַל פִּי שֶׁקָּרְבָה עוֹלָה תְּחִלָּה. וְאַף עַל גַּב דִּבְכָל מָקוֹם חַטָּאת קוֹדֶמֶת לְעוֹלָה, הַיְנוּ דַּוְקָא לְמִצְוָה אֲבָל לֹא לְעַכֵּב:

<B>בַּמֶּה דְבָרִים אֲמוּרִים בְּכֹהֵן נִמְלָךְ.</B> אַפִּרְקָא קַמָּא קָאֵי. וְהָכִי קָאָמְרִינַן, בַּמֶּה דְבָרִים אֲמוּרִים דְּחַטָּאת שֶׁנִּתְעָרְבָה בְעוֹלָה שֶׁכֻּלָּם יָמוּתוּ {א}, וְכֵן אַחַת בַּעֲשִׂיַּת כֹּהֵן:

<B>זֶה הַכְּלָל כָּל מָקוֹם שֶׁאַתָּה יָכוֹל לַחֲלֹק כוּ'.</B> כְּלוֹמַר, שֶׁקַּן אִשָּׁה אַחַת אֵינוֹ נֶחְלָק לִשְׁנַיִם אֶלָּא קַן אִשָּׁה אַחַת לְמַעְלָה וְקַן אִשָּׁה אַחֶרֶת לְמַטָּה, מֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל: <B>וְכָל מָקוֹם כוּ'.</B> כְּגוֹן אֶחָד לָזוֹ וּשְׁנַיִם לָזוֹ, שֶׁיֵּשׁ מֵחֵלֶק שֶׁל אִשָּׁה אַחַת לְמַעְלָה וּלְמַטָּה, הַמְרֻבֶּה כָּשֵׁר כִּדְפָרִישִׁית: 

~ פרק ג - משנה ג
<B>חַטָּאת לָזוֹ וְעוֹלָה לָזוֹ כוּ'.</B> הָא נַמִּי קָאֵי אַדְּתָנֵי לְעֵיל בְּפֶרֶק קַמָּא, עוֹלָה שֶׁנִּתְעָרְבָה בְחַטָּאת אוֹ אִפְּכָא, כֻּלָּן יָמוּתוּ. וְהָנֵי מִלֵּי בְּנִמְלָךְ, אֲבָל בְּלֹא נִמְלָךְ עַל תַּעֲרָבְתּוֹ וְעָשָׂה קֵן {ג} לְמַטָּה וְקֵן לְמַעְלָה, מֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל. וְאַף עַל גַּב דְּאִי אָתֵי לְאִמְּלוֹכֵי אָמְרִינַן לֵיהּ לֹא תַקְרִיב, כִּדְתַנְיָא יָמוּתוּ כֻלָּן, הַשְׁתָּא דְּלֹא נִמְלַךְ אָמְרִינַן שֶׁכָּל אֶחָד נַעֲשָׂה כְהִלְכָתוֹ וְכָשֵׁר: 

~ פרק ג - משנה ד
<B>חַטָּאת וְעוֹלָה וּסְתוּמָה וּמְפֹרֶשֶׁת.</B> שְׁתֵּי נָשִׁים שֶׁלָּקְחוּ שְׁלֹשָׁה קִנִּין, זוֹ צְרִיכָה עוֹלָה וְקֵן שָׁלֵם, דְּהַיְנוּ שְׁתֵּי עוֹלוֹת וְחַטָּאת אַחַת, וְזוֹ צְרִיכָה חַטָּאת וְקֵן שָׁלֵם, דְּהַיְנוּ שְׁתֵּי חַטָּאוֹת וְעוֹלָה אַחַת, וּפֵרְשׁוּ אֶחָד מִן הַקִּנִּין פְּרִידָה עוֹלָה לָזוֹ וּפְרִידָה חַטָּאת לָזוֹ, וְאֶחָד מִן הַקִּנִּין הִנִּיחוּ סְתוּמָה שֶׁלֹּא פֵרְשׁוּ אֵיזוֹ עוֹלָה וְאֵיזוֹ חַטָּאת, וְהַשְּׁלִישִׁי פֵּרְשׁוּ אֵיזוֹ עוֹלָה וְאֵיזוֹ חַטָּאת, אֲבָל בְּעָלִים לֹא נִתְפָּרְשׁוּ, הֲרֵי חַטָּאת לָזוֹ וְעוֹלָה לָזוֹ וְקֵן סְתוּמָה וְקֵן מְפֹרֶשֶׁת, וְצָרִיךְ לְהַקְרִיב שְׁנֵי הַקִּנִּין הַלָּלוּ סְתָם עַל שְׁתֵּיהֶן, הָעוֹלוֹת לְמַעְלָה וְחַטָּאוֹת לְמַטָּה, וְעוֹלָה לָזוֹ וְחַטָּאת לָזוֹ צָרִיךְ לְהַקְרִיב לְשֵׁם בְּעָלִים {ד}. וְאִם נְתָנָם הַכֹּהֵן חֶצְיָן לְמַטָּה וְחֶצְיָן לְמַעְלָה, אֵין כְּשֵׁרָה אֶלָּא סְתוּמָה, דְּהֵי מִנַּיְהוּ דַּעֲבַד עוֹלָה וְהֵי מִנַּיְהוּ דַּעֲבַד חַטָּאת, כָּשֵׁר, כֵּיוָן שֶׁכָּל קֵן וְקֵן הָיָה בִּפְנֵי עַצְמוֹ: <B>מִתְחַלֶּקֶת בֵּינֵיהֶם.</B> הוֹאִיל וּבְעֵרוּב לְקָחוּם, הָאַחַת יָצְאָה יְדֵי חַטָּאת וְהָאַחַת יָצְאָה יְדֵי עוֹלָה: 

~ פרק ג - משנה ה
<B>חַטָּאת שֶׁנִּתְעָרְבָה בְחוֹבָה.</B> כְּלוֹמַר הָא דִתְנִינַן לְעֵיל בְּפֶרֶק קַמָּא חַטָּאת שֶׁנִּתְעָרְבָה בְחוֹבָה אֵין כָּשֵׁר אֶלָּא מִנְיַן חַטָּאוֹת שֶׁבַּחוֹבָה, פְּעָמִים שֶׁעוֹלֶה אוֹתוֹ מִנְיָן לְמֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל, וּפְעָמִים שֶׁעוֹלֶה לְפָחוֹת מִמֶּחֱצָה, וּמְפָרֵשׁ וְאָזֵיל כֵּיצַד: <B>חוֹבָה שְׁנַיִם בְּחַטָּאת מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל.</B> פֵּרוּשׁ, שְׁנֵי קִנִּין חוֹבָה שֶׁיֵּשׁ בָּהֶן כִּפְלַיִם בַּחַטָּאת, כְּגוֹן שֶׁהִקְרִיב מֵהֶן עוֹלָה אַחַת וְנִשְׁתַּיְּרוּ שְׁתֵּי חַטָּאוֹת וְעוֹלָה אֶחָת: <B>מֶחֱצָה כָשֵׁר וּמֶחֱצָה פָסוּל.</B> שֶׁעוֹלָה אַחַת הַנִּשְׁאֶרֶת אֵינוֹ יָכוֹל לְהַקְרִיב שֶׁמָּא זֶהוּ שֶׁנִּקְבַּע לְחַטָּאת {ה}, וְחַטָּאת הַמְעֹרֶבֶת אֵינוֹ יָכוֹל לְהַקְרִיב שֶׁמָּא הִיא הָעוֹלָה הַנִּשְׁאֶרֶת {ו}, הִלְכָּךְ אֵינוֹ מַקְרִיב אֶלָּא שְׁתֵּי הַחַטָּאוֹת, וְהַחַטָּאת הַמְּעֹרֶבֶת וְהָעוֹלָה הַנִּשְׁאֶרֶת פְּסוּלִין, וְהַיְנוּ מֶחֱצָה כָּשֵׁר וּמֶחֱצָה פָּסוּל: <B>חַטָּאת שְׁנַיִם בְּחוֹבָה מִנְיָן שֶׁבַּחוֹבָה כָּשֵׁר.</B> פֵּרוּשׁ, כְּגוֹן שֶׁקָּרְבָה חַטָּאת אַחַת מִן הַשְּׁנֵי קִנִּין וְנִשְׁתַּיְּרוּ שְׁתֵּי עוֹלוֹת וְחַטָּאת אֶחָת {ז}, הֲרֵי הֵן אַרְבַּע פְּרִידוֹת עִם חַטָּאת שֶׁנִּתְעָרְבָה בֵינֵיהֶן, וַהֲרֵי אֵינוֹ יָכוֹל לְהַקְרִיב שְׁתֵּי הָעוֹלוֹת {ח} שֶׁמָּא יִקַּח חַטָּאת הַמְעֹרֶבֶת, וְגַם שְׁתֵּי חַטָּאוֹת {ט} לֹא יַקְרִיב שֶׁמָּא יִקַּח אוֹתוֹ שֶׁנִּקְבַּע לְעוֹלָה, הִלְכָּךְ אֵינוֹ מַקְרִיב אֶלָּא חַטָּאת אַחַת, הֲרֵי עוֹלֶה אוֹתוֹ מִנְיָן הַפָּחוֹת שֶׁבַּחוֹבָה לְפָחוֹת מִמֶּחֱצָה כָּשֵׁר. וּכְהַאי גַּוְנָא מְפָרְשִׁינַן נַמִּי וְכֵן עוֹלָה שֶׁנִּתְעָרְבָה בְחוֹבָה: 

~ פרק ג - משנה ו
<B>הֲרֵי עָלַי קֵן לִכְשֶׁאֵלֵד זָכָר.</B> בְּיוֹלֶדֶת עֲנִיָּה עָסְקִינַן. דְּאִלּוּ עֲשִׁירָה מְבִיאָה כֶבֶשׂ לְחוֹבָתָהּ: <B>וְהַכֹּהֵן צָרִיךְ לַעֲשׂוֹת שָׁלֹשׁ לְמַעְלָה.</B> שֶׁקַּן נְדָבָה שְׁתֵּיהֶן עוֹלוֹת וְקֵן שֶׁל חוֹבָה עוֹלָה לְמַעְלָה וְחַטָּאת לְמַטָּה, וְהוּא לֹא עָשָׂה כֵן, אֶלָּא שְׁתַּיִם לְמַעְלָה וּשְׁתַּיִם לְמַטָּה {י}, וְלֹא נִמְלַךְ בָּהּ לָדַעַת עַל מַה הֱבִיאָתָן, וְנִפְסְלָה פְּרִידָה אַחַת, הִלְכָּךְ צְרִיכָה שֶׁתָּבִיא פְּרִידָה אַחֶרֶת וְתַקְרִיבֶנָּהּ לְמַעְלָה: <B>מִמִּין אֶחָד.</B> כְּלוֹמַר, בַּמֶּה דְבָרִים אֲמוּרִים, כְּשֶׁהָיוּ מִמִּין אֶחָד. אֲבָל הָיוּ הַשְּׁנֵי קִנִּין שֶׁהֵבִיאָה מִשְּׁנֵי מִינִים, דְּהַיְנוּ קֵן אֶחָד תּוֹרִים וְקֵן אֶחָד בְּנֵי יוֹנָה, צְרִיכָה לְהָבִיא פְּרִידָה אַחַת תּוֹר וּפְרִידָה אַחַת מִבְּנֵי יוֹנָה, כְּדֵי לְהַשְׁלִים הַפְּרִידָה שֶׁנִּפְסְלָה, דְּלֹא יָדְעִינַן אִם מִן הַתּוֹרִים אִם מִן בְּנֵי יוֹנָה הָיְתָה, וּתְנַן לְעֵיל אֵין מְבִיאִים תּוֹרִים כְּנֶגֶד בְּנֵי יוֹנָה כוּ': <B>פֵּרְשָׁה נִדְרָהּ.</B> כְּשֶׁהֵבִיאָה שְׁנֵי קִנִּין לַכֹּהֵן פֵּרְשָׁה וְאָמְרָה אֵלּוּ לְנִדְרִי וְאֵלּוּ לְחוֹבָתִי: <B>מִמִּין אֶחָד.</B> כְּלוֹמַר אִם שְׁנֵי קִנִּין הַלָּלוּ הָיוּ מִמִּין אֶחָד דְּהַיְנוּ שְׁנֵיהֶן תּוֹרִים אוֹ שְׁנֵיהֶן בְּנֵי יוֹנָה, וְהִקְרִיב מֵהֶן הַכֹּהֵן שְׁתַּיִם לְמַעְלָה וּשְׁתַּיִם לְמַטָּה, אֵין כָּשֵׁר אֶלָּא פְּרִידָה אַחַת מִכָּל הָאַרְבַּע שֶׁהִקְרִיב, וְהִיא הָעוֹלָה שֶׁבַּחוֹבָה, מִידִי דַּהֲוֵי כְעוֹלָה שֶׁנִּתְעָרְבָה בְחוֹבָה דַּאֲמַרַן לְעֵיל אֵין כָּשֵׁר אֶלָּא מִנְיַן עוֹלוֹת שֶׁבַּחוֹבָה, וּצְרִיכָה לְהָבִיא עוֹד שָׁלֹשׁ פְּרִידוֹת, וּמַקְרִיב מֵהֶן אַחַת לְמַטָּה וְהִיא הַחַטָּאת שֶׁבַּחוֹבָה, וּשְׁתַּיִם לְמַעְלָה וְהוּא הַנֶּדֶר: <B>מִשְּׁנֵי מִינִים תָּבִיא אַרְבַּע.</B> אִם הֵבִיאָה שְׁנֵי קִנִּין מִשְּׁנֵי מִינִים וּפֵרְשָׁה נִדְרָהּ וְלֹא יָדַע הַכֹּהֵן אֵיזֶה שְׁתַּיִם מֵהֶן עָשָׂה לְמַעְלָה וְאֵיזֶה עָשָׂה לְמַטָּה, צְרִיכָה לְהָבִיא אַרְבַּע פְּרִידוֹת, דְּשֶׁמָּא הַשְּׁתֵּי תוֹרִים שֶׁהֵבִיאָה לְנִדְרָהּ וְהֵן עוֹלוֹת נַעֲשׂוּ לְמַטָּה וְנִשְׁאַר עָלֶיהָ הַנֶּדֶר כֻּלּוֹ, וּשְׁנֵי בְנֵי יוֹנָה שֶׁהֵן חוֹבָה נַעֲשׂוּ לְמַעְלָה וְנִשְׁאַר עָלֶיהָ בֶּן יוֹנָה אֶחָד לְחַטָּאת מֵחוֹבָתָהּ, אוֹ שֶׁמָּא שְׁנֵי בְנֵי יוֹנָה שֶׁל חוֹבָה נַעֲשׂוּ לְמַטָּה וְנִשְׁאַר עָלֶיהָ בֶּן יוֹנָה שֶׁל עוֹלָה מֵחוֹבָתָהּ: <B>קָבְעָה נִדְרָהּ.</B> כְּגוֹן שֶׁאָמְרָה בִשְׁעַת נִדְרָהּ הֲרֵי עָלַי קֵן מִמִּין פְּלוֹנִי. וּפֵרְשָׁה נִדְרָהּ דִּלְעֵיל, לֹא פֵרְשָׁה כְלוּם בִּשְׁעַת הַנֶּדֶר, אֶלָּא לְאַחַר שֶׁנָּדְרָה סְתָם הֲרֵי עָלַי קֵן הֵבִיאָה מֵאֵיזֶה מִין שֶׁרָצְתָה וְאָמְרָה אֵלּוּ לְנִדְרִי: <B>צְרִיכָה לְהָבִיא עוֹד חָמֵשׁ פְּרִידוֹת.</B> הָכָא מַיְרֵי כְּגוֹן שֶׁלְּאַחַר שֶׁקָּבְעָה נִדְרָהּ וְהֵבִיאָה נִדְרָהּ עִם חוֹבָתָהּ אֶל הַכֹּהֵן עָשָׂה הַכֹּהֵן שְׁתַּיִם לְמַעְלָה וּשְׁתַּיִם לְמַטָּה וְלֹא יָדַע אִם נִדְרָהּ עָשָׂה לְמַעְלָה וְחוֹבָתָהּ לְמַטָּה אִם לְהֶפֶךְ, וְהִיא גַם הִיא שָׁכְחָה אֵיזֶה מִין קָבְעָה לְנִדְרָהּ אִם תּוֹרִים אִם בְּנֵי יוֹנָה, צְרִיכָה לְהָבִיא עוֹד חָמֵשׁ פְּרִידוֹת, דְּשֶׁמָּא הַנֶּדֶר נַעֲשָׂה לְמַטָּה וְהַחוֹבָה לְמַעְלָה וְאֵין כָּשֵׁר אֶלָּא עוֹלָה שֶׁבַּחוֹבָה בִּלְבַד וְנִשְׁאַר עָלֶיהָ חַטָּאת שֶׁל חוֹבָה וּשְׁתֵּי עוֹלוֹת שֶׁל הַנֶּדֶר, וּמִפְּנֵי שֶׁשָּׁכְחָה בְּאֵיזֶה מִין קָבְעָה נִדְרָהּ צְרִיכָה לְהָבִיא שְׁתֵּי תוֹרִים וּשְׁנֵי בְנֵי יוֹנָה וְלַעֲשׂוֹת אַרְבַּעְתָּן לְמַעְלָה כְּדֵי לָצֵאת יְדֵי נִדְרָהּ, וּצְרִיכָה לְהָבִיא נַמִּי בֶּן יוֹנָה אוֹ תוֹר לְחַטָּאת לְהַשְׁלִים חוֹבָתָהּ, הֲרֵי חָמֵשׁ פְּרִידוֹת. וְהָנֵי מִלֵּי כְּשֶׁהֵבִיאָה מִתְּחִלָּה נִדְרָהּ וְחוֹבָתָהּ הַכֹּל מִמִּין אֶחָד, אֲבָל אִם יָדְעָה שֶׁמִּתְּחִלָּה הֵבִיאָה נִדְרָהּ וְחוֹבָתָהּ מִשְּׁנֵי מִינִים וְשָׁכְחָה מֵאֵיזֶה מִין קָבְעָה נִדְרָהּ, צְרִיכָה לְהָבִיא שֵׁשׁ פְּרִידוֹת, אַרְבַּע לְהַשְׁלִים נִדְרָהּ דְּהַיְנוּ שְׁתֵּי תוֹרִים וּשְׁנֵי בְנֵי יוֹנָה כְּדַאֲמַרַן לְעֵיל, כֵּיוָן שֶׁאֵינָהּ יוֹדַעַת בְּאֵיזֶה מִין קָבְעָה לְנִדְרָהּ וּצְרִיכָה לָצֵאת יְדֵי שְׁנֵיהֶם, וּשְׁתֵּי תוֹרִים אוֹ שְׁנֵי בְנֵי יוֹנָה לַעֲשׂוֹת אֶחָד חַטָּאת וְאֶחָד עוֹלָה לְהַשְׁלִים חוֹבָתָהּ. וְהָכָא לֹא אָמְרִינַן דְּדַי לָהּ בְּבֶן יוֹנָה אוֹ תוֹר לְחַטָּאת בִּלְבַד וְתַעֲלֶה לָהּ הָעוֹלָה שֶׁהִקְרִיבָה תְחִלָּה לְשֵׁם חוֹבָה כְּדַאֲמַרַן לְעֵיל, דְּבִשְׁלָמָא כְּשֶׁהֵבִיאָה מִמִּין אֶחָד אֵינָהּ צְרִיכָה לְהָבִיא לְעוֹלַת חוֹבָתָהּ, מִמַּה נַפְשָׁךְ, אִם אוֹתָן תּוֹרִין שֶׁקָּבְעָה לְנִדְרָהּ נַעֲשָׂה לְמַטָּה שְׁתֵּיהֶן, אִם כֵּן חוֹבָתָהּ לְמַעְלָה כֻּלָּהּ וְיָצְאָה יְדֵי חוֹבָתָהּ מֵעוֹלָה, וְאִם חוֹבָתָהּ נַעֲשֵׂית עוֹלָה לְמַטָּה וְחַטָּאת לְמַעְלָה, אִם כֵּן נִדְרָהּ נַמִּי נַעֲשֵׂית אַחַת לְמַעְלָה וְאַחַת לְמַטָּה וְיָצְאָה יְדֵי נִדְרָהּ מִתּוֹךְ אֶחָד, וּשְׁתֵּי תוֹרִים שֶׁהֵבִיאָה עַכְשָׁיו בָּאַחַת יָצְאָה יְדֵי נִדְרָהּ וּבָאַחַת יְדֵי עוֹלַת חוֹבָתָהּ. וְאִם חוֹבָתָהּ נַעֲשֵׂית כֻּלָּהּ לְמַטָּה וְהַחַטָּאת כְּשֵׁרָה וְהָעוֹלָה פְּסוּלָה, הַשְּׁתֵּי תוֹרִים שֶׁהֵבִיאָה עַכְשָׁיו בְּאַחַת הִשְׁלִימָה חוֹבָתָהּ וְהָאַחַת הִיא נְדָבָה, וְחַטָּאת שֶׁמְּבִיאָה עַכְשָׁיו תָּבִיא עַל הַסָּפֵק וְלֹא תֵּאָכֵל {יא}. וְאִם הָיוּ הָרִאשׁוֹנִים יוֹנִים תָּבִיא עַכְשָׁיו חַטָּאתָהּ בֶּן יוֹנָה, וְעוֹלָה אֵינָהּ צְרִיכָה מִמַּה נַפְשָׁךְ כְּמוֹ שֶׁאָמַרְנוּ בַתּוֹרִים. אֲבָל כְּשֶׁהֵבִיאָה שְׁנֵי מִינִים וְשָׁכְחָה אֵיזֶה מִין קָבְעָה לְנִדְרָהּ וְאֵיזֶה מִין קָבְעָה לְחוֹבָתָהּ {יב} וְגַם שָׁכְחָה מַה פֵּרְשָׁה, וְעָשָׂה הַכֹּהֵן כָּל מִין וּמִין אֶחָד לְמַעְלָה וְאֶחָד לְמַטָּה {יג} שֶׁהָיָה סָבוּר שֶׁשְּׁנֵיהֶן חוֹבוֹת סְתוּמוֹת, תָּבִיא אַרְבַּע לְנִדְרָהּ, שְׁתֵּי תוֹרִין וּשְׁנֵי יוֹנִים, לְפִי שֶׁאֵינָהּ יוֹדַעַת אֵיזֶה מִין קָבְעָה לְנִדְרָהּ, וְגַם שְׁנֵי מִינִין [צְרִיכָה לְחַטָּאת לְזַוְּגָהּ לְעוֹלַת חוֹבָתָהּ], לְפִי שֶׁאֵינָהּ יוֹדַעַת מֵאֵיזֶה מִין הָיְתָה {יד}, וַאֲנַן תּוֹר כְּנֶגֶד תּוֹר וְיוֹנָה כְּנֶגֶד יוֹנָה בָּעִינַן: <B>נְתָנָתַם לַכֹּהֵן וְאֵין יָדוּעַ כוּ'.</B> כְּגוֹן שֶׁקָּבְעָה נִדְרָהּ וְקָבְעָה חוֹבָתָהּ וְהֵבִיאָה לַכֹּהֵן קֵן לְנִדְרָהּ וְקֵן לְחוֹבָתָהּ וְאֵינָהּ יוֹדַעַת מֵאֵיזֶה מִין קָבְעָה נִדְרָהּ וּמֵאֵיזֶה מִין קָבְעָה חוֹבָתָהּ, אוֹ אִם נָתְנָה לַכֹּהֵן שְׁנֵי מִינִין אוֹ מִין אֶחָד וְהַכֹּהֵן אֵינוֹ יוֹדֵעַ אִם עָשָׂה הַכֹּל לְמַעְלָה אוֹ הַכֹּל לְמַטָּה אוֹ מֶחֱצָה לְמַעְלָה וּמֶחֱצָה לְמַטָּה, הֲרֵי זוֹ מְבִיאָה אַרְבַּע פְּרִידוֹת לְנִדְרָהּ, שְׁתֵּי תוֹרִין וּשְׁנֵי בְּנֵי יוֹנָה, וּמְבִיאָה גַם כֵּן שְׁתַּיִם לְחוֹבָתָהּ אַחַת חַטָּאת וְאַחַת עוֹלָה מֵאֵיזֶה מִשְּׁנֵי מִינִים שֶׁתִּרְצֶה, וּמְבִיאָה חַטָּאת {טו}: <B>אָמַר רַבִּי יְהוֹשֻׁעַ זֶהוּ שֶׁאָמְרוּ כוּ'.</B> כְּשֵׁם שֶׁהַכֶּבֶשׂ חַי אֵין יוֹצֵא מִמֶּנּוּ אֶלָּא קוֹל אֶחָד, וּכְשֶׁהוּא מֵת שֶׁהָיָה רָאוּי שֶׁיִּפְסֹק קוֹלוֹ יוֹצְאִים מִמֶּנּוּ שִׁבְעָה קוֹלוֹת כְּדִמְפָרֵשׁ וְאָזֵיל, הָכָא נַמִּי כְּשֶׁנָּדְרָה תְחִלָּה קֹדֶם שֶׁהֵבִיאָה כְלוּם לֹא הָיְתָה צְרִיכָה אֶלָּא קֵן לְנִדְרָהּ וְקֵן לְחוֹבָתָהּ, וּלְאַחַר שֶׁקָּבְעָה נִדְרָהּ וְהֵבִיאָה נִדְרָהּ וְחוֹבָתָהּ, לְפִי שֶׁאֵינָהּ יוֹדַעַת מַה קָּבְעָה וְאֵין יוֹדֵעַ הַכֹּהֵן מַה הִקְרִיב, צְרִיכָה לְהָבִיא אַרְבַּע פְּרִידוֹת לְנִדְרָהּ וְאַרְבַּע לְחוֹבָתָהּ כְּדַאֲמַרַן {טז}: <B>קַרְנָיו לַחֲצוֹצְרוֹת.</B> לְשׁוֹפָרוֹת. דְּקָרִינַן לְשִׁפּוּרָא חֲצוֹצַרְתָא {יז}: <B>אַף צַמְרוֹ.</B> לַעֲשׂוֹת מִמֶּנּוּ תְּכֵלֶת. עוֹשִׂים מִצַּמְרוֹ מְעִיל שֶׁכֻּלּוֹ תְכֵלֶת וְעַל שׁוּלָיו פַּעֲמוֹנִים שֶׁמַּשְׁמִיעִים קוֹל. וְתַנָּא קַמָּא תְּכֵלֶת לֹא קָא חֲשִׁיב, לְפִי שֶׁאֵין הַתְּכֵלֶת מַשְׁמִיעַ קוֹל אֶלָּא הַפַּעֲמוֹנִים שֶׁעַל שׁוּלָיו סָבִיב: <B>רַבִּי שִׁמְעוֹן בֶּן עֲקַשְׁיָא אוֹמֵר.</B> אַף כָּל זֶה דּוֹמֶה לְהַךְ מִלְּתָא, דִּכְשֵׁם שֶׁכְּשֶׁהוּא מֵת קוֹלוֹ שִׁבְעָה, כָּךְ זִקְנֵי תוֹרָה כָּל זְמַן שֶׁמַּזְקִינִים וּבָאִים לִידֵי תַּשּׁוּת כֹּחַ, הֵן מוֹסִיפִין חָכְמָה {יט}: <B>שֶׁנֶּאֱמַר מֵסִיר שָׂפָה לְנֶאֱמָנִים.</B> וּבְעַמֵּי הָאָרֶץ מִשְׁתַּעֵי קְרָא, דִּלְעֵיל מִנֵּיהּ כְּתִיב מֵסִיר לֵב רָאשֵׁי עַם הָאָרֶץ: <B>בִּישִׁישִׁים חָכְמָה.</B> הַיְנוּ תַּלְמִידֵי חֲכָמִים. דְּאִלּוּ עַמֵּי הָאָרֶץ, חַכְמַת מָה לָהֶם:
